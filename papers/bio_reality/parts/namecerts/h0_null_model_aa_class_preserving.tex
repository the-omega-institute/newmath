\paragraph{Null-model result.}

$$
\begin{aligned}
\mathrm{exact\_match\_count}&=0,\\
\widehat p_{\mathrm{aa\ class,exact}}&=0.0.
\end{aligned}
$$

$$
\begin{aligned}
\widehat p_{\mathrm{aa\ class,lower\ tail}}=\frac{7}{2000}=0.0035.
\end{aligned}
