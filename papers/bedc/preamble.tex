% Rendering rule: the target identifier is rendered without any literal `_`.
% Caller writes `\_` (LaTeX escaped underscore) inside the target name; the
% macro replaces every `\_` with a thin space, so `Foo\_bar` renders as
% `Foo bar`. Namespace dots stay literal.
%
% Usage:
%   \leanchecked{BEDC.BaseReflection.PackageReflection\_base}
%     → green  Lean4 ✓  BEDC.BaseReflection.PackageReflection base
%   \leansorryd{BEDC.BaseReflection.PsameBase\_inversion}{Prop→Type elim; inversion carrier redesign pending}
%     → orange Lean4 (sorry) BEDC.BaseReflection.PsameBase inversion --- rationale
%   \leanstmt{BEDC.BaseReflection.TokUnique.tokenReplacement}
%     → blue   Lean4 (statement) BEDC.BaseReflection.TokUnique.tokenReplacement
%   \leandef{BEDC.BaseReflection.GeneratedSameSig}
%     → gray   Lean4 (definition) BEDC.BaseReflection.GeneratedSameSig
