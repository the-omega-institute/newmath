% Rendering rule: the target identifier is rendered without any literal `_`.
% Caller writes `\_` (LaTeX escaped underscore) inside the target name; the
% macro replaces every `\_` with a thin space, so `Foo\_bar` renders as
% `Foo bar`. Namespace dots stay literal.
%
% Usage:
%   \leanchecked{BEDC.BaseReflection.PackageReflection\_base}
%     → green   Lean4 ✓  BEDC.BaseReflection.PackageReflection base
%        (canonical primary implementation; one per paper site)
%   \leanvariant{BEDC.BaseReflection.PackageReflection\_base\_with\_tokens}
%     → gray    └─ variant  BEDC.BaseReflection.PackageReflection base with tokens
%        (auxiliary version: typeclass-binder wrapper, weakened conclusion,
%        projection, or alternative proof of the same paper claim; multiple
%        per primary marker allowed)
%   \leandef{BEDC.BaseReflection.TokUnique.tokenReplacement}
%     → gray    Lean4 (definition) BEDC.BaseReflection.TokUnique.tokenReplacement

% Underscores inside \leantarget MUST be \_-escaped.

% The block is parsed by lean4/scripts/bedc_ci.py audit and consumed by
% lean4/scripts/critical_path.py to decide whether a chapter is retired
% from the horizon work queue. Both axes are mandatory.
\NewDocumentEnvironment{closurestatus}{m}{%
  \par\smallskip\noindent%
  \begingroup%
    \footnotesize\sffamily\color{purple!55!black}%
    \textbf{Closure status for $\NameCert_{#1}$.}\par
    \def\theoryclosure##1{\par\textbf{Theory closure:} $\TheoryGate(\NameCert_{#1},\,##1)$\par}%
    \def\scopeclosed##1{\textbf{Scope closed:} ##1\par}%
    \def\formalstatus##1{\textbf{Formal verification:} $\FormalStatus(\NameCert_{#1},\,##1)$\par}%
    \def\leantarget##1{\textbf{Lean target:} \texttt{\detokenize{##1}}\par}%
    \def\bridgestatus##1{\textbf{Bridge status:} ##1\par}%
    \def\notclaimed##1{\textbf{Not claimed:} ##1\par}%
    \def\upgradepath##1{\textbf{Upgrade path:} ##1\par}%
}{%
  \par\endgroup\smallskip
}
