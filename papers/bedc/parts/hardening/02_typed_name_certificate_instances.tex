\chapter{Typed Naming Certificate Instances}
\label{ch:hardening-typed-name-certificate-instances}

The BEDC kernel specifies the shape of a naming certificate. This chapter records first explicit instance templates. These are not complete reconstructions of traditional mathematics. They are interface seeds whose role is to show how the naming mechanism should be used.

\section{Certificate fields}

\begin{definition}[Typed naming certificate, expanded template]
\label{def:typed-naming-certificate-expanded-template}
A typed naming certificate for a derived interface \(N\) is a record
\[
  \NameCert(N)=
  (\SourceSpec_N,\PatternSpec_N,\ClassifierSpec_N,\StabCert_N,\LedgerPolicy_N).
\]
The fields have the following intended roles:
\begin{enumerate}
  \item \(\SourceSpec_N\): declares the permitted source histories or visible tokens.
  \item \(\PatternSpec_N\): states the behavior pattern being named.
  \item \(\ClassifierSpec_N\): states how the behavior is packaged or identified.
  \item \(\StabCert_N\): records closure, reuse, descent, composition, or seal stability.
  \item \(\LedgerPolicy_N\): specifies which gap ledger records the compression performed by the name.
\end{enumerate}
\end{definition}

\section{Unary repetition as \texorpdfstring{\(\NatUp\)}{Nat-up} seed}

\begin{definition}[Unary-source specification]
\label{def:unary-source-specification-hardening}
The unary-source specification is the class of histories generated by repeated extension with \(\bone\) from the empty history. It is not called a natural number object internally. It is a source pattern of repeated emission.
\end{definition}
\leanchecked{BEDC.FKernel.Unary.UnarySourceSpec\_induction}

\begin{definition}[Unary pattern]
The unary pattern is the stable behavior of repeated \(\bone\)-extension together with the ability to distinguish the empty history from an extended unary history by the generated history constructors.
\end{definition}

\begin{definition}[\texorpdfstring{\(\NatUp\)}{Nat-up} certificate seed]
The name \(\NatUp\) is licensed only when the following fields are supplied:
\begin{enumerate}
  \item source: unary histories;
  \item pattern: repeated \(\bone\)-extension;
  \item classifier: internal history-sameness restricted to unary histories;
  \item stability: closure under further \(\bone\)-extension and reuse as a counting interface;
  \item ledger: the originating unary history behind each visible counting token.
\end{enumerate}
\end{definition}

\begin{corollary}[\texorpdfstring{\(\NatUp\)}{Nat-up} is not primitive]
\label{cor:natup-certificate-seed-not-primitive}
The interface \(\NatUp\) is licensed by a naming certificate over unary histories. It is not an internal primitive of the kernel.
\end{corollary}
\leanchecked{BEDC.FKernel.Unary.nat\_up\_certificate\_seed\_not\_primitive}

\section{Continuation as \texorpdfstring{\(\AddUp\)}{Add-up} seed}

\begin{definition}[Continuation pattern]
\label{def:continuation-pattern-hardening}
The continuation pattern is the relation \(\Cont(h,k,r)\), read as: the history \(h\) continued by the history \(k\) yields result \(r\), up to internal history-sameness.
\end{definition}
\leanchecked{BEDC.FKernel.Cont.continuation\_pattern\_stability\_fields}

\begin{definition}[\texorpdfstring{\(\AddUp\)}{Add-up} certificate seed]
The name \(\AddUp\) is licensed only when the following fields are supplied:
\begin{enumerate}
  \item source: pairs of unary histories;
  \item pattern: relational continuation;
  \item classifier: internal history-sameness of continuation results;
  \item stability: left unit, right unit, determinacy up to \(\HSame\), and associativity up to \(\HSame\);
  \item ledger: the pair of source unary histories and the continuation witness.
\end{enumerate}
\end{definition}

\begin{theorem}[Addition-like behavior is a certificate, not an operation]
\label{thm:addition-like-behavior-certificate-not-operation}
The interface \(\AddUp\) is not admitted by writing a binary operation symbol. It is admitted by a naming certificate whose stability field contains unit and associativity witnesses for relational continuation.
\end{theorem}
\leanchecked{BEDC.FKernel.NameCert.add\_like\_behavior\_certified\_not\_operation}

\begin{proof}
The kernel contains histories and the relation \(\Cont\). It does not contain a primitive binary operation called addition. If continuation is shown to be stable on the unary source pattern and its behavior descends to the unary classification, then a derived interface may be named \(\AddUp\). The name records the evidence; it does not introduce a primitive operation.
\end{proof}

\section{Descent certificate for stable transformations}

\begin{definition}[Descent certificate]
\label{def:descent-certificate}
A descent certificate for a transformation pattern states that the pattern respects the classifier on the source and the classifier on the target. In explanatory notation, if \(a\) and \(a'\) are source-same, then their transformed outputs are target-same.
\end{definition}
\leandef{BEDC.FKernel.NameCert.DescentCertificate}
\leanchecked{BEDC.FKernel.NameCert.StableTransformation\_descentCertificate\_exists}

\begin{theorem}[Descent certificate respects source sameness]
\label{thm:descent-certificate-respects}
If a descent certificate maps a source classifier to a target classifier, then source-same inputs have target-same mapped outputs.
\end{theorem}
\leanchecked{BEDC.FKernel.NameCert.descentCertificate\_respects}
\leanvariant{BEDC.FKernel.NameCert.stableTransformation\_descent\_respects\_and\_ledger}
\leanvariant{BEDC.FKernel.NameCert.StableTransformation\_descent\_certificate\_and\_respects}
\leanvariant{BEDC.FKernel.NameCert.StableTransformation\_descentCertificate\_respects\_projection}

\begin{proof}
Unpack the descent certificate and apply its respect field to the source-sameness witness.
\end{proof}

\begin{theorem}[Stable transformations descend to packages]
\label{thm:stable-transformations-descend-to-packages}
If visible source package tokens introduce \(p\) and \(q\), and the source classifier identifies them, then any supplied stability witness that respects the package classifier transports that identification to the target package classifier.
\end{theorem}
\leanchecked{BEDC.FKernel.NameCert.stableTransformation\_descends\_to\_packages}
\leanvariant{BEDC.FKernel.Package.stable\_transformations\_descend\_to\_packages}

\begin{proof}
Apply the supplied descent witness to the two visible token introductions and the source package-sameness proof.
\end{proof}

\begin{corollary}[Function-like interfaces require descent]
\label{thm:function-like-interfaces-require-descent}
A function-like derived interface cannot be licensed by a raw transformation alone. It requires a descent certificate.
\end{corollary}
\leanchecked{BEDC.FKernel.NameCert.function\_like\_interfaces\_require\_descent}

\begin{corollary}[Function-like interfaces are derived]
\label{cor:function-like-interfaces-are-derived}
\label{cor:function-like-interfaces-derived-core}
A function-like interface obtained from a stable transformation carries a descent certificate, a respect proof, and a ledger witness.
\end{corollary}
