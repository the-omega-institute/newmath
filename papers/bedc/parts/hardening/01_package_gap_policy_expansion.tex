\chapter{Package and Gap Policy Expansion}
\label{ch:hardening-package-gap-policy-expansion}

The BEDC kernel separates asking, packaging, and gap recording. This chapter refines the public obligations of package and gap policies without changing the core declarations.

\section{Package policy as a public interface}

\begin{definition}[Package-policy interface]
\label{def:package-policy-interface}
A package policy for a bundle \(\Pi\) is a public contract specifying how generated answer histories are represented by visible package tokens. It is not a quotient object and it is not a set of equivalence classes. It consists of three obligations:
\begin{enumerate}
  \item \textbf{Existence:} every admitted signature history has an admitted package token.
  \item \textbf{Extensionality:} internally same signature histories admit internally same package tokens.
  \item \textbf{Grounding:} internally same package tokens are justified by internally same admitted signatures.
\end{enumerate}
\end{definition}
\leanchecked{BEDC.FKernel.Package.PackagePolicy\_iff\_fields}
\leanvariant{BEDC.FKernel.Package.packagePolicy\_field\_witnesses}
\leanvariant{BEDC.FKernel.Package.packagePolicy\_token\_exists}
\leanvariant{BEDC.FKernel.Package.packagePolicy\_nonempty\_token}
\leanvariant{BEDC.FKernel.Package.packagePolicy\_fields}
\leanvariant{BEDC.FKernel.Package.packagePolicy\_extensionality\_witness}

\begin{principle}[No quotient shortcut]
\label{prin:no-quotient-shortcut}
A package token is not introduced as a quotient class. It is introduced through a package policy. Quotient-like behavior is then a theorem, not a primitive.
\end{principle}
\leanchecked{BEDC.FKernel.Package.packagePolicy\_no\_quotient\_shortcut}

\begin{theorem}[Package tokens are signature-facing]
\label{thm:package-tokens-are-signature-facing}
Assume a package policy for \(\Pi\). If two source histories produce internally same signatures under \(\Pi\), then their visible package tokens are internally same under the package-sameness relation licensed by the package policy.
\end{theorem}
\leanchecked{BEDC.FKernel.Package.package\_tokens\_signature\_facing}
\leanvariant{BEDC.FKernel.Package.package\_tokens\_are\_signature\_facing}

\begin{proof}
The source histories first produce signatures. The internal sameness of signatures triggers the extensionality field of the package policy. Hence the package tokens assigned to these signatures are package-same. No quotient object is invoked.
\end{proof}

\section{Gap policy as provenance interface}

\begin{definition}[Gap-policy interface]
A gap policy for a package policy over domain \(D\) specifies a provenance membership relation
\[
  \InGap_{\Pi,D}(p,h),
\]
read as: history \(h\) is recorded in the gap ledger behind visible package token \(p\).
It must satisfy three obligations:
\begin{enumerate}
  \item \textbf{Coverage:} every admitted source history belongs to at least one visible gap ledger.
  \item \textbf{Separation:} if a source history belongs to two package ledgers, then those packages are package-same.
  \item \textbf{Signature determination:} membership is justified by a signature event and a package event for that signature.
\end{enumerate}
\end{definition}

\begin{theorem}[Gap ledgers are not optional]
For any nontrivial package policy, a gap policy is required to recover provenance. Without it, the visible package token has no internal record of which histories were compressed into it.
\end{theorem}

\begin{proof}
Packaging replaces source histories by visible tokens. If a visible token is used without a gap ledger, then the theory no longer contains an internal witness of the histories that led to the token. This violates the ledger-for-compression principle. Hence a gap policy is part of the admissible globalization interface.
\end{proof}

\begin{corollary}[Globalization has three layers]
\label{cor:globalization-has-three-layers}
A well-formed finite globalization interface is not merely a packaging operation. It consists of an asking policy, a package policy, and a gap policy.
\end{corollary}
\leanchecked{BEDC.FKernel.Gap.globalization\_has\_three\_layers}

\section{Composite gaps}

\begin{definition}[Composite gap ledger]
Let \(C\) compress histories into intermediate visible tokens and let \(D\) compress those intermediate tokens into final visible tokens. The composite gap relation is generated by a two-step witness:
\[
  \CompGap_{D\circ C}(z,h)
\]
when there exists an intermediate token \(y\) such that \(h\) is in the \(C\)-gap behind \(y\), and \(y\) is in the \(D\)-gap behind \(z\).
\end{definition}

\begin{theorem}[Composite coverage]
\label{thm:hardening-composite-coverage}
If \(C\) has coverage on the source domain and \(D\) has coverage on the intermediate image, then \(D\circ C\) has composite coverage on the source domain.
\end{theorem}
\leanchecked{BEDC.FKernel.Gap.hardening\_composite\_coverage}
\leanvariant{BEDC.FKernel.Gap.compGap\_coverage\_exact}
\leanvariant{BEDC.FKernel.Gap.compGap\_coverage\_with\_intermediate}
\leanvariant{BEDC.FKernel.Gap.compGap\_coverage\_with\_source\_and\_intermediate}

\begin{proof}
Let \(h\) be an admitted source history. By coverage of \(C\), there exists an intermediate token \(y\) with \(h\) in the \(C\)-gap behind \(y\). By coverage of \(D\), there exists a final token \(z\) with \(y\) in the \(D\)-gap behind \(z\). Thus the pair \((y,z)\) generates a composite gap witness placing \(h\) behind \(z\).
\end{proof}

\begin{remark}
Composite gap ledgers are the formal expression of the guiding slogan that a compressed result must retain a source map. The map may be direct or staged; the staged case is handled by composite gaps.
\end{remark}
