\section{Domain Policy and Transport Obligations}
\label{sec:po-domain-policy-abstract}

This chapter records the internal domain judgment \(\InDom(D,h)\) and the additional proof obligations needed for domain behavior.

\begin{definition}[Domain policy]
A source-domain name \(D\) is equipped with a domain policy, written
\[
  \DomainPolicy(D),
\]
when the following BEDC-level behavior is supplied.
\begin{enumerate}
  \item \textbf{Admission judgment.} Source membership is expressed only by \(\InDom(D,h)\).
  \item \textbf{History transport.} If \(\InDom(D,h)\) and \(\hsame(h,k)\), then \(\InDom(D,k)\).
  \item \textbf{No set-membership import.} No theorem in the active concrete layer may use the notation \(h\in D\) as an internal premise.
\end{enumerate}
\end{definition}

\begin{theorem}[Domain transport]
\label{thm:domain-transport}
Assume \(\DomainPolicy(D)\). If
\[
  \InDom(D,h)
  \quad\text{and}\quad
  \hsame(h,k),
\]
then
\[
  \InDom(D,k).
\]
\end{theorem}
\leanchecked{BEDC.FKernel.Gap.domain\_transport}
\leanvariant{BEDC.FKernel.Gap.domain\_transport\_refl}
\leanvariant{BEDC.FKernel.Gap.domain\_policy\_transport\_witnesses}
\leanvariant{BEDC.FKernel.Gap.DomainPolicy\_iff\_transport}
\leanvariant{BEDC.FKernel.Gap.domain\_transport\_symmetric}
\leanvariant{BEDC.FKernel.Gap.domain\_transport\_transitive}

\begin{proof}
This is the transport field of \(\DomainPolicy(D)\). The theorem is recorded separately because later gap arguments must cite it explicitly rather than silently treating \(D\) as a set.
\end{proof}

\begin{corollary}[Domain invariance of concrete source admission]
\label{cor:domain-invariance-of-concrete-source-admission}
Under \(\DomainPolicy(D)\), the source side of the concrete signature gap is invariant under internal history-sameness.
\end{corollary}
\leanchecked{BEDC.FKernel.Gap.domain\_invariance\_of\_concrete\_source\_admission}
\leanvariant{BEDC.FKernel.Gap.domain\_invariance}
\leanvariant{BEDC.FKernel.Gap.DomainPolicy\_transport\_and\_invariance}
\leanvariant{BEDC.FKernel.Gap.inGapSig\_domain\_transport\_with\_signature}
\leanvariant{BEDC.FKernel.Gap.inGapSig\_domain\_transport\_source}

\begin{proof}
Immediate from Theorem~\ref{thm:domain-transport} in both directions, using symmetry of \(\hsame\) for the reverse direction. The other parts of a concrete gap, namely signature generation and token introduction, require their own policies and are not bundled into domain transport.
\end{proof}

\begin{remark}[Active reading]
Active concrete gap statements use \(\InDom(D,h)\), not \(h\in D\). What is not automatic is any claim that an arbitrary domain name satisfies \(\DomainPolicy(D)\). That must be supplied for each concrete domain.
\end{remark}

\section{Concrete Domain Policy for the Unary Source}
\label{sec:po-domain-policy-unary}

\autoref{sec:po-domain-policy-abstract} introduces \(\mathsf{DomainPolicy}(D)\) as the condition allowing \(\InDom(D,h)\) to be transported along internal history-sameness. This section records the unary source instance.

\begin{definition}[Unary source domain]
Let \(\mathsf{UnarySource}\) be the source-domain name whose internal domain judgment is
\[
  \InDom(\mathsf{UnarySource},h)
  \quad\text{exactly when}\quad
  \Unary(h).
\]
This is not a set-membership statement. It is a domain judgment licensed by the unary-generation rules.
\end{definition}

\begin{lemma}[Unary is transported by history-sameness]
\label{lem:unary-transport}
If
\[
  \Unary(h),\qquad \hsame(h,k),
\]
then
\[
  \Unary(k).
\]
\end{lemma}
\leanchecked{BEDC.FKernel.Unary.unary\_transport}

\begin{proof}
Proceed by induction on the derivation of \(\Unary(h)\).

For the empty case, \(h\) is internally the empty history. Since \(\hsame(h,k)\), closed generation and no-confusion for \(\hsame\) force \(k\) to be internally the empty history. Hence \(\Unary(k)\).

For the step case, the unary derivation has the form \(\Unary(h_0)\Rightarrow\Unary(\Eone(h_0))\). From \(\hsame(\Eone(h_0),k)\), constructor inversion for \(\hsame\) yields a history \(k_0\) such that \(k\) is internally \(\Eone(k_0)\) and \(\hsame(h_0,k_0)\). By the induction hypothesis, \(\Unary(k_0)\). Therefore \(\Unary(\Eone(k_0))\), and by internal sameness this is the required unary judgment for \(k\).
\end{proof}

\begin{theorem}[Unary source satisfies DomainPolicy]
\label{thm:unary-domain-policy}
\[
  \mathsf{DomainPolicy}(\mathsf{UnarySource}).
\]
\end{theorem}
\leanchecked{BEDC.FKernel.Unary.unary\_domain\_policy}

\begin{proof}
Assume \(\InDom(\mathsf{UnarySource},h)\) and \(\hsame(h,k)\). By definition of \(\mathsf{UnarySource}\), the first premise is \(\Unary(h)\). Lemma~\ref{lem:unary-transport} gives \(\Unary(k)\), hence \(\InDom(\mathsf{UnarySource},k)\).
\end{proof}

\begin{corollary}[Unary-domain gaps may transport sources]
Any concrete gap theorem whose source domain is \(\mathsf{UnarySource}\) may use domain transport along \(\hsame\), provided no additional source-domain restriction is imposed.
\end{corollary}
\leanchecked{BEDC.FKernel.Unary.unary\_domain\_gap\_transport}

\begin{proof}
Immediate from Theorem~\ref{thm:unary-domain-policy} and Lemma~\ref{thm:domain-transport}.
\end{proof}
