\section{Consolidated Signature Gap Policy}
\label{sec:po-gap-policy-abstract}

This chapter restates the concrete signature gap and records the exact assumptions needed for coverage, separation, soundness, and completeness.

\begin{definition}[Policy-supported signature gap]
\label{def:policy-supported-signature-gap}
For a probe bundle $\Pi$, source domain $D$, package $p$, and history $h$, define
$$
  \InGapSig(\Pi,D,p,h)
$$
when there exist an answer history \(s\) and an evidence ledger \(\Delta\) such that
$$
  \InDom(D,h),
  \qquad
  \Sig(\Pi,h,s,\Delta),
  \qquad
  \TokIntro(\Pi,s,p).
$$
\end{definition}

\begin{theorem}[Policy-supported gap coverage]
\label{thm:gap-coverage-policy}
Assume $\AskPol(\Pi,D)$ and
$$
  \forall s,\;\exists p,\;\TokIntro(\Pi,s,p).
$$
If
$$
  \InDom(D,h),
$$
then there exists a token \(p\) such that
$$
  \InGapSig(\Pi,D,p,h).
$$
\end{theorem}
\leanchecked{BEDC.FKernel.Gap.policy\_gap\_coverage}
\leanvariant{BEDC.FKernel.Gap.gap\_coverage\_policy}

\begin{proof}
From $\AskPol(\Pi,D)$ and $\InDom(D,h)$, obtain a generated signature event $\Sig(\Pi,h,s,\Delta)$. From the explicit token-existence premise, obtain $p$ with $\TokIntro(\Pi,s,p)$. These data witness $\InGapSig(\Pi,D,p,h)$.
\end{proof}

\begin{theorem}[Policy-supported gap separation]
\label{thm:gap-separation-policy}
Assume $\AskPol(\Pi,D)$ and $\PkgTokPol(\Pi)$. If
$$
  \InGapSig(\Pi,D,p,h)
  \quad\text{and}\quad
  \InGapSig(\Pi,D,q,h),
$$
then
$$
  \psame(p,q).
$$
\end{theorem}

\begin{theorem}[Token-level Globalize exactness]
\label{thm:globalize-exact}
Assume $\PkgTokPol(\Pi)$, $\TokIntro(\Pi,s,p)$, and $\TokIntro(\Pi,t,q)$. Then $\psame(p,q)\Longleftrightarrow\hsame(s,t)$.
\end{theorem}

\begin{definition}[Classifier-preserving pullback ledger]
\label{def:classifier-preserving-pullback-ledger}
Let $D'$ and $D$ be source-domain names and let $\tau$ be a BEDC history operation written $\tau:D'\to D$. A classifier-preserving pullback ledger for $\tau$ over $\Pi$ consists of a pulled-back membership predicate ${\InGapSig}^{\tau}(\Pi,D',D,p,h')$ with the following clauses:
\begin{enumerate}
  \item source preservation: for every $h'$,
  $$
    \InDom(D',h')\Longrightarrow\InDom(D,\tau(h'));
  $$
  \item ledger classifier preservation: for every visible package token $p$ and source $h'$,
  $$
    {\InGapSig}^{\tau}(\Pi,D',D,p,h')
    \Longleftrightarrow
    \InDom(D',h')\;\text{and}\;\InGapSig(\Pi,D,p,\tau(h')).
  $$
\end{enumerate}
The classifier in the second clause is the package-gap ledger classifier in the sense of \autoref{def:typed-naming-certificate}: the visible token $p$ remains a token of the original bundle $\Pi$, and the pullback changes the source coordinate.
\end{definition}

\begin{theorem}[Package-token pullback separation]
\label{thm:package-token-pullback-separation}
Assume $\PkgPol(\Pi)$ and $\GapPol(\Pi,D)$. Let $\tau:D'\to D$ be equipped with a classifier-preserving pullback ledger over $\Pi$. If two pulled-back gap ledgers meet on the same $D'$-source,
$$
  {\InGapSig}^{\tau}(\Pi,D',D,p,h')
  \quad\text{and}\quad
  {\InGapSig}^{\tau}(\Pi,D',D,q,h'),
$$
then the visible package tokens are package-same in the original package interface of $\Pi$:
$$
  \psame(p,q).
$$
\end{theorem}

\begin{proof}
Unpack the two pulled-back memberships by the ledger classifier preservation clause of \autoref{def:classifier-preserving-pullback-ledger}. They give original ledger memberships
$$
  \InGapSig(\Pi,D,p,\tau(h'))
  \quad\text{and}\quad
  \InGapSig(\Pi,D,q,\tau(h')).
$$
These two memberships lie over the same target-domain source $\tau(h')$. Applying \autoref{thm:gap-separation-from-memberships} to $\GapPol(\Pi,D)$ and the displayed memberships gives $\psame(p,q)$. The $\PkgPol(\Pi)$ premise fixes the package-token interface whose visible tokens are compared; the pullback introduces no separate comparison relation.
\end{proof}

\begin{theorem}[Pulled-back gap policy preserves coverage and separation]
\label{thm:pulled-back-gap-policy-preserves-coverage-and-separation}
Assume $\GapPol(\Pi,D)$. Let $\tau:D'\to D$ be equipped with the source preservation and ledger classifier preservation clauses of \autoref{def:classifier-preserving-pullback-ledger}. Then the pulled-back membership predicate ${\InGapSig}^{\tau}(\Pi,D',D,p,h')$ satisfies the coverage and separation obligations over $D'$:
\begin{enumerate}
  \item for every source history $h'$,
  $$
    \InDom(D',h')\Longrightarrow\exists p,\;{\InGapSig}^{\tau}(\Pi,D',D,p,h');
  $$
  \item for every source history $h'$ and visible package tokens $p,q$,
  $$
    {\InGapSig}^{\tau}(\Pi,D',D,p,h')
    \Longrightarrow
    {\InGapSig}^{\tau}(\Pi,D',D,q,h')
    \Longrightarrow
    \psame(p,q).
  $$
\end{enumerate}
\end{theorem}
\leanchecked{BEDC.FKernel.Gap.PulledBackGapPolicy\_coverage\_separation}

\begin{proof}
For coverage, fix $h'$ and assume $\InDom(D',h')$. The source preservation clause of \autoref{def:classifier-preserving-pullback-ledger} gives $\InDom(D,\tau(h'))$. The coverage field of $\GapPol(\Pi,D)$ in \autoref{def:gap-policy} supplies a visible package token $p$ with
$$
  \InGapSig(\Pi,D,p,\tau(h')).
$$
Using the reverse direction of the ledger classifier preservation clause with $\InDom(D',h')$ and this original membership gives
$$
  {\InGapSig}^{\tau}(\Pi,D',D,p,h'),
$$
which proves pulled-back coverage.

For separation, fix $h'$, $p$, and $q$, and assume
$$
  {\InGapSig}^{\tau}(\Pi,D',D,p,h')
  \quad\text{and}\quad
  {\InGapSig}^{\tau}(\Pi,D',D,q,h').
$$
The forward direction of the ledger classifier preservation clause gives original gap memberships
$$
  \InGapSig(\Pi,D,p,\tau(h'))
  \quad\text{and}\quad
  \InGapSig(\Pi,D,q,\tau(h')).
$$
Applying \autoref{thm:gap-separation-from-memberships} to $\GapPol(\Pi,D)$ and these two memberships gives $\psame(p,q)$.
\end{proof}

\begin{theorem}[Composite pullback gap policy]
\label{thm:composite-pullback-gap-policy-preserves-coverage-and-separation}
Assume $\GapPol(\Pi,D)$. Let $\tau_1:D'\to D$ and $\tau_2:D''\to D'$ be BEDC history operations, each equipped with a classifier-preserving pullback ledger over $\Pi$ in the sense of \autoref{def:classifier-preserving-pullback-ledger}. Let the composite pulled-back membership predicate be the pullback of the $\tau_1$-ledger along $\tau_2$:
$$
  {\InGapSig}^{\tau_1\circ\tau_2}(\Pi,D'',D,p,h'')
  \Longleftrightarrow
  \InDom(D'',h'')\;\text{and}\;
  {\InGapSig}^{\tau_1}(\Pi,D',D,p,\tau_2(h'')).
$$
Then the operation $\tau_1\circ\tau_2:D''\to D$ carries a classifier-preserving pullback ledger over $\Pi$. Its composite membership predicate satisfies the coverage and separation obligations over $D''$:
\begin{enumerate}
  \item for every source history $h''$,
  $$
    \InDom(D'',h'')\Longrightarrow
    \exists p,\;{\InGapSig}^{\tau_1\circ\tau_2}(\Pi,D'',D,p,h'');
  $$
  \item for every source history $h''$ and visible package tokens $p,q$,
  $$
    {\InGapSig}^{\tau_1\circ\tau_2}(\Pi,D'',D,p,h'')
    \Longrightarrow
    {\InGapSig}^{\tau_1\circ\tau_2}(\Pi,D'',D,q,h'')
    \Longrightarrow
    \psame(p,q).
  $$
\end{enumerate}
\end{theorem}

\begin{proof}
First verify the two clauses of \autoref{def:classifier-preserving-pullback-ledger} for the composite operation. If $\InDom(D'',h'')$, the source-preservation clause for the $\tau_2$-ledger gives $\InDom(D',\tau_2(h''))$. Applying the source-preservation clause for the $\tau_1$-ledger gives
$$
  \InDom(D,\tau_1(\tau_2(h''))).
$$
Thus $\tau_1\circ\tau_2$ preserves admitted sources from $D''$ to $D$.

For classifier preservation, fix $p$ and $h''$. Suppose first that
$$
  {\InGapSig}^{\tau_1\circ\tau_2}(\Pi,D'',D,p,h'')
$$
holds. By the displayed definition, $\InDom(D'',h'')$ and
$$
  {\InGapSig}^{\tau_1}(\Pi,D',D,p,\tau_2(h''))
$$
hold. The classifier-preservation clause for the $\tau_1$-ledger gives
$$
  \InGapSig(\Pi,D,p,\tau_1(\tau_2(h''))).
$$
Hence the forward direction of the composite classifier clause holds.

Conversely, assume $\InDom(D'',h'')$ and
$$
  \InGapSig(\Pi,D,p,\tau_1(\tau_2(h''))).
$$
Source preservation for the $\tau_2$-ledger gives $\InDom(D',\tau_2(h''))$. The reverse direction of the classifier-preservation clause for the $\tau_1$-ledger then gives
$$
  {\InGapSig}^{\tau_1}(\Pi,D',D,p,\tau_2(h'')),
$$
and the displayed definition gives
$$
  {\InGapSig}^{\tau_1\circ\tau_2}(\Pi,D'',D,p,h'').
$$
Therefore the composite predicate satisfies
$$
  {\InGapSig}^{\tau_1\circ\tau_2}(\Pi,D'',D,p,h'')
  \Longleftrightarrow
  \InDom(D'',h'')\;\text{and}\;
  \InGapSig(\Pi,D,p,\tau_1(\tau_2(h''))),
$$
which is the classifier-preservation clause for $\tau_1\circ\tau_2$.

The composite operation is therefore equipped with the source-preservation and classifier-preservation clauses required by \autoref{def:classifier-preserving-pullback-ledger}. Applying \autoref{thm:pulled-back-gap-policy-preserves-coverage-and-separation} to $\GapPol(\Pi,D)$ and this composite ledger gives the displayed coverage and separation obligations on $D''$.
\end{proof}
