\section{Consolidated Signature Gap Policy}
\label{sec:po-gap-policy-abstract}

This chapter restates the concrete signature gap and records the exact assumptions needed for coverage, separation, soundness, and completeness.

\begin{definition}[Policy-supported signature gap]
\label{def:policy-supported-signature-gap}
For a probe bundle $\Pi$, source domain $D$, package $p$, and history $h$, define
$$
  \InGapSig(\Pi,D,p,h)
$$
when there exist an answer history \(s\) and an evidence ledger \(\Delta\) such that
$$
  \InDom(D,h),
  \qquad
  \Sig(\Pi,h,s,\Delta),
  \qquad
  \TokIntro(\Pi,s,p).
$$
\end{definition}

\begin{theorem}[Policy-supported gap coverage]
\label{thm:gap-coverage-policy}
Assume $\AskPol(\Pi,D)$ and
$$
  \forall s,\;\exists p,\;\TokIntro(\Pi,s,p).
$$
If
$$
  \InDom(D,h),
$$
then there exists a token \(p\) such that
$$
  \InGapSig(\Pi,D,p,h).
$$
\end{theorem}
\leanchecked{BEDC.FKernel.Gap.policy\_gap\_coverage}
\leanvariant{BEDC.FKernel.Gap.gap\_coverage\_policy}

\begin{proof}
From $\AskPol(\Pi,D)$ and $\InDom(D,h)$, obtain a generated signature event $\Sig(\Pi,h,s,\Delta)$. From the explicit token-existence premise, obtain $p$ with $\TokIntro(\Pi,s,p)$. These data witness $\InGapSig(\Pi,D,p,h)$.
\end{proof}

\begin{theorem}[Policy-supported gap separation]
\label{thm:gap-separation-policy}
Assume \(\DomainPolicy(D)\), \(\AskPol(\Pi,D)\), and \(\PkgTokPol(\Pi)\). If
\[
  \InGapSig(\Pi,D,p,h)
  \quad\text{and}\quad
  \InGapSig(\Pi,D,q,h),
\]
then
\[
  \psame(p,q).
\]
\end{theorem}

\begin{proof}
Unpack both gap memberships. We obtain signatures \(s,t\) and evidence ledgers \(\Delta,\Theta\) such that
\[
  \Sig(\Pi,h,s,\Delta),\qquad \TokIntro(\Pi,s,p),
\]
\[
  \Sig(\Pi,h,t,\Theta),\qquad \TokIntro(\Pi,t,q).
\]
By signature determinacy supplied through \(\AskPol(\Pi,D)\), \(\hsame(s,t)\). By package soundness in \(\PkgTokPol(\Pi)\), \(\psame(p,q)\).
\end{proof}

\begin{theorem}[Token-level Globalize exactness]
\label{thm:globalize-exact}
Assume $\PkgTokPol(\Pi)$, $\TokIntro(\Pi,s,p)$, and $\TokIntro(\Pi,t,q)$. Then $\psame(p,q)\Longleftrightarrow\hsame(s,t)$.
\end{theorem}
\leanchecked{BEDC.FKernel.Gap.policy\_globalize\_exact}

\begin{proof}
This is Theorem~\ref{thm:psame-iff-hsame}.
\end{proof}

\begin{corollary}[Active reading]
\label{cor:active-reading}
The concrete signature Globalize instance is presented only as a policy-supported construction. Its exactness depends on \(\DomainPolicy(D)\), \(\AskPol(\Pi,D)\), and \(\PkgTokPol(\Pi)\).
\end{corollary}
