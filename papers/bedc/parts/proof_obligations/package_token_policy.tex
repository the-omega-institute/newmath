\section{Package Token Policy and Reflection}
\label{sec:po-package-token-policy-abstract}

This chapter replaces the function-like token notation \(\Tok_{\Pi}(s)\) by the judgment \(\TokIntro(\Pi,s,p)\) and packages the necessary token behavior into a single proof interface.

\begin{definition}[Package token policy]
\label{def:package-token-policy}
For a probe bundle \(\Pi\), a package token policy, written
\[
  \PkgTokPol(\Pi),
\]
consists of the following clauses for the concrete signature-token instance.
\begin{enumerate}
  \item \textbf{Token existence.} Every generated answer history \(s\) that is intended to be visible admits some token \(p\) with \(\TokIntro(\Pi,s,p)\).
  \item \textbf{Soundness.} If \(\TokIntro(\Pi,s,p)\), \(\TokIntro(\Pi,t,q)\), and \(\hsame(s,t)\), then \(\psame(p,q)\).
  \item \textbf{Reflection.} If \(\TokIntro(\Pi,s,p)\), \(\TokIntro(\Pi,t,q)\), and \(\psame(p,q)\), then \(\hsame(s,t)\).
  \item \textbf{Package-sameness closure.} On introduced tokens, \(\psame\) is closed under reflexivity, symmetry, and transitivity.
  \item \textbf{No external token equality.} Visible tokens are compared by \(\psame\), not by any imported equality judgment.
\end{enumerate}
\end{definition}
\leanchecked{BEDC.FKernel.Package.PackageTokenPolicy\_iff\_fields}
\leanvariant{BEDC.FKernel.Package.PackageTokenPolicy\_iff\_soundness\_reflection}
\leanvariant{BEDC.FKernel.Package.PackageTokenPolicy\_fields}
\leanvariant{BEDC.FKernel.Package.packageTokenPolicy\_psame\_trans\_on\_introduced}

\begin{theorem}[Package tokens are signature-facing]
\label{thm:package-tokens-signature-facing}
Assume \(\PkgPol(\Pi)\). If
\[
  \TokIntro(\Pi,s,p),\quad
  \TokIntro(\Pi,t,q),\quad
  \hsame(s,t),
\]
then
\[
  \psame(p,q).
\]
\end{theorem}
\leanchecked{BEDC.FKernel.Package.PackagePolicy\_signature\_facing}
\leanchecked{BEDC.FKernel.Package.packagePolicy\_signature\_facing\_from\_fields}
\leanvariant{BEDC.FKernel.Package.packageTokenPolicy\_signature\_facing}

\begin{proof}
This is the extensionality field of the package policy applied to the two token-introduction premises and the history-sameness premise.
\end{proof}

\begin{theorem}[Package-sameness is equivalent to signature-sameness]
\label{thm:psame-iff-hsame}
\label{thm:packages-classify-introduced-signatures}
Assume \(\PkgTokPol(\Pi)\). If
\[
  \TokIntro(\Pi,s,p),
  \qquad
  \TokIntro(\Pi,t,q),
\]
then
\[
  \psame(p,q)
  \quad\Longleftrightarrow\quad
  \hsame(s,t).
\]
\end{theorem}
\leanchecked{BEDC.FKernel.Package.psame\_iff\_hsame}
\leanvariant{BEDC.FKernel.Package.PackageTokenPolicy\_sound\_complete}
\leanvariant{BEDC.FKernel.Package.packageTokenPolicy\_classifies\_introduced\_signatures}

\begin{proof}
The implication from right to left is the soundness clause of \(\PkgTokPol(\Pi)\). The implication from left to right is the reflection clause. No external token equality is used.
\end{proof}

\begin{corollary}[Concrete package completeness is policy-grounded]
\label{cor:concrete-package-completeness-policy-grounded}
The completeness direction of the concrete signature Globalize instance holds only under \(\PkgTokPol(\Pi)\). In particular, it is not a theorem for arbitrary package-sameness relations.
\end{corollary}
\leanchecked{BEDC.FKernel.Package.concrete\_package\_completeness\_policy\_grounded}
\leanvariant{BEDC.FKernel.Package.package\_completeness\_reflection\_field}

\begin{proof}
Completeness requires recovering \(\hsame(s,t)\) from \(\psame(p,q)\). This is exactly the reflection field of \(\PkgTokPol(\Pi)\).
\end{proof}

\begin{remark}[Proof-obligation boundary]
The token policy is deliberately narrow. It applies to the concrete signature-token package instance. Other package constructions must provide their own soundness and reflection certificates before they may inherit the same completeness theorem.
\end{remark}

\section{Inductive Package-Token Policy for Signature Packages}
\label{sec:po-inductive-package-token-policy-instance}

\autoref{sec:po-package-token-policy-abstract} introduces \(\mathsf{PackageTokenPolicy}(\Pi)\). This section removes one remaining ambiguity: package-sameness for the concrete signature-token instance should be generated by one rule only. Package reflection should then be a constructor-inversion theorem, not an extra informal assumption.

\begin{definition}[Concrete signature-token introduction]
For the concrete signature-token instance, the judgment
\[
  \TokIntro(\Pi,s,p)
\]
means that token \(p\) is introduced as a visible token for answer history \(s\) under probe bundle \(\Pi\).
\end{definition}

\begin{definition}[Inductive package-sameness for signature tokens]
\label{def:package-token-and-package-sameness}
For the concrete signature-token instance, \(\psame_{\Pi}^{\mathrm{sig}}(p,q)\) is generated only by the following rule:
\[
\frac{
  \TokIntro(\Pi,s,p)
  \qquad
  \TokIntro(\Pi,t,q)
  \qquad
  \hsame(s,t)
}{
  \psame_{\Pi}^{\mathrm{sig}}(p,q)
}.
\]
No other package-sameness constructors are admitted for this concrete instance.
\end{definition}
\leanchecked{BEDC.FKernel.Package.psame\_iff\_constructor\_witnesses}
\leanvariant{BEDC.FKernel.Package.psame\_left\_token\_witness}
\leanvariant{BEDC.FKernel.Package.psame\_right\_token\_witness}

\begin{lemma}[Package-sameness soundness]
\label{lem:psame-sound}
\label{lem:signature-package-extensionality}
If
\[
  \TokIntro(\Pi,s,p),\quad \TokIntro(\Pi,t,q),\quad \hsame(s,t),
\]
then
\[
  \psame_{\Pi}^{\mathrm{sig}}(p,q).
\]
\end{lemma}
\leanchecked{BEDC.FKernel.Package.psame\_sound}
\leanvariant{BEDC.FKernel.Package.signature\_package\_extensionality}

\begin{proof}
This is the single constructor of \(\psame_{\Pi}^{\mathrm{sig}}\).
\end{proof}

\begin{lemma}[Package-sameness reflection]
\label{lem:psame-reflect}
If
\[
  \TokIntro(\Pi,s,p),\quad \TokIntro(\Pi,t,q),\quad
  \psame_{\Pi}^{\mathrm{sig}}(p,q),
\]
then
\[
  \hsame(s,t).
\]
\end{lemma}
\leanchecked{BEDC.FKernel.Package.psame\_reflect}

\begin{proof}
By inversion on the derivation of \(\psame_{\Pi}^{\mathrm{sig}}(p,q)\). Since the concrete package-sameness relation has only the displayed constructor, any such derivation is generated from introduced signatures that are \(\hsame\). Therefore \(\hsame(s,t)\), up to the token-introduction premises used to identify the displayed tokens.
\end{proof}

\begin{theorem}[Concrete PackageTokenPolicy]
\label{thm:concrete-package-token-policy}
The concrete signature-token instance satisfies \(\mathsf{PackageTokenPolicy}(\Pi)\) whenever token introduction is available for generated signatures and package-sameness is interpreted as \(\psame_{\Pi}^{\mathrm{sig}}\).
\end{theorem}
\leanchecked{BEDC.FKernel.Package.concrete\_package\_token\_policy}

\begin{proof}
Token introduction is supplied by the chosen concrete token-introduction rule. Package-sameness introduction is Lemma~\ref{lem:psame-sound}. Package reflection is Lemma~\ref{lem:psame-reflect}. The no-extra-constructor clause holds by definition of \(\psame_{\Pi}^{\mathrm{sig}}\).
\end{proof}

\leanchecked{BEDC.FKernel.Package.concrete\_packageTokenPolicy}

\begin{corollary}[Exact signature package]
\label{cor:exact-signature-package}
For the concrete signature-token instance, \(\psame_{\Pi}^{\mathrm{sig}}(p,q)\) between introduced tokens is equivalent to internal sameness of the answer histories that introduced them.
\end{corollary}
\leanchecked{BEDC.FKernel.Package.psame\_reflect\_under\_tok\_unique}

\begin{proof}
The forward direction is Lemma~\ref{lem:psame-reflect}; the reverse direction is Lemma~\ref{lem:psame-sound}.
\end{proof}

\begin{remark}[Implementation note]
In a machine implementation, this chapter should be realized by defining package-sameness as an inductive relation with exactly one constructor. Reflection should then be proved by case analysis on that constructor.
\end{remark}

\section{Token-Introduction Uniqueness and Package Reflection}
\label{sec:po-token-intro-uniqueness-policy}

\autoref{sec:po-inductive-package-token-policy-instance} makes concrete package-sameness a one-constructor relation. That is necessary but not sufficient for the strongest reflection theorem. If the same visible token can be introduced by two unrelated signatures, then package-sameness of tokens does not determine the signatures uniquely. This section isolates the missing requirement.

\begin{definition}[Token-introduction uniqueness]
\label{def:token-unique}
For a fixed generated probe bundle \(\Pi\), define the token-introduction uniqueness obligation \(\mathsf{TokUnique}(\Pi)\) as the following principle:
\[
  \TokIntro(\Pi,s,p),\quad
  \TokIntro(\Pi,t,p)
  \quad\Longrightarrow\quad
  \hsame(s,t).
\]
This is an internal uniqueness condition up to history-sameness. It is not host-level equality of tokens or signatures.
\end{definition}
\leanchecked{BEDC.BaseReflection.TokUnique\_intro}
\leanvariant{BEDC.BaseReflection.TokUnique\_iff\_tokenReplacement}
\leandef{BEDC.FKernel.Package.TokUnique}
\leanchecked{BEDC.FKernel.Package.TokUnique\_replacement}
\leanvariant{BEDC.FKernel.Package.tokUnique\_replacement}

\begin{remark}[Why this is needed]
From a derivation of \(\psame_{\Pi}^{\mathrm{sig}}(p,q)\), inversion yields some generated signatures \(s_0,t_0\) introducing \(p,q\), together with \(\hsame(s_0,t_0)\). If the displayed premises use \(s,t\) instead, one still needs to compare \(s\) with \(s_0\) and \(t\) with \(t_0\). This is exactly the role of \(\mathsf{TokUnique}(\Pi)\).
\end{remark}

\begin{lemma}[Token replacement]
\label{lem:token-replacement-policy}
Assume \(\mathsf{TokUnique}(\Pi)\). If
\[
  \TokIntro(\Pi,s,p),\quad
  \TokIntro(\Pi,s_0,p),
\]
then
\[
  \hsame(s,s_0).
\]
\end{lemma}
\leandef{BEDC.BaseReflection.TokUnique.tokenReplacement}
\leanchecked{BEDC.BaseReflection.TokUnique\_replacement\_self}
\leanvariant{BEDC.BaseReflection.TokUnique\_replacement\_congr}

\begin{proof}
This is Definition~\ref{def:token-unique} applied to the two token-introduction judgments for the same token \(p\).
\end{proof}

\begin{lemma}[Symmetric token replacement]
\label{lem:token-replacement-policy-symm}
Assume \(\mathsf{TokUnique}(\Pi)\) and that \(\hsame\) is symmetric. If
\[
  \TokIntro(\Pi,s,p),\quad
  \TokIntro(\Pi,t,p),
\]
then
\[
  \hsame(t,s).
\]
\end{lemma}
\leanchecked{BEDC.BaseReflection.TokUnique\_replacement\_symm}
\leanvariant{BEDC.BaseReflection.TokUnique\_replacement\_trans}
\leanvariant{BEDC.FKernel.Package.tokUnique\_replacement\_symm}

\begin{proof}
Token-introduction uniqueness gives \(\hsame(s,t)\); symmetry of \(\hsame\) gives \(\hsame(t,s)\).
\end{proof}

\begin{theorem}[Concrete package reflection with token uniqueness]
\label{thm:package-reflection-token-unique}
Assume \(\mathsf{TokUnique}(\Pi)\). If
\[
  \TokIntro(\Pi,s,p),\quad
  \TokIntro(\Pi,t,q),\quad
  \psame_{\Pi}^{\mathrm{sig}}(p,q),
\]
then
\[
  \hsame(s,t).
\]
\end{theorem}
\leanchecked{BEDC.BaseReflection.PackageReflection\_token\_unique}
\leanvariant{BEDC.BaseReflection.PackageReflection\_token\_unique\_from\_inversion}
\leanvariant{BEDC.FKernel.Package.psame\_two\_step\_reflect\_under\_tok\_unique}

\begin{proof}
Invert the derivation of \(\psame_{\Pi}^{\mathrm{sig}}(p,q)\). Since the concrete signature-token package-sameness relation has only the constructor displayed in \autoref{sec:po-inductive-package-token-policy-instance}, the derivation supplies signatures \(s_0,t_0\) such that
\[
  \TokIntro(\Pi,s_0,p),\quad
  \TokIntro(\Pi,t_0,q),\quad
  \hsame(s_0,t_0).
\]
By Lemma~\ref{lem:token-replacement-policy}, the first token-introduction premise gives \(\hsame(s,s_0)\), and the second gives \(\hsame(t,t_0)\). Symmetry and transitivity of \(\hsame\) now yield \(\hsame(s,t)\).
\end{proof}

\begin{corollary}[Exact package-sameness under token uniqueness]
\label{cor:exact-package-sameness}
Assume \(\mathsf{TokUnique}(\Pi)\). For introduced tokens,
\[
  \psame_{\Pi}^{\mathrm{sig}}(p,q)
\]
if and only if their introducing signatures are internally history-same.
Under a package token policy, consecutive package-sameness judgments between introduced tokens compose to the outer package-sameness judgment.
\end{corollary}
\leanchecked{BEDC.BaseReflection.PsameBase\_iff\_hsame\_under\_tok\_unique}
\leanvariant{BEDC.BaseReflection.PsameBase\_equivalence\_under\_tok\_unique}
\leanvariant{BEDC.BaseReflection.exact\_package\_sameness}
\leanvariant{BEDC.FKernel.Package.psame\_symm\_under\_policy}
\leanvariant{BEDC.FKernel.Package.psame\_trans\_under\_policy}
\leanchecked{BEDC.FKernel.Package.packageTokenPolicy\_psame\_refl\_on\_introduced}
\leanvariant{BEDC.FKernel.Package.packageTokenPolicy\_psame\_equivalence\_on\_introduced}
\leanvariant{BEDC.FKernel.Package.psame\_trans\_under\_tok\_unique}

\begin{proof}
The forward direction is Theorem~\ref{thm:package-reflection-token-unique}. The reverse direction is the single constructor of \(\psame_{\Pi}^{\mathrm{sig}}\). Transitivity follows by reflecting both package-sameness premises to history-sameness, composing the histories, and applying package-token soundness.
\end{proof}

\begin{theorem}[PackageTokenPolicy from token uniqueness]
\label{thm:package-token-policy-from-token-unique}
The concrete signature-token package policy is exact when it includes token introduction, one-constructor package-sameness, and \(\mathsf{TokUnique}(\Pi)\).
\end{theorem}
\leanchecked{BEDC.FKernel.Package.packageTokenPolicy\_from\_reflection}
\leanvariant{BEDC.FKernel.Package.packageTokenPolicy\_from\_tokUnique}

\begin{proof}
Existence and soundness are supplied by token introduction and the package-sameness constructor. Reflection is Theorem~\ref{thm:package-reflection-token-unique}. The no-extra-constructor clause is the closed-generation clause for \(\psame_{\Pi}^{\mathrm{sig}}\).
\end{proof}

\section{Token-Uniqueness Modes}
\label{sec:po-token-uniqueness-modes}

This section makes \(\TokUnique(\Pi)\) explicit as an active policy field and separates two implementation modes so that the next machine sprint has a precise target.

\begin{definition}[Policy mode]
\label{def:policy-mode}
In policy mode, \(\TokUnique(\Pi)\) is a field of the concrete package-token policy:
\[
  \TokIntro(\Pi,s,p),\quad \TokIntro(\Pi,t,p)
  \quad\Longrightarrow\quad
  \hsame(s,t).
\]
Concrete Globalize exactness theorems may use this field as an assumption.
\end{definition}
\leandef{BEDC.BaseReflection.PolicyTokenMode}
\leanchecked{BEDC.BaseReflection.PolicyTokenMode\_iff\_TokUnique}

\begin{definition}[Canonical-constructor mode]
\label{def:canonical-mode}
In canonical-constructor mode, token introduction is implemented by a canonical token constructor, written schematically as
\[
  \TokCan(\Pi,s,p).
\]
The implementation must provide the no-confusion rule
\[
  \TokCan(\Pi,s,p),\quad \TokCan(\Pi,t,p)
  \quad\Longrightarrow\quad
  \hsame(s,t).
\]
If \(\TokIntro\) is defined by \(\TokCan\), then \(\TokUnique(\Pi)\) follows.
\end{definition}
\leandef{BEDC.BaseReflection.CanonicalTokenMode}

\begin{lemma}[Canonical mode implies token uniqueness]
\label{lem:canonical-implies-unique}
Assume \(\TokIntro(\Pi,s,p)\) is definitionally implemented by \(\TokCan(\Pi,s,p)\), and assume the no-confusion rule of Definition~\ref{def:canonical-mode}. Then \(\TokUnique(\Pi)\) holds.
\end{lemma}

\begin{proof}
Given \(\TokIntro(\Pi,s,p)\) and \(\TokIntro(\Pi,t,p)\), unfold the implementation to obtain \(\TokCan(\Pi,s,p)\) and \(\TokCan(\Pi,t,p)\). The no-confusion rule gives \(\hsame(s,t)\), which is exactly \(\TokUnique(\Pi)\).
\end{proof}
\leanchecked{BEDC.BaseReflection.CanonicalTokenMode\_implies\_TokUnique}
\leanvariant{BEDC.BaseReflection.CanonicalTokenMode\_tokenReplacement}
\leanvariant{BEDC.BaseReflection.CanonicalTokenMode\_replacement\_and\_unique}

\begin{principle}[Active mode]
\label{prin:active-token-mode}
The active paper theorem uses policy mode. Canonical-constructor mode is an implementation target, not a paper-level replacement, until a checked build records the implementation and no-confusion theorem.
\end{principle}
\leanchecked{BEDC.BaseReflection.active\_token\_mode\_reflects\_base}

\begin{corollary}[Stable package reflection contract]
\label{cor:stable-reflection-contract}
The package reflection theorem may be used later only in one of two forms:
\begin{enumerate}
  \item base-relation reflection under \(\TokUnique(\Pi)\);
  \item base-relation reflection under a checked canonical-constructor theorem implying \(\TokUnique(\Pi)\).
\end{enumerate}
It may be used for \(\psame_{\Pi}^{\mathrm{eq}}\) only through the checked closure-reflection theorem of Proposition~\ref{prop:conditional-closure-reflection-schema}; the exact Globalize export remains the base relation.
\end{corollary}
\leanchecked{BEDC.BaseReflection.StableReflectionContract\_from\_canonical}

\begin{proof}
The first form is Theorem~\ref{thm:base-reflection-active}. The second form follows from Lemma~\ref{lem:canonical-implies-unique}. The final exclusion is Corollary~\ref{cor:eq-not-exported}.
\end{proof}
\leanchecked{BEDC.BaseReflection.PackageReflection\_base\_from\_canonical}
\leanvariant{BEDC.BaseReflection.StableReflectionContract}
