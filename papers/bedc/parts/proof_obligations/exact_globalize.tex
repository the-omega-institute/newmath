\section{Exact Concrete Signature Globalize Under Explicit Policies}
\label{sec:po-exact-concrete-globalize-policy}

The concrete signature Globalize instance becomes exact only when all four policy layers are visible:
\[
  \DomainPolicy(D)
  +
  \AskPol(\Pi,D)
  +
  \mathsf{TokUnique}(\Pi)
  +
  \GapPol^{\mathrm{sig}}(\Pi,D).
\]
This chapter records the exactness theorem in its non-overclaiming form.

\begin{definition}[Policy-supported signature gap]
For histories admitted by \(D\), define
\[
  \InGapSig(\Pi,D,p,h)
\]
to mean that there exist an answer history \(s\) and an evidence ledger \(\Delta\) such that
\[
  \InDom(D,h),
  \qquad
  \Sig(\Pi,h,s,\Delta),
  \qquad
  \TokIntro(\Pi,s,p).
\]
\end{definition}

\begin{theorem}[Concrete gap coverage]
\label{thm:gap-coverage-globalize}
Assume \(\AskPol(\Pi,D)\) and that token introduction is available for every generated signature. If \(\InDom(D,h)\), then there exists a token \(p\) such that
\[
  \InGapSig(\Pi,D,p,h).
\]
\end{theorem}
\leanchecked{BEDC.FKernel.Gap.gap\_coverage\_globalize}

\begin{proof}
By the totality field of \(\AskPol(\Pi,D)\), the history \(h\) generates some signature \(s\) with evidence \(\Delta\). Token introduction supplies a token \(p\) with \(\TokIntro(\Pi,s,p)\). The three displayed judgments witness \(\InGapSig(\Pi,D,p,h)\).
\end{proof}

\begin{theorem}[Concrete gap separation]
\label{thm:gap-separation-globalize}
Assume \(\AskPol(\Pi,D)\), \(\mathsf{TokUnique}(\Pi)\), and the one-constructor concrete package-sameness relation. If
\[
  \InGapSig(\Pi,D,p,h)
  \quad\text{and}\quad
  \InGapSig(\Pi,D,q,h),
\]
then
\[
  \psame_{\Pi}^{\mathrm{sig}}(p,q).
\]
\end{theorem}
\leanchecked{BEDC.FKernel.Gap.gap\_separation\_globalize}

\begin{proof}
The two gap premises yield signatures \(s,t\) with
\[
  \Sig(\Pi,h,s,\Delta),\quad
  \Sig(\Pi,h,t,\Theta),
\]
and token-introduction judgments
\[
  \TokIntro(\Pi,s,p),\quad
  \TokIntro(\Pi,t,q).
\]
The determinacy field of \(\AskPol(\Pi,D)\) gives \(\hsame(s,t)\). The package-sameness constructor then gives \(\psame_{\Pi}^{\mathrm{sig}}(p,q)\).
\end{proof}

\begin{theorem}[Concrete Globalize soundness]
\label{thm:globalize-sound}
Assume \(\AskPol(\Pi,D)\). If admitted histories \(h,k\) generate internally same signatures, then any tokens introduced from those signatures are package-same.
\end{theorem}
\leanchecked{BEDC.BaseReflection.GeneratedSameSig\_psameBase}

\begin{proof}
Let \(s,t\) be generated signatures for \(h,k\), with \(\hsame(s,t)\). If \(p,q\) are introduced from \(s,t\), then the single constructor of \(\psame_{\Pi}^{\mathrm{sig}}\) yields \(\psame_{\Pi}^{\mathrm{sig}}(p,q)\).
\end{proof}

\begin{theorem}[Concrete Globalize completeness]
\label{thm:globalize-complete}
Assume \(\mathsf{TokUnique}(\Pi)\). If tokens introduced from generated signatures are package-same, then the generated signatures are internally history-same.
\end{theorem}
\leanchecked{BEDC.BaseReflection.PsameBase\_to\_GeneratedSameSig\_under\_tok\_unique}

\begin{proof}
This is exactly Theorem~\ref{thm:package-reflection-token-unique} applied to the token-introduction premises.
\end{proof}

\begin{corollary}[Exact concrete signature Globalize]
\label{cor:exact-globalize}
Under \(\DomainPolicy(D)\), \(\AskPol(\Pi,D)\), token introduction, \(\mathsf{TokUnique}(\Pi)\), and the one-constructor concrete package-sameness policy, the visible packages of \(\GlobSig(\Pi,D)\) classify admitted histories exactly by generated signature-sameness, with \(\InGapSig\) as the source ledger.
\end{corollary}
\leanchecked{BEDC.BaseReflection.ExactGlobalizeBase\_coverage}
\leanvariant{BEDC.FKernel.Gap.exact\_globalize\_classifies\_signatures}

\begin{proof}
Coverage is Theorem~\ref{thm:gap-coverage-globalize}. Separation is Theorem~\ref{thm:gap-separation-globalize}. Soundness and completeness are Theorems~\ref{thm:globalize-sound} and~\ref{thm:globalize-complete}. Domain transport is supplied by \(\DomainPolicy(D)\).
\end{proof}

\section{Exactness Export Contract}
\label{sec:po-exactness-export-contract-design}

This chapter records the exact theorem exported for active development. It is intentionally narrower than a full package algebra theorem.

\begin{theorem}[Exact concrete signature Globalize]
\label{thm:exact-concrete-globalize}
Assume:
\begin{enumerate}
  \item \(\DomainPolicy(D)\);
  \item \(\AskPol(\Pi,D)\);
  \item token introduction for generated signatures;
  \item \(\TokUnique(\Pi)\), or a checked canonical-constructor theorem implying it;
  \item the one-constructor base package-sameness relation \(\psame_{\Pi}^{\mathrm{base}}\);
  \item the policy-supported signature gap relation \(\InGapSig(\Pi,D,p,h)\).
\end{enumerate}
Then the visible packages of \(\GlobSig(\Pi,D)\) classify admitted histories exactly by generated signature-sameness, where exactness is stated with \(\psame_{\Pi}^{\mathrm{base}}\), not with \(\psame_{\Pi}^{\mathrm{eq}}\).
\end{theorem}
\leanchecked{BEDC.FKernel.Gap.exact\_concrete\_globalize}

\leanvariant{BEDC.BaseReflection.ExactGlobalizeBase\_exact}

\begin{proof}
Coverage is Theorem~\ref{thm:gap-coverage-globalize}. Separation is Theorem~\ref{thm:gap-separation-globalize}, with \(\psame_{\Pi}^{\mathrm{sig}}\) read as \(\psame_{\Pi}^{\mathrm{base}}\) by Definition~\ref{def:base-psame}. Soundness is the base constructor. Completeness is Theorem~\ref{thm:base-reflection-active}. Domain transport is supplied by \(\DomainPolicy(D)\). The theorem exports only the base relation because Corollary~\ref{cor:eq-not-exported} blocks implicit closure export.
\end{proof}

\begin{proposition}[What is intentionally not exported]
\label{prop:not-exported}
The following statements are not exported:
\begin{enumerate}
  \item \(\psame_{\Pi}^{\mathrm{eq}}\)-exactness;
  \item package-sameness transitivity without token-uniqueness hypotheses as a constructor-level theorem for \(\psame_{\Pi}^{\mathrm{base}}\);
  \item equivalence-closure exactness as a Globalize export;
  \item a checked \(\AddUp\) implementation artifact;
  \item a full natural-number or addition construction beyond the licensed unary-continuation interface.
\end{enumerate}
\end{proposition}

\begin{proof}
Each item requires an exactness predicate whose exported relation is not \(\psame_{\Pi}^{\mathrm{base}}\), an implementation artifact, or a derived-object construction that is outside the declared scope of this verification contract.
\end{proof}
\leanchecked{BEDC.BaseReflection.ExactGlobalizeBase\_soundness\_exports\_base}
\leanvariant{BEDC.BaseReflection.NotExported\_relation\_is\_base}
\leanvariant{BEDC.BaseReflection.PsameBase\_trans\_under\_tok\_unique}
\leandef{BEDC.BaseReflection.NotExported}
\leanchecked{BEDC.BaseReflection.NotExported\_from\_exact}
\leanvariant{BEDC.BaseReflection.NotExported\_from\_fields}
\leanvariant{BEDC.BaseReflection.NotExported\_classify\_iff}
\leanvariant{BEDC.BaseReflection.NotExported\_base\_to\_GeneratedSameSig}

\begin{protocol}[Export discipline]
\label{prot:export-discipline}
Modules cite Theorem~\ref{thm:exact-concrete-globalize} as the current exactness theorem. If a module needs \(\psame_{\Pi}^{\mathrm{eq}}\)-exactness, it must state an exactness predicate whose exported relation is \(\psame_{\Pi}^{\mathrm{eq}}\), or explicitly downgrade its result from exact reflection to closure-level compatibility.
\end{protocol}

\section{Checker-Friendly Exactness Predicate}
\label{sec:po-checker-friendly-exactness}

The exact concrete Globalize theorem intentionally states the active relation as \(\psame_{\Pi}^{\mathrm{base}}\). The remaining problem is not the relation boundary, but the phrase ``classify admitted histories exactly''. This chapter replaces that phrase by a predicate with explicit fields.

\begin{definition}[Generated signature sameness]
\label{def:generated-same-sig}
\label{thm:generated-same-sig}
For a generated probe bundle \(\Pi\), define
\[
\begin{aligned}
\GeneratedSameSig(\Pi,h,k) :\Longleftrightarrow\;&
\exists s,t,\Delta,\Theta,\;\Sig(\Pi,h,s,\Delta)\;\wedge\;\Sig(\Pi,k,t,\Theta)\;\wedge\;\hsame(s,t).
\end{aligned}
\]
This is the checker-facing predicate corresponding to ``the two admitted histories generate the same visible signature''.
\end{definition}
\leandef{BEDC.BaseReflection.GeneratedSameSig}
\leanchecked{BEDC.BaseReflection.GeneratedSameSig\_hsame}
\leanvariant{BEDC.BaseReflection.GeneratedSameSig\_from\_witnesses}
\leanvariant{BEDC.BaseReflection.GeneratedSameSig\_left\_witness}
\leanvariant{BEDC.BaseReflection.GeneratedSameSig\_right\_witness}
\leanvariant{BEDC.BaseReflection.GeneratedSameSig\_witnesses}
\leanvariant{BEDC.BaseReflection.GeneratedSameSig\_swap\_witnesses}
\leanvariant{BEDC.BaseReflection.GeneratedSameSig\_symm}
\leanvariant{BEDC.BaseReflection.GeneratedSameSig\_symm\_nonempty}
\leanvariant{BEDC.BaseReflection.GeneratedSameSig\_trans\_nonempty\_under\_determinacy}
\leanvariant{BEDC.BaseReflection.GeneratedSameSig\_trans\_under\_determinacy}
\leanvariant{BEDC.BaseReflection.GeneratedSameSig\_trans\_nonempty\_direct\_under\_determinacy}
\leanvariant{BEDC.BaseReflection.GeneratedSameSig\_chain\_witnesses\_under\_determinacy}
\leanvariant{BEDC.BaseReflection.GeneratedSameSig\_trans\_witness\_object\_under\_determinacy}

\begin{definition}[Base exact Globalize structure]
\label{def:exact-globalize-base}
For a bundle \(\Pi\) and a concrete domain policy \(D\), the proposition
\[
  \ExactGlobalizeBase(\Pi,D)
\]
is the conjunction of the following four fields.
\begin{enumerate}
  \item \textbf{Coverage.}
  \[
    \forall h,\; \InDom(D,h)\Rightarrow \exists p,\;\InGapSig(\Pi,D,p,h).
  \]
  \item \textbf{Base soundness.}
  \[
  \begin{aligned}
    \forall h,k,p,q,\;&\InGapSig(\Pi,D,p,h)\wedge\InGapSig(\Pi,D,q,k)\wedge\GeneratedSameSig(\Pi,h,k)\;\Rightarrow\\
    &\psame_{\Pi}^{\mathrm{base}}(p,q).
  \end{aligned}
  \]
  \item \textbf{Base completeness.}
  \[
  \begin{aligned}
    \forall h,k,p,q,\;&\InGapSig(\Pi,D,p,h)\wedge\InGapSig(\Pi,D,q,k)\wedge\psame_{\Pi}^{\mathrm{base}}(p,q)\;\Rightarrow\\
    &\GeneratedSameSig(\Pi,h,k).
  \end{aligned}
  \]
  \item \textbf{Classification iff.}
  \[
  \begin{aligned}
    \forall h,k,p,q,\;&\InGapSig(\Pi,D,p,h)\wedge\InGapSig(\Pi,D,q,k)\;\Rightarrow\\
    &\bigl(\psame_{\Pi}^{\mathrm{base}}(p,q)\Longleftrightarrow \GeneratedSameSig(\Pi,h,k)\bigr).
  \end{aligned}
  \]
\end{enumerate}
The fourth field may be implemented as a derived theorem from soundness and completeness; it is recorded here because it is the public export shape.
\end{definition}
\leanchecked{BEDC.BaseReflection.ExactGlobalizeBase\_iff\_fields}
\leanvariant{BEDC.BaseReflection.ExactGlobalizeBase\_soundness}
\leanvariant{BEDC.BaseReflection.ExactGlobalizeBase\_soundness\_nonempty}
\leanvariant{BEDC.BaseReflection.ExactGlobalizeBase\_completeness}
\leanvariant{BEDC.BaseReflection.ExactGlobalizeBase\_sound\_complete}
\leanvariant{BEDC.BaseReflection.ExactGlobalizeBase\_public\_export\_shape}
\leanvariant{BEDC.BaseReflection.ExactGlobalizeBase\_coverage\_and\_classification}

\begin{proposition}[Classification is derivable from the two directions]
\label{prop:iff-derived}
If coverage, base soundness, and base completeness hold, then for all \(h,k,p,q\) satisfying the two gap premises,
\[
  \psame_{\Pi}^{\mathrm{base}}(p,q)\Longleftrightarrow \GeneratedSameSig(\Pi,h,k).
\]
\end{proposition}
\leanchecked{BEDC.BaseReflection.ExactGlobalizeBase\_classify\_iff}
\leanvariant{BEDC.BaseReflection.ExactGlobalizeBase\_classify\_directions}
\leanvariant{BEDC.BaseReflection.ExactGlobalizeBase\_classify\_iff\_from\_directions}

\begin{proof}
The forward implication is base completeness. The reverse implication is base soundness. Coverage is not needed for this local equivalence, but it remains part of the global exactness structure because exact Globalize must also classify every admitted history, not only pairs already carrying visible packages.
\end{proof}

\begin{protocol}[Export replacement for prose exactness]
\label{prot:export-replacement}
Future modules should cite \(\ExactGlobalizeBase(\Pi,D)\) rather than a prose phrase such as ``visible packages classify histories exactly''. If a theorem needs only the local iff, it should cite the classification field. If it needs existence of a visible package for every admitted history, it must also cite coverage.
\end{protocol}

\section{Exact Globalize Base Contract}
\label{sec:po-exact-globalize-base-contract}

This chapter states the theorem replacing the prose exactness claim. The content is intentionally narrow: it is the exactness of the concrete signature Globalize instance at the base package relation.

\begin{theorem}[Exact Globalize base target]
\label{thm:exact-globalize-base-target}
\label{thm:globalize-exact-base}
Assume:
\begin{enumerate}
  \item \(\DomainPolicy(D)\);
  \item \(\AskPol(\Pi,D)\);
  \item signature generation and token introduction for generated signatures;
  \item \(\TokUnique(\Pi)\), or a checked canonical-constructor theorem implying it;
  \item the active base package relation \(\psame_{\Pi}^{\mathrm{base}}\);
  \item the policy-supported gap relation \(\InGapSig(\Pi,D,p,h)\);
  \item the base reflection theorem \texttt{PackageReflection.base}.
\end{enumerate}
Then
\[
  \ExactGlobalizeBase(\Pi,D).
\]
\end{theorem}
\leanchecked{BEDC.BaseReflection.ExactGlobalizeBase\_from\_fields}
\leanvariant{BEDC.BaseReflection.ExactGlobalizeBase\_fields}
\leanvariant{BEDC.BaseReflection.ExactGlobalizeBase\_coverage\_sound\_complete}
\leanvariant{BEDC.BaseReflection.ExactGlobalizeBase\_from\_fields\_classify\_iff}
\leanvariant{BEDC.BaseReflection.globalize\_exact\_base}
\leanvariant{BEDC.BaseReflection.Globalize.exactBase}
\leanvariant{BEDC.BaseReflection.ExactGlobalizeBase\_export\_from\_fields}
\leanvariant{BEDC.BaseReflection.no\_scaffold\_laundering\_fields\_and\_classifier}

\begin{proof}
Coverage is the policy-supported gap coverage theorem from this proof-obligation spine, with internal domain transport supplied by \(\DomainPolicy(D)\). Base soundness follows from generated signature sameness plus the single constructor of \(\psame_{\Pi}^{\mathrm{base}}\). Base completeness follows by applying base package reflection to the visible packages and then unpacking \(\GeneratedSameSig\). The local classification equivalence is Proposition~\ref{prop:iff-derived}.
\end{proof}

\begin{corollary}[No closure export]
\label{cor:no-closure-export}
Theorem~\ref{thm:exact-globalize-base-target} exports exactness only at \(\psame_{\Pi}^{\mathrm{base}}\). It does not export exactness at \(\psame_{\Pi}^{\mathrm{eq}}\).
\end{corollary}

\begin{proof}
The statement concludes \(\ExactGlobalizeBase(\Pi,D)\), whose relation field is \(\psame_{\Pi}^{\mathrm{base}}\) by Definition~\ref{def:exact-globalize-base}. The optional equivalence closure has the local reflection theorem of Proposition~\ref{prop:conditional-closure-reflection-schema}; that theorem does not replace the relation field of \(\ExactGlobalizeBase(\Pi,D)\).
\end{proof}
\leanchecked{BEDC.BaseReflection.ExactGlobalizeBase\_relation\_is\_base}
\leanvariant{BEDC.BaseReflection.ExactGlobalizeBase\_exports\_base\_relation}
\leanvariant{BEDC.BaseReflection.ExactGlobalizeBase\_no\_closure\_export}
\leanvariant{BEDC.BaseReflection.ExactGlobalizeBase\_no\_closure\_export\_directions}
\leanvariant{BEDC.BaseReflection.ExactGlobalizeBase\_no\_eqClosure\_export}

\begin{protocol}[Acceptance gate for a checked report]
\label{prot:checked-acceptance-gate}
This target may be upgraded from paper-level to checked only if the report includes all of the following:
\begin{enumerate}
  \item a proof-assistant build command and version;
  \item the public theorem name \texttt{Globalize.exactBase};
  \item no \texttt{sorry}, \texttt{admit}, or unchecked postulate in the proof of \texttt{Globalize.exactBase};
  \item no public theorem that replaces \(\hsame\) by host equality of histories;
  \item no public theorem that replaces \(\TokIntro\) by a host function without a relational wrapper;
  \item a list of every remaining axiom, postulate, or imported theorem used by the checked artifact.
\end{enumerate}
\end{protocol}
