\section{Exact Concrete Signature Globalize Under Explicit Policies}
\label{sec:po-exact-concrete-globalize-policy}

The concrete signature Globalize instance becomes exact only when the checked concrete inputs are visible:
$$
  \AskPol(\Pi,D)
  +
  \PkgTokPol(\Pi)
  +
  \forall h,s,\Delta,\;
  \InDom(D,h)\wedge\Sig(\Pi,h,s,\Delta)
  \Longrightarrow
  \exists p,\;\TokIntro(\Pi,s,p).
$$
This chapter records the exactness theorem in its non-overclaiming form. Domain transport is the separate policy recorded in Theorem~\ref{thm:domain-transport}.

The predicate $\InGapSig(\Pi,D,p,h)$ is Definition~\ref{def:internalized-signature-gap}.

\begin{theorem}[Concrete gap coverage]
\label{thm:gap-coverage-globalize}
Assume $\AskPol(\Pi,D)$ and the dependent generated-signature token condition: whenever $\InDom(D,h)$ and $\Sig(\Pi,h,s,\Delta)$, there exists a token $p$ with $\TokIntro(\Pi,s,p)$. If $\InDom(D,h)$, then there exists a token $p$ such that
$$
  \InGapSig(\Pi,D,p,h).
$$
\end{theorem}
\leanchecked{BEDC.FKernel.Gap.gap\_coverage\_globalize}

\begin{proof}
By the totality field of $\AskPol(\Pi,D)$, the admitted history $h$ generates some signature $s$ with evidence $\Delta$. The dependent token condition applied to $\InDom(D,h)$ and $\Sig(\Pi,h,s,\Delta)$ supplies a token $p$ with $\TokIntro(\Pi,s,p)$. The three displayed judgments witness $\InGapSig(\Pi,D,p,h)$.
\end{proof}

\begin{theorem}[Concrete gap separation]
\label{thm:concrete-gap-separation}
\label{thm:gap-separation-globalize}
Let \(\psame_{\Pi}^{\mathrm{sig}}\) be the concrete package-sameness relation generated by its signature-sameness constructor. Assume \(\AskPol(\Pi,D)\). If
$$
  \InGapSig(\Pi,D,p,h)
  \quad\text{and}\quad
  \InGapSig(\Pi,D,q,h),
$$
then
$$
  \psame_{\Pi}^{\mathrm{sig}}(p,q).
$$
\end{theorem}
\leanchecked{BEDC.FKernel.Gap.gap\_separation\_globalize}
\leanvariant{BEDC.FKernel.Gap.gap\_separation\_with\_domain\_pair}
\leanvariant{BEDC.FKernel.Gap.signature\_gap\_separation\_from\_witnesses}
\leanvariant{BEDC.FKernel.Gap.gap\_separation\_same\_signature\_witness\_pair}

\begin{proof}
The two gap premises yield the admitted-source witness $\InDom(D,h)$ and signatures $s,t$ with
$$
  \Sig(\Pi,h,s,\Delta),\quad
  \Sig(\Pi,h,t,\Theta),
$$
and token-introduction judgments
$$
  \TokIntro(\Pi,s,p),\quad
  \TokIntro(\Pi,t,q).
$$
Theorem~\ref{thm:signature-determinacy-under-asking-policy}, applied to $\AskPol(\Pi,D)$, the admitted-source witness, and the two generated signatures, gives $\hsame(s,t)$. The package-sameness constructor then gives $\psame_{\Pi}^{\mathrm{sig}}(p,q)$.
\end{proof}

\begin{theorem}[Concrete Globalize soundness]
\label{thm:globalize-sound}
Given a witness $G:\GeneratedSameSig(\Pi,h,k)$, if the left generated signature object of $G$ introduces token $p$ and the right generated signature object of $G$ introduces token $q$, then $\psame_{\Pi}^{\mathrm{base}}(p,q)$.
\end{theorem}
\leanchecked{BEDC.BaseReflection.GeneratedSameSig\_psameBase}

\begin{proof}
The witness $G$ carries the history-sameness component between its generated signature objects. The base package-sameness constructor applied to the two token-introduction premises and that component gives $\psame_{\Pi}^{\mathrm{base}}(p,q)$.
\end{proof}

\begin{theorem}[Concrete Globalize completeness]
\label{thm:globalize-complete}
Assume $\mathsf{TokUnique}(\Pi)$. Given generation judgments $\mathsf{SigGen}(\Pi,h,x,\Delta)$ and $\mathsf{SigGen}(\Pi,k,y,\Theta)$, token-introduction judgments $\TokIntro(\Pi,x,p)$ and $\TokIntro(\Pi,y,q)$, and $\psame_{\Pi}^{\mathrm{base}}(p,q)$, there is an element of $\mathsf{Nonempty}(\GeneratedSameSig(\Pi,h,k))$.
\end{theorem}
\leanchecked{BEDC.BaseReflection.PsameBase\_to\_GeneratedSameSig\_under\_tok\_unique}

\begin{proof}
Token-uniqueness reflection converts the base package-sameness premise, together with the two token-introduction judgments, into history-sameness of $x$ and $y$. Combining that reflected sameness with the two generation judgments fills the fields of a $\GeneratedSameSig(\Pi,h,k)$ witness.
\end{proof}

\section{Exactness Export Contract}
\label{sec:po-exactness-export-contract-design}

This chapter records the three-clause gap export used for active development. It is intentionally narrower than a full package algebra theorem.

\begin{theorem}[Local signature classification]
\label{thm:internalized-globalize-classifies-by-signatures}
Assume $\AskPol(\Pi,D)$, $\PkgTokPol(\Pi)$,
$$
  \InGapSig(\Pi,D,p,h)
  \quad\text{and}\quad
  \InGapSig(\Pi,D,q,k).
$$
Then package-sameness is classified by generated signatures:
$$
  \psame(p,q)
  \Longleftrightarrow
  \exists s,t,\Delta,\Theta,\;
  \Sig(\Pi,h,s,\Delta)
  \;\text{and}\;
  \Sig(\Pi,k,t,\Theta)
  \;\text{and}\;
  \hsame(s,t).
$$
\end{theorem}
\leanchecked{BEDC.FKernel.Gap.internalized\_globalize\_classifies\_by\_signatures}

\begin{proof}
The forward direction is the concrete completeness clause applied to the two gap memberships and the package-sameness premise. For the reverse direction, unpack the two $\InGapSig$ premises. The $p$-membership supplies a domain witness for $h$, a generated signature $u$, and the token-introduction witness for $p$; the $q$-membership supplies a domain witness for $k$, a generated signature $v$, and the token-introduction witness for $q$. Separately unpack the displayed existential into signatures $s$ from $h$ and $t$ from $k$, together with $\hsame(s,t)$. Asking-policy determinacy at $h$ gives $\hsame(s,u)$, and determinacy at $k$ gives $\hsame(t,v)$. By symmetry and transitivity, these three history-sameness facts compose to $\hsame(u,v)$. The package-token policy soundness field then consumes the two token-introduction witnesses and $\hsame(u,v)$ to conclude $\psame(p,q)$.
\end{proof}

\begin{theorem}[Exact Globalize base soundness export]
\label{thm:exact-globalize-base-soundness-exports-base}
Assume $\ExactGlobalizeBase(\Pi,D)$. If
$$
\InGapSig(\Pi,D,p,h),
\qquad
\InGapSig(\Pi,D,q,k),
\qquad
\GeneratedSameSig(\Pi,h,k),
$$
then $\psame_{\Pi}^{\mathrm{base}}(p,q)$.
\end{theorem}
\leanchecked{BEDC.BaseReflection.ExactGlobalizeBase\_soundness\_exports\_base}
\begin{proof}
This is the soundness projection of $\ExactGlobalizeBase(\Pi,D)$ applied to the two gap memberships and the generated-sameness witness.
\end{proof}

\begin{definition}[Not-exported boundary predicate]
\label{def:not-exported}
The predicate $\mathsf{NotExported}(\Pi,D,E)$ records the boundary attached to an exact base Globalize witness. It holds exactly when, for all histories $h,k$ and packages $p,q$, the two gap memberships
$$
\InGapSig(\Pi,D,p,h),
\qquad
\InGapSig(\Pi,D,q,k)
$$
bound the exported statement to
$$
\psame_{\Pi}^{\mathrm{base}}(p,q)
\Longleftrightarrow
\mathsf{Nonempty}(\GeneratedSameSig(\Pi,h,k)).
$$
\end{definition}
\leandef{BEDC.BaseReflection.NotExported}

\begin{proposition}[What is intentionally not exported]
\label{prop:not-exported}
The following statements are not exported:
\begin{enumerate}
  \item \(\psame_{\Pi}^{\mathrm{eq}}\)-exactness;
  \item package-sameness transitivity without token-uniqueness hypotheses as a constructor-level theorem for \(\psame_{\Pi}^{\mathrm{base}}\);
  \item equivalence-closure exactness as a Globalize export;
  \item $\AddUp$ certificate data as a clause of the exact Globalize export;
  \item a full natural-number or addition construction beyond the licensed unary-continuation interface.
\end{enumerate}
\end{proposition}

\begin{proof}
Each item requires an exactness predicate whose exported relation is not $\psame_{\Pi}^{\mathrm{base}}$, a derived-interface certificate carried as separate naming-certificate data, or a derived-object construction that is not a field of the exact Globalize predicate.
\end{proof}
\leanchecked{BEDC.BaseReflection.NotExported\_scope\_boundary}

\begin{protocol}[Export discipline]
\label{prot:export-discipline}
Modules cite Definition~\ref{def:exact-globalize-base} for the exactness predicate and Theorem~\ref{thm:exact-globalize-base-target} for its field constructor. If a module needs $\psame_{\Pi}^{\mathrm{eq}}$-exactness, it must state an exactness predicate whose exported relation is $\psame_{\Pi}^{\mathrm{eq}}$, or explicitly downgrade its result from exact reflection to closure-level compatibility.
\end{protocol}

\begin{definition}[Generated signature sameness]
\label{def:generated-same-sig}
\label{thm:generated-same-sig}
For a generated probe bundle $\Pi$, $\GeneratedSameSig(\Pi,h,k)$ is a witness object. Its fields are signatures $s,t$, evidence objects $\Delta,\Theta$, generation judgments $\Sig(\Pi,h,s,\Delta)$ and $\Sig(\Pi,k,t,\Theta)$, and a sameness witness $\hsame(s,t)$.

The existential reading is the inhabitation proposition for that object:
$$
\mathsf{Nonempty}(\GeneratedSameSig(\Pi,h,k))\Longleftrightarrow
\exists s,t,\Delta,\Theta,\;\Sig(\Pi,h,s,\Delta)\wedge\Sig(\Pi,k,t,\Theta)\wedge\hsame(s,t).
$$
Thus the checker-facing data corresponding to ``the two admitted histories generate the same visible signature'' is the witness object itself, while classification statements use its inhabitation.
\end{definition}

\begin{definition}[Base exact Globalize structure]
\label{def:exact-globalize-base}
For a bundle \(\Pi\) and a concrete domain policy \(D\), the proposition
$$
  \ExactGlobalizeBase(\Pi,D)
$$
is the conjunction of the following three structural fields.
\begin{enumerate}
  \item \textbf{Coverage.}
  $$
    \forall h,\; \InDom(D,h)\Rightarrow \exists p,\;\InGapSig(\Pi,D,p,h).
  $$
  \item \textbf{Base soundness.}
  $$
    \forall h,k,p,q,\; \InGapSig(\Pi,D,p,h)\Rightarrow\InGapSig(\Pi,D,q,k)\Rightarrow\GeneratedSameSig(\Pi,h,k)\Rightarrow \psame_{\Pi}^{\mathrm{base}}(p,q).
  $$
  \item \textbf{Base completeness.}
  $$
    \forall h,k,p,q,\; \InGapSig(\Pi,D,p,h)\Rightarrow\InGapSig(\Pi,D,q,k)\Rightarrow\psame_{\Pi}^{\mathrm{base}}(p,q)\Rightarrow \mathsf{Nonempty}(\GeneratedSameSig(\Pi,h,k)).
  $$
\end{enumerate}
The local classification equivalence is derived from soundness and completeness.
\end{definition}

\begin{proposition}[Classification is derivable from the two directions]
\label{prop:iff-derived}
If base soundness and base completeness hold, then for all $h,k,p,q$ satisfying
$$
  \InGapSig(\Pi,D,p,h)
  \quad\text{and}\quad
  \InGapSig(\Pi,D,q,k),
$$
one has
$$
  \psame_{\Pi}^{\mathrm{base}}(p,q)\Longleftrightarrow \mathsf{Nonempty}(\GeneratedSameSig(\Pi,h,k)).
$$
\end{proposition}
\leanchecked{BEDC.BaseReflection.ExactGlobalizeBase\_classify\_iff\_from\_directions}
\leanvariant{BEDC.BaseReflection.ExactGlobalizeBase\_classify\_directions}

\begin{proof}
The forward implication is base completeness. For the reverse implication, unpack the inhabited witness proposition and apply base soundness to the resulting $\GeneratedSameSig(\Pi,h,k)$ object. Coverage is not needed for this local equivalence, but it remains part of the global exactness structure because exact Globalize must also classify every admitted history, not only pairs already carrying visible packages.
\end{proof}

\begin{corollary}[Exact Globalize projections and classification]
\label{cor:exact-globalize-projections-and-classification}
From \(\ExactGlobalizeBase(\Pi,D)\) one obtains coverage, base soundness, base completeness, and the derived local classification equivalence
$$
  \InGapSig(\Pi,D,p,h)\wedge\InGapSig(\Pi,D,q,k)
  \quad\Rightarrow\quad
  \bigl(\psame_{\Pi}^{\mathrm{base}}(p,q)\Longleftrightarrow \mathsf{Nonempty}(\GeneratedSameSig(\Pi,h,k))\bigr).
$$
\end{corollary}
\leanchecked{BEDC.BaseReflection.ExactGlobalizeBase\_coverage\_soundness\_completeness\_and\_classify}

\begin{proof}
The first three clauses are the projections of \(\ExactGlobalizeBase(\Pi,D)\). The displayed equivalence is Proposition~\ref{prop:iff-derived}.
\end{proof}

\begin{protocol}[Exactness export discipline]
\label{prot:exactness-export-discipline}
Global existence cites Definition~\ref{def:exact-globalize-base}. Pairwise classification cites Proposition~\ref{prop:iff-derived} or Corollary~\ref{cor:exact-globalize-projections-and-classification} under displayed $\InGapSig$ witnesses. Coverage is used only when the statement must produce visible package tokens for admitted histories.
\end{protocol}

\section{Exact Globalize Base Contract}
\label{sec:po-exact-globalize-base-contract}

The theorem packages coverage, base soundness, and base completeness at the base package relation into the concrete signature Globalize exactness predicate.

\begin{theorem}[Exact Globalize base target]
\label{thm:exact-globalize-base-target}
\label{thm:globalize-exact-base}
Assume the following three field hypotheses:
\begin{enumerate}
  \item coverage:
  $$
    \forall h,\; \InDom(D,h)\Rightarrow \exists p,\;\InGapSig(\Pi,D,p,h);
  $$
  \item soundness:
  $$
    \forall h,k,p,q,\; \InGapSig(\Pi,D,p,h)\Rightarrow\InGapSig(\Pi,D,q,k)\Rightarrow\GeneratedSameSig(\Pi,h,k)\Rightarrow \psame_{\Pi}^{\mathrm{base}}(p,q);
  $$
  \item completeness:
  $$
    \forall h,k,p,q,\; \InGapSig(\Pi,D,p,h)\Rightarrow\InGapSig(\Pi,D,q,k)\Rightarrow\psame_{\Pi}^{\mathrm{base}}(p,q)\Rightarrow \mathsf{Nonempty}(\GeneratedSameSig(\Pi,h,k)).
  $$
\end{enumerate}
Then
$$
  \ExactGlobalizeBase(\Pi,D).
$$
\end{theorem}

\begin{corollary}[No closure export]
\label{cor:no-closure-export}
