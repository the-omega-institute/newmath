\label{ch:proof-obligations-psame-design}

\section{Concrete Package-Sameness Design Contract}
\label{ch:v154-psame-design-contract}

The v1.5.3 reflection theorem is correct for the one-constructor concrete package-sameness relation. It must not be silently transferred to an equivalence closure. This chapter fixes the active design.

\begin{definition}[Base concrete package-sameness]
\label{def:v154-base-psame}
For a fixed generated probe bundle \(\Pi\), the active concrete signature-token package-sameness relation is written
\[
  \psame_{\Pi}^{\mathrm{base}}(p,q).
\]
It is generated by exactly one constructor:
\[
  \TokIntro(\Pi,s,p),\quad
  \TokIntro(\Pi,t,q),\quad
  \hsame(s,t)
  \quad\Longrightarrow\quad
  \psame_{\Pi}^{\mathrm{base}}(p,q).
\]
The active notation \(\psame_{\Pi}^{\mathrm{sig}}\) is identified with this base relation:
\[
  \psame_{\Pi}^{\mathrm{sig}} := \psame_{\Pi}^{\mathrm{base}}.
\]
\end{definition}

\begin{principle}[No implicit equivalence closure]
\label{prin:v154-no-implicit-closure}
No theorem in the concrete signature Globalize instance may replace \(\psame_{\Pi}^{\mathrm{base}}\) by an equivalence closure without explicitly adding a closure-reflection theorem.
\end{principle}

\begin{definition}[Optional equivalence closure]
\label{def:v154-psame-eq}
If a future package-algebra layer needs an equivalence relation, it must introduce a separate relation
\[
  \psame_{\Pi}^{\mathrm{eq}}(p,q)
  := \EqCl(\psame_{\Pi}^{\mathrm{base}})(p,q).
\]
This relation is not the active relation for v1.5.4 concrete Globalize exactness.
\end{definition}

\begin{definition}[Closure-reflection obligation]
\label{def:v154-closure-reflection}
The optional equivalence-closure relation is reflection-safe only if the following additional theorem is proved:
\[
\begin{aligned}
  &\TokIntro(\Pi,s,p),\quad
    \TokIntro(\Pi,t,q),\quad
    \psame_{\Pi}^{\mathrm{eq}}(p,q)\\
  &\hspace{3.0cm}\Longrightarrow\quad \hsame(s,t).
\end{aligned}
\]
This obligation is named \(\ClosureReflect(\Pi)\).
\end{definition}

\begin{theorem}[Base reflection remains the active theorem]
\label{thm:v154-base-reflection-active}
Assume \(\TokUnique(\Pi)\). If
\[
  \TokIntro(\Pi,s,p),\quad
  \TokIntro(\Pi,t,q),\quad
  \psame_{\Pi}^{\mathrm{base}}(p,q),
\]
then
\[
  \hsame(s,t).
\]
\end{theorem}

\begin{proof}
This is Theorem~\ref{thm:v153-package-reflection-token-unique} with the notation of Definition~\ref{def:v154-base-psame}. The only constructor of \(\psame_{\Pi}^{\mathrm{base}}\) exposes introducing signatures \(s_0,t_0\) with \(\hsame(s_0,t_0)\). Token-introduction uniqueness compares \(s\) with \(s_0\) and \(t\) with \(t_0\); symmetry and transitivity of \(\hsame\) finish the proof.
\end{proof}

\begin{corollary}[Equivalence closure is not exported as exact]
\label{cor:v154-eq-not-exported}
The exact concrete Globalize theorem of v1.5.4 exports \(\psame_{\Pi}^{\mathrm{base}}\), not \(\psame_{\Pi}^{\mathrm{eq}}\). Exporting \(\psame_{\Pi}^{\mathrm{eq}}\) as exact is deferred until \(\ClosureReflect(\Pi)\) is checked.
\end{corollary}

\begin{proof}
By Principle~\ref{prin:v154-no-implicit-closure}, equivalence closure is a separate construction. Definition~\ref{def:v154-closure-reflection} records the extra theorem required for reflection through that closure. Since v1.5.4 records no checked proof of that theorem, only the base relation is exported as exact.
\end{proof}

\section{Base Reflection Schema}
\label{ch:v155-base-reflection-schema}

This chapter isolates the smallest constructor-inversion chain needed for base package reflection. It is the first proof-check target after v1.5.5.

\begin{definition}[Base package witness data]
\label{def:v155-psame-base-data}
For fixed \(\Pi,p,q\), define the witness predicate
\[
\PBaseData(\Pi,p,q) :\Longleftrightarrow
\exists s_0,t_0,\;\TokIntro(\Pi,s_0,p)\wedge\TokIntro(\Pi,t_0,q)\wedge\hsame(s_0,t_0).
\]
\end{definition}

\begin{lemma}[Base constructor]
\label{lem:v155-psame-base-constructor}
If
\[
  \TokIntro(\Pi,s,p),\quad \TokIntro(\Pi,t,q),\quad \hsame(s,t),
\]
then
\[
  \psame_{\Pi}^{\mathrm{base}}(p,q).
\]
\end{lemma}

\begin{proof}
This is the single constructor of Definition~\ref{def:v154-base-psame}. The important checker rule is that no second public constructor may be added without changing the versioned contract.
\end{proof}

\begin{lemma}[Base inversion]
\label{lem:v155-psame-base-inversion}
\label{thm:psame-base-inversion}
If \(\psame_{\Pi}^{\mathrm{base}}(p,q)\), then \(\PBaseData(\Pi,p,q)\).
\end{lemma}

\begin{proof}
Invert the one-constructor inductive proof of \(\psame_{\Pi}^{\mathrm{base}}(p,q)\). The inversion exposes introducing signatures \(s_0,t_0\) and a proof \(\hsame(s_0,t_0)\). This is exactly \(\PBaseData(\Pi,p,q)\).
\end{proof}

\begin{lemma}[Token replacement]
\label{lem:v155-token-replacement}
\label{thm:tok-unique-replacement}
Assume \(\TokUnique(\Pi)\). If
\[
  \TokIntro(\Pi,s,p),\quad \TokIntro(\Pi,t,p),
\]
then \(\hsame(s,t)\).
\end{lemma}

\begin{proof}
This is the token-uniqueness field applied at the same visible package token \(p\). In a canonical-constructor implementation it may instead be obtained by the no-confusion theorem described in Definition~\ref{def:v154-canonical-mode}.
\end{proof}

\begin{theorem}[Base package reflection]
\label{thm:v155-base-package-reflection}
\label{thm:package-reflection-base}
Assume \(\TokUnique(\Pi)\). If
\[
  \TokIntro(\Pi,s,p),\quad \TokIntro(\Pi,t,q),\quad \psame_{\Pi}^{\mathrm{base}}(p,q),
\]
then
\[
  \hsame(s,t).
\]
\end{theorem}

\begin{proof}
By Lemma~\ref{lem:v155-psame-base-inversion}, choose witnesses \(s_0,t_0\) such that
\[
  \TokIntro(\Pi,s_0,p),\quad \TokIntro(\Pi,t_0,q),\quad \hsame(s_0,t_0).
\]
Token replacement at \(p\) gives \(\hsame(s,s_0)\). Token replacement at \(q\) gives \(\hsame(t,t_0)\), hence by symmetry \(\hsame(t_0,t)\). Transitivity of \(\hsame\) gives
\[
  \hsame(s,s_0),\quad \hsame(s_0,t_0),\quad \hsame(t_0,t)
  \quad\Longrightarrow\quad \hsame(s,t).
\]
\end{proof}

\begin{protocol}[Public theorem names]
\label{prot:v155-public-base-reflection-names}
The checker-facing theorem names for this spine are:
\begin{center}
\begin{tabular}{ll}
\toprule
\textbf{Paper object} & \textbf{Suggested checker name}\\
\midrule
Base constructor & \texttt{PsameBase.intro}\\
Base inversion & \texttt{PsameBase.inversion}\\
Token replacement & \texttt{TokUnique.tokenReplacement}\\
Base package reflection & \texttt{PackageReflection.base}\\
\bottomrule
\end{tabular}
\end{center}
\end{protocol}

\section{Optional Closure Reflection Target}
\label{ch:v155-closure-reflection-target}

The equivalence closure of the base package relation is useful for later package algebra, but it must not be confused with the active exact relation. This chapter records the optional proof target without exporting it.

\begin{definition}[Equivalence closure target]
\label{def:v155-psame-eq-target}
The optional relation remains
\[
  \psame_{\Pi}^{\mathrm{eq}}(p,q) := \EqCl(\psame_{\Pi}^{\mathrm{base}})(p,q).
\]
It is not part of \(\ExactGlobalizeBase(\Pi,D)\).
\end{definition}

\begin{protocol}[Closure-reflection proof route]
\label{prot:v155-closure-reflection-route}
A future theorem \texttt{PackageReflection.eqClosure} should prove \(\ClosureReflect(\Pi)\) by induction over \(\EqCl(\psame_{\Pi}^{\mathrm{base}})\). The intended cases are:
\begin{enumerate}
  \item reflexive case: use two introductions of the same token and \(\TokUnique(\Pi)\);
  \item base-edge case: use Theorem~\ref{thm:v155-base-package-reflection};
  \item symmetric case: use symmetry of \(\hsame\);
  \item transitive case: use transitivity of \(\hsame\).
\end{enumerate}
\end{protocol}

\begin{proposition}[Conditional closure-reflection schema]
\label{prop:v155-conditional-closure-reflection-schema}
Assume \(\hsame\) is an equivalence relation, assume \(\TokUnique(\Pi)\), and assume \(\psame_{\Pi}^{\mathrm{eq}}\) is implemented as the ordinary reflexive-symmetric-transitive closure of \(\psame_{\Pi}^{\mathrm{base}}\). Then the proof route of Protocol~\ref{prot:v155-closure-reflection-route} is sufficient to prove
\[
\begin{aligned}
  &\TokIntro(\Pi,s,p),\quad \TokIntro(\Pi,t,q),\quad \psame_{\Pi}^{\mathrm{eq}}(p,q)\\
  &\hspace{3cm}\Longrightarrow\quad \hsame(s,t).
\end{aligned}
\]
\end{proposition}

\begin{proof}
Induct on the derivation of \(\psame_{\Pi}^{\mathrm{eq}}(p,q)\). The reflexive case compares two introductions of one token by token uniqueness. The base-edge case is Theorem~\ref{thm:v155-base-package-reflection}. The symmetric case reverses the induction result by symmetry of \(\hsame\). The transitive case composes the two induction results by transitivity of \(\hsame\).
\end{proof}

\begin{corollary}[Still not active]
\label{cor:v155-closure-still-not-active}
Proposition~\ref{prop:v155-conditional-closure-reflection-schema} is a paper-level proof schema. It does not activate \(\psame_{\Pi}^{\mathrm{eq}}\)-exactness unless a future checked report records the actual theorem and imports it into the export contract.
\end{corollary}

\begin{proof}
The active export discipline is Protocol~\ref{prot:v154-export-discipline} together with Corollary~\ref{cor:v155-no-closure-export}. A proof schema is not a checked artifact.
\end{proof}
