\chapter{Unary Right-Shift: Micro-Lemmas for the Additive License}
\label{ch:v153-unary-shift-micro-lemmas}

The v1.5.2 theorem \(\UnaryShift\) is the remaining proof obligation needed before the name \(\AddUp\) can replace the conservative name \(\UnaryContMonoidUp\). This chapter decomposes that obligation into smaller lemmas suitable for a local verification sprint.

\begin{lemma}[\(\Eone\)-congruence]
\label{lem:v153-eone-cong}
If \(\hsame(h,k)\), then
\[
  \hsame(\Eone(h),\Eone(k)).
\]
\end{lemma}

\begin{proof}
This is the \(\Eone\)-constructor rule for \(\hsame\).
\end{proof}

\begin{lemma}[Unary right-shift base]
\label{lem:v153-unary-shift-base}
If \(\ContR(\emp,\Eone(h),r)\), then there exists a history \(v\) such that
\[
  \ContR(\emp,h,v)
  \quad\text{and}\quad
  \hsame(r,\Eone(v)).
\]
\end{lemma}

\begin{proof}
By the left-unit law for \(\ContR\), the first premise yields \(\hsame(r,\Eone(h))\). Again by the left-unit law, \(\ContR(\emp,h,h)\). Take \(v=h\).
\end{proof}

\begin{lemma}[Unary right-shift step]
\label{lem:v153-unary-shift-step}
Assume the right-shift statement holds for a unary history \(k_0\). If
\[
  \Unary(k_0),
  \qquad
  \ContR(\Eone(k_0),\Eone(h),r'),
\]
then there exists a history \(v\) such that
\[
  \ContR(\Eone(k_0),h,v)
  \quad\text{and}\quad
  \hsame(r',\Eone(v)).
\]
\end{lemma}

\begin{proof}
Invert the continuation premise. The only active constructor for a left source headed by \(\Eone\) exposes a history \(u\) with
\[
  \ContR(k_0,\Eone(h),u)
  \quad\text{and}\quad
  \hsame(r',\Eone(u)).
\]
By the induction hypothesis applied to \(k_0\), there exists \(v_0\) such that
\[
  \ContR(k_0,h,v_0)
  \quad\text{and}\quad
  \hsame(u,\Eone(v_0)).
\]
The continuation step gives
\[
  \ContR(\Eone(k_0),h,\Eone(v_0)).
\]
Let \(v=\Eone(v_0)\). By Lemma~\ref{lem:v153-eone-cong}, \(\hsame(u,v)\) gives \(\hsame(\Eone(u),\Eone(v))\). Together with \(\hsame(r',\Eone(u))\) and transitivity, this yields \(\hsame(r',\Eone(v))\).
\end{proof}

\begin{theorem}[Unary right-shift]
\label{thm:v153-unary-shift}
Assume \(\Unary(k)\) and \(\ContR(k,\Eone(h),r')\). Then there exists a history \(v\) such that
\[
  \ContR(k,h,v)
  \quad\text{and}\quad
  \hsame(r',\Eone(v)).
\]
\end{theorem}

\begin{proof}
Induct on the derivation of \(\Unary(k)\). The empty case is Lemma~\ref{lem:v153-unary-shift-base}. The step case is Lemma~\ref{lem:v153-unary-shift-step} applied to the induction hypothesis.
\end{proof}

\begin{theorem}[Unary continuation commutativity]
\label{thm:v153-unary-commutativity}
If
\[
  \Unary(h),\quad \Unary(k),\quad
  \ContR(h,k,r),\quad \ContR(k,h,r'),
\]
then
\[
  \hsame(r,r').
\]
\end{theorem}

\begin{proof}
Induct on the derivation of \(\Unary(h)\). The empty case follows from the continuation unit laws. In the step case, write the left source as \(\Eone(h_0)\). Invert \(\ContR(\Eone(h_0),k,r)\), obtaining a history \(u\) such that \(\ContR(h_0,k,u)\) and \(\hsame(r,\Eone(u))\). Apply Theorem~\ref{thm:v153-unary-shift} to \(\ContR(k,\Eone(h_0),r')\), obtaining \(v\) with \(\ContR(k,h_0,v)\) and \(\hsame(r',\Eone(v))\). The induction hypothesis gives \(\hsame(u,v)\). Lemma~\ref{lem:v153-eone-cong} and transitivity of \(\hsame\) yield \(\hsame(r,r')\).
\end{proof}

\begin{corollary}[Activation of the additive name]
\label{cor:v153-add-activation}
Once Theorem~\ref{thm:v153-unary-commutativity} is included in the stability field of the additive naming certificate, the interface previously named \(\UnaryContMonoidUp\) is licensed to use the stronger name \(\AddUp\).
\end{corollary}

\begin{proof}
The monoid-like fields are unit, associativity, and determinacy of unary continuation. Theorem~\ref{thm:v153-unary-commutativity} supplies the commutativity field required by the additive name.
\end{proof}
