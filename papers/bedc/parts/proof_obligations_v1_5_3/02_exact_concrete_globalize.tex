\chapter{Exact Concrete Signature Globalize Under Explicit Policies}
\label{ch:v153-exact-concrete-globalize}

The concrete signature Globalize instance of v1.4 and v1.5 becomes exact only when all four policy layers are visible:
\[
  \DomainPolicy(D)
  +
  \AskPol(\Pi,D)
  +
  \mathsf{TokUnique}(\Pi)
  +
  \GapPol^{\mathrm{sig}}(\Pi,D).
\]
This chapter records the exactness theorem in its non-overclaiming form.

\begin{definition}[Policy-supported signature gap]
For histories admitted by \(D\), define
\[
  \InGapSig(\Pi,D,p,h)
\]
to mean that there exist an answer history \(s\) and an evidence ledger \(\Delta\) such that
\[
  \InDom(D,h),
  \qquad
  \Sig(\Pi,h,s,\Delta),
  \qquad
  \TokIntro(\Pi,s,p).
\]
\end{definition}

\begin{theorem}[Concrete gap coverage]
\label{thm:v153-gap-coverage}
Assume \(\AskPol(\Pi,D)\) and that token introduction is available for every generated signature. If \(\InDom(D,h)\), then there exists a token \(p\) such that
\[
  \InGapSig(\Pi,D,p,h).
\]
\end{theorem}

\begin{proof}
By the totality field of \(\AskPol(\Pi,D)\), the history \(h\) generates some signature \(s\) with evidence \(\Delta\). Token introduction supplies a token \(p\) with \(\TokIntro(\Pi,s,p)\). The three displayed judgments witness \(\InGapSig(\Pi,D,p,h)\).
\end{proof}

\begin{theorem}[Concrete gap separation]
\label{thm:v153-gap-separation}
Assume \(\AskPol(\Pi,D)\), \(\mathsf{TokUnique}(\Pi)\), and the one-constructor concrete package-sameness relation. If
\[
  \InGapSig(\Pi,D,p,h)
  \quad\text{and}\quad
  \InGapSig(\Pi,D,q,h),
\]
then
\[
  \psame_{\Pi}^{\mathrm{sig}}(p,q).
\]
\end{theorem}

\begin{proof}
The two gap premises yield signatures \(s,t\) with
\[
  \Sig(\Pi,h,s,\Delta),\quad
  \Sig(\Pi,h,t,\Theta),
\]
and token-introduction judgments
\[
  \TokIntro(\Pi,s,p),\quad
  \TokIntro(\Pi,t,q).
\]
The determinacy field of \(\AskPol(\Pi,D)\) gives \(\hsame(s,t)\). The package-sameness constructor then gives \(\psame_{\Pi}^{\mathrm{sig}}(p,q)\).
\end{proof}

\begin{theorem}[Concrete Globalize soundness]
\label{thm:v153-globalize-sound}
Assume \(\AskPol(\Pi,D)\). If admitted histories \(h,k\) generate internally same signatures, then any tokens introduced from those signatures are package-same.
\end{theorem}

\begin{proof}
Let \(s,t\) be generated signatures for \(h,k\), with \(\hsame(s,t)\). If \(p,q\) are introduced from \(s,t\), then the single constructor of \(\psame_{\Pi}^{\mathrm{sig}}\) yields \(\psame_{\Pi}^{\mathrm{sig}}(p,q)\).
\end{proof}

\begin{theorem}[Concrete Globalize completeness]
\label{thm:v153-globalize-complete}
Assume \(\mathsf{TokUnique}(\Pi)\). If tokens introduced from generated signatures are package-same, then the generated signatures are internally history-same.
\end{theorem}

\begin{proof}
This is exactly Theorem~\ref{thm:v153-package-reflection-token-unique} applied to the token-introduction premises.
\end{proof}

\begin{corollary}[Exact concrete signature Globalize]
\label{cor:v153-exact-globalize}
Under \(\DomainPolicy(D)\), \(\AskPol(\Pi,D)\), token introduction, \(\mathsf{TokUnique}(\Pi)\), and the one-constructor concrete package-sameness policy, the visible packages of \(\GlobSig(\Pi,D)\) classify admitted histories exactly by generated signature-sameness, with \(\InGapSig\) as the source ledger.
\end{corollary}

\begin{proof}
Coverage is Theorem~\ref{thm:v153-gap-coverage}. Separation is Theorem~\ref{thm:v153-gap-separation}. Soundness and completeness are Theorems~\ref{thm:v153-globalize-sound} and~\ref{thm:v153-globalize-complete}. Domain transport is supplied by \(\DomainPolicy(D)\).
\end{proof}
