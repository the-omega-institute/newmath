\chapter{Token-Introduction Uniqueness and Package Reflection}
\label{ch:v153-token-intro-uniqueness}

Chapter~\ref{ch:v152-inductive-package-token-policy} made concrete package-sameness a one-constructor relation. That is necessary but not sufficient for the strongest reflection theorem. If the same visible token can be introduced by two unrelated signatures, then package-sameness of tokens does not determine the signatures uniquely. This chapter isolates the missing requirement.

\begin{definition}[Token-introduction uniqueness]
\label{def:v153-token-unique}
For a fixed generated probe bundle \(\Pi\), define the token-introduction uniqueness obligation \(\mathsf{TokUnique}(\Pi)\) as the following principle:
\[
  \TokIntro(\Pi,s,p),\quad
  \TokIntro(\Pi,t,p)
  \quad\Longrightarrow\quad
  \hsame(s,t).
\]
This is an internal uniqueness condition up to history-sameness. It is not host-level equality of tokens or signatures.
\end{definition}

\begin{remark}[Why this is needed]
From a derivation of \(\psame_{\Pi}^{\mathrm{sig}}(p,q)\), inversion yields some generated signatures \(s_0,t_0\) introducing \(p,q\), together with \(\hsame(s_0,t_0)\). If the displayed premises use \(s,t\) instead, one still needs to compare \(s\) with \(s_0\) and \(t\) with \(t_0\). This is exactly the role of \(\mathsf{TokUnique}(\Pi)\).
\end{remark}

\begin{lemma}[Token replacement]
\label{lem:v153-token-replacement}
Assume \(\mathsf{TokUnique}(\Pi)\). If
\[
  \TokIntro(\Pi,s,p),\quad
  \TokIntro(\Pi,s_0,p),
\]
then
\[
  \hsame(s,s_0).
\]
\end{lemma}

\begin{proof}
This is Definition~\ref{def:v153-token-unique} applied to the two token-introduction judgments for the same token \(p\).
\end{proof}

\begin{theorem}[Concrete package reflection with token uniqueness]
\label{thm:v153-package-reflection-token-unique}
Assume \(\mathsf{TokUnique}(\Pi)\). If
\[
  \TokIntro(\Pi,s,p),\quad
  \TokIntro(\Pi,t,q),\quad
  \psame_{\Pi}^{\mathrm{sig}}(p,q),
\]
then
\[
  \hsame(s,t).
\]
\end{theorem}

\begin{proof}
Invert the derivation of \(\psame_{\Pi}^{\mathrm{sig}}(p,q)\). Since the concrete signature-token package-sameness relation has only the constructor displayed in Chapter~\ref{ch:v152-inductive-package-token-policy}, the derivation supplies signatures \(s_0,t_0\) such that
\[
  \TokIntro(\Pi,s_0,p),\quad
  \TokIntro(\Pi,t_0,q),\quad
  \hsame(s_0,t_0).
\]
By Lemma~\ref{lem:v153-token-replacement}, the first token-introduction premise gives \(\hsame(s,s_0)\), and the second gives \(\hsame(t,t_0)\). Symmetry and transitivity of \(\hsame\) now yield \(\hsame(s,t)\).
\end{proof}

\begin{corollary}[Exact package-sameness under token uniqueness]
\label{cor:v153-exact-package-sameness}
Assume \(\mathsf{TokUnique}(\Pi)\). For introduced tokens,
\[
  \psame_{\Pi}^{\mathrm{sig}}(p,q)
\]
if and only if their introducing signatures are internally history-same.
\end{corollary}

\begin{proof}
The forward direction is Theorem~\ref{thm:v153-package-reflection-token-unique}. The reverse direction is the single constructor of \(\psame_{\Pi}^{\mathrm{sig}}\).
\end{proof}

\begin{corollary}[Updated PackageTokenPolicy]
\label{cor:v153-updated-package-token-policy}
The concrete signature-token package policy is exact when it includes token introduction, one-constructor package-sameness, and \(\mathsf{TokUnique}(\Pi)\).
\end{corollary}

\begin{proof}
Existence and soundness are supplied by token introduction and the package-sameness constructor. Reflection is Theorem~\ref{thm:v153-package-reflection-token-unique}. The no-extra-constructor clause is the closed-generation clause for \(\psame_{\Pi}^{\mathrm{sig}}\).
\end{proof}
