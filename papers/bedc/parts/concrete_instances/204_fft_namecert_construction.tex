\chapter{A Concrete Naming Certificate for $\FftUp$}
\label{ch:concrete-instances-fft-namecert}

The Fast-Fourier-transform interface $\FftUp$ packages the $O(n \log n)$ divide-and-conquer evaluation of the discrete Fourier transform on length-$n$ data when $n$ has small prime factors. It depends on $\NameCert_{\FourierUp}$ (\autoref{ch:concrete-instances-fourier-namecert}) and $\NameCert_{\ComplexUp}$ (\autoref{ch:concrete-instances-complex-namecert}).

\section{Carrier definition}
\label{sec:fft-carrier}

\begin{definition}[FFT BHist source packet]
\label{def:fft-bhist-source-packet}
An $\FftUp$ source packet is a finite $\mathsf{BHist}$ schedule carrying a
$\ComplexUp$ scalar row, a $\FourierUp$ transform row, a finite
$\mathsf{ProbeBundle}$ of butterfly stages, $\mathsf{InBundle}$ membership
witnesses for every scheduled stage, and a factorization ledger naming the
split of the transform length. The carrier accepts exactly those packets whose
complex scalar rows and Fourier rows are accepted by their dependency
certificates and whose stage rows transport through the same $\hsame$ and
$\Cont$ evidence.
\end{definition}

\begin{theorem}[FFT carrier butterfly obligation]
\label{thm:fft-carrier-butterfly-obligation}
Every accepted $\FftUp$ carrier row projects to a finite butterfly schedule:
each stage is a $\mathsf{BHist}$ row in the displayed
$\mathsf{ProbeBundle}$, each membership witness is an
$\mathsf{InBundle}$ row, and each butterfly continuation is carried by
$\Cont$ from the $\ComplexUp$ scalar dependency and the $\FourierUp$ transform
dependency. Conversely, those rows repack as the FFT carrier.
\end{theorem}
\leanchecked{BEDC.Derived.FftUp.FftBHistSourcePacket\_butterfly\_schedule\_obligation}
\leanchecked{BEDC.Derived.FftUp.FftBHistSourcePacket\_carrier\_butterfly\_obligation}
\begin{proof}
Project the $\ComplexUp$ and $\FourierUp$ dependency rows from the packet.
They supply the scalar and transform histories used by every scheduled stage.

Read the finite schedule through the $\mathsf{ProbeBundle}$ field. Each
butterfly row is certified by an $\mathsf{InBundle}$ witness, so the schedule
is finite at the BEDC surface and does not depend on a host array traversal.

For the converse direction, place the dependency rows, stage membership rows,
and butterfly continuations into the packet constructor. The common $\hsame$
and $\Cont$ evidence transports the stage rows and yields the carried
$\FftUp$ endpoint.
\end{proof}

\begin{theorem}[FFT factorization classifier stability obligation]
\label{thm:fft-factorization-classifier-stability-obligation}
The factorization classifier row of an $\FftUp$ packet is stable under
accepted replacement of the complex scalar, Fourier transform, and finite
butterfly schedule rows. If two packets compare by the displayed $\hsame$
evidence, then the split-length ledger, the stage membership witnesses, and
the butterfly continuations classify the same FFT factorization surface.
\end{theorem}
\leanchecked{BEDC.Derived.FftUp.FftBHistSourcePacket\_factorization\_classifier\_stability}
\begin{proof}
Transport the scalar row through $\NameCert_{\ComplexUp}$ and the transform
row through $\NameCert_{\FourierUp}$. These two dependency transports fix the
source histories on which the finite schedule is read.

Next transport every scheduled butterfly stage along the same packet
comparison. The $\mathsf{InBundle}$ witnesses keep the stage list inside the
finite $\mathsf{ProbeBundle}$, while $\Cont$ transports the butterfly output
rows.

The factorization ledger is stored with those stage rows. Repacking the
transported dependencies, schedule, and ledger gives the same classifier
surface for the target FFT packet.
\end{proof}
\leanchecked{BEDC.Derived.FftUp.FftBHistSourcePacket\_factorization\_classifier\_stability\_obligation}

\begin{theorem}[FFT ledger exactness obligation]
\label{thm:fft-ledger-exactness-obligation}
The $\FftUp$ ledger consists exactly of the $\NameCert_{\ComplexUp}$
dependency row, the $\NameCert_{\FourierUp}$ dependency row, the finite
$\mathsf{ProbeBundle}$ schedule, the $\mathsf{InBundle}$ stage witnesses, the
butterfly continuation rows, the factorization ledger rows, and their
$\hsame$ and $\Cont$ transports. It excludes analytic error estimates,
floating-point bounds, asymptotic runtime claims, and standard bridge rows.
\end{theorem}
\leanchecked{BEDC.Derived.FftUp.FftBHistSourcePacket\_ledger\_exactness\_obligation}
\begin{proof}
Coverage reads any accepted FFT packet into the two dependency rows, the
finite schedule, the stage membership witnesses, the butterfly
continuations, and the factorization ledger.

Soundness repacks exactly those rows. The scalar and transform dependencies
justify the input histories, while $\mathsf{InBundle}$ and $\Cont$ justify the
finite schedule and every butterfly output.

Completeness is the absence of further ledger coordinates. Error estimates,
floating-point bounds, asymptotic analysis, and standard bridge data are not
fields of the packet, so they cannot be consumed by this certificate.
\end{proof}
\leanchecked{BEDC.Derived.FftUp.FftBHistSourcePacket\_factorization\_ledger\_transport\_determinacy}

\begin{theorem}[FFT root threshold unblock surface]
\label{thm:fft-root-threshold-unblock-surface}
The rows of \autoref{def:fft-bhist-source-packet},
\autoref{thm:fft-carrier-butterfly-obligation},
\autoref{thm:fft-factorization-classifier-stability-obligation}, and
\autoref{thm:fft-ledger-exactness-obligation} form the dependency-ready
finite-history surface for $\FftUp$. Downstream numerical-analysis
certificates may consume the complex dependency, Fourier dependency, finite
butterfly schedule, factorization ledger, and exact exclusion of analytic
error or bridge data, but must provide any such estimates or bridges outside
the FFT packet.
\end{theorem}
\leanchecked{BEDC.Derived.FftUp.FftBHistSourcePacket\_root\_threshold\_unblock\_surface}
\begin{proof}
The definition fixes the finite packet and the two dependency certificates.
The butterfly obligation turns the packet into carried $\mathsf{BHist}$ stage
rows over a certified finite schedule.

The factorization stability obligation supplies the classifier transport for
the split-length ledger and butterfly continuations. The ledger exactness
obligation then states that no analytic estimate or standard bridge row is
available inside the packet.

Together these rows expose the concrete FFT source surface needed by later
certificates while keeping numerical-analysis obligations separate.
\end{proof}

\autoref{thm:fft-factorization-classifier-stability-obligation}, the ledger
exactness obligation of \autoref{thm:fft-ledger-exactness-obligation}, and the
root threshold surface of \autoref{thm:fft-root-threshold-unblock-surface}

Consumers may read the $\NameCert_{\ComplexUp}$ and
$\NameCert_{\FourierUp}$ dependencies, the finite $\mathsf{ProbeBundle}$

rows, the factorization ledger, and the $\hsame/\Cont$ transport rows. Analytic

\end{theorem}
\leanchecked{BEDC.Derived.FftUp.FftBHistSourcePacket\_schedule\_public\_namecert\_export}

$\mathsf{ProbeBundle}$ schedule. Its membership rows and butterfly
continuations are the only stage data exposed to the BEDC surface.

Apply the butterfly carrier obligation to recover the finite stage witnesses
and their $\Cont$ rows from any accepted packet. Apply the factorization
classifier stability obligation to transport the split-length ledger and the
butterfly rows through the displayed $\hsame$ comparison.

The ledger exactness obligation states that the readable ledger consists
exactly of the two dependency rows, the schedule, the stage witnesses, the
butterfly continuations, the factorization ledger, and their transport
evidence. The root threshold surface repacks those rows as the public
$\NameCert_{\FftUp}$ interface, while analytic estimates, floating-point

\end{proof}

\begin{definition}[FFT length-one identity surface]
\label{def:fft-length-one-identity-surface}
For an accepted $\FftUp$ source packet of \autoref{def:fft-bhist-source-packet}, write $c$ for its $\ComplexUp$ scalar row, $F$ for its $\FourierUp$ transform row, $x$ for the carrier row produced from $c$ and $F$, $\beta$ for the scheduled butterfly row, $\nu$ for the factorization row, $\lambda$ for the ledger row, $y$ for the endpoint row, $\sigma$ for the stage name, and $\Pi$ for the finite schedule. The predicate $\mathsf{FftLenOneId}(P,x,y)$ holds when the following rows are displayed:
\begin{itemize}
  \item the schedule has length one, $\mathsf{BLEN}(\Pi)=1$;
  \item the carrier membership row of \autoref{thm:fft-carrier-butterfly-obligation} reads $\sigma$ as the unique scheduled stage and reads that stage back as the input row $x$;
  \item the singleton butterfly row supplied by $\NameCert_{\FourierUp}$ and $\NameCert_{\ComplexUp}$ classifies the butterfly ledger with the input row, $\lambda\sim_{\ComplexUp} x$;
  \item the length-one factorization ledger is neutral on that ledger, so the endpoint row classifies with the ledger row, $y\sim_{\ComplexUp}\lambda$.
\end{itemize}
This surface contains no analytic FFT equality, host array equality, floating-point estimate, asymptotic claim, or standard bridge row.
\end{definition}

\begin{lemma}[FFT length-one schedule stage exhaustion]
\label{lem:fft-length-one-schedule-stage-exhaustion}
Let $P$ be an accepted $\FftUp$ source packet of \autoref{def:fft-bhist-source-packet} with finite schedule $\Pi$. If $\mathsf{BLEN}(\Pi)=1$, then every scheduled butterfly stage read by the carrier obligation is the single stage named by the packet membership row. In particular, the packet exposes no second butterfly stage and no tail-stage membership row.
\end{lemma}
\begin{proof}
Apply \autoref{thm:fft-carrier-butterfly-obligation} to the accepted packet. This gives the displayed schedule membership row and the butterfly continuation rows inside the finite $\mathsf{ProbeBundle}$ schedule.

The length-one hypothesis and \autoref{thm:probe-bundle-length-one-iff-singleton} identify $\Pi$ with a singleton bundle. Then \autoref{thm:bundle-member-length-one-singleton} identifies the member supplied by the carrier obligation with that singleton head. A tail membership would be a membership row in $\Bnil$, which is impossible by the same singleton reading. Hence the carrier has exactly the one scheduled stage named by the packet membership row.
\end{proof}

\begin{theorem}[FFT length-one schedule classifies with the input row]
\label{thm:fft-length-one-schedule-input-classifier}
Let $P$ be an accepted $\FftUp$ source packet of \autoref{def:fft-bhist-source-packet}. If $\mathsf{FftLenOneId}(P,x,y)$ holds for its input row $x$ and endpoint row $y$, then the FFT output endpoint is classified with the input row:
$y\sim_{\ComplexUp}x$.
\end{theorem}
\begin{proof}
Read the accepted packet through \autoref{thm:fft-carrier-butterfly-obligation}. The packet yields the $\ComplexUp$ scalar dependency, the $\FourierUp$ transform dependency, the finite schedule membership row, the singleton butterfly continuation, and the continuation rows that lead to the endpoint.

By the first row of \autoref{def:fft-length-one-identity-surface}, the schedule has length one. The exhaustion lemma \autoref{lem:fft-length-one-schedule-stage-exhaustion} therefore removes every possible tail stage: the butterfly continuation seen by the carrier obligation is the unique scheduled row, and its stage readback is the input row $x$.

The singleton butterfly row in \autoref{def:fft-length-one-identity-surface} classifies the butterfly ledger with the input row, giving $\lambda\sim_{\ComplexUp}x$. Next apply \autoref{thm:fft-ledger-exactness-obligation}. The ledger contains exactly the two dependency rows, the singleton schedule membership row, the butterfly continuation, the factorization row, and their displayed transports; there is no further stage or hidden analytic row that can change the endpoint classification.

The neutral length-one factorization row in \autoref{def:fft-length-one-identity-surface} gives $y\sim_{\ComplexUp}\lambda$. Transitivity of the $\ComplexUp$ classifier supplied by the dependency certificate then composes $y\sim_{\ComplexUp}\lambda$ with $\lambda\sim_{\ComplexUp}x$, yielding $y\sim_{\ComplexUp}x$.
\end{proof}
