\begin{theorem}[ContinuousMap associative composition canonical package]
\label{thm:continuousmap-carrier-composition-associative-canonical-package}
For histories
$source,mid_1,mid_2,target,mapF,mapG,mapH,mapFG,mapGH,mapL,mapR,modF,modG,modH,modFG,modGH,modL,modR,certF,certG,certH,certFG,certGH,certL,certR,distF,distG,distH,displayedL,displayedR:\mathsf{BHist}$,
assume the three concrete continuous-map carriers
$$
\begin{array}{l}
\mathsf{ContinuousMapCarrier}(source,mapF,mid_1,modF,certF,distF),\\
\mathsf{ContinuousMapCarrier}(mid_1,mapG,mid_2,modG,certG,distG),\\
\mathsf{ContinuousMapCarrier}(mid_2,mapH,target,modH,certH,distH),
\end{array}
$$
the left-bracketed continuation ledgers
$$
\begin{array}{lll}
\ContR(mapF,mapG,mapFG),&
\ContR(modF,modG,modFG),&
\ContR(mid_2,modFG,certFG),\\
\ContR(mapFG,mapH,mapL),&
\ContR(modFG,modH,modL),&
\ContR(target,modL,certL),
\end{array}
$$
and the right-bracketed continuation ledgers
$$
\begin{array}{lll}
\ContR(mapG,mapH,mapGH),&
\ContR(modG,modH,modGH),&
\ContR(target,modGH,certGH),\\
\ContR(mapF,mapGH,mapR),&
\ContR(modF,modGH,modR),&
\ContR(target,modR,certR).
\end{array}
$$
Then the threefold composition data determine the associative canonical
certificate package:
\begin{enumerate}
\item the left-bracketed composite carrier closes over the canonical
source-target distance ledger,
$$
\mathsf{ContinuousMapCarrier}
(source,mapL,target,modL,certL,\mathsf{append}(source,target));
$$

\item the right-bracketed composite carrier closes over the same canonical
source-target distance ledger,
$$
\mathsf{ContinuousMapCarrier}
(source,mapR,target,modR,certR,\mathsf{append}(source,target));
$$

\item the two composite graph-map histories are classified together,
$$
\hsame(mapL,mapR);
$$

\item the two composite modulus histories are classified together,
$$
\hsame(modL,modR);
$$

\item every displayed carrier with either bracketed visible fields has the
canonical distance ledger:
$$
\begin{array}{l}
\mathsf{ContinuousMapCarrier}(source,mapL,target,modL,certL,displayedL)
\Rightarrow
\hsame(displayedL,\mathsf{append}(source,target)),\\
\mathsf{ContinuousMapCarrier}(source,mapR,target,modR,certR,displayedR)
\Rightarrow
\hsame(displayedR,\mathsf{append}(source,target)).
\end{array}
$$
\end{enumerate}
\end{theorem}
\begin{proof}
Build the left bracketing in two closure steps. First apply
Theorem~\ref{thm:continuousmap-carrier-comp-closed} to the
$source \to mid_1$ and $mid_1 \to mid_2$ carriers, using the graph, modulus,
and certificate ledgers ending at $certFG$. Apply the same theorem again to
that intermediate carrier and the $mid_2 \to target$ carrier, using the
ledgers ending at $certL$. This gives the left composite carrier with distance
$\mathsf{append}(source,target)$.

Build the right bracketing symmetrically. Apply
Theorem~\ref{thm:continuousmap-carrier-comp-closed} to the
$mid_1 \to mid_2$ and $mid_2 \to target$ carriers, using the ledgers ending at
$certGH$. Then compose the $source \to mid_1$ carrier with that intermediate
carrier using the ledgers ending at $certR$. This gives the right composite
carrier with the same displayed distance ledger
$\mathsf{append}(source,target)$.

For the graph-map row, apply continuation associativity up to history
sameness, Theorem~\ref{thm:continuation-associativity-up-to-hsame}, to
$\ContR(mapF,mapG,mapFG)$, $\ContR(mapFG,mapH,mapL)$,
$\ContR(mapG,mapH,mapGH)$, and $\ContR(mapF,mapGH,mapR)$. This is the paper
form of the checked kernel theorem
\texttt{continuation\_associativity\_up\_to\_hsame}; uniqueness of the
associative continuation endpoint is also recorded by
\texttt{cont\_assoc\_unique}.

For the modulus row, use the same continuation-associativity argument on
$\ContR(modF,modG,modFG)$, $\ContR(modFG,modH,modL)$,
$\ContR(modG,modH,modGH)$, and $\ContR(modF,modGH,modR)$. The resulting
classifier row is $\hsame(modL,modR)$.

Finally apply
Theorem~\ref{thm:continuousmap-carrier-comp-distance-deterministic} to the
left two-step composition data for any displayed left-bracketed carrier, and
apply the same theorem to the right two-step composition data for any
displayed right-bracketed carrier. The two applications give the two displayed
distance-determinacy clauses.
\end{proof}
\leanchecked{BEDC.Derived.ContinuousMapUp.ContinuousMapCarrier\_composition\_associative\_canonical\_package}

\begin{theorem}[ContinuousMap associative composition certificate determinacy]
\label{thm:continuousmap-composition-associative-certificate-deterministic}
Under the hypotheses of
\autoref{thm:continuousmap-carrier-composition-associative-canonical-package},
the two target-modulus certificate ledgers selected by the two bracketings
determine the same certificate history:
$$
\hsame(certL,certR).
$$
\end{theorem}
\begin{proof}
Apply
\autoref{thm:continuousmap-carrier-composition-associative-canonical-package}
to the three input carriers and the two displayed continuation spines. Its
modulus row gives $\hsame(modL,modR)$. Combine this modulus sameness with
reflexivity of the target history and the two target-modulus continuation
ledgers $\ContR(target,modL,certL)$ and $\ContR(target,modR,certR)$. The
continuation respect theorem for $\hsame$ then gives $\hsame(certL,certR)$.
\end{proof}
\leanchecked{BEDC.Derived.ContinuousMapUp.ContinuousMapCarrier\_composition\_associative\_certificate\_deterministic}

\begin{theorem}[ContinuousMap associative composition depth synchronization]
\label{thm:continuousmap-associative-composition-depth-synchronization}
For histories
$source,mid_1,mid_2,target,mapF,mapG,mapH,mapFG,mapGH,mapL,mapR,modF,modG,modH,modFG,modGH,modL,modR,certF,certG,certH,certFG,certGH,certL,certR,distF,distG,distH:\mathsf{BHist}$,
assume the three concrete continuous-map carriers
$$
\begin{array}{l}
\mathsf{ContinuousMapCarrier}(source,mapF,mid_1,modF,certF,distF),\\
\mathsf{ContinuousMapCarrier}(mid_1,mapG,mid_2,modG,certG,distG),\\
\mathsf{ContinuousMapCarrier}(mid_2,mapH,target,modH,certH,distH),
\end{array}
$$
the left-bracketed continuation ledgers
$$
\begin{array}{lll}
\ContR(mapF,mapG,mapFG),&
\ContR(modF,modG,modFG),&
\ContR(mid_2,modFG,certFG),\\
\ContR(mapFG,mapH,mapL),&
\ContR(modFG,modH,modL),&
\ContR(target,modL,certL),
\end{array}
$$
and the right-bracketed continuation ledgers
$$
\begin{array}{lll}
\ContR(mapG,mapH,mapGH),&
\ContR(modG,modH,modGH),&
\ContR(target,modGH,certGH),\\
\ContR(mapF,mapGH,mapR),&
\ContR(modF,modGH,modR),&
\ContR(target,modR,certR).
\end{array}
$$
Then both bracketings synchronize the endpoint depth with their displayed
composite graph histories:
$$
\begin{array}{l}
\mathsf{MetricDistanceDepth}(target)
=
\mathsf{MetricDistanceDepth}(source)+
\mathsf{MetricDistanceDepth}(mapL),\\
\mathsf{MetricDistanceDepth}(target)
=
\mathsf{MetricDistanceDepth}(source)+
\mathsf{MetricDistanceDepth}(mapR).
\end{array}
$$
\end{theorem}
\begin{proof}
First build the left intermediate carrier. Apply
Theorem~\ref{thm:continuousmap-carrier-comp-closed} to the first two
continuous-map carriers and the ledgers ending at $certFG$. Project its
continuous-function half, and project the continuous-function half of the
$mid_2 \to target$ carrier.

Apply Theorem~\ref{thm:continuousmap-comp-graph-depth-add} to those two
continuous-function carriers and the left-bracketed ledgers ending at $certL$.
This gives
$$
\mathsf{MetricDistanceDepth}(target)
=
\mathsf{MetricDistanceDepth}(source)+
\mathsf{MetricDistanceDepth}(mapL).
$$

Next build the right intermediate carrier. Apply
Theorem~\ref{thm:continuousmap-carrier-comp-closed} to the second and third
continuous-map carriers and the ledgers ending at $certGH$. Project its
continuous-function half, and project the continuous-function half of the
$source \to mid_1$ carrier.

Apply Theorem~\ref{thm:continuousmap-comp-graph-depth-add} to the
$source \to mid_1$ continuous-function carrier, the projected
$mid_1 \to target$ continuous-function carrier, and the right-bracketed ledgers
ending at $certR$. This gives
$$
\mathsf{MetricDistanceDepth}(target)
=
\mathsf{MetricDistanceDepth}(source)+
\mathsf{MetricDistanceDepth}(mapR).
$$

The two displayed equalities are exactly the depth-synchronization conclusion.
\end{proof}
\leanchecked{BEDC.Derived.ContinuousMapUp.ContinuousMapCarrier\_associative\_composition\_depth\_synchronization}

\begin{theorem}[ContinuousMap identity sandwich coherence]
\label{thm:continuousmap-identity-sandwich-coherence}
For histories
$source,target,map,modulus,cert,distance,map_{LF},map_L,map_{RT},map_R,mod_{LF},mod_L,mod_{RT},mod_R,cert_{LF},cert_L,cert_{RT},cert_R:\mathsf{BHist}$,
assume
$$
\mathsf{ContinuousMapCarrier}(source,map,target,modulus,cert,distance),
$$
and assume the left-bracketed identity-sandwich ledger spine
$$
\begin{array}{lll}
\ContR(\emp,map,map_{LF}),&
\ContR(\emp,modulus,mod_{LF}),&
\ContR(target,mod_{LF},cert_{LF}),\\
\ContR(map_{LF},\emp,map_L),&
\ContR(mod_{LF},\emp,mod_L),&
\ContR(target,mod_L,cert_L),
\end{array}
$$
and the right-bracketed identity-sandwich ledger spine
$$
\begin{array}{lll}
\ContR(map,\emp,map_{RT}),&
\ContR(modulus,\emp,mod_{RT}),&
\ContR(target,mod_{RT},cert_{RT}),\\
\ContR(\emp,map_{RT},map_R),&
\ContR(\emp,mod_{RT},mod_R),&
\ContR(target,mod_R,cert_R).
\end{array}
$$
Then the two bracketed identity sandwiches are coherent with the displayed
carrier:
\begin{enumerate}
\item the left-bracketed sandwich closes as
$$
\mathsf{ContinuousMapCarrier}
(source,map_L,target,mod_L,cert_L,\mathsf{append}(source,target));
$$

\item the right-bracketed sandwich closes as
$$
\mathsf{ContinuousMapCarrier}
(source,map_R,target,mod_R,cert_R,\mathsf{append}(source,target));
$$

\item the selected graph histories are the original graph history,
$$
\hsame(map_L,map)\land \hsame(map_R,map);
$$

\item the selected modulus histories are the original modulus history,
$$
\hsame(mod_L,modulus)\land \hsame(mod_R,modulus);
$$

\item the selected certificate histories and the original certificate history
are pairwise classified together,
$$
\hsame(cert_L,cert)\land \hsame(cert_R,cert)\land \hsame(cert_L,cert_R).
$$
\end{enumerate}
\end{theorem}
\leanchecked{BEDC.Derived.ContinuousMapUp.ContinuousMapCarrier\_identity\_sandwich\_coherence}
\begin{proof}
Apply \autoref{thm:continuousmap-empty-identity-carriers} to the displayed
$\mathsf{ContinuousMapCarrier}$. This supplies the source and target empty
identity carriers used as the outer factors of the sandwich.

For the left bracketing, first apply
\autoref{thm:continuousmap-carrier-comp-closed} to the source identity carrier
and the displayed carrier, using the ledgers ending at $cert_{LF}$. Apply the
same closure theorem again to that intermediate carrier and the target identity
carrier, using the ledgers ending at $cert_L$. This gives the left sandwich
carrier over $\mathsf{append}(source,target)$.

For the right bracketing, first compose the displayed carrier with the target
identity carrier, using the ledgers ending at $cert_{RT}$. Then compose the
source identity carrier with that intermediate carrier, using the ledgers ending
at $cert_R$. The same closure theorem gives the right sandwich carrier over the
same canonical source-target distance ledger.

Apply
\autoref{thm:continuousmap-carrier-composition-associative-canonical-package}
to the source identity carrier, the displayed carrier, and the target identity
carrier. Its graph and modulus rows identify the two bracketed selected
histories. Apply
\autoref{thm:continuousmap-composition-associative-certificate-deterministic}
to the same three-carrier spine to get $\hsame(cert_L,cert_R)$.

The remaining readback is by continuation units. The ledgers
$\ContR(\emp,map,map_{LF})$ and $\ContR(map_{LF},\emp,map_L)$ combine
\texttt{cont\_left\_unit\_result} and \texttt{cont\_right\_unit\_result} to
give $\hsame(map_L,map)$, and the right-bracketed graph ledgers give
$\hsame(map_R,map)$ in the same way. The identical argument applied to
$modulus$ gives the two modulus classifiers.

Project the original certificate ledger $\ContR(target,modulus,cert)$ from the
carrier. With target reflexivity, the modulus classifiers, and the displayed
target-modulus certificate ledgers, \texttt{cont\_respects\_hsame} identifies
$cert_L$ and $cert_R$ with $cert$. Together with the associative certificate
row, these are exactly the pairwise certificate classifiers.

The ten-field fit is: traditional target identity-sandwich coherence for
uniformly continuous maps, BEDC source $\mathsf{BHist}$ with $\ContR$, $\hsame$,
and $\mathsf{ContinuousMapCarrier}$, carrier $\mathsf{ContinuousMapCarrier}$,
classifier $\hsame$, stability from unit continuation and associative
composition, ledger graph/modulus/certificate histories, core theorem this
coherence package, exactness the canonical source-target distance rows, standard
bridge deferred, and status $\mathsf{ProofTarget}$.
\end{proof}
\leanchecked{BEDC.Derived.ContinuousMapUp.ContinuousMapCarrier\_identity\_sandwich\_coherence}
