\begin{theorem}[ContinuousMap associative composition canonical package]
\label{thm:continuousmap-carrier-composition-associative-canonical-package}
For histories
$source,mid_1,mid_2,target,mapF,mapG,mapH,mapFG,mapGH,mapL,mapR,modF,modG,modH,modFG,modGH,modL,modR,certF,certG,certH,certFG,certGH,certL,certR,distF,distG,distH,displayedL,displayedR:\mathsf{BHist}$,
assume the three concrete continuous-map carriers
$$
\begin{array}{l}
\mathsf{ContinuousMapCarrier}(source,mapF,mid_1,modF,certF,distF),\\
\mathsf{ContinuousMapCarrier}(mid_1,mapG,mid_2,modG,certG,distG),\\
\mathsf{ContinuousMapCarrier}(mid_2,mapH,target,modH,certH,distH),
\end{array}
$$
the left-bracketed continuation ledgers
$$
\begin{array}{lll}
\ContR(mapF,mapG,mapFG),&
\ContR(modF,modG,modFG),&
\ContR(mid_2,modFG,certFG),\\
\ContR(mapFG,mapH,mapL),&
\ContR(modFG,modH,modL),&
\ContR(target,modL,certL),
\end{array}
$$
and the right-bracketed continuation ledgers
$$
\begin{array}{lll}
\ContR(mapG,mapH,mapGH),&
\ContR(modG,modH,modGH),&
\ContR(target,modGH,certGH),\\
\ContR(mapF,mapGH,mapR),&
\ContR(modF,modGH,modR),&
\ContR(target,modR,certR).
\end{array}
$$
Then the threefold composition data determine the associative canonical
certificate package:
\begin{enumerate}
\item the left-bracketed composite carrier closes over the canonical
source-target distance ledger,
$$
\mathsf{ContinuousMapCarrier}
(source,mapL,target,modL,certL,\mathsf{append}(source,target));
$$

\item the right-bracketed composite carrier closes over the same canonical
source-target distance ledger,
$$
\mathsf{ContinuousMapCarrier}
(source,mapR,target,modR,certR,\mathsf{append}(source,target));
$$

\item the two composite graph-map histories are classified together,
$$
\hsame(mapL,mapR);
$$

\item the two composite modulus histories are classified together,
$$
\hsame(modL,modR);
$$

\item every displayed carrier with either bracketed visible fields has the
canonical distance ledger:
$$
\begin{array}{l}
\mathsf{ContinuousMapCarrier}(source,mapL,target,modL,certL,displayedL)
\Rightarrow
\hsame(displayedL,\mathsf{append}(source,target)),\\
\mathsf{ContinuousMapCarrier}(source,mapR,target,modR,certR,displayedR)
\Rightarrow
\hsame(displayedR,\mathsf{append}(source,target)).
\end{array}
$$
\end{enumerate}
\end{theorem}
\begin{proof}
Build the left bracketing in two closure steps. First apply
Theorem~\ref{thm:continuousmap-carrier-comp-closed} to the
$source \to mid_1$ and $mid_1 \to mid_2$ carriers, using the graph, modulus,
and certificate ledgers ending at $certFG$. Apply the same theorem again to
that intermediate carrier and the $mid_2 \to target$ carrier, using the
ledgers ending at $certL$. This gives the left composite carrier with distance
$\mathsf{append}(source,target)$.

Build the right bracketing symmetrically. Apply
Theorem~\ref{thm:continuousmap-carrier-comp-closed} to the
$mid_1 \to mid_2$ and $mid_2 \to target$ carriers, using the ledgers ending at
$certGH$. Then compose the $source \to mid_1$ carrier with that intermediate
carrier using the ledgers ending at $certR$. This gives the right composite
carrier with the same displayed distance ledger
$\mathsf{append}(source,target)$.

For the graph-map row, apply continuation associativity up to history
sameness, Theorem~\ref{thm:continuation-associativity-up-to-hsame}, to
$\ContR(mapF,mapG,mapFG)$, $\ContR(mapFG,mapH,mapL)$,
$\ContR(mapG,mapH,mapGH)$, and $\ContR(mapF,mapGH,mapR)$. This is the paper
form of the checked kernel theorem
\texttt{continuation\_associativity\_up\_to\_hsame}; uniqueness of the
associative continuation endpoint is also recorded by
\texttt{cont\_assoc\_unique}.

For the modulus row, use the same continuation-associativity argument on
$\ContR(modF,modG,modFG)$, $\ContR(modFG,modH,modL)$,
$\ContR(modG,modH,modGH)$, and $\ContR(modF,modGH,modR)$. The resulting
classifier row is $\hsame(modL,modR)$.

Finally apply
Theorem~\ref{thm:continuousmap-carrier-comp-distance-deterministic} to the
left two-step composition data for any displayed left-bracketed carrier, and
apply the same theorem to the right two-step composition data for any
displayed right-bracketed carrier. The two applications give the two displayed
distance-determinacy clauses.
\end{proof}

\begin{theorem}[ContinuousMap associative composition depth synchronization]
\label{thm:continuousmap-associative-composition-depth-synchronization}
For histories
$source,mid_1,mid_2,target,mapF,mapG,mapH,mapFG,mapGH,mapL,mapR,modF,modG,modH,modFG,modGH,modL,modR,certF,certG,certH,certFG,certGH,certL,certR,distF,distG,distH:\mathsf{BHist}$,
assume the three concrete continuous-map carriers
$$
\begin{array}{l}
\mathsf{ContinuousMapCarrier}(source,mapF,mid_1,modF,certF,distF),\\
\mathsf{ContinuousMapCarrier}(mid_1,mapG,mid_2,modG,certG,distG),\\
\mathsf{ContinuousMapCarrier}(mid_2,mapH,target,modH,certH,distH),
\end{array}
$$
the left-bracketed continuation ledgers
$$
\begin{array}{lll}
\ContR(mapF,mapG,mapFG),&
\ContR(modF,modG,modFG),&
\ContR(mid_2,modFG,certFG),\\
\ContR(mapFG,mapH,mapL),&
\ContR(modFG,modH,modL),&
\ContR(target,modL,certL),
\end{array}
$$
and the right-bracketed continuation ledgers
$$
\begin{array}{lll}
\ContR(mapG,mapH,mapGH),&
\ContR(modG,modH,modGH),&
\ContR(target,modGH,certGH),\\
\ContR(mapF,mapGH,mapR),&
\ContR(modF,modGH,modR),&
\ContR(target,modR,certR).
\end{array}
$$
Then both bracketings synchronize the endpoint depth with their displayed
composite graph histories:
$$
\begin{array}{l}
\mathsf{MetricDistanceDepth}(target)
=
\mathsf{MetricDistanceDepth}(source)+
\mathsf{MetricDistanceDepth}(mapL),\\
\mathsf{MetricDistanceDepth}(target)
=
\mathsf{MetricDistanceDepth}(source)+
\mathsf{MetricDistanceDepth}(mapR).
\end{array}
$$
\end{theorem}
\begin{proof}
First build the left intermediate carrier. Apply
Theorem~\ref{thm:continuousmap-carrier-comp-closed} to the first two
continuous-map carriers and the ledgers ending at $certFG$. Project its
continuous-function half, and project the continuous-function half of the
$mid_2 \to target$ carrier.

Apply Theorem~\ref{thm:continuousmap-comp-graph-depth-add} to those two
continuous-function carriers and the left-bracketed ledgers ending at $certL$.
This gives
$$
\mathsf{MetricDistanceDepth}(target)
=
\mathsf{MetricDistanceDepth}(source)+
\mathsf{MetricDistanceDepth}(mapL).
$$

Next build the right intermediate carrier. Apply
Theorem~\ref{thm:continuousmap-carrier-comp-closed} to the second and third
continuous-map carriers and the ledgers ending at $certGH$. Project its
continuous-function half, and project the continuous-function half of the
$source \to mid_1$ carrier.

Apply Theorem~\ref{thm:continuousmap-comp-graph-depth-add} to the
$source \to mid_1$ continuous-function carrier, the projected
$mid_1 \to target$ continuous-function carrier, and the right-bracketed ledgers
ending at $certR$. This gives
$$
\mathsf{MetricDistanceDepth}(target)
=
\mathsf{MetricDistanceDepth}(source)+
\mathsf{MetricDistanceDepth}(mapR).
$$

The two displayed equalities are exactly the depth-synchronization conclusion.
\end{proof}
