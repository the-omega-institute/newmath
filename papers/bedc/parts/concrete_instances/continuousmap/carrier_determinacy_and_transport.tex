\begin{theorem}[ContinuousMap target, certificate, and distance determinacy]
\label{thm:continuousmap-target-cert-distance-deterministic}
For histories $source$, $map$, $target$, $target'$, $modulus$, $cert$, $cert'$,
$distance$, and $distance'$, two concrete continuous-map carriers with the same
source, map, and modulus determine the target, certificate, and distance
ledgers up to $\hsame$:
$$
\begin{array}{l}
\mathsf{ContinuousMapCarrier}(source,map,target,modulus,cert,distance)
\Longrightarrow
\mathsf{ContinuousMapCarrier}(source,map,target',modulus,cert',distance')
\\
\Longrightarrow
\hsame(target,target')\land\hsame(cert,cert')\land
\hsame(distance,distance')\land\ContR(source,target,distance).
\end{array}
$$
\end{theorem}
\begin{proof}
The continuous-function components have the same source, map, and modulus, so
continuous-function target-certificate determinacy gives
$\hsame(target,target')$ and $\hsame(cert,cert')$. The two metric distance
witnesses then share source histories and target histories up to $\hsame$;
metric result determinacy identifies $distance$ and $distance'$. The
continuation ledger $\ContR(source,target,distance)$ is the final field of the
left metric distance witness.
\end{proof}
\leanchecked{BEDC.Derived.ContinuousMapUp.ContinuousMapCarrier\_target\_cert\_distance\_deterministic}

\begin{theorem}[ContinuousMap carrier canonical distance exactness]
\label{thm:continuousmap-carrier-canonical-distance-exactness}
For histories $source,map,target,modulus,cert,distance:\mathsf{BHist}$,
the concrete continuous-map carrier is exactly the continuous-function carrier
together with the canonical source-target distance classifier:
$$
\mathsf{ContinuousMapCarrier}(source,map,target,modulus,cert,distance)
\Longleftrightarrow
\mathsf{ContinuousFunctionCarrier}(source,map,target,modulus,cert)\land
\hsame(distance,\mathsf{append}(source,target)).
$$
\end{theorem}
\begin{proof}
In the forward direction, project the continuous-function carrier from the
given continuous-map carrier. Theorem~\ref{thm:continuousmap-carrier-canonical-distance-iff}
then supplies the carrier with distance ledger
$\mathsf{append}(source,target)$, and
Theorem~\ref{thm:continuousmap-target-cert-distance-deterministic} compares
the two carriers with the same source, map, target, modulus, and certificate,
identifying $distance$ with $\mathsf{append}(source,target)$.

Conversely, start from the displayed continuous-function carrier and the
classifier $\hsame(distance,\mathsf{append}(source,target))$.
Theorem~\ref{thm:continuousmap-carrier-canonical-distance-iff} builds the
canonical-distance carrier. Its metric component contains the continuation
$\ContR(source,target,\mathsf{append}(source,target))$ and unary membership of
the canonical distance ledger. Transport the continuation by
$\mathsf{BEDC.Derived.MetricUp.MetricDistanceWitness\_cont\_hsame\_transport}$,
transport unary membership of the distance ledger along $\hsame$, and repack
these fields as the required $\mathsf{MetricDistanceWitness}(source,target,distance)$.
Together with the original continuous-function carrier this is precisely
$\mathsf{ContinuousMapCarrier}(source,map,target,modulus,cert,distance)$.

The ten-field fit is: traditional target normalized continuous-map distance
ledger; BEDC source $\mathsf{ContinuousFunctionCarrier}$,
$\mathsf{MetricDistanceWitness}$, and $\ContR$; carrier
$\mathsf{ContinuousMapCarrier}$; classifier $\hsame$; stability obligations
distance determinacy and metric hsame transport; ledger the hidden distance
history; core theorem this iff; exactness the displayed iff; standard bridge
deferred; status $\mathsf{ProofTarget}$.
\end{proof}
\leanchecked{BEDC.Derived.ContinuousMapUp.ContinuousMapCarrier\_canonical\_distance\_exactness}

\begin{theorem}[ContinuousMap categorical canonical distance exactness]
\label{thm:continuousmap-carrier-categorical-canonical-distance-exactness}
For histories $source,map,target,modulus,cert,distance:\mathsf{BHist}$,
the continuous-map carrier is exactly the composite classifier obtained from
the categorical source-target hom row, the target-side continuity-modulus row,
and the canonical source-target distance ledger:
$$
\begin{array}{l}
\mathsf{ContinuousMapCarrier}(source,map,target,modulus,cert,distance)
\\[2pt]
\quad\Longleftrightarrow\quad
\mathsf{CategoryHomCarrier}(source,target,map)\land
\mathsf{ContinuousModulusWitness}(target,modulus,cert)\land
\hsame(distance,\mathsf{append}(source,target)).
\end{array}
$$
\end{theorem}
\begin{proof}
In the forward direction, apply
Theorem~\ref{thm:continuousmap-carrier-category-metric-decomposition} to the
given carrier. This yields the categorical hom row and the target-side
modulus row, together with the metric distance witness.

Apply
Theorem~\ref{thm:continuousmap-carrier-canonical-distance-exactness} to the
same carrier. Its distance component replaces the metric witness by the
canonical classifier
$\hsame(distance,\mathsf{append}(source,target))$, so the displayed
three-part conjunction follows.

Conversely, unpack
$\mathsf{CategoryHomCarrier}(source,target,map)$ into the unary source,
target, and map histories plus the graph continuation. Unpack
$\mathsf{ContinuousModulusWitness}(target,modulus,cert)$ into the target,
modulus, and certificate histories plus the modulus continuation. These rows
rebuild
$\mathsf{ContinuousFunctionCarrier}(source,map,target,modulus,cert)$.

Now use
Theorem~\ref{thm:continuousmap-carrier-canonical-distance-exactness} in the
reverse direction with the rebuilt continuous-function carrier and the
displayed classifier
$\hsame(distance,\mathsf{append}(source,target))$. The result is precisely
$\mathsf{ContinuousMapCarrier}(source,map,target,modulus,cert,distance)$.

The ten-field fit is: traditional target continuous maps as morphisms with a
canonical endpoint distance ledger; BEDC source
$\mathsf{CategoryUp}/\mathsf{ContinuousUp}/\mathsf{MetricUp}/
\mathsf{ContinuousMapUp}$; carrier $\mathsf{ContinuousMapCarrier}$;
classifier $\hsame$; stability supplied by the categorical hom row, the
modulus witness, and canonical distance exactness; ledger source, target, map,
modulus, certificate, and distance histories; core theorem the displayed
equivalence; exactness replacement of the metric witness by
$\hsame(distance,\mathsf{append}(source,target))$; standard bridge is the
same morphism-layer reading; downstream consumer $\ContinuousMapUp$ as the
morphism layer for compact-continuous certificates.
\end{proof}

\begin{theorem}[ContinuousMap carrier hsame-field transport with depth ledger]

cert,cert',distance,distance':\mathsf{BHist}$, suppose
$$
\mathsf{ContinuousMapCarrier}(source,map,target,modulus,cert,distance)
$$
and
$$
\hsame(source,source')\land
\hsame(map,map')\land
\hsame(target,target')\land
\hsame(modulus,modulus')\land
\hsame(cert,cert')\land
\hsame(distance,distance')
$$
hold. Then the transported histories again form a continuous-map carrier, and
the transported distance exposes the endpoint continuation and depth ledger:
$$
\begin{array}{l}
\mathsf{ContinuousMapCarrier}(source',map',target',modulus',cert',distance')
\\
\land\ \ContR(source',target',distance')
\\
\land\
\mathsf{MetricDistanceDepth}(distance')=
\mathsf{MetricDistanceDepth}(source')+\mathsf{MetricDistanceDepth}(target').
\end{array}
$$
\end{theorem}
\begin{proof}
Unpack the displayed $\mathsf{ContinuousMapCarrier}$ into its
$\mathsf{ContinuousFunctionCarrier}$ half and its
$\mathsf{MetricDistanceWitness}$ half. Apply
\autoref{thm:continuous-function-carrier-hsame-transport} to the five function
fields $source,map,target,modulus,cert$ and their displayed $\hsame$ images.
This gives the transported continuous-function half.

For the metric half, apply
\autoref{thm:metric-distance-witness-hsame-fields-transport} to
$source,target,distance$ and $source',target',distance'$. The resulting unary
fields and transported continuation repack as
$\mathsf{MetricDistanceWitness}(source',target',distance')$.

Apply
\autoref{thm:metric-distance-witness-hsame-transport-cont-depth-add} to the
same three $\hsame$ witnesses and the original metric witness. It supplies the
displayed $\ContR(source',target',distance')$ and the displayed depth equation.
Pairing the transported continuous-function half with the transported metric
witness gives the carrier component; pairing this with the continuation and
depth ledger gives the stated conjunction.

The ten-field fit is: traditional target stable uniformly continuous map under
classifier transport; BEDC source $\mathsf{ContinuousFunctionCarrier}$,
$\mathsf{MetricDistanceWitness}$, and $\hsame$; carrier
$\mathsf{ContinuousMapCarrier}$; classifier $\hsame$; stability obligation
six-field transport plus transported metric depth ledger; ledger hidden
distance history; core theorem this block; exactness transported carrier and
distance-depth equation; standard bridge deferred; status
$\mathsf{ProofTarget}$.
\end{proof}

\begin{theorem}[ContinuousMap function carrier metric graph exactness]
\label{thm:continuousmap-function-carrier-metric-graph-exactness}
For histories $source,map,target,modulus,cert,dist:\mathsf{BHist}$, if
$$
\mathsf{ContinuousFunctionCarrier}(source,map,target,modulus,cert)
$$
and
$$
\mathsf{MetricDistanceWitness}(source,map,dist)
$$
both hold, then the carrier exposes the graph continuation and the metric
distance endpoint is exactly the same history as the graph target:
$$
\mathsf{Cont}(source,map,target)\land \hsame(dist,target).
$$
\end{theorem}
\begin{proof}
Read back the continuous-function carrier to obtain its graph continuation
$\mathsf{Cont}(source,map,target)$. The metric witness supplies the continuation
$\mathsf{Cont}(source,map,dist)$ over the same source and map. Continuation
determinacy identifies the two endpoints, giving $\hsame(dist,target)$.
\end{proof}

\begin{theorem}[ContinuousMap empty map with empty distance boundaries]
\label{thm:continuousmap-empty-map-empty-distance-boundaries}
For histories $source$, $target$, $modulus$, and $cert$, a concrete
continuous-map carrier with empty graph map and empty distance ledger is
equivalent to empty source and target boundaries together with the empty-source
modulus witness:
$$
\mathsf{ContinuousMapCarrier}(source,\emp,target,modulus,cert,\emp)
\Longleftrightarrow
\hsame(source,\emp)\land\hsame(target,\emp)\land
\mathsf{ContinuousModulusWitness}(\emp,modulus,cert).
$$
\end{theorem}
\begin{proof}
In the forward direction, empty-map exactness for the continuous-function
component gives a modulus witness over the target, while empty-distance
exactness for the metric witness collapses both endpoint histories to
$\emp$. Transporting across the target collapse presents the modulus witness
over $\emp$.

Conversely, the empty endpoint data presents the graph carrier as the
empty-map exactness case, and metric empty-distance exactness builds the
distance witness over the two empty endpoints. Their conjunction is exactly the
concrete continuous-map carrier.
\end{proof}
\leanchecked{BEDC.Derived.ContinuousMapUp.ContinuousMapCarrier\_empty\_map\_empty\_distance\_boundaries\_iff}
