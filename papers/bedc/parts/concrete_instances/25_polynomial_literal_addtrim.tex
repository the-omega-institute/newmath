\begin{proposition}[Literal polynomial add-trim equality fails at cancelling singletons]
\label{prop:polynomial-literal-add-trim-equality-fails-cancelling-singletons}
Let $R$ be a scalar ring carrier with classifier $\sim_R$, zero endpoint $0_R$, addition $+$, and canonical trim $\mathsf{trim}_R$ as in \autoref{def:polynomial-pattern-specification} and \autoref{def:polynomial-stability-certificate}. Let $\mathsf{radd}_R$ and $\mathsf{PolySame}_R$ be as in \autoref{def:polynomial-raw-add-comparison-data}. Suppose $\neg(a\sim_R0_R)$, $\neg(b\sim_R0_R)$, and $(a+b)\sim_R0_R$.
Write $[c]:=\mathsf{cons}(c,\mathsf{nil})$, and set $p:=[a]$ and $q:=[b]$. Then literal finite-spine equality does not commute with raw addition and trimming:
$$
\begin{aligned}
\mathsf{trim}_R(\mathsf{radd}_R(p,q))&=\mathsf{nil},\\
\mathsf{radd}_R(\mathsf{trim}_R(p),\mathsf{trim}_R(q))&=[a+b],\\
\mathsf{trim}_R(\mathsf{radd}_R(p,q))&\ne \mathsf{radd}_R(\mathsf{trim}_R(p),\mathsf{trim}_R(q)).
\end{aligned}
$$
Nevertheless the two displayed outputs are polynomial-classified the same:
$$
\mathsf{PolySame}_R\bigl(\mathsf{trim}_R(\mathsf{radd}_R(p,q)),\mathsf{radd}_R(\mathsf{trim}_R(p),\mathsf{trim}_R(q))\bigr).
$$
For the canonical integer coefficient ring from \autoref{ch:concrete-instances-int-namecert} and \autoref{ch:concrete-instances-commring-namecert}, taking $a=1$ and $b=-1$ gives $p=[1]$, $q=[-1]$, $\mathsf{trim}_{\mathbb{Z}}(\mathsf{radd}_{\mathbb{Z}}(p,q))=\mathsf{nil}$, and $\mathsf{radd}_{\mathbb{Z}}(\mathsf{trim}_{\mathbb{Z}}(p),\mathsf{trim}_{\mathbb{Z}}(q))=[0]$.
\end{proposition}
\begin{proof}
The nonzero hypotheses make $p$ and $q$ normalized singleton representatives by \autoref{def:polynomial-normalized-singleton-representative}; the canonical trim fixes normalized representatives, so $\mathsf{trim}_R(p)=p$ and $\mathsf{trim}_R(q)=q$.
By the cons-cons and nil-nil clauses of $\mathsf{radd}_R$ in \autoref{def:polynomial-raw-add-comparison-data}, $\mathsf{radd}_R(\mathsf{trim}_R(p),\mathsf{trim}_R(q))=\mathsf{radd}_R([a],[b])=[a+b]$.
The equality $\mathsf{trim}_R(\mathsf{radd}_R(p,q))=\mathsf{nil}$ is the singleton cancellation conclusion of \autoref{prop:raw-polynomial-addition-leaves-singleton-normal-form}.
The displayed inequality follows because the left side is $\mathsf{nil}$ while the right side is a cons spine. Constructor no-confusion for the finite-spine carrier of \autoref{def:list-carrier} is the mixed nil/cons separation isolated in \autoref{thm:list-classifier-case-exactness} and \autoref{lem:list-classifier-nil-cons-no-confusion}.
For classifier comparison, unfold $\mathsf{PolySame}_R$. The first argument trims again to $\mathsf{nil}$. The second argument is $[a+b]$, and the hypothesis $(a+b)\sim_R0_R$ sends its singleton coefficient to the trailing-zero branch, so its trim is also $\mathsf{nil}$. The nil-nil clause of $\mathsf{ListClassifierSpec}$ gives the displayed $\mathsf{PolySame}_R$ witness.
The integer specialization uses the standard integer facts $1\ne0$, $-1\ne0$, and $1+(-1)=0$; \autoref{prop:commring-forgets-ring-certificate} keeps the ring operations and classifier when the commutative-ring source is read through $\RingUp$.
\end{proof}

\begin{definition}[Polynomial Horner evaluation]
\label{def:polynomial-horner-evaluation}
Let $R$ be a scalar carrier with classifier $\sim_R$, zero endpoint $0_R$, addition $+$, multiplication $\cdot$, canonical trim $\mathsf{trim}_R$, and raw addition $\mathsf{radd}_R$ from \autoref{def:polynomial-raw-add-comparison-data}. For a carried scalar point $\alpha$, define the Horner evaluation on finite coefficient spines from \autoref{def:list-carrier} by recursion
$$
\mathsf{eval}_{\alpha,R}(\mathsf{nil}):=0_R.
$$
$$
\mathsf{eval}_{\alpha,R}(\mathsf{cons}(a,x)):=a+\alpha\cdot\mathsf{eval}_{\alpha,R}(x).
$$
Define the normalized polynomial evaluation by
$$
\mathsf{Eval}_{\alpha,R}(x):=\mathsf{eval}_{\alpha,R}(\mathsf{trim}_R(x)).
$$
Define the normalized polynomial sum by
$$
\mathsf{PolyAdd}_R(x,y):=\mathsf{trim}_R(\mathsf{radd}_R(\mathsf{trim}_R(x),\mathsf{trim}_R(y))).
$$
On a normalized representative from \autoref{def:polynomial-carrier}, $\mathsf{Eval}_{\alpha,R}$ is the Horner evaluation itself, by trim idempotence from \autoref{def:polynomial-stability-certificate}.
\end{definition}

\begin{lemma}[Horner evaluation ignores classified zero tails]
\label{lem:horner-evaluation-zero-tail-invariance}
Let $R$ carry a $\CommRingUp(R)$ certificate, read as a $\RingUp(R)$ certificate by \autoref{prop:commring-forgets-ring-certificate}. Fix a carried scalar point $\alpha$. For finite coefficient spines $x,t:\mathsf{ListCarrier}(R)$, if $\mathsf{ZeroRem}_R(t)$ holds in the sense of \autoref{def:classified-zero-remainder-spine}, then
$$
\mathsf{eval}_{\alpha,R}(x{++}t)\sim_R\mathsf{eval}_{\alpha,R}(x).
$$
Consequently, for every finite coefficient spine $x$,
$$
\mathsf{Eval}_{\alpha,R}(x)\sim_R\mathsf{eval}_{\alpha,R}(x).
$$
\end{lemma}
\begin{proof}
First prove the endpoint claim $\mathsf{eval}_{\alpha,R}(t)\sim_R0_R$ for every classified zero tail $t$. Induct on the displayed $\mathsf{ZeroRem}_R(t)$ witness. The nil clause is the defining equation $\mathsf{eval}_{\alpha,R}(\mathsf{nil})=0_R$ together with zero self-classification from \autoref{def:ring-stability-certificate}. In the cons clause, write $t=\mathsf{cons}(a,t^\prime)$. The constructor gives $a\sim_R0_R$ and the induction hypothesis gives $\mathsf{eval}_{\alpha,R}(t^\prime)\sim_R0_R$. Multiplication congruence and the zero-multiplication row in \autoref{def:ring-stability-certificate} classify $\alpha\cdot\mathsf{eval}_{\alpha,R}(t^\prime)$ with $0_R$. Addition congruence then classifies $a+\alpha\cdot\mathsf{eval}_{\alpha,R}(t^\prime)$ with $0_R+0_R$, and the additive left-zero law reduces the latter endpoint to $0_R$.

For the displayed tail-invariance statement, induct on $x$. The nil case is the endpoint claim just proved. In the cons case, write $x=\mathsf{cons}(a,x^\prime)$. The induction hypothesis gives $\mathsf{eval}_{\alpha,R}(x^\prime{++}t)\sim_R\mathsf{eval}_{\alpha,R}(x^\prime)$. Multiplication congruence transports this comparison through left multiplication by $\alpha$, and addition congruence with source reflexivity on $a$ gives
$$
a+\alpha\cdot\mathsf{eval}_{\alpha,R}(x^\prime{++}t)\sim_Ra+\alpha\cdot\mathsf{eval}_{\alpha,R}(x^\prime).
$$
The two Horner recursion equations identify these endpoints with $\mathsf{eval}_{\alpha,R}(\mathsf{cons}(a,x^\prime{++}t))$ and $\mathsf{eval}_{\alpha,R}(\mathsf{cons}(a,x^\prime))$.

For the consequence, the trailing-zero removal clause in \autoref{def:polynomial-stability-certificate} presents each raw spine $x$ as $\mathsf{trim}_R(x){++}t$ with $\mathsf{ZeroRem}_R(t)$. Applying the first part gives $\mathsf{eval}_{\alpha,R}(x)\sim_R\mathsf{Eval}_{\alpha,R}(x)$, and symmetry of the scalar classifier reverses the displayed comparison.
\end{proof}

\begin{lemma}[Horner evaluation respects raw polynomial addition]
\label{lem:horner-evaluation-raw-additive}
Let $R$ carry a $\CommRingUp(R)$ certificate, read as a $\RingUp(R)$ certificate by \autoref{prop:commring-forgets-ring-certificate}. Fix a carried scalar point $\alpha$. For finite coefficient spines $x,y:\mathsf{ListCarrier}(R)$,
$$
\mathsf{eval}_{\alpha,R}(\mathsf{radd}_R(x,y))\sim_R
\mathsf{eval}_{\alpha,R}(x)+\mathsf{eval}_{\alpha,R}(y).
$$
\end{lemma}
\begin{proof}
Perform induction on the two finite spines, following the four recursion clauses for $\mathsf{radd}_R$ in \autoref{def:polynomial-raw-add-comparison-data}.

The nil-nil case compares $0_R$ with $0_R+0_R$, so it is exactly the additive left-zero law, read with symmetry when needed.

In the cons-nil case, write $x=\mathsf{cons}(a,x^\prime)$. The raw-addition recursion exposes the head $a+0_R$ and the tail $\mathsf{radd}_R(x^\prime,\mathsf{nil})$. The induction hypothesis classifies the tail evaluation with $\mathsf{eval}_{\alpha,R}(x^\prime)+0_R$, and \autoref{thm:ring-add-right-zero} reduces that endpoint to $\mathsf{eval}_{\alpha,R}(x^\prime)$. Multiplication congruence transports the tail comparison through left multiplication by $\alpha$. Additive congruence and the additive right-zero law reduce the exposed head $a+0_R$ to $a$, and the target endpoint $(\mathsf{eval}_{\alpha,R}(\mathsf{cons}(a,x^\prime)))+0_R$ to $\mathsf{eval}_{\alpha,R}(\mathsf{cons}(a,x^\prime))$.

The nil-cons case is symmetric. The raw head is $0_R+b$, which reduces to $b$ by the additive left-zero law, and the outer target $0_R+\mathsf{eval}_{\alpha,R}(\mathsf{cons}(b,y^\prime))$ reduces by the same law. The tail comparison is transported through multiplication by $\alpha$ exactly as in the cons-nil case.

In the cons-cons case, write $x=\mathsf{cons}(a,x^\prime)$ and $y=\mathsf{cons}(b,y^\prime)$. The induction hypothesis gives
$$
\mathsf{eval}_{\alpha,R}(\mathsf{radd}_R(x^\prime,y^\prime))\sim_R
\mathsf{eval}_{\alpha,R}(x^\prime)+\mathsf{eval}_{\alpha,R}(y^\prime).
$$
Multiplication congruence transports this through left multiplication by $\alpha$, and left distributivity from \autoref{def:ring-stability-certificate} classifies
$$
\alpha\cdot(\mathsf{eval}_{\alpha,R}(x^\prime)+\mathsf{eval}_{\alpha,R}(y^\prime))
$$
with
$$
\alpha\cdot\mathsf{eval}_{\alpha,R}(x^\prime)+\alpha\cdot\mathsf{eval}_{\alpha,R}(y^\prime).
$$
Addition congruence therefore transports the Horner endpoint for $\mathsf{radd}_R(x,y)$ to
$$
(a+b)+(\alpha\cdot\mathsf{eval}_{\alpha,R}(x^\prime)+\alpha\cdot\mathsf{eval}_{\alpha,R}(y^\prime)).
$$
Apply \autoref{thm:abgroup-middle-four-interchange} to the additive abelian-group operation, with the four additive blocks $a$, $b$, $\alpha\cdot\mathsf{eval}_{\alpha,R}(x^\prime)$, and $\alpha\cdot\mathsf{eval}_{\alpha,R}(y^\prime)$. This classifies the last endpoint with
$$
(a+\alpha\cdot\mathsf{eval}_{\alpha,R}(x^\prime))+(b+\alpha\cdot\mathsf{eval}_{\alpha,R}(y^\prime)),
$$
which is exactly $\mathsf{eval}_{\alpha,R}(x)+\mathsf{eval}_{\alpha,R}(y)$ by the Horner recursion.
\end{proof}

\begin{theorem}[Polynomial evaluation is additive]
\label{thm:polynomial-evaluation-additive-homomorphism}
Let $R$ carry a $\CommRingUp(R)$ certificate, read as a $\RingUp(R)$ certificate by \autoref{prop:commring-forgets-ring-certificate}. Fix a carried scalar point $\alpha$. For finite coefficient spines $p,q:\mathsf{ListCarrier}(R)$,
$$
\mathsf{Eval}_{\alpha,R}(\mathsf{PolyAdd}_R(p,q))\sim_R
\mathsf{Eval}_{\alpha,R}(p)+\mathsf{Eval}_{\alpha,R}(q).
$$
If $p$ and $q$ are normalized representatives in the polynomial carrier, the same assertion is the carrier formula
$$
\mathsf{eval}_{\alpha,R}(\mathsf{PolyAdd}_R(p,q))\sim_R
\mathsf{eval}_{\alpha,R}(p)+\mathsf{eval}_{\alpha,R}(q).
$$
\end{theorem}
\begin{proof}
Put $p_0=\mathsf{trim}_R(p)$ and $q_0=\mathsf{trim}_R(q)$. By \autoref{def:polynomial-horner-evaluation},
$$
\mathsf{PolyAdd}_R(p,q)=\mathsf{trim}_R(\mathsf{radd}_R(p_0,q_0)).
$$
Trim idempotence from \autoref{def:polynomial-stability-certificate} identifies $\mathsf{Eval}_{\alpha,R}(\mathsf{PolyAdd}_R(p,q))$ with $\mathsf{Eval}_{\alpha,R}(\mathsf{radd}_R(p_0,q_0))$. Applying \autoref{lem:horner-evaluation-zero-tail-invariance} gives
$$
\mathsf{Eval}_{\alpha,R}(\mathsf{radd}_R(p_0,q_0))\sim_R
\mathsf{eval}_{\alpha,R}(\mathsf{radd}_R(p_0,q_0)).
$$
By \autoref{lem:horner-evaluation-raw-additive},
$$
\mathsf{eval}_{\alpha,R}(\mathsf{radd}_R(p_0,q_0))\sim_R
\mathsf{eval}_{\alpha,R}(p_0)+\mathsf{eval}_{\alpha,R}(q_0).
$$
The right endpoint is $\mathsf{Eval}_{\alpha,R}(p)+\mathsf{Eval}_{\alpha,R}(q)$ by the definition of $p_0$, $q_0$, and normalized evaluation. Classifier transitivity composes the displayed comparisons.

If $p$ and $q$ are already normalized polynomial-carrier representatives, trim idempotence gives $p_0=p$ and $q_0=q$, and $\mathsf{Eval}_{\alpha,R}$ on each displayed representative is the unnormalized Horner evaluation. Substituting these equalities gives the carrier formula.
\end{proof}

\begin{lemma}[Horner evaluation respects structural classifier]
\label{lem:horner-evaluation-structural-classifier}
Let $R$ carry a $\CommRingUp(R)$ certificate, read as a $\RingUp(R)$ certificate by \autoref{prop:commring-forgets-ring-certificate}. Fix a carried scalar point $\alpha$. For finite coefficient spines $x,y:\mathsf{ListCarrier}(R)$, if
$$
\mathsf{ListClassifierSpec}(\sim_R,x,y)
$$
holds, then
$$
\mathsf{eval}_{\alpha,R}(x)\sim_R\mathsf{eval}_{\alpha,R}(y).
$$
\end{lemma}
\begin{proof}
Induct on the displayed structural classifier witness from \autoref{def:list-classifier-specification}. By \autoref{thm:list-classifier-case-exactness}, the only constructor cases are nil-nil and cons-cons; mixed nil-cons shapes have no classifier witness.

In the nil-nil case, the two Horner equations in \autoref{def:polynomial-horner-evaluation} identify both endpoints with $0_R$. The zero self-classification field retained in \autoref{def:ring-stability-certificate} gives $0_R\sim_R0_R$.

In the cons-cons case, write $x=\mathsf{cons}(a,x')$ and $y=\mathsf{cons}(b,y')$. The classifier constructor supplies $a\sim_R b$ and a tail witness $\mathsf{ListClassifierSpec}(\sim_R,x',y')$. The induction hypothesis gives
$$
\mathsf{eval}_{\alpha,R}(x')\sim_R\mathsf{eval}_{\alpha,R}(y').
$$
The carried point $\alpha$ is self-classified by the scalar certificate. Multiplication congruence from \autoref{def:ring-stability-certificate} transports the tail comparison through left multiplication by $\alpha$, giving
$$
\alpha\cdot\mathsf{eval}_{\alpha,R}(x')\sim_R
\alpha\cdot\mathsf{eval}_{\alpha,R}(y').
$$
Addition congruence combines this comparison with the head comparison $a\sim_Rb$:
$$
a+\alpha\cdot\mathsf{eval}_{\alpha,R}(x')\sim_R
b+\alpha\cdot\mathsf{eval}_{\alpha,R}(y').
$$
The cons recursion equation in \autoref{def:polynomial-horner-evaluation} identifies these two endpoints with $\mathsf{eval}_{\alpha,R}(\mathsf{cons}(a,x'))$ and $\mathsf{eval}_{\alpha,R}(\mathsf{cons}(b,y'))$, respectively.
\end{proof}

\begin{theorem}[Polynomial evaluation respects the polynomial classifier]
\label{thm:polynomial-evaluation-respects-polynomial-classifier}
Let $R$ carry a $\CommRingUp(R)$ certificate, read as a $\RingUp(R)$ certificate by \autoref{prop:commring-forgets-ring-certificate}. Fix a carried scalar point $\alpha$. For finite coefficient spines $p,q:\mathsf{ListCarrier}(R)$,
$$
\mathsf{PolySame}_R(p,q)
\Longrightarrow
\mathsf{Eval}_{\alpha,R}(p)\sim_R\mathsf{Eval}_{\alpha,R}(q).
$$
\end{theorem}
\begin{proof}
Unfold $\mathsf{PolySame}_R$ using \autoref{def:polynomial-raw-add-comparison-data}. A classifier witness for $p$ and $q$ is exactly
$$
\mathsf{ListClassifierSpec}(\sim_R,\mathsf{trim}_R(p),\mathsf{trim}_R(q)).
$$
Apply \autoref{lem:horner-evaluation-structural-classifier} to the two trimmed spines. This gives
$$
\mathsf{eval}_{\alpha,R}(\mathsf{trim}_R(p))\sim_R
\mathsf{eval}_{\alpha,R}(\mathsf{trim}_R(q)).
$$
The normalized evaluation definition in \autoref{def:polynomial-horner-evaluation} reads the left endpoint as $\mathsf{Eval}_{\alpha,R}(p)$ and the right endpoint as $\mathsf{Eval}_{\alpha,R}(q)$, giving the displayed implication.
\end{proof}

\begin{definition}[Polynomial raw Cauchy product]
\label{def:polynomial-raw-cauchy-product}
Let $R$ be a scalar carrier with classifier $\sim_R$, zero endpoint $0_R$, addition $+$, multiplication $\cdot$, canonical trim $\mathsf{trim}_R$, and raw addition $\mathsf{radd}_R$ from \autoref{def:polynomial-raw-add-comparison-data}. For a scalar coefficient $a$ and a finite coefficient spine $y:\mathsf{ListCarrier}(R)$, define scalar multiplication of coefficient spines by
$$
\mathsf{rscale}_R(a,\mathsf{nil}):=\mathsf{nil},
\qquad
\mathsf{rscale}_R(a,\mathsf{cons}(b,y)):=\mathsf{cons}(a\cdot b,\mathsf{rscale}_R(a,y)).
$$
Define the one-place coefficient shift by
$$
\mathsf{shift}_R(y):=\mathsf{cons}(0_R,y).
$$
The raw polynomial product is the finite recursion
$$
\mathsf{rmul}_R(\mathsf{nil},y):=\mathsf{nil},
$$
$$
\mathsf{rmul}_R(\mathsf{cons}(a,x),y):=
\mathsf{radd}_R(\mathsf{rscale}_R(a,y),\mathsf{shift}_R(\mathsf{rmul}_R(x,y))).
$$
The normalized polynomial product is
$$
\mathsf{PolyMul}_R(x,y):=\mathsf{trim}_R(\mathsf{rmul}_R(\mathsf{trim}_R(x),\mathsf{trim}_R(y))).
$$
This is the Cauchy product in low-coefficient-first Horner form: the cons clause expresses $(a+Xx)y=ay+X(xy)$.
\end{definition}

\begin{lemma}[Horner evaluation of scalar-multiplied coefficient spines]
\label{lem:horner-evaluation-raw-scalar-product}
Let $R$ carry a $\CommRingUp(R)$ certificate, read as a $\RingUp(R)$ certificate by \autoref{prop:commring-forgets-ring-certificate}. Fix a carried scalar point $\alpha$. For every carried scalar coefficient $a$ and every finite coefficient spine $y:\mathsf{ListCarrier}(R)$,
$$
\mathsf{eval}_{\alpha,R}(\mathsf{rscale}_R(a,y))\sim_R
 a\cdot\mathsf{eval}_{\alpha,R}(y).
$$
\end{lemma}
\begin{proof}
Induct on the finite spine $y$ from \autoref{def:list-carrier}. In the nil case, both the scalar-multiple spine and its Horner value are $0_R$. The target endpoint is $a\cdot0_R$, and the zero-multiplication row in \autoref{def:ring-stability-certificate}, with symmetry of $\sim_R$, gives the displayed comparison.

In the cons case, write $y=\mathsf{cons}(b,y')$. The recursion in \autoref{def:polynomial-raw-cauchy-product} and the Horner recursion in \autoref{def:polynomial-horner-evaluation} identify the left endpoint with
$$
a\cdot b+\alpha\cdot\mathsf{eval}_{\alpha,R}(\mathsf{rscale}_R(a,y')).
$$
The induction hypothesis and multiplication congruence classify this endpoint with
$$
a\cdot b+\alpha\cdot(a\cdot\mathsf{eval}_{\alpha,R}(y')).
$$
Multiplicative associativity, multiplicative commutativity from \autoref{def:commring-stability-certificate}, and multiplication congruence classify
$$
\alpha\cdot(a\cdot\mathsf{eval}_{\alpha,R}(y'))
$$
with
$$
a\cdot(\alpha\cdot\mathsf{eval}_{\alpha,R}(y')).
$$
Addition congruence transports this comparison under the common summand $a\cdot b$. The left distributive row in \autoref{def:ring-stability-certificate}, used in the reverse direction, then identifies
$$
a\cdot b+a\cdot(\alpha\cdot\mathsf{eval}_{\alpha,R}(y'))
$$
with
$$
a\cdot(b+\alpha\cdot\mathsf{eval}_{\alpha,R}(y')).
$$
The final endpoint is $a\cdot\mathsf{eval}_{\alpha,R}(\mathsf{cons}(b,y'))$ by the Horner recursion, and transitivity composes the comparisons.
\end{proof}

\begin{lemma}[Horner evaluation of coefficient shift]
\label{lem:horner-evaluation-coefficient-shift}
Let $R$ carry a $\CommRingUp(R)$ certificate, read as a $\RingUp(R)$ certificate by \autoref{prop:commring-forgets-ring-certificate}. Fix a carried scalar point $\alpha$. For every finite coefficient spine $x:\mathsf{ListCarrier}(R)$,

\mathsf{eval}_{\alpha,R}(\mathsf{shift}_R(x))\sim_R
\alpha\cdot\mathsf{eval}_{\alpha,R}(x).
$$
\end{lemma}
\begin{proof}
By \autoref{def:polynomial-raw-cauchy-product}, $\mathsf{shift}_R(x)=\mathsf{cons}(0_R,x)$. The Horner recursion from \autoref{def:polynomial-horner-evaluation} makes its evaluation
$$
0_R+\alpha\cdot\mathsf{eval}_{\alpha,R}(x).
$$
The additive left-zero row in \autoref{def:ring-stability-certificate} classifies this endpoint with $\alpha\cdot\mathsf{eval}_{\alpha,R}(x)$.
\end{proof}

\begin{lemma}[Horner evaluation respects raw polynomial multiplication]
\label{lem:horner-evaluation-raw-multiplicative}
Let $R$ carry a $\CommRingUp(R)$ certificate, read as a $\RingUp(R)$ certificate by \autoref{prop:commring-forgets-ring-certificate}. Fix a carried scalar point $\alpha$. For finite coefficient spines $x,y:\mathsf{ListCarrier}(R)$,
$$
\mathsf{eval}_{\alpha,R}(\mathsf{rmul}_R(x,y))\sim_R
\mathsf{eval}_{\alpha,R}(x)\cdot\mathsf{eval}_{\alpha,R}(y).
$$
\end{lemma}
\begin{proof}
Induct on the first spine $x$. In the nil case, $\mathsf{rmul}_R(\mathsf{nil},y)=\mathsf{nil}$, so the left endpoint evaluates to $0_R$. The target endpoint is $0_R\cdot\mathsf{eval}_{\alpha,R}(y)$, and the zero-multiplication row in \autoref{def:ring-stability-certificate}, with symmetry, gives the comparison.

For the cons case, write $x=\mathsf{cons}(a,x')$, put $u=\mathsf{eval}_{\alpha,R}(x')$, and put $v=\mathsf{eval}_{\alpha,R}(y)$. The raw product recursion in \autoref{def:polynomial-raw-cauchy-product} and the raw additivity theorem \autoref{lem:horner-evaluation-raw-additive} give
$$
\mathsf{eval}_{\alpha,R}(\mathsf{rmul}_R(\mathsf{cons}(a,x'),y))\sim_R
\mathsf{eval}_{\alpha,R}(\mathsf{rscale}_R(a,y))+
\mathsf{eval}_{\alpha,R}(\mathsf{shift}_R(\mathsf{rmul}_R(x',y))).
$$
By \autoref{lem:horner-evaluation-raw-scalar-product}, the first summand is classified with $a\cdot v$. By \autoref{lem:horner-evaluation-coefficient-shift}, the second summand is classified with
$$
\alpha\cdot\mathsf{eval}_{\alpha,R}(\mathsf{rmul}_R(x',y)).
$$
The induction hypothesis and multiplication congruence classify this with $\alpha\cdot(u\cdot v)$. Multiplicative associativity classifies $\alpha\cdot(u\cdot v)$ with $(\alpha\cdot u)\cdot v$. Addition congruence therefore gives
$$
\mathsf{eval}_{\alpha,R}(\mathsf{rmul}_R(\mathsf{cons}(a,x'),y))\sim_R
 a\cdot v+(\alpha\cdot u)\cdot v.
$$
The right distributive row retained by \autoref{prop:commring-forgets-ring-certificate}, equivalently derived from \autoref{thm:commring-right-distrib-from-left}, classifies
$$
(a+\alpha\cdot u)\cdot v
$$
with
$$
a\cdot v+(\alpha\cdot u)\cdot v.
$$
Using symmetry of this distributive comparison and the Horner recursion
$$
\mathsf{eval}_{\alpha,R}(\mathsf{cons}(a,x'))=a+\alpha\cdot u,
$$
transitivity gives the required cons-case endpoint.
\end{proof}

\begin{theorem}[Polynomial evaluation is multiplicative]
\label{thm:polynomial-evaluation-multiplicative-homomorphism}
Let $R$ carry a $\CommRingUp(R)$ certificate, read as a $\RingUp(R)$ certificate by \autoref{prop:commring-forgets-ring-certificate}. Fix a carried scalar point $\alpha$. For finite coefficient spines $p,q:\mathsf{ListCarrier}(R)$,
$$
\mathsf{Eval}_{\alpha,R}(\mathsf{PolyMul}_R(p,q))\sim_R
\mathsf{Eval}_{\alpha,R}(p)\cdot\mathsf{Eval}_{\alpha,R}(q).
$$
If $p$ and $q$ are normalized representatives in the polynomial carrier, the same assertion is the carrier formula
$$
\mathsf{eval}_{\alpha,R}(\mathsf{PolyMul}_R(p,q))\sim_R
\mathsf{eval}_{\alpha,R}(p)\cdot\mathsf{eval}_{\alpha,R}(q).
$$
\end{theorem}
\begin{proof}
Put $p_0=\mathsf{trim}_R(p)$ and $q_0=\mathsf{trim}_R(q)$. By \autoref{def:polynomial-raw-cauchy-product},
$$
\mathsf{PolyMul}_R(p,q)=\mathsf{trim}_R(\mathsf{rmul}_R(p_0,q_0)).
$$
Trim idempotence from \autoref{def:polynomial-stability-certificate} identifies the normalized evaluation of this endpoint with
$$
\mathsf{Eval}_{\alpha,R}(\mathsf{rmul}_R(p_0,q_0)).
$$
Applying \autoref{lem:horner-evaluation-zero-tail-invariance} gives
$$
\mathsf{Eval}_{\alpha,R}(\mathsf{rmul}_R(p_0,q_0))\sim_R
\mathsf{eval}_{\alpha,R}(\mathsf{rmul}_R(p_0,q_0)).
$$
By \autoref{lem:horner-evaluation-raw-multiplicative},
$$
\mathsf{eval}_{\alpha,R}(\mathsf{rmul}_R(p_0,q_0))\sim_R
\mathsf{eval}_{\alpha,R}(p_0)\cdot\mathsf{eval}_{\alpha,R}(q_0).
$$
The right endpoint is $\mathsf{Eval}_{\alpha,R}(p)\cdot\mathsf{Eval}_{\alpha,R}(q)$ by the definition of $p_0$, $q_0$, and normalized evaluation in \autoref{def:polynomial-horner-evaluation}. Classifier transitivity composes the displayed comparisons.

If $p$ and $q$ are already normalized polynomial-carrier representatives, trim idempotence gives $p_0=p$ and $q_0=q$. The product endpoint is normalized by its defining trim, so normalized evaluation on $\mathsf{PolyMul}_R(p,q)$ is the unnormalized Horner evaluation of that product. Substituting these equalities gives the carrier formula.
\end{proof}

\begin{lemma}[Zero-point Horner cons evaluation]
\label{lem:horner-evaluation-zero-cons-constant}
Let $R$ carry a $\RingUp(R)$ certificate with classifier $\sim_R$, zero endpoint $0_R$, addition $+$, multiplication $\cdot$, and Horner evaluation as in \autoref{def:polynomial-horner-evaluation}. For every carried scalar coefficient $a$ and every finite coefficient spine $t:\mathsf{ListCarrier}(R)$,
$$
\mathsf{eval}_{0_R,R}(\mathsf{cons}(a,t))\sim_R a.
$$
\end{lemma}
\begin{proof}
The cons recursion in \autoref{def:polynomial-horner-evaluation} identifies the left endpoint with
$$
a+0_R\cdot\mathsf{eval}_{0_R,R}(t).
$$
The zero-absorption theorem \autoref{thm:ring-zero-absorption-from-distributivity}, applied to $\mathsf{eval}_{0_R,R}(t)$, gives
$$
0_R\cdot\mathsf{eval}_{0_R,R}(t)\sim_R0_R.
$$
Classifier reflexivity supplies $a\sim_Ra$, and addition congruence from \autoref{def:ring-stability-certificate} gives
$$
a+0_R\cdot\mathsf{eval}_{0_R,R}(t)\sim_Ra+0_R.
$$
The additive right-zero theorem \autoref{thm:ring-add-right-zero} gives $a+0_R\sim_Ra$. Classifier transitivity composes the two comparisons.
\end{proof}

\begin{lemma}[Zero-point Horner evaluation ignores classified zero tails]
\label{lem:horner-evaluation-zero-tail-invariance-at-zero}
Let $R$ carry a $\RingUp(R)$ certificate with classifier $\sim_R$, zero endpoint $0_R$, addition $+$, multiplication $\cdot$, and Horner evaluation as in \autoref{def:polynomial-horner-evaluation}. For finite coefficient spines $x,t:\mathsf{ListCarrier}(R)$, if $\mathsf{ZeroRem}_R(t)$ holds in the sense of \autoref{def:classified-zero-remainder-spine}, then
$$
\mathsf{eval}_{0_R,R}(x{++}t)\sim_R\mathsf{eval}_{0_R,R}(x).
$$
\end{lemma}
\begin{proof}
First record the endpoint claim that every classified zero remainder evaluates to zero at $0_R$. If the zero remainder is $\mathsf{nil}$, the nil recursion in \autoref{def:polynomial-horner-evaluation} identifies its evaluation with $0_R$, and zero self-classification from \autoref{def:ring-stability-certificate} gives the claim. If the zero remainder is $\mathsf{cons}(c,t')$, the cons clause of \autoref{def:classified-zero-remainder-spine} supplies $c\sim_R0_R$. By \autoref{lem:horner-evaluation-zero-cons-constant},
$$
\mathsf{eval}_{0_R,R}(\mathsf{cons}(c,t'))\sim_Rc,
$$
and transitivity gives $\mathsf{eval}_{0_R,R}(\mathsf{cons}(c,t'))\sim_R0_R$.

Now split on the first constructor of $x$. If $x=\mathsf{nil}$, then $x{++}t=t$, so the endpoint claim gives the displayed comparison. If $x=\mathsf{cons}(a,x')$, then $x{++}t=\mathsf{cons}(a,x'{++}t)$. Applying \autoref{lem:horner-evaluation-zero-cons-constant} twice gives
$$
\mathsf{eval}_{0_R,R}(\mathsf{cons}(a,x'{++}t))\sim_Ra
$$
and
$$
\mathsf{eval}_{0_R,R}(\mathsf{cons}(a,x'))\sim_Ra.
$$
Symmetry of the second comparison and transitivity give the required comparison for the cons case.
\end{proof}

\begin{theorem}[Polynomial evaluation at zero returns the constant coefficient]
\label{thm:polynomial-evaluation-at-zero-constant-term}

\begin{lemma}[Polynomial raw addition structural swap]
\label{lem:polynomial-raw-add-structural-swap}
Let $R$ be a scalar carrier with classifier $\sim_R$, zero endpoint $0_R$, and addition $+$. Assume scalar additive commutativity:
$$
a+b\sim_R b+a
$$
for all scalar coefficients $a,b$. For finite coefficient spines $p,q:\mathsf{ListCarrier}(R)$,
$$
\mathsf{ListClassifierSpec}(\sim_R,\mathsf{radd}_R(p,q),\mathsf{radd}_R(q,p)).
$$
\end{lemma}
\begin{proof}
Induct simultaneously on the two finite coefficient spines, following the four recursion clauses of $\mathsf{radd}_R$ in \autoref{def:polynomial-raw-add-comparison-data}. The target classifier is the finite-spine structural classifier of \autoref{def:list-classifier-specification}, with matching constructors isolated by \autoref{thm:list-classifier-case-exactness}.

In the nil-nil case, both raw additions are $\mathsf{nil}$, so the nil clause of $\mathsf{ListClassifierSpec}$ gives the claim.

In the cons-nil case, write $p=\mathsf{cons}(a,p')$ and $q=\mathsf{nil}$. The recursion gives
$$
\mathsf{radd}_R(p,q)=\mathsf{cons}(a+0_R,\mathsf{radd}_R(p',\mathsf{nil}))
$$
and
$$
\mathsf{radd}_R(q,p)=\mathsf{cons}(0_R+a,\mathsf{radd}_R(\mathsf{nil},p')).
$$
Scalar additive commutativity gives $a+0_R\sim_R0_R+a$. The induction hypothesis applied to $p'$ and $\mathsf{nil}$ gives the tail classifier
$$
\mathsf{ListClassifierSpec}(\sim_R,\mathsf{radd}_R(p',\mathsf{nil}),\mathsf{radd}_R(\mathsf{nil},p')).
$$
The cons clause of $\mathsf{ListClassifierSpec}$ combines the head comparison and this tail comparison.

In the nil-cons case, write $p=\mathsf{nil}$ and $q=\mathsf{cons}(b,q')$. The recursion gives heads $0_R+b$ and $b+0_R$. Scalar additive commutativity gives $0_R+b\sim_Rb+0_R$, and the induction hypothesis at $\mathsf{nil}$ and $q'$ supplies the tail classifier. The cons clause rebuilds the displayed list-classifier witness.

In the cons-cons case, write $p=\mathsf{cons}(a,p')$ and $q=\mathsf{cons}(b,q')$. The recursion exposes heads $a+b$ and $b+a$, related by scalar additive commutativity. The induction hypothesis supplies
$$
\mathsf{ListClassifierSpec}(\sim_R,\mathsf{radd}_R(p',q'),\mathsf{radd}_R(q',p')).
$$
The cons clause combines the head and tail witnesses, giving the structural classifier for the two raw sums.
\end{proof}

\begin{lemma}[Raw polynomial addition is structurally associative]
\label{lem:polynomial-raw-add-structural-associativity}
Let $R$ be a scalar carrier with classifier $\sim_R$, zero endpoint $0_R$, addition $+$, and scalar certificate $\mathsf{cert}_R:\NameCert(R,\sim_R)$. Assume the additive fields retained in \autoref{def:ring-stability-certificate}: additive associativity, additive zero laws, classifier symmetry, classifier transitivity, and addition congruence. For finite coefficient spines $p,q,r:\mathsf{ListCarrier}(R)$,
$$
\mathsf{ListClassifierSpec}\bigl(\sim_R,
\mathsf{radd}_R(\mathsf{radd}_R(p,q),r),
\mathsf{radd}_R(p,\mathsf{radd}_R(q,r))\bigr).
$$
\end{lemma}
\begin{proof}
Induct on the finite triple spine $(p,q,r)$, using the maximum of the three finite lengths as the decreasing measure. The nil-nil-nil case unfolds both raw additions by \autoref{def:polynomial-raw-add-comparison-data} to $\mathsf{nil}$, and the nil clause of \autoref{def:list-classifier-specification} gives the classifier witness.

In every remaining constructor case, at least one of the three spines is a cons spine. Unfold the corresponding clauses of $\mathsf{radd}_R$ from \autoref{def:polynomial-raw-add-comparison-data}. Both associated raw sums expose a cons head, and their tails are exactly the two associated raw sums for the tail triple, with $\mathsf{nil}$ used as the tail of any absent spine. The induction hypothesis supplies the recursive tail classifier.

It remains to classify the exposed scalar heads. If all three heads are present, say $a,b,c$, the comparison is
$$
(a+b)+c\sim_R a+(b+c),
$$
which is scalar additive associativity. If exactly two heads are present, the missing head is the padding coefficient $0_R$, so the three possible comparisons
$$
(a+b)+0_R\sim_R a+(b+0_R),
$$
$$
(a+0_R)+c\sim_R a+(0_R+c),
$$
$$
(0_R+b)+c\sim_R 0_R+(b+c)
$$
are the same associativity law with one displayed argument equal to $0_R$. If only the left input has a head $a$, the comparison is
$$
(a+0_R)+0_R\sim_R a+0_R,
$$
which is the additive right-zero law applied to $a+0_R$. If only the middle input has a head $b$, the comparison is
$$
(0_R+b)+0_R\sim_R 0_R+(b+0_R),
$$
again additive associativity. If only the right input has a head $c$, the comparison is
$$
0_R+c\sim_R0_R+(0_R+c),
$$
obtained by reversing the additive left-zero law at the endpoint $0_R+c$ using classifier symmetry from $\mathsf{cert}_R$.

In each constructor case, the scalar head comparison and the tail classifier rebuild the cons clause of $\mathsf{ListClassifierSpec}$. These cases exhaust the finite-spine constructors by \autoref{thm:list-classifier-case-exactness}.
\end{proof}

\begin{theorem}[Polynomial raw multiplication commutativity from scalar commutativity]
\label{thm:polynomial-raw-multiplication-commutativity-from-commring-scalars}
Let $R$ carry a $\CommRingUp(R)$ certificate with classifier $\sim_R$, zero endpoint $0_R$, addition $+$, multiplication $\cdot$, scalar certificate $\mathsf{cert}_R:\NameCert(R,\sim_R)$, canonical trim $\mathsf{trim}_R$, and the raw operations $\mathsf{radd}_R$, $\mathsf{rscale}_R$, $\mathsf{shift}_R$, and $\mathsf{rmul}_R$ from \autoref{def:polynomial-raw-add-comparison-data} and \autoref{def:polynomial-raw-cauchy-product}. For finite coefficient spines $p,q:\mathsf{ListCarrier}(R)$,
$$
\mathsf{PolySame}_R(\mathsf{rmul}_R(p,q),\mathsf{rmul}_R(q,p)).
$$
\end{theorem}
\begin{proof}
Read the scalar source as a $\RingUp(R)$ source by \autoref{prop:commring-forgets-ring-certificate}. The retained rows used below are the $\NameCert$ reflexivity, symmetry, and transitivity fields, the additive abelian-group laws and addition congruence from \autoref{def:ring-stability-certificate}, multiplication congruence, and the multiplicative commutativity row from \autoref{def:commring-stability-certificate}.

First record the finite-spine consequences of the existing raw-addition package. The relation $\mathsf{PolySame}_R$ is reflexive, symmetric, and transitive after unfolding \autoref{def:polynomial-raw-add-comparison-data}; this is exactly structural list-classifier reflexivity, \autoref{thm:list-classifier-symm-from-namecert}, and \autoref{thm:list-classifier-transitivity} applied to the two trimmed representatives.

Raw addition is compatible with $\mathsf{PolySame}_R$ in both inputs. Indeed, \autoref{thm:polynomial-raw-add-two-sided-trim-compatibility} reduces both raw sums to sums of their trimmed inputs. The two displayed trimmed-input classifiers give a structural classifier between the two trimmed raw sums by induction on the two list-classifier witnesses in \autoref{def:list-classifier-specification}; the exposed heads are handled by addition congruence, with $0_R\sim_R0_R$ in the padding branches. Transitivity and symmetry of $\mathsf{PolySame}_R$ compose these three comparisons.

The shift operation preserves $\mathsf{PolySame}_R$. If $\mathsf{PolySame}_R(x,y)$ holds, write the trim decompositions supplied by \autoref{def:polynomial-stability-certificate} as $x=x_0{++}z$ and $y=y_0{++}w$, with $\mathsf{ListClassifierSpec}(\sim_R,x_0,y_0)$ and zero remainders $z,w$. Then
$$
\mathsf{shift}_R(x)=(\mathsf{cons}(0_R,x_0)){++}z,
\qquad
\mathsf{shift}_R(y)=(\mathsf{cons}(0_R,y_0)){++}w.
$$
The cons classifier built from $0_R\sim_R0_R$ and the classifier on $x_0,y_0$, followed by \autoref{lem:trim-invariance-common-classified-spine-zero-remainders}, gives $\mathsf{PolySame}_R(\mathsf{shift}_R(x),\mathsf{shift}_R(y))$. The same argument, with $0_R$ replaced by classified heads $h\sim_Rh'$, shows that $h\sim_Rh'$ and $\mathsf{PolySame}_R(t,t')$ imply
$$
\mathsf{PolySame}_R(\mathsf{cons}(h,t),\mathsf{cons}(h',t')).
$$

Zero remainders are closed under raw addition: induct on the two zero-remainder witnesses from \autoref{def:classified-zero-remainder-spine}. In the cons-cons branch, addition congruence sends $a+b$ to $0_R+0_R$, and the additive left-zero law reduces that endpoint to $0_R$; the cons-nil and nil-cons branches use the additive right-zero and left-zero laws. Hence $\mathsf{rmul}_R(s,\mathsf{nil})$ is a zero remainder for every finite spine $s$, by induction on $s$ using \autoref{def:polynomial-raw-cauchy-product}: the cons step combines $\mathsf{rscale}_R(a,\mathsf{nil})=\mathsf{nil}$, the zero-remainder closure of $\mathsf{shift}_R$, and raw-addition closure. Applying \autoref{lem:trim-invariance-common-classified-spine-zero-remainders} with common spine $\mathsf{nil}$ gives
$$
\mathsf{PolySame}_R(\mathsf{nil},\mathsf{rmul}_R(s,\mathsf{nil})).
$$

The key auxiliary recursion is right-cons expansion: for every scalar coefficient $a$ and finite spines $x,q$,
$$
\mathsf{PolySame}_R\bigl(\mathsf{rmul}_R(q,\mathsf{cons}(a,x)),
\mathsf{radd}_R(\mathsf{rscale}_R(a,q),\mathsf{shift}_R(\mathsf{rmul}_R(q,x)))\bigr).
$$
Induct on $q$. If $q=\mathsf{nil}$, the left side is $\mathsf{nil}$ and the right side is a raw sum of zero-remainder spines, so the zero-remainder comparison above applies. For the cons step write $q=\mathsf{cons}(b,y)$. Unfolding \autoref{def:polynomial-raw-cauchy-product} and the cons-cons branch of \autoref{def:polynomial-raw-add-comparison-data}, the two heads are $b\cdot a+0_R$ and $a\cdot b+0_R$; multiplicative commutativity and addition congruence classify them. The two tails are
$$
\mathsf{radd}_R(\mathsf{rscale}_R(b,x),\mathsf{rmul}_R(y,\mathsf{cons}(a,x)))
$$
and
$$
\mathsf{radd}_R(\mathsf{rscale}_R(a,y),\mathsf{radd}_R(\mathsf{rscale}_R(b,x),\mathsf{shift}_R(\mathsf{rmul}_R(y,x)))).
$$
The induction hypothesis and raw-addition compatibility compare the first tail with
$$
\mathsf{radd}_R(\mathsf{rscale}_R(b,x),\mathsf{radd}_R(\mathsf{rscale}_R(a,y),\mathsf{shift}_R(\mathsf{rmul}_R(y,x)))).
$$
Raw-addition associativity from \autoref{thm:polynomial-raw-addition-associativity} and raw-addition commutativity from \autoref{thm:polynomial-raw-add-commutativity-from-scalar-additive-commutativity}, transported through raw-addition compatibility, move the first two summands past each other and give the second tail. The cons preservation fact then proves the cons step.

Now induct on $p$. If $p=\mathsf{nil}$, then $\mathsf{rmul}_R(\mathsf{nil},q)=\mathsf{nil}$ by \autoref{def:polynomial-raw-cauchy-product}, while $\mathsf{rmul}_R(q,\mathsf{nil})$ is a zero remainder by the preceding paragraph; hence the displayed $\mathsf{PolySame}_R$ holds.

For the cons step write $p=\mathsf{cons}(a,x)$ and assume
$$
\mathsf{PolySame}_R(\mathsf{rmul}_R(x,q),\mathsf{rmul}_R(q,x)).
$$
Unfolding the left raw product gives
$$
\mathsf{rmul}_R(\mathsf{cons}(a,x),q)=
\mathsf{radd}_R(\mathsf{rscale}_R(a,q),\mathsf{shift}_R(\mathsf{rmul}_R(x,q))).
$$
Shift preservation and raw-addition compatibility transport the induction hypothesis to
$$
\mathsf{PolySame}_R\bigl(\mathsf{rmul}_R(\mathsf{cons}(a,x),q),
\mathsf{radd}_R(\mathsf{rscale}_R(a,q),\mathsf{shift}_R(\mathsf{rmul}_R(q,x)))\bigr).
$$
Right-cons expansion applied to $q,a,x$ gives the same middle endpoint classified with $\mathsf{rmul}_R(q,\mathsf{cons}(a,x))$. Symmetry and transitivity of $\mathsf{PolySame}_R$ compose the two comparisons, completing the induction.
\end{proof}

\begin{lemma}[Padded coefficient extensionality for the polynomial classifier]
\label{lem:padded-coefficient-extensionality-polynomial-classifier}
Let $R$ carry a $\RingUp(R)$ certificate with classifier $\sim_R$, zero endpoint $0_R$, scalar certificate $\mathsf{cert}_R:\NameCert(R,\sim_R)$, and canonical trim $\mathsf{trim}_R$ from \autoref{def:polynomial-pattern-specification} and \autoref{def:polynomial-stability-certificate}. Therefore

\mathsf{ListClassifierSpec}(\sim_R,x_0,y_0).
$$
Unfolding $\mathsf{PolySame}_R$ by \autoref{def:polynomial-raw-add-comparison-data} gives the displayed polynomial classifier comparison.
\end{proof}

\begin{lemma}[Raw Cauchy product coefficient formula]
\label{lem:raw-cauchy-product-coefficient-formula}
Let $R$ carry a $\CommRingUp(R)$ certificate with classifier $\sim_R$, zero endpoint $0_R$, addition $+$, multiplication $\cdot$, and the raw operations $\mathsf{radd}_R$, $\mathsf{rscale}_R$, $\mathsf{shift}_R$, and $\mathsf{rmul}_R$ from \autoref{def:polynomial-raw-add-comparison-data} and \autoref{def:polynomial-raw-cauchy-product}. For finite coefficient spines $x,y:\mathsf{ListCarrier}(R)$ and every $n:\NatUp$,
$$
\mathsf{rmul}_R(x,y)[n]\sim_R\mathsf{Cauchy}_R(x,y;n).
$$
\end{lemma}
\begin{proof}
Read the scalar source as a $\RingUp(R)$ source by \autoref{prop:commring-forgets-ring-certificate}. The recursion defining $\mathsf{radd}_R$ gives the padded coefficient formula $\mathsf{radd}_R(u,v)[n]\sim_Ru[n]+v[n]$ by induction on the two finite spines. The definitions of $\mathsf{rscale}_R$ and $\mathsf{shift}_R$ give
$$
\mathsf{rscale}_R(a,u)[n]\sim_Ra\cdot u[n],\qquad \mathsf{shift}_R(u)[0]\sim_R0_R,\qquad \mathsf{shift}_R(u)[n+1]\sim_Ru[n].
$$
These comparisons use only zero self-classification, additive congruence, and the additive zero laws from \autoref{def:ring-stability-certificate}.

Induct on the first spine $x$. If $x=\mathsf{nil}$, the raw product is $\mathsf{nil}$ by \autoref{def:polynomial-raw-cauchy-product}. Every summand in $\mathsf{Cauchy}_R(\mathsf{nil},y;n)$ has the form $0_R\cdot y[j]$ and is classified with $0_R$ by the zero-multiplication row in \autoref{def:ring-stability-certificate}. The zero-summand fold theorem \autoref{thm:zero-convolution-summands-fold-zero} classifies the Cauchy fold with $0_R$, which is the padded coefficient of $\mathsf{nil}$.

For the cons step, write $x=\mathsf{cons}(a,x^{\prime})$. At coefficient $0$, the raw product recursion gives the sum of $a\cdot y[0]$ and the zero head of the shifted recursive product; the additive right-zero law classifies this with $a\cdot y[0]$, which is $\mathsf{Cauchy}_R(\mathsf{cons}(a,x^{\prime}),y;0)$. At coefficient $n+1$, the same recursion and the auxiliary coefficient formulas give
$$
\mathsf{rmul}_R(\mathsf{cons}(a,x^{\prime}),y)[n+1]\sim_R a\cdot y[n+1]+\mathsf{rmul}_R(x^{\prime},y)[n].
$$
The induction hypothesis and additive congruence replace the second summand by $\mathsf{Cauchy}_R(x^{\prime},y;n)$. Splitting the finite index spine $i+j=n+1$ into the head case $i=0$ and the shifted cases $i=i^{\prime}+1$ identifies the resulting fold with $\mathsf{Cauchy}_R(\mathsf{cons}(a,x^{\prime}),y;n+1)$ by \autoref{thm:finite-additive-fold-congruence-coefficient-lists}.
\end{proof}

\begin{lemma}[Raw Cauchy product coefficient associativity]
\label{lem:raw-cauchy-product-coefficient-associativity}
Let $R$ carry a $\CommRingUp(R)$ certificate with classifier $\sim_R$, zero endpoint $0_R$, addition $+$, multiplication $\cdot$, and the raw Cauchy product data of \autoref{def:polynomial-raw-cauchy-product}. For finite coefficient spines $x,y,z:\mathsf{ListCarrier}(R)$ and every $n:\NatUp$,
$$
\mathsf{rmul}_R(\mathsf{rmul}_R(x,y),z)[n]\sim_R\mathsf{rmul}_R(x,\mathsf{rmul}_R(y,z))[n].
$$
\end{lemma}
\begin{proof}
Fix $n$. By \autoref{lem:raw-cauchy-product-coefficient-formula}, the left coefficient is classified with $\mathsf{Cauchy}_R(\mathsf{rmul}_R(x,y),z;n)$. For each outer index $s+k=n$, apply \autoref{lem:raw-cauchy-product-coefficient-formula} to replace $\mathsf{rmul}_R(x,y)[s]$ by $\mathsf{Cauchy}_R(x,y;s)$. Multiplication congruence and \autoref{thm:finite-additive-fold-congruence-coefficient-lists} lift these replacements through the outer finite fold. The right-multiplication half of \autoref{lem:scalar-product-distributes-across-finite-additive-folds} then classifies $\mathsf{Cauchy}_R(x,y;s)\cdot z[k]$ with the inner fold over $(i,j)$ satisfying $i+j=s$ of $(x[i]\cdot y[j])\cdot z[k]$. The nested finite index spine $i+j=s$, $s+k=n$ is the same finite triple spine $i+j+k=n$ in lexicographic order after deleting the bookkeeping index $s$. The finite fold swap certificate of \autoref{def:two-index-finite-fold-swap-certificate}, together with fold congruence, classifies the resulting nested fold with $\mathsf{Triple}_R(x,y,z;n)$. Thus the left raw product coefficient is classified with $\mathsf{Triple}_R(x,y,z;n)$.

The right coefficient is handled symmetrically. The coefficient formula first gives $\mathsf{Cauchy}_R(x,\mathsf{rmul}_R(y,z);n)$. Replacing each coefficient of $\mathsf{rmul}_R(y,z)$ by its Cauchy fold and applying the left-multiplication half of \autoref{lem:scalar-product-distributes-across-finite-additive-folds} gives the finite fold over triples $i+j+k=n$ of $x[i]\cdot(y[j]\cdot z[k])$. Scalar multiplication associativity from \autoref{def:ring-stability-certificate} classifies each displayed summand with $(x[i]\cdot y[j])\cdot z[k]$. Fold congruence therefore classifies the right coefficient with the same $\mathsf{Triple}_R(x,y,z;n)$. Symmetry and transitivity of the scalar classifier compose the two comparisons.
\end{proof}

\begin{theorem}[Polynomial multiplication associativity over commutative-ring scalars]
\label{thm:polynomial-multiplication-associativity-commring-scalars}
Let $R$ carry a $\CommRingUp(R)$ certificate, read as a $\RingUp(R)$ certificate by \autoref{prop:commring-forgets-ring-certificate}. Let the classifier be $\sim_R$, the zero endpoint be $0_R$, the scalar operations be $+$ and $\cdot$, the canonical trim be $\mathsf{trim}_R$ from \autoref{def:polynomial-stability-certificate}, and let $\mathsf{rmul}_R$ and $\mathsf{PolyMul}_R$ be the raw and normalized products of \autoref{def:polynomial-raw-cauchy-product}. For finite coefficient spines $p,q,r:\mathsf{ListCarrier}(R)$,
$$
\mathsf{PolySame}_R\bigl(\mathsf{PolyMul}_R(\mathsf{PolyMul}_R(p,q),r),\mathsf{PolyMul}_R(p,\mathsf{PolyMul}_R(q,r))\bigr).
$$
\end{theorem}
\begin{proof}
Put $p_0:=\mathsf{trim}_R(p)$, $q_0:=\mathsf{trim}_R(q)$, and $r_0:=\mathsf{trim}_R(r)$. Set $A:=\mathsf{PolyMul}_R(\mathsf{PolyMul}_R(p,q),r)$ and $B:=\mathsf{PolyMul}_R(p,\mathsf{PolyMul}_R(q,r))$. By \autoref{lem:padded-coefficient-extensionality-polynomial-classifier}, it is enough to prove $A[n]\sim_RB[n]$ for every $n:\NatUp$.

Trim preserves padded coefficients by the trailing-zero decomposition in \autoref{def:polynomial-stability-certificate}: for every finite spine $u$ and every $n$, $u[n]\sim_R\mathsf{trim}_R(u)[n]$. The coefficient formula \autoref{lem:raw-cauchy-product-coefficient-formula} is functorial in both inputs with respect to such pointwise padded-coefficient comparisons, since multiplication congruence compares each Cauchy summand and \autoref{thm:finite-additive-fold-congruence-coefficient-lists} lifts the summand comparisons through the finite fold.

Unfolding \autoref{def:polynomial-raw-cauchy-product} and using trim idempotence from \autoref{def:polynomial-stability-certificate}, the left normalized product has the same padded coefficients as $\mathsf{rmul}_R(\mathsf{trim}_R(\mathsf{rmul}_R(p_0,q_0)),r_0)$. By trim preservation and the functoriality just recorded, its $n$th coefficient is classified with $\mathsf{rmul}_R(\mathsf{rmul}_R(p_0,q_0),r_0)[n]$. Similarly, the right normalized product has the same padded coefficients as $\mathsf{rmul}_R(p_0,\mathsf{trim}_R(\mathsf{rmul}_R(q_0,r_0)))$, and its $n$th coefficient is classified with $\mathsf{rmul}_R(p_0,\mathsf{rmul}_R(q_0,r_0))[n]$.

Applying \autoref{lem:raw-cauchy-product-coefficient-associativity} to $p_0,q_0,r_0$ gives
$$
\mathsf{rmul}_R(\mathsf{rmul}_R(p_0,q_0),r_0)[n]\sim_R\mathsf{rmul}_R(p_0,\mathsf{rmul}_R(q_0,r_0))[n].
$$
Classifier symmetry and transitivity compose the two trim comparisons with this raw coefficient associativity comparison, yielding $A[n]\sim_RB[n]$ for the fixed $n$. Since $n$ was arbitrary, \autoref{lem:padded-coefficient-extensionality-polynomial-classifier} gives $\mathsf{PolySame}_R(A,B)$, which is the displayed associativity statement.
\end{proof}

\begin{theorem}[Polynomial singleton evaluation]
\label{thm:polynomial-singleton-evaluation}
Let $R$ carry a $\RingUp(R)$ certificate with classifier $\sim_R$, zero endpoint $0_R$, addition $+$, multiplication $\cdot$, canonical trim $\mathsf{trim}_R$, and Horner evaluation as in \autoref{def:polynomial-horner-evaluation}. For every carried scalar coefficient $c$ and every carried scalar point $\alpha$,
$$
\mathsf{Eval}_{\alpha,R}(\mathsf{cons}(c,\mathsf{nil}))\sim_R c.
$$
\end{theorem}
\begin{proof}
Put $s=\mathsf{cons}(c,\mathsf{nil})$. First compute the unnormalized Horner value. The cons and nil recursion clauses in \autoref{def:polynomial-horner-evaluation} identify
$$
\mathsf{eval}_{\alpha,R}(s)=c+\alpha\cdot0_R.
$$
The right zero-absorption half of \autoref{thm:ring-zero-absorption-from-distributivity} gives
$$
\alpha\cdot0_R\sim_R0_R.
$$
Classifier reflexivity for $c$ and addition congruence from \autoref{def:ring-stability-certificate} give
$$
c+\alpha\cdot0_R\sim_Rc+0_R.
$$
The additive right-zero theorem \autoref{thm:ring-add-right-zero} gives $c+0_R\sim_Rc$, and transitivity yields
$$
\mathsf{eval}_{\alpha,R}(s)\sim_Rc.
$$
By \autoref{def:polynomial-horner-evaluation}, $\mathsf{Eval}_{\alpha,R}(s)=\mathsf{eval}_{\alpha,R}(\mathsf{trim}_R(s))$. The one-coefficient trimming boundary in \autoref{def:polynomial-normalized-singleton-representative}, read with \autoref{def:polynomial-stability-certificate}, has exactly the empty zero branch and the retained singleton branch. In the empty zero branch, $c\sim_R0_R$ and $\mathsf{trim}_R(s)=\mathsf{nil}$; the nil Horner equation gives $\mathsf{Eval}_{\alpha,R}(s)=0_R$, and classifier symmetry gives $0_R\sim_Rc$. In the retained singleton branch, $\mathsf{trim}_R(s)=s$, so the raw computation above is the required normalized comparison. These are the two canonical trim branches for a singleton coefficient spine.
\end{proof}
