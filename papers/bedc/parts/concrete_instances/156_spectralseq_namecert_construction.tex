\chapter{A Concrete Naming Certificate for $\SpectralSeqUp$}
\label{ch:concrete-instances-spectralseq-namecert}

The spectral-sequence interface $\SpectralSeqUp$ packages a sequence of bigraded objects with differentials whose successive cohomologies converge to a target. It depends on $\NameCert_{\HomologyUp}$ (\autoref{ch:concrete-instances-homology-namecert}) and $\NameCert_{\AbelianCatUp}$ (\autoref{ch:concrete-instances-abeliancat-namecert}).

\section{Carrier definition}
\label{sec:spectralseq-carrier}

\begin{definition}[SpectralSeqUp BHist page carrier]
\label{def:spectralseq-bhist-page-carrier}
A $\SpectralSeqUp$ $\mathsf{BHist}$ page carrier is a finite page packet
$$
\mathcal S=(A,E,d,h,c,\lambda,\rho)
$$
over the source rows $\NameCert_{\AbelianCatUp}$ and
$\NameCert_{\HomologyUp}$. The row $A$ is the displayed abelian-category
surface in which the page objects and differential arrows live. The finite
family $E=(E_r^{p,q})$ is the page-object ledger; each $E_r^{p,q}$ is an
$A$-classified object row. The finite family $d=(d_r^{p,q})$ is the
differential ledger, with
$$
d_r^{p,q}:E_r^{p,q}\to E_r^{p+r,q-r+1}
$$
recorded as an $A$-classified morphism row. The row $h$ records the homology
readback identifying each successor page with the $\HomologyUp$ cycle row of
$d_r$ modulo its displayed boundary row. The row $c$ records the finite
convergence and filtration rows selected for the target object. The row
$\lambda$ is the $\Cont$ ledger ordering the finite page-to-page transitions,
and $\rho$ is the $\Pkg$ provenance row for the displayed source certificates,
page rows, differential rows, homology readbacks, convergence rows, and
continuation entries.
\end{definition}

\begin{definition}[SpectralSeqUp page classifier]
\label{def:spectralseq-page-classifier}
For two $\SpectralSeqUp$ page carriers
$\mathcal S=(A,E,d,h,c,\lambda,\rho)$ and
$\mathcal S'=(A',E',d',h',c',\lambda',\rho')$, the page classifier holds when
the following componentwise transport rows are displayed.
\begin{enumerate}
  \item $A\hsame A'$ as $\NameCert_{\AbelianCatUp}$ source surfaces.
  \item $E_r^{p,q}\hsame E_r'^{p,q}$ for every carried page object.
  \item $d_r^{p,q}\hsame d_r'^{p,q}$ for every differential row, with source
  and target read through the transported page rows.
  \item $h_r^{p,q}\hsame h_r'^{p,q}$ for every $\HomologyUp$ cycle-boundary
  readback row relating a differential to the successor page.
  \item $c\hsame c'$ for the displayed filtration and convergence rows.
  \item $\Cont(\lambda,\lambda')$ for the finite page-transition ledger.
  \item $\rho\hsame\rho'$ as $\Pkg$ provenance rows for exactly the listed
  page, differential, homology, convergence, and continuation entries.
\end{enumerate}
Thus the public classifier compares the finite spectral-sequence packet by
transporting the displayed rows themselves, not by importing a host exact
couple or an ambient equality of abelian categories.
\end{definition}

\section{The certificate}
\label{sec:spectralseq-certificate}

\begin{theorem}[SpectralSeqUp obligation surface]
\label{thm:spectralseq-obligation-surface}
Every $\SpectralSeqUp$ page carrier exposes exactly the source abelian-category
surface, the source homology surface, the finite page-object ledger, the page
classifier, the differential-square rows, the homology-page readback rows, the
finite convergence rows, the $\Cont$ transition ledger, and the $\Pkg$
provenance row. These rows form the public $\NameCert_{\SpectralSeqUp}$
obligation surface available to downstream $\HomologyUp$,
$\AbelianCatUp$, and derived-category consumers.
\end{theorem}
\begin{proof}
Project the carrier of \autoref{def:spectralseq-bhist-page-carrier}. The
projections give the abelian-category source row, the homology source row, the
finite page ledger, the differential ledger, the homology readback row, the
convergence row, the continuation ledger, and the provenance row.

The classifier of \autoref{def:spectralseq-page-classifier} is then read
component by component. It transports page objects, differential rows,
homology readbacks, convergence rows, continuation entries, and provenance
without adding any unlisted carrier.

Repack these projections as the named obligation surface. Since every exposed
entry is a projection of the displayed packet or the classifier attached to
that packet, the public certificate is exhausted by the listed rows.
\end{proof}

\begin{theorem}[SpectralSeqUp differential ledger stability]
\label{thm:spectralseq-differential-ledger}
If two $\SpectralSeqUp$ page carriers are related by the page classifier, then
each transported differential row $d_r^{p,q}$ remains an admissible
$\AbelianCatUp$ morphism row with the transported source and target pages, and
the corresponding successor-page homology readback remains classified by the
$\HomologyUp$ source surface. Finite $\Cont$ extension of the page-transition
ledger preserves the same property for the extended carried ledger.
\end{theorem}
\begin{proof}
Start with one differential row
$d_r^{p,q}:E_r^{p,q}\to E_r^{p+r,q-r+1}$ from the carrier. The page
classifier supplies $\hsame$ transport for the source page, target page, and
the differential row itself, so the transported row is again typed by the
transported $\AbelianCatUp$ source surface.

Project the homology readback row attached to this differential. The
classifier also transports the corresponding $\HomologyUp$ cycle-boundary
entry, so the successor page continues to be read as the displayed homology
page of the transported differential.

For a finite $\Cont$ extension, append the displayed transition row to
$\lambda$ and extend the provenance row by the same entry. The page packet is
then repacked with all previous differential and homology rows unchanged and
with the new transition recorded in the finite ledger.
\end{proof}

\begin{theorem}[SpectralSeqUp convergence boundary]
\label{thm:spectralseq-convergence-boundary}
The $\SpectralSeqUp$ certificate exports only the displayed page, differential,
homology-readback, filtration, convergence, $\Cont$, and $\Pkg$ rows of
\autoref{def:spectralseq-bhist-page-carrier}. It does not export a host exact
couple, a quotient object outside the displayed $\HomologyUp$ boundary rows,
or an ambient equality of abelian categories beyond the transported
$\NameCert_{\AbelianCatUp}$ source surface.
\end{theorem}
\begin{proof}
Unpack the carrier and inspect the convergence component $c$. Its public
content is the finite filtration and convergence ledger selected in the
$\mathsf{BHist}$ packet, together with the page and homology rows needed to
name its source.

The quotient information used by a successor page appears only through the
$\HomologyUp$ cycle-boundary readback row $h$. The theorem therefore reads no
separate quotient object: it uses the displayed boundary classifier already
carried by the homology source surface.

Finally, the abelian-category comparison is exactly the source-surface
transport in the page classifier. Since the carrier has no row naming a host
exact couple or an ambient category equality, those data cannot be exported by
the public $\NameCert_{\SpectralSeqUp}$ boundary.
\end{proof}

\begin{theorem}[SpectralSeqUp page-transition bridge]
\label{thm:spectralseq-page-transition-bridge}
For a $\SpectralSeqUp$ page carrier
$$
\mathcal S=(A,E,d,h,c,\lambda,\rho),
$$
the finite $\Cont$ ledger $\lambda$ of
\autoref{def:spectralseq-bhist-page-carrier}, transported through the page
classifier of \autoref{def:spectralseq-page-classifier}, gives a
$\mathsf{BHist}$-stable bridge between each successive page row
$E_r^{p,q}$ and the corresponding $\HomologyUp$ readback row carried by $h$.
The bridge exposes exactly the transported page-object rows, differential
rows, successor homology rows, $\Cont$ transition entries, and $\Pkg$
provenance rows already listed in the spectral-sequence packet.
\end{theorem}
\begin{proof}
Project the carrier to obtain the page ledger $E$, the differential ledger
$d$, the homology readback ledger $h$, the transition ledger $\lambda$, and
the provenance row $\rho$.

For each carried transition, apply the classifier clauses for the source page,
target page, differential row, and homology readback row. These clauses
transport the page endpoint and its successor readback componentwise through
$\hsame$ while the same finite $\Cont$ entry records the page-to-page step.

Finally repack the transported page endpoint, its differential, its
$\HomologyUp$ readback, the relevant continuation entry, and the provenance
row. Because every coordinate is projected from the original packet or from
the classifier transport, the resulting bridge is stable as a
$\mathsf{BHist}$ row and exports no extra exact-couple data.
\end{proof}

\begin{theorem}[SpectralSeqUp convergence-filtration bridge]
\label{thm:spectralseq-convergence-filtration-bridge}
For $\SpectralSeqUp$ page carriers related by the classifier of
\autoref{def:spectralseq-page-classifier}, the convergence row $c$ is stable
under $\hsame$ of the page packet and exposes exactly the finite filtration
boundary needed by a derived-functor consumer. The exposed boundary consists of
the transported convergence row, the page rows named by the finite filtration,
the endpoint readbacks supplied by $h$, the $\Cont$ ledger $\lambda$, and the
$\Pkg$ provenance row $\rho$.
\end{theorem}
\begin{proof}
Start with the classifier clause $c\hsame c'$. It transports the displayed
convergence and filtration rows as rows of the same page packet rather than as
ambient host objects.

Next combine that clause with the classifier clauses for $\lambda$ and $\rho$.
The continuation ledger records which finite page transitions support the
filtration boundary, while the provenance row keeps those transitions tied to
the same page, differential, homology, and convergence entries.

Apply \autoref{thm:spectralseq-convergence-boundary}. It excludes host exact
couples and quotient objects outside the displayed $\HomologyUp$ boundary
rows. The remaining exported data are precisely the finite filtration boundary
rows and their $\Cont$ and $\Pkg$ witnesses.
\end{proof}

\begin{theorem}[SpectralSeqUp DerivedFunctorUp consumer surface]
\label{thm:spectralseq-derivedfunctor-consumer-surface}
A downstream $\DerivedFunctorUp$ row consuming a $\SpectralSeqUp$ certificate
may use only the $\AbelianCatUp$ source row, the $\HomologyUp$ source row, the
page-object rows, the differential rows, the homology readbacks, the
convergence-filtration rows, the $\Cont$ transition ledger, and the $\Pkg$
provenance listed in this chapter. It consumes no host exact couple, ambient
quotient category, or equality of abelian categories outside the displayed
$\NameCert_{\AbelianCatUp}$ source transport.
\end{theorem}
\begin{proof}
Use \autoref{thm:spectralseq-obligation-surface} to enumerate the rows exported
by the spectral-sequence certificate. These are the source certificates, page
objects, differentials, homology readbacks, convergence rows, continuation
ledger, and provenance row.

Apply \autoref{thm:spectralseq-page-transition-bridge} for the page-to-page
bridge surface and \autoref{thm:spectralseq-convergence-filtration-bridge} for
the convergence-facing boundary. Together they provide the transition and
filtration rows needed by a derived-functor consumer.

Finally apply \autoref{thm:spectralseq-convergence-boundary}. Since the
certificate exports neither a host exact couple nor an ambient quotient
category, the downstream row can consume only the displayed BEDC rows and their
listed $\Cont$ and $\Pkg$ witnesses.
\end{proof}

\begin{theorem}[SpectralSeq boundary rows enter successor cycles]
\label{thm:spectralseq-boundary-row-successor-cycle-readback}
Let $\mathcal S=(A,E,d,h,c,\lambda,\rho)$ be a $\SpectralSeqUp$ $\mathsf{BHist}$ page carrier. Fix a carried differential row $d_r^{p,q}$, and write its $\HomologyUp$ readback on the BEDC history surface again as
$$
d_r^{p,q}:\mathsf{BHist}\to\mathsf{BHist}.
$$
Assume that the displayed differential-square row for this differential is
$$
\forall u:\mathsf{BHist},\ \hsame(d_r^{p,q}(d_r^{p,q}(u)),\mathsf{Empty}).
$$
If a representative $b$ lies in the displayed boundary row for this differential,
$$
B_{d_r^{p,q}}(b),
$$
then the successor-page readback row $h_r^{p,q}$ classifies the same representative through the cycle row:
$$
Z_{d_r^{p,q}}(b).
$$
Equivalently, every representative read by $h_r^{p,q}$ from the displayed boundary subrow of $d_r^{p,q}$ enters the successor surface $E_{r+1}^{p,q}$ only through the cycle row used by the homology-page quotient.
\end{theorem}
\begin{proof}
The carrier description in \autoref{def:spectralseq-bhist-page-carrier} gives, for the fixed differential row, a homology readback row identifying the successor page with the $\HomologyUp$ cycle row of $d_r^{p,q}$ modulo its displayed boundary row. The obligation surface in \autoref{thm:spectralseq-obligation-surface} exposes the differential-square row for the same displayed differential together with the matching homology-page readback row $h_r^{p,q}$.

Apply \autoref{thm:homology-boundary-carrier-cycle-closed} to the $\HomologyUp$ differential read back from $d_r^{p,q}$ and to the representative $b$. The boundary hypothesis supplies the boundary-carrier premise, while the displayed differential-square row supplies the universal square-zero premise. The conclusion is exactly $Z_{d_r^{p,q}}(b)$ before taking the boundary quotient. Since $h_r^{p,q}$ reads the successor page from that cycle row modulo the displayed boundary row, the boundary representative is admitted to the successor-page surface as a cycle representative.
\end{proof}

\begin{theorem}[SpectralSeqUp successor-page closure]
\label{thm:spectralseq-successor-page-closure}
Suppose a $\SpectralSeqUp$ page carrier supplies a differential row
$d_r^{p,q}$, its transported source and target page rows
$E_r^{p,q}$ and $E_r^{p+r,q-r+1}$, and the displayed $\HomologyUp$
cycle-boundary readback row identifying the successor page. Then the successor
page is repacked as a $\SpectralSeqUp$ page carrier over the same
$\NameCert_{\AbelianCatUp}$ source, with the $\Cont$ page-transition ledger
extended by the successor entry and the $\Pkg$ provenance retained for the
displayed page, differential, homology-readback, convergence, and transition
rows. No host quotient, host exact couple, or ambient category equality is
introduced by the page transition.
\end{theorem}
\leanchecked{BEDC.Derived.SpectralSeqUp.SpectralSeqBHistPageCarrier\_successor\_page\_closure}
\begin{proof}
Project the differential row, its source and target page rows, and the
homology readback row from the carrier of
\autoref{def:spectralseq-bhist-page-carrier}. These projections give the
only data used to name the successor page.

Use the page classifier of \autoref{def:spectralseq-page-classifier} to
transport the source page, target page, differential, and homology readback
componentwise. By \autoref{thm:spectralseq-differential-ledger}, the
transported differential remains an admissible $\AbelianCatUp$ morphism row,
and the transported readback remains classified by the displayed
$\HomologyUp$ source surface.

Repack the successor page by extending the finite $\Cont$ ledger with the
displayed page-transition entry and retaining the same $\Pkg$ provenance rows.
The boundary restriction of \autoref{thm:spectralseq-convergence-boundary}
excludes any host quotient or exact-couple object outside the carried
$\HomologyUp$ cycle-boundary row, so the repacked object is again a
$\SpectralSeqUp$ page carrier.
\end{proof}

\begin{definition}[SpectralSeqUp filtration ledger]
\label{def:spectralseq-filtration-ledger}
For a $\SpectralSeqUp$ page carrier
$\mathcal S=(A,E,d,h,c,\lambda,\rho)$, the filtration ledger is the finite
$\mathsf{BHist}$ packet $c$ already named in
\autoref{def:spectralseq-bhist-page-carrier}. It consists of the displayed
filtration rows for the target object, the convergence rows relating stable
page representatives to those filtration rows, the $\HomologyUp$ cycle and
boundary readbacks used to name each representative, and the $\Pkg$
provenance entries tying those rows to the carried page, differential, and
transition ledgers.

The public filtration ledger is therefore a subpacket of the carrier. It is
not a host exact couple, an external quotient object, or an ambient abutment
object outside the displayed BEDC rows.
\end{definition}

\begin{theorem}[SpectralSeqUp filtration transport]
\label{thm:spectralseq-filtration-transport}
Let $\mathcal S=(A,E,d,h,c,\lambda,\rho)$ and
$\mathcal S'=(A',E',d',h',c',\lambda',\rho')$ be $\SpectralSeqUp$ page
carriers related by the page classifier of
\autoref{def:spectralseq-page-classifier}. If $\lambda^+$ is obtained from
$\lambda$ by a finite $\Cont$ extension recording a displayed page
transition, then the filtration ledger $c$ transports to the filtration ledger
$c'$ and to the extended ledger over $\lambda^+$. The transport uses only the
classifier rows $c\hsame c'$, the transported page, differential, and homology
readback rows, and the finite $\Cont$ extension; it imports no host exact
couple.
\end{theorem}
\begin{proof}
Project the filtration ledger $c$ from
\autoref{def:spectralseq-filtration-ledger}. Its entries are already part of
the carrier of \autoref{def:spectralseq-bhist-page-carrier}.

Use the page classifier of \autoref{def:spectralseq-page-classifier}. It
supplies $c\hsame c'$ and transports the page objects, differentials, and
$\HomologyUp$ cycle-boundary readback rows that name the filtration entries.

For the finite extension of $\lambda$, apply
\autoref{thm:spectralseq-successor-page-closure}. The successor step extends
only the carried $\Cont$ transition ledger and retains the same $\Pkg$
provenance scope. By \autoref{thm:spectralseq-convergence-boundary}, there is
no exported exact couple or ambient quotient object to transport, so the
filtration ledger remains the displayed BEDC subpacket.
\end{proof}

\begin{theorem}[SpectralSeqUp abutment readback boundary]
\label{thm:spectralseq-abutment-readback-boundary}
Every abutment representative exported by the filtration ledger
$c$ of a $\SpectralSeqUp$ page carrier is read through the displayed
$\HomologyUp$ cycle-boundary rows and the $\Pkg$ provenance row of the carrier.
No representative is exported through a quotient object, exact-couple object,
or abutment carrier not named in \autoref{def:spectralseq-bhist-page-carrier}.
\end{theorem}
\begin{proof}
Unpack the filtration ledger using
\autoref{def:spectralseq-filtration-ledger}. Each exported representative is
attached to a convergence row in $c$ and to the provenance entry $\rho$ for the
same carried page packet.

The representative itself is read back through the homology component $h$ of
\autoref{def:spectralseq-bhist-page-carrier}. The boundary statement
\autoref{thm:spectralseq-convergence-boundary} identifies this as the
displayed $\HomologyUp$ cycle row modulo its displayed boundary row, not as a
separate host quotient.

Finally apply \autoref{thm:spectralseq-successor-page-closure} to the page
transition feeding the stable representative. The transition repacks the
successor page using the carried differential, homology readback,
filtration, $\Cont$, and $\Pkg$ rows. Thus every abutment readback remains
inside the displayed BEDC ledger.
\end{proof}
