\chapter{A Concrete Naming Certificate for \(\FullAxisUp\)}
\label{ch:concrete-instances-full-axis-seal-namecert}

This chapter records a prefix-thread seal interface over the \(\Ezero\)-spine. \(\FullAxisUp\) is NOT a metric or Cauchy completion: its boundary token \(\Eone(\Ezero(\emp))\) is a finite history, not a limit object, and the chapter explicitly disclaims any reading of the boundary as an infinity. The Cauchy / Real / Topology readings remain in the existing \(\CompleteMetricUp\) and related chapters; \(\FullAxisUp\) is a structural seal, not a metric completion.

\section{The prefix thread}

\begin{definition}[Prefix thread]
\label{def:prefix-thread}
A prefix thread \(T\) is a history-membership predicate \(T.\mathsf{member}\) that contains \(\emp\) and is closed under \(\Ezero\)-extension.
\end{definition}
\leandef{BEDC.Derived.AxisZeckendorf.FullAxis.PrefixThread}

\begin{definition}[Zero-spine prefix thread]
\label{def:zero-spine-prefix-thread}
The zero-spine prefix thread \(\mathsf{ZSpineThread}\) takes \(T.\mathsf{member} := \mathsf{ZeroSpine}\), with empty membership from Lemma~\ref{lem:zero-spine-empty-base} and \(\Ezero\)-step from Lemma~\ref{lem:zero-spine-closure-under-zero-extension}.
\end{definition}
\leandef{BEDC.Derived.AxisZeckendorf.FullAxis.zeroSpinePrefixThread}

\section{Seal certificate}

\begin{definition}[Seal certificate over a prefix thread]
\label{def:seal-certificate}
A seal certificate \(\mathsf{SealCert}(T)\) records the structural seal data for a prefix thread \(T\): directedness, monotonicity, prefix-closure, and a ledger policy. The first three fields are skeleton-grade; the ledger field is a horizon obligation queued for the formalize pipeline.
\end{definition}
\leandef{BEDC.Derived.AxisZeckendorf.FullAxis.SealCert}
\leanvariant{BEDC.Derived.AxisZeckendorf.FullAxis.zeroSpineSealCert}

\section{The boundary marker}

\begin{definition}[Boundary marker \(\mathtt{boundary\_01}\)]
\label{def:axiszeckendorf-boundary-01}
The boundary marker is the finite history
$$
  \mathtt{boundary\_01}
  := \Eone(\Ezero(\emp)).
$$
\end{definition}
\leandef{BEDC.Derived.AxisZeckendorf.FullAxis.boundary\_01}

\begin{theorem}[Boundary marker is not on the zero-spine]
\label{thm:boundary-01-not-zero-spine}
\(\mathsf{ZeroSpine}(\mathtt{boundary\_01})\) does not hold.
\end{theorem}
\leandef{BEDC.Derived.AxisZeckendorf.FullAxis.boundary\_01\_not\_zeroSpine}
\begin{proof}
\(\mathtt{boundary\_01}\) has top constructor \(\Eone\); apply zero-spine no-one-extension.
\end{proof}

\begin{remark}[Boundary marker is finite, not an infinity]
The boundary marker is a finite history of length two. It does NOT denote a limit, an infinity, or a transfinite stage. Any reading that promotes \(\mathtt{boundary\_01}\) to an infinity is explicitly rejected (cf.\ Chapter~\ref{ch:concrete-instances-axis-zeckendorf-cannot-claim}).
\end{remark}

\begin{theorem}[FullAxis prefix-thread boundary disjointness]
\label{thm:fullaxis-prefix-thread-boundary-disjointness}
For the zero-spine prefix thread of \autoref{def:zero-spine-prefix-thread}, the
finite boundary marker $\mathtt{boundary\_01}$ is outside the thread carrier.
Consequently it cannot serve as a thread-member witness, a prefix-thread
membership row, or a limit witness for the structural seal.
\end{theorem}
\begin{proof}
Unfold the zero-spine prefix thread of \autoref{def:zero-spine-prefix-thread}.
Its membership predicate is exactly $\mathsf{ZeroSpine}$, closed from $\emp$ by
repeated $\Ezero$-extension.

Apply \autoref{thm:boundary-01-not-zero-spine}. The finite marker
$\mathtt{boundary\_01}$ has an $\Eone$ head and therefore is not a member of the
zero-spine carrier.

Since \autoref{def:fullaxis-source-specification} reads source rows only from
thread membership, the boundary marker cannot be repacked as a source member.
The seal certificate therefore treats it only as a finite excluded boundary
token, not as a completion point or infinity.
\end{proof}

\begin{theorem}[FullAxis seal source exhaustion]
\label{thm:fullaxis-seal-source-exhaustion}
The $\FullAxisUp$ source, pattern, classifier, stability, and ledger rows are
exhausted by zero-spine prefix-thread membership together with the finite
boundary exclusion of
\autoref{thm:fullaxis-prefix-thread-boundary-disjointness}. They contain no
completion witness, no metric limit, and no thread-member row for
$\mathtt{boundary\_01}$.
\end{theorem}
\begin{proof}
Start with the zero-spine source. By
\autoref{def:fullaxis-source-specification}, the source rows are exactly the
members of the chosen prefix thread, and by
\autoref{def:zero-spine-prefix-thread} those members are precisely the
$\mathsf{ZeroSpine}$ histories.

The pattern and classifier rows add no outside carrier. The pattern row of
\autoref{def:fullaxis-pattern-specification} is repeated $\Ezero$-extension
inside the same thread, and the classifier row of
\autoref{def:fullaxis-classifier-specification} classifies only pairs already
present by joint thread membership.

Apply \autoref{thm:fullaxis-prefix-thread-boundary-disjointness}. The finite
marker $\mathtt{boundary\_01}$ is outside that carrier, so it cannot be read as a
thread member or as a limit witness. The licensing corollary
\autoref{cor:fullaxis-licensed-not-primitive} then repacks the displayed source,
pattern, classifier, stability, and ledger rows as the whole structural seal
surface.
\end{proof}

\label{thm:fullaxis-not-real-horizon}
\(\FullAxisUp\) does not entail any of the algebraic operations of \(\RealUp\) (Cauchy, limit uniqueness, field structure).

\end{remark}
