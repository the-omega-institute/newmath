\chapter{A Concrete Naming Certificate for $\HopfAlgUp$}
\label{ch:concrete-instances-hopfalg-namecert}

The Hopf-algebra interface $\HopfAlgUp$ packages a bialgebra carrier together with an antipode satisfying the convolution-inverse axiom. It depends on $\NameCert_{\RingUp}$ (\autoref{ch:concrete-instances-ring-namecert}) and $\NameCert_{\TensorProductUp}$ (\autoref{ch:concrete-instances-tensorproduct-namecert}).

\section{Carrier definition}
\label{sec:hopfalg-carrier}

\begin{definition}[HopfAlgUp BHist bialgebra carrier]\label{def:hopfalg-bhist-bialgebra-carrier}
Fix a $\NameCert_{\RingUp}$ row $R$ with multiplication ledger $\mu_R$ and unit ledger $\eta_R$, and fix a $\NameCert_{\TensorProductUp}$ row $T$ with tensor ledger $\otimes_T$. A concrete $\HopfAlgUp$ bialgebra carrier over $\mathsf{BHist}$ is the predicate
$$
\mathsf{HopfBialgCarrier}_{R,T}(h)
\Longleftrightarrow
\exists a,t,m,c,u,e:\mathsf{BHist}.\;
\hsame(h,a)\land
\mathsf{RingCarrier}_R(a)\land
\mathsf{TensorCarrier}_T(t)\land
\Cont(\otimes_T,\mu_R,m)\land
\Cont(t,c,a)\land
\Cont(u,\eta_R,a)\land
\Cont(c,e,a).
$$
Here $a$ is the bialgebra row, $t$ is the tensor-product row, $m$ is the multiplication provenance row, $c$ is the comultiplication provenance row, $u$ is the unit ledger, and $e$ is the counit ledger. The carrier is therefore anchored in $\mathsf{BHist}$ rows, with multiplication read through $\NameCert_{\RingUp}$ and tensor/comultiplication provenance read through $\NameCert_{\TensorProductUp}$.
\end{definition}

\begin{definition}[HopfAlgUp antipode ledger classifier]\label{def:hopfalg-antipode-ledger-classifier}
For two concrete Hopf-algebra rows
$$
H=(a,t,s,u,e)
\qquad\text{and}\qquad
H'=(a',t',s',u',e')
$$
over the same $\NameCert_{\RingUp}$ and $\NameCert_{\TensorProductUp}$ sources, define
$$
H\sim_{\HopfAlgUp}H'
\Longleftrightarrow
\hsame(a,a')\land
\hsame(t,t')\land
\hsame(s,s')\land
\hsame(u,u')\land
\hsame(e,e').
$$
Thus the classifier compares the bialgebra row, the tensor-product row, the antipode row, the unit ledger, and the counit ledger by $\hsame$.
\end{definition}

\section{The certificate}
\label{sec:hopfalg-certificate}

\begin{theorem}[HopfAlgUp tensor-product row stability]
\label{thm:hopfalg-tensor-product-row-stability}
Let $H=(a,t,s,u,e)$ be a concrete Hopf-algebra row satisfying \autoref{def:hopfalg-bhist-bialgebra-carrier}. If $H\sim_{\HopfAlgUp}H'$ with $H'=(a',t',s',u',e')$, then the tensor-product row of $H'$ is again a valid $\NameCert_{\TensorProductUp}$ row for the transported bialgebra carrier. In particular, the multiplication and comultiplication provenance rows read by $\Cont$ from $t$ transport to corresponding rows read from $t'$.
\end{theorem}
\leanchecked{BEDC.Derived.HopfAlgUp.HopfAlgBialgCarrier\_tensor\_product\_row\_stability}
\leanchecked{BEDC.Derived.HopfAlgUp.HopfBialgCarrier\_tensor\_product\_row\_stability}
\begin{proof}
Unpack the concrete carrier. The witness exposes the bialgebra row, the tensor-product row, and the two $\Cont$ ledgers that read multiplication and comultiplication provenance from the tensor source.

The classifier hypothesis supplies $t\hsame t'$ and $a\hsame a'$. Transport the tensor carrier and the two continuation ledgers along these $\hsame$ comparisons, using the $\hsame$ stability field of $\NameCert_{\TensorProductUp}$ for the tensor row and continuation transport for the displayed $\Cont$ rows.

The resulting rows have the same ring and tensor sources as the original package, with endpoints replaced by their transported representatives. Hence $t'$ remains the tensor-product row for the transported bialgebra carrier.
\end{proof}

\begin{theorem}[HopfAlgUp antipode convolution-inverse obligation]
\label{thm:hopfalg-antipode-convolution-inverse-obligation}
Let $H=(a,t,s,u,e)$ satisfy the concrete carrier and classifier of \autoref{def:hopfalg-bhist-bialgebra-carrier} and \autoref{def:hopfalg-antipode-ledger-classifier}. The antipode row $s$ is admissible for $\HopfAlgUp$ exactly when there are two $\mathsf{BHist}$ ledger rows
$$
\mathsf{ConvLeft}(a,t,s,u,e)
\qquad\text{and}\qquad
\mathsf{ConvRight}(a,t,s,u,e)
$$
whose $\Cont$ provenance composes the comultiplication row, the antipode row, the multiplication row, the unit ledger, and the counit ledger so that the two convolution composites land on the unit--counit endpoint.
\end{theorem}
\leanchecked{BEDC.Derived.HopfAlgUp.HopfBialgCarrier\_antipode\_convolution\_inverse\_obligation}
\begin{proof}
Start from the carrier witness. It gives the bialgebra row $a$, the tensor row $t$, the multiplication provenance read through $\NameCert_{\RingUp}$, the comultiplication provenance read through $\NameCert_{\TensorProductUp}$, and the unit and counit ledgers.

For the left convolution row, compose the comultiplication continuation with the tensor row that applies $s$ on the first component and the identity ledger on the second component. Then compose with the multiplication continuation and compare the result with the unit--counit ledger endpoint.

For the right convolution row, use the same continuation spine with $s$ placed on the second component. The two displayed rows are the two constructive obligations; no extra carrier is required beyond the $\mathsf{BHist}$ ledgers already unpacked from the concrete package.
\end{proof}

\begin{theorem}[HopfAlgUp NameCert obligation surface]
\label{thm:hopfalg-namecert-obligation-surface}
The data of \autoref{def:hopfalg-bhist-bialgebra-carrier}, \autoref{def:hopfalg-antipode-ledger-classifier}, \autoref{thm:hopfalg-tensor-product-row-stability}, and \autoref{thm:hopfalg-antipode-convolution-inverse-obligation} form the concrete obligation surface for a $\NameCert_{\HopfAlgUp}$ over $\mathsf{BHist}$. The surface consists of the carrier, the $\hsame$ classifier, tensor-row stability, multiplication and comultiplication provenance through $\Cont$, and the two antipode convolution-inverse ledger rows.
\end{theorem}
\begin{proof}
The carrier and classifier provide the object rows and equality policy. Reflexivity, symmetry, and transitivity are inherited componentwise from $\hsame$ over the bialgebra row, tensor row, antipode row, unit ledger, and counit ledger.

Tensor-row stability supplies the transport rule needed by the certificate: replacing a representative by an $\hsame$ representative preserves the tensor source and the multiplication/comultiplication provenance read from it. The ring multiplication row is retained from $\NameCert_{\RingUp}$, and the tensor row is retained from $\NameCert_{\TensorProductUp}$.

Finally, the antipode theorem packages the two convolution-inverse rows. These rows are exactly the Hopf-algebra-specific obligations beyond the bialgebra carrier, so the listed data are the $\NameCert_{\HopfAlgUp}$ obligation surface.
\end{proof}
\leanchecked{BEDC.Derived.HopfAlgUp.HopfAlgBialgCarrier\_namecert\_obligation\_surface}

\begin{lemma}[Two-step continuation middle cancellation]
\label{lem:two-step-continuation-middle-cancellation}
For histories $c,m,s,s',L,L',q,q':\mathsf{BHist}$, assume
$$
\Cont(c,s,L)\land\Cont(L,m,q)
$$
and
$$
\Cont(c,s',L')\land\Cont(L',m,q').
$$
If $\hsame(q,q')$, then $\hsame(s,s')$.
\end{lemma}
\begin{proof}
Apply the third cancellation clause of \autoref{thm:continuation-right-cancellation} to the two continuation spines
$$
c \xrightarrow{s} L \xrightarrow{m} q
\qquad\text{and}\qquad
c \xrightarrow{s'} L' \xrightarrow{m} q'.
$$
The shared prefix is $c$, the shared suffix is $m$, and the endpoint comparison is the displayed $\hsame(q,q')$. The cancellation clause returns $\hsame(s,s')$.
\end{proof}

\begin{theorem}[HopfAlgUp antipode uniqueness from convolution-inverse witnesses]
\label{thm:hopfalg-antipode-convolution-inverse-unique}
Fix a $\NameCert_{\RingUp}$ row $R$ and a $\NameCert_{\TensorProductUp}$ row $T$. Let $h:\mathsf{BHist}$ satisfy $\mathsf{HopfBialgCarrier}_{R,T}(h)$, and let $a,t,m,c,u,e:\mathsf{BHist}$ be carrier rows exposed by \autoref{def:hopfalg-bhist-bialgebra-carrier}. Let $s,s':\mathsf{BHist}$ be two antipode rows over these same carrier rows.

Assume both $s$ and $s'$ satisfy the full antipode obligation of \autoref{thm:hopfalg-antipode-convolution-inverse-obligation}: each has a $\mathsf{ConvLeft}$ packet and a $\mathsf{ConvRight}$ packet landing on a unit--counit endpoint. Write the left packets as $L$ and $L'$, and write their unit--counit endpoints as $q$ and $q'$. Assume the endpoint packets are represented by
$$
\Cont(u,e,q)
\qquad\text{and}\qquad
\Cont(u,e,q'),
$$
and that the left convolution packets expose the continuation faces
$$
\Cont(c,s,L)\land\Cont(L,m,q)
$$
and
$$
\Cont(c,s',L')\land\Cont(L',m,q').
$$
Then
$$
\hsame(s,s').
$$
Consequently, for $H=(a,t,s,u,e)$ and $H'=(a,t,s',u,e)$, the classifier of \autoref{def:hopfalg-antipode-ledger-classifier} gives $H\sim_{\HopfAlgUp}H'$.
\end{theorem}
\leanchecked{BEDC.Derived.HopfAlgUp.HopfAlgAntipodeClassifier\_convolution\_inverse\_unique}
\begin{proof}
The two unit--counit endpoint rows have the same source $u$ and the same tail $e$. Applying \autoref{thm:cont-transport-result-classified} to $\Cont(u,e,q)$ and $\Cont(u,e,q')$, with reflexivity on $u$, $e$, and $q$, gives the endpoint comparison $\hsame(q,q')$.

Now apply \autoref{lem:two-step-continuation-middle-cancellation} to the two displayed left convolution spines. This yields $\hsame(s,s')$.

The right convolution packets certify that both rows are full two-sided antipode witnesses for $\HopfAlgUp$; uniqueness is already forced by the shared left convolution face and the common unit--counit endpoint. Since the bialgebra row, tensor row, unit ledger, and counit ledger are the same in $H$ and $H'$, \autoref{def:hopfalg-antipode-ledger-classifier} then packages the componentwise comparison as $H\sim_{\HopfAlgUp}H'$.
\end{proof}
\leanvariant{BEDC.Derived.HopfAlgUp.HopfAlgBialgCarrier\_antipode\_convolution\_inverse\_unique}

\begin{theorem}[HopfAlgUp antipode classifier transport]
\label{thm:hopfalg-antipode-classifier-transport}
Let $H=(a,t,s,u,e)$ and $H'=(a',t',s',u',e')$ be two $\HopfAlgUp$ rows whose
ring, tensor, unit, and counit sources are paired by the carrier classifier.
Assume the left and right convolution packets of their antipode obligations
land on unit--counit endpoints that are themselves classified. Then
$\hsame(s,s')$, and the two five-tuples are related by the classifier of
\autoref{def:hopfalg-antipode-ledger-classifier}.
\end{theorem}
\begin{proof}
Start with the antipode obligation rows supplied by
\autoref{thm:hopfalg-antipode-convolution-inverse-obligation}. They expose
left and right convolution packets for $s$ and $s'$, and by hypothesis their
unit--counit endpoints are classified.

Transport the bialgebra, tensor, unit, and counit source rows along the given
carrier classifier comparisons. This places the two left convolution spines in
the same continuation shape, with classified source and endpoint rows.

Apply \autoref{lem:two-step-continuation-middle-cancellation} to the
transported left convolution packets. The classified unit--counit endpoints
give the endpoint comparison required by the cancellation lemma, so the
middle antipode rows satisfy $\hsame(s,s')$.

The right convolution packets ensure that both middle rows are full two-sided
antipode witnesses for the same transported Hopf algebra surface. Applying
\autoref{thm:hopfalg-antipode-convolution-inverse-unique} packages this
middle-row comparison with the source comparisons.

The classifier of \autoref{def:hopfalg-antipode-ledger-classifier} is exactly
the five-coordinate ledger classifier for $(a,t,s,u,e)$ and
$(a',t',s',u',e')$, so these comparisons give $H\sim_{\HopfAlgUp}H'$.
\end{proof}
\leanchecked{BEDC.Derived.HopfAlgUp.HopfAlgAntipodeClassifier\_transport}
