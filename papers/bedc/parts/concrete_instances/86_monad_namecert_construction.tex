\chapter{A Concrete Naming Certificate for $\MonadUp$}
\label{ch:concrete-instances-monad-namecert}

The monad interface $\MonadUp$ packages an endofunctor on a category with unit and multiplication natural transformations satisfying the monad laws. It depends on $\NameCert_{\NatTransUp}$ (\autoref{ch:concrete-instances-nattrans-namecert}) and $\NameCert_{\AdjunctionUp}$ (\autoref{ch:concrete-instances-adjunction-namecert}), and supplies the substrate for algebraic theories and computational effect models.

\section{Carrier definition}
\label{sec:monad-carrier}

\begin{theorem}[Monad adjunction endomorphism triangle results are empty]
\label{thm:monad-adjunction-endomorphism-triangle-results-empty}
For histories $p,a,unit,counit,left,right:\mathsf{BHist}$, suppose
$\mathsf{AdjunctionUnitCounitCarrier}(p,p,a,unit,counit,left,right)$ holds.
Then both triangle results are empty:
$$
\hsame(left,\emp)\land\hsame(right,\emp).
$$
\end{theorem}
\begin{proof}
Both the unit and counit components are endomorphism prefix components over
the same prefix $p$ and object $a$, so natural-transformation endomorphism
component exactness identifies each component with $\emp$. The two triangle
continuations are then compared with the empty right-unit continuation, first
in the unit-counit order and then in the counit-unit order.
\end{proof}
\leanchecked{BEDC.Derived.MonadUp.MonadAdjunctionEndomorphism\_triangle\_results\_empty}

\begin{theorem}[Monad adjunction endomorphism unit-counit middle is empty]
\label{thm:monad-adjunction-endomorphism-middle-between-unit-counit-empty}
For histories $p,a,unit,counit,left,right,bridge:\mathsf{BHist}$, suppose
$$
\mathsf{AdjunctionUnitCounitCarrier}(p,p,a,unit,counit,left,right)
$$
and $\ContR(unit,bridge,counit)$ hold. Then $bridge$ is empty:
$$
\hsame(bridge,\emp).
$$
\end{theorem}
\begin{proof}
The adjunction endomorphism carrier identifies both $unit$ and $counit$ with
$\emp$. With those endpoints collapsed, the displayed continuation is a
continuation from $\emp$ through $bridge$ back to $\emp$, and empty-result
inversion gives $bridge\hsame\emp$.
\end{proof}
\leanchecked{BEDC.Derived.MonadUp.MonadAdjunctionEndomorphism\_middle\_between\_unit\_counit\_empty}

\begin{theorem}[Monad adjunction endomorphism counit-unit middle is empty]
\label{thm:monad-adjunction-endomorphism-middle-between-counit-unit-empty}
For histories $p,a,unit,counit,left,right,bridge:\mathsf{BHist}$, suppose
$$
\mathsf{AdjunctionUnitCounitCarrier}(p,p,a,unit,counit,left,right)
$$
and $\ContR(counit,bridge,unit)$ hold. Then $bridge$ is empty:
$$
\hsame(bridge,\emp).
$$
\end{theorem}
\begin{proof}
The unit and counit are both endomorphism prefix components in the carrier, so
each is identified with $\emp$. After those endpoint identifications, the
displayed counit-unit continuation has empty endpoints, and empty-result
inversion identifies the middle history with $\emp$.
\end{proof}
\leanchecked{BEDC.Derived.MonadUp.MonadAdjunctionEndomorphism\_middle\_between\_counit\_unit\_empty}
