\chapter{A Concrete Naming Certificate for $\ContinuousMapUp$}
\label{ch:concrete-instances-continuous-namecert}

The continuous-map interface $\ContinuousMapUp$ has certificate obligations between two metric spaces from \autoref{ch:concrete-instances-metric-namecert}. The carrier is a BEDC graph witness between certified source and target point histories, together with a unary precision-indexed modulus relation. Totality over classifier-admitted source histories and single-valuedness up to the target classifier are certificate fields, so the public object is a relation-plus-ledger certificate rather than a host-level map or host-level numeric modulus.

\section{Carrier definition}

\begin{definition}[ContinuousMap carrier]
\label{def:continuous-carrier}
Use a graph relation between source point histories and target point histories, equipped with source and target metric ledgers, a totally bounded source witness, a unary precision ledger, and a modulus ledger relating requested source precision to target precision. The carrier includes totality for every source history accepted by the source classifier and single-valuedness up to the target classifier for any two target histories related to the same classifier-admitted source history.
\end{definition}
\leandef{BEDC.Derived.ContinuousUp.ContinuousFunctionCarrier}

\begin{definition}[Continuous modulus witness]
\label{def:continuous-modulus-witness}
For histories $source$, $modulus$, and $cert$,
$\mathsf{ContinuousModulusWitness}(source,modulus,cert)$ records the four
components
$$
\mathsf{UnaryHistory}(source),\quad
\mathsf{UnaryHistory}(modulus),\quad
\mathsf{UnaryHistory}(cert),\quad
\mathsf{Cont}(source,modulus,cert).
$$
\end{definition}
\leandef{BEDC.Derived.ContinuousUp.ContinuousModulusWitness}

\begin{theorem}[Continuous modulus witness empty-source exactness]
\label{thm:continuous-modulus-witness-empty-source-iff}
For histories $modulus$ and $target$, a one-step modulus witness whose source
is empty is exactly a unary target together with classifier equality between
the modulus and the target:
$$
\mathsf{ContinuousModulusWitness}(\emp,modulus,target)
\Longleftrightarrow
\mathsf{UnaryHistory}(target)\land\hsame(modulus,target).
$$
\end{theorem}
\begin{proof}
Unpack the witness. The continuation ledger is
$\ContR(\emp,modulus,target)$, so the left-unit continuation theorem identifies
the target with the modulus. This gives the forward classifier equality after
symmetry, while the target unary field is one of the witness fields.

Conversely, transport the target unary field across the classifier equality to
obtain a unary modulus. The source is unary by the empty-history constructor,
and the left-unit continuation equivalence rebuilds
$\ContR(\emp,modulus,target)$ from the same classifier equality.
\end{proof}
\leanchecked{BEDC.Derived.ContinuousUp.ContinuousModulusWitness\_empty\_source\_iff}

\begin{theorem}[Continuous function carrier composition closure]
\label{thm:continuous-function-carrier-comp-closed}
If $f$ carries a source history to a middle history with modulus $modF$, and
$g$ carries the middle history to a target history with modulus $modG$, then a
continuation composite $fg$ with modulus composite $modFG$ carries the source
history to the target history.
\end{theorem}
\begin{proof}
The two carrier witnesses give unary histories for the source, target, maps,
and moduli, plus continuation ledgers for the source-to-middle and
middle-to-target maps. Unary continuation closure gives unary histories for
the composite map and composite modulus. Case analysis on the continuation
constructors reduces the source-to-target continuation to associativity of
history append, while the supplied target-to-certificate ledger records the
composite modulus certificate.
\end{proof}
\leanchecked{BEDC.Derived.ContinuousUp.ContinuousFunctionCarrier\_comp\_closed}

\begin{theorem}[Continuous function carrier empty-map identity]
\label{thm:continuous-function-carrier-empty-map-identity}
For histories $source$, $modulus$, and $cert$, a certified modulus witness over
the source ledger determines the identity continuous-function carrier:
$$
\mathsf{ContinuousModulusWitness}(source,modulus,cert)
\Longrightarrow
\mathsf{ContinuousFunctionCarrier}(source,\emp,source,modulus,cert).
$$
\end{theorem}
\begin{proof}

and $cert$, together with the supplied certificate continuation
$\mathsf{Cont}(source,modulus,cert)$.

The empty map ledger is unary by the empty-history constructor. The graph part of
the identity carrier is the right-unit continuation
$\mathsf{Cont}(source,\emp,source)$.

Reassemble $\mathsf{ContinuousFunctionCarrier}$ from the unary source ledger
used for both source and target slots, the unary empty map ledger, the unary
modulus ledger, the right-unit graph continuation, and the supplied certificate
continuation.

The ten-field certificate fit is the following. The traditional target is the
identity uniformly continuous graph. The BEDC source is $BHist$ with
$\mathsf{Cont}$. The carrier is $\mathsf{ContinuousFunctionCarrier}$ and the
classifier is $\mathsf{hsame}$. The stability obligation is identity closure.
The ledger consists of source, empty map, modulus, and certificate histories.
The core theorem is this identity carrier, exactness is the displayed carrier,

\end{proof}

\begin{theorem}[Continuous function carrier empty-map exactness]
\label{thm:continuous-function-carrier-empty-map-iff}
For histories $source$, $target$, $modulus$, and $cert$, the empty graph map
inside $\mathsf{ContinuousFunctionCarrier}$ is exact:
$$
\mathsf{ContinuousFunctionCarrier}(source,\emp,target,modulus,cert)
\Longleftrightarrow
\hsame(target,source)\land
\mathsf{ContinuousModulusWitness}(target,modulus,cert).
$$
\end{theorem}
\begin{proof}
In the forward direction, unpack the carrier. The graph ledger is
$\mathsf{Cont}(source,\emp,target)$, so the continuation right-unit theorem gives
$\hsame(target,source)$. The target-modulus certificate ledger supplies
$\mathsf{Cont}(target,modulus,cert)$, and unary continuation closure supplies
the unary certificate history. Together with the target and modulus unary
fields, these assemble the displayed modulus witness.

Conversely, assume the target-source classifier and the modulus witness over
$target$. Transport the target unary ledger across $\hsame(target,source)$ to
obtain the source unary ledger. The empty map is unary, the graph ledger is the
right-unit continuation read from $\hsame(target,source)$, and the supplied
modulus witness gives the target, modulus, certificate, and certificate-ledger
fields of the carrier.
\end{proof}

$$
\begin{array}{l}
\hsame(source,source')\land\hsame(map,map')\land
\hsame(target,target')\land\hsame(modulus,modulus')\land
\hsame(cert,cert')

\Longrightarrow
\mathsf{ContinuousFunctionCarrier}(source',map',target',modulus',cert').
\end{array}

\begin{proof}
Unpack the concrete carrier witness into its four unary ledgers and two

\end{proof}

\begin{theorem}[Continuous function carrier prefix closure]
\label{thm:continuous-function-carrier-prefix-closed}
For histories $p$, $source$, $map$, $targetHist$, $modulus$, and $certHist$, a
visible unary prefix on the source, target, and certificate ledgers preserves
the full continuous-function carrier:

$$
\begin{array}{l}
\mathsf{UnaryHistory}(p)
\Longrightarrow
\mathsf{ContinuousFunctionCarrier}(source,map,targetHist,modulus,certHist)
\Longrightarrow
\\
\mathsf{ContinuousFunctionCarrier}(p\cdot source,map,p\cdot targetHist,modulus,
p\cdot certHist).
\end{array}
$$
\end{theorem}
\begin{proof}
Unpack the carrier into unary histories for the source, target, map, and
modulus, together with continuations $targetHist=source\cdot map$ and
$certHist=targetHist\cdot modulus$.

Unary append closure supplies the prefixed source and target histories
$p\cdot source$ and $p\cdot targetHist$. The map and modulus histories are
kept unchanged, so their unary witnesses are reused directly.

The graph ledger for the prefixed carrier follows by applying the prefix $p$ to
the continuation and reassociating append:

$$
(p\cdot source)\cdot map
=
p\cdot(source\cdot map)
=
p\cdot targetHist.
$$

The modulus-certificate ledger is transported in the same way:

$$
(p\cdot targetHist)\cdot modulus
=
p\cdot(targetHist\cdot modulus)
=
p\cdot certHist.
$$

The ten-field certificate fit is the following. The traditional target is
prefix-stable continuous graph evidence. The BEDC source is $BHist$ equipped
with $\mathsf{Cont}$. The carrier is $\mathsf{ContinuousFunctionCarrier}$ and
the classifier is $\mathsf{hsame}$. The stability obligation is prefix closure
of the graph and modulus-certificate ledgers. The ledger consists of source,
map, target, modulus, and certificate histories. The core theorem is this
prefix closure, exactness is the displayed carrier witness, the standard bridge
is deferred, and the status is $\mathsf{ProofTarget}$.
\end{proof}

\begin{theorem}[Continuous modulus witness prefix exactness]
\label{thm:continuous-modulus-witness-prefix-iff}
For histories $p,s,\delta,t$, a common visible unary prefix on the source and
target ledgers is exactly the same as unary evidence for the prefix together
with the unprefixed one-step modulus witness:

$$
\mathsf{ContinuousModulusWitness}(p\cdot s,\delta,p\cdot t)
\Longleftrightarrow
\mathsf{UnaryHistory}(p)\wedge
\mathsf{ContinuousModulusWitness}(s,\delta,t).
$$
\end{theorem}
\begin{proof}
For the forward direction, unpack the prefixed witness into unary histories for
$p\cdot s$, $\delta$, and $p\cdot t$, together with the continuation ledger

$$
(p\cdot s)\cdot\delta=p\cdot t.
$$

Unary append factorization on $p\cdot s$ recovers
$\mathsf{UnaryHistory}(p)$ and $\mathsf{UnaryHistory}(s)$. The same
factorization on $p\cdot t$ recovers $\mathsf{UnaryHistory}(t)$. Associativity
rewrites the displayed continuation ledger as
$p\cdot(s\cdot\delta)=p\cdot t$, and left cancellation on the continuation
ledger recovers $s\cdot\delta=t$. These data reassemble
$\mathsf{ContinuousModulusWitness}(s,\delta,t)$.

For the reverse direction, project $\mathsf{UnaryHistory}(p)$ and the
unprefixed witness from the conjunction. Unpacking the witness supplies unary
histories for $s$, $\delta$, and $t$, and the continuation ledger
$s\cdot\delta=t$.

Unary append closure gives unary histories for $p\cdot s$ and $p\cdot t$.
Prefixed continuation associativity transports the unprefixed ledger to
$(p\cdot s)\cdot\delta=p\cdot t$, so the prefixed witness is reconstructed.

The ten-field certificate fit is the following. The traditional target is
one-step modulus readback. The BEDC source is $BHist$ equipped with
$\mathsf{Cont}$. The carrier is $\mathsf{ContinuousModulusWitness}$ and the
classifier is $\mathsf{hsame}$. The stability obligation is prefix exactness.
The ledger consists of source, modulus, target, and prefix histories. The core
theorem is this iff, exactness is the displayed iff, the standard bridge is
deferred, and the status is $\mathsf{ProofTarget}$.
\end{proof}
\leanchecked{BEDC.Derived.ContinuousUp.ContinuousModulusWitness\_prefix\_iff}

exactness theorem to remove the common right suffix, then repack the graph
carrier fields.

For the reverse direction, turn the unsuffixed target-modulus certificate leg
into a continuous modulus witness, apply suffix exactness to append $p$ to the
modulus and certificate histories, and repack the original graph fields with
the suffixed certificate leg.
\end{proof}
\leanchecked{BEDC.Derived.ContinuousUp.ContinuousFunctionCarrier\_modulus\_suffix\_iff}

\begin{theorem}[Continuous function carrier prefixed graph-chain factorization]
\label{thm:continuous-function-carrier-prefixed-graph-chain-factorizes}
For histories $p,source,map,target,\delta_1,\delta_2,\delta,cert$, if
$\delta_1$ and $\delta_2$ are unary, $\mathsf{Cont}(\delta_1,\delta_2,\delta)$
holds, and
$$
\mathsf{ContinuousFunctionCarrier}
(p\cdot source,map,p\cdot target,\delta,p\cdot cert)
$$
holds, then there is a middle history $middle$ such that
$$
\mathsf{ContinuousFunctionCarrier}(source,map,target,\delta,cert)
\land
\mathsf{ContinuousModulusWitness}(target,\delta_1,middle)
\land
\mathsf{ContinuousModulusWitness}(middle,\delta_2,cert).
$$
\end{theorem}
\begin{proof}
Apply prefix exactness for continuous-function carriers to remove the common
visible prefix from the source, target, and certificate ledgers. This recovers
the unprefixed carrier and in particular the target-to-certificate continuation
$\mathsf{Cont}(target,\delta,cert)$.

Choose $middle=target\cdot\delta_1$. The first modulus witness is the canonical
continuation from $target$ through $\delta_1$ to this middle history. For the
second witness, combine $\mathsf{Cont}(\delta_1,\delta_2,\delta)$ with the
recovered target-to-certificate continuation and reassociate the BEDC append
spine, obtaining $\mathsf{Cont}(middle,\delta_2,cert)$.
\end{proof}
\leanchecked{BEDC.Derived.ContinuousUp.ContinuousFunctionCarrier\_prefixed\_graph\_chain\_factorizes}

\begin{theorem}[Continuous function carrier prefixed graph-chain first factorization]
\label{thm:continuous-function-carrier-prefixed-graph-chain-factorization}
For histories $p,source,map,target,\delta_1,\delta_2,\delta,cert$, if
$\delta_1$ and $\delta_2$ are unary, $\mathsf{Cont}(\delta_1,\delta_2,\delta)$
holds, and
$$
\mathsf{ContinuousFunctionCarrier}
(p\cdot source,map,p\cdot target,\delta,p\cdot cert)
$$
holds, then there is a middle history $middle$ such that
$$
\mathsf{ContinuousFunctionCarrier}(source,map,target,\delta_1,middle)
\land
\mathsf{ContinuousModulusWitness}(middle,\delta_2,cert).
$$
\end{theorem}
\begin{proof}
Apply prefix exactness for continuous-function carriers to remove the common
visible prefix. This exposes the unprefixed graph ledger
$\mathsf{Cont}(source,map,target)$ and the composite target-certificate ledger
$\mathsf{Cont}(target,\delta,cert)$.

Choose $middle=target\cdot\delta_1$. The first-stage carrier uses the original
source, map, and target carrier data, the graph ledger, the unary witness for
$\delta_1$, and the canonical continuation
$\mathsf{Cont}(target,\delta_1,middle)$. The second-stage modulus witness uses
the unary closure of $middle$, the unary witness for $\delta_2$, unary closure
of $cert$, and associativity of BEDC append to factor the composite ledger
through $\mathsf{Cont}(\delta_1,\delta_2,\delta)$.
\end{proof}
\leanchecked{BEDC.Derived.ContinuousUp.ContinuousFunctionCarrier\_prefixed\_graph\_chain\_factorization}
