\chapter{A Concrete Naming Certificate for $\ContinuousMapUp$}
\label{ch:concrete-instances-continuous-namecert}

The continuous-map interface $\ContinuousMapUp$ has certificate obligations between two metric spaces from \autoref{ch:concrete-instances-metric-namecert}. The carrier is a BEDC graph witness between certified source and target point histories, together with a unary precision-indexed modulus relation. Totality over classifier-admitted source histories and single-valuedness up to the target classifier are certificate fields, so the public object is a relation-plus-ledger certificate rather than a host-level map or host-level numeric modulus.

\section{Carrier definition}

\begin{definition}[ContinuousMap carrier]
\label{def:continuous-carrier}
Use a graph relation between source point histories and target point histories, equipped with source and target metric ledgers, a totally bounded source witness, a unary precision ledger, and a modulus ledger relating requested source precision to target precision. The carrier includes totality for every source history accepted by the source classifier and single-valuedness up to the target classifier for any two target histories related to the same classifier-admitted source history.
\end{definition}
\leandef{BEDC.Derived.ContinuousUp.ContinuousFunctionCarrier}

\begin{definition}[Continuous modulus witness]
\label{def:continuous-modulus-witness}
For histories $source$, $modulus$, and $cert$,
$\mathsf{ContinuousModulusWitness}(source,modulus,cert)$ records the four
components
$$
\mathsf{UnaryHistory}(source),\quad
\mathsf{UnaryHistory}(modulus),\quad
\mathsf{UnaryHistory}(cert),\quad
\mathsf{Cont}(source,modulus,cert).
$$
\end{definition}
\leandef{BEDC.Derived.ContinuousUp.ContinuousModulusWitness}

\begin{theorem}[Continuous modulus witness empty-source exactness]
\label{thm:continuous-modulus-witness-empty-source-iff}
For histories $modulus$ and $target$, a one-step modulus witness whose source
is empty is exactly a unary target together with classifier equality between
the modulus and the target:
$$
\mathsf{ContinuousModulusWitness}(\emp,modulus,target)
\Longleftrightarrow
\mathsf{UnaryHistory}(target)\land\hsame(modulus,target).
$$
\end{theorem}
\begin{proof}
Unpack the witness. The continuation ledger is
$\ContR(\emp,modulus,target)$, so the left-unit continuation theorem identifies
the target with the modulus. This gives the forward classifier equality after
symmetry, while the target unary field is one of the witness fields.

Conversely, transport the target unary field across the classifier equality to
obtain a unary modulus. The source is unary by the empty-history constructor,
and the left-unit continuation equivalence rebuilds
$\ContR(\emp,modulus,target)$ from the same classifier equality.
\end{proof}
\leanchecked{BEDC.Derived.ContinuousUp.ContinuousModulusWitness\_empty\_source\_iff}

\begin{theorem}[Continuous modulus witness empty-modulus exactness]
\label{thm:continuous-modulus-witness-empty-modulus-iff}

$$

\mathsf{UnaryHistory}(target)\land

\end{theorem}

Conversely, use the empty-history unary constructor for the modulus. The
supplied endpoint classifier, after symmetry, is exactly the right-unit
continuation equivalence needed to rebuild $\ContR(source,\emp,target)$.
Together with the two unary endpoint fields, this reassembles the witness.
\end{proof}

\begin{theorem}[Continuous function carrier composition closure]
\label{thm:continuous-function-carrier-comp-closed}
If $f$ carries a source history to a middle history with modulus $modF$, and
$g$ carries the middle history to a target history with modulus $modG$, then a
continuation composite $fg$ with modulus composite $modFG$ carries the source
history to the target history.
\end{theorem}
\begin{proof}
The two carrier witnesses give unary histories for the source, target, maps,
and moduli, plus continuation ledgers for the source-to-middle and
middle-to-target maps. Unary continuation closure gives unary histories for
the composite map and composite modulus. Case analysis on the continuation
constructors reduces the source-to-target continuation to associativity of
history append, while the supplied target-to-certificate ledger records the
composite modulus certificate.
\end{proof}

\begin{theorem}[Continuous function carrier prefix closure]
\label{thm:continuous-function-carrier-prefix-closed}
For histories $p$, $source$, $map$, $targetHist$, $modulus$, and $certHist$, a
visible unary prefix on the source, target, and certificate ledgers preserves
the full continuous-function carrier:

$$
\begin{array}{l}
\mathsf{UnaryHistory}(p)
\Longrightarrow
\mathsf{ContinuousFunctionCarrier}(source,map,targetHist,modulus,certHist)
\Longrightarrow
\\
\mathsf{ContinuousFunctionCarrier}(p\cdot source,map,p\cdot targetHist,modulus,
p\cdot certHist).
\end{array}
$$
\end{theorem}
\begin{proof}
Unpack the carrier into unary histories for the source, target, map, and
modulus, together with continuations $targetHist=source\cdot map$ and
$certHist=targetHist\cdot modulus$.

Unary append closure supplies the prefixed source and target histories
$p\cdot source$ and $p\cdot targetHist$. The map and modulus histories are
kept unchanged, so their unary witnesses are reused directly.

The graph ledger for the prefixed carrier follows by applying the prefix $p$ to
the continuation and reassociating append:

$$
(p\cdot source)\cdot map
=
p\cdot(source\cdot map)
=
p\cdot targetHist.
$$

The modulus-certificate ledger is transported in the same way:

$$
(p\cdot targetHist)\cdot modulus
=
p\cdot(targetHist\cdot modulus)
=
p\cdot certHist.
$$

The ten-field certificate fit is the following. The traditional target is
prefix-stable continuous graph evidence. The BEDC source is $BHist$ equipped
with $\mathsf{Cont}$. The carrier is $\mathsf{ContinuousFunctionCarrier}$ and
the classifier is $\mathsf{hsame}$. The stability obligation is prefix closure

map, target, modulus, and certificate histories. The core theorem is this

\end{proof}

rewrites the displayed continuation ledger as
$p\cdot(s\cdot\delta)=p\cdot t$, and left cancellation on the continuation
ledger recovers $s\cdot\delta=t$. These data reassemble
$\mathsf{ContinuousModulusWitness}(s,\delta,t)$.

unprefixed witness from the conjunction. Unpacking the witness supplies unary

\bigl(\mathsf{ContinuousModulusWitness}(source,\Ezero(modulus),target)

\begin{theorem}[Continuous modulus witness left e1 result cases]
\label{thm:continuous-modulus-witness-left-e1-result-cases}
For histories $source,modulus,target$, a one-step modulus witness from a
visible source endpoint to a visible target endpoint has only the empty modulus
case or a visible modulus tail:
$$
\mathsf{ContinuousModulusWitness}(\Eone(source),modulus,\Eone(target))
\Longrightarrow
\bigl(modulus=\emp\land\mathsf{UnaryHistory}(source)\land
\hsame(source,target)\bigr)
\lor
\bigl(\exists modulusTail:\mathsf{BHist}.\;
modulus=\Eone(modulusTail)\land
\mathsf{ContinuousModulusWitness}(\Eone(source),modulusTail,target)\bigr).
$$
\end{theorem}
\begin{proof}
Unpack the modulus witness and invert the continuation ledger with visible
result $\Eone(target)$. The empty modulus case strips the outer visible
constructor from the source-target equality and keeps the unary source
evidence. The zero-tagged modulus case is ruled out by constructor
no-confusion. In the visible modulus case, remove one $\Eone$ constructor from
the continuation ledger and repack the remaining fields as the tail witness.
\end{proof}
\leanchecked{BEDC.Derived.ContinuousUp.ContinuousModulusWitness\_left\_e1\_result\_cases}
