\chapter{A Concrete Naming Certificate for $\ContinuousMapUp$}
\label{ch:concrete-instances-continuous-namecert}

The continuous-map interface $\ContinuousMapUp$ has certificate obligations between two metric spaces from \autoref{ch:concrete-instances-metric-namecert}. The carrier is a BEDC graph witness between certified source and target point histories, together with a unary precision-indexed modulus relation. Totality over classifier-admitted source histories and single-valuedness up to the target classifier are certificate fields, so the public object is a relation-plus-ledger certificate rather than a host-level map or host-level numeric modulus.

\section{Carrier definition}

\begin{definition}[ContinuousMap carrier]
\label{def:continuous-carrier}
Use a graph relation between source point histories and target point histories, equipped with source and target metric ledgers, a totally bounded source witness, a unary precision ledger, and a modulus ledger relating requested source precision to target precision. The carrier includes totality for every source history accepted by the source classifier and single-valuedness up to the target classifier for any two target histories related to the same classifier-admitted source history.
\end{definition}

\section{The certificate}

\begin{definition}[ContinuousMap source specification]
\label{def:continuous-source-specification}
The source is a pair of certified metric spaces and the totally bounded subset discipline on the domain.
\end{definition}

\begin{definition}[ContinuousMap pattern specification]
\label{def:continuous-pattern-specification}
The pattern is graph-with-modulus formation over certified metric spaces and totally bounded source witnesses. A witness supplies the graph relation, the domain and codomain classifier ledgers, and the unary modulus relation that converts a target precision request into the source precision obligation needed for the graph.
\end{definition}

\begin{definition}[ContinuousMap classifier specification]
\label{def:continuous-classifier-specification}
The classifier relates two continuous-map witnesses when their graph outputs agree under the codomain classifier on every pair of source histories related by the source classifier, and when their unary modulus ledgers mutually refine each other for every precision history admitted by the precision classifier.
\end{definition}

\begin{definition}[ContinuousMap stability certificate]
\label{def:continuous-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item continuous-map classifier reflexivity
  \item continuous-map classifier symmetry
  \item continuous-map classifier transitivity
  \item modulus validity for uniform continuity
  \item graph and modulus replacement under the continuous-map classifier
  \item closure under identity
  \item closure under composition by composing moduli
\end{enumerate}
\end{definition}

\begin{theorem}[Continuous modulus chain factorization]
\label{thm:continuous-modulus-chain-factorizes}
For histories $s,\delta_1,\delta_2,t$, assume the two-link chain
$$
\mathsf{ContinuousModulusChain}(s,\delta_1,\delta_2,t).
$$
Then there is a history $m$ such that both one-step links are certified:
$$
\mathsf{ContinuousModulusWitness}(s,\delta_1,m)
\quad\text{and}\quad
\mathsf{ContinuousModulusWitness}(m,\delta_2,t).
$$
\end{theorem}
\begin{proof}
Unpack the chain into unary data for $s$, $\delta_1$, $\delta_2$, and $t$,
together with a middle history $m$ and continuation ledgers
$s\cdot\delta_1=m$ and $m\cdot\delta_2=t$.

The first ledger and the unary source and modulus data assemble
$$
\mathsf{ContinuousModulusWitness}(s,\delta_1,m).
$$

Unary continuation closure then makes the middle history unary. The second
ledger and the unary $\delta_2$ and target data assemble
$$
\mathsf{ContinuousModulusWitness}(m,\delta_2,t).
$$

Taking this middle history as the existential witness gives the displayed
factorization of the chain into its two certified one-step modulus witnesses.
\end{proof}

\begin{theorem}[Continuous modulus chain composite closure]
\label{thm:continuous-modulus-chain-composite-closed}
For unary histories $s,\delta_1,\delta_2,t$, a two-link modulus chain from
$s$ through $\delta_1$ and $\delta_2$ to $t$, together with a continuation
ledger composing $\delta_1$ and $\delta_2$ into $\delta$, determines a single
continuous modulus witness from $s$ to $t$ using $\delta$:
$$
\mathsf{ContinuousModulusChain}(s,\delta_1,\delta_2,t)
\Longrightarrow
\mathsf{Cont}(\delta_1,\delta_2,\delta)
\Longrightarrow
\mathsf{ContinuousModulusWitness}(s,\delta,t).
$$
\end{theorem}
\begin{proof}
Unpack the chain into unary endpoint and modulus data plus a middle history
with ledgers $m=s\cdot\delta_1$ and $t=m\cdot\delta_2$. The composite modulus
is unary by closure of unary continuation. The source-to-target continuation is
the associativity equality $(s\cdot\delta_1)\cdot\delta_2
=s\cdot(\delta_1\cdot\delta_2)$.
\end{proof}

\begin{theorem}[Prefixed continuous modulus composite closure]
\label{thm:continuous-modulus-prefixed-composite-closed}
For unary histories $p,s,\delta_1,\delta_2,t,\delta$, a unary prefix $p$, a
two-link modulus chain from $s$ through $\delta_1$ and $\delta_2$ to $t$, and a
continuation ledger composing $\delta_1$ and $\delta_2$ into $\delta$ determine
a prefixed continuous modulus witness:

$$
\begin{array}{l}
\mathsf{UnaryHistory}(p)
\Longrightarrow
\mathsf{ContinuousModulusChain}(s,\delta_1,\delta_2,t)
\Longrightarrow
\\
\mathsf{Cont}(\delta_1,\delta_2,\delta)
\Longrightarrow
\mathsf{ContinuousModulusWitness}(p\cdot s,\delta,p\cdot t).
\end{array}
$$

\end{theorem}
\begin{proof}
Unpack the chain into unary endpoint and modulus histories, together with the
continuation ledgers from $s$ to the middle history and from the middle history
to $t$.

The prefixed source and target histories are unary by unary append closure
applied to $p$ with $s$ and $t$. The composite modulus history is unary by unary
continuation closure applied to the two modulus histories and their composition
ledger.

The remaining continuation ledger is obtained by associating the concatenation
spine:

$$
(p\cdot s)\cdot\delta
=
p\cdot(s\cdot\delta)
=
p\cdot t.
$$

Thus the prefixed endpoints and the same composite modulus satisfy the public
continuous modulus witness predicate.
\end{proof}

\begin{definition}[ContinuousMap ledger policy]
\label{def:continuous-ledger-policy}
Record the graph relation, source and target metric ledgers, the totally bounded subset witness, the unary precision ledger, and the modulus ledger.
\end{definition}

\begin{definition}[$\ContinuousMapUp$ certificate obligations]
\label{def:continuous-certificate-obligations}
The carrier, source, pattern, classifier, stability, and ledger data listed above are the obligations required for the interface name $\ContinuousMapUp$.
\end{definition}

\paragraph{Primitive boundary.}
$\ContinuousMapUp$ is not a BEDC primitive. It is admitted only when those obligations are supplied.

\begin{remark}[Zero-axiom encoding note]
\label{rem:continuous-zero-axiom-note}
The encoding avoids extensional equality of maps and host $\texttt{Nat}$ by using graph classifier sameness and unary precision histories. The modulus object is a certified relation between precision histories, so no host-level numeric modulus is exposed at the certificate boundary.
\end{remark}
