\chapter{A Concrete Naming Certificate for \(\MatrixUp\)}
\label{ch:concrete-instances-matrix-namecert}

The matrix interface \(\MatrixUp\) has certificate obligations over a scalar ring from \autoref{ch:concrete-instances-ring-namecert} and two natural-number dimensions from \autoref{ch:concrete-instances-nat-namecert}. The carrier is functions \(\mathsf{Fin}(n) \times \mathsf{Fin}(m) \to R\); the classifier is pointwise scalar sameness on all index pairs. Matrix addition and multiplication are derived operations governed by the ring's certificate fields. It is not a primitive matrix type; it is admitted only as a derived function carrier.

\section{Carrier definition}

\begin{definition}[Matrix carrier]
\label{def:matrix-carrier}
Use functions \(\mathsf{Fin}(n)\times\mathsf{Fin}(m)\to R\) with \(n,m:\NatUp\) and scalar ring \(R\).
\end{definition}

\section{Concrete singleton empty-history matrix instance}
\label{sec:matrix-concrete-singleton-instance}

\begin{definition}[Concrete singleton empty-history matrix instance]
\label{def:singleton-empty-history-matrix-instance}
Let $A_{\emp}$ be the singleton history source of
Definition~\ref{def:list-singleton-history-source} at $a_0=\emp$. Use the
classifier
$$
\rho_{\emp}(h,k)\Longleftrightarrow
A_{\emp}(h)\land A_{\emp}(k)\land\hsame(h,k).
$$
The row-dimension witness and column-dimension witness are both represented by
the empty unary history $\emp$. The scalar side is the singleton empty-history
ring of Definition~\ref{def:singleton-empty-history-ring-instance}, and the
row and column vector layers are read through the singleton empty-history
vector-space instance of
Definition~\ref{def:singleton-empty-history-vecspace-instance}.

The matrix carrier is the same singleton source $A_{\emp}$ on
$\mathsf{BHist}$ codes. Define the displayed matrix operations by
$$
0_M:=\emp,\qquad I_M:=\emp,\qquad
M\oplus N:=\mathsf{append}(M,N),\qquad
M\odot N:=\mathsf{append}(M,N).
$$
The certificate fit has ten fields:
\begin{enumerate}
  \item Traditional target: the terminal singleton matrix algebra.
  \item BEDC source: $\mathsf{BHist}$ together with the singleton ring and
  $\VecSpaceUp$ layers.
  \item Carrier choice: $A_{\emp}$.
  \item Classifier choice: $\rho_{\emp}$.
  \item Stability obligations: closure, zero laws, identity laws,
  associativity, and congruence for the displayed operations.
  \item Ledger: the singleton endpoint together with row and column witnesses
  represented by $\emp$.
  \item Core theorem list:
  Theorem~\ref{thm:singleton-empty-history-matrix-laws}.
  \item Exactness predicate: singleton endpoint classification.
  \item Standard bridge: the terminal matrix object.
  \item Status: $\mathsf{ProofTarget}$.
\end{enumerate}
\end{definition}

\begin{theorem}[Matrix singleton classifier append split empty]
\label{thm:matrix-singleton-classifier-append-split-empty}
For $M,N,h:\mathsf{BHist}$,
$$
\rho_{\emp}(\mathsf{append}(M,N),h)
\quad\Longleftrightarrow\quad
M\hsame\emp\land N\hsame\emp\land A_{\emp}(h).
$$
\end{theorem}
\begin{proof}
The forward direction reads the left carrier component of
$\rho_{\emp}(\mathsf{append}(M,N),h)$ as
$\mathsf{append}(M,N)\hsame\emp$. Empty-result inversion for append splits this
endpoint witness into $M\hsame\emp$ and $N\hsame\emp$, while the right carrier
component supplies $A_{\emp}(h)$.

Conversely, $M\hsame\emp$ and $N\hsame\emp$ reconstruct the singleton carrier
witness for $\mathsf{append}(M,N)$. Together with $A_{\emp}(h)$, transitivity of
$\hsame$ through $\emp$ gives the classifier comparison.
\end{proof}
\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonClassifier\_append\_split\_empty\_iff}

\begin{theorem}[Matrix singleton add-mul classifier iff]
\label{thm:matrix-singleton-add-mul-classifier-iff}
For $M,N:\mathsf{BHist}$, the singleton matrix classifier relates the displayed
addition and multiplication results exactly when both inputs are singleton
empty endpoints:
$$
\rho_{\emp}(M\oplus N,M\odot N)
\quad\Longleftrightarrow\quad
M\hsame\emp\land N\hsame\emp.
$$
\end{theorem}
\begin{proof}
Both displayed operations are the same BEDC append spine
$\mathsf{append}(M,N)$. In the forward direction, the left carrier component
of $\rho_{\emp}(M\oplus N,M\odot N)$ says that this append spine is
$\hsame$ to $\emp$. Empty-result inversion for append splits it into the two
endpoint witnesses $M\hsame\emp$ and $N\hsame\emp$.

Conversely, the two endpoint witnesses reconstruct the singleton carrier for
$\mathsf{append}(M,N)$. Since $M\oplus N$ and $M\odot N$ unfold to that same
append spine, reflexivity of $\hsame$ supplies the classifier relation between
the displayed operation results.
\end{proof}
\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonAddMul\_classifier\_iff}

\begin{theorem}[Matrix singleton classifier visible right append exclusion]
\label{thm:matrix-singleton-classifier-append-visible-right-absurd}
For histories $h,p,q:\mathsf{BHist}$, a singleton matrix classifier witness
cannot target an append whose right factor has a visible head:
$$
\rho_{\emp}(h,\mathsf{append}(p,\mathsf{e0}(q)))\to\bot
\qquad\text{and}\qquad
\rho_{\emp}(h,\mathsf{append}(p,\mathsf{e1}(q)))\to\bot.
$$
\end{theorem}
\begin{proof}
Unfold the singleton matrix classifier. A classifier witness supplies the
target carrier endpoint
$\mathsf{append}(p,\mathsf{e0}(q))\hsame\emp$ in the first case, or
$\mathsf{append}(p,\mathsf{e1}(q))\hsame\emp$ in the second case. Empty-result
inversion for append forces the visible right factor itself to be $\emp$,
contradicting the disjointness of each visible constructor from $\emp$.
\end{proof}
\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonClassifier\_append\_visible\_right\_absurd}

\begin{theorem}[Matrix singleton classifier append pair carrier]
\label{thm:matrix-singleton-classifier-append-pair-carrier}
For $M,N,P,Q:\mathsf{BHist}$,
$$
\rho_{\emp}(\mathsf{append}(M,N),\mathsf{append}(P,Q))
\quad\Longleftrightarrow\quad
A_{\emp}(M)\land A_{\emp}(N)\land A_{\emp}(P)\land A_{\emp}(Q).
$$
\end{theorem}
\begin{proof}
The classifier exposes singleton carrier witnesses for both append endpoints.
Empty-result inversion for append splits these witnesses into the four displayed
endpoint carriers. Conversely, the four endpoint carriers rebuild the two empty
append endpoints, and transitivity of $\hsame$ through $\emp$ supplies the
classifier comparison between them.
\end{proof}
\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonClassifier\_append\_pair\_carrier\_iff}

\begin{theorem}[Matrix singleton classifier self-context split]
\label{thm:matrix-singleton-classifier-self-context-split}
For histories $a,x,b:\mathsf{BHist}$,
$$
\rho_{\emp}(\mathsf{append}(a,\mathsf{append}(x,a)),b)
\Longleftrightarrow
A_{\emp}(a)\land\rho_{\emp}(x,b).
$$
\end{theorem}
\begin{proof}
The forward implication splits the source carrier of the classifier twice by
empty-result inversion for append. This gives the carrier witness for the
repeated context $a$ and the carrier witness for the middle history $x$; the
target carrier field and transitivity through $\emp$ rebuild
$\rho_{\emp}(x,b)$.

Conversely, the carrier for $a$ and the source carrier inside
$\rho_{\emp}(x,b)$ rebuild the inner and outer empty append endpoints. The
target carrier is inherited from $\rho_{\emp}(x,b)$, and both endpoints compare
through the empty history.
\end{proof}
\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonClassifier\_append\_self\_context\_split\_iff}

\begin{theorem}[Matrix singleton classifier append-left cancellation]
\label{thm:matrix-singleton-classifier-append-left-cancellation}
For histories $P,Q,R:\mathsf{BHist}$,
$$
\rho_{\emp}(\mathsf{append}(R,P),\mathsf{append}(R,Q))
\Longleftrightarrow
A_{\emp}(R)\land \rho_{\emp}(P,Q).
$$
\end{theorem}
\begin{proof}
In the forward direction, the two carrier fields say that
$\mathsf{append}(R,P)$ and $\mathsf{append}(R,Q)$ are empty. Empty-result
inversion for append splits these witnesses into $R\hsame\emp$,
$P\hsame\emp$, and $Q\hsame\emp$. The shared left endpoint gives
$A_{\emp}(R)$, and the two remaining endpoint witnesses rebuild
$\rho_{\emp}(P,Q)$.

Conversely, unpack $A_{\emp}(R)$ and $\rho_{\emp}(P,Q)$. The shared left
endpoint and each singleton classified endpoint rebuild both empty append
endpoints. Transitivity of $\hsame$ through $\emp$ then supplies the classifier
comparison between the two appended histories.
\end{proof}
\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonClassifier\_append\_left\_cancel\_iff}

\begin{theorem}[Matrix singleton classifier append-right cancellation]
\label{thm:matrix-singleton-classifier-append-right-cancellation}
For histories $P,Q,R:\mathsf{BHist}$,
$$
\rho_{\emp}(\mathsf{append}(P,R),\mathsf{append}(Q,R))
\Longleftrightarrow
\rho_{\emp}(P,Q)\land A_{\emp}(R).
$$
\end{theorem}
\begin{proof}
In the forward direction, the two carrier fields say that
$\mathsf{append}(P,R)$ and $\mathsf{append}(Q,R)$ are empty. Empty-result
inversion for append splits these witnesses into $P\hsame\emp$,
$Q\hsame\emp$, and $R\hsame\emp$. The first two endpoint witnesses rebuild
$\rho_{\emp}(P,Q)$, and the shared right endpoint gives $A_{\emp}(R)$.

Conversely, unpack $\rho_{\emp}(P,Q)$ and $A_{\emp}(R)$. All three displayed
histories are singleton endpoints, so both appended histories reduce to
$\emp$, and the singleton classifier follows by reflexivity of $\hsame$.
\end{proof}
\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonClassifier\_append\_right\_cancel\_iff}

\begin{theorem}[Matrix singleton classifier append common context]
\label{thm:matrix-singleton-classifier-append-common-context}
For histories $L,P,Q,R:\mathsf{BHist}$,
$$
\rho_{\emp}(\mathsf{append}(L,\mathsf{append}(P,R)),
  \mathsf{append}(L,\mathsf{append}(Q,R)))
\Longleftrightarrow
A_{\emp}(L)\land \rho_{\emp}(P,Q)\land A_{\emp}(R).
$$
\end{theorem}
\begin{proof}
In the forward direction, the two classifier carrier fields say that both
common-context append endpoints are empty. Empty-result inversion first splits
off the shared left context $L$, then splits the middle append endpoints. This
yields $A_{\emp}(L)$, $A_{\emp}(P)$, $A_{\emp}(Q)$, and $A_{\emp}(R)$; the
middle comparison $\rho_{\emp}(P,Q)$ follows by transitivity through $\emp$.

Conversely, $A_{\emp}(L)$, $\rho_{\emp}(P,Q)$, and $A_{\emp}(R)$ reconstruct
both common-context append endpoints as empty. Transitivity through $\emp$
then supplies the singleton classifier relation between the displayed
endpoints.
\end{proof}
\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonClassifier\_append\_common\_context\_iff}

\begin{theorem}[Singleton empty-history matrix laws]
\label{thm:singleton-empty-history-matrix-laws}
For the singleton instance of
Definition~\ref{def:singleton-empty-history-matrix-instance}, the displayed
matrix carrier and operations satisfy the concrete $\MatrixUp$ law package:
\begin{enumerate}
  \item $0_M$ and $I_M$ are carried by $A_{\emp}$.
  \item If $A_{\emp}(M)$ and $A_{\emp}(N)$, then
  $A_{\emp}(M\oplus N)$ and $A_{\emp}(M\odot N)$.
  \item $0_M$ is a left and right unit for $\oplus$ under $\rho_{\emp}$.
  \item $I_M$ is a left and right identity for $\odot$ under $\rho_{\emp}$.
  \item $\oplus$ and $\odot$ are associative under $\rho_{\emp}$.
  \item $\rho_{\emp}$ is reflexive, symmetric, and transitive on carried
  matrix codes, and it is respected by $\oplus$ and $\odot$.
\end{enumerate}
\end{theorem}
\begin{proof}
The carrier endpoint is supplied by the singleton source witness from
Definition~\ref{def:list-singleton-history-source} at $a_0=\emp$. The row and
column witnesses use the same empty unary history, so the dimension ledger is
closed at the singleton endpoint.

For operation closure, carried inputs are classified as singleton endpoints.
The displayed operations are either $\emp$ or $\mathsf{append}$ applied to
singleton endpoints. The BEDC continuation unit laws cited in
Theorem~\ref{thm:concrete-history-monoid-laws} identify the resulting append
endpoint with $\emp$, so the singleton source again carries the result.

The zero laws for $\oplus$ and the identity laws for $\odot$ are the same
unit calculation. On the left and on the right, the operation compares
$\mathsf{append}(\emp,M)$ or $\mathsf{append}(M,\emp)$ with $M$ under
$\rho_{\emp}$, and the continuation unit laws provide the $\hsame$ component.
The endpoint components come from $A_{\emp}(M)$ and $A_{\emp}(\emp)$.

Associativity of $\oplus$ and $\odot$ uses the continuation associativity law
already used by the singleton monoid, ring, and vector-space chapters. Both
operation symbols are represented by $\mathsf{append}$, so the two
parenthesized products share a single BEDC append spine up to $\hsame$.

For classifier congruence, unpack the two $\rho_{\emp}$ hypotheses. Closure
keeps both composite endpoints in $A_{\emp}$, and the relation components are
combined by reflexivity, symmetry, and transitivity of $\hsame$ after the
append unit and associativity reductions. This gives the required
$\rho_{\emp}$ comparison for both $\oplus$ and $\odot$.
\end{proof}
\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonEmptyHistory\_laws}

\begin{theorem}[Singleton empty-history matrix endpoint collapse]
\label{thm:singleton-empty-history-matrix-endpoint-collapse}
With the notation of
Definition~\ref{def:singleton-empty-history-matrix-instance}, if
$A_{\emp}(M)$ and $A_{\emp}(N)$, then
\begin{enumerate}
  \item $\rho_{\emp}(M\oplus N,\emp)$.
  \item $\rho_{\emp}(M\odot N,\emp)$.
  \item $\rho_{\emp}(M\oplus N,M\odot N)$.
\end{enumerate}
\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:singleton-empty-history-matrix-laws} to the carried
inputs $M$ and $N$. Its closure clause puts both $M\oplus N$ and $M\odot N$
back in $A_{\emp}$, while the singleton carrier witnesses identify $M$ and
$N$ with the endpoint $\emp$.

By the displayed definitions, both operations are
$\mathsf{append}(M,N)$. Replace the carried inputs by their singleton endpoint
witnesses, then use the left and right continuation unit laws cited in
Theorem~\ref{thm:concrete-history-monoid-laws} to reduce the resulting append
history to $\emp$.

For $\rho_{\emp}(M\oplus N,\emp)$ and
$\rho_{\emp}(M\odot N,\emp)$, combine the closure witnesses with the
corresponding $\hsame$ endpoint calculation. Symmetry is used when the unit
witness is oriented from $\emp$ to the displayed operation rather than from
the displayed operation to $\emp$.

For $\rho_{\emp}(M\oplus N,M\odot N)$, compose the first endpoint comparison
with the symmetry of the second by transitivity of $\hsame$. The two carrier
witnesses are the closure witnesses already supplied by the law package, so
the classifier definition gives the required comparison.
\end{proof}
\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonEmptyHistory\_endpoint\_collapse}

The $\MatrixUp$ schema is instantiated by this concrete $\mathsf{BHist}$
carrier and operations, so formalization can target the endpoint-collapse and
law package rather than an abstract operator schema.

\begin{theorem}[Singleton empty-history matrix endpoint exactness]
\label{thm:singleton-empty-history-matrix-endpoint-exactness}
For the singleton instance of
Definition~\ref{def:singleton-empty-history-matrix-instance}, if
$A_{\emp}(M)$ and $A_{\emp}(N)$, then all displayed matrix endpoints are
classified by the singleton empty history:
\begin{enumerate}
  \item $\rho_{\emp}(M,\emp)$.
  \item $\rho_{\emp}(0_M,\emp)$.
  \item $\rho_{\emp}(I_M,\emp)$.
  \item $\rho_{\emp}(M\oplus N,\emp)$.
  \item $\rho_{\emp}(M\odot N,\emp)$.
\end{enumerate}
\end{theorem}
\begin{proof}
Unfold the singleton carrier $A_{\emp}$ and the classifier
$\rho_{\emp}$. The hypotheses say that $M$ and $N$ are carried by the
singleton endpoint, so the first classification is obtained by pairing the
two carrier witnesses with the corresponding singleton endpoint witness in
$\hsame$.

Apply Theorem~\ref{thm:singleton-empty-history-matrix-laws} to obtain closure
of $0_M$, $I_M$, $M\oplus N$, and $M\odot N$ in $A_{\emp}$. These closure
witnesses supply the carrier components required by the classifier for the
four remaining endpoints.

For the $\hsame$ components, compose the singleton endpoint witnesses with
symmetry and transitivity of $\hsame$. Thus each displayed endpoint is
identified with $\emp$, and repacking the carrier witnesses with the
$\hsame$ witness gives the five asserted instances of $\rho_{\emp}$.
\end{proof}

\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonEmptyHistory\_endpoint\_exactness}

\begin{theorem}[Matrix singleton classifier continuation commutativity closure]
\label{thm:matrix-singleton-classifier-continuation-comm-closed}
Let $M,N:\mathsf{BHist}$ be carried by the singleton matrix carrier
$A_{\emp}$. If $\ContR(M,N,L)$ and $\ContR(N,M,R)$, then both continuation
results are again carried by $A_{\emp}$, and the singleton matrix classifier
relates the two results:
$$
A_{\emp}(L)\land A_{\emp}(R)\land \rho_{\emp}(L,R).
$$
\end{theorem}
\begin{proof}
The two carrier hypotheses identify $M$ and $N$ with $\emp$. After these
identifications, the two continuation premises identify both $L$ and $R$ with
the BEDC append endpoint $\mathsf{append}(\emp,\emp)$, which is $\emp$. The
singleton carrier witnesses for $L$ and $R$ are therefore reflexive empty
history witnesses, and the classifier is rebuilt from those two carrier
components and $\hsame(\emp,\emp)$.
\end{proof}
\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonClassifier\_continuation\_comm\_closed}

\section{The certificate}

\begin{definition}[Matrix source specification]
\label{def:matrix-source-specification}
The source is the finite index carrier from \(\NatUp\) and the scalar source \(\RingUp(R)\).
\end{definition}

\begin{definition}[Matrix pattern specification]
\label{def:matrix-pattern-specification}
The pattern is finite table formation assigning a scalar witness to each index pair.
\end{definition}

\begin{definition}[Matrix classifier specification]
\label{def:matrix-classifier-specification}
The classifier relates matrices pointwise by scalar classifier at every index pair.
\end{definition}

\begin{definition}[Matrix stability certificate]
\label{def:matrix-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item pointwise reflexivity
  \item pointwise transitivity
  \item finite index closure
  \item addition closure by scalar addition
  \item multiplication closure by finite scalar sums and products
  \item identity matrix closure when dimensions match
\end{enumerate}
\end{definition}

\begin{definition}[Matrix ledger policy]
\label{def:matrix-ledger-policy}
Record the dimension witnesses, index discipline, and scalar ledger for each entry.
\end{definition}

\begin{definition}[\(\MatrixUp\) certificate obligations]
\label{def:matrix-certificate-obligations}
The carrier, source, pattern, classifier, stability, and ledger data listed above are the obligations required for the interface name \(\MatrixUp\).
\end{definition}

\paragraph{Primitive boundary.}
\(\MatrixUp\) is not a BEDC primitive. It is admitted only when those obligations are supplied.

\begin{remark}[Zero-axiom encoding note]
\label{rem:matrix-zero-axiom-note}
Avoids funext by using pointwise classifier sameness; finite indices are inductive, not quotient indices.
\end{remark}

\begin{lemma}[Finite Kronecker selector sum]
\label{lem:matrix-finite-kronecker-selector-sum}
Let $J=\mathsf{Fin}(p)$ be a finite index carrier licensed by the $\NatUp$
certificate of \autoref{ch:concrete-instances-nat-namecert}. Let $R$ be a
scalar carrier from the $\CommRingUp$ certificate of
\autoref{ch:concrete-instances-commring-namecert}, with classifier $\sim_R$,
operations $+_R,\cdot_R$, and scalars $0_R,1_R$. Assume a finite-index sum
certificate $\Sigma_J$ satisfying classifier congruence and the selector row:
if $j_0:J$, $t:J\to R$ is carried pointwise, $t(j_0)\sim_R u$, and
$t(j)\sim_R 0_R$ for every $j\ne j_0$, then $\Sigma_J(t)\sim_R u$.
For every pointwise carried scalar family $x:J\to R$ and every $j_0:J$,
$$
\Sigma_J(j\mapsto \delta_{j_0,j}\cdot_R x(j))\sim_R x(j_0)
\qquad\text{and}\qquad
\Sigma_J(j\mapsto x(j)\cdot_R \delta_{j,j_0})\sim_R x(j_0),
$$
where $\delta_{a,b}$ is $1_R$ when the finite-index equality certificate
identifies $a$ with $b$, and $0_R$ otherwise.
\end{lemma}
\begin{proof}
For the first displayed sum, split the finite index $j$. At the selected index
$j=j_0$, the Kronecker entry is $1_R$, so the scalar left-one law gives
$\delta_{j_0,j_0}\cdot_R x(j_0)\sim_R x(j_0)$. At every other index, the
Kronecker entry is $0_R$, so scalar left-zero absorption gives
$\delta_{j_0,j}\cdot_R x(j)\sim_R 0_R$. These are exactly the selected and
off-selected hypotheses of the selector row for $\Sigma_J$, applied to the
summand family $j\mapsto \delta_{j_0,j}\cdot_R x(j)$ with $u=x(j_0)$.

The second displayed sum uses the same selector-row hypothesis for
$j\mapsto x(j)\cdot_R \delta_{j,j_0}$. The scalar right-one law gives the
selected comparison at $j=j_0$, and scalar right-zero absorption gives the
comparison with $0_R$ away from $j_0$.
\end{proof}

\begin{theorem}[Matrix identity multiplication unit]
\label{thm:matrix-identity-multiplication-unit}
Let $n,m:\NatUp$ and let $R$ be a scalar $\CommRingUp$ certificate. Consider
the $\MatrixUp$ carrier of \autoref{def:matrix-carrier} with the pointwise
classifier of \autoref{def:matrix-classifier-specification}. Assume the
stability data of \autoref{def:matrix-stability-certificate} supplies finite
product sums over $\mathsf{Fin}(n)$ and $\mathsf{Fin}(m)$, the scalar zero and
one multiplication laws over $R$, and carried Kronecker identity matrices
$I_n$ and $I_m$ with entries
$$
(I_n)_{i,a}\sim_R\delta_{i,a}
\qquad\text{and}\qquad
(I_m)_{b,j}\sim_R\delta_{b,j}.
$$
If $M:\mathsf{Fin}(n)\times\mathsf{Fin}(m)\to R$ is carried, then the products
$I_n\cdot M$ and $M\cdot I_m$ are carried and
$$
I_n\cdot M\sim_{\mathrm{Mat}} M
\qquad\text{and}\qquad
M\cdot I_m\sim_{\mathrm{Mat}} M.
$$
\end{theorem}
\begin{proof}
The carriedness of $I_n$ and $I_m$ is part of the Kronecker identity closure
field, and $M$ is carried by hypothesis. Multiplication closure in
\autoref{def:matrix-stability-certificate} therefore carries both products.
It remains to prove the pointwise scalar comparisons required by
\autoref{def:matrix-classifier-specification}.

Fix $i:\mathsf{Fin}(n)$ and $b:\mathsf{Fin}(m)$. The left product entry is the
finite sum
$$
(I_n\cdot M)_{i,b}
\sim_R
\Sigma_{\mathsf{Fin}(n)}(a\mapsto (I_n)_{i,a}\cdot_R M_{a,b}).
$$
For each inner index $a$, identity-entry closure gives
$(I_n)_{i,a}\sim_R\delta_{i,a}$, and scalar classifier congruence for
$\cdot_R$ gives
$$
(I_n)_{i,a}\cdot_R M_{a,b}
\sim_R
\delta_{i,a}\cdot_R M_{a,b}.
$$
The finite-sum classifier congruence for $\Sigma_{\mathsf{Fin}(n)}$ yields
$$
\Sigma_{\mathsf{Fin}(n)}(a\mapsto (I_n)_{i,a}\cdot_R M_{a,b})
\sim_R
\Sigma_{\mathsf{Fin}(n)}(a\mapsto \delta_{i,a}\cdot_R M_{a,b}).
$$
Applying \autoref{lem:matrix-finite-kronecker-selector-sum} to the carried
family $a\mapsto M_{a,b}$ selects the unique inner index $a=i$, hence
$(I_n\cdot M)_{i,b}\sim_R M_{i,b}$.

For the right product, the entry at the same outer indices is
$$
(M\cdot I_m)_{i,b}
\sim_R
\Sigma_{\mathsf{Fin}(m)}(a\mapsto M_{i,a}\cdot_R (I_m)_{a,b}).
$$
For each inner index $a$, identity-entry closure gives
$(I_m)_{a,b}\sim_R\delta_{a,b}$, and scalar classifier congruence for
$\cdot_R$ gives
$$
M_{i,a}\cdot_R (I_m)_{a,b}
\sim_R
M_{i,a}\cdot_R \delta_{a,b}.
$$
The finite-sum classifier congruence for $\Sigma_{\mathsf{Fin}(m)}$ yields
$$
\Sigma_{\mathsf{Fin}(m)}(a\mapsto M_{i,a}\cdot_R (I_m)_{a,b})
\sim_R
\Sigma_{\mathsf{Fin}(m)}(a\mapsto M_{i,a}\cdot_R \delta_{a,b}).
$$
The right-hand selector part of
\autoref{lem:matrix-finite-kronecker-selector-sum} applied to the carried
family $a\mapsto M_{i,a}$ selects $a=b$, giving
$(M\cdot I_m)_{i,b}\sim_R M_{i,b}$.

Since these scalar comparisons hold for every row and column index, the
pointwise matrix classifier gives both displayed matrix identity laws. The
ledger in \autoref{def:matrix-ledger-policy} records the same row and column
witnesses throughout the two products, so no dimension data is changed.
\end{proof}

\begin{definition}[Two-index finite fold swap certificate]
\label{def:two-index-finite-fold-swap-certificate}
Let $R$ be a scalar $\RingUp$ carrier with classifier $\sim_R$, zero endpoint $0_R$, addition $+_R$, and finite additive fold from \autoref{def:finite-additive-scalar-fold}. Let $J$ and $K$ be finite index carriers, and write $\Sigma_J$ and $\Sigma_K$ for that fold applied to the finite spines enumerating $J$ and $K$. A two-index finite fold swap certificate for $(J,K)$ is the following stability field: for every pointwise carried scalar family $s:J\times K\to R$ and every pointwise classifier-compatible transpose $s^\tau:K\times J\to R$ satisfying
$$
s^\tau(k,j)\sim_R s(j,k)
$$
for all $j:J$ and $k:K$, the two iterated folds are scalar-classified:
$$
\Sigma_J(j\mapsto \Sigma_K(k\mapsto s(j,k)))
\sim_R
\Sigma_K(k\mapsto \Sigma_J(j\mapsto s^\tau(k,j))).
$$
\end{definition}

\begin{lemma}[Scalar product distributes across finite additive folds]
\label{lem:scalar-product-distributes-across-finite-additive-folds}
Let $R$ be a scalar $\RingUp$ carrier with classifier $\sim_R$, zero endpoint $0_R$, addition $+_R$, multiplication $\cdot_R$, and finite additive fold from \autoref{def:finite-additive-scalar-fold}. Assume the ring stability fields of \autoref{def:ring-stability-certificate}: additive congruence, multiplication congruence, left and right distributivity, and zero multiplication laws. For every carried scalar $c$ and every finite carried scalar spine $xs$, let $xs\cdot_R c$ denote the spine obtained by multiplying each entry of $xs$ on the right by $c$, and let $c\cdot_R xs$ denote the spine obtained by multiplying each entry on the left by $c$. Then
$$
\mathsf{fold}_{+,R}(xs)\cdot_R c
\sim_R
\mathsf{fold}_{+,R}(xs\cdot_R c)
$$
and
$$
c\cdot_R\mathsf{fold}_{+,R}(xs)
\sim_R
\mathsf{fold}_{+,R}(c\cdot_R xs).
$$
\end{lemma}
\begin{proof}
Induct on the finite spine $xs$. In the nil case, \autoref{def:finite-additive-scalar-fold} identifies the fold with $0_R$. The right and left zero multiplication laws from \autoref{def:ring-stability-certificate} identify $0_R\cdot_R c$ and $c\cdot_R0_R$ with $0_R$, which is the fold of the corresponding empty product spine.

In the cons case, write $xs=\mathsf{cons}(a,xs')$. For right multiplication, the fold equation gives $(a+_R\mathsf{fold}_{+,R}(xs'))\cdot_R c$. Right distributivity identifies this scalar with
$$
(a\cdot_R c)+_R(\mathsf{fold}_{+,R}(xs')\cdot_R c).
$$
The induction hypothesis identifies the second summand with $\mathsf{fold}_{+,R}(xs'\cdot_R c)$. Additive congruence then gives
$$
(a\cdot_R c)+_R\mathsf{fold}_{+,R}(xs'\cdot_R c),
$$
which is the cons fold of $xs\cdot_R c$.

The left multiplication proof is the same induction with left distributivity applied to $c\cdot_R(a+_R\mathsf{fold}_{+,R}(xs'))$. It yields
$$
(c\cdot_R a)+_R(c\cdot_R\mathsf{fold}_{+,R}(xs')),
$$
then uses the induction hypothesis and additive congruence to identify the result with the cons fold of $c\cdot_R xs$.
\end{proof}

\begin{theorem}[Matrix multiplication associativity from finite folds]
\label{thm:matrix-multiplication-associativity-finite-folds}
Let $R$ be a scalar $\RingUp$ carrier with classifier $\sim_R$, addition $+_R$, multiplication $\cdot_R$, zero endpoint $0_R$, and finite additive fold from \autoref{def:finite-additive-scalar-fold}. Let $n,m,p,q:\NatUp$, and consider the $\MatrixUp$ carriers of \autoref{def:matrix-carrier} with pointwise classifier from \autoref{def:matrix-classifier-specification}. Let
$$
A:\mathsf{Fin}(n)\times\mathsf{Fin}(m)\to R,
\qquad
B:\mathsf{Fin}(m)\times\mathsf{Fin}(p)\to R,
\qquad
C:\mathsf{Fin}(p)\times\mathsf{Fin}(q)\to R
$$
be carried matrices. Assume matrix multiplication is the finite-fold operation
$$
(X\cdot Y)_{u,w}:=\Sigma_V(v\mapsto X_{u,v}\cdot_RY_{v,w})
$$
for the matching inner finite index carrier $V$, using the multiplication closure field of \autoref{def:matrix-stability-certificate}. Assume the scalar ring stability fields of \autoref{def:ring-stability-certificate}, the fold congruence theorem \autoref{thm:finite-additive-fold-respects-scalar-classifier}, and the two-index finite fold swap certificate of \autoref{def:two-index-finite-fold-swap-certificate} for $\mathsf{Fin}(p)$ and $\mathsf{Fin}(m)$. Then the two matched products are carried and
$$
(A\cdot B)\cdot C\sim_{\mathrm{Mat}}A\cdot(B\cdot C).
$$
\end{theorem}
\begin{proof}
The multiplication closure item of \autoref{def:matrix-stability-certificate} carries $A\cdot B$ and $B\cdot C$, and then carries the two outer products $(A\cdot B)\cdot C$ and $A\cdot(B\cdot C)$. By \autoref{def:matrix-classifier-specification}, it remains to prove the scalar classifier comparison at each entry.

Fix $i:\mathsf{Fin}(n)$ and $l:\mathsf{Fin}(q)$, and set
$$
t(b,a):=A_{i,a}\cdot_R(B_{a,b}\cdot_R C_{b,l})
$$
for $a:\mathsf{Fin}(m)$ and $b:\mathsf{Fin}(p)$. Expanding the left-associated matrix product gives
$$
((A\cdot B)\cdot C)_{i,l}
\sim_R
\Sigma_{b:\mathsf{Fin}(p)}((A\cdot B)_{i,b}\cdot_R C_{b,l}).
$$
For each fixed $b$, expanding $(A\cdot B)_{i,b}$ and applying the right-multiplication half of \autoref{lem:scalar-product-distributes-across-finite-additive-folds} gives
$$
(A\cdot B)_{i,b}\cdot_R C_{b,l}
\sim_R
\Sigma_{a:\mathsf{Fin}(m)}((A_{i,a}\cdot_R B_{a,b})\cdot_R C_{b,l}).
$$
Scalar associativity from the multiplicative monoid part of \autoref{def:ring-stability-certificate} gives a pointwise comparison
$$
(A_{i,a}\cdot_R B_{a,b})\cdot_R C_{b,l}
\sim_R
A_{i,a}\cdot_R(B_{a,b}\cdot_R C_{b,l}).
$$
The finite-fold congruence theorem \autoref{thm:finite-additive-fold-respects-scalar-classifier} lifts these pointwise comparisons through the inner and outer folds. Hence
$$
((A\cdot B)\cdot C)_{i,l}
\sim_R
\Sigma_{b:\mathsf{Fin}(p)}\Sigma_{a:\mathsf{Fin}(m)}t(b,a).
$$

Expanding the right-associated matrix product gives
$$
(A\cdot(B\cdot C))_{i,l}
\sim_R
\Sigma_{a:\mathsf{Fin}(m)}(A_{i,a}\cdot_R(B\cdot C)_{a,l}).
$$
For each fixed $a$, expanding $(B\cdot C)_{a,l}$ and applying the left-multiplication half of \autoref{lem:scalar-product-distributes-across-finite-additive-folds} gives
$$
A_{i,a}\cdot_R(B\cdot C)_{a,l}
\sim_R
\Sigma_{b:\mathsf{Fin}(p)}A_{i,a}\cdot_R(B_{a,b}\cdot_R C_{b,l}).
$$
Again \autoref{thm:finite-additive-fold-respects-scalar-classifier} lifts the pointwise comparisons through the finite folds, so
$$
(A\cdot(B\cdot C))_{i,l}
\sim_R
\Sigma_{a:\mathsf{Fin}(m)}\Sigma_{b:\mathsf{Fin}(p)}t(b,a).
$$

Apply the fold swap certificate of \autoref{def:two-index-finite-fold-swap-certificate} to the family $t$ and its transpose $t^\tau(a,b):=t(b,a)$. This gives
$$
\Sigma_{b:\mathsf{Fin}(p)}\Sigma_{a:\mathsf{Fin}(m)}t(b,a)
\sim_R
\Sigma_{a:\mathsf{Fin}(m)}\Sigma_{b:\mathsf{Fin}(p)}t(b,a).
$$
Classifier transitivity composes the left expansion, this fold swap comparison, and the symmetry of the right expansion. Thus every entry of $(A\cdot B)\cdot C$ is scalar-classified with the corresponding entry of $A\cdot(B\cdot C)$. The pointwise matrix classifier of \autoref{def:matrix-classifier-specification} gives the displayed matrix classification, and the unchanged dimension witnesses are recorded by the ledger policy of \autoref{def:matrix-ledger-policy}.
\end{proof}

\begin{theorem}[Matrix multiplication classifier congruence]
\label{thm:matrix-multiplication-classifier-congruence}
Let $R$ be a scalar $\RingUp$ carrier with classifier $\sim_R$, multiplication $\cdot_R$, addition $+_R$, zero endpoint $0_R$, and finite additive fold from \autoref{def:finite-additive-scalar-fold}. Let $n,m,p:\NatUp$, and consider the $\MatrixUp$ carriers of \autoref{def:matrix-carrier} with pointwise classifier from \autoref{def:matrix-classifier-specification}. Let
$$
A,A':\mathsf{Fin}(n)\times\mathsf{Fin}(m)\to R,
\qquad
B,B':\mathsf{Fin}(m)\times\mathsf{Fin}(p)\to R
$$
be carried matrices over the same scalar source and the same matching inner dimension $m$. Assume matrix multiplication is the finite-fold operation
$$
(X\cdot Y)_{i,k}:=\Sigma_{\mathsf{Fin}(m)}(j\mapsto X_{i,j}\cdot_RY_{j,k})
$$
for the shared inner finite index carrier, using the multiplication closure field of \autoref{def:matrix-stability-certificate}. Assume the scalar multiplication congruence field of \autoref{def:ring-stability-certificate} and the fold congruence theorem \autoref{thm:finite-additive-fold-respects-scalar-classifier}. If
$$
A\sim_{\mathrm{Mat}}A'
\qquad\text{and}\qquad
B\sim_{\mathrm{Mat}}B',
$$
then $A\cdot B$ and $A'\cdot B'$ are carried $\MatrixUp$ matrices, and
$$
A\cdot B\sim_{\mathrm{Mat}}A'\cdot B'.
$$
\end{theorem}
\begin{proof}
The multiplication closure item in \autoref{def:matrix-stability-certificate} applies to the carried compatible pairs $(A,B)$ and $(A',B')$, so both products $A\cdot B$ and $A'\cdot B'$ are carried by the $\mathsf{Fin}(n)\times\mathsf{Fin}(p)$ matrix carrier.

By \autoref{def:matrix-classifier-specification}, it remains to prove the scalar classifier comparison entrywise. Fix $i:\mathsf{Fin}(n)$ and $k:\mathsf{Fin}(p)$. The matrix classifier hypotheses give, for every $j:\mathsf{Fin}(m)$,
$$
A_{i,j}\sim_R A'_{i,j}
\qquad\text{and}\qquad
B_{j,k}\sim_R B'_{j,k}.
$$
The multiplication congruence field in \autoref{def:ring-stability-certificate} therefore gives the summand comparison
$$
A_{i,j}\cdot_R B_{j,k}
\sim_R
A'_{i,j}\cdot_R B'_{j,k}
$$
for every inner index $j$.

The shared inner dimension supplies the same finite enumeration of $\mathsf{Fin}(m)$ on both sides. Hence these summand comparisons form a same-length finite-spine classifier witness between
$$
j\mapsto A_{i,j}\cdot_R B_{j,k}
\qquad\text{and}\qquad
j\mapsto A'_{i,j}\cdot_R B'_{j,k}.
$$
Applying \autoref{thm:finite-additive-fold-respects-scalar-classifier} to that finite-spine witness yields
$$
\Sigma_{\mathsf{Fin}(m)}(j\mapsto A_{i,j}\cdot_R B_{j,k})
\sim_R
\Sigma_{\mathsf{Fin}(m)}(j\mapsto A'_{i,j}\cdot_R B'_{j,k}).
$$
By the displayed definition of matrix multiplication, this is exactly
$$
(A\cdot B)_{i,k}\sim_R(A'\cdot B')_{i,k}.
$$
Since $i$ and $k$ were arbitrary, \autoref{def:matrix-classifier-specification} packages the entrywise scalar comparisons as $A\cdot B\sim_{\mathrm{Mat}}A'\cdot B'$. The dimension and scalar-source ledgers are unchanged and are recorded by \autoref{def:matrix-ledger-policy}.
\end{proof}
