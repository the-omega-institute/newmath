\chapter{A Concrete Naming Certificate for \(\MatrixUp\)}
\label{ch:concrete-instances-matrix-namecert}

The matrix interface \(\MatrixUp\) has certificate obligations over a scalar ring from \autoref{ch:concrete-instances-ring-namecert} and two natural-number dimensions from \autoref{ch:concrete-instances-nat-namecert}. The carrier is functions \(\mathsf{Fin}(n) \times \mathsf{Fin}(m) \to R\); the classifier is pointwise scalar sameness on all index pairs. Matrix addition and multiplication are derived operations governed by the ring's certificate fields. It is not a primitive matrix type; it is admitted only as a derived function carrier.

\section{Carrier definition}

\begin{definition}[Matrix carrier]
\label{def:matrix-carrier}
Use functions \(\mathsf{Fin}(n)\times\mathsf{Fin}(m)\to R\) with \(n,m:\NatUp\) and scalar ring \(R\).
\end{definition}

\section{Concrete singleton empty-history matrix instance}
\label{sec:matrix-concrete-singleton-instance}

\begin{definition}[Singleton empty-history matrix instance]
\label{def:singleton-empty-history-matrix-instance}
Let the matrix-code carrier be
$$
M_{\emp}(m):=\hsame(m,\emp).
$$
Define the matrix classifier by
$$
\mu_{\emp}(m,n):=
M_{\emp}(m)\land M_{\emp}(n)\land\hsame(m,n).
$$
Take both matrix dimensions to be the singleton unary endpoint supplied by
Definition~\ref{def:list-singleton-history-source}, and take the scalar source,
scalar classifier, and scalar operations from
Definition~\ref{def:singleton-empty-history-ring-instance}. The entry lookup is
constant:
$$
\mathsf{entry}_{\emp}(m,i,j):=\emp.
$$
The matrix operations are also constant at the empty-history endpoint:
$$
\mathsf{zero}_{\emp}:=\emp,\qquad
\mathsf{addM}_{\emp}(m,n):=\emp,\qquad
\mathsf{mulM}_{\emp}(m,n):=\emp,\qquad
\mathsf{idM}_{\emp}:=\emp.
$$
This is the one-element concrete $\mathsf{BHist}$ matrix instance over the
singleton empty-history scalar ring.
\end{definition}

\begin{theorem}[Singleton empty-history matrix laws]
\label{thm:singleton-empty-history-matrix-laws}
The tuple
$$
(M_{\emp},\mu_{\emp},\mathsf{entry}_{\emp},
\mathsf{zero}_{\emp},\mathsf{addM}_{\emp},
\mathsf{mulM}_{\emp},\mathsf{idM}_{\emp})
$$
satisfies the concrete $\MatrixUp$ law package: carrier inhabitance,
classifier reflexivity, classifier symmetry, classifier transitivity, entry
classification, addition closure and congruence, multiplication closure and
congruence, and identity-matrix closure with its left and right identity laws.
\end{theorem}
\begin{proof}
Carrier inhabitance is the reflexive endpoint witness $\hsame(\emp,\emp)$,
which gives $M_{\emp}(\emp)$. The same witness supplies reflexivity of
$\mu_{\emp}$, while symmetry and transitivity of $\mu_{\emp}$ are assembled
from the corresponding symmetry and transitivity rules for $\hsame$ together
with the displayed carrier fields.

For entry classification, every lookup
$\mathsf{entry}_{\emp}(m,i,j)$ is definitionally $\emp$. The singleton source
of Definition~\ref{def:list-singleton-history-source} gives the scalar carrier
witness for $\emp$, and Theorem~\ref{thm:singleton-empty-history-ring-laws}
supplies the scalar classifier laws at that endpoint.

Addition closure and multiplication closure are immediate because
$\mathsf{addM}_{\emp}(m,n)$ and $\mathsf{mulM}_{\emp}(m,n)$ both reduce to
$\emp$. Congruence for both operations compares two empty-history endpoints, so
the two result carriers are $M_{\emp}(\emp)$ and the relation component is
$\hsame(\emp,\emp)$.

The identity matrix code $\mathsf{idM}_{\emp}$ is also definitionally $\emp$,
so identity closure is the same singleton carrier witness. The left and right
identity laws compare $\mathsf{mulM}_{\emp}(\mathsf{idM}_{\emp},m)$ and
$\mathsf{mulM}_{\emp}(m,\mathsf{idM}_{\emp})$ with the carried matrix code
$m$; the operation side is $\emp$, and the input witness $M_{\emp}(m)$ gives
the required endpoint link under $\mu_{\emp}$.
\end{proof}

This concrete $\mathsf{BHist}$ instance lets future $\MatrixUp$ proofs target
the displayed operations rather than abstract operator fields.

\section{The certificate}

\begin{definition}[Matrix source specification]
\label{def:matrix-source-specification}
The source is the finite index carrier from \(\NatUp\) and the scalar source \(\RingUp(R)\).
\end{definition}

\begin{definition}[Matrix pattern specification]
\label{def:matrix-pattern-specification}
The pattern is finite table formation assigning a scalar witness to each index pair.
\end{definition}

\begin{definition}[Matrix classifier specification]
\label{def:matrix-classifier-specification}
The classifier relates matrices pointwise by scalar classifier at every index pair.
\end{definition}

\begin{definition}[Matrix stability certificate]
\label{def:matrix-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item pointwise reflexivity
  \item pointwise transitivity
  \item finite index closure
  \item addition closure by scalar addition
  \item multiplication closure by finite scalar sums and products
  \item identity matrix closure when dimensions match
\end{enumerate}
\end{definition}

\begin{definition}[Matrix ledger policy]
\label{def:matrix-ledger-policy}
Record the dimension witnesses, index discipline, and scalar ledger for each entry.
\end{definition}

\begin{definition}[\(\MatrixUp\) certificate obligations]
\label{def:matrix-certificate-obligations}
The carrier, source, pattern, classifier, stability, and ledger data listed above are the obligations required for the interface name \(\MatrixUp\).
\end{definition}

\paragraph{Primitive boundary.}
\(\MatrixUp\) is not a BEDC primitive. It is admitted only when those obligations are supplied.

\begin{remark}[Zero-axiom encoding note]
\label{rem:matrix-zero-axiom-note}
Avoids funext by using pointwise classifier sameness; finite indices are inductive, not quotient indices.
\end{remark}
