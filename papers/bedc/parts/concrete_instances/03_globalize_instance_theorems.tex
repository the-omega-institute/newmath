\chapter{The First Concrete Globalize Instance}
\label{ch:concrete-instances-globalize}

This chapter packages the two preceding chapters into a single concrete globalization instance.

\begin{definition}[Signature globalization instance]
Given a source domain specification \(D\), a generated probe bundle \(\Pi\), and \(\AskPol(\Pi,D)\), the signature globalization instance is the triple
\[
  \mathsf{Glob}^{\mathrm{sig}}(\Pi,D)
  :=
  \bigl(\Pkg^{\mathrm{sig}},\psame,\InGap^{\mathrm{sig}}\bigr).
\]
It is read as a visible package layer plus a provenance ledger layer.
\end{definition}

\begin{theorem}[Concrete globalization soundness]
Assume \(\AskPol(\Pi,D)\). If \(h,k\in D\) and \(\sameSig(\Pi,h,k)\), then any packages generated by the signatures of \(h\) and \(k\) are package-same.
\end{theorem}
\begin{proof}
By \(\sameSig(\Pi,h,k)\), there exist generated answer histories \(s,t\) and evidence ledgers such that
\[
  \Sig(\Pi,h,s,\Delta),\qquad \Sig(\Pi,k,t,\Theta),\qquad \hsame(s,t).
\]
If \(\Pkg^{\mathrm{sig}}(\Pi,s,p)\) and \(\Pkg^{\mathrm{sig}}(\Pi,t,q)\), package extensionality gives \(\psame(p,q)\).
\end{proof}

\begin{theorem}[Concrete globalization completeness]
Assume \(\AskPol(\Pi,D)\). If histories \(h,k\in D\) lie in package-same visible tokens of the signature globalization instance, then their generated signatures are signature-same.
\end{theorem}
\begin{proof}
Unfold the concrete gap membership. The memberships provide signatures \(s,t\) such that \(h\) generates \(s\), \(k\) generates \(t\), and the corresponding tokens are package-same. By the grounding part of the signature package policy, package-sameness of the visible tokens is grounded in \(\hsame(s,t)\). Hence \(\sameSig(\Pi,h,k)\).
\end{proof}

\begin{corollary}[Globalize equals signature classification in the concrete instance]
For \(\mathsf{Glob}^{\mathrm{sig}}(\Pi,D)\), visible package sameness coincides with signature sameness, up to the package and gap policies of the instance.
\end{corollary}

\begin{remark}[Why this matters]
The surrounding theory states that Globalize is classification by signatures. This chapter gives the first concrete finite-kernel inhabitant of that claim. It does not yet build real-like, circle-like, or fold-like interfaces; it only makes the finite signature globalization stand.
\end{remark}
