\chapter{The First Concrete Globalize Instance}
\label{ch:concrete-instances-globalize}

This chapter packages the two preceding chapters into a single concrete globalization instance.

\begin{definition}[Signature globalization instance]
Given a source domain specification \(D\), a generated probe bundle \(\Pi\), and \(\AskPol(\Pi,D)\), the signature globalization instance is the triple
\[
  \mathsf{Glob}^{\mathrm{sig}}(\Pi,D)
  :=
  \bigl(\Pkg^{\mathrm{sig}},\psame,\InGap^{\mathrm{sig}}\bigr).
\]
It is read as a visible package layer plus a provenance ledger layer.
\end{definition}
\leandef{BEDC.FKernel.Gap.signature\_globalization\_instance}

\begin{theorem}[Concrete globalization soundness]
\label{thm:concrete-globalize-soundness}
Assume \(\AskPol(\Pi,D)\) and \(\PkgTokPol(\Pi)\). If
$$
  \InGapSig(\Pi,D,p,h),
  \qquad
  \InGapSig(\Pi,D,q,k),
  \qquad
  \sameSig(\Pi,h,k),
$$
then \(\psame(p,q)\).
\end{theorem}
\leanchecked{BEDC.FKernel.Gap.concrete\_globalize\_soundness\_from\_sameSig}
\leanvariant{BEDC.FKernel.Gap.internalized\_globalize\_soundness\_for\_witnesses}
\begin{proof}
The two gap memberships provide introduced visible tokens for generated signatures of \(h\) and \(k\). By \(\sameSig(\Pi,h,k)\), there exist generated answer histories \(s,t\) and evidence ledgers such that
$$
  \Sig(\Pi,h,s,\Delta),\qquad \Sig(\Pi,k,t,\Theta),\qquad \hsame(s,t).
$$
The determinacy field of \(\AskPol(\Pi,D)\) aligns these signatures with the signatures carried by the two gap witnesses. Package soundness from \(\PkgTokPol(\Pi)\) gives \(\psame(p,q)\).
\end{proof}

\begin{theorem}[Concrete globalization completeness]
\label{thm:concrete-globalization-completeness}
Assume \(\PkgTokPol(\Pi)\). If
$$
  \InGapSig(\Pi,D,p,h),
  \qquad
  \InGapSig(\Pi,D,q,k),
  \qquad
  \psame(p,q),
$$
then \(\sameSig(\Pi,h,k)\).
\end{theorem}
\leanchecked{BEDC.FKernel.Gap.concrete\_globalize\_completeness\_sameSig}
\leanvariant{BEDC.FKernel.Gap.concrete\_globalization\_completeness}
\begin{proof}
Unfold the concrete gap memberships. They provide signatures \(s,t\) such that \(h\) generates \(s\), \(k\) generates \(t\), and \(p,q\) are introduced from those signatures. By the reflection field of \(\PkgTokPol(\Pi)\), package-sameness of the visible tokens is grounded in \(\hsame(s,t)\). Hence \(\sameSig(\Pi,h,k)\).
\end{proof}

\begin{corollary}[Globalize equals signature classification in the concrete instance]
\label{cor:globalize-equals-signature-classification-concrete}
Assume \(\AskPol(\Pi,D)\) and \(\PkgTokPol(\Pi)\). For introduced tokens with
$$
  \InGapSig(\Pi,D,p,h)
  \qquad\text{and}\qquad
  \InGapSig(\Pi,D,q,k),
$$
the concrete instance satisfies
$$
  \psame(p,q)\Longleftrightarrow \sameSig(\Pi,h,k).
$$
\end{corollary}
\leanchecked{BEDC.FKernel.Gap.globalize\_equals\_signature\_classification\_concrete}

\begin{remark}[Why this matters]
The surrounding theory states that Globalize is classification by signatures. This chapter gives the first concrete finite-kernel inhabitant of that claim. It does not yet build real-like, circle-like, or fold-like interfaces; it only makes the finite signature globalization stand.
\end{remark}
