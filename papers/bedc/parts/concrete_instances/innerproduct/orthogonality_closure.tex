\begin{theorem}[InnerProduct orthogonality symmetry row]
\label{thm:innerproduct-orthogonality-symmetry-row}
Let $I$ be an inner-product $\mathsf{BHist}$ carrier as in
\autoref{def:innerproduct-bhist-carrier}. For carried vector endpoints
$x,y:C_V$, the orthogonality row defined by the retained scalar-zero
classifier is symmetric:
$$
x\perp_I y
\quad\Longleftrightarrow\quad
y\perp_I x.
$$
\end{theorem}
\leanchecked{BEDC.Derived.InnerProductUp.InnerProductSingletonOrthogonal\_symm\_package}
\begin{proof}
Assume first that $x\perp_I y$. By
\autoref{def:innerproduct-orthogonality-row}, this is the scalar-zero
classification
$$
\langle x,y\rangle_{\InnerProductUp}\sim_{\mathbb{K}}0_{\mathbb{K}}.
$$
Apply the conjugate-symmetry row from
\autoref{thm:innerproduct-symmetry-positivity-row} with the two endpoints
ordered as $(y,x)$. It gives
$$
\langle y,x\rangle_{\InnerProductUp}
\sim_{\mathbb{K}}
\overline{\langle x,y\rangle_{\InnerProductUp}}.
$$
The scalar endpoint source retained in
\autoref{def:innerproduct-bhist-source} supplies classifier transport for the
displayed conjugation row. Applying that row to the scalar-zero classification
of $\langle x,y\rangle_{\InnerProductUp}$ gives
$$
\overline{\langle x,y\rangle_{\InnerProductUp}}
\sim_{\mathbb{K}}
\overline{0_{\mathbb{K}}}.
$$
The same scalar source fixes the zero endpoint under conjugation, the
scalar-side row used in \autoref{thm:innerproduct-zero-right-orthogonal-row},
so
$$
\overline{0_{\mathbb{K}}}\sim_{\mathbb{K}}0_{\mathbb{K}}.
$$
Composing these three scalar classifier witnesses by
\autoref{thm:namecert-equivalence-transitivity} gives
$$
\langle y,x\rangle_{\InnerProductUp}\sim_{\mathbb{K}}0_{\mathbb{K}}.
$$
By \autoref{def:innerproduct-orthogonality-row}, this is $y\perp_I x$.

For the converse implication, assume $y\perp_I x$. The same definition gives
$$
\langle y,x\rangle_{\InnerProductUp}\sim_{\mathbb{K}}0_{\mathbb{K}}.
$$
Apply the conjugate-symmetry row with endpoints ordered as $(x,y)$, transport
this scalar-zero witness through the displayed conjugation row, use the
zero-fix row for conjugation, and compose the scalar classifier witnesses by
\autoref{thm:namecert-equivalence-transitivity}. The result is
$$
\langle x,y\rangle_{\InnerProductUp}\sim_{\mathbb{K}}0_{\mathbb{K}},
$$
which is exactly $x\perp_I y$ by
\autoref{def:innerproduct-orthogonality-row}.
\end{proof}

\begin{theorem}[InnerProduct orthogonal additivity row]
\label{thm:innerproduct-orthogonal-additivity-row}
Let $I$ be an inner-product $\mathsf{BHist}$ carrier as in
\autoref{def:innerproduct-bhist-carrier}. For carried vector endpoints
$x,y,z:C_V$, if $x\perp_I y$ and $x\perp_I z$, then the displayed vector sum
$y+_V z$ is carried and
$$
x\perp_I (y+_V z).
$$
\end{theorem}
\begin{proof}
The embedded $\VecSpaceUp$ row in
\autoref{def:innerproduct-bhist-carrier} supplies closure of the displayed
vector addition, so $y+_V z$ is carried. Apply
\autoref{thm:innerproduct-orthogonality-symmetry-row} to the two hypotheses
$x\perp_I y$ and $x\perp_I z$. This gives
$$
y\perp_I x
\qquad\text{and}\qquad
z\perp_I x.
$$
By \autoref{def:innerproduct-orthogonality-row}, these two facts are exactly the
scalar-zero classifications
$$
\langle y,x\rangle_{\InnerProductUp}\sim_{\mathbb{K}}0_{\mathbb{K}}
\qquad\text{and}\qquad
\langle z,x\rangle_{\InnerProductUp}\sim_{\mathbb{K}}0_{\mathbb{K}}.
$$

Use the additive part of \autoref{thm:innerproduct-vecspace-linearity-row} in
the displayed first argument, with second endpoint $x$. It compares the
inner-product endpoint of the vector sum with the scalar sum:
$$
\langle y+_V z,x\rangle_{\InnerProductUp}
\sim_{\mathbb{K}}
\langle y,x\rangle_{\InnerProductUp}
+_{\mathsf{scal}}
\langle z,x\rangle_{\InnerProductUp}.
$$
The retained scalar source of \autoref{def:innerproduct-bhist-source} is the
$\FieldUp$ scalar coordinate. By \autoref{def:field-source-specification},
\autoref{prop:commring-forgets-ring-certificate}, and
\autoref{def:ring-stability-certificate}, its unchanged scalar classifier has
addition congruence, additive zero laws, and classifier transitivity. Therefore
the two scalar-zero witnesses above transport through scalar addition to give
$$
\langle y,x\rangle_{\InnerProductUp}
+_{\mathsf{scal}}
\langle z,x\rangle_{\InnerProductUp}
\sim_{\mathbb{K}}
0_{\mathbb{K}}+_{\mathsf{scal}}0_{\mathbb{K}}
\sim_{\mathbb{K}}
0_{\mathbb{K}}.
$$
This is the scalar addition component of
\autoref{thm:ring-zero-classifier-two-sided-ideal-package}, read in the retained
scalar classifier.

Composing the inner-product linearity comparison with this scalar-zero
classification by \autoref{thm:namecert-equivalence-transitivity} gives
$$
\langle y+_V z,x\rangle_{\InnerProductUp}
\sim_{\mathbb{K}}
0_{\mathbb{K}}.
$$
By \autoref{def:innerproduct-orthogonality-row}, this is
$(y+_V z)\perp_I x$. A final application of
\autoref{thm:innerproduct-orthogonality-symmetry-row}, now to the carried pair
$(y+_V z,x)$, gives $x\perp_I (y+_V z)$.
\end{proof}

\begin{theorem}[InnerProduct orthogonality right scalar-action closure]
\label{thm:innerproduct-orthogonality-right-scalar-action-closure}
Let $I$ be an inner-product $\mathsf{BHist}$ carrier as in
\autoref{def:innerproduct-bhist-carrier}. Let $x,y:C_V$ be carried vector
endpoints, and let $r:C_{\mathsf{scal}}$ be a carried scalar endpoint from the
retained scalar source. If $x\perp_I y$, then $r\cdot_V y$ is carried and
$$
x\perp_I (r\cdot_V y).
$$
\end{theorem}
\begin{proof}
By \autoref{thm:innerproduct-orthogonality-symmetry-row}, the hypothesis
$x\perp_I y$ gives $y\perp_I x$. Unpacking this conclusion with
\autoref{def:innerproduct-orthogonality-row} gives the scalar-zero classifier
witness
$$
\langle y,x\rangle_{\InnerProductUp}\sim_{\mathbb{K}}0_{\mathbb{K}}.
$$

The embedded $\VecSpaceUp$ row carries the scalar-action endpoint
$r\cdot_V y$: scalar-action closure is one of the module stability rows retained
by \autoref{def:vecspace-stability-certificate} and
\autoref{def:module-stability-certificate}. Hence the inner-product endpoint
$\langle r\cdot_V y,x\rangle_{\InnerProductUp}$ is defined in the same carrier.

Apply the scalar-action compatibility component of
\autoref{thm:innerproduct-vecspace-linearity-row} to the carried scalar $r$ and
vector endpoints $y$ and $x$. It gives a retained scalar-classifier comparison
between
$$
\langle r\cdot_V y,x\rangle_{\InnerProductUp}
\quad\text{and}\quad
r\cdot_{\mathbb{K}}\langle y,x\rangle_{\InnerProductUp}.
$$
Inside the retained scalar source, the ring rows listed in
\autoref{def:ring-stability-certificate} are exactly the hypotheses needed for
\autoref{thm:ring-zero-classifier-factor-absorption}. Applying that absorption
row to the zero-classified factor $\langle y,x\rangle_{\InnerProductUp}$ and the
multiplier $r$ gives
$$
r\cdot_{\mathbb{K}}\langle y,x\rangle_{\InnerProductUp}
\sim_{\mathbb{K}}0_{\mathbb{K}}.
$$

Compose the scalar-action compatibility comparison with this zero-product
classification by \autoref{thm:namecert-equivalence-transitivity}. The result is
$$
\langle r\cdot_V y,x\rangle_{\InnerProductUp}
\sim_{\mathbb{K}}0_{\mathbb{K}},
$$
which is $(r\cdot_V y)\perp_I x$ by
\autoref{def:innerproduct-orthogonality-row}. A second application of
\autoref{thm:innerproduct-orthogonality-symmetry-row}, now to the carried pair
$r\cdot_V y$ and $x$, gives $x\perp_I(r\cdot_V y)$.
\end{proof}

\begin{theorem}[InnerProduct orthogonal additive inverse closure]
\label{thm:innerproduct-orthogonal-additive-inverse-closure}
Let $I$ be an inner-product $\mathsf{BHist}$ carrier as in
\autoref{def:innerproduct-bhist-carrier}. For carried vector endpoints
$x,y:C_V$, if $x\perp_I y$, then the displayed additive inverse $-_V y$ is
carried and
$$
x\perp_I(-_V y).
$$
\end{theorem}
\begin{proof}
The embedded $\VecSpaceUp$ row retains its additive $\AbGroupUp$ carrier by
\autoref{cor:vecspace-additive-carrier-projection}, so the inverse-closure row
carries $-_V y$.

Let $\eta=-_{\mathbb{K}}1_{\mathbb{K}}$. The retained scalar source carries
$\eta$ by \autoref{def:field-source-specification} and \autoref{def:ring-carrier}.
The module rows of \autoref{def:module-stability-certificate} identify
$\eta\cdot_V y$ with the displayed inverse: scalar unit changes $y$ to
$1_{\mathbb{K}}\cdot_V y$, scalar distributivity over scalar addition changes
$(\eta+_{\mathbb{K}}1_{\mathbb{K}})\cdot_V y$ to
$(\eta\cdot_V y)+_V(1_{\mathbb{K}}\cdot_V y)$, the additive inverse law retained
by \autoref{def:ring-stability-certificate} collapses
$\eta+_{\mathbb{K}}1_{\mathbb{K}}$ to $0_{\mathbb{K}}$, and
\autoref{thm:module-zero-action-annihilation} collapses $0_{\mathbb{K}}\cdot_V y$
to $0_V$. Thus $\eta\cdot_V y$ is a left additive inverse of $y$. The displayed
inverse row gives $y+_V(-_V y)\sim_V 0_V$, and
\autoref{thm:group-left-right-inverse-uniqueness}, read in the additive
$\AbGroupUp$ reduct, gives $\eta\cdot_V y\sim_V -_V y$.

Apply \autoref{thm:innerproduct-orthogonality-right-scalar-action-closure} to
the carried scalar $\eta$ and the hypothesis $x\perp_I y$. It gives
$x\perp_I(\eta\cdot_V y)$. Transport this orthogonality witness along
$\eta\cdot_V y\sim_V -_V y$ by
\autoref{thm:innerproduct-orthogonality-transport-row}, keeping the left endpoint
fixed by reflexivity. The result is $x\perp_I(-_V y)$.
\end{proof}
