\begin{definition}[InnerProduct norm-metric seed source]
\label{def:innerproduct-norm-metric-seed-source}
The $\InnerProductUp$ norm-metric seed source is the $\mathsf{BHist}$ vector
carrier and classifier of \autoref{def:innerproduct-bhist-source} together
with the scalar-valued diagonal endpoint row
$$
\|x\|_{I}^{2}:=\langle x,x\rangle_{\InnerProductUp}.
$$
Its scalar-zero classifier row is the positive-definite exactness row of
\autoref{thm:innerproduct-positive-definite-obligation}: the diagonal
endpoint is classified with the displayed scalar zero exactly when the vector
endpoint is classified with the displayed vector zero. Its threshold law rows
are the form endpoint transport, classifier transport, scalar endpoint source,
and visible ledger rows exposed by
\autoref{def:innerproduct-root-threshold-source}.

This seed is therefore a BHist-rooted norm-squared interface. It names the
diagonal scalar endpoint and the zero exactness row needed by later
$\NormUp$ and $\HilbertUp$ consumers, while leaving ordinary norm extraction,
metric completion, Cauchy-limit selection, topology, and host Hilbert carrier
data outside the source.
\end{definition}

\begin{theorem}[InnerProduct norm-squared carrier row]
\label{thm:innerproduct-norm-squared-carrier-row}
For every carried vector endpoint $x$ of the $\InnerProductUp$ BHist source,
the norm-metric seed source carries the scalar endpoint
$\|x\|_{I}^{2}=\langle x,x\rangle_{\InnerProductUp}$. If
$x\sim_V x'$, then the two scalar endpoints
$\|x\|_{I}^{2}$ and $\|x'\|_{I}^{2}$ are classified by the retained scalar
classifier.
\end{theorem}
\begin{proof}
Unfold \autoref{def:innerproduct-norm-metric-seed-source}. The carried vector
endpoint $x$ lies in the vector carrier of
\autoref{def:innerproduct-bhist-source}, and the diagonal form row evaluates
at $(x,x)$ with codomain in the retained scalar endpoint source.

For transport, use the form endpoint transport row listed in
\autoref{thm:innerproduct-root-threshold-form-exactness}. Apply it to the two
vector classifier witnesses $x\sim_V x'$ and $x\sim_V x'$, once for each
argument of the diagonal pair.

The resulting scalar-classifier witness compares
$\langle x,x\rangle_{\InnerProductUp}$ with
$\langle x',x'\rangle_{\InnerProductUp}$. Repack these endpoints through the
definition of $\|-\|_{I}^{2}$ in the seed source.
\end{proof}

\begin{theorem}[InnerProduct norm zero classifier exactness]
\label{thm:innerproduct-norm-zero-classifier-exactness}
For every carried vector endpoint $x$ of the $\InnerProductUp$ BHist source,
the scalar endpoint $\|x\|_{I}^{2}$ is classified with the displayed scalar
zero endpoint exactly when $x$ is classified by $\sim_V$ with the displayed
vector zero endpoint $0_V$.
\end{theorem}
\begin{proof}
Unfold the norm-metric seed source so that $\|x\|_{I}^{2}$ is the diagonal
form endpoint $\langle x,x\rangle_{\InnerProductUp}$.

In the forward direction, assume that this scalar endpoint is classified with
the displayed scalar zero. Apply
\autoref{thm:innerproduct-positive-definite-obligation} to convert the
scalar-zero classification into the vector classifier witness $x\sim_V 0_V$.

In the reverse direction, assume $x\sim_V 0_V$. The same positive-definite
obligation supplies the converse scalar-zero classification for the diagonal
endpoint. Repack that scalar witness as the zero classification of
$\|x\|_{I}^{2}$ in \autoref{def:innerproduct-norm-metric-seed-source}.
\end{proof}

\begin{theorem}[InnerProduct norm-metric seed downstream boundary]
\label{thm:innerproduct-norm-metric-seed-downstream-boundary}
The downstream interfaces $\NormUp$ and $\HilbertUp$ may consume from the
$\InnerProductUp$ side only the norm-squared seed of
\autoref{def:innerproduct-norm-metric-seed-source} together with the existing
inner-product threshold rows of
\autoref{def:innerproduct-root-threshold-source}. They do not receive Banach
completeness, a Cauchy-limit selector, topology, or a host Hilbert carrier
from this seed.
\end{theorem}
\begin{proof}
For the norm-squared endpoint, use
\autoref{thm:innerproduct-norm-squared-carrier-row}. It supplies exactly the
scalar diagonal endpoint and its classifier transport along the vector
classifier.

For zero exactness, use
\autoref{thm:innerproduct-norm-zero-classifier-exactness}. It identifies
scalar-zero classification of the diagonal endpoint with vector-zero
classification and adds no ordinary norm value or completion row.

Finally compare the seed with the root threshold source. By
\autoref{def:innerproduct-root-threshold-source}, the remaining consumable
inner-product rows are the vector carrier and classifier, scalar endpoint
source, form law rows, positive-definite exactness, classifier transport,
orthogonality row, and ledger. Since neither this definition nor
\autoref{def:innerproduct-norm-metric-seed-source} lists Banach completeness,
Cauchy-limit selection, topology, or a host Hilbert carrier, those rows cannot
be consumed from the $\InnerProductUp$ seed boundary.
\end{proof}

\begin{theorem}[InnerProduct norm-squared scalar-action factor row]
\label{thm:innerproduct-norm-squared-scalar-action-factor-row}
For every carried scalar endpoint $r$ and carried vector endpoint $x$ of the
$\InnerProductUp$ BHist source, the norm-metric seed source carries the
diagonal endpoint $\|r\cdot_V x\|_{I}^{2}$, and the retained scalar classifier
classifies it with the scalar self-product acting on the original diagonal
endpoint:
$$
\|r\cdot_V x\|_{I}^{2}
\sim_{\mathbb{K}}
\bigl(r\cdot_{\mathbb{K}}\overline{r}\bigr)
\cdot_{\mathbb{K}}
\|x\|_{I}^{2}.
$$
Here $\overline{r}$ is the displayed scalar conjugate endpoint retained by the
$\InnerProductUp$ scalar source.
\end{theorem}
\begin{proof}
Unfold \autoref{def:innerproduct-norm-metric-seed-source}. The endpoint
$\|r\cdot_V x\|_{I}^{2}$ is the diagonal form value
$$
\langle r\cdot_V x,r\cdot_V x\rangle_{\InnerProductUp}.
$$
The vector-space component of \autoref{def:innerproduct-bhist-source} carries
$r\cdot_V x$, since scalar-action closure is one of the module stability rows
retained by \autoref{def:vecspace-stability-certificate} and
\autoref{def:module-stability-certificate}. Hence
\autoref{thm:innerproduct-norm-squared-carrier-row} applies to the carried
vector endpoint $r\cdot_V x$.

Apply the scalar-action compatibility component of
\autoref{thm:innerproduct-vecspace-linearity-row} in the scalar-linear
argument. It gives a retained scalar-classifier comparison
$$
\langle r\cdot_V x,r\cdot_V x\rangle_{\InnerProductUp}
\sim_{\mathbb{K}}
r\cdot_{\mathbb{K}}
\langle x,r\cdot_V x\rangle_{\InnerProductUp}.
$$
Apply the conjugate-linear scalar-action clause from the same theorem in the
other argument. The displayed conjugation row of the scalar source supplies the
comparison
$$
\langle x,r\cdot_V x\rangle_{\InnerProductUp}
\sim_{\mathbb{K}}
\overline{r}\cdot_{\mathbb{K}}
\langle x,x\rangle_{\InnerProductUp}.
$$
Scalar multiplication congruence in the retained scalar source, read through
\autoref{def:field-stability-certificate}, transports this second comparison
under left multiplication by $r$ and gives
$$
r\cdot_{\mathbb{K}}
\langle x,r\cdot_V x\rangle_{\InnerProductUp}
\sim_{\mathbb{K}}
r\cdot_{\mathbb{K}}
\bigl(
\overline{r}\cdot_{\mathbb{K}}
\langle x,x\rangle_{\InnerProductUp}
\bigr).
$$
The scalar associativity row from the same scalar source classifies the last
endpoint with
$$
\bigl(r\cdot_{\mathbb{K}}\overline{r}\bigr)
\cdot_{\mathbb{K}}
\langle x,x\rangle_{\InnerProductUp},
$$
using classifier symmetry from \autoref{def:lean-carrier-naming-certificate}
if the stored associativity row is oriented oppositely.

Finally repack
$\langle x,x\rangle_{\InnerProductUp}$ as $\|x\|_{I}^{2}$ by
\autoref{def:innerproduct-norm-metric-seed-source}. Successive applications of
\autoref{thm:namecert-equivalence-transitivity} compose the first scalar-action
comparison, the transported conjugate-linear comparison, the scalar
associativity comparison, and this repacking into the displayed retained
scalar-classifier row.
\end{proof}
