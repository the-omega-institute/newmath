\chapter{A Concrete Naming Certificate for \(\RatUp\)}
\label{ch:concrete-instances-rat-namecert}

The rational number interface $\RatUp$ starts from a carrier skeleton: a signed unary integer numerator, drawn from \autoref{ch:concrete-instances-int-namecert}, together with a positive unary denominator. Normalized representatives, coprimality, and arithmetic closure are separate source, pattern, classifier, and stability obligations later in the chapter. The classifier is structural equality on the normalized representative, not the equivalence relation $p_1 q_2 = p_2 q_1$ on unnormalized pairs, which would otherwise demand quotient-type machinery and $\mathsf{Quot.sound}$. It is not a primitive field; it is licensed only via the integer source and the normalization function.

\begin{remark}[Rational certificate scope]
\label{rem:concrete-instances-rat-namecert-roadmap}
This chapter records the rational licensing target through normalized
integer-positive-natural representatives and structural classifier data.
The checked core covers the rational history carrier, rational history classifier fields, ledger carrier, and semantic certificate. Normalized representatives are recorded by the source and pattern obligations, while coprimality and operation closure remain part of the paper specification.
\end{remark}

\section{Carrier definition}

\begin{definition}[Rat carrier]
\label{def:rat-carrier}
Use only the carrier skeleton: a signed unary integer numerator together with a unary denominator history that is not internally the same as $\emp$:
$$
  \mathsf{RatCarrier}(s,n,d)
  :=
  \mathsf{IntCarrier}(s,n)\land \mathsf{Unary}(d)\land \neg\hsame(d,\emp).
$$
Equivalently, this is an $\IntUp$ numerator with a positive unary-history denominator. At this skeleton layer, positivity is witnessed by writing the denominator as a one-extension of a unary tail; normalized representatives, coprimality, and arithmetic closure are supplied by later source, pattern, and stability obligations.
\end{definition}
\leandef{BEDC.Derived.RatUp.RatCarrier}

\begin{theorem}[Rat carrier positive denominator]
\label{thm:rat-carrier-positive-denominator}
If \(\mathsf{RatCarrier}(s,n,d)\), then the denominator is internally the same as \(\Eone(t)\) for some unary tail \(t\).
\end{theorem}
\leanchecked{BEDC.Derived.RatUp.RatCarrier\_positive\_denominator}
\begin{proof}
Unpack the carrier into its integer numerator, unary denominator, and non-emptiness clauses. The unary denominator is either \(\emp\) or a one-extension of a unary tail. The empty case contradicts non-emptiness, while the one-extension case gives the displayed positive-denominator witness.
\end{proof}

\begin{theorem}[Positive denominator yields a rational carrier]
\label{thm:rat-carrier-from-positive-denominator}
An \(\IntUp\) numerator together with a denominator internally same as
\(\Eone(t)\) for a unary tail \(t\) forms a \(\RatUp\) carrier.
\end{theorem}
\leanchecked{BEDC.Derived.RatUp.RatCarrier\_of\_int\_positive\_denominator}
\begin{proof}
Transport unary closure from the displayed \(\Eone\)-shape to the denominator
along \(\hsame\). Non-emptiness follows because an \(\Eone\)-history cannot be
internally same as \(\emp\).
\end{proof}

\begin{theorem}[Rat carrier denominator transport]
\label{thm:rat-carrier-denominator-hsame-transport}
If \(\mathsf{RatCarrier}(s,n,d)\) and \(d\hsame d'\), then
\(\mathsf{RatCarrier}(s,n,d')\).
\end{theorem}
\leanchecked{BEDC.Derived.RatUp.RatCarrier\_denominator\_hsame\_transport}
\begin{proof}
Extract the positive denominator witness from the rational carrier, transport
that witness across denominator sameness, and rebuild the rational carrier from
the unchanged integer numerator together with the transported positive
denominator.
\end{proof}

\begin{theorem}[Positive unary denominator transport]
\label{thm:rat-positive-unary-denominator-hsame-transport}
If a denominator history has a positive unary witness and is \(\hsame\)-equivalent to another history, then the other history has the same positive unary denominator witness.
\end{theorem}
\begin{proof}
Write the positive denominator as an \(\Eone\)-extension of a unary tail. Compose the reverse \(\hsame\) witness from the target history to the source history with the source witness to the \(\Eone\)-extension, and keep the unary tail witness.
\end{proof}
\leanchecked{BEDC.Derived.RatUp.PositiveUnaryDenominator\_hsame\_transport}

\begin{theorem}[Positive unary denominator is not empty]
\label{thm:rat-positive-unary-denominator-not-empty}
If a denominator has a positive unary witness, then it is not internally the
same as \(\emp\).
\end{theorem}
\begin{proof}
Unpack the positive denominator as internally the same as \(\Eone(t)\) for a
unary tail \(t\). Composing this witness with an assumed sameness to \(\emp\)
would make an \(\Eone\)-history internally the same as \(\emp\), contradicting
history constructor no-confusion.
\end{proof}
\leanchecked{BEDC.Derived.RatUp.PositiveUnaryDenominator\_not\_empty}

\begin{theorem}[Positive unary denominator is unary and nonempty]
\label{thm:positive-unary-denominator-unary-and-nonempty}
If a denominator history has a positive unary witness, then it is unary and cannot be internally the same as \(\emp\).
\end{theorem}
\leanchecked{BEDC.Derived.RatUp.PositiveUnaryDenominator\_unary\_and\_nonempty}
\begin{proof}
Write the positive denominator as internally the same as \(\Eone(t)\) with \(t\) unary. Unary closure under \(\Eone\), transported back along history sameness, gives unary membership of the denominator. If the denominator were internally same as \(\emp\), transitivity would make \(\Eone(t)\) internally same as \(\emp\), contradicting constructor separation.
\end{proof}

\begin{theorem}[Positive unary denominator carrier equivalence]
\label{thm:rat-carrier-iff-positive-unary-denominator}
The rational carrier condition is equivalent to an $\IntUp$ numerator together with a denominator that has a positive unary tail witness.
\end{theorem}
\begin{proof}
Unpack the carrier. A unary denominator that is not $\hsame$-empty cannot be the empty unary history, so unary case analysis gives a one-extension tail. Conversely, a denominator $\hsame$-equal to a one-extension of a unary tail is unary by transport, and it cannot be $\hsame$-empty by constructor no-confusion.
\end{proof}
\leanchecked{BEDC.Derived.RatUp.RatCarrier\_iff\_positive\_unary\_denominator}

\begin{theorem}[Positive denominator append unary tail]
\label{thm:rat-positive-denominator-append-unary-tail}
If a denominator has a positive unary witness and the appended tail is unary, then the continued denominator still has a positive unary witness.
\end{theorem}
\begin{proof}
Write the denominator as internally the same as \(\Eone(t)\) for a unary tail \(t\), and transport unary membership back to the denominator itself. The appended tail is either \(\emp\) or a one-step unary tail. The empty case keeps the original witness, while the one-step case takes the continued denominator without that final step as the new positive tail; closure follows from unary append closure.
\end{proof}
\leanchecked{BEDC.Derived.RatUp.PositiveUnaryDenominator\_append\_unary\_tail}

\begin{theorem}[Rat carrier append unary denominator closure]
\label{thm:rat-carrier-append-unary-denominator-closed}
If a rational carrier has denominator \(d\), then appending any unary history to
\(d\) preserves rational-carrier membership.
\end{theorem}
\begin{proof}
Use the carrier equivalence to expose the integer numerator carrier and the
positive unary denominator witness. Positive denominators are closed under
appending unary tails, and the rational carrier is rebuilt from the same
integer numerator and the resulting positive denominator.
\end{proof}
\leanchecked{BEDC.Derived.RatUp.RatCarrier\_append\_unary\_denominator\_closed}

\begin{theorem}[Positive denominator append unary prefix]
\label{thm:rat-positive-denominator-append-unary-prefix}
If $u$ is unary and $d$ has a positive unary denominator witness, then
$\mathsf{append}(u,d)$ has a positive unary denominator witness.
\end{theorem}
\begin{proof}
Write the positive denominator witness as an $\Eone$ tail, say $d$ is
internally the same as $\Eone(t)$ with $t$ unary. Appending the unary prefix
$u$ preserves unary structure by unary append closure, and composing the
displayed $\Eone$ witness with append compatibility gives the required
positive witness for $\mathsf{append}(u,d)$.
\end{proof}
\leanchecked{BEDC.Derived.RatUp.PositiveUnaryDenominator\_append\_unary\_prefix}

\begin{theorem}[Rat carrier prepend unary denominator closure]
\label{thm:rat-carrier-prepend-unary-denominator-closed}
If a rational carrier has denominator \(d\), then prepending any unary history
to \(d\) preserves rational-carrier membership.
\end{theorem}
\begin{proof}
Use the carrier equivalence to expose the integer numerator carrier and the
positive unary denominator witness. Positive denominators are closed under
unary prefix continuation, and the rational carrier is rebuilt from the same
integer numerator and the resulting positive denominator.
\end{proof}
\leanchecked{BEDC.Derived.RatUp.RatCarrier\_prepend\_unary\_denominator\_closed}

\begin{theorem}[Rat carrier hsame transport]
\label{thm:rat-carrier-hsame-transport}
If a rational carrier \(\mathsf{RatCarrier}(s,n,d)\) is transported along
\(\msame(s,t)\), \(\hsame(n,n')\), and \(\hsame(d,d')\), then the transported
triple is again a rational carrier.
\end{theorem}
\leanchecked{BEDC.Derived.RatUp.RatCarrier\_hsame\_transport}
\begin{proof}
Eliminate the sign sameness, then unpack the rational carrier. The numerator
is transported by the integer carrier transport theorem, denominator unary
closure is transported along \(\hsame(d,d')\), and denominator non-emptiness is
preserved by composing \(\hsame(d,d')\) with any empty-target witness.
\end{proof}

\section{The certificate}

\begin{definition}[Rat source specification]
\label{def:rat-source-specification}
The source is \(\IntUp \times \NatUp\) restricted to normalized denominator-positive pairs.
\end{definition}
\leandef{BEDC.Derived.RatUp.RatSourceSpec}

\begin{theorem}[Rat source specification yields the carrier]
\label{thm:rat-source-spec-to-carrier}
A rational source witness determines a rational carrier witness: the numerator
is an integer carrier, the denominator is unary, and the positive one-tail
witness excludes internal sameness with \(\emp\).
\end{theorem}
\leanchecked{BEDC.Derived.RatUp.RatSourceSpec\_to\_RatCarrier}
\begin{proof}
Unpack the source witness. The positive-denominator component gives a unary
tail \(t\) with \(d \hsame e_1(t)\). Unary closure under one-extension and
transport along \(\hsame\) give \(\mathsf{Unary}(d)\). If \(d\hsame \emp\),
then symmetry and transitivity give \(e_1(t)\hsame \emp\), contradicting
finite-kernel history no-confusion.
\end{proof}

\begin{definition}[Rat pattern specification]
\label{def:rat-pattern-specification}
The pattern is normalized numerator/denominator construction via a total normalization function on integer and positive-natural inputs.
\end{definition}

\begin{definition}[Rat classifier specification]
\label{def:rat-classifier-specification}
The classifier is structural sameness of normalized numerator and denominator representatives, tracked by mark sameness on signs and history sameness on the numerator and denominator histories.
\end{definition}
\leandef{BEDC.Derived.RatUp.RatClassifierSpec}

\begin{theorem}[Rat classifier reflexivity]
\label{thm:rat-classifier-reflexivity}
Every rational carrier witness classifies its representative with itself.
\end{theorem}
\begin{proof}
Reuse the carrier witness for both endpoints. The sign component is witnessed by reflexivity of \(\msame\), and the numerator and denominator components are witnessed by reflexivity of \(\hsame\).
\end{proof}
\leanchecked{BEDC.Derived.RatUp.RatClassifierSpec\_refl}

\begin{theorem}[Rat classifier exposes positive denominators]
\label{thm:rat-classifier-positive-denominators}
If two rational representatives are related by the rational classifier, then
both endpoint denominators carry positive unary denominator witnesses.
\end{theorem}
\begin{proof}
Eliminate the classifier witness to expose the two endpoint rational carriers.
Apply the rational-carrier positive-denominator theorem to each endpoint
carrier.
\end{proof}
\leanchecked{BEDC.Derived.RatUp.RatClassifierSpec\_positive\_denominators}

\begin{theorem}[Rat classifier transports positive denominators]
\label{thm:rat-classifier-denominator-positive-transport}
If two rational representatives are related by the rational classifier, and
each denominator is related by \(\hsame\) to a displayed denominator history,
then both displayed denominator histories carry positive unary denominator
witnesses.
\end{theorem}
\begin{proof}
Eliminate the classifier witness to expose the two endpoint rational carriers.
Each endpoint carrier gives a positive unary denominator witness, and the two
denominator \(\hsame\) links transport those witnesses to the displayed
denominator histories.
\end{proof}
\leanchecked{BEDC.Derived.RatUp.RatClassifierSpec\_denominator\_positive\_transport}

\begin{theorem}[Rat classifier denominator append closure]
\label{thm:rat-classifier-append-unary-denominators-closed}
If two rational representatives are classified the same, then appending
\(\hsame\)-matching unary histories to their denominators preserves the
rational classifier relation.
\end{theorem}
\begin{proof}
Unpack the classifier witness into the two endpoint carriers and the sign,
numerator, and denominator sameness fields. Close both endpoint denominator
carriers under unary append, transporting the unary-tail witness along the
tail sameness for the second endpoint. The resulting denominator sameness follows by
eliminating the original denominator and tail sameness witnesses and applying
reflexivity to the resulting append history.
\end{proof}
\leanchecked{BEDC.Derived.RatUp.RatClassifierSpec\_append\_unary\_denominators\_closed}

\begin{theorem}[Rat classifier denominator prepend closure]
\label{thm:rat-classifier-prepend-unary-denominators-closed}
If two rational representatives are classified the same, then prepending
\(\hsame\)-matching unary histories to their denominators preserves the
rational classifier relation.
\end{theorem}
\begin{proof}
Unpack the classifier witness into the two endpoint carriers and the sign,
numerator, and denominator sameness fields. Close both endpoint denominator
carriers under unary prepend, transporting unary membership of the first prefix
across prefix sameness for the second endpoint. The resulting denominator
sameness follows by eliminating the original denominator and prefix sameness
witnesses and applying reflexivity to the resulting append history.
\end{proof}
\leanchecked{BEDC.Derived.RatUp.RatClassifierSpec\_prepend\_unary\_denominators\_closed}

\begin{theorem}[Rat classifier transitivity]
\label{thm:rat-classifier-transitivity}
If one rational representative is classified the same as a second representative, and the second representative is classified the same as a third representative, then the first and third representatives are classified the same.
\end{theorem}
\leanchecked{BEDC.Derived.RatUp.RatClassifierSpec\_trans}
\begin{proof}
Unpack both classifier witnesses. The endpoint carriers are preserved, while sign sameness and the two history-sameness components compose by the finite-kernel transitivity theorems for \(\msame\) and \(\hsame\).
\end{proof}

\begin{theorem}[Rat classifier symmetry]
\label{thm:rat-classifier-symmetry}
If one rational representative is classified the same as a second representative,
then the second representative is classified the same as the first.
\end{theorem}
\begin{proof}
Unpack the endpoint carriers and the sign, numerator, and denominator sameness
fields. Rebuild the classifier in the reverse direction using symmetry of
\(\msame\) for signs and symmetry of \(\hsame\) for both history components.
\end{proof}
\leanchecked{BEDC.Derived.RatUp.RatClassifierSpec\_symm}

\begin{theorem}[Rat classifier component transport]
\label{thm:rat-classifier-component-transport}
If both rational classifier endpoints are transported componentwise by
\(\msame\) on signs and \(\hsame\) on numerator and denominator histories,
then the rational classifier transports to the componentwise transported
representatives.
\end{theorem}
\begin{proof}
Unpack the rational classifier witness. Transport each endpoint rational
carrier by the component transport theorem, then compose the old sign,
numerator, and denominator sameness links with the reversed source endpoint
transports and the target endpoint transports.
\end{proof}
\leanchecked{BEDC.Derived.RatUp.RatClassifierSpec\_component\_transport}

\begin{theorem}[Rat history classifier transitivity]
\label{thm:rat-history-classifier-trans}
If denominator histories $d$ and $e$ are related by the rational history
classifier, and $e$ and $f$ are related by the same classifier, then $d$ and
$f$ are related by the rational history classifier.
\end{theorem}
\begin{proof}
Unpack the two rational history classifier witnesses. Keep the left endpoint
carrier from the first witness and the right endpoint carrier from the second
witness, then compose the denominator history sameness proofs by transitivity
of $\hsame$.
\end{proof}
\leanchecked{BEDC.Derived.RatUp.RatHistoryClassifier\_trans}

\begin{theorem}[Rat history classifier endpoint transport]
\label{thm:rat-history-classifier-hsame-transport}
If denominator histories \(d\) and \(e\) are related by the rational history
classifier, and \(d\) and \(e\) are respectively internally the same as
\(d'\) and \(e'\), then \(d'\) and \(e'\) are related by the rational history
classifier.
\end{theorem}
\begin{proof}
Unpack the rational history classifier into endpoint rational-history carriers
and the stored \(\hsame(d,e)\) link. Transport both endpoint carriers across
the supplied history-sameness witnesses, then compose the reversed
\(\hsame(d,d')\) link, the stored classifier link, and \(\hsame(e,e')\).
\end{proof}
\leanchecked{BEDC.Derived.RatUp.RatHistoryClassifier\_hsame\_transport}

\begin{definition}[Rat stability certificate]
\label{def:rat-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item normalization produces positive denominator
  \item normalization produces coprime pair
  \item normalization is idempotent
  \item structural reflexivity and transitivity
  \item positive denominator closure under unary prefix continuation
  \item closure of rational operations after re-normalization
\end{enumerate}
\end{definition}

\begin{definition}[Rat ledger policy]
\label{def:rat-ledger-policy}
Record the raw rational-history carrier and an $\hsame$ link to the visible denominator history.
\end{definition}
\leandef{BEDC.Derived.RatUp.RatHistoryLedgerPolicy}

\begin{theorem}[Rat ledger visible carrier]
\label{thm:rat-ledger-visible-carrier}
If a raw rational history satisfies the ledger policy for a visible history, then the visible history is itself a rational-history carrier.
\end{theorem}
\leanchecked{BEDC.Derived.RatUp.RatHistoryLedgerPolicy\_visible\_carrier}
\begin{proof}
Unpack the ledger policy into a raw rational-history carrier and an $\hsame$ link from the raw history to the visible history. Transport the raw carrier across that link.
\end{proof}

\begin{theorem}[Rat ledger history-sameness transport]
\label{thm:rat-ledger-hsame-transport}
If a rational ledger relates raw and visible histories, then transporting the
raw endpoint and the visible endpoint along \(\hsame\) yields another rational
ledger relation.
\end{theorem}
\leanchecked{BEDC.Derived.RatUp.RatHistoryLedgerPolicy\_hsame\_transport}
\begin{proof}
Unpack the ledger into a raw carrier and a raw-visible link. Transport the raw
carrier across the raw-endpoint link, then compose the reversed raw transport
link, the stored ledger link, and the visible-endpoint link.
\end{proof}

\begin{theorem}[Rat ledger raw-visible classifier]
\label{thm:rat-ledger-raw-visible-classifier}
If $\mathsf{RatHistoryLedgerPolicy}(\rho,v)$ holds, then
$\mathsf{RatHistoryClassifier}(\rho,v)$ holds.
\end{theorem}
\leanchecked{BEDC.Derived.RatUp.RatHistoryLedgerPolicy\_raw\_visible\_classifier}
\begin{proof}
Unpack the ledger into the raw rational-history carrier for $\rho$ and the
stored $\hsame(\rho,v)$ link. The visible-carrier theorem supplies the
rational-history carrier for $v$, and these witnesses form the raw-visible
rational history classifier.
\end{proof}

\begin{definition}[\(\RatUp\) certificate obligations]
\label{def:rat-certificate-obligations}
The carrier, source, pattern, classifier, stability, and ledger data listed above are the obligations required for the interface name \(\RatUp\).
\end{definition}

\begin{theorem}[Concrete \(\RatUp\) naming certificate]
\label{thm:concrete-rat-namecert}
The rational-history carrier/classifier tuple forms a valid semantic naming certificate for $\RatUp$ at the history level.
\end{theorem}
\leanchecked{BEDC.Derived.RatUp.rat\_history\_semantic\_name\_certificate}

\begin{corollary}[\(\RatUp\) is licensed, not primitive]
\label{cor:rat-licensed-not-primitive}
\(\RatUp\) is licensed by normalized representatives, not by a quotient of pairs.
\end{corollary}

\(\RatUp\) is not a BEDC primitive. It is admitted only when those obligations are supplied.
