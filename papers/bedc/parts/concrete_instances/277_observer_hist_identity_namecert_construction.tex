\chapter{Naming Certificate for ObserverHistIdentityUp}
\label{ch:concrete-instances-observer-hist-identity}

The object $\ObserverHistIdentityUp$ names the bridge between subject-free observation and history identity in the finite kernel. Its certificate says that an observer row is nothing beyond a $\Hist$ source, a visible mark, and the generated $\Hist$ target, with identity judged by the kernel's own $\hsame$ and $\msame$ relations.

\section{Source spec for ObserverHistIdentityUp}
\label{sec:concrete-instances-observer-hist-identity-source-spec}

\begin{definition}[Source rows of ObserverHistIdentityUp]
\label{def:concrete-instances-observer-hist-identity-source-rows}
A source row for $\ObserverHistIdentityUp$ is a triple $(h,m,r)$ consisting of a source history $h : \Hist$, a visible mark $m : \Mark$, and a target history $r : \Hist$. The row is the raw event-flow datum from which the observer-history bridge is read: a history emits one distinction and records the resulting history.
\leandef{BEDC.Derived.ObserverHistIdentityUp.SourceRows}
\end{definition}

\section{Pattern spec}
\label{sec:concrete-instances-observer-hist-identity-pattern}

\begin{definition}[Pattern witness for ObserverHistIdentityUp]
\label{def:concrete-instances-observer-hist-identity-pattern}
The pattern of a source row $(h,m,r)$ is the internal extension witness $\Ext(h,m,r)$. Thus the row has the observer-history shape exactly when the target is generated from the source by the displayed mark, with no additional subject, clock, or equality principle outside $\Hist$, $\Mark$, and $\Ext$.
\leandef{BEDC.Derived.ObserverHistIdentityUp.Pattern}
\end{definition}

\section{Classifier spec}
\label{sec:concrete-instances-observer-hist-identity-classifier}

\begin{definition}[Classifier for ObserverHistIdentityUp]
\label{def:concrete-instances-observer-hist-identity-classifier}
Two source rows $(h,m,r)$ and $(h',m',r')$ name the same $\ObserverHistIdentityUp$ event when their sources are $\hsame$-equivalent, their marks are $\msame$-equivalent, and their targets are $\hsame$-equivalent. This classifier is componentwise and internal: it compares only histories and marks already present in the row.
\leandef{BEDC.Derived.ObserverHistIdentityUp.Classifier}
\end{definition}

\section{Stability certificate}
\label{sec:concrete-instances-observer-hist-identity-stability}

\begin{theorem}[Stability of ObserverHistIdentityUp]
\label{thm:concrete-instances-observer-hist-identity-stability}
If a source row satisfies the observer-history pattern and a second source row is classified with it componentwise, then the second row also satisfies the observer-history pattern. Equivalently, the observer-history bridge is closed under transport along $\hsame$ on source histories, $\msame$ on emitted marks, and $\hsame$ on target histories.
\leanchecked{BEDC.Derived.ObserverHistIdentityUp.stability}
\end{theorem}

\section{Ledger policy}
\label{sec:concrete-instances-observer-hist-identity-ledger}

\begin{definition}[Ledger for ObserverHistIdentityUp]
\label{def:concrete-instances-observer-hist-identity-ledger}
The ledger for $\ObserverHistIdentityUp$ keeps the source row together with its pattern witness. It therefore records the raw history before observation, the emitted mark, the generated history after observation, and the proof that this triple is an internal extension event. No external observer identity, host equality, or hidden selection function is stored.
\leandef{BEDC.Derived.ObserverHistIdentityUp.Ledger}
\end{definition}
