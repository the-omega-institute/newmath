\chapter{A Concrete Naming Certificate for $\YonedaUp$}
\label{ch:concrete-instances-yoneda-namecert}

The Yoneda interface $\YonedaUp$ packages, for each object of a category, the representable functor and the natural-isomorphism witness of the Yoneda lemma. It depends on $\NameCert_{\FunctorUp}$ (\autoref{ch:concrete-instances-functor-namecert}) and $\NameCert_{\NatTransUp}$ (\autoref{ch:concrete-instances-nattrans-namecert}), supplying the embedding into presheaf categories.

\section{Carrier definition}
\label{sec:yoneda-carrier}

\begin{theorem}[Yoneda representable empty component family characterization]
\label{thm:yoneda-representable-empty-component-family-iff}
For histories $p$ and $q$, an empty natural-transformation component exists
over every unary object exactly when $p$ and $q$ are unary prefixes with the
same history:
$$
\Bigl(\forall a,\ \mathsf{UnaryHistory}(a)\Rightarrow
\mathsf{NatTransPrefixComponentCarrier}(p,q,a,\emp)\Bigr)
\quad\Longleftrightarrow\quad
\mathsf{UnaryHistory}(p)\wedge\mathsf{UnaryHistory}(q)\wedge\HSame(p,q).
$$
\end{theorem}
\begin{proof}
For the forward direction, evaluate the component family at the empty unary
history and apply empty-component identity exactness. For the reverse direction,
the same exactness theorem rebuilds each object component from the common prefix
equality and the supplied unary object carrier.
\end{proof}
\leanchecked{BEDC.Derived.YonedaUp.YonedaRepresentable\_empty\_component\_family\_iff}

\begin{theorem}[Yoneda empty component family e1 target characterization]
\label{thm:yoneda-representable-empty-component-family-e1-target-iff}
For generated histories $p$ and $q$, the empty component family with target
$\Eone(q)$ is equivalent to unary membership of $q$ together with
$p\hsame\Eone(q)$:
$$
\Bigl(\forall a,\ \mathsf{UnaryHistory}(a)\Rightarrow
\mathsf{NatTransPrefixComponentCarrier}(p,\Eone(q),a,\emp)\Bigr)
\Longleftrightarrow
\mathsf{UnaryHistory}(q)\land p\hsame\Eone(q).
$$
\end{theorem}
\begin{proof}
The forward direction applies the empty component family characterization and
then inverts unary membership of the $\Eone$ target. Conversely, unary
membership of $q$ closes $\Eone(q)$ under unary construction, and the
$\hsame$ comparison transports unary membership back to $p$. The general empty
component family characterization then gives the displayed family.
\end{proof}
\leanchecked{BEDC.Derived.YonedaUp.YonedaRepresentable\_empty\_component\_family\_e1\_target\_iff}

\begin{theorem}[Yoneda empty component family displayed determinacy]
\label{thm:yoneda-representable-empty-component-family-displayed-deterministic}
If an empty representable component family and a displayed component family
both exist over every unary object for the same prefixes $p,q$, then
$\HSame(\emp,displayed)$.
\end{theorem}
\begin{proof}
Evaluate both families at the empty unary object. Their component carriers
share the same source and target hom-carrier endpoints, so morphism determinacy
identifies the empty displayed morphism with the supplied displayed component.
\end{proof}
\leanchecked{BEDC.Derived.YonedaUp.YonedaRepresentable\_empty\_component\_family\_displayed\_deterministic}

\begin{theorem}[Yoneda component family displayed determinacy]
\label{thm:yoneda-representable-component-family-displayed-deterministic}
If two displayed component families both exist over every unary object for the
same prefixes $p,q$, then their displayed histories are $\HSame$-equivalent.
\end{theorem}
\begin{proof}
Evaluate both families at the empty unary object. The two exposed component
carriers share the same prefixed hom-carrier source and target, so category
morphism determinacy identifies the two displayed component histories.
\end{proof}
\leanchecked{BEDC.Derived.YonedaUp.YonedaRepresentable\_component\_family\_displayed\_deterministic}

\begin{theorem}[Yoneda empty component family component determinacy]
\label{thm:yoneda-representable-empty-component-family-component-deterministic}
If $a$ is unary, an empty representable component family exists over every
unary object, and a displayed component over $a$ is supplied for the same
prefixes $p,q$, then $\HSame(\emp,displayed)$.
\end{theorem}
\begin{proof}
Evaluate the empty component family at the supplied unary object $a$. The
resulting empty component and the displayed component expose category
hom-carriers with the same prefixed source and target, so morphism determinacy
identifies their component histories.
\end{proof}
\leanchecked{BEDC.Derived.YonedaUp.YonedaRepresentable\_empty\_component\_family\_component\_deterministic}

\begin{theorem}[Yoneda representable empty component family target determinacy]
\label{thm:yoneda-representable-empty-component-family-target-deterministic}
For histories $p,q,r:\mathsf{BHist}$, if the empty natural-transformation
component family from $p$ to $q$ exists over every unary object and the empty
component family from $p$ to $r$ exists over every unary object, then $q$ and
$r$ have the same history:
$$
\Bigl(\forall a,\ \mathsf{UnaryHistory}(a)\Rightarrow
\mathsf{NatTransPrefixComponentCarrier}(p,q,a,\emp)\Bigr)\Rightarrow
\Bigl(\forall a,\ \mathsf{UnaryHistory}(a)\Rightarrow
\mathsf{NatTransPrefixComponentCarrier}(p,r,a,\emp)\Bigr)\Rightarrow
\HSame(q,r).
$$
\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:yoneda-representable-empty-component-family-iff} to
both component families. The first family gives $\HSame(p,q)$ and the second
gives $\HSame(p,r)$. Symmetry converts the first comparison to
$\HSame(q,p)$, and transitivity composes it with $\HSame(p,r)$.
\end{proof}
\leanchecked{BEDC.Derived.YonedaUp.YonedaRepresentable\_empty\_component\_family\_target\_deterministic}

\begin{theorem}[Yoneda representable empty component family source determinacy]
\label{thm:yoneda-representable-empty-component-family-source-deterministic}
For histories $p,q,r:\mathsf{BHist}$, if the empty natural-transformation
component family from $p$ to $q$ exists over every unary object and the empty
component family from $r$ to $q$ exists over every unary object, then $p$ and
$r$ have the same history:
$$
\Bigl(\forall a,\ \mathsf{UnaryHistory}(a)\Rightarrow
\mathsf{NatTransPrefixComponentCarrier}(p,q,a,\emp)\Bigr)\Rightarrow
\Bigl(\forall a,\ \mathsf{UnaryHistory}(a)\Rightarrow
\mathsf{NatTransPrefixComponentCarrier}(r,q,a,\emp)\Bigr)\Rightarrow
\HSame(p,r).
$$
\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:yoneda-representable-empty-component-family-iff} to
both component families. The first family gives $\HSame(p,q)$ and the second
gives $\HSame(r,q)$. Symmetry converts the second comparison to
$\HSame(q,r)$, and transitivity composes it with $\HSame(p,q)$.
\end{proof}
\leanchecked{BEDC.Derived.YonedaUp.YonedaRepresentable\_empty\_component\_family\_source\_deterministic}

\begin{theorem}[Yoneda empty component family boundary hsame equivalence]
\label{thm:yoneda-representable-empty-component-family-boundary-hsame-iff}
For histories $p,q,r,s:\mathsf{BHist}$, suppose that the empty component
family from $p$ to $q$ exists over every unary object, and likewise the empty
component family from $r$ to $s$ exists over every unary object. Then comparing
the sources is equivalent to comparing the targets:
$$
\HSame(p,r)\Longleftrightarrow\HSame(q,s).
$$
\end{theorem}
\begin{proof}
Apply the empty component family characterization to both families. The first
family gives $\HSame(p,q)$ and the second gives $\HSame(r,s)$. For the forward
direction, compose $\HSame(q,p)$, $\HSame(p,r)$, and $\HSame(r,s)$. For the
reverse direction, compose $\HSame(p,q)$, $\HSame(q,s)$, and $\HSame(s,r)$.
\end{proof}
\leanchecked{BEDC.Derived.YonedaUp.YonedaRepresentable\_empty\_component\_family\_boundary\_hsame\_iff}

\begin{theorem}[Yoneda empty component family boundary transport]
\label{thm:yoneda-empty-component-family-boundary-transport}
For histories $p,q,r,s:\mathsf{BHist}$, an empty component family from $p$ to
$q$ transports to an empty component family from $r$ to $s$ whenever
$p\hsame r$ and $q\hsame s$:
$$
\Bigl(\forall a,\ \mathsf{UnaryHistory}(a)\Rightarrow
\mathsf{NatTransPrefixComponentCarrier}(p,q,a,\emp)\Bigr)
\Rightarrow
\HSame(p,r)\Rightarrow\HSame(q,s)\Rightarrow
\Bigl(\forall a,\ \mathsf{UnaryHistory}(a)\Rightarrow
\mathsf{NatTransPrefixComponentCarrier}(r,s,a,\emp)\Bigr).
$$
\end{theorem}
\begin{proof}
Use the empty component family characterization to extract unary membership
for $p$ and $q$, together with $\HSame(p,q)$. Unary transport along the two
boundary equalities gives unary membership for $r$ and $s$, while transitivity
of $\HSame$ composes $\HSame(r,p)$, $\HSame(p,q)$, and $\HSame(q,s)$. The
empty-identity component characterization then rebuilds each component over
the supplied unary object.
\end{proof}
\leanchecked{BEDC.Derived.YonedaUp.YonedaRepresentable\_empty\_component\_family\_boundary\_transport}

\section{The certificate}
\label{sec:yoneda-certificate}

\begin{theorem}[Yoneda representable empty component family semantic certificate]
\label{thm:yoneda-representable-empty-component-family-semantic-namecert}
For a unary prefix history $p$, the empty component family predicate
$$
Y_p(q) :\Longleftrightarrow
\forall a,\ \mathsf{UnaryHistory}(a)\Rightarrow
\mathsf{NatTransPrefixComponentCarrier}(p,q,a,\emp)
$$
carries a semantic naming certificate with source, pattern, and ledger all
equal to $Y_p$, and classifier $\HSame$.
\end{theorem}
\begin{proof}
The carrier witness is $q=p$. For each unary object $a$, the empty-component
identity exactness theorem rebuilds
$\mathsf{NatTransPrefixComponentCarrier}(p,p,a,\emp)$ from unary membership of
$p$, unary membership of $a$, and reflexivity of $\HSame(p,p)$. The classifier
fields are the reflexivity, symmetry, and transitivity of $\HSame$, while
carrier transport along $\HSame(q,r)$ is obtained by replacing $q$ with $r$ in
the component family. Since source, pattern, and ledger are the same predicate,
the soundness rows are direct projections.
\end{proof}
\leanchecked{BEDC.Derived.YonedaUp.YonedaRepresentable\_empty\_component\_family\_semanticNameCert}

\paragraph{Capstone.}

\begin{theorem}[Prefix Yoneda embedding is fully faithful on empty components]
\label{thm:yoneda-prefix-empty-component-full-faithfulness}
For unary histories $p$ and $q$, the empty representable component family of
the prefix Yoneda construction is equivalent to prefix identity:
$$
\Bigl(\forall a,\ \mathsf{UnaryHistory}(a)\Rightarrow
\mathsf{NatTransPrefixComponentCarrier}(p,q,a,\emp)\Bigr)
\quad\Longleftrightarrow\quad
\HSame(p,q).
$$
Moreover, if displayed histories $\eta$ and $\theta$ both occur as component
families from $p$ to $q$ over all unary objects, then their displayed histories
are identical in the BEDC history classifier:
$$
\Bigl(\forall a,\ \mathsf{UnaryHistory}(a)\Rightarrow
\mathsf{NatTransPrefixComponentCarrier}(p,q,a,\eta)\Bigr)
\wedge
\Bigl(\forall a,\ \mathsf{UnaryHistory}(a)\Rightarrow
\mathsf{NatTransPrefixComponentCarrier}(p,q,a,\theta)\Bigr)
\Rightarrow \HSame(\eta,\theta).
$$
Thus the semantic family $Y_p$ classifies a unary target $q$ exactly by
$\HSame(p,q)$; on the empty-component skeleton this is the full and faithful
prefix embedding of $\CategoryUp$ through the certified $\FunctorUp$ and
$\NatTransUp$ interfaces.
\end{theorem}
\begin{proof}
For the equivalence, apply
\autoref{thm:yoneda-representable-empty-component-family-iff}. The forward
direction projects the exposed $\HSame(p,q)$ field from the unary-prefix
package. The reverse direction combines the ambient unary hypotheses on $p$ and
$q$ with $\HSame(p,q)$ and repacks them through the same characterization.

For displayed family uniqueness, apply
\autoref{thm:yoneda-representable-component-family-displayed-deterministic} to
the two supplied component families. This is the faithfulness clause: once the
source and target prefixes are fixed, the object-indexed displayed component
history is determined up to $\HSame$.

The fullness clause is the empty-component existence half of the first
paragraph. The functorial source of the construction is the semantic prefix
certificate of \autoref{thm:functor-prefix-semantic-namecert}; the natural
transformation classifier is supplied by
\autoref{thm:nattrans-prefix-component-classifier-semantic-namecert}. Together
with the semantic family certificate above, these identify $Y_p(q)$ with prefix
identity and leave no additional displayed component choice.
\end{proof}
\leanchecked{BEDC.Derived.YonedaUp.YonedaRepresentable\_prefix\_empty\_component\_full\_faithfulness}

\begin{theorem}[Prefix Yoneda empty-component standard bridge]
\label{thm:yoneda-prefix-empty-component-standard-bridge}
For unary histories $p$ and $q$, the BEDC predicate
$$
Y_p(q) :\Longleftrightarrow
\forall a,\ \mathsf{UnaryHistory}(a)\Rightarrow
\mathsf{NatTransPrefixComponentCarrier}(p,q,a,\emp)
$$
has a paper-side conservative bridge to the external unary-prefix presheaf
readback. The decode map reads $q$ as its finite unary prefix, and the encode
map sends an external unary word back to its BEDC unary history. Bridge
exactness identifies equality of external unary readbacks with $\HSame(p,q)$
under the carrier $Y_p(q)$, and the displayed component choice is unique up to
$\HSame$.
\end{theorem}
\begin{proof}
The source of the bridge is the semantic prefix certificate of
\autoref{thm:functor-prefix-semantic-namecert}. The component classifier used
by the readback is the natural-transformation certificate of
\autoref{thm:nattrans-prefix-component-classifier-semantic-namecert}.

Carrier exactness is exactly
\autoref{thm:yoneda-representable-empty-component-family-iff}: under the unary
hypotheses, $Y_p(q)$ is equivalent to $\HSame(p,q)$. Thus decoding $q$ to its
external unary prefix and encoding an external unary word back to a BEDC unary
history preserve and reflect the same prefix equality.

For displayed components, apply
\autoref{thm:yoneda-representable-component-family-displayed-deterministic}.
Any two component choices over the same unary source and target histories are
identified by $\HSame$, so the external readback has no additional component
choice beyond the BEDC classifier.
\end{proof}

\begin{theorem}[Prefix Yoneda empty-component mature interface]
\label{thm:yoneda-prefix-empty-component-mature-interface}
For unary histories $p$ and $q$, the mature scoped interface of the prefix
Yoneda object is exactly the prefix empty-component skeleton. Its carrier is
$Y_p(q)$, its classifier is $\HSame$, its full-faithfulness clause is
Theorem~\ref{thm:yoneda-prefix-empty-component-full-faithfulness}, its
conservative paper bridge is
Theorem~\ref{thm:yoneda-prefix-empty-component-standard-bridge}, and its
semantic certificate is
Theorem~\ref{thm:yoneda-representable-empty-component-family-semantic-namecert}.
The scope is restricted to unary prefix histories and empty components.
\end{theorem}
\begin{proof}
The carrier and classifier are fixed by the semantic certificate in
Theorem~\ref{thm:yoneda-representable-empty-component-family-semantic-namecert}:
the family $Y_p(q)$ is the empty-component representable family, and the
classifier it exposes is $\HSame$.

The fully faithful part is
Theorem~\ref{thm:yoneda-prefix-empty-component-full-faithfulness}. It identifies
the object-indexed carrier over all unary objects with $\HSame(p,q)$ and makes
the displayed component history unique up to $\HSame$.

The bridge part is
Theorem~\ref{thm:yoneda-prefix-empty-component-standard-bridge}. Its encode and
decode readings are conservative for the same unary-prefix carrier, so composing
the semantic certificate, the full-faithfulness clause, and this bridge gives
the stated mature interface without enlarging the domain. No step ranges beyond
unary prefix histories or beyond empty components.
\end{proof}
