\chapter{Certificate Obligations for $\RealUp$}
\label{ch:concrete-instances-real-namecert}

The real number interface $\RealUp$ is specified here as a certificate-obligation schema over sealed $\RegSeqRat$ stream names. These names are built from $\DyadicRatCore$ observations through $\StreamNameUp$, then sealed with regular-sequence data and a closeness classifier. The classical Dedekind / Cauchy completion is outside this schema because quotient and completion infrastructure is not exported by the mathlib-free BEDC kernel; the interface name is available only through the dyadic rational core, certified stream names, regular-sequence certificates, and their classifier.

\section{Carrier definition}

\begin{definition}[Real carrier]
\label{def:real-carrier}
Use a sealed $\RegSeqRat$ name. Its observations form a $\StreamNameUp$ stream over $\DyadicRatCore$, and its regularity certificate is exposed only through the $\RealUp$ seal.
\end{definition}

\begin{definition}[Constant real history carrier]
\label{def:real-constant-history-carrier}
A history is in the constant real history carrier when it is internally the same as an $\Eone$-sealed rational history:
$$
\mathsf{RealConstantHistoryCarrier}(h)
\quad\Longleftrightarrow\quad
\exists d,\ \hsame(h,\Eone(d))\ \wedge\ \mathsf{RatHistoryCarrier}(d).
$$
\end{definition}
\leandef{BEDC.Derived.RealUp.RealConstantHistoryCarrier}

\begin{theorem}[Constant real carrier at an e-one seal]
\label{thm:real-constant-history-carrier-e1-iff-rat}
For every history $d$,
$$
\mathsf{RealConstantHistoryCarrier}(\Eone(d))
\quad\Longleftrightarrow\quad
\mathsf{RatHistoryCarrier}(d).
$$
\end{theorem}
\begin{proof}
Unpack the constant carrier witness. One-constructor inversion turns
$\hsame(\Eone(d),\Eone(d'))$ into $\hsame(d,d')$, and rational history carrier
transport moves the witness back along that equality. The converse uses $d$
itself as the exposed rational history and reflexivity of $\hsame$.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealConstantHistoryCarrier\_e1\_iff\_rat}

\begin{theorem}[Constant real carrier e-zero exclusion]
\label{thm:real-constant-history-carrier-e0-absurd}
For every history tail $t$,
$$
\mathsf{RealConstantHistoryCarrier}(\Ezero(t))
\Longrightarrow
\bot.
$$
\end{theorem}
\begin{proof}
Unpack the constant real carrier witness. It displays $\Ezero(t)$ as
$\Eone(d)$ for some rational history $d$. BEDC history no-confusion between
zero-extension and one-extension constructors eliminates that display.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealConstantHistoryCarrier\_e0\_absurd}

\begin{theorem}[Constant real carrier empty exclusion]
\label{thm:real-constant-history-carrier-empty-absurd}
The empty history is not a constant real history carrier:
$$
\mathsf{RealConstantHistoryCarrier}(\emp)
\Longrightarrow
\bot.
$$
\end{theorem}
\begin{proof}
Unpack the constant real carrier witness. It displays $\emp$ as an
$\Eone$-sealed rational history, and BEDC history no-confusion between the
empty constructor and the one-extension constructor eliminates that display.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealConstantHistoryCarrier\_empty\_absurd}

\section{The certificate-obligation schema}

\begin{definition}[Real source specification]
\label{def:real-source-specification}
The source has three supplied layers: $\DyadicRatCore$ observations, $\StreamNameUp$ stream naming, and the $\RegSeqRat$ certificate over those observations.
\end{definition}

\begin{definition}[Real pattern specification]
\label{def:real-pattern-specification}
The pattern is sealed regular-sequence generation: every finite observation is a certified dyadic rational, stream observations are stable pointwise, and the $\RegSeqRat$ regularity witness supplies the dyadic Cauchy bound.
\end{definition}

\begin{definition}[Real classifier specification]
\label{def:real-classifier-specification}
The classifier is $\RegSeqRat$ closeness: two sealed real names classify together when their dyadic regular-sequence witnesses supply comparison certificates at every requested dyadic tolerance.
\end{definition}

\begin{definition}[Constant real history classifier]
\label{def:real-constant-history-classifier}
Two constant real histories classify together when each exposes an $\Eone$-sealed rational history and the exposed rational histories classify:
$$
\mathsf{RealConstantHistoryClassifier}(h,k)
\quad\Longleftrightarrow\quad
\exists d,e,\ \hsame(h,\Eone(d))\ \wedge\ \hsame(k,\Eone(e))
\ \wedge\ \mathsf{RatHistoryClassifier}(d,e).
$$
\end{definition}
\leandef{BEDC.Derived.RealUp.RealConstantHistoryClassifier}

\begin{theorem}[Constant real classifier at e-one seals]
\label{thm:real-constant-history-classifier-e1-iff-rat}
For all histories $d$ and $e$,
$$
\mathsf{RealConstantHistoryClassifier}(\Eone(d),\Eone(e))
\quad\Longleftrightarrow\quad
\mathsf{RatHistoryClassifier}(d,e).
$$
\end{theorem}
\begin{proof}
Eliminate the two seal witnesses. One-constructor inversion gives the two tail
$\hsame$ witnesses, and rational classifier transport carries the exposed
classifier back to the displayed tails. The reverse direction chooses the same
two rational histories and uses reflexivity at both seals.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealConstantHistoryClassifier\_e1\_iff\_rat}

\begin{theorem}[Constant real classifier endpoint carriers]
\label{thm:real-constant-history-classifier-endpoint-carriers}
If two constant real histories classify together, then both endpoints are in the
constant real history carrier:
$$
\mathsf{RealConstantHistoryClassifier}(h,k)
\Longrightarrow
\mathsf{RealConstantHistoryCarrier}(h)\land
\mathsf{RealConstantHistoryCarrier}(k).
$$
\end{theorem}
\begin{proof}
Unpack the classifier witness into rational histories $d,e$, endpoint displays
$\hsame(h,\Eone(d))$ and $\hsame(k,\Eone(e))$, and a rational history
classifier between $d$ and $e$. The rational classifier stores rational carrier
witnesses at both exposed endpoints, so pairing those carrier witnesses with the
two endpoint displays gives the two constant real history carriers.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealConstantHistoryClassifier\_endpoint\_carriers}

\begin{theorem}[Constant real classifier endpoint transport]
\label{thm:real-constant-history-classifier-endpoint-transport}
If constant real histories $h,k$ classify together, and $h\hsame h'$ and
$k\hsame k'$, then $h'$ and $k'$ classify together.
\end{theorem}
\begin{proof}
Unpack the two sealed rational witnesses. Symmetry and transitivity of
$\hsame$ move the displayed endpoint equalities from $h,k$ to $h',k'$, while
the rational classifier witness remains attached to the same sealed tails.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealConstantHistoryClassifier\_endpoint\_transport}

\begin{theorem}[Constant real classifier invalid endpoint exclusions]
\label{thm:real-constant-history-classifier-invalid-endpoint-absurd}
For every history $h$ and history tail $t$, the constant real-history classifier
excludes every endpoint shape that is not an $\Eone$-sealed rational history:
\begin{enumerate}
  \item $\mathsf{RealConstantHistoryClassifier}(\Ezero(t),h)\Longrightarrow\bot$;
  \item $\mathsf{RealConstantHistoryClassifier}(h,\Ezero(t))\Longrightarrow\bot$;
  \item $\mathsf{RealConstantHistoryClassifier}(\emp,h)\Longrightarrow\bot$;
  \item $\mathsf{RealConstantHistoryClassifier}(h,\emp)\Longrightarrow\bot$.
\end{enumerate}
In the ten-field fit, the traditional target is sealed constant-real endpoint
no-confusion; the BEDC source is $\mathsf{BHist}$ rational seals; the carrier is
$\mathsf{RealConstantHistoryCarrier}$; the classifier is
$\mathsf{RealConstantHistoryClassifier}$; the stability obligations are endpoint
carrier projection and carrier exclusions; the ledger is invalid endpoint shape;
the core theorem is this block; the exactness predicate is impossible
non-$\Eone$ endpoint; the standard bridge is constant rational real names; and
the status is $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
\begin{enumerate}
  \item Apply
  \autoref{thm:real-constant-history-classifier-endpoint-carriers} to the
  classifier witness. Its left projection gives
  $\mathsf{RealConstantHistoryCarrier}(\Ezero(t))$. Apply
  \autoref{thm:real-constant-history-carrier-e0-absurd}.

  \item Apply
  \autoref{thm:real-constant-history-classifier-endpoint-carriers} to the
  classifier witness. Its right projection gives
  $\mathsf{RealConstantHistoryCarrier}(\Ezero(t))$. Apply
  \autoref{thm:real-constant-history-carrier-e0-absurd}.

  \item Apply
  \autoref{thm:real-constant-history-classifier-endpoint-carriers} to the
  classifier witness. Its left projection gives
  $\mathsf{RealConstantHistoryCarrier}(\emp)$. Apply
  \autoref{thm:real-constant-history-carrier-empty-absurd}.

  \item Apply
  \autoref{thm:real-constant-history-classifier-endpoint-carriers} to the
  classifier witness. Its right projection gives
  $\mathsf{RealConstantHistoryCarrier}(\emp)$. Apply
  \autoref{thm:real-constant-history-carrier-empty-absurd}.
\end{enumerate}
\end{proof}

\begin{theorem}[Constant real classifier equivalence fields]
\label{thm:real-constant-history-classifier-equivalence-fields}
The constant real history classifier supplies the equivalence fields carried by
the constant-real certificate package:
\begin{enumerate}
  \item if $\mathsf{RealConstantHistoryCarrier}(h)$, then
  $\mathsf{RealConstantHistoryClassifier}(h,h)$;
  \item if $\mathsf{RealConstantHistoryClassifier}(h,k)$, then
  $\mathsf{RealConstantHistoryClassifier}(k,h)$;
  \item if $\mathsf{RealConstantHistoryClassifier}(h,k)$ and
  $\mathsf{RealConstantHistoryClassifier}(k,\ell)$, then
  $\mathsf{RealConstantHistoryClassifier}(h,\ell)$.
\end{enumerate}
In the ten-field fit, the traditional target is constant rational embedding
into $\RealUp$; the BEDC source is $BHist$ rational histories; the carrier is
$\mathsf{RealConstantHistoryCarrier}$; the classifier is
$\mathsf{RealConstantHistoryClassifier}$; the stability obligation is the
equivalence fields above; the ledger is the sealed rational tail; the core
theorem is this displayed package; exactness is inherited from the rational
classifier; the standard bridge is constant rational real names; and the status
is $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
For reflexivity, unpack $\mathsf{RealConstantHistoryCarrier}(h)$ as a sealed
rational tail $d$ with $\hsame(h,\Eone(d))$ and
$\mathsf{RatHistoryCarrier}(d)$. The rational carrier supplies the reflexive
rational classifier on $d$, using $\hsame$ reflexivity on the exposed tail.
Packing the same sealed tail twice and using the displayed endpoint witness on
both sides gives $\mathsf{RealConstantHistoryClassifier}(h,h)$.

For symmetry, unpack
$\mathsf{RealConstantHistoryClassifier}(h,k)$ into sealed rational tails $d,e$,
endpoint displays for $h$ and $k$, and
$\mathsf{RatHistoryClassifier}(d,e)$. Keep the same endpoint displays in the
opposite order and apply $\mathsf{RatHistoryClassifier\_symm}$ to the rational
classifier.

For transitivity, unpack the two classifiers into tails $d,e$ and $e',f$. The
two middle displays for $k$ compare $\Eone(e)$ and $\Eone(e')$ by $\hsame$
symmetry and transitivity; one-constructor inversion turns that comparison into
$\hsame(e,e')$. Transport the right rational classifier from $e'$ to $e$, then
compose it with the left rational classifier by
$\mathsf{RatHistoryClassifier\_trans}$. Repacking the outer endpoint displays
for $h$ and $\ell$ gives the required classifier.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealConstantHistoryClassifier\_equivalence\_fields}

\begin{definition}[Real stream classifiers]
\label{def:real-stream-classifiers}
For rational-observation streams $x,y:\mathbb{N}\to\Hist$, the stream classifier is pointwise rational-history classification. The finite prefix classifier records the same data recursively through a selected observation index.
\end{definition}
\leandef{BEDC.Derived.RealUp.RealStreamClassifier}
\leandef{BEDC.Derived.RealUp.RealStreamPrefixClassifier}

\begin{definition}[Real unary-observation stream classifier]
\label{def:real-unary-observation-stream-classifier}
For rational-observation maps $s,t:\Hist\to\Hist$, the unary-observation
stream classifier is pointwise rational-history classification over certified
unary BEDC observation histories:
$$
\mathsf{RealUnaryStreamClassifier}(s,t)
\quad\Longleftrightarrow\quad
\forall n,\ \mathsf{UnaryHistory}(n)\to
\mathsf{RatHistoryClassifier}(s(n),t(n)).
$$
\end{definition}

\begin{theorem}[Constant rational streams bridge StreamNameUp and RealUp]
\label{thm:streamname-real-constant-prefix-bridge}
For histories $d,e:\Hist$, the RatUp point classifier, the StreamNameUp
constant-stream classifier, the RealUp unary-observation stream classifier, and
the constant real-history classifier at e-one seals are equivalent:
$$
\mathsf{RatHistoryClassifier}(d,e)
\quad\Longleftrightarrow\quad
\mathsf{RatStreamNameClassifier}(\mathsf{RatConstStream}_d,
\mathsf{RatConstStream}_e).
$$
$$
\mathsf{RatStreamNameClassifier}(\mathsf{RatConstStream}_d,
\mathsf{RatConstStream}_e)
\quad\Longleftrightarrow\quad
\mathsf{RealUnaryStreamClassifier}(\mathsf{RatConstStream}_d,
\mathsf{RatConstStream}_e).
$$
$$
\mathsf{RealUnaryStreamClassifier}(\mathsf{RatConstStream}_d,
\mathsf{RatConstStream}_e)
\quad\Longleftrightarrow\quad
\mathsf{RealConstantHistoryClassifier}(\Eone(d),\Eone(e)).
$$
\end{theorem}
\begin{proof}
For the first equivalence, the forward direction is exactly the constant
classifier part of \autoref{thm:rat-stream-name-constant-witness}. Conversely,
unfold Definition~\ref{def:rat-stream-name-classifier} and read the pointwise
classifier at the empty unary observation history $\emp$ using its unary
witness; the two constant stream values are $d$ and $e$.

For the second equivalence, unfold
Definition~\ref{def:rat-stream-name-classifier} and
Definition~\ref{def:real-unary-observation-stream-classifier}. Both predicates
are the same pointwise requirement over certified unary observation histories,
so each direction is identity transport of the displayed witness.

For the last equivalence, unfold
Definition~\ref{def:real-unary-observation-stream-classifier} and
$\mathsf{RatConstStream}$. Reading at the empty unary observation gives
$\mathsf{RatHistoryClassifier}(d,e)$, while a rational point classifier supplies
the same pointwise witness at every unary observation.

Apply \autoref{thm:real-constant-history-classifier-e1-iff-rat} to identify that
rational point classifier with
$\mathsf{RealConstantHistoryClassifier}(\Eone(d),\Eone(e))$.
\end{proof}

The certificate fit has ten fields: traditional target constant rational real
names; BEDC source RatUp histories plus StreamNameUp unary observations; carrier
constant rational observation maps; classifier the displayed bridge; stability
inherited from RatUp classifier fields; ledger the constant rational history and
unary observation witness; core theorem
\autoref{thm:streamname-real-constant-prefix-bridge}; exactness the displayed
equivalence; standard bridge constant rational real sequence; status
$\mathsf{ProofTarget}$.

\begin{theorem}[Real stream classifier prefix closure]
\label{thm:real-stream-classifier-prefix}
If rational-observation streams are pointwise related by the rational history classifier, then every finite observation prefix is related by the recursively accumulated prefix classifier.
\end{theorem}
\begin{proof}
Induct on the selected prefix length. The base case is the pointwise classifier at zero. The successor case pairs the induction hypothesis for the shorter prefix with the pointwise classifier at the successor observation.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealStreamClassifier\_prefix}

\begin{theorem}[Real stream prefix classifier hsame transport]
\label{thm:real-stream-prefix-classifier-hsame-transport}
If two rational-observation streams are related pointwise by $\hsame$ on both
sides, then every classified finite prefix transports to the pointwise-related
streams.
\end{theorem}
\begin{proof}
Induct on the prefix length. The base observation is rational classifier
transport at index $0$. At a successor index, the induction hypothesis
transports the shorter prefix and rational classifier transport handles the
successor observation.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealStreamPrefixClassifier\_hsame\_transport}

\begin{theorem}[Real stream prefix classifier reflexivity]
\label{thm:real-stream-prefix-classifier-refl}
For every rational-observation stream $x:\mathbb{N}\to\Hist$, if
$\mathsf{RatHistoryCarrier}(x(i))$ for every observation index $i$, then
$\mathsf{RealStreamPrefixClassifier}(x,x,n)$ for every prefix index $n$.
\end{theorem}
\begin{proof}
Induct on $n$. At $0$, the carrier witness for $x(0)$ supplies the rational
history classifier for the endpoint, with $\hsame$ reflexivity providing the
identity transport on both sides.

At a successor index, pair the induction hypothesis for the shorter prefix with
the same rational classifier construction at the new endpoint $n+1$.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealStreamPrefixClassifier\_refl}

\begin{theorem}[Real stream prefix classifier symmetry]
\label{thm:real-stream-prefix-classifier-symm}
For rational-observation streams $x,y:\mathbb{N}\to\Hist$, if
$\mathsf{RealStreamPrefixClassifier}(x,y,n)$, then
$\mathsf{RealStreamPrefixClassifier}(y,x,n)$.
\end{theorem}
\begin{proof}
Induct on $n$. At $0$, apply rational-history classifier symmetry to the base
endpoint classifier.

At a successor index, reverse the shorter prefix by the induction hypothesis and
reverse the displayed successor endpoint by rational-history classifier
symmetry.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealStreamPrefixClassifier\_symm}

\begin{theorem}[Real stream prefix classifier endpoint projection]
\label{thm:real-stream-prefix-classifier-endpoint}
For rational-observation streams $x,y:\mathbb{N}\to\Hist$, if
$\mathsf{RealStreamPrefixClassifier}(x,y,n)$, then
$\mathsf{RatHistoryClassifier}(x(n),y(n))$.
\end{theorem}
\begin{proof}
Split on $n$. At $0$, the prefix witness is exactly the selected rational-history
classifier.

At a successor index, the recursive prefix witness is a pair; its right
component is the rational-history classifier at the selected endpoint.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealStreamPrefixClassifier\_endpoint}

\begin{theorem}[Real stream prefix classifier base of successor]
\label{thm:real-stream-prefix-classifier-base-of-successor}
For rational-observation streams $x,y:\mathbb{N}\to\Hist$, every nonempty
classified finite prefix contains the base observation classifier:
$$
\mathsf{RealStreamPrefixClassifier}(x,y,n+1)
\Longrightarrow
\mathsf{RealStreamPrefixClassifier}(x,y,0).
$$
\end{theorem}
\begin{proof}
Induct on $n$. At $0$, the successor prefix is the pair of the base classifier
and the first successor endpoint classifier, so project its left component. At
a further successor, recursively descend through the left prefix component until
the base observation classifier is exposed.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealStreamPrefixClassifier\_base\_of\_successor}

\begin{lemma}[Real stream prefix previous classifier with unary anchor]
\label{lem:real-stream-prefix-previous-with-unary}
For rational-observation streams \(x,y:\mathbb{N}\to\Hist\), if every left observation \(x(i)\) is unary, then every successor finite-prefix classifier contains both the previous prefix classifier and the unary anchor at the previous left endpoint:
$$
\mathsf{RealStreamPrefixClassifier}(x,y,n+1)
\Longrightarrow
\mathsf{RealStreamPrefixClassifier}(x,y,n)\wedge \mathsf{UnaryHistory}(x(n)).
$$
\end{lemma}
\begin{proof}
The recursive successor prefix is a pair whose left component is the previous prefix classifier. The unary side condition supplies \(\mathsf{UnaryHistory}(x(n))\) at the same previous endpoint.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealStreamPrefixClassifier\_previous\_with\_unary}

\begin{theorem}[Real stream prefix add-left previous classifier with unary anchor]
\label{thm:real-stream-prefix-add-left-previous-with-unary}
For rational-observation streams \(x,y:\mathbb{N}\to\Hist\), if every left observation \(x(i)\) is unary, then any prefix classified through \(m+n\) descends to the prefix classified through \(n\), while retaining the unary anchor at \(x(n)\):
$$
\mathsf{RealStreamPrefixClassifier}(x,y,m+n)
\Longrightarrow
\mathsf{RealStreamPrefixClassifier}(x,y,n)\wedge \mathsf{UnaryHistory}(x(n)).
$$
\end{theorem}
\begin{proof}
Induct on \(m\). At \(m=0\), the displayed prefix is the target prefix and the unary side condition supplies the endpoint anchor. At a successor \(m+1\), apply \autoref{lem:real-stream-prefix-previous-with-unary} to peel one recursive prefix layer, then apply the induction hypothesis to the exposed prefix through \(m+n\).
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealStreamPrefixClassifier\_add\_left\_previous\_with\_unary}

\begin{theorem}[Real stream finite-prefix exactness]
\label{thm:real-stream-finite-prefix-exactness}
For rational-observation streams $x,y:\mathbb{N}\to\Hist$,
$$
\mathsf{RealStreamClassifier}(x,y)
\quad\Longleftrightarrow\quad
\forall n,\ \mathsf{RealStreamPrefixClassifier}(x,y,n).
$$
\end{theorem}
\begin{proof}
For the forward direction, apply
\autoref{thm:real-stream-classifier-prefix}. The pointwise stream classifier
supplies exactly the rational-history classifier needed at each endpoint of the
recursive prefix construction.

For the reverse direction, assume every finite prefix is classified and fix an
observation index $n$. Apply the hypothesis at $n$, then use
\autoref{thm:real-stream-prefix-classifier-endpoint} to recover
$\mathsf{RatHistoryClassifier}(x(n),y(n))$. Since $n$ was arbitrary, these
endpoint classifiers are precisely $\mathsf{RealStreamClassifier}(x,y)$.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealStreamClassifier\_finite\_prefix\_exactness}

\begin{theorem}[Real stream classifier symmetry]
\label{thm:real-stream-classifier-symmetry}
For rational-observation streams $x,y:\mathbb{N}\to\Hist$, if
$\mathsf{RealStreamClassifier}(x,y)$, then
$\mathsf{RealStreamClassifier}(y,x)$.
\end{theorem}
\begin{proof}
Fix an observation index. The stream classifier supplies the rational-history
classifier at that index, and rational-history classifier symmetry reverses it.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealStreamClassifier\_symm}

\begin{theorem}[Real stream classifier transitivity]
\label{thm:real-stream-classifier-transitivity}
For rational-observation streams $x,y,z:\mathbb{N}\to\Hist$, if
$\mathsf{RealStreamClassifier}(x,y)$ and
$\mathsf{RealStreamClassifier}(y,z)$, then
$\mathsf{RealStreamClassifier}(x,z)$.
\end{theorem}
\begin{proof}
Fix an observation index. The two stream classifiers give the two
rational-history classifier witnesses at that index, and rational-history
classifier transitivity composes them.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealStreamClassifier\_trans}

\begin{theorem}[Real stream classifier certificate fields]
\label{thm:real-stream-classifier-certificate-fields}
For rational-observation streams $x,y,z:\mathbb{N}\to\Hist$, the pointwise
stream classifier supplies the following stability certificate fields:
\begin{enumerate}
  \item if $\forall n,\ \mathsf{RatHistoryCarrier}(x(n))$, then
  $\mathsf{RealStreamClassifier}(x,x)$;
  \item if $\mathsf{RealStreamClassifier}(x,y)$, then for every $n$ both
  $\mathsf{RatHistoryCarrier}(x(n))$ and $\mathsf{RatHistoryCarrier}(y(n))$;
  \item if $\mathsf{RealStreamClassifier}(x,y)$, then
  $\mathsf{RealStreamClassifier}(y,x)$;
  \item if $\mathsf{RealStreamClassifier}(x,y)$ and
  $\mathsf{RealStreamClassifier}(y,z)$, then
  $\mathsf{RealStreamClassifier}(x,z)$.
\end{enumerate}
\end{theorem}
\begin{proof}
For reflexivity, fix the stream $x$ and its pointwise rational-history carrier
witness. Apply \autoref{thm:real-stream-prefix-classifier-refl} at every finite
prefix. The reverse direction of
\autoref{thm:real-stream-finite-prefix-exactness} then repacks these prefix
witnesses as $\mathsf{RealStreamClassifier}(x,x)$.

For endpoint carrier readback, unfold the pointwise stream classifier and fix an
index $n$. The classifier at $n$ is a $\mathsf{RatHistoryClassifier}(x(n),y(n))$
witness, and unpacking the rational-history classifier gives both endpoint
carrier facts.

The symmetry field is exactly
\autoref{thm:real-stream-classifier-symmetry}. The transitivity field is exactly
\autoref{thm:real-stream-classifier-transitivity}. These four clauses are
therefore available as the certificate-level stability package for the stream
classifier.
\end{proof}

\begin{theorem}[Real stream prefix classifier transitivity]
\label{thm:real-stream-prefix-classifier-trans}
For rational-observation streams $x,y,z:\mathbb{N}\to\Hist$, if the finite
prefix classifier relates $x$ to $y$ through index $n$ and relates $y$
to $z$ through the same index, then it relates $x$ to $z$ through
index $n$.
\end{theorem}
\begin{proof}
Induct on the prefix index. At zero, use rational history classifier
transitivity. At a successor index, combine the induction hypothesis for the
shorter prefix with rational history classifier transitivity at the new
endpoint.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealStreamPrefixClassifier\_trans}

\begin{theorem}[Real stream prefix one-tail pair readback]
\label{thm:real-stream-prefix-e1-pair-readback}
If a classified finite rational-observation prefix displays its selected left and right endpoints as one-extension histories, then the displayed tails are unary and internally the same.
\end{theorem}
\begin{proof}
Split on the selected observation index. At zero, use the base rational-history classifier; at a successor, use the right endpoint of the prefix classifier. Transport the selected classifier to the displayed one-extension endpoints, then apply rational one-tail classifier exactness.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealStreamPrefixClassifier\_e1\_pair\_readback}

\begin{definition}[Real stability certificate]
\label{def:real-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item pointwise stream stability inherited from $\StreamNameUp$
  \item transport of the $\RegSeqRat$ regularity witness along stream classification
  \item classifier reflexivity, symmetry, and transitivity by dyadic tolerance comparison
  \item compatibility of constant dyadic embedding with the closeness classifier
  \item closure of real operations only when supplied as certified $\RegSeqRat$ transformations
\end{enumerate}
\end{definition}

\begin{definition}[Real ledger policy]
\label{def:real-ledger-policy}
Record finite stream observations, dyadic comparison witnesses, regularity certificates, and classifier comparison certificates when sameness is used.
\end{definition}

\begin{definition}[RealUp certificate obligations]
\label{def:real-certificate-obligations}
The carrier, source, pattern, classifier, stability, and ledger data listed above are the required fields for using the interface name $\RealUp$.
\end{definition}

\paragraph{Primitive boundary.}
$\RealUp$ is not a BEDC primitive. It is admitted only when those obligations are supplied.

\begin{remark}[Zero-axiom encoding note]
\label{rem:real-zero-axiom-note}
The interface avoids quotient machinery. It does not use $\mathsf{Quot.sound}$; sameness is carried by dyadic comparison certificates and the $\RegSeqRat$ classifier.
\end{remark}
