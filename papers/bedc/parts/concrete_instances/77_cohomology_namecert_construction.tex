\chapter{A Concrete Naming Certificate for $\CohomologyUp$}
\label{ch:concrete-instances-cohomology-namecert}

The cohomology interface $\CohomologyUp$ packages a cochain complex with the dual kernel/image quotients computing cohomology groups. It depends on $\NameCert_{\HomologyUp}$ (\autoref{ch:concrete-instances-homology-namecert}) and supplies the structural substrate for de~Rham, sheaf, and \v{C}ech cohomology certificates.

\section{Carrier definition}
\label{sec:cohomology-carrier}

\begin{theorem}[Cohomology cocycle axis right cancel]
\label{thm:cohomology-cocycle-axis-right-cancel}
Let $d:\mathsf{BHist}\to\mathsf{BHist}$ preserve append up to $\hsame$. If
the shifted history $h\cdot axis$ is a $d$-cocycle, then $h$ is a
$d$-cocycle.
\end{theorem}
\begin{proof}
Append preservation rewrites the shifted differential as
$\mathsf{append}(d(h),d(axis))$. BEDC append-empty inversion reads the left
component back from the empty appended history, giving $d(h)\hsame\emp$.
\end{proof}
\leanchecked{BEDC.Derived.CohomologyUp.CohomologyCocycle\_axis\_right\_cancel}

\begin{theorem}[Cohomology cocycle axis context cancel]
\label{thm:cohomology-cocycle-axis-context-cancel}
Let $d:\mathsf{BHist}\to\mathsf{BHist}$ preserve append up to $\hsame$. If
the two-sided shifted history $left\cdot(h\cdot right)$ is a $d$-cocycle,
then $h$ is a $d$-cocycle.
\end{theorem}
\begin{proof}
Apply left-axis cancellation to the outer context. This recovers the
right-axis shifted cocycle for $h\cdot right$. Applying right-axis
cancellation to that shifted cocycle gives $d(h)\hsame\emp$.
\end{proof}

\begin{theorem}[Cohomology cocycle bilateral axis cancel]
\label{thm:cohomology-cocycle-axis-bilateral-cancel}
Let $d:\mathsf{BHist}\to\mathsf{BHist}$ preserve append up to $\hsame$. If
the doubly shifted history $(leftAxis\cdot h)\cdot rightAxis$ is a
$d$-cocycle, then $h$ is a $d$-cocycle.
\end{theorem}
\begin{proof}
Right-axis cancellation removes the trailing $rightAxis$ from the displayed
cocycle and yields a cocycle on $leftAxis\cdot h$. Left-axis cancellation then
removes the leading $leftAxis$ and gives the core cocycle condition on $h$.
\end{proof}

\begin{theorem}[Cohomology cocycle append closure]
\label{thm:cohomology-cocycle-append-closed}
Let $d:\mathsf{BHist}\to\mathsf{BHist}$ preserve append up to $\hsame$, and
let the fixed cohomology axis satisfy $d(axis)\hsame\emp$. If the shifted
histories $h\cdot axis$ and $k\cdot axis$ are $d$-cocycles, then the shifted
append $(h\cdot k)\cdot axis$ is also a $d$-cocycle.
\end{theorem}
\begin{proof}
Apply append preservation to the two shifted cocycle hypotheses. Append-empty
inversion gives $d(h)\hsame\emp$ and $d(k)\hsame\emp$, since the axis
derivative is empty. Append preservation for $h\cdot k$ then makes
$d(h\cdot k)$ empty, and one final append-preservation step with the same
axis gives the displayed shifted cocycle.
\end{proof}
\leanchecked{BEDC.Derived.CohomologyUp.CohomologyCocycle\_append\_closed}

\begin{theorem}[Cohomology cocycle prepend axis closure]
\label{thm:cohomology-cocycle-prepend-axis-closed}
Let $d:\mathsf{BHist}\to\mathsf{BHist}$ preserve append up to $\hsame$, and
let the fixed cohomology axis satisfy $d(axis)\hsame\emp$. If the shifted
histories $axis\cdot h$ and $axis\cdot k$ are $d$-cocycles, then the shifted
append $axis\cdot(h\cdot k)$ is also a $d$-cocycle.
\end{theorem}
\begin{proof}
Append preservation rewrites the two shifted cocycle hypotheses as empty
append histories $\mathsf{append}(d(axis),d(h))$ and
$\mathsf{append}(d(axis),d(k))$. Append-empty inversion extracts
$d(h)\hsame\emp$ and $d(k)\hsame\emp$. The core append closure makes
$d(h\cdot k)$ empty, and the axis cocycle row then rebuilds the displayed
left-axis cocycle through the append law.
\end{proof}
\leanchecked{BEDC.Derived.CohomologyUp.CohomologyCocycle\_prepend\_axis\_closed}

\begin{theorem}[Cohomology cocycle append hsame transport]
\label{thm:cohomology-cocycle-append-hsame-transport}
Let $d:\mathsf{BHist}\to\mathsf{BHist}$ preserve append up to $\hsame$ and
be congruent for $\hsame$, and let the fixed cohomology axis satisfy
$d(axis)\hsame\emp$. If the shifted histories $h\cdot axis$ and
$k\cdot axis$ are $d$-cocycles and
$\hsame((h\cdot k)\cdot axis,r)$, then $r$ is a $d$-cocycle.
\end{theorem}
\begin{proof}
The append closure theorem gives the cocycle predicate for
$(h\cdot k)\cdot axis$. Congruence for $d$ transports the endpoint
comparison to the corresponding differential histories, and symmetry plus
transitivity of $\hsame$ carries the cocycle predicate to $r$.
\end{proof}
\leanchecked{BEDC.Derived.CohomologyUp.CohomologyCocycle\_append\_hsame\_transport}

\begin{theorem}[Cohomology cocycle core append closure]
\label{thm:cohomology-cocycle-append-core-closed}
Let $d:\mathsf{BHist}\to\mathsf{BHist}$ preserve append up to $\hsame$. If
$h$ and $k$ are $d$-cocycles in the core history sense, then their append
$h\cdot k$ is also a $d$-cocycle.
\end{theorem}
\begin{proof}
Convert the two core cocycle hypotheses to empty equalities for $d(h)$ and
$d(k)$. Append-empty introduction makes $d(h)\cdot d(k)$ empty. Append
preservation transports this empty result back to $d(h\cdot k)$.
\end{proof}
\leanchecked{BEDC.Derived.CohomologyUp.CohomologyCocycle\_append\_core\_closed}

\begin{theorem}[Cohomology cocycle append empty equivalence]
\label{thm:cohomology-cocycle-append-empty-iff}
Let $d:\mathsf{BHist}\to\mathsf{BHist}$ preserve append up to $\hsame$. For
histories $h,k:\mathsf{BHist}$, the appended history $h\cdot k$ is a
$d$-cocycle exactly when both $h$ and $k$ are $d$-cocycles:
$$
d(h\cdot k)\hsame\emp
\quad\Longleftrightarrow\quad
d(h)\hsame\emp\ \wedge\ d(k)\hsame\emp.
$$
\end{theorem}
\begin{proof}
Append preservation transports the displayed cocycle on $h\cdot k$ to the
empty appended history $\mathsf{append}(d(h),d(k))$, and BEDC
append-empty inversion splits this into the two component cocycle equations.
Conversely, the core append closure theorem combines the two component
cocycles into the appended cocycle.
\end{proof}
\leanchecked{BEDC.Derived.CohomologyUp.CohomologyCocycle\_append\_empty\_iff}

\begin{theorem}[Cohomology cocycle left-shift append empty equivalence]
\label{thm:cohomology-cocycle-left-shift-append-empty-iff}
Let $d:\mathsf{BHist}\to\mathsf{BHist}$ preserve append up to $\hsame$. For
histories $axis,h,k:\mathsf{BHist}$, the left-shifted appended history
$axis\cdot(h\cdot k)$ is a $d$-cocycle exactly when the axis and both tail
components are $d$-cocycles:
$$
d(axis\cdot(h\cdot k))\hsame\emp
\quad\Longleftrightarrow\quad
d(axis)\hsame\emp\ \wedge\ d(h)\hsame\emp\ \wedge\ d(k)\hsame\emp.
$$
\end{theorem}
\begin{proof}
Apply append-empty equivalence to split the left axis from the tail
$h\cdot k$. Apply the same equivalence again to split the tail into $h$ and
$k$. The converse first rebuilds the tail cocycle and then rebuilds the
left-shifted cocycle.
\end{proof}
\leanchecked{BEDC.Derived.CohomologyUp.CohomologyCocycle\_left\_shift\_append\_empty\_iff}

\begin{theorem}[Cohomology cocycle axis context append closure]
\label{thm:cohomology-cocycle-axis-context-append-closed}
Let $d:\mathsf{BHist}\to\mathsf{BHist}$ preserve append up to $\hsame$, and
let $left$ and $right$ be $d$-cocycles. If the two-sided shifted histories
$left\cdot(h\cdot right)$ and $left\cdot(k\cdot right)$ are $d$-cocycles,
then the two-sided shifted append $left\cdot((h\cdot k)\cdot right)$ is also
a $d$-cocycle.
\end{theorem}
\begin{proof}
Two-sided context cancellation applied to the two hypotheses gives core
cocycles for $h$ and $k$. Core append closure makes $h\cdot k$ a core
cocycle. Append preservation with the right context, then with the left
context, combines these three cocycle rows into the displayed two-sided
shifted cocycle.
\end{proof}
\leanchecked{BEDC.Derived.CohomologyUp.CohomologyCocycle\_axis\_context\_append\_closed}
