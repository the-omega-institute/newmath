\chapter{A Concrete Naming Certificate for $\CohomologyUp$}
\label{ch:concrete-instances-cohomology-namecert}

The cohomology interface $\CohomologyUp$ packages a cochain complex with the dual kernel/image quotients computing cohomology groups. It depends on $\NameCert_{\HomologyUp}$ (\autoref{ch:concrete-instances-homology-namecert}) and supplies the structural substrate for de~Rham, sheaf, and \v{C}ech cohomology certificates.

\section{Carrier definition}
\label{sec:cohomology-carrier}

\begin{theorem}[Cohomology cocycle axis right cancel]
\label{thm:cohomology-cocycle-axis-right-cancel}
Let $d:\mathsf{BHist}\to\mathsf{BHist}$ preserve append up to $\hsame$. If
the shifted history $h\cdot axis$ is a $d$-cocycle, then $h$ is a
$d$-cocycle.
\end{theorem}
\begin{proof}
Append preservation rewrites the shifted differential as
$\mathsf{append}(d(h),d(axis))$. BEDC append-empty inversion reads the left
component back from the empty appended history, giving $d(h)\hsame\emp$.
\end{proof}
\leanchecked{BEDC.Derived.CohomologyUp.CohomologyCocycle\_axis\_right\_cancel}

\end{proof}

\begin{theorem}[Cohomology cocycle append closure]
\label{thm:cohomology-cocycle-append-closed}
Let $d:\mathsf{BHist}\to\mathsf{BHist}$ preserve append up to $\hsame$, and
let the fixed cohomology axis satisfy $d(axis)\hsame\emp$. If the shifted
histories $h\cdot axis$ and $k\cdot axis$ are $d$-cocycles, then the shifted
append $(h\cdot k)\cdot axis$ is also a $d$-cocycle.
\end{theorem}
\begin{proof}
Apply append preservation to the two shifted cocycle hypotheses. Append-empty
inversion gives $d(h)\hsame\emp$ and $d(k)\hsame\emp$, since the axis
derivative is empty. Append preservation for $h\cdot k$ then makes
$d(h\cdot k)$ empty, and one final append-preservation step with the same
axis gives the displayed shifted cocycle.
\end{proof}
\leanchecked{BEDC.Derived.CohomologyUp.CohomologyCocycle\_append\_closed}

\begin{theorem}[Cohomology cocycle prepend axis closure]
\label{thm:cohomology-cocycle-prepend-axis-closed}

\begin{proof}

$d(h)\hsame\emp$ and $d(k)\hsame\emp$. The core append closure makes
$d(h\cdot k)$ empty, and the axis cocycle row then rebuilds the displayed
left-axis cocycle through the append law.
\end{proof}
\leanchecked{BEDC.Derived.CohomologyUp.CohomologyCocycle\_prepend\_axis\_closed}

\begin{theorem}[Cohomology cocycle continuation context cancel]
\label{thm:cohomology-cocycle-continuation-context-cancel}
Let $d:\mathsf{BHist}\to\mathsf{BHist}$ preserve append up to $\hsame$ and
respect $\hsame$ on histories. If $\Cont(left,h,mid)$ and
$\Cont(mid,right,r)$ compose a two-sided context around $h$, then any
$d$-cocycle on $r$ restricts to a $d$-cocycle on $h$.
\end{theorem}
\begin{proof}
The two continuation ledgers identify $r$ with $(left\cdot h)\cdot right$.
Associativity rewrites this as $left\cdot(h\cdot right)$, and congruence for
$d$ transports the cocycle witness to that reassociated context. Axis context
cancellation then removes the visible left and right factors.
\end{proof}
\leanchecked{BEDC.Derived.CohomologyUp.CohomologyCocycle\_continuation\_context\_cancel}

\begin{theorem}[Cohomology cocycle continuation context empty iff]
\label{thm:cohomology-cocycle-continuation-context-empty-iff}
Let $d:\mathsf{BHist}\to\mathsf{BHist}$ preserve append up to $\hsame$. If
$\Cont(left,h,mid)$ and $\Cont(mid,right,r)$ compose a two-sided continuation
context, then $r$ is a $d$-cocycle exactly when the three exposed components
$left$, $h$, and $right$ are $d$-cocycles:
$$
d(r)\hsame\emp
\quad\Longleftrightarrow\quad
d(left)\hsame\emp\ \wedge\ d(h)\hsame\emp\ \wedge\ d(right)\hsame\emp.
$$
\end{theorem}
\begin{proof}
The two continuation ledgers identify $r$ with $(left\cdot h)\cdot right$.
Append-empty equivalence first splits the outer append into the cocycle on
$left\cdot h$ and the cocycle on $right$, and then splits $left\cdot h$ into
the two cocycles on $left$ and $h$. Conversely, core append closure combines
$left$ with $h$ and then combines that result with $right$.
\end{proof}
\leanchecked{BEDC.Derived.CohomologyUp.CohomologyCocycle\_continuation\_context\_empty\_iff}

\section{The certificate}
\label{sec:cohomology-certificate}

\begin{definition}[Cohomology cocycle predicate]
\label{def:cohomology-cocycle-predicate}
For a BEDC cohomology differential $d:\mathsf{BHist}\to\mathsf{BHist}$, write
$$
C_d(h) :\Longleftrightarrow d(h)\hsame\emp.
$$
\end{definition}

\begin{lemma}[Cohomology cocycle hsame transport]
\label{lem:cohomology-cocycle-hsame-transport}
Let $d:\mathsf{BHist}\to\mathsf{BHist}$ respect $\hsame$. If $C_d(h)$ and $\hsame(h,k)$, then $C_d(k)$.
\end{lemma}
\leanchecked{BEDC.Derived.CohomologyUp.CohomologyCocycle\_hsame\_transport}
\begin{proof}
By the respect-$\hsame$ hypothesis on $d$, $\hsame(h,k)$ implies $\hsame(d(h),d(k))$. Symmetry of that comparison, followed by the cocycle row $d(h)\hsame\emp$, gives $d(k)\hsame\emp$.
\end{proof}

\begin{theorem}[Cohomology cocycle predicate semantic name certificate]
\label{thm:cohomology-cocycle-predicate-semantic-namecert}
Let $d:\mathsf{BHist}\to\mathsf{BHist}$ preserve append up to $\hsame$ and respect $\hsame$. Fix $axis:\mathsf{BHist}$ with $C_d(axis)$. Then the cocycle predicate $C_d$, used simultaneously as source, pattern, and ledger, with classifier $\hsame$, carries a semantic naming certificate:
$$
\mathsf{SemanticNameCert}(C_d,C_d,C_d,\hsame).
$$
Its cohomology stability witnesses are the bilateral axis-cancel row of \autoref{thm:cohomology-cocycle-axis-bilateral-cancel} and the append-closure row of \autoref{thm:cohomology-cocycle-append-closed}.
\end{theorem}
\begin{proof}
The source is inhabited by the chosen axis cocycle. The classifier fields in \autoref{def:lean-carrier-naming-certificate} are the finite-kernel reflexivity, symmetry, and transitivity laws for $\hsame$. Carrier transport for the core certificate is \autoref{lem:cohomology-cocycle-hsame-transport}. Therefore $C_d$ with classifier $\hsame$ forms the core $\NameCert(C_d,\hsame)$.

Since source, pattern, and ledger are all the same predicate $C_d$, both semantic soundness maps are identity maps on a source witness.
