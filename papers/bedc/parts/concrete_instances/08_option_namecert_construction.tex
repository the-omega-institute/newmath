\section{Tagged option map relation R-corner classifier coherence}
\label{sec:option-maprel-r-corner-classifier-coherence}

\begin{theorem}[Tagged option map relation R-corner classifier coherence]
\label{thm:tagged-option-maprel-r-corner-classifier-coherence}
This theorem fits the per-object template as follows: the traditional target is
right-corner option-map endpoint determinacy; the BEDC source is
$\mathsf{BHist}$ with one $\mathsf{DescentCertificate}$; the carrier is the
tagged option endpoint shape; the classifiers are
$\mathsf{TaggedOptionHistoryClassifier}_{S,\sim_S}$ and
$\mathsf{TaggedOptionHistoryClassifier}_{T,\sim_T}$; the stability obligations
are shared-target source reflection and shared-source target normal form; the
ledger records the three R-corner legs; the core theorem is simultaneous source
and target endpoint classification; exactness is absent-or-present exhaustion on
both axes; the standard bridge is $\mathsf{none}$/$\mathsf{some}$ map readback;
and the status is $\mathsf{ProofTarget}$.

Let $S,T:\mathsf{BHist}\to\mathsf{Prop}$ be BEDC history source and target
carriers, let $\sim_S$ and $\sim_T$ be their classifiers, let
$$
\delta:\mathsf{DescentCertificate}(\mathsf{BHist},\mathsf{BHist},\sim_S,\sim_T),
$$
and assume payload preservation
$$
\rho_{\delta}:\forall a.\;S(a)\to T(\delta.\mathsf{map}(a)).
$$
Assume $\mathsf{TaggedOptionSourceHsameCompatible}(S,\sim_S)$ and reflected
source readback from visible target equality,
$$
\mathsf{TaggedOptionPayloadDescentReflectsSource}_{\delta}(S,\sim_S).
$$
For histories $h,h',k,k'$, if
$$
\mathsf{TaggedOptionMapRel}_{\delta}(S,T,\rho_{\delta};h,k),
$$
$$
\mathsf{TaggedOptionMapRel}_{\delta}(S,T,\rho_{\delta};h',k),
$$
and
$$
\mathsf{TaggedOptionMapRel}_{\delta}(S,T,\rho_{\delta};h',k'),
$$
then
$$
\mathsf{TaggedOptionHistoryClassifier}_{S,\sim_S}(h,h')
\land
\mathsf{TaggedOptionHistoryClassifier}_{T,\sim_T}(k,k').
$$
\end{theorem}
\begin{proof}
First compare the two legs with shared target endpoint $k$. Apply
Theorem~\ref{thm:tagged-option-maprel-shared-target-source-reflection-normal-form}
to the map-relation witnesses
$\mathsf{TaggedOptionMapRel}_{\delta}(S,T,\rho_{\delta};h,k)$ and
$\mathsf{TaggedOptionMapRel}_{\delta}(S,T,\rho_{\delta};h',k)$. This gives
either simultaneous empty presentations of $h,h',k$ or a present-payload
package ending in
$\mathsf{TaggedOptionHistoryClassifier}_{S,\sim_S}(h,h')$.

In an absent branch of this shared-target comparison, the empty presentations of
$h$ and $h'$ assemble the absent branch of
$\mathsf{TaggedOptionHistoryClassifier}_{S,\sim_S}(h,h')$. In its present
branch, retain the source classifier supplied by the reflected normal form.

Next compare the two legs with shared source endpoint $h'$. Apply
Theorem~\ref{thm:tagged-option-maprel-shared-source-target-normal-form} to the
map-relation witnesses
$\mathsf{TaggedOptionMapRel}_{\delta}(S,T,\rho_{\delta};h',k)$ and
$\mathsf{TaggedOptionMapRel}_{\delta}(S,T,\rho_{\delta};h',k')$. The result
gives either simultaneous empty presentations of $h',k,k'$ or a present-payload
package ending in
$\mathsf{TaggedOptionHistoryClassifier}_{T,\sim_T}(k,k')$.

In an absent branch of this shared-source comparison, the empty presentations of
$k$ and $k'$ assemble the absent branch of
$\mathsf{TaggedOptionHistoryClassifier}_{T,\sim_T}(k,k')$. In its present
branch, retain the target classifier supplied by the normal form.

Combining the source classifier obtained from the shared-target comparison with
the target classifier obtained from the shared-source comparison yields the
displayed conjunction.
\end{proof}

\section{Tagged option payload descent image target classifier closure}
\label{sec:option-payload-descent-image-target-classifier-closure}

\begin{theorem}[Tagged option payload descent image classifier target classifier closure]
\label{thm:tagged-option-payload-descent-image-classifier-target-classifier-closure}
This theorem fits the per-object template as follows: the traditional target is
functorial $\mathsf{OptionUp}$ target classifier closure; the BEDC source is
$\mathsf{BHist}$ with a descent certificate; the carrier is the transported
target tagged-option carrier; the classifier is
$\mathsf{TaggedOptionHistoryClassifier}_{T,\sim_T}$; the stability obligation is
branch exactness plus descent respect; the ledger is absent or transported
present payload provenance; the core theorem is the displayed classifier
implication; exactness is the branch split; the standard bridge is
$\mathsf{none}$/$\mathsf{some}$ classifier closure; and the status is
$\mathsf{ProofTarget}$.

Let $S,T:\mathsf{BHist}\to\mathsf{Prop}$ be BEDC history carriers, let
$\sim_S$ and $\sim_T$ be classifiers, let
$\delta:\mathsf{DescentCertificate}(\mathsf{BHist},\mathsf{BHist},\sim_S,\sim_T)$,
and assume payload preservation
$$
\rho_{\delta}:\forall a.\;S(a)\to T(\delta.\mathsf{map}(a)).
$$
Let $\mathsf{cert}:\NameCert(S,\sim_S)$ and assume
$\mathsf{TaggedOptionSourceHsameCompatible}(S,\sim_S)$. For target endpoints
$k,k':\mathsf{BHist}$,
$$
\mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
(S,T,\rho_{\delta};k,k')
\to
\mathsf{TaggedOptionHistoryClassifier}_{T,\sim_T}(k,k').
$$
\end{theorem}
\begin{proof}
Apply
Theorem~\ref{thm:tagged-option-payload-descent-image-classifier-branch-exactness}
to the displayed carrier, classifier, descent, preservation, certificate,
compatibility, and endpoint data. The resulting branch split is either
absent-absent target presentation or transported present-payload presentation.

In the absent branch, the split supplies $\hsame(k,\emp)$ and
$\hsame(k',\emp)$. Use the absent-absent constructor of
$\mathsf{TaggedOptionHistoryClassifier}_{T,\sim_T}$.

In the present branch, keep the witnesses $a,b:\mathsf{BHist}$, the source
carrier witnesses, $a\sim_S b$, the two transported target carrier witnesses,
and the two target endpoint presentations. The descent certificate's respect
field gives
$$
\delta.\mathsf{respects}(a\sim_S b):
\delta.\mathsf{map}(a)\sim_T\delta.\mathsf{map}(b).
$$

Use the present-present constructor of
$\mathsf{TaggedOptionHistoryClassifier}_{T,\sim_T}$ with target witnesses
$\delta.\mathsf{map}(a)$ and $\delta.\mathsf{map}(b)$, their two $T$ witnesses,
the transported classifier witness, and the two endpoint presentations.
\end{proof}

\section{Option composite image classifier normalized present intermediate classifier determinacy}
\label{sec:option-composite-image-classifier-normalized-present-intermediate-classifier-determinacy}

\begin{theorem}[Tagged option composite image classifier normalized present intermediate classifier determinacy]
\label{thm:tagged-option-composite-image-classifier-normalized-present-intermediate-classifier-determinacy}
This theorem fits the per-object template as follows: the traditional target is
classifier-level readback for the intermediate public endpoints in functorial
$\mathsf{OptionUp}$; the BEDC source is $\mathsf{BHist}$ with two composable
$\mathsf{DescentCertificate}$ maps; the carrier is the selected intermediate
$\Eone$ endpoint shape; the classifier is
$\mathsf{TaggedOptionHistoryClassifier}_{T,\sim_T}$; the stability obligations
are selected visible public factorization, present-payload determinacy,
$\mathsf{TaggedOptionSourceHsameCompatible}(T,\sim_T)$, and the equivalence
fields of $\mathsf{cert}_T$; the ledger records selected intermediate endpoints
and competing visible intermediate payloads; the core theorem is
classifier-level determinacy; exactness says every competing visible
intermediate pair lies in the same $\sim_T$ class; the standard bridge is
some-payload functorial readback; and the status is $\mathsf{ProofTarget}$. Let
the carriers $S,T,U$, classifiers $\sim_S,\sim_T,\sim_U$, descents
$\delta,\epsilon$, preservation fields $\rho_{\delta},\rho_{\epsilon}$,
endpoint-stability field $\eta_{\epsilon}$, certificates
$\mathsf{cert}_S,\mathsf{cert}_T,\mathsf{cert}_U$, compatibility fields,
endpoints $m,m',u,u'$, and selected visible payloads $x,y$ satisfy exactly the
hypotheses of
Theorem~\ref{thm:tagged-option-composite-image-classifier-normalized-selected-visible-public-factorization}.
Then there exist $k,k',p,q,a,b:\mathsf{BHist}$ such that
\begin{enumerate}
\item the ten selected visible public factorization clauses of
Theorem~\ref{thm:tagged-option-composite-image-classifier-normalized-selected-visible-public-factorization}
hold for $k,k',p,q,a,b$;
\item the selected intermediate endpoints are classified by
$$
\mathsf{TaggedOptionHistoryClassifier}_{T,\sim_T}(k,k');
$$
\item every competing visible intermediate pair lies in the selected
intermediate classifier class:
$$
\forall r,s:\mathsf{BHist}.\;
T(r)\to T(s)\to
\hsame(k,\Eone(r))\to \hsame(k',\Eone(s))\to r\sim_T s.
$$
\end{enumerate}
\end{theorem}
\begin{proof}
Apply
Theorem~\ref{thm:tagged-option-composite-image-classifier-normalized-present-intermediate-payload-determinacy}
to the displayed carrier, classifier, descent, preservation, stability,
certificate, compatibility, endpoint, and selected visible payload data. This
supplies $k,k',p,q,a,b$, the ten selected visible public factorization clauses,
and the two present-payload determinacy functions.

The present branch of the ten factorization clauses gives $T(p)$, $T(q)$,
$\hsame(k,\Eone(p))$, $\hsame(k',\Eone(q))$, and $p\sim_T q$. These witnesses
assemble
$$
\mathsf{TaggedOptionHistoryClassifier}_{T,\sim_T}(k,k')
$$
by the present constructor of the tagged option history classifier.

Now fix competing visible intermediate payloads $r,s:\mathsf{BHist}$ with
$T(r)$, $T(s)$, $\hsame(k,\Eone(r))$, and $\hsame(k',\Eone(s))$. The two
determinacy functions yield $\hsame(p,r)$ and $\hsame(q,s)$.

Target hsame compatibility converts these to $p\sim_T r$ and $q\sim_T s$.
Using the symmetry field of $\mathsf{cert}_T$ on $p\sim_T r$ gives
$r\sim_T p$; two applications of the transitivity field of $\mathsf{cert}_T$
compose $r\sim_T p$, $p\sim_T q$, and $q\sim_T s$ into $r\sim_T s$.
\end{proof}
