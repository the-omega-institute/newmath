\chapter{A Concrete Naming Certificate for \(\OptionUp\)}
\label{ch:concrete-instances-option-namecert}

The optional value interface \(\OptionUp\) is licensed over any source carrier already endowed with a \(\NameCert\). The carrier is the inductive sum of an absent constructor and a present constructor wrapping a witness of the source. It is not a primitive type former; it is licensed only via its underlying inductive carrier and the classifier that lifts the source classifier through the present constructor.

\section{Carrier definition}

\begin{definition}[Option carrier]
\label{def:option-carrier}
For a licensed source carrier \(A\), define an inductive carrier with constructors \(\mathsf{none}\) and \(\mathsf{some}(a)\).
\end{definition}
\leanchecked{BEDC.Derived.OptionUp.OptionCarrier\_cases}

\section{The certificate}

\begin{definition}[Option source specification]
\label{def:option-source-specification}
The source is the source NameCert for \(A\), lifted through the optional carrier.
\end{definition}
\leandef{BEDC.Derived.OptionUp.OptionHistoryCarrier}

\begin{theorem}[Option history carrier transport]
\label{thm:option-history-carrier-hsame-transport}
If two BEDC histories are internally the same, then optional history-carrier
membership transfers from the first history to the second.
\end{theorem}
\begin{proof}
Invert the optional carrier witness. In the absent case, compose the symmetric
history-sameness link with the stored empty-history witness. In the present
case, constructor inversion of the history-sameness witness identifies the two
histories, so the source witness is reused.
\end{proof}
\leanchecked{BEDC.Derived.OptionUp.OptionHistoryCarrier\_hsame\_transport}

\begin{definition}[Option pattern specification]
\label{def:option-pattern-specification}
The pattern is absent-or-present generation: one null constructor and one wrapper constructor over source witnesses.
\end{definition}

\begin{definition}[Option classifier specification]
\label{def:option-classifier-specification}
The classifier relates \(\mathsf{none}\) only to \(\mathsf{none}\), and relates \(\mathsf{some}(a)\) to \(\mathsf{some}(a')\) exactly when the source classifier relates \(a\) and \(a'\).
\end{definition}

\begin{theorem}[Option history classifier endpoint transport]
\label{thm:option-history-classifier-hsame-transport}
If the optional history classifier relates \(h\) to \(k\), and
\(\hsame(h,h')\) and \(\hsame(k,k')\), then it relates \(h'\) to \(k'\).
\end{theorem}
\begin{proof}
Unpack the classifier into its endpoint carrier witnesses and the stored
\(\hsame(h,k)\) link. Transport both carriers along the endpoint sameness
links, then compose symmetry of \(\hsame(h,h')\), the stored link, and
\(\hsame(k,k')\).
\end{proof}
\leanchecked{BEDC.Derived.OptionUp.OptionHistoryClassifier\_hsame\_transport}

\begin{theorem}[Option history classifier symmetry]
\label{thm:option-history-classifier-symmetry}
For any source predicate on BEDC histories, the optional history classifier is
symmetric.
\end{theorem}
\begin{proof}
Unpack the classifier into endpoint carrier witnesses and the stored
\(\hsame(h,k)\) link. Swap the endpoint witnesses and reverse the stored link
by symmetry of \(\hsame\).
\end{proof}
\leanchecked{BEDC.Derived.OptionUp.OptionHistoryClassifier\_symm}

\begin{theorem}[Option history classifier transitivity]
\label{thm:option-history-classifier-transitivity}
For any source predicate on BEDC histories, the optional history classifier is
transitive.
\end{theorem}
\begin{proof}
Unpack the two classifier witnesses. The source and target carrier witnesses
are preserved, and the underlying \(\hsame\) links compose across the shared
middle history.
\end{proof}
\leanchecked{BEDC.Derived.OptionUp.OptionHistoryClassifier\_trans}

\begin{theorem}[Option history classifier empty left iff]
\label{thm:option-history-classifier-empty-left-iff}
For any source predicate on BEDC histories, the optional history classifier
relates the empty history on the left to \(k\) exactly when \(k\) is
history-same as the empty history.
\end{theorem}
\begin{proof}
In the forward direction, unpack the classifier and reverse its stored
\(\hsame(\emp,k)\) link. In the reverse direction, both endpoint carrier
witnesses are supplied by the absent-history branch, and the classifier link is
the symmetric form of the assumed \(\hsame(k,\emp)\).
\end{proof}
\leanchecked{BEDC.Derived.OptionUp.OptionHistoryClassifier\_empty\_left\_iff}

\begin{theorem}[Option classifier \(\hsame\) symmetry]
\label{thm:option-classifier-hsame-symm}
For optional BEDC histories, the option classifier induced by \(\hsame\) is symmetric.
\end{theorem}
\begin{proof}
The none--none case returns the existing classifier witness. The mixed constructor cases contradict the classifier definition, and the some--some case reverses the underlying BEDC history witness by \(\hsame\) symmetry.
\end{proof}
\leanchecked{BEDC.Derived.OptionUp.OptionClassifierSpec\_hsame\_symm}

\begin{definition}[Option stability certificate]
\label{def:option-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item reflexivity for none
  \item lifted reflexivity for some
  \item lifted transitivity for some
  \item constructor separation between none and some
  \item closure under the present constructor
\end{enumerate}
\end{definition}

\begin{definition}[Option ledger policy]
\label{def:option-ledger-policy}
The ledger keeps a raw optional-history carrier witness and an $\hsame$ link from the raw history to the visible history.
\end{definition}
\leandef{BEDC.Derived.OptionUp.OptionHistoryLedgerPolicy}

\begin{theorem}[Option visible ledger carrier]
\label{thm:option-ledger-policy-visible-carrier}
If source-history membership transports across $\hsame$, then every option ledger policy exposes a visible optional-history carrier witness.
\end{theorem}
\begin{proof}
Unpack the raw carrier and the raw-to-visible history link. In the absent case, compose the symmetric visible-to-raw link with the raw empty-history witness. In the present case, transport the raw source witness along the raw-to-visible link.
\end{proof}
\leanchecked{BEDC.Derived.OptionUp.OptionHistoryLedgerPolicy\_visible\_carrier}

\begin{theorem}[Option raw-visible ledger classifier]
\label{thm:option-ledger-raw-visible-classifier}
If source-history membership transports across $\hsame$, then every option
ledger policy classifies its raw history with its visible history.
\end{theorem}
\begin{proof}
Unpack the ledger into the raw carrier witness and the raw-to-visible
history-sameness link. The visible carrier is obtained from the visible-carrier
theorem, and the stored link supplies the classifier equality field.
\end{proof}
\leanchecked{BEDC.Derived.OptionUp.OptionHistoryLedgerPolicy\_raw\_visible\_classifier}

\begin{theorem}[Option ledger history-same transport]
\label{thm:option-ledger-hsame-transport}
If both raw histories and both visible histories are related by $\hsame$, then
an option ledger policy transports from the first raw-visible pair to the
second.
\end{theorem}
\begin{proof}
Unpack the ledger into its raw carrier witness and raw-to-visible history link.
The raw carrier transports along the raw $\hsame$ witness. The transported
raw-to-visible link is the composite of the symmetric raw link, the stored
ledger link, and the visible link.
\end{proof}
\leanchecked{BEDC.Derived.OptionUp.OptionHistoryLedgerPolicy\_hsame\_transport}

\begin{theorem}[Concrete \(\OptionUp\) naming certificate]
\label{thm:concrete-option-namecert}
The lifted tuple forms a valid NameCert for \(\OptionUp(A)\).
\end{theorem}
\begin{proof}
The history carrier contains the empty history and any source history. The classifier records both carrier witnesses and the BEDC history equality witness. Reflexivity, symmetry, transitivity, and carrier transport are the corresponding \(\hsame\) fields, while pattern and ledger soundness are carrier projections.
\end{proof}
\leanchecked{BEDC.Derived.OptionUp.option\_history\_semantic\_name\_certificate}

\begin{corollary}[\(\OptionUp\) is licensed, not primitive]
\label{cor:option-licensed-not-primitive}
\(\OptionUp\) is licensed as an inductive wrapper over \(A\), not as a primitive type former.
\end{corollary}
