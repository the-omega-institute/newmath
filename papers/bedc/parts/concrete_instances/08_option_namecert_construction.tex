\begin{theorem}[Tagged option payload descent image classifier mixed visible-tag separation]
\label{thm:tagged-option-payload-descent-image-classifier-mixed-visible-tag-separation}
This theorem fits the per-object template as follows: the traditional target is
image option no-confusion under payload descent; the BEDC source is
$\mathsf{BHist}$ with a $\mathsf{DescentCertificate}$; the carrier is
$\mathsf{TaggedOptionPayloadDescentImageCarrier}$; the classifier is
$\mathsf{TaggedOptionPayloadDescentImageClassifier}$; the stability obligation is
branch exactness plus generated-history no-confusion; the ledger is the visible
absent/present target tag; the core theorem is the two mixed-case
contradictions; exactness permits only absent/absent or present/present target
pairs; the standard bridge is $\mathsf{none}$/$\mathsf{some}$ under payload
descent; and the status is $\mathsf{ProofTarget}$.

Let $S,T:\mathsf{BHist}\to\mathsf{Prop}$ be BEDC history source carriers, let
$\sim_S$ and $\sim_T$ be source classifiers, let
$\delta:\mathsf{DescentCertificate}(\mathsf{BHist},\mathsf{BHist},\sim_S,\sim_T)$,
and assume payload preservation
$$
\rho_{\delta}:\forall a.\;S(a)\to T(\delta.\mathsf{map}(a)).
$$
Let $\mathsf{cert}:\NameCert(S,\sim_S)$, assume source compatibility
$\mathsf{TaggedOptionSourceHsameCompatible}(S,\sim_S)$, let
$k,k':\mathsf{BHist}$ be target endpoints, and let $u,v:\mathsf{BHist}$ be
target payloads. Both mixed visible-tag clauses hold:
$$
\mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
(S,T,\rho_{\delta};k,k')\to
\hsame(k,\emp)\to T(v)\to \hsame(k',\Eone(v))\to\bot
$$
and
$$
\mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
(S,T,\rho_{\delta};k,k')\to
T(u)\to \hsame(k,\Eone(u))\to \hsame(k',\emp)\to\bot.
$$
\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:tagged-option-payload-descent-image-classifier-branch-exactness}
to the image-classifier witness. Branch exactness leaves only the absent/absent
target pair or a transported present/present target pair.

For the clause with $\hsame(k,\emp)$ and $\hsame(k',\Eone(v))$, the
absent/absent branch contradicts the displayed present endpoint on the right by
empty-versus-$\Eone$ no-confusion. The present/present branch contradicts the
displayed absent endpoint on the left by the same no-confusion principle.

For the clause with $\hsame(k,\Eone(u))$ and $\hsame(k',\emp)$, the
absent/absent branch contradicts the displayed present endpoint on the left.
The present/present branch contradicts the displayed absent endpoint on the
right. These two contradictions exclude all cases supplied by branch exactness.
\end{proof}
\begin{theorem}[Tagged option payload descent image classifier nonempty present-pair inversion]
\label{thm:tagged-option-payload-descent-image-classifier-nonempty-present-pair-inversion}
This theorem fits the per-object template as follows: the traditional target is
nonempty option image branch inversion; the BEDC source is $\mathsf{BHist}$ with
a $\mathsf{DescentCertificate}$; the carrier is
$\mathsf{TaggedOptionPayloadDescentImageCarrier}$; the classifier is
$\mathsf{TaggedOptionPayloadDescentImageClassifier}$; the stability obligation is
branch exactness plus endpoint non-absence; the ledger is the synchronized
transported payload pair; the core theorem is present-pair recovery; exactness is
elimination of the absent pair; the standard bridge is $\mathsf{none}$/$\mathsf{some}$
under payload descent; and the status is $\mathsf{ProofTarget}$.

Let $S,T:\mathsf{BHist}\to\mathsf{Prop}$ be BEDC history source carriers, let
$\sim_S$ and $\sim_T$ be source classifiers, let
$\delta:\mathsf{DescentCertificate}(\mathsf{BHist},\mathsf{BHist},\sim_S,\sim_T)$,
and assume payload preservation
$$
\rho_{\delta}:\forall a.\;S(a)\to T(\delta.\mathsf{map}(a)).
$$
Let $\mathsf{cert}:\NameCert(S,\sim_S)$ and assume
$\mathsf{TaggedOptionSourceHsameCompatible}(S,\sim_S)$. For target endpoints
$k,k':\mathsf{BHist}$, endpoint non-absence on either side forces the
synchronized transported present pair. The left non-absence clause is
$$
\mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
(S,T,\rho_{\delta};k,k')
\to
\neg\hsame(k,\emp)
\to
\exists a,b:\mathsf{BHist}.\;
S(a)\land S(b)\land a\sim_S b\land
T(\delta.\mathsf{map}(a))\land T(\delta.\mathsf{map}(b))\land
\hsame(k,\Eone(\delta.\mathsf{map}(a)))\land
\hsame(k',\Eone(\delta.\mathsf{map}(b))).
$$
The right non-absence clause is
$$
\mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
(S,T,\rho_{\delta};k,k')
\to
\neg\hsame(k',\emp)
\to
\exists a,b:\mathsf{BHist}.\;
S(a)\land S(b)\land a\sim_S b\land
T(\delta.\mathsf{map}(a))\land T(\delta.\mathsf{map}(b))\land
\hsame(k,\Eone(\delta.\mathsf{map}(a)))\land
\hsame(k',\Eone(\delta.\mathsf{map}(b))).
$$
\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:tagged-option-payload-descent-image-classifier-branch-exactness}
to the image-classifier witness. Branch exactness supplies either the
absent/absent target pair or an exposed transported present/present target pair.

For the left non-absence clause, the absent/absent branch contains
$\hsame(k,\emp)$ and therefore contradicts the premise $\neg\hsame(k,\emp)$.
The present/present branch already contains payloads $a,b$ with $S(a)$, $S(b)$,
$a\sim_S b$, the two transported target-carrier witnesses, and the displayed
present endpoint presentations. Return those witnesses.

For the right non-absence clause, the same branch exactness split applies. The
absent/absent branch contains $\hsame(k',\emp)$ and contradicts the premise
$\neg\hsame(k',\emp)$, while the present/present branch is exactly the required
synchronized transported payload pair.
\end{proof}

\begin{theorem}[Tagged option payload descent image classifier reflective transitive visible payload pair coherence]
\label{thm:tagged-option-payload-descent-image-classifier-reflective-transitive-visible-pair-coherence}
This theorem fits the per-object template as follows: the traditional target is
reflective transitive visible readback coherence for constructor-tagged
$\mathsf{OptionUp}$ under payload descent; the BEDC source is $\mathsf{BHist}$
with a $\mathsf{DescentCertificate}$; the carrier is
$\mathsf{TaggedOptionPayloadDescentImageCarrier}$; the classifier is
$\mathsf{TaggedOptionPayloadDescentImageClassifier}$; the stability obligation
is reflective transitivity plus visible endpoint determinacy; the ledger is the
two displayed target payload pairs; the core theorem is the shared-source-pair
coherence below; exactness says a transitive visible image-classifier path has a
single source payload pair for both visible presentations; the standard bridge
is $\mathsf{none}$/$\mathsf{some}$ payload readback; and the status is
$\mathsf{ProofTarget}$.

Let $S,T:\mathsf{BHist}\to\mathsf{Prop}$ be BEDC history source carriers, let
$\sim_S$ and $\sim_T$ be source classifiers, let
$\delta:\mathsf{DescentCertificate}(\mathsf{BHist},\mathsf{BHist},\sim_S,\sim_T)$,
and assume payload preservation
$$
\rho_{\delta}:\forall a.\;S(a)\to T(\delta.\mathsf{map}(a)).
$$
Let $\mathsf{cert}:\NameCert(S,\sim_S)$, assume
$\mathsf{TaggedOptionSourceHsameCompatible}(S,\sim_S)$, and assume
$$
\mathsf{TaggedOptionPayloadDescentReflectsSource}_{\delta}(S,\sim_S).
$$
For target endpoints $k,\ell,m:\mathsf{BHist}$ and visible target payloads
$u,v,u',v':\mathsf{BHist}$, if
$$
\mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
(S,T,\rho_{\delta};k,\ell)
$$
and
$$
\mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
(S,T,\rho_{\delta};\ell,m)
$$
and
$$
T(u)\land T(v)\land
\hsame(k,\Eone(u))\land
\hsame(m,\Eone(v))
$$
and
$$
T(u')\land T(v')\land
\hsame(k,\Eone(u'))\land
\hsame(m,\Eone(v')),
$$
then
$$
\exists a,b:\mathsf{BHist}.\;
S(a)\land S(b)\land a\sim_S b\land
T(\delta.\mathsf{map}(a))\land T(\delta.\mathsf{map}(b))\land
\hsame(u,\delta.\mathsf{map}(a))\land
\hsame(v,\delta.\mathsf{map}(b))\land
\hsame(u',\delta.\mathsf{map}(a))\land
\hsame(v',\delta.\mathsf{map}(b)).
$$
\end{theorem}
\begin{proof}
First combine the two image-classifier witnesses. Apply
Theorem~\ref{thm:tagged-option-payload-descent-image-classifier-reflective-transitivity}
to the witnesses at $(k,\ell)$ and $(\ell,m)$, using the source hsame
compatibility and source-reflection hypotheses. This gives the direct
image-classifier witness from $k$ to $m$.

Next apply
Theorem~\ref{thm:tagged-option-payload-descent-image-classifier-visible-payload-readback}
to the direct witness and the first visible target payload pair. The result is a
source pair $a,b$ with $S(a)$, $S(b)$, $a\sim_S b$, transported target carrier
facts, and the readback identifications
$\hsame(u,\delta.\mathsf{map}(a))$ and
$\hsame(v,\delta.\mathsf{map}(b))$.

It remains to compare the primed visible presentation with the unprimed one.
At the left endpoint $k$, the two witnesses present $k$ as both $\Eone(u)$ and
$\Eone(u')$; $\Eone$ no-confusion, symmetry of $\hsame$, and transitivity of
$\hsame$ give $\hsame(u',\delta.\mathsf{map}(a))$. At the right endpoint $m$,
the same argument applied to $\Eone(v)$ and $\Eone(v')$ gives
$\hsame(v',\delta.\mathsf{map}(b))$. Repack the original source pair together
with these four payload identifications.
\end{proof}
