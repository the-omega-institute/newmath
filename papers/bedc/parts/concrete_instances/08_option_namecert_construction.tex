\section{Option history classifier unary two-sided context}
\label{sec:option-history-classifier-unary-two-sided-context}

\begin{theorem}[Option history carrier unary two-sided context]
\label{thm:option-history-carrier-unary-two-sided-context}
Let $p,q,h:\mathsf{BHist}$. If $\mathsf{UnaryHistory}(p)$ and
$\mathsf{UnaryHistory}(q)$ hold, then
$$
\mathsf{OptionHistoryCarrier}_{\mathsf{UnaryHistory}}(h)
\Longleftrightarrow
\mathsf{OptionHistoryCarrier}_{\mathsf{UnaryHistory}}
  (\mathsf{append}(p,\mathsf{append}(h,q))).
$$
\end{theorem}
\begin{proof}
For the forward direction, first apply
Theorem~\ref{thm:option-history-carrier-unary-suffix-closed} with suffix $q$.
Then apply Theorem~\ref{thm:option-history-carrier-unary-prefix-closed} with
prefix $p$.

For the reverse direction, split the contextual nullable carrier witness.

In the absent branch, use $\mathsf{hsame}\_\mathsf{iff}\_\mathsf{eq}$ to turn
the empty contextual endpoint witness into equality. Apply
$\mathsf{append}\_\mathsf{eq}\_\mathsf{empty}\_\mathsf{iff}$ to the outer
append, and then again to the inner append, recovering $\hsame(h,\emp)$.

In the source branch, use
$\mathsf{unary}\_\mathsf{append}\_\mathsf{right}\_\mathsf{factor}$ on the
outer append to recover $\mathsf{UnaryHistory}(\mathsf{append}(h,q))$. Then
use $\mathsf{unary}\_\mathsf{append}\_\mathsf{left}\_\mathsf{factor}$ on the
inner append to recover $\mathsf{UnaryHistory}(h)$.

These two recovered branches are exactly the nullable carrier alternatives for
$h$.
\end{proof}

\begin{theorem}[Option history classifier unary two-sided context]
\label{thm:option-history-classifier-unary-two-sided-context}
This theorem fits the ten-field template as follows: the traditional target is
context-stable nullable optional history classification; the BEDC source is
$\mathsf{BHist}$ with $\mathsf{UnaryHistory}$, $\mathsf{append}$, and
$\hsame$; the carrier is
$\mathsf{OptionHistoryCarrier}_{\mathsf{UnaryHistory}}$; the classifier is
$\mathsf{OptionHistoryClassifier}_{\mathsf{UnaryHistory}}$; the stability
obligations are unary prefix and suffix closure together with cancellation;
the ledger is the hidden common unary context; the core theorem is the
displayed equivalence; exactness is recovery of the uncloaked classifier by
append cancellation and unary factor readback; the standard bridge is option
values under common unary context; and the status is $\mathsf{ProofTarget}$.

Let $p,q,h,k:\mathsf{BHist}$. If $\mathsf{UnaryHistory}(p)$ and
$\mathsf{UnaryHistory}(q)$ hold, then
$$
\mathsf{OptionHistoryClassifier}_{\mathsf{UnaryHistory}}(h,k)
\Longleftrightarrow
\mathsf{OptionHistoryClassifier}_{\mathsf{UnaryHistory}}
  (\mathsf{append}(p,\mathsf{append}(h,q)),
   \mathsf{append}(p,\mathsf{append}(k,q))).
$$
\end{theorem}
\begin{proof}
For the forward direction, first apply
Theorem~\ref{thm:option-history-classifier-unary-suffix-closed} with suffix
$q$ to place the common unary right context on both endpoints. Then apply
Theorem~\ref{thm:option-history-classifier-unary-prefix-closed} with prefix
$p$ to place the common unary left context.

For the reverse direction, unpack the contextual nullable classifier into its
two endpoint carrier witnesses and the contextual $\hsame$ field. Each
endpoint carrier is read back from the surrounding context. In an absent
branch, use $\hsame\_\mathsf{iff}\_\mathsf{eq}$ and
$\mathsf{append}\_\mathsf{eq}\_\mathsf{empty}\_\mathsf{iff}$ twice to recover
the corresponding uncloaked empty endpoint witness.

In a source branch, use
$\mathsf{unary}\_\mathsf{append}\_\mathsf{right}\_\mathsf{factor}$ on the
outer append and
$\mathsf{unary}\_\mathsf{append}\_\mathsf{left}\_\mathsf{factor}$ on the inner
append to recover $\mathsf{UnaryHistory}(h)$ or
$\mathsf{UnaryHistory}(k)$ as required by the nullable carrier.

It remains to recover the uncloaked classifier relation. Convert the
contextual $\hsame$ field to equality, cancel the common left prefix by
$\mathsf{append}\_\mathsf{left}\_\mathsf{cancel}$, and then cancel the common
right suffix by $\mathsf{append}\_\mathsf{right}\_\mathsf{cancel}$. Repack the
two recovered endpoint carriers together with the recovered $\hsame(h,k)$.
\end{proof}
\leanchecked{BEDC.Derived.OptionUp.OptionHistoryClassifier\_unary\_two\_sided\_context\_iff}

\section{Tagged option map relation R-corner classifier coherence}
\label{sec:option-maprel-r-corner-classifier-coherence}

\begin{theorem}[Tagged option map relation R-corner classifier coherence]
\label{thm:tagged-option-maprel-r-corner-classifier-coherence}
This theorem fits the per-object template as follows: the traditional target is
right-corner option-map endpoint determinacy; the BEDC source is
$\mathsf{BHist}$ with one $\mathsf{DescentCertificate}$; the carrier is the
tagged option endpoint shape; the classifiers are
$\mathsf{TaggedOptionHistoryClassifier}_{S,\sim_S}$ and
$\mathsf{TaggedOptionHistoryClassifier}_{T,\sim_T}$; the stability obligations
are shared-target source reflection and shared-source target normal form; the
ledger records the three R-corner legs; the core theorem is simultaneous source
and target endpoint classification; exactness is absent-or-present exhaustion on
both axes; the standard bridge is $\mathsf{none}$/$\mathsf{some}$ map readback;
and the status is $\mathsf{ProofTarget}$.

Let $S,T:\mathsf{BHist}\to\mathsf{Prop}$ be BEDC history source and target
carriers, let $\sim_S$ and $\sim_T$ be their classifiers, let
$$
\delta:\mathsf{DescentCertificate}(\mathsf{BHist},\mathsf{BHist},\sim_S,\sim_T),
$$
and assume payload preservation
$$
\rho_{\delta}:\forall a.\;S(a)\to T(\delta.\mathsf{map}(a)).
$$
Assume $\mathsf{TaggedOptionSourceHsameCompatible}(S,\sim_S)$ and reflected
source readback from visible target equality,
$$
\mathsf{TaggedOptionPayloadDescentReflectsSource}_{\delta}(S,\sim_S).
$$
For histories $h,h',k,k'$, if
$$
\mathsf{TaggedOptionMapRel}_{\delta}(S,T,\rho_{\delta};h,k),
$$
$$
\mathsf{TaggedOptionMapRel}_{\delta}(S,T,\rho_{\delta};h',k),
$$
and
$$
\mathsf{TaggedOptionMapRel}_{\delta}(S,T,\rho_{\delta};h',k'),
$$
then
$$
\mathsf{TaggedOptionHistoryClassifier}_{S,\sim_S}(h,h')
\land
\mathsf{TaggedOptionHistoryClassifier}_{T,\sim_T}(k,k').
$$
\end{theorem}
\begin{proof}
First compare the two legs with shared target endpoint $k$. Apply
Theorem~\ref{thm:tagged-option-maprel-common-target-source-classification} to
the map-relation witnesses
$\mathsf{TaggedOptionMapRel}_{\delta}(S,T,\rho_{\delta};h,k)$ and
$\mathsf{TaggedOptionMapRel}_{\delta}(S,T,\rho_{\delta};h',k)$, using the
reflected source readback hypothesis. This gives
$\mathsf{TaggedOptionHistoryClassifier}_{S,\sim_S}(h,h')$.

Next compare the two legs with shared source endpoint $h'$. Apply
Theorem~\ref{thm:tagged-option-maprel-common-source-target-classification} to
the map-relation witnesses
$\mathsf{TaggedOptionMapRel}_{\delta}(S,T,\rho_{\delta};h',k)$ and
$\mathsf{TaggedOptionMapRel}_{\delta}(S,T,\rho_{\delta};h',k')$, using source
hsame compatibility. This gives
$\mathsf{TaggedOptionHistoryClassifier}_{T,\sim_T}(k,k')$.

Pairing these two classifier conclusions yields the displayed conjunction.
\end{proof}

\begin{theorem}[Tagged option common-right visible payload classification]
\label{thm:tagged-option-history-classifier-common-right-visible-payload-classification}
Let $S:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history carrier and let
$R:\mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$ be a symmetric and
transitive payload classifier. Assume that visible $\hsame$ between source
payloads is reflected into $R$. For histories $h,r,k,a,c,b$, if
$$
\mathsf{TaggedOptionHistoryClassifier}_{S,R}(h,k)
\quad\hbox{and}\quad
\mathsf{TaggedOptionHistoryClassifier}_{S,R}(r,k),
$$
and $a,c,b$ are displayed source payloads with
$$
\hsame(h,\Eone(a)),\qquad
\hsame(r,\Eone(c)),\qquad
\hsame(k,\Eone(b)),
$$
then $R(a,c)$.
\end{theorem}
\begin{proof}
Use right-visible branch inversion on both classifiers at the common right
presentation $\hsame(k,\Eone(b))$. The first classifier exposes a left payload
classified with a right payload history-same to $b$, while the second exposes a
right payload chain with the same visible $b$.

Invert the two visible $\Eone$ presentations at $h$ and $r$ to compare the
displayed payloads $a$ and $c$ with the exposed classifier payloads. The
source-$\hsame$ reflection converts these payload comparisons into $R$-legs.
Transitivity composes $a$ to the common visible payload $b$ through the first
classifier. The second classifier composes $c$ to the same $b$; symmetry turns
that leg around, and one final transitivity step yields $R(a,c)$.
\end{proof}
\leanchecked{BEDC.Derived.OptionUp.TaggedOptionHistoryClassifier\_common\_right\_visible\_payload\_classification}

\section{Tagged option payload descent image target classifier closure}

\begin{theorem}[Tagged option payload descent image classifier target classifier closure]
\label{thm:tagged-option-payload-descent-image-classifier-target-classifier-closure}
This theorem fits the per-object template as follows: the traditional target is
functorial $\mathsf{OptionUp}$ target classifier closure; the BEDC source is
$\mathsf{BHist}$ with a descent certificate; the carrier is the transported
target tagged-option carrier; the classifier is
$\mathsf{TaggedOptionHistoryClassifier}_{T,\sim_T}$; the stability obligation is
branch exactness plus descent respect; the ledger is absent or transported
present payload provenance; the core theorem is the displayed classifier
implication; exactness is the branch split; the standard bridge is
$\mathsf{none}$/$\mathsf{some}$ classifier closure; and the status is
$\mathsf{ProofTarget}$.

Let $S,T:\mathsf{BHist}\to\mathsf{Prop}$ be BEDC history carriers, let
$\sim_S$ and $\sim_T$ be classifiers, let
$\delta:\mathsf{DescentCertificate}(\mathsf{BHist},\mathsf{BHist},\sim_S,\sim_T)$,
and assume payload preservation
$$
\rho_{\delta}:\forall a.\;S(a)\to T(\delta.\mathsf{map}(a)).
$$
Let $\mathsf{cert}:\NameCert(S,\sim_S)$ and assume
$\mathsf{TaggedOptionSourceHsameCompatible}(S,\sim_S)$. For target endpoints
$k,k':\mathsf{BHist}$,
$$
\mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
(S,T,\rho_{\delta};k,k')
\to
\mathsf{TaggedOptionHistoryClassifier}_{T,\sim_T}(k,k').
$$
\end{theorem}
\begin{proof}
Apply
Theorem~\ref{thm:tagged-option-payload-descent-image-classifier-branch-exactness}
to the displayed carrier, classifier, descent, preservation, certificate,
compatibility, and endpoint data. The resulting branch split is either
absent-absent target presentation or transported present-payload presentation.

In the absent branch, the split supplies $\hsame(k,\emp)$ and
$\hsame(k',\emp)$. Use the absent-absent constructor of
$\mathsf{TaggedOptionHistoryClassifier}_{T,\sim_T}$.

In the present branch, keep the witnesses $a,b:\mathsf{BHist}$, the source
carrier witnesses, $a\sim_S b$, the two transported target carrier witnesses,
and the two target endpoint presentations. The descent certificate's respect
field gives
$$
\delta.\mathsf{respects}(a\sim_S b):
\delta.\mathsf{map}(a)\sim_T\delta.\mathsf{map}(b).
$$

Use the present-present constructor of
$\mathsf{TaggedOptionHistoryClassifier}_{T,\sim_T}$ with target witnesses
$\delta.\mathsf{map}(a)$ and $\delta.\mathsf{map}(b)$, their two $T$ witnesses,
the transported classifier witness, and the two endpoint presentations.
\end{proof}

$$

Now fix competing visible intermediate payloads $r,s:\mathsf{BHist}$ with
$T(r)$, $T(s)$, $\hsame(k,\Eone(r))$, and $\hsame(k',\Eone(s))$. The two

$r\sim_T p$; two applications of the transitivity field of $\mathsf{cert}_T$
compose $r\sim_T p$, $p\sim_T q$, and $q\sim_T s$ into $r\sim_T s$.
\end{proof}

\begin{theorem}[Tagged option composite image classifier normalized intermediate factor coherence]
\label{thm:tagged-option-composite-image-classifier-normalized-intermediate-factor-coherence}
Let $T,U:\mathsf{BHist}\to\mathsf{Prop}$ be BEDC history carriers, let
$\sim_T$ and $\sim_U$ be their classifiers, let
$$
\epsilon:\mathsf{DescentCertificate}(\mathsf{BHist},\mathsf{BHist},\sim_T,\sim_U),
$$
and assume payload preservation
$$
\rho_{\epsilon}:\forall a.\;T(a)\to U(\epsilon.\mathsf{map}(a)).
$$
Assume reflected source readback from visible public equality,
$$
\mathsf{TaggedOptionPayloadDescentReflectsSource}_{\epsilon}(T,\sim_T).
$$
For histories $k,l,k',l',u,u'$, if
$$
\mathsf{TaggedOptionMapRel}_{\epsilon}(T,U,\rho_{\epsilon};k,u),
$$
$$
\mathsf{TaggedOptionMapRel}_{\epsilon}(T,U,\rho_{\epsilon};l,u),
$$
$$
\mathsf{TaggedOptionMapRel}_{\epsilon}(T,U,\rho_{\epsilon};k',u'),
$$
and
$$
\mathsf{TaggedOptionMapRel}_{\epsilon}(T,U,\rho_{\epsilon};l',u'),
$$
then
$$
\mathsf{TaggedOptionHistoryClassifier}_{T,\sim_T}(k,l)
\land
\mathsf{TaggedOptionHistoryClassifier}_{T,\sim_T}(k',l').
$$
\end{theorem}
\begin{proof}
First compare the two intermediate factors over the normalized public endpoint
$u$. Apply
Theorem~\ref{thm:tagged-option-maprel-shared-target-source-reflection-normal-form}
to the two map-relation witnesses
$\mathsf{TaggedOptionMapRel}_{\epsilon}(T,U,\rho_{\epsilon};k,u)$ and
$\mathsf{TaggedOptionMapRel}_{\epsilon}(T,U,\rho_{\epsilon};l,u)$. This gives
either simultaneous empty presentations of $k,l,u$ or a present-payload package
ending in $\mathsf{TaggedOptionHistoryClassifier}_{T,\sim_T}(k,l)$.

Next compare the two intermediate factors over the normalized public endpoint
$u'$. Apply
Theorem~\ref{thm:tagged-option-maprel-shared-target-source-reflection-normal-form}
to the two map-relation witnesses
$\mathsf{TaggedOptionMapRel}_{\epsilon}(T,U,\rho_{\epsilon};k',u')$ and
$\mathsf{TaggedOptionMapRel}_{\epsilon}(T,U,\rho_{\epsilon};l',u')$. This gives
either simultaneous empty presentations of $k',l',u'$ or a present-payload
package ending in $\mathsf{TaggedOptionHistoryClassifier}_{T,\sim_T}(k',l')$.

In an absent branch of the comparison over $u$, the empty presentations of $k$
and $l$ assemble the absent branch of
$\mathsf{TaggedOptionHistoryClassifier}_{T,\sim_T}(k,l)$. In its present branch,
retain the classifier supplied by the reflected normal form.

In an absent branch of the comparison over $u'$, the empty presentations of $k'$
and $l'$ assemble the absent branch of
$\mathsf{TaggedOptionHistoryClassifier}_{T,\sim_T}(k',l')$. In its present
branch, retain the classifier supplied by the reflected normal form.

Combining the classifier obtained over $u$ with the classifier obtained over
$u'$ yields the displayed conjunction.
\end{proof}
