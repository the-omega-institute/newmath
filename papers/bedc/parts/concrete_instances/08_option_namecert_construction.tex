\begin{theorem}[Tagged option map relation R-corner classifier coherence]
\label{thm:tagged-option-maprel-r-corner-classifier-coherence}
This theorem fits the per-object template as follows: the traditional target is
right-corner option-map endpoint determinacy; the BEDC source is
$\mathsf{BHist}$ with one $\mathsf{DescentCertificate}$; the carrier is the
tagged option endpoint shape; the classifiers are
$\mathsf{TaggedOptionHistoryClassifier}_{S,\sim_S}$ and
$\mathsf{TaggedOptionHistoryClassifier}_{T,\sim_T}$; the stability obligations
are shared-target source reflection and shared-source target normal form; the
ledger records the three R-corner legs; the core theorem is simultaneous source
and target endpoint classification; exactness is absent-or-present exhaustion on
both axes; the standard bridge is $\mathsf{none}$/$\mathsf{some}$ map readback;
and the status is $\mathsf{ProofTarget}$.

Let $S,T:\mathsf{BHist}\to\mathsf{Prop}$ be BEDC history source and target
carriers, let $\sim_S$ and $\sim_T$ be their classifiers, let
$$
\delta:\mathsf{DescentCertificate}(\mathsf{BHist},\mathsf{BHist},\sim_S,\sim_T),
$$
and assume payload preservation
$$
\rho_{\delta}:\forall a.\;S(a)\to T(\delta.\mathsf{map}(a)).
$$
Assume $\mathsf{TaggedOptionSourceHsameCompatible}(S,\sim_S)$ and reflected
source readback from visible target equality,
$$
\mathsf{TaggedOptionPayloadDescentReflectsSource}_{\delta}(S,\sim_S).
$$
For histories $h,h',k,k'$, if
$$
\mathsf{TaggedOptionMapRel}_{\delta}(S,T,\rho_{\delta};h,k),
$$
$$
\mathsf{TaggedOptionMapRel}_{\delta}(S,T,\rho_{\delta};h',k),
$$
and
$$
\mathsf{TaggedOptionMapRel}_{\delta}(S,T,\rho_{\delta};h',k'),
$$
then
$$
\mathsf{TaggedOptionHistoryClassifier}_{S,\sim_S}(h,h')
\land
\mathsf{TaggedOptionHistoryClassifier}_{T,\sim_T}(k,k').
$$
\end{theorem}
\begin{proof}
First compare the two legs with shared target endpoint $k$. Apply
Theorem~\ref{thm:tagged-option-maprel-common-target-source-classification} to
the map-relation witnesses
$\mathsf{TaggedOptionMapRel}_{\delta}(S,T,\rho_{\delta};h,k)$ and
$\mathsf{TaggedOptionMapRel}_{\delta}(S,T,\rho_{\delta};h',k)$, using the
reflected source readback hypothesis. This gives
$\mathsf{TaggedOptionHistoryClassifier}_{S,\sim_S}(h,h')$.

Next compare the two legs with shared source endpoint $h'$. Apply
Theorem~\ref{thm:tagged-option-maprel-common-source-target-classification} to
the map-relation witnesses
$\mathsf{TaggedOptionMapRel}_{\delta}(S,T,\rho_{\delta};h',k)$ and
$\mathsf{TaggedOptionMapRel}_{\delta}(S,T,\rho_{\delta};h',k')$, using source
hsame compatibility. This gives
$\mathsf{TaggedOptionHistoryClassifier}_{T,\sim_T}(k,k')$.

Pairing these two classifier conclusions yields the displayed conjunction.
\end{proof}

\begin{theorem}[Tagged option common-right visible payload classification]
\label{thm:tagged-option-history-classifier-common-right-visible-payload-classification}
Let $S:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history carrier and let
$R:\mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$ be a symmetric and
transitive payload classifier. Assume that visible $\hsame$ between source
payloads is reflected into $R$. For histories $h,r,k,a,c,b$, if
$$
\mathsf{TaggedOptionHistoryClassifier}_{S,R}(h,k)
\quad\hbox{and}\quad
\mathsf{TaggedOptionHistoryClassifier}_{S,R}(r,k),
$$
and $a,c,b$ are displayed source payloads with
$$
\hsame(h,\Eone(a)),\qquad
\hsame(r,\Eone(c)),\qquad
\hsame(k,\Eone(b)),
$$
then $R(a,c)$.
\end{theorem}
\begin{proof}
Use right-visible branch inversion on both classifiers at the common right
presentation $\hsame(k,\Eone(b))$. The first classifier exposes a left payload
classified with a right payload history-same to $b$, while the second exposes a
right payload chain with the same visible $b$.

Invert the two visible $\Eone$ presentations at $h$ and $r$ to compare the
displayed payloads $a$ and $c$ with the exposed classifier payloads. The
source-$\hsame$ reflection converts these payload comparisons into $R$-legs.
Transitivity composes $a$ to the common visible payload $b$ through the first
classifier. The second classifier composes $c$ to the same $b$; symmetry turns
that leg around, and one final transitivity step yields $R(a,c)$.
\end{proof}

\begin{theorem}[Tagged option present-chain visible endpoint classification]
\label{thm:tagged-option-present-chain-visible-endpoint-classification}
Let $S:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history carrier and let
$R:\mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$ be a transitive payload
classifier. Assume that visible $\hsame$ between source payloads is reflected
into $R$. For histories $h,k,r,a,c$, if
$$
\mathsf{TaggedOptionHistoryClassifier}_{S,R}(h,k)
\quad\hbox{and}\quad
\mathsf{TaggedOptionHistoryClassifier}_{S,R}(k,r),
$$
if $S(a)$ and $S(c)$ hold, and if the endpoints are visibly present as
$$
\hsame(h,\Eone(a)),\qquad \hsame(r,\Eone(c)),
$$
then $R(a,c)$.
\end{theorem}
\begin{proof}
Compose the two tagged classifiers by transitivity of $R$ to classify $h$ and
$r$. The absent branch would present $h$ as both $\emp$ and $\Eone(a)$, which is
impossible by BEDC constructor no-confusion. In the present branch, invert the
two visible $\Eone$ presentations to compare $a$ and $c$ with the exposed
payload witnesses. The source-$\hsame$ reflection supplies the two outer
$R$-legs, and transitivity composes them with the exposed payload classifier.
\end{proof}
