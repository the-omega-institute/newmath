\begin{theorem}[Tagged option payload descent image classifier reflective transitive normalized image closure]
\label{thm:tagged-option-payload-descent-image-classifier-reflective-transitive-normalized-image-closure}
Let $S,T:\mathsf{BHist}\to\mathsf{Prop}$ be BEDC history source and target
carriers, let $\sim_S$ and $\sim_T$ be source and target classifiers, let
$\delta:\mathsf{DescentCertificate}(\mathsf{BHist},\mathsf{BHist},\sim_S,\sim_T)$,
and assume payload preservation
$$
\rho_{\delta}:\forall a.\;S(a)\to T(\delta.\mathsf{map}(a)).
$$
Let $\mathsf{cert}:\NameCert(S,\sim_S)$, assume source compatibility
$\mathsf{TaggedOptionSourceHsameCompatible}(S,\sim_S)$, and assume
$$
\mathsf{TaggedOptionPayloadDescentReflectsSource}_{\delta}(S,\sim_S).
$$
For target endpoints $k,\ell,m,p,q:\mathsf{BHist}$, if
$$
\mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
(S,T,\rho_{\delta};k,\ell)
$$
and
$$
\mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
(S,T,\rho_{\delta};\ell,m)
$$
and
$$
\hsame(k,p)\land \hsame(m,q),
$$
then
$$
\mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
(S,T,\rho_{\delta};p,q).
$$
\end{theorem}
\begin{proof}
First compose the two image-classifier witnesses. Apply
Theorem~\ref{thm:tagged-option-payload-descent-image-classifier-reflective-transitivity}
to the witnesses at $(k,\ell)$ and $(\ell,m)$, using source compatibility and
source reflection. This gives the direct classifier from $k$ to $m$.

Next normalize the public endpoints. Apply
Theorem~\ref{thm:tagged-option-payload-descent-image-classifier-hsame-transport}
with $\hsame(k,p)$ and $\hsame(m,q)$. The result is the displayed classifier at
the normalized endpoint pair $(p,q)$.
\end{proof}

$\mathsf{TaggedOptionPayloadDescentImageCarrier}$; the classifier is

classifier exposure, and paired branch readback; the ledger is the normalized
absent pair or transported present pair; the core theorem is the displayed
public endpoint conjunction and exclusive branch split; exactness is normalized
public endpoint readback; the standard bridge is $\mathsf{none}$/$\mathsf{some}$
readback; and the status is $\mathsf{ProofTarget}$.

Let $S,T:\mathsf{BHist}\to\mathsf{Prop}$ be BEDC history source and target
carriers, let $\sim_S$ and $\sim_T$ be source and target classifiers, let
$\delta:\mathsf{DescentCertificate}(\mathsf{BHist},\mathsf{BHist},\sim_S,\sim_T)$,
and assume payload preservation
$$
\rho_{\delta}:\forall a.\;S(a)\to T(\delta.\mathsf{map}(a)).
$$
Let $\mathsf{cert}:\NameCert(S,\sim_S)$, assume source compatibility
$\mathsf{TaggedOptionSourceHsameCompatible}(S,\sim_S)$, and assume
$$
\mathsf{TaggedOptionPayloadDescentReflectsSource}_{\delta}(S,\sim_S).
$$
For target endpoints $k,\ell,m,p,q:\mathsf{BHist}$, if
$$
\mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
(S,T,\rho_{\delta};k,\ell)
$$
and
$$
\mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
(S,T,\rho_{\delta};\ell,m)
$$
and
$$
\hsame(k,p)\land \hsame(m,q),
$$
then
$$
\mathsf{TaggedOptionPayloadDescentImageCarrier}_{\delta}
(S,T,\rho_{\delta};p)
\land
\mathsf{TaggedOptionPayloadDescentImageCarrier}_{\delta}
(S,T,\rho_{\delta};q)
\land

$$
and either

or
$$
\exists a,b:\mathsf{BHist}.\;
S(a)\land S(b)\land a\sim_S b\land
T(\delta.\mathsf{map}(a))\land T(\delta.\mathsf{map}(b))\land
\hsame(p,\Eone(\delta.\mathsf{map}(a)))\land
\hsame(q,\Eone(\delta.\mathsf{map}(b)))\land
\neg\hsame(p,\emp)\land \neg\hsame(q,\emp)\land
\bigl(\forall a',b':\mathsf{BHist}.\;
S(a')\to S(b')\to a'\sim_S b'\to
T(\delta.\mathsf{map}(a'))\to T(\delta.\mathsf{map}(b'))\to
\hsame(p,\Eone(\delta.\mathsf{map}(a')))\to
\hsame(q,\Eone(\delta.\mathsf{map}(b')))\to
\hsame(\delta.\mathsf{map}(a),\delta.\mathsf{map}(a'))\land
\hsame(\delta.\mathsf{map}(b),\delta.\mathsf{map}(b'))\bigr).
$$
\end{theorem}
\begin{proof}
First obtain normalized image closure. Apply
Theorem~\ref{thm:tagged-option-payload-descent-image-classifier-reflective-transitive-normalized-image-closure}
to the two image-classifier witnesses and the endpoint equalities. This gives
the direct classifier at $(p,q)$.

Apply
Theorem~\ref{thm:tagged-option-payload-descent-image-classifier-certificate}
to the normalized classifier. This exposes the two normalized image carriers at
$p$ and $q$ and the target tagged classifier
$\mathsf{TaggedOptionHistoryClassifier}_{T,\sim_T}(p,q)$.

Finally apply
Theorem~\ref{thm:tagged-option-payload-descent-image-classifier-pair-readback-single-valued}
to the same normalized classifier. Its absent branch gives the normalized empty
pair and rules out every transported present pair at $(p,q)$; its present branch
gives the displayed source pair, transported target witnesses, endpoint
presentations, non-absence clauses, and paired single-valuedness. Conjoin these
branch data with the exposed public carrier and classifier fields.
\end{proof}

\begin{theorem}[Tagged option payload descent image classifier reflective transitive normalized branch occupancy equivalence]
\label{thm:tagged-option-payload-descent-image-classifier-reflective-transitive-normalized-branch-occupancy-equivalence}
Let $S,T:\mathsf{BHist}\to\mathsf{Prop}$ be BEDC history source and target
carriers, let $\sim_S$ and $\sim_T$ be source and target classifiers, let
$\delta:\mathsf{DescentCertificate}(\mathsf{BHist},\mathsf{BHist},\sim_S,\sim_T)$,
and assume payload preservation
$$
\rho_{\delta}:\forall a.\;S(a)\to T(\delta.\mathsf{map}(a)).
$$
Let $\mathsf{cert}:\NameCert(S,\sim_S)$, assume source compatibility
$\mathsf{TaggedOptionSourceHsameCompatible}(S,\sim_S)$, and assume
$$
\mathsf{TaggedOptionPayloadDescentReflectsSource}_{\delta}(S,\sim_S).
$$
For target endpoints $k,\ell,m,p,q:\mathsf{BHist}$, if
$$
\mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
(S,T,\rho_{\delta};k,\ell)
$$
and
$$
\mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
(S,T,\rho_{\delta};\ell,m)
$$
and
$$
\hsame(k,p)\land \hsame(m,q),
$$
then
$$
\bigl(\hsame(p,\emp)\Longleftrightarrow\hsame(q,\emp)\bigr)
\land
\bigl((\exists u:\mathsf{BHist}.\;T(u)\land\hsame(p,\Eone(u)))
\Longleftrightarrow
(\exists v:\mathsf{BHist}.\;T(v)\land\hsame(q,\Eone(v)))\bigr)
\land
\bigl(\hsame(p,\emp)\to
\neg\exists u:\mathsf{BHist}.\;T(u)\land\hsame(p,\Eone(u))\bigr)
\land
\bigl(\hsame(q,\emp)\to
\neg\exists v:\mathsf{BHist}.\;T(v)\land\hsame(q,\Eone(v))\bigr).
$$
\end{theorem}
\begin{proof}
Apply
Theorem~\ref{thm:tagged-option-payload-descent-image-classifier-reflective-transitive-transported-visible-branch-inversion}
to the two image-classifier witnesses and the endpoint equalities. This gives
the four normalized one-sided clauses at $(p,q)$.

The first and third clauses give the empty-occupancy equivalence directly. The
left absent clause sends $\hsame(p,\emp)$ to $\hsame(q,\emp)$, and the right
absent clause sends $\hsame(q,\emp)$ to $\hsame(p,\emp)$.

For the forward visible-occupancy direction, unpack a witness
$u:\mathsf{BHist}$ with $T(u)$ and $\hsame(p,\Eone(u))$. The left visible clause
returns source witnesses $a,b$ and a transported right endpoint
$\hsame(q,\Eone(\delta.\mathsf{map}(b)))$ together with
$T(\delta.\mathsf{map}(b))$. Choosing $v=\delta.\mathsf{map}(b)$ gives the
right visible witness.

For the reverse visible-occupancy direction, unpack a witness
$v:\mathsf{BHist}$ with $T(v)$ and $\hsame(q,\Eone(v))$. The right visible
clause returns source witnesses $a,b$ and a transported left endpoint
$\hsame(p,\Eone(\delta.\mathsf{map}(a)))$ together with
$T(\delta.\mathsf{map}(a))$. Choosing $u=\delta.\mathsf{map}(a)$ gives the left
visible witness.

It remains to exclude simultaneous empty and visible occupancy at each
normalized endpoint. If $\hsame(p,\emp)$ and $\hsame(p,\Eone(u))$ both hold,
compose them by symmetry and transitivity to obtain $\hsame(\emp,\Eone(u))$,
contradicting BEDC history no-confusion via $not\_hsame\_emp\_e1$.

If $\hsame(q,\emp)$ and $\hsame(q,\Eone(v))$ both hold, compose the visible
presentation with the empty presentation to obtain $\hsame(\Eone(v),\emp)$,
contradicting BEDC history no-confusion via $not\_hsame\_e1\_empty$.
\end{proof}

$\mathsf{DescentCertificate}$; the carrier is
$\mathsf{TaggedOptionPayloadDescentImageCarrier}$; the classifier is
$\mathsf{TaggedOptionPayloadDescentImageClassifier}$; the obligations are
reflective transitivity, endpoint transport, and direct visible-payload
exactness; the ledger is the normalized visible payload endpoints; the core
theorem is the displayed equivalence; exactness is source-payload explanation of
normalized visible target payloads; the standard bridge is $\mathsf{none}$/$\mathsf{some}$

Let $S,T:\mathsf{BHist}\to\mathsf{Prop}$ be BEDC history source and target
carriers, let $\sim_S$ and $\sim_T$ be source and target classifiers, let
$\delta:\mathsf{DescentCertificate}(\mathsf{BHist},\mathsf{BHist},\sim_S,\sim_T)$,

\rho_{\delta}:\forall a.\;S(a)\to T(\delta.\mathsf{map}(a)).

For target endpoints $k,\ell,m,p,q:\mathsf{BHist}$ and visible target payloads
$u,v:\mathsf{BHist}$, if
$$
\mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
(S,T,\rho_{\delta};k,\ell)
$$
and
$$
\mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
(S,T,\rho_{\delta};\ell,m)

$$

T(u)\land T(v)\land

Theorem~\ref{thm:tagged-option-payload-descent-image-classifier-visible-payload-exactness}

left-to-right direction keeps the four visible facts and supplies source

visible facts. These are exactly $T(u)$, $T(v)$, $\hsame(p,\Eone(u))$, and

equivalence.
\end{proof}

\begin{theorem}[Tagged option composite image classifier visible payload classification]
\label{thm:tagged-option-composite-image-classifier-visible-payload-classification}
This theorem fits the per-object template as follows: the traditional target is
functorial $\mathsf{OptionUp}$ visible payload classification; the BEDC source is
$\mathsf{BHist}$ with two composable $\mathsf{DescentCertificate}$ maps; the
carrier is $\mathsf{TaggedOptionPayloadDescentImageCarrier}$; the classifier is
$\mathsf{TaggedOptionPayloadDescentImageClassifier}$; the stability obligation is
visible factorization followed by target-classifier transport; the ledger records
final visible payloads and intermediate payload readbacks; the core theorem is
the displayed factorization together with $p\sim_T q$ and $x\sim_U y$; exactness
is composite classified final payload recovery through the intermediate payloads;
the standard bridge is $\mathsf{none}$/$\mathsf{some}$ functorial payload
classification; and the status is $\mathsf{ProofTarget}$.

Let $S,T,U:\mathsf{BHist}\to\mathsf{Prop}$ be BEDC history source, intermediate,
and target carriers, let $\sim_S$, $\sim_T$, and $\sim_U$ be their classifiers,
and let
$$
\delta:\mathsf{DescentCertificate}(\mathsf{BHist},\mathsf{BHist},\sim_S,\sim_T)
$$
and
$$
\epsilon:\mathsf{DescentCertificate}(\mathsf{BHist},\mathsf{BHist},\sim_T,\sim_U).
$$
Assume payload preservation fields
$$
\rho_{\delta}:\forall a.\;S(a)\to T(\delta.\mathsf{map}(a))
$$
and
$$
\rho_{\epsilon}:\forall b.\;T(b)\to U(\epsilon.\mathsf{map}(b)).
$$
Assume endpoint stability for the second descent:
$$
\eta_{\epsilon}:\forall x\,y.\;
T(x)\to T(y)\to \hsame(x,y)\to
\hsame(\epsilon.\mathsf{map}(x),\epsilon.\mathsf{map}(y)).
$$
Write $\epsilon\circ\delta$ for the descent certificate whose map field is
$a\mapsto\epsilon.\mathsf{map}(\delta.\mathsf{map}(a))$ and whose respects field
sends $r:a\sim_S b$ to
$\epsilon.\mathsf{respects}(\delta.\mathsf{respects}(r))$. Define
$$
\rho_{\epsilon\circ\delta}(a,Sa)=
\rho_{\epsilon}(\delta.\mathsf{map}(a),\rho_{\delta}(a,Sa)).
$$
Let $\mathsf{cert}_S:\NameCert(S,\sim_S)$,
$\mathsf{cert}_T:\NameCert(T,\sim_T)$, and
$\mathsf{cert}_U:\NameCert(U,\sim_U)$. Assume
$\mathsf{TaggedOptionSourceHsameCompatible}(S,\sim_S)$,
$\mathsf{TaggedOptionSourceHsameCompatible}(T,\sim_T)$, and
$\mathsf{TaggedOptionSourceHsameCompatible}(U,\sim_U)$. For final endpoints
$m,m':\mathsf{BHist}$ and visible final payloads $x,y:\mathsf{BHist}$, if
$$
\mathsf{TaggedOptionPayloadDescentImageClassifier}_{\epsilon\circ\delta}
(S,U,\rho_{\epsilon\circ\delta};m,m')
$$
and
$$
U(x)\land U(y)\land
\hsame(m,\Eone(x))\land
\hsame(m',\Eone(y)),
$$
then there exist $k,k',p,q,a,b:\mathsf{BHist}$ such that
\begin{enumerate}
\item $\mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
(S,T,\rho_{\delta};k,k')$;
\item $\mathsf{TaggedOptionMapRel}_{\epsilon}(T,U,\rho_{\epsilon};k,m)$ and
$\mathsf{TaggedOptionMapRel}_{\epsilon}(T,U,\rho_{\epsilon};k',m')$;
\item $T(p)$, $T(q)$, $U(\epsilon.\mathsf{map}(p))$, and
$U(\epsilon.\mathsf{map}(q))$;
\item $\hsame(k,\Eone(p))$, $\hsame(k',\Eone(q))$,
$\hsame(x,\epsilon.\mathsf{map}(p))$, and
$\hsame(y,\epsilon.\mathsf{map}(q))$;
\item $S(a)$, $S(b)$, $a\sim_S b$,
$T(\delta.\mathsf{map}(a))$, and $T(\delta.\mathsf{map}(b))$;
\item $\hsame(p,\delta.\mathsf{map}(a))$ and
$\hsame(q,\delta.\mathsf{map}(b))$;
\item $p\sim_T q$;
\item $x\sim_U y$.
\end{enumerate}
\end{theorem}
\begin{proof}
Apply
Theorem~\ref{thm:tagged-option-composite-image-classifier-visible-payload-factorization}
to the composite image-classifier witness and the visible final presentations.
This supplies $k,k',p,q,a,b$ and the six displayed factorization clauses:
the intermediate image-classifier witness, the two $\epsilon$ map-relation
witnesses, the carrier data, the four visible readbacks, the source classified
payload pair, and the two $\delta$ payload readbacks.

Use $\delta.\mathsf{respects}$ on $a\sim_S b$ to obtain
$\delta.\mathsf{map}(a)\sim_T\delta.\mathsf{map}(b)$. Target compatibility for
$T$ turns $\hsame(p,\delta.\mathsf{map}(a))$ into
$p\sim_T\delta.\mathsf{map}(a)$ and
$\hsame(q,\delta.\mathsf{map}(b))$ into
$q\sim_T\delta.\mathsf{map}(b)$. Symmetry from $\mathsf{cert}_T$ reverses the
second relation, and transitivity from $\mathsf{cert}_T$ composes the three
relations into $p\sim_T q$.

Now apply $\epsilon.\mathsf{respects}$ to $p\sim_T q$, giving
$\epsilon.\mathsf{map}(p)\sim_U\epsilon.\mathsf{map}(q)$. Compatibility for $U$
turns $\hsame(x,\epsilon.\mathsf{map}(p))$ into
$x\sim_U\epsilon.\mathsf{map}(p)$ and
$\hsame(y,\epsilon.\mathsf{map}(q))$ into
$y\sim_U\epsilon.\mathsf{map}(q)$. Symmetry and transitivity from
$\mathsf{cert}_U$ compose these three relations into $x\sim_U y$.

Package the original six factorization clauses with the two classifier
consequences $p\sim_T q$ and $x\sim_U y$.
\end{proof}
