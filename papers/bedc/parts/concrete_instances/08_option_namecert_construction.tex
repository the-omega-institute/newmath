\begin{theorem}[Tagged option carrier single-valued visible readback]
\label{thm:tagged-option-history-carrier-single-valued-visible-readback}
This theorem fits the per-object template as follows: the traditional target is
tagged optional carrier readback; the BEDC source is $\mathsf{BHist}$ with
$\hsame$; the carrier is $\mathsf{TaggedOptionHistoryCarrier}$; the classifier
is unused; the stability obligation is constructor inversion plus branch
exclusion; the ledger is the visible absent/present tag and present payload; the
core theorem is carrier single-valued visible readback; exactness is existence,
opposite-branch exclusion, and payload uniqueness; the standard bridge is
$\mathsf{none}$/$\mathsf{some}$; and the status is $\mathsf{ProofTarget}$. For
every source carrier $S:\mathsf{BHist}\to\mathsf{Prop}$ and history
$h:\mathsf{BHist}$, if $\mathsf{TaggedOptionHistoryCarrier}_S(h)$ holds, then
either
$$
\hsame(h,\emp)\land
\neg\exists a:\mathsf{BHist}.\; S(a)\land\hsame(h,\Eone(a))
$$
or
$$
\exists a:\mathsf{BHist}.\;
S(a)\land
\hsame(h,\Eone(a))\land
\neg\hsame(h,\emp)\land
\bigl(\forall a':\mathsf{BHist}.\;
S(a')\to \hsame(h,\Eone(a'))\to \hsame(a,a')\bigr).
$$
\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:tagged-option-history-carrier-exclusive-branch-partition}
to the carrier witness. This gives the displayed absent branch or a visible
present payload branch.

In the absent branch, keep the equality $\hsame(h,\emp)$ and the negative
source-present clause supplied by the carrier partition. These are exactly the
two fields required for the first displayed alternative.

In the present branch, keep the payload $a$, the source witness $S(a)$, the
endpoint link $\hsame(h,\Eone(a))$, and the non-absence proof supplied by the
carrier partition. For any other visible present payload $a'$, apply
Theorem~\ref{thm:tagged-option-history-present-payload-determinism} to the two
endpoint links $\hsame(h,\Eone(a))$ and $\hsame(h,\Eone(a'))$. The resulting
$\hsame(a,a')$ gives the payload uniqueness field, completing the second
displayed alternative.
\end{proof}

$$
\hsame(h,\emp)\land
\hsame(k,\emp)\land
\bigl(\neg\exists a:\mathsf{BHist}.\;

\bigl(\neg\exists b:\mathsf{BHist}.\;

$$
or
$$
\exists a,b:\mathsf{BHist}.\;

\hsame(k,\Eone(b))\land

\neg\hsame(k,\emp).

\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:tagged-option-history-classifier-branch-exactness} to
the classifier witness $\gamma$. This splits the proof into the exact
absent-absent branch and the exact present-present branch.

In the absent-absent branch, keep the two displayed endpoint equalities. Apply
Theorem~\ref{thm:tagged-option-history-classifier-endpoint-carriers} to
$\gamma$ to obtain carrier witnesses for both endpoints. Then apply
Theorem~\ref{thm:tagged-option-history-carrier-exclusive-branch-partition} to
each endpoint carrier. The partition result rules out any source payload whose
visible present endpoint is $\hsame$-equal to the displayed absent endpoint, so
the two negative existential clauses follow.

In the present-present branch, keep the payload histories $a,b$, the source
witnesses $S(a)$ and $S(b)$, the endpoint equalities, and the source classifier
witness $a\sim_S b$ exposed by branch exactness. Apply
Theorem~\ref{thm:tagged-option-history-classifier-endpoint-carriers} to
$\gamma$, and then apply
Theorem~\ref{thm:tagged-option-history-carrier-exclusive-branch-partition} to
both endpoint carriers. The supplied present payload witnesses rule out the
absent alternatives, so the present alternatives of the two carrier partitions
give $\neg\hsame(h,\emp)$ and $\neg\hsame(k,\emp)$, completing the displayed
present-present branch.
\end{proof}

\begin{theorem}[Tagged option payload descent image classifier pair readback is single-valued]
\label{thm:tagged-option-payload-descent-image-classifier-pair-readback-single-valued}
This theorem fits the per-object template as follows: the traditional target is
image option pair readback for constructor-tagged $\mathsf{OptionUp}$; the BEDC
source is $\mathsf{BHist}$ with a $\mathsf{DescentCertificate}$; the carrier is
$\mathsf{TaggedOptionPayloadDescentImageCarrier}$; the classifier is
$\mathsf{TaggedOptionPayloadDescentImageClassifier}$; the stability obligation is
branch exactness plus constructor no-confusion; the ledger is the pair of
transported visible payloads; the core theorem is paired exclusive
single-valued readback; exactness is the absent pair or the transported present
pair; the standard bridge is $\mathsf{none}$/$\mathsf{some}$ under payload
descent; and the status is $\mathsf{ProofTarget}$.

Let $S,T:\mathsf{BHist}\to\mathsf{Prop}$ be BEDC history source carriers, let
$\sim_S$ and $\sim_T$ be source classifiers, let
$\delta:\mathsf{DescentCertificate}(\mathsf{BHist},\mathsf{BHist},\sim_S,\sim_T)$,
and assume payload preservation
$$
\rho_{\delta}:\forall a.\;S(a)\to T(\delta.\mathsf{map}(a)).
$$
Let $\mathsf{cert}:\NameCert(S,\sim_S)$ and assume
$\mathsf{TaggedOptionSourceHsameCompatible}(S,\sim_S)$. For target endpoints
$k,k':\mathsf{BHist}$, if
$$
\mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
(S,T,\rho_{\delta};k,k')
$$
holds, then either
$$
\hsame(k,\emp)\land
\hsame(k',\emp)\land
\neg\exists a,b:\mathsf{BHist}.\;
S(a)\land S(b)\land a\sim_S b\land
T(\delta.\mathsf{map}(a))\land T(\delta.\mathsf{map}(b))\land
\hsame(k,\Eone(\delta.\mathsf{map}(a)))\land
\hsame(k',\Eone(\delta.\mathsf{map}(b)))
$$
or
$$
\exists a,b:\mathsf{BHist}.\;
S(a)\land S(b)\land a\sim_S b\land
T(\delta.\mathsf{map}(a))\land T(\delta.\mathsf{map}(b))\land
\hsame(k,\Eone(\delta.\mathsf{map}(a)))\land
\hsame(k',\Eone(\delta.\mathsf{map}(b)))\land
\neg\hsame(k,\emp)\land
\neg\hsame(k',\emp)\land
\bigl(\forall a',b':\mathsf{BHist}.\;
S(a')\to S(b')\to a'\sim_S b'\to
T(\delta.\mathsf{map}(a'))\to T(\delta.\mathsf{map}(b'))\to
\hsame(k,\Eone(\delta.\mathsf{map}(a')))\to
\hsame(k',\Eone(\delta.\mathsf{map}(b')))\to
\hsame(\delta.\mathsf{map}(a),\delta.\mathsf{map}(a'))\land
\hsame(\delta.\mathsf{map}(b),\delta.\mathsf{map}(b'))\bigr).
$$
\end{theorem}
\begin{proof}
Apply
Theorem~\ref{thm:tagged-option-payload-descent-image-classifier-branch-exactness}
to the image-classifier witness. This splits the target pair into the exact
absent pair or a transported present pair with source payloads $a,b$.

In the absent branch, keep the two endpoint presentations
$\hsame(k,\emp)$ and $\hsame(k',\emp)$. If a transported present pair also
existed at the same endpoints, the left endpoint would be presented both as
$\emp$ and as $\Eone(\delta.\mathsf{map}(a))$, contradicting
empty-versus-$\Eone$ no-confusion. This gives the displayed negative pair
clause.

In the present branch, keep the exposed payloads $a,b$, their source witnesses,
the source classifier witness, the two transported target witnesses, and the two
visible endpoint presentations. A simultaneous absent presentation of either
endpoint would contradict the corresponding visible present presentation by the
same constructor no-confusion argument, so the two non-absent clauses follow.

It remains to prove single-valuedness for any other transported present pair
$a',b'$ at the same endpoints. Apply
Theorem~\ref{thm:tagged-option-history-present-payload-determinism} to the two
left endpoint presentations of $k$ to obtain
$\hsame(\delta.\mathsf{map}(a),\delta.\mathsf{map}(a'))$.

Apply the same theorem to the two right endpoint presentations of $k'$ to obtain
$\hsame(\delta.\mathsf{map}(b),\delta.\mathsf{map}(b'))$. These two equalities
are exactly the final paired single-valuedness clause.
\end{proof}

Let $S,T:\mathsf{BHist}\to\mathsf{Prop}$ be BEDC history source carriers, let
$\sim_S$ and $\sim_T$ be source classifiers, let
$\delta:\mathsf{DescentCertificate}(\mathsf{BHist},\mathsf{BHist},\sim_S,\sim_T)$,
and assume payload preservation
$$
\rho_{\delta}:\forall a.\;S(a)\to T(\delta.\mathsf{map}(a)).
$$
Let $\mathsf{cert}:\NameCert(S,\sim_S)$, assume
$\mathsf{TaggedOptionSourceHsameCompatible}(S,\sim_S)$, and assume
$$
\mathsf{TaggedOptionPayloadDescentReflectsSource}_{\delta}(S,\sim_S).
$$
Then the option payload descent image classifier supplies the following five
certificate-facing fields.
\begin{enumerate}
  \item For every target endpoint $k:\mathsf{BHist}$,
  $$
  \mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
  (S,T,\rho_{\delta};k,k)
  \Longleftrightarrow
  \mathsf{TaggedOptionPayloadDescentImageCarrier}_{\delta}
  (S,T,\rho_{\delta};k).
  $$
  \item For target endpoints $k,k':\mathsf{BHist}$,
  $$
  \mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
  (S,T,\rho_{\delta};k,k')
  \to
  \mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
  (S,T,\rho_{\delta};k',k).
  $$
  \item For target endpoints $k,\ell,m:\mathsf{BHist}$,
  $$
  \mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
  (S,T,\rho_{\delta};k,\ell)
  \to
  \mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
  (S,T,\rho_{\delta};\ell,m)
  \to
  \mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
  (S,T,\rho_{\delta};k,m).
  $$
  \item For target endpoints $k,k',u,u':\mathsf{BHist}$,
  $$
  \mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
  (S,T,\rho_{\delta};k,k')
  \to
  \hsame(k,u)
  \to
  \hsame(k',u')
  \to
  \mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
  (S,T,\rho_{\delta};u,u').
  $$
  \item For target endpoints $k,k':\mathsf{BHist}$,
  $$
  \mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
  (S,T,\rho_{\delta};k,k')
  \to
  \mathsf{TaggedOptionPayloadDescentImageCarrier}_{\delta}
  (S,T,\rho_{\delta};k)
  \land
  \mathsf{TaggedOptionPayloadDescentImageCarrier}_{\delta}
  (S,T,\rho_{\delta};k').
  $$
\end{enumerate}
\end{theorem}
\begin{proof}
For the self-exactness field, apply
Theorem~\ref{thm:tagged-option-payload-descent-image-classifier-self-exactness}
at the chosen endpoint $k$. Its two directions are exactly the displayed
equivalence between the self-classifier witness and image-carrier membership.

For symmetry and carrier exposure, apply
Theorem~\ref{thm:tagged-option-payload-descent-image-classifier-stability-fields}.
Its second field gives the reversed classifier witness for classified endpoints,
and its third field gives the two image-carrier witnesses exposed by any
classifier witness.

For transitivity, apply
Theorem~\ref{thm:tagged-option-payload-descent-image-classifier-reflective-transitivity}
to the two classifier witnesses at $(k,\ell)$ and $(\ell,m)$, using the stated
source hsame compatibility and source-reflection hypotheses.

For endpoint transport, apply
Theorem~\ref{thm:tagged-option-payload-descent-image-classifier-hsame-transport}
to the classifier witness at $(k,k')$ and the two displayed $\hsame$ endpoint
witnesses. Its conclusion is the transported classifier witness at $(u,u')$.
\end{proof}

\begin{theorem}[Tagged option payload descent image semantic certificate under reflection]
\label{thm:tagged-option-payload-descent-image-classifier-semantic-certificate}
Let $S,T:\mathsf{BHist}\to\mathsf{Prop}$ be BEDC history source carriers, let
$\sim_S$ and $\sim_T$ be source classifiers, let
$\delta:\mathsf{DescentCertificate}(\mathsf{BHist},\mathsf{BHist},\sim_S,\sim_T)$,
and assume payload preservation
$$
\rho_{\delta}:\forall a.\;S(a)\to T(\delta.\mathsf{map}(a)).
$$
Let $\mathsf{cert}:\NameCert(S,\sim_S)$, assume
$\mathsf{TaggedOptionSourceHsameCompatible}(S,\sim_S)$, and assume
$$
\mathsf{TaggedOptionPayloadDescentReflectsSource}_{\delta}(S,\sim_S).
$$
Then the payload descent image carrier and image classifier form the image-level
semantic certificate
$$
\mathsf{SemanticNameCert}(
\mathsf{TaggedOptionPayloadDescentImageCarrier}_{\delta}(S,T,\rho_{\delta}),
\mathsf{TaggedOptionPayloadDescentImageCarrier}_{\delta}(S,T,\rho_{\delta}),
\mathsf{TaggedOptionPayloadDescentImageCarrier}_{\delta}(S,T,\rho_{\delta}),
\mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}(S,T,\rho_{\delta})).
$$
\end{theorem}
\begin{proof}
Build the core $\NameCert$ for the image carrier and image classifier. For
inhabitation, use the absent branch of
Theorem~\ref{thm:tagged-option-payload-descent-image-branch-exactness} at
$\emp$: reflexivity of $\hsame$ gives the displayed absent endpoint, and the
reverse direction of branch exactness gives the image-carrier witness at
$\emp$.

For reflexivity of the core classifier, apply
Theorem~\ref{thm:tagged-option-payload-descent-image-classifier-self-exactness}
to the image-carrier witness at the endpoint. For symmetry, use the symmetry
field of
Theorem~\ref{thm:tagged-option-payload-descent-image-classifier-stability-fields}.

For transitivity, use
Theorem~\ref{thm:tagged-option-payload-descent-image-classifier-reflective-transitivity}
with the displayed reflection hypothesis. For carrier respect, use the endpoint
carrier exposure field of
Theorem~\ref{thm:tagged-option-payload-descent-image-classifier-stability-fields}
and project the second image-carrier endpoint from the classified pair.

These five clauses assemble
$\NameCert(\mathsf{TaggedOptionPayloadDescentImageCarrier}_{\delta}
(S,T,\rho_{\delta}),
\mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
(S,T,\rho_{\delta}))$.

Finally, set the source, pattern, and ledger predicates in
$\mathsf{SemanticNameCert}$ all equal to the same payload descent image carrier.
The pattern and ledger soundness maps are identity projections, so the core
$\NameCert$ just built repacks as the displayed semantic certificate.
\end{proof}
