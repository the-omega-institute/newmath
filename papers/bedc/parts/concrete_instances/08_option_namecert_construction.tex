\section{Payload descent image classifier reflective transitive normalized public endpoint readback}
\label{sec:option-image-classifier-reflective-transitive-normalized-public-endpoint-readback}

\begin{theorem}[Tagged option payload descent image classifier reflective transitive normalized public endpoint readback]
\label{thm:tagged-option-payload-descent-image-classifier-reflective-transitive-normalized-public-endpoint-readback}
This theorem fits the per-object template as follows: the traditional target is
endpoint-normalized two-step public readback for constructor-tagged
$\mathsf{OptionUp}$; the BEDC source is $\mathsf{BHist}$ with a
$\mathsf{DescentCertificate}$; the carrier is
$\mathsf{TaggedOptionPayloadDescentImageCarrier}$; the classifier is
$\mathsf{TaggedOptionPayloadDescentImageClassifier}$; the stability obligations
are reflective transitivity, endpoint transport, carrier exposure, target
classifier exposure, and paired branch readback; the ledger is the normalized
absent pair or transported present pair; the core theorem is the displayed
public endpoint conjunction and exclusive branch split; exactness is normalized
public endpoint readback; the standard bridge is $\mathsf{none}$/$\mathsf{some}$
readback; and the status is $\mathsf{ProofTarget}$.

Let $S,T:\mathsf{BHist}\to\mathsf{Prop}$ be BEDC history source and target
carriers, let $\sim_S$ and $\sim_T$ be source and target classifiers, let
$\delta:\mathsf{DescentCertificate}(\mathsf{BHist},\mathsf{BHist},\sim_S,\sim_T)$,
and assume payload preservation
$$
\rho_{\delta}:\forall a.\;S(a)\to T(\delta.\mathsf{map}(a)).
$$
Let $\mathsf{cert}:\NameCert(S,\sim_S)$, assume source compatibility
$\mathsf{TaggedOptionSourceHsameCompatible}(S,\sim_S)$, and assume
$$
\mathsf{TaggedOptionPayloadDescentReflectsSource}_{\delta}(S,\sim_S).
$$
For target endpoints $k,\ell,m,p,q:\mathsf{BHist}$, if
$$
\mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
(S,T,\rho_{\delta};k,\ell)
$$
and
$$
\mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
(S,T,\rho_{\delta};\ell,m)
$$
and
$$
\hsame(k,p)\land \hsame(m,q),
$$
then
$$
\mathsf{TaggedOptionPayloadDescentImageCarrier}_{\delta}
(S,T,\rho_{\delta};p)
\land
\mathsf{TaggedOptionPayloadDescentImageCarrier}_{\delta}
(S,T,\rho_{\delta};q)
\land
\mathsf{TaggedOptionHistoryClassifier}_{T,\sim_T}(p,q)
$$
and either
$$
\hsame(p,\emp)\land \hsame(q,\emp)\land
\neg\exists a,b:\mathsf{BHist}.\;
S(a)\land S(b)\land a\sim_S b\land
T(\delta.\mathsf{map}(a))\land T(\delta.\mathsf{map}(b))\land
\hsame(p,\Eone(\delta.\mathsf{map}(a)))\land
\hsame(q,\Eone(\delta.\mathsf{map}(b)))
$$
or
$$
\exists a,b:\mathsf{BHist}.\;
S(a)\land S(b)\land a\sim_S b\land
T(\delta.\mathsf{map}(a))\land T(\delta.\mathsf{map}(b))\land
\hsame(p,\Eone(\delta.\mathsf{map}(a)))\land
\hsame(q,\Eone(\delta.\mathsf{map}(b)))\land
\neg\hsame(p,\emp)\land \neg\hsame(q,\emp)\land
\bigl(\forall a',b':\mathsf{BHist}.\;
S(a')\to S(b')\to a'\sim_S b'\to
T(\delta.\mathsf{map}(a'))\to T(\delta.\mathsf{map}(b'))\to
\hsame(p,\Eone(\delta.\mathsf{map}(a')))\to
\hsame(q,\Eone(\delta.\mathsf{map}(b')))\to
\hsame(\delta.\mathsf{map}(a),\delta.\mathsf{map}(a'))\land
\hsame(\delta.\mathsf{map}(b),\delta.\mathsf{map}(b'))\bigr).
$$
\end{theorem}
\begin{proof}
First compose the two image-classifier witnesses. Apply
Theorem~\ref{thm:tagged-option-payload-descent-image-classifier-reflective-transitivity}
to the witnesses at $(k,\ell)$ and $(\ell,m)$, using source compatibility and
source reflection. This gives the direct classifier from $k$ to $m$.

Next normalize the public endpoints. Apply
Theorem~\ref{thm:tagged-option-payload-descent-image-classifier-hsame-transport}
with $\hsame(k,p)$ and $\hsame(m,q)$. The result is a direct classifier at the
normalized endpoint pair $(p,q)$.

Apply
Theorem~\ref{thm:tagged-option-payload-descent-image-classifier-certificate}
to the normalized classifier. This exposes the two normalized image carriers at
$p$ and $q$ and the target tagged classifier
$\mathsf{TaggedOptionHistoryClassifier}_{T,\sim_T}(p,q)$.

Finally apply
Theorem~\ref{thm:tagged-option-payload-descent-image-classifier-pair-readback-single-valued}
to the same normalized classifier. Its absent branch gives the normalized empty
pair and rules out every transported present pair at $(p,q)$; its present branch
gives the displayed source pair, transported target witnesses, endpoint
presentations, non-absence clauses, and paired single-valuedness. Conjoin these
branch data with the exposed public carrier and classifier fields.
\end{proof}
