\begin{theorem}[Rat denominator continuation common-context cancellation]
\label{thm:field-rat-denominator-continuation-common-context-cancellation}
For all histories $p,q,d,e,pd,pe,left,right:\mathsf{BHist}$, if
$\mathsf{UnaryHistory}(p)$, $\mathsf{UnaryHistory}(q)$,
$\mathsf{RatHistoryCarrier}(d)$, $\mathsf{RatHistoryCarrier}(e)$,
$\Cont(p,d,pd)$, $\Cont(p,e,pe)$, $\Cont(pd,q,left)$,
$\Cont(pe,q,right)$, and $\mathsf{RatHistoryClassifier}(left,right)$, then
$\mathsf{RatHistoryClassifier}(d,e)$.
\end{theorem}
\begin{proof}
Unpack the final Rat classifier. Its comparison component gives
$left\hsame right$.

Apply the continuation common-context cancellation theorem
\texttt{BEDC.FKernel.Cont.cont\_cancel\_common\_context} to the two matched
prefix witnesses, the two matched suffix witnesses, and the comparison
$left\hsame right$. The shared prefix $p$ and shared suffix $q$ cancel, leaving
the endpoint comparison $d\hsame e$.

Combine this endpoint comparison with the original denominator carriers
$\mathsf{RatHistoryCarrier}(d)$ and $\mathsf{RatHistoryCarrier}(e)$. These
three pieces are exactly the fields of
$\mathsf{RatHistoryClassifier}(d,e)$.
\end{proof}

\begin{theorem}[Rat denominator matched-context classifier exactness]
\label{thm:field-rat-denominator-matched-context-classifier-exactness}
For all histories $p,q,d,e,pd,pe,left,right:\mathsf{BHist}$, if
$\mathsf{UnaryHistory}(p)$, $\mathsf{UnaryHistory}(q)$,
$\mathsf{RatHistoryCarrier}(d)$, $\mathsf{RatHistoryCarrier}(e)$,
$\Cont(p,d,pd)$, $\Cont(p,e,pe)$, $\Cont(pd,q,left)$, and
$\Cont(pe,q,right)$, then
$$
\mathsf{RatHistoryClassifier}(d,e)\Longleftrightarrow
\mathsf{RatHistoryClassifier}(left,right).
$$
\end{theorem}
\begin{proof}
For the forward direction, apply
\autoref{thm:rat-history-classifier-unary-denominator-context-closed} to
$\mathsf{RatHistoryClassifier}(d,e)$ with the common unary prefix $p$ and
common unary tail $q$. This gives the classifier for the append representatives
obtained by placing the same matched context around $d$ and $e$.

Continuation determinacy and associativity compare those representatives with
the displayed endpoints $left$ and $right$, using the premises
$\Cont(p,d,pd)$, $\Cont(p,e,pe)$, $\Cont(pd,q,left)$, and
$\Cont(pe,q,right)$. Transporting the classifier across these endpoint
comparisons gives
$\mathsf{RatHistoryClassifier}(left,right)$.

For the reverse direction, apply
\autoref{thm:field-rat-denominator-continuation-common-context-cancellation}
to the same unary-context and continuation premises. It cancels the common
prefix and suffix context from $\mathsf{RatHistoryClassifier}(left,right)$ and
returns $\mathsf{RatHistoryClassifier}(d,e)$.
\end{proof}

\begin{theorem}[Rat denominator continuation common-context ledger closure]
\label{thm:field-rat-denominator-continuation-common-context-ledger-closure}
For all histories $p,p',q,q',\rho,v,pr,pv,left,right:\mathsf{BHist}$, if
$\mathsf{RatHistoryLedgerPolicy}(\rho,v)$, $\mathsf{UnaryHistory}(p)$,
$\mathsf{UnaryHistory}(p')$, $\mathsf{UnaryHistory}(q)$,
$\mathsf{UnaryHistory}(q')$, $p\hsame p'$, $q\hsame q'$,
$\Cont(p,\rho,pr)$, $\Cont(p',v,pv)$, $\Cont(pr,q,left)$, and
$\Cont(pv,q',right)$, then $\mathsf{RatHistoryLedgerPolicy}(left,right)$.
\end{theorem}
\begin{proof}
First apply \autoref{thm:rat-ledger-unary-denominator-context-closed} to the
ledger $\mathsf{RatHistoryLedgerPolicy}(\rho,v)$, the matching unary prefixes
$p,p'$, and the matching unary suffixes $q,q'$. This gives the canonical ledger
between $\mathsf{append}(p,\mathsf{append}(\rho,q))$ and
$\mathsf{append}(p',\mathsf{append}(v,q'))$.

Next compare the canonical endpoints with the displayed two-step continuation
endpoints. Continuation determinacy identifies the intermediate histories
$pr$ and $pv$ with the corresponding prefix appends, and
\texttt{BEDC.FKernel.Cont.cont\_assoc\_hsame} identifies the two displayed
suffix continuations with the canonical common-context bracketings.

Finally transport the canonical ledger across those endpoint $\hsame$ links by
\autoref{thm:rat-ledger-hsame-transport}. The resulting ledger is exactly
$\mathsf{RatHistoryLedgerPolicy}(left,right)$.
\end{proof}

\begin{theorem}[Rat denominator continuation common-context exactness package]
\label{thm:field-rat-denominator-continuation-common-context-exactness-package}
For all histories $p,q,d,e,pd,pe,left,right:\mathsf{BHist}$, if
$\mathsf{UnaryHistory}(p)$, $\mathsf{UnaryHistory}(q)$,
$\mathsf{RatHistoryCarrier}(d)$, $\mathsf{RatHistoryCarrier}(e)$,
$\Cont(p,d,pd)$, $\Cont(p,e,pe)$, $\Cont(pd,q,left)$, and
$\Cont(pe,q,right)$, then both equivalences hold:
$$
\mathsf{RatHistoryClassifier}(left,right)\Longleftrightarrow
\mathsf{RatHistoryClassifier}(d,e).
$$
$$
\mathsf{RatHistoryLedgerPolicy}(left,right)\Longleftrightarrow
\mathsf{RatHistoryLedgerPolicy}(d,e).
$$
\end{theorem}
\begin{proof}
For the classifier implication from $left,right$ back to $d,e$, apply
\autoref{thm:field-rat-denominator-continuation-common-context-cancellation}
to the same unary-context and continuation premises. It cancels the common
prefix and suffix context from $\mathsf{RatHistoryClassifier}(left,right)$ and
returns $\mathsf{RatHistoryClassifier}(d,e)$.

For the classifier implication from $d,e$ to $left,right$, apply
\autoref{thm:rat-history-classifier-unary-denominator-context-closed} to
$\mathsf{RatHistoryClassifier}(d,e)$ with the common unary prefix $p$ and common
unary suffix $q$. This gives the classifier for the canonical endpoints obtained
by placing $p$ and $q$ around $d$ and $e$.

Continuation determinacy compares those canonical endpoints with the displayed
endpoints $left$ and $right$, using the premises $\Cont(p,d,pd)$,
$\Cont(p,e,pe)$, $\Cont(pd,q,left)$, and $\Cont(pe,q,right)$. Transporting the
classifier across these endpoint comparisons by
\autoref{thm:rat-history-classifier-hsame-transport} gives
$\mathsf{RatHistoryClassifier}(left,right)$.

For the ledger implication from $left,right$ back to $d,e$, first convert
$\mathsf{RatHistoryLedgerPolicy}(left,right)$ to the raw-visible classifier by
\autoref{thm:rat-ledger-raw-visible-classifier}. The cancellation theorem then
returns $\mathsf{RatHistoryClassifier}(d,e)$, whose comparison component is
$d\hsame e$. Together with the input carrier $\mathsf{RatHistoryCarrier}(d)$,
this is precisely $\mathsf{RatHistoryLedgerPolicy}(d,e)$.

For the ledger implication from $d,e$ to $left,right$, apply
\autoref{thm:rat-ledger-unary-denominator-context-closed} to
$\mathsf{RatHistoryLedgerPolicy}(d,e)$ with the common unary prefix $p$ and
common unary suffix $q$. Continuation determinacy compares the canonical ledger
endpoints with $left$ and $right$, and
\autoref{thm:rat-ledger-hsame-transport} transports the closed ledger across
those endpoint comparisons.
\end{proof}

$p,p',q,q',d,e,pd,pe,left,right:\mathsf{BHist}$, if
$\mathsf{UnaryHistory}(p)$, $\mathsf{UnaryHistory}(p')$,
$\mathsf{UnaryHistory}(q)$, $\mathsf{UnaryHistory}(q')$, $p\hsame p'$,
$q\hsame q'$, $\mathsf{RatHistoryCarrier}(d)$,
$\mathsf{RatHistoryCarrier}(e)$, $\Cont(p,d,pd)$, $\Cont(p',e,pe)$,
$\Cont(pd,q,left)$, and $\Cont(pe,q',right)$, then
$$
\mathsf{RatHistoryClassifier}(d,e)
\Longleftrightarrow
\mathsf{RatHistoryClassifier}(left,right).
$$
\end{theorem}
\begin{proof}
For the forward direction, assume $\mathsf{RatHistoryClassifier}(d,e)$.
Apply \autoref{thm:rat-history-classifier-unary-denominator-context-closed}
to the unary prefixes $p,p'$, the unary suffixes $q,q'$, and the witnesses
$p\hsame p'$ and $q\hsame q'$. This yields the classifier for the append
histories obtained by placing the two history-same contexts around $d$ and
$e$.

The displayed endpoints are obtained from those append histories by the
two-step continuation presentations. Continuation determinacy identifies
$pd$ and $pe$ with the corresponding prefix appends, while continuation
associativity identifies the final suffix continuations with the append
bracketings. Transport the classifier across these endpoint $\hsame$ links by
\autoref{thm:rat-history-classifier-hsame-transport} to obtain
$\mathsf{RatHistoryClassifier}(left,right)$.

For the reverse direction, assume
$\mathsf{RatHistoryClassifier}(left,right)$. Transport the continuation
presentation along $p\hsame p'$ and $q\hsame q'$ using
\texttt{BEDC.FKernel.Cont.cont\_hsame\_transport}; this gives a common-context
presentation with the same displayed endpoints up to $\hsame$.

Transport the endpoint classifier across those endpoint comparisons by
\autoref{thm:rat-history-classifier-hsame-transport}. Then apply
\autoref{thm:field-rat-denominator-continuation-common-context-cancellation}
to cancel the common unary prefix and suffix. Repacking the two Rat carriers
with the recovered endpoint comparison gives
$\mathsf{RatHistoryClassifier}(d,e)$.
\end{proof}

\begin{theorem}[Rat denominator continuation is semigroup-only on the positive carrier]
\label{thm:field-rat-denominator-continuation-semigroup-only}
For the carrier $\mathsf{RatHistoryCarrier}$ and continuation relation $\Cont$,
Theorems~\ref{thm:field-rat-denominator-continuation-carrier-closure} and
\ref{thm:field-rat-denominator-continuation-semigroup-laws} provide carrier
closure and classifier associativity on positive Rat denominators. The empty
history is not an internal carrier endpoint: $\mathsf{RatHistoryCarrier}(\emp)$
implies $\mathsf{False}$. Consequently the endpoint laws in
Theorem~\ref{thm:field-rat-denominator-continuation-external-boundary-laws}
are external empty-history endpoint laws. They do not provide an internal
$\MonoidUp$ unit or an internal $\FieldUp$ unit on the Rat denominator carrier.
\end{theorem}
\begin{proof}
The semigroup clauses are exactly the content of the displayed closure and
semigroup-law theorems. Carrier closure keeps every realized continuation result
inside $\mathsf{RatHistoryCarrier}$ when both inputs are positive Rat
denominators, and the semigroup law identifies the two continuation bracketings
by the Rat history classifier.

For the unit boundary, suppose $\mathsf{RatHistoryCarrier}(\emp)$. Reflexivity
of $\hsame$ at $\emp$ gives $\emp\hsame\emp$. Applying
Theorem~\ref{thm:rat-history-carrier-not-empty} to the assumed carrier witness
and this reflexive endpoint equality yields $\mathsf{False}$. Thus the
empty-history clauses in
Theorem~\ref{thm:field-rat-denominator-continuation-external-boundary-laws}
can be used only as external endpoint laws, not as an internal carrier unit.
\end{proof}

\begin{theorem}[Rat denominator continuation has no internal carrier unit]
\label{thm:field-rat-denominator-continuation-no-internal-unit}
For every history $u:\mathsf{BHist}$, the conjunction of
$\mathsf{RatHistoryCarrier}(u)$ with the internal right-unit classifier law
$$
\forall d,r:\mathsf{BHist}.\;
\mathsf{RatHistoryCarrier}(d)\to
\Cont(d,u,r)\to
\mathsf{RatHistoryClassifier}(r,d)
$$
implies $\mathsf{False}$. Consequently no carried $u$ can be a two-sided
internal continuation unit on $\mathsf{RatHistoryCarrier}$ under
$\mathsf{RatHistoryClassifier}$.
\end{theorem}
\begin{proof}
Assume a carried candidate $u$ and the displayed right-unit classifier law.
Let $d_1:=\Eone(\emp)$. The positive-denominator construction gives
$\mathsf{RatHistoryCarrier}(d_1)$.

Use the canonical continuation witness
$\Cont(d_1,u,\mathsf{append}(d_1,u))$. Applying the assumed right-unit law at
$d_1$ and $r:=\mathsf{append}(d_1,u)$ gives
$\mathsf{RatHistoryClassifier}(\mathsf{append}(d_1,u),d_1)$, whose comparison
component is $\mathsf{append}(d_1,u)\hsame d_1$.

Transport the continuation result along this comparison to obtain
$\Cont(d_1,u,d_1)$. The kernel theorem
\texttt{BEDC.FKernel.Cont.cont\_right\_unit\_unique} then forces
$u\hsame\emp$.

Finally apply Theorem~\ref{thm:rat-history-carrier-not-empty} to the assumed
carrier witness $\mathsf{RatHistoryCarrier}(u)$ and the equality
$u\hsame\emp$. This contradiction proves that no carried endpoint can realize
the required right-unit law, and a two-sided internal unit would contain that
right-unit law as one of its clauses.
\end{proof}

satisfies
the following package clauses:

  $\mathsf{RatHistoryCarrier}(de)$, $\mathsf{RatHistoryCarrier}(ef)$,
  $\mathsf{RatHistoryCarrier}(left)$, $\mathsf{RatHistoryCarrier}(right)$, and
  $\mathsf{RatHistoryClassifier}(left,right)$;
  \item for all histories $d,d',u,u',r,s:\mathsf{BHist}$, if
  $\mathsf{RatHistoryClassifier}(d,d')$, $\mathsf{UnaryHistory}(u)$,
  $\mathsf{UnaryHistory}(u')$, $u\hsame u'$, $\Cont(d,u,r)$, and

  \item for all histories $d,d',e,r,r':\mathsf{BHist}$, if

  $\mathsf{UnaryHistory}(u')$, $u\hsame u'$, $\Cont(\rho,u,r)$, and
  $\Cont(v,u',s)$, then $\mathsf{RatHistoryLedgerPolicy}(r,s)$;
  \item for every history $u:\mathsf{BHist}$,
  $\mathsf{RatHistoryCarrier}(u)$ together with the right-unit classifier law
  $$
  \forall d,r:\mathsf{BHist}.\;
  \mathsf{RatHistoryCarrier}(d)\to
  \Cont(d,u,r)\to
  \mathsf{RatHistoryClassifier}(r,d)
  $$
  implies $\mathsf{False}$. Hence no carried endpoint can serve as a right or
  two-sided internal unit for this continuation instance.
\end{enumerate}
\end{theorem}
\begin{proof}
For the first clause, apply
\autoref{thm:field-rat-denominator-continuation-semigroup-laws}. It gives the
two intermediate carriers, the two outer carriers, and $left\hsame right$.
Combining the two outer carriers with this comparison yields
$\mathsf{RatHistoryClassifier}(left,right)$.

The same clause records the carrier closure supplied by
\autoref{thm:field-rat-denominator-continuation-carrier-closure}. The
classifier statement for the two bracketings is the semigroup associativity
component, while
\autoref{thm:field-rat-denominator-continuation-commutative-classifier} records
the corresponding commutative classifier coherence for swapped carried
continuations.

For classifier preservation under matching continuation tails, use
\autoref{thm:field-rat-denominator-continuation-classifier-closure}. For the
shared carried right endpoint form, use
\autoref{thm:field-rat-classifier-continuation-right-closed}. The compatible
associative version with classified triples is exactly
\autoref{thm:field-rat-denominator-continuation-associative-classifier-congruence}.

For the ledger clause, apply
\autoref{thm:field-rat-denominator-continuation-ledger-closure} to the displayed
ledger-related endpoints, matching unary tails, matching tail comparison, and
continuation witnesses. The conclusion is the stated preservation of
$\mathsf{RatHistoryLedgerPolicy}$.

For the final obstruction clause, apply
\autoref{thm:field-rat-denominator-continuation-no-internal-unit}. Its statement
is precisely the contradiction obtained from a carried endpoint satisfying the
right-unit classifier law, and any two-sided internal unit contains that
right-unit clause.
\end{proof}

\begin{theorem}[Rat denominator continuation has no internal one-sided carrier unit]
\label{thm:field-rat-denominator-continuation-no-internal-one-sided-unit}
For every history $u:\mathsf{BHist}$, if $\mathsf{RatHistoryCarrier}(u)$, then
each internal one-sided carrier-unit classifier law is impossible. The left-unit
classifier law

$$
\forall d,r:\mathsf{BHist}.\;
\mathsf{RatHistoryCarrier}(d)\to
\Cont(u,d,r)\to
\mathsf{RatHistoryClassifier}(r,d)
$$

implies $\mathsf{False}$, and the right-unit classifier law

$$
\forall d,r:\mathsf{BHist}.\;
\mathsf{RatHistoryCarrier}(d)\to
\Cont(d,u,r)\to
\mathsf{RatHistoryClassifier}(r,d)
$$

implies $\mathsf{False}$. Hence Rat denominator continuation supplies no
carried one-sided unit under $\mathsf{RatHistoryClassifier}$.
\end{theorem}
\begin{proof}
Assume a carried candidate $u$ and the displayed left-unit classifier law. Let
$d_1:=\Eone(\emp)$. The positive-denominator construction gives
$\mathsf{RatHistoryCarrier}(d_1)$.

Use the canonical continuation witness
$\Cont(u,d_1,\mathsf{append}(u,d_1))$. Applying the assumed left-unit law at
$d_1$ and $r:=\mathsf{append}(u,d_1)$ gives
$\mathsf{RatHistoryClassifier}(\mathsf{append}(u,d_1),d_1)$, whose comparison
component is $\mathsf{append}(u,d_1)\hsame d_1$.

Transport the continuation result along this comparison to obtain
$\Cont(u,d_1,d_1)$. The kernel theorem
\texttt{BEDC.FKernel.Cont.cont\_left\_unit\_unique} then forces
$u\hsame\emp$.

Finally apply Theorem~\ref{thm:rat-history-carrier-not-empty} to the assumed
carrier witness $\mathsf{RatHistoryCarrier}(u)$ and the equality $u\hsame\emp$.
This proves that the left-unit classifier law implies $\mathsf{False}$.

For the right-unit classifier law, apply
Theorem~\ref{thm:field-rat-denominator-continuation-no-internal-unit} to the
same carried history $u$ and the same carrier witness. Its conclusion is
$\mathsf{False}$.
\end{proof}

\begin{theorem}[Rat denominator continuation is not a FieldUp multiplication]
\label{thm:field-rat-denominator-continuation-not-field-multiplication}
For every proposed carried unit $u:\mathsf{BHist}$, suppose
$\mathsf{RatHistoryCarrier}(u)$ and suppose the proposed carried
multiplication on the Rat denominator carrier is the continuation relation
$\Cont$. If the corresponding $\FieldUp(\mathsf{RatUp})$ right-unit law under
$\mathsf{RatHistoryClassifier}$ is
$$
\forall d,r:\mathsf{BHist}.\;
\mathsf{RatHistoryCarrier}(d)\to
\Cont(d,u,r)\to
\mathsf{RatHistoryClassifier}(r,d),
$$
then the assumptions imply $\mathsf{False}$. Consequently a
$\FieldUp(\mathsf{RatUp})$ certificate cannot take Rat-denominator
continuation itself as its carried multiplication together with a carried
unit.
\end{theorem}
\begin{proof}
Unpack the attempted $\FieldUp$ multiplication data for the carried operation
$\Cont$. The right-unit field supplies exactly the displayed classifier law:
for every Rat denominator $d$ and every continuation result $r$ from
$\Cont(d,u,r)$, the result must classify as the original denominator $d$ under
$\mathsf{RatHistoryClassifier}$.

Apply
Theorem~\ref{thm:field-rat-denominator-continuation-no-internal-unit} to the
same history $u$, using the carried-unit witness
$\mathsf{RatHistoryCarrier}(u)$ and this unpacked right-unit field. Its
conclusion is $\mathsf{False}$.

Theorem~\ref{thm:field-rat-denominator-continuation-semigroup-only} records the
positive part left intact by this contradiction: continuation on Rat
denominators still has carrier closure and semigroup-level associativity, but
those clauses do not supply an internal $\MonoidUp$ unit or the unit required by
$\FieldUp$ multiplication.
\end{proof}

\begin{theorem}[Rat denominator continuation commutative-semigroup obstruction package]
\label{thm:field-rat-denominator-continuation-commutative-semigroup-obstruction-package}
For the concrete BEDC carrier $\mathsf{RatHistoryCarrier}$, continuation
relation $\Cont$, and classifier $\mathsf{RatHistoryClassifier}$, Rat
denominator continuation satisfies the following four-field package.
\begin{enumerate}
\item Carrier closure: if $\mathsf{RatHistoryCarrier}(d)$,
$\mathsf{RatHistoryCarrier}(e)$, and $\Cont(d,e,r)$, then
$\mathsf{RatHistoryCarrier}(r)$.
\item Classifier associativity: if $d,e,f,de,ef,\ell,r:\mathsf{BHist}$ are
related by $\Cont(d,e,de)$, $\Cont(e,f,ef)$, $\Cont(de,f,\ell)$, and
$\Cont(d,ef,r)$, with $d,e,f$ carried by $\mathsf{RatHistoryCarrier}$, then the
two bracketed results are carried and satisfy $\ell\hsame r$.
\item Classifier commutativity: if $\mathsf{RatHistoryCarrier}(d)$,
$\mathsf{RatHistoryCarrier}(e)$, $\Cont(d,e,r)$, and $\Cont(e,d,s)$, then
$\mathsf{RatHistoryClassifier}(r,s)$.
\item Internal right-unit impossibility: for every $u:\mathsf{BHist}$, the
conjunction of $\mathsf{RatHistoryCarrier}(u)$ with
$$
\forall d,r:\mathsf{BHist}.\;
\mathsf{RatHistoryCarrier}(d)\to
\Cont(d,u,r)\to
\mathsf{RatHistoryClassifier}(r,d)
$$
implies $\mathsf{False}$.
\end{enumerate}
Consequently Rat-denominator continuation is a concrete commutative
semigroup-level operation under $\mathsf{RatHistoryClassifier}$ with no internal
carried unit, and therefore cannot serve as the carried multiplication of
$\FieldUp(\mathsf{RatUp})$.
\end{theorem}
\begin{proof}
The carrier-closure field is exactly
\autoref{thm:field-rat-denominator-continuation-carrier-closure}; it uses only
the concrete carrier $\mathsf{RatHistoryCarrier}$ and the concrete continuation
relation $\Cont$.

The associativity field is
\autoref{thm:field-rat-denominator-continuation-semigroup-laws}. It supplies the
two intermediate carriers, the two outer result carriers, and the classifier
comparison between the left- and right-bracketed continuations.

The commutativity field is
\autoref{thm:field-rat-denominator-continuation-commutative-classifier}. It
compares the two swapped denominator continuations by
$\mathsf{RatHistoryClassifier}$.

The internal right-unit obstruction is
\autoref{thm:field-rat-denominator-continuation-no-internal-unit}. Combining
that obstruction with the closure, associativity, and commutativity fields gives
precisely a commutative semigroup-level continuation operation, not a carried
field multiplication with an internal unit.
\end{proof}

component is $\mathsf{append}(u,d_1)\hsame d_1$.

Transport the continuation result along this comparison to obtain
$\Cont(u,d_1,d_1)$. The kernel theorem
\texttt{BEDC.FKernel.Cont.cont\_left\_unit\_unique} then forces
$u\hsame\emp$.

Finally apply Theorem~\ref{thm:rat-history-carrier-not-empty} to the assumed
carrier witness $\mathsf{RatHistoryCarrier}(u)$ and the equality
$u\hsame\emp$. This contradiction proves that no carried endpoint can realize
the required left-unit law.
\end{proof}
