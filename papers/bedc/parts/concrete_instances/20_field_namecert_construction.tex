\section{RatUp boundary for the singleton field schema}
\label{sec:field-singleton-rat-boundary}

\begin{theorem}[Singleton field schema is disjoint from the RatUp history carrier]
\label{thm:field-singleton-schema-disjoint-from-rat-history-carrier}
For every $h:\mathsf{BHist}$, if $A_{\emp}(h)$ is the singleton source from
Definition~\ref{def:singleton-empty-history-field-schema-instance}, then
$\mathsf{RatHistoryCarrier}(h)$ implies $\mathsf{False}$. Equivalently, no
RatUp history-carrier endpoint is an endpoint of the singleton empty-history
$\FieldUp$ schema.
\end{theorem}
\begin{proof}
Assume $A_{\emp}(h)$ and $\mathsf{RatHistoryCarrier}(h)$. Unfolding the
singleton carrier component gives the endpoint equality $h\hsame\emp$.

The RatUp carrier witness is separated from the empty denominator endpoint by
Theorem~\ref{thm:rat-history-carrier-not-empty}. Applying that theorem to the
carrier witness and the equality $h\hsame\emp$ gives the required
contradiction.
\end{proof}
