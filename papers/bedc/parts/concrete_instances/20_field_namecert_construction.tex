\begin{theorem}[RatUp FieldUp affine center normal subgroup]
\label{thm:ratup-fieldup-affine-center-normal-subgroup}
For Definition~\ref{def:ratup-fieldup-concrete-exit-object}, the concrete
affine center predicate $\mathsf{Center}_F$ on certified affine automorphisms
forms a normal subgroup package inside $\mathsf{AffAut}_F$. Namely:
\begin{enumerate}
\item $\mathsf{Center}_F(\mathrm{id}_F)$.
\item If $\mathsf{Center}_F(Z)$ and $\mathsf{Center}_F(W)$, then
$\mathsf{Center}_F(Z\circ_{\mathrm{aff}}W)$.
\item If $\mathsf{Center}_F(Z)$, then
$\mathsf{Center}_F(\iota_F(Z))$.
\item If $Z\equiv_{\mathrm{aff}}W$ and $\mathsf{Center}_F(Z)$, then
$\mathsf{Center}_F(W)$.
\item If $\mathsf{Center}_F(Z)$, then for every
$T:\mathsf{AffAut}_F$,
$\mathsf{Center}_F(\mathrm{Inn}_T(Z))$.
\end{enumerate}
\end{theorem}
\begin{proof}
Fix the canonical exit action and inverse package of
\autoref{thm:ratup-fieldup-canonical-exit-action-inverse-package}. This fixes
the carrier $\mathsf{AffAut}_F$, the product $\circ_{\mathrm{aff}}$, the
identity $\mathrm{id}_F$, the inverse row $\iota_F(-)$, the classifier
$\equiv_{\mathrm{aff}}$, and the inner action $\mathrm{Inn}_T$ from
\autoref{thm:ratup-fieldup-affine-inner-conjugation-exit}.

The identity member is central by the identity rows in
\autoref{thm:field-affine-automorphism-group-laws}. For product closure, fix a
test automorphism $T$ and central elements $Z,W$. Associativity from
\autoref{thm:field-affine-automorphism-composition-associative} exposes
$W\circ_{\mathrm{aff}}T$ and then $Z\circ_{\mathrm{aff}}T$ as adjacent
subproducts. Apply the two center equations and transport both replacements by
\autoref{thm:field-affine-pointwise-classifier-operation-congruence}; the same
associativity theorem repacks the result as
$T\circ_{\mathrm{aff}}(Z\circ_{\mathrm{aff}}W)$.

For inverse closure, use
\autoref{thm:ratup-fieldup-affine-center-fixed-point-criterion}. If
$\mathsf{Center}_F(Z)$, then every $\mathrm{Inn}_T(Z)$ is classifier-equal to
$Z$. The inverse-preservation clause of
\autoref{thm:ratup-fieldup-affine-inner-conjugation-exit}, together with
classifier congruence, transports this fixed-point equation through
$\iota_F(-)$ and gives
$\mathrm{Inn}_T(\iota_F(Z))\equiv_{\mathrm{aff}}\iota_F(Z)$ for every $T$.
The fixed-point criterion returns $\mathsf{Center}_F(\iota_F(Z))$.

Classifier transport is pointwise: if $Z\equiv_{\mathrm{aff}}W$, then
operation congruence replaces $Z$ by $W$ on both sides of the defining
commutation equation for each test automorphism.

Finally assume $\mathsf{Center}_F(Z)$ and fix a conjugating element $T$. By
\autoref{thm:ratup-fieldup-affine-center-fixed-point-criterion}, $Z$ is fixed
by every inner affine action. For any test element $S$,
\autoref{thm:ratup-fieldup-affine-inner-action-composition-exit} rewrites
$\mathrm{Inn}_S(\mathrm{Inn}_T(Z))$ as
$\mathrm{Inn}_{S\circ_{\mathrm{aff}}T}(Z)$, which is classifier-equal to $Z$.
The same criterion also identifies $\mathrm{Inn}_T(Z)$ with $Z$, and classifier
transitivity yields
$\mathrm{Inn}_S(\mathrm{Inn}_T(Z))\equiv_{\mathrm{aff}}\mathrm{Inn}_T(Z)$.
Applying the fixed-point criterion once more gives
$\mathsf{Center}_F(\mathrm{Inn}_T(Z))$.
\end{proof}

\begin{theorem}[RatUp FieldUp affine centralizer normalizer subgroup]
\label{thm:ratup-fieldup-affine-centralizer-normalizer-subgroup}
For Definition~\ref{def:ratup-fieldup-concrete-exit-object}, fix a certified
affine automorphism $A:\mathsf{AffAut}_F$ in the concrete $\FieldUp(\RatUp)$
exit object. Let $C_F(A)$ be the affine centralizer carrier from
\autoref{def:ratup-fieldup-affine-centralizer-carrier}, and let $N_F(C_F(A))$
be the two-sided normalizer carrier from
\autoref{def:ratup-fieldup-affine-centralizer-normalizer-carrier}. Then
$N_F(C_F(A))$ is the concrete affine normalizer subgroup of $C_F(A)$:
\begin{enumerate}
\item $N_F(C_F(A))(\mathrm{id}_F)$.
\item If $N_F(C_F(A))(T)$ and $N_F(C_F(A))(U)$, then
$N_F(C_F(A))(T\circ_{\mathrm{aff}}U)$.
\item If $N_F(C_F(A))(T)$, then $N_F(C_F(A))(\iota_F(T))$.
\item If $N_F(C_F(A))(T)$ and $T\equiv_{\mathrm{aff}}U$, then
$N_F(C_F(A))(U)$.
\end{enumerate}
\end{theorem}
\begin{proof}
Fix the canonical affine exit action and inverse package from
\autoref{thm:ratup-fieldup-canonical-exit-action-inverse-package}. This fixes
$\mathsf{AffAut}_F$, $\circ_{\mathrm{aff}}$, $\mathrm{id}_F$, $\iota_F(-)$,
and $\equiv_{\mathrm{aff}}$ for the concrete $\FieldUp(\RatUp)$ exit object.

For the identity element, fix $X:\mathsf{AffAut}_F$. The identity clause of
\autoref{thm:ratup-fieldup-affine-inner-action-composition-exit} identifies
$\mathrm{Conj}_{\mathrm{id}_F}(X)$ with $X$. Hence centralizer membership is
preserved and reflected by identity conjugation, and the identity row of
\autoref{thm:ratup-fieldup-affine-centralizer-subgroup} supplies the required
carrier witness.

For product closure, assume $N_F(C_F(A))(T)$ and $N_F(C_F(A))(U)$, and fix
$X:\mathsf{AffAut}_F$. If $C_F(A)(X)$, the two-sided definition first gives
$C_F(A)(\mathrm{Conj}_U(X))$ and then gives
$C_F(A)(\mathrm{Conj}_T(\mathrm{Conj}_U(X)))$. The composition clause of
\autoref{thm:ratup-fieldup-affine-inner-action-composition-exit} identifies
the composite conjugation with conjugation by $T\circ_{\mathrm{aff}}U$, so the
forward half gives $C_F(A)(\mathrm{Conj}_{T\circ_{\mathrm{aff}}U}(X))$.

Conversely assume $C_F(A)(\mathrm{Conj}_{T\circ_{\mathrm{aff}}U}(X))$. The
same composition clause reads this as
$C_F(A)(\mathrm{Conj}_T(\mathrm{Conj}_U(X)))$. The reverse half of the
two-sided definition for $T$ gives $C_F(A)(\mathrm{Conj}_U(X))$, and the
reverse half for $U$ gives $C_F(A)(X)$.

For inverse closure, assume $N_F(C_F(A))(T)$ and fix $X:\mathsf{AffAut}_F$.
Let $Y=\mathrm{Conj}_{\iota_F(T)}(X)$. The inverse antihomomorphism package of
\autoref{thm:ratup-fieldup-affine-inverse-antihom-exit} and the conjugation
equation exactness of
\autoref{thm:ratup-fieldup-affine-conjugation-equation-exactness} identify
$\mathrm{Conj}_T(Y)$ with $X$ under $\equiv_{\mathrm{aff}}$.

If $C_F(A)(X)$, transport this membership across the exactness comparison to
$C_F(A)(\mathrm{Conj}_T(Y))$. The reverse half of the normalizer equivalence for
$T$, applied to $Y$, gives $C_F(A)(Y)$. If $C_F(A)(Y)$, the forward half for
$T$ and the same double-conjugation exactness return $C_F(A)(X)$. Thus
conjugation by $\iota_F(T)$ preserves and reflects centralizer membership, and
the inverse row gives the carrier witness for $\iota_F(T)$.

For classifier transport, assume $N_F(C_F(A))(T)$ and
$T\equiv_{\mathrm{aff}}U$. For each $X$, affine operation congruence from
\autoref{thm:field-affine-pointwise-classifier-operation-congruence} transports
the conjugation word
$(T\circ_{\mathrm{aff}}X)\circ_{\mathrm{aff}}\iota_F(T)$ to
$(U\circ_{\mathrm{aff}}X)\circ_{\mathrm{aff}}\iota_F(U)$. The preservation and
reflection clauses for $T$ therefore give the same two-sided centralizer
equivalence for $U$, and the carrier witness is transported along
$T\equiv_{\mathrm{aff}}U$.
\end{proof}

Per-object template: traditional target normalizer subgroup; BEDC source
concrete $\FieldUp(\RatUp)$ affine automorphism group plus $\GroupUp$; carrier
$N_F(C_F(A))$; classifier $\equiv_{\mathrm{aff}}$; stability identity, product,
inverse, and transport; ledger fixed element and conjugating witnesses;
exactness preservation-reflection of centralizer membership; standard bridge
ordinary normalizer subgroup; status $\mathsf{ProofTarget}$.

\begin{theorem}[RatUp FieldUp affine centralizer normalizer action automorphism]
\label{thm:ratup-fieldup-affine-centralizer-normalizer-action-automorphism}
For Definition~\ref{def:ratup-fieldup-concrete-exit-object}, fix a certified
affine automorphism $A:\mathsf{AffAut}_F$ in the concrete $\FieldUp(\RatUp)$
exit object. Let $C_F(A)$ be the affine centralizer carrier from
\autoref{def:ratup-fieldup-affine-centralizer-carrier}, let $N_F(C_F(A))$ be
the normalizer carrier from
\autoref{def:ratup-fieldup-affine-centralizer-normalizer-carrier}, and define
the restricted action by
$$
\mathrm{Act}_T(X):=\mathrm{Conj}_T(X)
=(T\circ_{\mathrm{aff}}X)\circ_{\mathrm{aff}}\iota_F(T).
$$
If $N_F(C_F(A))(T)$ and $C_F(A)(X),C_F(A)(Y)$, then conjugation by $T$ is an
automorphism of the centralizer classifier:
\begin{enumerate}
\item $N_F(C_F(A))(\iota_F(T))$.
\item $C_F(A)(\mathrm{Act}_{\iota_F(T)}(X))$.
\item
$$
\mathrm{Act}_{\iota_F(T)}(\mathrm{Act}_T(X))\equiv_{\mathrm{aff}}X
\quad\text{and}\quad
\mathrm{Act}_T(\mathrm{Act}_{\iota_F(T)}(X))\equiv_{\mathrm{aff}}X.
$$
\item
$$
\mathrm{Act}_T(X)\equiv_{\mathrm{aff}}\mathrm{Act}_T(Y)
\Longleftrightarrow X\equiv_{\mathrm{aff}}Y.
$$
\end{enumerate}
\end{theorem}
\begin{proof}
The inverse normalizer witness is the inverse-closure row of
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-subgroup}. Applied to
$N_F(C_F(A))(T)$, it gives $N_F(C_F(A))(\iota_F(T))$.

Carrier preservation for the inverse action is then exactly the preservation
part of \autoref{thm:ratup-fieldup-affine-centralizer-normalizer-action},
specialized to the normalizer element $\iota_F(T)$ and the centralizer element
$X$. This gives $C_F(A)(\mathrm{Act}_{\iota_F(T)}(X))$.

For cancellation, use
\autoref{thm:ratup-fieldup-affine-conjugation-exactness} with conjugating
element $T$ and acted element $X$. Its inverse-action clauses identify the two
double conjugations by $T$ and $\iota_F(T)$ with $X$ under
$\equiv_{\mathrm{aff}}$, which are precisely the two displayed
$\mathrm{Act}$-cancellation laws.

For classifier exactness, apply the same theorem to the pair $X,Y$. Its
classifier-exact conjugation clause gives
$$
\mathrm{Conj}_T(X)\equiv_{\mathrm{aff}}\mathrm{Conj}_T(Y)
\Longleftrightarrow X\equiv_{\mathrm{aff}}Y.
$$
Replacing $\mathrm{Conj}_T(-)$ by the definition of $\mathrm{Act}_T(-)$ gives
the stated equivalence on the centralizer carrier.
\end{proof}

Per-object template: traditional target normalizer action by automorphisms;
BEDC source concrete $\FieldUp(\RatUp)$ plus $\GroupUp$; carrier $C_F(A)$;
classifier $\equiv_{\mathrm{aff}}$; stability inverse action, classifier
exactness, and cancellation; ledger fixed element, normalizer witness, acted
element, and inverse witness; core theorem list
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-subgroup},
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-action}, and
\autoref{thm:ratup-fieldup-affine-conjugation-exactness}; exactness predicate
classifier-exact automorphism action on the centralizer; standard bridge
ordinary normalizer action by automorphisms; status $\mathsf{ProofTarget}$.

\begin{theorem}[RatUp FieldUp affine centralizer normalizer action]
\label{thm:ratup-fieldup-affine-centralizer-normalizer-action}
For Definition~\ref{def:ratup-fieldup-concrete-exit-object}, fix a certified
affine automorphism $A:\mathsf{AffAut}_F$ in the concrete
$\FieldUp(\RatUp)$ exit object. Let $C_F(A)$ be the affine centralizer carrier

\autoref{def:ratup-fieldup-affine-centralizer-normalizer-carrier}. For
$T:\mathsf{AffAut}_F$ define the restricted normalizer action on centralizer
elements by
$$
\mathrm{Act}_T(X):=\mathrm{Conj}_T(X)
=(T\circ_{\mathrm{aff}}X)\circ_{\mathrm{aff}}\iota_F(T).
$$
If $N_F(C_F(A))(T)$ and $C_F(A)(X)$, then
$$
C_F(A)(\mathrm{Act}_T(X)).
$$
The restricted action also obeys the identity and composition laws:
$$
\mathrm{Act}_{\mathrm{id}_F}(X)\equiv_{\mathrm{aff}}X.
$$
For $N_F(C_F(A))(T)$, $N_F(C_F(A))(U)$, and $C_F(A)(X)$,
$$
\mathrm{Act}_T(\mathrm{Act}_U(X))\equiv_{\mathrm{aff}}
\mathrm{Act}_{T\circ_{\mathrm{aff}}U}(X).
$$
\end{theorem}
\begin{proof}
Unfold the normalizer carrier from
\autoref{def:ratup-fieldup-affine-centralizer-normalizer-carrier}. Its
preservation clause says exactly that a normalizing affine automorphism sends
each $C_F(A)$-element to another $C_F(A)$-element by the concrete conjugation
$\mathrm{Conj}_T(-)$. Since $\mathrm{Act}_T(X)$ is defined to be this conjugate,
$N_F(C_F(A))(T)$ and $C_F(A)(X)$ give
$C_F(A)(\mathrm{Act}_T(X))$.

For the identity law, apply
\autoref{thm:ratup-fieldup-affine-inner-action-composition-exit} with
conjugating parameter $\mathrm{id}_F$ and acted element $X$. The identity clause
of that theorem gives
$\mathrm{Act}_{\mathrm{id}_F}(X)\equiv_{\mathrm{aff}}X$ under the same affine
classifier.

For composition, first use the preservation clause of $U$ to obtain
$C_F(A)(\mathrm{Act}_U(X))$. Then use the preservation clause of $T$ to keep the
outer acted element inside $C_F(A)$. The composition clause of
\autoref{thm:ratup-fieldup-affine-inner-action-composition-exit}, applied to
$T$, $U$, and $X$, identifies the two affine words as
$$
\mathrm{Act}_T(\mathrm{Act}_U(X))\equiv_{\mathrm{aff}}
\mathrm{Act}_{T\circ_{\mathrm{aff}}U}(X).
$$

The carrier membership clauses come from the two normalizer hypotheses, while
the classifier comparison is the concrete inner-action law already supplied for
the $\FieldUp(\RatUp)$ affine exit object. The inverse witnesses are the
$\iota_F(T)$ and $\iota_F(U)$ rows fixed by the canonical affine inverse
package, and the acted elements remain in the same centralizer carrier
throughout.
\end{proof}

Per-object template: traditional target normalizer action on centralizer; BEDC

\begin{theorem}[RatUp FieldUp restricted normalizer action automorphism and kernel exactness]

\autoref{thm:ratup-fieldup-canonical-exit-action-inverse-package} identify the

\end{proof}

\begin{theorem}[RatUp FieldUp affine centralizer normalizer action representation]

the restricted action by
$$
\mathrm{Act}_T(X):=\mathrm{Conj}_T(X)
=(T\circ_{\mathrm{aff}}X)\circ_{\mathrm{aff}}\iota_F(T).
$$
The assignment
$$
T\longmapsto \mathrm{Act}_T
$$
from $N_F(C_F(A))$ to classifier-exact automorphisms of $C_F(A)$ is a concrete
homomorphic action representation. Concretely, if
$N_F(C_F(A))(T)$, $N_F(C_F(A))(U)$, and $C_F(A)(X),C_F(A)(Y)$, then:
\begin{enumerate}
\item $C_F(A)(\mathrm{Act}_T(X))$.
\item
$$
\mathrm{Act}_T(X)\equiv_{\mathrm{aff}}\mathrm{Act}_T(Y)
\Longleftrightarrow X\equiv_{\mathrm{aff}}Y.
$$
\item
$$
\mathrm{Act}_{\mathrm{id}_F}(X)\equiv_{\mathrm{aff}}X.
$$
\item
$$
\mathrm{Act}_T(\mathrm{Act}_U(X))\equiv_{\mathrm{aff}}
\mathrm{Act}_{T\circ_{\mathrm{aff}}U}(X).
$$
\item $N_F(C_F(A))(\iota_F(T))$, and the inverse action is two-sided:
$$
\mathrm{Act}_{\iota_F(T)}(\mathrm{Act}_T(X))\equiv_{\mathrm{aff}}X
\quad\text{and}\quad
\mathrm{Act}_T(\mathrm{Act}_{\iota_F(T)}(X))\equiv_{\mathrm{aff}}X.
$$
\end{enumerate}
\end{theorem}
\begin{proof}
The carrier and homomorphism laws are supplied by
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-action}. Its
preservation clause gives $C_F(A)(\mathrm{Act}_T(X))$, its identity clause gives
the action of $\mathrm{id}_F$, and its composition clause gives the displayed
law for $T\circ_{\mathrm{aff}}U$.

The subgroup certificate
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-subgroup} supplies the
normalizer bookkeeping for the representing source: identity membership, product
closure, and the inverse witness $N_F(C_F(A))(\iota_F(T))$.

For the target automorphism attached to $T$, apply
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-action-automorphism}.
Its classifier-exactness clause gives the displayed equivalence on
$\equiv_{\mathrm{aff}}$, and its inverse-action clauses give the two-sided
cancellation formulas.

Finally,
\autoref{thm:ratup-fieldup-affine-conjugation-exactness} identifies these
clauses with concrete affine conjugation in the $\FieldUp(\RatUp)$ exit object.
Thus the map $T\mapsto\mathrm{Act}_T$ lands in the classifier-exact
automorphisms of $C_F(A)$ and respects the normalizer multiplication by the
stated composition law.
\end{proof}

\begin{theorem}[RatUp FieldUp affine centralizer normalizer action certificate]
\label{thm:ratup-fieldup-affine-centralizer-normalizer-action-certificate}

\autoref{def:ratup-fieldup-affine-centralizer-normalizer-carrier}. For each
$T:\mathsf{AffAut}_F$ define
$$
\mathrm{Act}_T(X):=(T\circ_{\mathrm{aff}}X)\circ_{\mathrm{aff}}\iota_F(T).
$$

$T\equiv_{\mathrm{aff}}U$, and $X\equiv_{\mathrm{aff}}Y$, then the restricted
action is classifier well-defined:
$$
\mathrm{Act}_T(X)\equiv_{\mathrm{aff}}\mathrm{Act}_U(Y).
$$
Moreover, for every $T$ with $N_F(C_F(A))(T)$, the inverse affine automorphism
is again a normalizer element,
$$
N_F(C_F(A))(\iota_F(T)),
$$
and for every $X$ with $C_F(A)(X)$ the two restricted actions cancel:
$$
\mathrm{Act}_{\iota_F(T)}(\mathrm{Act}_T(X))\equiv_{\mathrm{aff}}X
\quad\text{and}\quad
\mathrm{Act}_T(\mathrm{Act}_{\iota_F(T)}(X))\equiv_{\mathrm{aff}}X.
$$
\end{theorem}
\begin{proof}
Fix the concrete affine exit package supplied by
\autoref{thm:ratup-fieldup-canonical-exit-action-inverse-package}. It fixes the
carrier $\mathsf{AffAut}_F$, the operation $\circ_{\mathrm{aff}}$, the inverse
operation $\iota_F(-)$, and the classifier $\equiv_{\mathrm{aff}}$ for the
concrete $\FieldUp(\RatUp)$ affine exit object.

For classifier well-definedness, first apply
\autoref{thm:field-affine-pointwise-classifier-operation-congruence} to the
left action factor and the acted element. This transports
$T\circ_{\mathrm{aff}}X$ to $U\circ_{\mathrm{aff}}Y$ under the affine
classifier. The same congruence theorem, applied to the right inverse factors
and using the inverse determinacy encoded by
\autoref{thm:ratup-fieldup-affine-inner-conjugation-exit}, transports the full
conjugation word
$(T\circ_{\mathrm{aff}}X)\circ_{\mathrm{aff}}\iota_F(T)$ to
$(U\circ_{\mathrm{aff}}Y)\circ_{\mathrm{aff}}\iota_F(U)$. This is exactly
$\mathrm{Act}_T(X)\equiv_{\mathrm{aff}}\mathrm{Act}_U(Y)$.

For the inverse normalizer clause, apply the inverse row of
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-subgroup} to
$N_F(C_F(A))(T)$. It gives $N_F(C_F(A))(\iota_F(T))$.

Finally, use
\autoref{thm:ratup-fieldup-affine-conjugation-equation-exactness} with
conjugating element $T$ and acted element $X$. Its two inverse-action
cancellation clauses identify conjugation by $\iota_F(T)$ after conjugation by
$T$, and conjugation by $T$ after conjugation by $\iota_F(T)$, with the original
acted element $X$ under $\equiv_{\mathrm{aff}}$. Since $\mathrm{Act}$ is the
restricted notation for these conjugation words on $C_F(A)$, the displayed
restricted cancellation laws follow.
\end{proof}
