\begin{definition}[RatUp FieldUp concrete exit object]
\label{def:ratup-fieldup-concrete-exit-object}
For $\mathsf{RatUpFieldSource}_F$, the concrete
$\FieldUp(\RatUp)$ exit object is the finite row package over
$\mathsf{BHist}$ whose carrier and classifier rows are
$\mathsf{RatHistoryCarrier}$, $\mathsf{RatHistoryClassifier}$,
$\mathsf{RatDenomUnitCarrier}$, and
$\mathsf{RatDenomUnitClassifier}$.

Its operation rows are exactly the concrete table rows for $\mathsf{append}$,
$\Cont$, the denominator unit, the contextual action $\Gamma$, the
context-pair action, affine rows, inverse and nonzero rows, one-sided
multiplication cancellation exactness, two-sided product exactness, the
same-endpoint plus crossed singleton-classifier noncollapse rows, and the
selector-indexed bracketing rows generated by $E_s(h,k,\ell)$: selected
endpoint carrier coverage, selected endpoint denominator-classifier coverage,
and selector-indexed singleton-classifier exclusion under arbitrary contexts.
No schematic $\mathsf{mul}$, $\mathsf{add}$, or $\mathsf{neg}$ parameter is
part of this object.
\end{definition}

\begin{lemma}[RatUp FieldUp BMark selector endpoint-pair cases]
\label{lem:ratup-fieldup-bracketing-selector-pair-cases}

$x:=\mu_D^u(\mu_D^u(h,k),\ell)$ and

$s,t:\mathsf{BMark}$, the selected endpoint pair
$(E_s(h,k,\ell),E_t(h,k,\ell))$ is covered by the four generated rows
$(x,x)$, $(y,y)$, $(x,y)$, and $(y,x)$ up to $\hsame$ on both endpoints.
\end{lemma}
\begin{proof}
Perform generated case analysis on the selector pair $(s,t)$. In every branch,
unfold the endpoint selector of
\autoref{def:ratup-fieldup-bracketing-endpoint-selector}; the remaining endpoint
comparison is reflexivity of $\hsame$.

In the $(\mathsf{BMark.b0},\mathsf{BMark.b0})$ branch, both selected endpoints
unfold to $x$, so the row $(x,x)$ supplies the pair witness.

In the $(\mathsf{BMark.b1},\mathsf{BMark.b1})$ branch, both selected endpoints
unfold to $y$, so the row $(y,y)$ supplies the pair witness.

In the $(\mathsf{BMark.b0},\mathsf{BMark.b1})$ branch, the left endpoint
unfolds to $x$ and the right endpoint unfolds to $y$, giving the row $(x,y)$.

In the $(\mathsf{BMark.b1},\mathsf{BMark.b0})$ branch, the left endpoint
unfolds to $y$ and the right endpoint unfolds to $x$, giving the row $(y,x)$.
\end{proof}

\label{lem:ratup-fieldup-bracketing-endpoint-selector-exhaustion}

\end{lemma}
\begin{proof}
Apply $\mathsf{BMark\_generated\_cases}$ to $s$ and then to $t$. This gives
four closed generated selector pairs.

\begin{enumerate}
  \item If $s=\mathsf{BMark.b0}$ and $t=\mathsf{BMark.b0}$, unfold
  \autoref{def:ratup-fieldup-bracketing-endpoint-selector}. Then
  $E_s(h,k,\ell)=x$ and $E_t(h,k,\ell)=x$, so the selected pair is $(x,x)$.

  \item If $s=\mathsf{BMark.b0}$ and $t=\mathsf{BMark.b1}$, unfold
  \autoref{def:ratup-fieldup-bracketing-endpoint-selector}. Then
  $E_s(h,k,\ell)=x$ and $E_t(h,k,\ell)=y$, so the selected pair is $(x,y)$.

  \item If $s=\mathsf{BMark.b1}$ and $t=\mathsf{BMark.b0}$, unfold
  \autoref{def:ratup-fieldup-bracketing-endpoint-selector}. Then
  $E_s(h,k,\ell)=y$ and $E_t(h,k,\ell)=x$, so the selected pair is $(y,x)$.

  \item If $s=\mathsf{BMark.b1}$ and $t=\mathsf{BMark.b1}$, unfold
  \autoref{def:ratup-fieldup-bracketing-endpoint-selector}. Then
  $E_s(h,k,\ell)=y$ and $E_t(h,k,\ell)=y$, so the selected pair is $(y,y)$.
\end{enumerate}

These four branches exhaust the generated $\mathsf{BMark}$ selectors, so no

\end{proof}

\begin{theorem}[RatUp FieldUp selector-complete denominator strict support package]
\label{thm:ratup-fieldup-selector-complete-denominator-strict-support-package}

histories $h,k,\ell:\mathsf{BHist}$ with $C_D^u(h)$, $C_D^u(k)$, and
$C_D^u(\ell)$, let $E_s(h,k,\ell)$ be the endpoint selector of
\autoref{def:ratup-fieldup-bracketing-endpoint-selector}. If
$\mathsf{ThreeSupp}_D^u(h,k,\ell)$ holds, then for all selectors
$s,t:\mathsf{BMark}$ the selector-complete denominator instance of
\autoref{def:ratup-fieldup-selector-complete-denominator-instance} supplies
the strict-support endpoint package
$$
R_D^u(E_s(h,k,\ell),E_t(h,k,\ell))
\land
\mathsf{fieldSingletonEmptyNonZero}(E_s(h,k,\ell))
\land
\mathsf{fieldSingletonEmptyNonZero}(E_t(h,k,\ell)).
$$
\end{theorem}
\begin{proof}
Apply \autoref{thm:ratup-fieldup-selector-complete-denominator-well-formed-exit-slice} to the
same concrete denominator tuple, carrier hypotheses, support witness, and
selectors $s,t:\mathsf{BMark}$. Its classifier row gives
$R_D^u(E_s(h,k,\ell),E_t(h,k,\ell))$.

For the left strict-support endpoint, specialize the obstruction row of
\autoref{thm:ratup-fieldup-selector-complete-denominator-well-formed-exit-slice} to the selector
pair $(s,s)$ and to contexts $L=R=\emp$. Assume a forbidden witness
$\mathsf{fieldSingletonEmptyClassifier}(E_s(h,k,\ell),\emp)$. The empty-left
append unit transports this witness to a singleton-empty classifier on
$\mathsf{append}(\emp,E_s(h,k,\ell))$ against itself, contradicting the
specialized obstruction. By
\autoref{def:field-singleton-empty-nonzero}, this is exactly
$\mathsf{fieldSingletonEmptyNonZero}(E_s(h,k,\ell))$.

For the right strict-support endpoint, repeat the same argument with the
selector pair $(t,t)$ and contexts $L=R=\emp$. A forbidden classifier
$\mathsf{fieldSingletonEmptyClassifier}(E_t(h,k,\ell),\emp)$ transports along
the empty-left append unit to the obstructed classifier on
$\mathsf{append}(\emp,E_t(h,k,\ell))$ against itself. The resulting
contradiction is the required
$\mathsf{fieldSingletonEmptyNonZero}(E_t(h,k,\ell))$.

Combining the classifier row with the two singleton-empty complement witnesses
gives the displayed strict-support endpoint package.
\end{proof}

\begin{theorem}[RatUp FieldUp selector-complete exit package]
\label{thm:ratup-fieldup-selector-complete-exit-package}
For the concrete denominator tuple $(C_D^u,R_D^u,\mu_D^u,e_D^u,N_D^u)$ and
histories $h,k,\ell:\mathsf{BHist}$ with $C_D^u(h)$, $C_D^u(k)$, and

\autoref{def:ratup-fieldup-bracketing-endpoint-selector}. If
$\mathsf{ThreeSupp}_D^u(h,k,\ell)$ holds, then the selector-complete
denominator instance of
\autoref{def:ratup-fieldup-selector-complete-denominator-instance} supplies a
concrete $\FieldUp(\mathsf{RatUp})$ exit-package row over
$\mathsf{RatUpFieldSource}_F$: for every $s,t:\mathsf{BMark}$ and every
$L,R:\mathsf{BHist}$ it provides selected endpoint carrier coverage, the
denominator classifier row
$R_D^u(E_s(h,k,\ell),E_t(h,k,\ell))$, and the contextual singleton-classifier
noncollapse row
$$
\mathsf{fieldSingletonEmptyClassifier}
(\mathsf{append}(L,E_s(h,k,\ell)),\mathsf{append}(R,E_t(h,k,\ell)))\to\bot.
$$
This selector row is one of the concrete denominator rows consumed by
\autoref{thm:ratup-fieldup-algebra-hierarchy-exit-witness}, and it is read in
the denominator coordinate of $\mathsf{RatUpFieldSource}_F$. No schematic
$\mathsf{mul}$, $\mathsf{add}$, or $\mathsf{neg}$ parameter enters the row.
\end{theorem}
\begin{proof}
Apply \autoref{thm:ratup-fieldup-selector-complete-denominator-laws} to the
concrete denominator coordinate of $\mathsf{RatUpFieldSource}_F$, the three
carrier hypotheses, and
$\mathsf{ThreeSupp}_D^u(h,k,\ell)$. This gives, for arbitrary
$s,t:\mathsf{BMark}$ and $L,R:\mathsf{BHist}$, the selected endpoint carrier
rows, the selected endpoint classifier row, and the contextual singleton-empty
obstruction.

Apply \autoref{thm:ratup-fieldup-algebra-hierarchy-exit-witness}. It records
the denominator classifier coverage and singleton-classifier noncollapse rows
as concrete rows inside the $\FieldUp(\mathsf{RatUp})$ hierarchy exit package
over $\mathsf{RatUpFieldSource}_F$.

The row is therefore an admissible concrete exit-package component. Its data
are the selected denominator endpoints inside $\mathsf{BHist}$, the carrier
coverage, the denominator classifier coverage, and contextual noncollapse over
the source-aligned denominator coordinate, so no schematic $\mathsf{mul}$,
$\mathsf{add}$, or $\mathsf{neg}$ operator parameter is introduced.
\end{proof}

\begin{theorem}[RatUp FieldUp carrier-aware selector-complete concrete instance]
\label{thm:ratup-fieldup-carrier-aware-selector-complete-concrete-instance}
For the concrete denominator tuple $(C_D^u,R_D^u,\mu_D^u,e_D^u,N_D^u)$ and
histories $h,k,\ell:\mathsf{BHist}$ with $C_D^u(h)$, $C_D^u(k)$, and
$C_D^u(\ell)$, let $E_s(h,k,\ell)$ be the endpoint selector of
\autoref{def:ratup-fieldup-bracketing-endpoint-selector}. If
$\mathsf{ThreeSupp}_D^u(h,k,\ell)$ holds, then for every pair of selectors
$s,t:\mathsf{BMark}$ and every pair of contexts $L,R:\mathsf{BHist}$, the
selector-complete denominator instance supplies all carrier-aware rows
\begin{enumerate}
  \item $C_D^u(E_s(h,k,\ell))$;
  \item $C_D^u(E_t(h,k,\ell))$;
  \item $R_D^u(E_s(h,k,\ell),E_t(h,k,\ell))$;
  \item
  $$
  \mathsf{fieldSingletonEmptyClassifier}
  (\mathsf{append}(L,E_s(h,k,\ell)),\mathsf{append}(R,E_t(h,k,\ell)))\to\bot.
  $$
\end{enumerate}
Together these rows form a concrete carrier-aware $\FieldUp(\RatUp)$ exit row
over $\mathsf{RatUpFieldSource}_F$.
\end{theorem}
\begin{proof}
Apply \autoref{thm:ratup-fieldup-bracketing-selector-exit-subpackage} to the
concrete denominator tuple, the carrier hypotheses on $h,k,\ell$, and the
support witness $\mathsf{ThreeSupp}_D^u(h,k,\ell)$. For the chosen
$s,t:\mathsf{BMark}$ and contexts $L,R:\mathsf{BHist}$, it supplies the two
selected-endpoint carrier rows, the denominator-classifier row, and the
contextual singleton-empty obstruction.

Apply \autoref{thm:ratup-fieldup-selector-complete-exit-package} to the same
concrete tuple, carrier hypotheses, and support witness. This places the same
selected denominator row inside the concrete $\FieldUp(\RatUp)$ exit package
over $\mathsf{RatUpFieldSource}_F$.

The selector-complete denominator instance therefore carries both endpoint
coverage and package membership: the two selected endpoints remain in $C_D^u$,
the selected pair lies in $R_D^u$, and every contextual singleton-empty
classifier on the selected pair collapses to $\bot$.
\end{proof}

\begin{theorem}[RatUp FieldUp selector-complete denominator row exhaustion]
\label{thm:ratup-fieldup-selector-complete-denominator-row-exhaustion}
For the concrete denominator tuple $(C_D^u,R_D^u,\mu_D^u,e_D^u,N_D^u)$ and
histories $h,k,\ell:\mathsf{BHist}$ with $C_D^u(h)$, $C_D^u(k)$, and
$C_D^u(\ell)$, let $E_s(h,k,\ell)$ be the endpoint selector of
\autoref{def:ratup-fieldup-bracketing-endpoint-selector}. Fix contexts
$L,R:\mathsf{BHist}$ and assume $\mathsf{ThreeSupp}_D^u(h,k,\ell)$. Set
$x:=\mu_D^u(\mu_D^u(h,k),\ell)$ and
$y:=\mu_D^u(h,\mu_D^u(k,\ell))$. Then the selected denominator row of
Definition~\ref{def:ratup-fieldup-selector-complete-denominator-instance} is
exhausted by exactly the following four ordered endpoint rows:
\begin{enumerate}
  \item $(x,x)$, with carrier rows $C_D^u(x)$ and $C_D^u(x)$, classifier row
  $R_D^u(x,x)$, and obstruction
  $$
  \mathsf{fieldSingletonEmptyClassifier}
  (\mathsf{append}(L,x),\mathsf{append}(R,x))\to\bot;
  $$
  \item $(x,y)$, with carrier rows $C_D^u(x)$ and $C_D^u(y)$, classifier row
  $R_D^u(x,y)$, and obstruction
  $$
  \mathsf{fieldSingletonEmptyClassifier}
  (\mathsf{append}(L,x),\mathsf{append}(R,y))\to\bot;
  $$
  \item $(y,x)$, with carrier rows $C_D^u(y)$ and $C_D^u(x)$, classifier row
  $R_D^u(y,x)$, and obstruction
  $$
  \mathsf{fieldSingletonEmptyClassifier}
  (\mathsf{append}(L,y),\mathsf{append}(R,x))\to\bot;
  $$
  \item $(y,y)$, with carrier rows $C_D^u(y)$ and $C_D^u(y)$, classifier row
  $R_D^u(y,y)$, and obstruction
  $$
  \mathsf{fieldSingletonEmptyClassifier}
  (\mathsf{append}(L,y),\mathsf{append}(R,y))\to\bot.
  $$
\end{enumerate}
For every selector pair $s,t:\mathsf{BMark}$, the row selected by
$(E_s(h,k,\ell),E_t(h,k,\ell))$ is one of these four rows, and its two carrier
rows, $R_D^u$ classifier row, and singleton-empty contextual obstruction are
the corresponding rows supplied by the selector-complete $\FieldUp(\mathsf{RatUp})$
exit package. Conversely, the four displayed ordered endpoint rows are realized
by the selector pairs $(\mathsf{BMark.b0},\mathsf{BMark.b0})$,
$(\mathsf{BMark.b0},\mathsf{BMark.b1})$,
$(\mathsf{BMark.b1},\mathsf{BMark.b0})$, and
$(\mathsf{BMark.b1},\mathsf{BMark.b1})$.
\end{theorem}
\begin{proof}
Apply Lemma~\ref{lem:ratup-fieldup-bracketing-endpoint-selector-exhaustion} to
the arbitrary selectors $s,t:\mathsf{BMark}$. It gives the four generated
selector branches and identifies their selected endpoint pairs as $(x,x)$,
$(x,y)$, $(y,x)$, and $(y,y)$.

Apply Theorem~\ref{thm:ratup-fieldup-selector-complete-denominator-well-formed-exit-slice}
to the concrete denominator tuple, the three carrier hypotheses, the support
witness $\mathsf{ThreeSupp}_D^u(h,k,\ell)$, the selected pair $s,t$, and the
contexts $L,R$. This supplies the two carrier rows, the $R_D^u$ classifier row,
and the singleton-empty contextual obstruction for the selected denominator
row.

Apply Theorem~\ref{thm:ratup-fieldup-selector-complete-exit-package} to the
same data. The classifier row and the contextual obstruction are recorded as
concrete $\FieldUp(\mathsf{RatUp})$ exit-package rows over
$\mathsf{RatUpFieldSource}_F$, so each of the four cases is an admissible row
of the concrete exit package.

It remains to account for the converse realization. Specialize
Lemma~\ref{lem:ratup-fieldup-bracketing-endpoint-selector-exhaustion} to the
four selector pairs
$(\mathsf{BMark.b0},\mathsf{BMark.b0})$,
$(\mathsf{BMark.b0},\mathsf{BMark.b1})$,
$(\mathsf{BMark.b1},\mathsf{BMark.b0})$, and
$(\mathsf{BMark.b1},\mathsf{BMark.b1})$. Unfolding
\autoref{def:ratup-fieldup-bracketing-endpoint-selector} gives respectively
the endpoint rows $(x,x)$, $(x,y)$, $(y,x)$, and $(y,y)$, and the preceding
well-formedness and exit-package applications attach the corresponding four
rows displayed in the statement.
\end{proof}
