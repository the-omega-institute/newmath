\begin{definition}[RatUp FieldUp concrete exit object]
\label{def:ratup-fieldup-concrete-exit-object}
For $\mathsf{RatUpFieldSource}_F$, the concrete
$\FieldUp(\RatUp)$ exit object is the finite row package over
$\mathsf{BHist}$ whose carrier and classifier rows are
$\mathsf{RatHistoryCarrier}$, $\mathsf{RatHistoryClassifier}$,
$\mathsf{RatDenomUnitCarrier}$, and
$\mathsf{RatDenomUnitClassifier}$.

Its operation rows are exactly the concrete table rows for $\mathsf{append}$,
$\Cont$, the denominator unit, the contextual action $\Gamma$, the
context-pair action, affine rows, inverse and nonzero rows, one-sided
multiplication cancellation exactness, two-sided product exactness, the
same-endpoint plus crossed singleton-classifier noncollapse rows, and the
selector-indexed bracketing rows generated by $E_s(h,k,\ell)$: selected
endpoint carrier coverage, selected endpoint denominator-classifier coverage,
and selector-indexed singleton-classifier exclusion under arbitrary contexts.
No schematic $\mathsf{mul}$, $\mathsf{add}$, or $\mathsf{neg}$ parameter is
part of this object.
\end{definition}

\begin{theorem}[RatUp FieldUp concrete exit object is complete]
\label{thm:ratup-fieldup-concrete-exit-object-complete}
For $\mathsf{RatUpFieldSource}_F$,
Definition~\ref{def:ratup-fieldup-concrete-exit-object} is a complete
paper-side $\FieldUp(\RatUp)$ exit certificate over $\mathsf{BHist}$. Its
source, carrier, classifier, stability obligations, ledger, core theorem list,
exactness predicate, standard-bridge slot, and status are supplied by
\autoref{thm:ratup-fieldup-concrete-schema-instance},
\autoref{thm:ratup-fieldup-concrete-operation-exit-exhaustion}, and
\autoref{thm:ratup-fieldup-algebra-hierarchy-exit-witness}.

Every row in this certificate is concrete BEDC data carried by the displayed
$\mathsf{BHist}$ source tuple and denominator endpoints. No schematic
$\mathsf{mul}$, $\mathsf{add}$, or $\mathsf{neg}$ operator parameter remains.
\end{theorem}
\begin{proof}
Apply \autoref{thm:ratup-fieldup-concrete-schema-instance}. It fixes the single
$\mathsf{BHist}$ source tuple for $\mathsf{RatUpFieldSource}_F$ and identifies
the carrier, classifier, source, stability, ledger, and exactness rows consumed
by the concrete certificate instance.

Next apply
\autoref{thm:ratup-fieldup-concrete-operation-exit-exhaustion}. This covers
every operation row named by
Definition~\ref{def:ratup-fieldup-concrete-exit-object}, including the carrier
and classifier anchors, denominator rows, contextual rows, affine rows, inverse
and nonzero rows, product exactness, and singleton-classifier noncollapse rows.

Finally apply \autoref{thm:ratup-fieldup-algebra-hierarchy-exit-witness}. It
packages those rows as the Phase Algebra Hierarchy paper certificate for
$\FieldUp(\RatUp)$, with the bridge slot and status recorded as part of the
same paper-side object. Thus the concrete exit object is complete and has no
remaining schematic operation parameter.
\end{proof}

\begin{theorem}[RatUp FieldUp concrete exit object hierarchy exactness]
\label{thm:ratup-fieldup-concrete-exit-object-hierarchy-exactness}
For $\mathsf{RatUpFieldSource}_F$, the finite row package of
Definition~\ref{def:ratup-fieldup-concrete-exit-object} is exactly the concrete
$\FieldUp(\RatUp)$ hierarchy exit package. Its carrier and classifier rows are
the concrete $\mathsf{RatHistory}$ and $\mathsf{RatDenomUnit}$ rows, and its
operation and exactness rows are precisely those certified by
\autoref{thm:ratup-fieldup-concrete-operation-exit-exhaustion} and
\autoref{thm:ratup-fieldup-algebra-hierarchy-exit-witness}.
\end{theorem}
\begin{proof}
Apply \autoref{thm:ratup-fieldup-concrete-operation-exit-exhaustion}. It
exhausts the row inventory of
Definition~\ref{def:ratup-fieldup-concrete-exit-object}: the carrier and
classifier anchoring rows, the $\CommRingUp$ base rows, denominator
multiplication and unit rows, contextual and context-pair rows, affine rows,
inverse and nonzero rows, two-sided product exactness, and same-endpoint plus
crossed singleton-classifier noncollapse rows.

Apply \autoref{thm:ratup-fieldup-algebra-hierarchy-exit-witness}. It packages
the same concrete rows as the algebra-hierarchy exit witness for
$\FieldUp(\RatUp)$ over $\mathsf{RatUpFieldSource}_F$, including the ten-field
certificate, product classifier and nonzero transport, five-factor ledger,
cross-endpoint exclusions, and bracketing-complete classifier package.

Because both cited theorems use the same source tuple, $\mathsf{BHist}$ carrier,
denominator endpoints, and operation-row inventory, the finite package defined
in Definition~\ref{def:ratup-fieldup-concrete-exit-object} and the concrete
$\FieldUp(\RatUp)$ hierarchy exit package have the same rows and the same
exactness obligations.
\end{proof}

\begin{theorem}[RatUp FieldUp BMark-selected bracketing exit subpackage]
\label{thm:ratup-fieldup-bracketing-selector-exit-subpackage}
For the concrete denominator tuple $(C_D^u,R_D^u,\mu_D^u,e_D^u,N_D^u)$ and
histories $h,k,\ell:\mathsf{BHist}$ with $C_D^u(h)$, $C_D^u(k)$, and
$C_D^u(\ell)$, let $E_s(h,k,\ell)$ be the endpoint selector of
\autoref{def:ratup-fieldup-bracketing-endpoint-selector}. If
$\mathsf{ThreeSupp}_D^u(h,k,\ell)$ holds, then for all selectors
$s,t:\mathsf{BMark}$ and all contexts $L,R:\mathsf{BHist}$, the selected
endpoints form the following concrete exit subpackage:
\begin{enumerate}
  \item $C_D^u(E_s(h,k,\ell))$ and $C_D^u(E_t(h,k,\ell))$;
  \item $R_D^u(E_s(h,k,\ell),E_t(h,k,\ell))$;
  \item
  $$
  \mathsf{fieldSingletonEmptyClassifier}
  (\mathsf{append}(L,E_s(h,k,\ell)),\mathsf{append}(R,E_t(h,k,\ell)))\to\bot.
  $$
\end{enumerate}
\end{theorem}
\begin{proof}
Set $x:=\mu_D^u(\mu_D^u(h,k),\ell)$ and
$y:=\mu_D^u(h,\mu_D^u(k,\ell))$. Apply
\autoref{thm:ratup-fieldup-bracketing-complete-classifier-exit-certificate}
to the same concrete denominator tuple, carrier hypotheses, endpoint
definitions, and $\mathsf{ThreeSupp}_D^u(h,k,\ell)$ witness. This supplies
$C_D^u(x)$, $C_D^u(y)$, $R_D^u(x,y)$, and the four ordered endpoint-pair
singleton-classifier exclusions.

For carrier coverage, use the two generated $\mathsf{BMark}$ cases for $s$.
In the $\mathsf{BMark.b0}$ branch, the selected endpoint is $x$, so
$C_D^u(x)$ transports to $C_D^u(E_s(h,k,\ell))$. In the
$\mathsf{BMark.b1}$ branch, the selected endpoint is $y$, so $C_D^u(y)$
transports to $C_D^u(E_s(h,k,\ell))$. The same two-case argument for $t$
gives $C_D^u(E_t(h,k,\ell))$.

The displayed denominator-classifier row is exactly
\autoref{thm:ratup-fieldup-bracketing-selector-denominator-classifier-coverage}
applied to the same carrier data and support witness. It packages the
reflexive, symmetric, and crossed endpoint classifier rows across all selected
pairs $s,t:\mathsf{BMark}$.

The singleton-classifier rejection under arbitrary contexts is exactly
\autoref{thm:ratup-fieldup-bracketing-choice-classifier-exclusion} applied to
the same selector endpoints. Thus the selected pair $(s,t)$ and the contexts
$(L,R)$ carry both the internal denominator classifier row and the external
singleton-classifier exclusion required by the concrete exit subpackage.
\end{proof}

Per-object template: traditional target selector-indexed bracketing exit
subpackage for $\FieldUp(\mathsf{RatUp})$; BEDC source $\mathsf{BHist}$,
$\mathsf{BMark}$, $\mu_D^u$, $R_D^u$, and $\mathsf{ThreeSupp}_D^u$; carrier
$C_D^u$; classifier $R_D^u$; stability obligations the bracketing-complete
certificate plus selector coverage and selector exclusion; ledger the selected
endpoint pair $(s,t)$ and contexts $(L,R)$; exactness predicate total
selected-endpoint coverage and noncollapse; standard bridge deferred; status
$\mathsf{ProofTarget}$.

\begin{theorem}[RatUp FieldUp selector-complete denominator laws]
\label{thm:ratup-fieldup-selector-complete-denominator-laws}
For the concrete denominator tuple $(C_D^u,R_D^u,\mu_D^u,e_D^u,N_D^u)$ and
histories $h,k,\ell:\mathsf{BHist}$ with $C_D^u(h)$, $C_D^u(k)$, and
$C_D^u(\ell)$, let $E_s(h,k,\ell)$ be the endpoint selector of
\autoref{def:ratup-fieldup-bracketing-endpoint-selector}. If
$\mathsf{ThreeSupp}_D^u(h,k,\ell)$ holds, then for all selectors
$s,t:\mathsf{BMark}$ and all contexts $L,R:\mathsf{BHist}$ the
selector-complete denominator instance of
\autoref{def:ratup-fieldup-selector-complete-denominator-instance} satisfies
the two laws
$$
R_D^u(E_s(h,k,\ell),E_t(h,k,\ell)),
$$
and
$$
\mathsf{fieldSingletonEmptyClassifier}
(\mathsf{append}(L,E_s(h,k,\ell)),\mathsf{append}(R,E_t(h,k,\ell)))\to\bot.
$$
\end{theorem}
\begin{proof}
Apply
\autoref{thm:ratup-fieldup-bracketing-selector-denominator-classifier-coverage}
to the concrete denominator tuple, the three carrier hypotheses, and
$\mathsf{ThreeSupp}_D^u(h,k,\ell)$. This gives the selected endpoint
classifier row $R_D^u(E_s(h,k,\ell),E_t(h,k,\ell))$ for arbitrary
$s,t:\mathsf{BMark}$.

Apply \autoref{thm:ratup-fieldup-bracketing-choice-classifier-exclusion} to
the same denominator tuple, carrier hypotheses, support witness, selectors,
and contexts $L,R:\mathsf{BHist}$. This gives the obstruction row excluding
the displayed singleton-empty classifier on
$\mathsf{append}(L,E_s(h,k,\ell))$ and
$\mathsf{append}(R,E_t(h,k,\ell))$.

These two rows are exactly the selector slice consumed by
\autoref{thm:ratup-fieldup-algebra-hierarchy-exit-witness}: one row supplies
denominator classifier coverage for every selected endpoint pair, and the
other supplies contextual singleton-classifier noncollapse for the same
selector interface.
\end{proof}
