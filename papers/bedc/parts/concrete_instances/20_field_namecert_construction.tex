\begin{definition}[RatUp FieldUp affine normalizer action kernel carrier]
\label{def:ratup-fieldup-affine-normalizer-action-kernel-carrier}
For Definition~\ref{def:ratup-fieldup-concrete-exit-object}, fix a certified
affine automorphism $A:\mathsf{AffAut}_F$ in the concrete $\FieldUp(\RatUp)$
exit object. Let $C_F(A)$ be the affine centralizer carrier from
\autoref{def:ratup-fieldup-affine-centralizer-carrier}, let $N_F(C_F(A))$ be
the normalizer carrier from
\autoref{def:ratup-fieldup-affine-centralizer-normalizer-carrier}, and write
the restricted normalizer action as
$$
\mathrm{Act}_T(X):=(T\circ_{\mathrm{aff}}X)\circ_{\mathrm{aff}}\iota_F(T).
$$
The pointwise kernel carrier of this action is the predicate $K_A$ on certified
affine automorphisms given by
$$
K_A(T):=N_F(C_F(A))(T)\ \text{and}\quad
\forall X:\mathsf{AffAut}_F,\ C_F(A)(X)\Rightarrow
\mathrm{Act}_T(X)\equiv_{\mathrm{aff}}X.
$$
\end{definition}

\begin{theorem}[RatUp FieldUp affine normalizer action kernel normal subgroup]
\label{thm:ratup-fieldup-affine-normalizer-action-kernel-normal-subgroup}
For Definition~\ref{def:ratup-fieldup-concrete-exit-object}, fix
$A:\mathsf{AffAut}_F$. The carrier $K_A$ of
Definition~\ref{def:ratup-fieldup-affine-normalizer-action-kernel-carrier} is a
normal subgroup package inside the affine normalizer $N_F(C_F(A))$:
\begin{enumerate}
\item $K_A(\mathrm{id}_F)$.
\item If $K_A(T)$ and $K_A(U)$, then $K_A(T\circ_{\mathrm{aff}}U)$.
\item If $K_A(T)$, then $K_A(\iota_F(T))$.
\item If $K_A(T)$, $N_F(C_F(A))(U)$, and $T\equiv_{\mathrm{aff}}U$, then
$K_A(U)$.
\item If $K_A(T)$ and $N_F(C_F(A))(S)$, then
$K_A(S\circ_{\mathrm{aff}}T\circ_{\mathrm{aff}}\iota_F(S))$.
\end{enumerate}
\end{theorem}
\begin{proof}
The carrier part of each clause is supplied by
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-subgroup}. Its
identity, product, inverse, and classifier-transport rows keep the first four
displayed witnesses inside $N_F(C_F(A))$; for the conjugation word, use the
inverse row on $S$ and the product row twice.

For the identity element, use the identity law in
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-action}. For every
$X$ with $C_F(A)(X)$, it gives
$\mathrm{Act}_{\mathrm{id}_F}(X)\equiv_{\mathrm{aff}}X$, so the pointwise
kernel predicate holds.

For products, assume $K_A(T)$ and $K_A(U)$ and fix $X$ with $C_F(A)(X)$. The
composition law in
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-action} rewrites
$\mathrm{Act}_{T\circ_{\mathrm{aff}}U}(X)$ as
$\mathrm{Act}_T(\mathrm{Act}_U(X))$. The $U$-kernel hypothesis identifies the
inner acted element with $X$, and the $T$-kernel hypothesis then identifies the
outer action with $X$.

For inverses, start with $K_A(T)$. The inverse theorem
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-action-inverse}
identifies $\mathrm{Act}_{\iota_F(T)}$ as a two-sided inverse of
$\mathrm{Act}_T$ on $C_F(A)$. Since $\mathrm{Act}_T$ is pointwise the classifier
identity there, the inverse action is also pointwise the classifier identity.

For classifier transport, assume $K_A(T)$, $N_F(C_F(A))(U)$, and
$T\equiv_{\mathrm{aff}}U$. The action classifier congruence theorem
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-action-classifier-congruence}
transports the pointwise equality
$\mathrm{Act}_T(X)\equiv_{\mathrm{aff}}X$ to
$\mathrm{Act}_U(X)\equiv_{\mathrm{aff}}X$ for each $X$ in $C_F(A)$.

Finally assume $K_A(T)$ and $N_F(C_F(A))(S)$. For any $X$ with $C_F(A)(X)$,
apply the composition law twice to rewrite
$$
\mathrm{Act}_{S\circ_{\mathrm{aff}}T\circ_{\mathrm{aff}}\iota_F(S)}(X)
$$
as
$$
\mathrm{Act}_S(\mathrm{Act}_T(\mathrm{Act}_{\iota_F(S)}(X))).
$$

The inverse theorem keeps $\mathrm{Act}_{\iota_F(S)}(X)$ inside $C_F(A)$. Apply
the $K_A(T)$ hypothesis to this middle acted element, obtaining
$\mathrm{Act}_T(\mathrm{Act}_{\iota_F(S)}(X))
\equiv_{\mathrm{aff}}\mathrm{Act}_{\iota_F(S)}(X)$. Acting by $S$ on both sides
and using classifier congruence reduces the displayed conjugation action to
$\mathrm{Act}_S(\mathrm{Act}_{\iota_F(S)}(X))$.

The right inverse cancellation clause of
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-action-inverse}
identifies this last term with $X$. Hence
$S\circ_{\mathrm{aff}}T\circ_{\mathrm{aff}}\iota_F(S)$ lies in $K_A$, giving
normality inside the concrete affine normalizer.
\end{proof}

\begin{theorem}[RatUp FieldUp restricted normalizer action kernel fiber exactness]
\label{thm:ratup-fieldup-restricted-normalizer-action-kernel-fiber-exactness}
For Definition~\ref{def:ratup-fieldup-concrete-exit-object}, fix a certified
affine automorphism $A:\mathsf{AffAut}_F$. Let $C_F(A)$ be its concrete affine
centralizer, let $N_F(C_F(A))$ be the corresponding normalizer, let
$\mathrm{Act}_T(X)$ denote the restricted normalizer action, and let $K_A$ be
the kernel carrier of
Definition~\ref{def:ratup-fieldup-affine-normalizer-action-kernel-carrier}. If
$N_F(C_F(A))(S)$ and $N_F(C_F(A))(T)$, then
$$
K_A(\iota_F(S)\circ_{\mathrm{aff}}T)
\Longleftrightarrow
\forall X:\mathsf{AffAut}_F,\ C_F(A)(X)\Rightarrow
\mathrm{Act}_S(X)\equiv_{\mathrm{aff}}\mathrm{Act}_T(X).
$$
\end{theorem}
\begin{proof}
The normalizer subgroup theorem
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-subgroup} places
$\iota_F(S)$ in $N_F(C_F(A))$ and then places
$\iota_F(S)\circ_{\mathrm{aff}}T$ in $N_F(C_F(A))$. Thus the displayed left
side is a genuine instance of the kernel predicate from
Definition~\ref{def:ratup-fieldup-affine-normalizer-action-kernel-carrier}.

For the forward direction, assume
$K_A(\iota_F(S)\circ_{\mathrm{aff}}T)$ and fix $X$ with $C_F(A)(X)$. The
representation law in
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-action-representation}
rewrites the action of $\iota_F(S)\circ_{\mathrm{aff}}T$ on $X$ as
$\mathrm{Act}_{\iota_F(S)}(\mathrm{Act}_T(X))$. The kernel hypothesis identifies
this element with $X$.

Apply $\mathrm{Act}_S$ to that equality, using the classifier-exact action row
of
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-action-representation}.
The inverse-action cancellation rows from
\autoref{thm:ratup-fieldup-restricted-normalizer-action-automorphism-kernel-exactness}
reduce
$\mathrm{Act}_S(\mathrm{Act}_{\iota_F(S)}(\mathrm{Act}_T(X)))$ to
$\mathrm{Act}_T(X)$. Hence $\mathrm{Act}_T(X)$ and $\mathrm{Act}_S(X)$ are
classifier-equal, which is the required pointwise equality after symmetry of
$\equiv_{\mathrm{aff}}$.

For the reverse direction, assume the pointwise equality of the two restricted
actions on every $X$ with $C_F(A)(X)$. The carrier part of
$K_A(\iota_F(S)\circ_{\mathrm{aff}}T)$ is the normalizer product constructed in
the first paragraph.

It remains to show pointwise fixed action. Fix $X$ with $C_F(A)(X)$. By the
representation law, $\mathrm{Act}_{\iota_F(S)\circ_{\mathrm{aff}}T}(X)$ is
classifier-equal to $\mathrm{Act}_{\iota_F(S)}(\mathrm{Act}_T(X))$. Transport the
assumed pointwise equality
$\mathrm{Act}_S(X)\equiv_{\mathrm{aff}}\mathrm{Act}_T(X)$ through the
classifier-exact action $\mathrm{Act}_{\iota_F(S)}$.

The result identifies
$\mathrm{Act}_{\iota_F(S)}(\mathrm{Act}_T(X))$ with
$\mathrm{Act}_{\iota_F(S)}(\mathrm{Act}_S(X))$. The remaining inverse-action
cancellation row from
\autoref{thm:ratup-fieldup-restricted-normalizer-action-automorphism-kernel-exactness}
reduces this last element to $X$. Therefore
$\iota_F(S)\circ_{\mathrm{aff}}T$ fixes every centralizer element and lies in
$K_A$.
\end{proof}

\begin{proof}

\autoref{thm:field-rat-denominator-continuation-not-field-multiplication} to the

\begin{theorem}[RatUp FieldUp affine centralizer normalizer orbit action invariance]
\label{thm:ratup-fieldup-affine-centralizer-normalizer-orbit-action-invariance}
For Definition~\ref{def:ratup-fieldup-concrete-exit-object}, fix a certified
affine automorphism $A:\mathsf{AffAut}_F$ in the concrete $\FieldUp(\RatUp)$
exit object. Let $C_F(A)$ be the affine centralizer carrier, let
$N_F(C_F(A))$ be its normalizer carrier, and let $\sim_{N_F(A)}$ be the orbit
relation of
\autoref{def:ratup-fieldup-affine-centralizer-normalizer-orbit}. For carried
centralizer elements $X,Y:\mathsf{AffAut}_F$ and a normalizer element
$S:\mathsf{AffAut}_F$, if $C_F(A)(X)$, $C_F(A)(Y)$,
$X\sim_{N_F(A)}Y$, and $N_F(C_F(A))(S)$, then
$$
\mathrm{Act}_S(X)\sim_{N_F(A)}\mathrm{Act}_S(Y).
$$
\end{theorem}
\begin{proof}
Choose an orbit witness $T:\mathsf{AffAut}_F$ for $X\sim_{N_F(A)}Y$, so
$N_F(C_F(A))(T)$ and
$\mathrm{Act}_T(X)\equiv_{\mathrm{aff}}Y$. The inverse row of
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-subgroup} gives
$N_F(C_F(A))(\iota_F(S))$, and the product row gives the composite witness
$$
N_F(C_F(A))(S\circ_{\mathrm{aff}}T\circ_{\mathrm{aff}}\iota_F(S)).
$$

It remains to identify the action of this composite witness on
$\mathrm{Act}_S(X)$. Apply the composition rows of
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-action} to reassociate
the restricted action, and use
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-action-inverse} to
cancel the adjacent $\iota_F(S)$ and $S$ actions. The result is classifier-equal
to $\mathrm{Act}_S(\mathrm{Act}_T(X))$.

Now transport
$\mathrm{Act}_T(X)\equiv_{\mathrm{aff}}Y$ through the action by $S$ using
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-action-classifier-congruence}.
Classifier transitivity identifies the composite action on
$\mathrm{Act}_S(X)$ with $\mathrm{Act}_S(Y)$. Hence
$S\circ_{\mathrm{aff}}T\circ_{\mathrm{aff}}\iota_F(S)$ is an orbit witness for
$\mathrm{Act}_S(X)\sim_{N_F(A)}\mathrm{Act}_S(Y)$.
\end{proof}

\autoref{def:ratup-fieldup-affine-centralizer-normalizer-orbit}. For carried

centralizer carrier of $\mathrm{Act}_{S\circ_{\mathrm{aff}}T}(X)$, and the same
theorem applied to $S$ gives the carrier of $\mathrm{Act}_S(Y)$.

For the classifier comparison, use the composition row of

\begin{theorem}[RatUp FieldUp affine centralizer normalizer orbit action reflection]
\label{thm:ratup-fieldup-affine-centralizer-normalizer-orbit-action-iff}
For Definition~\ref{def:ratup-fieldup-concrete-exit-object}, fix a certified
affine automorphism $A:\mathsf{AffAut}_F$ in the concrete $\FieldUp(\RatUp)$
exit object. Let $C_F(A)$ be the affine centralizer carrier, let
$N_F(C_F(A))$ be its normalizer carrier, and let $\sim_{N_F(A)}$ be the orbit
relation of
\autoref{def:ratup-fieldup-affine-centralizer-normalizer-orbit}. For carried
centralizer elements $X,Y:\mathsf{AffAut}_F$ and a normalizer element
$S:\mathsf{AffAut}_F$, if $C_F(A)(X)$, $C_F(A)(Y)$, and $N_F(C_F(A))(S)$,
then
$$
X\sim_{N_F(A)}Y
\Longleftrightarrow
\mathrm{Act}_S(X)\sim_{N_F(A)}\mathrm{Act}_S(Y).
$$
\end{theorem}
\begin{proof}
The forward direction is exactly
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-orbit-action-invariance}
with the acting element $S$.

For the reverse direction, assume
$\mathrm{Act}_S(X)\sim_{N_F(A)}\mathrm{Act}_S(Y)$. The inverse row of
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-subgroup} gives
$N_F(C_F(A))(\iota_F(S))$.

The carrier clauses in
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-action-inverse} place
$\mathrm{Act}_S(X)$ and $\mathrm{Act}_S(Y)$ back in $C_F(A)$, so the orbit
invariance theorem applies to these acted endpoints.

Apply
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-orbit-action-invariance}
to the assumed orbit relation with the acting element $\iota_F(S)$. This yields
$$
\mathrm{Act}_{\iota_F(S)}(\mathrm{Act}_S(X))
\sim_{N_F(A)}
\mathrm{Act}_{\iota_F(S)}(\mathrm{Act}_S(Y)).
$$

The cancellation clauses of
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-action-inverse}
identify the left endpoint with $X$ and the right endpoint with $Y$ in the
affine classifier. The endpoint transport theorem
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-orbit-classifier-transport}
then replaces both acted endpoints by $X$ and $Y$, giving
$X\sim_{N_F(A)}Y$.
\end{proof}
