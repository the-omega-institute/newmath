\begin{theorem}[RatUp FieldUp source has a non-singleton endpoint]
\label{thm:ratup-fieldup-source-non-singleton-witness}
Let $d_1:=\Eone(\emp)$. For the concrete source tuple
$\mathsf{RatUpFieldSource}_F$ of
\autoref{def:ratup-fieldup-concrete-source-tuple}, the field carrier accepts
$d_1$, while the singleton empty-history carrier rejects it:
$$
\mathsf{RatHistoryCarrier}(d_1)
$$
and
$$
\mathsf{fieldSingletonEmptyCarrier}(d_1)\to\bot.
$$
Consequently no carrier-coverage map from the $\FieldUp(\mathsf{RatUp})$
source carrier to the singleton empty-history carrier can exist.
\end{theorem}
\begin{proof}
Unfold the carrier coordinate of $\mathsf{RatUpFieldSource}_F$ in
\autoref{def:ratup-fieldup-concrete-source-tuple}. It is exactly
$\mathsf{RatHistoryCarrier}$, so the source endpoint question is the concrete
RatUp carrier question.

Apply
\autoref{thm:field-rat-history-carrier-has-non-singleton-boundary-witness} at
$d_1=\Eone(\emp)$. It supplies the displayed
$\mathsf{RatHistoryCarrier}(d_1)$ witness and the rejection map
$\mathsf{fieldSingletonEmptyCarrier}(d_1)\to\bot$.

Now suppose a carrier-coverage map from the $\FieldUp(\mathsf{RatUp})$ source
carrier to $\mathsf{fieldSingletonEmptyCarrier}$ were given. Evaluating it at
$d_1$ and the displayed RatUp carrier witness gives
$\mathsf{fieldSingletonEmptyCarrier}(d_1)$, contradicting the rejection map.
\end{proof}

\begin{theorem}[RatUp FieldUp source coordinate coverage obstruction]
\label{thm:ratup-fieldup-source-coordinate-coverage-obstruction}
Let $d_1:=\Eone(\emp)$. For the concrete source tuple
$\mathsf{RatUpFieldSource}_F$ of
\autoref{def:ratup-fieldup-concrete-source-tuple}, the diagonal endpoint
supplies all three source-coordinate witnesses
$$
\mathsf{RatHistoryCarrier}(d_1),
$$
$$
\mathsf{RatHistoryClassifier}(d_1,d_1),
$$
and
$$
\mathsf{RatHistoryLedgerPolicy}(d_1,d_1).
$$
Each of the following coverage maps is contradictory:
\begin{enumerate}
  \item carrier coverage
  $$
  \forall h:\mathsf{BHist},\mathsf{RatHistoryCarrier}(h)\to
  \mathsf{fieldSingletonEmptyCarrier}(h);
  $$
  \item classifier coverage
  $$
  \forall h k:\mathsf{BHist},\mathsf{RatHistoryClassifier}(h,k)\to
  \mathsf{FieldSingletonClassifier}(h,k);
  $$
  \item ledger endpoint coverage
  $$
  \forall \rho v:\mathsf{BHist},\mathsf{RatHistoryLedgerPolicy}(\rho,v)\to
  \mathsf{FieldSingletonCarrier}(\rho)\land
  \mathsf{FieldSingletonCarrier}(v).
  $$
\end{enumerate}
Thus the concrete source tuple cannot be collapsed to the singleton
$\FieldUp$ boundary in any of its carrier, classifier, or ledger coordinates.
\end{theorem}
\begin{proof}
The carrier coordinate is the one isolated in
\autoref{thm:ratup-fieldup-source-non-singleton-witness}. It gives the endpoint
witness $\mathsf{RatHistoryCarrier}(d_1)$ and rules out any carrier-coverage
map from $\mathsf{RatHistoryCarrier}$ to
$\mathsf{fieldSingletonEmptyCarrier}$.

The classifier and ledger coordinates are supplied by
\autoref{thm:field-rat-classifier-ledger-singleton-empty-coverage-obstruction}.
That theorem gives the diagonal classifier witness
$\mathsf{RatHistoryClassifier}(d_1,d_1)$ and the diagonal ledger witness
$\mathsf{RatHistoryLedgerPolicy}(d_1,d_1)$.

The same theorem also rejects the displayed classifier coverage map and the
displayed ledger endpoint coverage map. Combining these two rejections with the
carrier rejection from the first paragraph leaves no coverage coordinate by
which $\mathsf{RatUpFieldSource}_F$ can be read as the singleton $\FieldUp$
boundary.
\end{proof}

\begin{theorem}[RatUp FieldUp contextual action support transports across context-pair classifiers]
\label{thm:field-rat-denominator-contextual-action-pair-transport-support}
For the concrete denominator tuple
$(C_D^u,R_D^u,\mu_D^u,e_D^u,N_D^u)$ and the contextual action $\Gamma$ of
\autoref{def:ratup-fieldup-denominator-contextual-action-instance}, suppose
$C_D^u(h)$ and suppose that $(l,r)$, $(s,t)$, $(l',r')$, and $(s',t')$ are
carried denominator context pairs. If
$\mathsf{PairClass}_D^u((l,r),(s,t))$ and
$\mathsf{PairClass}_D^u((l',r'),(s',t'))$, and if
$p\hsame\emp$, $q\hsame\emp$, $p'\hsame\emp$, and $q'\hsame\emp$, then strict
support is invariant under the transported two-layer contextual action:
$$
N_D^u(\Gamma_{p',q',l',r'}(\Gamma_{p,q,l,r}(h)))
\Longleftrightarrow
N_D^u(\Gamma_{p',q',s',t'}(\Gamma_{p,q,s,t}(h))).
$$
\end{theorem}
\begin{proof}
Apply
\autoref{thm:ratup-fieldup-denominator-contextual-action-composition-closure}
to the left pair of carried contexts and the empty endpoint hypotheses. Its
support row reads the left nested action as the disjunction of the common core
$N_D^u(h)$ with the four strict-support terms from $(l,r)$ and $(l',r')$.

Apply the same composition-closure theorem to $(s,t)$ and $(s',t')$. This reads
the right nested action as the same core term $N_D^u(h)$ together with the four
strict-support terms from the transported context pairs.

Next use
\autoref{thm:field-rat-denominator-context-pair-product-strict-support} on both
context-pair products. The four support terms on the left are exactly strict
support of $\mu_D^u(\mu_D^u(l',l),\mu_D^u(r,r'))$, while the four support terms
on the right are strict support of
$\mu_D^u(\mu_D^u(s',s),\mu_D^u(t,t'))$.

The componentwise pair classifiers are the hypotheses required by
\autoref{thm:field-rat-denominator-context-pair-product-strict-support-transport}.
They transport the product strict-support term between the two context-pair
products.

For the forward direction, use the left composition readback. A core-support
witness $N_D^u(h)$ is inserted directly into the right composition readback. A
context-product support witness is transported across the product pair and then
inserted into the corresponding four context disjuncts on the right.

The reverse direction is symmetric. Use the right composition readback, transport
the product-pair support term back across the same classifier equivalence, keep
the common core support fixed, and close with the left composition readback.
\end{proof}

\begin{theorem}[RatUp FieldUp contextual action pair transport carrier and support]
\label{thm:field-rat-denominator-contextual-action-pair-transport-carrier-support}
For the concrete denominator tuple
$(C_D^u,R_D^u,\mu_D^u,e_D^u,N_D^u)$ and the contextual action $\Gamma$ of
\autoref{def:ratup-fieldup-denominator-contextual-action-instance}, suppose
$C_D^u(h)$ and suppose that $(l,r)$, $(s,t)$, $(l',r')$, and $(s',t')$ are
carried denominator context pairs. If
$\mathsf{PairClass}_D^u((l,r),(s,t))$ and
$\mathsf{PairClass}_D^u((l',r'),(s',t'))$, and if
$p\hsame\emp$, $q\hsame\emp$, $p'\hsame\emp$, and $q'\hsame\emp$, then both
transported nested contextual actions are carried:
$$
C_D^u(\Gamma_{p',q',l',r'}(\Gamma_{p,q,l,r}(h)))
\land
C_D^u(\Gamma_{p',q',s',t'}(\Gamma_{p,q,s,t}(h))).
$$
Their strict support is transported by the same context-pair classifiers:
$$
N_D^u(\Gamma_{p',q',l',r'}(\Gamma_{p,q,l,r}(h)))
\Longleftrightarrow
N_D^u(\Gamma_{p',q',s',t'}(\Gamma_{p,q,s,t}(h))).
$$
\end{theorem}
\begin{proof}
Apply
\autoref{thm:ratup-fieldup-denominator-contextual-action-composition-closure}
to the carried contexts $(l,r)$ and $(l',r')$ with the empty endpoint
hypotheses. This gives the left carrier conclusion and reads the left nested
action as the five-term strict-support expression with core $N_D^u(h)$.

Apply the same composition-closure theorem to $(s,t)$ and $(s',t')$. This gives
the right carrier conclusion and reads the right nested action as the
corresponding five-term strict-support expression with the same core term.
Together these two applications establish the displayed carrier conjunction.

The context tails are compressed by
\autoref{thm:field-rat-denominator-context-pair-product-strict-support}. On the
left, the four context support terms are strict support of
$\mu_D^u(\mu_D^u(l',l),\mu_D^u(r,r'))$; on the right, they are strict support of
$\mu_D^u(\mu_D^u(s',s),\mu_D^u(t,t'))$.

Use
\autoref{thm:field-rat-denominator-context-pair-product-strict-support-transport}
under the two $\mathsf{PairClass}_D^u$ hypotheses. It transports the compressed
product strict-support term between the two context-pair products.

For the forward direction of the strict-support equivalence, read the left
nested action through the left composition theorem. A core witness
$N_D^u(h)$ is kept fixed, while a context-product witness is transported to the
right product and expanded through the right product-support readback. The right
composition readback then yields support of the right nested action.

The reverse direction is the same argument with left and right interchanged.
The right composition readback supplies either the common core witness or a
right product-support witness; the latter is transported back across the
product classifier equivalence and expanded on the left before closing with the
left composition readback.
\end{proof}

Per-object template: traditional target transported contextual-action
stability; BEDC source $\mathsf{BHist}$, $\mathsf{RatDenomUnitCarrier}$,
$\mathsf{RatDenomUnitClassifier}$, $\mathsf{append}$, empty endpoint, and
$\Gamma$; stability obligations carrier closure plus strict-support transport;
core theorem
\autoref{thm:field-rat-denominator-contextual-action-pair-transport-carrier-support};
exactness predicate the displayed support iff; standard bridge deferred; status
$\mathsf{ProofTarget}$.

$\Gamma$ and affine coordinate $\mathsf{AffAut}_F$ as fixed in
Definition~\ref{def:concrete-ratup-fieldup-certificate-instance}.

The source anchor is
\autoref{thm:ratup-fieldup-certificate-rows-source-anchored}. It reads the
$\CommRingUp$ base coordinate through
\autoref{thm:ratup-fieldup-assembled-certificate-boundary} and pins the field
carrier, field classifier, denominator tuple, contextual action, context-pair
monoid, and affine coordinate to the same concrete source tuple.

apart-zero predicate, so the inverse classifier, inverse stability, and

\autoref{thm:ratup-fieldup-denominator-total-normalization-support-collapse-row}.

\begin{theorem}[RatUp FieldUp assembled support-neutrality certificate]
\label{thm:ratup-fieldup-assembled-support-neutrality-certificate}
For the concrete denominator tuple $(C_D^u,R_D^u,\mu_D^u,e_D^u,N_D^u)$, the
contextual action $\Gamma$ of
\autoref{def:ratup-fieldup-denominator-contextual-action-instance}, and the
named source tuple $\mathsf{RatUpFieldSource}_F$, the assembled
$\FieldUp(\mathsf{RatUp})$ certificate includes the support-neutrality row
below. Under the hypotheses of
\autoref{thm:ratup-fieldup-denominator-total-normalization-support-collapse-row},
the nested action is carried:
$$
C_D^u(\Gamma_{p',q',l',r'}(\Gamma_{p,q,l,r}(h))).
$$
It is classified by the collected action:
$$
R_D^u(\Gamma_{p',q',l',r'}(\Gamma_{p,q,l,r}(h)),
\Gamma_{p',q',\mu_D^u(l',l),\mu_D^u(r,r')}(h)).
$$
It satisfies strict-support neutrality:
$$
N_D^u(\Gamma_{p',q',l',r'}(\Gamma_{p,q,l,r}(h)))
\Longleftrightarrow
N_D^u(h).
$$
For any second endpoint $k$ under the same collected carried contexts, and for
$s,t,s',t'$ satisfying $s\hsame\emp$, $t\hsame\emp$, $s'\hsame\emp$, and
$t'\hsame\emp$, the collected action remains classifier-exact against that
endpoint:
$$
R_D^u(\Gamma_{s,t,\mu_D^u(l',l),\mu_D^u(r,r')}(h),
\Gamma_{s',t',\mu_D^u(l',l),\mu_D^u(r,r')}(k))
\Longleftrightarrow R_D^u(h,k).
$$
\end{theorem}
\begin{proof}
The assembled boundary is fixed by
\autoref{thm:ratup-fieldup-assembled-certificate-boundary}. It places
$\mathsf{RatUpFieldSource}_F$, the displayed denominator tuple, and the action
$\Gamma$ inside the concrete $\FieldUp(\mathsf{RatUp})$ certificate governed by
\autoref{def:field-certificate-obligations}.

Within that boundary, use
\autoref{thm:ratup-fieldup-denominator-total-normalization-support-collapse-row}.
Its hypotheses are exactly the carried-denominator, empty external-context, and
strict-support hypotheses named in the statement. The theorem supplies the
carried nested action and its classifier comparison with the collected action.

The same row supplies the strict-support equivalence
$N_D^u(\Gamma_{p',q',l',r'}(\Gamma_{p,q,l,r}(h)))\Longleftrightarrow N_D^u(h)$.
It also supplies classifier exactness for the collected action against the
second endpoint $k$. These four conclusions are therefore available as one
support-neutrality row of the assembled FieldUp certificate.
\end{proof}

\begin{theorem}[RatUp FieldUp exit certificate has nontrivial carrier and support-neutral action]
\label{thm:ratup-fieldup-exit-certificate-nontrivial-support-neutral}
For Definition~\ref{def:concrete-ratup-fieldup-certificate-instance} and
$d_1:=\Eone(\emp)$, the concrete $\FieldUp(\mathsf{RatUp})$ exit certificate has
both a nontrivial source carrier and a support-neutral contextual action.
Specifically:
\begin{enumerate}
  \item $\mathsf{RatHistoryCarrier}(d_1)$ holds;
  \item $\mathsf{fieldSingletonEmptyCarrier}(d_1)\to\bot$;
  \item no carrier-coverage map from the $\FieldUp(\mathsf{RatUp})$ source
  carrier to $\mathsf{fieldSingletonEmptyCarrier}$ exists;
  \item under the hypotheses of
  \autoref{thm:ratup-fieldup-assembled-support-neutrality-certificate}, the
  nested contextual action is carried, classified by the collected action,
  support-neutral, and classifier-exact against the second endpoint.
\end{enumerate}
Equivalently, the same named certificate contains the endpoint obstruction
$$
\mathsf{RatHistoryCarrier}(d_1)
$$
with
$$
\mathsf{fieldSingletonEmptyCarrier}(d_1)\to\bot
$$
and the contextual-action row
$$
C_D^u(\Gamma_{p',q',l',r'}(\Gamma_{p,q,l,r}(h))),
$$
$$
R_D^u(\Gamma_{p',q',l',r'}(\Gamma_{p,q,l,r}(h)),
\Gamma_{p',q',\mu_D^u(l',l),\mu_D^u(r,r')}(h)),
$$
$$
N_D^u(\Gamma_{p',q',l',r'}(\Gamma_{p,q,l,r}(h)))
\Longleftrightarrow N_D^u(h),
$$
and, for the second endpoint $k$ with empty external contexts,
$$
R_D^u(\Gamma_{s,t,\mu_D^u(l',l),\mu_D^u(r,r')}(h),
\Gamma_{s',t',\mu_D^u(l',l),\mu_D^u(r,r')}(k))
\Longleftrightarrow R_D^u(h,k).
$$
\end{theorem}
\begin{proof}
The concrete certificate is fixed by
\autoref{thm:ratup-fieldup-semantic-namecert}. It identifies
Definition~\ref{def:concrete-ratup-fieldup-certificate-instance} as the
semantic naming certificate for the displayed $\FieldUp(\mathsf{RatUp})$
source tuple, so the carrier, classifier, denominator tuple, and contextual
action are the concrete rows named in the statement.

The carrier obstruction comes from
\autoref{thm:ratup-fieldup-source-non-singleton-witness}. With
$d_1:=\Eone(\emp)$ it supplies $\mathsf{RatHistoryCarrier}(d_1)$, the map
$\mathsf{fieldSingletonEmptyCarrier}(d_1)\to\bot$, and the contradiction
against any carrier-coverage map into the singleton empty-history carrier.

Finally apply
\autoref{thm:ratup-fieldup-assembled-support-neutrality-certificate}. Under
its carried-context and strict-support hypotheses, it supplies the nested
carried action, the collected-action classifier comparison, strict-support
neutrality, and classifier exactness against the second endpoint. These rows
belong to the same certificate fixed above, so the non-singleton carrier and
support-neutral action are simultaneous properties of the concrete exit
certificate.
\end{proof}

\begin{theorem}[RatUp FieldUp transported support-neutrality certificate]
\label{thm:ratup-fieldup-transported-support-neutrality-row}
For the concrete denominator tuple $(C_D^u,R_D^u,\mu_D^u,e_D^u,N_D^u)$ and the
contextual action $\Gamma$ of
\autoref{def:ratup-fieldup-denominator-contextual-action-instance}, suppose the
hypotheses of
\autoref{thm:ratup-fieldup-assembled-support-neutrality-certificate} hold for
the context pairs $(l,r)$ and $(l',r')$. Suppose also that the transported
context pairs $(s,t)$ and $(s',t')$ are carried and that the pair classifiers
required by
\autoref{thm:field-rat-denominator-contextual-action-pair-transport-carrier-support}
compare $(l,r)$ with $(s,t)$ and $(l',r')$ with $(s',t')$. Then the transported
nested action is carried:
$$
C_D^u(\Gamma_{p',q',s',t'}(\Gamma_{p,q,s,t}(h))).
$$
Its strict support is neutral against the original denominator history:
$$
N_D^u(\Gamma_{p',q',s',t'}(\Gamma_{p,q,s,t}(h)))
\Longleftrightarrow
N_D^u(h).
$$
\end{theorem}
\begin{proof}
Apply
\autoref{thm:field-rat-denominator-contextual-action-pair-transport-carrier-support}
to the carried context pairs and the two pair classifiers. The transported
carrier conclusion is the second carried component, and the same theorem gives
the strict-support equivalence between the original nested action and the
transported nested action.

Apply
\autoref{thm:ratup-fieldup-assembled-support-neutrality-certificate} to the
original context pairs $(l,r)$ and $(l',r')$. This supplies
$N_D^u(\Gamma_{p',q',l',r'}(\Gamma_{p,q,l,r}(h)))\Longleftrightarrow N_D^u(h)$.

For the forward direction, read support of the transported nested action back
through the transport equivalence, then use the assembled support-neutrality
row to obtain $N_D^u(h)$.

For the reverse direction, use assembled support-neutrality to obtain support
of the original nested action, then move it across the transport equivalence to
the transported context-pair representative.
\end{proof}
