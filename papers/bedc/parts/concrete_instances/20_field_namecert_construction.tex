\begin{theorem}[RatUp FieldUp selector-pair strict exit determinacy]
\label{thm:ratup-fieldup-selector-pair-strict-exit-determinacy}
For the concrete denominator tuple $(C_D^u,R_D^u,\mu_D^u,e_D^u,N_D^u)$ and
histories $h,k,\ell:\mathsf{BHist}$ with $C_D^u(h)$, $C_D^u(k)$, and
$C_D^u(\ell)$, let $E_q(h,k,\ell)$ be the endpoint selector of
\autoref{def:ratup-fieldup-bracketing-endpoint-selector} for each
$q:\mathsf{BMark}$. If $\mathsf{ThreeSupp}_D^u(h,k,\ell)$ holds, then for all
selector pairs $s,t,u,v:\mathsf{BMark}$ and contexts $L,R:\mathsf{BHist}$ the
following concrete $\FieldUp(\RatUp)$ exit package is determined:
\begin{enumerate}
  \item
  $$
  R_D^u(E_s(h,k,\ell),E_u(h,k,\ell))
  \land
  R_D^u(E_t(h,k,\ell),E_v(h,k,\ell));
  $$
  \item
  $$
  \mathsf{RatHistoryClassifier}(E_s(h,k,\ell),E_u(h,k,\ell))
  \land
  \mathsf{RatHistoryClassifier}(E_t(h,k,\ell),E_v(h,k,\ell));
  $$
  \item
  $$
  C_D^u(E_s(h,k,\ell))
  \land
  C_D^u(E_t(h,k,\ell))
  \land
  R_D^u(E_s(h,k,\ell),E_t(h,k,\ell))
  \land
  \left(
  \mathsf{fieldSingletonEmptyClassifier}
  (\mathsf{append}(L,E_s(h,k,\ell)),\mathsf{append}(R,E_t(h,k,\ell)))\to\bot
  \right)
  \land
  \mathsf{fieldSingletonEmptyNonZero}(E_s(h,k,\ell))
  \land
  \mathsf{fieldSingletonEmptyNonZero}(E_t(h,k,\ell));
  $$
  \item
  $$
  C_D^u(E_u(h,k,\ell))
  \land
  C_D^u(E_v(h,k,\ell))
  \land
  R_D^u(E_u(h,k,\ell),E_v(h,k,\ell))
  \land
  \left(
  \mathsf{fieldSingletonEmptyClassifier}
  (\mathsf{append}(L,E_u(h,k,\ell)),\mathsf{append}(R,E_v(h,k,\ell)))\to\bot
  \right)
  \land
  \mathsf{fieldSingletonEmptyNonZero}(E_u(h,k,\ell))
  \land
  \mathsf{fieldSingletonEmptyNonZero}(E_v(h,k,\ell)).
  $$
\end{enumerate}
\end{theorem}
\begin{proof}
Apply \autoref{thm:ratup-fieldup-selector-complete-strict-exit-package} to the
selector pair $(s,u)$ and contexts $L,R:\mathsf{BHist}$. Its strict exit row
contains the denominator classifier row
$R_D^u(E_s(h,k,\ell),E_u(h,k,\ell))$.

Apply the same strict exit package to the selector pair $(t,v)$ and the same
contexts. This supplies $R_D^u(E_t(h,k,\ell),E_v(h,k,\ell))$.

Apply \autoref{thm:ratup-fieldup-bracketing-selector-rat-classifier-coverage} to
the selected pairs $(s,u)$ and $(t,v)$. This gives
$\mathsf{RatHistoryClassifier}(E_s(h,k,\ell),E_u(h,k,\ell))$ and
$\mathsf{RatHistoryClassifier}(E_t(h,k,\ell),E_v(h,k,\ell))$.

Next apply \autoref{thm:ratup-fieldup-selector-complete-strict-exit-package} to
the selector pair $(s,t)$ and contexts $L,R:\mathsf{BHist}$. The resulting
strict exit row is exactly the third displayed conjunction.

Finally apply the same strict exit package to the selector pair $(u,v)$ and the
same contexts. Its row is the fourth displayed conjunction, and together with
the two componentwise classifier rows it assembles the stated deterministic
selector-pair package.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.ratup\_fieldup\_selector\_pair\_strict\_exit\_determinacy}

$R_D^u$ plus $\mathsf{RatHistoryClassifier}$; stability obligations selector
strict exit row coverage and Rat classifier coverage; ledger the two selector
pairs and contexts $L,R$; core theorem this block; exactness predicate
componentwise selector determinacy; status $\mathsf{ProofTarget}$.

\begin{theorem}[RatUp FieldUp bracketing selector complete obstruction package]
\label{thm:ratup-fieldup-bracketing-selector-complete-obstruction-package}
For the concrete denominator tuple $(C_D^u,R_D^u,\mu_D^u,e_D^u,N_D^u)$ and
histories $h,k,\ell:\mathsf{BHist}$ with $C_D^u(h)$, $C_D^u(k)$, and
$C_D^u(\ell)$, let $E_s(h,k,\ell)$ be the endpoint selector of
\autoref{def:ratup-fieldup-bracketing-endpoint-selector}. If
$\mathsf{ThreeSupp}_D^u(h,k,\ell)$ holds, then for all selectors
$s,t:\mathsf{BMark}$ and contexts $L,R:\mathsf{BHist}$, setting
$a:=E_s(h,k,\ell)$ and $b:=E_t(h,k,\ell)$ gives one concrete selector
obstruction package with the following rows:
\begin{enumerate}
  \item $a\hsame b$;
  \item $R_D^u(a,b)$;
  \item $\mathsf{RatHistoryClassifier}(a,b)$;
  \item
  $$
  \mathsf{fieldSingletonEmptyClassifier}
  (\mathsf{append}(L,a),\mathsf{append}(R,b))\to\bot;
  $$
  \item
  $$
  \mathsf{fieldSingletonEmptyCarrier}(\mathsf{append}(L,a))\to\bot;
  $$
  \item
  $$
  \mathsf{fieldSingletonEmptyNonZero}(\mathsf{append}(L,a));
  $$
  \item
  $$
  \mathsf{fieldSingletonEmptyCarrier}(\mathsf{append}(a,R))\to\bot;
  $$
  \item
  $$
  \mathsf{fieldSingletonEmptyNonZero}(\mathsf{append}(a,R));
  $$
  \item
  $$
  \mathsf{fieldSingletonEmptyCarrier}(\mathsf{append}(L,b))\to\bot;
  $$
  \item
  $$
  \mathsf{fieldSingletonEmptyNonZero}(\mathsf{append}(L,b));
  $$
  \item
  $$
  \mathsf{fieldSingletonEmptyCarrier}(\mathsf{append}(b,R))\to\bot;
  $$
  \item
  $$
  \mathsf{fieldSingletonEmptyNonZero}(\mathsf{append}(b,R)).
  $$
\end{enumerate}
\end{theorem}
\begin{proof}
First apply
\autoref{thm:ratup-fieldup-bracketing-selector-independence-package} to the
same denominator tuple, carrier witnesses, support witness, selector pair, and
the context $L:\mathsf{BHist}$. This supplies $a\hsame b$ and
$\mathsf{RatHistoryClassifier}(a,b)$ as the common selector-independent rows.

Next apply \autoref{thm:ratup-fieldup-bracketing-selector-exit-subpackage} to
the selector pair $s,t:\mathsf{BMark}$ and the contexts $L,R:\mathsf{BHist}$.
Its exit row gives $R_D^u(a,b)$ and the singleton-classifier exclusion for
$(\mathsf{append}(L,a),\mathsf{append}(R,b))$.

We now build the contextual nonzero rows for the left endpoint. Apply
\autoref{thm:ratup-fieldup-selector-contextual-nonzero-package} to the selector
$s$ with context $L$ to obtain the carrier exclusion and
$\mathsf{fieldSingletonEmptyNonZero}$ witness for $\mathsf{append}(L,a)$.
Apply the same theorem to $s$ with context $R$ to obtain the corresponding
right-context rows for $\mathsf{append}(a,R)$.

Finally apply
\autoref{thm:ratup-fieldup-selector-contextual-nonzero-package} to the selector
$t$ with context $L$, and then with context $R$. These two applications give the
carrier exclusions and $\mathsf{fieldSingletonEmptyNonZero}$ witnesses for
$\mathsf{append}(L,b)$ and $\mathsf{append}(b,R)$. Replacing $a$ and $b$ by the
displayed selector endpoints assembles the concrete obstruction package.
\end{proof}

\begin{enumerate}
  \item the carrier, classifier, denominator, and contextual-action rows of

  \item the operation-table coverage of
  \autoref{thm:ratup-fieldup-concrete-operation-exit-exhaustion};
  \item the completed exit object of

  \item the final-row semantic certificate of

  \item the nonzero and unit-obstruction rows of
  \autoref{thm:ratup-fieldup-operation-table-final-inventory-nonzero-coverage}.

Thus a downstream Phase Algebra Hierarchy consumer can read
$\mathcal E_F$ as a single concrete row-coverage certificate for the canonical
$\mathsf{BHist}$ source.
\end{theorem}
\begin{proof}
First take the concrete schema instance of
\autoref{thm:ratup-fieldup-concrete-schema-instance}. It fixes the shared

tuple, and contextual action rows used by the $\FieldUp(\RatUp)$ construction.

\autoref{thm:ratup-fieldup-concrete-operation-exit-exhaustion} with the
completed exit object of
\autoref{thm:ratup-fieldup-concrete-exit-object-complete}. These two cited
certificates identify the same selectors and the same concrete object
$\mathcal E_F$ rather than introducing a second exit target.

Finally append the final-row semantic certificate from
\autoref{thm:ratup-fieldup-final-row-inventory-semantic-certificate} and the
nonzero and unit-obstruction rows from
\autoref{thm:ratup-fieldup-operation-table-final-inventory-nonzero-coverage}.

certificate.

contextual action, selectors, or exit object. It only reads the rows already
provided by the cited concrete $\mathsf{BHist}$ certificates.
\end{proof}
