\begin{theorem}[RatUp FieldUp proof-target exit package]
\label{thm:ratup-fieldup-proof-target-exit-package}
For the concrete source tuple $\mathsf{RatUpFieldSource}_F$, the
$\FieldUp(\RatUp)$ proof target is exhausted by concrete $\mathsf{BHist}$ rows.
Its exit package is the constructive composition of the paper certificate
assembly of \autoref{thm:ratup-fieldup-semantic-namecert}, the source anchoring
of \autoref{thm:ratup-fieldup-certificate-rows-source-anchored}, the concrete
schema instance of \autoref{thm:ratup-fieldup-concrete-schema-instance}, the
concrete exit certificate package of
\autoref{thm:ratup-fieldup-concrete-exit-certificate-package}, the transported
exit-row contextual separation of
\autoref{thm:ratup-fieldup-transported-exit-row-contextual-separation-package},
and the normalized contextual separation of
\autoref{thm:ratup-fieldup-normalized-contextual-separation-package}. Thus the
displayed $\mathsf{BHist}$ source tuple carries the exit package for
$\FieldUp(\RatUp)$, and no exit row remains as a schematic $\mathsf{mul}$,
$\mathsf{add}$, or $\mathsf{neg}$ operator parameter.
\end{theorem}
\begin{proof}
Start with \autoref{thm:ratup-fieldup-semantic-namecert}. It supplies the
paper-side carrier, source, classifier, stability, ledger, and exactness rows
for the concrete source tuple $\mathsf{RatUpFieldSource}_F$ over
$\mathsf{BHist}$.

Next apply \autoref{thm:ratup-fieldup-certificate-rows-source-anchored}. It
identifies every row consumed by the concrete certificate with the same source
tuple: the rational history carrier and classifier, the denominator tuple, the
contextual action $\Gamma$, the context-pair monoid, and the affine coordinate.

The schema and exit boundary are then fixed by
\autoref{thm:ratup-fieldup-concrete-schema-instance} and
\autoref{thm:ratup-fieldup-concrete-exit-certificate-package}. Together these
theorem sites repack the source-anchored rows as the public
$\FieldUp(\RatUp)$ certificate instance and its concrete exit certificate.

Finally use
\autoref{thm:ratup-fieldup-transported-exit-row-contextual-separation-package}
and \autoref{thm:ratup-fieldup-normalized-contextual-separation-package}. The
first theorem supplies the transported noncollapse and contextual rejection
rows, while the second records the normalized row package in the same concrete
source. Combining these clauses leaves only rows carried by
$\mathsf{RatUpFieldSource}_F$, so the exit package has no schematic
$\mathsf{mul}$, $\mathsf{add}$, or $\mathsf{neg}$ operator coordinate.
\end{proof}

\begin{theorem}[RatUp FieldUp right-nested threefold contextual boundary]
\label{thm:ratup-fieldup-right-nested-threefold-contextual-boundary}
For the concrete denominator tuple $(C_D^u,R_D^u,\mu_D^u,e_D^u,N_D^u)$ and
histories $h,k,\ell:\mathsf{BHist}$ with $C_D^u(h)$, $C_D^u(k)$, and
$C_D^u(\ell)$, let $y:=\mu_D^u(h,\mu_D^u(k,\ell))$. If
$\mathsf{ThreeSupp}_D^u(h,k,\ell)$ holds, then the right-nested
$\FieldUp(\mathsf{RatUp})$ denominator endpoint carries the following
contextual boundary:
\begin{enumerate}
  \item for every context history $r:\mathsf{BHist}$,
  $\mathsf{fieldSingletonEmptyCarrier}(\mathsf{append}(r,y))\to\bot$;
  \item for every context history $r:\mathsf{BHist}$,
  $\mathsf{fieldSingletonEmptyCarrier}(\mathsf{append}(y,r))\to\bot$;
  \item for every context history $r:\mathsf{BHist}$,
  $\mathsf{fieldSingletonEmptyNonZero}(\mathsf{append}(r,y))$;
  \item for every context history $r:\mathsf{BHist}$,
  $\mathsf{fieldSingletonEmptyNonZero}(\mathsf{append}(y,r))$;
  \item for all $L,R:\mathsf{BHist}$,
  $$
  \mathsf{fieldSingletonEmptyClassifier}(\mathsf{append}(L,y),\mathsf{append}(R,y))
  \to\bot.
  $$
\end{enumerate}
\end{theorem}
\begin{proof}
Apply
\autoref{thm:ratup-fieldup-right-nested-threefold-support-non-singleton-exactness}
to the same concrete denominator tuple, carried histories, and support witness.
The right-nested endpoint data recovers $\mathsf{RatHistoryCarrier}(y)$ and the
singleton rejection map $\mathsf{fieldSingletonEmptyCarrier}(y)\to\bot$.

Fix a context history $r:\mathsf{BHist}$. Apply
\autoref{thm:field-singleton-empty-append-rat-carrier-absurd} with context $r$
and Rat carrier endpoint $y$ to reject
$\mathsf{fieldSingletonEmptyCarrier}(\mathsf{append}(r,y))$. Apply
\autoref{thm:field-singleton-empty-append-left-rat-carrier-absurd} with the same
two histories in the opposite append order to reject
$\mathsf{fieldSingletonEmptyCarrier}(\mathsf{append}(y,r))$.

For the right-append inverse-domain witness, unfold
$\mathsf{fieldSingletonEmptyNonZero}(\mathsf{append}(r,y))$ as rejection of the
singleton classifier against $\emp$. Any such classifier witness contains the
source-carrier component
$\mathsf{fieldSingletonEmptyCarrier}(\mathsf{append}(r,y))$, contradicting the
right-append rejection.

For the left-append inverse-domain witness, unfold
$\mathsf{fieldSingletonEmptyNonZero}(\mathsf{append}(y,r))$ as rejection of the
singleton classifier against $\emp$. Its source-carrier component is
$\mathsf{fieldSingletonEmptyCarrier}(\mathsf{append}(y,r))$, contradicting the
left-append rejection.

Finally package $\mathsf{RatHistoryClassifier}(y,y)$ from the recovered Rat
carrier witness on both sides and reflexivity of $\hsame$. For arbitrary
$L,R:\mathsf{BHist}$, apply
\autoref{thm:field-singleton-empty-classifier-append-rat-classifier-absurd}.
It rejects
$\mathsf{fieldSingletonEmptyClassifier}(\mathsf{append}(L,y),\mathsf{append}(R,y))$.
\end{proof}
