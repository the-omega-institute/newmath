\chapter{A Concrete Naming Certificate for \(\FieldUp\)}
\label{ch:concrete-instances-field-namecert}

The field interface $\FieldUp$ extends \autoref{ch:concrete-instances-commring-namecert} with the multiplicative inverse field. The inverse domain is a unary certificate $\mathsf{NonZero}_F(a)$ derived from constructive apartness of $a$ from $0$; inverse formation is licensed only by an explicit witness $p:\mathsf{NonZero}_F(a)$. This proof-relevant discipline replaces the classical "$a \ne 0$ implies $a$ invertible" formulation, which would otherwise require LEM. The rational carrier from \autoref{ch:concrete-instances-rat-namecert} supplies the candidate carrier and classifier source; a $\FieldUp$ certificate is present only when commutative-ring structure, constructive nonzero evidence, inverse closure, inverse laws indexed by nonzero witnesses, and the ledger are supplied over that source.

\section{Carrier definition}

\begin{definition}[Field carrier]
\label{def:field-carrier}
Use a commutative ring setup extended with a constructive apartness predicate and define $\mathsf{NonZero}_F(a)$ as the unary certificate obtained from apartness of $a$ from $0$. The inverse datum has type $\mathsf{inv}(a,p)$ for $p:\mathsf{NonZero}_F(a)$, so no inverse is formed without the nonzero witness.
\end{definition}

\section{The certificate}

\begin{definition}[Field source specification]
\label{def:field-source-specification}
The source is a certified commutative ring carrier plus constructive apartness from zero, packaged through the derived predicate $\mathsf{NonZero}_F$.
\end{definition}

\begin{definition}[Field pattern specification]
\label{def:field-pattern-specification}
The pattern is commutative-ring expression formation extended by inverse formation only from a pair $(a,p)$ with $p:\mathsf{NonZero}_F(a)$.
\end{definition}

\begin{definition}[Field classifier specification]
\label{def:field-classifier-specification}
The classifier is ring sameness $\sim_F$. Nonzero evidence is stable under classifier replacement: from $a\sim_F b$ and $p:\mathsf{NonZero}_F(a)$ the certificate supplies transported evidence $q:\mathsf{NonZero}_F(b)$.
\end{definition}

\begin{definition}[Field stability certificate]
\label{def:field-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item all commutative-ring stability fields
  \item contradiction of $\mathsf{NonZero}_F(0)$
  \item stability of $\mathsf{NonZero}_F$ under the carrier classifier
  \item inverse closure for explicit witnesses $p:\mathsf{NonZero}_F(a)$
  \item left and right multiplicative inverse laws indexed by each witness $p:\mathsf{NonZero}_F(a)$
\end{enumerate}
\end{definition}

\begin{theorem}[Field inverse congruence from apartness transport]
\label{thm:field-inverse-congruence-from-apartness}
For a BEDC history-level $\FieldUp$ candidate with BHist elements, commutative-ring classifier $\sim_F$, unary nonzero certificate $\mathsf{NonZero}_F$, inverse operation $\mathsf{inv}(a,p)$ indexed by $p:\mathsf{NonZero}_F(a)$, classifier transport of nonzero evidence, and left and right inverse laws indexed by nonzero witnesses, if $a\sim_F b$ and $p:\mathsf{NonZero}_F(a)$ while $q:\mathsf{NonZero}_F(b)$ is the transported witness along $a\sim_F b$, then $\mathsf{inv}(a,p)\sim_F\mathsf{inv}(b,q)$.
\end{theorem}
\begin{proof}
Transport the left inverse law for $\mathsf{inv}(a,p)$ across $a\sim_F b$ to obtain a left inverse witness over $b$. Combine this transported witness with the right inverse law for $\mathsf{inv}(b,q)$ at $b$. The group left/right inverse uniqueness argument, applied inside the multiplicative group of nonzero field elements, identifies the transported inverse with the chosen inverse at $b$.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.field\_inverse\_congruence\_from\_apartness}

\begin{theorem}[Field inverse cancel from apartness-indexed inverses]
\label{thm:field-inverse-cancel-from-apartness}
For a BEDC history-level \(\FieldUp\) candidate with proof-indexed inverses,
if \(\mathsf{inv}(a,p)\hsame\mathsf{inv}(b,q)\), then \(a\hsame b\).
\end{theorem}
\begin{proof}
Transport the left inverse law for \(a\) across the equality of
proof-indexed inverses to make \(a\) a right inverse of the common inverse
\(\mathsf{inv}(b,q)\). The right inverse law for \(b\) makes \(b\) the
matching left inverse. Group left/right inverse uniqueness identifies \(b\)
with \(a\), and symmetry gives \(a\hsame b\).
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.field\_inverse\_cancel\_from\_apartness}

\begin{theorem}[Field inverse of an indexed inverse]
\label{thm:field-inverse-involutive-from-apartness}
For a BEDC history-level $\FieldUp$ candidate with multiplication on
$\mathsf{BHist}$, unit $1$, proof-indexed inverse $\mathsf{inv}(a,p)$,
associativity, left and right identity, multiplication congruence, and left
inverse laws indexed by nonzero witnesses, if
$p_a:\mathsf{NonZero}_F(a)$ and
$p_{\mathsf{inv}}:\mathsf{NonZero}_F(\mathsf{inv}(a,p_a))$, then
$$
\mathsf{inv}(\mathsf{inv}(a,p_a),p_{\mathsf{inv}})\hsame a.
$$
\end{theorem}
\begin{proof}
Set the middle element in the inverse-uniqueness argument to
$\mathsf{inv}(a,p_a)$. The left inverse law for $p_{\mathsf{inv}}$ gives
$$
\mathsf{inv}(\mathsf{inv}(a,p_a),p_{\mathsf{inv}})\mathsf{inv}(a,p_a)\hsame 1.
$$

The left inverse law for $p_a$ gives
$$
\mathsf{inv}(a,p_a)a\hsame 1.
$$

Associativity, multiplication congruence, and the identity laws give the
standard left/right inverse uniqueness chain
$x\hsame x\,1\hsame x(yz)\hsame (xy)z\hsame 1\,z\hsame z$, with
$x=\mathsf{inv}(\mathsf{inv}(a,p_a),p_{\mathsf{inv}})$,
$y=\mathsf{inv}(a,p_a)$, and $z=a$. Hence the two outer witnesses are
identified, which is the stated equality.
\end{proof}

\begin{theorem}[Field inverse product reversal from apartness-indexed inverses]
\label{thm:field-inverse-product-reverse-from-apartness}
For a BEDC history-level $\FieldUp$ candidate with proof-indexed inverses,
explicit witnesses $p_{ab}:\mathsf{NonZero}_F(ab)$,
$p_a:\mathsf{NonZero}_F(a)$, and $p_b:\mathsf{NonZero}_F(b)$ determine
the product inverse by
$$
\mathsf{inv}(ab,p_{ab})\hsame \mathsf{inv}(b,p_b)\mathsf{inv}(a,p_a).
$$
\end{theorem}
\begin{proof}
The reversed product $\mathsf{inv}(b,p_b)\mathsf{inv}(a,p_a)$ is a right
inverse of $ab$: reassociate to expose $b\mathsf{inv}(b,p_b)$, collapse it
to the unit, and finish with the right inverse law for $a$. The indexed left
inverse of $ab$ and group uniqueness of left and right inverse witnesses
identify the chosen inverse of $ab$ with this reversed product.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.field\_inverse\_product\_reverse\_from\_apartness}

\begin{theorem}[Field multiplication by an indexed right inverse cancels]
\label{thm:field-mul-inverse-right-cancel-from-apartness}
For a BEDC history-level $\FieldUp$ candidate with proof-indexed inverses,
any witness $p_b:\mathsf{NonZero}_F(b)$ determines
$$
(a b)\mathsf{inv}(b,p_b)\hsame a.
$$
\end{theorem}
\begin{proof}
Reassociate $(a b)\mathsf{inv}(b,p_b)$ to
$a(b\mathsf{inv}(b,p_b))$. The apartness-indexed right inverse law collapses
the tail to the unit, and the right identity law leaves $a$.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.field\_mul\_inverse\_right\_cancel\_from\_apartness}

\begin{theorem}[Field right-multiplication equation solution from apartness]
\label{thm:field-right-mul-equation-solution-from-apartness}
For a BEDC history-level $\FieldUp$ candidate with proof-indexed inverses,
any witness $p_b:\mathsf{NonZero}_F(b)$ makes $xb\hsame a$ imply
$x\hsame a\mathsf{inv}(b,p_b)$.
\end{theorem}
\begin{proof}
First cancel the indexed right inverse in $(x b)\mathsf{inv}(b,p_b)$ to
identify it with $x$. Then transport the supplied equality $x b\hsame a$
through right multiplication by $\mathsf{inv}(b,p_b)$.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.field\_right\_mul\_equation\_solution\_from\_apartness}

\begin{theorem}[Field multiplication by an indexed left inverse cancels]
\label{thm:field-mul-inverse-left-cancel-from-apartness}
For a BEDC history-level $\FieldUp$ candidate with proof-indexed inverses,
any witness $p_a:\mathsf{NonZero}_F(a)$ determines
$$
\mathsf{inv}(a,p_a)(a b)\hsame b.
$$
\end{theorem}
\begin{proof}
Reassociate $\mathsf{inv}(a,p_a)(a b)$ to
$(\mathsf{inv}(a,p_a)a)b$. The apartness-indexed left inverse law collapses
the head to the unit, and the left identity law leaves $b$.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.field\_mul\_inverse\_left\_cancel\_from\_apartness}

\begin{theorem}[Field left-multiplication equation solution from apartness]
\label{thm:field-left-mul-equation-solution-from-apartness}
For a BEDC history-level $\FieldUp$ candidate with proof-indexed inverses,
any witness $p_a:\mathsf{NonZero}_F(a)$ makes $a x\hsame c$ imply
$x\hsame \mathsf{inv}(a,p_a)c$.
\end{theorem}
\begin{proof}
First use Theorem~\ref{thm:field-mul-inverse-left-cancel-from-apartness}
to identify $x$ with $\mathsf{inv}(a,p_a)(a x)$.

Then transport the supplied equality $a x\hsame c$ through left multiplication
by $\mathsf{inv}(a,p_a)$, yielding
$\mathsf{inv}(a,p_a)(a x)\hsame \mathsf{inv}(a,p_a)c$.
\end{proof}

\begin{theorem}[Field middle multiplication equation solution from apartness]
\label{thm:field-middle-mul-equation-solution-from-apartness}
For a BEDC history-level $\FieldUp$ candidate with proof-indexed inverses,
explicit witnesses $p_a:\mathsf{NonZero}_F(a)$ and
$p_b:\mathsf{NonZero}_F(b)$ make
$$
(a x)b\hsame c
$$
imply
$$
x\hsame \mathsf{inv}(a,p_a)(c\mathsf{inv}(b,p_b)).
$$
\end{theorem}
\begin{proof}
First solve the right factor equation using the indexed right inverse of $b$,
obtaining $a x\hsame c\mathsf{inv}(b,p_b)$. Then cancel the left factor
with the indexed left inverse of $a$, transporting the right-factor solution
through left multiplication by $\mathsf{inv}(a,p_a)$.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.field\_middle\_mul\_equation\_solution\_from\_apartness}

\begin{definition}[Field ledger policy]
\label{def:field-ledger-policy}
Record the ring ledger and the $\mathsf{NonZero}_F$ witness used for each inverse.
\end{definition}

\begin{definition}[\(\FieldUp\) certificate obligations]
\label{def:field-certificate-obligations}
The carrier, source, pattern, classifier, stability, and ledger data listed above are the obligations required for the interface name \(\FieldUp\).
\end{definition}

\paragraph{Per-object certificate record.}
\begin{enumerate}
  \item Traditional target: constructive field.
  \item BEDC source: $\RatUp$ plus $\CommRingUp$ data.
  \item Carrier choice: the $\RatUp$ carrier with $\mathsf{NonZero}_F(a)$ derived from apartness from $0$.
  \item Classifier choice: the rational classifier plus $\mathsf{NonZero}_F$ transport.
  \item Stability obligations: commutative-ring laws, contradiction of $\mathsf{NonZero}_F(0)$, nonzero transport, inverse closure, and inverse laws indexed by nonzero witnesses.
  \item Ledger: ring expression tree plus the $\mathsf{NonZero}_F$ witness for each inverse.
  \item Core theorem list: inverse closure, left inverse, right inverse, nonzero stability, inverse congruence from apartness transport, inverse involution under explicit nonzero evidence, right-multiplication equation solving, left-multiplication equation solving, middle multiplication equation solving, and distributivity inherited from $\CommRingUp$.
  \item Exactness predicate: coverage, soundness, and completeness for admitted field packages.
  \item Standard bridge: deferred conservative bridge.
  \item Status: ProofTarget.
\end{enumerate}

\paragraph{Primitive boundary.}
\(\FieldUp\) is not a BEDC primitive. It is admitted only when those obligations are supplied.

\begin{remark}[Zero-axiom encoding note]
\label{rem:field-zero-axiom-note}
Replaces LEM-based \(a\ne0\) reasoning with explicit apartness witnesses.
\end{remark}
