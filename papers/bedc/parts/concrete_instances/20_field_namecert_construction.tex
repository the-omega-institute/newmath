\begin{theorem}[RatUp FieldUp transported strict-support non-singleton endpoint]
\label{thm:ratup-fieldup-transported-strict-support-non-singleton}
For the concrete denominator tuple $(C_D^u,R_D^u,\mu_D^u,e_D^u,N_D^u)$ and the
contextual action $\Gamma$ of
\autoref{def:ratup-fieldup-denominator-contextual-action-instance}, let
$x:=\Gamma_{p',q',s',t'}(\Gamma_{p,q,s,t}(h))$. Under the hypotheses of
\autoref{thm:ratup-fieldup-transported-support-neutrality-row}, suppose also
that $N_D^u(h)$ holds. Then the concrete $\FieldUp(\mathsf{RatUp})$ exit
certificate contains all of the following endpoint data:
\begin{enumerate}
  \item $C_D^u(x)$;
  \item $N_D^u(x)$;
  \item $\mathsf{RatHistoryCarrier}(x)$;
  \item $\mathsf{fieldSingletonEmptyCarrier}(x)\to\bot$.
\end{enumerate}
Thus the transported strict-support endpoint of the
$\FieldUp(\mathsf{RatUp})$ exit certificate cannot collapse into the singleton
empty-history field carrier.
\end{theorem}
\begin{proof}
Apply
\autoref{thm:ratup-fieldup-exit-transported-support-nondegenerate} at the same
transported endpoint. It gives the carried endpoint clause $C_D^u(x)$ and the
strict-support equivalence
$$
N_D^u(x)\Longleftrightarrow N_D^u(h).
$$

Use the reverse direction of this equivalence with the displayed witness
$N_D^u(h)$. This produces the transported strict-support witness $N_D^u(x)$.

Now read the denominator tuple row from
\autoref{thm:field-rat-denominator-fieldup-contextual-multiplicative-boundary}.
There $N_D^u(y)=\mathsf{RatHistoryCarrier}(y)$ for every denominator history
$y$, so the transported support witness gives $\mathsf{RatHistoryCarrier}(x)$.

Finally apply
\autoref{thm:field-singleton-schema-disjoint-from-rat-history-carrier} to
$x$. Any witness of $\mathsf{fieldSingletonEmptyCarrier}(x)$, together with the
Rat history carrier witness just obtained, yields contradiction. This is the
displayed map $\mathsf{fieldSingletonEmptyCarrier}(x)\to\bot$.
\end{proof}

\begin{theorem}[RatUp FieldUp transported strict-support contextual singleton exclusion]
\label{thm:ratup-fieldup-transported-strict-support-contextual-singleton-exclusion}
For the concrete denominator tuple $(C_D^u,R_D^u,\mu_D^u,e_D^u,N_D^u)$ and the
contextual action $\Gamma$ of
\autoref{def:ratup-fieldup-denominator-contextual-action-instance}, let
$x:=\Gamma_{p',q',s',t'}(\Gamma_{p,q,s,t}(h))$. Under the hypotheses of
\autoref{thm:ratup-fieldup-transported-support-neutrality-row}, suppose also
that $N_D^u(h)$ holds. Then for every context history $r:\mathsf{BHist}$ both
boundary collapses are impossible:
\begin{enumerate}
  \item $\mathsf{fieldSingletonEmptyCarrier}(\mathsf{append}(r,x))\to\bot$;
  \item $\mathsf{fieldSingletonEmptyCarrier}(\mathsf{append}(x,r))\to\bot$.
\end{enumerate}
\end{theorem}
\begin{proof}
Apply
\autoref{thm:ratup-fieldup-transported-strict-support-non-singleton} at the
same transported endpoint. The endpoint data supplied there includes
$\mathsf{RatHistoryCarrier}(x)$.

For the right-factor append boundary, apply
\autoref{thm:field-singleton-empty-append-rat-carrier-absurd} with the context
history $r$ and the RatUp factor $x$. The displayed Rat carrier witness turns
any witness of
$\mathsf{fieldSingletonEmptyCarrier}(\mathsf{append}(r,x))$ into
contradiction.

For the left-factor append boundary, apply
\autoref{thm:field-singleton-empty-append-left-rat-carrier-absurd} with the
same RatUp factor $x$ and context history $r$. The same carrier witness turns
any witness of $\mathsf{fieldSingletonEmptyCarrier}(\mathsf{append}(x,r))$
into contradiction.
\end{proof}

\begin{theorem}[RatUp FieldUp transported strict-support separation package]
\label{thm:ratup-fieldup-transported-strict-support-separation-package}
For the concrete denominator tuple $(C_D^u,R_D^u,\mu_D^u,e_D^u,N_D^u)$ and the
contextual action $\Gamma$ of
\autoref{def:ratup-fieldup-denominator-contextual-action-instance}, let
$x:=\Gamma_{p',q',s',t'}(\Gamma_{p,q,s,t}(h))$. Under the hypotheses of
\autoref{thm:ratup-fieldup-transported-support-neutrality-row}, suppose also
that $N_D^u(h)$ holds. Then the concrete $\FieldUp(\mathsf{RatUp})$ exit
certificate contains the following separation package:
\begin{enumerate}
  \item $C_D^u(x)$;
  \item $N_D^u(x)$;
  \item $\mathsf{RatHistoryCarrier}(x)$;
  \item $\mathsf{fieldSingletonEmptyCarrier}(x)\to\bot$;
  \item $x\hsame\emp\to\bot$.
\end{enumerate}
\end{theorem}
\begin{proof}
First apply
\autoref{thm:ratup-fieldup-transported-strict-support-non-singleton} with the
same denominator tuple, action, endpoint, and hypotheses. This recovers
$C_D^u(x)$, $N_D^u(x)$, $\mathsf{RatHistoryCarrier}(x)$, and
$\mathsf{fieldSingletonEmptyCarrier}(x)\to\bot$.

It remains to separate the endpoint from the kernel empty history itself. Apply
\autoref{thm:rat-history-carrier-not-empty} to the recovered
$\mathsf{RatHistoryCarrier}(x)$ witness. The resulting map sends any equality
$x\hsame\emp$ to contradiction, giving $x\hsame\emp\to\bot$.

Collect these five clauses as the displayed transported strict-support
separation package.
\end{proof}

\begin{theorem}[RatUp FieldUp concrete exit certificate package]
\label{thm:ratup-fieldup-concrete-exit-certificate-package}

source tuple $\mathsf{RatUpFieldSource}_F$, the concrete
$\FieldUp(\mathsf{RatUp})$ exit certificate simultaneously contains the
following certificate data:
\begin{enumerate}
  \item the obligation rows of \autoref{def:field-certificate-obligations};
  \item the denominator multiplicative coverage of
  \autoref{thm:field-rat-denominator-fieldup-multiplicative-obligation-coverage};
  \item the concrete exit-row closure of
  \autoref{thm:ratup-fieldup-concrete-exit-row-closure};
  \item the empty-context nondegenerate witness of
  \autoref{thm:ratup-fieldup-empty-context-nondegenerate-witness};
  \item the transported strict-support non-singleton endpoint of
  \autoref{thm:ratup-fieldup-transported-strict-support-non-singleton}.
\end{enumerate}
Hence the exit package is a concrete $\mathsf{BHist}$-anchored
$\mathsf{ProofTarget}$ over $\mathsf{RatUpFieldSource}_F$, not an abstract
operator schema.
\end{theorem}
\begin{proof}
Begin with \autoref{def:field-certificate-obligations}. It fixes the rows that a
$\FieldUp(\mathsf{RatUp})$ certificate must carry: source, carrier, classifier,
ring base, apartness, inverse, product, denominator, contextual denominator,
affine, ledger, exactness, and status rows.

Next apply
\autoref{thm:field-rat-denominator-fieldup-multiplicative-obligation-coverage}.
This supplies the denominator multiplicative row coverage inside the same field
certificate language, so the denominator component is read as a certificate row
rather than as an external monoid fact.

Apply \autoref{thm:ratup-fieldup-concrete-exit-row-closure}. It repacks the
semantic certificate rows, source anchoring rows, support-sensitive coverage, and
non-singleton endpoint data as one concrete exit-row closure over
$\mathsf{RatUpFieldSource}_F$.

The nondegenerate empty-context component is supplied by
\autoref{thm:ratup-fieldup-empty-context-nondegenerate-witness}. It gives the
wrapped rational endpoint, its rational-history carrier witness, its singleton
empty rejection map, and the nonzero-complement clause under empty singleton
wrappers.

Finally apply
\autoref{thm:ratup-fieldup-transported-strict-support-non-singleton}. This gives
the transported endpoint clauses $C_D^u(x)$, $N_D^u(x)$,
$\mathsf{RatHistoryCarrier}(x)$, and
$\mathsf{fieldSingletonEmptyCarrier}(x)\to\bot$.

Combining these five cited sites yields a single concrete
$\mathsf{BHist}$-anchored certificate package with status
$\mathsf{ProofTarget}$. Its data are the rows of the displayed
$\FieldUp(\mathsf{RatUp})$ instance, so the conclusion is a concrete exit
certificate for $\mathsf{RatUpFieldSource}_F$ and not an abstract operator
schema.
\end{proof}

\begin{theorem}[RatUp FieldUp transported exit row nondegenerate closure]
\label{thm:ratup-fieldup-transported-exit-row-nondegenerate-closure}
For the concrete source tuple $\mathsf{RatUpFieldSource}_F$, the concrete
denominator tuple $(C_D^u,R_D^u,\mu_D^u,e_D^u,N_D^u)$, and the contextual
action $\Gamma$ of
\autoref{def:ratup-fieldup-denominator-contextual-action-instance}, let
$x:=\Gamma_{p',q',s',t'}(\Gamma_{p,q,s,t}(h))$. Under the hypotheses of
\autoref{thm:ratup-fieldup-transported-support-neutrality-row}, suppose also
that $N_D^u(h)$ holds. Then the single concrete
$\FieldUp(\mathsf{RatUp})$ exit row package simultaneously contains:
\begin{enumerate}
  \item the semantic certificate assembly and source anchoring rows recorded by
  \autoref{thm:ratup-fieldup-concrete-exit-row-closure};
  \item the transported carrier clause $C_D^u(x)$ and the strict-support
  equivalence
  $$
  N_D^u(x)\Longleftrightarrow N_D^u(h);
  $$
  \item the endpoint facts $C_D^u(x)$, $N_D^u(x)$,
  $\mathsf{RatHistoryCarrier}(x)$, and
  $\mathsf{fieldSingletonEmptyCarrier}(x)\to\bot$;
  \item the separation clause that no carrier-coverage map
  $$
  \kappa:\mathsf{RatHistoryCarrier}(x)\to
  \mathsf{fieldSingletonEmptyCarrier}(x)
  $$
  exists for the transported strict-support endpoint of this exit row.
\end{enumerate}
Thus the transported endpoint, its concrete exit-row closure, and its
source-anchored certificate rows are one nondegenerate
$\FieldUp(\mathsf{RatUp})$ exit package.
\end{theorem}
\begin{proof}
First fix the row package. Apply
\autoref{thm:ratup-fieldup-concrete-exit-row-closure}. It identifies the
concrete $\FieldUp(\mathsf{RatUp})$ exit rows over
$\mathsf{RatUpFieldSource}_F$ and records the semantic certificate assembly
together with the source anchoring rows.

Next read the transported endpoint. Apply
\autoref{thm:ratup-fieldup-exit-transported-support-nondegenerate} under the
same hypotheses. This supplies the transported carrier clause $C_D^u(x)$ and
the strict-support equivalence
$$
N_D^u(x)\Longleftrightarrow N_D^u(h).
$$
Using the assumed witness $N_D^u(h)$ gives $N_D^u(x)$.

Now use the preceding endpoint theorem
\autoref{thm:ratup-fieldup-transported-strict-support-non-singleton}. It gives
the concrete Rat carrier witness $\mathsf{RatHistoryCarrier}(x)$ and the
singleton rejection map
$\mathsf{fieldSingletonEmptyCarrier}(x)\to\bot$ for the same endpoint.

It remains to rule out coverage by the singleton empty-history source. Suppose
that a carrier-coverage map
$$
\kappa:\mathsf{RatHistoryCarrier}(x)\to
\mathsf{fieldSingletonEmptyCarrier}(x)
$$
exists at this transported strict-support endpoint. Applying $\kappa$ to the
witness $\mathsf{RatHistoryCarrier}(x)$ gives
$\mathsf{fieldSingletonEmptyCarrier}(x)$, contradicting the displayed rejection
map. Hence no such carrier-coverage map exists.
\end{proof}
