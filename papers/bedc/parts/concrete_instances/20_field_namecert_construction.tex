\section{Rat denominator empty-unit continuation monoid instance}
\label{sec:field-rat-denominator-empty-unit-continuation-monoid-instance}

\begin{definition}[Rat denominator empty-unit continuation monoid instance]
\label{def:field-rat-denominator-empty-unit-continuation-monoid-instance}
The empty-unit denominator continuation monoid data for
$\FieldUp(\mathsf{RatUp})$ are the concrete tuple
$(C_D^u,R_D^u,\mu_D^u,e_D^u)$ defined by
$$
C_D^u(h)\Longleftrightarrow\mathsf{RatDenomUnitCarrier}(h),
$$
$$
R_D^u(h,k)\Longleftrightarrow\mathsf{RatDenomUnitClassifier}(h,k),
$$
$$
\mu_D^u(h,k):=\mathsf{append}(h,k),
$$
and $e_D^u:=\emp$. For every pair of histories $h,k:\mathsf{BHist}$, the
multiplication is equipped with the canonical continuation witness
$$
\Cont(h,k,\mathsf{append}(h,k)).
$$
\end{definition}

\begin{theorem}[Rat denominator empty-unit continuation monoid laws]
\label{thm:field-rat-denominator-empty-unit-continuation-monoid-laws}
The concrete tuple
$(C_D^u,R_D^u,\mu_D^u,e_D^u)$ of
\autoref{def:field-rat-denominator-empty-unit-continuation-monoid-instance}
satisfies the denominator empty-unit continuation monoid laws:
\begin{enumerate}
  \item The unit is carried: $C_D^u(e_D^u)$.
  \item The multiplication is closed: if $C_D^u(h)$ and $C_D^u(k)$, then
  $C_D^u(\mu_D^u(h,k))$.
  \item The left and right unit products are classified with their endpoint:
  for every $h$ with $C_D^u(h)$,
  $R_D^u(\mu_D^u(e_D^u,h),h)$ and
  $R_D^u(\mu_D^u(h,e_D^u),h)$.
  \item The multiplication is associative under $R_D^u$: if
  $C_D^u(h)$, $C_D^u(k)$, and $C_D^u(\ell)$, then
  $R_D^u(\mu_D^u(\mu_D^u(h,k),\ell),
  \mu_D^u(h,\mu_D^u(k,\ell)))$.
  \item The multiplication respects classified endpoints: if $R_D^u(h,h')$
  and $R_D^u(k,k')$, then
  $R_D^u(\mu_D^u(h,k),\mu_D^u(h',k'))$.
\end{enumerate}
\end{theorem}
\leanchecked{BEDC.Derived.FieldUp.field\_rat\_denominator\_empty\_unit\_continuation\_monoid\_laws}
\begin{proof}
Unfold $C_D^u$, $R_D^u$, $\mu_D^u$, and $e_D^u$. The displayed multiplication
is the external append operation, so each product carries the canonical
continuation witness $\Cont(h,k,\mathsf{append}(h,k))$.

The carried empty unit, append closure, and the two one-sided unit
classifications are exactly the first four clauses of
\autoref{thm:field-rat-denominator-continuation-adjoined-unit-laws} applied to
those canonical continuation witnesses.

For associativity, use the same adjoined-unit theorem with the four canonical
continuation witnesses for the two inner and two outer products. Its fifth
clause gives the classifier comparison between the two bracketed append
results.

It remains to show multiplication congruence. From $R_D^u(h,h')$ and
$R_D^u(k,k')$, unfold the classifier to obtain the four carried endpoint
witnesses together with $h\hsame h'$ and $k\hsame k'$. Apply
\autoref{thm:external-append-hsame-congruence} to the two $\hsame$ comparisons;
append closure for $C_D^u$ supplies the two carried product witnesses. These
three pieces are exactly
$R_D^u(\mu_D^u(h,k),\mu_D^u(h',k'))$.

Thus the denominator continuation fields are concrete BEDC data: carrier
$\mathsf{RatDenomUnitCarrier}$, classifier
$\mathsf{RatDenomUnitClassifier}$, multiplication $\mathsf{append}$, and unit
$\emp$, with no abstract operator field being used.
\end{proof}

\section{Rat denominator unit endpoint exactness}
\label{sec:field-rat-denominator-unit-endpoint-exactness}

\begin{theorem}[Rat denominator continuation right-unit endpoint iff empty]
\label{thm:field-rat-denominator-continuation-right-unit-endpoint-iff-empty}
For every history $u:\mathsf{BHist}$, the right-unit classifier law is
equivalent to external emptiness:
$$
\left(
\forall d,r:\mathsf{BHist}.\;
\mathsf{RatHistoryCarrier}(d)\to
\Cont(d,u,r)\to
\mathsf{RatHistoryClassifier}(r,d)
\right)
\Longleftrightarrow
u\hsame\emp.
$$
\end{theorem}
\leanchecked{BEDC.Derived.FieldUp.field\_rat\_denominator\_continuation\_right\_unit\_law\_endpoint\_empty\_iff}
\begin{proof}
For the forward direction, apply
Theorem~\ref{thm:field-rat-denominator-continuation-unit-endpoint-empty} to
the displayed right-unit classifier law. Its first endpoint-forcing clause
returns $u\hsame\emp$.

Conversely, assume $u\hsame\emp$, a carrier witness
$\mathsf{RatHistoryCarrier}(d)$, and a displayed continuation
$\Cont(d,u,r)$. Transport the second continuation endpoint along
$u\hsame\emp$ to obtain a continuation presentation $\Cont(d,\emp,r)$ with
the same displayed result. The right external-boundary law in
Theorem~\ref{thm:field-rat-denominator-continuation-external-boundary-laws}
then yields $\mathsf{RatHistoryClassifier}(r,d)$.
\end{proof}

\begin{theorem}[Rat denominator continuation left-unit endpoint iff empty]

\left(

\end{theorem}
\leanchecked{BEDC.Derived.FieldUp.field\_rat\_denominator\_continuation\_left\_unit\_law\_endpoint\_empty\_iff}

Conversely, assume $u\hsame\emp$, a carrier witness

\end{proof}

\begin{theorem}[Rat denominator external unit separation package]
\label{thm:field-rat-denominator-external-unit-separation-package}
For the concrete denominator carrier, classifier, and continuation operation
given by
$$
C_D(h)\Longleftrightarrow\mathsf{RatHistoryCarrier}(h),
$$
$$
R_D(h,k)\Longleftrightarrow\mathsf{RatHistoryClassifier}(h,k),
$$
with operation represented by $\Cont$, every history $u:\mathsf{BHist}$
satisfies the following three clauses.
\begin{enumerate}
\item The right classifier-unit law
$$
\forall d,r:\mathsf{BHist}.\;
C_D(d)\to
\Cont(d,u,r)\to
R_D(r,d)
$$
is equivalent to $u\hsame\emp$.
\item The left classifier-unit law
$$
\forall d,r:\mathsf{BHist}.\;
C_D(d)\to
\Cont(u,d,r)\to
R_D(r,d)
$$
is equivalent to $u\hsame\emp$.
\item If $C_D(u)$ holds, then either one-sided classifier-unit law displayed
above implies $\mathsf{False}$.
\end{enumerate}
\end{theorem}
\leanchecked{BEDC.Derived.FieldUp.field\_rat\_denominator\_external\_unit\_separation\_package}
\begin{proof}
The displayed definitions of $C_D$ and $R_D$ identify the right classifier-unit
law with the concrete law in
\autoref{thm:field-rat-denominator-continuation-right-unit-endpoint-iff-empty}.
That theorem gives the first equivalence exactly.

The same unpacking identifies the left classifier-unit law with the concrete
law in
\autoref{thm:field-rat-denominator-continuation-left-unit-endpoint-iff-empty}.
That theorem gives the second equivalence.

For the carried exclusion, assume $C_D(u)$. This is precisely
$\mathsf{RatHistoryCarrier}(u)$. The right one-sided law contradicts
\autoref{thm:field-rat-denominator-continuation-no-internal-unit}, while the
left one-sided law contradicts
\autoref{thm:field-rat-denominator-continuation-no-internal-left-unit}.

Equivalently, either endpoint equivalence first yields $u\hsame\emp$, and then
\autoref{thm:rat-history-carrier-not-empty} excludes the simultaneous carried
witness. Thus the empty endpoint is external to the positive Rat denominator
carrier even though it is exact for the classifier-unit boundary laws.
\end{proof}

\begin{theorem}[Rat denominator concrete continuation is cancellative and unitless]
\label{thm:field-rat-denominator-concrete-cancellative-unitless-semigroup}
For the concrete data $C_D$, $R_D$, and $\mu_D$ of
\autoref{def:field-rat-denominator-concrete-continuation-instance}, the
denominator-side fragment consumed by the $\FieldUp(\mathsf{RatUp})$
certificate has the following form.
\begin{enumerate}
  \item $C_D$, $R_D$, and $\mu_D$ form the carried commutative semigroup
  fragment: multiplication closes on carried denominators, commutativity and
  associativity hold up to $R_D$, binary classifier congruence is respected in
  both inputs, and the Rat history ledger is closed under classified carried
  right tails.
  \item For all histories $p,q,d,e,pd,pe,left,right:\mathsf{BHist}$, if
  $\mathsf{UnaryHistory}(p)$, $\mathsf{UnaryHistory}(q)$, $C_D(d)$, $C_D(e)$,
  $\Cont(p,d,pd)$, $\Cont(p,e,pe)$, $\Cont(pd,q,left)$, and
  $\Cont(pe,q,right)$, then the common unary context is exact for both the
  classifier and the ledger:
  $$
  R_D(left,right)\Longleftrightarrow R_D(d,e).
  $$
  $$
  \mathsf{RatHistoryLedgerPolicy}(left,right)\Longleftrightarrow
  \mathsf{RatHistoryLedgerPolicy}(d,e).
  $$
  \item For every endpoint $u:\mathsf{BHist}$, if either one-sided
  classifier-unit law
  $$
  \forall d,r:\mathsf{BHist}.\;
  C_D(d)\to
  \Cont(d,u,r)\to
  R_D(r,d)
  $$
  or
  $$
  \forall d,r:\mathsf{BHist}.\;
  C_D(d)\to
  \Cont(u,d,r)\to
  R_D(r,d)
  $$
  holds, then $u\hsame\emp$. If also $C_D(u)$, then $\mathsf{False}$.
\end{enumerate}
\end{theorem}
\begin{proof}
Unfold the concrete data of
\autoref{def:field-rat-denominator-concrete-continuation-instance}, so that
$C_D$ is $\mathsf{RatHistoryCarrier}$, $R_D$ is
$\mathsf{RatHistoryClassifier}$, and $\mu_D$ is append with its canonical
continuation witness.

The semigroup fragment is exactly
\autoref{thm:field-rat-denominator-concrete-continuation-instance-laws}. Its
clauses give carrier closure, commutativity, associativity, binary classifier
congruence, and ledger closure for the displayed concrete operation.

For the common-context clause, apply
\autoref{thm:field-rat-denominator-continuation-common-context-exactness-package}
to the two unary contexts and the four displayed continuation witnesses. After
the same unfolding of $C_D$ and $R_D$, its two conclusions are precisely the
classifier cancellation equivalence and the ledger exactness equivalence in the
statement.

For the endpoint obstruction, use
\autoref{thm:field-rat-denominator-continuation-right-unit-endpoint-iff-empty}
for the right-unit classifier law and
\autoref{thm:field-rat-denominator-continuation-left-unit-endpoint-iff-empty}
for the left-unit classifier law. Either case yields $u\hsame\emp$. If a
carrier witness $C_D(u)$ is also present, unfold $C_D$ and apply
\autoref{thm:rat-history-carrier-not-empty} to that witness and the comparison
$u\hsame\emp$, giving $\mathsf{False}$.
\end{proof}

\begin{theorem}[Rat denominator continuation exact external-unit package]
\label{thm:field-rat-denominator-concrete-external-unit-package}
For the concrete Rat denominator continuation data
$(C_D,R_D,\mu_D)$ of
\autoref{def:field-rat-denominator-concrete-continuation-instance}, the
following package holds.
\begin{enumerate}
  \item If $C_D(d)$ and $C_D(e)$, then $C_D(\mu_D(d,e))$.
  \item If $C_D(d)$ and $C_D(e)$, then
  $R_D(\mu_D(d,e),\mu_D(e,d))$.
  \item If $C_D(d)$, $C_D(e)$, and $C_D(f)$, then
  $R_D(\mu_D(\mu_D(d,e),f),\mu_D(d,\mu_D(e,f)))$.
  \item If $R_D(d,d')$ and $R_D(e,e')$, then
  $R_D(\mu_D(d,e),\mu_D(d',e'))$.
  \item If $\mathsf{RatHistoryLedgerPolicy}(\rho,v)$, $C_D(e)$, and
  $R_D(e,e')$, then
  $\mathsf{RatHistoryLedgerPolicy}(\mu_D(\rho,e),\mu_D(v,e'))$.
  \item For every endpoint $u:\mathsf{BHist}$,

  $$
  \left(
  \forall d,r:\mathsf{BHist}.\;
  C_D(d)\to
  \Cont(d,u,r)\to
  R_D(r,d)
  \right)
  \Longleftrightarrow
  u\hsame\emp.
  $$

  \item For every endpoint $u:\mathsf{BHist}$,

  $$
  \left(
  \forall d,r:\mathsf{BHist}.\;
  C_D(d)\to
  \Cont(u,d,r)\to
  R_D(r,d)
  \right)
  \Longleftrightarrow
  u\hsame\emp.
  $$

  \item If $C_D(u)$, then neither displayed one-sided classifier-unit law can
  hold for $u$.
  \item If $u$ and $v$ each satisfy either displayed one-sided classifier-unit
  law, then $u\hsame v$.
\end{enumerate}
\end{theorem}
\begin{proof}
Unfold the concrete names $C_D$, $R_D$, and $\mu_D$ only as needed. The first
five clauses are exactly the first law group in
\autoref{thm:field-rat-denominator-concrete-continuation-instance-laws}:
carrier closure, commutative classifier, associative classifier, binary
classifier congruence, and ledger closure.

For the right external-unit exactness clause, read $C_D$ as
$\mathsf{RatHistoryCarrier}$ and $R_D$ as $\mathsf{RatHistoryClassifier}$. Apply
\autoref{thm:field-rat-denominator-continuation-right-unit-endpoint-iff-empty}
to the displayed right-unit law.

For the left external-unit exactness clause, use the same identifications and
apply
\autoref{thm:field-rat-denominator-continuation-left-unit-endpoint-iff-empty}.

For carried-endpoint exclusion, suppose $C_D(u)$ and one of the displayed
one-sided classifier-unit laws holds. The corresponding exactness clause gives
$u\hsame\emp$. Since $C_D(u)$ is $\mathsf{RatHistoryCarrier}(u)$, rational
history carrier nonemptiness,
\autoref{thm:rat-history-carrier-not-empty}, contradicts $u\hsame\emp$.

Finally, take two one-sided classifier-unit endpoints $u$ and $v$. The
right/right, left/left, and right/left cases are the three clauses of
\autoref{thm:field-rat-denominator-continuation-unit-endpoint-unique}. The
left/right case follows by applying the mixed clause in the opposite order and
then using symmetry of $\hsame$.
\end{proof}

\textbf{Traditional target}: denominator-side continuation package for
$\FieldUp(\mathsf{RatUp})$ with an external unit endpoint; \textbf{BEDC source}:
$\mathsf{RatHistoryCarrier}$, $\mathsf{RatHistoryClassifier}$, $\Cont$,
$\mathsf{append}$, and $\mathsf{RatHistoryLedgerPolicy}$; \textbf{carrier
choice}: $C_D$ from
\autoref{def:field-rat-denominator-concrete-continuation-instance};
\textbf{classifier choice}: $R_D$ from
\autoref{def:field-rat-denominator-concrete-continuation-instance};
\textbf{stability obligation}: carrier closure, commutative classifier,
associative classifier, classifier congruence, ledger closure, and endpoint
uniqueness; \textbf{ledger}: the Rat history ledger policy transported through
$\mu_D$; \textbf{core theorem}:
\autoref{thm:field-rat-denominator-concrete-external-unit-package};
\textbf{exactness predicate}: the right and left external-unit iff clauses
identifying the endpoint with $\emp$; \textbf{standard bridge}: deferred;
\textbf{status}: $\mathsf{ProofTarget}$.

\begin{definition}[Rat denominator external-unit envelope]
\label{def:field-rat-denominator-unit-envelope}
The Rat denominator external-unit envelope over the concrete continuation
instance of
\autoref{def:field-rat-denominator-concrete-continuation-instance} has carrier,
classifier, operation, and unit
$$
C_D^\ast(h)\Longleftrightarrow
\mathsf{RatHistoryCarrier}(h)\lor h\hsame\emp,
$$
$$
R_D^\ast(h,k)\Longleftrightarrow
\left(
\mathsf{RatHistoryCarrier}(h)\land
\mathsf{RatHistoryCarrier}(k)\land
\mathsf{RatHistoryClassifier}(h,k)
\right)
\lor
\left(h\hsame\emp\land k\hsame\emp\right),
$$
$$
\mu_D^\ast(h,k):=\mathsf{append}(h,k),
$$
with external unit $e_D^\ast:=\emp$. The first classifier branch is the original
Rat denominator classifier guarded by carried denominator evidence. The second
branch is the single external endpoint clause, so mixed carried-empty pairs are
not classified merely by adjoining $e_D^\ast$.
\end{definition}

\begin{theorem}[Rat denominator external-unit envelope classifier exactness]
\label{thm:field-rat-denominator-unit-envelope-classifier-exactness}
For all histories $h,k:\mathsf{BHist}$, the classifier $R_D^\ast$ of
\autoref{def:field-rat-denominator-unit-envelope} splits exactly as follows.
\begin{enumerate}
  \item If $\mathsf{RatHistoryCarrier}(h)$ and
  $\mathsf{RatHistoryCarrier}(k)$, then
  $$
  R_D^\ast(h,k)\Longleftrightarrow
  \mathsf{RatHistoryClassifier}(h,k).
  $$
  \item If $\mathsf{RatHistoryCarrier}(h)$ and $k\hsame\emp$, then
  $\neg R_D^\ast(h,k)$.
  \item If $h\hsame\emp$ and $\mathsf{RatHistoryCarrier}(k)$, then
  $\neg R_D^\ast(h,k)$.
  \item If $h\hsame\emp$ and $k\hsame\emp$, then $R_D^\ast(h,k)$.
\end{enumerate}
\end{theorem}
\begin{proof}
Unfold $R_D^\ast$. In the carried-carried case, the forward implication cannot
come from the empty-empty branch, because that branch would make the carried
endpoint $h$ $\hsame$ to $\emp$, contradicting
$\mathsf{RatHistoryCarrier\_not\_empty}$. The remaining branch is exactly the
guarded $\mathsf{RatHistoryClassifier}(h,k)$ branch. The reverse implication
uses the two displayed carrier witnesses together with that guarded branch.

For a carried-left and empty-right pair, the guarded carried branch would give
$\mathsf{RatHistoryCarrier}(k)$, and
$\mathsf{RatHistoryCarrier\_not\_empty}$ rejects it against $k\hsame\emp$. The
empty-empty branch would make $h\hsame\emp$, which is rejected by the same
nonemptiness theorem applied to $\mathsf{RatHistoryCarrier}(h)$.

The empty-left and carried-right pair is symmetric. The guarded carried branch
would make the left endpoint a Rat carrier and hence nonempty, while the
empty-empty branch contradicts the carried right endpoint through
$\mathsf{RatHistoryCarrier\_not\_empty}$.

If both endpoints are $\hsame$ to $\emp$, choose the second branch in the
definition of $R_D^\ast$. This is precisely the external endpoint clause.
\end{proof}

\begin{theorem}[Rat denominator external-unit envelope monoid laws]
\label{thm:field-rat-denominator-unit-envelope-monoid-laws}
The data
$(C_D^\ast,R_D^\ast,\mu_D^\ast,e_D^\ast)$ of
\autoref{def:field-rat-denominator-unit-envelope} satisfy the following laws.
\begin{enumerate}
  \item Carrier inhabitance: $C_D^\ast(e_D^\ast)$.
  \item Closure: if $C_D^\ast(h)$ and $C_D^\ast(k)$, then
  $C_D^\ast(\mu_D^\ast(h,k))$.
  \item Associativity under the classifier: if $C_D^\ast(h)$, $C_D^\ast(k)$,
  and $C_D^\ast(l)$, then
  $R_D^\ast(\mu_D^\ast(\mu_D^\ast(h,k),l),
  \mu_D^\ast(h,\mu_D^\ast(k,l)))$.
  \item Left and right unit laws:
  $R_D^\ast(\mu_D^\ast(e_D^\ast,h),h)$ and
  $R_D^\ast(\mu_D^\ast(h,e_D^\ast),h)$ for every $h$ with $C_D^\ast(h)$.
  \item Binary classifier congruence: if $R_D^\ast(h,h')$ and
  $R_D^\ast(k,k')$, then
  $R_D^\ast(\mu_D^\ast(h,k),\mu_D^\ast(h',k'))$.
  \item Restriction: on carried denominators, $C_D^\ast$, $R_D^\ast$, and
  $\mu_D^\ast$ restrict to $C_D$, $R_D$, and $\mu_D$ from
  \autoref{def:field-rat-denominator-concrete-continuation-instance}.
  \item Obstruction: if a carried endpoint $u$ satisfies either one-sided
  classifier-unit law on carried denominators, then $\mathsf{False}$.
\end{enumerate}
\end{theorem}
\leanchecked{BEDC.Derived.FieldUp.field\_rat\_denominator\_unit\_envelope\_monoid\_laws}
\begin{proof}
For carrier inhabitance, use the right disjunct in $C_D^\ast$ with reflexivity
at $\emp$. The carried/carried closure case is exactly
\autoref{thm:field-rat-denominator-concrete-continuation-instance-laws}; the
carried/empty and empty/carried cases reduce by the append unit laws and then
transport the carried witness by
$\mathsf{RatHistoryCarrier\_hsame\_transport}$. The empty/empty case is the
right carrier disjunct at $\emp$.

For associativity, split the three envelope carrier witnesses into their carried
or external-empty alternatives. The carried/carried/carried branch is the
associativity clause of
\autoref{thm:field-rat-denominator-concrete-continuation-instance-laws}. Every
branch containing an external-empty endpoint uses the append unit laws to remove
that endpoint, uses $\mathsf{append\_assoc}$ for the common bracketing spine,
and then repacks either the transported carried classifier or the reflexive
empty-empty classifier branch.

For the unit laws, split $C_D^\ast(h)$. In the carried branch,
$\mathsf{append}(\emp,h)$ and $\mathsf{append}(h,\emp)$ are transported back to
$h$ by the append unit laws, and
$\mathsf{RatHistoryClassifier\_hsame\_transport}$ supplies the guarded carried
classifier branch. In the external-empty branch, both displayed products are
$\hsame$ to $\emp$, so the empty-empty branch of $R_D^\ast$ applies.

For binary classifier congruence, split the two classifier witnesses. The
carried/carried classifier branch invokes the congruence clause of
\autoref{thm:field-rat-denominator-concrete-continuation-instance-laws}. If one
input classifier is the empty-empty branch, the append unit laws collapse that
side, and $\mathsf{RatHistoryClassifier\_hsame\_transport}$ transports the
remaining carried classifier; if both are empty-empty, both appended products
are $\hsame$ to $\emp$.

The restriction clause is immediate from the guarded carried branch of
$R_D^\ast$ and the definitions of $C_D^\ast$ and $\mu_D^\ast$ when the inputs
are carried denominators.

Finally, a carried endpoint satisfying a right or left one-sided classifier-unit
law on the carried denominator fragment is ruled out by
\autoref{thm:field-rat-denominator-continuation-no-one-sided-internal-unit}.
Thus adjoining the external endpoint gives a unit-bearing envelope, not an
internal $\FieldUp$ multiplication on the positive Rat denominator carrier.
\end{proof}

\begin{theorem}[Rat denominator external-unit envelope unit uniqueness]
\label{thm:field-rat-denominator-unit-envelope-unit-unique}
For every history $u:\mathsf{BHist}$ with $C_D^\ast(u)$, the unit laws of the
external envelope classify $u$ as the named endpoint $e_D^\ast$:
\begin{enumerate}
  \item if

  $$
  \forall h:\mathsf{BHist}.\;
  C_D^\ast(h)\to
  R_D^\ast(\mu_D^\ast(h,u),h),
  $$

  then $R_D^\ast(u,e_D^\ast)$;
  \item if

  $$
  \forall h:\mathsf{BHist}.\;
  C_D^\ast(h)\to
  R_D^\ast(\mu_D^\ast(u,h),h),
  $$

  then $R_D^\ast(u,e_D^\ast)$.
\end{enumerate}
\end{theorem}
\begin{proof}
Split the carrier witness $C_D^\ast(u)$ into the carried Rat denominator branch
and the external-empty branch. In the external-empty branch, $u\hsame\emp$ and
$e_D^\ast=\emp$, so the empty-empty branch of $R_D^\ast$ immediately gives
$R_D^\ast(u,e_D^\ast)$ for either one-sided law.

Assume now that $u$ is carried and that the right envelope unit law is
displayed. Restrict this law to a carried Rat denominator $d$. The closure part
of \autoref{thm:field-rat-denominator-unit-envelope-monoid-laws} makes
$\mu_D^\ast(d,u)$ carried, while
\autoref{thm:rat-history-carrier-not-empty} rules out the empty-empty branch
against the carried input $d$.

Unfolding the guarded branch of $R_D^\ast$ therefore gives the Rat denominator
classifier for the canonical append product. For any displayed continuation
$\Cont(d,u,r)$, compare it with the canonical append continuation witness and
transport the classifier to obtain
$\mathsf{RatHistoryClassifier}(r,d)$. This is exactly the right-unit
continuation law, so
\autoref{thm:field-rat-denominator-continuation-right-unit-endpoint-iff-empty}
yields $u\hsame\emp$.

The empty comparison $u\hsame\emp$, together with $e_D^\ast=\emp$, gives the
empty-empty classifier branch of $R_D^\ast(u,e_D^\ast)$.

The left envelope unit law is symmetric. Restrict it to carried Rat
denominators, unfold the carried branch of
$R_D^\ast(\mu_D^\ast(u,d),d)$, use the canonical append continuation witness to
obtain the displayed left-unit continuation law, and apply
\autoref{thm:field-rat-denominator-continuation-left-unit-endpoint-iff-empty}.
The resulting comparison $u\hsame\emp$ again repacks through the empty-empty
branch as $R_D^\ast(u,e_D^\ast)$.
\end{proof}

\begin{theorem}[Rat denominator external-unit envelope carried classifier exactness]
\label{thm:field-rat-denominator-unit-envelope-carried-classifier-exactness}
For all histories $h,k:\mathsf{BHist}$, if
$\mathsf{RatHistoryCarrier}(h)$ and $\mathsf{RatHistoryCarrier}(k)$, then
$$
R_D^\ast(h,k)\Longleftrightarrow\mathsf{RatHistoryClassifier}(h,k).
$$
\end{theorem}
\begin{proof}
Unfold $R_D^\ast$ from
\autoref{def:field-rat-denominator-unit-envelope}. The reverse direction uses
the guarded carried branch: the two displayed carrier witnesses for $h$ and
$k$, together with $\mathsf{RatHistoryClassifier}(h,k)$, assemble the first
disjunct of the envelope classifier.

For the forward direction, split the envelope classifier witness. In the
guarded carried branch, the third component is already
$\mathsf{RatHistoryClassifier}(h,k)$, so projection gives the desired Rat
classifier.

It remains to reject the empty-empty branch on carried inputs. That branch
contains $h\hsame\emp$; applying
\autoref{thm:rat-history-carrier-not-empty} to the carried witness for $h$ and
this endpoint comparison yields $\mathsf{False}$, from which the Rat classifier
conclusion follows constructively.
\end{proof}

This is the exactness predicate promised for restriction back to the carried
Rat denominator continuation instance.

\textbf{Traditional target}: unit-enveloped denominator monoid fragment for
$\FieldUp(\mathsf{RatUp})$; \textbf{BEDC source}:
$\mathsf{RatHistoryCarrier}$, $\mathsf{RatHistoryClassifier}$, $\Cont$, and
$\mathsf{append}$; \textbf{carrier choice}: the external-unit envelope
$C_D^\ast$; \textbf{classifier choice}: the carried Rat classifier plus the
empty-empty endpoint classifier; \textbf{stability obligation}: carrier closure,
associativity, unit laws, classifier congruence, restriction to the carried
continuation instance, unit endpoint uniqueness, and one-sided carried-unit
obstruction; \textbf{ledger}: the external unit bit plus continuation witness;
\textbf{core theorems}:
\autoref{thm:field-rat-denominator-unit-envelope-monoid-laws}
and
\autoref{thm:field-rat-denominator-unit-envelope-unit-unique};
\textbf{exactness predicate}: restriction back to the carried Rat denominator
continuation instance, with the adjoined endpoint as the unique envelope unit
up to $R_D^\ast$; \textbf{standard bridge}: deferred; \textbf{status}:
$\mathsf{ProofTarget}$.

\begin{theorem}[Rat denominator external-unit envelope is strict]
\label{thm:field-rat-denominator-unit-envelope-strict}
For the concrete carried data $C_D$, $R_D$, and $\mu_D$ of
\autoref{def:field-rat-denominator-concrete-continuation-instance} and the
envelope data $C_D^\ast$, $R_D^\ast$, $\mu_D^\ast$, and $e_D^\ast$ of
\autoref{def:field-rat-denominator-unit-envelope}, the adjoined endpoint is
strictly external and the carried fragment is recovered exactly.
\begin{enumerate}
  \item $C_D^\ast(e_D^\ast)$ and $\neg C_D(e_D^\ast)$.
  \item For every $h:\mathsf{BHist}$, if $C_D(h)$, then
  $\neg R_D^\ast(h,e_D^\ast)$ and $\neg R_D^\ast(e_D^\ast,h)$.
  \item For every $h,k:\mathsf{BHist}$, if $C_D(h)$ and $C_D(k)$, then
  $$
  R_D^\ast(h,k)\Longleftrightarrow R_D(h,k).
  $$
  On the same carried inputs, the displayed product is the original append
  product:
  $$
  \mu_D^\ast(h,k)=\mu_D(h,k)=\mathsf{append}(h,k).
  $$
\end{enumerate}
\end{theorem}
\begin{proof}
Carrier inhabitance is the first clause of
\autoref{thm:field-rat-denominator-unit-envelope-monoid-laws}. For the
exclusion from the carried fragment, unfold
\autoref{def:field-rat-denominator-concrete-continuation-instance} and
\autoref{def:field-rat-denominator-unit-envelope}: $e_D^\ast$ is $\emp$ and
$C_D$ is $\mathsf{RatHistoryCarrier}$. Then
\autoref{thm:rat-history-carrier-not-empty} rules out a carried witness at the
empty history.

For the mixed right endpoint, assume $C_D(h)$ and
$R_D^\ast(h,e_D^\ast)$. Unfolding $R_D^\ast$ gives either the guarded carried
branch or the empty-empty branch. The guarded carried branch includes
$C_D(e_D^\ast)$, contradicting the endpoint exclusion already proved. The
empty-empty branch gives $h\hsame\emp$, contradicting
\autoref{thm:rat-history-carrier-not-empty} applied to the carried witness
$C_D(h)$. The mixed left endpoint is identical after swapping the two inputs.

It remains to recover the carried classifier. Suppose $C_D(h)$ and $C_D(k)$.
If $R_D^\ast(h,k)$ holds, unfold $R_D^\ast$. The empty-empty branch would give
$h\hsame\emp$, contradicting the same non-emptiness theorem, so the guarded
carried branch remains and yields $R_D(h,k)$.

Conversely, from $C_D(h)$, $C_D(k)$, and $R_D(h,k)$, the guarded carried branch
of $R_D^\ast(h,k)$ applies directly. Finally, unfolding the two displayed
operations shows that $\mu_D^\ast$ and $\mu_D$ both return the append history
on carried inputs.
\end{proof}

\begin{theorem}[Rat denominator external-unit envelope mixed endpoint separation]
\label{thm:field-rat-denominator-unit-envelope-mixed-endpoint-separation}
For all $d,e:\mathsf{BHist}$, the external-unit classifier $R_D^u$ has no
mixed carried-empty endpoint branch:
$$
(\mathsf{RatHistoryCarrier}(d)\land e\hsame\emp)
\to (R_D^u(d,e)\to \mathsf{False}),
$$
and
$$
(d\hsame\emp\land \mathsf{RatHistoryCarrier}(e))
\to (R_D^u(d,e)\to \mathsf{False}).
$$
\end{theorem}
\begin{proof}
Unfold $R_D^u$ and then
$\mathsf{RatDenomUnitClassifier}$. In the first displayed implication, the
classifier supplies $d\hsame e$. Combining this comparison with $e\hsame\emp$
by symmetry and transitivity of $\hsame$ from
\autoref{thm:history-sameness-equivalence} gives $d\hsame\emp$.

Together with $\mathsf{RatHistoryCarrier}(d)$, rational history carrier
nonemptiness, \autoref{thm:rat-history-carrier-not-empty}, gives
$\mathsf{False}$.

The second displayed implication is symmetric. The classifier again supplies
$d\hsame e$; together with $d\hsame\emp$ and the equivalence laws for $\hsame$,
this gives $e\hsame\emp$. The carried endpoint
$\mathsf{RatHistoryCarrier}(e)$ is then contradicted by
\autoref{thm:rat-history-carrier-not-empty}.
\end{proof}

\textbf{Traditional target}: mixed-endpoint separation for the unit envelope;
\textbf{BEDC source}: $\mathsf{RatHistoryCarrier}$,
$\mathsf{RatHistoryClassifier}$, $\Cont$, and $\emp$; \textbf{carrier choice}:
$C_D^u$; \textbf{classifier choice}: $R_D^u$; \textbf{stability
obligation}: mixed-branch separation; \textbf{ledger}: the external-unit bit;
\textbf{core theorem}:
\autoref{thm:field-rat-denominator-unit-envelope-mixed-endpoint-separation};
\textbf{exactness predicate}: absence of mixed carried-empty classification;
\textbf{standard bridge}: deferred; \textbf{status}: $\mathsf{ProofTarget}$.

\begin{theorem}[Rat denominator external-unit envelope MonoidUp certificate]
\label{thm:field-rat-denominator-unit-envelope-monoid-namecert}
The concrete tuple $(C_D^\ast,R_D^\ast,\mu_D^\ast,e_D^\ast)$ of
\autoref{def:field-rat-denominator-unit-envelope}, with carrier
$C_D^\ast$, classifier $R_D^\ast$, operation $\mathsf{append}$, and unit
$\emp$, supplies the concrete $\mathsf{BHist}$-level $\MonoidUp$ naming
certificate for the unit-enveloped Rat denominator continuation fragment.
Its certificate ledger is the external-unit bit together with the continuation
witness used by the guarded carried branch of $R_D^\ast$.
\end{theorem}
\begin{proof}
Apply \autoref{thm:field-rat-denominator-unit-envelope-monoid-laws}. Its
carrier inhabitance clause supplies the exported unit witness for $e_D^\ast$,
and its closure clause supplies the carrier witness for every exported
$\mathsf{append}$ product in the envelope.

The same theorem supplies the left and right unit laws, associativity under
$R_D^\ast$, and binary classifier congruence. These are exactly the stability
fields required by \autoref{def:monoid-certificate-obligations} for a
$\MonoidUp$ interface over a BEDC carrier and classifier.

It remains to identify the certificate packaging. The carrier is the concrete
predicate $C_D^\ast$ on $\mathsf{BHist}$; the classifier is the concrete
relation $R_D^\ast$; the source operation is the native BEDC
$\mathsf{append}$ operation through $\mu_D^\ast$; and the named unit is
$\emp$. Thus \autoref{thm:concrete-monoid-namecert} packages these fields as a
$\mathsf{NameCert}$ for $\MonoidUp(C_D^\ast)$.

The ledger is concrete rather than schematic. The right carrier disjunct records
whether the endpoint is the adjoined external unit, while the guarded carried
classifier branch records the continuation witness that relates the native
Rat denominator fragment to the envelope product. This is the required
$\mathsf{BHist}$-anchored certificate consumed at the $\FieldUp(\mathsf{RatUp})$
boundary.
\end{proof}
