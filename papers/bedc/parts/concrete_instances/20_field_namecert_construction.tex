\chapter{A Concrete Naming Certificate for \(\FieldUp\)}
\label{ch:concrete-instances-field-namecert}

The field interface $\FieldUp$ extends \autoref{ch:concrete-instances-commring-namecert} with the multiplicative inverse field. The inverse domain is a unary certificate $\mathsf{NonZero}_F(a)$ derived from constructive apartness of $a$ from $0$; inverse formation is licensed only by an explicit witness $p:\mathsf{NonZero}_F(a)$. This proof-relevant discipline replaces the classical "$a \ne 0$ implies $a$ invertible" formulation, which would otherwise require LEM. The rational carrier from \autoref{ch:concrete-instances-rat-namecert} supplies the candidate carrier and classifier source; a $\FieldUp$ certificate is present only when commutative-ring structure, constructive nonzero evidence, inverse closure, inverse laws indexed by nonzero witnesses, and the ledger are supplied over that source.

\section{Carrier definition}

\begin{definition}[Field carrier]
\label{def:field-carrier}
Use a commutative ring setup extended with a constructive apartness predicate and define $\mathsf{NonZero}_F(a)$ as the unary certificate obtained from apartness of $a$ from $0$. The inverse datum has type $\mathsf{inv}(a,p)$ for $p:\mathsf{NonZero}_F(a)$, so no inverse is formed without the nonzero witness.
\end{definition}

\section{Concrete singleton empty-history field schema instance}
\label{sec:field-concrete-singleton-instance}

\begin{definition}[Singleton empty-history field schema instance]
\label{def:singleton-empty-history-field-schema-instance}
Let $A_{\emp}$ be the singleton history source of
Definition~\ref{def:list-singleton-history-source} at $a_0=\emp$. Define the
classifier
$$
\rho_{\emp}(h,k)\Longleftrightarrow
A_{\emp}(h)\land A_{\emp}(k)\land\hsame(h,k).
$$
Take $\mathsf{add}$, $\mathsf{neg}$, $0$, $\mathsf{mul}$, and $1$ to be the
constant empty-history operations of
Theorem~\ref{thm:singleton-empty-history-commring-laws}. Define
$\mathsf{NonZero}_{\bot}(h):=\bot$. The proof-indexed inverse operation is the
empty function on witnesses $p:\mathsf{NonZero}_{\bot}(h)$: from such a
witness, elimination of $\bot$ supplies the required history endpoint.
\end{definition}

\begin{theorem}[Singleton empty-history field schema laws]
\label{thm:singleton-empty-history-field-schema-laws}
The tuple
$$
(A_{\emp},\rho_{\emp},\mathsf{add},0,\mathsf{neg},\mathsf{mul},1,
\mathsf{NonZero}_{\bot},\mathsf{inv})
$$
satisfies the $\FieldUp$ schema laws over the concrete $\mathsf{BHist}$
singleton carrier.
\end{theorem}
\begin{proof}
The commutative-ring fields are exactly the fields supplied by
Theorem~\ref{thm:singleton-empty-history-commring-laws} for the displayed
carrier, classifier, and constant empty-history operations.

The nonzero exclusion at $0$ is immediate: a witness of
$\mathsf{NonZero}_{\bot}(0)$ is a witness of $\bot$.

Nonzero transport along $\rho_{\emp}$ is vacuous. Given
$\rho_{\emp}(h,k)$ and $p:\mathsf{NonZero}_{\bot}(h)$, eliminate $p$ to obtain
the required witness at $k$.

Inverse closure and inverse congruence are also vacuous. Both clauses start
from an explicit witness $p:\mathsf{NonZero}_{\bot}(h)$, so elimination of
$p$ supplies the requested carrier endpoint or classifier witness.

The left and right inverse laws have the same proof shape. Each law is indexed
by a witness $p:\mathsf{NonZero}_{\bot}(h)$; eliminating $p$ closes the law
without forming a classical inverse outside its licensed domain.
\end{proof}

\begin{theorem}[Singleton empty-history FieldUp schema certificate]
\label{thm:singleton-empty-history-field-schema-certificate}
The tuple from
Definition~\ref{def:singleton-empty-history-field-schema-instance}, together
with the law package from
Theorem~\ref{thm:singleton-empty-history-field-schema-laws}, supplies the
carrier, source, pattern, classifier, stability, and ledger fields required by
Definition~\ref{def:field-certificate-obligations} over the concrete

\begin{enumerate}
  \item Traditional target: the singleton proof-indexed $\FieldUp$ schema.
  \item BEDC source: $\mathsf{BHist}$ with $A_{\emp}$.
  \item Carrier choice: $A_{\emp}$.
  \item Classifier choice: $\rho_{\emp}$.
  \item Stability obligations: commutative-ring laws plus vacuous nonzero
  transport, inverse closure, and inverse laws.
  \item Ledger: the singleton endpoint and the empty nonzero witness domain.
  \item Core theorem list: Theorem~\ref{thm:singleton-empty-history-field-schema-laws}.
  \item Exactness predicate: singleton endpoint classification.
  \item Standard bridge: only the terminal proof-indexed schema.
  \item Status: ProofTarget.
\end{enumerate}
\end{theorem}
\begin{proof}
The source and carrier fields are the singleton history source
$A_{\emp}$ from Definition~\ref{def:singleton-empty-history-field-schema-instance}.
The classifier field is $\rho_{\emp}$, whose only endpoints are the empty
histories admitted by $A_{\emp}$.

For the algebraic stability fields, use
Theorem~\ref{thm:singleton-empty-history-commring-laws}. It supplies the
commutative-ring part for the constant empty-history operations on the same
carrier and classifier.

For the field-specific clauses, apply
Theorem~\ref{thm:singleton-empty-history-field-schema-laws}. Nonzero transport,
inverse closure, inverse congruence, and the left and right inverse laws all
begin with a witness $p:\mathsf{NonZero}_{\bot}(h)$.

Since $\mathsf{NonZero}_{\bot}(h)$ is $\bot$, eliminating $p$ closes every
inverse and nonzero subfield without adding a classical nonzero test. The
ledger therefore records only the singleton endpoint and the empty witness
domain, giving the required concrete $\mathsf{BHist}$ FieldUp naming
certificate template.
\end{proof}

\section{The certificate}

\begin{definition}[Field source specification]
\label{def:field-source-specification}
The source is a certified commutative ring carrier plus constructive apartness from zero, packaged through the derived predicate $\mathsf{NonZero}_F$.
\end{definition}

\begin{definition}[Field pattern specification]
\label{def:field-pattern-specification}
The pattern is commutative-ring expression formation extended by inverse formation only from a pair $(a,p)$ with $p:\mathsf{NonZero}_F(a)$.
\end{definition}

\begin{definition}[Field classifier specification]
\label{def:field-classifier-specification}
The classifier is ring sameness $\sim_F$. Nonzero evidence is stable under classifier replacement: from $a\sim_F b$ and $p:\mathsf{NonZero}_F(a)$ the certificate supplies transported evidence $q:\mathsf{NonZero}_F(b)$.
\end{definition}

\begin{definition}[Field stability certificate]
\label{def:field-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item all commutative-ring stability fields
  \item contradiction of $\mathsf{NonZero}_F(0)$
  \item stability of $\mathsf{NonZero}_F$ under the carrier classifier
  \item inverse closure for explicit witnesses $p:\mathsf{NonZero}_F(a)$
  \item left and right multiplicative inverse laws indexed by each witness $p:\mathsf{NonZero}_F(a)$
\end{enumerate}
\end{definition}

\begin{theorem}[Field inverse congruence from apartness transport]
\label{thm:field-inverse-congruence-from-apartness}
For a BEDC history-level $\FieldUp$ candidate with BHist elements, commutative-ring classifier $\sim_F$, unary nonzero certificate $\mathsf{NonZero}_F$, inverse operation $\mathsf{inv}(a,p)$ indexed by $p:\mathsf{NonZero}_F(a)$, classifier transport of nonzero evidence, and left and right inverse laws indexed by nonzero witnesses, if $a\sim_F b$ and $p:\mathsf{NonZero}_F(a)$ while $q:\mathsf{NonZero}_F(b)$ is the transported witness along $a\sim_F b$, then $\mathsf{inv}(a,p)\sim_F\mathsf{inv}(b,q)$.
\end{theorem}
\begin{proof}
Transport the left inverse law for $\mathsf{inv}(a,p)$ across $a\sim_F b$ to obtain a left inverse witness over $b$. Combine this transported witness with the right inverse law for $\mathsf{inv}(b,q)$ at $b$. The group left/right inverse uniqueness argument, applied inside the multiplicative group of nonzero field elements, identifies the transported inverse with the chosen inverse at $b$.
\end{proof}

\begin{theorem}[Field inverse cancel from apartness-indexed inverses]
\label{thm:field-inverse-cancel-from-apartness}
For a BEDC history-level \(\FieldUp\) candidate with proof-indexed inverses,
if \(\mathsf{inv}(a,p)\hsame\mathsf{inv}(b,q)\), then \(a\hsame b\).
\end{theorem}
\begin{proof}
Transport the left inverse law for \(a\) across the equality of
proof-indexed inverses to make \(a\) a right inverse of the common inverse
\(\mathsf{inv}(b,q)\). The right inverse law for \(b\) makes \(b\) the
matching left inverse. Group left/right inverse uniqueness identifies \(b\)
with \(a\), and symmetry gives \(a\hsame b\).
\end{proof}

\begin{theorem}[Field inverse of an indexed inverse]
\label{thm:field-inverse-involutive-from-apartness}
For a BEDC history-level $\FieldUp$ candidate with multiplication on
$\mathsf{BHist}$, unit $1$, proof-indexed inverse $\mathsf{inv}(a,p)$,
associativity, left and right identity, multiplication congruence, and left
inverse laws indexed by nonzero witnesses, if
$p_a:\mathsf{NonZero}_F(a)$ and
$p_{\mathsf{inv}}:\mathsf{NonZero}_F(\mathsf{inv}(a,p_a))$, then
$$
\mathsf{inv}(\mathsf{inv}(a,p_a),p_{\mathsf{inv}})\hsame a.
$$
\end{theorem}
\begin{proof}
Set the middle element in the inverse-uniqueness argument to
$\mathsf{inv}(a,p_a)$. The left inverse law for $p_{\mathsf{inv}}$ gives
$$
\mathsf{inv}(\mathsf{inv}(a,p_a),p_{\mathsf{inv}})\mathsf{inv}(a,p_a)\hsame 1.
$$

The left inverse law for $p_a$ gives
$$
\mathsf{inv}(a,p_a)a\hsame 1.
$$

Associativity, multiplication congruence, and the identity laws give the
standard left/right inverse uniqueness chain
$x\hsame x\,1\hsame x(yz)\hsame (xy)z\hsame 1\,z\hsame z$, with
$x=\mathsf{inv}(\mathsf{inv}(a,p_a),p_{\mathsf{inv}})$,
$y=\mathsf{inv}(a,p_a)$, and $z=a$. Hence the two outer witnesses are
identified, which is the stated equality.
\end{proof}

\begin{theorem}[Field inverse product reversal from apartness-indexed inverses]
\label{thm:field-inverse-product-reverse-from-apartness}
For a BEDC history-level $\FieldUp$ candidate with proof-indexed inverses,
explicit witnesses $p_{ab}:\mathsf{NonZero}_F(ab)$,
$p_a:\mathsf{NonZero}_F(a)$, and $p_b:\mathsf{NonZero}_F(b)$ determine
the product inverse by
$$
\mathsf{inv}(ab,p_{ab})\hsame \mathsf{inv}(b,p_b)\mathsf{inv}(a,p_a).
$$
\end{theorem}
\begin{proof}
The reversed product $\mathsf{inv}(b,p_b)\mathsf{inv}(a,p_a)$ is a right
inverse of $ab$: reassociate to expose $b\mathsf{inv}(b,p_b)$, collapse it
to the unit, and finish with the right inverse law for $a$. The indexed left
inverse of $ab$ and group uniqueness of left and right inverse witnesses
identify the chosen inverse of $ab$ with this reversed product.
\end{proof}

\begin{theorem}[Field multiplication cancellation from apartness-certified factors]
\label{thm:field-nonzero-mul-cancel-from-apartness}
For a BEDC history-level $\FieldUp$ candidate with proof-indexed inverses,
right and left multiplication by apartness-certified nonzero factors are
cancellative. Any witness $p_b:\mathsf{NonZero}_F(b)$ makes
$$
xb\hsame yb
\quad\Longrightarrow\quad
x\hsame y,
$$
and any witness $p_a:\mathsf{NonZero}_F(a)$ makes
$$
ax\hsame ay
\quad\Longrightarrow\quad
x\hsame y.
$$
\end{theorem}
\begin{proof}
For the right clause, apply
Theorem~\ref{thm:field-right-mul-equation-solution-from-apartness} to
$xb\hsame yb$. This gives
$x\hsame (yb)\mathsf{inv}(b,p_b)$. Theorem~\ref{thm:field-mul-inverse-right-cancel-from-apartness}
identifies $(yb)\mathsf{inv}(b,p_b)$ with $y$, so transitivity of
$\hsame$ gives $x\hsame y$.

For the left clause, apply
Theorem~\ref{thm:field-left-mul-equation-solution-from-apartness} to
$ax\hsame ay$. This gives
$x\hsame \mathsf{inv}(a,p_a)(ay)$. Theorem~\ref{thm:field-mul-inverse-left-cancel-from-apartness}
identifies $\mathsf{inv}(a,p_a)(ay)$ with $y$, and transitivity of
$\hsame$ finishes.
\end{proof}

\begin{theorem}[Field two-sided multiplication exactness from apartness]
\label{thm:field-two-sided-mul-exact-from-apartness}
For a BEDC history-level $\FieldUp$ candidate with proof-indexed inverses,
multiplication congruence, and witnesses $p_a:\mathsf{NonZero}_F(a)$ and
$p_b:\mathsf{NonZero}_F(b)$, every $x,y:\mathsf{BHist}$ satisfy
$$
(a x)b\hsame(a y)b
\Longleftrightarrow
x\hsame y.
$$
\end{theorem}
\begin{proof}
For the forward direction, apply
Theorem~\ref{thm:field-nonzero-mul-cancel-from-apartness} with the
right-factor witness $p_b$. This cancels the certified factor $b$ and gives
$a x\hsame a y$.

Apply Theorem~\ref{thm:field-nonzero-mul-cancel-from-apartness} again with
the left-factor witness $p_a$. This cancels the certified factor $a$ and
gives $x\hsame y$.

For the reverse direction, start from $x\hsame y$. Multiplication congruence
transports this classifier equality through left multiplication by $a$, giving
$a x\hsame a y$.

Apply multiplication congruence once more through right multiplication by $b$.
The result is exactly
$$
(a x)b\hsame(a y)b.
$$
\end{proof}

\begin{theorem}[Field two-sided zero product exactness from apartness]
\label{thm:field-two-sided-zero-product-cancel-from-apartness}
For a BEDC history-level $\FieldUp$ candidate with proof-indexed inverses,
ring zero absorption, multiplication congruence, and witnesses
$p_a:\mathsf{NonZero}_F(a)$ and $p_b:\mathsf{NonZero}_F(b)$, every
$x:\mathsf{BHist}$ satisfies
$$
(a x)b\hsame0
\Longleftrightarrow
x\hsame0.
$$
\end{theorem}
\begin{proof}
For the forward direction, apply
Theorem~\ref{thm:field-right-mul-equation-solution-from-apartness} with the
right-factor witness $p_b$ to the hypothesis $(a x)b\hsame0$. This solves the
right multiplication equation and gives
$a x\hsame0\mathsf{inv}(b,p_b)$.

Use Theorem~\ref{thm:ring-mul-zero-absorption} to identify
$0\mathsf{inv}(b,p_b)$ with $0$. Transitivity of $\hsame$ therefore gives
$a x\hsame0$.

Now apply the checked left inverse cancellation chain with the left-factor
witness $p_a$ to $a x\hsame0$. The left solver gives
$x\hsame\mathsf{inv}(a,p_a)0$.

Use Theorem~\ref{thm:ring-mul-zero-absorption} again to identify
$\mathsf{inv}(a,p_a)0$ with $0$. Transitivity of $\hsame$ yields
$x\hsame0$.

For the reverse direction, start from $x\hsame0$. Multiplication congruence
transports this equality through left multiplication by $a$, giving
$a x\hsame a0$.

Apply Theorem~\ref{thm:ring-mul-zero-absorption} to collapse $a0\hsame0$.
Transitivity gives $a x\hsame0$.

Multiplication congruence now transports through right multiplication by $b$,
giving $(a x)b\hsame0b$. A second application of
Theorem~\ref{thm:ring-mul-zero-absorption} collapses $0b\hsame0$, and
transitivity yields $(a x)b\hsame0$.
\end{proof}

\begin{theorem}[Field one-sided zero product cancellation from apartness]
\label{thm:field-one-sided-zero-product-cancel-from-apartness}
For a BEDC history-level $\FieldUp$ candidate with proof-indexed inverses,
ring zero absorption, multiplication congruence, and indexed inverse laws,
every $a,b:\mathsf{BHist}$ satisfies the following clauses.
\begin{enumerate}
\item If $p_a:\mathsf{NonZero}_F(a)$ and $a b\hsame0$, then $b\hsame0$.
\item If $p_b:\mathsf{NonZero}_F(b)$ and $a b\hsame0$, then $a\hsame0$.
\end{enumerate}
\end{theorem}
\begin{proof}
For the left-factor clause, apply
Theorem~\ref{thm:field-left-mul-equation-solution-from-apartness} to the
equation $a b\hsame0$ with witness $p_a$. This gives
$b\hsame\mathsf{inv}(a,p_a)0$.

Use Theorem~\ref{thm:ring-mul-zero-absorption} to identify
$\mathsf{inv}(a,p_a)0$ with $0$. Transitivity of $\hsame$ yields
$b\hsame0$.

For the right-factor clause, apply
Theorem~\ref{thm:field-right-mul-equation-solution-from-apartness} to the
same equation with witness $p_b$. This gives
$a\hsame0\mathsf{inv}(b,p_b)$.

Use Theorem~\ref{thm:ring-mul-zero-absorption} to identify
$0\mathsf{inv}(b,p_b)$ with $0$. Transitivity of $\hsame$ yields
$a\hsame0$.
\end{proof}

\begin{theorem}[Field nonzero factors exclude zero product]
\label{thm:field-nonzero-factors-exclude-zero-product}
For a BEDC history-level $\FieldUp$ candidate with proof-indexed inverses,
ring zero absorption, multiplication congruence, indexed inverse laws,
classifier transport of $\mathsf{NonZero}_F$, and contradiction of
$\mathsf{NonZero}_F(0)$, every $a,b:\mathsf{BHist}$ with witnesses
$p_a:\mathsf{NonZero}_F(a)$ and $p_b:\mathsf{NonZero}_F(b)$ satisfies
$$
\neg(a b\hsame0).
$$
\end{theorem}
\begin{proof}
Assume $a b\hsame0$. Apply
Theorem~\ref{thm:field-one-sided-zero-product-cancel-from-apartness} with the
left-factor witness $p_a$. The left-factor clause gives $b\hsame0$.

Transport the witness $p_b:\mathsf{NonZero}_F(b)$ along this classifier
equality. The stability clause for $\mathsf{NonZero}_F$ supplies
$\mathsf{NonZero}_F(0)$.

The field stability certificate includes contradiction of
$\mathsf{NonZero}_F(0)$, so the transported evidence closes the goal with
$\bot$.

The symmetric route uses $p_b$ in the right-factor clause and transports
$p_a$ after deriving $a\hsame0$.
\end{proof}

\begin{theorem}[Field equal-squares signed factor exclusion]
\label{thm:field-equal-squares-signed-factor-exclusion}
For a BEDC history-level $\FieldUp$ candidate with proof-indexed inverses,
ring zero absorption, multiplication congruence, the $\CommRingUp$ hypotheses
required by Theorem~\ref{thm:commring-difference-of-squares-zero-exact}, and
every $a,b:\mathsf{BHist}$ with $a\cdot a\hsame b\cdot b$, the following
constructive clauses hold.
\begin{enumerate}
\item If $p_+:\mathsf{NonZero}_F(a+b)$, then $a+(-b)\hsame0$.
\item If $p_-:\mathsf{NonZero}_F(a+(-b))$, then $a+b\hsame0$.
\end{enumerate}
\end{theorem}
\begin{proof}
Apply the reverse direction of
Theorem~\ref{thm:commring-difference-of-squares-zero-exact} to the square
equality. The signed factor product is therefore classified as zero:
$$
(a+b)\cdot(a+(-b))\hsame0.
$$

For the first clause, use the witness $p_+:\mathsf{NonZero}_F(a+b)$ and
apply Theorem~\ref{thm:field-one-sided-zero-product-cancel-from-apartness} to
the signed factor product with left factor $a+b$ and right factor $a+(-b)$.
The left-factor clause gives
$a+(-b)\hsame0$.

For the second clause, use the witness $p_-:\mathsf{NonZero}_F(a+(-b))$ and
apply Theorem~\ref{thm:field-one-sided-zero-product-cancel-from-apartness} to
the same signed factor product. The right-factor clause gives $a+b\hsame0$.
\end{proof}

Per-object certificate: this is a Core theorem list item for constructive
field square-factor reasoning. Source: $\FieldUp$ plus $\CommRingUp$. Carrier:
$\mathsf{BHist}$. Classifier: $\hsame$. Ledger: the chosen nonzero signed
factor witness $p_+$ or $p_-$. Status: ProofTarget.

\begin{theorem}[Field affine left multiplication equation exactness from apartness]
\label{thm:field-affine-left-equation-exact-from-apartness}
For a BEDC history-level $\FieldUp$ candidate with operations
$\mathsf{add},\mathsf{mul},\mathsf{neg},\mathsf{zero},\mathsf{one}$ on
$\mathsf{BHist}$, additive abelian-group laws for
$\mathsf{add},\mathsf{neg},\mathsf{zero}$, addition congruence,
multiplication associativity and congruence, and left and right inverse laws
indexed by $p_a:\mathsf{NonZero}_F(a)$, every
$x,b,c:\mathsf{BHist}$ satisfy
$$
\mathsf{add}(\mathsf{mul}(a,x),b)\hsame c
\Longleftrightarrow
x\hsame
\mathsf{mul}(\mathsf{inv}(a,p_a),\mathsf{add}(c,\mathsf{neg}(b))).
$$
\end{theorem}
\begin{proof}
For the forward direction, transport
$\mathsf{add}(\mathsf{mul}(a,x),b)\hsame c$ through right addition by
$\mathsf{neg}(b)$. Addition congruence gives an equality between
$\mathsf{add}(\mathsf{add}(\mathsf{mul}(a,x),b),\mathsf{neg}(b))$ and
$\mathsf{add}(c,\mathsf{neg}(b))$.

Associativity, the additive inverse law for $b$, and the additive identity law
collapse the left side to $\mathsf{mul}(a,x)$. Apply
Theorem~\ref{thm:field-left-mul-equation-solution-from-apartness} to obtain
the displayed solution for $x$ with right side
$\mathsf{add}(c,\mathsf{neg}(b))$.

For the reverse direction, transport the assumed displayed solution through
left multiplication by $a$. Multiplication congruence, associativity, the
indexed inverse law for $p_a$, and the multiplicative identity law collapse the
right side to $\mathsf{add}(c,\mathsf{neg}(b))$, so
$\mathsf{mul}(a,x)\hsame\mathsf{add}(c,\mathsf{neg}(b))$.

Add $b$ on the right and use addition congruence. Associativity, additive
commutativity for the middle pair when needed, the additive inverse law for
$b$, and the additive identity law collapse
$\mathsf{add}(\mathsf{add}(c,\mathsf{neg}(b)),b)$ back to $c$. This gives
$\mathsf{add}(\mathsf{mul}(a,x),b)\hsame c$.
\end{proof}
