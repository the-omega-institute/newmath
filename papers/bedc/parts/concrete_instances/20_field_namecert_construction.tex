\section{RatUp boundary for the singleton field schema}
\label{sec:field-singleton-rat-boundary}

\begin{theorem}[Singleton field schema is disjoint from the RatUp history carrier]
\label{thm:field-singleton-schema-disjoint-from-rat-history-carrier}
For every $h:\mathsf{BHist}$, if $A_{\emp}(h)$ is the singleton source from
Definition~\ref{def:singleton-empty-history-field-schema-instance}, then
$\mathsf{RatHistoryCarrier}(h)$ implies $\mathsf{False}$. Equivalently, no
RatUp history-carrier endpoint is an endpoint of the singleton empty-history
$\FieldUp$ schema.
\end{theorem}
\begin{proof}
Assume $A_{\emp}(h)$ and $\mathsf{RatHistoryCarrier}(h)$. Unfolding the
singleton carrier component gives the endpoint equality $h\hsame\emp$.

The RatUp carrier witness is separated from the empty denominator endpoint by
Theorem~\ref{thm:rat-history-carrier-not-empty}. Applying that theorem to the
carrier witness and the equality $h\hsame\emp$ gives the required
contradiction.
\end{proof}

\begin{theorem}[Singleton field schema is disjoint from RatUp classifier endpoints]
\label{thm:field-singleton-schema-disjoint-from-rat-history-classifier}
For every $h,k:\mathsf{BHist}$, if $A_{\emp}(h)$ holds for the singleton
source from Definition~\ref{def:singleton-empty-history-field-schema-instance}
and $\mathsf{RatHistoryClassifier}(h,k)$ holds, then $\mathsf{False}$.
Likewise, if $A_{\emp}(k)$ and $\mathsf{RatHistoryClassifier}(h,k)$ hold, then
$\mathsf{False}$.
\end{theorem}
\begin{proof}
Unfold the Rat history classifier. Its witness contains
$\mathsf{RatHistoryCarrier}(h)$, $\mathsf{RatHistoryCarrier}(k)$, and the
displayed relation $h\hsame k$.

For the left endpoint, assume $A_{\emp}(h)$. Apply
Theorem~\ref{thm:field-singleton-schema-disjoint-from-rat-history-carrier} to
$A_{\emp}(h)$ and the carrier witness $\mathsf{RatHistoryCarrier}(h)$.

For the right endpoint, assume $A_{\emp}(k)$. Apply
Theorem~\ref{thm:field-singleton-schema-disjoint-from-rat-history-carrier} to
$A_{\emp}(k)$ and the carrier witness $\mathsf{RatHistoryCarrier}(k)$.
\end{proof}

Traditional target terminal singleton field boundary; BEDC source singleton
$\mathsf{BHist}$ $\FieldUp$ schema plus RatUp history carrier; carrier choices
$A_{\emp}$ and $\mathsf{RatHistoryCarrier}$; classifier choices $\rho_{\emp}$
and Rat history classifier; stability obligation nonempty denominator
separation; ledger empty endpoint versus positive-denominator evidence; core
theorems
Theorem~\ref{thm:field-singleton-schema-disjoint-from-rat-history-carrier} and
Theorem~\ref{thm:field-singleton-schema-disjoint-from-rat-history-classifier};
exactness predicate no shared endpoint; standard bridge terminal schema is not
the rational field; status ProofTarget.

\begin{theorem}[Classifier-complement singleton FieldUp instance is RatUp-disjoint]
\label{thm:field-singleton-empty-concrete-instance-rat-disjoint}
For the concrete source $A_{\emp}$, classifier $\rho_{\emp}$, proof domain
$\mathsf{fieldSingletonEmptyNonZero}$, and constant operations
$\mathsf{fieldSingletonEmptyMul}$, $\mathsf{fieldSingletonEmptyOne}$, and
$\mathsf{fieldSingletonEmptyInv}$ from
Definition~\ref{def:field-singleton-empty-nonzero}, the $\FieldUp$ schema laws
hold over the concrete $\mathsf{BHist}$ singleton carrier, and every endpoint
admitted by $A_{\emp}$ is disjoint from $\mathsf{RatHistoryCarrier}$.

Its ten-field template is:
\begin{enumerate}
  \item Traditional target: one-element proof-indexed $\FieldUp$ schema.
  \item BEDC source: $\mathsf{BHist}$ singleton source plus RatUp boundary.
  \item Carrier: $A_{\emp}$.
  \item Classifier: $\rho_{\emp}$.
  \item Stability obligations:
  Theorem~\ref{thm:field-singleton-empty-schema-laws}.
  \item Ledger: the singleton endpoint and empty nonzero witness domain.
  \item Core theorem list:
  Theorem~\ref{thm:field-singleton-empty-schema-laws} and
  Theorem~\ref{thm:field-singleton-schema-disjoint-from-rat-history-carrier}.
  \item Exactness predicate: singleton endpoint classification plus
  RatUp-disjointness.
  \item Standard bridge: terminal proof-indexed schema only.
  \item Status: $\mathsf{ProofTarget}$.
\end{enumerate}
\end{theorem}
\begin{proof}
First apply Theorem~\ref{thm:field-singleton-empty-schema-laws} to the displayed
tuple
$$
(A_{\emp},\rho_{\emp},\mathsf{fieldSingletonEmptyNonZero},
\mathsf{fieldSingletonEmptyMul},\mathsf{fieldSingletonEmptyOne},
\mathsf{fieldSingletonEmptyInv}).
$$
This supplies the concrete law package over the singleton $\mathsf{BHist}$
carrier.

For the RatUp boundary, take any displayed endpoint witness $A_{\emp}(h)$ and
any witness $\mathsf{RatHistoryCarrier}(h)$. Applying
Theorem~\ref{thm:field-singleton-schema-disjoint-from-rat-history-carrier} to
these two witnesses gives $\mathsf{False}$, so no admitted singleton endpoint is
a RatUp history-carrier endpoint.
\end{proof}

\section{Field affine explicit inverse}
\label{sec:field-affine-explicit-inverse}

\begin{theorem}[Field two-sided affine expression has explicit inverse]
\label{thm:field-affine-two-sided-explicit-inverse}
For a BEDC history-level $\FieldUp$ candidate with the same operations,
additive laws, multiplication laws, congruence data, and indexed inverse laws
as in
\autoref{thm:field-affine-two-sided-equation-exact-from-apartness}, let
$p_a:\mathsf{NonZero}_F(a)$ and $p_b:\mathsf{NonZero}_F(b)$ be the two
nonzero witnesses for the left and right affine factors. For each
$c:\mathsf{BHist}$ define
$$
s(c)=
\mathsf{mul}(\mathsf{inv}(a,p_a),
\mathsf{mul}(\mathsf{add}(c,\mathsf{neg}(d)),\mathsf{inv}(b,p_b))).
$$
Then every $c,x,y:\mathsf{BHist}$ satisfies the following three clauses.
\begin{enumerate}
\item $\mathsf{add}(\mathsf{mul}(\mathsf{mul}(a,s(c)),b),d)\hsame c$.
\item If
$\mathsf{add}(\mathsf{mul}(\mathsf{mul}(a,x),b),d)\hsame c$, then
$x\hsame s(c)$.
\item If
$$
\mathsf{add}(\mathsf{mul}(\mathsf{mul}(a,x),b),d)\hsame
\mathsf{add}(\mathsf{mul}(\mathsf{mul}(a,y),b),d),
$$
then $x\hsame y$.
\end{enumerate}
\end{theorem}
\begin{proof}
For the displayed preimage, instantiate
\autoref{thm:field-affine-two-sided-equation-exact-from-apartness} at
$x=s(c)$. The reverse implication only needs the reflexive classifier
equality $s(c)\hsame s(c)$, so it produces
$\mathsf{add}(\mathsf{mul}(\mathsf{mul}(a,s(c)),b),d)\hsame c$.

For uniqueness of any solution, use the forward implication of the same
exactness theorem. An assumed classification of the affine expression at
$x$ with $c$ is transported directly to $x\hsame s(c)$.

For equality of two affine expressions, apply
\autoref{thm:field-affine-two-sided-map-classifier-exact-from-apartness} to the
assumed classifier equality. Its conclusion is exactly $x\hsame y$.
\end{proof}

\begin{theorem}[Field equal-squares signed factors cannot both be nonzero]
\label{thm:field-equal-squares-signed-factors-not-both-nonzero}
For a BEDC history-level $\FieldUp$ candidate with the hypotheses of
\autoref{thm:field-equal-squares-signed-factor-exclusion} and
\autoref{thm:field-nonzero-factors-exclude-zero-product}, every
$a,b:\mathsf{BHist}$ satisfies
$$
a\cdot a\hsame b\cdot b
\quad\Longrightarrow\quad
\mathsf{NonZero}_F(a+b)\to
\mathsf{NonZero}_F(a+(-b))\to\mathsf{False}.
$$
The traditional target is constructive signed-factor square reasoning; the
BEDC source is $\FieldUp$ together with $\CommRingUp$; the carrier is
$\mathsf{BHist}$; the classifier is $\hsame$; the ledger consists of the two
explicit signed-factor nonzero witnesses; the core theorem is this block; the
exactness predicate is impossibility of simultaneous nonzero signed factors
under equal squares; the status is $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Assume $a\cdot a\hsame b\cdot b$ and witnesses
$p_+:\mathsf{NonZero}_F(a+b)$ and
$p_-:\mathsf{NonZero}_F(a+(-b))$.

First apply \autoref{thm:commring-difference-of-squares-zero-exact} to the
square equality. The signed-factor product is classified as zero:
$$
(a+b)\cdot(a+(-b))\hsame0.
$$

Now apply \autoref{thm:field-nonzero-factors-exclude-zero-product} to the two
nonzero witnesses $p_+$ and $p_-$ together with this displayed zero product.
That theorem says that such a zero product is impossible when both factors are
explicitly nonzero, so the assumed pair of signed-factor witnesses yields
$\mathsf{False}$.
\end{proof}

\begin{theorem}[Field affine two-sided equation solutions are unique from apartness]
\label{thm:field-affine-two-sided-equation-solution-unique-from-apartness}
For a BEDC history-level $\FieldUp$ candidate with operations
$\mathsf{add},\mathsf{mul},\mathsf{neg},\mathsf{zero},\mathsf{one}$ on
$\mathsf{BHist}$, additive group laws for
$\mathsf{add},\mathsf{neg},\mathsf{zero}$, addition congruence,
multiplication associativity, left and right multiplicative identity laws,
multiplication congruence, and indexed inverse laws for witnesses
$p_a:\mathsf{NonZero}_F(a)$ and $p_b:\mathsf{NonZero}_F(b)$, every
$x,y,d,c:\mathsf{BHist}$ satisfies
$$
\mathsf{add}(\mathsf{mul}(\mathsf{mul}(a,x),b),d)\hsame c
\quad\Longrightarrow\quad
\mathsf{add}(\mathsf{mul}(\mathsf{mul}(a,y),b),d)\hsame c
\quad\Longrightarrow\quad
x\hsame y.
$$
\end{theorem}
\begin{proof}
Assume the two displayed affine equations. Use symmetry of the second equation
and transitivity of $\hsame$ to compose the first equation with it. This gives
the classifier equality between the two affine outputs:
$$
\mathsf{add}(\mathsf{mul}(\mathsf{mul}(a,x),b),d)\hsame
\mathsf{add}(\mathsf{mul}(\mathsf{mul}(a,y),b),d).
$$

Apply \autoref{thm:field-affine-two-sided-map-classifier-exact-from-apartness}
with the same certified nonzero factors $p_a,p_b$ and offset $d$. The affine
injectivity theorem removes the shared affine map and yields $x\hsame y$.
\end{proof}

Per-object certificate: Traditional target: affine equation uniqueness. BEDC
source: $\FieldUp$ over $\mathsf{BHist}$ plus additive group laws. Carrier
choice: $\mathsf{BHist}$. Classifier choice: $\hsame$. Stability obligations:
the operations and laws required by affine two-sided injectivity. Ledger: the
two certified factors and offset. Core theorem list entry:
Theorem~\ref{thm:field-affine-two-sided-equation-solution-unique-from-apartness}.
Exactness predicate: two certified solutions of the same affine equation are
classified equal. Standard bridge and downstream use: canonical solution
extraction for FieldUp equation solving. Status: $\mathsf{ProofTarget}$.

\begin{theorem}[Field product nonzero excludes zero factors]
\label{thm:field-product-nonzero-excludes-zero-factors}
For a BEDC history-level $\FieldUp$ candidate with ring zero absorption,
multiplication congruence, classifier transport of $\mathsf{NonZero}_F$,
and contradiction of $\mathsf{NonZero}_F(0)$, every $a,b:\mathsf{BHist}$
with
$$
p_{ab}:\mathsf{NonZero}_F(\mathsf{mul}(a,b))
$$
satisfies both $a\hsame0\to\bot$ and $b\hsame0\to\bot$.
\end{theorem}
\begin{proof}
Assume $a\hsame0$. Multiplication congruence sends
$\mathsf{mul}(a,b)$ to $\mathsf{mul}(0,b)$, and
Theorem~\ref{thm:ring-mul-zero-absorption} identifies
$\mathsf{mul}(0,b)$ with $0$. Transitivity gives
$\mathsf{mul}(a,b)\hsame0$.

Transport $p_{ab}:\mathsf{NonZero}_F(\mathsf{mul}(a,b))$ along this
classifier equality. The transport clause supplies
$\mathsf{NonZero}_F(0)$, and the field zero-contradiction clause closes
the left-factor obstruction.

The right-factor obstruction is symmetric. From $b\hsame0$, multiplication
congruence sends $\mathsf{mul}(a,b)$ to $\mathsf{mul}(a,0)$, and
Theorem~\ref{thm:ring-mul-zero-absorption} identifies that product with
$0$. Transporting the same witness $p_{ab}$ to $\mathsf{NonZero}_F(0)$
again contradicts the field zero clause.
\end{proof}

$\mathsf{NonZero}_F(0)$. Ledger: product witness
$p_{ab}:\mathsf{NonZero}_F(\mathsf{mul}(a,b))$ and the tested zero-factor

\begin{theorem}[Field two-sided product nonzero excludes middle zero]
\label{thm:field-two-sided-product-nonzero-excludes-middle-zero}
For a BEDC history-level $\FieldUp$ candidate with the hypotheses of
\autoref{thm:field-two-sided-zero-product-cancel-from-apartness}, classifier
transport of $\mathsf{NonZero}_F$, and contradiction of
$\mathsf{NonZero}_F(0)$, every $a,b,x:\mathsf{BHist}$ with outer witnesses
$p_a:\mathsf{NonZero}_F(a)$ and $p_b:\mathsf{NonZero}_F(b)$ and product
witness
$$
p_{axb}:\mathsf{NonZero}_F(\mathsf{mul}(\mathsf{mul}(a,x),b)),
$$
satisfies $x\hsame0\to\bot$.
\end{theorem}
\begin{proof}
Assume $x\hsame0$. Use the reverse direction of
\autoref{thm:field-two-sided-zero-product-cancel-from-apartness} with the
outer witnesses $p_a$ and $p_b$. This derives
$$
\mathsf{mul}(\mathsf{mul}(a,x),b)\hsame0.
$$

Transport
$p_{axb}:\mathsf{NonZero}_F(\mathsf{mul}(\mathsf{mul}(a,x),b))$ along this
classifier equality. The nonzero transport clause supplies
$\mathsf{NonZero}_F(0)$.

The field zero-contradiction clause rejects $\mathsf{NonZero}_F(0)$, so the
assumed middle classifier $x\hsame0$ yields $\bot$.
\end{proof}

Per-object certificate: Traditional target: two-sided nonzero-product
middle-factor obstruction. BEDC source: $\FieldUp$ plus $\RingUp$ zero
absorption. Carrier choice: $\mathsf{BHist}$. Classifier choice: $\hsame$.
Stability obligations: two-sided zero-product exactness, nonzero transport,
and zero contradiction. Ledger: the two outer nonzero witnesses $p_a,p_b$
plus $p_{axb}$. Core theorem list:
Theorem~\ref{thm:field-two-sided-product-nonzero-excludes-middle-zero}.
Exactness predicate: obstruction to middle zero. Standard bridge: deferred.
Status: ProofTarget.
