\begin{theorem}[RatUp FieldUp bracketing selector independence package]
\label{thm:ratup-fieldup-bracketing-selector-independence-package}
For the concrete denominator tuple $(C_D^u,R_D^u,\mu_D^u,e_D^u,N_D^u)$ and
histories $h,k,\ell:\mathsf{BHist}$ with $C_D^u(h)$, $C_D^u(k)$, and
$C_D^u(\ell)$, let $E_s(h,k,\ell)$ be the endpoint selector of
\autoref{def:ratup-fieldup-bracketing-endpoint-selector}. If
$\mathsf{ThreeSupp}_D^u(h,k,\ell)$ holds, then for all selectors
$s,t:\mathsf{BMark}$ and every context $r:\mathsf{BHist}$, setting
$a:=E_s(h,k,\ell)$ and $b:=E_t(h,k,\ell)$ gives the following package:
\begin{enumerate}
  \item
  $$
  a\hsame b;
  $$
  \item
  $$
  \mathsf{RatHistoryClassifier}(a,b);
  $$
  \item
  $$
  \mathsf{fieldSingletonEmptyCarrier}(\mathsf{append}(r,a))\to\bot;
  $$
  \item
  $$
  \mathsf{fieldSingletonEmptyCarrier}(\mathsf{append}(a,r))\to\bot;
  $$
  \item
  $$
  \mathsf{fieldSingletonEmptyNonZero}(\mathsf{append}(r,a));
  $$
  \item
  $$
  \mathsf{fieldSingletonEmptyNonZero}(\mathsf{append}(a,r)).
  $$
\end{enumerate}
\end{theorem}
\begin{proof}
First apply \autoref{thm:ratup-fieldup-bracketing-endpoint-selector-same} to the
selectors $s,t:\mathsf{BMark}$. This gives the endpoint sameness row
$E_s(h,k,\ell)\hsame E_t(h,k,\ell)$.

Next apply
\autoref{thm:ratup-fieldup-bracketing-selector-rat-classifier-coverage} to the
same carrier witnesses and the same $\mathsf{ThreeSupp}_D^u(h,k,\ell)$
witness. This gives the Rat classifier row
$\mathsf{RatHistoryClassifier}(E_s(h,k,\ell),E_t(h,k,\ell))$.

Finally apply \autoref{thm:ratup-fieldup-selector-contextual-nonzero-package} to
the selected endpoint $E_s(h,k,\ell)$ and the context $r:\mathsf{BHist}$. Its
four rows give the two one-sided singleton-carrier exclusions and the two
corresponding $\mathsf{fieldSingletonEmptyNonZero}$ witnesses. Replacing the
temporary names $a$ and $b$ by their selector definitions assembles the displayed
package.
\end{proof}
