\begin{definition}[RatUp FieldUp concrete exit object]
\label{def:ratup-fieldup-concrete-exit-object}
For $\mathsf{RatUpFieldSource}_F$, the concrete
$\FieldUp(\RatUp)$ exit object is the finite row package over
$\mathsf{BHist}$ whose carrier and classifier rows are
$\mathsf{RatHistoryCarrier}$, $\mathsf{RatHistoryClassifier}$,
$\mathsf{RatDenomUnitCarrier}$, and
$\mathsf{RatDenomUnitClassifier}$.

Its operation rows are exactly the concrete table rows for $\mathsf{append}$,
$\Cont$, the denominator unit, the contextual action $\Gamma$, the
context-pair action, affine rows, inverse and nonzero rows, one-sided
multiplication cancellation exactness, two-sided product exactness, the
same-endpoint plus crossed singleton-classifier noncollapse rows, and the
selector-indexed bracketing rows generated by $E_s(h,k,\ell)$: selected
endpoint carrier coverage, selected endpoint denominator-classifier coverage,
and selector-indexed singleton-classifier exclusion under arbitrary contexts.
No schematic $\mathsf{mul}$, $\mathsf{add}$, or $\mathsf{neg}$ parameter is
part of this object.
\end{definition}

\autoref{thm:ratup-fieldup-concrete-operation-exit-exhaustion} and

\begin{proof}

$\FieldUp(\RatUp)$ over $\mathsf{RatUpFieldSource}_F$, including the ten-field

cross-endpoint exclusions, and bracketing-complete classifier package.

Because both cited theorems use the same source tuple, $\mathsf{BHist}$ carrier,
denominator endpoints, and operation-row inventory, the finite package defined
in Definition~\ref{def:ratup-fieldup-concrete-exit-object} and the concrete

\end{proof}

\begin{lemma}[RatUp FieldUp BMark selector endpoint-pair cases]
\label{lem:ratup-fieldup-bracketing-selector-pair-cases}
For the concrete denominator tuple $(C_D^u,R_D^u,\mu_D^u,e_D^u,N_D^u)$ and
histories $h,k,\ell:\mathsf{BHist}$, let $E_s(h,k,\ell)$ be the endpoint
selector of \autoref{def:ratup-fieldup-bracketing-endpoint-selector}. Set
$x:=\mu_D^u(\mu_D^u(h,k),\ell)$ and
$y:=\mu_D^u(h,\mu_D^u(k,\ell))$. Then for all selectors
$s,t:\mathsf{BMark}$, the selected endpoint pair
$(E_s(h,k,\ell),E_t(h,k,\ell))$ is covered by the four generated rows
$(x,x)$, $(y,y)$, $(x,y)$, and $(y,x)$ up to $\hsame$ on both endpoints.
\end{lemma}
\begin{proof}
Perform generated case analysis on the selector pair $(s,t)$. In every branch,
unfold the endpoint selector of
\autoref{def:ratup-fieldup-bracketing-endpoint-selector}; the remaining endpoint
comparison is reflexivity of $\hsame$.

In the $(\mathsf{BMark.b0},\mathsf{BMark.b0})$ branch, both selected endpoints
unfold to $x$, so the row $(x,x)$ supplies the pair witness.

In the $(\mathsf{BMark.b1},\mathsf{BMark.b1})$ branch, both selected endpoints
unfold to $y$, so the row $(y,y)$ supplies the pair witness.

In the $(\mathsf{BMark.b0},\mathsf{BMark.b1})$ branch, the left endpoint
unfolds to $x$ and the right endpoint unfolds to $y$, giving the row $(x,y)$.

In the $(\mathsf{BMark.b1},\mathsf{BMark.b0})$ branch, the left endpoint
unfolds to $y$ and the right endpoint unfolds to $x$, giving the row $(y,x)$.
\end{proof}

\begin{theorem}[RatUp FieldUp BMark-selected bracketing exit subpackage]
\label{thm:ratup-fieldup-bracketing-selector-exit-subpackage}
For the concrete denominator tuple $(C_D^u,R_D^u,\mu_D^u,e_D^u,N_D^u)$ and
histories $h,k,\ell:\mathsf{BHist}$ with $C_D^u(h)$, $C_D^u(k)$, and
$C_D^u(\ell)$, let $E_s(h,k,\ell)$ be the endpoint selector of
\autoref{def:ratup-fieldup-bracketing-endpoint-selector}. If
$\mathsf{ThreeSupp}_D^u(h,k,\ell)$ holds, then for all selectors
$s,t:\mathsf{BMark}$ and all contexts $L,R:\mathsf{BHist}$, the selected
endpoints form the following concrete exit subpackage:
\begin{enumerate}
  \item $C_D^u(E_s(h,k,\ell))$ and $C_D^u(E_t(h,k,\ell))$;
  \item $R_D^u(E_s(h,k,\ell),E_t(h,k,\ell))$;
  \item
  $$
  \mathsf{fieldSingletonEmptyClassifier}
  (\mathsf{append}(L,E_s(h,k,\ell)),\mathsf{append}(R,E_t(h,k,\ell)))\to\bot.
  $$
\end{enumerate}
\end{theorem}
\begin{proof}
Set $x:=\mu_D^u(\mu_D^u(h,k),\ell)$ and
$y:=\mu_D^u(h,\mu_D^u(k,\ell))$. Apply
\autoref{thm:ratup-fieldup-bracketing-complete-classifier-exit-certificate}
to the same concrete denominator tuple, carrier hypotheses, endpoint
definitions, and $\mathsf{ThreeSupp}_D^u(h,k,\ell)$ witness. This supplies
$C_D^u(x)$, $C_D^u(y)$, $R_D^u(x,y)$, and the four ordered endpoint-pair
singleton-classifier exclusions.

For carrier coverage, use the two generated $\mathsf{BMark}$ cases for $s$.
In the $\mathsf{BMark.b0}$ branch, the selected endpoint is $x$, so
$C_D^u(x)$ transports to $C_D^u(E_s(h,k,\ell))$. In the
$\mathsf{BMark.b1}$ branch, the selected endpoint is $y$, so $C_D^u(y)$
transports to $C_D^u(E_s(h,k,\ell))$. The same two-case argument for $t$
gives $C_D^u(E_t(h,k,\ell))$.

The displayed denominator-classifier row is exactly
\autoref{thm:ratup-fieldup-bracketing-selector-denominator-classifier-coverage}
applied to the same carrier data and support witness. It packages the
reflexive, symmetric, and crossed endpoint classifier rows across all selected
pairs $s,t:\mathsf{BMark}$.

The singleton-classifier rejection under arbitrary contexts is exactly
\autoref{thm:ratup-fieldup-bracketing-choice-classifier-exclusion} applied to
the same selector endpoints. Thus the selected pair $(s,t)$ and the contexts
$(L,R)$ carry both the internal denominator classifier row and the external
singleton-classifier exclusion required by the concrete exit subpackage.
\end{proof}

\begin{lemma}[RatUp FieldUp bracketing endpoint selector exhaustion]
\label{lem:ratup-fieldup-bracketing-endpoint-selector-exhaustion}
For histories $h,k,\ell:\mathsf{BHist}$, set
$x:=\mu_D^u(\mu_D^u(h,k),\ell)$ and
$y:=\mu_D^u(h,\mu_D^u(k,\ell))$. For every pair of selectors
$s,t:\mathsf{BMark}$, the selected endpoint pair
$(E_s(h,k,\ell),E_t(h,k,\ell))$ is one of
$(x,x)$, $(x,y)$, $(y,x)$, and $(y,y)$, with the corresponding displayed
equalities between $E_s(h,k,\ell)$, $E_t(h,k,\ell)$ and the selected endpoints
$x,y$.
\end{lemma}
\begin{proof}
Apply $\mathsf{BMark\_generated\_cases}$ to $s$ and then to $t$. This gives
four closed generated selector pairs.

\begin{enumerate}
  \item If $s=\mathsf{BMark.b0}$ and $t=\mathsf{BMark.b0}$, unfold
  \autoref{def:ratup-fieldup-bracketing-endpoint-selector}. Then
  $E_s(h,k,\ell)=x$ and $E_t(h,k,\ell)=x$, so the selected pair is $(x,x)$.

  \item If $s=\mathsf{BMark.b0}$ and $t=\mathsf{BMark.b1}$, unfold
  \autoref{def:ratup-fieldup-bracketing-endpoint-selector}. Then
  $E_s(h,k,\ell)=x$ and $E_t(h,k,\ell)=y$, so the selected pair is $(x,y)$.

  \item If $s=\mathsf{BMark.b1}$ and $t=\mathsf{BMark.b0}$, unfold
  \autoref{def:ratup-fieldup-bracketing-endpoint-selector}. Then
  $E_s(h,k,\ell)=y$ and $E_t(h,k,\ell)=x$, so the selected pair is $(y,x)$.

  \item If $s=\mathsf{BMark.b1}$ and $t=\mathsf{BMark.b1}$, unfold
  \autoref{def:ratup-fieldup-bracketing-endpoint-selector}. Then
  $E_s(h,k,\ell)=y$ and $E_t(h,k,\ell)=y$, so the selected pair is $(y,y)$.
\end{enumerate}

These four branches exhaust the generated $\mathsf{BMark}$ selectors, so no
other selected endpoint pair is available.
\end{proof}

This case-exhaustion lemma is consumed by
\autoref{thm:ratup-fieldup-bracketing-choice-classifier-exclusion},
\autoref{thm:ratup-fieldup-bracketing-selector-denominator-classifier-coverage},
and \autoref{thm:ratup-fieldup-selector-complete-denominator-well-formed-exit-slice}.

\begin{theorem}[RatUp FieldUp selector-complete denominator well-formed exit slice]

Definition~\ref{def:ratup-fieldup-concrete-exit-object}.

\begin{theorem}[RatUp FieldUp selector-complete denominator strict support package]
\label{thm:ratup-fieldup-selector-complete-denominator-strict-support-package}
For the concrete denominator tuple $(C_D^u,R_D^u,\mu_D^u,e_D^u,N_D^u)$ and
histories $h,k,\ell:\mathsf{BHist}$ with $C_D^u(h)$, $C_D^u(k)$, and
$C_D^u(\ell)$, let $E_s(h,k,\ell)$ be the endpoint selector of
\autoref{def:ratup-fieldup-bracketing-endpoint-selector}. If
$\mathsf{ThreeSupp}_D^u(h,k,\ell)$ holds, then for all selectors
$s,t:\mathsf{BMark}$ the selector-complete denominator instance of
\autoref{def:ratup-fieldup-selector-complete-denominator-instance} supplies
the strict-support endpoint package
$$
R_D^u(E_s(h,k,\ell),E_t(h,k,\ell))
\land
\mathsf{fieldSingletonEmptyNonZero}(E_s(h,k,\ell))
\land
\mathsf{fieldSingletonEmptyNonZero}(E_t(h,k,\ell)).
$$
\end{theorem}
\begin{proof}
Apply \autoref{thm:ratup-fieldup-selector-complete-denominator-laws} to the
same concrete denominator tuple, carrier hypotheses, support witness, and
selectors $s,t:\mathsf{BMark}$. Its classifier row gives
$R_D^u(E_s(h,k,\ell),E_t(h,k,\ell))$.

For the left strict-support endpoint, specialize the obstruction row of
\autoref{thm:ratup-fieldup-selector-complete-denominator-laws} to the selector
pair $(s,s)$ and to contexts $L=R=\emp$. Assume a forbidden witness
$\mathsf{fieldSingletonEmptyClassifier}(E_s(h,k,\ell),\emp)$. The empty-left
append unit transports this witness to a singleton-empty classifier on
$\mathsf{append}(\emp,E_s(h,k,\ell))$ against itself, contradicting the
specialized obstruction. By
\autoref{def:field-singleton-empty-nonzero}, this is exactly
$\mathsf{fieldSingletonEmptyNonZero}(E_s(h,k,\ell))$.

For the right strict-support endpoint, repeat the same argument with the
selector pair $(t,t)$ and contexts $L=R=\emp$. A forbidden classifier
$\mathsf{fieldSingletonEmptyClassifier}(E_t(h,k,\ell),\emp)$ transports along
the empty-left append unit to the obstructed classifier on
$\mathsf{append}(\emp,E_t(h,k,\ell))$ against itself. The resulting
contradiction is the required
$\mathsf{fieldSingletonEmptyNonZero}(E_t(h,k,\ell))$.

Combining the classifier row with the two singleton-empty complement witnesses
gives the displayed strict-support endpoint package.
\end{proof}
