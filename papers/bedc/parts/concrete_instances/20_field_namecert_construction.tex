\begin{theorem}[Field singleton-empty operation context nonzero obstruction]
\label{thm:field-singleton-empty-operation-context-nonzero-obstruction}
For histories $L,R,h,k,out:\mathsf{BHist}$ with
$\mathsf{fieldSingletonEmptyCarrier}(L)$,
$\mathsf{fieldSingletonEmptyCarrier}(R)$, and
$\mathsf{fieldSingletonEmptyNonZero}(out)$, singleton-empty operation contexts
reject proof-indexed nonzero outputs. The multiplication endpoint satisfies
$$
\mathsf{fieldSingletonEmptyClassifier}
  (\mathsf{append}(L,\mathsf{fieldSingletonEmptyMul}(h,k)),
   \mathsf{append}(R,out))
\Longrightarrow
\mathsf{False}.
$$
The unit endpoint satisfies
$$
\mathsf{fieldSingletonEmptyClassifier}
  (\mathsf{append}(L,\mathsf{fieldSingletonEmptyOne}),
   \mathsf{append}(R,out))
\Longrightarrow
\mathsf{False}.
$$
For every proof witness
$p:\mathsf{fieldSingletonEmptyNonZero}(h)$, the inverse endpoint satisfies
$$
\mathsf{fieldSingletonEmptyClassifier}
  (\mathsf{append}(L,\mathsf{fieldSingletonEmptyInv}(h,p)),
   \mathsf{append}(R,out))
\Longrightarrow
\mathsf{False}.
$$
\end{theorem}
\begin{proof}
Apply \autoref{thm:field-singleton-empty-operation-context-normalization} to
the multiplication, unit, or proof-indexed inverse classifier in the displayed
context. In each case the contextual classifier into $\mathsf{append}(R,out)$
is converted into $\mathsf{fieldSingletonEmptyCarrier}(out)$.

The carrier assertion is the empty endpoint input for
\autoref{thm:field-singleton-nonzero-empty-endpoint-absurd}. Combining it with
the assumed witness $\mathsf{fieldSingletonEmptyNonZero}(out)$ gives
$\mathsf{False}$ for each of the three operation contexts.
\end{proof}

\section{Apart-zero continuation exactness}
\label{sec:field-apartzero-continuation-exactness}

\begin{theorem}[Field apart-zero nested continuation empty-context exactness]
\label{thm:field-apartzero-nested-continuation-empty-context-iff}
For histories $l,h,r,u,v:\mathsf{BHist}$ with $\ContR(l,h,u)$ and
$\ContR(u,r,v)$, if $l\hsame\emp$ and $r\hsame\emp$, then
$$
\mathsf{ApartZero}_F(v)
\Longleftrightarrow
\mathsf{ApartZero}_F(h).
$$
\end{theorem}
\begin{proof}
The two continuation witnesses identify $u$ with $\mathsf{append}(l,h)$ and
$v$ with $\mathsf{append}(u,r)$. Continuation associativity and result
transport compare $v$ with the nested append endpoint
$\mathsf{append}(l,\mathsf{append}(h,r))$.

Transport $\mathsf{ApartZero}_F$ across this $\hsame$ comparison. The result
is exactly the bilateral empty-context statement of
\autoref{thm:field-apartzero-empty-context-iff}, applied with the witnesses
$l\hsame\emp$ and $r\hsame\emp$.
\end{proof}

\begin{theorem}[Field apart-zero nested continuation factor split]
\label{thm:field-apartzero-nested-continuation-factor-split}
For histories $l,h,r,u,v,q,w:\mathsf{BHist}$ with $\ContR(l,h,u)$,
$\ContR(u,r,v)$, and $\ContR(v,q,w)$, if $l\hsame\emp$ and $r\hsame\emp$,
then
$$
\mathsf{ApartZero}_F(w)
\Longleftrightarrow
\mathsf{ApartZero}_F(h)\lor\mathsf{ApartZero}_F(q).
$$
\end{theorem}
\begin{proof}
Apply \autoref{thm:field-apartzero-continuation-endpoint-split} to the tail
continuation witness $\ContR(v,q,w)$. This gives
$\mathsf{ApartZero}_F(w)$ exactly when
$\mathsf{ApartZero}_F(v)\lor\mathsf{ApartZero}_F(q)$ holds.

Apply
\autoref{thm:field-apartzero-nested-continuation-empty-context-iff} to the
first two continuation witnesses and the exterior empty-context witnesses
$l\hsame\emp$ and $r\hsame\emp$. Substituting this equivalence for the first
disjunct yields the displayed factor split.
\end{proof}

Per-object template: traditional target continuation-tail apartness factor
split under empty exterior context; BEDC source $\mathsf{BHist}$, $\ContR$,
and $\mathsf{ApartZero}_F$; carrier apart-zero histories; classifier $\hsame$
transport; stability obligations nested empty-context exactness plus
continuation endpoint split; ledger the two empty-context witnesses and tail
$q$; exactness predicate the displayed iff; standard bridge ordinary
product/context apartness factorization; status $\mathsf{ProofTarget}$.

\begin{theorem}[Field apart-zero two-tail nested continuation factor split]
\label{thm:field-apartzero-two-tail-nested-continuation-factor-split}
For histories $l,h,r,u,v,q,w,s,z:\mathsf{BHist}$ with $\ContR(l,h,u)$,
$\ContR(u,r,v)$, $\ContR(v,q,w)$, and $\ContR(w,s,z)$, if
$l\hsame\emp$ and $r\hsame\emp$, then
$$
\mathsf{ApartZero}_F(z)
\Longleftrightarrow
\mathsf{ApartZero}_F(h)\lor\mathsf{ApartZero}_F(q)\lor\mathsf{ApartZero}_F(s).
$$
\end{theorem}
\begin{proof}
Apply \autoref{thm:field-apartzero-continuation-endpoint-split} to the final
tail continuation witness $\ContR(w,s,z)$. This gives
$\mathsf{ApartZero}_F(z)$ exactly when
$\mathsf{ApartZero}_F(w)\lor\mathsf{ApartZero}_F(s)$ holds.

Apply \autoref{thm:field-apartzero-nested-continuation-factor-split} to the
first three continuation witnesses and the exterior witnesses $l\hsame\emp$
and $r\hsame\emp$. This identifies $\mathsf{ApartZero}_F(w)$ with
$\mathsf{ApartZero}_F(h)\lor\mathsf{ApartZero}_F(q)$.

For the forward direction, case-analyze the endpoint disjunction. The
$\mathsf{ApartZero}_F(s)$ case supplies the right tail disjunct. In the
$\mathsf{ApartZero}_F(w)$ case, case-analyze the nested split and place the
resulting witness in the $h$ or $q$ disjunct.

For the reverse direction, case-analyze the displayed three-way disjunction.
The $h$ and $q$ cases first rebuild $\mathsf{ApartZero}_F(w)$ by the reverse
direction of the nested split, and then rebuild $\mathsf{ApartZero}_F(z)$ by
the reverse direction of the endpoint split. The $s$ case uses the reverse
direction of the endpoint split directly.
\end{proof}

Per-object template: object $\FieldUp$ apart-zero multi-tail factorization;
source $\mathsf{BHist}$ plus $\ContR$; carrier apart-zero histories;
classifier $\hsame$ transport; transport predicate $\mathsf{ApartZero}_F$;
ledger the empty exterior witnesses $l\hsame\emp$ and $r\hsame\emp$; tail
data the two tail factors $q$ and $s$; core theorem this block plus the
one-tail split and endpoint split; exactness predicate the displayed iff;
status $\mathsf{ProofTarget}$.

\begin{theorem}[Field apart-zero three-tail nested continuation factor split]
\label{thm:field-apartzero-three-tail-nested-continuation-factor-split}
For histories $l,h,r,u,v,q,w,s,z,t,y:\mathsf{BHist}$ with
$\ContR(l,h,u)$, $\ContR(u,r,v)$, $\ContR(v,q,w)$, $\ContR(w,s,z)$, and
$\ContR(z,t,y)$, if $l\hsame\emp$ and $r\hsame\emp$, then
$$
\mathsf{ApartZero}_F(y)
\Longleftrightarrow
\mathsf{ApartZero}_F(h)\lor\mathsf{ApartZero}_F(q)\lor
\mathsf{ApartZero}_F(s)\lor\mathsf{ApartZero}_F(t).
$$
\end{theorem}
\begin{proof}
Apply \autoref{thm:field-apartzero-continuation-endpoint-split} to the
endpoint continuation witness $\ContR(z,t,y)$. This splits
$\mathsf{ApartZero}_F(y)$ into
$\mathsf{ApartZero}_F(z)\lor\mathsf{ApartZero}_F(t)$.

Apply
\autoref{thm:field-apartzero-two-tail-nested-continuation-factor-split} to
the first four continuation witnesses and the exterior empty-context
witnesses $l\hsame\emp$ and $r\hsame\emp$. This rewrites
$\mathsf{ApartZero}_F(z)$ as the three-way disjunction on $h,q,s$.

For the forward direction, case-analyze the endpoint split. The
$\mathsf{ApartZero}_F(t)$ case supplies the final tail disjunct. In the
$\mathsf{ApartZero}_F(z)$ case, case-analyze the two-tail nested split and
place the resulting evidence in the $h$, $q$, or $s$ disjunct.

For the reverse direction, case-analyze the displayed four-way disjunction.
The $h$, $q$, and $s$ cases first rebuild $\mathsf{ApartZero}_F(z)$ by the
reverse direction of the two-tail nested split, and then rebuild
$\mathsf{ApartZero}_F(y)$ by the reverse direction of the endpoint split. The
$t$ case uses the reverse direction of the endpoint split directly.
\end{proof}

Per-object template: traditional target finite product/context apartness

and two-tail nested split; ledger exterior empty witnesses plus tails $q,s,t$;

\begin{theorem}[Field apart-zero two-tail contextual singleton result exclusion]
\label{thm:field-apartzero-two-tail-contextual-singleton-result-exclusion}
For histories $L,R,l,h,r,u,v,q,w,s,z,endpoint:\mathsf{BHist}$ with
$\mathsf{FieldSingletonCarrier}(L)$, $\mathsf{FieldSingletonCarrier}(R)$,
$\ContR(l,h,u)$, $\ContR(u,r,v)$, $\ContR(v,q,w)$, $\ContR(w,s,z)$,
$l\hsame\emp$, $r\hsame\emp$, and
$\mathsf{FieldSingletonClassifier}(\mathsf{append}(L,z),\mathsf{append}(R,endpoint))$,
one has
$$
(\mathsf{ApartZero}_F(h)\to\bot)
\land
(\mathsf{ApartZero}_F(q)\to\bot)
\land
(\mathsf{ApartZero}_F(s)\to\bot).
$$
\end{theorem}
\begin{proof}
Use \autoref{thm:field-singleton-classifier-append-context-cancel} with the
two singleton-context carrier witnesses for $L$ and $R$. The forward direction
of the cancellation equivalence turns the displayed contextual classifier into
$\mathsf{FieldSingletonClassifier}(z,endpoint)$.

Apply
\autoref{thm:field-apartzero-continuation-singleton-result-factor-absurd} to
the final continuation witness $\ContR(w,s,z)$ and the classifier just
obtained. This gives exclusions
$\mathsf{ApartZero}_F(w)\to\bot$ and
$\mathsf{ApartZero}_F(s)\to\bot$.

It remains to rule out the two exposed earlier factors. Apply
\autoref{thm:field-apartzero-two-tail-nested-continuation-factor-split} to the
first three continuation witnesses and the exterior witnesses $l\hsame\emp$
and $r\hsame\emp$. The reverse direction sends either
$\mathsf{ApartZero}_F(h)$ or $\mathsf{ApartZero}_F(q)$ to
$\mathsf{ApartZero}_F(w)$, contradicting the exclusion of $w$. Together with
the direct exclusion of $s$, these three maps are exactly the displayed
conjunction.
\end{proof}

Per-object template: $\FieldUp$ apart-zero singleton-result exclusion; source
$\mathsf{BHist}$ plus $\ContR$; carrier predicate $\mathsf{ApartZero}_F$;
classifier $\hsame$ transport; ledger the hidden singleton contexts; exactness
predicate the displayed conjunction; status $\mathsf{ProofTarget}$.

\begin{theorem}[Field apart-zero nested continuation singleton-result factor exclusion]
\label{thm:field-apartzero-nested-continuation-singleton-result-factor-absurd}
For histories $l,h,r,u,v,q,w,endpoint:\mathsf{BHist}$ with
$\ContR(l,h,u)$, $\ContR(u,r,v)$, and $\ContR(v,q,w)$, if
$l\hsame\emp$, $r\hsame\emp$, and
$\mathsf{FieldSingletonClassifier}(w,endpoint)$, then
$$
\bigl(\mathsf{ApartZero}_F(h)\Rightarrow\mathsf{False}\bigr)\land
\bigl(\mathsf{ApartZero}_F(q)\Rightarrow\mathsf{False}\bigr).
$$
\end{theorem}
\begin{proof}
Apply \autoref{thm:field-singleton-classifier-empty-endpoints} to the
classifier witness $\mathsf{FieldSingletonClassifier}(w,endpoint)$. The
result gives in particular the endpoint comparison $w\hsame\emp$.

Assume $\mathsf{ApartZero}_F(h)$. The reverse direction of
\autoref{thm:field-apartzero-nested-continuation-factor-split}, applied to
the left disjunct and the three continuation witnesses, produces
$\mathsf{ApartZero}_F(w)$. Applying this apart-zero witness to
$w\hsame\emp$ gives $\mathsf{False}$.

Assume $\mathsf{ApartZero}_F(q)$. The same reverse direction of the nested
factor split, now applied to the right disjunct, again produces
$\mathsf{ApartZero}_F(w)$. Applying it to $w\hsame\emp$ gives
$\mathsf{False}$. Pairing the two implications gives the displayed
exclusion.
\end{proof}

Per-object template: traditional target zero-result factor exclusion for
nested field multiplication; BEDC source $\mathsf{BHist}$, $\ContR$,
$\mathsf{FieldSingletonClassifier}$, and $\mathsf{ApartZero}_F$; carrier
apart-zero histories; classifier $\hsame$ transport; ledger the hidden
singleton result endpoint and the two exposed factors; core theorem list
the nested continuation factor split plus singleton endpoint classifier;
exactness predicate the displayed paired exclusion; standard bridge ordinary
zero product excludes nonzero factors under a singleton result; status
$\mathsf{ProofTarget}$.
