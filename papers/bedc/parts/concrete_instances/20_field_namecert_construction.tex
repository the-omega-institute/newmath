\section{Rat denominator empty-unit continuation monoid instance}
\label{sec:field-rat-denominator-empty-unit-continuation-monoid-instance}

\begin{definition}[Rat denominator empty-unit continuation monoid instance]
\label{def:field-rat-denominator-empty-unit-continuation-monoid-instance}
The empty-unit denominator continuation monoid data for
$\FieldUp(\mathsf{RatUp})$ are the concrete tuple
$(C_D^u,R_D^u,\mu_D^u,e_D^u)$ defined by
$$
C_D^u(h)\Longleftrightarrow\mathsf{RatDenomUnitCarrier}(h),
$$
$$
R_D^u(h,k)\Longleftrightarrow\mathsf{RatDenomUnitClassifier}(h,k),
$$
$$
\mu_D^u(h,k):=\mathsf{append}(h,k),
$$
and $e_D^u:=\emp$. For every pair of histories $h,k:\mathsf{BHist}$, the
multiplication is equipped with the canonical continuation witness
$$
\Cont(h,k,\mathsf{append}(h,k)).
$$
\end{definition}

\begin{theorem}[Rat denominator empty-unit continuation monoid laws]
\label{thm:field-rat-denominator-empty-unit-continuation-monoid-laws}
The concrete tuple
$(C_D^u,R_D^u,\mu_D^u,e_D^u)$ of
\autoref{def:field-rat-denominator-empty-unit-continuation-monoid-instance}
satisfies the denominator empty-unit continuation monoid laws:
\begin{enumerate}
  \item The unit is carried: $C_D^u(e_D^u)$.
  \item The multiplication is closed: if $C_D^u(h)$ and $C_D^u(k)$, then
  $C_D^u(\mu_D^u(h,k))$.
  \item The left and right unit products are classified with their endpoint:
  for every $h$ with $C_D^u(h)$,
  $R_D^u(\mu_D^u(e_D^u,h),h)$ and
  $R_D^u(\mu_D^u(h,e_D^u),h)$.
  \item The multiplication is associative under $R_D^u$: if
  $C_D^u(h)$, $C_D^u(k)$, and $C_D^u(\ell)$, then
  $R_D^u(\mu_D^u(\mu_D^u(h,k),\ell),
  \mu_D^u(h,\mu_D^u(k,\ell)))$.
  \item The multiplication respects classified endpoints: if $R_D^u(h,h')$
  and $R_D^u(k,k')$, then
  $R_D^u(\mu_D^u(h,k),\mu_D^u(h',k'))$.
\end{enumerate}
\end{theorem}
\begin{proof}
Unfold $C_D^u$, $R_D^u$, $\mu_D^u$, and $e_D^u$. The displayed multiplication
is the external append operation, so each product carries the canonical
continuation witness $\Cont(h,k,\mathsf{append}(h,k))$.

The carried empty unit, append closure, and the two one-sided unit
classifications are exactly the first four clauses of
\autoref{thm:field-rat-denominator-continuation-adjoined-unit-laws} applied to
those canonical continuation witnesses.

For associativity, use the same adjoined-unit theorem with the four canonical
continuation witnesses for the two inner and two outer products. Its fifth
clause gives the classifier comparison between the two bracketed append
results.

It remains to show multiplication congruence. From $R_D^u(h,h')$ and
$R_D^u(k,k')$, unfold the classifier to obtain the four carried endpoint
witnesses together with $h\hsame h'$ and $k\hsame k'$. Apply
\autoref{thm:external-append-hsame-congruence} to the two $\hsame$ comparisons;
append closure for $C_D^u$ supplies the two carried product witnesses. These
three pieces are exactly
$R_D^u(\mu_D^u(h,k),\mu_D^u(h',k'))$.

Thus the denominator continuation fields are concrete BEDC data: carrier
$\mathsf{RatDenomUnitCarrier}$, classifier
$\mathsf{RatDenomUnitClassifier}$, multiplication $\mathsf{append}$, and unit
$\emp$, with no abstract operator field being used.
\end{proof}

\section{Rat denominator unit endpoint exactness}
\label{sec:field-rat-denominator-unit-endpoint-exactness}

\begin{theorem}[Rat denominator continuation right-unit endpoint iff empty]
\label{thm:field-rat-denominator-continuation-right-unit-endpoint-iff-empty}
For every history $u:\mathsf{BHist}$, the right-unit classifier law is
equivalent to external emptiness:
$$
\left(
\forall d,r:\mathsf{BHist}.\;
\mathsf{RatHistoryCarrier}(d)\to
\Cont(d,u,r)\to
\mathsf{RatHistoryClassifier}(r,d)
\right)
\Longleftrightarrow
u\hsame\emp.
$$
\end{theorem}
\leanchecked{BEDC.Derived.FieldUp.field\_rat\_denominator\_continuation\_right\_unit\_law\_endpoint\_empty\_iff}
\begin{proof}
For the forward direction, apply
Theorem~\ref{thm:field-rat-denominator-continuation-unit-endpoint-empty} to
the displayed right-unit classifier law. Its first endpoint-forcing clause
returns $u\hsame\emp$.

Conversely, assume $u\hsame\emp$, a carrier witness
$\mathsf{RatHistoryCarrier}(d)$, and a displayed continuation
$\Cont(d,u,r)$. Transport the second continuation endpoint along
$u\hsame\emp$ to obtain a continuation presentation $\Cont(d,\emp,r)$ with
the same displayed result. The right external-boundary law in
Theorem~\ref{thm:field-rat-denominator-continuation-external-boundary-laws}
then yields $\mathsf{RatHistoryClassifier}(r,d)$.
\end{proof}

\begin{theorem}[Rat denominator continuation left-unit endpoint iff empty]
\label{thm:field-rat-denominator-continuation-left-unit-endpoint-iff-empty}
For every history $u:\mathsf{BHist}$, the left-unit classifier law is equivalent
to external emptiness:
$$
\left(
\forall d,r:\mathsf{BHist}.\;
\mathsf{RatHistoryCarrier}(d)\to
\Cont(u,d,r)\to
\mathsf{RatHistoryClassifier}(r,d)
\right)
\Longleftrightarrow
u\hsame\emp.
$$
\end{theorem}
\leanchecked{BEDC.Derived.FieldUp.field\_rat\_denominator\_continuation\_left\_unit\_law\_endpoint\_empty\_iff}
\begin{proof}
For the forward direction, apply
Theorem~\ref{thm:field-rat-denominator-continuation-unit-endpoint-empty} to
the displayed left-unit classifier law. Its second endpoint-forcing clause
returns $u\hsame\emp$.

Conversely, assume $u\hsame\emp$, a carrier witness
$\mathsf{RatHistoryCarrier}(d)$, and a displayed continuation
$\Cont(u,d,r)$. Transport the first continuation endpoint along
$u\hsame\emp$ to obtain $\Cont(\emp,d,r)$. The left external-boundary law in
Theorem~\ref{thm:field-rat-denominator-continuation-external-boundary-laws}
classifies the displayed result as
$\mathsf{RatHistoryClassifier}(r,d)$.
\end{proof}

\begin{theorem}[Rat denominator external unit separation package]
\label{thm:field-rat-denominator-external-unit-separation-package}
For the concrete denominator carrier, classifier, and continuation operation
given by
$$
C_D(h)\Longleftrightarrow\mathsf{RatHistoryCarrier}(h),
$$
$$
R_D(h,k)\Longleftrightarrow\mathsf{RatHistoryClassifier}(h,k),
$$
with operation represented by $\Cont$, every history $u:\mathsf{BHist}$
satisfies the following three clauses.
\begin{enumerate}
\item The right classifier-unit law
$$
\forall d,r:\mathsf{BHist}.\;
C_D(d)\to
\Cont(d,u,r)\to
R_D(r,d)
$$
is equivalent to $u\hsame\emp$.
\item The left classifier-unit law
$$
\forall d,r:\mathsf{BHist}.\;
C_D(d)\to
\Cont(u,d,r)\to
R_D(r,d)
$$
is equivalent to $u\hsame\emp$.
\item If $C_D(u)$ holds, then either one-sided classifier-unit law displayed
above implies $\mathsf{False}$.
\end{enumerate}
\end{theorem}
\begin{proof}
The displayed definitions of $C_D$ and $R_D$ identify the right classifier-unit
law with the concrete law in
\autoref{thm:field-rat-denominator-continuation-right-unit-endpoint-iff-empty}.
That theorem gives the first equivalence exactly.

The same unpacking identifies the left classifier-unit law with the concrete
law in
\autoref{thm:field-rat-denominator-continuation-left-unit-endpoint-iff-empty}.
That theorem gives the second equivalence.

For the carried exclusion, assume $C_D(u)$. This is precisely
$\mathsf{RatHistoryCarrier}(u)$. The right one-sided law contradicts
\autoref{thm:field-rat-denominator-continuation-no-internal-unit}, while the
left one-sided law contradicts
\autoref{thm:field-rat-denominator-continuation-no-internal-left-unit}.

Equivalently, either endpoint equivalence first yields $u\hsame\emp$, and then
\autoref{thm:rat-history-carrier-not-empty} excludes the simultaneous carried
witness. Thus the empty endpoint is external to the positive Rat denominator
carrier even though it is exact for the classifier-unit boundary laws.
\end{proof}

\textbf{Traditional target}: external unit boundary for the Rat denominator
continuation fragment; \textbf{BEDC source}: $\mathsf{RatHistoryCarrier}$,
$\mathsf{RatHistoryClassifier}$, and $\Cont$; \textbf{carrier}:
$C_D$; \textbf{classifier}: $R_D$; \textbf{stability obligation}: endpoint
exactness plus carried-unit exclusion; \textbf{ledger}: continuation witness;
\textbf{core theorem}:
\autoref{thm:field-rat-denominator-external-unit-separation-package};
\textbf{exactness predicate}: external-unit iff empty plus no internal carried
unit; \textbf{standard bridge}: deferred; \textbf{status}:
$\mathsf{ProofTarget}$.

\textbf{Traditional target}: denominator-side continuation unit endpoint

\textbf{carrier choice}: Rat denominators; \textbf{classifier choice}: Rat

\section{Concrete Rat denominator external-unit package}

\begin{theorem}[Rat denominator concrete continuation is cancellative and unitless]
\label{thm:field-rat-denominator-concrete-cancellative-unitless-semigroup}
For the concrete data $C_D$, $R_D$, and $\mu_D$ of
\autoref{def:field-rat-denominator-concrete-continuation-instance}, the
denominator-side fragment consumed by the $\FieldUp(\mathsf{RatUp})$
certificate has the following form.
\begin{enumerate}
  \item $C_D$, $R_D$, and $\mu_D$ form the carried commutative semigroup
  fragment: multiplication closes on carried denominators, commutativity and
  associativity hold up to $R_D$, binary classifier congruence is respected in
  both inputs, and the Rat history ledger is closed under classified carried
  right tails.
  \item For all histories $p,q,d,e,pd,pe,left,right:\mathsf{BHist}$, if
  $\mathsf{UnaryHistory}(p)$, $\mathsf{UnaryHistory}(q)$, $C_D(d)$, $C_D(e)$,
  $\Cont(p,d,pd)$, $\Cont(p,e,pe)$, $\Cont(pd,q,left)$, and
  $\Cont(pe,q,right)$, then the common unary context is exact for both the
  classifier and the ledger:
  $$
  R_D(left,right)\Longleftrightarrow R_D(d,e).
  $$
  $$
  \mathsf{RatHistoryLedgerPolicy}(left,right)\Longleftrightarrow
  \mathsf{RatHistoryLedgerPolicy}(d,e).
  $$
  \item For every endpoint $u:\mathsf{BHist}$, if either one-sided
  classifier-unit law
  $$
  \forall d,r:\mathsf{BHist}.\;
  C_D(d)\to
  \Cont(d,u,r)\to
  R_D(r,d)
  $$
  or
  $$
  \forall d,r:\mathsf{BHist}.\;
  C_D(d)\to
  \Cont(u,d,r)\to
  R_D(r,d)
  $$
  holds, then $u\hsame\emp$. If also $C_D(u)$, then $\mathsf{False}$.
\end{enumerate}
\end{theorem}
\begin{proof}
Unfold the concrete data of
\autoref{def:field-rat-denominator-concrete-continuation-instance}, so that
$C_D$ is $\mathsf{RatHistoryCarrier}$, $R_D$ is
$\mathsf{RatHistoryClassifier}$, and $\mu_D$ is append with its canonical
continuation witness.

The semigroup fragment is exactly
\autoref{thm:field-rat-denominator-concrete-continuation-instance-laws}. Its
clauses give carrier closure, commutativity, associativity, binary classifier
congruence, and ledger closure for the displayed concrete operation.

For the common-context clause, apply
\autoref{thm:field-rat-denominator-continuation-common-context-exactness-package}
to the two unary contexts and the four displayed continuation witnesses. After
the same unfolding of $C_D$ and $R_D$, its two conclusions are precisely the
classifier cancellation equivalence and the ledger exactness equivalence in the
statement.

For the endpoint obstruction, use
\autoref{thm:field-rat-denominator-continuation-right-unit-endpoint-iff-empty}
for the right-unit classifier law and
\autoref{thm:field-rat-denominator-continuation-left-unit-endpoint-iff-empty}
for the left-unit classifier law. Either case yields $u\hsame\emp$. If a
carrier witness $C_D(u)$ is also present, unfold $C_D$ and apply
\autoref{thm:rat-history-carrier-not-empty} to that witness and the comparison
$u\hsame\emp$, giving $\mathsf{False}$.
\end{proof}

\begin{theorem}[Rat denominator continuation exact external-unit package]
\label{thm:field-rat-denominator-concrete-external-unit-package}
For the concrete Rat denominator continuation data
$(C_D,R_D,\mu_D)$ of
\autoref{def:field-rat-denominator-concrete-continuation-instance}, the
following package holds.
\begin{enumerate}
  \item If $C_D(d)$ and $C_D(e)$, then $C_D(\mu_D(d,e))$.
  \item If $C_D(d)$ and $C_D(e)$, then
  $R_D(\mu_D(d,e),\mu_D(e,d))$.
  \item If $C_D(d)$, $C_D(e)$, and $C_D(f)$, then
  $R_D(\mu_D(\mu_D(d,e),f),\mu_D(d,\mu_D(e,f)))$.
  \item If $R_D(d,d')$ and $R_D(e,e')$, then
  $R_D(\mu_D(d,e),\mu_D(d',e'))$.
  \item If $\mathsf{RatHistoryLedgerPolicy}(\rho,v)$, $C_D(e)$, and
  $R_D(e,e')$, then
  $\mathsf{RatHistoryLedgerPolicy}(\mu_D(\rho,e),\mu_D(v,e'))$.
  \item For every endpoint $u:\mathsf{BHist}$,

  $$
  \left(
  \forall d,r:\mathsf{BHist}.\;
  C_D(d)\to
  \Cont(d,u,r)\to
  R_D(r,d)
  \right)
  \Longleftrightarrow
  u\hsame\emp.
  $$

  \item For every endpoint $u:\mathsf{BHist}$,

  $$
  \left(
  \forall d,r:\mathsf{BHist}.\;
  C_D(d)\to
  \Cont(u,d,r)\to
  R_D(r,d)
  \right)
  \Longleftrightarrow
  u\hsame\emp.
  $$

  \item If $C_D(u)$, then neither displayed one-sided classifier-unit law can
  hold for $u$.
  \item If $u$ and $v$ each satisfy either displayed one-sided classifier-unit
  law, then $u\hsame v$.
\end{enumerate}
\end{theorem}
\begin{proof}
Unfold the concrete names $C_D$, $R_D$, and $\mu_D$ only as needed. The first
five clauses are exactly the first law group in
\autoref{thm:field-rat-denominator-concrete-continuation-instance-laws}:
carrier closure, commutative classifier, associative classifier, binary
classifier congruence, and ledger closure.

For the right external-unit exactness clause, read $C_D$ as
$\mathsf{RatHistoryCarrier}$ and $R_D$ as $\mathsf{RatHistoryClassifier}$. Apply
\autoref{thm:field-rat-denominator-continuation-right-unit-endpoint-iff-empty}
to the displayed right-unit law.

For the left external-unit exactness clause, use the same identifications and
apply
\autoref{thm:field-rat-denominator-continuation-left-unit-endpoint-iff-empty}.

For carried-endpoint exclusion, suppose $C_D(u)$ and one of the displayed
one-sided classifier-unit laws holds. The corresponding exactness clause gives
$u\hsame\emp$. Since $C_D(u)$ is $\mathsf{RatHistoryCarrier}(u)$, rational
history carrier nonemptiness,
\autoref{thm:rat-history-carrier-not-empty}, contradicts $u\hsame\emp$.

Finally, take two one-sided classifier-unit endpoints $u$ and $v$. The
right/right, left/left, and right/left cases are the three clauses of
\autoref{thm:field-rat-denominator-continuation-unit-endpoint-unique}. The
left/right case follows by applying the mixed clause in the opposite order and
then using symmetry of $\hsame$.
\end{proof}

\textbf{Traditional target}: denominator-side continuation package for
$\FieldUp(\mathsf{RatUp})$ with an external unit endpoint; \textbf{BEDC source}:
$\mathsf{RatHistoryCarrier}$, $\mathsf{RatHistoryClassifier}$, $\Cont$,
$\mathsf{append}$, and $\mathsf{RatHistoryLedgerPolicy}$; \textbf{carrier
choice}: $C_D$ from
\autoref{def:field-rat-denominator-concrete-continuation-instance};
\textbf{classifier choice}: $R_D$ from
\autoref{def:field-rat-denominator-concrete-continuation-instance};
\textbf{stability obligation}: carrier closure, commutative classifier,
associative classifier, classifier congruence, ledger closure, and endpoint
uniqueness; \textbf{ledger}: the Rat history ledger policy transported through
$\mu_D$; \textbf{core theorem}:
\autoref{thm:field-rat-denominator-concrete-external-unit-package};
\textbf{exactness predicate}: the right and left external-unit iff clauses
identifying the endpoint with $\emp$; \textbf{standard bridge}: deferred;
\textbf{status}: $\mathsf{ProofTarget}$.

\begin{theorem}[Classifier-complement singleton FieldUp concrete instance]
\label{thm:field-singleton-empty-concrete-instance-certificate}
Over $\mathsf{BHist}$, the concrete BEDC tuple
$$
(\mathsf{fieldSingletonEmptyCarrier},
\mathsf{fieldSingletonEmptyClassifier},
\mathsf{fieldSingletonEmptyNonZero},
\mathsf{fieldSingletonEmptyMul},
\mathsf{fieldSingletonEmptyOne},
\mathsf{fieldSingletonEmptyInv})
$$
assembles the singleton classifier-complement $\FieldUp$ instance as follows:
the carrier is inhabited by $\emp$; the carrier and classifier carry a
$\mathsf{SemanticNameCert}$; $\mathsf{fieldSingletonEmptyNonZero}(\emp)$ is
contradictory; $\mathsf{fieldSingletonEmptyNonZero}$ transports along
$\mathsf{fieldSingletonEmptyClassifier}$; proof-indexed inverses remain in the
carrier; and the left and right products against a proof-indexed inverse are
classified as $\mathsf{fieldSingletonEmptyOne}$.
\end{theorem}
\begin{proof}
The carrier and classifier certificate is the checked
\leanchecked{BEDC.Derived.FieldUp.fieldSingletonEmptyCarrier\_semanticNameCert}.
It packages the singleton carrier inhabitance, classifier reflexivity,
symmetry, transitivity, carrier transport, and the source, pattern, and ledger
soundness fields for the same concrete predicate and classifier.

The field-specific law package is the checked
\leanchecked{BEDC.Derived.FieldUp.field\_singleton\_empty\_schema\_laws}.
Its clauses are exactly the contradiction of
$\mathsf{fieldSingletonEmptyNonZero}(\emp)$, nonzero transport along the
classifier, inverse closure, and the two proof-indexed inverse laws for
$\mathsf{fieldSingletonEmptyMul}$, $\mathsf{fieldSingletonEmptyOne}$, and
$\mathsf{fieldSingletonEmptyInv}$.

Finally,
\autoref{thm:field-singleton-empty-nonzero-carrier-complement} gives the paper
reading of the nonzero predicate as the complement of the empty singleton
classifier. Thus the displayed tuple is not merely a list of separate checked
components: it is the concrete $\mathsf{BHist}$ classifier-complement
singleton instance with its naming certificate and its proof-indexed field-law
package aligned on the same carrier, classifier, predicate, and operations.
\end{proof}
