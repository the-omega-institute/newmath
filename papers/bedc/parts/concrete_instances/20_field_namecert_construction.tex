\begin{theorem}[RatUp FieldUp five-factor nonzero factor ledger]
\label{thm:ratup-fieldup-five-factor-nonzero-factor-ledger}
For a BEDC history-level $\FieldUp(\mathsf{RatUp})$ candidate over
$\mathsf{BHist}$ satisfying the hypotheses of
\autoref{thm:field-four-factor-product-nonzero-iff-factors} and
\autoref{thm:field-product-nonzero-of-nonzero-factors}, together with the
zero-factor exclusion hypotheses used by
\autoref{thm:field-product-nonzero-excludes-zero-factors}, assume the
apartness-to-nonzero bridge
$$
\alpha_h:\bigl(h\hsame0\to\bot\bigr)\to\mathsf{NonZero}_F(h)
$$
for each history term occurring below. Then every
$a,b,c,d,e:\mathsf{BHist}$ satisfies
$$
\mathsf{NonZero}_F(
\mathsf{mul}(\mathsf{mul}(\mathsf{mul}(\mathsf{mul}(a,b),c),d),e))
\Longleftrightarrow
\mathsf{NonZero}_F(a)\wedge\mathsf{NonZero}_F(b)\wedge
\mathsf{NonZero}_F(c)\wedge\mathsf{NonZero}_F(d)\wedge
\mathsf{NonZero}_F(e).
$$
\end{theorem}
\begin{proof}
For the forward direction, assume
$p_{abcde}:\mathsf{NonZero}_F(
\mathsf{mul}(\mathsf{mul}(\mathsf{mul}(\mathsf{mul}(a,b),c),d),e))$.
Apply \autoref{thm:field-product-nonzero-excludes-zero-factors} to the outer
binary product with left factor
$\mathsf{mul}(\mathsf{mul}(\mathsf{mul}(a,b),c),d)$ and right factor $e$. This
gives apartness of the fourfold prefix from $0$ and apartness of $e$ from $0$.

Convert those two apartness witnesses through $\alpha_h$. The left conversion
gives
$p_{abcd}:\mathsf{NonZero}_F(\mathsf{mul}(\mathsf{mul}(\mathsf{mul}(a,b),c),d))$,
and the right conversion gives $\mathsf{NonZero}_F(e)$.

Apply \autoref{thm:field-four-factor-product-nonzero-iff-factors} to
$p_{abcd}$. It supplies the four factor witnesses for $a$, $b$, $c$, and $d$.
Pair these four witnesses with the converted witness for $e$ in the displayed
order.

For the reverse direction, assume witnesses
$p_a:\mathsf{NonZero}_F(a)$, $p_b:\mathsf{NonZero}_F(b)$,
$p_c:\mathsf{NonZero}_F(c)$, $p_d:\mathsf{NonZero}_F(d)$, and
$p_e:\mathsf{NonZero}_F(e)$. Apply
\autoref{thm:field-four-factor-product-nonzero-iff-factors} in the reverse
direction to obtain
$p_{abcd}:\mathsf{NonZero}_F(\mathsf{mul}(\mathsf{mul}(\mathsf{mul}(a,b),c),d))$.

Apply \autoref{thm:field-product-nonzero-of-nonzero-factors} to $p_{abcd}$ and
$p_e$. The resulting witness is exactly the nonzero certificate for the
left-nested fivefold product, completing the equivalence.
\end{proof}

Per-object template: traditional target RatUp FieldUp five-factor nonzero
factor ledger; BEDC source $\FieldUp(\mathsf{RatUp})$ over $\mathsf{BHist}$
with product nonzero exclusion and product nonzero closure; carrier
$\mathsf{BHist}$; classifier $\hsame$ with $\mathsf{NonZero}_F$; stability
obligations four-factor exactness, binary product closure, zero-factor
exclusion, and apartness-to-nonzero transport; ledger factor witnesses
$p_a,p_b,p_c,p_d,p_e$, prefix witness $p_{abcd}$, and fivefold product witness
$p_{abcde}$; core theorem
\autoref{thm:ratup-fieldup-five-factor-nonzero-factor-ledger}; exactness
predicate nonzero of the left-nested fivefold product equivalent to the five
factor nonzero witnesses; standard bridge deferred; status
$\mathsf{ProofTarget}$.

\begin{theorem}[RatUp FieldUp singleton-empty operation context normalization]
\label{thm:field-singleton-empty-operation-context-normalization}

$\mathsf{fieldSingletonEmptyCarrier}(L)$ and
$\mathsf{fieldSingletonEmptyCarrier}(R)$, the lower-case singleton-empty
$\FieldUp$ operation slots are contextual $\mathsf{BHist}$ endpoints:
$$
\mathsf{fieldSingletonEmptyClassifier}
  (\mathsf{append}(L,\mathsf{fieldSingletonEmptyMul}(h,k)),
   \mathsf{append}(R,out))
\Longleftrightarrow
\mathsf{fieldSingletonEmptyCarrier}(out).
$$
The same contextual normalization holds for the terminal unit endpoint:
$$
\mathsf{fieldSingletonEmptyClassifier}
  (\mathsf{append}(L,\mathsf{fieldSingletonEmptyOne}),
   \mathsf{append}(R,out))
\Longleftrightarrow
\mathsf{fieldSingletonEmptyCarrier}(out).
$$
Finally, for every proof witness
$p:\mathsf{fieldSingletonEmptyNonZero}(h)$, the inverse endpoint satisfies
$$
\mathsf{fieldSingletonEmptyClassifier}
  (\mathsf{append}(L,\mathsf{fieldSingletonEmptyInv}(h,p)),
   \mathsf{append}(R,out))
\Longleftrightarrow
\mathsf{fieldSingletonEmptyCarrier}(out).
$$
\end{theorem}
\begin{proof}
Unfold the three lower-case operation endpoints. In the singleton-empty
$\FieldUp$ construction, $\mathsf{fieldSingletonEmptyMul}(h,k)$,
$\mathsf{fieldSingletonEmptyOne}$, and
$\mathsf{fieldSingletonEmptyInv}(h,p)$ are all the endpoint $\emp$.

For the forward direction in each displayed equivalence, use the two context
carrier witnesses for $L$ and $R$. Empty-result inversion for append removes
the left endpoint $\mathsf{append}(L,\emp)$ and then removes the right context
from $\mathsf{append}(R,out)$, leaving precisely
$\mathsf{fieldSingletonEmptyCarrier}(out)$.

For the reverse direction, begin with the displayed output carrier. Append-empty
reconstruction places the output behind the carried context $R$, and the left
context witness reconstructs the empty operation endpoint behind $L$.
Classifier transitivity through $\emp$ gives the required singleton-empty
classifier in each of the multiplication, unit, and proof-indexed inverse
cases.
\end{proof}

\begin{theorem}[RatUp FieldUp right-nested threefold contextual boundary]
\label{thm:ratup-fieldup-right-nested-threefold-contextual-boundary}
For the concrete denominator tuple $(C_D^u,R_D^u,\mu_D^u,e_D^u,N_D^u)$ and
histories $h,k,\ell:\mathsf{BHist}$ with $C_D^u(h)$, $C_D^u(k)$, and
$C_D^u(\ell)$, let $y:=\mu_D^u(h,\mu_D^u(k,\ell))$. If
$\mathsf{ThreeSupp}_D^u(h,k,\ell)$ holds, then the right-nested
$\FieldUp(\mathsf{RatUp})$ denominator endpoint carries the following
contextual boundary:
\begin{enumerate}
  \item the right-nested endpoint rows $C_D^u(y)$,
  $\mathsf{RatHistoryCarrier}(y)$, and
  $\mathsf{RatHistoryClassifier}(y,y)$;
  \item for every context history $r:\mathsf{BHist}$,
  $\mathsf{fieldSingletonEmptyCarrier}(\mathsf{append}(r,y))\to\bot$;
  \item for every context history $r:\mathsf{BHist}$,
  $\mathsf{fieldSingletonEmptyCarrier}(\mathsf{append}(y,r))\to\bot$;
  \item for every context history $r:\mathsf{BHist}$,
  $\mathsf{fieldSingletonEmptyNonZero}(\mathsf{append}(r,y))$;
  \item for every context history $r:\mathsf{BHist}$,
  $\mathsf{fieldSingletonEmptyNonZero}(\mathsf{append}(y,r))$;
  \item for all $L,R:\mathsf{BHist}$,
  $$
  \mathsf{fieldSingletonEmptyClassifier}(\mathsf{append}(L,y),\mathsf{append}(R,y))
  \to\bot.
  $$
\end{enumerate}
\end{theorem}
\begin{proof}
Apply
\autoref{thm:ratup-fieldup-right-nested-threefold-support-non-singleton-exactness}
to the same concrete denominator tuple, carried histories, and support witness.
It supplies $C_D^u(y)$ and, under
$\mathsf{ThreeSupp}_D^u(h,k,\ell)$, recovers
$\mathsf{RatHistoryCarrier}(y)$ and the singleton rejection map
$\mathsf{fieldSingletonEmptyCarrier}(y)\to\bot$.

The recovered Rat carrier witness on both endpoints, together with reflexivity
of $\hsame$, packages $\mathsf{RatHistoryClassifier}(y,y)$ as the endpoint
classifier row in the theorem statement.

Fix a context history $r:\mathsf{BHist}$. Apply
\autoref{thm:field-singleton-empty-append-rat-carrier-absurd} with context $r$
and Rat carrier endpoint $y$ to reject
$\mathsf{fieldSingletonEmptyCarrier}(\mathsf{append}(r,y))$. Apply
\autoref{thm:field-singleton-empty-append-left-rat-carrier-absurd} with the same
two histories in the opposite append order to reject
$\mathsf{fieldSingletonEmptyCarrier}(\mathsf{append}(y,r))$.

For the right-append inverse-domain witness, unfold
$\mathsf{fieldSingletonEmptyNonZero}(\mathsf{append}(r,y))$ as rejection of the
singleton classifier against $\emp$. Any such classifier witness contains the
source-carrier component

\begin{theorem}[RatUp FieldUp bracketing-independent threefold contextual boundary]

  \item for every context history $r:\mathsf{BHist}$, the four contextual

  \begin{enumerate}

    \item $\mathsf{fieldSingletonEmptyCarrier}(\mathsf{append}(x,r))\to\bot$;
    \item $\mathsf{fieldSingletonEmptyCarrier}(\mathsf{append}(r,y))\to\bot$;
    \item $\mathsf{fieldSingletonEmptyCarrier}(\mathsf{append}(y,r))\to\bot$.
  \end{enumerate}
  \item for every context history $r:\mathsf{BHist}$, the four inverse-domain
  witnesses:
  \begin{enumerate}
    \item $\mathsf{fieldSingletonEmptyNonZero}(\mathsf{append}(r,x))$;
    \item $\mathsf{fieldSingletonEmptyNonZero}(\mathsf{append}(x,r))$;
    \item $\mathsf{fieldSingletonEmptyNonZero}(\mathsf{append}(r,y))$;
    \item $\mathsf{fieldSingletonEmptyNonZero}(\mathsf{append}(y,r))$.
  \end{enumerate}
  \item for all $L,R:\mathsf{BHist}$, the two contextual
  singleton-classifier exclusions
  $$
  \mathsf{fieldSingletonEmptyClassifier}(\mathsf{append}(L,x),\mathsf{append}(R,x))
  \to\bot
  $$
  and
  $$
  \mathsf{fieldSingletonEmptyClassifier}(\mathsf{append}(L,y),\mathsf{append}(R,y))
  \to\bot.
  $$
\end{enumerate}
\end{theorem}
\begin{proof}

\autoref{thm:ratup-fieldup-threefold-support-non-singleton-exactness} to the
carried histories $h,k,\ell$. This supplies the left-nested carrier row

\autoref{thm:ratup-fieldup-right-nested-threefold-support-non-singleton-exactness}

$R_D^u(x,y)$, the strict-support comparison for $y$, the Rat carrier witness for

\autoref{thm:ratup-fieldup-threefold-support-contextual-nonzero-package}. This
packages the two corresponding inverse-domain witnesses for
$\mathsf{append}(r,x)$ and $\mathsf{append}(x,r)$.

\autoref{thm:ratup-fieldup-right-nested-threefold-contextual-boundary}. Its

two inverse-domain witnesses for the right-nested bracketing, and the
contextual singleton-classifier exclusion for arbitrary left and right contexts

left-nested contextual separation package already contains the arbitrary
$L,R:\mathsf{BHist}$ singleton-classifier rejection, so the two endpoint
classifier exclusions stand side by side with the carrier and inverse-domain
