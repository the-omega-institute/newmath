\begin{theorem}[Field singleton-empty operation context nonzero obstruction]
\label{thm:field-singleton-empty-operation-context-nonzero-obstruction}

$\mathsf{fieldSingletonEmptyCarrier}(L)$,
$\mathsf{fieldSingletonEmptyCarrier}(R)$, and
$\mathsf{fieldSingletonEmptyNonZero}(out)$, singleton-empty operation contexts
reject proof-indexed nonzero outputs. The multiplication endpoint satisfies
$$
\mathsf{fieldSingletonEmptyClassifier}
  (\mathsf{append}(L,\mathsf{fieldSingletonEmptyMul}(h,k)),
   \mathsf{append}(R,out))
\Longrightarrow
\mathsf{False}.
$$
The unit endpoint satisfies
$$
\mathsf{fieldSingletonEmptyClassifier}
  (\mathsf{append}(L,\mathsf{fieldSingletonEmptyOne}),
   \mathsf{append}(R,out))
\Longrightarrow
\mathsf{False}.
$$
For every proof witness
$p:\mathsf{fieldSingletonEmptyNonZero}(h)$, the inverse endpoint satisfies
$$
\mathsf{fieldSingletonEmptyClassifier}
  (\mathsf{append}(L,\mathsf{fieldSingletonEmptyInv}(h,p)),
   \mathsf{append}(R,out))
\Longrightarrow
\mathsf{False}.
$$
\end{theorem}
\begin{proof}
Apply \autoref{thm:field-singleton-empty-operation-context-normalization} to
the multiplication, unit, or proof-indexed inverse classifier in the displayed
context. In each case the contextual classifier into $\mathsf{append}(R,out)$
is converted into $\mathsf{fieldSingletonEmptyCarrier}(out)$.

The carrier assertion is the empty endpoint input for
\autoref{thm:field-singleton-nonzero-empty-endpoint-absurd}. Combining it with
the assumed witness $\mathsf{fieldSingletonEmptyNonZero}(out)$ gives
$\mathsf{False}$ for each of the three operation contexts.
\end{proof}

\section{Apart-zero continuation exactness}
\label{sec:field-apartzero-continuation-exactness}

\begin{theorem}[Field apart-zero nested continuation empty-context exactness]
\label{thm:field-apartzero-nested-continuation-empty-context-iff}
For histories $l,h,r,u,v:\mathsf{BHist}$ with $\ContR(l,h,u)$ and
$\ContR(u,r,v)$, if $l\hsame\emp$ and $r\hsame\emp$, then
$$
\mathsf{ApartZero}_F(v)
\Longleftrightarrow
\mathsf{ApartZero}_F(h).
$$
\end{theorem}
\begin{proof}
The two continuation witnesses identify $u$ with $\mathsf{append}(l,h)$ and
$v$ with $\mathsf{append}(u,r)$. Continuation associativity and result
transport compare $v$ with the nested append endpoint
$\mathsf{append}(l,\mathsf{append}(h,r))$.

Transport $\mathsf{ApartZero}_F$ across this $\hsame$ comparison. The result
is exactly the bilateral empty-context statement of
\autoref{thm:field-apartzero-empty-context-iff}, applied with the witnesses
$l\hsame\emp$ and $r\hsame\emp$.
\end{proof}

\begin{theorem}[Field apart-zero nested continuation factor split]
\label{thm:field-apartzero-nested-continuation-factor-split}
For histories $l,h,r,u,v,q,w:\mathsf{BHist}$ with $\ContR(l,h,u)$,
$\ContR(u,r,v)$, and $\ContR(v,q,w)$, if $l\hsame\emp$ and $r\hsame\emp$,
then
$$
\mathsf{ApartZero}_F(w)
\Longleftrightarrow
\mathsf{ApartZero}_F(h)\lor\mathsf{ApartZero}_F(q).
$$
\end{theorem}
\begin{proof}
Apply \autoref{thm:field-apartzero-continuation-endpoint-split} to the tail
continuation witness $\ContR(v,q,w)$. This gives
$\mathsf{ApartZero}_F(w)$ exactly when
$\mathsf{ApartZero}_F(v)\lor\mathsf{ApartZero}_F(q)$ holds.

Apply
\autoref{thm:field-apartzero-nested-continuation-empty-context-iff} to the
first two continuation witnesses and the exterior empty-context witnesses
$l\hsame\emp$ and $r\hsame\emp$. Substituting this equivalence for the first
disjunct yields the displayed factor split.
\end{proof}

Per-object template: traditional target continuation-tail apartness factor
split under empty exterior context; BEDC source $\mathsf{BHist}$, $\ContR$,
and $\mathsf{ApartZero}_F$; carrier apart-zero histories; classifier $\hsame$
transport; stability obligations nested empty-context exactness plus
continuation endpoint split; ledger the two empty-context witnesses and tail
$q$; exactness predicate the displayed iff; standard bridge ordinary
product/context apartness factorization; status $\mathsf{ProofTarget}$.
