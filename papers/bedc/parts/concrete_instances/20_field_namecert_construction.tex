\begin{theorem}[Rat denominator continuation common-context cancellation]
\label{thm:field-rat-denominator-continuation-common-context-cancellation}
For all histories $p,q,d,e,pd,pe,left,right:\mathsf{BHist}$, if
$\mathsf{UnaryHistory}(p)$, $\mathsf{UnaryHistory}(q)$,
$\mathsf{RatHistoryCarrier}(d)$, $\mathsf{RatHistoryCarrier}(e)$,
$\Cont(p,d,pd)$, $\Cont(p,e,pe)$, $\Cont(pd,q,left)$,
$\Cont(pe,q,right)$, and $\mathsf{RatHistoryClassifier}(left,right)$, then
$\mathsf{RatHistoryClassifier}(d,e)$.
\end{theorem}
\begin{proof}
Unpack the final Rat classifier. Its comparison component gives
$left\hsame right$.

Apply the continuation common-context cancellation theorem
\texttt{BEDC.FKernel.Cont.cont\_cancel\_common\_context} to the two matched
prefix witnesses, the two matched suffix witnesses, and the comparison
$left\hsame right$. The shared prefix $p$ and shared suffix $q$ cancel, leaving
the endpoint comparison $d\hsame e$.

Combine this endpoint comparison with the original denominator carriers
$\mathsf{RatHistoryCarrier}(d)$ and $\mathsf{RatHistoryCarrier}(e)$. These
three pieces are exactly the fields of
$\mathsf{RatHistoryClassifier}(d,e)$.
\end{proof}

\begin{theorem}[Rat denominator continuation common-context classifier exactness]
\label{thm:field-rat-denominator-continuation-common-context-classifier-exactness}
For all histories $p,q,d,e,pd,pe,left,right:\mathsf{BHist}$, if
$\mathsf{UnaryHistory}(p)$, $\mathsf{UnaryHistory}(q)$,
$\mathsf{RatHistoryCarrier}(d)$, $\mathsf{RatHistoryCarrier}(e)$,
$\Cont(p,d,pd)$, $\Cont(p,e,pe)$, $\Cont(pd,q,left)$, and
$\Cont(pe,q,right)$, then
$$
\mathsf{RatHistoryClassifier}(d,e)\Longleftrightarrow
\mathsf{RatHistoryClassifier}(left,right).
$$
\end{theorem}
\begin{proof}
For the forward direction, apply the Rat unary denominator context closure
theorem to $\mathsf{RatHistoryClassifier}(d,e)$ with the common unary prefix
$p$ and common unary suffix $q$. This gives the classifier for the canonical
history endpoints obtained by placing $p$ and $q$ around $d$ and $e$.

Continuation determinacy compares those canonical endpoints with the displayed
endpoints $left$ and $right$, using the premises $\Cont(p,d,pd)$,
$\Cont(p,e,pe)$, $\Cont(pd,q,left)$, and $\Cont(pe,q,right)$. Transporting the
classifier across these endpoint comparisons gives
$\mathsf{RatHistoryClassifier}(left,right)$.

For the reverse direction, apply
\autoref{thm:field-rat-denominator-continuation-common-context-cancellation}
to the same unary-context and continuation premises. It cancels the common
prefix and suffix context from $\mathsf{RatHistoryClassifier}(left,right)$ and
returns $\mathsf{RatHistoryClassifier}(d,e)$.
\end{proof}

denominators; \textbf{classifier choice}: Rat history classifier;
\textbf{stability obligation}: common-context classifier exactness; \textbf{ledger}:
matched unary prefix and suffix contexts; \textbf{core theorem}:

\textbf{exactness predicate}: classifier equivalence under matched continuation
contexts; \textbf{standard bridge}: deferred; \textbf{status}:

\begin{theorem}[Rat denominator continuation is semigroup-only on the positive carrier]
\label{thm:field-rat-denominator-continuation-semigroup-only}
For the carrier $\mathsf{RatHistoryCarrier}$ and continuation relation $\Cont$,
Theorems~\ref{thm:field-rat-denominator-continuation-carrier-closure} and
\ref{thm:field-rat-denominator-continuation-semigroup-laws} provide carrier
closure and classifier associativity on positive Rat denominators. The empty
history is not an internal carrier endpoint: $\mathsf{RatHistoryCarrier}(\emp)$
implies $\mathsf{False}$. Consequently the endpoint laws in
Theorem~\ref{thm:field-rat-denominator-continuation-empty-identity-and-assoc}
are external empty-history endpoint laws. They do not provide an internal
$\MonoidUp$ unit or an internal $\FieldUp$ unit on the Rat denominator carrier.
\end{theorem}
\begin{proof}
The semigroup clauses are exactly the content of the displayed closure and
semigroup-law theorems. Carrier closure keeps every realized continuation result
inside $\mathsf{RatHistoryCarrier}$ when both inputs are positive Rat
denominators, and the semigroup law identifies the two continuation bracketings
by the Rat history classifier.

For the unit boundary, suppose $\mathsf{RatHistoryCarrier}(\emp)$. Reflexivity
of $\hsame$ at $\emp$ gives $\emp\hsame\emp$. Applying
Theorem~\ref{thm:rat-history-carrier-not-empty} to the assumed carrier witness
and this reflexive endpoint equality yields $\mathsf{False}$. Thus the
empty-history clauses in
Theorem~\ref{thm:field-rat-denominator-continuation-empty-identity-and-assoc}
can be used only as external endpoint laws, not as an internal carrier unit.
\end{proof}
