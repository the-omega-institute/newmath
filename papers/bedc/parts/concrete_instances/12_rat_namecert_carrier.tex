\chapter{A Concrete Naming Certificate for \(\RatUp\)}
\label{ch:concrete-instances-rat-namecert}

The rational number interface $\RatUp$ starts from a carrier skeleton: a signed unary integer numerator, drawn from \autoref{ch:concrete-instances-int-namecert}, together with a positive unary denominator. Normalized representatives, coprimality, and arithmetic closure are separate source, pattern, classifier, and stability obligations later in the chapter. The classifier is structural equality on the normalized representative, not the equivalence relation $p_1 q_2 = p_2 q_1$ on unnormalized pairs, which would otherwise demand quotient-type machinery and $\mathsf{Quot.sound}$. It is not a primitive field; it is licensed only via the integer source and the normalization function.

\begin{remark}[Rational certificate scope]
\label{rem:concrete-instances-rat-namecert-roadmap}
This chapter records the rational licensing target through normalized
integer-positive-natural representatives and structural classifier data.
The checked core covers the rational history carrier, rational history classifier fields, ledger carrier, and semantic certificate. Normalized representatives are recorded by the source and pattern obligations, while coprimality and operation closure remain part of the paper specification.
\end{remark}

\section{Carrier definition}

\begin{definition}[Rat carrier]
\label{def:rat-carrier}
Use only the carrier skeleton: a signed unary integer numerator together with a unary denominator history that is not internally the same as $\emp$:
$$
  \mathsf{RatCarrier}(s,n,d)
  :=
  \mathsf{IntCarrier}(s,n)\land \mathsf{Unary}(d)\land \neg\hsame(d,\emp).
$$
Equivalently, this is an $\IntUp$ numerator with a positive unary-history denominator. At this skeleton layer, positivity is witnessed by writing the denominator as a one-extension of a unary tail; normalized representatives, coprimality, and arithmetic closure are supplied by later source, pattern, and stability obligations.
\end{definition}
\leandef{BEDC.Derived.RatUp.RatCarrier}

\begin{theorem}[Rat carrier positive denominator]
\label{thm:rat-carrier-positive-denominator}
If \(\mathsf{RatCarrier}(s,n,d)\), then the denominator is internally the same as \(\Eone(t)\) for some unary tail \(t\).
\end{theorem}
\leanchecked{BEDC.Derived.RatUp.RatCarrier\_positive\_denominator}
\begin{proof}
Unpack the carrier into its integer numerator, unary denominator, and non-emptiness clauses. The unary denominator is either \(\emp\) or a one-extension of a unary tail. The empty case contradicts non-emptiness, while the one-extension case gives the displayed positive-denominator witness.
\end{proof}

\begin{theorem}[Positive denominator yields a rational carrier]
\label{thm:rat-carrier-from-positive-denominator}
An \(\IntUp\) numerator together with a denominator internally same as
\(\Eone(t)\) for a unary tail \(t\) forms a \(\RatUp\) carrier.
\end{theorem}
\leanchecked{BEDC.Derived.RatUp.RatCarrier\_of\_int\_positive\_denominator}
\begin{proof}
Transport unary closure from the displayed \(\Eone\)-shape to the denominator
along \(\hsame\). Non-emptiness follows because an \(\Eone\)-history cannot be
internally same as \(\emp\).
\end{proof}

\begin{theorem}[Rat carrier denominator transport]
\label{thm:rat-carrier-denominator-hsame-transport}
If \(\mathsf{RatCarrier}(s,n,d)\) and \(d\hsame d'\), then
\(\mathsf{RatCarrier}(s,n,d')\).
\end{theorem}
\leanchecked{BEDC.Derived.RatUp.RatCarrier\_denominator\_hsame\_transport}
\begin{proof}
Extract the positive denominator witness from the rational carrier, transport
that witness across denominator sameness, and rebuild the rational carrier from
the unchanged integer numerator together with the transported positive
denominator.
\end{proof}

\begin{theorem}[Positive unary denominator transport]
\label{thm:rat-positive-unary-denominator-hsame-transport}
If a denominator history has a positive unary witness and is \(\hsame\)-equivalent to another history, then the other history has the same positive unary denominator witness.
\end{theorem}
\begin{proof}
Write the positive denominator as an \(\Eone\)-extension of a unary tail. Compose the reverse \(\hsame\) witness from the target history to the source history with the source witness to the \(\Eone\)-extension, and keep the unary tail witness.
\end{proof}
\leanchecked{BEDC.Derived.RatUp.PositiveUnaryDenominator\_hsame\_transport}

\begin{theorem}[One-extension positive denominator equivalence]
\label{thm:rat-positive-unary-denominator-eone-iff-unary}
A one-extension history has a positive unary denominator witness exactly when
its tail is unary.
\end{theorem}
\begin{proof}
For the forward direction, eliminate the positive-denominator witness
\(\hsame(\Eone(t),\Eone(u))\) with unary tail \(u\). One-constructor inversion
gives \(\hsame(t,u)\), and unary membership transports back to \(t\). For the
reverse direction, use \(t\) itself as the displayed positive tail and the
reflexive one-extension sameness witness.
\end{proof}
\leanchecked{BEDC.Derived.RatUp.PositiveUnaryDenominator\_e1\_iff\_unary}

\begin{theorem}[Zero-extension cannot be a positive denominator]
\label{thm:rat-positive-denominator-e0-absurd}
A zero-extension history has no positive unary denominator witness.
\end{theorem}
\begin{proof}
Unpack the positive denominator witness. It would identify an
\(\Ezero\)-history with an \(\Eone\)-history, contradicting finite-kernel
history constructor no-confusion.
\end{proof}
\leanchecked{BEDC.Derived.RatUp.PositiveUnaryDenominator\_e0\_absurd}

\begin{theorem}[Positive unary denominator is not empty]
\label{thm:rat-positive-unary-denominator-not-empty}
If a denominator has a positive unary witness, then it is not internally the
same as \(\emp\).
\end{theorem}
\begin{proof}
Unpack the positive denominator as internally the same as \(\Eone(t)\) for a
unary tail \(t\). Composing this witness with an assumed sameness to \(\emp\)
would make an \(\Eone\)-history internally the same as \(\emp\), contradicting
history constructor no-confusion.
\end{proof}
\leanchecked{BEDC.Derived.RatUp.PositiveUnaryDenominator\_not\_empty}

\begin{theorem}[Positive unary denominator is unary and nonempty]
\label{thm:positive-unary-denominator-unary-and-nonempty}
If a denominator history has a positive unary witness, then it is unary and cannot be internally the same as \(\emp\).
\end{theorem}
\leanchecked{BEDC.Derived.RatUp.PositiveUnaryDenominator\_unary\_and\_nonempty}
\begin{proof}
Write the positive denominator as internally the same as \(\Eone(t)\) with \(t\) unary. Unary closure under \(\Eone\), transported back along history sameness, gives unary membership of the denominator. If the denominator were internally same as \(\emp\), transitivity would make \(\Eone(t)\) internally same as \(\emp\), contradicting constructor separation.
\end{proof}

\begin{theorem}[Positive unary denominator carrier equivalence]
\label{thm:rat-carrier-iff-positive-unary-denominator}
The rational carrier condition is equivalent to an $\IntUp$ numerator together with a denominator that has a positive unary tail witness.
\end{theorem}
\begin{proof}
Unpack the carrier. A unary denominator that is not $\hsame$-empty cannot be the empty unary history, so unary case analysis gives a one-extension tail. Conversely, a denominator $\hsame$-equal to a one-extension of a unary tail is unary by transport, and it cannot be $\hsame$-empty by constructor no-confusion.
\end{proof}
\leanchecked{BEDC.Derived.RatUp.RatCarrier\_iff\_positive\_unary\_denominator}

\begin{theorem}[Positive denominator append unary tail]
\label{thm:rat-positive-denominator-append-unary-tail}
If a denominator has a positive unary witness and the appended tail is unary, then the continued denominator still has a positive unary witness.
\end{theorem}
\begin{proof}
Write the denominator as internally the same as \(\Eone(t)\) for a unary tail \(t\), and transport unary membership back to the denominator itself. The appended tail is either \(\emp\) or a one-step unary tail. The empty case keeps the original witness, while the one-step case takes the continued denominator without that final step as the new positive tail; closure follows from unary append closure.
\end{proof}
\leanchecked{BEDC.Derived.RatUp.PositiveUnaryDenominator\_append\_unary\_tail}

\begin{theorem}[Positive denominator append unary prefix]
\label{thm:rat-positive-denominator-append-unary-prefix}

\end{theorem}
\begin{proof}

internally the same as $\Eone(t)$ with $t$ unary. Appending the unary prefix

positive witness for $\mathsf{append}(u,d)$.
\end{proof}
\leanchecked{BEDC.Derived.RatUp.PositiveUnaryDenominator\_append\_unary\_prefix}

\begin{theorem}[Positive unary denominator append unary left iff]
\label{thm:positive-unary-denominator-append-unary-left-iff}
If $u$ is unary, then a continued denominator
$\mathsf{append}(u,r)$ has a positive unary denominator witness exactly when
either $r$ is internally empty and $u$ already has such a witness, or $r$
itself has such a witness.
\end{theorem}
\begin{proof}
Case-split on the right history $r$. The empty case reduces the continuation
to $u$. A zero-extension case is impossible because no zero-extension is a
positive denominator. In the one-extension case, one-constructor readback gives
unary membership of the continued tail, and unary append factorization recovers
the right tail witness. Conversely, the empty-right alternative reduces by
endpoint equality, while a positive witness on $r$ is preserved by unary
prefix closure.
\end{proof}
\leanchecked{BEDC.Derived.RatUp.PositiveUnaryDenominator\_append\_unary\_left\_iff}

\begin{theorem}[Positive unary denominator append unary left split]
\label{thm:positive-unary-denominator-append-unary-left-split}
If $p$ is unary and $\mathsf{append}(p,d)$ has a positive unary denominator
witness, then either $p$ already has a positive unary denominator witness or
$d$ has one.
\end{theorem}
\begin{proof}
Split on the visible shape of the unary prefix. The empty prefix transports the
positive witness across left-unit append. A zero-headed prefix contradicts unary
history membership. A one-headed prefix is itself a positive unary denominator.
\end{proof}
\leanchecked{BEDC.Derived.RatUp.PositiveUnaryDenominator\_append\_unary\_left\_split}

\begin{theorem}[Rat carrier hsame transport]
\label{thm:rat-carrier-hsame-transport}
If a rational carrier \(\mathsf{RatCarrier}(s,n,d)\) is transported along
\(\msame(s,t)\), \(\hsame(n,n')\), and \(\hsame(d,d')\), then the transported
triple is again a rational carrier.
\end{theorem}
\leanchecked{BEDC.Derived.RatUp.RatCarrier\_hsame\_transport}
\begin{proof}
Eliminate the sign sameness, then unpack the rational carrier. The numerator
is transported by the integer carrier transport theorem, denominator unary
closure is transported along \(\hsame(d,d')\), and denominator non-emptiness is
preserved by composing \(\hsame(d,d')\) with any empty-target witness.
\end{proof}
