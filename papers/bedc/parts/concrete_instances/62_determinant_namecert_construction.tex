\chapter{A Concrete Naming Certificate for $\DeterminantUp$}
\label{ch:concrete-instances-determinant-namecert}

The determinant interface $\DeterminantUp$ packages the alternating multilinear form on square matrices that detects invertibility. It depends on $\NameCert_{\MatrixUp}$ (\autoref{ch:concrete-instances-matrix-namecert}) and $\NameCert_{\CommRingUp}$ (\autoref{ch:concrete-instances-commring-namecert}), and supplies the invariant used by the eigenvalue and characteristic-polynomial certificates.

\section{Carrier definition}
\label{sec:determinant-carrier}

\begin{definition}[Singleton matrix determinant certificate]
\label{def:singleton-matrix-determinant-certificate}
Use the singleton empty-history matrix carrier and classifier from
$\MatrixUp$,
$$
\mathsf{MatrixSingletonCarrier},\qquad
\mathsf{MatrixSingletonClassifier},
$$
together with the singleton empty-history commutative-ring carrier and
classifier from $\CommRingUp$,
$$
\mathsf{CommRingSingletonCarrier},\qquad
\mathsf{CommRingSingletonClassifier}.
$$
The determinant readback is the constant scalar endpoint
$$
\det_{\emp}(M):=\emp.
$$
Let $\mathsf{MatrixSingletonOne}$ denote the singleton matrix identity and
$\mathsf{CommRingSingletonOne}$ the singleton commutative-ring unit. Define
the endpoint relation by
$$
\Delta_{\emp}(M,r)
\Longleftrightarrow
\mathsf{MatrixSingletonClassifier}(M,\mathsf{MatrixSingletonOne})
\land
\mathsf{CommRingSingletonClassifier}(r,\mathsf{CommRingSingletonOne}).
$$
The certificate fit has ten fields:
\begin{enumerate}
  \item Traditional target: the terminal determinant on the one-by-one
  terminal matrix algebra.
  \item BEDC source: the singleton $\MatrixUp$ instance over the singleton
  $\CommRingUp$ scalar instance.
  \item Carrier choice: $\mathsf{MatrixSingletonCarrier}$.
  \item Classifier choice: $\mathsf{MatrixSingletonClassifier}$, with scalar
  readback classified by $\mathsf{CommRingSingletonClassifier}$.
  \item Stability obligations: identity stability, classifier transport,
  multiplicativity, and row-swap sign collapse at the singleton endpoint.
  \item Ledger: the singleton matrix endpoint together with the singleton
  scalar endpoint.
  \item Core theorem list:
  Theorem~\ref{thm:singleton-matrix-determinant-endpoint-correspondence},
  Theorem~\ref{thm:singleton-empty-history-matrix-laws}, and
  Theorem~\ref{thm:singleton-empty-history-commring-laws}.
  \item Exactness predicate: $\Delta_{\emp}(M,r)$.
  \item Standard bridge: the determinant of the terminal one-by-one matrix
  algebra.
  \item Status: $\mathsf{ProofTarget}$.
\end{enumerate}
\end{definition}

\begin{theorem}[Singleton matrix determinant endpoint correspondence]
\label{thm:singleton-matrix-determinant-endpoint-correspondence}
For histories $M,r:\mathsf{BHist}$, the determinant endpoint relation satisfies
$$
\Delta_{\emp}(M,r)
\Longleftrightarrow
\mathsf{MatrixSingletonCarrier}(M)
\land
\mathsf{CommRingSingletonClassifier}(r,\mathsf{CommRingSingletonOne}).
$$
Moreover, for singleton-carried histories $M,N:\mathsf{BHist}$:
\begin{enumerate}
  \item $\det_{\emp}(\mathsf{MatrixSingletonOne})$ is classified with
  $\mathsf{CommRingSingletonOne}$ by
  $\mathsf{CommRingSingletonClassifier}$.
  \item If $\mathsf{MatrixSingletonClassifier}(M,N)$, then
  $\mathsf{CommRingSingletonClassifier}(\det_{\emp}(M),\det_{\emp}(N))$.
  \item $\mathsf{CommRingSingletonClassifier}
  (\det_{\emp}(\mathsf{MatrixSingletonMul}(M,N)),
  \mathsf{CommRingSingletonMul}(\det_{\emp}(M),\det_{\emp}(N)))$.
\end{enumerate}
\end{theorem}
\begin{proof}
For the forward direction of the displayed equivalence, unfold
$\Delta_{\emp}$. The matrix component classifies $M$ with
$\mathsf{MatrixSingletonOne}$, so its carrier field gives
$\mathsf{MatrixSingletonCarrier}(M)$. The scalar component is already the
displayed $\mathsf{CommRingSingletonClassifier}$ witness.

For the reverse direction, apply
Theorem~\ref{thm:singleton-empty-history-matrix-endpoint-collapse} with the
carried matrix $M$ and the carried identity endpoint supplied by
Theorem~\ref{thm:singleton-empty-history-matrix-laws}. Equivalently,
Theorem~\ref{thm:singleton-empty-history-matrix-endpoint-collapse} collapses
the singleton matrix endpoint to $\emp$, and the identity law in
Theorem~\ref{thm:singleton-empty-history-matrix-laws} identifies
$\mathsf{MatrixSingletonOne}$ with the same endpoint. Together with the given
scalar classifier witness, these two components repackage
$\Delta_{\emp}(M,r)$.

For the determinant of the identity, $\det_{\emp}$ is constantly $\emp$ and
$\mathsf{CommRingSingletonOne}$ is the singleton scalar endpoint. The
commutative-ring singleton laws of
Theorem~\ref{thm:singleton-empty-history-commring-laws} classify these two
scalar histories at the same endpoint.

Classifier transport is immediate from the same scalar collapse. If
$\mathsf{MatrixSingletonClassifier}(M,N)$, both determinant readbacks are
$\emp$, so the commutative-ring singleton laws package the scalar comparison.

For multiplicativity, the matrix product endpoint is first collapsed by
Theorem~\ref{thm:singleton-empty-history-matrix-endpoint-collapse}. The two
scalar sides are then $\emp$ and
$\mathsf{CommRingSingletonMul}(\emp,\emp)$; the singleton commutative-ring
laws identify the multiplication result with
$\mathsf{CommRingSingletonOne}$, so the final scalar classifier witness is
again packaged at $\emp$.
\end{proof}

\begin{theorem}[Determinant append-pair scalar classifier]
\label{thm:determinant-singleton-det-append-pair-scalar-classifier}
For histories $M,N,r:\mathsf{BHist}$, the scalar classifier of the singleton
determinant readback on $\mathsf{append}(M,N)$ is equivalent to singleton
matrix carriers for $M$ and $N$ together with the singleton commutative-ring
carrier for $r$.
\end{theorem}
\begin{proof}
Unfold the determinant readback as an append with the empty suffix. In the
forward direction, the scalar classifier supplies emptiness of
$\mathsf{append}(\mathsf{append}(M,N),\emp)$; append-empty inversion first
removes the determinant suffix and then splits the matrix pair into carrier
witnesses for $M$ and $N$. The target scalar carrier is the second endpoint
field of the classifier. Conversely, the two matrix carrier witnesses make
$\mathsf{append}(M,N)$ empty, the determinant suffix keeps it empty, and this
empty determinant endpoint classifies with the carried scalar endpoint $r$.
\end{proof}
\leanchecked{BEDC.Derived.DeterminantUp.DeterminantSingletonDet\_append\_pair\_scalar\_classifier\_iff}

\begin{theorem}[Determinant visible matrix scalar classifier exclusion]
\label{thm:determinant-singleton-det-visible-matrix-scalar-classifier-absurd}
For histories $M,r:\mathsf{BHist}$, neither visible matrix endpoint can feed
the singleton determinant readback into a scalar classifier:
$$
\mathsf{CommRingSingletonClassifier}
(\mathsf{DeterminantSingletonDet}(\Ezero(M)),r)\Longrightarrow\bot,
\qquad
\mathsf{CommRingSingletonClassifier}
(\mathsf{DeterminantSingletonDet}(\Eone(M)),r)\Longrightarrow\bot.
$$
\end{theorem}
\begin{proof}
Unfold the singleton determinant readback as append with the empty suffix and
read the left carrier field of the scalar classifier. The field forces the
visible determinant endpoint to be history-same to $\emp$. Constructor
no-confusion for $\Ezero$ and $\Eone$ contradicts these two possible carrier
clauses.
\end{proof}
\leanchecked{BEDC.Derived.DeterminantUp.DeterminantSingletonDet\_visible\_matrix\_scalar\_classifier\_absurd}

\begin{theorem}[Determinant singleton classifier semantic name certificate]
\label{thm:determinant-singleton-classifier-semantic-namecert}
The determinant singleton classifier over $\mathsf{MatrixSingletonCarrier}$ is
a semantic name certificate:
$$
\mathsf{SemanticNameCert}(
\mathsf{MatrixSingletonCarrier},
\mathsf{MatrixSingletonCarrier},
\mathsf{MatrixSingletonCarrier},
\mathsf{DeterminantSingletonClassifier}).
$$
\end{theorem}
\begin{proof}
Use the matrix singleton certificate supplied by
Theorem~\ref{thm:singleton-empty-history-matrix-laws} for carrier inhabitation,
reflexivity, symmetry, transitivity, and carrier transport on the matrix
component. The determinant readback is the constant singleton scalar endpoint,
and the commutative-ring singleton certificate supplied by
Theorem~\ref{thm:singleton-empty-history-commring-laws} classifies the scalar
readbacks. Pairing these two classifier components gives the determinant
classifier fields. Pattern and ledger soundness are the identity inclusions
because all three semantic predicates are
$\mathsf{MatrixSingletonCarrier}$.
\end{proof}

\begin{theorem}[Singleton determinant threefold multiplicativity]
\label{thm:determinant-singleton-triple-multiplicativity}
For singleton-carried matrix histories $M,N,P:\mathsf{BHist}$, the
left-bracketed matrix product has the scalar determinant classifier
$$
\mathsf{CommRingSingletonClassifier}\bigl(
\mathsf{DeterminantSingletonDet}(
\mathsf{MatrixSingletonMul}(\mathsf{MatrixSingletonMul}(M,N),P)),
\mathsf{CommRingSingletonMul}(
\mathsf{CommRingSingletonMul}(
\mathsf{DeterminantSingletonDet}(M),
\mathsf{DeterminantSingletonDet}(N)),
\mathsf{DeterminantSingletonDet}(P))\bigr).
$$
The right-bracketed matrix product has the analogous classifier
$$
\mathsf{CommRingSingletonClassifier}\bigl(
\mathsf{DeterminantSingletonDet}(
\mathsf{MatrixSingletonMul}(M,\mathsf{MatrixSingletonMul}(N,P))),
\mathsf{CommRingSingletonMul}(
\mathsf{DeterminantSingletonDet}(M),
\mathsf{CommRingSingletonMul}(
\mathsf{DeterminantSingletonDet}(N),
\mathsf{DeterminantSingletonDet}(P)))\bigr).
$$
Moreover, the two scalar products above are classified with one another by
$\mathsf{CommRingSingletonClassifier}$.
\end{theorem}
\begin{proof}
The singleton matrix laws in
Theorem~\ref{thm:singleton-empty-history-matrix-laws} close
$\mathsf{MatrixSingletonMul}(M,N)$ and then
$\mathsf{MatrixSingletonMul}(\mathsf{MatrixSingletonMul}(M,N),P)$ under
$\mathsf{MatrixSingletonCarrier}$. The same laws also close
$\mathsf{MatrixSingletonMul}(N,P)$ and
$\mathsf{MatrixSingletonMul}(M,\mathsf{MatrixSingletonMul}(N,P))$, and their
associativity field classifies the two matrix products at the singleton
endpoint.

Apply
Theorem~\ref{thm:singleton-matrix-determinant-endpoint-correspondence} to the
left-bracketed product and then use
Theorem~\ref{thm:determinant-singleton-classifier-semantic-namecert} to keep
the determinant readback stable under the singleton matrix classifier. The
result is the displayed classifier from the determinant of
$\mathsf{MatrixSingletonMul}(\mathsf{MatrixSingletonMul}(M,N),P)$ to the
left-bracketed scalar product.

The right-bracketed statement is obtained by the same argument applied to
$\mathsf{MatrixSingletonMul}(M,\mathsf{MatrixSingletonMul}(N,P))$. Matrix
associativity transports the product endpoint, and determinant readback
stability gives the corresponding scalar classifier.

Finally, Theorem~\ref{thm:singleton-empty-history-commring-laws} closes
$\mathsf{CommRingSingletonMul}$ on the three determinant readbacks and
classifies the left- and right-bracketed scalar products by commutative-ring
associativity.
\end{proof}

\begin{definition}[Determinant singleton row-swap endpoint certificate]
\label{def:determinant-singleton-row-swap-endpoint-certificate}
For histories $M,S:\mathsf{BHist}$, write
$$
\mathsf{DetSwap}_{\emp}(M,S)
\Longleftrightarrow
\mathsf{MatrixSingletonCarrier}(M)\land
\mathsf{MatrixSingletonClassifier}(S,M).
$$
This is the row-swap endpoint certificate at the singleton determinant surface:
the endpoint $S$ is read through the same singleton matrix classifier as the
source matrix $M$.
\end{definition}

\begin{theorem}[Determinant singleton row-swap sign collapse]
\label{thm:determinant-singleton-row-swap-sign-collapse}
For histories $M,S:\mathsf{BHist}$, every singleton row-swap endpoint
certificate
$$
\mathsf{DetSwap}_{\emp}(M,S)
$$
supplies the signed determinant classifier
$$
\mathsf{CommRingSingletonClassifier}
(\det_{\emp}(S),\mathsf{CommRingSingletonNeg}(\det_{\emp}(M)))
$$
and hence the unsigned determinant classifier
$$
\mathsf{CommRingSingletonClassifier}(\det_{\emp}(S),\det_{\emp}(M)).
$$
\end{theorem}
\begin{proof}
Unpack $\mathsf{DetSwap}_{\emp}(M,S)$. Its second component is the matrix
classifier $\mathsf{MatrixSingletonClassifier}(S,M)$, and its first component
keeps $M$ in the singleton matrix carrier. Applying the classifier-transport
clause in \autoref{thm:singleton-matrix-determinant-endpoint-correspondence}
to $S$ and $M$ gives
$$
\mathsf{CommRingSingletonClassifier}(\det_{\emp}(S),\det_{\emp}(M)).
$$
Thus both determinant readbacks are singleton scalar endpoints.

Apply \autoref{thm:commring-singleton-neg-empty-endpoint} with
$h=\det_{\emp}(M)$ and $t=\det_{\emp}(S)$. Since $\det_{\emp}(S)$ is a
singleton scalar endpoint and $\mathsf{CommRingSingletonNeg}(\det_{\emp}(M))$
is the empty singleton scalar endpoint, the singleton commutative-ring
classifier relates $\mathsf{CommRingSingletonNeg}(\det_{\emp}(M))$ to
$\det_{\emp}(S)$. Symmetry of the scalar classifier, supplied by
\autoref{thm:singleton-empty-history-commring-laws}, gives the displayed
signed orientation
$$
\mathsf{CommRingSingletonClassifier}
(\det_{\emp}(S),\mathsf{CommRingSingletonNeg}(\det_{\emp}(M))).
$$

Apply the same negation endpoint theorem with $t=\det_{\emp}(M)$ to classify
$\mathsf{CommRingSingletonNeg}(\det_{\emp}(M))$ with $\det_{\emp}(M)$.
Transitivity of the singleton scalar classifier from
\autoref{thm:singleton-empty-history-commring-laws} composes this comparison
with the signed classifier above, yielding
$$
\mathsf{CommRingSingletonClassifier}(\det_{\emp}(S),\det_{\emp}(M)).
$$
\end{proof}
