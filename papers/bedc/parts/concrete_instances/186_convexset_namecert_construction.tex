\chapter{A Concrete Naming Certificate for $\ConvexSetUp$}
\label{ch:concrete-instances-convexset-namecert}

The convex-set interface $\ConvexSetUp$ packages a subset of a vector-space carrier closed under affine combinations with nonnegative coefficients summing to one. It depends on $\NameCert_{\VecSpaceUp}$ (\autoref{ch:concrete-instances-vecspace-namecert}) and $\NameCert_{\AffineSpaceUp}$ (\autoref{ch:concrete-instances-affinespace-namecert}).

\section{The certificate}

\begin{definition}[ConvexSet binary affine-combination row]
\label{def:convexset-binary-affine-combination-row}
For a $\ConvexSetUp$ certificate over a scalar $\FieldUp(F)$ and a vector
carrier from $\VecSpaceUp(F,V)$, write $C$ for the convex carrier,
$\sim_F$ for the scalar classifier, $+_F$ for scalar addition, $1_F$ for the
scalar unit, $\mathsf{NonNeg}_F$ for the nonnegative-scalar predicate, $\odot$
for the scalar action, and $+_V$ for vector addition. The binary
affine-combination row is the certificate field saying that, for all
$a,b:F$ and all carried points $x,y:V$,
$$
C(x)\land C(y)\land \mathsf{NonNeg}_F(a)\land \mathsf{NonNeg}_F(b)
\land (a+_F b)\sim_F 1_F
\Longrightarrow
C\bigl((a\odot x)+_V(b\odot y)\bigr).
$$
This is the two-point instance of the convex closure advertised in the
$\ConvexSetUp$ carrier description, using the scalar and action data supplied
by the $\VecSpaceUp$ and $\AffineSpaceUp$ dependencies.
\end{definition}

\begin{theorem}[ConvexSet midpoint closure from affine-combination row]
\label{thm:convexset-midpoint-affine-combination-closure}
Let a $\ConvexSetUp$ certificate be given with notation as in
\autoref{def:convexset-binary-affine-combination-row}. If $x$ and $y$ are
carried by the convex carrier, $h$ is a nonnegative scalar, and the scalar sum
$h+_F h$ is classified as the scalar unit,
$$
C(x)\land C(y)\land \mathsf{NonNeg}_F(h)\land (h+_F h)\sim_F 1_F,
$$
then the affine midpoint endpoint is carried by the same convex carrier:
$$
C\bigl((h\odot x)+_V(h\odot y)\bigr).
$$
\end{theorem}
\leanchecked{BEDC.Derived.ConvexSetUp.ConvexSetSingletonAffineSpine\_midpoint\_closure}
\begin{proof}
Apply the binary affine-combination row of
\autoref{def:convexset-binary-affine-combination-row} with both scalar
coefficients equal to $h$. The two point-carrier hypotheses supply $C(x)$ and
$C(y)$. The single nonnegative hypothesis supplies both coefficient
nonnegativity clauses, since the row asks separately for
$\mathsf{NonNeg}_F(h)$ on the left coefficient and on the right coefficient.
The displayed classifier hypothesis is exactly the coefficient-sum clause
$(h+_F h)\sim_F 1_F$. The row therefore returns
$C\bigl((h\odot x)+_V(h\odot y)\bigr)$, which is the claimed midpoint carrier
closure.
\end{proof}

\begin{theorem}[ConvexSet affine-combination row order symmetry]
\label{thm:convexset-affine-combination-row-order-symmetric}
Let a $\ConvexSetUp$ certificate be given over a scalar $\FieldUp(F)$ and a
vector carrier from $\VecSpaceUp(F,V)$, with notation as in
\autoref{def:convexset-binary-affine-combination-row}. For scalars $a,b:F$
and points $x,y:V$, assume the scalar additive commutativity row inherited from
the scalar field certificate supplies the comparison
$$
(b+_F a)\sim_F(a+_F b).
$$
If
$$
C(x)\land C(y)\land \mathsf{NonNeg}_F(a)\land \mathsf{NonNeg}_F(b)
\land (a+_F b)\sim_F 1_F,
$$
then the swapped affine endpoint is carried by the same convex carrier:
$$
C\bigl((b\odot y)+_V(a\odot x)\bigr).
$$
\end{theorem}
\begin{proof}
The scalar part of the field certificate retains the additive abelian-group
rows listed through \autoref{def:ring-stability-certificate} and
\autoref{def:abgroup-stability-certificate}, so the displayed scalar additive
commutativity comparison is an available classifier edge for these two
coefficients. The scalar classifier is a naming-certificate classifier, and
\autoref{thm:namecert-equivalence-transitivity} composes this comparison with
the coefficient-sum hypothesis $(a+_F b)\sim_F1_F$. Hence
$$
(b+_F a)\sim_F1_F.
$$
Apply the binary affine-combination row of
\autoref{def:convexset-binary-affine-combination-row} with left coefficient
$b$, right coefficient $a$, left point $y$, and right point $x$. The carrier
hypotheses provide $C(y)$ and $C(x)$ in the required order. The nonnegativity
hypotheses provide $\mathsf{NonNeg}_F(b)$ and $\mathsf{NonNeg}_F(a)$ in the
required order. The transitive scalar comparison above is exactly the reversed
coefficient-sum premise. The row therefore returns
$C\bigl((b\odot y)+_V(a\odot x)\bigr)$.
\end{proof}

\begin{definition}[ConvexSet finite affine-spine data]
\label{def:convexset-finite-affine-spine-data}
Let a $\ConvexSetUp$ certificate be given with notation as in
\autoref{def:convexset-binary-affine-combination-row}. Also write $0_F$ for
the scalar zero and $\cdot_F$ for scalar multiplication. For a finite scalar
spine $\lambda:\mathsf{ListCarrier}(F)$, write $\mathsf{sum}_F(\lambda)$ for
the finite recursive scalar sum
$$
\mathsf{sum}_F(\mathsf{nil})=0_F,
\qquad
\mathsf{sum}_F(\mathsf{cons}(a,\lambda'))=a+_F\mathsf{sum}_F(\lambda').
$$
For a finite point spine $xs:\mathsf{ListCarrier}(V)$, a finite scalar-weight
spine $\lambda:\mathsf{ListCarrier}(F)$, and an endpoint $z:V$, define
$\mathsf{ConvAffSpine}_C(xs,\lambda,z)$ to mean the conjunction of the
following two data.

First, the global finite-weight readback says that the point spine and weight
spine have the same finite positions, every displayed point is carried, every
displayed scalar weight is nonnegative, and the finite scalar sum is classified
as the scalar unit:
$$
\exists \theta:\mathsf{Pos}(xs)\cong\mathsf{Pos}(\lambda).\;
\Bigl(\forall i:\mathsf{Pos}(xs).\;
C(\mathsf{entry}_{xs}(i))\land
\mathsf{NonNeg}_F(\mathsf{entry}_{\lambda}(\theta(i)))\Bigr)
\land \mathsf{sum}_F(\lambda)\sim_F1_F.
$$
Second, the endpoint $z$ is obtained from those leaves by the least local
binary bracketing relation $\mathsf{Bracket}_C(xs,\lambda,z)$ generated by the
following clauses:
\begin{enumerate}
  \item singleton: if $C(x)$, $\mathsf{NonNeg}_F(a)$, and $a\sim_F1_F$, then
  $$
  \mathsf{Bracket}_C(\mathsf{cons}(x,\mathsf{nil}),
  \mathsf{cons}(a,\mathsf{nil}),x);
  $$
  \item binary extension: if $C(x)$, $\mathsf{NonNeg}_F(a)$,
  $\mathsf{NonNeg}_F(b)$, $(a+_F b)\sim_F1_F$, and
  $\mathsf{Bracket}_C(ys,\mu,y)$, then
  $$
  \mathsf{Bracket}_C\bigl(\mathsf{cons}(x,ys),
  \mathsf{cons}(a,b\cdot_F\mu),
  (a\odot x)+_V(b\odot y)\bigr),
  $$
  where $b\cdot_F\mu$ denotes pointwise scalar multiplication of the tail
  weight spine.
\end{enumerate}
Thus $\mathsf{ConvAffSpine}_C(xs,\lambda,z)$ records both the visible finite
unit-sum weight spine and the recursively bracketed affine endpoint built from
binary affine-combination rows.
\end{definition}
\leandef{BEDC.Derived.ConvexSetUp.ConvexSetSingletonAffineSpine}

\begin{theorem}[ConvexSet finite affine-spine closure]
\label{thm:convexset-finite-affine-spine-closure}
Let a $\ConvexSetUp$ certificate be given over a scalar $\FieldUp(F)$ and a
vector carrier from $\VecSpaceUp(F,V)$, with notation as in
\autoref{def:convexset-binary-affine-combination-row}. For every finite point
spine $xs:\mathsf{ListCarrier}(V)$, finite scalar-weight spine
$\lambda:\mathsf{ListCarrier}(F)$, and endpoint $z:V$, if
$$
\mathsf{ConvAffSpine}_C(xs,\lambda,z)
$$
holds, then the recursively bracketed affine endpoint is carried by the convex
carrier:
$$
C(z).
$$
\end{theorem}
\begin{proof}
Unpack \autoref{def:convexset-finite-affine-spine-data}. The finite-weight
readback supplies the advertised carried-point, nonnegative-weight, and
unit-sum scalar spine. The carrier conclusion is proved by induction on the
local bracketing witness $\mathsf{Bracket}_C(xs,\lambda,z)$.

In the singleton clause, the endpoint is the displayed point $x$, and the
clause itself supplies $C(x)$. Hence the endpoint is carried.

In the binary-extension clause, the induction hypothesis applied to the tail
bracketing witness gives $C(y)$. The current clause supplies $C(x)$,
$\mathsf{NonNeg}_F(a)$, $\mathsf{NonNeg}_F(b)$, and the local unit-sum
comparison $(a+_F b)\sim_F1_F$. Applying the binary affine-combination row of
\autoref{def:convexset-binary-affine-combination-row} to the carried points
$x,y$ and coefficients $a,b$ gives
$$
C\bigl((a\odot x)+_V(b\odot y)\bigr).
$$
This endpoint is exactly the endpoint named by the binary-extension clause.
The induction therefore covers every recursively bracketed finite affine-spine
witness, so $C(z)$ holds.
\end{proof}
\leanchecked{BEDC.Derived.ConvexSetUp.ConvexSetSingletonAffineSpine\_closure}

\begin{definition}[ConvexSet pointwise intersection carrier]
\label{def:convexset-pointwise-intersection-carrier}
Let two $\ConvexSetUp$ certificates be given over the same scalar $\FieldUp(F)$ and the same vector carrier from $\VecSpaceUp(F,V)$, using the same scalar classifier $\sim_F$, scalar addition $+_F$, scalar unit $1_F$, nonnegative predicate $\mathsf{NonNeg}_F$, scalar action $\odot$, and vector addition $+_V$. Write $C,D:V\to\mathsf{Prop}$ for their convex carriers. The pointwise intersection carrier $C\cap_{\ConvexSetUp}D$ is the predicate on $V$ defined by
$$
(C\cap_{\ConvexSetUp}D)(z)\Longleftrightarrow C(z)\land D(z).
$$
\end{definition}
\leandef{BEDC.Derived.ConvexSetUp.ConvexSetPointwiseIntersection}

\begin{theorem}[ConvexSet pointwise intersection affine-combination closure]
\label{thm:convexset-pointwise-intersection-affine-combination-closure}
Let two $\ConvexSetUp$ certificates be given as in \autoref{def:convexset-pointwise-intersection-carrier}. Assume each certificate supplies the binary affine-combination row of \autoref{def:convexset-binary-affine-combination-row} for its own convex carrier, and write $K:=C\cap_{\ConvexSetUp}D$. For all $a,b:F$ and $x,y:V$, if
$$
K(x)\land K(y)\land \mathsf{NonNeg}_F(a)\land \mathsf{NonNeg}_F(b)
\land (a+_F b)\sim_F 1_F,
$$
then
$$
K\bigl((a\odot x)+_V(b\odot y)\bigr).
$$
Thus the pointwise intersection satisfies the same nonnegative unit-sum binary affine-combination closure row over the common $\FieldUp$ and $\VecSpaceUp$ data.
\end{theorem}
\leanchecked{BEDC.Derived.ConvexSetUp.ConvexSetPointwiseIntersection\_affine\_combination\_closure}
\begin{proof}
Unfold \autoref{def:convexset-pointwise-intersection-carrier}. From $K(x)$ obtain $C(x)$ and $D(x)$, and from $K(y)$ obtain $C(y)$ and $D(y)$. The scalar hypotheses are shared by the two input certificates because the scalar classifier, scalar addition, scalar unit, nonnegative predicate, scalar action, and vector addition are the common data named in the definition.

Apply the binary affine-combination row of \autoref{def:convexset-binary-affine-combination-row} for the carrier $C$ to the coefficients $a,b$ and the carried points $x,y$. The projected carrier hypotheses, the two nonnegativity hypotheses, and the unit-sum classifier hypothesis give
$$
C\bigl((a\odot x)+_V(b\odot y)\bigr).
$$
Apply the same row for the carrier $D$ with the same coefficients and points. The other projected carrier hypotheses and the same scalar hypotheses give
$$
D\bigl((a\odot x)+_V(b\odot y)\bigr).
$$
Pairing these two carrier conclusions and folding \autoref{def:convexset-pointwise-intersection-carrier} gives
$$
K\bigl((a\odot x)+_V(b\odot y)\bigr),
$$
which is exactly the displayed binary affine-combination closure row for the intersection carrier.
\end{proof}

\begin{definition}[ConvexSet classifier image carrier]
\label{def:convexset-linear-image-carrier}
Let $F$ carry a $\FieldUp(F)$ certificate, let $V$ and $W$ carry $\VecSpaceUp(F,V)$ and $\VecSpaceUp(F,W)$ certificates over the same scalar source, and let $C:V\to\mathsf{Prop}$ be the convex carrier of a $\ConvexSetUp$ certificate on $V$. Write $\sim_W$ for the codomain vector classifier, and let $f:V\to W$ be a carried $\LinearMapUp(V,W)$ map. The classifier image of $C$ along $f$ is the predicate $f[C]_{\sim_W}:W\to\mathsf{Prop}$ defined by
$$
f[C]_{\sim_W}(w)\Longleftrightarrow
\exists x:V.\; C(x)\land f(x)\sim_W w.
$$
The witness records a source point carried by the convex carrier and a codomain classifier comparison from its $f$-output to the displayed codomain endpoint.
\end{definition}

\begin{theorem}[ConvexSet linear image affine-combination closure]
\label{thm:convexset-linear-image-affine-combination-closure}
Let the data be as in \autoref{def:convexset-linear-image-carrier}. Write $\sim_F$, $+_F$, $1_F$, and $\mathsf{NonNeg}_F$ for the scalar classifier, scalar addition, scalar unit, and nonnegative predicate. Write $+_V$ and $\odot_V$ for the source vector addition and scalar action, and write $+_W$, $\odot_W$, and $\sim_W$ for the codomain vector addition, scalar action, and classifier. Assume the source $\ConvexSetUp$ certificate supplies the binary affine-combination row of \autoref{def:convexset-binary-affine-combination-row}. For carried scalars $a,b:F$ and codomain endpoints $u,v:W$, if
$$
f[C]_{\sim_W}(u)\land f[C]_{\sim_W}(v)\land
\mathsf{NonNeg}_F(a)\land\mathsf{NonNeg}_F(b)\land
(a+_F b)\sim_F1_F,
$$
then
$$
f[C]_{\sim_W}\bigl((a\odot_W u)+_W(b\odot_W v)\bigr).
$$
Thus the pointwise classifier image $f[C]_{\sim_W}$ satisfies the same binary nonnegative unit-sum affine-combination closure row in the codomain vector-space carrier.
\end{theorem}
\begin{proof}
Unpack \autoref{def:convexset-linear-image-carrier} for the two image hypotheses. Choose source points $x,y:V$ with
$$
C(x),\qquad C(y),\qquad f(x)\sim_Wu,\qquad f(y)\sim_Wv.
$$
Because the $\ConvexSetUp$ carrier is a subset of the source $\VecSpaceUp$ carrier, $x$ and $y$ are carried source inputs. Set
$$
s:=(a\odot_V x)+_V(b\odot_V y).
$$
The nonnegativity hypotheses and the scalar unit-sum comparison are exactly the coefficient hypotheses required by \autoref{def:convexset-binary-affine-combination-row}. Applying that row to $x,y$ and $a,b$ gives $C(s)$.

It remains to compare the $f$-output of this source witness with the displayed codomain affine combination. The carried $\LinearMapUp(V,W)$ map supplies the function-with-law rows of \autoref{def:linearmap-carrier}; read through \autoref{def:vecspace-carrier} and \autoref{def:module-linearmap-certificate-package}, these include output closure, additive preservation, and scalar compatibility. Source scalar-action closure and source additive closure make the two summands of $s$ and $s$ carried source inputs. Additive preservation gives
$$
f(s)\sim_W f(a\odot_V x)+_W f(b\odot_V y).
$$
Scalar compatibility gives
$$
f(a\odot_V x)\sim_W a\odot_W f(x),
\qquad
f(b\odot_V y)\sim_W b\odot_W f(y).
$$
The target additive congruence row from \autoref{def:module-stability-certificate} transports these two comparisons to
$$
f(a\odot_V x)+_W f(b\odot_V y)
\sim_W
(a\odot_W f(x))+_W(b\odot_W f(y)).
$$

The image comparisons $f(x)\sim_Wu$ and $f(y)\sim_Wv$ are codomain classifier comparisons. Scalar-action congruence in the codomain vector-space module, using scalar reflexivity on the carried scalars $a$ and $b$, gives
$$
a\odot_W f(x)\sim_W a\odot_W u,
\qquad
b\odot_W f(y)\sim_W b\odot_W v.
$$
A second use of target additive congruence gives
$$
(a\odot_W f(x))+_W(b\odot_W f(y))
\sim_W
(a\odot_W u)+_W(b\odot_W v).
$$
Classifier transitivity, as packaged by \autoref{thm:namecert-equivalence-transitivity}, composes the three codomain comparisons and yields
$$
f(s)\sim_W(a\odot_W u)+_W(b\odot_W v).
$$
The source witness $s$, the convex-carrier proof $C(s)$, and this codomain classifier comparison repack by \autoref{def:convexset-linear-image-carrier} as
$$
f[C]_{\sim_W}\bigl((a\odot_W u)+_W(b\odot_W v)\bigr).
$$
\end{proof}

\begin{theorem}[ConvexSet linear image finite affine-spine closure]
\label{thm:convexset-linear-image-finite-affine-spine-closure}
Let the data be as in \autoref{def:convexset-linear-image-carrier}. Write $\sim_F$, $+_F$, $1_F$, and $\mathsf{NonNeg}_F$ for the scalar classifier, scalar addition, scalar unit, and nonnegative predicate. Write $+_W$, $\odot_W$, and $\sim_W$ for the codomain vector addition, scalar action, and classifier. Assume the source $\ConvexSetUp$ certificate supplies the binary affine-combination row of \autoref{def:convexset-binary-affine-combination-row}. For every finite codomain point spine $ys:\mathsf{ListCarrier}(W)$, finite scalar-weight spine $\lambda:\mathsf{ListCarrier}(F)$, and endpoint $z:W$, if
$$
\mathsf{ConvAffSpine}_{f[C]_{\sim_W}}(ys,\lambda,z)
$$
holds in the sense of \autoref{def:convexset-finite-affine-spine-data} with carrier predicate $f[C]_{\sim_W}$, then
$$
f[C]_{\sim_W}(z).
$$
Thus the classifier image carrier is closed under every finite affine spine whose weights are nonnegative and whose finite scalar sum is classified as $1_F$.
\end{theorem}
\begin{proof}
By \autoref{thm:convexset-linear-image-affine-combination-closure}, the image carrier $f[C]_{\sim_W}$ satisfies the codomain binary affine-combination row
$$
f[C]_{\sim_W}(u)\land f[C]_{\sim_W}(v)\land
\mathsf{NonNeg}_F(a)\land\mathsf{NonNeg}_F(b)\land(a+_F b)\sim_F1_F
\Longrightarrow
f[C]_{\sim_W}\bigl((a\odot_Wu)+_W(b\odot_Wv)\bigr).
$$
This row is obtained from the source convex row and the carried $\LinearMapUp(V,W)$ data, whose additivity, scalar compatibility, and codomain congruence rows are read through \autoref{def:linearmap-carrier}, \autoref{def:module-linearmap-certificate-package}, and \autoref{def:module-stability-certificate}.

Unpack the displayed hypothesis using \autoref{def:convexset-finite-affine-spine-data}. The finite readback gives the common finite position data, nonnegative weights, and unit-sum scalar comparison. The endpoint carrier conclusion is proved by induction on the local bracketing witness $\mathsf{Bracket}_{f[C]_{\sim_W}}(ys,\lambda,z)$.

In the singleton clause, the endpoint is the displayed codomain point $u$, and the clause itself supplies $f[C]_{\sim_W}(u)$. Hence the endpoint lies in the image carrier.

In the binary-extension clause, the clause supplies $f[C]_{\sim_W}(u)$, the coefficient hypotheses $\mathsf{NonNeg}_F(a)$ and $\mathsf{NonNeg}_F(b)$, the local unit-sum comparison $(a+_F b)\sim_F1_F$, and a tail bracketing witness with endpoint $v$. The induction hypothesis applied to that tail witness gives $f[C]_{\sim_W}(v)$. Applying the binary image row above to $u,v$ and $a,b$ gives
$$
f[C]_{\sim_W}\bigl((a\odot_Wu)+_W(b\odot_Wv)\bigr),
$$
which is exactly the endpoint named by the binary-extension clause. Therefore every local bracketing witness over $f[C]_{\sim_W}$ has its endpoint in the same classifier image carrier, and the original endpoint $z$ satisfies $f[C]_{\sim_W}(z)$.
\end{proof}
