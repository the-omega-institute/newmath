\section{Carrier definition}

\begin{definition}[POSet carrier]
\label{def:poset-carrier}
Use a preorder setup extended with antisymmetry under the carrier classifier.
\end{definition}

\begin{theorem}[POSetUp BHist carrier transport obligation]
\label{thm:poset-bhist-carrier-transport-obligation}
Let $h$ and $k$ be carried $\mathsf{BHist}$ endpoints in the concrete unary
prefix preorder carrier.  If $h\hsame k$, the preorder carrier transport row
is present, and the $\POSetUp$ antisymmetry row is carried with that preorder
package, then transporting the endpoint from $h$ to $k$ leaves it in the
$\POSetUp$ carrier.  The transported package keeps the preorder carrier row
and repacks the same antisymmetry field; it does not introduce a quotient
carrier or a representative class.
\end{theorem}
\begin{proof}
Apply the carrier transport row of
\autoref{thm:preorder-prefix-order-laws-hsame}.  It transports the displayed
unary $\mathsf{BHist}$ carrier along the classifier row $h\hsame k$, so the
preorder carrier component remains available at the target endpoint.

The $\POSetUp$ carrier of \autoref{def:poset-carrier} is precisely that
preorder carrier together with an antisymmetry row under the same classifier.
Repack the transported preorder carrier with the displayed antisymmetry field.
The endpoint remains a carried $\mathsf{BHist}$ endpoint, and no host quotient
object is used.
\end{proof}

\begin{theorem}[POSetUp classifier exactness obligation]
\label{thm:poset-classifier-exactness-obligation}
For carried unary $\mathsf{BHist}$ histories $h$ and $k$, the $\POSetUp$
classifier row between them is exact for the bidirectional prefix comparison:
the classifier holds exactly when the classifier row is present and the two
directed preorder comparison rows $h\le_{\mathsf{pref}}k$ and
$k\le_{\mathsf{pref}}h$ collapse by the antisymmetry field.  The reverse
comparison rows are obtained from preorder reflexivity and transport along the
classifier.
\end{theorem}
\begin{proof}
For the forward comparison-to-classifier direction, assume the two directed
prefix witnesses.  The antisymmetry field of
\autoref{thm:poset-prefix-antisymmetry} collapses them to $\hsame(h,k)$, which
is the carrier classifier named by \autoref{def:poset-carrier}.

For the classifier-to-comparison direction, assume $\hsame(h,k)$.  The
preorder package of \autoref{thm:preorder-prefix-order-laws-hsame} supplies
reflexivity at each carried endpoint and transports comparison endpoints along
the classifier.  Transport the reflexive comparison at $h$ to obtain
$h\le_{\mathsf{pref}}k$, and transport the reflexive comparison at $k$ to
obtain $k\le_{\mathsf{pref}}h$.

These two constructions identify the classifier boundary with the
bidirectional comparison boundary, with antisymmetry providing the collapse
and the preorder rows providing the transported reflexive comparisons.
\end{proof}

\begin{theorem}[POSetUp antisymmetry ledger obligation]
\label{thm:poset-ledger-antisymmetry-obligation}
An antisymmetry use in the concrete $\POSetUp$ ledger records exactly the two
comparison witnesses $h\le_{\mathsf{pref}}k$ and
$k\le_{\mathsf{pref}}h$, their carried $\mathsf{BHist}$ endpoints, and the
resulting $\hsame(h,k)$ classifier row.  The ledger contains no host quotient,
chosen class representative, or external equality witness.
\end{theorem}
\begin{proof}
Project the two comparison witnesses from the preorder ledger supplied by
\autoref{thm:preorder-prefix-order-laws-hsame}.  Their endpoints are the same
carried unary $\mathsf{BHist}$ histories used by the $\POSetUp$ carrier.

Apply the antisymmetry row
\autoref{thm:poset-prefix-antisymmetry} to the projected comparison pair.  It
produces the classifier row $\hsame(h,k)$ and no additional object.

Repack the ledger entry as the ordered pair of comparison witnesses, the two
displayed endpoints, and the resulting classifier row.  This is exactly the
ledger policy described by the $\POSetUp$ certificate surface.
\end{proof}

\begin{theorem}[POSet root carrier obligation]
\label{thm:poset-root-carrier-obligation}
The root $\POSetUp$ carrier obligation is a $\PreorderUp$ carrier over
$\mathsf{BHist}$, equipped with the same prefix relation and with an
antisymmetry row under the carrier classifier. No quotient carrier is added:
the carried endpoints remain displayed $\mathsf{BHist}$ histories, and the
only collapse allowed by the partial-order field is the classifier conclusion
$\hsame(h,k)$.
\end{theorem}
\begin{proof}
The carrier clause is exactly \autoref{def:poset-carrier}: start with the
preorder carrier and add the antisymmetry field as an extra row, not as a new
endpoint space.

For the concrete root instance, \autoref{thm:preorder-prefix-order-laws-hsame}
supplies the $\mathsf{BHist}$ carrier, the prefix preorder relation, and the
$\hsame$ classifier boundary. The partial-order coordinate is supplied by
\autoref{thm:poset-prefix-antisymmetry}, which turns opposing prefix
comparisons into $\hsame(h,k)$.

Thus the carrier remains the inherited preorder carrier, while the
antisymmetry data is an obligation row attached to that carrier.
\end{proof}

\begin{theorem}[POSet root classifier obligation]
\label{thm:poset-root-classifier-obligation}
The root $\POSetUp$ classifier is the inherited $\hsame$-based classifier on
carried $\mathsf{BHist}$ endpoints, and it is stable under the preorder
relation: classifier-equivalent endpoints may be read through the same prefix
comparison rows without changing the carried $\POSetUp$ object.
\end{theorem}
\begin{proof}
By \autoref{def:poset-carrier}, a $\POSetUp$ object does not replace the
preorder classifier. It extends the preorder package by demanding
antisymmetry under that classifier.

Apply \autoref{thm:preorder-prefix-order-laws-hsame} to obtain the inherited
$\hsame$ classifier together with the prefix-order stability rows. Those rows
are the classifier transport surface for comparison endpoints.

The additional $\POSetUp$ row from \autoref{thm:poset-prefix-antisymmetry} is
compatible with the same classifier, because its conclusion is again
$\hsame(h,k)$. Hence classifier transport remains the inherited preorder
transport.
\end{proof}

\begin{theorem}[POSet root exactness obligation]
\label{thm:poset-root-exactness-obligation}
For carried unary prefix endpoints $h$ and $k$, the opposing comparison rows
$h\le_{\mathsf{pref}}k$ and $k\le_{\mathsf{pref}}h$ are exactly the
$\POSetUp$ classifier row on those endpoints: the forward direction is
antisymmetry, and the reverse direction is the inherited reflexive prefix row
transported along $\hsame$.
\end{theorem}
\begin{proof}
For the forward direction, assume the two comparison rows. The concrete
antisymmetry theorem \autoref{thm:poset-prefix-antisymmetry} collapses the
opposing prefix witnesses to $\hsame(h,k)$, which is the classifier row
specified by \autoref{def:poset-carrier}.

For the reverse direction, assume the classifier row $\hsame(h,k)$. The
preorder law package of \autoref{thm:preorder-prefix-order-laws-hsame}
provides reflexive prefix comparisons on carried endpoints and transports
comparison endpoints along $\hsame$.

Transporting the reflexive comparison on $h$ gives $h\le_{\mathsf{pref}}k$,
and transporting the reflexive comparison on $k$ gives
$k\le_{\mathsf{pref}}h$. The two directions identify the exact comparison
boundary of the root $\POSetUp$ row.
\end{proof}
