\chapter{Naming Certificate for $\mathsf{CompactUniformContinuityUp}$}
\label{ch:concrete-instances-compact-uniform-continuity}

The compact uniform-continuity interface $\mathsf{CompactUniformContinuityUp}$ names the constructive Heine--Cantor bridge: a continuous map over a compact metric carrier determines a uniform-modulus ledger. Its source is the shared $\ContinuousMapUp$ and $\CompactMetricUp$ certificate data, and its public object is the uniform precision row kept inside the BEDC history layer.

\section{Source spec for $\mathsf{CompactUniformContinuityUp}$}
\label{sec:concrete-instances-compact-uniform-continuity-source-spec}

\begin{definition}[Source rows of $\mathsf{CompactUniformContinuityUp}$]
\label{def:concrete-instances-compact-uniform-continuity-source-rows}
A source row for $\mathsf{CompactUniformContinuityUp}$ is a finite BEDC packet
$$
(\mathsf{source},\mathsf{map},\mathsf{target},\mathsf{modulus},\mathsf{cert},
\mathsf{distance},\varepsilon,\mathcal{P}_{\varepsilon},s,M,\ell,
\mu_{\varepsilon},\mathsf{u}_{\varepsilon})
$$
where the first six histories are the $\ContinuousMapUp$ carrier row, $\varepsilon$ is the positive tolerance history, $\mathcal{P}_{\varepsilon}$ is the compact-metric $\mathsf{ProbeBundle}$ net at that tolerance, $s,M,\ell$ are the compact-metric stream, modulus, and limit witnesses, and $\mu_{\varepsilon},\mathsf{u}_{\varepsilon}$ are the produced uniform precision history and its ledger certificate.
\end{definition}
\leandef{BEDC.Derived.CompactUniformContinuityUp.SourceRows}

\section{Pattern spec}
\label{sec:concrete-instances-compact-uniform-continuity-pattern}

\begin{definition}[Pattern witness for $\mathsf{CompactUniformContinuityUp}$]
\label{def:concrete-instances-compact-uniform-continuity-pattern}
The pattern of a compact uniform-continuity row is the conjunction of three visible certificate clauses: the continuous graph is accepted by $\ContinuousMapUp$, the shared source is accepted by $\CompactMetricUp$ through the displayed finite-net and complete-limit rows, and the tolerance history continues by $\Cont$ to the uniform precision certificate
$$
\Cont(\varepsilon,\mu_{\varepsilon},\mathsf{u}_{\varepsilon}).
$$
Thus the uniform modulus is named by a ledger relation on histories, not by an external selector from tolerances to radii.
\end{definition}
\leandef{BEDC.Derived.CompactUniformContinuityUp.Pattern}

\section{Classifier spec}
\label{sec:concrete-instances-compact-uniform-continuity-classifier}

\begin{definition}[Classifier for $\mathsf{CompactUniformContinuityUp}$]
\label{def:concrete-instances-compact-uniform-continuity-classifier}
Two source rows classify as the same $\mathsf{CompactUniformContinuityUp}$ object when their continuous-map histories, compact-metric tolerance and limit histories, stream and stream-modulus families, uniform precision histories, and uniform ledger certificates agree by the displayed $\hsame$ rows. The classifier therefore identifies presentations that expose the same compact subset, the same continuous graph, and the same uniform-modulus ledger up to BEDC history sameness.
\end{definition}
\leandef{BEDC.Derived.CompactUniformContinuityUp.Classifier}

\section{Stability certificate}
\label{sec:concrete-instances-compact-uniform-continuity-stability}

\begin{theorem}[Stability of $\mathsf{CompactUniformContinuityUp}$]
\label{thm:concrete-instances-compact-uniform-continuity-stability}
If a source row satisfies the compact uniform-continuity pattern, then every classified presentation of the same row also satisfies the pattern. In particular, the continuous-map carrier transports componentwise, the compact-metric finite-net and complete-limit certificate transports along the tolerance, stream, modulus, and limit classifiers, and the uniform precision ledger remains a $\Cont$ row after classifier transport. This is the bridge closure needed for restriction to a compact subcarrier: the compact source and continuous graph may be re-presented through classified histories without changing the named uniform-modulus object.
\end{theorem}
\leanchecked{BEDC.Derived.CompactUniformContinuityUp.stability}

\begin{proof}
Unfold the pattern row. The first component is the $\ContinuousMapUp$ carrier, so componentwise $\hsame$ transport of source, graph, target, modulus, certificate, and distance preserves the same carrier row. The second component is the $\CompactMetricUp$ certificate, and its checked transport theorem preserves the finite-net and complete-limit package under the displayed tolerance, stream, modulus, and limit classifiers. The final component is the uniform precision ledger; transporting $\varepsilon$, $\mu_{\varepsilon}$, and $\mathsf{u}_{\varepsilon}$ by $\hsame$ preserves the same $\Cont$ row.
\end{proof}

\section{Ledger policy}
\label{sec:concrete-instances-compact-uniform-continuity-ledger}

\begin{definition}[Ledger for $\mathsf{CompactUniformContinuityUp}$]
\label{def:concrete-instances-compact-uniform-continuity-ledger}
The ledger keeps the entire source row: the continuous-map carrier histories, the compact-metric tolerance and $\mathsf{ProbeBundle}$ net, the stream, stream-modulus, and limit witnesses, and the uniform precision and certificate histories. Traceability is therefore local to the finite compact-subset enumeration and the pointwise-continuity data from which the uniform modulus is read back; no subsequence selector, open-cover choice function, or external modulus function is recorded.
\end{definition}
