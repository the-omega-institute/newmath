\chapter{A Concrete Naming Certificate for \(\NatTransUp\)}
\label{ch:concrete-instances-nattrans-namecert}

The natural transformation interface \(\NatTransUp\) has certificate obligations between two functors from \autoref{ch:concrete-instances-functor-namecert} sharing source and target categories. The carrier is, for each object \(X\) of the source category, a morphism in the target category from the first functor's image of \(X\) to the second functor's image of \(X\), satisfying the naturality square under the codomain category's morphism classifier. It is not a primitive 2-cell; it is admitted only as an object-indexed family of morphisms with the naturality field.

\section{Carrier definition}

\begin{definition}[NatTrans carrier]
\label{def:nattrans-carrier}
Use an object-indexed family of codomain morphisms \(\eta_X:F(X)\to G(X)\) plus naturality witnesses for every source morphism.
\end{definition}
\leandef{BEDC.Derived.NatTransUp.NatTransPrefixComponentCarrier}

\begin{theorem}[NatTrans prefix component empty identity exactness]
\label{thm:nattrans-prefix-component-carrier-empty-identity-iff}
For prefix components over unary histories, the empty component from
$\mathsf{append}(p,a)$ to $\mathsf{append}(q,a)$ exists exactly when $p$, $q$,
and $a$ are unary histories and $\HSame(p,q)$.
\end{theorem}
\begin{proof}
Unfold the prefix component carrier to its category hom-carrier over
$\mathsf{append}(p,a)$ and $\mathsf{append}(q,a)$. Category empty-identity
exactness identifies the two prefixed endpoints; cancellation of the shared
suffix $a$ recovers $\HSame(p,q)$. Conversely, unary append closure supplies
the two prefixed endpoint carriers, and the equality of $p$ and $q$ transports
the category empty identity carrier to the component carrier.
\end{proof}
\leanchecked{BEDC.Derived.NatTransUp.NatTransPrefixComponentCarrier\_empty\_identity\_iff}

\begin{theorem}[NatTrans empty identity prefix transitivity]
\label{thm:nattrans-prefix-component-empty-identity-prefix-trans}
For prefix components over unary histories, if the empty component identifies
\(\mathsf{append}(p,a)\) with \(\mathsf{append}(q,a)\) and the empty component
also identifies \(\mathsf{append}(q,a)\) with \(\mathsf{append}(r,a)\), then
\(\HSame(p,r)\).
\end{theorem}
\begin{proof}
Apply empty-identity exactness to the two component witnesses. The first
component gives \(\HSame(p,q)\), and the second gives \(\HSame(q,r)\).
Transitivity of history sameness identifies \(p\) and \(r\).
\end{proof}
\leanchecked{BEDC.Derived.NatTransUp.NatTransPrefixComponentCarrier\_empty\_identity\_prefix\_trans}

\begin{theorem}[NatTrans prefix component endomorphism component empty exactness]
\label{thm:nattrans-prefix-component-endomorphism-component-empty}
For prefix components whose source and target prefixes agree, a component
history \(\eta\) is carried exactly when the prefix, object, and component
histories are unary and the component is internally empty:
$$
\mathsf{NatTransPrefixComponentCarrier}(p,p,a,\eta)
\quad\Longleftrightarrow\quad
\mathsf{UnaryHistory}(p)\wedge\mathsf{UnaryHistory}(a)\wedge
\mathsf{UnaryHistory}(\eta)\wedge\eta\hsame\emp.
$$
\end{theorem}
\leanchecked{BEDC.Derived.NatTransUp.NatTransPrefixComponentCarrier\_endomorphism\_component\_empty\_iff}
\begin{proof}
Unfold the prefix component into its category hom-carrier
\(\mathsf{CategoryHomCarrier}(\mathsf{append}(p,a),\mathsf{append}(p,a),\eta)\).
The continuation field has matching source and target, hence right-unit
uniqueness forces \(\eta\hsame\emp\), while the component fields provide unary
membership of \(p\), \(a\), and \(\eta\). Conversely, an empty component is
rebuilt by the prefix empty-identity exactness theorem with reflexivity of
\(\hsame(p,p)\).
\end{proof}

\begin{proof}

source prefix extracts unary membership of $p$, and the prefix sameness
specializes to the equality $q=\Eone(p)$. Conversely, substituting this equality
rebuilds the one-headed target prefix, and empty-identity exactness packages
the empty component.

$$
\mathsf{NatTransPrefixComponentCarrier}(\emp,q,a,\eta)
\quad\Longleftrightarrow\quad
\mathsf{UnaryHistory}(q)\wedge
\mathsf{UnaryHistory}(a)\wedge
\mathsf{CategoryHomCarrier}(a,\mathsf{append}(q,a),\eta).

\begin{proof}

$\mathsf{append}(\emp,a)$, which is history-same as $a$ by the left unit law for

unprefixed source form. The converse uses the same endpoint equality in the
opposite direction and packages the empty prefix carrier.
\end{proof}

\begin{theorem}[NatTrans prefix component zero-headed component impossibility]
\label{thm:nattrans-prefix-component-carrier-zero-headed-component-absurd}
For histories $p,q,a,\eta$, a prefix component carrier has no zero-headed
source prefix, target prefix, object, or component morphism:
$$
\mathsf{NatTransPrefixComponentCarrier}(p,q,a,\eta)
\Longrightarrow
\left(
\exists z,\ p=\Ezero(z)
\right)
\lor
\left(
\exists z,\ q=\Ezero(z)
\right)
\lor
\left(
\exists z,\ a=\Ezero(z)
\right)
\lor
\left(
\exists z,\ \eta=\Ezero(z)
\right)
\Longrightarrow
\mathsf{False}.
$$
\end{theorem}
\begin{proof}
Unfold the prefix component carrier. The first three fields give unary-history
witnesses for $p$, $q$, and $a$, and the underlying category hom-carrier gives
the unary-history witness for the component morphism $\eta$. A zero-headed
component in any of those positions contradicts unary zero-extension
separation.
\end{proof}

\begin{theorem}[NatTrans prefix component vertical composition closure]
\label{thm:nattrans-prefix-component-carrier-vert-comp-closed}
For prefix components over unary histories, if \(\eta\) is a component from
\(\mathsf{append}(p,a)\) to \(\mathsf{append}(q,a)\), \(\theta\) is a component
from \(\mathsf{append}(q,a)\) to \(\mathsf{append}(r,a)\), and
\(\ContR(\eta,\theta,\theta\eta)\), then the composite component is a component
from \(\mathsf{append}(p,a)\) to \(\mathsf{append}(r,a)\).
\end{theorem}
\begin{proof}
Unpack the two prefix-component carrier witnesses. The outer prefix and object
unary witnesses come from the first component and the target prefix witness
comes from the second component. Category hom-carrier composition closure
composes the two component morphisms across the shared middle object
\(\mathsf{append}(q,a)\).
\end{proof}

\begin{theorem}[NatTrans prefix component empty vertical composite exactness]
\label{thm:nattrans-prefix-component-carrier-vert-comp-empty-iff}

$\mathsf{append}(p,a)$ to $\mathsf{append}(q,a)$ and $\theta$ is a component
from $\mathsf{append}(q,a)$ to $\mathsf{append}(r,a)$, then the empty vertical
composite is equivalent to both component histories being empty and the
adjacent prefixes agreeing:
$$
\ContR(\eta,\theta,\emp)
\quad\Longleftrightarrow\quad
\HSame(\eta,\emp)\wedge\HSame(\theta,\emp)\wedge\HSame(p,q)\wedge\HSame(q,r).
$$
\end{theorem}
\begin{proof}
Unfold the two prefix components to their category hom-carriers over the
shared object suffix $a$. Category empty-composite exactness gives the two
empty component histories and the two endpoint equalities. Right cancellation
removes the common suffix $a$ from both endpoint equalities. Conversely, the
prefix equalities rebuild the component endpoint equalities, and category
empty-composite exactness supplies the empty vertical continuation ledger.
\end{proof}

determinacy identifies the prefixed middle endpoints
\(\mathsf{append}(q,a)\) and \(\mathsf{append}(q',a)\). Cancellation of the
shared suffix \(a\) gives \(\HSame(q,q')\).
\end{proof}
\leanchecked{BEDC.Derived.NatTransUp.NatTransPrefixComponentCarrier\_vert\_comp\_middle\_prefix\_deterministic}

\begin{theorem}[NatTrans prefix component vertical composite source-prefix determinacy]
\label{thm:nattrans-prefix-component-vertical-composite-source-prefix-determinacy}
For prefix components over unary histories, if \(\eta\) is a component from
\(\mathsf{append}(p,a)\) to \(\mathsf{append}(q,a)\), \(\theta\) is a component
from \(\mathsf{append}(q,a)\) to \(\mathsf{append}(r,a)\),
\(\ContR(\eta,\theta,\eta\theta)\), and the same composite history
\(\eta\theta\) is displayed as a component from \(\mathsf{append}(s,a)\) to
\(\mathsf{append}(r,a)\), then \(\HSame(p,s)\).
\end{theorem}
\begin{proof}
First close the vertical composite of \(\eta\) and \(\theta\), obtaining a
component carrier from prefix \(p\) to prefix \(r\) over the object \(a\). Apply
source-prefix determinacy to this composed carrier and the displayed carrier
with source prefix \(s\), both targeting prefix \(r\).
\end{proof}

\end{proof}

\begin{theorem}[Prefix identity naturality square]
\label{thm:nattrans-prefix-identity-naturality-square}

both composites are hom-carriers from \(\mathsf{append}(p,a)\) to
\(\mathsf{append}(p,b)\), and the two composites are history-same.
\end{theorem}
\begin{proof}
The left empty continuation identifies the left composite with \(f\), and
the right empty continuation identifies the right composite with \(f\). The
given hom-carrier therefore supplies both unprefixed sides of the square.
Prefix hom-carrier preservation sends both sides across \(p\), and
transitivity of history sameness identifies the composites.
\end{proof}

\section{The certificate}

\begin{definition}[NatTrans source specification]
\label{def:nattrans-source-specification}
The source is two certified functors sharing the same certified source and target categories.
\end{definition}

\begin{definition}[NatTrans pattern specification]
\label{def:nattrans-pattern-specification}
The pattern is component-family formation with one codomain morphism per source object.
\end{definition}

\begin{definition}[NatTrans classifier specification]
\label{def:nattrans-classifier-specification}
The classifier relates natural transformations componentwise by the target category morphism classifier.
\end{definition}

\begin{definition}[NatTrans stability certificate]
\label{def:nattrans-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item componentwise reflexivity
  \item componentwise transitivity
  \item naturality square validity for every source morphism
  \item stability of naturality under component classifier replacement
  \item vertical composition closure
  \item identity natural transformation closure
\end{enumerate}
\end{definition}

\begin{definition}[NatTrans ledger policy]
\label{def:nattrans-ledger-policy}
Record the functor ledgers, component morphism ledgers, and naturality derivations.
\end{definition}

\begin{definition}[\(\NatTransUp\) certificate obligations]
\label{def:nattrans-certificate-obligations}
The carrier, source, pattern, classifier, stability, and ledger data listed above are the obligations required for the interface name \(\NatTransUp\).
\end{definition}

\paragraph{Primitive boundary.}
\(\NatTransUp\) is not a BEDC primitive. It is admitted only when those obligations are supplied.

\begin{remark}[Zero-axiom encoding note]
\label{rem:nattrans-zero-axiom-note}
Avoids funext by componentwise classifier sameness.
\end{remark}

\begin{theorem}[NatTrans vertical composition naturality]
\label{thm:nattrans-vertical-composition-naturality}
Let \(F\), \(G\), and \(H\) be certified \(\FunctorUp\) carriers with the same certified source category and target category. Let \(\alpha:F\Rightarrow G\) and \(\beta:G\Rightarrow H\) be admitted \(\NatTransUp\) carriers. For every source morphism \(f:C\to C'\), the vertical component family \((\beta\circ\alpha)_X=\beta_X\circ\alpha_X\) satisfies the target morphism-classifier naturality square
\[
  H(f)\circ(\beta\circ\alpha)_C
  \sim
  (\beta\circ\alpha)_{C'}\circ F(f).
\]
\end{theorem}
\begin{proof}
The shared source and target categories are part of the \(\NatTransUp\) source specification in \autoref{def:nattrans-source-specification}. By the functor carrier definition in \autoref{def:functor-carrier}, the symbols \(F(f)\), \(G(f)\), and \(H(f)\) denote the corresponding target-category morphism images. By \autoref{def:nattrans-carrier}, each admitted natural transformation is an object-indexed component family with a naturality witness for every source morphism.

The \(\NatTransUp\) stability certificate in \autoref{def:nattrans-stability-certificate} includes vertical composition closure. Applying that field to \(\alpha\) and \(\beta\) gives an admitted \(\NatTransUp\) carrier from \(F\) to \(H\) whose component at \(X\) is \(\beta_X\circ\alpha_X\). The same stability certificate lists naturality square validity for every source morphism, so applying that naturality field to this composite carrier and to \(f:C\to C'\) gives
\[
  H(f)\circ(\beta\circ\alpha)_C
  \sim
  (\beta\circ\alpha)_{C'}\circ F(f).
\]
This is the displayed classifier witness for the composite component family.
\end{proof}

\begin{definition}[Opposite natural transformation certificate data]
\label{def:opposite-nattrans-certificate-data}
Let $\mathcal C$ and $\mathcal D$ be $\CategoryUp$ certificates, let $F,G:\mathcal C\to\mathcal D$ be $\FunctorUp$ certificates, and let
$\alpha:F\Rightarrow G$ be a $\NatTransUp$ certificate with component family $\alpha_X:F_0(X)\to G_0(X)$ in $\mathcal D$. Over the
opposite-functor certificates of \autoref{def:opposite-functor-certificate-data}, define $\alpha^{\mathrm{op}}:G^{\mathrm{op}}\Rightarrow F^{\mathrm{op}}$ by the same raw component family
$$
(\alpha^{\mathrm{op}})_X:=\alpha_X.
$$
The component is typed in $\mathcal D^{\mathrm{op}}$ because the hom-carrier of \autoref{def:opposite-category-certificate-data} reads the original
morphism $\alpha_X:F_0(X)\to G_0(X)$ as a displayed morphism $G_0(X)\to F_0(X)$. For a displayed morphism $u:X\to Y$ in
$\mathcal C^{\mathrm{op}}$, the naturality ledger is the classifier-symmetric form of the original naturality ledger for the same raw morphism
$u:Y\to X$ in $\mathcal C$, with both target composites read through opposite composition.
\end{definition}

\begin{theorem}[Opposite natural transformation certificate]
\label{thm:opposite-natural-transformation-certificate}
If $\mathcal C$ and $\mathcal D$ satisfy the $\CategoryUp$ certificate obligations, $F,G:\mathcal C\to\mathcal D$ satisfy the $\FunctorUp$
certificate obligations, and $\alpha:F\Rightarrow G$ satisfies the $\NatTransUp$ certificate obligations of \autoref{def:nattrans-certificate-obligations},
then the data of \autoref{def:opposite-nattrans-certificate-data} satisfy the $\NatTransUp$ certificate obligations from $G^{\mathrm{op}}$ to
$F^{\mathrm{op}}$ over $\mathcal C^{\mathrm{op}}$ and $\mathcal D^{\mathrm{op}}$.
\end{theorem}
\begin{proof}
By \autoref{thm:opposite-category-certificate}, the source and target opposite data are certified categories. By \autoref{thm:opposite-functor-certificate},
$G^{\mathrm{op}}$ and $F^{\mathrm{op}}$ are certified functors with common source $\mathcal C^{\mathrm{op}}$ and common target $\mathcal D^{\mathrm{op}}$.
This supplies the source specification required by \autoref{def:nattrans-source-specification}.

For the carrier field, let $X$ be a source object. The component field of $\alpha$ gives $\alpha_X:F_0(X)\to G_0(X)$ in $\mathcal D$.
Since the opposite hom-carrier reverses endpoints and leaves the raw morphism classifier unchanged, the same raw morphism is a component
$(\alpha^{\mathrm{op}})_X:G^{\mathrm{op}}_0(X)\to F^{\mathrm{op}}_0(X)$ in $\mathcal D^{\mathrm{op}}$. Thus the carrier and pattern fields of
\autoref{def:nattrans-carrier} and \autoref{def:nattrans-pattern-specification} are the original component family with the endpoints read oppositely.

The classifier field is inherited componentwise. The target morphism classifier of $\mathcal D^{\mathrm{op}}$ is the target morphism classifier
of $\mathcal D$ on the same raw component representatives, so componentwise reflexivity, transitivity, and classifier replacement are exactly the
corresponding fields listed in \autoref{def:nattrans-classifier-specification} and \autoref{def:nattrans-stability-certificate} for $\alpha$.

It remains to check naturality. Let $u:X\to Y$ be displayed in $\mathcal C^{\mathrm{op}}$. The same raw morphism is $u:Y\to X$ in
$\mathcal C$. Naturality of $\alpha$ in $\mathcal D$ gives the target morphism-classifier comparison
$$
G_1(u)\circ_{\mathcal D}\alpha_Y \sim_{\mathcal D} \alpha_X\circ_{\mathcal D}F_1(u).
$$
Applying symmetry of the codomain morphism classifier gives
$$
\alpha_X\circ_{\mathcal D}F_1(u) \sim_{\mathcal D} G_1(u)\circ_{\mathcal D}\alpha_Y.
$$
By \autoref{def:opposite-category-certificate-data} and \autoref{def:opposite-functor-certificate-data}, these two original composites are exactly
$$
F^{\mathrm{op}}_1(u)\circ_{\mathcal D^{\mathrm{op}}}(\alpha^{\mathrm{op}})_X
$$
and
$$
(\alpha^{\mathrm{op}})_Y\circ_{\mathcal D^{\mathrm{op}}}G^{\mathrm{op}}_1(u),
$$
respectively. Since the morphism classifier is unchanged by the opposite category data, this is the required naturality square in $\mathcal D^{\mathrm{op}}$.

The remaining stability and ledger clauses are inherited without changing raw data. Identity components and vertical component composites are the same raw
component histories, with their endpoints read in the opposite target category; the naturality argument above applies to each displayed source morphism. The
ledger policy of \autoref{def:nattrans-ledger-policy} therefore records the same functor ledgers, component morphism ledgers, and naturality derivations,
now indexed by the two opposite-functor certificates.
\end{proof}
