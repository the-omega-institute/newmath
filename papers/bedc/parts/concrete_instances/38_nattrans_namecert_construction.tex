\chapter{A Concrete Naming Certificate for \(\NatTransUp\)}
\label{ch:concrete-instances-nattrans-namecert}

The natural transformation interface \(\NatTransUp\) has certificate obligations between two functors from \autoref{ch:concrete-instances-functor-namecert} sharing source and target categories. The carrier is, for each object \(X\) of the source category, a morphism in the target category from the first functor's image of \(X\) to the second functor's image of \(X\), satisfying the naturality square under the codomain category's morphism classifier. It is not a primitive 2-cell; it is admitted only as an object-indexed family of morphisms with the naturality field.

\section{Carrier definition}

\begin{definition}[NatTrans carrier]
\label{def:nattrans-carrier}
Use an object-indexed family of codomain morphisms \(\eta_X:F(X)\to G(X)\) plus naturality witnesses for every source morphism.
\end{definition}
\leandef{BEDC.Derived.NatTransUp.NatTransPrefixComponentCarrier}

\begin{theorem}[NatTrans prefix component empty identity exactness]
\label{thm:nattrans-prefix-component-carrier-empty-identity-iff}
For prefix components over unary histories, the empty component from
$\mathsf{append}(p,a)$ to $\mathsf{append}(q,a)$ exists exactly when $p$, $q$,
and $a$ are unary histories and $\HSame(p,q)$.
\end{theorem}
\begin{proof}
Unfold the prefix component carrier to its category hom-carrier over
$\mathsf{append}(p,a)$ and $\mathsf{append}(q,a)$. Category empty-identity
exactness identifies the two prefixed endpoints; cancellation of the shared
suffix $a$ recovers $\HSame(p,q)$. Conversely, unary append closure supplies
the two prefixed endpoint carriers, and the equality of $p$ and $q$ transports
the category empty identity carrier to the component carrier.
\end{proof}
\leanchecked{BEDC.Derived.NatTransUp.NatTransPrefixComponentCarrier\_empty\_identity\_iff}

\begin{theorem}[NatTrans empty identity prefix transitivity]
\label{thm:nattrans-prefix-component-empty-identity-prefix-trans}
For prefix components over unary histories, if the empty component identifies
\(\mathsf{append}(p,a)\) with \(\mathsf{append}(q,a)\) and the empty component
also identifies \(\mathsf{append}(q,a)\) with \(\mathsf{append}(r,a)\), then
\(\HSame(p,r)\).
\end{theorem}
\begin{proof}
Apply empty-identity exactness to the two component witnesses. The first
component gives \(\HSame(p,q)\), and the second gives \(\HSame(q,r)\).
Transitivity of history sameness identifies \(p\) and \(r\).
\end{proof}
\leanchecked{BEDC.Derived.NatTransUp.NatTransPrefixComponentCarrier\_empty\_identity\_prefix\_trans}

\begin{theorem}[NatTrans prefix component vertical composition closure]
\label{thm:nattrans-prefix-component-carrier-vert-comp-closed}
For prefix components over unary histories, if \(\eta\) is a component from
\(\mathsf{append}(p,a)\) to \(\mathsf{append}(q,a)\), \(\theta\) is a component
from \(\mathsf{append}(q,a)\) to \(\mathsf{append}(r,a)\), and
\(\ContR(\eta,\theta,\theta\eta)\), then the composite component is a component
from \(\mathsf{append}(p,a)\) to \(\mathsf{append}(r,a)\).
\end{theorem}
\begin{proof}
Unpack the two prefix-component carrier witnesses. The outer prefix and object
unary witnesses come from the first component and the target prefix witness
comes from the second component. Category hom-carrier composition closure
composes the two component morphisms across the shared middle object
\(\mathsf{append}(q,a)\).
\end{proof}
\leanchecked{BEDC.Derived.NatTransUp.NatTransPrefixComponentCarrier\_vert\_comp\_closed}

\begin{theorem}[NatTrans prefix component vertical composition public readback]
\label{thm:nattrans-prefix-component-carrier-vert-comp-public-readback}
For prefix components over unary histories, if \(\eta\) is a component from
\(\mathsf{append}(p,a)\) to \(\mathsf{append}(q,a)\), \(\theta\) is a component
from \(\mathsf{append}(q,a)\) to \(\mathsf{append}(r,a)\), and
\(\ContR(\eta,\theta,\theta\eta)\), then the composite is a component from
\(\mathsf{append}(p,a)\) to \(\mathsf{append}(r,a)\), and every displayed
component over the same source prefix, target prefix, and object is \(\HSame\)
to that composite.
\end{theorem}
\begin{proof}
The vertical composition closure theorem supplies the displayed composite
component carrier. Any displayed component with the same prefixed source and
target unpacks to a category hom-carrier with the same endpoints. Category
morphism determinacy compares the underlying hom-carriers, yielding the public
\(\HSame\) readback for the composite.
\end{proof}
\leanchecked{BEDC.Derived.NatTransUp.NatTransPrefixComponentCarrier\_vert\_comp\_public\_readback}

\begin{theorem}[NatTrans prefix component vertical composition right factor]
\label{thm:nattrans-prefix-component-carrier-vert-comp-right-factor}
For prefix components over unary histories, if $\eta$ is a component from
$\mathsf{append}(p,a)$ to $\mathsf{append}(q,a)$,
$\ContR(\eta,\theta,\theta\eta)$ composes $\eta$ with $\theta$, and the
composite is displayed as a component from $\mathsf{append}(p,a)$ to
$\mathsf{append}(r,a)$, then $\theta$ is a component from
$\mathsf{append}(q,a)$ to $\mathsf{append}(r,a)$.
\end{theorem}
\begin{proof}
Unpack the left component and the displayed composite to their underlying
category hom-carriers. The category right-factor theorem recovers the missing
hom-carrier for $\theta$ between the middle endpoint
$\mathsf{append}(q,a)$ and the displayed target endpoint
$\mathsf{append}(r,a)$. The prefix carrier for $q$, the prefix carrier for
$r$, and the object carrier for $a$ reassemble the NatTrans prefix component
carrier for $\theta$.
\end{proof}
\leanchecked{BEDC.Derived.NatTransUp.NatTransPrefixComponentCarrier\_vert\_comp\_right\_factor}

\begin{theorem}[NatTrans prefix component vertical composition right identity closure]
\label{thm:nattrans-prefix-component-vert-comp-right-identity-closed}
For prefix components over unary histories, if $\eta$ is a component from
$\mathsf{append}(p,a)$ to $\mathsf{append}(q,a)$ and
$\ContR(\eta,\emp,\eta')$, then $\eta'$ is again such a component and
$\HSame(\eta',\eta)$.
\end{theorem}
\begin{proof}
The BEDC continuation right-unit witness $\ContR(\eta,\emp,\eta)$ and the
displayed continuation $\ContR(\eta,\emp,\eta')$ have the same source and
tail. Continuation determinacy gives $\HSame(\eta',\eta)$. Transport the
unpacked prefix-component carrier for $\eta$ along this history sameness to
obtain the component carrier for $\eta'$.
\end{proof}
\leanchecked{BEDC.Derived.NatTransUp.NatTransPrefixComponentCarrier\_vert\_comp\_right\_identity\_closed}

\begin{theorem}[NatTrans prefix component vertical composition associativity closure]
\label{thm:nattrans-prefix-component-carrier-vert-comp-assoc-closed}
For prefix components over unary histories, let \(\eta\), \(\theta\), and
\(\iota\) be consecutive components from \(p\) to \(q\), \(q\) to \(r\), and
\(r\) to \(s\) at the object \(a\). If \(\eta\theta\), \(\theta\iota\),
\((\eta\theta)\iota\), and \(\eta(\theta\iota)\) are supplied by the BEDC
continuation relation, then the two bracketed composites are components from
\(p\) to \(s\) at \(a\), and they are \(\HSame\).
\end{theorem}
\begin{proof}
Unpack the three prefix-component carrier witnesses to their underlying
category hom-carriers between
\(\mathsf{append}(p,a)\), \(\mathsf{append}(q,a)\),
\(\mathsf{append}(r,a)\), and \(\mathsf{append}(s,a)\). Category hom-carrier
associativity closure gives hom-carrier witnesses for both bracketed
composites and identifies them by \(\HSame\). Repackage the shared source
prefix, target prefix, and object unary witnesses around those two underlying
hom-carriers.
\end{proof}
\leanchecked{BEDC.Derived.NatTransUp.NatTransPrefixComponentCarrier\_vert\_comp\_assoc\_closed}

\begin{theorem}[NatTrans vertical composite target-prefix determinacy]

\begin{theorem}[Prefix identity naturality square]
\label{thm:nattrans-prefix-identity-naturality-square}
For the unary-continuation category, let \(p\) be unary and let \(f\) be a
hom-carrier from \(a\) to \(b\). If the two square composites are obtained by
composing \(f\) with the empty continuation on the left and on the right, then
both composites are hom-carriers from \(\mathsf{append}(p,a)\) to
\(\mathsf{append}(p,b)\), and the two composites are history-same.
\end{theorem}
\begin{proof}
The left empty continuation identifies the left composite with \(f\), and
the right empty continuation identifies the right composite with \(f\). The
given hom-carrier therefore supplies both unprefixed sides of the square.
Prefix hom-carrier preservation sends both sides across \(p\), and
transitivity of history sameness identifies the composites.
\end{proof}
\leanchecked{BEDC.Derived.NatTransUp.NatTransPrefixIdentity\_naturality\_square}

\begin{theorem}[Prefix identity component square closure]
\label{thm:nattrans-prefix-identity-component-square-closed}
For the unary-continuation category, if \(p\) and \(a\) are unary,
\(\ContR(\emp,\emp,id)\), \(\ContR(id,\emp,left)\), and
\(\ContR(\emp,id,right)\), then \(left\) and \(right\) are hom-carriers from
\(\mathsf{append}(p,a)\) to \(\mathsf{append}(p,a)\), and
\(\HSame(left,right)\).
\end{theorem}
\begin{proof}
The prefix identity hom-carrier theorem gives the carrier witness for \(id\)
at the prefixed object. The right unit continuation identifies \(left\) with
\(id\), and the left unit continuation identifies \(right\) with \(id\).
Transporting the identity carrier witness across these equalities gives both
legs of the component square, and transitivity of \(\HSame\) closes the
square.
\end{proof}
\leanchecked{BEDC.Derived.NatTransUp.NatTransPrefixIdentity\_identity\_component\_square\_closed}

\begin{theorem}[NatTrans prefix component identity semantic name certificate]
\label{thm:nattrans-prefix-component-identity-semantic-namecert}
For unary histories $p,a$, the identity prefix-component predicate

\end{proof}

\section{The certificate}

\begin{definition}[NatTrans source specification]
\label{def:nattrans-source-specification}
The source is two certified functors sharing the same certified source and target categories.
\end{definition}

\begin{definition}[NatTrans pattern specification]
\label{def:nattrans-pattern-specification}
The pattern is component-family formation with one codomain morphism per source object.
\end{definition}

\begin{definition}[NatTrans classifier specification]
\label{def:nattrans-classifier-specification}
The classifier relates natural transformations componentwise by the target category morphism classifier.
\end{definition}

\begin{definition}[NatTrans stability certificate]
\label{def:nattrans-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item componentwise reflexivity
  \item componentwise transitivity
  \item naturality square validity for every source morphism
  \item stability of naturality under component classifier replacement
  \item vertical composition closure
  \item identity natural transformation closure
\end{enumerate}
\end{definition}

\begin{definition}[NatTrans ledger policy]
\label{def:nattrans-ledger-policy}
Record the functor ledgers, component morphism ledgers, and naturality derivations.
\end{definition}

\begin{definition}[\(\NatTransUp\) certificate obligations]
\label{def:nattrans-certificate-obligations}
The carrier, source, pattern, classifier, stability, and ledger data listed above are the obligations required for the interface name \(\NatTransUp\).
\end{definition}

\paragraph{Primitive boundary.}
\(\NatTransUp\) is not a BEDC primitive. It is admitted only when those obligations are supplied.

\begin{remark}[Zero-axiom encoding note]
\label{rem:nattrans-zero-axiom-note}
Avoids funext by componentwise classifier sameness.
\end{remark}
