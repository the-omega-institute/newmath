\begin{definition}[SOne carrier]

Use pairs $(x,y)$ of Bishop real witnesses with a certified unit equation witness $x^2 + y^2 \approx 1$ under $\RealClassifier$.
\end{definition}

\begin{theorem}[S1Up root obligation surface]
\label{thm:s1-root-obligation-surface}
An accepted $\SOneUp$ carrier over $\mathsf{BHist}$ consists of two $\RealUp$
coordinate rows, the product-source row inherited from $\ProdUp$, the
$\RealUp$ classifier transports for both coordinates, a displayed radius row
asserting that $x^2+y^2$ is classified with the $\RealUp$ unit, and a
$\NameCert$ ledger recording the $\RealUp$ and product witnesses.

The radius row is a classifier obligation over the displayed real-coordinate
data. It is not a host equality and it does not add a primitive circle carrier
outside the real/product source rows.
\end{theorem}
\leanchecked{BEDC.Derived.S1Up.SOneHistoryCarrier\_root\_obligation\_surface}
\begin{proof}
Unpack the two $\RealUp$ coordinate certificates. They supply the two
$\mathsf{BHist}$ component rows and the componentwise classifier transports
used by the circle carrier.

Use the $\ProdUp$ product carrier to pair those rows into one source packet.
The resulting packet records the real-coordinate witnesses and the product
source row in the same $\mathsf{BHist}$ ledger.

Finally record the radius condition $x^2+y^2$ against the $\RealUp$ unit as a
displayed classifier obligation. Since the condition is stored as a row in the
$\NameCert$ ledger, the carrier exposes only the real, product, radius, and
ledger witnesses described above.
\end{proof}

\mathsf{RatHistoryCarrier}(d_x)\land
\hsame(y,\Eone(d_y))\land

\end{theorem}
\begin{proof}

history carriers. The product carrier supplies the two component histories and
their continuation ledger into $h$. Each real-constant carrier then exposes a
rational denominator tail together with the displayed $\Eone$ presentation of

$\mathsf{Cont}(x,y,p)$ and $\mathsf{Cont}(x',y',p)$; after transporting along
$\hsame(x,x')$, continuation left-cancellation identifies $y$ with $y'$. The
equation components are both classified against the fixed
$\mathsf{SOneUnitHistory}$ endpoint, so unit-equation determinacy identifies
each equation witness with that endpoint, and transitivity gives
$\hsame(e,e')$.
\end{proof}

\begin{theorem}[SOne shared-point component equivalence]
\label{thm:sone-shared-point-component-iff}
For histories $x,x',y,y',e,e',p$, two concrete $\SOneUp$ carrier witnesses over
the same displayed point ledger have the following shared-point component
equivalence:
$$
\mathsf{SOneHistoryCarrier}(x,y,e,p)
\Longrightarrow
\mathsf{SOneHistoryCarrier}(x',y',e',p)
\Longrightarrow
(\hsame(x,x') \Longrightarrow \hsame(y,y')\land\hsame(e,e'))
\land
(\hsame(y,y') \Longrightarrow \hsame(x,x')\land\hsame(e,e')).
$$
\end{theorem}
\begin{proof}
Assume the two displayed carrier witnesses
$\mathsf{SOneHistoryCarrier}(x,y,e,p)$ and
$\mathsf{SOneHistoryCarrier}(x',y',e',p)$.

For the first branch, suppose $\hsame(x,x')$. Apply
\autoref{thm:sone-history-carrier-coordinate-pair-right-deterministic} to the
two carrier witnesses and the first-coordinate equality. The hypotheses use the
same displayed point ledger $p$, so right determinacy recovers both
$\hsame(y,y')$ and $\hsame(e,e')$; the equation component is recovered against
the fixed $\mathsf{SOneUnitHistory}$ ledger exposed by the carrier.

For the second branch, suppose $\hsame(y,y')$. Apply
\autoref{thm:sone-history-carrier-coordinate-pair-deterministic} to the same
two carrier witnesses and the second-coordinate equality. The shared displayed
point ledger is again $p$, so coordinate-pair determinacy recovers
$\hsame(x,x')$ and the same equation-ledger comparison $\hsame(e,e')$ against
$\mathsf{SOneUnitHistory}$.

Pair the two implications.
\end{proof}
\leanchecked{BEDC.Derived.S1Up.SOneHistoryCarrier\_shared\_point\_component\_iff}

\begin{definition}[SOne displayed completion readback]
\label{def:sone-displayed-completion-readback}
Let $\alpha$ be a displayed BEDC source map on $\BHist$. A row
$$
\mathsf{SOneDisplayedCompletionReadback}_{\alpha}(a,x,y,e,p,c)
$$
means that the source row $a$ is read through a concrete $\SOneUp$ carrier and
that the $\CompletionUp$ endpoint is represented by the same visible point
ledger:
$$
\begin{aligned}
\mathsf{SOneDisplayedCompletionReadback}_{\alpha}(a,x,y,e,p,c)
:\Longleftrightarrow{}&
\mathsf{SOneHistoryCarrier}(x,y,e,p)\land{}\\
&\hsame(\alpha(a),p)\land \hsame(c,p).
\end{aligned}
$$
The predicate has no field for a host quotient representative: the only
completion readback data are the displayed point ledger $p$ and the two
$\hsame$ comparisons tying $p$ to the source image and to the public endpoint.
\end{definition}

\begin{proposition}[SOne completion readback preserves source separation]
\label{prop:sone-completion-readback-source-separation}
Let $\alpha$ be a displayed BEDC source map on $\BHist$ whose additive
and cancellation rows supply the injectivity judgment
$$
\forall u,v:\BHist.\;
\hsame(\alpha(u),\alpha(v))\Longrightarrow\hsame(u,v).
$$
Then displayed $\SOneUp$-$\CompletionUp$ readback is injective on source rows
and therefore preserves source separation:
$$
\begin{aligned}
&\mathsf{SOneDisplayedCompletionReadback}_{\alpha}(a,x,y,e,p,c)\land
\mathsf{SOneDisplayedCompletionReadback}_{\alpha}(a',x',y',e',p',c')\\
&\Longrightarrow
\bigl(\hsame(c,c')\Longrightarrow\hsame(a,a')\bigr)\land
\bigl((\hsame(a,a')\Longrightarrow\bot)\Longrightarrow
(\hsame(c,c')\Longrightarrow\bot)\bigr).
\end{aligned}
$$
\end{proposition}
\begin{proof}
Unpack the two displayed completion-readback rows of
\autoref{def:sone-displayed-completion-readback}. The first row gives
$\mathsf{SOneHistoryCarrier}(x,y,e,p)$, $\hsame(\alpha(a),p)$, and
$\hsame(c,p)$. The second gives
$\mathsf{SOneHistoryCarrier}(x',y',e',p')$, $\hsame(\alpha(a'),p')$, and
$\hsame(c',p')$.

By \autoref{def:s1-carrier} and
\autoref{thm:sone-history-carrier-public-readback}, the ledgers $p$ and $p'$
are the public point histories exposed by concrete $\SOneUp$ carriers: each is
read from the displayed real-coordinate carrier, the unit-equation ledger, and
the BEDC continuation row. Thus the endpoint comparisons above are comparisons
with visible $\BHist$ representatives, not comparisons inside an
external completion quotient.

Assume $\hsame(c,c')$. Compose the displayed comparisons by symmetry and
transitivity of $\hsame$:
$$
\alpha(a)\;\hsame\;p\;\hsame\;c\;\hsame\;c'\;\hsame\;p'\;\hsame\;\alpha(a').
$$
This gives $\hsame(\alpha(a),\alpha(a'))$. Applying the injectivity judgment
for $\alpha$ gives $\hsame(a,a')$, proving the readback-injectivity clause.

For source separation, suppose $\hsame(a,a')\Longrightarrow\bot$ and suppose
$\hsame(c,c')$. The injectivity clause just proved yields $\hsame(a,a')$, which
contradicts the displayed source-separation hypothesis. Hence
$\hsame(c,c')\Longrightarrow\bot$. The argument uses only the concrete
$\SOneUp$ carrier readback, $\hsame$ symmetry and transitivity, and the
source-map injectivity row supplied by additive cancellation.
\end{proof}

\begin{definition}[SOne visible finite-rank and prime-support ledgers]
\label{def:sone-visible-finite-rank-prime-support-ledgers}
Fix a concrete $\SOneUp$ history-carrier witness
$\mathsf{SOneHistoryCarrier}(x,y,e,p)$. A \emph{visible $\SOneUp$ bridge
ledger} is the BEDC ledger surface obtained from the carrier fields of
\autoref{thm:s1-root-obligation-surface}: the two $\RealUp$ coordinate rows,
the $\ProdUp$ product row, the unit-equation classifier row, and the displayed
point-continuation row for $p$.

A \emph{bounded-rank certificate} on this visible ledger is a pair
$(b,R_{\le b})$ where $b$ is a displayed BEDC rank row and $R_{\le b}$ asserts
that every prime-support row read from the visible bridge ledger factors
through rank at most $b$.

A \emph{cofinal prime-support evidence row} on the same visible ledger is a
predicate $C_{\mathrm{prime}}$ such that, for every displayed BEDC rank row $b$,
there is a visible prime-support row $q_b$ in the point-continuation ledger of
$p$ whose support does not factor through rank at most $b$.

A \emph{finite-rank host homologization certificate admitted through the
$\SOneUp$ carrier/readback bridge} is a host-side finite-rank homologization
claim whose theorem-preservation and no-host-leak fields translate its public
$\SOneUp$ consequence to a bounded-rank certificate on the visible ledger,
rather than to a host equality or an unrecorded host quotient.
\end{definition}

\begin{lemma}[SOne finite-rank host certificate induces bounded visible rank]
\label{lem:sone-finite-rank-host-induces-bounded-visible-rank}
For a concrete carrier witness $\mathsf{SOneHistoryCarrier}(x,y,e,p)$, every
finite-rank host homologization certificate admitted through the $\SOneUp$
carrier/readback bridge induces a bounded-rank certificate on the visible
$\SOneUp$ bridge ledger of
\autoref{def:sone-visible-finite-rank-prime-support-ledgers}.
\end{lemma}
\begin{proof}
The carrier surface is fixed by \autoref{thm:s1-root-obligation-surface}: the
visible data are the two real-coordinate rows, the product row, the unit row,
and the point-continuation ledger. The public readback theorem
\autoref{thm:sone-history-carrier-public-readback} exposes exactly these
BEDC-side rows and no additional host carrier.

An admitted host homologization certificate can therefore reach the public
$\SOneUp$ surface only through the standard bridge discipline: its preserved
host claim must be listed in the theorem-preservation field, and the
no-host-leak field of \autoref{def:acceptance-bridge-no-host-leak} forbids the
host's finite-rank object, host quotient, or host equality from becoming a
BEDC export. Hence the finite-rank content has only one BEDC-visible form: an
explicit displayed rank bound $b$ together with a ledger assertion that all
prime-support rows visible in the $\SOneUp$ point ledger factor through rank at
most $b$. This is precisely a bounded-rank certificate on the visible bridge
ledger.
\end{proof}

\begin{theorem}[SOne finite-rank host homologization obstruction]
\label{thm:sone-finite-rank-host-homologization-obstruction}
Let $\mathsf{SOneHistoryCarrier}(x,y,e,p)$ be a concrete $\SOneUp$
history-carrier witness. If the visible $\SOneUp$ bridge ledger carries
cofinal prime-support evidence in the sense of
\autoref{def:sone-visible-finite-rank-prime-support-ledgers}, then no
finite-rank host homologization certificate can be admitted through the
$\SOneUp$ carrier/readback bridge for that ledger.
\end{theorem}
\begin{proof}
Suppose that such a finite-rank host homologization certificate were admitted.
By \autoref{lem:sone-finite-rank-host-induces-bounded-visible-rank}, it induces
a bounded-rank certificate $(b,R_{\le b})$ on the visible $\SOneUp$ bridge
ledger. Thus every visible prime-support row in the point-continuation ledger
of $p$ must factor through rank at most $b$.

Apply the cofinal prime-support evidence to this same displayed bound $b$. It
provides a visible prime-support row $q_b$ in the point-continuation ledger of
$p$ whose support does not factor through rank at most $b$.

The row $q_b$ is visible in the same ledger to which $R_{\le b}$ applies, so
$R_{\le b}$ says that $q_b$ factors through rank at most $b$. Cofinality says
that the same row does not factor through rank at most $b$. This contradiction
blocks the induced bounded-rank certificate, and therefore blocks the admitted
finite-rank host homologization certificate that would induce it.
\end{proof}
