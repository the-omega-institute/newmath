\section{The certificate}

\begin{definition}[SOne source specification]
\label{def:s1-source-specification}
The source is the $\ProdUp$ pair of two $\RealUp$ witnesses together with the real-algebra data needed to form $x^2 + y^2$ and the displayed unit witness.
\end{definition}
\leandef{BEDC.Derived.S1Up.SOneSourceSpec}
\leanchecked{BEDC.Derived.S1Up.SOneSourceSpec\_history\_carrier\_fields}

\begin{definition}[SOne pattern specification]
\label{def:s1-pattern-specification}
The pattern is unit-circle formation from a real pair and a certified proof object for the equation $x^2 + y^2 \approx 1$.
\end{definition}

\begin{definition}[SOne classifier specification]
\label{def:s1-classifier-specification}
The concrete $\mathsf{BHist}$ classifier relates two
$\mathsf{SOneHistoryCarrier}$ witnesses $(x,y,e,p)$ and $(x',y',e',p')$
exactly when $\hsame(x,x')$ and $\hsame(y,y')$ hold componentwise. Under those
component witnesses, the equation and point ledgers are determined by
\autoref{thm:s1-component-classifier-ledger-determinacy}, and they are
transported by \autoref{thm:s1-equation-witness-transport} and
\autoref{thm:s1-carrier-hsame-transport}.
\end{definition}

\begin{definition}[SOne component classifier relation]
\label{def:sone-component-classifier-relation}
For histories $x,y,e,p,x',y',e',p'$, the concrete relation
$\mathsf{SOneComponentClassifier}((x,y,e,p),(x',y',e',p'))$ means the
conjunction:
$$
\mathsf{SOneHistoryCarrier}(x,y,e,p)
\wedge
\mathsf{SOneHistoryCarrier}(x',y',e',p')
\wedge
\hsame(x,x') \wedge \hsame(y,y').
$$
Thus the classifier exposes exactly the two real component comparisons. The
equation ledger and the point ledger are not classifier fields; they are
recovered by \autoref{thm:s1-component-classifier-ledger-determinacy}.
\end{definition}
\leandef{BEDC.Derived.S1Up.SOneComponentClassifier}

\begin{theorem}[SOne component classifier stability]
\label{thm:sone-component-classifier-stability}
On the carrier $\mathsf{SOneHistoryCarrier}$, the relation
$\mathsf{SOneComponentClassifier}$ supplies the stability row required by the
semantic $\NameCert$ for $\SOneUp$. Concretely:
\begin{enumerate}
  \item it is reflexive, symmetric, and transitive by the equivalence theorem
  for $\hsame$ on the two exposed real components;
  \item any classifier witness projects the two carrier rows and determines
  $\hsame(e,e')$ and $\hsame(p,p')$ from its exposed component comparisons;
  \item the carrier transports from $(x,y,e,p)$ to $(x',y',e',p')$ along the
  resulting four coordinate $\hsame$ witnesses.
\end{enumerate}
\end{theorem}
\leanchecked{BEDC.Derived.S1Up.SOneComponentClassifier\_stability}
\begin{proof}
Reflexivity, symmetry, and transitivity are inherited componentwise from
\autoref{thm:history-sameness-equivalence}. Reflexivity uses the two diagonal
component witnesses; symmetry reverses both exposed witnesses; transitivity
composes the two exposed witnesses through the middle carrier.

For the hidden ledgers, unpack the classifier witness into four fields: the
source carrier row $\mathsf{SOneHistoryCarrier}(x,y,e,p)$, the target carrier row
$\mathsf{SOneHistoryCarrier}(x',y',e',p')$, and the exposed component witnesses
$\hsame(x,x')$ and $\hsame(y,y')$. Apply
\autoref{thm:s1-component-classifier-ledger-determinacy} to these four fields;
it returns exactly the equation witness $\hsame(e,e')$ and the point witness
$\hsame(p,p')$.

The equation ledger is stable by
\autoref{thm:s1-equation-witness-transport}. With the component ledgers and the
determined point ledger, \autoref{thm:s1-carrier-hsame-transport} then
transports the whole carrier from $(x,y,e,p)$ to $(x',y',e',p')$.

The certificate fit has ten fields: the traditional target is unit-circle
component classification; the BEDC source is $\mathsf{BHist}$ with
$\mathsf{RealConstantHistoryCarrier}$,
$\mathsf{RealConstantHistoryClassifier}$, $\ProdUp$, and $\mathsf{Cont}$; the
carrier is $\mathsf{SOneHistoryCarrier}$; the classifier is
$\mathsf{SOneComponentClassifier}$; the stability obligations are equivalence
plus carrier transport; the ledger data are the equation and point witnesses;
the core theorem is this stability theorem; exactness is hidden-ledger
determinacy; the standard bridge is deferred; and the status is
$\mathsf{ProofTarget}$.
\end{proof}

\begin{definition}[SOne stability certificate]
\label{def:s1-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item reflexivity from reflexivity of the two real components and identity transport of the equation witness
  \item symmetry by symmetry of the two real components and reverse transport of the equation witness
  \item transitivity by transitivity of the two real components and composition of equation transport
  \item compatibility with the $\ProdUp$ classifier over $\RealUp \times \RealUp$
  \item preservation of the certified equation $x^2 + y^2 \approx 1$
\end{enumerate}
\end{definition}

\begin{theorem}[SOne equation-witness transport]
\label{thm:s1-equation-witness-transport}
For histories $x,y,e,e',p$, the concrete $\SOneUp$ history carrier is stable
under replacement of the displayed unit-equation witness along the equation
ledger:
$$
\mathsf{SOneHistoryCarrier}(x,y,e,p)
\Longrightarrow
\hsame(e,e')
\Longrightarrow
\mathsf{SOneHistoryCarrier}(x,y,e',p).
$$
\end{theorem}
\begin{proof}
Unpack $\mathsf{SOneHistoryCarrier}(x,y,e,p)$ into the two
$\mathsf{RealConstantHistoryCarrier}$ component fields, the
$\mathsf{RealConstantHistoryClassifier}(e,\mathsf{SOneUnitHistory})$ field,
and the continuation witness $\mathsf{Cont}(x,y,p)$.

The equation ledger gives $\hsame(e,e')$. Combine it with
$\hsame(\mathsf{SOneUnitHistory},\mathsf{SOneUnitHistory})$ by reflexivity and
apply the endpoint-transport theorem for the real-constant history classifier.
This produces
$\mathsf{RealConstantHistoryClassifier}(e',\mathsf{SOneUnitHistory})$ while the
two component carriers and the continuation ledger remain unchanged.

Repack the four fields into $\mathsf{SOneHistoryCarrier}(x,y,e',p)$. The
certificate fit is the traditional target unit-circle equation transport; the
BEDC source $\mathsf{BHist}$ with $\mathsf{RealConstantHistoryClassifier}$; the
carrier $\mathsf{SOneHistoryCarrier}$; the classifier $\hsame$ on the equation
ledger; the stability obligation equation-witness transport; the ledger
transported unit-equation proof; the core theorem equation-witness replacement;
the exactness claim that the same $\SOneUp$ carrier is preserved; the standard
bridge deferred outside this theorem; and the status $\mathsf{ProofTarget}$.
\end{proof}
\leanchecked{BEDC.Derived.S1Up.SOneHistoryCarrier\_equation\_witness\_transport}

\begin{theorem}[SOne carrier hsame transport]
\label{thm:s1-carrier-hsame-transport}
For histories $x,y,e,p,x',y',e',p'$, the concrete $\SOneUp$ history carrier is
stable under componentwise $\hsame$ transport on all exposed carrier
coordinates:
$$
\mathsf{SOneHistoryCarrier}(x,y,e,p)
\Longrightarrow
\hsame(x,x')
\Longrightarrow
\hsame(y,y')
\Longrightarrow
\hsame(e,e')
\Longrightarrow
\hsame(p,p')
\Longrightarrow
\mathsf{SOneHistoryCarrier}(x',y',e',p').
$$
\end{theorem}
\begin{proof}
Unpack $\mathsf{SOneHistoryCarrier}(x,y,e,p)$ into the two
$\mathsf{RealConstantHistoryCarrier}$ component fields, the
$\mathsf{RealConstantHistoryClassifier}(e,\mathsf{SOneUnitHistory})$ field, and
the continuation witness $\mathsf{Cont}(x,y,p)$.

Transport the first and second real-constant carrier fields along
$\hsame(x,x')$ and $\hsame(y,y')$. Each step composes the exposed endpoint
equality of the real-constant carrier with the supplied history equality, so the
same real-constant carrier predicate is obtained at $x'$ and at $y'$.

Transport the equation classifier by applying
$\mathsf{RealConstantHistoryClassifier\_endpoint\_transport}$ to $\hsame(e,e')$
and the reflexive witness
$\hsame(\mathsf{SOneUnitHistory},\mathsf{SOneUnitHistory})$. This gives
$\mathsf{RealConstantHistoryClassifier}(e',\mathsf{SOneUnitHistory})$.

Transport the continuation ledger by the kernel continuation $\hsame$ transport
theorem, using the component witnesses for $x$, $y$, and $p$. The result is
$\mathsf{Cont}(x',y',p')$.

Repack the two transported real-constant carriers, the transported equation
classifier, and the transported continuation ledger into
$\mathsf{SOneHistoryCarrier}(x',y',e',p')$. The certificate fit is the
traditional target unit-circle carrier transport; the BEDC source
$\mathsf{BHist}$ with real-constant carrier, real-constant classifier, and
$\mathsf{Cont}$; the carrier $\mathsf{SOneHistoryCarrier}$; the classifier
$\hsame$ on the exposed coordinates; the stability obligation carrier
transport; the ledger transported endpoints and equation proof; the core
theorem this carrier-transport statement; the exactness claim that the same
carrier predicate is preserved; the standard bridge deferred outside this
theorem; and the status $\mathsf{ProofTarget}$.
\end{proof}
\leanchecked{BEDC.Derived.S1Up.SOneHistoryCarrier\_coordinate\_transport}


\begin{theorem}[SOne transported public readback]
\label{thm:s1-public-readback-transport}
For histories $x,y,e,p,x',y',e',p'$, a concrete $\SOneUp$ history carrier
and componentwise transported BEDC history data expose the public readback at
the transported tuple:
$$
\mathsf{SOneHistoryCarrier}(x,y,e,p)
\Longrightarrow
\hsame(x,x')
\Longrightarrow
\hsame(y,y')
\Longrightarrow
\hsame(e,e')
\Longrightarrow
\hsame(p,p')
\Longrightarrow
\mathsf{SOneProductHistoryCarrier}(p')\land
\hsame(e',\mathsf{SOneUnitHistory})\land
\exists d_x,d_y.\;
\hsame(x',\Eone(d_x))\land
\mathsf{RatHistoryCarrier}(d_x)\land
\hsame(y',\Eone(d_y))\land
\mathsf{RatHistoryCarrier}(d_y)\land
\ContR(x',y',p').
$$
\end{theorem}
\begin{proof}
First apply \autoref{thm:s1-carrier-hsame-transport} to the displayed carrier
and the four supplied $\hsame$ witnesses. This produces the transported carrier
$\mathsf{SOneHistoryCarrier}(x',y',e',p')$.

Next apply \autoref{thm:sone-history-carrier-public-readback} to that
transported carrier. Its conclusion is exactly the product carrier at $p'$, the
unit-equation readback for $e'$, the two rational component witnesses for $x'$
and $y'$, and the transported continuation ledger.

The ten-field fit is: traditional target transported unit-circle public
readback; BEDC source $\SOneUp$ over $\mathsf{BHist}$, $\RealUp$, $\ProdUp$, and
$\mathsf{Cont}$; carrier $\mathsf{SOneHistoryCarrier}$; classifier coordinate
$\hsame$; stability coordinate transport; ledger transported equation and point
witnesses; core theorem this readback; exactness public carrier and
unit-equation exposure; standard bridge deferred; status $\mathsf{ProofTarget}$.
\end{proof}

\begin{theorem}[SOne semantic NameCert field assembly]
\label{thm:sone-semantic-namecert-field-assembly}
The concrete carrier $\mathsf{SOneHistoryCarrier}$, the component classifier
relation of \autoref{def:sone-component-classifier-relation}, the source and
pattern specifications of \autoref{def:s1-source-specification} and
\autoref{def:s1-pattern-specification}, the stability certificate of
\autoref{def:s1-stability-certificate}, and the ledger policy of
\autoref{def:s1-ledger-policy} supply the five semantic certificate rows needed
for the $\SOneUp$ $\NameCert$: carrier and public readback, hidden
equation/point ledger determinacy, equation replacement, full coordinate
carrier transport, and transported public readback.
\end{theorem}
\begin{proof}
The carrier and public-readback row is
\autoref{thm:sone-history-carrier-public-readback}. It reads a concrete
$\mathsf{SOneHistoryCarrier}(x,y,e,p)$ as a product carrier, the fixed
unit-equation history, rational witnesses for $x$ and $y$, and the continuation
ledger at $p$.

For the classifier row, unpack
\autoref{def:sone-component-classifier-relation}. A classifier witness supplies
the source carrier row, the target carrier row, and the two component witnesses
$\hsame(x,x')$ and $\hsame(y,y')$. Applying
\autoref{thm:s1-component-classifier-ledger-determinacy} to these four projected
fields determines the hidden ledgers $\hsame(e,e')$ and $\hsame(p,p')$.

The stability row first replaces the equation witness through
\autoref{thm:s1-equation-witness-transport}. With the two public component
witnesses and the two hidden ledgers now in hand,
\autoref{thm:s1-carrier-hsame-transport} transports the full concrete carrier
from $(x,y,e,p)$ to $(x',y',e',p')$.

The final readback row is
\autoref{thm:s1-public-readback-transport}. It applies the transported carrier
to expose the product carrier at $p'$, the transported unit-equation witness,
the rational component witnesses at $x'$ and $y'$, and the continuation ledger
$\ContR(x',y',p')$. These five rows are exactly the semantic payload required by
the concrete $\NameCert$ data for $\SOneUp$.
\end{proof}

\begin{definition}[SOne public constructor packet]
\label{def:sone-public-constructor-packet}
A public constructor packet for $\SOneUp$ is a finite $\mathsf{BHist}$ packet
whose displayed rows are exactly two $\RealUp$ component histories $x$ and $y$,
the $\ProdUp$ point row $p$ for the ordered pair, the unit-equation witness row
$e$ certifying $x^{2}+y^{2}\approx 1$, the componentwise $\hsame(x,x')$ and
$\hsame(y,y')$ transport rows used by the classifier, the $\Cont$ ledger
carrying the point and equation endpoints, and the $\NameCert_{\SOneUp}$
provenance row. The packet has no field for an angle parameter, quotient
representative, connectedness witness, compactness witness, or standard circle
bridge.
\end{definition}

\begin{theorem}[SOne public constructor scope]
\label{thm:sone-public-constructor-scope}
Every accepted public constructor packet of
\autoref{def:sone-public-constructor-packet} reads only the two $\RealUp$
component rows, the $\ProdUp$ point row, the unit-equation witness, the
componentwise $\hsame$ classifier rows, the $\Cont$ continuation ledger, and the
$\NameCert_{\SOneUp}$ provenance row. Conversely, those displayed rows repack
into the same public constructor surface whenever the carrier and equation
transport obligations of \autoref{thm:sone-history-semantic-namecert} are
present.
\end{theorem}
\begin{proof}
Project the packet to its displayed rows. The definition lists only the two real
components, the product point, the equation witness, the componentwise
classifier transports, the continuation ledger, and the certificate provenance.

For the repacking direction, first use
\autoref{thm:s1-public-namecert-obligation-assembly} to align the displayed
rows with the public $\NameCert_{\SOneUp}$ fields. Then apply
\autoref{thm:sone-history-semantic-namecert} to supply the carrier, classifier,
stability, and ledger envelope.

The packet has no row for angle parametrisation, quotient representatives,
connectedness, compactness, or a standard bridge. A public read therefore cannot
leave the displayed BEDC rows, and the same rows reconstruct the public
constructor surface.
\end{proof}

\begin{theorem}[SOne public endpoint exactness]
\label{thm:sone-public-endpoint-exactness}
Let $P$ and $P'$ be accepted public constructor packets for $\SOneUp$. If their
two real component rows are related componentwise by $\hsame$, then the
unit-equation witness rows and product-point ledger rows exposed by $P$ and
$P'$ are the same public $\SOneUp$ endpoints under the classifier of
\autoref{def:sone-component-classifier-relation}. No hidden angle, quotient, or
host-circle endpoint can distinguish the two accepted packets.
\end{theorem}
\leanchecked{BEDC.Derived.S1Up.SOneComponentClassifier\_public\_endpoint\_exactness}
\begin{proof}
Read both packets through \autoref{thm:sone-public-constructor-scope}. The only
classifier data available are the two component rows and their componentwise
$\hsame$ witnesses.

Apply \autoref{thm:s1-component-classifier-ledger-determinacy}. The component
classifier determines the equation ledger and the point ledger from those two
real-component comparisons.

Finally transport the carrier with
\autoref{thm:s1-equation-witness-transport} and
\autoref{thm:s1-carrier-hsame-transport}. The transported equation witness and
product point are exactly the public endpoints read by the two packets, and no
undisplayed host-circle row remains available to separate them.
\end{proof}

\begin{theorem}[SOne public constructor compatibility]
\label{thm:s1-public-constructor-compatibility}
Every public $\SOneUp$ constructor row built from two $\RealUp$ component
rows, the $\ProdUp$ point readback, the unit-equation witness, componentwise
$\hsame$, and the $\Cont$ point ledger repacks into the same public
$\NameCert_{\SOneUp}$ surface. Conversely, every public constructor read from
$\NameCert_{\SOneUp}$ exposes only those component, product, equation,
transport, continuation, and provenance rows. It exposes no connectedness,
compactness, angle parametrisation, or standard circle bridge.
\end{theorem}
\begin{proof}
Start with the displayed public constructor row. Apply
\autoref{thm:s1-public-namecert-obligation-assembly} to identify the public
$\NameCert_{\SOneUp}$ fields as the two $\RealUp$ component rows, the $\ProdUp$
point row, the unit-equation witness, componentwise $\hsame$, and the $\Cont$
point ledger.

The semantic certificate is then supplied by
\autoref{thm:sone-history-semantic-namecert}. Its carrier, classifier, source,
stability, and ledger rows give the certificate envelope into which the public
constructor fields repack.

For the reverse read, project the public surface through
\autoref{thm:s1-product-real-ledger-obligation-surface},
\autoref{thm:s1-equation-transport-obligation-surface}, and
\autoref{thm:sone-component-classifier-public-readback}. These results recover
exactly the product readback, equation transport, component comparisons, and
continuation ledgers available to a public constructor.

No further row is present in the carrier or in the semantic certificate.
Therefore the public constructor interface is compatible with the unit-circle
equation and exposes only the listed BEDC history rows, while connectedness,
compactness, angle parametrisation, and standard circle bridges remain outside
the exported surface.
\end{proof}

\end{remark}

\begin{theorem}[SOne public constructor standard bridge boundary]
\label{thm:sone-public-constructor-standard-bridge}
Any standard-circle comparison admitted by this chapter consumes exactly the
public constructor packet rows of
\autoref{def:sone-public-constructor-packet}: two $\RealUp$ component histories,
the $\ProdUp$ point row, the unit-equation witness, componentwise $\hsame$
transports, the continuation ledger, and certificate provenance. Conversely,
those rows supply the BEDC side of a standard-circle bridge. The consumable
surface excludes angle quotients, trigonometric libraries, covering-space data,
winding-number data, and topological-group structure.
\end{theorem}
\leanchecked{BEDC.Derived.S1Up.SOneUp\_StdBridge}
\begin{proof}
First project an admitted standard-circle comparison to the public constructor
surface. By \autoref{def:sone-public-constructor-packet}, its exported rows are
exactly the two real component histories, product point row, unit-equation
witness, componentwise transports, continuation ledger, and provenance fields.

Next apply \autoref{thm:s1-public-constructor-compatibility} and
\autoref{thm:sone-history-semantic-namecert}. These identify the displayed rows
with the public $\NameCert_{\SOneUp}$ carrier, classifier, stability, and ledger
envelope. Conversely, the public readback and constructor interface of
\autoref{thm:sone-history-carrier-public-readback},
\autoref{thm:s1-public-readback-transport}, and
\autoref{thm:sone-component-classifier-public-readback} repacks them as the
BEDC side of a standard-circle comparison.

The cited projections never manufacture angle representatives, trigonometric
operations, covering-space lifts, winding-number ledgers, or a topological-group
law. Those structures remain outside the bridge boundary.
\end{proof}
