\chapter{A Concrete Naming Certificate for $\WeylGroupUp$}
\label{ch:concrete-instances-weylgroup-namecert}

The Weyl-group interface $\WeylGroupUp$ packages the finite reflection group generated by the simple roots of a root system. It depends on $\NameCert_{\RootSystemUp}$ (\autoref{ch:concrete-instances-rootsystem-namecert}) and $\NameCert_{\GroupUp}$ (\autoref{ch:concrete-instances-group-namecert}), and is the discrete symmetry datum on which character-theoretic certificates rest.

\section{Carrier definition}
\label{sec:weylgroup-carrier}

\begin{definition}[WeylGroup BHist source row]
\label{def:weylgroup-bhist-source-row}
The paper-side $\WeylGroupUp$ source row is the product of the
$\RootSystemUp$ finite-support carrier of
\autoref{def:rootsystem-bhist-certificate-surface} and the singleton
$\GroupUp$ history carrier of
\autoref{def:concrete-singleton-history-group-instance}. A Weyl word is a
$\mathsf{BHist}$ endpoint generated by a finite list of simple-reflection rows
from the root source, with word composition read by $\Cont$ and append in the
group source. Two Weyl words are compared by the restricted $\hsame$
classifier that preserves the carried root endpoint, the singleton
$\GroupUp$ word endpoint, and the finite root-row ledger.
\end{definition}

\section{The certificate}
\label{sec:weylgroup-certificate}

\begin{theorem}[WeylGroup RootSystem source obligation]
\label{thm:weylgroup-root-system-source-obligation}
A $\WeylGroupUp$ certificate source is exactly a scoped
$\RootSystemUp$ finite-support certificate together with the $\GroupUp$
carrier and classifier rows used to compose reflected-root histories. The
source exposes the finite root rows, the reflection-closure rows, the Cartan
ledger rows, and the singleton composition scope; it does not add a separate
host-level equality relation on roots or actions.
\end{theorem}
\begin{proof}
Project the $\RootSystemUp$ finite-support certificate. Its public rows give
the finite root carrier, reflection closure for displayed simple reflections,
and the Cartan integer endpoints used to read relation data.

Project the $\GroupUp$ source beside it. The carrier and classifier rows
provide the singleton composition scope in which reflected-root histories are
composed. Combining these projections gives exactly the source surface named
by the statement, with all equality routed through the displayed BEDC
classifiers.
\end{proof}

\begin{theorem}[WeylGroup simple reflection carrier obligation]
\label{thm:weylgroup-simple-reflection-carrier-obligation}
Each carried simple root row exports a simple-reflection generator history
whose endpoint is a $\mathsf{BHist}$ group-carried action on root rows. The
generator is carried by the $\GroupUp$ composition surface, and its action on
roots is transported by the $\RootSystemUp$ reflection-closure row.
\end{theorem}
\begin{proof}
Start with a carried simple root row from the finite root support of
$\RootSystemUp$. The reflection-closure row supplies the reflected root
endpoint for every carried input root, so the simple reflection is a displayed
operation on root rows rather than an external map.

Transport that displayed operation through the $\GroupUp$ carrier. The
singleton composition scope packages the reflected-root endpoint as a
$\mathsf{BHist}$ action history, and the group classifier records that this
history is carried by the certified group surface.
\end{proof}

\begin{theorem}[WeylGroup Coxeter ledger obligation]
\label{thm:weylgroup-coxeter-ledger-obligation}
The $\WeylGroupUp$ ledger records finite generator words, Cartan integer
endpoints, and Coxeter relation rows without quotienting by host equality.
Relation witnesses are packaged as $\mathsf{BHist}$ rows and descend through
$\hsame$ classifier transport.
\end{theorem}
\begin{proof}
Read finite support from the $\RootSystemUp$ source. Since the simple roots
are displayed as finite rows, the generator words are finite lists of carried
simple-reflection histories and have no need for an ambient free group outside
the certificate.

Read the Cartan endpoints from the same source and attach the Coxeter relation
rows to the corresponding finite words. The relation witnesses are then
transported through $\hsame$ and the public classifiers, so the ledger records
classified rows rather than quotient classes imposed by the host theory.
\end{proof}

\begin{theorem}[WeylGroup action classifier stability obligation]
\label{thm:weylgroup-action-classifier-stability-obligation}
If two generator words are related by the Coxeter ledger classifier, then
their induced actions on carried roots are related by the $\RootSystemUp$
public classifier. The stability is computed inside the displayed word ledger
using simple-reflection carrier transport, the $\GroupUp$ associativity and
unit rows, and $\RootSystemUp$ classifier transport.
\end{theorem}
\begin{proof}
Induct over the displayed finite-word ledger. In the empty word case, the
$\GroupUp$ unit row identifies the induced action with the carried root row,
and the $\RootSystemUp$ classifier witnesses stability of that root.

In the generator step, use the simple-reflection carrier obligation to
transport the head generator across the carried root row. The induction
hypothesis transports the remaining word, and the $\GroupUp$ associativity
row rebrackets the composed action without leaving the displayed carrier.

For a Coxeter relation row, read the relation witness from the ledger and
transport it through $\hsame$. The resulting endpoints are compared by the
$\RootSystemUp$ public classifier, which gives the required stability of the
two induced root actions.
\end{proof}

\begin{theorem}[WeylGroup root-system group carrier row]
\label{thm:weylgroup-rootsystem-group-carrier-row}
The source row of \autoref{def:weylgroup-bhist-source-row} extracts the
$\WeylGroupUp$ carrier and classifier from exactly two displayed surfaces:
the finite nonzero $\RootSystemUp$ carrier with its public root classifier, and
the singleton $\GroupUp$ BHist carrier with its append operation, empty unit,
constant inverse, and classifier $R_{\emp}$. A carried Weyl endpoint is
therefore a root-system source row together with a singleton group word row,
and its classifier is the restricted $\hsame$ comparison on those two rows.
\end{theorem}
\begin{proof}
Project the root side from
\autoref{thm:rootsystem-nonzero-finiteness-obligation} and
\autoref{thm:rootsystem-carrier-classifier-obligation}. These rows supply the
finite root support, the inherited vector classifier, and the finite-set
membership classifier that can be transported through $\hsame$.

Project the word side from
\autoref{thm:group-singleton-history-semantic-namecert} and
\autoref{thm:concrete-singleton-history-group-laws}. They supply the singleton
history carrier, the append operation, the unit row, inverse closure, and the
group classifier over $\mathsf{BHist}$.

The Weyl carrier is the conjunction of these two source reads. The restricted
classifier is the NameCert classifier transport applied simultaneously to the
root row and the group word row, so no host-level equality of root-system
presentations or group words is introduced.
\end{proof}

\begin{theorem}[WeylGroup simple-reflection word closure row]
\label{thm:weylgroup-simple-reflection-word-closure-row}

simple-reflection generator $s_\alpha$ determines a carried Weyl word endpoint.
If two carried Weyl words are represented by endpoints $w_0$ and $w_1$, and
$\Cont(w_0,w_1,w_{01})$ records their append product, then $w_{01}$ is again a
carried Weyl word acting on carried roots by the displayed reflection closure
rows of $\RootSystemUp$.
\end{theorem}

\autoref{thm:rootsystem-reflection-closure-obligation} supplies, for carried
roots $\alpha$ and $\beta$, a carried root $s_\alpha(\beta)$ and the $\Cont$

Read the same generator as a singleton $\GroupUp$ word endpoint using

For a composite word, use continuation associativity from

\begin{theorem}[WeylGroup Coxeter word ledger row]
\label{thm:weylgroup-coxeter-word-ledger-row}
Coxeter relations between simple-reflection words in the
\autoref{def:weylgroup-bhist-source-row} surface are ledger rows, not
host-level equalities of words. For each displayed pair of simple roots
$\alpha,\beta$, the Cartan endpoint ledger of $\RootSystemUp$ records the
integer relation data that licenses the corresponding braid row, and the
$\GroupUp$ singleton word source records the two append spines being compared
under the restricted $\hsame$ classifier.
\end{theorem}
\leanchecked{BEDC.Derived.WeylGroupUp.WeylGroupCoxeterWordLedger\_row}

Read the relation parameter from
\autoref{thm:rootsystem-cartan-ledger-obligation}. The Cartan endpoint is an
$\IntUp$-carried history, and its transport is already stated as ledger
stability under the public root classifier.

Build the two braid-word spines by iterating the simple-reflection word row
from \autoref{thm:weylgroup-simple-reflection-word-closure-row}. The group
singleton laws and continuation associativity keep the two append spines
inside the singleton $\GroupUp$ word carrier while preserving the displayed
ledger entries for each generator occurrence.

The comparison is then a recorded classifier row between the two carried
spines. It names the Coxeter relation as BEDC provenance data from the Cartan
ledger and the two word ledgers; it does not quotient the Weyl-word carrier by
an external equality relation.
\end{proof}

\begin{theorem}[WeylGroup action classifier stability row]
\label{thm:weylgroup-action-classifier-stability-row}
The Weyl action classifier is stable under $\hsame$ on both inputs. If root
rows are transported by the $\RootSystemUp$ public classifier and Weyl words
are transported by the restricted $\GroupUp$ word classifier of
\autoref{def:weylgroup-bhist-source-row}, then the displayed action endpoints
on carried roots are transported by the same $\RootSystemUp$ classifier, with
all generator, append, Coxeter-ledger, and dependency rows preserved.
\end{theorem}

\begin{proof}
Transport the root input through
\autoref{thm:rootsystem-carrier-classifier-obligation}. This preserves the
finite-support row, the inherited vector representative, and the nonzero
witness used by the reflection closure theorem.

Transport the word input through
\autoref{thm:weylgroup-rootsystem-group-carrier-row}. The word classifier is
the restricted $\hsame$ row supplied by the singleton $\GroupUp$ source and
the NameCert classifier transport, so append products and unit insertions stay
within the same carried word scope.

Apply the simple-reflection closure row generator by generator, using
\autoref{thm:weylgroup-simple-reflection-word-closure-row}. Reassociate
composites by the continuation associativity and unit rows, and read Coxeter

\begin{theorem}[WeylGroup public NameCert export]
\label{thm:weylgroup-public-namecert-export}
The rows of \autoref{def:weylgroup-bhist-source-row},
\autoref{thm:weylgroup-rootsystem-group-carrier-row},
\autoref{thm:weylgroup-simple-reflection-word-closure-row},
\autoref{thm:weylgroup-coxeter-word-ledger-row}, and
\autoref{thm:weylgroup-action-classifier-stability-row} package the public
paper-side $\NameCert_{\WeylGroupUp}$ over the $\RootSystemUp$
finite-support classifier and the $\GroupUp$ singleton word classifier. The
export exposes the carried root input, the singleton word input, the action
endpoint, the Coxeter ledger rows, and the action-stability row as the public
Weyl-group certificate boundary, with no host word equality and no Coxeter
quotient.
\end{theorem}
\begin{proof}
Project the root-system component from
\autoref{def:weylgroup-bhist-source-row} and
\autoref{thm:weylgroup-rootsystem-group-carrier-row}. This supplies the
finite-support root carrier, its public classifier, and the endpoint
transport row inherited from $\RootSystemUp$.

Project the word component from the same source row and the singleton
$\GroupUp$ carrier. The simple-reflection word closure theorem supplies the
generator action rows, and the Coxeter word ledger theorem records braid data
as carried ledger entries rather than as an equality quotient on host words.

Finally apply the action classifier stability row. It transports carried
roots and carried words through the public classifiers already projected, so
the root-system carrier, group carrier, word closure, Coxeter ledger, and
action-stability rows repack exactly as the public
$\NameCert_{\WeylGroupUp}$ boundary. No additional equality principle or
Coxeter quotient is introduced.
\end{proof}

\begin{theorem}[WeylGroup finite reflection-group bridge]
\label{thm:weylgroup-standard-reflection-group-bridge}
The public $\NameCert_{\WeylGroupUp}$ surface of
\autoref{thm:weylgroup-public-namecert-export} carries a paper-side bridge
with the standard finite reflection-word presentation. In the forward
direction, every BEDC carried Weyl word from
\autoref{def:weylgroup-bhist-source-row} has a finite reflection-word
interpretation obtained by reading the $\RootSystemUp$ finite root rows and
composing the displayed simple reflections through the $\GroupUp$ singleton
word carrier. If two BEDC words are related by the Coxeter ledger rows, then
their interpreted reflection words induce the same action on carried roots
under the $\RootSystemUp$ public classifier.

In the reverse direction, every finite word in displayed simple reflections
whose letters are $\RootSystemUp$ root rows has a BEDC word endpoint. The
endpoint is obtained by the simple-reflection word closure row and by iterated
$\Cont$ append through the singleton $\GroupUp$ carrier, so the standard word
reads back into the same public $\NameCert_{\WeylGroupUp}$ boundary.
\end{theorem}
\begin{proof}
For the forward direction, read the BEDC word through
\autoref{thm:weylgroup-rootsystem-group-carrier-row}. This separates the
finite $\RootSystemUp$ root-row carrier from the singleton $\GroupUp$ word
carrier while preserving the restricted classifier on carried Weyl endpoints.

Next interpret each displayed generator by
\autoref{thm:weylgroup-simple-reflection-word-closure-row}. The resulting
finite list of simple reflections is composed inside the singleton word
carrier. When the input BEDC words are Coxeter-ledger-related,
\autoref{thm:weylgroup-coxeter-word-ledger-row} supplies the recorded braid
rows, and \autoref{thm:weylgroup-action-classifier-stability-row} transports
the two induced actions to the same $\RootSystemUp$ public classifier class.

For the reverse direction, start from a finite standard word whose letters
are displayed root rows. Apply
\autoref{thm:weylgroup-simple-reflection-word-closure-row} to the first
letter, then iterate the append clause using $\Cont$ for the remaining
letters. The root-system and group carrier rows from
\autoref{thm:weylgroup-rootsystem-group-carrier-row} keep every intermediate
endpoint inside the public source surface.

Finally package the interpreted word and the readback endpoint by
\autoref{thm:weylgroup-public-namecert-export}. The bridge therefore changes
only the paper-side reading between BEDC carried words and finite
reflection-words; it does not add a host word quotient or advance the Lean
verification axis.
\end{proof}

\begin{theorem}[WeylGroup root-system terminal standard bridge]
\label{thm:weylgroup-root-system-terminal-standard-bridge}
The public $\WeylGroupUp$ surface of
\autoref{thm:weylgroup-public-namecert-export} supplies a paper-level
$\StdBridge(\WeylGroupUp,T_W^\ast)$, where $T_W^\ast$ is the terminal
host-facing Weyl-word action on the displayed $\RootSystemUp$ support.
The encode field sends the unique host Weyl word to the BEDC empty singleton
group word over the carried root row. The decode field sends every carried
Weyl word to the unique host word. The decode-encode round trip is terminal
equality in $T_W^\ast$, and the encode-decode round trip is the restricted
$\GroupUp$ singleton classifier together with the $\RootSystemUp$ public
classifier. The theorem-preservation field is the finite list consisting of
the root-system group carrier row, simple-reflection word closure, Coxeter word
ledger, action classifier stability, and public NameCert export. The
no-host-leak field confines the host terminal object to this bridge theorem.
\end{theorem}
\begin{proof}
We check the six fields of \autoref{sec:acceptance-bridge-six-fields}.
\begin{enumerate}
\item The encode field is the constant map from the unique word of
$T_W^\ast$ to the empty singleton $\GroupUp$ word carried over a displayed
$\RootSystemUp$ row. The word endpoint is public by
\autoref{thm:group-singleton-terminal-standard-bridge}, and the root row is
public by \autoref{thm:rootsystem-namecert-obligation-surface}.

\item The decode field sends any carried Weyl word of
\autoref{def:weylgroup-bhist-source-row} to the unique host Weyl word. Since
$T_W^\ast$ is terminal, classifier-respect on the host side is terminal
equality.

\item The decode-encode round trip starts from the unique host word, encodes
it as the empty singleton word over the carried root row, and decodes back to
the same host word. This is terminal equality in $T_W^\ast$.

\item The encode-decode round trip starts from a carried Weyl word. The word
component is classified by the restricted singleton $\GroupUp$ classifier as
in \autoref{thm:group-singleton-terminal-standard-bridge}; the root component
is classified by the public $\RootSystemUp$ classifier of
\autoref{thm:rootsystem-namecert-obligation-surface}. Together these are
exactly the restricted classifier of
\autoref{thm:weylgroup-rootsystem-group-carrier-row}.

\item The theorem-preservation list is finite and explicit. Its rows are
\autoref{thm:weylgroup-rootsystem-group-carrier-row},
\autoref{thm:weylgroup-simple-reflection-word-closure-row},
\autoref{thm:weylgroup-coxeter-word-ledger-row},
\autoref{thm:weylgroup-action-classifier-stability-row}, and
\autoref{thm:weylgroup-public-namecert-export}. The checked action transport
row records the same carrier, word, ledger, and classifier dependencies.

\item The no-host-leak field follows because the host terminal object appears
only in the encode, decode, and terminal-equality clauses above. The exported
surface remains the BEDC $\RootSystemUp$ support, singleton $\GroupUp$ word
carrier, Coxeter ledger rows, action classifier stability row, and public
NameCert boundary.
\end{enumerate}
\end{proof}
