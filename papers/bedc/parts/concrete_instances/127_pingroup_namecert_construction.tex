\chapter{A Concrete Naming Certificate for $\PinGroupUp$}
\label{ch:concrete-instances-pingroup-namecert}

The Pin-group interface $\PinGroupUp$ packages the double cover of the full orthogonal group, extending Spin by a reflection generator. It depends on $\NameCert_{\SpinGroupUp}$ (\autoref{ch:concrete-instances-spingroup-namecert}) and $\NameCert_{\CliffordUp}$ (\autoref{ch:concrete-instances-clifford-namecert}).

\section{Carrier definition}
\label{sec:pingroup-carrier}

\begin{definition}[PinGroupUp BHist carrier]
\label{def:pingroup-bhist-carrier}
A $\PinGroupUp$ $\mathsf{BHist}$ carrier is a finite packet
$$
\Pi=(S,C,R,\mu,\iota,\lambda,\rho)
$$
whose source rows are the displayed $\NameCert_{\SpinGroupUp}$ and
$\NameCert_{\CliffordUp}$ surfaces. The row $S$ is the selected
$\SpinGroupUp$ carrier row. The row $C$ is the selected $\CliffordUp$ carrier
row. The row $R$ is the reflection-generator row, displayed as a Clifford
carrier entry whose source and target are read only through the same $C$ row.
The row $\mu$ records the finite Clifford-product ledger combining carried
spin rows with the displayed reflection row. The row $\iota$ records the
inclusion of the spin row into the Clifford row. The row $\lambda$ is the
$\Cont$ ledger transporting only the listed rows, and $\rho$ is the $\Pkg$
provenance row for the spin, Clifford, reflection, product, inclusion, and
continuation entries.

Two such packets are compared by the public classifier only when the
corresponding $S$, $C$, $R$, $\mu$, $\iota$, and $\rho$ rows are related by
$\hsame$, and the continuation ledgers are related by $\Cont$. No ambient
orthogonal group, external Clifford algebra, or standard double-cover object is
part of the carrier.
\end{definition}

\section{The certificate}
\label{sec:pingroup-certificate}

\begin{theorem}[PinGroupUp Clifford-reflection product closure]
\label{thm:pingroup-clifford-reflection-product-closure}
Let $\Pi=(S,C,R,\mu,\iota,\lambda,\rho)$ be a $\PinGroupUp$ $\mathsf{BHist}$
carrier. If $x$ is carried by the spin row $S$ and the reflection-generator
row $R$ is carried by the Clifford row $C$, then the displayed Clifford
product row $\mu(x,R)$ is again a carried $\PinGroupUp$ row. Its Clifford
source is $C$, its reflection source is $R$, and its transport is exhausted by
the $\hsame$ comparisons and $\Cont$ entries already present in $\Pi$.
\end{theorem}
\begin{proof}
Project the spin carrier row, Clifford carrier row, reflection row, product
ledger, continuation ledger, and provenance row from
\autoref{def:pingroup-bhist-carrier}. These projections give all endpoints
used by the product row $\mu(x,R)$.

The Clifford endpoint is read through $C$, and the reflection endpoint is read
through $R$. The public classifier compares both endpoints only by $\hsame$,
so transported product endpoints remain in the displayed finite packet.

It remains to repack the product row. Append the product entry already named
by $\mu$ to the finite $\mathsf{BHist}$ ledger and transport the same entry by
the displayed $\Cont$ row. The resulting packet has the same source rows,
reflection row, and provenance row, hence it is again a $\PinGroupUp$ carrier.
\end{proof}

\begin{theorem}[PinGroupUp Spin and Clifford source scope]
\label{thm:pingroup-spin-clifford-source-scope}
Every source row admitted by a $\PinGroupUp$ carrier is one of the displayed
$\SpinGroupUp$ or $\CliffordUp$ source rows of
\autoref{def:pingroup-bhist-carrier}. The reflection-generator row, the
Clifford-product row, and the spin-inclusion row are internal rows over those
two sources and do not introduce a third source certificate.
\end{theorem}
\begin{proof}
Induct over the finite $\mathsf{BHist}$ ledger underlying the packet. The
initial source entries are exactly the spin row $S$ and the Clifford row $C$
from \autoref{def:pingroup-bhist-carrier}.

For a reflection-generator entry, the carrier definition reads its source and
target through $C$, so the entry contributes no new source. For a spin
inclusion entry, the domain is $S$ and the codomain is $C$.

For a Clifford-product entry, the ledger row is $\mu$ and its endpoints are
the already carried spin and reflection rows. The continuation step preserves
the same finite source list by $\Cont$, so the induction closes with only the
two displayed source certificates.
\end{proof}

\begin{theorem}[PinGroupUp NameCert obligation surface]
\label{thm:pingroup-namecert-obligation-surface}
The public $\NameCert_{\PinGroupUp}$ obligation surface consists exactly of
the $\mathsf{BHist}$ carrier of \autoref{def:pingroup-bhist-carrier}, its
componentwise $\hsame$ classifier, the reflection-generator row, the
Clifford-product closure row, the spin-inclusion row, and the finite
$\Cont$/$\Pkg$ ledger scope. It claims no data beyond the displayed
$\SpinGroupUp$ and $\CliffordUp$ source certificates.
\end{theorem}
\begin{proof}
Unpack the carrier packet. The projections expose the spin source, Clifford
source, reflection generator, Clifford product, spin inclusion, continuation
ledger, and provenance row.

Apply \autoref{thm:pingroup-clifford-reflection-product-closure} to identify
the product row as a carried $\PinGroupUp$ row. Apply
\autoref{thm:pingroup-spin-clifford-source-scope} to restrict all source rows
to the displayed spin and Clifford surfaces.

The classifier contains only componentwise $\hsame$ transport and $\Cont$
transport of the finite ledger. Repacking those projections gives precisely
the named obligation surface, with no additional carrier or source certificate
available to downstream consumers.
\end{proof}
