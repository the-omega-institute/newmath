\chapter{A Concrete Naming Certificate for $\CompleteMetricUp$}
\label{ch:concrete-instances-completemetric-namecert}

The complete-metric interface $\CompleteMetricUp$ packages a metric carrier in which every regular Cauchy sequence converges to an explicit limit with explicit modulus. It is the L10 sub-milestone certified by the real-completeness capstone (\autoref{ch:capstones-real-completeness}) and the substrate for $\NameCert_{\BanachUp}$.

\section{Carrier definition}
\label{sec:completemetric-carrier}

\begin{definition}[Singleton complete metric instance]
\label{def:singleton-complete-metric-instance}
The singleton complete metric instance is the metric carrier on histories
$$
C_{\emp}(x):=\hsame(x,\emp).
$$
Its classifier is
$$
x\sim_C y \quad:\!\!\Longleftrightarrow\quad
C_{\emp}(x)\land C_{\emp}(y)\land \hsame(x,y).
$$
A regular Cauchy name for this carrier is a stream of observed histories
$\gamma$ whose every observed history $\gamma_n$ is carried by $C_{\emp}$,
together with the regularity ledger asserting that every displayed pair
$\gamma_m,\gamma_n$ admits the empty-distance metric row
$\mathsf{MetricDistanceWitness}(\gamma_m,\gamma_n,\emp)$ after transport along
their carried endpoint rows. The limit selector sends every such name to
$\emp$. Its modulus is the empty history, and its convergence distance witness
at every observation is
$\mathsf{MetricDistanceWitness}(\gamma_n,\emp,\emp)$.
\end{definition}

\begin{definition}[Complete metric limit witness]
\label{def:completemetric-limit-witness}
Let $X$ be a certified metric carrier over $\mathsf{BHist}$, with point
classifier, distance rows encoded by $\mathsf{MetricDistanceWitness}$, and
the $\RatUp$ comparison ledgers used for visible tolerances. A complete metric
limit witness for a regular Cauchy stream name $s$ is the BEDC relation
$$
\mathsf{CompleteMetricLimitWitness}(X,s,\mu,\ell,L)
$$
whose accepted row consists of the following ledgers:
\begin{enumerate}
  \item a $\StreamNameUp$ regular Cauchy stream name $s$ whose observations
  are carried by $X$;
  \item an explicit modulus history $\mu$ certifying the regular Cauchy
  bound for $s$ at each positive $\RatUp$ tolerance;
  \item a limit history $\ell$ carried by $X$;
  \item for every positive tolerance history $\varepsilon$, a threshold
  observation history $N_{\varepsilon}$ displayed by $\mu$;
  \item for every observation history $n$ beyond $N_{\varepsilon}$, a
  distance history $d_{n,\varepsilon}$ together with the row
  $\mathsf{MetricDistanceWitness}(s(n),\ell,d_{n,\varepsilon})$ and a
  $\RatUp$ comparison ledger certifying
  $d_{n,\varepsilon}\leq\varepsilon$.
\end{enumerate}
The witness is therefore a finite-schema history package: the carrier,
stream, modulus, limit, metric-distance rows, and rational comparisons are all
visible BEDC ledgers, while convergence is recorded by the family of displayed
late-observation bounds.
\end{definition}

\section{The certificate}
\label{sec:completemetric-certificate}

\begin{theorem}[Singleton complete metric laws]
\label{thm:singleton-complete-metric-laws}
The instance of \autoref{def:singleton-complete-metric-instance} satisfies the
complete-metric laws required by $\CompleteMetricUp$: every $C_{\emp}$ regular
Cauchy name converges to the selected limit $\emp$, the selected limit is
stable under the classifier $\sim_C$, any two displayed limits are $\hsame$,
and the convergence ledger is compatible with metric empty-distance exactness
from \autoref{thm:metric-distance-witness-empty-distance-iff}. This
construction is the $\CompleteMetricUp$ companion to
\autoref{thm:singleton-metric-totally-bounded-laws}.
\end{theorem}
\leanchecked{BEDC.Derived.CompleteMetricUp.SingletonCompleteMetric\_laws}
\begin{proof}
Unfold $C_{\emp}$. Each observed history $\gamma_n$ of a regular Cauchy name
comes with a carried endpoint row $\hsame(\gamma_n,\emp)$. The selected limit
$\emp$ is carried by reflexivity of $\hsame$, and the empty modulus asks for no
additional history beyond this endpoint ledger.

For convergence, apply
\autoref{thm:metric-distance-witness-empty-distance-iff} to the endpoint rows
$\hsame(\gamma_n,\emp)$ and $\hsame(\emp,\emp)$. This identifies the
empty-distance metric row
$\mathsf{MetricDistanceWitness}(\gamma_n,\emp,\emp)$ with the classifier data
already carried by the stream observation and the selected limit.

Stability under $\sim_C$ is obtained by unpacking the classifier. A transported
candidate limit is carried by $C_{\emp}$, hence it is $\hsame$ to $\emp$.
Transitivity of $\hsame$ transports the convergence ledger from $\emp$ to that
candidate, while the empty-distance exactness theorem supplies the same
metric row after the endpoint transport.

Finally, suppose $\ell_1$ and $\ell_2$ are two displayed limits for the same
regular Cauchy name. Each displayed limit is carried by $C_{\emp}$, so
$\hsame(\ell_1,\emp)$ and $\hsame(\ell_2,\emp)$ hold. Symmetry and transitivity
of $\hsame$ give $\hsame(\ell_1,\ell_2)$, proving uniqueness of displayed
limits inside the singleton classifier.
\end{proof}

\begin{theorem}[Complete metric limit witness stability]
\label{thm:completemetric-limit-witness-stability}
Let $X$ be a certified metric carrier with $\StreamNameUp$ observation
transport, $\mathsf{MetricDistanceWitness}$ transport, and $\RatUp$
comparison transport. If
$$
\mathsf{CompleteMetricLimitWitness}(X,s,\mu,\ell,L)
$$
holds, and if the stream observations of $s$, the modulus history $\mu$, the
limit history $\ell$, every displayed threshold, and every displayed distance
history are transported by $\hsame$ together with the corresponding $\RatUp$
positivity and comparison rows, then the transported ledgers form a complete
metric limit witness
$$
\mathsf{CompleteMetricLimitWitness}(X,s',\mu',\ell',L')
$$
for the transported regular Cauchy stream name $s'$ and transported limit
history $\ell'$.
\end{theorem}
\leanchecked{BEDC.Derived.CompleteMetricUp.CompleteMetricLimitWitness\_hsame\_transport}
\begin{proof}
Unpack the witness row. The stream component is pointwise: for each certified
observation history, $\StreamNameUp$ transport carries the observed history
$s(n)$ to the corresponding observed history $s'(n')$, and the regular Cauchy
modulus row is moved along the supplied $\hsame$ history for $\mu$.

Fix a positive tolerance history $\varepsilon'$ transported from
$\varepsilon$. The threshold ledger $N_{\varepsilon}$ is carried to
$N'_{\varepsilon'}$, and the $\RatUp$ classifier transports the positivity
row for the tolerance together with the order row expressing that an
observation is late enough.

For such a late transported observation, the original witness supplies
$\mathsf{MetricDistanceWitness}(s(n),\ell,d_{n,\varepsilon})$ and the bound
$d_{n,\varepsilon}\leq\varepsilon$. Metric-distance transport moves this row
to
$$
\mathsf{MetricDistanceWitness}(s'(n'),\ell',d'_{n',\varepsilon'}).
$$

The remaining ledger is the comparison. Apply the $\RatUp$ classifier
transport to the displayed bound and repack the result as
$d'_{n',\varepsilon'}\leq\varepsilon'$. Since every field of the carrier row
has been transported and the quantification over late observations is
preserved pointwise, the transported package is again a complete metric limit
witness.
\end{proof}

\begin{theorem}[CompleteMetricUp limit witness NameCert target]
\label{thm:completemetric-limit-witness-namecert-target}
Let $X$ be a certified $\MetricSpaceUp$ carrier over $\mathsf{BHist}$. The
carrier of \autoref{def:completemetric-limit-witness}, the classifier given
by transported stream observations, modulus histories, limit histories,
threshold ledgers, distance histories, and $\RatUp$ comparison rows, the
stability theorem
\autoref{thm:completemetric-limit-witness-stability}, and the ledger policy
recording $\StreamNameUp$ observations, $\mathsf{MetricDistanceWitness}$ rows,
and $\RatUp$ bounds form the paper-level $\NameCert$ target for
$\CompleteMetricUp$ over $X$.
\end{theorem}
\begin{proof}
The source object is the certified metric carrier $X$ together with its
$\MetricSpaceUp$ point classifier and distance-row interface. The BEDC pattern
is the regular Cauchy stream name $s$, the modulus history $\mu$, the carried
limit history $\ell$, and the family $L$ of late-observation bound ledgers in
\autoref{def:completemetric-limit-witness}.

The classifier is the componentwise transport relation on this package. It
requires transported $\StreamNameUp$ observations, transported modulus and
limit histories, transported threshold histories for each positive tolerance,
transported distance histories, and transported $\RatUp$ positivity and
comparison rows. These are exactly the histories named in the public
complete-metric witness.

For the stability field, apply
\autoref{thm:completemetric-limit-witness-stability}. It takes a displayed
complete metric limit witness and the componentwise classifier data just
listed, then repacks them as a complete metric limit witness for the
transported stream and transported limit.

The ledger policy is therefore visible and finite-schema: it records
$\StreamNameUp$ observation rows, $\mathsf{MetricDistanceWitness}$ rows from
late observations to the selected limit, and $\RatUp$ comparison rows showing
that each displayed distance is below the requested positive tolerance. The
remaining entries of the ten-field template are inherited from the same
metric carrier: target name $\CompleteMetricUp$, carrier $X$, exactness by
metric endpoint transport, and no separate standard bridge. Thus the listed
fields assemble the paper-level $\NameCert$ target for $\CompleteMetricUp$.
\end{proof}

\begin{theorem}[Complete metric limit witness uniqueness]
\label{thm:completemetric-limit-witness-uniqueness}
Let $X$ be a separated certified $\MetricSpaceUp$ carrier over $\mathsf{BHist}$
with point classifier $\sim_X$. Assume the metric certificate supplies symmetry,
the triangle comparison row, the positive-half $\RatUp$ comparison ledger, and
zero-distance exactness in the following ledger form: if carried histories $a,b$
have, for every positive $\RatUp$ tolerance $\varepsilon$, a distance history
$e_{\varepsilon}$ with
$$
\mathsf{MetricDistanceWitness}(a,b,e_{\varepsilon})
\quad\text{and}\quad
 e_{\varepsilon}\leq\varepsilon,
$$
then $a\sim_X b$. If two displayed complete-metric limit witnesses converge the
same regular Cauchy stream name $s$,
$$
\mathsf{CompleteMetricLimitWitness}(X,s,\mu_0,\ell_0,L_0)
\quad\text{and}\quad
\mathsf{CompleteMetricLimitWitness}(X,s,\mu_1,\ell_1,L_1),
$$
then their limit histories are point-classifier equal:
$$
\ell_0\sim_X\ell_1.
$$
\end{theorem}
\leanchecked{BEDC.Derived.CompleteMetricUp.CompleteMetricLimitWitness\_singleton\_uniqueness}
\begin{proof}
Unpack the two rows of \autoref{def:completemetric-limit-witness}. The rows
state that $\ell_0$ and $\ell_1$ are carried by $X$, and that each positive
$\RatUp$ tolerance supplies a threshold after which the observations of the
same stream $s$ are within that tolerance of the displayed limit.

Fix a positive tolerance $\varepsilon$. Choose its positive half
$\rho_{\varepsilon}$ together with the comparison ledger
$\rho_{\varepsilon}+\rho_{\varepsilon}\leq\varepsilon$. The first limit witness
supplies a threshold $N^0_{\rho}$ such that every observation $n$ beyond it has
a distance row
$$
\mathsf{MetricDistanceWitness}(s(n),\ell_0,d^0_{n,\rho})
\quad\text{with}\quad
d^0_{n,\rho}\leq\rho_{\varepsilon}.
$$
The second witness supplies a threshold $N^1_{\rho}$ such that every observation
$n$ beyond it has a distance row
$$
\mathsf{MetricDistanceWitness}(s(n),\ell_1,d^1_{n,\rho})
\quad\text{with}\quad
d^1_{n,\rho}\leq\rho_{\varepsilon}.
$$
The common-tail ledger of the regular $\StreamNameUp$ component gives an
observation $n_{\varepsilon}$ beyond both thresholds.

Apply metric symmetry to the first displayed distance row at
$n_{\varepsilon}$, obtaining the same bound for a row from $\ell_0$ to
$s(n_{\varepsilon})$. Apply the triangle comparison row to this symmetric row
and to the second displayed row from $s(n_{\varepsilon})$ to $\ell_1$. This
produces a distance history $e_{\varepsilon}$ with
$$
\mathsf{MetricDistanceWitness}(\ell_0,\ell_1,e_{\varepsilon})
$$
and a comparison ledger
$$
e_{\varepsilon}\leq d^0_{n_{\varepsilon},\rho}+d^1_{n_{\varepsilon},\rho}.
$$
The two convergence bounds and the half-tolerance comparison compose in
$\RatUp$ to give
$$
e_{\varepsilon}\leq\varepsilon.
$$

Since $\varepsilon$ was an arbitrary positive $\RatUp$ tolerance, the pair
$\ell_0,\ell_1$ has displayed distance below every positive tolerance. The
zero-distance exactness field of the separated metric carrier therefore returns
$\ell_0\sim_X\ell_1$.
\end{proof}
