\chapter{A Concrete Naming Certificate for $\SubtypeUp$}

The subtype interface $\SubtypeUp$ packages the carrier of elements of a parent $\NameCert_{\SetUp}$ satisfying a decidable predicate. It depends on $\NameCert_{\SetUp}$ (\autoref{ch:concrete-instances-set-namecert}) and supplies the kernel-licit refinement type used throughout the algebra ladder for sub-objects.

\begin{definition}[Subtype restricted classifier]
\label{def:subtype-restricted-classifier}
Let $S$ be the certified point type of a parent $\NameCert_{\SetUp}$, with
parent classifier $\sim_S$. Let $P:S\to\mathsf{Prop}$ be a decidable predicate
stable under the parent classifier, meaning that for $x,y:S$,
$$
x\sim_S y\Longrightarrow P(x)\Longrightarrow P(y).
$$
The $\SubtypeUp$ carrier cut out by $P$ is
$$
S_P=\{(x,p)\mid x:S\ \text{and}\ p:P(x)\}.
$$
For subtype elements $(x,p_x)$ and $(y,p_y)$, the subtype classifier is the
restriction of the parent classifier:
$$
(x,p_x)\sim_{S_P}(y,p_y)\quad:\Longleftrightarrow\quad x\sim_S y.
$$
\end{definition}

\begin{theorem}[Subtype classifier reflects parent classifier]
\label{thm:subtype-classifier-reflects-parent-classifier}
Let $S$ be a parent $\NameCert_{\SetUp}$ with classifier $\sim_S$, and let
$P:S\to\mathsf{Prop}$ be a decidable predicate stable under $\sim_S$. For all
$x,y:S$ and witnesses $p_x:P(x)$ and $p_y:P(y)$, the classifier of
\autoref{def:subtype-restricted-classifier} satisfies
$$
(x,p_x)\sim_{S_P}(y,p_y)
\Longleftrightarrow
x\sim_S y.
$$
Equivalently, a parent classifier witness between predicate-bearing endpoints
induces the subtype classifier witness, and every subtype classifier witness
projects to the parent classifier witness.
\end{theorem}
\begin{proof}
Unfold the classifier clause in \autoref{def:subtype-restricted-classifier}.
The forward implication is projection: a subtype classifier witness is a parent
classifier witness between the first components. The reverse implication is
inclusion: a witness of $x\sim_S y$ is, by the displayed definition, a witness
of $(x,p_x)\sim_{S_P}(y,p_y)$ once the two subtype endpoints have been formed.
The decidability and stability assumptions are the formation and transport rows
for the $\SubtypeUp$ carrier; the restricted classifier itself carries no data
beyond the parent comparison.
\end{proof}

\begin{theorem}[Subtype carrier obligation row]
\label{thm:subtype-carrier-obligation-row}
Let $S:\mathsf{BHist}\to\mathsf{Prop}$ be the carrier of a parent $\NameCert_{\SetUp}$, and let $P:\mathsf{BHist}\to\mathsf{Prop}$ be a decidable predicate. In the predicate presentation of \autoref{def:subtype-restricted-classifier}, the carrier obligation for $\SubtypeUp$ is exactly
$$
S_P(h)\Longleftrightarrow S(h)\land P(h).
$$
Consequently a displayed endpoint $s_0:\mathsf{BHist}$ with $S(s_0)$ and $P(s_0)$ supplies the inhabitedness row
$$
S_P(s_0).
$$
\end{theorem}
\begin{proof}
Unfold the carrier clause of \autoref{def:subtype-restricted-classifier}. A subtype endpoint is formed by pairing a parent-carrier witness with a predicate witness, so its predicate presentation is the conjunction $S(h)\land P(h)$.

For inhabitedness, apply the same clause at $s_0$. The two displayed witnesses $S(s_0)$ and $P(s_0)$ are precisely the two components required to form $S_P(s_0)$.
\end{proof}

\begin{theorem}[Subtype classifier projection obligation]
\label{thm:subtype-classifier-projection-obligation}
Let $S:\mathsf{BHist}\to\mathsf{Prop}$ be the carrier of a parent $\NameCert_{\SetUp}$ with classifier $\sim_S$, and let $P:\mathsf{BHist}\to\mathsf{Prop}$. For all predicate-bearing endpoints $h,k:\mathsf{BHist}$, the classifier row for $\SubtypeUp$ is exactly
$$
h\sim_{S_P}k\Longleftrightarrow h\sim_S k.
$$
Thus parent projection and restricted-classifier introduction are the two classifier rows: a subtype classifier witness projects to the parent comparison, and a parent comparison between already formed subtype endpoints introduces the restricted classifier witness.
\end{theorem}
\begin{proof}
Unfold the classifier clause in \autoref{def:subtype-restricted-classifier}. The restricted classifier stores no comparison datum apart from the parent comparison $h\sim_S k$.

The forward direction is projection from the restricted witness to that parent witness. The reverse direction is introduction: once the endpoints are already certified by the carrier row, the parent witness is read as the restricted-classifier witness.
\end{proof}

\begin{theorem}[Subtype NameCert transport obligation]
\label{thm:subtype-namecert-transport-obligation}
Let $S:\mathsf{BHist}\to\mathsf{Prop}$ be the carrier of a parent $\NameCert_{\SetUp}$ with classifier $\sim_S$, and let $\mathsf{cert}:\NameCert(S,\sim_S)$. Let $P:\mathsf{BHist}\to\mathsf{Prop}$ be stable under the parent classifier, meaning that for all $h,k:\mathsf{BHist}$,
$$
S(h)\land P(h)\land h\sim_S k\Longrightarrow P(k).
$$
In the predicate presentation of \autoref{def:subtype-restricted-classifier}, parent carrier transport from $\mathsf{cert}$ together with predicate stability gives
$$
S_P(h)\to h\sim_{S_P}k\to S_P(k).
$$
This is the transport row consumed by \autoref{thm:subtype-restricted-classifier-namecert}.
\end{theorem}
\begin{proof}
Unfold $S_P(h)$ by \autoref{thm:subtype-carrier-obligation-row}. This gives $S(h)$ and $P(h)$. Unfold the restricted classifier by \autoref{thm:subtype-classifier-projection-obligation}; the comparison is the parent witness $h\sim_S k$.

The carrier-respect field of the parent $\NameCert(S,\sim_S)$ transports $S(h)$ along $h\sim_S k$ and gives $S(k)$. Apply the displayed predicate-stability hypothesis to $S(h)$, $P(h)$, and the same parent comparison to obtain $P(k)$.

Pair $S(k)$ with $P(k)$ using the carrier row for $S_P(k)$. The resulting witness is the required transport row for the restricted certificate.
\end{proof}

\begin{lemma}[Subtype carrier closed under restricted classifier]
\label{lem:subtype-carrier-restricted-classifier-closure}
Let $S:\mathsf{BHist}\to\mathsf{Prop}$ be the carrier of a parent $\NameCert_{\SetUp}$ with classifier $\sim_S$, and let $\mathsf{cert}:\NameCert(S,\sim_S)$. Let $P:\mathsf{BHist}\to\mathsf{Prop}$ be a decidable predicate stable under the parent classifier, in the sense that for all $x,y:\mathsf{BHist}$,
$$
S(x)\land P(x)\land x\sim_S y\Longrightarrow P(y).
$$
In the predicate presentation of \autoref{def:subtype-restricted-classifier}, write
$$
S_P(h)\Longleftrightarrow S(h)\land P(h),
\qquad
h\sim_{S_P}k\Longleftrightarrow h\sim_S k.
$$
If $S_P(h)$ and $h\sim_{S_P}k$, then $S_P(k)$.
\end{lemma}
\leanchecked{BEDC.Derived.SubtypeUp.SubtypeRestrictedCarrier\_classifier\_closed}
\begin{proof}
Unfold $S_P(h)$ to obtain $S(h)$ and $P(h)$. Unfold the restricted classifier to read the comparison as $h\sim_S k$. The carrier-transport field of $\mathsf{cert}$, one of the fields listed in \autoref{def:lean-carrier-naming-certificate}, transports $S(h)$ along this comparison and gives $S(k)$. Stability of $P$ under the same parent comparison gives $P(k)$. Pairing $S(k)$ with $P(k)$ gives $S_P(k)$.
\end{proof}

\begin{theorem}[Subtype restricted classifier NameCert]
\label{thm:subtype-restricted-classifier-namecert}
Under the hypotheses and notation of \autoref{lem:subtype-carrier-restricted-classifier-closure}, assume that the subtype carrier has a displayed endpoint, so there is an $s_0:\mathsf{BHist}$ with $S(s_0)$ and $P(s_0)$. Then the $\SubtypeUp$ carrier $S_P$ with the restricted classifier $\sim_{S_P}$ forms a Lean carrier naming certificate:
$$
\NameCert(S_P,\sim_{S_P}).
$$
\end{theorem}
\leanchecked{BEDC.Derived.SubtypeUp.SubtypeRestrictedClassifier\_nameCert}
\begin{proof}
Inhabitedness is witnessed by $s_0$, since $S(s_0)$ and $P(s_0)$ are exactly the two components of $S_P(s_0)$.

For reflexivity, let $h$ be carried by $S_P$. Then $S(h)$ holds, so the reflexivity field of $\mathsf{cert}$ gives $h\sim_S h$. By the classifier clause of \autoref{def:subtype-restricted-classifier}, this is $h\sim_{S_P}h$.

For symmetry, a restricted witness $h\sim_{S_P}k$ is the parent witness $h\sim_S k$. The symmetry field of $\mathsf{cert}$ gives $k\sim_S h$, hence $k\sim_{S_P}h$.

For transitivity, restricted witnesses $h\sim_{S_P}k$ and $k\sim_{S_P}r$ are parent witnesses $h\sim_S k$ and $k\sim_S r$. The transitivity field of $\mathsf{cert}$ composes them to $h\sim_S r$, which is the restricted witness $h\sim_{S_P}r$.

Carrier transport for $S_P$ is exactly \autoref{lem:subtype-carrier-restricted-classifier-closure}. These five rows are the inhabitedness, reflexivity, symmetry, transitivity, and carrier-transport fields required by \autoref{def:lean-carrier-naming-certificate}.
\end{proof}
