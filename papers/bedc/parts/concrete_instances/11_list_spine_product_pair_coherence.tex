\begin{definition}[List cons-boundary coherence]
\label{def:list-cons-boundary-coherence}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier and let
$\sim_A$ be a source classifier. Define
$\mathsf{ListConsBoundaryCoherent}_{A,\sim_A}$ to mean that any two displayed
cons presentations of the same list endpoint classify the displayed heads and
tails. Concretely, for histories $m,a,a',t,t',p,p'$ and finite spines $xs,xs'$,
assume $A(a)$, $A(a')$, $\mathsf{ListSpineRep}_A(t,xs)$,
$\mathsf{ListSpineRep}_A(t',xs')$, $\ContR(a,t,p)$, $\ContR(a',t',p')$,
$\hsame(m,\Eone(p))$, and $\hsame(m,\Eone(p'))$. The required outputs are
exactly $a\sim_A a'$ and $\mathsf{ListClassifierSpec}(\sim_A,xs,xs')$.
\end{definition}

\begin{theorem}[List spine coherence from cons-boundary coherence]
\label{thm:list-spine-coherence-from-cons-boundary-coherence}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier and let
$\sim_A$ be a source classifier. If
$\mathsf{ListConsBoundaryCoherent}_{A,\sim_A}$ holds, then
$\mathsf{ListSpineRepCoherent}_{A,\sim_A}$ holds.
\end{theorem}
\begin{proof}
Compare two finite-spine presentations of the same history by cases on the
displayed spines. The nil-nil case is the nil constructor of
$\mathsf{ListClassifierSpec}$. The mixed nil-cons cases contradict
Theorem~\ref{thm:list-spine-rep-nil-cons-no-confusion}. In the cons-cons case,
both presentations expose the same list endpoint together with displayed head
histories, tail spines, tail representations, continuation witnesses, and
endpoint witnesses. Definition~\ref{def:list-cons-boundary-coherence} supplies
exactly the head classifier witness and the tail finite-spine classifier
witness, and these are the two constructor arguments needed for
$\mathsf{ListClassifierSpec}(\sim_A,-,-)$ on the whole cons spines.
\end{proof}

\begin{theorem}[List spine coherence iff cons-boundary coherence]
\label{thm:list-spine-coherence-iff-cons-boundary-coherence}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier and let
$\sim_A$ be a source classifier. Then
$\mathsf{ListSpineRepCoherent}_{A,\sim_A}$ holds if and only if
$\mathsf{ListConsBoundaryCoherent}_{A,\sim_A}$ holds.
\end{theorem}
\begin{proof}
For the forward implication, assume
$\mathsf{ListSpineRepCoherent}_{A,\sim_A}$. Given two displayed cons
presentations of the same endpoint, view them as two
$\mathsf{ListSpineRep}_A$ presentations with cons spines. Spine coherence
returns a $\mathsf{ListClassifierSpec}(\sim_A,-,-)$ witness relating those two
cons spines. The cons-cons branch of
Theorem~\ref{thm:list-classifier-case-exactness} inverts that witness and
exposes exactly the required head classifier $a\sim_A a'$ together with the
tail classifier $\mathsf{ListClassifierSpec}(\sim_A,xs,xs')$. These are the two
fields of Definition~\ref{def:list-cons-boundary-coherence}. The reverse
implication is precisely
Theorem~\ref{thm:list-spine-coherence-from-cons-boundary-coherence}.
\end{proof}

\begin{theorem}[List spine bridge endpoint exposure and symmetry]
\label{thm:list-spine-bridge-endpoint-exposure-symmetry}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier and let
$\sim_A$ be a source classifier. The compact per-object fit is: traditional
target finite-list classifier stability; BEDC source $A$ over $\mathsf{BHist}$
with $\NameCert(A,\sim_A)$; carrier $\mathsf{ListHistoryCarrier}_A$; classifier
$\mathsf{ListSpineBridgeClassifier}_{A,\sim_A}$; stability obligations endpoint
exposure and symmetry; ledger finite spine representation; core theorem the two
displayed clauses below; exactness via $\mathsf{ListSpineRep}$; standard bridge
finite-spine readback; status $\mathsf{ProofTarget}$. For all histories $h$ and
$k$, the bridge classifier satisfies endpoint exposure:
$$
\mathsf{ListSpineBridgeClassifier}_{A,\sim_A}(h,k)
\to
\mathsf{ListHistoryCarrier}_A(h)\land \mathsf{ListHistoryCarrier}_A(k).
$$
Moreover, for any $\mathsf{cert}:\NameCert(A,\sim_A)$, the bridge classifier is
symmetric:
$$
\mathsf{ListSpineBridgeClassifier}_{A,\sim_A}(h,k)
\to
\mathsf{ListSpineBridgeClassifier}_{A,\sim_A}(k,h).
$$
\end{theorem}
\begin{proof}
Unpack a bridge witness into finite spines $xs$ and $ys$, displayed
representations $\mathsf{ListSpineRep}_A(h,xs)$ and
$\mathsf{ListSpineRep}_A(k,ys)$, and the finite-spine classifier witness
$\mathsf{ListClassifierSpec}(\sim_A,xs,ys)$. Endpoint exposure applies the
soundness direction of Theorem~\ref{thm:list-spine-representation-exactness} to
the two displayed representations. For symmetry, keep the two representations,
apply Theorem~\ref{thm:list-classifier-symm-from-namecert} to the source
certificate $\mathsf{cert}$ and the finite-spine classifier witness, then rebuild
the bridge witness with the represented spines swapped.
\end{proof}

\begin{theorem}[List public-object certificate from cons-boundary coherence]
\label{thm:list-public-object-certificate-from-cons-boundary-coherence}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier, let
$\sim_A$ be a source classifier, and let
$\mathsf{cert}:\NameCert(A,\sim_A)$. The ten-field per-object fit is source
carrier $A$, source classifier $\sim_A$, source certificate $\mathsf{cert}$,
list carrier $\mathsf{ListHistoryCarrier}_A$, list pattern
$\mathsf{ListHistoryPattern}_A$, list ledger $\mathsf{ListHistoryLedger}_A$,
finite-spine representation $\mathsf{ListSpineRep}_A$, finite-spine classifier
$\mathsf{ListClassifierSpec}(\sim_A,-,-)$, bridge classifier
$\mathsf{ListSpineHistoryClassifier}_{A,\sim_A}$, and semantic certificate
target
$$
\mathsf{SemanticNameCert}(\mathsf{ListHistoryCarrier}_A,
\mathsf{ListHistoryPattern}_A,
\mathsf{ListHistoryLedger}_A,
\mathsf{ListSpineHistoryClassifier}_{A,\sim_A}).
$$
If $\mathsf{ListConsBoundaryCoherent}_{A,\sim_A}$ holds, then this semantic
certificate target holds.
\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:list-spine-coherence-from-cons-boundary-coherence} to
the cons-boundary coherence witness, obtaining
$\mathsf{ListSpineRepCoherent}_{A,\sim_A}$. Then apply
Theorem~\ref{thm:concrete-list-namecert} with the source certificate
$\mathsf{cert}$ and this spine-coherence witness. The resulting certificate is
exactly the displayed semantic target for the list public object.
\end{proof}
