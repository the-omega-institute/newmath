\begin{definition}[List cons-boundary coherence]
\label{def:list-cons-boundary-coherence}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier and let
$\sim_A$ be a source classifier. Define
$\mathsf{ListConsBoundaryCoherent}_{A,\sim_A}$ as the predicate
$$
\forall m\,a\,a'\,t\,t'\,p\,p'\,xs\,xs'.\;
A(a)\to A(a')\to
\mathsf{ListSpineRep}_A(t,xs)\to
\mathsf{ListSpineRep}_A(t',xs')\to
\ContR(a,t,p)\to \ContR(a',t',p')\to
\hsame(m,\Eone(p))\to \hsame(m,\Eone(p'))\to
((a\sim_A a')\land \mathsf{ListClassifierSpec}(\sim_A,xs,xs')).
$$
\end{definition}

\begin{theorem}[List spine coherence from cons-boundary coherence]
\label{thm:list-spine-coherence-from-cons-boundary-coherence}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier and let
$\sim_A$ be a source classifier. If
$\mathsf{ListConsBoundaryCoherent}_{A,\sim_A}$ holds, then
$\mathsf{ListSpineRepCoherent}_{A,\sim_A}$ holds.
\end{theorem}
\begin{proof}
Compare two finite-spine presentations of the same history by cases on the
displayed spines. The nil-nil case is the nil constructor of
$\mathsf{ListClassifierSpec}$. The mixed nil-cons cases contradict
Theorem~\ref{thm:list-spine-rep-nil-cons-no-confusion}. In the cons-cons case,
both presentations expose the same list endpoint together with displayed head
histories, tail spines, tail representations, continuation witnesses, and
endpoint witnesses. Definition~\ref{def:list-cons-boundary-coherence} supplies
the conjunction
$(a\sim_A a')\land \mathsf{ListClassifierSpec}(\sim_A,xs,xs')$; its left
conjunct is the head classifier witness, and its right conjunct is the tail
finite-spine classifier witness. These are the two constructor arguments for
$\mathsf{ListClassifierSpec}(\sim_A,-,-)$ on the whole cons spines.
\end{proof}

\begin{theorem}[List spine coherence iff cons-boundary coherence]
\label{thm:list-spine-coherence-iff-cons-boundary-coherence}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier and let
$\sim_A$ be a source classifier. Then
$\mathsf{ListSpineRepCoherent}_{A,\sim_A}$ holds if and only if
$\mathsf{ListConsBoundaryCoherent}_{A,\sim_A}$ holds.
\end{theorem}
\begin{proof}
For the forward implication, assume
$\mathsf{ListSpineRepCoherent}_{A,\sim_A}$. Given two displayed cons
presentations of the same endpoint, view them as two
$\mathsf{ListSpineRep}_A$ presentations with cons spines. Spine coherence
returns a $\mathsf{ListClassifierSpec}(\sim_A,-,-)$ witness relating those two
cons spines. The cons-cons branch of
Theorem~\ref{thm:list-classifier-case-exactness} inverts that witness and
exposes exactly the required head classifier $a\sim_A a'$ together with the
tail classifier $\mathsf{ListClassifierSpec}(\sim_A,xs,xs')$. These are the two
fields of Definition~\ref{def:list-cons-boundary-coherence}. The reverse
implication is precisely
Theorem~\ref{thm:list-spine-coherence-from-cons-boundary-coherence}.
\end{proof}

\begin{definition}[Pair frame]
\label{def:list-pair-frame}
For histories $u,t:\mathsf{BHist}$, define $\mathsf{PairFrame}(u,t)$ by
structural recursion on the first BEDC history:
$$
\begin{array}{rcl}
\mathsf{PairFrame}(\emp,t) &=& \Ezero(t),\\
\mathsf{PairFrame}(\Ezero(a),t) &=& \Eone(\Ezero(\mathsf{PairFrame}(a,t))),\\
\mathsf{PairFrame}(\Eone(a),t) &=& \Eone(\Eone(\mathsf{PairFrame}(a,t))).
\end{array}
$$
The outer $\Eone$ marks a nonempty first component, while the inner
$\Ezero/\Eone$ tag records the constructor of that first component.
\end{definition}

\begin{lemma}[Pair frame injectivity]
\label{lem:list-pair-frame-injectivity}
For histories $u,u',t,t':\mathsf{BHist}$, if
$$
\hsame(\mathsf{PairFrame}(u,t),\mathsf{PairFrame}(u',t')),
$$
then $\hsame(u,u')\land\hsame(t,t')$.
\end{lemma}
\begin{proof}
Proceed by structural induction and case analysis on $u$ and $u'$. The mixed
$\emp/\Ezero$ and $\emp/\Eone$ cases contradict BEDC constructor no-confusion
because $\mathsf{PairFrame}(\emp,t)$ begins with $\Ezero$, while the nonempty
cases begin with $\Eone$. The mixed $\Ezero/\Eone$ cases contradict
no-confusion after inverting the outer $\Eone$ and comparing the inner tags. In
the matching $\emp/\emp$ case, constructor inversion gives $\hsame(t,t')$. In
the matching $\Ezero/\Ezero$ and $\Eone/\Eone$ cases, invert the outer
$\Eone$, then the matching inner tag, and apply the induction hypothesis to the
recursive pair-frame equality. Combining the exposed recursive head equality
with the constructor equality gives $\hsame(u,u')$, and the induction
hypothesis also gives $\hsame(t,t')$.
\end{proof}

\begin{theorem}[Framed list endpoint no-confusion and cons inversion]
\label{thm:framed-list-endpoint-no-confusion-cons-inversion}
The ten-field fit is: traditional target finite-list endpoint readback; BEDC
source $\mathsf{BHist}$ with $\hsame$; carrier framed finite spines; classifier
endpoint $\hsame$; stability obligations nil/cons separation and cons inversion;
ledger constructor frame; core theorem this block; exactness framed endpoint
exposes list shape; standard bridge finite spines; status
$\mathsf{ProofTarget}$. For all histories $a,b:\mathsf{BHist}$ and finite
spines $xs,ys$,
$$
\neg\hsame(\mathsf{FramedListEndpoint}(\mathsf{nil}),
\mathsf{FramedListEndpoint}(\mathsf{cons}(a,xs))),
$$
$$
\neg\hsame(\mathsf{FramedListEndpoint}(\mathsf{cons}(a,xs)),
\mathsf{FramedListEndpoint}(\mathsf{nil})),
$$
and
$$
\hsame(\mathsf{FramedListEndpoint}(\mathsf{cons}(a,xs)),
\mathsf{FramedListEndpoint}(\mathsf{cons}(b,ys)))
$$
implies both $\hsame(a,b)$ and
$\hsame(\mathsf{FramedListEndpoint}(xs),\mathsf{FramedListEndpoint}(ys))$.
\end{theorem}
\begin{proof}
The nil endpoint has outer constructor $\Ezero$, while every cons endpoint has
outer constructor $\Eone$, so the two mixed cases contradict BEDC history
no-confusion in the $\Ezero/\Eone$ and $\Eone/\Ezero$ directions. For the
cons-cons clause, invert the outer $\Eone$ equality and obtain equality of the
two displayed pair frames. Lemma~\ref{lem:list-pair-frame-injectivity} applies:
its proof is an induction on the first frame argument, with constructor
no-confusion separating the empty and tagged nonempty cases and with recursive
inversion exposing the first and second frame arguments. The two projections of
that lemma give $\hsame(a,b)$ and
$\hsame(\mathsf{FramedListEndpoint}(xs),\mathsf{FramedListEndpoint}(ys))$.
\end{proof}

\begin{theorem}[List spine bridge source refinement]
\label{thm:list-spine-bridge-source-refinement}
Let $A,B:\mathsf{BHist}\to\mathsf{Prop}$ be BEDC history source carriers, and
let $\sim_A$ and $\sim_B$ be source classifiers. The ten-field fit is:
traditional target finite-list source refinement; BEDC source $\mathsf{BHist}$
with $\mathsf{ListSpineRep}$; carrier $\mathsf{ListHistoryCarrier}$; classifier
$\mathsf{ListSpineBridgeClassifier}$; stability obligation carrier and
classifier preservation under source refinement; ledger finite spine
representation; core theorem the three displayed implications below; exactness
inherited from $\mathsf{ListSpineRep}$; standard bridge finite-spine readback;
status $\mathsf{ProofTarget}$. Assume the source carrier and source classifier
of $A$ are included in those of $B$:
$$
\forall h:\mathsf{BHist}.\; A(h)\to B(h)
$$
and
$$
\forall x\,y:\mathsf{BHist}.\; x\sim_A y\to x\sim_B y.
$$
Then the displayed list-spine data transport from $A$ to $B$:
$$
\mathsf{ListSpineRep}_A(h,xs)
\to
\mathsf{ListSpineRep}_B(h,xs),
$$
$$
\mathsf{ListSpineBridgeClassifier}_{A,\sim_A}(h,k)
\to
\mathsf{ListSpineBridgeClassifier}_{B,\sim_B}(h,k),
$$
and
$$
\mathsf{ListSpineHistoryClassifier}_{A,\sim_A}(h,k)
\to
\mathsf{ListSpineHistoryClassifier}_{B,\sim_B}(h,k).
$$
\end{theorem}
\begin{proof}
Proceed by induction on the displayed $\mathsf{ListSpineRep}_A(h,xs)$ witness.
The nil representation keeps the same endpoint equality. In the cons case, the
carrier inclusion turns the recorded source datum $A(a)$ into $B(a)$, while the
induction hypothesis transports the tail representation and the continuation
and endpoint witnesses are reused unchanged. For the finite-spine classifier
component, induct on the displayed $\mathsf{ListClassifierSpec}(\sim_A,xs,ys)$
witness and apply the classifier inclusion at each cons head. A bridge witness is
then repacked with the transported left and right spines and the transported
finite-spine classifier. The certificate classifier clause follows by unfolding
that it is the bridge classifier and rebuilding the same endpoints.
\end{proof}

\begin{lemma}[List spine representation endpoint transport]
\label{lem:list-spine-representation-endpoint-transport}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier. For all
histories $h$ and $h'$ and every finite list spine $xs$,
$\mathsf{ListSpineRep}_A(h,xs)$ and $\hsame(h,h')$ imply
$\mathsf{ListSpineRep}_A(h',xs)$.
\end{lemma}
\begin{proof}
Proceed by induction on the displayed finite spine. The empty spine is transported
by composing the endpoint empty witness with $\hsame(h,h')$. For a cons spine,
keep the recorded head payload and tail representation, and compose the outer
history endpoint witness with $\hsame(h,h')$; the payload and tail witnesses are
unchanged.
\end{proof}

\begin{theorem}[List spine bridge endpoint transport]
\label{thm:list-spine-bridge-endpoint-transport}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier and let
$\sim_A$ be a source classifier. The ten-field per-object fit is: traditional
target finite-list classifier stability; BEDC source $A$ over $\mathsf{BHist}$;
source classifier $\sim_A$; carrier $\mathsf{ListHistoryCarrier}_A$; classifier
$\mathsf{ListSpineBridgeClassifier}_{A,\sim_A}$; stability endpoint transport;
ledger finite-spine presentation; exactness via $\mathsf{ListSpineRep}$; standard
bridge finite spines; status $\mathsf{ProofTarget}$. For all histories $h$, $k$,
$h'$, and $k'$,
$$
\mathsf{ListSpineBridgeClassifier}_{A,\sim_A}(h,k)
\to
\hsame(h,h')
\to
\hsame(k,k')
\to
\mathsf{ListSpineBridgeClassifier}_{A,\sim_A}(h',k').
$$
\end{theorem}
\begin{proof}
Unpack the bridge witness into finite spines $xs$ and $ys$, the represented
endpoints $\mathsf{ListSpineRep}_A(h,xs)$ and
$\mathsf{ListSpineRep}_A(k,ys)$, and the finite-spine classifier witness. Apply
Lemma~\ref{lem:list-spine-representation-endpoint-transport} to transport the
two displayed endpoint representations along $\hsame(h,h')$ and $\hsame(k,k')$.
The finite-spine classifier witness is independent of the represented endpoints,
so the bridge witness is rebuilt over $h'$ and $k'$ with the same spines.
\end{proof}

\begin{theorem}[List spine bridge nil-cons separation under coherence]
\label{thm:list-spine-bridge-nil-cons-separation-under-coherence}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier, let
$\sim_A$ be a source classifier, let $\mathsf{cert}:\NameCert(A,\sim_A)$, and
assume $\mathsf{ListSpineRepCoherent}_{A,\sim_A}$. For all histories $h$ and $k$,
all source histories $a$, and all finite spines $xs$, the bridge classifier
separates the nil and cons displayed endpoints:
$$
\mathsf{ListSpineRep}_A(h,\mathsf{nil})
\to
\mathsf{ListSpineRep}_A(k,\mathsf{cons}(a,xs))
\to
\mathsf{ListSpineBridgeClassifier}_{A,\sim_A}(h,k)
\to
\bot.
$$
It also separates the symmetric cons-vs-nil orientation:
$$
\mathsf{ListSpineRep}_A(h,\mathsf{cons}(a,xs))
\to
\mathsf{ListSpineRep}_A(k,\mathsf{nil})
\to
\mathsf{ListSpineBridgeClassifier}_{A,\sim_A}(h,k)
\to
\bot.
$$
\end{theorem}
\begin{proof}
Unpack the bridge witness into finite spines $u$ and $v$, endpoint presentations
$\mathsf{ListSpineRep}_A(h,u)$ and $\mathsf{ListSpineRep}_A(k,v)$, and a
finite-spine classifier $\mathsf{ListClassifierSpec}(\sim_A,u,v)$. In the
nil-vs-cons orientation, apply the coherence datum at endpoint $h$ to relate the
displayed $\mathsf{nil}$ spine to $u$, and at endpoint $k$ to relate $v$ to the
displayed $\mathsf{cons}(a,xs)$ spine. Compose these two finite-spine classifier
witnesses with the bridge classifier witness using transitivity supplied by
$\mathsf{cert}:\NameCert(A,\sim_A)$. The resulting
$\mathsf{ListClassifierSpec}(\sim_A,\mathsf{nil},\mathsf{cons}(a,xs))$ contradicts
Lemma~\ref{lem:list-classifier-nil-cons-no-confusion}. The cons-vs-nil
orientation is identical after swapping the two endpoint coherence comparisons
and using the symmetric nil-cons no-confusion clause.
\end{proof}

\begin{theorem}[List public-object certificate from cons-boundary coherence]
\label{thm:list-public-object-certificate-from-cons-boundary-coherence}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier, let
$\sim_A$ be a source classifier, and let
$\mathsf{cert}:\NameCert(A,\sim_A)$. The ten-field per-object fit is:
traditional target finite lists; BEDC source $A$ with $\NameCert(A,\sim_A)$;
carrier $\mathsf{ListHistoryCarrier}_A$; classifier
$\mathsf{ListSpineHistoryClassifier}_{A,\sim_A}$; stability supplied by the
source certificate and $\mathsf{ListConsBoundaryCoherent}_{A,\sim_A}$; ledger
$\mathsf{ListHistoryLedger}_A$; core theorem list
Theorem~\ref{thm:list-spine-coherence-from-cons-boundary-coherence} and
Theorem~\ref{thm:concrete-list-namecert}; exactness through the spine bridge
$\mathsf{ListSpineRep}_A$; standard bridge finite spines; status
$\mathsf{ProofTarget}$ under
$\mathsf{ListConsBoundaryCoherent}_{A,\sim_A}$. The semantic certificate target
is
$$
\mathsf{SemanticNameCert}(\mathsf{ListHistoryCarrier}_A,
\mathsf{ListHistoryPattern}_A,
\mathsf{ListHistoryLedger}_A,
\mathsf{ListSpineHistoryClassifier}_{A,\sim_A}).
$$
If $\mathsf{ListConsBoundaryCoherent}_{A,\sim_A}$ holds, then this semantic
certificate target holds.
\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:list-spine-coherence-from-cons-boundary-coherence} to
the cons-boundary coherence witness, obtaining
$\mathsf{ListSpineRepCoherent}_{A,\sim_A}$. Then apply
Theorem~\ref{thm:concrete-list-namecert} with the source certificate
$\mathsf{cert}$ and this spine-coherence witness. The resulting certificate is
exactly the displayed semantic target for the list public object.
\end{proof}

\begin{definition}[Singleton history source for lists]
\label{def:list-singleton-history-source}
For a fixed BEDC history $a_0:\mathsf{BHist}$, define the singleton history
source carrier $A_{a_0}:\mathsf{BHist}\to\mathsf{Prop}$ by
$$
A_{a_0}(x)\Longleftrightarrow\hsame(x,a_0).
$$
\end{definition}

\begin{theorem}[List singleton source framed carrier certificate]
\label{thm:list-singleton-source-framed-carrier-certificate}
Fix a BEDC history $a_0:\mathsf{BHist}$ and let $A_{a_0}$ be the singleton
history source of Definition~\ref{def:list-singleton-history-source}. The
ten-field per-object fit is: traditional target framed carrier certificate for
finite lists over one BEDC history class; BEDC source $\mathsf{BHist}$ with
$\hsame$; carrier
$\mathsf{FramedListHistoryCarrier}_{A_{a_0}}$; classifier
$\mathsf{FramedListBridgeClassifier}_{A_{a_0},\hsame}$; stability from
$\mathsf{ListSourceHsameCompatible}(A_{a_0},\hsame)$ and framed endpoint
coherence; ledger finite framed spine; core theorem
Corollary~\ref{cor:framed-list-carrier-certificate}; exactness
through framed spine representation; standard bridge finite spines; status
$\mathsf{ProofTarget}$. The framed semantic certificate
$$
\mathsf{SemanticNameCert}(\mathsf{FramedListHistoryCarrier}_{A_{a_0}},
\mathsf{FramedListHistoryCarrier}_{A_{a_0}},
\mathsf{FramedListHistoryCarrier}_{A_{a_0}},
\mathsf{FramedListBridgeClassifier}_{A_{a_0},\hsame})
$$
hold.
\end{theorem}
\begin{proof}
Theorem~\ref{thm:singleton-history-source-namecert} supplies the source
certificate $\NameCert(A_{a_0},\hsame)$. The compatibility predicate
$\mathsf{ListSourceHsameCompatible}(A_{a_0},\hsame)$ is immediate: the final displayed input is already the required $\hsame(a,b)$ witness. Apply
Corollary~\ref{cor:framed-list-carrier-certificate} with the singleton source
certificate and this compatibility witness.
\end{proof}

\begin{theorem}[List singleton source spine coherence]
\label{thm:list-singleton-source-spine-coherence}
For any BEDC history $a_0$, let $A_{a_0}$ be the singleton history source of
Definition~\ref{def:list-singleton-history-source}. Then
$\mathsf{ListSpineRepCoherent}_{A_{a_0},\hsame}$ holds.
\end{theorem}
\begin{proof}
Compare two singleton-source spine presentations of the same endpoint by cases
on the displayed spines. The mixed nil/cons cases contradict
Theorem~\ref{thm:list-spine-rep-nil-cons-no-confusion}; the nil/nil case is the nil finite-spine classifier. In the cons/cons case, one-constructor and
continuation-pair inversion expose comparable tails. Each displayed head lies in
$A_{a_0}$, so both heads are $\hsame$ to $a_0$ and hence $\hsame$ to each other;
apply the induction hypothesis to the exposed tail presentations and rebuild
the $\mathsf{ListClassifierSpec}(\hsame,-,-)$ cons witness.
\end{proof}
