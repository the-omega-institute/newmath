\chapter{A Concrete Naming Certificate for \(\AxisAddUp\)}
\label{ch:concrete-instances-axisadd-namecert}

The additive seed on the \(\Ezero\)-spine is licensed by continuation of zero-spine histories. It is not a primitive operation. The full arithmetic algebra (associativity, commutativity, determinacy) over \(\AxisAddUp\) is left as a horizon for the formalize pipeline; this chapter pins down the certificate skeleton and the kernel facts already available from the finite-kernel layer.

\section{Continuation on the zero-spine}

\begin{definition}[Additive source specification (zero-spine)]
\label{def:axisadd-source-specification}
The source specification \(\mathsf{AxisAddSource}\) consists of ordered pairs \((h, k)\) of zero-spine histories.
\end{definition}
\leandef{BEDC.Derived.AxisZeckendorf.AxisAdd.AxisAddSourceSpec}

\begin{definition}[Additive pattern specification (zero-spine)]
\label{def:axisadd-pattern-specification}
The pattern specification \(\mathsf{AxisAddPattern}\) is the continuation predicate
$$
  \ContR(h, k, r)
$$
together with \(\mathsf{ZeroSpine}(h)\) and \(\mathsf{ZeroSpine}(k)\).
\end{definition}
\leandef{BEDC.Derived.AxisZeckendorf.AxisAdd.AxisAddPatternSpec}

\begin{definition}[Additive classifier specification (zero-spine)]
\label{def:axisadd-classifier-specification}
The classifier specification \(\mathsf{AxisAddClassifier}\) identifies two additive results \(r_1\) and \(r_2\) of \((h, k)\) when both share the same zero-spine pattern witness and \(\HSame(r_1, r_2)\) holds.
\end{definition}
\leandef{BEDC.Derived.AxisZeckendorf.AxisAdd.AxisAddClassifierSpec}

\begin{theorem}[Left unit on the zero-spine]
\label{thm:axisadd-left-unit}
For every zero-spine history \(h\), continuation by the empty history on the left produces \(h\):
$$
  \ContR(\emp, h, h).
$$
\end{theorem}
\begin{proof}
This is the kernel left-unit theorem of \(\ContR\) restricted to a zero-spine input.
\end{proof}

\begin{definition}[Additive stability certificate (zero-spine)]
\label{def:axisadd-stability-certificate}
The stability certificate \(\mathsf{AxisAddStability}\) records the kernel left-unit on the zero-spine. The remaining fields (right unit, associativity, commutativity, determinacy, and zero-spine closure of the result) are horizon obligations whose paper-side proofs are queued for the codex formalize pipeline.
\end{definition}
\leandef{BEDC.Derived.AxisZeckendorf.AxisAdd.AxisAddStabilityCert}
\leanvariant{BEDC.Derived.AxisZeckendorf.AxisAdd.axisAddStabilityCert}

\begin{definition}[Additive ledger policy (zero-spine)]
\label{def:axisadd-ledger-policy}
The ledger policy \(\mathsf{AxisAddLedger}\) records the input pair \((h, k)\) and the continuation witness behind any visible additive result.
\end{definition}
\leandef{BEDC.Derived.AxisZeckendorf.AxisAdd.AxisAddLedgerPolicy}

\section{The certificate}

\begin{theorem}[Concrete \(\AxisAddUp\) naming certificate]
\label{thm:concrete-axisadd-namecert}
The tuple
$$
  \mathsf{NameCert}_{\AxisAddUp}
  :=
  (\mathsf{AxisAddSource},
   \mathsf{AxisAddPattern},
   \mathsf{AxisAddClassifier},
   \mathsf{AxisAddStability},
   \mathsf{AxisAddLedger})
$$
forms a valid naming certificate for the derived additive interface \(\AxisAddUp\) on the \(\Ezero\)-spine.
\end{theorem}
\leandef{BEDC.Derived.AxisZeckendorf.AxisAdd.axisAdd\_namecert}
\begin{proof}
Source, pattern, classifier, stability, and ledger fields are supplied by the preceding definitions. The stability field includes the kernel left-unit fact and explicitly carries the remaining algebraic obligations as horizon entries.
\end{proof}

\begin{corollary}[\(\AxisAddUp\) is licensed, not primitive]
\label{cor:axisaddup-licensed-not-primitive}
The interface \(\AxisAddUp\) is licensed by \(\mathsf{NameCert}_{\AxisAddUp}\). It is not a primitive operation of the finite kernel.
\end{corollary}

\section{Horizon obligations}

The following theorems record the zero-spine closure already checked in Lean and the remaining algebraic horizon.

\begin{theorem}[Continuation preserves zero-spine]
\label{thm:axisadd-cont-preserves-zerospine}
If \(\mathsf{ZeroSpine}(h)\), \(\mathsf{ZeroSpine}(k)\), and \(\ContR(h, k, r)\), then \(\mathsf{ZeroSpine}(r)\).
\end{theorem}
\leanchecked{BEDC.Derived.AxisZeckendorf.AxisAdd.AxisAddCont\_result\_zeroSpine}
\leanvariant{BEDC.Derived.AxisZeckendorf.AxisAdd.AxisAdd\_cont\_result\_zeroSpine}

\begin{theorem}[Right unit on the zero-spine]
\label{thm:axisadd-right-unit}
For every zero-spine history \(h\), \(\ContR(h, \emp, h)\).
\end{theorem}

\begin{theorem}[Associativity on the zero-spine]
\label{thm:axisadd-associativity}
\(\AxisAddUp\) is associative up to \(\HSame\) on zero-spine histories.
\end{theorem}

\begin{theorem}[Commutativity on the zero-spine]
\label{thm:axisadd-commutativity}
\(\AxisAddUp\) is commutative up to \(\HSame\) on zero-spine histories.
\end{theorem}

\begin{remark}[Distinct from \(\AddUp\)]
\(\AxisAddUp\) and \(\AddUp\) act on different carriers: the first uses \(\mathsf{ZeroSpine}\), the second uses \(\mathsf{Unary}\). The two are kernel-distinct, by Theorem~\ref{thm:axisnat-natup-kernel-distinct}; identification across the two carriers requires the bridge of Chapter~\ref{ch:concrete-instances-unary-direction-bridge}.
\end{remark}
