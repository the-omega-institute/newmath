\chapter{A Concrete Naming Certificate for \(\GammaUp\)}
\label{ch:concrete-instances-gamma-namecert}
\origin{human}
\leanchecked{BEDC.Derived.GammaUp.GammaTasteGate\_single\_carrier\_alignment}

This chapter gives a concrete naming-certificate construction for the constructive Gamma-function interface \(\GammaUp\) at the paper-proof level. The construction sits above \(\NatUp\) (\autoref{ch:concrete-instances-nat-namecert}) for the unary factorial, \(\PrimeUp\) (\autoref{ch:concrete-instances-prime-namecert}) for the explicit prime-power decomposition of factorials, \(\RealAnalyticUp\) (\autoref{ch:concrete-instances-real-analytic-namecert}) for the constructive \(\exp\) and \(\log\) used in the analytic extension, \(\ComplexUp\) (\autoref{ch:concrete-instances-complex-namecert}) for the codomain, and \(\HolomorphicUp\) (\autoref{ch:concrete-instances-holomorphic-namecert}) for the meromorphic-extension classifier. The construction proceeds in three layers: integer-point factorial (\(\Gamma(n+1) = n!\)), Weierstrass-product extension to \(\ComplexUp \setminus \{0, -1, -2, \ldots\}\) with explicit unary modulus, and the recurrence \(\Gamma(s+1) = s\,\Gamma(s)\). All convergence witnesses are unary-indexed total functions; the construction uses no \(\mathsf{Classical.choice}\), no \(\mathsf{Quot.sound}\), and no \(\mathsf{propext}\), so the certificate's Lean targets fall under the project's axiom-purity gate.

\section{The pole locus and the Gamma domain}
\label{sec:gamma-domain}

\begin{definition}[Negative-integer pole locus]
\label{def:gamma-pole-locus}
The pole-locus predicate \(\mathsf{GammaPole}(s)\) on \(\ComplexUp\)-history holds when there exists a unary natural \(n\) with \(\hsame(s, -n)\), where \(-n\) is the explicit signed-integer encoding of \autoref{ch:concrete-instances-int-namecert}. Equivalently, the imaginary part of \(s\) equals \(\emp\) and the real part is a non-positive integer.
\end{definition}
\leandef{BEDC.Derived.GammaUp.GammaPoleLocus}

\begin{theorem}[Gamma pole locus has complex carrier witness]
\label{thm:gamma-pole-locus-complex-carrier}
If \(s\) lies in the Gamma pole locus through a unary witness \(n\), then the same witness presents \(s\) as \(\hsame\) to the complex history with real component \(\Eone(n)\) and imaginary component \(\Eone(\emp)\), and \(s\) is carried by \(\ComplexUp\).
\end{theorem}
\begin{proof}
The unary witness \(n\) gives a positive rational denominator for \(\Eone(n)\), while \(\emp\) gives the positive rational denominator for \(\Eone(\emp)\). The product carrier forms the corresponding complex history, and the pole-locus \(\hsame\) witness transports that carrier back to \(s\).
\end{proof}
\leanchecked{BEDC.Derived.GammaUp.GammaPoleLocus\_complex\_carrier\_witness}

\begin{theorem}[Gamma pole locus is not empty]
\label{thm:gamma-pole-locus-not-empty}
If \(s\) lies in the Gamma pole locus, then \(s\) cannot be \(\hsame\) to
\(\emp\).
\end{theorem}
\begin{proof}
The pole-locus witness induces a \(\ComplexUp\) carrier for \(s\) by the
complex-carrier witness theorem. The complex carrier is non-empty, so an
\(\hsame(s,\emp)\) assumption contradicts that carrier.
\end{proof}
\leanchecked{BEDC.Derived.GammaUp.GammaPoleLocus\_not\_empty}

\begin{theorem}[Gamma pole locus is stable under unary append as a complex carrier]
\label{thm:gamma-pole-locus-append-unary-complex-carrier}
If $s$ lies in the Gamma pole locus and $q$ is unary, then
$\mathsf{append}(s,q)$ is carried by $\ComplexUp$ and cannot be $\hsame$ to
$\emp$.
\end{theorem}
\begin{proof}
The pole-locus witness gives a $\ComplexUp$ carrier for $s$. Complex carriers
are closed under unary append, so $\mathsf{append}(s,q)$ is again a complex
carrier. The complex-carrier non-emptiness theorem excludes any comparison of
that appended history with $\emp$.
\end{proof}
\leanchecked{BEDC.Derived.GammaUp.GammaPoleLocus\_append\_unary\_complex\_carrier}

\begin{theorem}[Gamma pole locus hsame transport witness]
\label{thm:gamma-pole-locus-hsame-transport-witness}
If $s\hsame t$ and $s$ lies in the Gamma pole locus, then $t$ lies in the
same pole locus. Moreover, the transported pole claim exposes a concrete unary
witness $n$ with
$$
t\hsame \mathsf{append}(\Eone(n),\Eone(\emp)).
$$
\end{theorem}
\begin{proof}
The pole-locus witness for $s$ supplies a unary history $n$ and an equality
from $s$ to the explicit append history. Compose the symmetry of $s\hsame t$
with that equality. The same unary $n$ then witnesses both the pole-locus
claim for $t$ and the displayed append-history equality.
\end{proof}
\leanchecked{BEDC.Derived.GammaUp.GammaPoleLocus\_hsame\_transport\_witness}

\begin{theorem}[Gamma pole locus witness uniqueness]
\label{thm:gamma-pole-locus-witness-unique}
If the same complex history \(s\) is displayed as a Gamma pole by unary
witnesses \(n\) and \(m\), then \(n\hsame m\).
\end{theorem}
\begin{proof}
The two displayed pole equalities identify the append histories
\(\mathsf{append}(\Eone(n),\Eone(\emp))\) and
\(\mathsf{append}(\Eone(m),\Eone(\emp))\). Cancelling the common right
component \(\Eone(\emp)\) gives \(\Eone(n)\hsame\Eone(m)\), and constructor
inversion gives \(n\hsame m\).
\end{proof}
\leanchecked{BEDC.Derived.GammaUp.GammaPoleLocus\_witness\_unique}

\begin{theorem}[Gamma pole locus suffix source and witness determinacy]
\label{thm:gamma-pole-locus-suffix-source-witness-deterministic}
If $s$ and $t$ lie in the Gamma pole locus and the same suffix $q$ continues
both histories to the same endpoint $sq$, then $s\hsame t$. Moreover, the two
pole displays expose unary witnesses $n$ and $m$ with $n\hsame m$.
\end{theorem}
\begin{proof}
Continuation right cancellation on the two ledgers gives $s\hsame t$. Transport
the pole-locus witness for $t$ back to the source $s$, then apply Gamma
pole-locus witness uniqueness against the original witness for $s$. The two
unary witnesses and their equality form the displayed witness package.
\end{proof}
\leanchecked{BEDC.Derived.GammaUp.GammaPoleLocus\_suffix\_source\_witness\_deterministic}

\begin{definition}[Gamma domain]
\label{def:gamma-domain}
\(\mathsf{GammaDom}(s)\) holds when \(\mathsf{ComplexUp}(s)\) and \(\neg \mathsf{GammaPole}(s)\). The negation is constructive: \(\mathsf{GammaDom}(s)\) is witnessed by an explicit apartness datum \((k, w)\) where \(k\) is a unary precision and \(w\) is a strict-prefix witness showing that the distance from \(s\) to the closest non-positive integer exceeds \(2^{-k}\).
\end{definition}
\leandef{BEDC.Derived.GammaUp.GammaDomainCore}

\begin{theorem}[Gamma domain core is non-empty]
\label{thm:gamma-domain-core-not-empty}
If a Gamma-domain core witness carries \(s\) as a complex history and records
constructive apartness from the pole locus, then \(s\) cannot be \(\hsame\) to
\(\emp\).
\end{theorem}
\begin{proof}
Unpack the Gamma-domain core witness and read its \(\ComplexUp\) carrier for
\(s\). The complex carrier is non-empty, so any \(\hsame(s,\emp)\) assumption
contradicts the carrier.
\end{proof}
\leanchecked{BEDC.Derived.GammaUp.GammaDomainCore\_not\_empty}

\begin{theorem}[Gamma domain core transports exclusions]
\label{thm:gamma-domain-core-hsame-transport-exclusions}
If \(s\hsame t\) and \(s\) is in the Gamma-domain core, then \(t\) is excluded
from the pole locus and \(t\hsame\emp\) is impossible.
\end{theorem}
\begin{proof}
Transport any pole-locus witness for \(t\) back along the symmetric
\(\hsame\)-comparison to obtain a pole-locus witness for \(s\), contradicting
the domain-core apartness field. For the empty-history exclusion, compose
\(s\hsame t\) with \(t\hsame\emp\) and apply the non-empty core theorem.
\end{proof}
\leanchecked{BEDC.Derived.GammaUp.GammaDomainCore\_hsame\_transport\_exclusions}

\begin{theorem}[Gamma domain core hsame transport]
\label{thm:gamma-domain-core-hsame-transport}
If \(s\hsame t\) and \(s\) is in the Gamma-domain core with apartness datum
\(apart\), then \(t\) is in the Gamma-domain core with the same apartness datum,
and \(t\hsame\emp\) is impossible.
\end{theorem}
\begin{proof}
Transport the complex-history carrier of \(s\) along \(s\hsame t\). The unary
apartness datum is unchanged. The pole-locus and empty-history exclusions are
the two exclusion clauses transported by the preceding theorem.
\end{proof}
\leanchecked{BEDC.Derived.GammaUp.GammaDomainCore\_hsame\_transport}

\begin{theorem}[Gamma domain core hsame transport nonempty]
\label{thm:gamma-domain-core-hsame-transport-not-empty}
If \(s\hsame t\) and \(s\) is in the Gamma-domain core, then \(t\) is in the
same domain core and \(t\hsame\emp\) is impossible.
\end{theorem}
\begin{proof}
Transport the \(\ComplexUp\) carrier from \(s\) to \(t\). A pole-locus witness
for \(t\) transports back along the symmetric \(\hsame\)-comparison to a
pole-locus witness for \(s\), contradicting the apartness field of the original
domain witness. Together with the unchanged unary apartness datum, these fields
rebuild the domain core for \(t\), and the non-empty theorem excludes
\(t\hsame\emp\).
\end{proof}
\leanchecked{BEDC.Derived.GammaUp.GammaDomainCore\_hsame\_transport\_not\_empty}

\begin{lemma}[Apartness from the pole locus is decidable]
\label{lem:gamma-pole-apartness-decidable}
For every complex history \(s\) and unary precision \(k\), the question ``does \(s\) lie within \(2^{-k}\) of \(\mathsf{GammaPole}\)?'' is decidable. The decision either supplies a near-pole witness \(\hsame(\mathsf{Re}(s), -n)\) up to precision \(k\) for some specific unary \(n\), or supplies the apartness datum required by \autoref{def:gamma-domain}.
\end{lemma}
\begin{proof}
Compare the rational real-part history \(\mathsf{Re}(s)\) against the finite list of negative-integer candidates \(\{0, -1, -2, \ldots, -K\}\) where \(K\) is the unary length of \(\mathsf{Re}(s)\) at precision \(k\). All comparisons are decidable on rational representations by \(\mathsf{NatUnaryStrictPrefix}\) (\autoref{ch:concrete-instances-nat-namecert}). The imaginary-part check uses the same strict-prefix decidability.
\end{proof}
\leanchecked{BEDC.Derived.GammaUp.GammaPoleLocus\_shape\_partition}

\section{Gamma at unary positive integers via factorial}
\label{sec:gamma-positive-integers}

\begin{definition}[Factorial bridge]
\label{def:gamma-factorial}
For unary \(n \geq \emp\), \(\mathsf{Factorial}(n, m)\) holds when \(m\) is the unary product \(1 \cdot 2 \cdots n\) computed by unary recursion: \(\mathsf{Factorial}(\emp, \Eone(\emp))\) and \(\mathsf{Factorial}(\Eone(p), m)\) iff there exists \(m'\) with \(\mathsf{Factorial}(p, m')\) and \(m = \Eone(p) \cdot m'\) under unary multiplication (\(\PrimeUp\), \autoref{ch:concrete-instances-prime-namecert}).
\end{definition}
\leandef{BEDC.Derived.GammaUp.GammaFactorial}

\begin{theorem}[Gamma at positive integers equals factorial]
\label{thm:gamma-positive-integer}
For unary \(n \geq \emp\), \(\mathsf{Gamma}(n + \Eone(\emp), z)\) holds with \(\hsame(z, m)\) where \(\mathsf{Factorial}(n, m)\). Concretely, \(\Gamma(1) = 1\), \(\Gamma(2) = 1\), \(\Gamma(3) = 2\), and in general \(\Gamma(n+1) = n!\).
\end{theorem}
\begin{proof}
Direct unfolding of \autoref{def:gamma-factorial} together with the recurrence \(\Gamma(s+1) = s\,\Gamma(s)\) of \autoref{thm:gamma-recurrence} below, applied unary-inductively starting from \(\Gamma(1) = 1\) (the empty-product base case).
\end{proof}
\leanchecked{BEDC.Derived.GammaUp.GammaFactorial\_unary\_result}

\begin{remark}[Factorial used by the zeta functional equation]
The functional equation \(\zeta(s) = 2^s \pi^{s-1} \sin(\pi s/2) \Gamma(1-s) \zeta(1-s)\) (\autoref{ch:concrete-instances-zeta-continuation}) requires \(\Gamma(1-s)\) at \(s = -2n\) (negative even integers), which is \(\Gamma(1+2n) = (2n)!\). \autoref{thm:gamma-positive-integer} supplies this value as a closed unary expression.
\end{remark}

\section{Gamma extension via Weierstrass product}
\label{sec:gamma-weierstrass}

\begin{definition}[Weierstrass partial product]
\label{def:gamma-weierstrass-part}
For complex history \(s\) with \(\mathsf{GammaDom}(s)\) and unary index \(N\), the Weierstrass partial product predicate \(\mathsf{GammaPart}(s, N, P_N)\) holds when \(P_N\) is the explicit complex history
$$
P_N \;=\; \frac{1}{s} \prod_{n=1}^{N} \left(1 + \frac{1}{n}\right)^{s} \left(1 + \frac{s}{n}\right)^{-1},
$$
computed by unary recursion using complex multiplication, division (legitimate by \(\mathsf{GammaDom}(s)\) supplying \(\mathsf{CplxNonZero}\) at every factor), and the constructive complex \(\exp\) of \autoref{ch:concrete-instances-real-analytic-namecert} for the rational power \((1 + 1/n)^s = \exp(s \log(1 + 1/n))\).
\end{definition}
\leandef{BEDC.Derived.GammaFunctionUp.GammaWeierstrassPart}

\begin{lemma}[Weierstrass product Cauchy modulus on the Gamma domain]
\label{lem:gamma-weierstrass-cauchy}
For every \(s\) with \(\mathsf{GammaDom}(s)\) at apartness precision \(k_0\), the partial products \(P_N\) form a regular complex Cauchy sequence with explicit modulus \(N(k) := \max(k_0, k) + |\mathsf{Re}(s)|^2 + |\mathsf{Im}(s)|^2\): for all \(M, N \geq N(k)\), the modulus of \(P_M - P_N\) is below \(2^{-k}\).
\end{lemma}
\begin{proof}
Take logarithms of consecutive partial products: \(\log(P_{N+1}/P_N) = s \log(1 + 1/N) - \log(1 + s/N)\). Taylor-expand both logarithms via \(\log(1 + u) = u - u^2/2 + O(u^3)\) using the constructive \(\log\) of \autoref{def:real-analytic-log}; the leading \(s/N\) terms cancel and the residual is of order \(|s|^2/N^2\), giving an explicit geometric tail bound. The modulus \(N(k)\) is the precision at which the geometric tail falls below \(2^{-k}\).
\end{proof}
\leanchecked{BEDC.Derived.GammaFunctionUp.GammaWeierstrassCauchyModulus\_constant}

\begin{definition}[Constructive Gamma]
\label{def:gamma}
For \(s\) with \(\mathsf{GammaDom}(s)\), \(\mathsf{Gamma}(s, z)\) holds when \(z\) is the complex limit certificate of \autoref{lem:gamma-weierstrass-cauchy} via the \(\ComplexLimitUp\) construction of \autoref{ch:concrete-instances-complex-limit-namecert}, applied to the partial products \((P_N)_N\). The witness includes the apartness precision \(k_0\), the modulus \(N\), and the limit \(z\).
\end{definition}
\leandef{BEDC.Derived.GammaFunctionUp.Gamma}

\begin{theorem}[Gamma is holomorphic on the Gamma domain]
\label{thm:gamma-holomorphic}
\(\mathsf{Gamma} : \mathsf{GammaDom} \to \ComplexUp\) is holomorphic in the sense of \(\HolomorphicUp\) (\autoref{ch:concrete-instances-holomorphic-namecert}) on every open disk \(\mathsf{OpenDisk}(z_0, r)\) entirely contained in \(\mathsf{GammaDom}\).
\end{theorem}
\begin{proof}
Term-wise complex differentiability of each Weierstrass factor follows from the constructive \(\exp\) and \(\log\) being holomorphic in their arguments (\autoref{ch:concrete-instances-real-analytic-namecert}). Uniform convergence of the partial products on a compact sub-domain of \(\mathsf{GammaDom}\) is the geometric-tail bound of \autoref{lem:gamma-weierstrass-cauchy}. Apply the holomorphic-limit closure of \(\HolomorphicUp\) (\autoref{ch:concrete-instances-holomorphic-namecert}).
\end{proof}
\leanchecked{BEDC.Derived.GammaFunctionUp.Gamma\_holomorphic\_limit\_certificate\_readback}

\section{Functional equation \(\Gamma(s+1) = s\,\Gamma(s)\)}
\label{sec:gamma-recurrence}

\begin{theorem}[Gamma recurrence]
\label{thm:gamma-recurrence}
For every \(s\) with \(\mathsf{GammaDom}(s)\) and \(\mathsf{GammaDom}(s + \Eone(\emp))\), if \(\mathsf{Gamma}(s, z)\) and \(\mathsf{Gamma}(s + \Eone(\emp), w)\), then \(\mathsf{ComplexClassifier}(w, s \cdot z)\). Equivalently, \(\Gamma(s+1) = s\,\Gamma(s)\) up to \(\ComplexClassifier\).
\end{theorem}
\begin{proof}
Compare the Weierstrass partial products of \(\Gamma(s+1)\) and \(s \cdot \Gamma(s)\): the index-shift identity \(P_N(s+1) = (s/(N+1)) \cdot s \cdot P_N(s) \cdot \text{(small correction)}\) emerges term by term from the explicit factor formula. The small correction is bounded by \autoref{lem:gamma-weierstrass-cauchy}; the limits agree by uniqueness of \(\ComplexLimitUp\).
\end{proof}
\leanchecked{BEDC.Derived.GammaFunctionUp.Gamma\_recurrence\_limit\_pair\_carriers}

\begin{theorem}[Gamma at \(1\)]
\label{thm:gamma-at-one}
\(\mathsf{Gamma}(\Eone(\emp), z)\) gives \(\mathsf{ComplexClassifier}(z, \Eone(\emp))\): \(\Gamma(1) = 1\).
\end{theorem}
\begin{proof}
Specialize \autoref{def:gamma-weierstrass-part} at \(s = 1\): the product becomes \(\frac{1}{1} \prod_{n=1}^{N} (1 + 1/n) (1 + 1/n)^{-1} = 1\) telescopically. The limit is \(1\) by direct unfolding.
\end{proof}
\leanchecked{BEDC.Derived.GammaFunctionUp.GammaAtOne\_constant\_certificate}

\begin{theorem}[Gamma Weierstrass Cauchy modulus hsame transport]
\label{thm:gamma-weierstrass-cauchy-modulus-hsame-transport}
Let $s$ and $t$ be complex histories with $s\hsame t$. Suppose $s$ carries a Gamma-domain core witness whose apartness datum has precision $k_0$, and set
$$
N_s(k):=\max(k_0,k)+|\mathsf{Re}(s)|^2+|\mathsf{Im}(s)|^2.
$$
Then $t$ carries a Gamma-domain core witness with the same apartness datum. Let $P^t$ be the unary-indexed Weierstrass partial-product family obtained from \autoref{def:gamma-weierstrass-part} at $t$. The pair $(P^t,N_s)$ satisfies $\mathsf{CplxRegSeq}(P^t,N_s)$ in the sense of \autoref{def:cplx-regseq}. Equivalently, for every unary precision $k$ and all unary indices $m,n$ accepted above $N_s(k)$, the complex distance from $P^t_m$ to $P^t_n$ is bounded by the unary representation of $2^{-k}$.
\end{theorem}
\begin{proof}
Apply \autoref{thm:gamma-domain-core-hsame-transport} to the comparison $s\hsame t$. The transported witness keeps the original apartness datum, hence keeps the precision $k_0$, while replacing the complex carrier and pole-exclusion fields by their transports to $t$.

Define
$$
N_t(k):=\max(k_0,k)+|\mathsf{Re}(t)|^2+|\mathsf{Im}(t)|^2.
$$
By \autoref{lem:gamma-weierstrass-cauchy}, the Weierstrass partial products at $t$ form a regular complex Cauchy sequence with modulus $N_t$.

It remains only to replace $N_t$ by $N_s$. The comparison $s\hsame t$, together with the product-classifier reading of complex histories in \autoref{def:complex-classifier-specification}, transports the real and imaginary readbacks of $s$ to those of $t$. The absolute-value, squaring, unary maximum, and unary addition continuations used in the displayed modulus are classifier-respecting, so for every unary $k$ there is a history-sameness comparison
$$
N_s(k)\hsame N_t(k).
$$
Now fix $k$ and unary indices $m,n$ accepted above $N_s(k)$ in the threshold clause of \autoref{def:cplx-regseq}. In the equality branch, compose the equality with $N_s(k)\hsame N_t(k)$. In the strict-prefix branch, unpack \autoref{def:nat-unary-strict-prefix} and transport the same nonempty unary tail across the endpoint comparison by \autoref{thm:nat-unary-strict-prefix-cont-hsame-transport}. Thus $m$ and $n$ are accepted above $N_t(k)$ as well. The regularity witness for $(P^t,N_t)$ supplies the required complex-distance bound, and the distance history is unchanged. Since $k,m,n$ were arbitrary, $(P^t,N_s)$ is regular with the transported source modulus.
\end{proof}

\begin{theorem}[Gamma limit certificate hsame transport]
\label{thm:gamma-limit-certificate-hsame-transport}
Let $s,t,z$ be complex histories with $s\hsame t$. Suppose $D_s$ is a $\mathsf{GammaDom}(s)$ core witness with apartness precision $k_0$, and suppose $\mathsf{Gamma}(s,z)$ is witnessed in the sense of \autoref{def:gamma} by the Weierstrass partial-product family $P^s$ of \autoref{def:gamma-weierstrass-part}, the Cauchy modulus
$$
N_s(k):=\max(k_0,k)+|\mathsf{Re}(s)|^2+|\mathsf{Im}(s)|^2,
$$
and a complex-limit modulus $M_s$. Then $\mathsf{Gamma}(t,z)$ holds with the Gamma-domain core witness transported from $D_s$ by \autoref{thm:gamma-domain-core-hsame-transport}, with the Weierstrass partial-product family $P^t$ at $t$, with the same Cauchy modulus $N_s$, and with the same limit point $z$.
\end{theorem}
\begin{proof}
Unpack the displayed Gamma witness by \autoref{def:gamma}. It consists of the domain witness $D_s$, the partial-product family $P^s$, the regularity supplied by \autoref{lem:gamma-weierstrass-cauchy}, and a limit row
$$
\mathsf{CplxLim}(P^s,N_s,z,M_s)
$$
in the sense of \autoref{def:cplx-lim}.

Apply \autoref{thm:gamma-domain-core-hsame-transport} to $s\hsame t$ and $D_s$. This gives the target domain witness $D_t$ with the same apartness datum and the same precision $k_0$. Let $P^t$ be the partial-product family obtained from \autoref{def:gamma-weierstrass-part} using $D_t$. By \autoref{thm:gamma-weierstrass-cauchy-modulus-hsame-transport}, $(P^t,N_s)$ is a regular complex sequence, so the target Cauchy modulus is the source modulus $N_s$.

For every unary index $n$, the partial products $P^s_n$ and $P^t_n$ are history-same. This is unary induction on $n$. In the base product, the constants are identical and the only source-dependent endpoint is transported by $s\hsame t$. In the successor step, the induction hypothesis compares the preceding products; the displayed source comparison transports the factors $1+s/n$, their inverses, and $\exp(s\log(1+1/n))$ to the corresponding factors at $t$. The rational-pair complex operations are classifier-respecting by \autoref{def:complex-stability-certificate}, and the analytic $\exp$ and $\log$ rows are classifier-respecting in the construction of \autoref{ch:concrete-instances-real-analytic-namecert}. Continuation determinacy then identifies the two successor products.

The pointwise comparison $P^s_n\hsame P^t_n$ lets \autoref{thm:complex-limit-sequence-hsame-transport} carry the source limit row to
$$
\mathsf{CplxLim}(P^t,N_s,z,M_s).
$$
If the target limit certificate is presented through another endpoint $z_t$, \autoref{thm:cplx-lim-unique} and \autoref{def:cplx-lim-classifier-specification} classify $z_t$ with $z$ componentwise, and \autoref{lem:cplx-lim-hsame} transports the limit row to the endpoint $z$. Repacking \autoref{def:gamma} with $D_t$, $P^t$, $N_s$, $M_s$, and this limit row gives $\mathsf{Gamma}(t,z)$.
\end{proof}
