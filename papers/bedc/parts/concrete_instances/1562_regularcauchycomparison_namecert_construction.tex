\chapter{Regular Cauchy Comparison NameCert Construction}
\label{ch:concrete-instances-regular-cauchy-comparison-namecert}
\origin{human}

The $\RegularCauchyComparisonUp$ interface records the finite comparison layer used by the Real completeness proof target of \autoref{sec:real-completeness-limit-theorem}. It compares two regular Cauchy names through shared finite windows, displayed dyadic observations, rational tolerance rows, and the downstream $\RealUp$ seal classifier. The packet does not select a host limit, quotient regular streams, or invoke ambient real completeness.

\begin{definition}[Regular Cauchy comparison carrier]
\label{def:regular-cauchy-comparison-carrier}
A $\RegularCauchyComparisonUp$ carrier is a finite $\BHist$ packet
$$
K=(X,Y,W,D,\tau,L,E,H,C,P,N)
$$
with the following visible rows:
\begin{enumerate}
\item $X$ and $Y$ are the two displayed $\RegSeqRatUp$ regular Cauchy names under comparison;
\item $W$ is the shared $\StreamNameUp$ window schedule used by both names for every inspected precision request;
\item $D$ is the displayed $\DyadicRatCoreUp$ observation surface for the two windows named by $W$;
\item $\tau$ is the rational tolerance row against which the dyadic observations are compared;
\item $L$ is the finite ledger recording which window reads, dyadic observations, and tolerance rows have been inspected;
\item $E$ is the $\RealUp$ seal-provenance row reached only after the common window and dyadic comparison rows have been displayed;
\item $H$ stores componentwise $\hsame$ transports for the two names, the shared schedule, dyadic observations, tolerance row, finite ledger, and seal row;
\item $C$ stores the displayed $\Cont$ routes by which consumer reads reach those same finite rows;
\item $P$ is the $\Pkg$ provenance over $\RegSeqRatUp$, $\StreamNameUp$, $\DyadicRatCoreUp$, $\RealUp$, $\BHist$, $\hsame$, $\Cont$, and $\NameCert$;
\item $N$ is the local $\NameCert_{\RegularCauchyComparisonUp}$ row.
\end{enumerate}
The carrier contains no quotient of regular names, no host real equality test, no selected limit, and no completeness witness outside the displayed packet.
\end{definition}

\begin{definition}[Regular Cauchy comparison classifier]
\label{def:regular-cauchy-comparison-classifier}
For accepted $\RegularCauchyComparisonUp$ packets $K$ and $K'$, the classifier $\mathsf{RegularCauchyComparisonClassifier}(K,K')$ holds exactly when all displayed finite comparison windows refine to the same transported dyadic tolerance row and the $\RealUp$ seal rows classify identically after those refinements.

Thus the classifier reads only the two $\RegSeqRatUp$ names, the shared $\StreamNameUp$ schedule, displayed $\DyadicRatCoreUp$ observations, rational tolerance row, finite ledger, $\RealUp$ seal provenance, $\BHist$, $\hsame$, $\Cont$, $\Pkg$, and $\NameCert$ rows. It does not compare host sequences, quotient representatives, selected limits, or unrecorded real carriers.
\end{definition}

\begin{theorem}[Regular Cauchy comparison NameCert obligations]
\label{thm:regular-cauchy-comparison-namecert-obligations}
The local $\NameCert_{\RegularCauchyComparisonUp}$ surface consists of five finite rows:
\begin{enumerate}
\item carrier admission: accepted packets are exactly the finite $\BHist$ carriers of \autoref{def:regular-cauchy-comparison-carrier};
\item classifier stability: \autoref{def:regular-cauchy-comparison-classifier} is preserved by displayed $\hsame$ transports and $\Cont$ routes on the two $\RegSeqRatUp$ names, shared $\StreamNameUp$ schedule, dyadic observations, tolerance row, ledger, seal row, package provenance, and local naming row;
\item window exactness: every comparison request reads both names through the same displayed finite schedule $W$ before inspecting the dyadic observation row $D$;
\item tolerance exactness: the dyadic observations are compared only against the rational tolerance row $\tau$ recorded in the packet ledger;
\item seal handoff: the $\RealUp$ classifier is consumed only after the common-window comparison and tolerance rows have been displayed.
\end{enumerate}
This surface is a comparison certificate for the Real completeness capstone and does not assert real completeness.
\end{theorem}

\begin{theorem}[Regular Cauchy comparison window exactness]
\label{thm:regular-cauchy-comparison-window-exactness}
For every accepted $\RegularCauchyComparisonUp$ packet
$$
K=(X,Y,W,D,\tau,L,E,H,C,P,N)
$$
and every displayed comparison request, both $\RegSeqRatUp$ names are read through the same finite $\StreamNameUp$ schedule $W$, the selected observations are exactly the $\DyadicRatCoreUp$ rows in $D$, and the tolerance comparison is exactly the rational row $\tau$ recorded in $L$. No consumer read may split the two names across unrelated windows or replace $\tau$ by an unrecorded tolerance.
\end{theorem}

\begin{theorem}[Regular Cauchy comparison Real classifier handoff]
\label{thm:regular-cauchy-comparison-real-classifier-handoff}
Every $\RealUp$ classifier read exported by an accepted $\RegularCauchyComparisonUp$ packet factors through the two $\RegSeqRatUp$ names, the shared $\StreamNameUp$ window schedule, the displayed $\DyadicRatCoreUp$ observations, the rational tolerance row, the finite ledger, $\RealUp$ seal provenance, $\BHist$, $\hsame$, $\Cont$, $\Pkg$, and $\NameCert$ rows. The handoff supplies the finite comparison layer used by \autoref{thm:real-completeness-limit-exists-unique} without adding quotient stream equality, a selected host limit, host real equality, or an ambient completeness theorem.
\end{theorem}
