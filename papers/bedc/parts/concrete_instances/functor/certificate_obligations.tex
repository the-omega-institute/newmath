\begin{theorem}[Functor prefix hom-carrier composite middle-prefix determinacy]
\label{thm:functor-prefix-hom-carrier-comp-middle-prefix-deterministic}
For histories $p,q,a,b,c,f,g,fg$, suppose $f$ is a category hom-carrier from
$\mathsf{append}(p,a)$ to $\mathsf{append}(p,b)$, $g$ is a category
hom-carrier from $\mathsf{append}(q,b)$ to $\mathsf{append}(p,c)$, and
$\ContR(f,g,fg)$ holds. If $fg$ is displayed as a category hom-carrier from
$\mathsf{append}(p,a)$ to $\mathsf{append}(p,c)$, then $\hsame(p,q)$.
\end{theorem}
\begin{proof}
Category hom-carrier composite middle-object determinacy identifies
$\mathsf{append}(p,b)$ with $\mathsf{append}(q,b)$. Cancelling the common right
suffix $b$ gives $\hsame(p,q)$.
\end{proof}
\leanchecked{BEDC.Derived.FunctorUp.FunctorPrefixHomCarrier\_comp\_middle\_prefix\_deterministic}

\begin{theorem}[Functor prefix hom-carrier composition family public readback]
\label{thm:functor-prefix-hom-carrier-comp-family-public-readback}
For histories $a,b,c,f,g,fg,displayed$, suppose every unary prefix $p$ carries
$f$ as a hom-carrier from $\mathsf{append}(p,a)$ to $\mathsf{append}(p,b)$,
every unary prefix carries $g$ from $\mathsf{append}(p,b)$ to
$\mathsf{append}(p,c)$, $\ContR(f,g,fg)$ holds, and every unary prefix carries
$displayed$ from $\mathsf{append}(p,a)$ to $\mathsf{append}(p,c)$. Then
$$
\HSame(fg,displayed).
$$
\end{theorem}
\begin{proof}
Evaluate the three prefix-indexed carrier families at the empty unary prefix.
The two factor carriers close under category composition, and morphism
determinacy compares the resulting composite carrier with the displayed carrier.
\end{proof}
\leanchecked{BEDC.Derived.FunctorUp.FunctorPrefixHomCarrier\_comp\_family\_public\_readback}

\begin{theorem}[Functor prefix hom-carrier composite zero-headed component impossibility]
\label{thm:functor-prefix-hom-carrier-comp-zero-headed-component-absurd}
For histories $p,a,b,c,f,g,fg$, two prefixed hom-carriers composing over the
middle object $b$ cannot have any zero-headed prefix, object, factor morphism,
or composite morphism component:
$$
\mathsf{CategoryHomCarrier}(\mathsf{append}(p,a),\mathsf{append}(p,b),f)
\land
\mathsf{CategoryHomCarrier}(\mathsf{append}(p,b),\mathsf{append}(p,c),g)
\land
\ContR(f,g,fg)
\Longrightarrow
\mathsf{False}
$$
whenever one of $p,a,b,c,f,g,fg$ is displayed as \(\Ezero(w)\).
\end{theorem}
\begin{proof}
Close the two hom-carriers under composition to obtain the carrier for
$fg$. The zero-headed prefix, source, middle, and left morphism cases reduce to
the left factor exclusion; the target and right morphism cases reduce to the
right factor exclusion; the composite morphism case reduces to the closed
composite carrier.
\end{proof}
\leanchecked{BEDC.Derived.FunctorUp.FunctorPrefixHomCarrier\_comp\_zero\_headed\_component\_absurd}

\begin{theorem}[Functor prefix composite nonempty target component split]
\label{thm:functor-prefix-hom-carrier-comp-nonempty-target-component-split}
For prefixed hom-carriers
$$
\mathsf{CategoryHomCarrier}(\mathsf{append}(p,a),\mathsf{append}(p,b),f)
$$
and
$$
\mathsf{CategoryHomCarrier}(\mathsf{append}(p,b),\mathsf{append}(p,c),g),
$$
if $\ContR(f,g,fg)$ and $fg$ is not empty, then the displayed target

$$
(\hsame(p,\emp)\to\bot)\vee(\hsame(c,\emp)\to\bot).
$$
\end{theorem}
\begin{proof}
The category composite nonempty-target obstruction applies to the two
prefixed hom-carriers and their continuation composite, giving
$\hsame(\mathsf{append}(p,c),\emp)\to\bot$. BEDC append nonemptiness then
separates the prefixed target into a nonempty prefix component or a nonempty
target-tail component.
\end{proof}
\leanchecked{BEDC.Derived.FunctorUp.FunctorPrefixHomCarrier\_comp\_nonempty\_target\_component\_split}

\section{The certificate}

\begin{definition}[Functor source specification]
\label{def:functor-source-specification}
The source is a pair of certified categories.
\end{definition}

\begin{definition}[Functor pattern specification]
\label{def:functor-pattern-specification}
The pattern is structure-preserving mapping formation over object and morphism carriers.
\end{definition}

\begin{definition}[Functor classifier specification]
\label{def:functor-classifier-specification}
The classifier relates functors pointwise on object images and morphism images using the codomain category classifiers.
\end{definition}

\begin{definition}[Functor stability certificate]
\label{def:functor-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item pointwise object-map classifier reflexivity
  \item pointwise object-map classifier symmetry
  \item pointwise object-map classifier transitivity
  \item pointwise morphism-map classifier reflexivity
  \item pointwise morphism-map classifier symmetry
  \item pointwise morphism-map classifier transitivity
  \item identity preservation
  \item composition preservation
  \item stability under source and target classifier replacement
\end{enumerate}
\end{definition}

\begin{definition}[Functor ledger policy]
\label{def:functor-ledger-policy}
Record the two category ledgers and the preservation witnesses.
\end{definition}

\begin{definition}[\(\FunctorUp\) certificate obligations]
\label{def:functor-certificate-obligations}
The carrier, source, pattern, classifier, stability, and ledger data listed above are the obligations required for the interface name \(\FunctorUp\).
\end{definition}

\begin{theorem}[Concrete prefix functor certificate]
\label{thm:functor-prefix-concrete-certificate}
For every unary history $p$, the object assignment
$a\mapsto \mathsf{append}(p,a)$ and morphism assignment $f\mapsto f$ form a
concrete $\FunctorUp$ certificate on the unary-continuation $\CategoryUp$
carrier. The carrier field $\mathsf{PrefixFunctorCarrier}(p)$ is supplied by
\autoref{thm:prefix-functor-carrier-from-unary-prefix}; this same carrier
theorem packages the prefix preservation, prefix reflection, and prefixed
composition fields. Public readback is supplied by
\autoref{thm:prefix-functor-carrier-comp-public-readback}.
\end{theorem}
\begin{proof}
Use the roadmap ten-field template for the prefix endofunctor target.
\begin{enumerate}
  \item The traditional target is the prefix endofunctor on the unary
  continuation category, and the BEDC source is the checked
  unary-continuation $\CategoryUp$ carrier.
  \item The carrier choice is $\mathsf{PrefixFunctorCarrier}(p)$. Apply
  \autoref{thm:prefix-functor-carrier-from-unary-prefix} to the unary history
  $p$ to obtain this field.
  \item The classifier choice is pointwise: objects are compared by the
  unary-continuation object classifier, and morphisms are compared by the
  displayed hom-carrier classifier. No quotient or external equality is added.
  \item The stability fields are identity preservation, composition
  preservation, and source-target classifier transport. These are the carrier
  fields packaged by \autoref{thm:prefix-functor-carrier-from-unary-prefix}.
  \item The ledger records that the morphism map is the identity map on
  displayed morphism histories, while the object ledger stores only the fixed
  unary prefix $p$.
  \item The core theorem list and exactness predicate are the displayed
  carrier, identity, composition, transport, and readback clauses. The standard
  bridge is the public comparison of the prefixed composite with the original
  composite, supplied by
  \autoref{thm:prefix-functor-carrier-comp-public-readback}. The resulting
  status is a concrete paper certificate for $\FunctorUp$.
\end{enumerate}
\end{proof}

\begin{theorem}[Semantic $\FunctorUp$ prefix certificate]
\label{thm:functor-prefix-semantic-namecert}
For every unary history $p$, the source, pattern, classifier, stability, and
ledger data of
Definitions~\ref{def:functor-source-specification}--\ref{def:functor-ledger-policy},
together with \autoref{thm:functor-prefix-concrete-certificate}, form a
semantic naming certificate for the prefix endofunctor interface $\FunctorUp$
over the unary-continuation $\CategoryUp$. The carrier is
$\mathsf{PrefixFunctorCarrier}(p)$; the object component sends $a$ to
$\mathsf{append}(p,a)$; the hom component sends each displayed morphism history
to itself; the classifier is pointwise $\hsame$ on object histories and hom
histories; and the ledger records the fixed prefix $p$ together with the
identity map on displayed morphism histories.
\end{theorem}
\begin{proof}
Start from the concrete certificate of
\autoref{thm:functor-prefix-concrete-certificate}. Its carrier field is
$\mathsf{PrefixFunctorCarrier}(p)$, supplied by
\autoref{thm:prefix-functor-carrier-from-unary-prefix}, so the semantic carrier
row is the same object-level package rather than a separate witness.

The classifier row is pointwise. Object images are compared by $\hsame$ after
the prefix operation, and hom images are compared by $\hsame$ on the displayed
hom histories because the morphism map is the identity map. The source and
pattern rows are therefore exactly the source, pattern, and classifier
specifications already listed for $\FunctorUp$.

The stability rows are the checked preservation clauses consumed by the
concrete certificate. The prefix carrier theorem
\autoref{thm:prefix-functor-carrier-from-unary-prefix} packages identity
preservation, composition preservation, and $\hsame$ transport, while
\autoref{thm:prefix-functor-carrier-comp-public-readback} supplies the public
readback clause.

Finally, the ledger row contains no hidden quotient data. It records the unary
prefix $p$ on objects and records that morphism histories are read through the
identity map. These rows are precisely the ten-field certificate data, so they
assemble as the semantic naming certificate for the prefix $\FunctorUp$
interface over $\CategoryUp$.
\end{proof}

\begin{theorem}[Prefix functor two-sided identity law]
\label{thm:functor-prefix-hom-carrier-two-sided-identity-law}
For histories $p,a,b,f,left,right$, suppose $p$ is unary, $f$ is a
category hom-carrier from $a$ to $b$, $\ContR(\emp,f,left)$ holds, and
$\ContR(f,\emp,right)$ holds. Then
$$
\mathsf{CategoryHomCarrier}(\mathsf{append}(p,a),\mathsf{append}(p,b),left)
\wedge
\mathsf{CategoryHomCarrier}(\mathsf{append}(p,a),\mathsf{append}(p,b),right)
\wedge
\HSame(left,f)\wedge\HSame(right,f)\wedge\HSame(left,right).
$$
\end{theorem}
\begin{proof}
First transport the displayed hom-carrier for $f$ across the unary prefix.
Apply \autoref{thm:prefix-functor-carrier-from-unary-prefix} to the unary
witness for $p$ and to the carrier from $a$ to $b$. This gives $f$ as a
hom-carrier from $\mathsf{append}(p,a)$ to $\mathsf{append}(p,b)$.

Next apply \autoref{thm:category-hom-carrier-two-sided-identity-law} to this
prefixed carrier and to the two continuation ledgers
$\ContR(\emp,f,left)$ and $\ContR(f,\emp,right)$.

Project the resulting conjunction. The first two projections are the prefixed
carrier conclusions for $left$ and $right$; the remaining projections are
$\HSame(left,f)$, $\HSame(right,f)$, and $\HSame(left,right)$.
\end{proof}
\leanchecked{BEDC.Derived.FunctorUp.FunctorPrefixHomCarrier\_two\_sided\_identity\_law}

\begin{definition}[Opposite functor certificate data]
\label{def:opposite-functor-certificate-data}
Let $\mathcal C$ and $\mathcal D$ be $\CategoryUp$ certificates, and let
$F:\mathcal C\to\mathcal D$ denote a $\FunctorUp$ certificate with object map
$F_0$ and morphism map $F_1$. Over the opposite category data of
\autoref{def:opposite-category-certificate-data}, define
$F^{\mathrm{op}}:\mathcal C^{\mathrm{op}}\to\mathcal D^{\mathrm{op}}$ by
$$
F^{\mathrm{op}}_0(a):=F_0(a),
\qquad
F^{\mathrm{op}}_1(u):=F_1(u).
$$
A displayed morphism $u:a\to b$ in $\mathcal C^{\mathrm{op}}$ is read as the
same raw morphism $u:b\to a$ in $\mathcal C$; therefore
$F^{\mathrm{op}}_1(u)$ is read as $F_1(u):F_0(b)\to F_0(a)$ in
$\mathcal D$, hence as a displayed morphism $F_0(a)\to F_0(b)$ in
$\mathcal D^{\mathrm{op}}$.
\end{definition}

\begin{theorem}[Opposite functor certificate]
\label{thm:opposite-functor-certificate}
If $\mathcal C$ and $\mathcal D$ satisfy the $\CategoryUp$ certificate
obligations and $F:\mathcal C\to\mathcal D$ satisfies the $\FunctorUp$
certificate obligations of \autoref{def:functor-certificate-obligations}, then
the data of \autoref{def:opposite-functor-certificate-data} satisfy the
$\FunctorUp$ certificate obligations from $\mathcal C^{\mathrm{op}}$ to
$\mathcal D^{\mathrm{op}}$.
\end{theorem}
\begin{proof}
By \autoref{thm:opposite-category-certificate}, the source and codomain
opposite data are certified categories, so the source specification required by
\autoref{def:functor-source-specification} is present. The carrier and pattern
fields are the same two maps $F_0$ and $F_1$, only with the hom-carrier
endpoints read through the opposite category convention.

For hom-carrier landing, let $u:a\to b$ be displayed in
$\mathcal C^{\mathrm{op}}$. By
\autoref{def:opposite-category-certificate-data}, this is the original
hom-carrier assertion $u:b\to a$ in $\mathcal C$. The carrier field of $F$
sends it to a displayed morphism $F_1(u):F_0(b)\to F_0(a)$ in $\mathcal D$.
Unfolding the codomain opposite hom-carrier reads that same assertion as
$F_1(u):F_0(a)\to F_0(b)$ in $\mathcal D^{\mathrm{op}}$, which is exactly the
landing field for $F^{\mathrm{op}}$.

The classifier specification is inherited pointwise. Object classifiers are
unchanged by the opposite construction, and the morphism classifier of
$\mathcal D^{\mathrm{op}}$ is the morphism classifier of $\mathcal D$ on the
same displayed morphism representatives. Thus the pointwise object-map,
morphism-map, reflexivity, and transitivity clauses from
\autoref{def:functor-classifier-specification} and
\autoref{def:functor-stability-certificate} are the original clauses for $F$.

For identity preservation, the identity at $a$ in $\mathcal C^{\mathrm{op}}$
is the identity at $a$ in $\mathcal C$, and the identity at $F_0(a)$ in
$\mathcal D^{\mathrm{op}}$ is the identity at $F_0(a)$ in $\mathcal D$. The
identity preservation witness for $F$ therefore gives the required
$\mathcal D^{\mathrm{op}}$ morphism-classifier witness for $F^{\mathrm{op}}$.

For composition preservation, let $u:a\to b$ and $v:b\to c$ be displayed in
$\mathcal C^{\mathrm{op}}$, and let $w$ be their displayed composite there.
Unfolding opposite composition gives the original source ledger
$$
\mathsf{Comp}_{\mathcal C}(v,u,w).
$$
The composition preservation field of $F$ gives the codomain classifier
comparison between $F_1(w)$ and the composite selected by
$\mathsf{Comp}_{\mathcal D}(F_1(v),F_1(u),-)$. By the definition of opposite
composition in $\mathcal D^{\mathrm{op}}$, that is precisely the composite
selected by
$$
\mathsf{Comp}_{\mathcal D^{\mathrm{op}}}(F_1(u),F_1(v),-).
$$
Since the morphism classifier is unchanged, this is the required composition
preservation witness for $F^{\mathrm{op}}$.

Stability under source and target classifier replacement is also inherited.
A source-endpoint replacement in the opposite category is a target-endpoint
replacement in the original category, and conversely; the original stability
field applies with the endpoint order swapped. The ledger policy of
\autoref{def:functor-ledger-policy} records the same category ledgers and the
same preservation witnesses, read through the two opposite-category
certificates.
\end{proof}

\begin{definition}[Split monomorphism in a certified category]
\label{def:category-split-monomorphism}
Let $\mathcal C$ be a $\CategoryUp$ certificate. A displayed morphism
$f:a\to b$ is a split monomorphism when there are a displayed morphism
$g:b\to a$, a source identity display $i_a:a\to a$, and a composite display
$c:a\to a$ such that $\mathsf{HomCarrier}_{\mathcal C}(a,b,f)$,
$\mathsf{HomCarrier}_{\mathcal C}(b,a,g)$, $i_a$ is the identity witness at
$a$, $\mathsf{Comp}_{\mathcal C}(f,g,c)$ records $g\circ_{\mathcal C} f$,
and $c\sim_{\mathcal C} i_a$ in the morphism classifier at $(a,a)$. The
morphism $g$ is a left-inverse witness for $f$.
\end{definition}

\begin{theorem}[Functor preserves split monomorphisms]
\label{thm:functor-preserves-split-monomorphism}
Let $\mathcal C$ and $\mathcal D$ be $\CategoryUp$ certificates, and let
$F:\mathcal C\to\mathcal D$ be a $\FunctorUp$ certificate with object map
$F_0$ and morphism map $F_1$. If $f:a\to b$ is a split monomorphism in
$\mathcal C$ with left-inverse witness $g:b\to a$, then
$F_1(f):F_0(a)\to F_0(b)$ is a split monomorphism in $\mathcal D$ with
left-inverse witness $F_1(g):F_0(b)\to F_0(a)$.
\end{theorem}
\begin{proof}
Unpack the source split-monomorphism data from
\autoref{def:category-split-monomorphism}. Thus $f$ and $g$ are displayed
hom-carriers in $\mathcal C$, $i_a$ is the chosen identity display at $a$, and
there is a composite display $c$ with $\mathsf{Comp}_{\mathcal C}(f,g,c)$ and
$c\sim_{\mathcal C} i_a$.

The hom-carrier landing field of the functor carrier sends $f$ and $g$ to
hom-carriers $F_1(f):F_0(a)\to F_0(b)$ and
$F_1(g):F_0(b)\to F_0(a)$ in $\mathcal D$. Apply the composition preservation
field of \autoref{def:functor-stability-certificate} to the source composite
ledger $\mathsf{Comp}_{\mathcal C}(f,g,c)$. It supplies a codomain composite
display $d$ with
$$
\mathsf{Comp}_{\mathcal D}(F_1(f),F_1(g),d)
$$
and a morphism-classifier comparison between $d$ and $F_1(c)$.

The morphism-classifier stability field for $F$ transports the source
comparison $c\sim_{\mathcal C} i_a$ to a codomain comparison
$F_1(c)\sim_{\mathcal D}F_1(i_a)$. Identity preservation for $F$ identifies
$F_1(i_a)$ with the identity display $i_{F_0(a)}$ of $\mathcal D$. Symmetry
and transitivity of the codomain morphism classifier, supplied by the
$\CategoryUp$ stability laws, compose these comparisons to obtain
$d\sim_{\mathcal D} i_{F_0(a)}$.

The displayed morphism $F_1(g)$, the composite display $d$, and the classifier
comparison with $i_{F_0(a)}$ form the split-monomorphism witness for $F_1(f)$
in $\mathcal D$.
\end{proof}

\begin{definition}[Split epimorphism in a certified category]
\label{def:category-split-epimorphism}
Let $\mathcal C$ be a $\CategoryUp$ certificate. A displayed morphism
$f:a\to b$ is a split epimorphism when there are a displayed morphism
$g:b\to a$, a target identity display $i_b:b\to b$, and a composite display
$c:b\to b$ such that $\mathsf{HomCarrier}_{\mathcal C}(a,b,f)$,
$\mathsf{HomCarrier}_{\mathcal C}(b,a,g)$, $i_b$ is the identity witness at
$b$, $\mathsf{Comp}_{\mathcal C}(g,f,c)$ records $f\circ_{\mathcal C} g$, and
$c\sim_{\mathcal C} i_b$ in the morphism classifier at $(b,b)$. The morphism
$g$ is a right-inverse witness for $f$.
\end{definition}

\begin{theorem}[Functor preserves split epimorphisms]
\label{thm:functor-preserves-split-epimorphism}
Let $\mathcal C$ and $\mathcal D$ be $\CategoryUp$ certificates, and let
$F:\mathcal C\to\mathcal D$ be a $\FunctorUp$ certificate with object map
$F_0$ and morphism map $F_1$. If $f:a\to b$ is a split epimorphism in
$\mathcal C$ with right-inverse witness $g:b\to a$, then
$F_1(f):F_0(a)\to F_0(b)$ is a split epimorphism in $\mathcal D$ with
right-inverse witness $F_1(g):F_0(b)\to F_0(a)$.
\end{theorem}
\begin{proof}
Unpack the source split-epimorphism data from
\autoref{def:category-split-epimorphism}. Thus $f$ and $g$ are displayed
hom-carriers in $\mathcal C$, $i_b$ is the chosen identity display at $b$, and
there is a composite display $c$ with $\mathsf{Comp}_{\mathcal C}(g,f,c)$ and
$c\sim_{\mathcal C} i_b$.

The hom-carrier landing field of the functor carrier sends $f$ and $g$ to
hom-carriers $F_1(f):F_0(a)\to F_0(b)$ and
$F_1(g):F_0(b)\to F_0(a)$ in $\mathcal D$. Apply the composition preservation
field of \autoref{def:functor-stability-certificate} to the source composite
ledger $\mathsf{Comp}_{\mathcal C}(g,f,c)$. It supplies a codomain composite
display $d$ with
$$
\mathsf{Comp}_{\mathcal D}(F_1(g),F_1(f),d)
$$
and a morphism-classifier comparison between $d$ and $F_1(c)$.

The morphism-classifier stability field for $F$ transports the source
comparison $c\sim_{\mathcal C} i_b$ to a codomain comparison
$F_1(c)\sim_{\mathcal D}F_1(i_b)$. Identity preservation for $F$ identifies
$F_1(i_b)$ with the identity display $i_{F_0(b)}$ of $\mathcal D$. Symmetry and
transitivity of the codomain morphism classifier, supplied by the $\CategoryUp$
stability laws, compose these comparisons to obtain
$d\sim_{\mathcal D} i_{F_0(b)}$.

The displayed morphism $F_1(g)$, the composite display $d$, the codomain target
identity display $i_{F_0(b)}$, and the classifier comparison with
$i_{F_0(b)}$ form the split-epimorphism witness for $F_1(f)$ in $\mathcal D$.
\end{proof}

\begin{definition}[Split isomorphism witness pair]
\label{def:category-split-isomorphism-witness-pair}
Let $\mathcal C$ be a $\CategoryUp$ certificate. A displayed morphism
$f:a\to b$ carries a split isomorphism witness pair when it carries a
split-monomorphism witness in the sense of
\autoref{def:category-split-monomorphism} with left-inverse witness
$g_L:b\to a$ and a split-epimorphism witness in the sense of
\autoref{def:category-split-epimorphism} with right-inverse witness
$g_R:b\to a$. Thus the package stores the two oriented composite certificates
$$
g_L\circ_{\mathcal C} f\sim_{\mathcal C}\mathsf{id}_a,
\qquad
f\circ_{\mathcal C} g_R\sim_{\mathcal C}\mathsf{id}_b.
$$
The two inverse-side witnesses are part of the data; no equality between
$g_L$ and $g_R$ is required by this package.
\end{definition}

\begin{theorem}[Functor preserves split isomorphism witness pairs]
\label{thm:functor-preserves-split-isomorphism-witness-pair}
Let $\mathcal C$ and $\mathcal D$ be $\CategoryUp$ certificates, and let
$F:\mathcal C\to\mathcal D$ be a $\FunctorUp$ certificate with object map
$F_0$ and morphism map $F_1$. If $f:a\to b$ carries a split isomorphism
witness pair in $\mathcal C$ with left-inverse witness $g_L:b\to a$ and
right-inverse witness $g_R:b\to a$, then
$F_1(f):F_0(a)\to F_0(b)$ carries a split isomorphism witness pair in
$\mathcal D$ whose left-inverse witness is $F_1(g_L)$ and whose right-inverse
witness is $F_1(g_R)$.
\end{theorem}
\begin{proof}
Unpack the witness pair by
\autoref{def:category-split-isomorphism-witness-pair}. Its left-oriented field
is a split-monomorphism witness for $f$ with left-inverse witness $g_L$, and
its right-oriented field is a split-epimorphism witness for $f$ with
right-inverse witness $g_R$.

Apply \autoref{thm:functor-preserves-split-monomorphism} to the left-oriented
field. This gives a split-monomorphism witness in $\mathcal D$ for
$F_1(f):F_0(a)\to F_0(b)$, with left-inverse witness
$F_1(g_L):F_0(b)\to F_0(a)$. Its codomain composite certificate is classified
with the identity display at $F_0(a)$.

Apply \autoref{thm:functor-preserves-split-epimorphism} to the right-oriented
field. This gives a split-epimorphism witness in $\mathcal D$ for the same
forward morphism $F_1(f):F_0(a)\to F_0(b)$, with right-inverse witness
$F_1(g_R):F_0(b)\to F_0(a)$. Its codomain composite certificate is classified
with the identity display at $F_0(b)$.

The two transported fields have the same displayed forward morphism and the
same image endpoints. Packing them by
\autoref{def:category-split-isomorphism-witness-pair} gives the required split
isomorphism witness pair in $\mathcal D$.
\end{proof}
