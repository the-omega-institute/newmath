\begin{theorem}[Prefix functor visible morphism strict prefix]
\label{thm:functor-prefix-e1-visible-morphism-strict-prefix}
For histories $p,a,r,k$, a prefixed visible morphism from
$\mathsf{append}(p,\Eone(a))$ to $\mathsf{append}(p,\Eone(r))$ with morphism
$\Eone(k)$, together with nonempty $k$, exposes a strict unary prefix from
$\Eone(a)$ to $r$:
$$
\mathsf{CategoryHomCarrier}(\mathsf{append}(p,\Eone(a)),
\mathsf{append}(p,\Eone(r)),\Eone(k))
\Longrightarrow
k\neq \emp
\Longrightarrow
\mathsf{NatUnaryStrictPrefix}(\Eone(a),r).
$$
\end{theorem}
\begin{proof}
Apply prefixed visible endpoint morphism exactness. Its forward direction
recovers unary membership of $k$ and the continuation $\ContR(\Eone(a),k,r)$.
The nonempty witness for $k$ is therefore exactly the strict-prefix tail.
\end{proof}
\leanchecked{BEDC.Derived.FunctorUp.FunctorPrefixHomCarrier\_e1\_visible\_morphism\_strict\_prefix}

\begin{theorem}[Prefix functor visible morphism endpoint contradiction]
\label{thm:functor-prefix-e1-visible-morphism-endpoint-hsame-absurd}
For histories $p,a,r,k$, a prefixed visible morphism with nonempty visible tail
cannot have its unprefixed source endpoint $\hsame$ to the target endpoint:
$$
\mathsf{CategoryHomCarrier}(\mathsf{append}(p,\Eone(a)),
\mathsf{append}(p,\Eone(r)),\Eone(k))
\Longrightarrow
k\neq \emp
\Longrightarrow
\hsame(\Eone(a),r)
\Longrightarrow
\bot.
$$
\end{theorem}
\begin{proof}
The strict-prefix readback supplies a nonempty unary tail from $\Eone(a)$ to
$r$. The endpoint $\hsame$ comparison would force this tail to be empty by
strict-prefix endpoint cancellation, contradicting the displayed nonemptiness.
\end{proof}
\leanchecked{BEDC.Derived.FunctorUp.FunctorPrefixHomCarrier\_e1\_visible\_morphism\_endpoint\_hsame\_absurd}

\begin{theorem}[Prefix functor visible morphism unique strict boundary]
\label{thm:functor-prefix-visible-morphism-unique-strict-boundary}
For histories $p,a,r,k,k'$, two prefixed visible morphisms between the same
one-headed endpoints have the same visible payload, and any nonempty first
payload exposes the strict endpoint boundary:
$$
\mathsf{CategoryHomCarrier}(\mathsf{append}(p,\Eone(a)),
\mathsf{append}(p,\Eone(r)),\Eone(k))
\Longrightarrow
\mathsf{CategoryHomCarrier}(\mathsf{append}(p,\Eone(a)),
\mathsf{append}(p,\Eone(r)),\Eone(k'))
\Longrightarrow
k\neq\emp
\Longrightarrow
\hsame(k,k')\wedge\mathsf{NatUnaryStrictPrefix}(\Eone(a),r)\wedge
(\hsame(\Eone(a),r)\to\bot).
$$
\end{theorem}
\begin{proof}
Apply category morphism determinacy to the two displayed hom-carriers. This
gives $\hsame(\Eone(k),\Eone(k'))$, and one-headed no-confusion descends it to
$\hsame(k,k')$.

Apply the strict-prefix readback theorem to the first displayed carrier and
the nonempty witness for $k$. This supplies
$\mathsf{NatUnaryStrictPrefix}(\Eone(a),r)$.

Finally, apply the endpoint contradiction theorem to the same carrier and
nonempty witness. It gives the implication
$\hsame(\Eone(a),r)\to\bot$, and the three facts assemble the displayed
conjunction.
\end{proof}
\leanchecked{BEDC.Derived.FunctorUp.FunctorPrefixHomCarrier\_e1\_visible\_morphism\_unique\_strict\_boundary}

\begin{theorem}[Prefix functor e-one visible morphism payload determinacy]
\label{thm:functor-prefix-e1-visible-morphism-payload-deterministic}
For histories $p,p',a,a',r,r',k,k'$, if the three displayed endpoints agree,
$$
\hsame(p,p'),\qquad \hsame(a,a'),\qquad \hsame(r,r'),
$$
then two prefixed one-headed visible morphisms with those transported endpoints
have the same visible payload:
$$
\mathsf{CategoryHomCarrier}(\mathsf{append}(p,\Eone(a)),
\mathsf{append}(p,\Eone(r)),\Eone(k))
\Longrightarrow
\mathsf{CategoryHomCarrier}(\mathsf{append}(p',\Eone(a')),
\mathsf{append}(p',\Eone(r')),\Eone(k'))
\Longrightarrow
\hsame(k,k').
$$
\end{theorem}
\begin{proof}
Use the three $\hsame$ rows to identify the prefix, source tail, and target
tail displays. Category morphism determinacy then gives
$\hsame(\Eone(k),\Eone(k'))$, and the e-one constructor inversion descends this
to $\hsame(k,k')$.
\end{proof}
\leanchecked{BEDC.Derived.FunctorUp.FunctorPrefixHomCarrier\_e1\_visible\_morphism\_payload\_deterministic}
