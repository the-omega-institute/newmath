\begin{theorem}[Metric distance witness right e1 result shape]
\label{thm:metric-distance-witness-right-e1-result-shape}
For histories $x,y,d$, every concrete metric distance witness whose right
endpoint is visibly one-headed has a visibly one-headed distance ledger, and
the exposed distance tail is again the metric witness for the exposed right
endpoint:
$$
\mathsf{MetricDistanceWitness}(x,\Eone(y),d)
\Longrightarrow
\exists k:\mathsf{BHist}.\;
\hsame(d,\Eone(k))\land
\mathsf{MetricDistanceWitness}(x,y,k).
$$
\end{theorem}
\begin{proof}
Unpack the distance witness into unary carrier fields and the continuation
ledger. The continuation result has the form
$d=x\cdot\Eone(y)$, which is definitionally
$\Eone(x\cdot y)$. Choose $k=x\cdot y$. Unary inversion on the visible right
endpoint gives the carrier for $y$, unary append closure gives the carrier for
$k$, and the canonical append equality gives the descended continuation
ledger.
\end{proof}
\leanchecked{BEDC.Derived.MetricUp.MetricDistanceWitness\_right\_e1\_result\_shape}

\begin{theorem}[Metric distance witness right e1 result tail exactness]
\label{thm:metric-distance-witness-right-e1-result-tail-hsame-exact}
For histories $x,y,d,k$, if a distance witness has a visibly one-headed right
endpoint and its displayed result is history-same as $\Eone(k)$, then $k$ is
the exact descended distance tail:
$$
\mathsf{MetricDistanceWitness}(x,\Eone(y),d)\Longrightarrow
\hsame(d,\Eone(k))\Longrightarrow
\mathsf{MetricDistanceWitness}(x,y,k).
$$
\end{theorem}
\begin{proof}
Apply right e-one result shape to expose an intermediate tail $k_0$ together
with $\hsame(d,\Eone(k_0))$ and the descended metric witness. Compose this
classifier with the displayed result classifier, invert the resulting
one-headed $\hsame$ comparison, and transport the tail witness along the
exposed tail equality.
\end{proof}

\begin{theorem}[Metric distance witness right e1 result tail determinism]
\label{thm:metric-distance-witness-right-e1-result-tail-deterministic}
For histories $x,y,d,d'$, two metric distance witnesses with the same source,
the same visible right endpoint, and visible results determine the same result
tail:
$$
\mathsf{MetricDistanceWitness}(x,\Eone(y),\Eone(d))
\Longrightarrow
\mathsf{MetricDistanceWitness}(x,\Eone(y),\Eone(d'))
\Longrightarrow
\hsame(d,d').
$$
\end{theorem}
\begin{proof}
Project the continuation ledgers from the two metric witnesses. Continuation
determinacy identifies the visible results
$\hsame(\Eone(d),\Eone(d'))$, and one-constructor inversion descends this
classifier to the tail relation $\hsame(d,d')$.
\end{proof}

\begin{theorem}[Metric distance witness right e1 source shape]
\label{thm:metric-distance-witness-right-e1-source-shape}
For histories $x,y,d$, every concrete metric distance witness whose right
endpoint is visibly one-headed exposes a visible distance tail and the matching
empty-or-visible source alternative:
$$
\mathsf{MetricDistanceWitness}(x,\Eone(y),d)
\Longrightarrow
\exists k:\mathsf{BHist}.\;
\hsame(d,\Eone(k))\land
\Bigl(
(x=\emp\land\mathsf{UnaryHistory}(y)\land\hsame(k,y))
\lor
\exists x_1:\mathsf{BHist}.\;
x=\Eone(x_1)\land
\mathsf{MetricDistanceWitness}(\Eone(x_1),y,k)
\Bigr).
$$
\end{theorem}
\begin{proof}
Apply \autoref{thm:metric-distance-witness-right-e1-result-shape} to the
given witness. This produces a history $k$, a displayed result comparison
$\hsame(d,\Eone(k))$, and the descended witness
$\mathsf{MetricDistanceWitness}(x,y,k)$.

We next rebuild the displayed-result witness with result $\Eone(k)$. Keep the
original source and right endpoint carriers, close the descended tail carrier
under $\Eone$, and transport the original continuation field along
$\hsame(d,\Eone(k))$.

Apply
\autoref{thm:metric-distance-witness-right-one-headed-source-cases} to this
displayed-result witness. In the empty branch, keep the same $k$ and record
$x=\emp$, the unary carrier for $y$, and $\hsame(k,y)$. In the visible branch,
keep the same $k$ and the exposed $x_1$, together with
$x=\Eone(x_1)$ and the descended metric witness.
\end{proof}
\leanchecked{BEDC.Derived.MetricUp.MetricDistanceWitness\_right\_e1\_source\_shape}

\begin{theorem}[Metric distance witness right e1 result iff]
\label{thm:metric-distance-witness-right-e1-result-iff}
For histories $x,y,d$, a distance witness whose right endpoint and result both
carry one visible constructor is equivalent to the descended tail witness:
$$
\mathsf{MetricDistanceWitness}(x,\Eone(y),\Eone(d))
\Longleftrightarrow
\mathsf{MetricDistanceWitness}(x,y,d).
$$
\end{theorem}
\begin{proof}
The forward direction is right one-headed result exactness. For the reverse
direction, keep the source unary carrier, close the right endpoint and result
carriers under $\Eone$, and lift the continuation ledger by the one-step
continuation rule.
\end{proof}

$$
\mathsf{MetricDistanceWitness}(\Eone(x),\Eone(y),\Eone(d))

\label{thm:metric-distance-witness-e1-pair-result-iff}

$$
\mathsf{MetricDistanceWitness}(\Eone(x),\Eone(y),\Eone(d))
\Longleftrightarrow
\mathsf{UnaryHistory}(x)\land\mathsf{UnaryHistory}(y)\land
\hsame(d,\Eone(x)\cdot y).
$$
\end{theorem}
\begin{proof}

Theorem~\ref{thm:metric-distance-witness-e1-pair-result-tail-readback}. It
descends the visible target and result constructors and returns exactly the
two unary tail carriers and the displayed continuation ledger.

For the reverse direction, start from the unary tail carriers and the
$\hsame$ ledger. Unary closure builds the source carrier
$\mathsf{UnaryHistory}(\Eone(x))$, and the continuation ledger transports the
append result so that $d$ is the concrete distance from $\Eone(x)$ to $y$.
This gives the central witness
$\mathsf{MetricDistanceWitness}(\Eone(x),y,d)$.

Apply
Theorem~\ref{thm:metric-distance-witness-left-e1-visible-target-step} to this
central witness. The visible target constructor and the matching visible
result constructor are restored, producing
$\mathsf{MetricDistanceWitness}(\Eone(x),\Eone(y),\Eone(d))$.
\end{proof}
