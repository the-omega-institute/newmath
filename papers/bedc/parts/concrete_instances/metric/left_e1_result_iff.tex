\begin{theorem}[Metric distance witness left e1 result iff]
\label{thm:metric-distance-witness-left-e1-result-iff}
For histories $x,y,d$, a distance witness from a visible left endpoint to a
visible result is equivalent to the empty middle case or to a visible middle
tail carrying the remaining witness:
$$
\mathsf{MetricDistanceWitness}(\Eone(x),y,\Eone(d))
\Longleftrightarrow
\bigl(y=\emp\land\mathsf{UnaryHistory}(x)\land\hsame(x,d)\bigr)
\lor
\bigl(\exists y_1:\mathsf{BHist}.\;
y=\Eone(y_1)\land
\mathsf{MetricDistanceWitness}(\Eone(x),y_1,d)\bigr).
$$
\end{theorem}
\begin{proof}
The forward direction is the visible-result case split. For the reverse
direction, the empty middle branch rebuilds the witness from unary closure
under $\Eone$ and the right-unit continuation law. The visible middle branch
adds one visible constructor back to the middle endpoint and result ledger by
the one-step continuation rule.
\end{proof}
\leanchecked{BEDC.Derived.MetricUp.MetricDistanceWitness\_left\_e1\_result\_iff}

\begin{theorem}[Metric distance witness left e1 result depth addition]
\label{thm:metric-distance-witness-left-e1-result-depth-add}
For histories $x,y,d$, every visible-result distance witness
$$
\mathsf{MetricDistanceWitness}(\Eone(x),y,\Eone(d))
$$
descends the metric depth identity to the visible tails:
$$
\mathsf{MetricDistanceDepth}(d)
=
\mathsf{MetricDistanceDepth}(x)+\mathsf{MetricDistanceDepth}(y).
$$
\end{theorem}
\begin{proof}
Apply metric-distance depth addition to the displayed witness. The resulting
identity has a visible constructor on the source side and on the result side.
Normalize the natural-number expression with successor addition, then cancel
the shared successor.
\end{proof}
\leanchecked{BEDC.Derived.MetricUp.MetricDistanceWitness\_left\_e1\_result\_depth\_add}

\begin{theorem}[Metric distance witness left e1 result empty target iff]
\label{thm:metric-distance-witness-left-e1-result-empty-target-iff}
For histories $x,y,d$, a metric distance witness from $\Eone(x)$ to target
$y$ with visible result $\Eone(d)$ identifies $y$ with the empty history
exactly when the exposed source tail is unary and agrees with the result tail:
$$
\mathsf{MetricDistanceWitness}(\Eone(x),y,\Eone(d))
\Longrightarrow
\bigl(\hsame(y,\emp)\Longleftrightarrow
\mathsf{UnaryHistory}(x)\land\hsame(x,d)\bigr).
$$
\end{theorem}
\begin{proof}
Split the target shape. The empty target branch is the right-unit continuation
case and reads back the unary source tail and the tail equality. The \(e0\)
target contradicts the unary target carrier. The visible target branch cannot
also have \(x\hsame d\): transporting the result to \(\Eone(x)\) would make
\(\Cont(\Eone(x),\Eone(y_1),\Eone(x))\), so right-unit uniqueness would
identify \(\Eone(y_1)\) with \(\emp\), contradicting constructor
disjointness.
\end{proof}
\leanchecked{BEDC.Derived.MetricUp.MetricDistanceWitness\_left\_e1\_result\_empty\_target\_iff}

\begin{theorem}[Metric distance witness left e1 result hsame source]
\label{thm:metric-distance-witness-left-e1-result-hsame-source}
For histories $x,x',y,d$, a distance witness from $\Eone(x)$ to a visible
result keeps the same empty-or-visible alternative after transporting the
displayed source payload along $\hsame(x,x')$:
$$
\mathsf{MetricDistanceWitness}(\Eone(x),y,\Eone(d))
\Longrightarrow
\hsame(x,x')
\Longrightarrow
\bigl(y=\emp\land\mathsf{UnaryHistory}(x')\land\hsame(x',d)\bigr)
\lor
\bigl(\exists y_1:\mathsf{BHist}.\;
y=\Eone(y_1)\land
\mathsf{MetricDistanceWitness}(\Eone(x'),y_1,d)\bigr).
$$
\end{theorem}
\begin{proof}
Case-reduce the source $\hsame$ witness. The transported payload is then
definitionally the original payload, and the established visible-result split
gives the displayed empty target or visible target branch.
\end{proof}
\leanchecked{BEDC.Derived.MetricUp.MetricDistanceWitness\_left\_e1\_result\_hsame\_source}

\begin{theorem}[Metric distance witness left e1 result hsame target]
\label{thm:metric-distance-witness-left-e1-result-hsame-target}
For histories $x,y,y',d$, a distance witness from a visible left endpoint to a
visible result keeps the same empty-or-visible target alternative after
transporting the target presentation along $\hsame(y,y')$:
$$
\mathsf{MetricDistanceWitness}(\Eone(x),y,\Eone(d))
\Longrightarrow
\hsame(y,y')
\Longrightarrow
\bigl(y'=\emp\land\mathsf{UnaryHistory}(x)\land\hsame(x,d)\bigr)
\lor
\bigl(\exists y_1:\mathsf{BHist}.\;
y'=\Eone(y_1)\land
\mathsf{MetricDistanceWitness}(\Eone(x),y_1,d)\bigr).
$$
\end{theorem}
\begin{proof}
Apply the visible-result case split. In the empty branch, empty-history
inversion transports the displayed target to $\emp$. In the visible branch,
one-constructor inversion of $\hsame(\Eone(y_0),y')$ exposes
$y'=\Eone(y_1)$ and $y_0\hsame y_1$; unary and continuation transport
carry the tail metric witness to the displayed target tail.
\end{proof}
\leanchecked{BEDC.Derived.MetricUp.MetricDistanceWitness\_left\_e1\_result\_hsame\_target}

\begin{theorem}[Metric distance witness left e1 result hsame result]
\label{thm:metric-distance-witness-left-e1-result-hsame-result}
For histories $x,y,d,d'$, a distance witness from a visible left endpoint to a
visible result keeps the same empty-or-visible target alternative after
transporting the displayed result tail along $\hsame(d,d')$:
$$
\mathsf{MetricDistanceWitness}(\Eone(x),y,\Eone(d))
\Longrightarrow
\hsame(d,d')
\Longrightarrow
\bigl(y=\emp\land\mathsf{UnaryHistory}(x)\land\hsame(x,d')\bigr)
\lor
\bigl(\exists y_1:\mathsf{BHist}.\;
y=\Eone(y_1)\land
\mathsf{MetricDistanceWitness}(\Eone(x),y_1,d')\bigr).
$$
\end{theorem}
\begin{proof}
First lift the result-tail agreement through the visible constructor, obtaining
$\hsame(\Eone(d),\Eone(d'))$. The metric hsame transport field is then applied
with reflexive source and target transports, so the original witness repacks as
$\mathsf{MetricDistanceWitness}(\Eone(x),y,\Eone(d'))$.

Apply Theorem~\ref{thm:metric-distance-witness-left-e1-result-iff} to this
transported witness. The empty target branch gives
$\mathsf{UnaryHistory}(x)$ and $\hsame(x,d')$; the visible target branch gives a
tail $y_1$ with $y=\Eone(y_1)$ and
$\mathsf{MetricDistanceWitness}(\Eone(x),y_1,d')$.
\end{proof}
\leanchecked{BEDC.Derived.MetricUp.MetricDistanceWitness\_left\_e1\_result\_hsame\_result}

\begin{theorem}[Metric distance witness left e1 result tail determinism]
\label{thm:metric-distance-witness-left-e1-result-tail-deterministic}
For histories $x,y,d,d'$, two visible-result distance witnesses with the same
visible left endpoint and target determine the same result tail:
$$
\mathsf{MetricDistanceWitness}(\Eone(x),y,\Eone(d))
\Longrightarrow
\mathsf{MetricDistanceWitness}(\Eone(x),y,\Eone(d'))
\Longrightarrow
\hsame(d,d').
$$
\end{theorem}
\begin{proof}
Unpack the two distance witnesses only to their continuation ledgers. BEDC
continuation determinacy gives
$\hsame(\Eone(d),\Eone(d'))$ for the two visible results. One-constructor
inversion for $\Eone$ then descends this classifier to $\hsame(d,d')$.
\end{proof}
\leanchecked{BEDC.Derived.MetricUp.MetricDistanceWitness\_left\_e1\_result\_tail\_deterministic}

\begin{theorem}[Metric distance witness left e1 result shape]
\label{thm:metric-distance-witness-left-e1-result-shape}
For histories $x,y,d$, every distance witness from $\Eone(x)$ has a visible
result shape:
$$
\mathsf{MetricDistanceWitness}(\Eone(x),y,d)
\Longrightarrow
\exists k:\mathsf{BHist}.\;
\hsame(d,\Eone(k))\land
\Bigl(
  (y=\emp\land\mathsf{UnaryHistory}(x)\land\hsame(x,k))
  \lor
  \exists y_1:\mathsf{BHist}.\;
    y=\Eone(y_1)\land
    \mathsf{MetricDistanceWitness}(\Eone(x),y_1,k)
\Bigr).
$$
\end{theorem}
\begin{proof}
Decompose the result history. An empty result is incompatible with a visible
left endpoint and the continuation equation; an $\Ezero$ result is incompatible
with the unary distance carrier. In the $\Eone(k)$ case, the established
visible-result split gives either the empty target boundary or the visible
target tail witness.
\end{proof}
\leanchecked{BEDC.Derived.MetricUp.MetricDistanceWitness\_left\_e1\_result\_shape}
