\begin{theorem}[Metric distance witness left e1 visible target step]
\label{thm:metric-distance-witness-left-e1-visible-target-step}
For histories $x,y,d$, a distance witness from $\Eone(x)$ to $y$ with result
$d$ remains a distance witness after placing the target and result under one
visible $\Eone$ constructor:
$$
\mathsf{MetricDistanceWitness}(\Eone(x),y,d)
\Longrightarrow
\mathsf{MetricDistanceWitness}(\Eone(x),\Eone(y),\Eone(d)).
$$
\end{theorem}
\begin{proof}
Unpack the witness into unary carriers and the continuation ledger. Unary
closure supplies carriers for $\Eone(y)$ and $\Eone(d)$, and the continuation
step rule lifts the same ledger from target $y$ and result $d$ to target
$\Eone(y)$ and result $\Eone(d)$.
\end{proof}

\begin{theorem}[Metric distance witness left e1 visible target result shape]
\label{thm:metric-distance-witness-left-e1-visible-target-result-shape}
For histories $x,y,d$, any distance witness from $\Eone(x)$ to $\Eone(y)$ has
a visible result tail, and that tail is again the distance witness from
$\Eone(x)$ to $y$:
$$
\mathsf{MetricDistanceWitness}(\Eone(x),\Eone(y),d)
\Longrightarrow
\exists k.\;
\hsame(d,\Eone(k))\land
\mathsf{MetricDistanceWitness}(\Eone(x),y,k).
$$
\end{theorem}
\begin{proof}
Apply the left visible-result shape split to the displayed witness. The
empty-target branch would identify $\Eone(y)$ with $\emp$, contradicting
constructor no-confusion. The remaining visible branch supplies the result
tail $k$ and the descended witness with target $y$.
\end{proof}

\begin{theorem}[Metric distance witness left e1 visible target result tail exactness]
\label{thm:metric-distance-witness-left-e1-visible-target-result-tail-hsame-exact}
For histories $x,y,d,k$, if a distance witness from $\Eone(x)$ to
$\Eone(y)$ has displayed result history-same as $\Eone(k)$, then $k$ is the
descended distance result from $\Eone(x)$ to $y$:
$$
\mathsf{MetricDistanceWitness}(\Eone(x),\Eone(y),d)
\Longrightarrow
\hsame(d,\Eone(k))
\Longrightarrow
\mathsf{MetricDistanceWitness}(\Eone(x),y,k).
$$
\end{theorem}
\begin{proof}
Apply the visible-target result shape theorem to expose a result tail $k_0$,
a displayed comparison $\hsame(d,\Eone(k_0))$, and the descended witness.
Composing this comparison with the assumed comparison to $\Eone(k)$ gives a
one-headed history-sameness relation between $\Eone(k_0)$ and $\Eone(k)$.
Constructor inversion descends it to $\hsame(k_0,k)$, and tail transport
returns the displayed witness at $k$.
\end{proof}

\begin{theorem}[Metric distance witness left e1 source step]
\label{thm:metric-distance-witness-left-e1-source-step}
For histories $x,y,d$, a distance witness from $x$ to $y$ with result $d$
remains a distance witness after placing the source and result under one
visible $\Eone$ constructor:
$$
\mathsf{MetricDistanceWitness}(x,y,d)
\Longrightarrow
\mathsf{MetricDistanceWitness}(\Eone(x),y,\Eone(d)).
$$
\end{theorem}
\begin{proof}
Unpack the witness into unary carriers and the continuation ledger
$d=x\cdot y$. Unary closure supplies the new source and result carriers. Since
$y$ is unary, the append law rewrites $\Eone(x)\cdot y$ as
$\Eone(x\cdot y)$, so the lifted result $\Eone(d)$ satisfies the required
continuation equation.
\end{proof}

\begin{theorem}[Metric distance witness left e1 source iff]
\label{thm:metric-distance-witness-left-e1-source-iff}
For histories $x,y,d$, placing the source and result under one visible
$\Eone$ constructor is reversible:
$$
\mathsf{MetricDistanceWitness}(\Eone(x),y,\Eone(d))
\Longleftrightarrow
\mathsf{MetricDistanceWitness}(x,y,d).
$$
\end{theorem}
\begin{proof}
The reverse direction is
Theorem~\ref{thm:metric-distance-witness-left-e1-source-step}. For the forward
direction, split on the target history. The empty target case is the right
empty-boundary exactness of the metric witness. A zero-headed target contradicts
the unary target carrier recorded in the witness. In the one-headed target case,
the result equation descends through the visible constructor, and unary append
commutes the visible source constructor across the unary target tail, giving the
original continuation ledger.
\end{proof}
\leanchecked{BEDC.Derived.MetricUp.MetricDistanceWitness\_left\_e1\_source\_iff}

\begin{theorem}[Metric distance witness left e1 source result iff]
\label{thm:metric-distance-witness-left-e1-source-result-iff}

$$
\mathsf{MetricDistanceWitness}(\Eone(x),y,\Eone(d))
\Longleftrightarrow
\mathsf{MetricDistanceWitness}(x,y,d).
$$
\end{theorem}
\begin{proof}
For the forward direction, unpack the displayed witness. Unary inversion on
the source and result gives unary carriers for $x$ and $d$. The continuation
ledger identifies $\Eone(d)$ with $\Eone(x)\cdot y$, and the unary append law
for the target rewrites the right side to $\Eone(x\cdot y)$; constructor
inversion exposes the descended continuation ledger $d=x\cdot y$.

For the reverse direction, apply
\autoref{thm:metric-distance-witness-left-e1-source-step}.
\end{proof}
