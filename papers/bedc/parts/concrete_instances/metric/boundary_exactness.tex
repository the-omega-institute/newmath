\begin{theorem}[Metric distance witness empty-source visible-distance target exactness]
\label{thm:metric-distance-witness-empty-source-e1-distance-target-exactness}
For histories $target,d$, a concrete distance witness from the empty source
whose distance ledger is visibly one-headed identifies the target with that
same one-headed ledger and exposes unary evidence for the ledger tail:
$$
\mathsf{MetricDistanceWitness}(\emp,target,\Eone(d))
\Longrightarrow
target=\Eone(d)\land\mathsf{UnaryHistory}(d).
$$
\end{theorem}
\begin{proof}
Apply empty-left exactness to obtain unary evidence for $target$ and the
classifier $\hsame(\Eone(d),target)$. One-constructor inversion on this
classifier presents $target$ as $\Eone(d)$ up to the exposed tail equality.
Substitution into the unary target witness and unary one-step inversion give
$\mathsf{UnaryHistory}(d)$.
\end{proof}
\leanchecked{BEDC.Derived.MetricUp.MetricDistanceWitness\_empty\_source\_e1\_distance\_target\_exactness}

\begin{theorem}[Metric nonempty distance has a nonempty endpoint]
\label{thm:metric-distance-witness-nonempty-distance-endpoint-nonempty}
For histories $x,y,d:\mathsf{BHist}$, if
$\mathsf{MetricDistanceWitness}(x,y,d)$ holds and $d$ is not history-same as
$\emp$, then at least one endpoint is not history-same as $\emp$:
$$
\mathsf{MetricDistanceWitness}(x,y,d)\to
(d\hsame\emp\to\mathsf{False})\to
\bigl((x\hsame\emp\to\mathsf{False})
\vee (y\hsame\emp\to\mathsf{False})\bigr).
$$
\end{theorem}
\leanchecked{BEDC.Derived.MetricUp.MetricDistanceWitness\_nonempty\_distance\_endpoint\_nonempty}
\begin{proof}
Case-split the two endpoint histories. A visible endpoint constructor is
immediately disjoint from $\emp$, giving the required side of the disjunction.
In the remaining empty/empty case, the continuation field of the metric witness
identifies the distance result with $\emp$, contradicting the assumed
nonempty distance.
\end{proof}

\begin{theorem}[Metric distance witness zero-headed component impossibility]
\label{thm:metric-distance-witness-e0-component-absurd}
For histories $x,y,d:\mathsf{BHist}$, no concrete metric distance witness has
a zero-headed source, target, or distance component:
$$
\begin{array}{l}
\bigl(\mathsf{MetricDistanceWitness}(\Ezero(x),y,d)\to\mathsf{False}\bigr)
\land
\bigl(\mathsf{MetricDistanceWitness}(x,\Ezero(y),d)\to\mathsf{False}\bigr)
\land
\bigl(\mathsf{MetricDistanceWitness}(x,y,\Ezero(d))\to\mathsf{False}\bigr).
\end{array}
$$
\end{theorem}
\leanchecked{BEDC.Derived.MetricUp.MetricDistanceWitness\_e0\_component\_absurd}
\begin{proof}
Unpack the concrete witness. Its three component fields are unary-history
evidence for the source, target, and distance histories. BEDC unary histories
are generated only from $\emp$ by $\Eone$, so each displayed $\Ezero$ component
contradicts unary no-zero-extension.
\end{proof}
