\begin{theorem}[Metric distance witness empty-source visible-distance target exactness]
\label{thm:metric-distance-witness-empty-source-e1-distance-target-exactness}
For histories $target,d$, a concrete distance witness from the empty source
whose distance ledger is visibly one-headed identifies the target with that
same one-headed ledger and exposes unary evidence for the ledger tail:
$$
\mathsf{MetricDistanceWitness}(\emp,target,\Eone(d))
\Longrightarrow
target=\Eone(d)\land\mathsf{UnaryHistory}(d).
$$
\end{theorem}
\begin{proof}
Apply empty-left exactness to obtain unary evidence for $target$ and the
classifier $\hsame(\Eone(d),target)$. One-constructor inversion on this
classifier presents $target$ as $\Eone(d)$ up to the exposed tail equality.
Substitution into the unary target witness and unary one-step inversion give
$\mathsf{UnaryHistory}(d)$.
\end{proof}
\leanchecked{BEDC.Derived.MetricUp.MetricDistanceWitness\_empty\_source\_e1\_distance\_target\_exactness}

\begin{theorem}[Metric nonempty distance has a nonempty endpoint]
\label{thm:metric-distance-witness-nonempty-distance-endpoint-nonempty}
For histories $x,y,d:\mathsf{BHist}$, if
$\mathsf{MetricDistanceWitness}(x,y,d)$ holds and $d$ is not history-same as
$\emp$, then at least one endpoint is not history-same as $\emp$:
$$
\mathsf{MetricDistanceWitness}(x,y,d)\to
(d\hsame\emp\to\mathsf{False})\to
\bigl((x\hsame\emp\to\mathsf{False})
\vee (y\hsame\emp\to\mathsf{False})\bigr).
$$
\end{theorem}
\leanchecked{BEDC.Derived.MetricUp.MetricDistanceWitness\_nonempty\_distance\_endpoint\_nonempty}
\begin{proof}
Case-split the two endpoint histories. A visible endpoint constructor is
immediately disjoint from $\emp$, giving the required side of the disjunction.
In the remaining empty/empty case, the continuation field of the metric witness
identifies the distance result with $\emp$, contradicting the assumed
nonempty distance.
\end{proof}

\begin{theorem}[Metric nonempty distance endpoint positive depth]
\label{thm:metric-distance-witness-nonempty-distance-endpoint-positive-depth}
For histories $x,y,d:\mathsf{BHist}$, if
$\mathsf{MetricDistanceWitness}(x,y,d)$ holds and $d$ is not history-same as
$\emp$, then at least one endpoint has positive structural metric depth:
$$
\mathsf{MetricDistanceWitness}(x,y,d)\to
(d\hsame\emp\to\mathsf{False})\to
\bigl(0<\mathsf{MetricDistanceDepth}(x)
\vee 0<\mathsf{MetricDistanceDepth}(y)\bigr).
$$
\end{theorem}
\begin{proof}
Apply the nonempty-distance endpoint theorem to obtain a nonempty endpoint.
The positive-depth classifier for metric histories converts that endpoint
nonemptiness into positive structural depth on the corresponding side of the
disjunction.
\end{proof}
\leanchecked{BEDC.Derived.MetricUp.MetricDistanceWitness\_nonempty\_distance\_endpoint\_positive\_depth}

\begin{theorem}[Metric nonempty distance endpoint e-one cases]
\label{thm:metric-distance-witness-nonempty-distance-endpoint-e1-cases}
For histories $x,y,d:\mathsf{BHist}$, if
$\mathsf{MetricDistanceWitness}(x,y,d)$ holds and the distance endpoint is not
history-same as $\emp$, then at least one endpoint is visibly one-headed with a
unary tail:
$$
\begin{array}{l}
\mathsf{MetricDistanceWitness}(x,y,d)\to
(d\hsame\emp\to\mathsf{False})\to\\
\qquad
\bigl(\exists x_0,\ x=\Eone(x_0)\land\mathsf{UnaryHistory}(x_0)\bigr)
\vee
\bigl(\exists y_0,\ y=\Eone(y_0)\land\mathsf{UnaryHistory}(y_0)\bigr).
\end{array}
$$
\end{theorem}
\begin{proof}
First apply the endpoint non-emptiness theorem to the metric witness and the
nonempty distance assumption. On whichever side is nonempty, use the endpoint's
unary-history carrier from the metric witness and apply unary nonempty
one-tail readback.
\end{proof}
\leanchecked{BEDC.Derived.MetricUp.MetricDistanceWitness\_nonempty\_distance\_endpoint\_e1\_cases}

\begin{theorem}[Metric empty-source nonempty distance target e-one cases]
\label{thm:metric-distance-witness-empty-source-nonempty-distance-target-e1-cases}
For histories $target,dist:\mathsf{BHist}$, if a concrete metric distance
witness starts at the empty source and its distance endpoint is not
history-same as $\emp$, then the target endpoint is visibly one-headed with a
unary tail:
$$
\mathsf{MetricDistanceWitness}(\emp,target,dist)\to
(dist\hsame\emp\to\mathsf{False})\to
\exists targetTail,\ target=\Eone(targetTail)\land
\mathsf{UnaryHistory}(targetTail).
$$
\end{theorem}
\begin{proof}
Apply the nonempty-distance endpoint e-one cases theorem. The source-side
visible branch would identify $\emp$ with a one-headed history, which is
impossible by constructor disjointness. The remaining branch is exactly the
target e-one readback.
\end{proof}

\begin{theorem}[Metric distance witness zero-headed component impossibility]
\label{thm:metric-distance-witness-e0-component-absurd}
For histories $x,y,d:\mathsf{BHist}$, no concrete metric distance witness has
a zero-headed source, target, or distance component:
$$
\begin{array}{l}
\bigl(\mathsf{MetricDistanceWitness}(\Ezero(x),y,d)\to\mathsf{False}\bigr)
\land
\bigl(\mathsf{MetricDistanceWitness}(x,\Ezero(y),d)\to\mathsf{False}\bigr)
\land
\bigl(\mathsf{MetricDistanceWitness}(x,y,\Ezero(d))\to\mathsf{False}\bigr).
\end{array}
$$
\end{theorem}
\leanchecked{BEDC.Derived.MetricUp.MetricDistanceWitness\_e0\_component\_absurd}
\begin{proof}
Unpack the concrete witness. Its three component fields are unary-history
evidence for the source, target, and distance histories. BEDC unary histories
are generated only from $\emp$ by $\Eone$, so each displayed $\Ezero$ component
contradicts unary no-zero-extension.
\end{proof}

\begin{theorem}[Metric empty distance visible endpoint impossibility]
\label{thm:metric-distance-witness-empty-distance-visible-endpoint-absurd}
For histories $x,y,d,tx,ty$, if a metric distance witness has a result
history-same as $\emp$, then neither endpoint can be history-same as a visible
one-headed history:
$$
\begin{array}{l}
\mathsf{MetricDistanceWitness}(x,y,d)\to
\hsame(d,\emp)\to\\
\qquad
\bigl(\hsame(x,\Eone(tx))\to\mathsf{False}\bigr)\land
\bigl(\hsame(y,\Eone(ty))\to\mathsf{False}\bigr).
\end{array}
$$
\end{theorem}
\begin{proof}
Transport the continuation result of the metric witness along the displayed
distance classifier, obtaining an empty-distance witness over the same
endpoints. Empty-distance exactness identifies both endpoints with $\emp$.
Composing either endpoint comparison with a visible one-headed classifier
contradicts the empty/e-one no-confusion theorem.
\end{proof}
\leanchecked{BEDC.Derived.MetricUp.MetricDistanceWitness\_empty\_distance\_visible\_endpoint\_absurd}

\begin{theorem}[Metric empty-distance composite endpoint collapse]
\label{thm:metric-distance-witness-empty-distance-composite-endpoint-collapse}
For histories $x,y,z,d_{xy},d_{yz}$, if
$\mathsf{MetricDistanceWitness}(x,y,d_{xy})$ and
$\mathsf{MetricDistanceWitness}(y,z,d_{yz})$ both have distance histories
classified as $\emp$, then all three endpoints classify together:
$$
d_{xy}\hsame\emp\to d_{yz}\hsame\emp\to
\HSame(x,y)\wedge\HSame(y,z)\wedge\HSame(x,z).
$$
\end{theorem}
\begin{proof}
Transport each distance witness to the empty-distance case. Empty-distance
exactness identifies the corresponding endpoint pair with $\emp$, and
transitivity through $\emp$ gives the three endpoint comparisons.
\end{proof}
\leanchecked{BEDC.Derived.MetricUp.MetricDistanceWitness\_empty\_distance\_composite\_endpoint\_collapse}

\begin{theorem}[Metric right-boundary visible context distance source]
\label{thm:metric-distance-witness-right-boundary-visible-context-distance-source}
For histories $p,q,x,d:\mathsf{BHist}$, a visible-context distance witness with
empty central right endpoint identifies the central distance with the visible
source and gives the same structural depth:
$$
\mathsf{MetricDistanceWitness}(p\cdot x,\emp\cdot q,(p\cdot d)\cdot q)
\Longrightarrow
\hsame(d,x)\land
\mathsf{MetricDistanceDepth}(d)=\mathsf{MetricDistanceDepth}(x).
$$
\end{theorem}
\begin{proof}
Right-boundary visible-context exactness extracts the central classifier
$\hsame(d,x)$. Case analysis on that classifier identifies the two histories,
and the metric-depth equality follows by reflexivity.
\end{proof}
\leanchecked{BEDC.Derived.MetricUp.MetricDistanceWitness\_right\_boundary\_visible\_context\_distance\_source}
