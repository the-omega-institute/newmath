\begin{theorem}[Metric distance witness visible-context empty distance exactness]
\label{thm:metric-distance-witness-visible-context-empty-distance-iff}
For histories $p,q,x,y$, a visible-context distance witness whose central
distance ledger is empty is exactly unary evidence for the visible prefix and
suffix together with empty classification of both central endpoints:
$$
\mathsf{MetricDistanceWitness}(p\cdot x,y\cdot q,(p\cdot\emp)\cdot q)
\Longleftrightarrow
\mathsf{UnaryHistory}(p)\ \wedge\ \mathsf{UnaryHistory}(q)\ \wedge\
\hsame(x,\emp)\ \wedge\ \hsame(y,\emp).
$$
\end{theorem}
\begin{proof}
Apply visible-context exactness to peel off the prefix and suffix. The central
witness is then $\mathsf{MetricDistanceWitness}(x,y,\emp)$, which empty-distance
exactness identifies with $\hsame(x,\emp)$ and $\hsame(y,\emp)$. Conversely,
the two empty endpoint classifiers build the central empty-distance witness,
and visible-context exactness reattaches the same prefix and suffix.
\end{proof}
\leanchecked{BEDC.Derived.MetricUp.MetricDistanceWitness\_visible\_context\_empty\_distance\_iff}

\begin{theorem}[Metric visible-context empty-distance continuation witness depth package]
\label{thm:metric-distance-witness-visible-context-empty-distance-continuation-witness-depth}
For histories $p,q,x,y,l,r,m,o:\mathsf{BHist}$, suppose
$$
\mathsf{MetricDistanceWitness}(p\cdot x,y\cdot q,(p\cdot\emp)\cdot q)
$$
holds together with $\mathsf{UnaryHistory}(l)$, $\mathsf{UnaryHistory}(r)$,
$\ContR(l,x,m)$, and $\ContR(m,r,o)$. Then the direct continuation carries a
metric witness whose depth is induced by the two visible endpoints:
$$
\mathsf{MetricDistanceWitness}(l,r,o)\wedge
\mathsf{MetricDistanceDepth}(o)=
\mathsf{MetricDistanceDepth}(l)+\mathsf{MetricDistanceDepth}(r).
$$
\end{theorem}
\begin{proof}
Apply the visible-context empty-distance continuation splice to the distance
witness and the two continuation ledgers. This gives the direct continuation
$\ContR(l,r,o)$.

The unary carriers for $l$ and $r$ then close the continuation under the metric
distance witness constructor, so the direct continuation packages as
$\mathsf{MetricDistanceWitness}(l,r,o)$.

Finally, metric witness depth additivity applied to this witness gives
$\mathsf{MetricDistanceDepth}(o)=
\mathsf{MetricDistanceDepth}(l)+\mathsf{MetricDistanceDepth}(r)$. Pairing the
witness with that displayed equation is the required conjunction.
\end{proof}

\begin{theorem}[Metric visible-context empty-distance two-sided endpoint witness transport]
\label{thm:metric-visible-context-empty-distance-two-sided-endpoint-witness-transport}
For histories $p,q,x,y,l,r,m,n,o:\mathsf{BHist}$, suppose
$$
\mathsf{MetricDistanceWitness}(p\cdot x,y\cdot q,(p\cdot\emp)\cdot q)
$$
holds together with $\mathsf{UnaryHistory}(l)$, $\mathsf{UnaryHistory}(r)$,
$\ContR(l,x,m)$, $\ContR(y,r,n)$, and $\ContR(l,r,o)$. Then the two reduced
endpoint continuations carry the same metric distance witness:
$$
\mathsf{MetricDistanceWitness}(m,n,o).
$$
\end{theorem}
\begin{proof}
Apply \autoref{thm:metric-visible-context-empty-distance-endpoint-continuation-units}
to the visible-context empty-distance witness and the two endpoint continuation
ledgers. This gives $\hsame(m,l)$ and $\hsame(n,r)$.

The unary endpoint hypotheses for $l$ and $r$, together with the direct
continuation ledger $\ContR(l,r,o)$, build the concrete direct metric witness
$\mathsf{MetricDistanceWitness}(l,r,o)$.

Now use
\autoref{thm:metric-distance-witness-hsame-fields-transport}. Transport the
first endpoint along the symmetry of $\hsame(m,l)$, the second endpoint along
the symmetry of $\hsame(n,r)$, and the result endpoint along reflexivity of
$\hsame(o,o)$. The continuation ledger transports with the same three
relations, so the transported fields package as
$\mathsf{MetricDistanceWitness}(m,n,o)$.
\end{proof}
\leanchecked{BEDC.Derived.MetricUp.MetricDistanceWitness\_visible\_context\_empty\_distance\_two\_sided\_endpoint\_witness\_transport}
