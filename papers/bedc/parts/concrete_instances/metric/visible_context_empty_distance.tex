\begin{theorem}[Metric distance witness visible-context empty distance exactness]
\label{thm:metric-distance-witness-visible-context-empty-distance-iff}
For histories $p,q,x,y$, a visible-context distance witness whose central
distance ledger is empty is exactly unary evidence for the visible prefix and
suffix together with empty classification of both central endpoints:
$$
\mathsf{MetricDistanceWitness}(p\cdot x,y\cdot q,(p\cdot\emp)\cdot q)
\Longleftrightarrow
\mathsf{UnaryHistory}(p)\ \wedge\ \mathsf{UnaryHistory}(q)\ \wedge\
\hsame(x,\emp)\ \wedge\ \hsame(y,\emp).
$$
\end{theorem}
\begin{proof}
Apply visible-context exactness to peel off the prefix and suffix. The central
witness is then $\mathsf{MetricDistanceWitness}(x,y,\emp)$, which empty-distance
exactness identifies with $\hsame(x,\emp)$ and $\hsame(y,\emp)$. Conversely,
the two empty endpoint classifiers build the central empty-distance witness,
and visible-context exactness reattaches the same prefix and suffix.
\end{proof}
\leanchecked{BEDC.Derived.MetricUp.MetricDistanceWitness\_visible\_context\_empty\_distance\_iff}

\begin{theorem}[Metric visible-context empty-distance continuation splice]
\label{thm:metric-distance-witness-visible-context-empty-distance-continuation-splice}
For histories $p,q,x,y,l,r,m,o$, suppose
$$
\mathsf{MetricDistanceWitness}(p\cdot x,y\cdot q,(p\cdot\emp)\cdot q)
$$
and continuations $\ContR(l,x,m)$ and $\ContR(m,r,o)$ hold. Then the empty
central endpoint exposed by the witness splices the continuation directly:
$$
\ContR(l,r,o).
$$
\end{theorem}
\begin{proof}
Apply visible-context empty-distance exactness to the witness. This gives
$\hsame(x,\emp)$. Substitute the empty central endpoint into
$\ContR(l,x,m)$, so the first continuation identifies $m$ with $l$. The second
continuation $\ContR(m,r,o)$ is therefore the desired direct continuation
$\ContR(l,r,o)$.
\end{proof}
\leanchecked{BEDC.Derived.MetricUp.MetricDistanceWitness\_visible\_context\_empty\_distance\_continuation\_splice}

\begin{theorem}[Metric visible-context empty-distance witness splice]
\label{thm:metric-distance-witness-visible-context-empty-distance-witness-splice}
For histories $p,q,x,y,l,r,m,o$, suppose
$$
\mathsf{MetricDistanceWitness}(p\cdot x,y\cdot q,(p\cdot\emp)\cdot q)
$$
and unary endpoint evidence for $l$ and $r$ hold. If
$\ContR(l,x,m)$ and $\ContR(m,r,o)$ hold, then the spliced continuation is a
metric distance witness:
$$
\mathsf{MetricDistanceWitness}(l,r,o).
$$
\end{theorem}
\begin{proof}
The visible-context empty-distance continuation splice supplies
$\ContR(l,r,o)$. Together with the unary endpoint hypotheses, unary
continuation closure gives the unary carrier for $o$. These three fields are
exactly the concrete metric distance witness from $l$ to $r$ with output $o$.
\end{proof}
\leanchecked{BEDC.Derived.MetricUp.MetricDistanceWitness\_visible\_context\_empty\_distance\_witness\_splice}
