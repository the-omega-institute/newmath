\begin{definition}[Metric ball budget predicate]
\label{def:metric-ball-budget-predicate}
For a certified $\MetricSpaceUp$ setup on a carrier $X$ with distance
$d:X\times X\to\RealUp$, a ball-budget structure supplies an ordered-real
comparison $\le_{\mathsf R}$ on $\RealUp$, an addition operation $+$ on
$\RealUp$, a transitivity field for $\le_{\mathsf R}$, and a two-sided
addition-monotonicity field saying that $a\le_{\mathsf R} a'$ and
$b\le_{\mathsf R} b'$ entail $a+b\le_{\mathsf R} a'+b'$. Relative to this
chosen budget structure, the closed ball-budget predicate is
$$
\mathsf{MetricBallBudget}_d(x,y;r) :\Longleftrightarrow d(x,y)\le_{\mathsf R} r.
$$
\end{definition}
\leandef{BEDC.Derived.MetricUp.MetricBallBudget}

\begin{theorem}[Metric ball budget composition]
\label{thm:metric-ball-budget-composition}
Let $X$ be a carrier equipped with the $\MetricSpaceUp$ certificate obligations
of \autoref{def:metric-certificate-obligations}, and let
$d:X\times X\to\RealUp$ be its certified distance. Assume also the ball-budget
ordered-real fields of \autoref{def:metric-ball-budget-predicate}: the
comparison $\le_{\mathsf R}$, addition $+$, budget-transitivity, and two-sided
budget-addition-monotonicity. For $x,y,z:X$ and budgets $r,s:\RealUp$,
$$
\mathsf{MetricBallBudget}_d(x,y;r)\Longrightarrow
\mathsf{MetricBallBudget}_d(y,z;s)\Longrightarrow
\mathsf{MetricBallBudget}_d(x,z;r+s).
$$
Equivalently,
$$
d(x,y)\le_{\mathsf R} r\Longrightarrow d(y,z)\le_{\mathsf R} s\Longrightarrow d(x,z)\le_{\mathsf R} r+s.
$$
\end{theorem}
\begin{proof}
Unfold \autoref{def:metric-ball-budget-predicate}. The two premises are the
budget comparisons $d(x,y)\le_{\mathsf R} r$ and
$d(y,z)\le_{\mathsf R} s$. The triangle-inequality field of
\autoref{def:metric-stability-certificate} gives
$$
d(x,z)\le_{\mathsf R} d(x,y)+d(y,z).
$$

Apply the supplied two-sided budget-addition-monotonicity field to the two
budget premises. This gives
$$
d(x,y)+d(y,z)\le_{\mathsf R} r+s.
$$

Apply the supplied budget-transitivity field to the triangle inequality and
the displayed budget bound. This yields $d(x,z)\le_{\mathsf R} r+s$.
Refolding \autoref{def:metric-ball-budget-predicate} gives the closed
ball-budget conclusion.
\end{proof}

\begin{theorem}[Metric distance witness standard bridge]
\label{thm:metric-distance-witness-standard-bridge}
For the public $\MetricUp$ distance-witness surface over unary
$\mathsf{BHist}$ endpoints $x,y$, there is a conservative standard
interpretation whose BEDC carrier is
$$
\{\,d\mid \mathsf{MetricDistanceWitness}(x,y,d)\,\},
$$
whose classifier is $\hsame$, whose forward row reads such a witness as the
finite append-distance presentation between the two unary endpoints, and whose
reverse row reconstructs the BEDC witness from that finite append presentation.
The bridge is only for this generated distance-witness surface; it does not
assert arbitrary host real metrics.
\end{theorem}
\begin{proof}
The carrier and classifier come from
\autoref{thm:metric-distance-witness-semantic-namecert}: the public BEDC
objects are exactly the $d$ satisfying
$\mathsf{MetricDistanceWitness}(x,y,d)$, and the classifier is $\hsame$.

For the forward row, apply
\autoref{thm:metric-distance-witness-visible-context-iff} to read the visible
unary context as the finite append-distance presentation between the endpoints.
\autoref{thm:metric-distance-witness-visible-context-determinacy} ensures that
this presentation is determined by the visible context, while
\autoref{thm:metric-distance-witness-symmetric-classifier} transports the same
presentation across the symmetric endpoint order.

For the reverse row, read the finite append-distance presentation back through
\autoref{thm:metric-distance-witness-visible-context-iff}. The resulting BEDC
witness is classified by the same $\hsame$ row, and
\autoref{thm:metric-ball-budget-composition} supplies the budget-compatible
composition law needed by the public distance-witness surface.

Thus the two rows form a conservative paper-side standard bridge for the
generated $\mathsf{MetricDistanceWitness}$ surface only. No completeness,
compactness, topology, or arbitrary host-real metric structure is introduced.
\end{proof}

\begin{theorem}[MetricSpaceUp public distance surface exhaustion]
\label{thm:metricspace-public-distance-surface-exhaustion}
Every public distance read of $\MetricSpaceUp$ projects exactly to the
$\mathsf{BHist}$ endpoint rows, the $\mathsf{MetricDistanceWitness}$ row, the
$\mathsf{MetricBallBudget}$ row, the $\hsame$ classifier row, the finite
$\Cont$ provenance row, and the $\RealUp$ budget rows of
\autoref{def:metricspace-public-distance-surface}. Conversely, any such finite
family repacks as the same public distance surface.
\end{theorem}
\begin{proof}
Project a public read through the distance-witness semantic certificate of
\autoref{thm:metric-distance-witness-semantic-namecert}. This fixes the
$\mathsf{BHist}$ endpoints, the concrete
$\mathsf{MetricDistanceWitness}(x,y,d)$ row, and the $\hsame$ classifier row.

Next apply \autoref{thm:metric-ball-budget-composition}. The only budget rows
available to the public read are the displayed $\RealUp$ comparison, addition,
transitivity, and monotonicity fields used to compose ball-budget witnesses.

Finally use \autoref{thm:metric-distance-witness-standard-bridge} and
\autoref{thm:metric-distance-witness-visible-context-determinacy}. The bridge
reads the same finite witness surface in the standard direction, and visible
context determinacy prevents an additional host distance row from entering the
packet. Repacking the projected rows gives exactly
\autoref{def:metricspace-public-distance-surface}.
\end{proof}
