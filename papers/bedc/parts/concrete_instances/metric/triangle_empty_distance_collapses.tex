\begin{theorem}[Metric triangle right distance empty collapse]
\label{thm:metric-distance-witness-triangle-right-distance-empty-collapse}
For histories $x,y,z,d_{xy},d_{yz},d_{xyz}$, if the right distance in a
metric triangle is history-same to $\emp$, then the shared endpoint and right
endpoint collapse and the remaining displayed distances read back to the left
endpoint:
$$
\begin{array}{l}
\mathsf{MetricDistanceWitness}(x,y,d_{xy})\Longrightarrow
\mathsf{MetricDistanceWitness}(y,z,d_{yz})\Longrightarrow
\mathsf{MetricDistanceWitness}(d_{xy},z,d_{xyz})\Longrightarrow\\
\hsame(d_{yz},\emp)\Longrightarrow
\hsame(y,\emp)\land\hsame(z,\emp)\land
\hsame(d_{xy},x)\land\hsame(d_{xyz},x).
\end{array}
$$
\end{theorem}
\begin{proof}
Empty-distance exactness for the witness from $y$ to $z$ gives
$y\hsame\emp$ and $z\hsame\emp$. Collapsing $y$ in the first witness and using
empty-right exactness identifies $d_{xy}$ with $x$. Collapsing $z$ in the outer
witness identifies $d_{xyz}$ with $d_{xy}$, and transitivity composes this with
the previous readback to obtain $d_{xyz}\hsame x$.
\end{proof}
\leanchecked{BEDC.Derived.MetricUp.MetricDistanceWitness\_triangle\_right\_distance\_empty\_collapse}

\begin{theorem}[Metric triangle left distance empty visible-context closure]
\label{thm:metric-distance-witness-triangle-left-distance-empty-visible-context-closed}
For histories $p,q,x,y,z,d_{xy},d_{yz},d_{xyz}$, if the left distance in a
metric triangle is history-same to $\emp$, then the triangle closes through
the visible context with empty central source:
$$
\begin{array}{l}
\mathsf{UnaryHistory}(p)\Longrightarrow\mathsf{UnaryHistory}(q)\Longrightarrow
\mathsf{MetricDistanceWitness}(x,y,d_{xy})\Longrightarrow\\
\mathsf{MetricDistanceWitness}(y,z,d_{yz})\Longrightarrow
\mathsf{MetricDistanceWitness}(d_{xy},z,d_{xyz})\Longrightarrow
\hsame(d_{xy},\emp)\Longrightarrow\\
\mathsf{MetricDistanceWitness}(p\cdot\emp,z\cdot q,(p\cdot d_{xyz})\cdot q).
\end{array}
$$
\end{theorem}
\begin{proof}
The left-distance empty collapse identifies the left endpoints with $\emp$ and
reads $d_{xyz}$ back to $z$. Empty-left exactness therefore builds the central
witness $\mathsf{MetricDistanceWitness}(\emp,z,d_{xyz})$. The visible-context
iff then appends the certified unary ledgers $p$ and $q$ around this central
witness.
\end{proof}
\leanchecked{BEDC.Derived.MetricUp.MetricDistanceWitness\_triangle\_left\_distance\_empty\_visible\_context\_closed}

\begin{theorem}[Metric triangle right distance empty visible-context closure]

the visible context with empty central target:
$$
\begin{array}{l}
\mathsf{UnaryHistory}(p)\Longrightarrow\mathsf{UnaryHistory}(q)\Longrightarrow

\mathsf{MetricDistanceWitness}(d_{xy},z,d_{xyz})\Longrightarrow
\hsame(d_{yz},\emp)\Longrightarrow\\
\mathsf{MetricDistanceWitness}(p\cdot x,\emp\cdot q,(p\cdot d_{xyz})\cdot q).

$$
\end{theorem}
\begin{proof}

The visible-context iff then appends the certified unary ledgers $p$ and $q$

\end{proof}
