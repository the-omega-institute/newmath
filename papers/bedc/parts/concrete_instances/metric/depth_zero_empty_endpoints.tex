\begin{theorem}[Metric distance depth append-left cancellation]
\label{thm:metric-distance-depth-append-left-cancel}

depths after appending the same left prefix cancels that prefix:
$$
\mathsf{MetricDistanceDepth}(p\cdot d)
=
\mathsf{MetricDistanceDepth}(p\cdot e)
\Longrightarrow
\mathsf{MetricDistanceDepth}(d)=\mathsf{MetricDistanceDepth}(e).
$$
\end{theorem}
\begin{proof}
The structural depth of a left append decomposes as the depth of the common
prefix plus the depth of the tail. Rewriting both displayed sides by this
depth-add equation reduces the claim to left cancellation for addition in
\(\mathbb{N}\), yielding equality of the two tail depths.
\end{proof}

\begin{theorem}[Metric strict prefix depth decomposition]
\label{thm:metric-distance-depth-strict-prefix-decomposes}
For histories $h,k:\mathsf{BHist}$, a strict unary prefix witness from $h$ to
$k$ exposes a nonempty unary tail whose continuation reconstructs $k$, and the
metric depth of $k$ splits as the depth of $h$ plus the depth of that tail:
$$
\mathsf{NatUnaryStrictPrefix}(h,k)
\Longrightarrow
\exists tail:\mathsf{BHist}.\;
\mathsf{UnaryHistory}(tail)\land (tail=\emp\to\mathsf{False})\land
\ContR(h,tail,k)\land
\mathsf{MetricDistanceDepth}(k)
=
\mathsf{MetricDistanceDepth}(h)+\mathsf{MetricDistanceDepth}(tail).
$$
\end{theorem}
\begin{proof}
Unpack the strict-prefix witness. It already supplies a unary nonempty tail
and a continuation ledger from $h$ through that tail to $k$. Inverting the
continuation identifies $k$ with $h\cdot tail$. Induction on the concrete tail
then proves the structural depth equation for append, giving the displayed
sum formula.
\end{proof}
