\begin{theorem}[Metric depth-zero distance has empty endpoints]
\label{thm:metric-distance-witness-depth-zero-empty-endpoints}
For histories $x,y,d$, a concrete distance witness whose distance ledger has
structural depth zero has empty endpoints:
$$
\mathsf{MetricDistanceWitness}(x,y,d)\to
\mathsf{MetricDistanceDepth}(d)=0\to
\hsame(x,\emp)\land\hsame(y,\emp).
$$
\end{theorem}
\begin{proof}
Classify the displayed distance history. The empty distance case is exactly the
empty-distance witness exactness theorem. A zero-headed or one-headed distance
has successor depth, contradicting the displayed zero-depth equation by
constructor no-confusion for natural numbers.
\end{proof}
\leanchecked{BEDC.Derived.MetricUp.MetricDistanceWitness\_depth\_zero\_empty\_endpoints}

\begin{theorem}[Metric empty endpoints have depth-zero distance]
\label{thm:metric-distance-witness-empty-endpoints-depth-zero}
For histories $x,y,d$, a concrete distance witness whose endpoints are both
history-same to $\emp$ has structural distance depth zero:
$$
\mathsf{MetricDistanceWitness}(x,y,d)\to
\hsame(x,\emp)\to
\hsame(y,\emp)\to
\mathsf{MetricDistanceDepth}(d)=0.
$$
\end{theorem}
\begin{proof}
Eliminate the two endpoint sameness witnesses, reducing both endpoints to
$\emp$. The continuation ledger inside the distance witness then reduces the
distance history itself to $\emp$, and the depth definition evaluates to zero.
\end{proof}
\leanchecked{BEDC.Derived.MetricUp.MetricDistanceWitness\_empty\_endpoints\_depth\_zero}

\begin{theorem}[Metric distance depth zero iff empty]
\label{thm:metric-distance-depth-zero-iff-empty}
For a history $d$, structural metric distance depth is zero exactly when $d$
classifies with the empty history:
$$
\mathsf{MetricDistanceDepth}(d)=0
\Longleftrightarrow
\hsame(d,\emp).
$$
\end{theorem}
\begin{proof}
Case-analyze the history $d$. The empty case is reflexive. A zero-headed or
one-headed history has successor depth, so it cannot satisfy the zero-depth
equation. Conversely, a history-sameness proof with $\emp$ reduces the depth
equation to the empty reflexive case.
\end{proof}
\leanchecked{BEDC.Derived.MetricUp.MetricDistanceDepth\_zero\_iff\_empty}

\begin{theorem}[Metric distance positive depth iff nonempty]
\label{thm:metric-distance-depth-positive-iff-nonempty}
For a history $d$, positive structural metric distance depth is equivalent to
not being history-same as $\emp$:
$$
0<\mathsf{MetricDistanceDepth}(d)
\Longleftrightarrow
(d\hsame\emp\to\mathsf{False}).
$$
\end{theorem}
\begin{proof}
Case-analyze $d$. The empty case has depth zero and cannot satisfy the
nonempty predicate. A zero-headed or one-headed history has successor depth,
and constructor separation rules out history-sameness with $\emp$.
\end{proof}
\leanchecked{BEDC.Derived.MetricUp.MetricDistanceDepth\_positive\_iff\_nonempty}

\begin{theorem}[Metric distance depth append-left equality emptiness]
\label{thm:metric-distance-depth-append-left-eq-iff-empty}
For histories $p$ and $d$, appending $p$ on the left preserves structural
metric distance depth exactly when $p$ is the empty history:
$$
\mathsf{MetricDistanceDepth}(p\cdot d)=\mathsf{MetricDistanceDepth}(d)
\Longleftrightarrow
\hsame(p,\emp).
$$
\end{theorem}
\begin{proof}
Induct on the displayed tail $d$. The empty tail is precisely
\autoref{thm:metric-distance-depth-zero-iff-empty}. A zero-headed or one-headed
tail reduces the equality through successor injectivity, and the reverse
direction rebuilds the same successor equality after the left prefix is
identified with $\emp$.
\end{proof}
\leanchecked{BEDC.Derived.MetricUp.MetricDistanceDepth\_append\_left\_eq\_iff\_empty}

\begin{theorem}[Metric visible-context depth-zero collapse]
\label{thm:metric-distance-witness-visible-context-depth-zero-collapse}
For histories $p,q,x,y,d$, a visible-context distance witness whose central
distance has structural depth zero collapses the central distance and both
central endpoints to the empty history:
$$
\mathsf{MetricDistanceWitness}(p\cdot x,y\cdot q,(p\cdot d)\cdot q)
\to
\mathsf{MetricDistanceDepth}(d)=0
\to
\hsame(d,\emp)\land\hsame(x,\emp)\land\hsame(y,\emp).
$$
\end{theorem}
\begin{proof}
Apply visible-context exactness to peel away the prefix $p$ and suffix $q$,
recovering the central witness $\mathsf{MetricDistanceWitness}(x,y,d)$. The
depth-zero endpoint theorem gives $\hsame(x,\emp)$ and $\hsame(y,\emp)$, while
the depth-zero classifier theorem gives $\hsame(d,\emp)$.
\end{proof}

\begin{theorem}[Metric visible-context positive central depth has a nonempty endpoint]
\label{thm:metric-distance-witness-visible-context-positive-depth-nonempty-endpoint}
For histories $p,q,x,y,d$, a visible-context distance witness whose central
distance has positive structural depth exposes a nonempty central endpoint:
$$
\mathsf{MetricDistanceWitness}(p\cdot x,y\cdot q,(p\cdot d)\cdot q)
\to
\bigl(\mathsf{MetricDistanceDepth}(d)=0\to\mathsf{False}\bigr)
\to
\bigl((\hsame(x,\emp)\to\mathsf{False})
\vee(\hsame(y,\emp)\to\mathsf{False})\bigr).
$$
\end{theorem}
\begin{proof}
Apply visible-context exactness to peel away the displayed prefix $p$ and
suffix $q$, obtaining the central witness
$\mathsf{MetricDistanceWitness}(x,y,d)$.

It remains to convert positive structural depth into nonempty-distance input.
If $\hsame(d,\emp)$ held, the depth-zero classifier theorem would give
$\mathsf{MetricDistanceDepth}(d)=0$, contradicting the positive-depth
hypothesis.

Apply the nonempty-distance endpoint theorem to the central witness and this
derived nonempty-distance contradiction. The result is exactly the displayed
disjunction: either the left central endpoint is not history-same to $\emp$, or
the right central endpoint is not history-same to $\emp$.
\end{proof}
