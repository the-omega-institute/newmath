\begin{definition}[PublicKey decrypt-encrypt transcript interface]
\label{def:publickey-decrypt-encrypt-transcript-interface}
Let $\mathcal P$ be a $\PublicKeyUp$ certificate over the group-side naming certificate of \autoref{ch:concrete-instances-group-namecert} and the number-field plaintext certificate of \autoref{ch:concrete-instances-numfield-namecert}. Write $\mathsf{Msg}_{\mathcal P}(m)$ for its plaintext carrier and $\mu_{\mathcal P}(m,m')$ for its plaintext classifier. The plaintext certificate is a carrier naming certificate in the sense of \autoref{def:lean-carrier-naming-certificate}; in the concrete rational number-field surface it is supplied by \autoref{thm:numfield-reflexive-namecert-obligation-surface}.

Write $\mathsf{Pub}_{\mathcal P}(pk)$, $\mathsf{Sec}_{\mathcal P}(sk)$, and $\mathsf{Ciph}_{\mathcal P}(c)$ for the public-key, secret-key, and ciphertext carrier rows. The group-side rows named by \autoref{def:group-stability-certificate} type the key-generation and encryption witnesses. Write
$$
\mathsf{PKKeyGen}_{\mathcal P}(pk,sk)
$$
for the row witness that $pk$ and $sk$ are a generated public-key pair, write
$$
\mathsf{PKEncrypt}_{\mathcal P}(pk,m,c)
$$
for the row witness that $c$ is the encryption transcript of plaintext $m$ under public key $pk$, and write
$$
\mathsf{PKDecrypt}_{\mathcal P}(sk,c,d)
$$
for the row witness that secret key $sk$ decrypts ciphertext $c$ to plaintext endpoint $d$.

A certified encryption transcript is the predicate
$$
\mathsf{PKCertifiedEnc}_{\mathcal P}(pk,m,c)
\Longleftrightarrow
\mathsf{Pub}_{\mathcal P}(pk)\land
\mathsf{Msg}_{\mathcal P}(m)\land
\mathsf{Ciph}_{\mathcal P}(c)\land
\mathsf{PKEncrypt}_{\mathcal P}(pk,m,c).
$$
The decrypt-encrypt exactness row of $\mathcal P$ is
$$
\mathsf{PKDecryptEncryptExact}_{\mathcal P}
\Longleftrightarrow
\forall pk,sk,m,c:\mathsf{BHist}.\;
\mathsf{PKKeyGen}_{\mathcal P}(pk,sk)\land
\mathsf{PKCertifiedEnc}_{\mathcal P}(pk,m,c)
\Longrightarrow
\exists d:\mathsf{BHist}.\;
\mathsf{PKDecrypt}_{\mathcal P}(sk,c,d)\land
\mu_{\mathcal P}(d,m).
$$
\end{definition}
\leandef{BEDC.Derived.PublicKeyUp.PKCertifiedEnc}
\leandef{BEDC.Derived.PublicKeyUp.PKDecryptEncryptExact}

\begin{theorem}[PublicKey decrypt-encrypt correctness]
\label{thm:publickey-decrypt-encrypt-correctness}
Let $\mathcal P$ be a $\PublicKeyUp$ certificate with transcript rows as in \autoref{def:publickey-decrypt-encrypt-transcript-interface}, and assume $\mathsf{PKDecryptEncryptExact}_{\mathcal P}$. For histories $pk$, $sk$, $m$, and $c$, if
$$
\mathsf{PKKeyGen}_{\mathcal P}(pk,sk)
\qquad\text{and}\qquad
\mathsf{PKCertifiedEnc}_{\mathcal P}(pk,m,c),
$$
then decrypting $c$ with $sk$ returns a carried plaintext endpoint in the message-classifier class of $m$:
$$
\exists d:\mathsf{BHist}.\;
\mathsf{Msg}_{\mathcal P}(d)\land
\mathsf{PKDecrypt}_{\mathcal P}(sk,c,d)\land
\mu_{\mathcal P}(d,m).
$$
\end{theorem}
\leanchecked{BEDC.Derived.PublicKeyUp.PublicKeyDecryptEncrypt\_correctness}
\begin{proof}
Unfold \autoref{def:publickey-decrypt-encrypt-transcript-interface} only as far as the certified-encryption predicate and the decrypt-encrypt exactness row. The certified-encryption hypothesis gives the carried plaintext fact $\mathsf{Msg}_{\mathcal P}(m)$ together with the encryption transcript under $pk$.

Apply $\mathsf{PKDecryptEncryptExact}_{\mathcal P}$ to the generated keypair row and the certified-encryption transcript. This gives a history $d$ with
$$
\mathsf{PKDecrypt}_{\mathcal P}(sk,c,d)
\qquad\text{and}\qquad
\mu_{\mathcal P}(d,m).
$$
The plaintext classifier is supplied by a carrier naming certificate. Its symmetry field reverses $\mu_{\mathcal P}(d,m)$ to $\mu_{\mathcal P}(m,d)$, and its carrier-transport field carries $\mathsf{Msg}_{\mathcal P}(m)$ along that comparison to obtain $\mathsf{Msg}_{\mathcal P}(d)$.

Using the same witness $d$, the carried plaintext fact, the decryption row, and the classifier comparison prove the displayed existential statement.
\end{proof}
\leanchecked{BEDC.Derived.PublicKeyUp.PublicKey\_decrypt\_encrypt\_correctness}

\begin{definition}[PublicKey decryption output determinacy row]
\label{def:publickey-decrypt-output-determinacy-row}
Let $\mathcal P$ be a $\PublicKeyUp$ certificate with transcript rows as in \autoref{def:publickey-decrypt-encrypt-transcript-interface}. The decrypt-output determinacy row of $\mathcal P$ is the predicate
$$
\mathsf{PKDecryptOutputDet}_{\mathcal P}
\Longleftrightarrow
\forall pk,sk,c,d_1,d_2:\mathsf{BHist}.\;
\mathsf{PKKeyGen}_{\mathcal P}(pk,sk)\land
\mathsf{PKDecrypt}_{\mathcal P}(sk,c,d_1)\land
\mathsf{PKDecrypt}_{\mathcal P}(sk,c,d_2)
\Longrightarrow
\mu_{\mathcal P}(d_1,d_2).
$$
It records that decryption for a generated secret key and a fixed ciphertext has a plaintext endpoint determined up to the plaintext classifier of the certificate.
\end{definition}

\begin{theorem}[PublicKey decryption output determinacy]
\label{thm:publickey-decrypt-output-determinacy}
Let $\mathcal P$ be a $\PublicKeyUp$ certificate with transcript rows as in \autoref{def:publickey-decrypt-encrypt-transcript-interface}, and assume $\mathsf{PKDecryptOutputDet}_{\mathcal P}$. For histories $pk$, $sk$, $c$, $d_1$, and $d_2$, if
$$
\mathsf{PKKeyGen}_{\mathcal P}(pk,sk)
\qquad\text{and}\qquad
\mathsf{PKDecrypt}_{\mathcal P}(sk,c,d_1)
\qquad\text{and}\qquad
\mathsf{PKDecrypt}_{\mathcal P}(sk,c,d_2),
$$
then the two decryption endpoints are plaintext-classifier equal:
$$
\mu_{\mathcal P}(d_1,d_2).
$$
\end{theorem}
\begin{proof}
By \autoref{def:publickey-decrypt-output-determinacy-row}, the row $\mathsf{PKDecryptOutputDet}_{\mathcal P}$ is the universal BEDC-history statement that any two decryption endpoints for the same generated keypair row and ciphertext are related by $\mu_{\mathcal P}$. Instantiate that universal statement at the displayed histories $pk$, $sk$, $c$, $d_1$, and $d_2$. The three hypotheses are exactly its antecedent, so the instantiated implication gives $\mu_{\mathcal P}(d_1,d_2)$.
\end{proof}
