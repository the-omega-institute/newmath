\chapter{Inter-Hist Locality Name Certificate}
\label{ch:concrete-instances-inter-hist-locality}

The object $\mathsf{InterHistLocalityUp}$ packages the local two-history
comparison surface used when BEDC speaks about relations between independent
histories.  The certificate does not add a global clock, a shared frame, or a
host-level equality oracle.  It records only two histories, a displayed
locality witness, the two continuation endpoints generated from that witness,
and the classifier that compares those endpoints.

\section{Source}
\label{sec:concrete-instances-inter-hist-locality-source}

\begin{definition}[Source rows for $\mathsf{InterHistLocalityUp}$]
\label{def:concrete-instances-inter-hist-locality-source}
A source row is a tuple
$(h_a,h_b,w,e_a,e_b,\ell_a,\ell_b)$ where $h_a$ and $h_b$ are the two
component histories, $w$ is the locality witness inspected by both histories,
$e_a$ and $e_b$ are the visible endpoints, and
$\ell_a : \Cont(h_a,w,e_a)$ and $\ell_b : \Cont(h_b,w,e_b)$ are the displayed
finite continuation witnesses.  The row is the whole admissible input surface;
no global synchronization datum is part of the source.
\end{definition}
\leandef{BEDC.Derived.InterHistLocalityUp.SourceRows}

\section{Pattern}
\label{sec:concrete-instances-inter-hist-locality-pattern}

\begin{definition}[Local two-endpoint pattern]
\label{def:concrete-instances-inter-hist-locality-pattern}
The pattern accepted by $\mathsf{InterHistLocalityUp}$ is exactly the paired
continuation shape
$$
\begin{aligned}
  \Cont(h_a,w,e_a) \quad\text{and}\quad \Cont(h_b,w,e_b).
\end{aligned}
$$
Thus the two histories are compared only after the same local witness has been
displayed inside the finite BEDC ledger.  The pattern is intentionally weaker
than a multi-observer physics semantics: it is the locality interface on which
such semantics may later be built.
\end{definition}
\leandef{BEDC.Derived.InterHistLocalityUp.Pattern}

\section{Classifier}
\label{sec:concrete-instances-inter-hist-locality-classifier}

\begin{definition}[Locality classifier]
\label{def:concrete-instances-inter-hist-locality-classifier}
Two source rows classify together when their corresponding visible endpoints
are history-same:
$$
\begin{aligned}
  e_a \hsame e'_a \quad\text{and}\quad e_b \hsame e'_b.
\end{aligned}
$$
The classifier deliberately ignores any host identity of the raw row record.
Only the two endpoint judgments exposed by the locality ledger are public.
\end{definition}
\leandef{BEDC.Derived.InterHistLocalityUp.Classifier}

\section{Stability}
\label{sec:concrete-instances-inter-hist-locality-stability}

\begin{theorem}[Locality is stable under common history extension]
\label{thm:concrete-instances-inter-hist-locality-stability}
If two inter-Hist locality rows classify together, then extending both rows by
the same finite continuation suffix preserves the paired locality pattern and
preserves the classifier.  The operation extends the shared witness from $w$ to
$w \mathbin{\mathrm{append}} x$ and extends the two endpoints from $e_a,e_b$ to
$e_a \mathbin{\mathrm{append}} x,e_b \mathbin{\mathrm{append}} x$.
\end{theorem}
\leanchecked{BEDC.Derived.InterHistLocalityUp.stability}

\begin{proof}
The proof is the associativity of finite continuation, applied separately to
the two component histories, together with congruence of $\hsame$ under the
same appended suffix.  No global frame, quotient, choice operator, or
propositional extensionality principle is used.
\end{proof}

\section{Ledger}
\label{sec:concrete-instances-inter-hist-locality-ledger}

\begin{definition}[Ledger for $\mathsf{InterHistLocalityUp}$]
\label{def:concrete-instances-inter-hist-locality-ledger}
The ledger retains the original source row: the two histories, the locality
witness, both visible endpoints, and both continuation witnesses.  This is the
traceability policy for the bridge.  A public inter-Hist locality judgment can
always be audited back to the finite source data that generated its two
classifier endpoints.
\end{definition}
\leandef{BEDC.Derived.InterHistLocalityUp.Ledger}
