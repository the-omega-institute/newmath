\chapter{A Concrete Naming Certificate for $\DistributionUp$}
\label{ch:concrete-instances-distribution-namecert}

The distribution interface $\DistributionUp$ packages the pushforward measure of a random variable on its target space. It depends on $\NameCert_{\RandomVarUp}$ (\autoref{ch:concrete-instances-randomvar-namecert}) and $\NameCert_{\MeasureUp}$ (\autoref{ch:concrete-instances-measure-namecert}).

\section{Carrier definition}
\label{sec:distribution-carrier}

\begin{definition}[Distribution pushforward carrier]
\label{def:distribution-pushforward-carrier}
A carried $\DistributionUp$ datum consists of a carried $\RandomVarUp$ map $X:S\to T$ as in \autoref{sec:randomvar-carrier}, a $\ProbSpaceUp$ source carrier whose measurable-event family and probability measure are written $\mathcal{A}_S$ and $\mu_S$, and a target $\MeasureUp$ carrier whose measurable-event family is written $\mathcal{A}_T$. For each target measurable event $B\in\mathcal{A}_T$, the $\RandomVarUp$ measurable-preimage row supplies a source measurable event
$$
X^{-1}(B)\in\mathcal{A}_S.
$$
The pushed-forward endpoint is the RealUp-valued row
$$
\mu_X(B):=\mu_S(X^{-1}(B)).
$$
Thus the $\DistributionUp$ carrier is the target measurable family $\mathcal{A}_T$ together with the endpoint assignment $B\mapsto\mu_X(B)$ whose source is the probability measure of the carried source.
\end{definition}
\leandef{BEDC.Derived.DistributionUp.DistributionPushforwardSourceSpec}

\begin{definition}[Distribution pushforward classifier]
\label{def:distribution-pushforward-classifier}
Let $(\mathcal{A}_T,\mu_X)$ and $(\mathcal{A}'_T,\mu'_Y)$ be two displayed $\DistributionUp$ pushforward rows carried by target event classifiers $\sim_{\mathcal{A}_T}$ and by the $\RealUp$ classifier $\sim_R$ of \autoref{def:real-classifier-specification}. The rows are classified when, for every pair of displayed target measurable events $B\in\mathcal{A}_T$ and $B'\in\mathcal{A}'_T$ with
$$
B\sim_{\mathcal{A}_T}B',
$$
their pushed-forward endpoints agree under the real classifier:
$$
\mu_X(B)\sim_R\mu'_Y(B').
$$
Equivalently, the event classifier decides which target events may be compared, and the RealUp classifier decides equality of their measured pushforward endpoints.
\end{definition}
\leandef{BEDC.Derived.DistributionUp.DistributionPushforwardClassifier}
\leanchecked{BEDC.Derived.DistributionUp.DistributionPushforwardClassifier\_row}

\section{The certificate}
\label{sec:distribution-certificate}

\begin{theorem}[Distribution pushforward row]
\label{thm:distribution-pushforward-row}
Let $X:S\to T$ be a carried $\RandomVarUp$ map from a $\ProbSpaceUp$ source to a target $\MeasureUp$ carrier, and let $\mu_X$ be the $\DistributionUp$ endpoint assignment of \autoref{def:distribution-pushforward-carrier}. Write $\mathcal{A}_T$ for the target measurable-event family, $\mu_S$ for the source probability measure, and $x\sim_R y$ for the $\RealUp$ classifier. Then for every target measurable event $B\in\mathcal{A}_T$,
$$
\mu_X(B)\sim_R\mu_S(X^{-1}(B)).
$$
\end{theorem}
\leanchecked{BEDC.Derived.DistributionUp.DistributionPushforward\_row}
\begin{proof}
By \autoref{def:distribution-pushforward-carrier}, the endpoint $\mu_X(B)$ is defined by applying the source probability measure to the measurable preimage supplied by the $\RandomVarUp$ row:
$$
\mu_X(B):=\mu_S(X^{-1}(B)).
$$

The measurable-preimage row ensures that $X^{-1}(B)$ is a legitimate source measurable event before the source measure is read. The source $\ProbSpaceUp$ carrier supplies the measure row $\mu_S$ on that event.

The displayed equality is therefore an endpoint identity in $\RealUp$. Reflexivity of the $\RealUp$ classifier gives
$$
\mu_S(X^{-1}(B))\sim_R\mu_S(X^{-1}(B)).
$$
Transport along the definitional carrier equality reads this as
$$
\mu_X(B)\sim_R\mu_S(X^{-1}(B)).
$$
\end{proof}
\leanchecked{BEDC.Derived.DistributionUp.DistributionPushforwardCarrier\_row}

\begin{theorem}[Distribution NameCert obligation packet]
\label{thm:distribution-namecert-obligation-packet}
The $\DistributionUp$ pushforward surface supplies the NameCert obligation packet for a carried random variable $X:S\to T$ over a source $\ProbSpaceUp$ carrier and a target $\MeasureUp$ carrier. Its ten-field roadmap fit is:
\begin{enumerate}
\item \textbf{Traditional target}: the distribution law of $X$, regarded as the pushforward probability measure on target measurable events.
\item \textbf{BEDC source}: the $\RandomVarUp$ measurable-preimage certificate, the source $\ProbSpaceUp$ measure rows, the target $\MeasureUp$ event classifier, and the shared $\RealUp$ classifier.
\item \textbf{Carrier choice}: the target measurable family $\mathcal{A}_T$ equipped with the endpoint assignment $B\mapsto\mu_S(X^{-1}(B))$ from \autoref{def:distribution-pushforward-carrier}.
\item \textbf{Classifier choice}: the row classifier of \autoref{def:distribution-pushforward-classifier}, combining target-event classification with RealUp endpoint classification.
\item \textbf{Stability obligations}: target-event transport through the $\MeasureUp$ classifier, source-preimage transport through the $\RandomVarUp$ measurable-preimage rows, source-measure transport through the source $\MeasureUp$ ledger, and reflexive, symmetric, transitive, and congruent use of the $\RealUp$ classifier.
\item \textbf{Ledger}: the source event $X^{-1}(B)$ is retained as the hidden witness explaining each target endpoint, while the public row exposes only the target event $B$ and the real endpoint $\mu_X(B)$.
\item \textbf{Core theorem list}: the pushforward row of \autoref{thm:distribution-pushforward-row}, total-mass inheritance \autoref{thm:distribution-pushforward-total-mass-unit}, empty-event inheritance \autoref{thm:distribution-empty-target-event-zero-mass}, finite additivity \autoref{thm:distribution-pushforward-finite-disjoint-union-additivity}, monotonicity \autoref{thm:distribution-pushforward-monotone-events}, probability bounds \autoref{thm:distribution-pushforward-probability-bounds}, complement-mass decomposition \autoref{thm:distribution-complement-mass-decomposition}, countable disjoint sigma-additivity \autoref{thm:distribution-pushforward-countable-disjoint-sigma-additivity}, and continuity from below \autoref{thm:distribution-pushforward-continuity-from-below}.
\item \textbf{Exactness predicate}: for every target measurable event $B$, the row exactness predicate is $\mu_X(B)\sim_R\mu_S(X^{-1}(B))$, with target-event classifier transport and source-event ledger transport preserving the displayed endpoint.
\item \textbf{Standard bridge}: the ordinary measure-theoretic pushforward law $X_\ast\mu_S$ on the target measurable space, read conservatively through the target event classifier and the RealUp endpoint classifier.
\item \textbf{Status}: the packet records the carrier and certificate obligations at the paper level, with Lean formalization reserved for the $\BEDC.Derived.DistributionUp$ namespace.
\end{enumerate}
\end{theorem}
\leanchecked{BEDC.Derived.DistributionUp.DistributionPushforwardCarrier\_semantic\_name\_certificate}
\begin{proof}
The carrier and classifier fields are exactly \autoref{def:distribution-pushforward-carrier} and \autoref{def:distribution-pushforward-classifier}. The carrier uses the $\RandomVarUp$ measurable-preimage row to make $X^{-1}(B)$ a source measurable event, and then reads the source $\ProbSpaceUp$ measure row at that event.

The stability and exactness fields follow from the same chain. Target-event transport is supplied by the $\MeasureUp$ event classifier; source-preimage transport is supplied by the RandomVarUp measurable-preimage certificate; source-measure transport is read through the source $\MeasureUp$ ledger; endpoint comparison is composed by RealUp classifier reflexivity and transitivity.

The ledger field records that the source event is kept as hidden witness data for each public target endpoint. This is precisely the information consumed by the downstream mass, empty-event, additivity, monotonicity, sigma-additivity, and continuity-from-below rows in this chapter.

The standard bridge is the usual pushforward measure surface, but only as a conservative readback of the displayed BEDC rows. No host quotient or host event equality is imported into the carrier; all comparisons pass through the target-event and RealUp classifiers.
\end{proof}

\begin{theorem}[Distribution pushforward total mass is the probability unit]
\label{thm:distribution-pushforward-total-mass-unit}
Let $X$ be a carried $\RandomVarUp$ map as in \autoref{sec:randomvar-carrier}, from a $\ProbSpaceUp$ source as in \autoref{sec:probspace-carrier} to a $\MeasureUp$ target as in \autoref{sec:measure-carrier}. Write $\Omega_S$ for the source total event, $\Omega_T$ for the target total event, $\mathcal{A}_S,\mathcal{A}_T$ for the source and target measurable-event families, $\mu_S$ for the source measure, $\mu_X$ for the $\DistributionUp$ pushforward measure as in \autoref{sec:distribution-carrier}, $1_R$ for the probability unit, and $x\sim_R y$ for the $\RealUp$ classifier of \autoref{def:real-classifier-specification}. Suppose:
\begin{itemize}
\item $\Omega_S\in\mathcal{A}_S$ and $\Omega_T\in\mathcal{A}_T$;
\item the $\RandomVarUp$ certificate row of \autoref{sec:randomvar-certificate} identifies the target-total preimage as the source total event,
$$
X^{-1}(\Omega_T)=\Omega_S;
$$
\item the $\DistributionUp$ pushforward row of \autoref{sec:distribution-certificate} gives, at every target measurable event $B\in\mathcal{A}_T$,
$$
\mu_X(B)\sim_R\mu_S(X^{-1}(B));
$$
\item the $\ProbSpaceUp$ total-mass row of \autoref{sec:probspace-certificate} gives
$$
\mu_S(\Omega_S)\sim_R1_R.
$$
\end{itemize}
Then
$$
\mu_X(\Omega_T)\sim_R1_R.
$$
\end{theorem}
\leanchecked{BEDC.Derived.DistributionUp.DistributionPushforward\_total\_mass\_unit}
\leanvariant{BEDC.Derived.DistributionUp.DistributionPushforwardWitness\_total\_mass\_unit}
\begin{proof}
Apply the pushforward row to the target total event $B:=\Omega_T$. Since $\Omega_T\in\mathcal{A}_T$, this gives
$$
\mu_X(\Omega_T)\sim_R\mu_S(X^{-1}(\Omega_T)).
$$
The random-variable target-total row identifies $X^{-1}(\Omega_T)$ with $\Omega_S$ as a measurable source event, and source-event transport through \autoref{thm:semantic-namecert-pattern-ledger-transport} permits the source-measure value to be read at the transported event. Thus the preceding comparison reads
$$
\mu_X(\Omega_T)\sim_R\mu_S(\Omega_S).
$$
The probability-space total-mass row gives
$$
\mu_S(\Omega_S)\sim_R1_R.
$$
The relation $\sim_R$ is the $\RealUp$ classifier, hence its transitivity field composes these two classifications and yields
$$
\mu_X(\Omega_T)\sim_R1_R.
$$
\end{proof}

\begin{theorem}[Distribution pushforward finite disjoint-union additivity]
\label{thm:distribution-pushforward-finite-disjoint-union-additivity}
Let $X:S\to T$ be a carried $\RandomVarUp$ map as in \autoref{sec:randomvar-carrier}, from a $\ProbSpaceUp$ source as in \autoref{sec:probspace-carrier} to a $\MeasureUp$ target as in \autoref{sec:measure-carrier}. Write $\mathcal{A}_S$ and $\mathcal{A}_T$ for the source and target measurable-event families, $\mu_S$ for the source measure, $\mu_X$ for the $\DistributionUp$ pushforward measure as in \autoref{sec:distribution-carrier}, and $x\sim_R y$ for the $\RealUp$ classifier of \autoref{def:real-classifier-specification}. Let $B,C,U_T$ be target events such that $B,C,U_T\in\mathcal{A}_T$ and the displayed target binary-union row reads $U_T$ as $B\cup C$:
$$
t\in U_T\quad\Longleftrightarrow\quad t\in B\ \text{or}\ t\in C.
$$
Assume $B$ and $C$ are disjoint. Define
$$
A_B:=X^{-1}(B),\qquad A_C:=X^{-1}(C),\qquad A_U:=X^{-1}(U_T).
$$
Let $U_S$ be the source binary union of $A_B$ and $A_C$ supplied by the source $\MeasureUp$ sigma-algebra row. Suppose:
\begin{itemize}
\item the $\RandomVarUp$ measurable-preimage row and the source sigma-algebra row satisfy the hypotheses of \autoref{thm:randomvar-preimage-disjoint-binary-union-exactness} for $B,C,U_T$;
\item the $\DistributionUp$ pushforward row of \autoref{sec:distribution-certificate} gives, at every target measurable event $E\in\mathcal{A}_T$,
$$
\mu_X(E)\sim_R\mu_S(X^{-1}(E));
$$
\item the source $\MeasureUp$ certificate and the shared $\RealUp$ additive $\AbGroupUp$ row satisfy the hypotheses of \autoref{thm:measure-finite-disjoint-union-additivity} for the disjoint measurable pair $A_B,A_C$ and its measurable binary union $U_S$;
\item source-event transport through \autoref{thm:semantic-namecert-pattern-ledger-transport} permits $\mu_S$ to be read across the source event classifier supplied by \autoref{def:measure-classifier-specification}.
\end{itemize}
Then
$$
\mu_X(U_T)\sim_R\mu_X(B)+\mu_X(C).
$$
\end{theorem}
\begin{proof}
Apply \autoref{thm:randomvar-preimage-disjoint-binary-union-exactness} to the displayed target union $U_T=B\cup C$. It gives
$$
A_B,A_C,A_U\in\mathcal{A}_S,
$$
identifies $U_S$ as a measurable source binary union, classifies the target-union preimage with that source union,
$$
A_U\sim_{\mathcal{A}_S}U_S,
$$
and shows that $A_B$ and $A_C$ are disjoint source events.

Apply the pushforward row to $U_T$, $B$, and $C$. This gives
$$
\mu_X(U_T)\sim_R\mu_S(A_U),
$$
$$
\mu_X(B)\sim_R\mu_S(A_B),\qquad \mu_X(C)\sim_R\mu_S(A_C).
$$
The source-event comparison $A_U\sim_{\mathcal{A}_S}U_S$, read through the source measure ledger by \autoref{thm:semantic-namecert-pattern-ledger-transport}, gives
$$
\mu_S(A_U)\sim_R\mu_S(U_S).
$$

Since $A_B$ and $A_C$ are disjoint measurable source events and $U_S$ is their measurable binary union, \autoref{thm:measure-finite-disjoint-union-additivity} applied to the source $\MeasureUp$ carrier yields
$$
\mu_S(U_S)\sim_R\mu_S(A_B)+\mu_S(A_C).
$$
Transitivity of the $\RealUp$ classifier composes the three comparisons into
$$
\mu_X(U_T)\sim_R\mu_S(A_B)+\mu_S(A_C).
$$

By symmetry of $\sim_R$, the pushforward comparisons for $B$ and $C$ also read
$$
\mu_S(A_B)\sim_R\mu_X(B),\qquad \mu_S(A_C)\sim_R\mu_X(C).
$$
The additive congruence field of the shared $\RealUp$ additive $\AbGroupUp$ row transports these two comparisons through $+$, giving
$$
\mu_S(A_B)+\mu_S(A_C)\sim_R\mu_X(B)+\mu_X(C).
$$
A final transitivity step gives
$$
\mu_X(U_T)\sim_R\mu_X(B)+\mu_X(C).
$$
\end{proof}
\leanchecked{BEDC.Derived.DistributionUp.DistributionPushforward\_finite\_disjoint\_union\_additivity}

\begin{theorem}[Distribution nonnegative value inheritance]
\label{thm:distribution-nonnegative-value-inheritance}
Let $X:S\to T$ be a carried $\RandomVarUp$ map as in \autoref{sec:randomvar-carrier}, from a $\ProbSpaceUp$ source as in \autoref{sec:probspace-carrier} to a $\MeasureUp$ target as in \autoref{sec:measure-carrier}. Write $\mathcal{A}_S$ and $\mathcal{A}_T$ for the source and target measurable-event families, $\mu_S$ for the source measure, $\mu_X$ for the $\DistributionUp$ pushforward measure as in \autoref{sec:distribution-carrier}, $x\sim_R y$ for the $\RealUp$ classifier of \autoref{def:real-classifier-specification}, and $\mathsf{NonNeg}_R$ for the nonnegative predicate exposed by the shared $\RealAlgOrder$ row. Let $B$ be a target event with $B\in\mathcal{A}_T$, and define
$$
A_B:=X^{-1}(B).
$$
Suppose:
\begin{itemize}
\item the $\RandomVarUp$ measurable-preimage row of \autoref{sec:randomvar-certificate} sends the target measurable event $B$ to the source measurable event $A_B$, so $A_B\in\mathcal{A}_S$;
\item the $\DistributionUp$ pushforward row of \autoref{sec:distribution-certificate} gives, at the target event $B$,
$$
\mu_X(B)\sim_R\mu_S(A_B);
$$
\item the source $\MeasureUp$ component of the $\ProbSpaceUp$ source exposes the nonnegative value row of \autoref{thm:measure-nonnegative-value-obligation}, and the same $\RealAlgOrder$ nonnegative predicate is stable under the $\RealUp$ classifier.
\end{itemize}
Then
$$
\mathsf{NonNeg}_R(\mu_X(B)).
$$
\end{theorem}
\begin{proof}
The measurable-preimage row makes $A_B$ a displayed measurable event in the source measurable family. Reading the source probability space through its underlying $\MeasureUp$ component, \autoref{thm:measure-nonnegative-value-obligation} supplies
$$
\mathsf{NonNeg}_R(\mu_S(A_B)).
$$
The pushforward row identifies the pushed-forward endpoint with the source endpoint:
$$
\mu_X(B)\sim_R\mu_S(A_B).
$$
By symmetry of the $\RealUp$ classifier, this also reads
$$
\mu_S(A_B)\sim_R\mu_X(B).
$$
Stability of the $\RealAlgOrder$ nonnegative predicate under $\sim_R$ transports the displayed source nonnegative row across this classifier comparison, giving
$$
\mathsf{NonNeg}_R(\mu_X(B)).
$$
\end{proof}

\begin{theorem}[Distribution pushforward monotone under target inclusion]
\label{thm:distribution-pushforward-monotone-under-target-inclusion}
Let $X:S\to T$ be a carried $\RandomVarUp$ map as in \autoref{sec:randomvar-carrier}, from a $\ProbSpaceUp$ source as in \autoref{sec:probspace-carrier} to a $\MeasureUp$ target as in \autoref{sec:measure-carrier}. Write $\mathcal{A}_S$ and $\mathcal{A}_T$ for the source and target measurable-event families, $\mu_S$ for the source measure, $\mu_X$ for the $\DistributionUp$ pushforward measure as in \autoref{sec:distribution-carrier}, and $x\sim_R y$ for the $\RealUp$ classifier of \autoref{def:real-classifier-specification}. Let $B$ and $C$ be target events in one carried target fiber with $B,C\in\mathcal{A}_T$, and suppose the public target readback displays $B\subseteq C$. Define
$$
A_B:=X^{-1}(B),\qquad A_C:=X^{-1}(C).
$$
Suppose:
\begin{itemize}
\item the target complement/relative-difference row of \autoref{def:measure-complement-difference-row}, the $\RandomVarUp$ measurable-preimage row, and the source complement/relative-difference row satisfy the hypotheses of \autoref{thm:randomvar-preimage-relative-difference-exactness} for the ambient target event $C$ and subevent $B$;
\item the $\DistributionUp$ pushforward row of \autoref{sec:distribution-certificate} gives, at every target measurable event $E\in\mathcal{A}_T$,
$$
\mu_X(E)\sim_R\mu_S(X^{-1}(E));
$$
\item the source $\MeasureUp$ certificate satisfies the hypotheses of \autoref{thm:measure-monotone-under-measurable-inclusion} for the displayed source inclusion $A_B\subseteq A_C$ and its measurable relative-difference row;
\item the RealAlgOrder order row on $\RealUp$ is stable under $\sim_R$ on both endpoints.
\end{itemize}
Then
$$
\mu_X(B)\le_R\mu_X(C).
$$
\end{theorem}
\leanchecked{BEDC.Derived.DistributionUp.DistributionPushforward\_monotone\_under\_target\_inclusion}
\begin{proof}
Apply \autoref{thm:randomvar-preimage-relative-difference-exactness} with the ambient target event $A:=C$ and the subevent $B$. It gives
$$
A_B,A_C\in\mathcal{A}_S
$$
and reads the target inclusion $B\subseteq C$ as the source inclusion
$$
A_B\subseteq A_C.
$$
It also supplies the source relative-difference ledger: the event $A_C\setminus A_B$ is measurable, is disjoint from $A_B$, and its displayed binary union with $A_B$ is classified as $A_C$ in the source event classifier.

These are exactly the inclusion-decomposition inputs for \autoref{thm:measure-monotone-under-measurable-inclusion} in the source $\MeasureUp$ carrier. Hence
$$
\mu_S(A_B)\le_R\mu_S(A_C).
$$

Apply the pushforward row to $B$ and $C$. Since both are target measurable,
$$
\mu_X(B)\sim_R\mu_S(A_B),\qquad \mu_X(C)\sim_R\mu_S(A_C).
$$
By symmetry of the $\RealUp$ classifier, the first comparison also reads $\mu_S(A_B)\sim_R\mu_X(B)$. Transport the left endpoint of the source order comparison across this classifier equality, and then transport the right endpoint across $\mu_S(A_C)\sim_R\mu_X(C)$ using stability of the RealAlgOrder order row. The resulting comparison is
$$
\mu_X(B)\le_R\mu_X(C).
$$
\end{proof}

\begin{theorem}[Distribution pushforward relative-difference additivity]
\label{thm:distribution-pushforward-relative-difference-additivity}
Let $X:S\to T$ be a carried $\RandomVarUp$ map as in \autoref{sec:randomvar-carrier}, from a $\ProbSpaceUp$ source as in \autoref{sec:probspace-carrier} to a target $\MeasureUp$ carrier as in \autoref{sec:measure-carrier}. Write $\mathcal{A}_S$ and $\mathcal{A}_T$ for the source and target measurable-event families, $\mu_S$ for the source measure, $\mu_X$ for the $\DistributionUp$ pushforward measure as in \autoref{sec:distribution-carrier}, and $x\sim_R y$ for the $\RealUp$ classifier of \autoref{def:real-classifier-specification}. Let $A$ and $B$ be target events in one carried target fiber with $A,B\in\mathcal{A}_T$, and suppose the public target readback displays $B\subseteq A$. Suppose the target complement/relative-difference row of \autoref{def:measure-complement-difference-row} supplies a measurable target event
$$
D_T:=A\setminus B
$$
with public membership readback
$$
t\in D_T\quad\Longleftrightarrow\quad t\in A\ \text{and}\ t\notin B.
$$
Define source preimages
$$
A_S:=X^{-1}(A),\qquad B_S:=X^{-1}(B),\qquad D_X:=X^{-1}(D_T).
$$
Let $D_S:=A_S\setminus B_S$ be the source relative-difference event supplied by the source $\MeasureUp$ complement/relative-difference row once the source inclusion $B_S\subseteq A_S$ is displayed. Suppose:
\begin{itemize}
\item the $\RandomVarUp$ measurable-preimage row and the source complement/relative-difference row satisfy the hypotheses of \autoref{thm:randomvar-preimage-relative-difference-exactness} for $A,B,D_T$;
\item the $\DistributionUp$ pushforward row of \autoref{sec:distribution-certificate} gives, at every target measurable event $E\in\mathcal{A}_T$,
$$
\mu_X(E)\sim_R\mu_S(X^{-1}(E));
$$
\item the source $\MeasureUp$ certificate and the shared $\RealUp$ additive $\AbGroupUp$ row satisfy the hypotheses of \autoref{thm:measure-relative-difference-additivity} for the displayed source inclusion $B_S\subseteq A_S$;
\item source-event transport through \autoref{thm:semantic-namecert-pattern-ledger-transport} permits $\mu_S$ to be read across the source event classifier supplied by \autoref{def:measure-classifier-specification};
\item the shared $\RealUp$ additive $\AbGroupUp$ row supplies addition congruence for $\sim_R$.
\end{itemize}
Then
$$
\mu_X(D_T)\sim_R\mu_S(D_S)
$$
and
$$
\mu_X(A)\sim_R\mu_X(B)+\mu_X(D_T).
$$
\end{theorem}

\begin{theorem}[Distribution pushforward countable disjoint-union sigma-additivity]
\label{thm:distribution-pushforward-countable-disjoint-sigma-additivity}
Let $X:S\to T$ be a carried $\RandomVarUp$ map from a $\ProbSpaceUp$ source to a $\MeasureUp$ target. Write $\mathcal{A}_S$ and $\mathcal{A}_T$ for the source and target measurable-event families, $\mu_S$ for the source measure, $\mu_X$ for the $\DistributionUp$ pushforward measure, $E\simeq_S F$ for the source event classifier, and $x\sim_R y$ for the $\RealUp$ classifier.
Let $(B_n)_{n\in\mathbb N}$ be target events with $B_n\in\mathcal{A}_T$ for every $n$, and let $U_T\in\mathcal{A}_T$ be a displayed target countable union:
$$
t\in U_T\Longleftrightarrow \exists n\in\mathbb N,\, t\in B_n.
$$
Assume that the target row is pairwise disjoint: if $i\neq j$, no carried target point has simultaneous membership in $B_i$ and $B_j$.
For each $n$, put $A_n:=X^{-1}(B_n)$ and put $A_U:=X^{-1}(U_T)$. Suppose the following rows are available.
\begin{enumerate}
\item The $\RandomVarUp$ measurable-preimage row sends every target measurable event in this display to a source measurable event and gives the public readback
$$
 s\in X^{-1}(E)\Longleftrightarrow X(s)\in E.
$$
\item The source countable-union row supplies the source measurable event
$$
U_S:=\bigvee\nolimits^S_{n\in\mathbb N}A_n\in\mathcal{A}_S,
$$
and the countable preimage-union bridge supplies the source event classifier comparison
$$
A_U\simeq_S U_S.
$$
\item The source $\MeasureUp$ surface is sigma-additive on displayed pairwise-disjoint source-measurable countable unions:
$$
\mu_S(U_S)\sim_R\sum_{n\in\mathbb N}\mu_S(A_n).
$$
\item The $\DistributionUp$ pushforward row reads every target measurable event $E\in\mathcal{A}_T$ by
$$
\mu_X(E)\sim_R\mu_S(X^{-1}(E)).
$$
\item The source measure is readable through the source event classifier:
$$
E\simeq_S F\Longrightarrow \mu_S(E)\sim_R\mu_S(F).
$$
\item The countable $\RealUp$ sum endpoint is stable under pointwise $\RealUp$ classifier comparison:
$$
\bigl(\forall n\in\mathbb N,\, r_n\sim_R q_n\bigr)\Longrightarrow\sum_{n\in\mathbb N}r_n\sim_R\sum_{n\in\mathbb N}q_n.
$$
\end{enumerate}
Then the pushed-forward target measure classifies the displayed target countable disjoint union by the countable sum of the pushed-forward component measures:
$$
\mu_X(U_T)\sim_R\sum_{n\in\mathbb N}\mu_X(B_n).
$$
\end{theorem}
\leanchecked{BEDC.Derived.DistributionUp.DistributionPushforward\_countable\_disjoint\_sigma\_additivity}
\begin{proof}
Invert the measurable-preimage row at $U_T$ and at each $B_n$. This gives $A_U\in\mathcal{A}_S$, gives $A_n\in\mathcal{A}_S$ for every $n$, and gives the readbacks
$$
s\in A_U\Longleftrightarrow X(s)\in U_T
$$
and
$$
s\in A_n\Longleftrightarrow X(s)\in B_n.
$$
The source countable-union row supplies $U_S\in\mathcal{A}_S$, and the countable preimage-union bridge gives
$$
A_U\simeq_S U_S.
$$
The source sequence $(A_n)_{n\in\mathbb N}$ is pairwise disjoint. Indeed, let $i\neq j$ and suppose that a carried source point $s$ has witnesses $s\in A_i$ and $s\in A_j$. The preimage readback for $B_i$ gives $X(s)\in B_i$, and the preimage readback for $B_j$ gives $X(s)\in B_j$. This contradicts the displayed target disjointness of $B_i$ and $B_j$.
Hence the hypotheses of the source sigma-additivity row apply to $(A_n)$ and $U_S$, giving
$$
\mu_S(U_S)\sim_R\sum_{n\in\mathbb N}\mu_S(A_n).
$$
Apply the pushforward row to the target union $U_T$. Since $U_T\in\mathcal{A}_T$,
$$
\mu_X(U_T)\sim_R\mu_S(A_U).
$$
Read the source event comparison $A_U\simeq_S U_S$ through the source measure classifier-transport row. This gives
$$
\mu_S(A_U)\sim_R\mu_S(U_S).
$$
By transitivity of the $\RealUp$ classifier,
$$
\mu_X(U_T)\sim_R\mu_S(U_S).
$$
Composing this comparison with the source sigma-additivity comparison yields
$$
\mu_X(U_T)\sim_R\sum_{n\in\mathbb N}\mu_S(A_n).
$$
For each $n\in\mathbb N$, the pushforward row applied to the target measurable event $B_n$ gives
$$
\mu_X(B_n)\sim_R\mu_S(A_n).
$$
By symmetry of the $\RealUp$ classifier,
$$
\mu_S(A_n)\sim_R\mu_X(B_n).
$$
The countable-sum stability row applied to the pointwise comparisons gives
$$
\sum_{n\in\mathbb N}\mu_S(A_n)\sim_R\sum_{n\in\mathbb N}\mu_X(B_n).
$$
A final transitivity step for the $\RealUp$ classifier gives
$$
\mu_X(U_T)\sim_R\sum_{n\in\mathbb N}\mu_X(B_n).
$$
\end{proof}

\begin{theorem}[Distribution pushforward continuity from below]\label{thm:distribution-pushforward-continuity-from-below}
Let $X:S\to T$ be a carried $\RandomVarUp$ map as in $\autoref{sec:randomvar-carrier}$, from a $\ProbSpaceUp$ source as in $\autoref{sec:probspace-carrier}$ to a target $\MeasureUp$ carrier as in $\autoref{sec:measure-carrier}$. Write $\mathcal A_S$ and $\mathcal A_T$ for the source and target measurable-event families, $\mu_S$ for the source measure, $\mu_X$ for the $\DistributionUp$ pushforward measure as in $\autoref{sec:distribution-carrier}$, and $x\sim_R y$ for the $\RealUp$ classifier of $\autoref{def:real-classifier-specification}$. Let $(B_n)_{n\in\mathbb N}$ be target events with $B_n\in\mathcal A_T$ for every $n$, and suppose the target readback supplies both
$$
B_n\subseteq B_{n+1}\quad\text{for every }n\in\mathbb N,
$$
and
$$
B:=\bigcup_{n\in\mathbb N} B_n
\qquad\text{with}
\qquad
t\in B\Longleftrightarrow \exists n\in\mathbb N,\ t\in B_n.
$$
For each $n$, let $A_n:=X^{-1}(B_n)$ and let $A_B:=X^{-1}(B)$. Suppose the hypotheses of $\autoref{thm:randomvar-countable-preimage-union-exactness}$ and $\autoref{thm:measure-continuity-from-below-package}$ are satisfied for this data, including source measurability and the source countable-union row
$$
U_S:=\bigvee\nolimits^S_{n\in\mathbb N} A_n\in\mathcal A_S
\qquad\text{and}\qquad
A_B\sim_{\mathcal A_S}U_S.
$$
Then
$$
\mu_X(B)\sim_R\lim_{n\to\infty}\mu_X(B_n).
$$
\end{theorem}
\leanchecked{BEDC.Derived.DistributionUp.DistributionPushforward\_continuity\_from\_below\_prefix\_readback}
\begin{proof}
For each $n$, apply the measurable-preimage readback to get
$$
A_n\in\mathcal A_S,\qquad A_B\in\mathcal A_S.
$$
From the target inclusion row $B_n\subseteq B_{n+1}$ and the preimage readbacks
$$
s\in A_n\Longleftrightarrow X(s)\in B_n,
\qquad s\in A_{n+1}\Longleftrightarrow X(s)\in B_{n+1},
$$
it follows pointwise that $A_n\subseteq A_{n+1}$. Hence $(A_n)_{n\in\mathbb N}$ is an increasing source measurable sequence.

Apply $\autoref{thm:measure-continuity-from-below-package}$ to $(A_n)$ and its source union $U_S$. This provides a source disjoint layer chain with prefix readbacks
$$
\Sigma_\mu(D_0,\dots,D_n)\sim_R\mu_S(A_n)\quad\text{for every }n\in\mathbb N,
$$
and the endpoint comparison
$$
\mu_S(U_S)\sim_R\sum_{n\in\mathbb N}\mu_S(D_n).
$$

By the $\DistributionUp$ pushforward row of $\autoref{sec:distribution-certificate}$,
$$
\mu_X(B)\sim_R\mu_S(A_B),
$$
and for each $n$
$$
\mu_X(B_n)\sim_R\mu_S(A_n).
$$
Composing the first comparison with $A_B\sim_{\mathcal A_S}U_S$ through source measure transport gives
$$
\mu_X(B)\sim_R\mu_S(U_S).
$$
Combining this with the continuity-from-below endpoint row yields
$$
\mu_X(B)\sim_R\sum_{n\in\mathbb N}\mu_S(D_n).
$$
Together with the prefix readbacks, each finite prefix is also transport-equivalent to the pushed-forward target prefix value:
$$
\Sigma_\mu(D_0,\dots,D_n)\sim_R\mu_X(B_n)\quad\text{for every }n\in\mathbb N.
$$
By the continuity-from-below readback convention for this displayed chain, the endpoint on the right is exactly the monotone limit of the target pushed-forward prefix sequence, hence
$$
\mu_X(B)\sim_R\lim_{n\to\infty}\mu_X(B_n).
$$
\end{proof}

\begin{theorem}[Distribution pushforward probability package]\label{thm:distribution-pushforward-borel-probability-package}
Let $X:S\to T$ be a carried $\RandomVarUp$ map as in $\autoref{sec:randomvar-carrier}$, from a $\ProbSpaceUp$ source as in $\autoref{sec:probspace-carrier}$ to a target $\MeasureUp$ carrier as in $\autoref{sec:measure-carrier}$. \end{theorem}
\begin{proof}
Under the simultaneous hypotheses, the unit row is the row supplied by $\autoref{thm:distribution-pushforward-total-mass-unit}$, and the empty-event row is supplied by $\autoref{thm:distribution-empty-target-event-zero-mass}$. Nonnegativity of every target measurable value is the component row from $\autoref{thm:distribution-nonnegative-value-inheritance}$.

The order-sensitive rows are also part of the assumed component surface. Monotonicity is the row of $\autoref{thm:distribution-pushforward-monotone-events}$, finite disjoint binary additivity is the row of $\autoref{thm:distribution-pushforward-finite-disjoint-union-additivity}$, and relative-difference additivity is the row of $\autoref{thm:distribution-pushforward-relative-difference-additivity}$. The complement-mass row is supplied by $\autoref{thm:distribution-complement-mass-decomposition}$.

For countable structure, the assumed surface includes $\autoref{thm:distribution-pushforward-countable-disjoint-sigma-additivity}$ for disjoint target families and $\autoref{thm:distribution-pushforward-continuity-from-below}$ for increasing target families. These nine rows have the same carried map $X$, the same target event family $\mathcal A_T$, and the same displayed pushforward row $\mu_X$, so collecting the assumed component rows gives the announced probability-measure package on the target event surface.
\end{proof}
