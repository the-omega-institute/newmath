\begin{theorem}[Singleton evaluation append-input classifier readback]
\label{thm:linearmap-singleton-eval-append-input-classifier}
For histories $f,x,y,h$, classifying the singleton evaluation of an appended
input against $h$ is exactly the singleton carrier condition on $h$:
$$
\mathsf{LinearMapSingletonClassifier}
  (\mathsf{LinearMapSingletonEval}(f,x\cdot y),h)
\Longleftrightarrow
\mathsf{LinearMapSingletonCarrier}(h).
$$
\end{theorem}
\begin{proof}
The singleton evaluator always returns $\emp$. In the forward direction, the
classifier already contains the carrier witness for its right endpoint. In the
reverse direction, the carrier witness for $h$ combines with reflexivity at
$\emp$ and symmetry of the displayed carrier equality to form the restricted
singleton classifier.
\end{proof}
\leanchecked{BEDC.Derived.LinearMapUp.LinearMapSingletonEval\_append\_input\_classifier\_iff}

\label{thm:linearmap-singleton-eval-append-context-classifier}
For histories $p,f,x,h:\mathsf{BHist}$, classifying a singleton evaluation
after an appended context is exactly the pair of singleton carrier conditions
on the context and the target:
$$
\mathsf{LinearMapSingletonClassifier}
  (\mathsf{append}(p,\mathsf{LinearMapSingletonEval}(f,x)),h)
\Longleftrightarrow
\mathsf{LinearMapSingletonCarrier}(p)\land
\mathsf{LinearMapSingletonCarrier}(h).

\end{theorem}
\begin{proof}

carrier field classifies $\mathsf{append}(p,\emp)$ with $\emp$, and
empty-append inversion exposes $p\hsame\emp$; the right classifier carrier
gives $h\hsame\emp$.

Conversely, $p\hsame\emp$ combines with reflexivity at the evaluator result to
classify $\mathsf{append}(p,\mathsf{LinearMapSingletonEval}(f,x))$ with
$\emp$. The target carrier for $h$ and symmetry on that carrier witness then
assemble the restricted singleton classifier.
\end{proof}

\label{thm:linearmap-singleton-eval-continuation-classifier}

$$
\ContR(\mathsf{LinearMapSingletonEval}(f,x),y,r),
$$
then classifying the continuation result $r$ against $h$ is equivalent to
the two exposed endpoints $y$ and $h$ lying in the singleton carrier:
$$
\mathsf{LinearMapSingletonClassifier}(r,h)
\Longleftrightarrow
\mathsf{LinearMapSingletonCarrier}(y)\land
\mathsf{LinearMapSingletonCarrier}(h).
$$
\end{theorem}
\leanchecked{BEDC.Derived.LinearMapUp.LinearMapSingletonEval\_continuation\_classifier\_iff}
\begin{proof}
The singleton evaluator returns $\emp$, so the continuation ledger displays
$r$ as $\mathsf{append}(\emp,y)$. In the forward direction, the classifier
field $r\hsame\emp$ and the left-unit append equation identify $y$ with
$\emp$, while the right classifier carrier gives the singleton witness for
$h$. Conversely, singleton witnesses for $y$ and $h$ identify $r$ with
$\emp$ through the same continuation ledger and left-unit append equation,
and the two singleton carrier witnesses assemble the restricted classifier.
\end{proof}
