\section{The certificate}

\begin{definition}[LinearMap source specification]
\label{def:linearmap-source-specification}
The source is a pair of vector spaces over the same certified field.
\end{definition}

\begin{definition}[LinearMap pattern specification]
\label{def:linearmap-pattern-specification}
The pattern is function-with-law formation: map data together with linearity witnesses.
\end{definition}

\begin{definition}[LinearMap classifier specification]
\label{def:linearmap-classifier-specification}
The classifier relates two linear maps when their outputs are codomain-classified the same for every input witness.
\end{definition}

\begin{definition}[LinearMap stability certificate]
\label{def:linearmap-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item pointwise classifier reflexivity
  \item pointwise classifier transitivity
  \item preservation of zero derived from linearity
  \item additivity preservation
  \item scalar-compatibility preservation
  \item closure under identity and composition
\end{enumerate}
\end{definition}

\begin{definition}[LinearMap ledger policy]
\label{def:linearmap-ledger-policy}
Record the function identifier, its linearity witnesses, and codomain ledgers for pointwise comparisons.
\end{definition}

\begin{definition}[\(\LinearMapUp\) certificate obligations]
\label{def:linearmap-certificate-obligations}
The carrier, source, pattern, classifier, stability, and ledger data listed above are the obligations required for the interface name \(\LinearMapUp\).
\end{definition}

\paragraph{Primitive boundary.}
\(\LinearMapUp\) is not a BEDC primitive. It is admitted only when those obligations are supplied.

\begin{remark}[Zero-axiom encoding note]
\label{rem:linearmap-zero-axiom-note}
Avoids funext by using pointwise classifier sameness instead of equality of functions.
\end{remark}

\begin{theorem}[LinearMap terminal standard bridge]
\label{thm:linearmap-terminal-standard-bridge}
The singleton empty-history public map-code surface of
\autoref{def:singleton-empty-history-linearmap-instance} and
\autoref{thm:singleton-empty-history-linearmap-public-code-exactness}
interprets in the terminal one-object, one-arrow linear-hom space. Every
carried public code maps to the unique hom, evaluation sends the unique
carried vector to itself, composition sends the unique composable hom pair to
the unique hom, and the return interpretation is classified by
$\lambda_{\emp}$ and $\rho_{\emp}$.
\end{theorem}
\begin{proof}
The public-code exactness theorem identifies every carried public map code
with the empty-history identity code under $\lambda_{\emp}$, and it gives the
same collapse for the zero code, the composed code, and the evaluation
readback. Thus the identity, zero, composition, and evaluation rows all land in
the same singleton code class.

The terminal standard object has one object endpoint and one hom endpoint.
After the public surface has collapsed to the empty-history code class, both
directions of the bridge are forced: a carried public code can only interpret
as the unique terminal hom, and the unique terminal hom can only be represented
by the carried empty-history public code.

For vectors, the singleton carrier supplies only the empty endpoint, so
evaluation of the unique hom at the unique carried vector returns that vector.
The map-code comparison is the $\lambda_{\emp}$ classification already supplied
by exactness, and the vector readback is the singleton $\rho_{\emp}$
classification. No host function extensionality is used; the argument is only
constructor collapse and carrier-aware classifier transport on the singleton
surface.
\end{proof}

\begin{theorem}[LinearMap pointwise classifier equivalence]
\label{thm:linearmap-pointwise-classifier-equivalence}
Let $F$ carry a $\FieldUp(F)$ certificate, and let $V$ and $W$ carry $\VecSpaceUp(F,V)$ and $\VecSpaceUp(F,W)$ certificates. Write $C_V$ and $C_W$ for source and codomain carrier predicates, and write $\sim_W$ for a codomain classifier supplied by $\NameCert(C_W,\sim_W)$. Let $\mathsf{Lin}_{V,W}(f)$ be the carried $\LinearMapUp(V,W)$ map-code predicate from \autoref{def:linearmap-carrier}; in particular, a witness of $\mathsf{Lin}_{V,W}(f)$ includes output closure $C_V(x)\Longrightarrow C_W(f(x))$. Define the pointwise $\LinearMapUp$ classifier by
$$
f\sim_{\LinearMapUp(V,W)}g
:\Longleftrightarrow
\mathsf{Lin}_{V,W}(f)\land\mathsf{Lin}_{V,W}(g)\land
\forall x,\; C_V(x)\Longrightarrow f(x)\sim_W g(x).
$$
Then $\sim_{\LinearMapUp(V,W)}$ is reflexive, symmetric, and transitive on carried linear-map codes:
$$
\begin{aligned}
&\mathsf{Lin}_{V,W}(f)\Longrightarrow f\sim_{\LinearMapUp(V,W)}f,\\
&f\sim_{\LinearMapUp(V,W)}g\Longrightarrow g\sim_{\LinearMapUp(V,W)}f,\\
&f\sim_{\LinearMapUp(V,W)}g\land g\sim_{\LinearMapUp(V,W)}h
\Longrightarrow f\sim_{\LinearMapUp(V,W)}h.
\end{aligned}
$$
\end{theorem}
\begin{proof}
By \autoref{def:lean-carrier-naming-certificate}, the codomain $\NameCert(C_W,\sim_W)$ supplies reflexivity for carried codomain outputs, symmetry for $\sim_W$, and transitivity for composable $\sim_W$ witnesses. By \autoref{def:linearmap-classifier-specification}, the $\LinearMapUp$ classifier is the pointwise codomain comparison over all carried source inputs, together with the two carried map-code endpoints.

For reflexivity, assume $\mathsf{Lin}_{V,W}(f)$ and let $x$ be a carried source input. Output closure in the linear-map carrier gives $C_W(f(x))$. Codomain reflexivity gives $f(x)\sim_W f(x)$. The two endpoint carrier fields are both the given $\mathsf{Lin}_{V,W}(f)$ witness, so the pointwise classifier relates $f$ to itself.

For symmetry, assume $f\sim_{\LinearMapUp(V,W)}g$. Its endpoint fields give $\mathsf{Lin}_{V,W}(f)$ and $\mathsf{Lin}_{V,W}(g)$. For each carried $x$, the pointwise field gives $f(x)\sim_W g(x)$; codomain symmetry gives $g(x)\sim_W f(x)$. Swapping the endpoint carrier fields and using these reversed pointwise comparisons gives $g\sim_{\LinearMapUp(V,W)}f$.

For transitivity, assume $f\sim_{\LinearMapUp(V,W)}g$ and $g\sim_{\LinearMapUp(V,W)}h$. The first endpoint field of the first witness carries $f$, and the second endpoint field of the second witness carries $h$. For each carried $x$, the two pointwise fields give
$$
f(x)\sim_W g(x),
\qquad
g(x)\sim_W h(x).
$$
Codomain transitivity gives $f(x)\sim_W h(x)$. These pointwise composites, together with the carried endpoints $f$ and $h$, form $f\sim_{\LinearMapUp(V,W)}h$. Thus the classifier attached to \autoref{def:linearmap-classifier-specification} satisfies the reflexive, symmetric, and transitive rows required by the stability and certificate obligations in \autoref{def:linearmap-stability-certificate} and \autoref{def:linearmap-certificate-obligations}.
\end{proof}
