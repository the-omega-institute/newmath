\begin{theorem}[Singleton evaluation continuation visible target exclusion]
\label{thm:linearmap-singleton-eval-continuation-visible-target-exclusion}
For histories $f,x,y,r,p:\mathsf{BHist}$, if
$$
\ContR(\mathsf{LinearMapSingletonEval}(f,x),y,r),
$$
then both visible target classifier claims are impossible:
$$
\mathsf{LinearMapSingletonClassifier}(r,\Ezero(p))\Longrightarrow\bot,
\qquad
\mathsf{LinearMapSingletonClassifier}(r,\Eone(p))\Longrightarrow\bot.
$$
The ten-field fit is: traditional target terminal linear-map continuation
result exclusion; BEDC source $\mathsf{BHist}$, $\ContR$, and the singleton
$\LinearMapUp$ evaluator; carrier $\mathsf{LinearMapSingletonCarrier}$;
classifier $\mathsf{LinearMapSingletonClassifier}$; stability from
continuation classifier readback and constructor no-confusion; ledger
continuation result endpoint; core theorem this block; exactness visible-target
impossibility; standard bridge terminal linear hom; status
$\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:linearmap-singleton-eval-continuation-classifier} to
the displayed continuation with target $\Ezero(p)$. A classifier witness for
$\Ezero(p)$ then forces
$\mathsf{LinearMapSingletonCarrier}(\Ezero(p))$.

Apply the same continuation classifier readback with target $\Eone(p)$. A
classifier witness for $\Eone(p)$ forces
$\mathsf{LinearMapSingletonCarrier}(\Eone(p))$.

Unfolding the singleton carrier makes each witness a history-sameness proof
from the visible endpoint to $\emp$. Constructor no-confusion for $\Ezero$ and
$\Eone$ contradicts those histories being empty, so both displayed classifier
claims imply $\bot$.
\end{proof}
\leanchecked{BEDC.Derived.LinearMapUp.LinearMapSingletonEval\_continuation\_visible\_target\_classifier\_absurd}

\begin{theorem}[Singleton evaluation continuation visible input exclusion]
\label{thm:linearmap-singleton-eval-continuation-visible-input-exclusion}
For histories $f,x,p,r,h:\mathsf{BHist}$, each visible continuation input
prevents a singleton-classified continuation result:
$$
\ContR(\mathsf{LinearMapSingletonEval}(f,x),\Ezero(p),r)
\Longrightarrow
\bigl(\mathsf{LinearMapSingletonClassifier}(r,h)\Longrightarrow\bot\bigr),
$$
and
$$
\ContR(\mathsf{LinearMapSingletonEval}(f,x),\Eone(p),r)
\Longrightarrow
\bigl(\mathsf{LinearMapSingletonClassifier}(r,h)\Longrightarrow\bot\bigr).
$$
The ten-field fit is: traditional target terminal linear-map continuation
input exclusion; BEDC source $\mathsf{BHist}$, $\ContR$, and the singleton
evaluator; carrier $\mathsf{LinearMapSingletonCarrier}$; classifier
$\mathsf{LinearMapSingletonClassifier}$; stability from continuation
classifier readback plus constructor separation; ledger continuation input
endpoint; core theorem this block; exactness visible-input impossibility;
standard bridge terminal linear hom; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:linearmap-singleton-eval-continuation-classifier} to
the first displayed continuation with continuation input $\Ezero(p)$. A
classifier witness for $r$ against $h$ then forces
$\mathsf{LinearMapSingletonCarrier}(\Ezero(p))$.

Apply the same continuation classifier readback to the second displayed
continuation with continuation input $\Eone(p)$. A classifier witness for $r$
against $h$ forces $\mathsf{LinearMapSingletonCarrier}(\Eone(p))$.

Unfolding the singleton carrier turns each carrier witness into a
history-sameness proof from the visible continuation input to $\emp$.
Constructor no-confusion for $\Ezero$ and $\Eone$ contradicts those histories
being empty, so each visible continuation input case implies $\bot$.
\end{proof}
\leanchecked{BEDC.Derived.LinearMapUp.LinearMapSingletonEval\_continuation\_visible\_input\_classifier\_absurd}
\leanchecked{BEDC.Derived.LinearMapUp.LinearMapSingletonEval\_continuation\_visible\_input\_absurd}
