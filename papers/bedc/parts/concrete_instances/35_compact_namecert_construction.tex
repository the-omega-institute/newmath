\chapter{A Concrete Naming Certificate for \(\CompactUp\)}
\label{ch:concrete-instances-compact-namecert}

The compactness interface \(\CompactUp\) has certificate obligations for subsets of metric spaces from \autoref{ch:concrete-instances-metric-namecert}. The BEDC-licensed carrier is a subset predicate equipped with explicit locatedness witnesses and finite-net witnesses: locatedness gives, for any point and rational \(\epsilon\), a decision whether the point is within \(\epsilon\) of the subset; finite-net data gives, for any rational \(\epsilon\), a finite \(\epsilon\)-net.

\section{Carrier definition}

\begin{definition}[Compact carrier]
\label{def:compact-carrier}
Use a metric-space subset predicate equipped with locatedness and total-boundedness witness functions.
\end{definition}
\leandef{BEDC.Derived.CompactUp.CompactWitnessCarrier}

\begin{theorem}[Compact witness carrier finite extension closure]
\label{thm:compact-witness-carrier-finite-extension-closed}
If a compact witness carrier records unary subset, locatedness, and finite-net
histories through locatedness and compactness ledgers, then appending a unary
finite-net refinement preserves the same witness shape.
\end{theorem}
\begin{proof}
The existing carrier supplies unary histories for the subset, locatedness, and
finite-net ledgers together with continuations through the located and compact
witnesses. Unary continuation closure gives the refined finite-net carrier.
Continuation associativity combines the old compactness ledger with the
refinement ledger, and determinism identifies the associated witness with the
named refined finite-net history.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactWitnessCarrier\_finite\_extension\_closed}

\begin{definition}[Compact finite refinement chain]
\label{def:compact-finite-refinement-chain}
The compact finite refinement chain relation is written

$$
\mathsf{CompactFiniteRefinementChain}(finite,compact,finalFinite,finalCompact).
$$

It is the inductive BEDC relation on paired finite-net and compactness ledgers
generated by the following constructors:
\begin{enumerate}
  \item the base constructor relates $(finite,compact)$ to itself;
  \item the step constructor says that, from a prior chain to $(f,c)$,
  $\mathsf{UnaryHistory}(e)$, $\Cont(f,e,f')$, and $\Cont(c,e,c')$,
  one obtains a chain to $(f',c')$.
\end{enumerate}
\end{definition}

\begin{theorem}[Compact finite refinement chain transitivity]
\label{thm:compact-finite-refinement-chain-transitivity}
For finite-net ledgers $finite$, $midFinite$, and $finalFinite$ and compactness
ledgers $compact$, $midCompact$, and $finalCompact$, if

$$
\mathsf{CompactFiniteRefinementChain}(finite,compact,midFinite,midCompact)
$$

and

$$
\mathsf{CompactFiniteRefinementChain}(midFinite,midCompact,finalFinite,finalCompact),
$$

then

$$
\mathsf{CompactFiniteRefinementChain}(finite,compact,finalFinite,finalCompact).
$$
\end{theorem}
\begin{proof}
Induct on the second compact finite refinement chain.

In the base case, the second chain ends at its starting pair
$(midFinite,midCompact)$. The desired chain is exactly the first chain.

In the step case, the second chain is obtained from a prior pair $(f,c)$ by a
unary refinement history $e$ together with witnesses $\mathsf{UnaryHistory}(e)$,
$\Cont(f,e,f')$, and $\Cont(c,e,c')$. Apply the induction hypothesis to compose
the first chain with the prior second-chain segment, producing a chain from
$(finite,compact)$ to $(f,c)$.

Apply the step constructor of
Definition~\ref{def:compact-finite-refinement-chain} to that composed chain and
the displayed unary-history and continuation witnesses. This gives the required
chain from $(finite,compact)$ to $(f',c')$, which is the final pair in the step
case.
\end{proof}

\begin{theorem}[Compact witness carrier finite refinement chain closure]
\label{thm:compact-witness-carrier-finite-refinement-chain-closed}
If

$$
\mathsf{CompactWitnessCarrier}(subset,located,finite,intermediate,compact)
$$

and

$$
\mathsf{CompactFiniteRefinementChain}(finite,compact,finalFinite,finalCompact),
$$

then

$$
\mathsf{CompactWitnessCarrier}(subset,located,finalFinite,intermediate,finalCompact).
$$

\end{theorem}
\begin{proof}
Induct on the given compact finite refinement chain.

In the base case, the final finite-net ledger and final compactness ledger are
the original ledgers $finite$ and $compact$. The desired carrier is exactly the
given compact witness carrier.

In the step case, the induction hypothesis gives the carrier at the prior pair
$(f,c)$. Apply
Theorem~\ref{thm:compact-witness-carrier-finite-extension-closed} to that
carrier, the unary refinement history $e$, and the continuations
$\Cont(f,e,f')$ and $\Cont(c,e,c')$. The result is the required compact witness
carrier at $(f',c')$.
\end{proof}

\begin{theorem}[Compact witness carrier iterated finite refinement closure]
\label{thm:compact-witness-carrier-iterated-finite-refinement-closed}
For histories $subset$, $located$, $finite$, $extra_0$, $extra_1$,
$intermediate$, $compact$, $finite_0$, $compact_0$, $finite_1$, and
$compact_1$, assume:
\begin{enumerate}
  \item
  $$
  \mathsf{CompactWitnessCarrier}(subset,located,finite,intermediate,compact);
  $$
  \item $\mathsf{UnaryHistory}(extra_0)$ and
  $\mathsf{UnaryHistory}(extra_1)$;
  \item $\Cont(finite,extra_0,finite_0)$ and
  $\Cont(compact,extra_0,compact_0)$;
  \item $\Cont(finite_0,extra_1,finite_1)$ and
  $\Cont(compact_0,extra_1,compact_1)$.
\end{enumerate}
Then
$$
\mathsf{CompactWitnessCarrier}(subset,located,finite_1,intermediate,compact_1)
$$
\end{theorem}
\begin{proof}
First apply Theorem~\ref{thm:compact-witness-carrier-finite-extension-closed} to
the carrier with finite-net ledger $finite$ and compactness ledger $compact$,
using $extra_0$ as the unary refinement history. This produces the intermediate
carrier.

$$
\mathsf{CompactWitnessCarrier}(subset,located,finite_0,intermediate,compact_0)
$$

Apply the same theorem again to that intermediate carrier, now using $extra_1$
as the unary refinement history and the continuations from $finite_0$ to
$finite_1$ and from $compact_0$ to $compact_1$. The resulting carrier is:

$$
\mathsf{CompactWitnessCarrier}(subset,located,finite_1,intermediate,compact_1)
$$
\end{proof}

\begin{theorem}[Compact witness carrier located extension closure]
\label{thm:compact-witness-carrier-located-extension-closed}
If a compact witness carrier records a subset through locatedness and finite-net
histories, then appending a unary locatedness refinement preserves the compact
witness shape after reassociating the subset ledger.
\end{theorem}
\begin{proof}
The carrier supplies unary subset, locatedness, and finite-net histories,
together with a continuation from the subset through the locatedness witness to
the intermediate witness. Unary continuation closure makes the refined
locatedness history unary. Continuation associativity shifts the subset ledger
across the locatedness refinement, and determinism identifies the associated
locatedness witness with the displayed refinement. The final finite-net ledger
then gives the compact witness endpoint.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactWitnessCarrier\_located\_extension\_closed}

\section{The certificate}

\begin{definition}[Compact source specification]
\label{def:compact-source-specification}
The source is a certified metric space plus a subset predicate over its carrier.
\end{definition}

\begin{definition}[Compact pattern specification]
\label{def:compact-pattern-specification}
The pattern is compact-subset formation by attaching locatedness and finite-net data for each rational precision.
\end{definition}

\begin{definition}[Compact classifier specification]
\label{def:compact-classifier-specification}
The classifier relates compact witnesses when their membership predicates agree by explicit bidirectional evidence under carrier sameness.
\end{definition}

\begin{definition}[Compact stability certificate]
\label{def:compact-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item membership stability under carrier classifier
  \item locatedness witness validity
  \item total-bounded finite-net validity
  \item monotonicity of precision witnesses where required
  \item stability under compact-witness classifier
\end{enumerate}
\end{definition}

\begin{theorem}[Compact net witness prefix closure]
\label{thm:compact-net-witness-prefix-closed}
For unary histories $p,c,\epsilon,n$, if $n$ is a compact net witness obtained
by continuing center $c$ with precision $\epsilon$, then prefixing the center
and net by $p$ preserves the same precision witness:
$$
\mathsf{CompactNetWitness}(c,\epsilon,n)
\Longrightarrow
\mathsf{CompactNetWitness}(p \cdot c,\epsilon,p \cdot n).
$$
\end{theorem}
\begin{proof}
Unpack the witness into unary center, precision, and net histories together
with the continuation ledger $n=c\cdot\epsilon$. Unary closure gives unary
histories for $p\cdot c$ and $p\cdot n$. The prefixed continuation ledger is
the associativity equality $(p\cdot c)\cdot\epsilon=p\cdot(c\cdot\epsilon)$.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactNetWitness\_prefix\_closed}

\begin{theorem}[Compact net witness prefix exactness]
\label{thm:compact-net-witness-prefix-iff}
For histories $p,c,\epsilon,n$, a prefixed compact-net witness is equivalent
to unary prefix evidence together with the unprefixed witness:
$$
\mathsf{CompactNetWitness}(p \cdot c,\epsilon,p \cdot n)
\Longleftrightarrow
\mathsf{UnaryHistory}(p)\wedge
\mathsf{CompactNetWitness}(c,\epsilon,n).
$$
\end{theorem}
\begin{proof}
For the forward direction, unpack the prefixed witness. Unary append
factorization recovers the unary prefix and the unary center and net histories,
while prefix cancellation on the continuation ledger recovers
$\mathsf{Cont}(c,\epsilon,n)$. For the reverse direction, apply prefix closure
to the unary prefix and the original compact-net witness.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactNetWitness\_prefix\_iff}

\begin{definition}[Compact ledger policy]
\label{def:compact-ledger-policy}
Record the subset predicate, locatedness decisions, and finite net ledgers.
\end{definition}

\begin{definition}[\(\CompactUp\) certificate obligations]
\label{def:compact-certificate-obligations}
The carrier, source, pattern, classifier, stability, and ledger data listed above are the obligations required for the interface name \(\CompactUp\).
\end{definition}

\paragraph{Primitive boundary.}
\(\CompactUp\) is not a BEDC primitive. It is admitted only when those obligations are supplied.

\begin{remark}[Zero-axiom encoding note]
\label{rem:compact-zero-axiom-note}
Open-cover compactness is not the exported carrier for \(\CompactUp\); the exported carrier is locatedness plus explicit finite-net data.
\end{remark}
