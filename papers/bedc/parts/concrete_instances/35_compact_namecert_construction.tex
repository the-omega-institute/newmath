\chapter{A Concrete Naming Certificate for \(\CompactUp\)}
\label{ch:concrete-instances-compact-namecert}

The compactness interface \(\CompactUp\) has certificate obligations for subsets of metric spaces from \autoref{ch:concrete-instances-metric-namecert}. The BEDC-licensed carrier is a subset predicate equipped with explicit locatedness witnesses and finite-net witnesses: locatedness gives, for any point and rational \(\epsilon\), a decision whether the point is within \(\epsilon\) of the subset; finite-net data gives, for any rational \(\epsilon\), a finite \(\epsilon\)-net.

\section{Carrier definition}

\begin{definition}[Compact carrier]
\label{def:compact-carrier}
Use a metric-space subset predicate equipped with locatedness and total-boundedness witness functions.
\end{definition}
\leandef{BEDC.Derived.CompactUp.CompactWitnessCarrier}

\begin{theorem}[Compact witness carrier finite extension closure]
\label{thm:compact-witness-carrier-finite-extension-closed}
If a compact witness carrier records unary subset, locatedness, and finite-net
histories through locatedness and compactness ledgers, then appending a unary
finite-net refinement preserves the same witness shape.
\end{theorem}
\begin{proof}
The existing carrier supplies unary histories for the subset, locatedness, and
finite-net ledgers together with continuations through the located and compact
witnesses. Unary continuation closure gives the refined finite-net carrier.
Continuation associativity combines the old compactness ledger with the
refinement ledger, and determinism identifies the associated witness with the
named refined finite-net history.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactWitnessCarrier\_finite\_extension\_closed}

\begin{definition}[Compact finite refinement chain]
\label{def:compact-finite-refinement-chain}
The compact finite refinement chain relation is written

$$
\mathsf{CompactFiniteRefinementChain}(finite,compact,finalFinite,finalCompact).
$$

It is the inductive BEDC relation on paired finite-net and compactness ledgers
generated by the following constructors:
\begin{enumerate}
  \item the base constructor relates $(finite,compact)$ to itself;
  \item the step constructor says that, from a prior chain to $(f,c)$,
  $\mathsf{UnaryHistory}(e)$, $\Cont(f,e,f')$, and $\Cont(c,e,c')$,
  one obtains a chain to $(f',c')$.
\end{enumerate}
\end{definition}
\leandef{BEDC.Derived.CompactUp.CompactFiniteRefinementChain}

\begin{theorem}[Compact finite refinement chain transitivity]
\label{thm:compact-finite-refinement-chain-transitivity}
For finite-net ledgers $finite$, $midFinite$, and $finalFinite$ and compactness
ledgers $compact$, $midCompact$, and $finalCompact$, if

$$
\mathsf{CompactFiniteRefinementChain}(finite,compact,midFinite,midCompact)
$$

and

$$
\mathsf{CompactFiniteRefinementChain}(midFinite,midCompact,finalFinite,finalCompact),
$$

then

$$
\mathsf{CompactFiniteRefinementChain}(finite,compact,finalFinite,finalCompact).
$$
\end{theorem}
\begin{proof}
Induct on the second compact finite refinement chain.

In the base case, the second chain ends at its starting pair
$(midFinite,midCompact)$. The desired chain is exactly the first chain.

In the step case, the second chain is obtained from a prior pair $(f,c)$ by a
unary refinement history $e$ together with witnesses $\mathsf{UnaryHistory}(e)$,
$\Cont(f,e,f')$, and $\Cont(c,e,c')$. Apply the induction hypothesis to compose
the first chain with the prior second-chain segment, producing a chain from
$(finite,compact)$ to $(f,c)$.

Apply the step constructor of
Definition~\ref{def:compact-finite-refinement-chain} to that composed chain and
the displayed unary-history and continuation witnesses. This gives the required
chain from $(finite,compact)$ to $(f',c')$, which is the final pair in the step
case.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactFiniteRefinementChain\_transitivity}

\begin{theorem}[Compact finite refinement chain prefix closure]
\label{thm:compact-finite-refinement-chain-prefix-closed}
For a prefix ledger $p$ and finite-net and compactness ledgers
$finite,compact,finalFinite,finalCompact$, if

$$
\mathsf{CompactFiniteRefinementChain}(finite,compact,finalFinite,finalCompact),
$$

then

$$
\mathsf{CompactFiniteRefinementChain}(p\cdot finite,p\cdot compact,
p\cdot finalFinite,p\cdot finalCompact).
$$
\end{theorem}
\begin{proof}
Induct on the compact finite refinement chain. The base constructor is stable
under prefixing because both endpoints remain identical pairs. In the step
case, apply the induction hypothesis to the prior chain and rebuild the step
constructor. The two continuation witnesses are obtained by applying the prefix
operation to the original continuation equalities and reassociating append.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactFiniteRefinementChain\_prefix\_closed}

\begin{theorem}[Compact finite refinement chain final endpoints unary]
\label{thm:compact-finite-refinement-chain-final-endpoints-unary}
If

$$
\mathsf{CompactFiniteRefinementChain}(finite,compact,finalFinite,finalCompact),
$$

and the starting finite-net and compactness ledgers are unary histories, then
the final finite-net and compactness ledgers are unary histories:

$$
\mathsf{UnaryHistory}(finite)\land\mathsf{UnaryHistory}(compact)
\Longrightarrow
\mathsf{UnaryHistory}(finalFinite)\land
\mathsf{UnaryHistory}(finalCompact).
$$
\end{theorem}
\begin{proof}
Induct on the compact finite refinement chain. In the base case, the final
endpoints are the starting endpoints, so the supplied unary-history witnesses
are the desired witnesses. In the step case, the induction hypothesis gives
unary-history witnesses for the prior finite-net and compactness ledgers. The
step constructor supplies a unary refinement history and two continuation
witnesses. Closure of unary histories under continuation gives unary histories
for both final ledgers.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactFiniteRefinementChain\_final\_endpoints\_unary}

\begin{theorem}[Compact finite refinement chain initial endpoints unary]
\label{thm:compact-finite-refinement-chain-initial-endpoints-unary}
If
$$
\mathsf{CompactFiniteRefinementChain}(finite,compact,finalFinite,finalCompact),
$$
and the final finite-net and compactness ledgers are unary histories, then the
initial finite-net and compactness ledgers are unary histories:
$$
\mathsf{UnaryHistory}(finalFinite)\land\mathsf{UnaryHistory}(finalCompact)
\Longrightarrow
\mathsf{UnaryHistory}(finite)\land\mathsf{UnaryHistory}(compact).
$$
\end{theorem}
\begin{proof}
Induct on the compact finite refinement chain. The base case is immediate. In
the step case, factor the final unary-history witnesses through the two
continuation witnesses by unary left-factor closure, then apply the induction
hypothesis to the prior chain.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactFiniteRefinementChain\_initial\_endpoints\_unary}

\begin{theorem}[Compact finite refinement chain endpoint hsame]
\label{thm:compact-finite-refinement-chain-endpoint-hsame}
If
$$
\mathsf{CompactFiniteRefinementChain}(finite,compact,finalFinite,finalCompact)
$$
and $\hsame(finite,compact)$, then $\hsame(finalFinite,finalCompact)$.
\end{theorem}
\begin{proof}
Induct on the compact finite refinement chain. The base case is the supplied
starting endpoint equality. In the step case, the induction hypothesis gives
equality of the current finite and compact endpoints; the common unary
refinement tail and the two continuation witnesses transport that equality to
the next endpoints by continuation respect for $\hsame$.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactFiniteRefinementChain\_endpoint\_hsame}

\begin{theorem}[Compact finite refinement chain endpoint hsame reflects]
\label{thm:compact-finite-refinement-chain-endpoint-hsame-reflects}
If
$$
\mathsf{CompactFiniteRefinementChain}(finite,compact,finalFinite,finalCompact)
$$
and $\hsame(finalFinite,finalCompact)$, then $\hsame(finite,compact)$.
\end{theorem}
\begin{proof}
Induct on the compact finite refinement chain. The base case is the supplied
final endpoint equality. In the step case, both endpoints were extended by the
same refinement tail. Common-suffix cancellation for the two continuation
witnesses reflects $\hsame$ from the final endpoints to the preceding endpoints,
and the induction hypothesis reflects it to the starting endpoints.
\end{proof}

\begin{theorem}[Compact finite refinement chain endpoint hsame iff]
\label{thm:compact-finite-refinement-chain-endpoint-hsame-iff}
If
$$
\mathsf{CompactFiniteRefinementChain}(finite,compact,finalFinite,finalCompact)
$$
holds, then
$$
\hsame(finalFinite,finalCompact)\Longleftrightarrow\hsame(finite,compact).
$$
\end{theorem}
\begin{proof}
Use the forward endpoint-transport theorem for one implication and the endpoint
reflection theorem for the other. The same finite refinement chain supplies the
common continuation structure needed by both directions.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactFiniteRefinementChain\_endpoint\_hsame\_iff}

\begin{theorem}[Compact finite refinement chain empty final reflects empty initial]
\label{thm:compact-finite-refinement-chain-empty-final-reflects-empty-initial}
If a compact finite refinement chain ends with both final ledgers empty, then
both initial ledgers are empty:
$$
\mathsf{CompactFiniteRefinementChain}(finite,compact,finalFinite,finalCompact)
\land \hsame(finalFinite,\emp)\land\hsame(finalCompact,\emp)
\Longrightarrow \hsame(finite,\emp)\land\hsame(compact,\emp).
$$
\end{theorem}
\begin{proof}
Induct on the refinement chain. In a step, transport the final empty equality
back across each displayed continuation, use empty-result inversion to identify
the prior ledger with $\emp$, and apply the induction hypothesis.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactFiniteRefinementChain\_empty\_final\_reflects\_empty\_initial}

\begin{theorem}[Compact finite refinement chain nonempty forward]
\label{thm:compact-finite-refinement-chain-nonempty-forward}
If a compact finite refinement chain starts from a finite-net ledger apart from
$\emp$, then its final finite-net ledger is apart from $\emp$:
$$
\mathsf{CompactFiniteRefinementChain}(finite,compact,finalFinite,finalCompact)
\land ((finite\hsame\emp)\to\bot)
\Longrightarrow
((finalFinite\hsame\emp)\to\bot).
$$
\end{theorem}
\begin{proof}
Induct on the refinement chain. The base case is the starting apartness
predicate. In a step, a collapse of the final finite ledger to $\emp$ identifies
the appended prior finite ledger and refinement tail with $\emp$. Empty-append
inversion gives a collapse of the prior finite ledger, contradicting the
induction hypothesis.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactFiniteRefinementChain\_nonempty\_forward}

\begin{theorem}[Compact finite refinement chain compact nonempty forward]
\label{thm:compact-finite-refinement-chain-compact-nonempty-forward}
If a compact finite refinement chain starts from a compact ledger apart from

$$
\mathsf{CompactFiniteRefinementChain}(finite,compact,finalFinite,finalCompact)
\land ((compact\hsame\emp)\to\bot)
\Longrightarrow
((finalCompact\hsame\emp)\to\bot).
$$
\end{theorem}
\begin{proof}
Induct on the refinement chain. The base case is the starting compact
apartness predicate. In a step, a collapse of the final compact ledger to
$\emp$ makes the appended prior compact ledger empty, and empty-append
inversion gives the contradiction by the induction hypothesis.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactFiniteRefinementChain\_compact\_nonempty\_forward}

\begin{theorem}[Compact finite refinement chain continuation witness]
\label{thm:compact-finite-refinement-chain-continuation-witness}
If
$$
\mathsf{CompactFiniteRefinementChain}(finite,compact,finalFinite,finalCompact)
$$
holds, then there is a unary refinement history $extra$ that realizes both final
endpoints as continuations from their corresponding starts:
$$
\mathsf{UnaryHistory}(extra)\land
\Cont(finite,extra,finalFinite)\land
\Cont(compact,extra,finalCompact).
$$
\end{theorem}
\begin{proof}
Induct on the finite refinement chain. The base case uses the empty history as
the common continuation tail and the right-unit continuation law on both
endpoints. In the step case, the induction hypothesis supplies a common
accumulated tail. Associate that tail with the step refinement on the finite
side and on the compact side. Continuation determinism identifies the two
associated tails, unary continuation closure proves the accumulated tail unary,
and the associated continuations give the two displayed endpoint ledgers.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactFiniteRefinementChain\_continuation\_witness}

\begin{theorem}[Compact witness carrier finite refinement chain closure]
\label{thm:compact-witness-carrier-finite-refinement-chain-closed}
If

$$
\mathsf{CompactWitnessCarrier}(subset,located,finite,intermediate,compact)
$$

and

$$
\mathsf{CompactFiniteRefinementChain}(finite,compact,finalFinite,finalCompact),
$$

then

$$
\mathsf{CompactWitnessCarrier}(subset,located,finalFinite,intermediate,finalCompact).
$$

\end{theorem}
\begin{proof}
Induct on the given compact finite refinement chain.

In the base case, the final finite-net ledger and final compactness ledger are
the original ledgers $finite$ and $compact$. The desired carrier is exactly the
given compact witness carrier.

In the step case, the induction hypothesis gives the carrier at the prior pair
$(f,c)$. Apply
Theorem~\ref{thm:compact-witness-carrier-finite-extension-closed} to that
carrier, the unary refinement history $e$, and the continuations
$\Cont(f,e,f')$ and $\Cont(c,e,c')$. The result is the required compact witness
carrier at $(f',c')$.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactWitnessCarrier\_finite\_refinement\_chain\_closed}

\begin{theorem}[Compact witness carrier empty final refinement reflects empty initial]
\label{thm:compact-witness-carrier-empty-final-refinement-reflects-empty-initial}
For a compact witness carrier, if a finite refinement chain sends the finite
and compact ledgers to empty final ledgers, then all five carrier ledgers are
empty:
$$
\begin{array}{l}
\mathsf{CompactWitnessCarrier}(subset,located,finite,intermediate,compact)
\land
\mathsf{CompactFiniteRefinementChain}(finite,compact,finalFinite,finalCompact)
\\
\land\ \hsame(finalFinite,\emp)\land\hsame(finalCompact,\emp)
\Longrightarrow
\hsame(subset,\emp)\land\hsame(located,\emp)\land\hsame(finite,\emp)\land
\hsame(intermediate,\emp)\land\hsame(compact,\emp).
\end{array}
$$
\end{theorem}
\begin{proof}
First apply the finite-chain empty reflection theorem to the refinement chain.
Then use empty-result inversion on the compactness continuation
$\Cont(intermediate,finite,compact)$ and on the locatedness continuation
$\Cont(subset,located,intermediate)$.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactWitnessCarrier\_empty\_final\_refinement\_reflects\_empty\_initial}

\begin{theorem}[Compact witness carrier finite refinement chain endpoints unary]
\label{thm:compact-witness-carrier-finite-refinement-chain-endpoints-unary}
If

$$
\mathsf{CompactWitnessCarrier}(subset,located,finite,intermediate,compact)
$$

and

$$
\mathsf{CompactFiniteRefinementChain}(finite,compact,finalFinite,finalCompact),
$$

then the final finite-net and compactness ledgers are unary histories:

$$
\mathsf{UnaryHistory}(finalFinite)\land\mathsf{UnaryHistory}(finalCompact).
$$
\end{theorem}
\begin{proof}
The compact witness carrier supplies a unary finite-net ledger. Its subset and
located unary witnesses, together with the first continuation ledger, give a
unary intermediate ledger; combining that with the finite-net unary witness and
the second continuation ledger gives a unary compactness ledger. Applying the
finite refinement chain endpoint theorem to these two starting endpoints gives
the displayed final endpoint unary witnesses.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactWitnessCarrier\_finite\_refinement\_chain\_endpoints\_unary}

\begin{theorem}[Compact witness carrier iterated finite refinement closure]
\label{thm:compact-witness-carrier-iterated-finite-refinement-closed}
For histories $subset$, $located$, $finite$, $extra_0$, $extra_1$,
$intermediate$, $compact$, $finite_0$, $compact_0$, $finite_1$, and
$compact_1$, assume:
\begin{enumerate}
  \item
  $$
  \mathsf{CompactWitnessCarrier}(subset,located,finite,intermediate,compact);
  $$
  \item $\mathsf{UnaryHistory}(extra_0)$ and
  $\mathsf{UnaryHistory}(extra_1)$;
  \item $\Cont(finite,extra_0,finite_0)$ and
  $\Cont(compact,extra_0,compact_0)$;
  \item $\Cont(finite_0,extra_1,finite_1)$ and
  $\Cont(compact_0,extra_1,compact_1)$.
\end{enumerate}
Then
$$
\mathsf{CompactWitnessCarrier}(subset,located,finite_1,intermediate,compact_1)
$$
\end{theorem}
\begin{proof}
First apply Theorem~\ref{thm:compact-witness-carrier-finite-extension-closed} to
the carrier with finite-net ledger $finite$ and compactness ledger $compact$,
using $extra_0$ as the unary refinement history. This produces the intermediate
carrier.

$$
\mathsf{CompactWitnessCarrier}(subset,located,finite_0,intermediate,compact_0)
$$

Apply the same theorem again to that intermediate carrier, now using $extra_1$
as the unary refinement history and the continuations from $finite_0$ to
$finite_1$ and from $compact_0$ to $compact_1$. The resulting carrier is:

$$
\mathsf{CompactWitnessCarrier}(subset,located,finite_1,intermediate,compact_1)
$$
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactWitnessCarrier\_iterated\_finite\_refinement\_closed}

\begin{theorem}[Compact witness carrier located extension closure]
\label{thm:compact-witness-carrier-located-extension-closed}
If a compact witness carrier records a subset through locatedness and finite-net
histories, then appending a unary locatedness refinement preserves the compact
witness shape after reassociating the subset ledger.
\end{theorem}
\begin{proof}
The carrier supplies unary subset, locatedness, and finite-net histories,
together with a continuation from the subset through the locatedness witness to
the intermediate witness. Unary continuation closure makes the refined
locatedness history unary. Continuation associativity shifts the subset ledger
across the locatedness refinement, and determinism identifies the associated
locatedness witness with the displayed refinement. The final finite-net ledger
then gives the compact witness endpoint.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactWitnessCarrier\_located\_extension\_closed}

\begin{definition}[Compact located refinement chain]
\label{def:compact-located-refinement-chain}
The compact located refinement chain relation is written

$$
\mathsf{CompactLocatedRefinementChain}(finite,located,intermediate,compact,finalLocated,finalIntermediate,finalCompact).
$$

It is the inductive BEDC relation on locatedness, intermediate, and compactness
ledgers with the finite-net ledger held fixed, generated by the following
constructors:
\begin{enumerate}
  \item the base constructor relates $(located,intermediate,compact)$ to itself
  while keeping $finite$ fixed;
  \item the step constructor says that, from a prior chain to $(l,i,c)$,
  $\mathsf{UnaryHistory}(e)$, $\Cont(l,e,l')$, $\Cont(i,e,i')$, and
  $\Cont(i',finite,c')$, one obtains a chain to $(l',i',c')$.
\end{enumerate}
\end{definition}
\leandef{BEDC.Derived.CompactUp.CompactLocatedRefinementChain}

\begin{theorem}[Compact located refinement chain transitivity]
\label{thm:compact-located-refinement-chain-transitivity}
For a fixed finite-net ledger $finite$, locatedness ledgers $located$,
$midLocated$, and $finalLocated$, intermediate ledgers $intermediate$,
$midIntermediate$, and $finalIntermediate$, and compactness ledgers $compact$,
$midCompact$, and $finalCompact$, if

$$
\mathsf{CompactLocatedRefinementChain}(finite,located,intermediate,compact,midLocated,midIntermediate,midCompact)
$$

and

$$
\mathsf{CompactLocatedRefinementChain}(finite,midLocated,midIntermediate,midCompact,finalLocated,finalIntermediate,finalCompact),
$$

then

$$
\mathsf{CompactLocatedRefinementChain}(finite,located,intermediate,compact,finalLocated,finalIntermediate,finalCompact).
$$
\end{theorem}
\begin{proof}
Induct on the second compact located refinement chain.

In the base case, the second chain ends at its starting triple
$(midLocated,midIntermediate,midCompact)$. The desired chain is exactly the first
chain.

In the step case, the second chain is obtained from a prior triple $(l,i,c)$ by
a unary refinement history $e$ together with witnesses $\mathsf{UnaryHistory}(e)$,
$\Cont(l,e,l')$, $\Cont(i,e,i')$, and $\Cont(i',finite,c')$. Apply the induction
hypothesis to compose the first chain with the prior second-chain segment,
producing a chain from $(located,intermediate,compact)$ to $(l,i,c)$.

Apply the step constructor of
Definition~\ref{def:compact-located-refinement-chain} to that composed chain and
the displayed unary-history and continuation witnesses. This gives the required
chain from $(located,intermediate,compact)$ to $(l',i',c')$, which is the final
triple in the step case.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactLocatedRefinementChain\_transitivity}

\begin{theorem}[Compact located refinement chain endpoint unary hsame iff]
\label{thm:compact-located-refinement-chain-endpoint-unary-hsame-iff}
If a compact located refinement chain carries unary starting histories
\(finite,located,intermediate,compact\), then its final located,
intermediate, and compact endpoints are unary, and
\(\hsame(finalLocated,finalIntermediate)\) holds exactly when
\(\hsame(located,intermediate)\) holds.

\begin{proof}
Apply \autoref{thm:compact-located-refinement-chain-final-endpoints-unary}

\begin{theorem}[Compact located refinement chain continuation witness]
\label{thm:compact-located-refinement-chain-continuation-witness}
If
$$
\mathsf{CompactLocatedRefinementChain}(finite,located,intermediate,compact,
finalLocated,finalIntermediate,finalCompact)
$$
holds and $\Cont(intermediate,finite,compact)$ is the starting compactness
ledger, then there is a unary refinement history $extra$ such that
$$
\mathsf{UnaryHistory}(extra)\land
\Cont(located,extra,finalLocated)\land
\Cont(intermediate,extra,finalIntermediate)\land
\Cont(finalIntermediate,finite,finalCompact).
$$
\end{theorem}
\begin{proof}
Induct on the located refinement chain. The base case uses the empty history as
the common locatedness and intermediate tail, together with the supplied
starting compactness ledger. In the step case, the induction hypothesis supplies
an accumulated tail for locatedness and the intermediate ledger. Associate that
tail with the step refinement on both ledgers. Determinism identifies the two
associated tails, unary continuation closure proves the accumulated tail unary,
and the step compactness continuation gives the final compact endpoint ledger.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactLocatedRefinementChain\_continuation\_witness}
\begin{theorem}[Compact located refinement chain final compact determinacy]\label{thm:compact-located-refinement-chain-final-compact-deterministic}
If two compact located refinement chains share the same starting compactness ledger and the same final located and intermediate endpoints, then their final compact endpoints are $\hsame$-equal.
\end{theorem}
\begin{proof}
Apply the continuation witness theorem to both chains using the same starting compactness ledger. Both witnesses end in $\Cont(finalIntermediate,finite,-)$, so continuation determinism identifies the final compact endpoints.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactLocatedRefinementChain\_final\_compact\_deterministic}
\begin{theorem}[Compact witness carrier located refinement chain closure]
\label{thm:compact-witness-carrier-located-refinement-chain-closed}
If
$$
\mathsf{CompactWitnessCarrier}(subset,located,finite,intermediate,compact)
$$
and
$$
\mathsf{CompactLocatedRefinementChain}(finite,located,intermediate,compact,finalLocated,finalIntermediate,finalCompact),
$$
then
$$
\mathsf{CompactWitnessCarrier}(subset,finalLocated,finite,finalIntermediate,finalCompact).
$$
\end{theorem}
\begin{proof}
Induct on the compact located refinement chain.

In the base case, the final locatedness, intermediate, and compactness ledgers
are the original ledgers $located$, $intermediate$, and $compact$. The desired
carrier is exactly the given compact witness carrier.

In the step case, the induction hypothesis gives the carrier at the prior triple
$(l,i,c)$. Apply
Theorem~\ref{thm:compact-witness-carrier-located-extension-closed} to that
carrier, the unary refinement history $e$, and the continuations
$\Cont(l,e,l')$, $\Cont(i,e,i')$, and $\Cont(i',finite,c')$.

The result is the required compact witness carrier at $(l',i',c')$.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactWitnessCarrier\_located\_refinement\_chain\_closed}

\begin{theorem}[Compact witness carrier located finite refinement chain closure]
\label{thm:compact-witness-carrier-located-finite-refinement-chain-closed}
If

$$
\mathsf{CompactWitnessCarrier}(subset,located,finite,intermediate,compact),
$$

and

$$
\mathsf{CompactLocatedRefinementChain}(finite,located,intermediate,compact,
finalLocated,finalIntermediate,finalCompact),
$$

and

$$
\mathsf{CompactFiniteRefinementChain}(finite,finalCompact,finalFinite,finalCompact'),
$$

then

$$
\mathsf{CompactWitnessCarrier}(subset,finalLocated,finalFinite,finalIntermediate,
finalCompact').
$$
\end{theorem}
\begin{proof}
First apply
Theorem~\ref{thm:compact-witness-carrier-located-refinement-chain-closed} to the
initial carrier and the located refinement chain. This gives

$$
\mathsf{CompactWitnessCarrier}(subset,finalLocated,finite,finalIntermediate,
finalCompact).
$$

Then apply
Theorem~\ref{thm:compact-witness-carrier-finite-refinement-chain-closed} to that
carrier and the finite refinement chain from $(finite,finalCompact)$ to
$(finalFinite,finalCompact')$. The result is the displayed compact witness
carrier.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactWitnessCarrier\_located\_finite\_refinement\_chain\_closed}

\begin{theorem}[Compact witness carrier located finite refinement chain final endpoints unary]
\label{thm:compact-witness-carrier-located-finite-refinement-chain-final-endpoints-unary}
Assume
$$
\mathsf{CompactWitnessCarrier}(subset,located,finite,intermediate,compact),
$$
a located refinement chain from $(located,intermediate,compact)$ to
$(finalLocated,finalIntermediate,finalCompact)$ over the same finite witness,
and a finite refinement chain from $(finite,finalCompact)$ to
$(finalFinite,finalCompact')$. Then
$$
\mathsf{UnaryHistory}(finalLocated)\land
\mathsf{UnaryHistory}(finalIntermediate)\land
\mathsf{UnaryHistory}(finalFinite)\land
\mathsf{UnaryHistory}(finalCompact').
$$
\end{theorem}
\begin{proof}
Unpack the compact witness carrier. The carrier provides unary witnesses for
$subset$, $located$, and $finite$ together with the continuation ledgers
$\Cont(subset,located,intermediate)$ and
$\Cont(intermediate,finite,compact)$. Closure of unary histories under
$\Cont$ gives unary witnesses for $intermediate$ and $compact$.

Apply the located-chain endpoint theorem to the located chain and these four
initial endpoint witnesses. This gives unary witnesses for $finalLocated$,
$finalIntermediate$, and $finalCompact$. Apply the finite-chain endpoint theorem
to the finite chain, the original unary witness for $finite$, and the obtained
unary witness for $finalCompact$. The result supplies unary witnesses for
$finalFinite$ and $finalCompact'$, which together form the displayed
conjunction.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactWitnessCarrier\_located\_finite\_refinement\_chain\_final\_endpoints\_unary}
