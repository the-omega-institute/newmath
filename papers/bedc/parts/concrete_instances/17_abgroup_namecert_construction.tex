\chapter{A Concrete Naming Certificate for \(\AbGroupUp\)}
\label{ch:concrete-instances-abgroup-namecert}

The abelian group interface \(\AbGroupUp\) extends \autoref{ch:concrete-instances-group-namecert} with the commutativity field \(a \cdot b = b \cdot a\) under the carrier's classifier. It is the abelian specialization of the group certificate; the integer additive group from \autoref{ch:concrete-instances-int-namecert} is the canonical concrete instance.

\section{Carrier definition}

\begin{definition}[AbGroup carrier]
\label{def:abgroup-carrier}
Use a group setup extended with commutativity of the binary operation.
\end{definition}

\section{The certificate}

\begin{definition}[AbGroup source specification]
\label{def:abgroup-source-specification}
The source is a certified group carrier plus a commutativity field.
\end{definition}

\begin{definition}[AbGroup pattern specification]
\label{def:abgroup-pattern-specification}
The pattern is group expression formation under an operation whose operands may be exchanged by certificate law.
\end{definition}

\begin{definition}[AbGroup classifier specification]
\label{def:abgroup-classifier-specification}
The classifier is the group carrier classifier.
\end{definition}

\begin{definition}[AbGroup stability certificate]
\label{def:abgroup-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item all group stability fields
  \item commutativity under classifier
  \item compatibility of commutativity with inverse and multiplication congruence
\end{enumerate}
\end{definition}

\begin{theorem}[AbGroup left-right multiplication swap]
\label{thm:abgroup-mul-left-right-swap}
For BEDC histories equipped with an associative multiplication, commutativity,
and multiplication congruence under \(\hsame\), the expression
\(a\cdot(b\cdot c)\) is internally the same as \(b\cdot(a\cdot c)\).
\end{theorem}
\begin{proof}
Reassociate \(a\cdot(b\cdot c)\) to \((a\cdot b)\cdot c\), transport the
left factor across commutativity, and reassociate \((b\cdot a)\cdot c\) back to
\(b\cdot(a\cdot c)\).
\end{proof}
\leanchecked{BEDC.Derived.AbGroupUp.abgroup\_mul\_left\_right\_swap}

\begin{theorem}[AbGroup middle-four interchange]
\label{thm:abgroup-middle-four-interchange}
For BEDC histories equipped with an associative multiplication, commutativity,
and multiplication congruence under \(\hsame\),
$$
\hsame((a\cdot b)\cdot(c\cdot d),(a\cdot c)\cdot(b\cdot d)).
$$
\end{theorem}
\begin{proof}
Reassociate \((a\cdot b)\cdot(c\cdot d)\) to
\(a\cdot(b\cdot(c\cdot d))\). Apply the abelian left-right swap to the right
factor, obtaining \(a\cdot(c\cdot(b\cdot d))\), and reassociate back to
\((a\cdot c)\cdot(b\cdot d)\).
\end{proof}
\leanchecked{BEDC.Derived.AbGroupUp.abgroup\_mul\_middle\_four}

\begin{theorem}[AbGroup conjugation collapse]
\label{thm:abgroup-conjugation-collapse}
For BEDC histories equipped with associative and commutative multiplication,
a right identity, multiplication congruence, and right inverses, every
expression $a\cdot(b\cdot a^{-1})$ is internally the same as $b$.
\end{theorem}
\begin{proof}
Use the abelian left-right swap to rewrite $a\cdot(b\cdot a^{-1})$ as
$b\cdot(a\cdot a^{-1})$. Collapse $a\cdot a^{-1}$ to the identity by the
right-inverse law, then use the right-identity law for $b$.
\end{proof}
\leanchecked{BEDC.Derived.AbGroupUp.abgroup\_conjugation\_collapse}

\begin{theorem}[AbGroup inverse conjugation collapse]
\label{thm:abgroup-inverse-conjugation-collapse}
For BEDC histories equipped with associative and commutative multiplication,
a right identity, multiplication congruence, and left inverses, every
expression $a^{-1}\cdot(b\cdot a)$ is internally the same as $b$.
\end{theorem}
\begin{proof}
Use the abelian left-right swap to rewrite $a^{-1}\cdot(b\cdot a)$ as
$b\cdot(a^{-1}\cdot a)$. Collapse $a^{-1}\cdot a$ to the identity by the
left-inverse law, then use the right-identity law for $b$.
\end{proof}
\leanchecked{BEDC.Derived.AbGroupUp.abgroup\_inverse\_conjugation\_collapse}

\begin{theorem}[AbGroup inverse preserves multiplication order]
\label{thm:abgroup-inverse-mul}
For BEDC histories equipped with group laws, commutative multiplication, and
multiplication congruence under $\hsame$, the inverse of a product is
internally the product of the two inverses in the same order:
$$
\hsame((x\cdot y)^{-1},x^{-1}\cdot y^{-1}).
$$
\end{theorem}
\begin{proof}
The group inverse multiplication theorem identifies $(x\cdot y)^{-1}$ with
$y^{-1}\cdot x^{-1}$. Commutativity then exchanges the two inverse factors,
yielding $x^{-1}\cdot y^{-1}$.
\end{proof}
\leanchecked{BEDC.Derived.AbGroupUp.abgroup\_inverse\_mul}

\begin{theorem}[AbGroup right factor swap]
\label{thm:abgroup-mul-right-factor-swap}
For BEDC histories equipped with an associative multiplication, commutativity,
and multiplication congruence under $\hsame$,
$$
\hsame((a\cdot b)\cdot c,(a\cdot c)\cdot b).
$$
\end{theorem}
\begin{proof}
Reassociate $(a\cdot b)\cdot c$ to $a\cdot(b\cdot c)$, commute the inner
factor $b\cdot c$ to $c\cdot b$ under multiplication congruence, and reassociate
back to $(a\cdot c)\cdot b$.
\end{proof}
\leanchecked{BEDC.Derived.AbGroupUp.abgroup\_mul\_right\_factor\_swap}

\begin{theorem}[AbGroup common left factor collection]
\label{thm:abgroup-mul-common-left-factor-collect}
For BEDC histories equipped with an associative multiplication, commutativity,
and multiplication congruence under $\hsame$,
$$
\hsame((a\cdot b)\cdot(a\cdot c),a\cdot(a\cdot(b\cdot c))).
$$
\end{theorem}
\begin{proof}
Reassociate the left side to $a\cdot(b\cdot(a\cdot c))$. Apply the abelian
left-right multiplication swap inside the tail to turn $b\cdot(a\cdot c)$
into $a\cdot(b\cdot c)$, and transport that tail equality through left
multiplication by $a$.
\end{proof}
\leanchecked{BEDC.Derived.AbGroupUp.abgroup\_mul\_common\_left\_factor\_collect}

\begin{theorem}[AbGroup common right factor collection]
\label{thm:abgroup-mul-common-right-factor-collect}
For BEDC histories equipped with an associative multiplication, commutativity,
and multiplication congruence under $\hsame$,
$$
\hsame((b\cdot a)\cdot(c\cdot a),a\cdot(a\cdot(b\cdot c))).
$$
\end{theorem}
\begin{proof}
Commute the exposed right factors to rewrite the left side as
\((a\cdot b)\cdot(a\cdot c)\). The common left factor collection theorem then
collects the two occurrences of \(a\) and leaves the residual factor
\(b\cdot c\).
\end{proof}
\leanchecked{BEDC.Derived.AbGroupUp.abgroup\_mul\_common\_right\_factor\_collect}

\begin{theorem}[AbGroup balanced multiplication cancellation]
\label{thm:abgroup-mul-balanced-cancel}
For BEDC histories equipped with group laws, commutativity, and multiplication
congruence under $\hsame$, if $a\cdot b\hsame c\cdot a$, then $b\hsame c$.
\end{theorem}
\begin{proof}
Commute the right endpoint $c\cdot a$ to $a\cdot c$. The hypothesis then has
the common left factor $a$ on both sides, so group left cancellation removes
that factor and yields $b\hsame c$.
\end{proof}
\leanchecked{BEDC.Derived.AbGroupUp.abgroup\_mul\_balanced\_cancel}

\begin{theorem}[AbGroup middle-four double cancellation]
\label{thm:abgroup-middle-four-double-cancel}
For BEDC histories equipped with group laws, commutativity, and multiplication
congruence under $\hsame$, if
$$
\hsame((a\cdot b)\cdot(c\cdot d),(a\cdot c)\cdot(u\cdot d)),
$$
then $\hsame(b,u)$.
\end{theorem}
\begin{proof}
The rule depends on the AbGroup middle-four theorem and on the Group left and
right cancellation theorems; it is not merely a renamed form of any one
theorem. Apply Theorem~\ref{thm:abgroup-middle-four-interchange} to the left
endpoint of the hypothesis and compose by symmetry and transitivity of
$\hsame$. This turns the supplied comparison into
$$
\hsame((a\cdot c)\cdot(b\cdot d),(a\cdot c)\cdot(u\cdot d)).
$$

Now apply group left cancellation, Theorem~\ref{thm:group-left-cancellation},
to the common left factor $a\cdot c$. The remaining comparison is
$$
\hsame(b\cdot d,u\cdot d).
$$

Finally apply group right cancellation, Theorem~\ref{thm:group-right-cancellation},
to the common right factor $d$, yielding $\hsame(b,u)$.
\end{proof}

\begin{theorem}[Nonempty history append cannot be right-absorbed]
\label{thm:abgroup-history-append-nonempty-left-ne-right}
For every BEDC history $h$ and right tail $k$, neither
$\mathsf{append}(\mathsf{e0}(h),k)=k$ nor
$\mathsf{append}(\mathsf{e1}(h),k)=k$ holds.
\end{theorem}
\begin{proof}
Induct on the right tail $k$. The empty case is constructor disjointness
against $\emp$. In the nonempty cases, append exposes the same outer
constructor as the tail, so constructor injectivity reduces the claim to the
induction hypothesis on the inner tail.
\end{proof}
\leanchecked{BEDC.Derived.AbGroupUp.history\_append\_nonempty\_left\_ne\_right}

\begin{definition}[AbGroup ledger policy]
\label{def:abgroup-ledger-policy}
Record group expression provenance and the commutativity law when used.
\end{definition}

\begin{definition}[\(\AbGroupUp\) certificate obligations]
\label{def:abgroup-certificate-obligations}
The carrier, source, pattern, classifier, stability, and ledger data listed above are the obligations required for the interface name \(\AbGroupUp\).
\end{definition}

\paragraph{Primitive boundary.}
\(\AbGroupUp\) is not a BEDC primitive. It is admitted only when those obligations are supplied.
