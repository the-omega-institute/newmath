\chapter{A Concrete Naming Certificate for \(\AbGroupUp\)}
\label{ch:concrete-instances-abgroup-namecert}

The abelian group interface \(\AbGroupUp\) extends \autoref{ch:concrete-instances-group-namecert} with the commutativity field \(a \cdot b = b \cdot a\) under the carrier's classifier. It is the abelian specialization of the group certificate; the integer additive group from \autoref{ch:concrete-instances-int-namecert} is the canonical concrete instance.

\section{Carrier definition}

\begin{definition}[AbGroup carrier]
\label{def:abgroup-carrier}
Use a group setup extended with commutativity of the binary operation.
\end{definition}

\section{Concrete singleton history abelian group instance}
\label{sec:abgroup-concrete-singleton-history-instance}

\begin{definition}[Concrete singleton history abelian group instance]
\label{def:concrete-singleton-history-abgroup-instance}
Let $A_{\emp}$ and $R_{\emp}$ be the singleton carrier and classifier from
Definition~\ref{def:concrete-singleton-history-group-instance}. Define
the operations by
$$
\mathsf{mul}(h,k):=\mathsf{append}(h,k),\qquad
\mathsf{inv}(h):=\emp,\qquad
e:=\emp.
$$
The resulting tuple $(A_{\emp},R_{\emp},\mathsf{mul},e,\mathsf{inv})$ is the
singleton history abelian group instance.
\end{definition}

\begin{theorem}[Concrete singleton history abelian group laws]
\label{thm:concrete-singleton-history-abgroup-laws}
The tuple $(A_{\emp},R_{\emp},\mathsf{mul},e,\mathsf{inv})$ satisfies the
concrete group laws of Theorem~\ref{thm:concrete-singleton-history-group-laws}
and the abelian commutativity law
$R_{\emp}(\mathsf{mul}(h,k),\mathsf{mul}(k,h))$ for all histories $h,k$ with
$A_{\emp}(h)$ and $A_{\emp}(k)$.
\end{theorem}
\begin{proof}
The group fields are exactly the concrete singleton history group laws supplied
by Theorem~\ref{thm:concrete-singleton-history-group-laws}.

For commutativity, suppose $A_{\emp}(h)$ and $A_{\emp}(k)$. These singleton
carrier witnesses identify both $h$ and $k$ with $\emp$ under $\hsame$.

The continuation unit laws reduce $\mathsf{append}(h,k)$ and
$\mathsf{append}(k,h)$ to the singleton class of $\emp$. Symmetry and
transitivity of $\hsame$ then assemble the endpoint relation
$\hsame(\mathsf{mul}(h,k),\mathsf{mul}(k,h))$.

The multiplication-closure clauses from the concrete group laws supply
$A_{\emp}(\mathsf{mul}(h,k))$ and $A_{\emp}(\mathsf{mul}(k,h))$. Together with
the endpoint relation, these are exactly the classifier witness
$R_{\emp}(\mathsf{mul}(h,k),\mathsf{mul}(k,h))$.
\end{proof}

Thus the $\AbGroupUp$ schema is instantiated by this concrete $\mathsf{BHist}$
carrier and operation, so later Lean work can formalize BHist-anchored abelian
group laws in $\mathsf{BEDC.Derived.AbGroupUp}$ rather than another abstract
operator schema.

\end{proof}

\begin{theorem}[AbGroup centralizer and commutator collapse]
\label{thm:abgroup-centralizer-commutator-collapse}
For BEDC histories equipped with the $\GroupUp$ associativity, unit,
multiplication-congruence, and two-sided inverse laws, plus $\AbGroupUp$
commutativity under $\hsame$, fix $a:\mathsf{BHist}$. Let $Z_a$ be the
centralizer carrier of \autoref{def:group-centralizer-carrier}, and let
$\mathsf{comm}(x,a)$ be the commutator expression from
\autoref{thm:group-commutator-trivial-iff-commutes}. Then every
$x:\mathsf{BHist}$ satisfies:
\begin{enumerate}
\item $Z_a(x)$.
\item $\mathsf{comm}(x,a)\hsame e$.
\item $\hsame((a\cdot x)\cdot a^{-1},x)$.
\end{enumerate}
The ten-field fit is: traditional target abelian groups have total
centralizers; BEDC source $\AbGroupUp$ over $\mathsf{BHist}$ plus the
$\GroupUp$ centralizer carrier; carrier $Z_a$; classifier $\hsame$; stability
from commutativity, commutator exactness, and conjugation collapse; ledger
fixed element, commuting witness, commutator spine, and inverse witness; core
theorem the three displayed clauses backed by
\autoref{def:group-centralizer-carrier},
\autoref{thm:group-commutator-trivial-iff-commutes}, and
\autoref{thm:abgroup-conjugation-collapse}; exactness predicate total
centralizer iff trivial commutator; standard bridge ordinary abelian
centralizer collapse; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Commutativity supplies the comparison $x\cdot a\hsame a\cdot x$. By the
definition of $Z_a$ in \autoref{def:group-centralizer-carrier}, this is exactly
the centralizer clause for the chosen $x$.

Apply \autoref{thm:group-commutator-trivial-iff-commutes} to the same ordered
pair $(x,a)$. The commuting witness obtained above is transported to the
trivial commutator fiber, giving $\mathsf{comm}(x,a)\hsame e$.

Finally apply \autoref{thm:abgroup-conjugation-collapse} with middle element
$x$. The abelian conjugation spine collapses $(a\cdot x)\cdot a^{-1}$ to $x$,
which gives the inner-conjugation fixed-point clause.
\end{proof}

\begin{theorem}[AbGroup middle-four double cancellation]
\label{thm:abgroup-middle-four-double-cancel}
For BEDC histories equipped with group laws, commutativity, and multiplication
congruence under $\hsame$, if
$$
\hsame((a\cdot b)\cdot(c\cdot d),(a\cdot c)\cdot(u\cdot d)),
$$
then $\hsame(b,u)$.
\end{theorem}
\begin{proof}
The rule depends on the AbGroup middle-four theorem and on the Group left and
right cancellation theorems; it is not merely a renamed form of any one
theorem. Apply Theorem~\ref{thm:abgroup-middle-four-interchange} to the left
endpoint of the hypothesis and compose by symmetry and transitivity of
$\hsame$. This turns the supplied comparison into
$$
\hsame((a\cdot c)\cdot(b\cdot d),(a\cdot c)\cdot(u\cdot d)).
$$

Now apply group left cancellation, Theorem~\ref{thm:group-left-cancellation},
to the common left factor $a\cdot c$. The remaining comparison is
$$
\hsame(b\cdot d,u\cdot d).
$$

Finally apply group right cancellation, Theorem~\ref{thm:group-right-cancellation},
to the common right factor $d$, yielding $\hsame(b,u)$.
\end{proof}

\begin{theorem}[AbGroup transported middle-four double cancellation]
\label{thm:abgroup-middle-four-transported-double-cancel}

$$
\hsame((a\cdot b)\cdot(c\cdot d),(a\cdot u)\cdot(c\cdot v))

\end{theorem}
\begin{proof}

After symmetry and transitivity of $\hsame$, the comparison has common leading

$$
\hsame((a\cdot c)\cdot(b\cdot d),(a\cdot c)\cdot(u\cdot v)).

$$
\hsame((a\cdot c)\cdot(b\cdot d),(a\cdot c)\cdot(u\cdot d)).

Apply Theorem~\ref{thm:group-left-cancellation} to remove the common left
factor $a\cdot c$. The remaining comparison is
$$
\hsame(b\cdot d,u\cdot d).
$$

Finally apply Theorem~\ref{thm:group-right-cancellation} to remove the common
right factor $d$, yielding $\hsame(b,u)$.
\end{proof}

\begin{theorem}[Nonempty history append cannot be right-absorbed]
\label{thm:abgroup-history-append-nonempty-left-ne-right}
For every BEDC history $h$ and right tail $k$, neither
$\mathsf{append}(\mathsf{e0}(h),k)=k$ nor
$\mathsf{append}(\mathsf{e1}(h),k)=k$ holds.
\end{theorem}
\begin{proof}
Induct on the right tail $k$. The empty case is constructor disjointness
against $\emp$. In the nonempty cases, append exposes the same outer
constructor as the tail, so constructor injectivity reduces the claim to the
induction hypothesis on the inner tail.
\end{proof}
