\section{Field affine explicit inverse}
\label{sec:field-affine-explicit-inverse}

\begin{theorem}[Field two-sided affine expression has explicit inverse]
\label{thm:field-affine-two-sided-explicit-inverse}
For a BEDC history-level $\FieldUp$ candidate with the same operations,
additive laws, multiplication laws, congruence data, and indexed inverse laws
as in
\autoref{thm:field-affine-two-sided-equation-exact-from-apartness}, let
$p_a:\mathsf{NonZero}_F(a)$ and $p_b:\mathsf{NonZero}_F(b)$ be the two
nonzero witnesses for the left and right affine factors. For each
$c:\mathsf{BHist}$ define
$$
s(c)=
\mathsf{mul}(\mathsf{inv}(a,p_a),
\mathsf{mul}(\mathsf{add}(c,\mathsf{neg}(d)),\mathsf{inv}(b,p_b))).
$$
Then every $c,x,y:\mathsf{BHist}$ satisfies the following three clauses.
\begin{enumerate}
\item $\mathsf{add}(\mathsf{mul}(\mathsf{mul}(a,s(c)),b),d)\hsame c$.
\item If
$\mathsf{add}(\mathsf{mul}(\mathsf{mul}(a,x),b),d)\hsame c$, then
$x\hsame s(c)$.
\item If
$$
\mathsf{add}(\mathsf{mul}(\mathsf{mul}(a,x),b),d)\hsame
\mathsf{add}(\mathsf{mul}(\mathsf{mul}(a,y),b),d),
$$
then $x\hsame y$.
\end{enumerate}
\end{theorem}
\begin{proof}
For the displayed preimage, instantiate
\autoref{thm:field-affine-two-sided-equation-exact-from-apartness} at
$x=s(c)$. The reverse implication only needs the reflexive classifier
equality $s(c)\hsame s(c)$, so it produces
$\mathsf{add}(\mathsf{mul}(\mathsf{mul}(a,s(c)),b),d)\hsame c$.

For uniqueness of any solution, use the forward implication of the same
exactness theorem. An assumed classification of the affine expression at
$x$ with $c$ is transported directly to $x\hsame s(c)$.

For equality of two affine expressions, apply
\autoref{thm:field-affine-two-sided-map-classifier-exact-from-apartness} to the
assumed classifier equality. Its conclusion is exactly $x\hsame y$.
\end{proof}

\begin{theorem}[Field two-sided affine map is a certified automorphism]
\label{thm:field-affine-two-sided-certified-automorphism}
For a BEDC history-level $\FieldUp$ candidate with the same operations,
additive laws, multiplication laws, congruence data, and indexed inverse laws
as in
\autoref{thm:field-affine-two-sided-equation-exact-from-apartness}, let
$p_a:\mathsf{NonZero}_F(a)$ and $p_b:\mathsf{NonZero}_F(b)$ be the two
nonzero witnesses for the left and right affine factors. Define
$$
T(x)=\mathsf{add}(\mathsf{mul}(\mathsf{mul}(a,x),b),d)
$$
and, for each $c:\mathsf{BHist}$, define
$$
s(c)=
\mathsf{mul}(\mathsf{inv}(a,p_a),
\mathsf{mul}(\mathsf{add}(c,\mathsf{neg}(d)),\mathsf{inv}(b,p_b))).
$$
Then every $x,y,c:\mathsf{BHist}$ satisfies
$$
T(s(c))\hsame c,
\qquad
s(T(x))\hsame x,
\qquad
T(x)\hsame T(y)\Longleftrightarrow x\hsame y.
$$
\end{theorem}
\begin{proof}
For the left inverse clause, instantiate
\autoref{thm:field-affine-two-sided-equation-exact-from-apartness} at
$x=s(c)$. The reflexive classifier equality $s(c)\hsame s(c)$ enters the
reverse implication and yields $T(s(c))\hsame c$.

For the right inverse clause, instantiate the same exactness theorem with the
target classifier value $T(x)$. The reflexive equality $T(x)\hsame T(x)$ enters
the forward implication and yields $x\hsame s(T(x))$. Symmetry of $\hsame$
gives the displayed orientation $s(T(x))\hsame x$.

Finally apply
\autoref{thm:field-affine-two-sided-map-classifier-exact-from-apartness} with
the same witnesses $p_a,p_b$ and offset $d$. Its two directions are precisely
the classifier-preservation and classifier-reflection clauses
$T(x)\hsame T(y)\Longleftrightarrow x\hsame y$.
\end{proof}

Per-object certificate: traditional target affine automorphism over a field.
BEDC source: $\FieldUp$ over $\mathsf{BHist}$. Carrier:
$\mathsf{BHist}$. Classifier: $\hsame$. Stability obligations:
\autoref{thm:field-affine-two-sided-equation-exact-from-apartness} and
\autoref{thm:field-affine-two-sided-map-classifier-exact-from-apartness}.
Ledger: the two nonzero factor witnesses $p_a,p_b$ and the offset $d$. Core
theorem: \autoref{thm:field-affine-two-sided-certified-automorphism}. Exactness
predicate: inverse-pair plus classifier iff. Standard bridge: deferred. Status:
$\mathsf{ProofTarget}$.

\begin{theorem}[Field two-sided affine automorphism composition with explicit product witnesses]
\label{thm:field-affine-two-sided-automorphism-composition}
For a BEDC history-level $\FieldUp$ candidate over $\mathsf{BHist}$ with the
additive group laws, multiplication associativity, left and right
multiplicative unit laws, multiplication and addition congruence,
distributivity fields inherited from $\CommRingUp$, indexed inverse laws,
classifier transport of $\mathsf{NonZero}_F$, and contradiction of
$\mathsf{NonZero}_F(0)$, let
$p_a:\mathsf{NonZero}_F(a)$, $p_b:\mathsf{NonZero}_F(b)$,
$p_c:\mathsf{NonZero}_F(c)$, and $p_e:\mathsf{NonZero}_F(e)$ be explicit
nonzero witnesses. Define
$$
T_1(x)=\mathsf{add}(\mathsf{mul}(\mathsf{mul}(a,x),b),d)
$$
and
$$
T_2(y)=\mathsf{add}(\mathsf{mul}(\mathsf{mul}(c,y),e),f).
$$
Let $\mathsf{ApartZero}_F(h)$ abbreviate $h\hsame0\to\bot$, and assume
conversion maps
$$
\alpha_{ca}:\mathsf{ApartZero}_F(\mathsf{mul}(c,a))\to
\mathsf{NonZero}_F(\mathsf{mul}(c,a)),
\qquad
\alpha_{be}:\mathsf{ApartZero}_F(\mathsf{mul}(b,e))\to
\mathsf{NonZero}_F(\mathsf{mul}(b,e)).
$$
Let $q_{ca}$ and $q_{be}$ be the product apartness witnesses obtained from
\autoref{thm:field-nonzero-factors-exclude-zero-product} for the pairs
$(p_c,p_a)$ and $(p_b,p_e)$, and set
$p_{ca}:=\alpha_{ca}(q_{ca})$ and $p_{be}:=\alpha_{be}(q_{be})$. Define
$$
C(x)=
\mathsf{add}(
\mathsf{mul}(\mathsf{mul}(\mathsf{mul}(c,a),x),\mathsf{mul}(b,e)),
\mathsf{add}(\mathsf{mul}(\mathsf{mul}(c,d),e),f)).
$$
Then every $x:\mathsf{BHist}$ satisfies
$$
T_2(T_1(x))\hsame C(x).
$$
Moreover, with the product witnesses $p_{ca}$ and $p_{be}$, the displayed
composite $C$ is a certified two-sided affine automorphism by
\autoref{thm:field-affine-two-sided-certified-automorphism}.
\end{theorem}
\begin{proof}
Expand $T_2(T_1(x))$ and use distributivity to separate the translated part
coming from $d$ from the term containing $x$. Addition congruence transports
the two pieces through the outer sum, while multiplication congruence transports
the inner affine equality through the left factor $c$ and the right factor
$e$.

Multiplication associativity then collects the $x$-term into
$\mathsf{mul}(\mathsf{mul}(\mathsf{mul}(c,a),x),\mathsf{mul}(b,e))$. The same
associativity and distributivity data identify the translated remainder as
$\mathsf{add}(\mathsf{mul}(\mathsf{mul}(c,d),e),f)$. This gives the displayed
normal form $T_2(T_1(x))\hsame C(x)$.

It remains to provide the two nonzero indices required by
\autoref{thm:field-affine-two-sided-certified-automorphism}. Apply
\autoref{thm:field-nonzero-factors-exclude-zero-product} to $p_c,p_a$ to obtain
$q_{ca}:\mathsf{ApartZero}_F(\mathsf{mul}(c,a))$, and to $p_b,p_e$ to obtain
$q_{be}:\mathsf{ApartZero}_F(\mathsf{mul}(b,e))$. The conversion maps
$\alpha_{ca}$ and $\alpha_{be}$ turn these apartness witnesses into
$p_{ca}$ and $p_{be}$.

Now instantiate
\autoref{thm:field-affine-two-sided-certified-automorphism} with left factor
$\mathsf{mul}(c,a)$, right factor $\mathsf{mul}(b,e)$, offset
$\mathsf{add}(\mathsf{mul}(\mathsf{mul}(c,d),e),f)$, and witnesses
$p_{ca},p_{be}$. The theorem supplies the inverse-pair clauses and classifier
iff for $C$, so the normalized composite is again certified.
\end{proof}

Per-object certificate: traditional target affine automorphism composition.
BEDC source: $\FieldUp+\CommRingUp$ over $\mathsf{BHist}$. Carrier:
$\mathsf{BHist}$. Classifier: $\hsame$. Stability obligations: composition
normal form plus product apartness conversion. Ledger: the four factor
witnesses $p_a,p_b,p_c,p_e$ and the two product witnesses $p_{ca},p_{be}$.
Core theorem: \autoref{thm:field-affine-two-sided-automorphism-composition}.
Exactness predicate: composite certified automorphism. Standard bridge:
deferred. Status: $\mathsf{ProofTarget}$.

\begin{theorem}[Field affine composite has explicit two-sided inverse]
\label{thm:field-affine-composite-two-sided-inverse}
For a BEDC history-level $\FieldUp$ candidate under the hypotheses of
\autoref{thm:field-affine-two-sided-automorphism-composition}, let
$p_a:\mathsf{NonZero}_F(a)$, $p_b:\mathsf{NonZero}_F(b)$,
$p_{a'}:\mathsf{NonZero}_F(a')$, and $p_{b'}:\mathsf{NonZero}_F(b')$ be the
four nonzero witnesses for the two affine certificates. Define
$$
T_1(x)=\mathsf{add}(\mathsf{mul}(\mathsf{mul}(a,x),b),d)
$$
and
$$
T_2(x)=\mathsf{add}(\mathsf{mul}(\mathsf{mul}(a',x),b'),d').
$$
For each $r:\mathsf{BHist}$ define
$$
s_1(r)=
\mathsf{mul}(\mathsf{inv}(a,p_a),
\mathsf{mul}(\mathsf{add}(r,\mathsf{neg}(d)),\mathsf{inv}(b,p_b)))
$$
and
$$
s_2(r)=
\mathsf{mul}(\mathsf{inv}(a',p_{a'}),
\mathsf{mul}(\mathsf{add}(r,\mathsf{neg}(d')),\mathsf{inv}(b',p_{b'}))).
$$
Then every $x,y,q:\mathsf{BHist}$ satisfies the following three clauses.
\begin{enumerate}
\item

$$
T_2(T_1(x))\hsame T_2(T_1(y)) \Longleftrightarrow x\hsame y.
$$
\item

$$
T_2(T_1(s_1(s_2(q))))\hsame q.
$$
\item

$$
s_1(s_2(T_2(T_1(x))))\hsame x.
$$
\end{enumerate}
\end{theorem}
\begin{proof}
Instantiate
\autoref{thm:field-affine-two-sided-automorphism-composition} with
$c=a'$, $e=b'$, $f=d'$, and the product conversion maps for the collected
factors. Its collected-factor normal form, together with
\autoref{thm:field-affine-two-sided-certified-automorphism}, supplies the
classifier exactness for the composite, giving the displayed iff clause.

For the left inverse clause, first use the left inverse of the $T_2$
certificate at $q$, which gives $T_2(s_2(q))\hsame q$. Then use the left
inverse of the $T_1$ certificate at $s_2(q)$, giving
$$
T_1(s_1(s_2(q)))\hsame s_2(q).
$$
Congruence of $T_2$ transports this equality through the outer map, and
transitivity yields
$$
T_2(T_1(s_1(s_2(q))))\hsame q.
$$

For the right inverse clause, the right inverse of the $T_2$ certificate gives
$$
s_2(T_2(T_1(x)))\hsame T_1(x).
$$
Apply the left inverse clause of the $T_1$ certificate to
$s_2(T_2(T_1(x)))$, obtaining
$$
T_1(s_1(s_2(T_2(T_1(x)))))\hsame s_2(T_2(T_1(x))).
$$
Compose these two equalities to classify
$$
T_1(s_1(s_2(T_2(T_1(x)))))
$$
with $T_1(x)$, and reflect the result through the $T_1$ classifier exactness to
conclude $s_1(s_2(T_2(T_1(x))))\hsame x$.
\end{proof}

Per-object certificate: traditional target affine automorphism inverse for
composition; BEDC source $\FieldUp$ over $\mathsf{BHist}$; carrier
$\mathsf{BHist}$; classifier $\hsame$; stability obligations the two affine
certificates plus composition classifier exactness; ledger four nonzero
witnesses and two offsets; core theorem
\autoref{thm:field-affine-composite-two-sided-inverse}; exactness predicate
two-sided inverse plus classifier iff; standard bridge deferred; status
$\mathsf{ProofTarget}$.

\begin{theorem}[Field affine composite inverse agrees with reverse-composed inverses]
\label{thm:field-affine-composite-inverse-reverse-coherence}
For a BEDC history-level $\FieldUp$ candidate with the operations and laws used
by
\autoref{thm:field-affine-two-sided-automorphism-composition}, let
$p_a:\mathsf{NonZero}_F(a)$, $p_b:\mathsf{NonZero}_F(b)$,
$p_{a'}:\mathsf{NonZero}_F(a')$, and $p_{b'}:\mathsf{NonZero}_F(b')$ be the
nonzero witnesses for the two component affine maps
$$
T_1(x)=\mathsf{add}(\mathsf{mul}(\mathsf{mul}(a,x),b),d)
$$
and
$$
T_2(x)=\mathsf{add}(\mathsf{mul}(\mathsf{mul}(a',x),b'),d').
$$
Put
$A=\mathsf{mul}(a',a)$, $B=\mathsf{mul}(b,b')$, and
$D=\mathsf{add}(\mathsf{mul}(\mathsf{mul}(a',d),b'),d')$. Let
$\mathsf{ApartZero}_F(h)$ abbreviate $h\hsame0\to\bot$, and assume conversion
maps
$$
\alpha_A:\mathsf{ApartZero}_F(A)\to\mathsf{NonZero}_F(A),
\qquad
\alpha_B:\mathsf{ApartZero}_F(B)\to\mathsf{NonZero}_F(B).
$$
Let $q_A$ and $q_B$ be the product apartness witnesses obtained from
\autoref{thm:field-nonzero-factors-exclude-zero-product} for the pairs
$(p_{a'},p_a)$ and $(p_b,p_{b'})$, and set $p_A:=\alpha_A(q_A)$ and
$p_B:=\alpha_B(q_B)$. For each
$c:\mathsf{BHist}$ define
$$
s_1(c)=
\mathsf{mul}(\mathsf{inv}(a,p_a),
\mathsf{mul}(\mathsf{add}(c,\mathsf{neg}(d)),\mathsf{inv}(b,p_b)))
$$
and
$$
s_2(c)=
\mathsf{mul}(\mathsf{inv}(a',p_{a'}),
\mathsf{mul}(\mathsf{add}(c,\mathsf{neg}(d')),\mathsf{inv}(b',p_{b'}))).
$$
For each $q:\mathsf{BHist}$ define the composite explicit inverse
$$
s_{21}(q)=
\mathsf{mul}(\mathsf{inv}(A,p_A),
\mathsf{mul}(\mathsf{add}(q,\mathsf{neg}(D)),\mathsf{inv}(B,p_B))).
$$
Then every $q:\mathsf{BHist}$ satisfies
$$
s_{21}(q)\hsame s_1(s_2(q)).
$$
\end{theorem}
\begin{proof}
Let
$$
C(x)=\mathsf{add}(\mathsf{mul}(\mathsf{mul}(A,x),B),D).
$$
Apply
\autoref{thm:field-affine-two-sided-explicit-inverse} to the composite
coefficients $A,B,D$ with witnesses $p_A,p_B$. Its left inverse clause gives
$C(s_{21}(q))\hsame q$.

Next instantiate
\autoref{thm:field-affine-two-sided-certified-automorphism} for the first
component with witnesses $p_a,p_b$ and offset $d$, and again for the second
component with witnesses $p_{a'},p_{b'}$ and offset $d'$. The first certificate
gives $T_1(s_1(s_2(q)))\hsame s_2(q)$, and the second gives
$T_2(s_2(q))\hsame q$.

Use the classifier-preservation direction of the second certificate to
transport the first displayed equality through $T_2$. Transitivity with
$T_2(s_2(q))\hsame q$ yields
$T_2(T_1(s_1(s_2(q))))\hsame q$.

By
\autoref{thm:field-affine-two-sided-automorphism-composition} with the
substitution $c=a'$, $e=b'$, and $f=d'$, the left side of this equation is
classified by $C(s_1(s_2(q)))$ with the collected coefficients $A,B,D$, and the
certified automorphism status uses the converted witnesses $p_A,p_B$. Hence
$s_1(s_2(q))$ also solves the same composite affine equation $C(x)\hsame q$.

Finally apply
\autoref{thm:field-affine-two-sided-equation-solution-unique-from-apartness}
to the two solutions $s_{21}(q)$ and $s_1(s_2(q))$ of $C(x)\hsame q$. The
unique-solution conclusion is exactly
$s_{21}(q)\hsame s_1(s_2(q))$.
\end{proof}

exactness, certified component automorphisms, composite coefficient collection,

\begin{theorem}[Field affine composite has reverse explicit inverse]
\label{thm:field-affine-composite-reverse-explicit-inverse}
Under the hypotheses and notation of
\autoref{thm:field-affine-two-sided-certified-automorphism} and
\autoref{thm:field-affine-two-sided-automorphism-composition}, define
$$
T(x)=\mathsf{add}(\mathsf{mul}(\mathsf{mul}(a,x),b),d)
$$
and
$$
U(z)=\mathsf{add}(\mathsf{mul}(\mathsf{mul}(c,z),e),f).
$$
Let $s_T$ and $s_U$ be the displayed explicit inverse maps associated to $T$
and $U$. Then every $x,y,q:\mathsf{BHist}$ satisfies
$$
U(T(s_T(s_U(q))))\hsame q,
\qquad
s_T(s_U(U(T(x))))\hsame x,
\qquad
U(T(x))\hsame U(T(y))\Longleftrightarrow x\hsame y.
$$
\end{theorem}
\begin{proof}
First instantiate
\autoref{thm:field-affine-two-sided-automorphism-composition} with
the factors and offsets defining $T$ and $U$. The composition theorem identifies
$U\circ T$ as the certified affine composite, and
\autoref{thm:field-affine-two-sided-certified-automorphism} supplies the
classifier-iff clause for that certified composite.

For the displayed left preimage, apply the left-inverse clause for $T$ at
$s_U(q)$, then apply the left-inverse clause for $U$ at $q$. Congruence of $U$
and transitivity of $\hsame$ identify the transported composite with
$U(T(s_T(s_U(q))))\hsame q$.

For the right inverse, apply the right-inverse clause of
\autoref{thm:field-affine-two-sided-certified-automorphism} to $U$ at the point
$T(x)$. This gives
$$
s_U(U(T(x)))\hsame T(x).
$$

Use the solution-uniqueness clause of
\autoref{thm:field-affine-two-sided-explicit-inverse} for $T$ at the target
$T(x)$, with the reflexive classification of $T(x)$ as its input. This gives
$x\hsame s_T(T(x))$, hence $s_T(T(x))\hsame x$ by symmetry. Transport the
displayed equality through the explicit formula for $s_T$, and finish by
transitivity of $\hsame$ to obtain $s_T(s_U(U(T(x))))\hsame x$.
\end{proof}

\begin{definition}[Field affine automorphism package]
\label{def:field-affine-automorphism-package}
For a BEDC history-level $\FieldUp$ candidate over $\mathsf{BHist}$, a field
affine certificate is data
$(a,b,d,p_a,p_b)$ with $a,b,d:\mathsf{BHist}$,
$p_a:\mathsf{NonZero}_F(a)$, and $p_b:\mathsf{NonZero}_F(b)$. Its displayed
map is
$$
T_{a,b,d}(x)=\mathsf{add}(\mathsf{mul}(\mathsf{mul}(a,x),b),d).
$$
For two displayed affine maps $T$ and $U$, the affine pointwise classifier is
defined by
$$
T\equiv_{\mathrm{aff}} U
\quad\Longleftrightarrow\quad
\text{every }x:\mathsf{BHist}\text{ satisfies }T(x)\hsame U(x).
$$
\end{definition}

\begin{theorem}[Field affine automorphism package laws]
\label{thm:field-affine-automorphism-group-package}
For a BEDC history-level $\FieldUp+\CommRingUp$ candidate over
$\mathsf{BHist}$, assume the explicit nonzero witness
$p_1:\mathsf{NonZero}_F(\mathsf{one})$ and the inverse-coefficient nonzero
witnesses required by
\autoref{thm:field-affine-inverse-is-certified-affine}. Then the affine
certificates of \autoref{def:field-affine-automorphism-package} satisfy the
$\GroupUp$-style law package under $\equiv_{\mathrm{aff}}$: the identity
certificate is
$(\mathsf{one},\mathsf{one},\mathsf{zero},p_1,p_1)$, the coefficient-collected
composition is the composite supplied by
\autoref{thm:field-affine-two-sided-automorphism-composition}, and the inverse
of $(a,b,d,p_a,p_b)$ is the normal form supplied by
\autoref{thm:field-affine-inverse-is-certified-affine}. These operations are
closed on certified affine automorphisms, composition is associative under the
pointwise classifier, and every certificate satisfies the left and right
identity laws and the left and right inverse laws under
$\equiv_{\mathrm{aff}}$.
\end{theorem}
\begin{proof}
The identity certificate uses the unit and zero laws already carried by the
$\FieldUp+\CommRingUp$ setup. For every $x:\mathsf{BHist}$, the displayed map
with coefficients $(\mathsf{one},\mathsf{one},\mathsf{zero})$ reduces by the
left and right multiplicative unit laws and the additive zero law to $x$, so it
is the identity under $\equiv_{\mathrm{aff}}$.

Composition closure is exactly
\autoref{thm:field-affine-two-sided-automorphism-composition}. Given two
certified affine automorphisms, that theorem collects the factors and offset
into the displayed affine normal form and returns the product nonzero witnesses
needed to certify the composite. The resulting certified map is also covered by
\autoref{thm:field-affine-two-sided-certified-automorphism}.

Associativity is checked pointwise. For three certificates, compare the two
parenthesized triple composites by expanding both through the same displayed
coefficient collection classifier from
\autoref{thm:field-affine-two-sided-automorphism-composition}; associativity of
the $\CommRingUp$ multiplication and addition data identifies the collected
left factor, right factor, and offset. Hence the two triple composites are
related by $\equiv_{\mathrm{aff}}$.

Inverse closure is
\autoref{thm:field-affine-inverse-is-certified-affine}: the explicit inverse
map is classifier-equal to a certified affine normal form whose coefficients
use the supplied inverse-coefficient witnesses. The two-sided inverse laws then
follow from \autoref{thm:field-affine-two-sided-certified-automorphism} for a
single certificate and from
\autoref{thm:field-affine-composite-reverse-explicit-inverse} for a composite,
which identifies the reverse explicit inverse with the collected composite
inverse. Together these clauses give the stated $\GroupUp$-style law package.
\end{proof}

\begin{theorem}[Field threefold certified affine automorphisms compose]
\label{thm:field-affine-threefold-certified-automorphism-composition}
For a BEDC history-level $\FieldUp$ candidate over $\mathsf{BHist}$ with the
operations, congruence laws, distributivity laws, indexed inverse laws,
classifier transport of $\mathsf{NonZero}_F$, and product-apartness conversion
maps used in
\autoref{thm:field-affine-two-sided-automorphism-composition}, let
$p_{a_1}:\mathsf{NonZero}_F(a_1)$, $p_{b_1}:\mathsf{NonZero}_F(b_1)$,
$p_{a_2}:\mathsf{NonZero}_F(a_2)$, $p_{b_2}:\mathsf{NonZero}_F(b_2)$,
$p_{a_3}:\mathsf{NonZero}_F(a_3)$, and
$p_{b_3}:\mathsf{NonZero}_F(b_3)$ be the six side-factor witnesses. Define
the component maps
$$
T_1(x)=\mathsf{add}(\mathsf{mul}(\mathsf{mul}(a_1,x),b_1),d_1),
\qquad
T_2(x)=\mathsf{add}(\mathsf{mul}(\mathsf{mul}(a_2,x),b_2),d_2),
\qquad
T_3(x)=\mathsf{add}(\mathsf{mul}(\mathsf{mul}(a_3,x),b_3),d_3).
$$
First collect the factors for $T_2\circ T_1$ by
$$
A_{21}=\mathsf{mul}(a_2,a_1),\qquad
B_{21}=\mathsf{mul}(b_1,b_2),\qquad
D_{21}=\mathsf{add}(\mathsf{mul}(\mathsf{mul}(a_2,d_1),b_2),d_2).
$$
Then collect the factors for $T_3\circ T_2\circ T_1$ by
$$
A_{321}=\mathsf{mul}(a_3,A_{21}),\qquad
B_{321}=\mathsf{mul}(B_{21},b_3),\qquad
D_{321}=\mathsf{add}(\mathsf{mul}(\mathsf{mul}(a_3,D_{21}),b_3),d_3).
$$
Let $p_{A_{21}},p_{B_{21}},p_{A_{321}},p_{B_{321}}$ be the collected nonzero
witnesses obtained from the six side-factor witnesses by the product-apartness
theorem
\autoref{thm:field-product-apartness-inverse-product-reverse} and the
available product-apartness conversion maps. Define the collected normal form
$$
C_{321}(x)=
\mathsf{add}(\mathsf{mul}(\mathsf{mul}(A_{321},x),B_{321}),D_{321}).
$$
For each $q:\mathsf{BHist}$ define the component inverse maps
$$

\end{proof}
