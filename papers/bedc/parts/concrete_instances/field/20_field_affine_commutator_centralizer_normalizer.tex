\begin{definition}[RatUp FieldUp affine centralizer normalizer carrier]
\label{def:ratup-fieldup-affine-centralizer-normalizer-carrier}
For Definition~\ref{def:ratup-fieldup-concrete-exit-object}, fix a certified
affine automorphism $A:\mathsf{AffAut}_F$ in the concrete
$\FieldUp(\RatUp)$ exit object, and let $C_F(A)$ be the carrier from
\autoref{def:ratup-fieldup-affine-centralizer-carrier}. The concrete affine
normalizer carrier of $C_F(A)$ is the predicate
$$
N_F(C_F(A))(T)
:\Longleftrightarrow
T:\mathsf{AffAut}_F
\quad\text{and}\quad
\forall X:\mathsf{AffAut}_F,\ C_F(A)(X)\Rightarrow
C_F(A)(\mathrm{Conj}_T(X)),
$$
where
$$
\mathrm{Conj}_T(X):=(T\circ_{\mathrm{aff}}X)\circ_{\mathrm{aff}}\iota_F(T)
$$
uses the affine product and inverse row fixed by
\autoref{thm:ratup-fieldup-canonical-exit-action-inverse-package}.
\end{definition}

\begin{theorem}[RatUp FieldUp affine centralizer normalizer inclusion]
\label{thm:ratup-fieldup-affine-centralizer-normalizer-inclusion}
For Definition~\ref{def:ratup-fieldup-concrete-exit-object}, fix a certified
affine automorphism $A:\mathsf{AffAut}_F$ in the concrete
$\FieldUp(\RatUp)$ exit object. Let $C_F(A)$ be the carrier from
\autoref{def:ratup-fieldup-affine-centralizer-carrier}, and let
$N_F(C_F(A))$ be the carrier from
\autoref{def:ratup-fieldup-affine-centralizer-normalizer-carrier}. Then every
centralizer element normalizes the centralizer:
$$
C_F(A)(T)\Rightarrow N_F(C_F(A))(T).
$$
Equivalently, if $T$ and $X$ commute with $A$, then the inner affine conjugate
of $X$ by $T$ also commutes with $A$:
$$
C_F(A)(T)\quad\text{and}\quad C_F(A)(X)
\quad\Rightarrow\quad
C_F(A)((T\circ_{\mathrm{aff}}X)\circ_{\mathrm{aff}}\iota_F(T)).
$$
\end{theorem}
\begin{proof}
Fix the canonical affine exit action from
\autoref{thm:ratup-fieldup-canonical-exit-action-inverse-package}. This fixes
$\mathsf{AffAut}_F$, $\circ_{\mathrm{aff}}$, $\mathrm{id}_F$, $\iota_F(-)$,
and $\equiv_{\mathrm{aff}}$ for the concrete $\FieldUp(\RatUp)$ exit object.

Assume $C_F(A)(T)$ and let $X$ satisfy $C_F(A)(X)$. Apply
\autoref{thm:ratup-fieldup-affine-centralizer-subgroup} to the fixed element
$A$. Its inverse closure gives $C_F(A)(\iota_F(T))$, and its product closure
first gives $C_F(A)(T\circ_{\mathrm{aff}}X)$ and then gives
$C_F(A)((T\circ_{\mathrm{aff}}X)\circ_{\mathrm{aff}}\iota_F(T))$.

The displayed product is the concrete inner conjugate
$\mathrm{Conj}_T(X)$ in the notation of
\autoref{def:ratup-fieldup-affine-centralizer-normalizer-carrier}. The carrier
certification and classifier stability of this inner conjugate are exactly the
action clauses supplied by
\autoref{thm:ratup-fieldup-affine-inner-conjugation-exit}.

Reading the final membership through
\autoref{thm:ratup-fieldup-affine-commutator-centralizer-exactness} gives the
equivalent commutator form: the conjugated element has trivial commutator with
$A$, hence it commutes with $A$ under the concrete affine classifier. This is
precisely $N_F(C_F(A))(T)$.
\end{proof}
