\begin{theorem}[Field apart-zero nested continuation empty-context exactness]
\label{thm:field-apartzero-nested-continuation-empty-context-iff}
For histories $l,h,r,u,v:\mathsf{BHist}$ with $\ContR(l,h,u)$ and
$\ContR(u,r,v)$, if $l\hsame\emp$ and $r\hsame\emp$, then
$$
\mathsf{ApartZero}_F(v)
\Longleftrightarrow
\mathsf{ApartZero}_F(h).
$$
\end{theorem}
\begin{proof}
The two continuation witnesses identify $u$ with $\mathsf{append}(l,h)$ and
$v$ with $\mathsf{append}(u,r)$. Continuation associativity and result
transport compare $v$ with the nested append endpoint
$\mathsf{append}(l,\mathsf{append}(h,r))$.

Transport $\mathsf{ApartZero}_F$ across this $\hsame$ comparison. The result
is exactly the bilateral empty-context statement of
\autoref{thm:field-apartzero-empty-context-iff}, applied with the witnesses
$l\hsame\emp$ and $r\hsame\emp$.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.FieldApartZero\_nested\_continuation\_empty\_context\_iff}

\begin{theorem}[Field apart-zero nested continuation factor split]
\label{thm:field-apartzero-nested-continuation-factor-split}
For histories $l,h,r,u,v,q,w:\mathsf{BHist}$ with $\ContR(l,h,u)$,
$\ContR(u,r,v)$, and $\ContR(v,q,w)$, if $l\hsame\emp$ and $r\hsame\emp$,
then
$$
\mathsf{ApartZero}_F(w)
\Longleftrightarrow
\mathsf{ApartZero}_F(h)\lor\mathsf{ApartZero}_F(q).
$$
\end{theorem}
\begin{proof}
Apply \autoref{thm:field-apartzero-continuation-endpoint-split} to the tail
continuation witness $\ContR(v,q,w)$. This gives
$\mathsf{ApartZero}_F(w)$ exactly when
$\mathsf{ApartZero}_F(v)\lor\mathsf{ApartZero}_F(q)$ holds.

Apply
\autoref{thm:field-apartzero-nested-continuation-empty-context-iff} to the
first two continuation witnesses and the exterior empty-context witnesses
$l\hsame\emp$ and $r\hsame\emp$. Substituting this equivalence for the first
disjunct yields the displayed factor split.
\end{proof}

\begin{theorem}[Field apart-zero two-tail nested continuation factor split]
\label{thm:field-apartzero-two-tail-nested-continuation-factor-split}
For histories $l,h,r,u,v,q,w,s,z:\mathsf{BHist}$ with $\ContR(l,h,u)$,
$\ContR(u,r,v)$, $\ContR(v,q,w)$, and $\ContR(w,s,z)$, if
$l\hsame\emp$ and $r\hsame\emp$, then
$$
\mathsf{ApartZero}_F(z)
\Longleftrightarrow
\mathsf{ApartZero}_F(h)\lor\mathsf{ApartZero}_F(q)\lor\mathsf{ApartZero}_F(s).
$$
\end{theorem}
\begin{proof}
Apply \autoref{thm:field-apartzero-continuation-endpoint-split} to the final
tail continuation witness $\ContR(w,s,z)$. This gives
$\mathsf{ApartZero}_F(z)$ exactly when
$\mathsf{ApartZero}_F(w)\lor\mathsf{ApartZero}_F(s)$ holds.

Apply \autoref{thm:field-apartzero-nested-continuation-factor-split} to the
first three continuation witnesses and the exterior witnesses $l\hsame\emp$
and $r\hsame\emp$. This identifies $\mathsf{ApartZero}_F(w)$ with
$\mathsf{ApartZero}_F(h)\lor\mathsf{ApartZero}_F(q)$.

For the forward direction, case-analyze the endpoint disjunction. The
$\mathsf{ApartZero}_F(s)$ case supplies the right tail disjunct. In the
$\mathsf{ApartZero}_F(w)$ case, case-analyze the nested split and place the
resulting witness in the $h$ or $q$ disjunct.

For the reverse direction, case-analyze the displayed three-way disjunction.
The $h$ and $q$ cases first rebuild $\mathsf{ApartZero}_F(w)$ by the reverse
direction of the nested split, and then rebuild $\mathsf{ApartZero}_F(z)$ by
the reverse direction of the endpoint split. The $s$ case uses the reverse
direction of the endpoint split directly.
\end{proof}

\begin{theorem}[Field apart-zero nested continuation singleton-result factor exclusion]
\label{thm:field-apartzero-nested-continuation-singleton-result-factor-exclusion}
For histories $l,h,r,u,v,q,w,s,z,endpoint:\mathsf{BHist}$ with
$\ContR(l,h,u)$, $\ContR(u,r,v)$, $\ContR(v,q,w)$, and $\ContR(w,s,z)$, if
$l\hsame\emp$, $r\hsame\emp$, and
$\mathsf{FieldSingletonClassifier}(z,endpoint)$, then
$$
(\mathsf{ApartZero}_F(h)\to\bot)
\land
(\mathsf{ApartZero}_F(q)\to\bot)
\land
(\mathsf{ApartZero}_F(s)\to\bot).
$$
\end{theorem}
\begin{proof}
Apply
\autoref{thm:field-apartzero-continuation-singleton-result-factor-absurd} to
the final continuation witness $\ContR(w,s,z)$ together with the terminal
singleton-result certificate. This gives exclusions for
$\mathsf{ApartZero}_F(w)$ and $\mathsf{ApartZero}_F(s)$.

To exclude $\mathsf{ApartZero}_F(h)$, assume such a witness. The reverse
direction of
\autoref{thm:field-apartzero-nested-continuation-factor-split}, applied to the
first three continuation witnesses and the exterior empty-context witnesses,
turns the left factor witness into $\mathsf{ApartZero}_F(w)$. This contradicts
the exclusion for $w$.

The argument for $\mathsf{ApartZero}_F(q)$ is the same, using the right nested
factor witness in the reverse direction of the same split theorem. Pair the two
nested exclusions with the already obtained exclusion for $s$ to obtain the
displayed triple exclusion.
\end{proof}

\mathsf{ApartZero}_F(h)\lor\mathsf{ApartZero}_F(q)\lor

\begin{proof}

$\mathsf{ApartZero}_F(y)$ into
$\mathsf{ApartZero}_F(z)\lor\mathsf{ApartZero}_F(t)$.

\autoref{thm:field-apartzero-two-tail-nested-continuation-factor-split} to
the first four continuation witnesses and the exterior empty-context
witnesses $l\hsame\emp$ and $r\hsame\emp$. This rewrites
$\mathsf{ApartZero}_F(z)$ as the three-way disjunction on $h,q,s$.

For the forward direction, case-analyze the endpoint split. The
$\mathsf{ApartZero}_F(t)$ case supplies the final tail disjunct. In the
$\mathsf{ApartZero}_F(z)$ case, case-analyze the two-tail nested split and

For the reverse direction, case-analyze the displayed four-way disjunction.
The $h$, $q$, and $s$ cases first rebuild $\mathsf{ApartZero}_F(z)$ by the

$\mathsf{ApartZero}_F(y)$ by the reverse direction of the endpoint split. The

\begin{theorem}[Field apart-zero three-tail contextual singleton result exclusion]
\label{thm:field-apartzero-three-tail-contextual-singleton-result-exclusion}
For histories $L,R,l,h,r,u,v,q,w,s,z,t,y,endpoint:\mathsf{BHist}$ with
$\mathsf{FieldSingletonCarrier}(L)$, $\mathsf{FieldSingletonCarrier}(R)$,
$\ContR(l,h,u)$, $\ContR(u,r,v)$, $\ContR(v,q,w)$, $\ContR(w,s,z)$,
$\ContR(z,t,y)$, $l\hsame\emp$, $r\hsame\emp$, and
$\mathsf{FieldSingletonClassifier}(\mathsf{append}(L,y),\mathsf{append}(R,endpoint))$,
one has
$$
(\mathsf{ApartZero}_F(h)\to\bot)
\land
(\mathsf{ApartZero}_F(q)\to\bot)
\land
(\mathsf{ApartZero}_F(s)\to\bot)
\land
(\mathsf{ApartZero}_F(t)\to\bot).
$$
\end{theorem}
\begin{proof}
Use \autoref{thm:field-singleton-classifier-append-context-cancel} with the
singleton-carrier witnesses for $L$ and $R$. The forward direction of the
cancellation equivalence turns the contextual classifier into
$\mathsf{FieldSingletonClassifier}(y,endpoint)$.

Apply \autoref{thm:field-singleton-classifier-empty-endpoints} to that
singleton endpoint classifier. The resulting endpoint comparison gives
$y\hsame\emp$.

Now assume one of the four factor witnesses
$\mathsf{ApartZero}_F(h)$, $\mathsf{ApartZero}_F(q)$,
$\mathsf{ApartZero}_F(s)$, or $\mathsf{ApartZero}_F(t)$. The reverse direction
of
\autoref{thm:field-apartzero-three-tail-nested-continuation-factor-split},
applied to the corresponding disjunct and the five continuation witnesses,
rebuilds $\mathsf{ApartZero}_F(y)$.

Applying this apart-zero witness to $y\hsame\emp$ gives $\bot$ in each case.
Pairing the four implications gives the displayed exclusion.
\end{proof}

\begin{definition}[Field apart-zero tail-factor spine]
\label{def:field-apartzero-tail-factor-spine}
For histories $y,x:\mathsf{BHist}$, let
$\mathsf{FieldApartTailFactor}(y,x)$ be the least relation on
$\mathsf{BHist}$ generated by the following constructors:
\begin{enumerate}
\item If $\ContR(l,h,u)$, $\ContR(u,r,v)$, $l\hsame\emp$, and
$r\hsame\emp$, then $\mathsf{FieldApartTailFactor}(v,h)$.
\item If $\mathsf{FieldApartTailFactor}(w,x)$ and $\ContR(w,q,z)$, then
$\mathsf{FieldApartTailFactor}(z,x)$.
\item If $\ContR(w,q,z)$, then $\mathsf{FieldApartTailFactor}(z,q)$.
\end{enumerate}
\end{definition}

\begin{lemma}[Field apart-zero tail-factor empty endpoint exclusion]
\label{lem:field-apartzero-tail-factor-empty-endpoint-exclusion}
For histories $y,x:\mathsf{BHist}$, if
$\mathsf{FieldApartTailFactor}(y,x)$, $y\hsame\emp$, and
$\mathsf{ApartZero}_F(x)$, then $\bot$.
\end{lemma}
\begin{proof}
Induct on the generated tail-factor witness. In the empty-context generator,
the nested empty-context apart-zero equivalence transports
$\mathsf{ApartZero}_F(x)$ to the result endpoint, contradicting
$y\hsame\emp$. In the carried-tail generator, empty-result inversion for the
last continuation gives an empty previous endpoint, so the induction
hypothesis applies. In the exposed-factor generator, empty-result inversion
gives the exposed factor itself as $\emp$, contradicting
$\mathsf{ApartZero}_F(x)$.
\end{proof}

\begin{lemma}[Field apart-zero tail-factor hsame transport]
\label{lem:field-apartzero-tail-factor-hsame-transport}
For histories $y,y',x,x':\mathsf{BHist}$, if
$\mathsf{FieldApartTailFactor}(y,x)$, $y\hsame y'$, and $x\hsame x'$, then
$\mathsf{FieldApartTailFactor}(y',x')$.
\end{lemma}
\begin{proof}
Induct on the generated witness
$\mathsf{FieldApartTailFactor}(y,x)$.

In the empty-context constructor, transport the first continuation witness along
$x\hsame x'$ and reflexivity of its other endpoints using
\autoref{thm:cont-hsame-transport}. Transport the second continuation witness
along reflexivity of its source and factor endpoint, and along $y\hsame y'$ for
the result. The two exterior witnesses $l\hsame\emp$ and $r\hsame\emp$ are
unchanged, so constructor 1 rebuilds
$\mathsf{FieldApartTailFactor}(y',x')$.

In the carried-tail constructor, apply the induction hypothesis to the stored
factor witness, the reflexive comparison on its carried result, and the supplied
$x\hsame x'$. Transport the final continuation result along $y\hsame y'$ by
\autoref{thm:cont-hsame-transport}, then reuse constructor 2.

In the exposed-factor constructor, transport the continuation argument by
$x\hsame x'$ and its result by $y\hsame y'$ using
\autoref{thm:cont-hsame-transport}. Constructor 3 then gives the transported
generated factor witness.
\end{proof}

\begin{lemma}[Field apart-zero tail-factor result closure]
\label{lem:field-apartzero-tail-factor-result-closure}
For histories $y,x:\mathsf{BHist}$, if
$\mathsf{FieldApartTailFactor}(y,x)$, then
$$
\mathsf{ApartZero}_F(x)\Rightarrow\mathsf{ApartZero}_F(y).
$$
\end{lemma}
\begin{proof}
Induct on the generated witness
$\mathsf{FieldApartTailFactor}(y,x)$.

In the initial empty-context constructor, apply
\autoref{thm:field-apartzero-nested-continuation-empty-context-iff} to the
two continuation witnesses and the two exterior comparisons. The reverse
direction transports $\mathsf{ApartZero}_F(h)$ to
$\mathsf{ApartZero}_F(v)$.

In the carried-tail constructor, the induction hypothesis first transports
$\mathsf{ApartZero}_F(x)$ to $\mathsf{ApartZero}_F(w)$. Then apply the
reverse direction of
\autoref{thm:field-apartzero-continuation-endpoint-split} to the left
endpoint disjunct, obtaining $\mathsf{ApartZero}_F(z)$.

In the exposed-factor constructor, apply the reverse direction of
\autoref{thm:field-apartzero-continuation-endpoint-split} to the right factor
disjunct. This sends $\mathsf{ApartZero}_F(q)$ to
$\mathsf{ApartZero}_F(z)$.
\end{proof}

\begin{theorem}[Field apart-zero three-tail tail-factor coverage]
\label{thm:field-apartzero-three-tail-tail-factor-coverage}
For histories $l,h,r,u,v,q,w,s,z,t,y:\mathsf{BHist}$ with
$\ContR(l,h,u)$, $\ContR(u,r,v)$, $\ContR(v,q,w)$, $\ContR(w,s,z)$, and
$\ContR(z,t,y)$, if $l\hsame\emp$ and $r\hsame\emp$, then
$$
\mathsf{FieldApartTailFactor}(y,h)
\land
\mathsf{FieldApartTailFactor}(y,q)
\land
\mathsf{FieldApartTailFactor}(y,s)
\land
\mathsf{FieldApartTailFactor}(y,t).
$$
\end{theorem}
\begin{proof}
We first build the hidden initial factor. The initial tail-factor constructor
applied to $\ContR(l,h,u)$, $\ContR(u,r,v)$, $l\hsame\emp$, and
$r\hsame\emp$ gives $\mathsf{FieldApartTailFactor}(v,h)$. Carry this witness
successively across $\ContR(v,q,w)$, $\ContR(w,s,z)$, and $\ContR(z,t,y)$ by
the carried-tail constructor to obtain $\mathsf{FieldApartTailFactor}(y,h)$.

Next expose the $q$ factor at the first continuation after the initial
empty-context block. The exposed-factor constructor applied to
$\ContR(v,q,w)$ gives $\mathsf{FieldApartTailFactor}(w,q)$. Carry it across
$\ContR(w,s,z)$ and $\ContR(z,t,y)$ to obtain
$\mathsf{FieldApartTailFactor}(y,q)$.

The $s$ factor is exposed in the same generated spine at the next step.
Apply the exposed-factor constructor to $\ContR(w,s,z)$, giving
$\mathsf{FieldApartTailFactor}(z,s)$, and then carry this witness across
$\ContR(z,t,y)$ to obtain $\mathsf{FieldApartTailFactor}(y,s)$.

Finally, apply the exposed-factor constructor directly to $\ContR(z,t,y)$.
This gives $\mathsf{FieldApartTailFactor}(y,t)$. Pair the four generated
witnesses in the displayed order.
\end{proof}

Per-object template: traditional target finite product factor coverage; BEDC

carrier generated tail-factor witnesses; classifier $\hsame$; ledger the
exposed factors $h,q,s,t$; core theorem this block plus
