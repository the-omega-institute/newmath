\begin{theorem}[RatUp FieldUp affine commutator fixedness criterion]

$T,S:\mathsf{AffAut}_F$ be certified affine automorphisms in the concrete
$\FieldUp(\RatUp)$ exit object. With the affine group instance of
Definition~\ref{def:ratup-fieldup-affine-automorphism-group-instance}, define
$$
\mathsf{AffComm}_F(T,S):=(T\circ_{\mathrm{aff}}S)\circ_{\mathrm{aff}}
(\iota_F(T)\circ_{\mathrm{aff}}\iota_F(S)).
$$
Then the affine commutator is trivial exactly when the two affine
automorphisms commute:
$$
\mathsf{AffComm}_F(T,S)\equiv_{\mathrm{aff}}\mathrm{id}_F
\Longleftrightarrow
T\circ_{\mathrm{aff}}S\equiv_{\mathrm{aff}}S\circ_{\mathrm{aff}}T.
$$
Equivalently, the same commutator is trivial exactly when $S$ is fixed by
inner conjugation by $T$:
$$
\mathsf{AffComm}_F(T,S)\equiv_{\mathrm{aff}}\mathrm{id}_F
\Longleftrightarrow
\mathrm{Inn}_T(S)\equiv_{\mathrm{aff}}S.
$$
\end{theorem}
\begin{proof}

Definition~\ref{def:ratup-fieldup-affine-automorphism-group-instance}. It
reads the group carrier as $\mathsf{AffAut}_F$, the product as
$\circ_{\mathrm{aff}}$, the unit as $\mathrm{id}_F$, the inverse as
$\iota_F(-)$, and the classifier as $\equiv_{\mathrm{aff}}$.

concrete group package with $a=T$ and $b=S$. Its history-level commutator
becomes the displayed term $\mathsf{AffComm}_F(T,S)$, and its classifier
statement becomes the first displayed equivalence.

It remains to compare commutation with fixedness under inner conjugation.
Expand $\mathrm{Inn}_T(S)$ as
$T\circ_{\mathrm{aff}}S\circ_{\mathrm{aff}}\iota_F(T)$. If
$T\circ_{\mathrm{aff}}S\equiv_{\mathrm{aff}}S\circ_{\mathrm{aff}}T$, compose
the comparison on the right by $\iota_F(T)$, reassociate, and collapse
$T\circ_{\mathrm{aff}}\iota_F(T)$ to $\mathrm{id}_F$. The right identity law

$\iota_F(T)\circ_{\mathrm{aff}}T$ to $\mathrm{id}_F$. The right identity law

identity laws from the affine action certificate. The cancellation steps are
the concrete common-factor exactness clauses of
\autoref{thm:ratup-fieldup-affine-automorphism-cancellation-exactness-exit}.
\end{proof}

\begin{theorem}[RatUp FieldUp affine centralizers are conjugation-equivariant]
\label{thm:ratup-fieldup-affine-centralizer-conjugation-equivariance}
For Definition~\ref{def:ratup-fieldup-concrete-exit-object}, let
$A,T,X:\mathsf{AffAut}_F$ be certified affine automorphisms in the concrete
$\FieldUp(\RatUp)$ exit object. Define the inner affine conjugation action by
$$
\mathrm{Inn}_T(Y)=(T\circ_{\mathrm{aff}}Y)\circ_{\mathrm{aff}}\iota_F(T).
$$
Let $C_F(A)(X)$ be the centralizer membership predicate of
Definition~\ref{def:ratup-fieldup-affine-centralizer-carrier}. Then
centralizer membership is transported by inner affine conjugation:
$$
C_F(A)(X)
\Longleftrightarrow
C_F(\mathrm{Inn}_T(A))(\mathrm{Inn}_T(X)).
$$
Equivalently,
$$
X\circ_{\mathrm{aff}}A\equiv_{\mathrm{aff}}A\circ_{\mathrm{aff}}X
\Longleftrightarrow
\mathrm{Inn}_T(X)\circ_{\mathrm{aff}}\mathrm{Inn}_T(A)
\equiv_{\mathrm{aff}}
\mathrm{Inn}_T(A)\circ_{\mathrm{aff}}\mathrm{Inn}_T(X).
$$
\end{theorem}
\begin{proof}
Fix the canonical affine group package from
\autoref{thm:ratup-fieldup-canonical-exit-action-inverse-package}. This fixes
$\mathsf{AffAut}_F$, $\circ_{\mathrm{aff}}$, $\mathrm{id}_F$, $\iota_F(-)$,
and $\equiv_{\mathrm{aff}}$, and
\autoref{thm:ratup-fieldup-affine-automorphism-action-certificate} supplies
the associativity, inverse, cancellation, and classifier congruence rows used
below.

For the forward direction, assume $C_F(A)(X)$, namely
$X\circ_{\mathrm{aff}}A\equiv_{\mathrm{aff}}A\circ_{\mathrm{aff}}X$.
Left-compose and right-compose this classifier equation by the conjugating
factor $T$ and its inverse witness $\iota_F(T)$. The classifier congruence row
keeps equality stable under these certified affine products.

Reassociate the two transported composites by
\autoref{thm:field-affine-automorphism-composition-associative}. The adjacent
factors $\iota_F(T)\circ_{\mathrm{aff}}T$ collapse by the inverse law in
\autoref{thm:field-affine-automorphism-group-laws}, and the identity rows
leave exactly
$$
\mathrm{Inn}_T(X)\circ_{\mathrm{aff}}\mathrm{Inn}_T(A)
\equiv_{\mathrm{aff}}
\mathrm{Inn}_T(A)\circ_{\mathrm{aff}}\mathrm{Inn}_T(X).
$$
This is $C_F(\mathrm{Inn}_T(A))(\mathrm{Inn}_T(X))$ by
Definition~\ref{def:ratup-fieldup-affine-centralizer-carrier}. The same
normal form is the concrete affine conjugation exactness asserted by
\autoref{thm:ratup-fieldup-affine-inner-conjugation-exit}.

For the reverse direction, apply the forward construction with
$\iota_F(T)$ as the conjugating automorphism to the centralizer witness for
$\mathrm{Inn}_T(A)$ and $\mathrm{Inn}_T(X)$. The inverse antihomomorphism and
double-inverse clauses of
\autoref{thm:ratup-fieldup-affine-inverse-antihom-exit} identify the resulting
double conjugates with $A$ and $X$ under $\equiv_{\mathrm{aff}}$.

Finally use classifier transitivity and symmetry to replace the double
conjugate centralizer equation by
$X\circ_{\mathrm{aff}}A\equiv_{\mathrm{aff}}A\circ_{\mathrm{aff}}X$. This is
precisely $C_F(A)(X)$, so the two displayed centralizer-membership conditions
are equivalent.
\end{proof}
