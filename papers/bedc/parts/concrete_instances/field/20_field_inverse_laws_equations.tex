\begin{theorem}[Field right-multiplication empty equation solutions are unique from apartness]
\label{thm:field-right-mul-empty-equation-solution-unique-from-apartness}
For a BEDC history-level $\FieldUp$ candidate with proof-indexed inverses, if
$p_b:\mathsf{NonZero}_F(b)$ and both $x$ and $y$ solve the empty endpoint
right-multiplication equation
$$
x b\hsame \emp
\qquad\text{and}\qquad
y b\hsame \emp,
$$
then $x\hsame y$.
\end{theorem}
\begin{proof}
Apply the right-multiplication solution theorem to both displayed equations at
the concrete endpoint $\emp$. Both solutions are classified with the same
normal form $\emp\,\mathsf{inv}(b,p_b)$. Symmetry on the second classification
and transitivity of $\hsame$ identify $x$ and $y$.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.field\_right\_mul\_empty\_equation\_solution\_unique\_from\_apartness}

\begin{theorem}[Field multiplication by an indexed left inverse cancels]
\label{thm:field-mul-inverse-left-cancel-from-apartness}
For a BEDC history-level $\FieldUp$ candidate with proof-indexed inverses,
any witness $p_a:\mathsf{NonZero}_F(a)$ determines
$$
\mathsf{inv}(a,p_a)(a b)\hsame b.
$$
\end{theorem}
\begin{proof}
Reassociate $\mathsf{inv}(a,p_a)(a b)$ to
$(\mathsf{inv}(a,p_a)a)b$. The apartness-indexed left inverse law collapses
the head to the unit, and the left identity law leaves $b$.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.field\_mul\_inverse\_left\_cancel\_from\_apartness}

$\mathsf{inv}(a,p_a)(a x)\hsame \mathsf{inv}(a,p_a)c$.

any witness $p_a:\mathsf{NonZero}_F(a)$ makes the left-multiplication
equation and its displayed solution equivalent:
$$

\Longleftrightarrow
x\hsame \mathsf{inv}(a,p_a)c.
$$
\end{theorem}
\begin{proof}
For the forward direction, apply
Theorem~\ref{thm:field-left-mul-equation-solution-from-apartness}. This is

multiplication by $a$. The transported right side is
$a(\mathsf{inv}(a,p_a)c)$.

Reassociate the factors as $(a\mathsf{inv}(a,p_a))c$. The indexed right
inverse law collapses $a\mathsf{inv}(a,p_a)$ to the unit, and the left
identity law leaves $c$.
\end{proof}
