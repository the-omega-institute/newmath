\begin{theorem}[Rat denominator context-pair product strict support]
\label{thm:field-rat-denominator-context-pair-product-strict-support}
For carried denominator context pairs $(l,r)$ and $(l',r')$, strict support of
their product is exactly the disjunction of the four component supports:
$$
N_D^u(\mu_D^u(\mu_D^u(l',l),\mu_D^u(r,r')))
\Longleftrightarrow
N_D^u(l')\lor N_D^u(l)\lor N_D^u(r)\lor N_D^u(r').
$$
\end{theorem}
\begin{proof}
Apply bilateral strict-support exactness to the left product
$\mu_D^u(l',l)$ and the carried right product $\mu_D^u(r,r')$. The remaining
right-product support term is then opened by denominator product support
exactness for $r$ and $r'$. Reassociating the resulting disjunction yields the
four displayed component supports, and the reverse implication closes the same
two product-support equivalences in the opposite order.
\end{proof}

\begin{theorem}[Rat denominator context-pair product strict support transport]
\label{thm:field-rat-denominator-context-pair-product-strict-support-transport}
If $(l,r)$ and $(s,t)$ are componentwise $R_D^u$-classified, and
$(l',r')$ and $(s',t')$ are componentwise $R_D^u$-classified, then strict
support is invariant across the two product endpoints:
$$
N_D^u(\mu_D^u(\mu_D^u(l',l),\mu_D^u(r,r')))
\Longleftrightarrow
N_D^u(\mu_D^u(\mu_D^u(s',s),\mu_D^u(t,t'))).
$$
\end{theorem}
\begin{proof}
The component classifiers expose $\HSame(l,s)$, $\HSame(r,t)$,
$\HSame(l',s')$, and $\HSame(r',t')$. Continuation congruence composes these
through the two product layers, giving an $\HSame$ witness between the two
displayed product endpoints. Rat denominator support transports across that
endpoint witness in both directions.
\end{proof}

\begin{theorem}[RatUp FieldUp denominator context-pair product certificate row]
\label{thm:ratup-fieldup-denominator-context-pair-product-certificate-row}
For the concrete denominator context-pair structure of
\autoref{def:ratup-fieldup-denominator-context-pair-monoid}, let $(l,r)$ and
$(l',r')$ be carried denominator context pairs. Let $(m,s)$ and $(m',s')$ be
carried comparison pairs with
$\mathsf{PairClass}_D^u((l,r),(m,s))$ and
$\mathsf{PairClass}_D^u((l',r'),(m',s'))$. Then the concrete product
$(l,r)\star(l',r')$ preserves carried membership, respects the componentwise
$R_D^u$ classifiers against $(m,s)\star(m',s')$, and has bundled strict support
exactly
$$
N_D^u(\mu_D^u(\mu_D^u(l',l),\mu_D^u(r,r')))
\Longleftrightarrow
\mathsf{PairSupp}_D^u(l,r)\lor\mathsf{PairSupp}_D^u(l',r').
$$
The transported product endpoint has the same bundled strict support:
$$
N_D^u(\mu_D^u(\mu_D^u(l',l),\mu_D^u(r,r')))
\Longleftrightarrow
N_D^u(\mu_D^u(\mu_D^u(m',m),\mu_D^u(s,s'))).
$$
Reversing the two acting pairs preserves the component classifiers,
$$
R_D^u(\mu_D^u(l',l),\mu_D^u(l,l'))\quad\text{and}\quad
R_D^u(\mu_D^u(r,r'),\mu_D^u(r',r)),
$$
and the reversed bundled support has the same readback:
$$
N_D^u(\mu_D^u(\mu_D^u(l,l'),\mu_D^u(r',r)))
\Longleftrightarrow
\mathsf{PairSupp}_D^u(l,r)\lor\mathsf{PairSupp}_D^u(l',r').
$$
\end{theorem}
\begin{proof}
Apply
\autoref{thm:ratup-fieldup-denominator-context-pair-product-operation-stable}.
Its carrier clause gives carried membership for the two components of
$(l,r)\star(l',r')$, and its classifier clause gives componentwise $R_D^u$
respect against $(m,s)\star(m',s')$.

The strict support of the concrete product is the support clause of the same
operation row. Equivalently, apply
\autoref{thm:field-rat-denominator-context-pair-product-strict-support} and
regroup the four component supports into
$\mathsf{PairSupp}_D^u(l,r)\lor\mathsf{PairSupp}_D^u(l',r')$.

The comparison endpoint has the same strict support by
\autoref{thm:field-rat-denominator-context-pair-product-strict-support-transport},
using the four component classifiers supplied by the two pair-class
hypotheses.

For the reversed product, apply
\autoref{thm:ratup-fieldup-denominator-context-pair-commutative-support}. It
gives the displayed left and right component classifiers. Applying the same
commutative-support theorem with the acting pairs exchanged gives the reversed
support readback, and commutativity of disjunction identifies it with the
displayed pair-support disjunction.
\end{proof}
