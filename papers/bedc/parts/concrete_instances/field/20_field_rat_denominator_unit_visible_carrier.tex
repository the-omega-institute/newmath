\begin{theorem}[Rat denominator unit carrier zero-headed absurd]
\label{thm:rat-denom-unit-carrier-e0-absurd}
For every tail $t:\mathsf{BHist}$, the empty-unit denominator carrier cannot
hold on a zero-headed history:
$$
\mathsf{RatDenomUnitCarrier}(\Ezero(t))\Longrightarrow \bot.
$$
\end{theorem}
\leanchecked{BEDC.Derived.FieldUp.RatDenomUnitCarrier\_e0\_absurd}
\begin{proof}
Unfold the empty-unit denominator carrier. Its empty-endpoint branch would make
$\Ezero(t)$ equal to $\emp$, contradicting constructor disjointness. Its Rat
branch is impossible because the Rat denominator carrier has no zero-headed
inhabitant.
\end{proof}

\begin{theorem}[Rat denominator unit carrier one-headed unary iff]
\label{thm:field-rat-denom-unit-carrier-one-headed-unary-iff}
For every tail $t:\mathsf{BHist}$, the empty-unit denominator carrier sees a
one-headed history exactly when the tail is unary:
$$
\mathsf{RatDenomUnitCarrier}(\Eone(t))
\Longleftrightarrow
\mathsf{UnaryHistory}(t).
$$
\end{theorem}
\leanchecked{BEDC.Derived.FieldUp.RatDenomUnitCarrier\_e1\_tail\_unary\_iff}
\begin{proof}
Unfold the empty-unit denominator carrier. Its empty-endpoint branch would give
$\Eone(t)\hsame\emp$, contradicting constructor disjointness. The remaining
branch is $\mathsf{RatHistoryCarrier}(\Eone(t))$, which is equivalent to
$\mathsf{UnaryHistory}(t)$ by the one-headed Rat carrier characterization.
\end{proof}

\begin{theorem}[Rat denominator unit classifier one-headed tail iff]
\label{thm:rat-denom-unit-classifier-e1-tail-iff}
For histories $d,e:\mathsf{BHist}$, the empty-unit denominator classifier on
two one-headed histories is exactly the pair of unary tail witnesses together
with tail sameness:
$$
\mathsf{RatDenomUnitClassifier}(\Eone(d),\Eone(e))
\Longleftrightarrow
\mathsf{UnaryHistory}(d)\wedge
\mathsf{UnaryHistory}(e)\wedge d\hsame e.
$$
\end{theorem}
\leanchecked{BEDC.Derived.FieldUp.RatDenomUnitClassifier\_e1\_tail\_iff}
\begin{proof}
Unfold the classifier into its two empty-unit carrier fields and the endpoint
sameness field. Each carrier field is converted by the one-headed carrier
characterization above, and the endpoint sameness field is converted by
constructor injectivity for $\Eone$.
\end{proof}

\begin{theorem}[Rat denominator unit classifier visible cases iff]
\label{thm:rat-denom-unit-classifier-visible-cases-iff}
For histories $h,k:\mathsf{BHist}$, the empty-unit denominator classifier is
equivalent to either the empty--empty branch or a matched pair of one-headed
unary denominator tails:
$$
\begin{aligned}
&\mathsf{RatDenomUnitClassifier}(h,k)\\
&\Longleftrightarrow
\left(h\hsame\emp\land k\hsame\emp\right)\\
&\quad{}\lor
\exists h_t,k_t:\mathsf{BHist}.\;
h=\Eone(h_t)\land k=\Eone(k_t)\land \mathsf{UnaryHistory}(h_t)\\
&\qquad{}\land \mathsf{UnaryHistory}(k_t)\land h_t\hsame k_t.
\end{aligned}
$$
\end{theorem}
\begin{proof}
Unpack the classifier into two carrier witnesses and an endpoint comparison.
Each carrier is split by the visible carrier equivalence. Mixed empty and
one-headed endpoints contradict constructor disjointness. The one-headed pair
branch converts endpoint sameness through $\Eone$ injectivity, and the reverse
direction rebuilds the two carriers and the classifier comparison from the
displayed tail data.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.RatDenomUnitClassifier\_visible\_cases\_iff}
