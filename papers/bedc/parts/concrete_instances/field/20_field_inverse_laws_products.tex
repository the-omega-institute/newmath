\begin{theorem}[Field inverse product reversal from apartness-indexed inverses]
\label{thm:field-inverse-product-reverse-from-apartness}
For a BEDC history-level $\FieldUp$ candidate with proof-indexed inverses,
explicit witnesses $p_{ab}:\mathsf{NonZero}_F(ab)$,
$p_a:\mathsf{NonZero}_F(a)$, and $p_b:\mathsf{NonZero}_F(b)$ determine
the product inverse by
$$
\mathsf{inv}(ab,p_{ab})\hsame \mathsf{inv}(b,p_b)\mathsf{inv}(a,p_a).
$$
\end{theorem}
\begin{proof}
The reversed product $\mathsf{inv}(b,p_b)\mathsf{inv}(a,p_a)$ is a right
inverse of $ab$: reassociate to expose $b\mathsf{inv}(b,p_b)$, collapse it
to the unit, and finish with the right inverse law for $a$. The indexed left
inverse of $ab$ and group uniqueness of left and right inverse witnesses
identify the chosen inverse of $ab$ with this reversed product.
\end{proof}

\begin{theorem}[Field inverse product reversal right inverse excludes empty]
\label{thm:field-inverse-product-reverse-right-inverse-apart-empty-from-one-apartness}
For a BEDC history-level $\FieldUp$ candidate with proof-indexed inverses,
assume associativity, left identity, multiplication congruence, indexed right
inverse laws, and one-apartness from empty. For
$p_a:\mathsf{NonZero}_F(a)$ and $p_b:\mathsf{NonZero}_F(b)$,
$$
\mathsf{mul}(\mathsf{mul}(a,b),
  \mathsf{mul}(\mathsf{inv}(b,p_b),\mathsf{inv}(a,p_a)))\hsame0\to\bot.
$$
\end{theorem}
\begin{proof}
Reassociate the product to expose
$b\mathsf{inv}(b,p_b)$ inside the middle factor. The right inverse law for
$b$ reduces that middle product to $\mathsf{inv}(a,p_a)$ by the left identity,
and multiplication congruence then reduces the whole expression to
$a\mathsf{inv}(a,p_a)$. The right inverse law for $a$ identifies the product
with one, so any classifier equality from the product to empty would transport
one to empty, contradicting one-apartness.
\end{proof}

$$
\mathsf{NonZero}_F(
  \mathsf{mul}(\mathsf{inv}(b,p_b),\mathsf{inv}(a,p_a))).

\autoref{thm:field-inverse-product-reverse-from-apartness} identifies this
inverse with the reversed product
$\mathsf{mul}(\mathsf{inv}(b,p_b),\mathsf{inv}(a,p_a))$.
Transporting the nonzero witness across that classifier equality gives the
displayed witness.
\end{proof}

\begin{theorem}[Field inverse product reversal apartness transports across products]
\label{thm:field-inverse-product-reverse-apart-empty-transport-from-product-hsame}
For a BEDC history-level $\FieldUp$ candidate with proof-indexed inverses,
suppose
$$
\mathsf{mul}(a,b)\hsame\mathsf{mul}(a',b').
$$
Given nonzero witnesses for both products and for all four factors, apartness

$\mathsf{mul}(\mathsf{inv}(b,p_b),\mathsf{inv}(a,p_a))$ transports to

\autoref{thm:field-inverse-product-reverse-from-apartness} identifies each

product as empty would classify the source reversed product as empty and
contradict the supplied source apartness.
\end{proof}

\begin{theorem}[Field inverse product classifier exactness from apartness]
\label{thm:field-inverse-product-classifier-exact-from-apartness}
For a BEDC history-level $\FieldUp$ candidate over $\mathsf{BHist}$ satisfying
the hypotheses of
\autoref{thm:field-product-nonzero-of-nonzero-factors},
\autoref{thm:field-inverse-classifier-exact-from-apartness}, and
\autoref{thm:field-inverse-product-reverse-from-apartness}, assume a bridge
$$
\alpha:\bigl(h\hsame0\to\bot\bigr)\to\mathsf{NonZero}_F(h)
$$
and witnesses $p_a:\mathsf{NonZero}_F(a)$,
$p_b:\mathsf{NonZero}_F(b)$, $p_c:\mathsf{NonZero}_F(c)$, and
$p_d:\mathsf{NonZero}_F(d)$. Let $p_{ab}$ and $p_{cd}$ be the product nonzero
witnesses for $\mathsf{mul}(a,b)$ and $\mathsf{mul}(c,d)$ supplied by
\autoref{thm:field-product-nonzero-of-nonzero-factors}. Then
$$
\mathsf{mul}(a,b)\hsame\mathsf{mul}(c,d) \Longleftrightarrow
\mathsf{mul}(\mathsf{inv}(b,p_b),\mathsf{inv}(a,p_a))\hsame
\mathsf{mul}(\mathsf{inv}(d,p_d),\mathsf{inv}(c,p_c)).
$$
\end{theorem}
\begin{proof}
First build the product witnesses. Apply
\autoref{thm:field-product-nonzero-of-nonzero-factors} to $p_a$ and $p_b$,
using the bridge $\alpha$, to obtain
$p_{ab}:\mathsf{NonZero}_F(\mathsf{mul}(a,b))$. Apply the same theorem to
$p_c$ and $p_d$ to obtain
$p_{cd}:\mathsf{NonZero}_F(\mathsf{mul}(c,d))$.

For the forward direction, apply
\autoref{thm:field-inverse-classifier-exact-from-apartness} to the two product
endpoints with witnesses $p_{ab}$ and $p_{cd}$. The resulting inverse equality
has endpoints $\mathsf{inv}(\mathsf{mul}(a,b),p_{ab})$ and
$\mathsf{inv}(\mathsf{mul}(c,d),p_{cd})$. Apply
\autoref{thm:field-inverse-product-reverse-from-apartness} on both sides, and
transport across these two reversal equalities by symmetry and transitivity of
$\hsame$.

For the reverse direction, start with the displayed reversed-product equality.
Use the same two product reversal equalities, now in the opposite transport
orientation, to obtain
$\mathsf{inv}(\mathsf{mul}(a,b),p_{ab})\hsame
\mathsf{inv}(\mathsf{mul}(c,d),p_{cd})$. The reverse implication of
\autoref{thm:field-inverse-classifier-exact-from-apartness} then gives
$\mathsf{mul}(a,b)\hsame\mathsf{mul}(c,d)$.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.field\_inverse\_product\_classifier\_exact\_from\_apartness}

\begin{theorem}[Field inverse three-factor product reversal from apartness-indexed inverses]
\label{thm:field-inverse-three-factor-product-reversal-from-apartness}

multiplication congruence, left and right identity, proof-indexed inverse laws,
product nonzero exclusion, and conversion maps
$$
\alpha_{ab}:\mathsf{ApartZero}_F(\mathsf{mul}(a,b))\to
\mathsf{NonZero}_F(\mathsf{mul}(a,b)),
\quad
\alpha_a:\mathsf{ApartZero}_F(a)\to\mathsf{NonZero}_F(a),
$$
$$
\alpha_b:\mathsf{ApartZero}_F(b)\to\mathsf{NonZero}_F(b),
\quad
\alpha_c:\mathsf{ApartZero}_F(c)\to\mathsf{NonZero}_F(c),
$$
where $\mathsf{ApartZero}_F(h):=(h\hsame0\to\bot)$, any witness
$$
p_{abc}:\mathsf{NonZero}_F(\mathsf{mul}(\mathsf{mul}(a,b),c))
$$
induces witnesses $p_{ab}$, $p_a$, $p_b$, and $p_c$ such that
$$
\mathsf{inv}(\mathsf{mul}(\mathsf{mul}(a,b),c),p_{abc})\hsame
\mathsf{mul}(\mathsf{inv}(c,p_c),
\mathsf{mul}(\mathsf{inv}(b,p_b),\mathsf{inv}(a,p_a))).
$$
\end{theorem}
\begin{proof}
Apply \autoref{thm:field-product-nonzero-excludes-zero-factors} to
$p_{abc}$ for the binary product $(\mathsf{mul}(a,b),c)$. This gives
$q_{ab}:\mathsf{ApartZero}_F(\mathsf{mul}(a,b))$ and
$q_c:\mathsf{ApartZero}_F(c)$; set $p_{ab}:=\alpha_{ab}(q_{ab})$ and
$p_c:=\alpha_c(q_c)$.

Apply the same theorem to $p_{ab}$ for the binary product $(a,b)$. The two
resulting apartness witnesses $q_a:\mathsf{ApartZero}_F(a)$ and
$q_b:\mathsf{ApartZero}_F(b)$ determine $p_a:=\alpha_a(q_a)$ and
$p_b:=\alpha_b(q_b)$.

Now apply \autoref{thm:field-inverse-product-reverse-from-apartness} to
$(\mathsf{mul}(a,b),c)$ with witnesses $p_{abc}$, $p_{ab}$, and $p_c$. Apply it
again to $(a,b)$ with witnesses $p_{ab}$, $p_a$, and $p_b$. Multiplication
congruence transports the second equality into the right factor of the first
one, and associativity plus transitivity of $\hsame$ gives the displayed
three-factor reversed product.
\end{proof}
