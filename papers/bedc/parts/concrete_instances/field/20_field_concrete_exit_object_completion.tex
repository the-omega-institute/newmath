\begin{definition}[RatUp FieldUp concrete exit object]
\label{def:ratup-fieldup-concrete-exit-object}
For $\mathsf{RatUpFieldSource}_F$ and a finite row package $\mathcal E$ over
$\mathsf{BHist}$, write
$$
\mathsf{RatUpFieldExitObject}_F(\mathcal E)
$$
for the concrete predicate asserting that $\mathcal E$ is the conjunction of
the following rows:
\begin{enumerate}
  \item the carrier and classifier rows
  $\mathsf{RatHistoryCarrier}$, $\mathsf{RatHistoryClassifier}$,
  $\mathsf{RatDenomUnitCarrier}$, and
  $\mathsf{RatDenomUnitClassifier}$;
  \item the concrete schema instance row for $\mathsf{RatUpFieldSource}_F$;
  \item the concrete operation-exit exhaustion row for $\mathsf{append}$,
  $\Cont$, the denominator unit, the contextual action $\Gamma$, the
  context-pair action, affine rows, inverse and nonzero rows, one-sided
  multiplication cancellation exactness, two-sided product exactness, and the
  same-endpoint plus crossed singleton-classifier noncollapse rows;
  \item the algebra-hierarchy exit witness row for $\FieldUp(\RatUp)$;
  \item the selector inverse-domain coverage rows generated by
  $E_s(h,k,\ell)$: selected endpoint carrier coverage, selected endpoint
  denominator-classifier coverage, strict endpoint nonzero witnesses, and
  selector-indexed singleton-classifier exclusion under arbitrary contexts.
\end{enumerate}

The concrete $\FieldUp(\RatUp)$ exit object is the paper package $\mathcal E$
satisfying $\mathsf{RatUpFieldExitObject}_F(\mathcal E)$. No schematic
$\mathsf{mul}$, $\mathsf{add}$, or $\mathsf{neg}$ parameter is part of this
predicate.
\end{definition}

\begin{theorem}[RatUp FieldUp concrete exit object is complete]
\label{thm:ratup-fieldup-concrete-exit-object-complete}
For $\mathsf{RatUpFieldSource}_F$, the paper package $\mathcal E_F$ determined
by Definition~\ref{def:ratup-fieldup-concrete-exit-object} satisfies
$$
\mathsf{RatUpFieldExitObject}_F(\mathcal E_F).
$$
Moreover $\mathcal E_F$ is exactly the paper-side $\FieldUp(\RatUp)$ exit
package over $\mathsf{BHist}$: its source, carrier, classifier, stability
obligations, ledger, core theorem list, exactness predicate, standard-bridge
slot, and status are precisely the rows named in
Definition~\ref{def:ratup-fieldup-concrete-exit-object}.
\end{theorem}
\begin{proof}
Apply \autoref{thm:ratup-fieldup-concrete-schema-instance}. It fixes the
single $\mathsf{BHist}$ source tuple for $\mathsf{RatUpFieldSource}_F$ and
identifies the carrier, classifier, source, stability, ledger, and exactness
rows consumed by the concrete package $\mathcal E_F$.

Next apply
\autoref{thm:ratup-fieldup-concrete-operation-exit-exhaustion}. This covers
every operation row named in
Definition~\ref{def:ratup-fieldup-concrete-exit-object}, including the carrier
and classifier anchors, denominator rows, contextual rows, affine rows, inverse
and nonzero rows, product exactness, and singleton-classifier noncollapse rows.

Apply \autoref{thm:ratup-fieldup-algebra-hierarchy-exit-witness}. It packages
those rows as the Phase Algebra Hierarchy paper certificate for
$\FieldUp(\RatUp)$, with the bridge slot and status recorded as part of the
same paper-side object.

Finally apply
\autoref{thm:ratup-fieldup-concrete-exit-selector-inverse-domain-coverage}. It
supplies the selector inverse-domain coverage rows generated by
$E_s(h,k,\ell)$, including selected endpoint carrier coverage, denominator
classifier coverage, strict endpoint nonzero witnesses, and contextual
singleton-classifier exclusion.

These cited inputs cover exactly the conjuncts defining
$\mathsf{RatUpFieldExitObject}_F(\mathcal E_F)$. Hence the predicate holds, and
because the algebra-hierarchy witness uses the same $\mathsf{BHist}$ source and
row inventory, $\mathcal E_F$ is exactly the paper-side
$\FieldUp(\RatUp)$ exit package over $\mathsf{BHist}$.
\end{proof}

\begin{theorem}[RatUp FieldUp concrete exit selector inverse-domain coverage]
\label{thm:ratup-fieldup-concrete-exit-selector-inverse-domain-coverage}
For $\mathsf{RatUpFieldSource}_F$, the concrete exit object of
Definition~\ref{def:ratup-fieldup-concrete-exit-object} contains the strict
inverse-domain selector row for the concrete tuple
$(C_D^u,R_D^u,\mu_D^u,e_D^u,N_D^u)$. Explicitly, for histories
$h,k,\ell:\mathsf{BHist}$ with $C_D^u(h)$, $C_D^u(k)$, $C_D^u(\ell)$, and
$\mathsf{ThreeSupp}_D^u(h,k,\ell)$, selectors $s,t:\mathsf{BMark}$, and
contexts $L,R:\mathsf{BHist}$, the concrete exit object contains the row
$$
C_D^u(E_s(h,k,\ell))
\land
C_D^u(E_t(h,k,\ell))
\land
R_D^u(E_s(h,k,\ell),E_t(h,k,\ell))
\land
\mathsf{fieldSingletonEmptyNonZero}(E_s(h,k,\ell))
\land
\mathsf{fieldSingletonEmptyNonZero}(E_t(h,k,\ell))
\land
\left(
\mathsf{fieldSingletonEmptyClassifier}
(\mathsf{append}(L,E_s(h,k,\ell)),\mathsf{append}(R,E_t(h,k,\ell)))\to\bot
\right).
$$
\end{theorem}
\begin{proof}
Apply \autoref{thm:ratup-fieldup-selector-complete-strict-exit-package} to the
same concrete denominator tuple, carrier hypotheses, support witness, selector
pair, and context pair. It supplies the displayed strict selector row, including
the two selected endpoint carrier rows, the denominator classifier row, the two
strict endpoint nonzero witnesses, and the contextual singleton-classifier
obstruction.

By Definition~\ref{def:ratup-fieldup-concrete-exit-object}, selector
inverse-domain coverage is one conjunct of
$\mathsf{RatUpFieldExitObject}_F$ for the concrete exit package. Therefore the
strict row supplied by the selector package is one of the concrete exit-object
rows for $\mathsf{RatUpFieldSource}_F$.
\end{proof}
