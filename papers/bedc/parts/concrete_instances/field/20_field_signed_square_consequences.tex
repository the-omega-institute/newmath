\section{Signed square-factor consequences}
\label{sec:field-signed-square-factor-consequences}

\begin{theorem}[Field equal-squares signed factor endpoint collapse]
\label{thm:field-equal-squares-signed-factor-endpoint-collapse}
For a BEDC history-level $\FieldUp$ candidate with the hypotheses of
\autoref{thm:field-equal-squares-signed-factor-exclusion}, together with the
additive associativity, additive right identity, addition congruence, and
additive right-inverse laws needed to instantiate
\autoref{thm:group-right-cancellation} for $\mathsf{add}$, every
$a,b:\mathsf{BHist}$ with $a\cdot a\hsame b\cdot b$ satisfies the two
endpoint clauses:
\begin{enumerate}
\item if $p_+:\mathsf{NonZero}_F(a+b)$, then $a\hsame b$;
\item if $p_-:\mathsf{NonZero}_F(a+(-b))$, then $a\hsame -b$.
\end{enumerate}
Traditional target signed square-root ambiguity; BEDC source $\FieldUp +
\CommRingUp + \GroupUp$; carrier $\mathsf{BHist}$; classifier $\hsame$;
operations $\mathsf{add},\mathsf{mul},\mathsf{neg},0,1$; stability obligations
signed-factor exclusion and additive right cancellation; ledger the chosen
signed-factor nonzero witness; core theorem this block; exactness predicate
endpoint collapse under a certified signed factor; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Assume $a\cdot a\hsame b\cdot b$.

For the $p_+:\mathsf{NonZero}_F(a+b)$ branch, apply
\autoref{thm:field-equal-squares-signed-factor-exclusion} to the square
equality and the witness $p_+$. The first signed-factor clause gives
$a+(-b)\hsame0$.

The additive right-inverse law also gives $b+(-b)\hsame0$. By symmetry and
transitivity of $\hsame$, the two displayed sums are classified together:
$$
a+(-b)\hsame b+(-b).
$$
Now instantiate \autoref{thm:group-right-cancellation} for the additive group
operation $\mathsf{add}$. Cancelling the common right summand $-b$ yields
$a\hsame b$.

For the $p_-:\mathsf{NonZero}_F(a+(-b))$ branch, apply
\autoref{thm:field-equal-squares-signed-factor-exclusion} to the same square
equality and the witness $p_-$. The second signed-factor clause gives
$a+b\hsame0$.

The additive left-inverse law gives $(-b)+b\hsame0$. Comparing this equality
with $a+b\hsame0$ and using symmetry and transitivity of $\hsame$ yields
$$
a+b\hsame (-b)+b.
$$
A second instance of additive right cancellation removes the common right
summand $b$, leaving $a\hsame -b$.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.field\_equal\_squares\_signed\_factor\_endpoint\_collapse}

\begin{theorem}[Field equal-squares signed branch determinacy]
\label{thm:field-equal-squares-signed-branch-determinacy}
For a BEDC history-level $\FieldUp$ candidate with the joint hypotheses needed
to instantiate
\autoref{thm:field-equal-squares-signed-factor-endpoint-collapse} and
\autoref{thm:field-equal-squares-signed-factors-not-both-nonzero}, every
$a,b:\mathsf{BHist}$ with
$$
a\cdot a\hsame b\cdot b
$$
satisfies the two deterministic signed-branch clauses:
\begin{enumerate}
\item if $p_+:\mathsf{NonZero}_F(a+b)$, then $a\hsame b$ and
$\mathsf{NonZero}_F(a+(-b))\to\mathsf{False}$;
\item if $p_-:\mathsf{NonZero}_F(a+(-b))$, then $a\hsame -b$ and
$\mathsf{NonZero}_F(a+b)\to\mathsf{False}$.
\end{enumerate}
Traditional target signed square-root branch choice; BEDC source $\FieldUp$
with the signed-factor and additive-cancellation hypotheses; carrier
$\mathsf{BHist}$; classifier $\hsame$; ledger the chosen signed-factor witness;
core theorem this block; exactness predicate endpoint identification together
with exclusion of the opposite signed witness; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Assume $a\cdot a\hsame b\cdot b$.

For the $p_+:\mathsf{NonZero}_F(a+b)$ branch, apply
\autoref{thm:field-equal-squares-signed-factor-endpoint-collapse} to the square
equality and the witness $p_+$. Its positive signed-factor endpoint clause
gives $a\hsame b$.

To exclude the opposite witness, assume
$q_-:\mathsf{NonZero}_F(a+(-b))$. Apply
\autoref{thm:field-equal-squares-signed-factors-not-both-nonzero} to the same
square equality, the chosen witness $p_+$, and the hypothetical witness $q_-$.
The result is $\mathsf{False}$, so the positive branch carries both the endpoint
and the opposite-witness contradiction.

For the $p_-:\mathsf{NonZero}_F(a+(-b))$ branch, apply
\autoref{thm:field-equal-squares-signed-factor-endpoint-collapse} to the square
equality and the witness $p_-$. Its negative signed-factor endpoint clause
gives $a\hsame -b$.

To exclude the opposite witness, assume $q_+:\mathsf{NonZero}_F(a+b)$. Apply
\autoref{thm:field-equal-squares-signed-factors-not-both-nonzero} to the same
square equality, the witness $q_+$, and the chosen witness $p_-$. This produces
$\mathsf{False}$ and completes the negative branch package.
\end{proof}

\begin{theorem}[Field product nonzero excludes zero factors]
\label{thm:field-product-nonzero-excludes-zero-factors}
For a BEDC history-level $\FieldUp$ candidate with ring zero absorption,
multiplication congruence, classifier transport of $\mathsf{NonZero}_F$,
and contradiction of $\mathsf{NonZero}_F(0)$, every $a,b:\mathsf{BHist}$
with
$$
p_{ab}:\mathsf{NonZero}_F(\mathsf{mul}(a,b))
$$
satisfies both $a\hsame0\to\bot$ and $b\hsame0\to\bot$.
\end{theorem}
\begin{proof}
Assume $a\hsame0$. Multiplication congruence sends
$\mathsf{mul}(a,b)$ to $\mathsf{mul}(0,b)$, and
Theorem~\ref{thm:ring-mul-zero-absorption} identifies
$\mathsf{mul}(0,b)$ with $0$. Transitivity gives
$\mathsf{mul}(a,b)\hsame0$.

Transport $p_{ab}:\mathsf{NonZero}_F(\mathsf{mul}(a,b))$ along this
classifier equality. The transport clause supplies
$\mathsf{NonZero}_F(0)$, and the field zero-contradiction clause closes
the left-factor obstruction.

The right-factor obstruction is symmetric. From $b\hsame0$, multiplication
congruence sends $\mathsf{mul}(a,b)$ to $\mathsf{mul}(a,0)$, and
Theorem~\ref{thm:ring-mul-zero-absorption} identifies that product with
$0$. Transporting the same witness $p_{ab}$ to $\mathsf{NonZero}_F(0)$
again contradicts the field zero clause.
\end{proof}

\begin{theorem}[Field two-sided product nonzero excludes middle zero without outer apartness]
\label{thm:field-two-sided-product-nonzero-excludes-middle-zero}
For a BEDC history-level $\FieldUp$ candidate with the hypotheses of
\autoref{thm:field-product-nonzero-excludes-zero-factors}, together with the
ring zero-absorption and multiplication-congruence fields used there, every
$a,b,x:\mathsf{BHist}$ with product witness
$$
p_{axb}:\mathsf{NonZero}_F(\mathsf{mul}(\mathsf{mul}(a,x),b)),
$$
satisfies $x\hsame0\to\bot$.
\end{theorem}
\begin{proof}
Assume $x\hsame0$. Multiplication congruence sends
$\mathsf{mul}(a,x)$ to $\mathsf{mul}(a,0)$, and
Theorem~\ref{thm:ring-mul-zero-absorption} identifies
$\mathsf{mul}(a,0)$ with $0$. Transitivity gives
$$
\mathsf{mul}(a,x)\hsame0.
$$

Apply \autoref{thm:field-product-nonzero-excludes-zero-factors} to the
product with left factor $\mathsf{mul}(a,x)$, right factor $b$, and witness
$p_{axb}$. Its left zero-factor obstruction rejects the displayed classifier,
so the assumed middle classifier $x\hsame0$ yields $\bot$.
\end{proof}

\begin{theorem}[Field certified three-factor product excludes zero factors]
\label{thm:field-three-factor-product-nonzero-excludes-zero-factors}
For a BEDC history-level $\FieldUp$ candidate with the hypotheses of
\autoref{thm:field-product-nonzero-excludes-zero-factors}, including the
ring zero-absorption and multiplication congruence hypotheses used there,
every $a,x,b:\mathsf{BHist}$ with
$$
p_{axb}:\mathsf{NonZero}_F(\mathsf{mul}(\mathsf{mul}(a,x),b))
$$
satisfies
$$
(a\hsame0\to\bot)\wedge(x\hsame0\to\bot)\wedge(b\hsame0\to\bot).
$$
\end{theorem}
\begin{proof}
Apply \autoref{thm:field-product-nonzero-excludes-zero-factors} to the
binary product $(\mathsf{mul}(a,x),b)$ and the witness $p_{axb}$. This gives
an obstruction to $\mathsf{mul}(a,x)\hsame0$ and an obstruction to
$b\hsame0$. The second obstruction is exactly the right-factor clause.

Assume $a\hsame0$. Multiplication congruence sends $\mathsf{mul}(a,x)$ to
$\mathsf{mul}(0,x)$, and \autoref{thm:ring-mul-zero-absorption} classifies
$\mathsf{mul}(0,x)$ as $0$. By transitivity, $\mathsf{mul}(a,x)\hsame0$,
contradicting the first obstruction.

Assume $x\hsame0$. Multiplication congruence sends $\mathsf{mul}(a,x)$ to
$\mathsf{mul}(a,0)$, and \autoref{thm:ring-mul-zero-absorption} classifies
$\mathsf{mul}(a,0)$ as $0$. By transitivity, $\mathsf{mul}(a,x)\hsame0$
again contradicts the first obstruction.
\end{proof}

Per-object certificate: Traditional target: zero-factor exclusion for
certified triple products. BEDC source: $\FieldUp$ plus $\RingUp$ zero
absorption over $\mathsf{BHist}$. Carrier choice: $\mathsf{BHist}$.
Classifier choice: $\hsame$. Stability obligations: product nonzero transport
and zero absorption. Ledger: $p_{axb}$ and the tested zero classifier. Core
theorem list:
Theorem~\ref{thm:field-three-factor-product-nonzero-excludes-zero-factors}.
Exactness predicate: no zero factor under a certified nonzero threefold
product. Standard bridge: deferred. Status: ProofTarget.
