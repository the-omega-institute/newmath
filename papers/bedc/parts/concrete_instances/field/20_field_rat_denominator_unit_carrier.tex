\begin{theorem}[Rat denominator unit semantic NameCert]
\label{thm:rat-denom-unit-semantic-namecert}
For carrier, source, pattern, and ledger all equal to
$\mathsf{RatDenomUnitCarrier}$, and classifier
$\mathsf{RatDenomUnitClassifier}$, the denominator empty-unit interface
satisfies $\mathsf{SemanticNameCert}$.
\end{theorem}
\begin{proof}
The source is inhabited because $\emp$ belongs to
$\mathsf{RatDenomUnitCarrier}$ by the empty-unit branch.

Reflexivity, symmetry, transitivity, and carrier transport are the
corresponding $\hsame$ fields inside
$\mathsf{RatDenomUnitClassifier}$.

Pattern soundness and ledger soundness are projections from the source carrier,
because source, pattern, and ledger are all the same carrier predicate.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.RatDenomUnitClassifier\_semanticNameCert}

\begin{theorem}[Rat denominator unit classifier append left factor classifier]
\label{thm:rat-denom-unit-classifier-append-left-factor-classifier}
For histories $h,h',k,k':\mathsf{BHist}$, if
$$
\mathsf{RatDenomUnitClassifier}(\mathsf{append}(h,k),\mathsf{append}(h',k'))
$$
and $h\hsame h'$, then
$$
\mathsf{RatDenomUnitClassifier}(k,k').
$$
\end{theorem}
\begin{proof}
The classifier on the appended histories supplies denominator-unit carriers for
both appended endpoints and their $\hsame$ comparison. Carrier factorization
recovers denominator-unit carriers for $k$ and $k'$. Replacing the right
endpoint by the same left prefix using $h\hsame h'$ leaves a shared-prefix
append comparison, and finite-kernel left cancellation gives $k\hsame k'$.
These three fields assemble the suffix classifier.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.RatDenomUnitClassifier\_append\_left\_factor\_classifier}

\begin{theorem}[Rat denominator unit classifier append factor hsame iff]
\label{thm:rat-denom-unit-classifier-append-factor-hsame-iff}
For histories $h,h',k,k':\mathsf{BHist}$, if
$$
\mathsf{RatDenomUnitClassifier}(\mathsf{append}(h,k),\mathsf{append}(h',k')),
$$
then
$$
h\hsame h' \Longleftrightarrow k\hsame k'.
$$
\end{theorem}
\begin{proof}
The forward direction applies the append left factor classifier theorem and
projects the suffix $\hsame$ field from the resulting classifier. The reverse
direction applies the corresponding right factor classifier theorem and
projects the prefix $\hsame$ field.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.RatDenomUnitClassifier\_append\_factor\_hsame\_iff}
