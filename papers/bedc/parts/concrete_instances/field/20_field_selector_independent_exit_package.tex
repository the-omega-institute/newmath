\begin{theorem}[RatUp FieldUp exit object operation-table lock]
\label{thm:ratup-fieldup-exit-object-operation-table-lock}
For $\mathsf{RatUpFieldSource}_F$, let $\mathcal E_F$ be the concrete exit
object of Definition~\ref{def:ratup-fieldup-concrete-exit-object}. The same
$\mathsf{BHist}$ row package simultaneously satisfies
$$
\mathsf{RatUpFieldExitObject}_F(\mathcal E_F),
$$
the concrete operation-table coverage of
\autoref{thm:ratup-fieldup-assembled-certificate-operation-table-coverage}, and
the selector-independent final boundary row of
\autoref{thm:ratup-fieldup-concrete-exit-selector-independent-final-boundary}.
Equivalently, $\mathcal E_F$ is a completed concrete exit object whose carrier,
classifier, denominator, contextual, affine, inverse, nonzero, and final
boundary rows are all borne by the one concrete $\mathsf{BHist}$ package.
\end{theorem}
\begin{proof}
Apply \autoref{thm:ratup-fieldup-concrete-exit-object-complete}. This gives
$\mathsf{RatUpFieldExitObject}_F(\mathcal E_F)$ for the paper package
determined by Definition~\ref{def:ratup-fieldup-concrete-exit-object}.

Apply
\autoref{thm:ratup-fieldup-assembled-certificate-operation-table-coverage}. It
identifies the carrier, classifier, denominator tuple, contextual action,
context-pair monoid, affine row source, inverse rows, nonzero rows, and
exit-noncollapse rows as rows of the same concrete $\mathsf{BHist}$ package.

Finally apply
\autoref{thm:ratup-fieldup-concrete-exit-selector-independent-final-boundary}.
This supplies the selector-independent final boundary row for that concrete
exit object. Conjoining these three cited results yields a single completed
$\FieldUp(\RatUp)$ exit object that carries the operation table and the
selector-independent boundary simultaneously.
\end{proof}

\begin{theorem}[RatUp FieldUp algebra hierarchy selector-independent exit]
\label{thm:ratup-fieldup-algebra-hierarchy-selector-independent-exit}
For $\mathsf{RatUpFieldSource}_F$, the concrete denominator tuple
$(C_D^u,R_D^u,\mu_D^u,e_D^u,N_D^u)$, histories
$h,k,\ell:\mathsf{BHist}$ with $C_D^u(h)$, $C_D^u(k)$, and $C_D^u(\ell)$,
selectors $s,t,u,v:\mathsf{BMark}$, and contexts
$L,R,r:\mathsf{BHist}$, if $\mathsf{ThreeSupp}_D^u(h,k,\ell)$ holds, then one
concrete $\mathsf{BHist}$ package contains all of the following rows:
\begin{enumerate}
  \item every row of
  \autoref{thm:ratup-fieldup-algebra-hierarchy-exit-witness};
  \item every shared certificate and operation-table row of
  \autoref{thm:ratup-fieldup-assembled-certificate-operation-table-coverage};
  \item every selector-independent final boundary row of
  \autoref{thm:ratup-fieldup-selector-independent-final-boundary}.
\end{enumerate}
Equivalently, the concrete source tuple, carrier, classifier, denominator
tuple, and endpoint selectors determine a single
$\FieldUp(\RatUp)$ exit certificate that serves the Phase Algebra Hierarchy
interface independently of the certified selector pair.
\end{theorem}
\begin{proof}
First instantiate
\autoref{thm:ratup-fieldup-algebra-hierarchy-exit-witness} at the concrete
source tuple $\mathsf{RatUpFieldSource}_F$. This gives the paper algebra
hierarchy rows over the carrier $\mathsf{BHist}$, with the RatUp history
classifier and the concrete denominator endpoints.

Next instantiate
\autoref{thm:ratup-fieldup-assembled-certificate-operation-table-coverage} at
the same source tuple and the operation table of
Definition~\ref{def:ratup-fieldup-concrete-operation-table}. Its rows use the
same carrier, classifier, denominator coordinate, contextual action,
context-pair coordinate, and affine coordinate as the hierarchy package.

Finally instantiate
\autoref{thm:ratup-fieldup-selector-independent-final-boundary} with the given
histories, denominator-carrier witnesses, support witness, selectors
$s,t,u,v$, and contexts $L,R,r$. This supplies the two final-boundary packages
and the componentwise $R_D^u$ and $\mathsf{RatHistoryClassifier}$ rows for the
selected endpoints.

The three instantiations name the same concrete source tuple
$\mathsf{RatUpFieldSource}_F$, the same $\mathsf{BHist}$ carrier, the same RatUp
classifier, and the same concrete denominator tuple. Conjunction introduction
therefore packages their row families as one concrete
$\FieldUp(\RatUp)$ algebra hierarchy exit certificate.
\end{proof}
