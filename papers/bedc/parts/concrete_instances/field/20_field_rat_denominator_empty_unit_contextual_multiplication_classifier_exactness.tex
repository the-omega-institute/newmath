\begin{theorem}[Rat denominator empty-unit contextual multiplication classifier exactness]
\label{thm:field-rat-denominator-empty-unit-contextual-multiplication-classifier-exactness}
For histories $p,q,p',q',h,k,t:\mathsf{BHist}$, if
$p\hsame\emp$, $q\hsame\emp$, $p'\hsame\emp$, $q'\hsame\emp$,
$C_D^u(h)$, $C_D^u(k)$, and $C_D^u(t)$, then fixed denominator
multiplication is classifier-exact inside empty external contexts on both
sides:
$$
R_D^u(\mathsf{append}(p,\mathsf{append}(\mu_D^u(h,t),q)),
\mathsf{append}(p',\mathsf{append}(\mu_D^u(k,t),q')))
\Longleftrightarrow R_D^u(h,k),
$$
and
$$
R_D^u(\mathsf{append}(p,\mathsf{append}(\mu_D^u(t,h),q)),
\mathsf{append}(p',\mathsf{append}(\mu_D^u(t,k),q')))
\Longleftrightarrow R_D^u(h,k).
$$
\end{theorem}
\begin{proof}
For the right multiplication law, apply
\autoref{thm:field-rat-denominator-unit-classifier-empty-context-exactness} to
the four hypotheses $p\hsame\emp$, $q\hsame\emp$, $p'\hsame\emp$, and
$q'\hsame\emp$. This strips the empty external contexts from the displayed
classifier and leaves the comparison
$R_D^u(\mu_D^u(h,t),\mu_D^u(k,t))$.

The carried hypotheses $C_D^u(h)$, $C_D^u(k)$, and $C_D^u(t)$ now match the
input of
\autoref{thm:field-rat-denominator-empty-unit-right-multiplication-classifier-exactness}.
That theorem identifies the stripped comparison with $R_D^u(h,k)$, giving the
first equivalence.

For the left multiplication law, the same empty-context exactness theorem
again removes the external histories and leaves
$R_D^u(\mu_D^u(t,h),\mu_D^u(t,k))$.

The same carrier hypotheses then match
\autoref{thm:field-rat-denominator-empty-unit-left-multiplication-classifier-exactness},
which identifies this stripped comparison with $R_D^u(h,k)$. This gives the
second equivalence.
\end{proof}
