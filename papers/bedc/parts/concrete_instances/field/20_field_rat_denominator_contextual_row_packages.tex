Finally apply

\begin{theorem}[RatUp FieldUp denominator contextual action support collapse]
\label{thm:ratup-fieldup-denominator-contextual-action-support-collapse}
For the concrete tuple $(C_D^u,R_D^u,\mu_D^u,e_D^u,N_D^u)$ and the contextual
action $\Gamma$ of
\autoref{def:ratup-fieldup-denominator-contextual-action-instance}, suppose
$C_D^u(h)$, $C_D^u(l)$, $C_D^u(r)$, $C_D^u(l')$, and $C_D^u(r')$. Suppose also
that $p\hsame\emp$, $q\hsame\emp$, $p'\hsame\emp$, and $q'\hsame\emp$, and that
$\neg N_D^u(l)$, $\neg N_D^u(r)$, $\neg N_D^u(l')$, and $\neg N_D^u(r')$. Then
the nested action is carried:
$$
C_D^u(\Gamma_{p',q',l',r'}(\Gamma_{p,q,l,r}(h))).
$$
It is classified by the collected denominator-context action:
$$
R_D^u(\Gamma_{p',q',l',r'}(\Gamma_{p,q,l,r}(h)),
\Gamma_{p',q',\mu_D^u(l',l),\mu_D^u(r,r')}(h)).
$$
Under the same no-extra-support hypotheses, strict support collapses to the
support of the acted-on history:
$$
N_D^u(\Gamma_{p',q',l',r'}(\Gamma_{p,q,l,r}(h)))
\Longleftrightarrow
N_D^u(h).
$$
\end{theorem}
\begin{proof}
Apply
\autoref{thm:ratup-fieldup-denominator-contextual-action-unit-composition-laws}
to the displayed carried history, carried denominator contexts, and empty
external endpoint equations. The theorem gives the carried nested action and the
classifier comparison with the collected denominator-context action.

It also gives the five-term strict-support readback
$$
N_D^u(\Gamma_{p',q',l',r'}(\Gamma_{p,q,l,r}(h)))
\Longleftrightarrow
N_D^u(l')\lor N_D^u(l)\lor N_D^u(h)\lor N_D^u(r)\lor N_D^u(r').
$$

For the forward direction of the displayed collapse, the four context-support
disjuncts contradict the hypotheses
$\neg N_D^u(l')$, $\neg N_D^u(l)$, $\neg N_D^u(r)$, and $\neg N_D^u(r')$.
The only remaining disjunct is $N_D^u(h)$.

For the reverse direction, insert $N_D^u(h)$ into the middle disjunct of the
five-term readback and close by the same equivalence. Thus the contextual action
adds no strict support when all four carried contexts have no strict support.
\end{proof}

\begin{theorem}[RatUp FieldUp denominator contextual action total normalization package]
\label{thm:ratup-fieldup-denominator-contextual-action-total-normalization-package}
For the concrete denominator tuple
$(C_D^u,R_D^u,\mu_D^u,e_D^u,N_D^u)$ and the contextual action $\Gamma$ of
\autoref{def:ratup-fieldup-denominator-contextual-action-instance}, the
$\FieldUp(\mathsf{RatUp})$ denominator action package contains the following
single certificate-level normalization record.

\begin{enumerate}
\item The empty action is the classifier unit. If $C_D^u(h)$, then
$$
C_D^u(\Gamma_{\emp,\emp,\emp,\emp}(h)),
$$
$$
R_D^u(\Gamma_{\emp,\emp,\emp,\emp}(h),h),
$$
and
$$
N_D^u(\Gamma_{\emp,\emp,\emp,\emp}(h))
\Longleftrightarrow
N_D^u(h).
$$

\item The two-layer contextual action normalizes to one collected carried action.
If $C_D^u(h)$, $C_D^u(k)$, $C_D^u(l)$, $C_D^u(r)$, $C_D^u(l')$, and
$C_D^u(r')$, and if all displayed external endpoints are $\hsame\emp$, then
$$
R_D^u(\Gamma_{p',q',l',r'}(\Gamma_{p,q,l,r}(h)),
\Gamma_{p',q',\mu_D^u(l',l),\mu_D^u(r,r')}(h)).
$$
For any second endpoint $k$ under the same collected carried contexts,
denominator classifier exactness is
$$
R_D^u(\Gamma_{s,t,\mu_D^u(l',l),\mu_D^u(r,r')}(h),
\Gamma_{s',t',\mu_D^u(l',l),\mu_D^u(r,r')}(k))
\Longleftrightarrow R_D^u(h,k).
$$
The nested endpoint has the five-term strict-support readback
$$
N_D^u(\Gamma_{p',q',l',r'}(\Gamma_{p,q,l,r}(h)))
\Longleftrightarrow
N_D^u(l')\lor N_D^u(l)\lor N_D^u(h)\lor N_D^u(r)\lor N_D^u(r').
$$

\item The threefold empty-external action normalizes to one collected action.
For $C_D^u(h)$ and carried contexts $l_0,r_0,l_1,r_1,l_2,r_2$, let
$$
H_3:=
\Gamma_{\emp,\emp,l_2,r_2}
(\Gamma_{\emp,\emp,l_1,r_1}(\Gamma_{\emp,\emp,l_0,r_0}(h))).
$$
Then
$$
C_D^u(H_3),
$$
$$
R_D^u\bigl(H_3,
\Gamma_{\emp,\emp,\mu_D^u(l_2,\mu_D^u(l_1,l_0)),
\mu_D^u(\mu_D^u(r_0,r_1),r_2)}(h)\bigr),
$$
and
$$
N_D^u(H_3)
\Longleftrightarrow
N_D^u(l_2)\lor N_D^u(l_1)\lor N_D^u(l_0)\lor N_D^u(h)
\lor N_D^u(r_0)\lor N_D^u(r_1)\lor N_D^u(r_2).
$$

\item The same record includes the surrounding row-closure assembly, the one-step
action-certificate rows, the exact readback rows, and the bilateral cancellation
fields listed in
\autoref{thm:ratup-fieldup-denominator-contextual-row-closure-package}.
\end{enumerate}
\end{theorem}
\begin{proof}
The unit and two-layer action laws are exactly the concrete rows of
\autoref{thm:ratup-fieldup-denominator-contextual-action-unit-composition-laws}.
They supply the carried empty action, classifier identity, unchanged strict
support, the collected two-layer classifier comparison, and the five-term
strict-support readback.

For classifier exactness of the collected two-layer endpoint, apply
\autoref{thm:field-rat-denominator-normalized-contextual-action-exactness} with
the inner and outer carried contexts ordered so that the collected left context is
$\mu_D^u(l',l)$ and the collected right context is $\mu_D^u(r,r')$. Its
normalized row gives the displayed denominator classifier equivalence, and its
support rows agree with the five-term readback already recorded above.

The threefold row is
\autoref{thm:field-rat-denominator-contextual-action-threefold-normalization}.
It packages the carried threefold endpoint, the classifier comparison with the
single collected action, and the seven-term support readback for the empty
external layers.

Finally use
\autoref{thm:ratup-fieldup-denominator-contextual-row-closure-package}. That
theorem supplies the surrounding row package, action certificate, exact readback,
and bilateral cancellation fields over the same concrete denominator tuple and
the same action $\Gamma$, so the displayed action-normalization clauses form one
certificate-level FieldUp denominator package.
\end{proof}
