\begin{theorem}[RatUp FieldUp denominator contextual action composition]
\label{thm:field-rat-denominator-contextual-action-composition}
For histories $h,k,l,r,l',r',p,q,p',q',s,t,s',t':\mathsf{BHist}$, assume
$C_D^u(h)$, $C_D^u(k)$, $C_D^u(l)$, $C_D^u(r)$, $C_D^u(l')$,
$C_D^u(r')$, and
$$
p\hsame\emp,\quad q\hsame\emp,\quad p'\hsame\emp,\quad q'\hsame\emp,
\quad s\hsame\emp,\quad t\hsame\emp,\quad s'\hsame\emp,\quad t'\hsame\emp.
$$
Then the concrete action $\Gamma$ of
\autoref{def:ratup-fieldup-denominator-contextual-action-instance} is exact
under two carried contextual applications:
$$
R_D^u(\Gamma_{p,q,l,r}(\Gamma_{s,t,l',r'}(h)),
\Gamma_{p',q',l,r}(\Gamma_{s',t',l',r'}(k)))
\Longleftrightarrow R_D^u(h,k).
$$
Its strict-support readbacks are
$$
N_D^u(\Gamma_{p,q,l,r}(\Gamma_{s,t,l',r'}(h)))
\Longleftrightarrow
N_D^u(l)\lor N_D^u(l')\lor N_D^u(h)\lor N_D^u(r')\lor N_D^u(r)
$$
and
$$
N_D^u(\Gamma_{p',q',l,r}(\Gamma_{s',t',l',r'}(k)))
\Longleftrightarrow
N_D^u(l)\lor N_D^u(l')\lor N_D^u(k)\lor N_D^u(r')\lor N_D^u(r).
$$
\end{theorem}
\begin{proof}
Apply
\autoref{thm:ratup-fieldup-denominator-contextual-action-certificate} to the
inner actions with carried contexts $l',r'$ and external contexts
$s,t,s',t'$. This gives the classifier equivalence between the inner endpoints
and $h,k$, and it reads the inner supports as
$N_D^u(l')\lor N_D^u(h)\lor N_D^u(r')$ and
$N_D^u(l')\lor N_D^u(k)\lor N_D^u(r')$.

The carrier part used in the same action proof puts both inner endpoints back
inside $C_D^u$: carried bilateral product closure handles $B_{l',r'}(h)$ and
$B_{l',r'}(k)$, and empty external endpoint transport preserves the displayed
carrier. These two carrier witnesses are the inputs needed for the outer action.

Apply
\autoref{thm:ratup-fieldup-denominator-contextual-action-certificate} again to
the outer action with carried contexts $l,r$ and external contexts
$p,q,p',q'$. Its classifier row identifies the displayed nested classifier with
the classifier of the two inner endpoints. Composing this equivalence with the
inner classifier equivalence gives the stated comparison with $R_D^u(h,k)$.

For strict support, the outer action reads
$\Gamma_{p,q,l,r}(\Gamma_{s,t,l',r'}(h))$ as
$N_D^u(l)\lor N_D^u(\Gamma_{s,t,l',r'}(h))\lor N_D^u(r)$. Substitute the inner
support readback and reassociate the finite disjunction to obtain
$N_D^u(l)\lor N_D^u(l')\lor N_D^u(h)\lor N_D^u(r')\lor N_D^u(r)$.

The second support statement is identical with $k$ in place of $h$ and
$s',t'$ in place of $s,t$. The same outer readback, inner readback, and
reassociation give
$N_D^u(l)\lor N_D^u(l')\lor N_D^u(k)\lor N_D^u(r')\lor N_D^u(r)$.
\end{proof}

\begin{theorem}[RatUp FieldUp denominator normalized contextual action exactness]
\label{thm:field-rat-denominator-normalized-contextual-action-exactness}
For histories $h,k,l,r,l',r',p,q,p',q',s,t,s',t':\mathsf{BHist}$, assume
$C_D^u(h)$, $C_D^u(k)$, $C_D^u(l)$, $C_D^u(r)$, $C_D^u(l')$,
$C_D^u(r')$, and
$$
p\hsame\emp,\quad q\hsame\emp,\quad p'\hsame\emp,\quad q'\hsame\emp,
\quad s\hsame\emp,\quad t\hsame\emp,\quad s'\hsame\emp,\quad t'\hsame\emp.
$$
Then the single normalized actions obtained from the combined carried contexts
are exact for the denominator classifier:
$$
R_D^u(\Gamma_{p,q,\mu_D^u(l,l'),\mu_D^u(r',r)}(h),
\Gamma_{p',q',\mu_D^u(l,l'),\mu_D^u(r',r)}(k))
\Longleftrightarrow R_D^u(h,k).
$$
The normalized strict-support readbacks are
$$
N_D^u(\Gamma_{p,q,\mu_D^u(l,l'),\mu_D^u(r',r)}(h))
\Longleftrightarrow
N_D^u(l)\lor N_D^u(l')\lor N_D^u(h)\lor N_D^u(r')\lor N_D^u(r)
$$
and
$$
N_D^u(\Gamma_{p',q',\mu_D^u(l,l'),\mu_D^u(r',r)}(k))
\Longleftrightarrow
N_D^u(l)\lor N_D^u(l')\lor N_D^u(k)\lor N_D^u(r')\lor N_D^u(r).
$$
\end{theorem}
\begin{proof}
Apply
\autoref{thm:ratup-fieldup-denominator-contextual-action-composition-closure}
to the $h$ endpoint with inner carried contexts $l',r'$, outer carried contexts
$l,r$, inner external endpoints $s,t$, and outer external endpoints $p,q$. The
classifier row identifies the nested action
$\Gamma_{p,q,l,r}(\Gamma_{s,t,l',r'}(h))$ with the normalized action
$\Gamma_{p,q,\mu_D^u(l,l'),\mu_D^u(r',r)}(h)$.

Its support row reads the nested endpoint as the displayed five-term
disjunction. Strict-support transport along the concrete classifier $R_D^u$
then gives the same readback for the normalized $h$ endpoint.

Apply the same closure theorem to the $k$ endpoint with inner external
endpoints $s',t'$ and outer external endpoints $p',q'$. This gives the
classifier comparison between
$\Gamma_{p',q',l,r}(\Gamma_{s',t',l',r'}(k))$ and
$\Gamma_{p',q',\mu_D^u(l,l'),\mu_D^u(r',r)}(k)$.

The corresponding support row, transported along $R_D^u$, gives the
strict-support readback containing $N_D^u(k)$ in the middle position.

Now use
\autoref{thm:field-rat-denominator-contextual-action-composition} on the two
nested endpoints. It supplies the equivalence between their denominator
classifier and $R_D^u(h,k)$. Transport this equivalence across the two closure
comparisons by symmetry and transitivity for the concrete classifier $R_D^u$.

The two support statements are the strict-support readbacks already obtained
from the endpoint closure applications, with the finite disjunctions
reassociated to the displayed order.
\end{proof}

\begin{theorem}[RatUp FieldUp denominator contextual action threefold normalization]
\label{thm:field-rat-denominator-contextual-action-threefold-normalization}
For histories $h,l_0,r_0,l_1,r_1,l_2,r_2:\mathsf{BHist}$, assume
$C_D^u(h)$, $C_D^u(l_0)$, $C_D^u(r_0)$, $C_D^u(l_1)$, $C_D^u(r_1)$,
$C_D^u(l_2)$, and $C_D^u(r_2)$. Let
$$
H_3:=
\Gamma_{\emp,\emp,l_2,r_2}
(\Gamma_{\emp,\emp,l_1,r_1}(\Gamma_{\emp,\emp,l_0,r_0}(h))).
$$
Then the threefold concrete denominator contextual action is carried:
$$
C_D^u(H_3).
$$
It is classified by the single normalized carried action
$$
R_D^u\bigl(H_3,
\Gamma_{\emp,\emp,\mu_D^u(l_2,\mu_D^u(l_1,l_0)),
\mu_D^u(\mu_D^u(r_0,r_1),r_2)}(h)\bigr).
$$
Its strict-support readback is exactly
$$
N_D^u(H_3)
\Longleftrightarrow
N_D^u(l_2)\lor N_D^u(l_1)\lor N_D^u(l_0)\lor N_D^u(h)
\lor N_D^u(r_0)\lor N_D^u(r_1)\lor N_D^u(r_2).
$$
\end{theorem}
\begin{proof}
First apply
\autoref{thm:ratup-fieldup-denominator-contextual-action-composition-closure}
to the inner two contextual layers with carried contexts $l_0,r_0$ and
$l_1,r_1$. Since all external endpoints are $\emp$, the inner composite is
carried, classified by
$\Gamma_{\emp,\emp,\mu_D^u(l_1,l_0),\mu_D^u(r_0,r_1)}(h)$, and has
five-term support readback
$N_D^u(l_1)\lor N_D^u(l_0)\lor N_D^u(h)\lor N_D^u(r_0)\lor N_D^u(r_1)$.

Next apply the same closure theorem to the outer layer with carried contexts
$l_2,r_2$ and to the normalized inner action as the carried core. The carrier
row gives $C_D^u(H_3)$, while the classifier row reduces the threefold term to
an outer action on the normalized two-layer core.

Use the denominator multiplicative certificate rows of
\autoref{thm:field-rat-denominator-fieldup-multiplicative-certificate-record}
to reassociate the carried append products. The left contexts collect as
$\mu_D^u(l_2,\mu_D^u(l_1,l_0))$, and the right contexts collect as
$\mu_D^u(\mu_D^u(r_0,r_1),r_2)$, so the classifier comparison is exactly the
displayed single normalized action.

For strict support, the outer closure row reads the support of $H_3$ as
$N_D^u(l_2)$, the support of the inner two-layer action, and $N_D^u(r_2)$.
Substitute the inner support readback from the first step and reassociate the
finite constructive disjunction. This gives the displayed seven-term
strict-support formula in the stated order.
\end{proof}
