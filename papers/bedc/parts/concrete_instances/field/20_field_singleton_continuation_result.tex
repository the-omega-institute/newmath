\begin{theorem}[Field singleton carrier continuation visible-left exclusion]
\label{thm:field-singleton-carrier-continuation-visible-left-absurd}
For histories $p,Q,R:\mathsf{BHist}$, no continuation whose left input is
headed by a visible constructor can land in the singleton field carrier:
$$
\bigl(\ContR(\Ezero(p),Q,R)\Rightarrow
  \mathsf{FieldSingletonCarrier}(R)\Rightarrow\bot\bigr)
\land
\bigl(\ContR(\Eone(p),Q,R)\Rightarrow
  \mathsf{FieldSingletonCarrier}(R)\Rightarrow\bot\bigr).
$$
\end{theorem}
\begin{proof}
Transport a displayed continuation result through the singleton carrier to
$\emp$. Empty-result inversion then identifies the visible left input with
$\emp$, contradicting BEDC history no-confusion for $\Ezero$ or $\Eone$.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.FieldSingletonCarrier\_continuation\_visible\_left\_absurd}

\begin{theorem}[Field singleton classifier continuation empty result iff]
\label{thm:field-singleton-classifier-continuation-empty-result-iff}
For histories $P,Q,R:\mathsf{BHist}$ with $\ContR(P,Q,R)$, the singleton
field classifier relates $P$ and $Q$ exactly when the continuation result $R$
is singleton-classified with $\emp$:
$$
\mathsf{FieldSingletonClassifier}(P,Q)
\Longleftrightarrow
\mathsf{FieldSingletonClassifier}(R,\emp).
$$
\end{theorem}
\begin{proof}
The forward implication is the continuation result classifier. Conversely, the
result-side classifier identifies $R$ with $\emp$. Transporting the displayed
continuation to the empty result and applying empty-result inversion identifies
both $P$ and $Q$ with $\emp$, which reconstructs the singleton classifier.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.FieldSingletonClassifier\_continuation\_empty\_result\_iff}

$$

\begin{proof}

Thus $R$ is singleton-carried, and it compares with $P$ through $\emp$.
Conversely, the result-left classifier identifies $R$ with $\emp$. Transport
the displayed continuation to empty result and apply empty-result inversion;
this recovers singleton carrier witnesses for both $P$ and $Q$, which rebuild
the input classifier.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.FieldSingletonClassifier\_continuation\_result\_left\_iff}

For histories $P,Q,R:\mathsf{BHist}$ with $\ContR(P,Q,R)$, the result-side

$$

\mathsf{FieldSingletonCarrier}(P)\land

\end{theorem}
\begin{proof}
For the forward direction, the result classifier identifies $R$ with $\emp$;
transporting the continuation to empty result and inverting it identifies

\begin{theorem}[Field singleton carrier continuation context empty iff]
\label{thm:field-singleton-carrier-continuation-context-empty-iff}
For histories $L,h,M,R,O:\mathsf{BHist}$ with
$\ContR(L,h,M)$ and $\ContR(M,R,O)$, the final result is a singleton carrier
exactly when the three exposed factors are singleton carriers:
$$
\mathsf{FieldSingletonCarrier}(O)
\Longleftrightarrow
\mathsf{FieldSingletonCarrier}(L)\land
\mathsf{FieldSingletonCarrier}(h)\land
\mathsf{FieldSingletonCarrier}(R).
$$
\end{theorem}
\begin{proof}
For the forward direction, transport the second continuation result to
$\emp$ and invert the empty result; this gives singleton carriers for $M$
and $R$. Transport the first continuation result from $M$ to $\emp$ and
invert again, exposing singleton carriers for $L$ and $h$. Conversely,
singleton carriers for $L$ and $h$ transport the first continuation result
to the singleton carrier, and the resulting middle carrier together with the
carrier for $R$ transports the second continuation result to the singleton
carrier.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.FieldSingletonCarrier\_continuation\_context\_empty\_iff}

\begin{theorem}[Field singleton classifier continuation append split]
\label{thm:field-singleton-classifier-continuation-append-split}
For histories $L,P,Q,R,T:\mathsf{BHist}$ with $\ContR(P,Q,T)$, singleton
classification of $\mathsf{append}(L,T)$ against $R$ is equivalent to the four
displayed endpoints lying in the singleton carrier:
$$
\mathsf{FieldSingletonClassifier}(\mathsf{append}(L,T),R)
\Longleftrightarrow
\mathsf{FieldSingletonCarrier}(L)\land
\mathsf{FieldSingletonCarrier}(P)\land
\mathsf{FieldSingletonCarrier}(Q)\land
\mathsf{FieldSingletonCarrier}(R).
$$
\end{theorem}
\begin{proof}
The forward direction splits the empty appended endpoint into carrier witnesses
for $L$ and $T$. Transporting the continuation ledger to the empty result and
inverting it gives carrier witnesses for $P$ and $Q$, while the classifier
itself supplies the carrier witness for $R$. Conversely, carrier witnesses for
$P$ and $Q$ make the continuation result $T$ singleton, and combining that
with the carrier of $L$ rebuilds the left appended endpoint. Both classifier
endpoints then compare through $\emp$.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.FieldSingletonClassifier\_continuation\_append\_split\_iff}

\mathsf{FieldSingletonCarrier}(Q)\land

\label{lem:field-singleton-classifier-continuation-context-empty-result}

carriers and
$$
\ContR(\mathsf{append}(L,P),\mathsf{append}(Q,R),T),
$$
then singleton classification of the continuation result against $\emp$ is
equivalent to singleton classification of the exposed middle endpoints:
$$
\mathsf{FieldSingletonClassifier}(T,\emp)
\Longleftrightarrow
\mathsf{FieldSingletonClassifier}(P,Q).

\end{lemma}
\begin{proof}
For the forward implication, transport the displayed continuation to the empty
result and invert it. The two appended endpoints split at $\emp$, exposing
singleton carriers for $P$ and $Q$, and the two carriers compare through
$\emp$. Conversely, singleton carriers for $P$ and $Q$, together with the
assumed singleton carriers for $L$ and $R$, make both appended continuation
endpoints empty; the continuation result is therefore singleton and is
classified with $\emp$.
\end{proof}
