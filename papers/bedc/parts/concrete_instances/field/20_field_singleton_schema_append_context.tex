\begin{theorem}[Field singleton classifier append context cancellation]
\label{thm:field-singleton-classifier-append-context-cancel}
For histories $L,R,Q,S:\mathsf{BHist}$, if both displayed contexts lie in the
singleton carrier $A_{\emp}$, then appending $L$ to the first endpoint and
$R$ to the second endpoint does not change the singleton classifier:
$$
\mathsf{FieldSingletonCarrier}(L)\Rightarrow
\mathsf{FieldSingletonCarrier}(R)\Rightarrow
\bigl(
\mathsf{FieldSingletonClassifier}(\mathsf{append}(L,Q),\mathsf{append}(R,S))
\Longleftrightarrow
\mathsf{FieldSingletonClassifier}(Q,S)
\bigr).
$$
\end{theorem}
\begin{proof}
In the forward direction, the appended endpoint carriers split by
empty-result inversion for append. The right pieces give
$Q\hsame\emp$ and $S\hsame\emp$, and transitivity through $\emp$ rebuilds the
base classifier.

Conversely, combine the two context carrier witnesses with the endpoint
carrier fields in $\mathsf{FieldSingletonClassifier}(Q,S)$. Empty-result
inversion for append rebuilds singleton carrier witnesses for the two appended
endpoints, and transitivity through $\emp$ supplies their classifier
comparison.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.FieldSingletonClassifier\_append\_context\_cancel\_iff}

\begin{theorem}[Field singleton classifier append endpoint split]
\label{thm:field-singleton-classifier-append-endpoint-split}
For histories $L,Q,R,S:\mathsf{BHist}$,
$\mathsf{FieldSingletonClassifier}(\mathsf{append}(L,Q),\mathsf{append}(R,S))$ is equivalent to the four carrier witnesses
$\mathsf{FieldSingletonCarrier}(L)$, $\mathsf{FieldSingletonCarrier}(Q)$, $\mathsf{FieldSingletonCarrier}(R)$, and $\mathsf{FieldSingletonCarrier}(S)$.
\end{theorem}
\begin{proof}
The two classifier carrier fields identify both append endpoints with
$\emp$, and empty-result inversion for append splits these witnesses into the
four component carriers. Conversely, the four component carrier witnesses
rebuild the two appended carrier witnesses, and transitivity through $\emp$
supplies their classifier comparison.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.FieldSingletonClassifier\_append\_endpoint\_split\_iff}

\begin{theorem}[Field singleton carrier empty append context]
\label{thm:field-singleton-carrier-append-context-empty}
For histories $L,R,h:\mathsf{BHist}$, if $L$ and $R$ lie in the singleton
carrier $A_{\emp}$, then surrounding $h$ by the displayed left and right append
contexts does not change singleton-carrier membership:
$$
\mathsf{FieldSingletonCarrier}(L)\Rightarrow
\mathsf{FieldSingletonCarrier}(R)\Rightarrow
\bigl(
\mathsf{FieldSingletonCarrier}(\mathsf{append}(L,\mathsf{append}(h,R)))
\Longleftrightarrow
\mathsf{FieldSingletonCarrier}(h)
\bigr).
$$
\end{theorem}
\begin{proof}
In the forward direction, the singleton carrier witness identifies the nested
append with $\emp$. Empty-result inversion for append first splits off $L$ and
then splits $\mathsf{append}(h,R)$, recovering $h\hsame\emp$.

Conversely, combine the carrier witness for $h$ with the carrier witness for
$R$ to rebuild the inner empty append. Combining that result with the carrier
witness for $L$ rebuilds the displayed outer empty append.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.FieldSingletonCarrier\_append\_context\_empty\_iff}

\begin{theorem}[Field singleton classifier empty append context]
\label{thm:field-singleton-classifier-append-context-empty}
For histories $L,R,h,k:\mathsf{BHist}$, if $L$ and $R$ lie in the singleton
carrier $A_{\emp}$, then adding $L$ on the left of the first endpoint and $R$
on the right of the second endpoint preserves and reflects the singleton field
classifier:
$$
\mathsf{FieldSingletonCarrier}(L)\Rightarrow
\mathsf{FieldSingletonCarrier}(R)\Rightarrow
\bigl(
\mathsf{FieldSingletonClassifier}(\mathsf{append}(L,h),\mathsf{append}(k,R))
\Longleftrightarrow
\mathsf{FieldSingletonClassifier}(h,k)
\bigr).
$$
\end{theorem}
\begin{proof}
In the forward direction, the classifier supplies singleton-carrier witnesses
for both appended endpoints. Empty-result inversion for append recovers
$h\hsame\emp$ from the left endpoint and $k\hsame\emp$ from the right endpoint,
and transitivity through $\emp$ supplies the classifier comparison between
$h$ and $k$.

Conversely, the two carrier fields of $\mathsf{FieldSingletonClassifier}(h,k)$,
together with the carrier witnesses for $L$ and $R$, rebuild the two appended
singleton-carrier witnesses. Since both appended endpoints are carried to
$\emp$, transitivity through $\emp$ gives the appended classifier comparison.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.FieldSingletonClassifier\_append\_context\_empty\_iff}

\begin{theorem}[Singleton empty nonzero append right cancellation]
\label{thm:field-singleton-empty-nonzero-append-right-cancel}
For histories $P,Q:\mathsf{BHist}$, if $P$ lies in the singleton empty
carrier, then appending $P$ on the right does not change the
classifier-complement singleton nonzero predicate:
$$
\mathsf{fieldSingletonEmptyCarrier}(P)\Rightarrow
\bigl(
\mathsf{fieldSingletonEmptyNonZero}(\mathsf{append}(Q,P))
\Longleftrightarrow
\mathsf{fieldSingletonEmptyNonZero}(Q)
\bigr).
$$
\end{theorem}
\begin{proof}
In the forward direction, suppose $Q$ were classified against $\emp$. Combining
that endpoint witness with the displayed carrier witness for $P$ rebuilds a
singleton empty classifier witness for $\mathsf{append}(Q,P)$, contradicting
the appended complement predicate.

Conversely, any classifier witness for $\mathsf{append}(Q,P)$ supplies
$\mathsf{append}(Q,P)\hsame\emp$. Empty-result inversion for append extracts
$Q\hsame\emp$, and this reconstructs the forbidden classifier witness for
$Q$ against $\emp$.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.fieldSingletonEmptyNonZero\_append\_right\_cancel\_iff}

\begin{theorem}[Singleton empty nonzero append left cancellation]
\label{thm:field-singleton-empty-nonzero-append-left-cancel}
For histories $P,Q:\mathsf{BHist}$, if $P$ lies in the singleton empty
carrier, then appending $P$ on the left does not change the
classifier-complement singleton nonzero predicate:
$$
\mathsf{fieldSingletonEmptyCarrier}(P)\Rightarrow
\bigl(
\mathsf{fieldSingletonEmptyNonZero}(\mathsf{append}(P,Q))
\Longleftrightarrow
\mathsf{fieldSingletonEmptyNonZero}(Q)
\bigr).
$$
\end{theorem}
\begin{proof}
The forward implication turns a forbidden classifier witness for $Q$ into one
for $\mathsf{append}(P,Q)$ by pairing the carrier witness for $P$ with the
empty endpoint witness for $Q$.

For the reverse implication, a classifier witness for
$\mathsf{append}(P,Q)$ identifies the append result with $\emp$. Empty-result
inversion splits this into carrier witnesses for both factors, and the right
factor witness rebuilds the forbidden singleton empty classifier for $Q$.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.fieldSingletonEmptyNonZero\_append\_left\_cancel\_iff}
