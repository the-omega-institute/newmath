\begin{definition}[Rat denominator continuation with adjoined empty unit]
\label{def:field-rat-denominator-unit-carrier}
For each history $h:\mathsf{BHist}$, define the empty-unit denominator carrier
by

$$
\mathsf{RatDenomUnitCarrier}(h)\Longleftrightarrow
h\hsame\emp\lor\mathsf{RatHistoryCarrier}(h).
$$

For histories $h,k:\mathsf{BHist}$, define the corresponding endpoint
classifier by

$$
\mathsf{RatDenomUnitClassifier}(h,k)\Longleftrightarrow
\mathsf{RatDenomUnitCarrier}(h)\land
\mathsf{RatDenomUnitCarrier}(k)\land h\hsame k.
$$
\end{definition}
\leandef{BEDC.Derived.FieldUp.RatDenomUnitCarrier}
\leandef{BEDC.Derived.FieldUp.RatDenomUnitClassifier}

\begin{theorem}[Rat denominator unit carrier history-sameness transport]
\label{thm:rat-denom-unit-carrier-hsame-transport}
For histories $h,k:\mathsf{BHist}$, if $h\hsame k$ and
$\mathsf{RatDenomUnitCarrier}(h)$, then
$\mathsf{RatDenomUnitCarrier}(k)$.
\end{theorem}
\begin{proof}
Split the carrier disjunction for $h$. In the empty branch, compose
$h\hsame k$ with the displayed comparison $h\hsame\emp$ and symmetry or
transitivity of $\hsame$ to obtain $k\hsame\emp$. This selects the empty
branch of $\mathsf{RatDenomUnitCarrier}(k)$.

In the Rat branch, apply Rat carrier transport along the $\hsame$ comparison
from $h$ to $k$. The resulting $\mathsf{RatHistoryCarrier}(k)$ witness selects
the Rat branch of the denominator unit carrier.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.RatDenomUnitCarrier\_hsame\_transport}

\begin{theorem}[Rat denominator unit carrier continuation closure]
\label{thm:rat-denom-unit-carrier-continuation-closed}
For histories $h,k,r:\mathsf{BHist}$, if
$\mathsf{RatDenomUnitCarrier}(h)$, $\mathsf{RatDenomUnitCarrier}(k)$, and
$\Cont(h,k,r)$, then $\mathsf{RatDenomUnitCarrier}(r)$.
\end{theorem}
\begin{proof}
Split the two input carriers. The empty/empty case uses the two continuation
unit laws to identify the result with $\emp$, hence the result lies in the
empty branch.

In the two mixed cases, the corresponding unit law identifies the continuation
result with the nonempty input; use
$\mathsf{cont\_respects\_hsame}$ and Rat carrier transport to move the Rat
carrier witness to $r$.

In the Rat/Rat case, apply the Rat denominator continuation carrier-closure
theorem directly. The resulting $\mathsf{RatHistoryCarrier}(r)$ witness selects
the Rat branch of $\mathsf{RatDenomUnitCarrier}(r)$.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.RatDenomUnitCarrier\_continuation\_closed}

\begin{theorem}[Rat denominator unit carrier nonempty endpoint exactness]
\label{thm:rat-denom-unit-carrier-nonempty-rat}
For every history $h$, if $\mathsf{RatDenomUnitCarrier}(h)$ holds and
$h\hsame\emp$ is impossible, then $h$ is a genuine Rat denominator endpoint:
$$
\mathsf{RatHistoryCarrier}(h).
$$
\end{theorem}
\begin{proof}
Unfold the carrier disjunction. The empty-unit branch is exactly
$h\hsame\emp$, contradicting the supplied nonempty endpoint condition. The
other branch is the required $\mathsf{RatHistoryCarrier}(h)$ witness.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.RatDenomUnitCarrier\_nonempty\_rat}

\begin{theorem}[Rat denominator unit continuation exposes a nonempty Rat endpoint]
\label{thm:rat-denom-unit-continuation-nonempty-rat-endpoint}
For histories $h,k,r$, if $h$ and $k$ lie in
$\mathsf{RatDenomUnitCarrier}$ and $\Cont(h,k,r)$ has a result not
$\hsame$ to $\emp$, then at least one input endpoint is a genuine Rat
denominator endpoint:
$$
\mathsf{RatHistoryCarrier}(h)\lor\mathsf{RatHistoryCarrier}(k).
$$
\end{theorem}
\begin{proof}
Split both carrier witnesses into their empty-unit and Rat endpoint branches.
If both inputs are empty-unit endpoints, continuation transport along those
$\hsame$ witnesses identifies the result with $\emp$, contradicting the
nonempty result condition. In either mixed branch, the Rat side is already the
required disjunct.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.RatDenomUnitCarrier\_continuation\_nonempty\_rat\_endpoint}

\begin{theorem}[Rat denominator unit-envelope classifier is append-closed]
\label{thm:field-rat-denominator-unit-envelope-classifier-append-closed}
For histories $h,h',k,k':\mathsf{BHist}$, suppose each displayed pair lies in
the denominator unit envelope: either both endpoints are
$\mathsf{RatHistoryCarrier}$ and related by $\mathsf{RatHistoryClassifier}$,
or both endpoints are $\hsame$ to $\emp$. Then
$\mathsf{append}(h,k)$ and $\mathsf{append}(h',k')$ lie in the same envelope.
\end{theorem}
\begin{proof}
Split the two envelope hypotheses. In the carried/carried branch, the right
classifier exposes a unary denominator tail and a tail $\hsame$ comparison, so
Rat denominator classifier append closure applies to the left classifier. In
the mixed branches, the empty side reduces by the left or right unit law for
append, and classifier transport carries the nonempty classifier across those
$\hsame$ comparisons. The empty/empty branch follows by eliminating the four
empty endpoint equalities and applying reflexivity at $\emp$.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.field\_rat\_denominator\_unit\_envelope\_classifier\_append\_closed}

\begin{theorem}[Rat denominator unit-envelope classifier empty-right exactness]
\label{thm:field-rat-denominator-unit-envelope-classifier-empty-right-exactness}
For every history $h:\mathsf{BHist}$, the unit-envelope classifier with empty
right endpoint is exactly empty endpoint sameness:
$$
\mathsf{FieldRatDenominatorUnitEnvelopeClassifier}(h,\emp)
\Longleftrightarrow h\hsame\emp.
$$
\end{theorem}
\begin{proof}
Split the concrete unit-envelope classifier. In the carried branch, the right
endpoint would be both $\mathsf{RatHistoryCarrier}$ and $\emp$, which contradicts
$\mathsf{RatHistoryCarrier\_not\_empty}$. In the empty branch, the left endpoint
comparison is already present. Conversely, $h\hsame\emp$ selects the empty-empty
branch with reflexivity at $\emp$.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.FieldRatDenominatorUnitEnvelopeClassifier\_empty\_right\_iff}

\begin{theorem}[Rat denominator unit-envelope classifier empty-left exactness]
\label{thm:field-rat-denominator-unit-envelope-classifier-empty-left-endpoint-iff}
For histories $h,k:\mathsf{BHist}$, if the left endpoint is empty, then the
unit-envelope classifier is exactly empty endpoint sameness on the right:
$$
h\hsame\emp
\Longrightarrow
\left(
\mathsf{FieldRatDenominatorUnitEnvelopeClassifier}(h,k)
\Longleftrightarrow k\hsame\emp
\right).
$$
\end{theorem}
\begin{proof}
Split the concrete unit-envelope classifier. In the carried branch, the left
endpoint would be both $\mathsf{RatHistoryCarrier}$ and $\emp$, contradicting
$\mathsf{RatHistoryCarrier\_not\_empty}$. In the empty branch, the right
endpoint comparison is already present. Conversely, $k\hsame\emp$ together
with the displayed left endpoint comparison selects the empty-empty branch.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.FieldRatDenominatorUnitEnvelopeClassifier\_empty\_left\_endpoint\_iff}

\begin{theorem}[Rat denominator unit-envelope classifier is continuation-closed]
\label{thm:field-rat-denominator-unit-envelope-classifier-continuation-closed}
For histories $h,h',k,k',r,r':\mathsf{BHist}$, if
$$
\mathsf{FieldRatDenominatorUnitEnvelopeClassifier}(h,h'),\quad
\mathsf{FieldRatDenominatorUnitEnvelopeClassifier}(k,k'),\quad
\Cont(h,k,r),\quad
\Cont(h',k',r'),
$$
then
$$
\mathsf{FieldRatDenominatorUnitEnvelopeClassifier}(r,r').
$$
\end{theorem}
\begin{proof}
Split the two classifier witnesses. In the carried-carried branch, Rat
denominator continuation classifier closure supplies a carried classifier for
the two results. In each mixed branch, continuation congruence against the empty
endpoint unit identifies the result with the carried endpoint on that side; Rat
carrier transport then supplies carried result witnesses, and continuation
congruence supplies the result comparison. In the empty-empty branch, both
continuations transport to the empty-left unit continuation at $\emp$, so both
results are history-same to $\emp$ and the empty branch applies.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.FieldRatDenominatorUnitEnvelopeClassifier\_continuation\_closed}

\begin{theorem}[Rat denominator continuation adjoined-unit laws]
\label{thm:field-rat-denominator-continuation-adjoined-unit-laws}
The concrete carrier $\mathsf{RatDenomUnitCarrier}$ and classifier
$\mathsf{RatDenomUnitClassifier}$ satisfy the denominator continuation laws
needed to use $\emp$ as the endpoint unit:
\begin{enumerate}
\item $\mathsf{RatDenomUnitCarrier}(\emp)$.
\item If $\mathsf{RatDenomUnitCarrier}(h)$,
$\mathsf{RatDenomUnitCarrier}(k)$, and $\Cont(h,k,r)$, then
$\mathsf{RatDenomUnitCarrier}(r)$.
\item If $\mathsf{RatDenomUnitCarrier}(h)$ and $\Cont(\emp,h,r)$, then
$\mathsf{RatDenomUnitClassifier}(r,h)$.
\item If $\mathsf{RatDenomUnitCarrier}(h)$ and $\Cont(h,\emp,r)$, then
$\mathsf{RatDenomUnitClassifier}(r,h)$.
\item For all histories
$h,k,\ell,hk,k\ell,left,right:\mathsf{BHist}$, if
$\mathsf{RatDenomUnitCarrier}(h)$, $\mathsf{RatDenomUnitCarrier}(k)$,
$\mathsf{RatDenomUnitCarrier}(\ell)$, $\Cont(h,k,hk)$,
$\Cont(k,\ell,k\ell)$, $\Cont(hk,\ell,left)$, and
$\Cont(h,k\ell,right)$, then
$\mathsf{RatDenomUnitClassifier}(left,right)$.
\end{enumerate}
\end{theorem}
\leanchecked{BEDC.Derived.FieldUp.field\_rat\_denominator\_continuation\_adjoined\_unit\_laws}
\begin{proof}
The empty endpoint is carried by the left disjunct of
Definition~\ref{def:field-rat-denominator-unit-carrier}, using reflexivity of
$\hsame$ at $\emp$.

For closure, split both input carrier witnesses into their empty-endpoint and
positive-denominator alternatives. In the positive/positive case, the result is
carried by
\texttt{BEDC.Derived.FieldUp.RatContinuation.RatHistoryCarrier\_continuation\_closed}.

If the left input is $\hsame\emp$, transport the continuation relation to an
empty-left continuation and use
\texttt{BEDC.FKernel.Cont.cont\_left\_unit\_result}; the result is classified
with the right endpoint. If the right endpoint is positive, transport that
positive carrier across the resulting $\hsame$ comparison by
\texttt{BEDC.Derived.RatUp.RatHistoryCarrier\_hsame\_transport}; if it is
empty, the result belongs to the empty disjunct. The cases with the right input
$\hsame\emp$ are symmetric, using
\texttt{BEDC.FKernel.Cont.Units.cont\_right\_unit\_result}.

The two displayed unit laws are the corresponding one-sided cases of the same
argument. The theorem
\texttt{BEDC.FKernel.Cont.cont\_left\_unit\_result} gives $r\hsame h$ from
$\Cont(\emp,h,r)$, and
\texttt{BEDC.FKernel.Cont.Units.cont\_right\_unit\_result} gives the same
comparison from $\Cont(h,\emp,r)$. In each case, the carrier part of the
classifier is obtained from the assumed carrier for $h$ and the closure
argument above for $r$.

For associativity, first apply closure to the two inner continuations to obtain
$\mathsf{RatDenomUnitCarrier}(hk)$ and
$\mathsf{RatDenomUnitCarrier}(k\ell)$. Apply closure again to the two outer
continuations to obtain carrier witnesses for $left$ and $right$.

Finally use \texttt{BEDC.FKernel.Cont.cont\_assoc\_hsame} on the four displayed
continuation witnesses. Its comparison $left\hsame right$, together with the
two carrier witnesses just constructed, is exactly
$\mathsf{RatDenomUnitClassifier}(left,right)$.
\end{proof}

\begin{theorem}[Rat denominator unit classifier is closed under continuation]
\label{thm:rat-denom-unit-classifier-continuation-closed}
If two pairs of endpoints are classified by
$\mathsf{RatDenomUnitClassifier}$ and the two displayed continuations match
the left endpoints and the right endpoints,
$$
\mathsf{RatDenomUnitClassifier}(h,h')\land
\mathsf{RatDenomUnitClassifier}(k,k')\land
\Cont(h,k,r)\land\Cont(h',k',r'),
$$
then the two continuation results are again classified:
$$
\mathsf{RatDenomUnitClassifier}(r,r').
$$
\end{theorem}
\leanchecked{BEDC.Derived.FieldUp.RatDenomUnitClassifier\_continuation\_closed}

\begin{theorem}[Rat denominator unit classifier append commutativity]
\label{thm:field-rat-denom-unit-classifier-append-comm}
If two histories \(h,k:\mathsf{BHist}\) are carried by the adjoined empty-unit
denominator carrier, then the two append products classify each other:
$$
\mathsf{RatDenomUnitClassifier}
\bigl(\mathsf{append}(h,k),\mathsf{append}(k,h)\bigr).
$$
\end{theorem}
\begin{proof}
Split the two carrier witnesses into the empty endpoint branch and the
rational-denominator branch. The empty--empty and mixed cases use the
continuation left and right unit laws to compare each append result with the
nonempty endpoint, or with \(\emp\) in the empty--empty case. When both
endpoints are rational denominator histories, use the denominator continuation
commutative classifier theorem on the two canonical append continuations.
In every case, continuation closure supplies the carried evidence for both
append results.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.RatDenomUnitClassifier\_append\_comm}
