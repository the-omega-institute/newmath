\begin{theorem}[Field square zero rejects nonzero factor]
\label{thm:field-square-zero-rejects-factor-nonzero}
For a BEDC history-level $\FieldUp$ candidate with the hypotheses of
\autoref{thm:field-one-sided-zero-product-cancel-from-apartness}, together
with classifier transport of $\mathsf{NonZero}_F$ and contradiction of
$\mathsf{NonZero}_F(0)$, every $a:\mathsf{BHist}$ satisfying
$$
\mathsf{mul}(a,a)\hsame0
$$
has $\mathsf{NonZero}_F(a)\to\bot$. Moreover, for
$$
\mathsf{ApartZero}_F(h) := (h\hsame0\to\bot),
$$
the same displayed square-zero classifier refutes
$\mathsf{ApartZero}_F(\mathsf{mul}(a,a))$.
\end{theorem}
\begin{proof}
Assume $z:\mathsf{mul}(a,a)\hsame0$ and
$p_a:\mathsf{NonZero}_F(a)$. Apply
\autoref{thm:field-one-sided-zero-product-cancel-from-apartness} to $z$ with
the left-factor witness $p_a$. The left-factor clause returns $a\hsame0$.

Transport $p_a$ along this classifier equality. The classifier-transport
stability for $\mathsf{NonZero}_F$ gives a witness of
$\mathsf{NonZero}_F(0)$.

The field certificate supplies contradiction of $\mathsf{NonZero}_F(0)$, so
the transported witness closes the nonzero-factor clause with $\bot$.

For the apartness clause, take
$q:\mathsf{ApartZero}_F(\mathsf{mul}(a,a))$. Since
$\mathsf{ApartZero}_F(h)$ means $h\hsame0\to\bot$, direct application of $q$
to the displayed square-zero classifier $z$ gives $\bot$.
\end{proof}

\begin{theorem}[Field square nonzero iff factor nonzero]
\label{thm:field-square-nonzero-iff-factor-nonzero}
For a BEDC history-level $\FieldUp$ candidate satisfying the hypotheses of
\autoref{thm:field-product-nonzero-excludes-zero-factors} and
\autoref{thm:field-product-nonzero-of-nonzero-factors}, including a
constructive bridge family
$$
\alpha_t:\bigl(t\hsame0\to\bot\bigr)\to\mathsf{NonZero}_F(t),
$$
every $a:\mathsf{BHist}$ satisfies
$$
\mathsf{NonZero}_F(\mathsf{mul}(a,a))
\Longleftrightarrow
\mathsf{NonZero}_F(a).
$$
\end{theorem}
\begin{proof}
For the forward direction, assume
$p_{aa}:\mathsf{NonZero}_F(\mathsf{mul}(a,a))$.
Apply \autoref{thm:field-product-nonzero-excludes-zero-factors} to the square
product with both factors equal to $a$. The left-factor clause gives
$a\hsame0\to\bot$.

Apply the bridge-family instance $\alpha_a$ to this apartness witness. The
result is $\mathsf{NonZero}_F(a)$.

For the reverse direction, assume $p_a:\mathsf{NonZero}_F(a)$. Apply
\autoref{thm:field-product-nonzero-of-nonzero-factors} to the same witness
$p_a$ in both factor slots. The resulting product witness is exactly
$\mathsf{NonZero}_F(\mathsf{mul}(a,a))$.
\end{proof}
