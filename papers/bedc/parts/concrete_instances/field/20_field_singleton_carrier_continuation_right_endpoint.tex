\begin{theorem}[Field singleton carrier continuation right endpoint iff]
\label{thm:field-singleton-carrier-continuation-right-endpoint-iff}
For histories $P,Q,R:\mathsf{BHist}$, if $P$ is a singleton field carrier
endpoint and $\ContR(P,Q,R)$, then $R$ is a singleton field carrier endpoint
exactly when $Q$ is a singleton field carrier endpoint:
$$
\mathsf{FieldSingletonCarrier}(P)\Rightarrow
\ContR(P,Q,R)\Rightarrow
\bigl(\mathsf{FieldSingletonCarrier}(R)\Longleftrightarrow
\mathsf{FieldSingletonCarrier}(Q)\bigr).
$$
\end{theorem}
\begin{proof}
The carrier witness for $P$ identifies the left endpoint with $\emp$. The
displayed continuation therefore has the same result as the right endpoint
$Q$. Transporting singleton-carrier membership along this equality gives the
forward implication, while singleton carrier continuation closure gives the
reverse implication.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.FieldSingletonCarrier\_continuation\_right\_endpoint\_iff}

\begin{theorem}[Field singleton classifier continuation left endpoint transport]
\label{thm:field-singleton-classifier-continuation-left-endpoint-transport}
For histories $P,Q,R,S:\mathsf{BHist}$, if $Q$ is a singleton field carrier
endpoint and $\ContR(P,Q,R)$, then singleton classification from the result
endpoint to $S$ is equivalent to singleton classification from the left
endpoint to $S$:
$$
\mathsf{FieldSingletonCarrier}(Q)\Rightarrow
\ContR(P,Q,R)\Rightarrow
\bigl(\mathsf{FieldSingletonClassifier}(R,S)\Longleftrightarrow
\mathsf{FieldSingletonClassifier}(P,S)\bigr).
$$
\end{theorem}
\begin{proof}
The carrier witness for $Q$ identifies the continuation tail with $\emp$, so
the continuation result is internally the same as $P$. Unfolding the singleton
classifier on both sides reduces the comparison to transporting the left
carrier field and the endpoint-sameness field along this result equality.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.FieldSingletonClassifier\_continuation\_left\_endpoint\_transport\_iff}
