\begin{theorem}[RatUp FieldUp affine normalizer action kernel normal subgroup]
\label{thm:ratup-fieldup-affine-normalizer-action-kernel-normal-subgroup}
For Definition~\ref{def:ratup-fieldup-concrete-exit-object}, fix
$A:\mathsf{AffAut}_F$. The carrier $K_A$ of
Definition~\ref{def:ratup-fieldup-affine-normalizer-action-kernel-carrier} is a
normal subgroup package inside the affine normalizer $N_F(C_F(A))$:
\begin{enumerate}
\item $K_A(\mathrm{id}_F)$.
\item If $K_A(T)$ and $K_A(U)$, then $K_A(T\circ_{\mathrm{aff}}U)$.
\item If $K_A(T)$, then $K_A(\iota_F(T))$.
\item If $K_A(T)$, $N_F(C_F(A))(U)$, and $T\equiv_{\mathrm{aff}}U$, then
$K_A(U)$.
\item If $K_A(T)$ and $N_F(C_F(A))(S)$, then
$K_A(S\circ_{\mathrm{aff}}T\circ_{\mathrm{aff}}\iota_F(S))$.
\end{enumerate}
\end{theorem}
\begin{proof}
The carrier part of each clause is supplied by
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-subgroup}. Its
identity, product, inverse, and classifier-transport rows keep the first four
displayed witnesses inside $N_F(C_F(A))$; for the conjugation word, use the
inverse row on $S$ and the product row twice.

For the identity element, use the identity law in
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-action}. For every
$X$ with $C_F(A)(X)$, it gives
$\mathrm{Act}_{\mathrm{id}_F}(X)\equiv_{\mathrm{aff}}X$, so the pointwise
kernel predicate holds.

For products, assume $K_A(T)$ and $K_A(U)$ and fix $X$ with $C_F(A)(X)$. The
composition law in
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-action} rewrites
$\mathrm{Act}_{T\circ_{\mathrm{aff}}U}(X)$ as
$\mathrm{Act}_T(\mathrm{Act}_U(X))$. The $U$-kernel hypothesis identifies the
inner acted element with $X$, and the $T$-kernel hypothesis then identifies the
outer action with $X$.

For inverses, start with $K_A(T)$. The inverse theorem
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-action-inverse}
identifies $\mathrm{Act}_{\iota_F(T)}$ as a two-sided inverse of
$\mathrm{Act}_T$ on $C_F(A)$. Since $\mathrm{Act}_T$ is pointwise the classifier
identity there, the inverse action is also pointwise the classifier identity.

For classifier transport, assume $K_A(T)$, $N_F(C_F(A))(U)$, and
$T\equiv_{\mathrm{aff}}U$. The action classifier congruence theorem
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-action-classifier-congruence}
transports the pointwise equality
$\mathrm{Act}_T(X)\equiv_{\mathrm{aff}}X$ to
$\mathrm{Act}_U(X)\equiv_{\mathrm{aff}}X$ for each $X$ in $C_F(A)$.

Finally assume $K_A(T)$ and $N_F(C_F(A))(S)$. For any $X$ with $C_F(A)(X)$,
apply the composition law twice to rewrite
$$
\mathrm{Act}_{S\circ_{\mathrm{aff}}T\circ_{\mathrm{aff}}\iota_F(S)}(X)
$$
as
$$
\mathrm{Act}_S(\mathrm{Act}_T(\mathrm{Act}_{\iota_F(S)}(X))).
$$

The inverse theorem keeps $\mathrm{Act}_{\iota_F(S)}(X)$ inside $C_F(A)$. Apply
the $K_A(T)$ hypothesis to this middle acted element, obtaining
$\mathrm{Act}_T(\mathrm{Act}_{\iota_F(S)}(X))
\equiv_{\mathrm{aff}}\mathrm{Act}_{\iota_F(S)}(X)$. Acting by $S$ on both sides
and using classifier congruence reduces the displayed conjugation action to
$\mathrm{Act}_S(\mathrm{Act}_{\iota_F(S)}(X))$.

The right inverse cancellation clause of
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-action-inverse}
identifies this last term with $X$. Hence
$S\circ_{\mathrm{aff}}T\circ_{\mathrm{aff}}\iota_F(S)$ lies in $K_A$, giving
normality inside the concrete affine normalizer.
\end{proof}

\end{theorem}

\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-action-representation}
rewrites the action of $\iota_F(S)\circ_{\mathrm{aff}}T$ on $X$ as
$\mathrm{Act}_{\iota_F(S)}(\mathrm{Act}_T(X))$. The kernel hypothesis identifies
this element with $X$.

Apply $\mathrm{Act}_S$ to that equality, using the classifier-exact action row
of
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-action-representation}.
The inverse-action cancellation rows from
\autoref{thm:ratup-fieldup-restricted-normalizer-action-automorphism-kernel-exactness}

$\mathrm{Act}_S(\mathrm{Act}_{\iota_F(S)}(\mathrm{Act}_T(X)))$ to
$\mathrm{Act}_T(X)$. Hence $\mathrm{Act}_T(X)$ and $\mathrm{Act}_S(X)$ are
classifier-equal, which is the required pointwise equality after symmetry of

For the reverse direction, assume the pointwise equality of the two restricted

It remains to show pointwise fixed action. Fix $X$ with $C_F(A)(X)$. By the
representation law, $\mathrm{Act}_{\iota_F(S)\circ_{\mathrm{aff}}T}(X)$ is
classifier-equal to $\mathrm{Act}_{\iota_F(S)}(\mathrm{Act}_T(X))$. Transport the
assumed pointwise equality
$\mathrm{Act}_S(X)\equiv_{\mathrm{aff}}\mathrm{Act}_T(X)$ through the

The result identifies
$\mathrm{Act}_{\iota_F(S)}(\mathrm{Act}_T(X))$ with
$\mathrm{Act}_{\iota_F(S)}(\mathrm{Act}_S(X))$. The remaining inverse-action

\autoref{thm:ratup-fieldup-restricted-normalizer-action-automorphism-kernel-exactness}
reduces this last element to $X$. Therefore
$\iota_F(S)\circ_{\mathrm{aff}}T$ fixes every centralizer element and lies in
$K_A$.
\end{proof}

\begin{theorem}[RatUp FieldUp affine normalizer same-action classifier stability]

\begin{enumerate}
  \item if $N_F(C_F(A))(S)$, then $S\equiv_{\rho_A}S$;
  \item if $N_F(C_F(A))(S)$, $N_F(C_F(A))(T)$, and

  \item if $N_F(C_F(A))(S)$, $N_F(C_F(A))(T)$, $N_F(C_F(A))(U)$,
  $S\equiv_{\rho_A}T$, and $T\equiv_{\rho_A}U$, then
  $S\equiv_{\rho_A}U$;
  \item if $N_F(C_F(A))(S)$, $N_F(C_F(A))(T)$,
  $S\equiv_{\rho_A}T$, $S\equiv_{\mathrm{aff}}S'$, and
  $T\equiv_{\mathrm{aff}}T'$, then the normalizer carrier transports to
  $N_F(C_F(A))(S')$ and $N_F(C_F(A))(T')$, and
  $S'\equiv_{\rho_A}T'$.
\end{enumerate}

and stable under endpoint replacement by the affine classifier without adding a
quotient type.
\end{theorem}
\begin{proof}
Use
\autoref{thm:ratup-fieldup-affine-normalizer-same-action-classifier-determinacy}
as the translation between $S\equiv_{\rho_A}T$ and pointwise equality of the
restricted actions on $C_F(A)$.

For reflexivity, fix $X:\mathsf{AffAut}_F$ with $C_F(A)(X)$. The two restricted
actions are the same expression, so affine classifier reflexivity gives
$\mathrm{Act}_S(X)\equiv_{\mathrm{aff}}\mathrm{Act}_S(X)$. Determinacy packs
this pointwise row into $S\equiv_{\rho_A}S$.

For symmetry, determinacy expands $S\equiv_{\rho_A}T$ into
$\mathrm{Act}_S(X)\equiv_{\mathrm{aff}}\mathrm{Act}_T(X)$ for every carried
centralizer element $X$. Affine classifier symmetry reverses each displayed
comparison, and determinacy repacks the reversed pointwise row as
$T\equiv_{\rho_A}S$.

For transitivity, expand the two hypotheses by determinacy. At each carried
centralizer element $X$, affine classifier transitivity composes
$\mathrm{Act}_S(X)\equiv_{\mathrm{aff}}\mathrm{Act}_T(X)$ with
$\mathrm{Act}_T(X)\equiv_{\mathrm{aff}}\mathrm{Act}_U(X)$. Determinacy then
packs the resulting pointwise comparison into $S\equiv_{\rho_A}U$.

For endpoint replacement, first use the normalizer carrier transport rows on
$S\equiv_{\mathrm{aff}}S'$ and $T\equiv_{\mathrm{aff}}T'$ to place $S'$ and
$T'$ in $N_F(C_F(A))$. Determinacy expands $S\equiv_{\rho_A}T$ to the pointwise
restricted-action equality between $S$ and $T$.

Now fix $X$ with $C_F(A)(X)$. Apply
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-action-classifier-congruence}
to the acting endpoint comparison $S\equiv_{\mathrm{aff}}S'$ and the reflexive
endpoint comparison $X\equiv_{\mathrm{aff}}X$; this compares
$\mathrm{Act}_S(X)$ with $\mathrm{Act}_{S'}(X)$. Apply the same congruence row
to $T\equiv_{\mathrm{aff}}T'$ and the same carried $X$; this compares
$\mathrm{Act}_T(X)$ with $\mathrm{Act}_{T'}(X)$.

Affine classifier symmetry and transitivity compose these two endpoint
comparisons with the pointwise equality between $\mathrm{Act}_S(X)$ and
$\mathrm{Act}_T(X)$. The result is
$\mathrm{Act}_{S'}(X)\equiv_{\mathrm{aff}}\mathrm{Act}_{T'}(X)$ for every
carried centralizer element $X$, and determinacy packs it as
$S'\equiv_{\rho_A}T'$.
\end{proof}

\begin{theorem}[RatUp FieldUp affine normalizer point stabilizer subgroup]
\label{thm:ratup-fieldup-affine-centralizer-normalizer-point-stabilizer-subgroup}
For Definition~\ref{def:ratup-fieldup-concrete-exit-object}, fix a certified
affine automorphism $A:\mathsf{AffAut}_F$ and a carried centralizer element
$X:\mathsf{AffAut}_F$ with $C_F(A)(X)$. The carrier
$\mathrm{Stab}_{A,X}$ of
Definition~\ref{def:ratup-fieldup-affine-centralizer-normalizer-point-stabilizer}
is a subgroup package inside $N_F(C_F(A))$:
\begin{enumerate}
\item $\mathrm{Stab}_{A,X}(\mathrm{id}_F)$.
\item If $\mathrm{Stab}_{A,X}(T)$ and $\mathrm{Stab}_{A,X}(U)$, then
$\mathrm{Stab}_{A,X}(T\circ_{\mathrm{aff}}U)$.
\item If $\mathrm{Stab}_{A,X}(T)$, then
$\mathrm{Stab}_{A,X}(\iota_F(T))$.
\item If $\mathrm{Stab}_{A,X}(T)$, $N_F(C_F(A))(U)$, and
$T\equiv_{\mathrm{aff}}U$, then $\mathrm{Stab}_{A,X}(U)$.
\end{enumerate}
\end{theorem}
\begin{proof}
The normalizer component of each row is supplied by
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-subgroup}. Its
identity, product, inverse, and classifier-transport clauses keep the displayed
affine words inside $N_F(C_F(A))$.

For the identity row, use the identity law in
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-action}. Applied to
the fixed carried point $X$, it gives
$\mathrm{Act}_{\mathrm{id}_F}(X)\equiv_{\mathrm{aff}}X$.

For product closure, assume $\mathrm{Stab}_{A,X}(T)$ and
$\mathrm{Stab}_{A,X}(U)$. The composition law in
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-action} identifies
the product action on $X$ with
$\mathrm{Act}_T(\mathrm{Act}_U(X))$. The $U$-stabilizer row fixes $X$, and the
acting-coordinate part of
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-action-classifier-congruence}
transports the remaining action to $\mathrm{Act}_T(X)$. The $T$-stabilizer row
then identifies the result with $X$.

For inverse closure, assume $\mathrm{Stab}_{A,X}(T)$. The inverse theorem
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-action-inverse}
supplies cancellation between the actions of $T$ and $\iota_F(T)$ on the
centralizer carrier. Since $T$ fixes $X$ up to $\equiv_{\mathrm{aff}}$, endpoint
transport by
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-action-classifier-congruence}
turns the inverse cancellation row into
$\mathrm{Act}_{\iota_F(T)}(X)\equiv_{\mathrm{aff}}X$.

For classifier transport, assume $\mathrm{Stab}_{A,X}(T)$,
$N_F(C_F(A))(U)$, and $T\equiv_{\mathrm{aff}}U$. The endpoint and acting
coordinates in
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-action-classifier-congruence}
transport $\mathrm{Act}_T(X)\equiv_{\mathrm{aff}}X$ to
$\mathrm{Act}_U(X)\equiv_{\mathrm{aff}}X$. This gives the transported
stabilizer row inside the normalizer.
\end{proof}

\begin{theorem}[RatUp FieldUp normalizer kernel collapses orbit action]
\label{thm:ratup-fieldup-normalizer-kernel-collapses-orbit-action}
For Definition~\ref{def:ratup-fieldup-concrete-exit-object}, fix a certified
affine automorphism $A:\mathsf{AffAut}_F$ in the concrete $\FieldUp(\RatUp)$
exit object. Let $C_F(A)$ be the affine centralizer carrier, let
$N_F(C_F(A))$ be its normalizer carrier, let $K_A$ be the kernel carrier of
Definition~\ref{def:ratup-fieldup-affine-normalizer-action-kernel-carrier}, let
$\mathrm{Act}$ be the restricted normalizer action, and let
$\sim_{N_F(A)}$ be the orbit relation of
\autoref{def:ratup-fieldup-affine-centralizer-normalizer-orbit}. For carried
centralizer elements $X,Y:\mathsf{AffAut}_F$, a normalizer element
$S:\mathsf{AffAut}_F$, and a kernel element $T:\mathsf{AffAut}_F$, if
$C_F(A)(X)$, $C_F(A)(Y)$, $X\sim_{N_F(A)}Y$, $N_F(C_F(A))(S)$, and $K_A(T)$,
then $C_F(A)(\mathrm{Act}_{S\circ_{\mathrm{aff}}T}(X))$,
$C_F(A)(\mathrm{Act}_S(Y))$, and
$$
\mathrm{Act}_{S\circ_{\mathrm{aff}}T}(X)\sim_{N_F(A)}\mathrm{Act}_S(Y).
$$
\end{theorem}
\begin{proof}
The kernel hypothesis contains the carrier witness $N_F(C_F(A))(T)$. Combine
it with $N_F(C_F(A))(S)$ by the product row of
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-subgroup}; this
certifies $S\circ_{\mathrm{aff}}T$ as a normalizer element. The action theorem
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-action} then gives the
centralizer carrier of $\mathrm{Act}_{S\circ_{\mathrm{aff}}T}(X)$, and the same
theorem applied to $S$ gives the carrier of $\mathrm{Act}_S(Y)$.

For the classifier comparison, use the composition row of
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-action} to identify
$\mathrm{Act}_{S\circ_{\mathrm{aff}}T}(X)$ with
$\mathrm{Act}_S(\mathrm{Act}_T(X))$. Since $K_A(T)$ is pointwise trivial on
carried centralizer elements, it identifies $\mathrm{Act}_T(X)$ with $X$. The
action classifier congruence theorem transports this equality through the
action by $S$, so
$\mathrm{Act}_{S\circ_{\mathrm{aff}}T}(X)$ is classifier-equal to
$\mathrm{Act}_S(X)$.

Apply
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-orbit-action-invariance}
to the original orbit witness $X\sim_{N_F(A)}Y$ and the normalizer element $S$.
It gives $\mathrm{Act}_S(X)\sim_{N_F(A)}\mathrm{Act}_S(Y)$. Transport the left
endpoint across the classifier equality above using the orbit classifier
congruence and transport rows, yielding the displayed orbit relation for
$\mathrm{Act}_{S\circ_{\mathrm{aff}}T}(X)$ and $\mathrm{Act}_S(Y)$.
\end{proof}

Per-object template: traditional target kernel collapse for the induced

$\equiv_{\mathrm{aff}}$; stability kernel factors act trivially after passing

\mathrm{Act}_{S\circ_{\mathrm{aff}}T}(X)\sim_{N_F(A)}\mathrm{Act}_S(Y).
