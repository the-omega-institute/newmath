$$
\mathsf{SemanticNameCert}
\bigl(h\mapsto
  \mathsf{FieldApartZero}(\mathsf{append}(L,\mathsf{append}(h,R)))\bigr)
\bigl(h\mapsto
  \mathsf{FieldApartZero}(\mathsf{append}(L,\mathsf{append}(h,R)))\bigr)
\bigl(h\mapsto
  \mathsf{FieldApartZero}(\mathsf{append}(L,\mathsf{append}(h,R)))\bigr)
\bigl(h,k\mapsto h\hsame k\bigr).
$$
Its source, pattern, and ledger policy are all the displayed contextual
apart-zero predicate, and its classifier is $\hsame$.
\end{theorem}
\begin{proof}
Apply \autoref{thm:ratup-fieldup-canonical-machine-checkable-exit} to fix
Definition~\ref{def:ratup-fieldup-concrete-exit-object} as the concrete
$\mathsf{BHist}$-anchored $\FieldUp(\RatUp)$ exit object over
$\mathsf{RatUpFieldSource}_F$. This pins the row to the canonical exit
certificate rather than to a schematic field interface.

Next apply
\autoref{thm:field-apartzero-two-sided-context-semantic-namecert} to the
histories $L$ and $R$ and the apart-zero witness $p$. Its source, pattern,
ledger policy, and classifier are exactly the four displayed arguments of
$\mathsf{SemanticNameCert}$.

Combining the fixed canonical exit object with that semantic certificate gives
row membership for the contextual apart-zero predicate inside the concrete
$\FieldUp(\RatUp)$ exit certificate.
\end{proof}

\begin{theorem}[RatUp FieldUp affine centralizer normalizer SubgroupUp tower certificate]
\label{thm:ratup-fieldup-affine-centralizer-normalizer-subgroup-tower-certificate}
For Definition~\ref{def:ratup-fieldup-concrete-exit-object}, fix a certified
affine automorphism $A:\mathsf{AffAut}_F$ in the concrete $\FieldUp(\RatUp)$
exit object. Let $C_F(A)$ be the affine centralizer carrier from
\autoref{def:ratup-fieldup-affine-centralizer-carrier}, let $N_F(C_F(A))$ be
the two-sided affine normalizer carrier from
\autoref{def:ratup-fieldup-affine-centralizer-normalizer-carrier}, and define
$$
\mathrm{Act}_T(X)=(T\circ_{\mathrm{aff}}X)\circ_{\mathrm{aff}}\iota_F(T).
$$
Then the inclusion
$$
C_F(A)\leq N_F(C_F(A))
$$
forms the concrete quotient-ready $\SubgroupUp$ tower whose lower subgroup rows
are the centralizer identity, product, inverse, and classifier-transport rows,
whose upper subgroup rows are the normalizer identity, product, inverse, and
classifier-transport rows, whose carrier inclusion is
$$
C_F(A)(T)\Rightarrow N_F(C_F(A))(T),
$$
and whose action rows state that every $T$ in $N_F(C_F(A))$ preserves $C_F(A)$
under $\mathrm{Act}_T$, the identity and product action laws hold on the
centralizer carrier, the classifier well-definedness condition for
$\mathrm{Act}$ holds on normalizer and centralizer representatives, and the
inverse action by $\iota_F(T)$ cancels $\mathrm{Act}_T$ on both sides.
\end{theorem}
\begin{proof}
The lower carrier is the concrete affine centralizer of $A$. Apply
\autoref{thm:ratup-fieldup-affine-centralizer-subgroup}; it supplies the
identity, product, inverse, and classifier-transport rows for $C_F(A)$.

The upper carrier is the two-sided affine normalizer of this centralizer. Apply
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-subgroup}; it supplies
the corresponding subgroup rows for $N_F(C_F(A))$. The carrier inclusion is
exactly \autoref{thm:ratup-fieldup-affine-centralizer-normalizer-inclusion}.

It remains to expose the quotient-ready action data. Apply
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-action} for
normalizer preservation of the centralizer and for the identity and composition
action laws. Apply
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-action-certificate} for
classifier well-definedness of $\mathrm{Act}$ and for inverse-action
cancellation on the restricted centralizer carrier.

These rows are precisely the concrete $\SubgroupUp$ input demanded by the
quotient-group construction: a lower subgroup, an upper subgroup, an inclusion
of carriers, and an action certificate by upper representatives on lower
representatives.
\end{proof}

Per-object template: traditional target normal-subgroup input for quotient
groups; BEDC source concrete $\FieldUp(\RatUp)$ together with $\SubgroupUp$;
carrier pair $C_F(A),N_F(C_F(A))$; classifier $\equiv_{\mathrm{aff}}$; ledger
fixed affine element, conjugators, inverse rows, and action witnesses; exactness
as the normalizer action on the centralizer; status $\mathsf{ProofTarget}$.

$\mathsf{fieldSingletonEmptyCarrier}(L)$,

$\mathsf{fieldSingletonEmptyCarrier}(U)$,
$\mathsf{fieldSingletonEmptyCarrier}(V)$, and
$\mathsf{fieldSingletonEmptyNonZero}

$$
\mathsf{fieldSingletonEmptyClassifier}
  (\mathsf{append}(L,\mathsf{fieldSingletonEmptyMul}(h,k)),
   \mathsf{append}(R,out))
\Longrightarrow
\mathsf{False}.
$$

$$
\mathsf{fieldSingletonEmptyClassifier}
  (\mathsf{append}(L,\mathsf{fieldSingletonEmptyOne}),
   \mathsf{append}(R,out))

\mathsf{False}.

proof-indexed inverse endpoint satisfies
$$
\mathsf{fieldSingletonEmptyClassifier}
  (\mathsf{append}(L,\mathsf{fieldSingletonEmptyInv}(h,p)),
   \mathsf{append}(R,out))
\Longrightarrow
\mathsf{False}.
$$
\end{theorem}
\begin{proof}
First cancel the carried output contexts. Apply
\autoref{thm:field-singleton-empty-nonzero-append-context-cancel-iff} to

$\mathsf{fieldSingletonEmptyCarrier}(out)$.
