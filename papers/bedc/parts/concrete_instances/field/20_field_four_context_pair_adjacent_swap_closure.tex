\begin{theorem}[RatUp FieldUp four context-pair adjacent-swap closure]
\label{thm:ratup-fieldup-four-context-pair-adjacent-swap-closure}
For the concrete denominator tuple
$(C_D^u,R_D^u,\mu_D^u,e_D^u,N_D^u)$ and the contextual action $\Gamma$ of
\autoref{def:ratup-fieldup-denominator-contextual-action-instance}, let
$h,l_0,r_0,l_1,r_1,l_2,r_2,l_3,r_3:\mathsf{BHist}$. Suppose $C_D^u(h)$ and
suppose that $(l_0,r_0)$, $(l_1,r_1)$, $(l_2,r_2)$, and $(l_3,r_3)$ are
carried denominator context pairs. Let
$$
H_{0123}:=
\Gamma_{\emp,\emp,l_3,r_3}
(\Gamma_{\emp,\emp,l_2,r_2}
(\Gamma_{\emp,\emp,l_1,r_1}
(\Gamma_{\emp,\emp,l_0,r_0}(h)))).
$$
Let
$$
H_{0213}:=
\Gamma_{\emp,\emp,l_3,r_3}
(\Gamma_{\emp,\emp,l_1,r_1}
(\Gamma_{\emp,\emp,l_2,r_2}
(\Gamma_{\emp,\emp,l_0,r_0}(h)))).
$$
Then the middle-swap histories are denominator-classified:
$$
R_D^u(H_{0123},H_{0213}).
$$
Their strict-support readbacks agree:
$$
N_D^u(H_{0123})
\Longleftrightarrow
N_D^u(H_{0213}).
$$
\end{theorem}
\begin{proof}
First carry the action of $(l_0,r_0)$ on $h$. Apply
\autoref{thm:field-rat-denominator-contextual-action-composition-support-laws}
with empty endpoints and the carried pair data to keep
$\Gamma_{\emp,\emp,l_0,r_0}(h)$ inside $C_D^u$.

Next apply
\autoref{thm:ratup-fieldup-context-pair-adjacent-swap-exit-row} over that
carried base, using $(l_1,r_1)$ and $(l_2,r_2)$ as the adjacent pair to be
exchanged and $(l_3,r_3)$ as the common outer pair. This gives the displayed
classifier between $H_{0123}$ and $H_{0213}$.

The contextual action pair transport rows carry the same classifier through the
common outer context and preserve the strict-support predicate on transported
denominator histories. Applying the support part of the adjacent-swap row, then
reading it through the common outer transport, gives the displayed support
equivalence.
\end{proof}

\autoref{thm:rat-denom-unit-contextual-action-three-pair-reverse-order-coherence},
\autoref{thm:ratup-fieldup-context-pair-adjacent-swap-exit-row}, and
\autoref{thm:field-rat-denominator-contextual-action-composition-support-laws};
exactness predicate the displayed classifier and support iff; standard bridge
deferred; status $\mathsf{ProofTarget}$.

\begin{theorem}[RatUp FieldUp four context-pair adjacent-swap coverage]
\label{thm:ratup-fieldup-four-context-pair-adjacent-swap-coverage}
For the concrete denominator tuple
$(C_D^u,R_D^u,\mu_D^u,e_D^u,N_D^u)$ and the contextual action $\Gamma$ of
\autoref{def:ratup-fieldup-denominator-contextual-action-instance}, let
$h,l_0,r_0,l_1,r_1,l_2,r_2,l_3,r_3:\mathsf{BHist}$. Suppose $C_D^u(h)$ and
suppose that $(l_0,r_0)$, $(l_1,r_1)$, $(l_2,r_2)$, and $(l_3,r_3)$ are
carried denominator context pairs. Let
$$
H_{0123}:=
\Gamma_{\emp,\emp,l_3,r_3}
(\Gamma_{\emp,\emp,l_2,r_2}
(\Gamma_{\emp,\emp,l_1,r_1}
(\Gamma_{\emp,\emp,l_0,r_0}(h)))).
$$
Let
$$
H_{1023}:=
\Gamma_{\emp,\emp,l_3,r_3}
(\Gamma_{\emp,\emp,l_2,r_2}
(\Gamma_{\emp,\emp,l_0,r_0}
(\Gamma_{\emp,\emp,l_1,r_1}(h)))).
$$
Let
$$
H_{0213}:=
\Gamma_{\emp,\emp,l_3,r_3}
(\Gamma_{\emp,\emp,l_1,r_1}
(\Gamma_{\emp,\emp,l_2,r_2}
(\Gamma_{\emp,\emp,l_0,r_0}(h)))).
$$
Let
$$
H_{0132}:=
\Gamma_{\emp,\emp,l_2,r_2}
(\Gamma_{\emp,\emp,l_3,r_3}
(\Gamma_{\emp,\emp,l_1,r_1}
(\Gamma_{\emp,\emp,l_0,r_0}(h)))).
$$
Then the three adjacent-generator rows from $H_{0123}$ are classified:
$$
R_D^u(H_{0123},H_{1023})
\land
R_D^u(H_{0123},H_{0213})
\land
R_D^u(H_{0123},H_{0132}).
$$
The corresponding strict-support readbacks agree:
$$
N_D^u(H_{0123})
\Longleftrightarrow
N_D^u(H_{1023}).
$$
$$
N_D^u(H_{0123})
\Longleftrightarrow
N_D^u(H_{0213}).
$$
$$
N_D^u(H_{0123})
\Longleftrightarrow
N_D^u(H_{0132}).
$$
\end{theorem}
\begin{proof}
For the first adjacent generator, apply
\autoref{thm:ratup-fieldup-context-pair-adjacent-swap-exit-row} to the carried
pairs $(l_0,r_0)$, $(l_1,r_1)$, and $(l_2,r_2)$ over $h$. Then carry the
result through the common outer context $(l_3,r_3)$ using the contextual action
pair transport classifier row
\autoref{thm:rat-denom-unit-contextual-action-pair-transport-classifier}. The
carrier premises needed for the outer action come from
\autoref{thm:field-rat-denominator-contextual-action-composition-support-laws}.

For the middle adjacent generator, use
\autoref{thm:ratup-fieldup-four-context-pair-adjacent-swap-closure}. It is
exactly the displayed comparison between $H_{0123}$ and $H_{0213}$ under the
same four carried context pairs.

For the last adjacent generator, first apply
\autoref{thm:field-rat-denominator-contextual-action-composition-support-laws}
to carry the common two-step base
$\Gamma_{\emp,\emp,l_1,r_1}(\Gamma_{\emp,\emp,l_0,r_0}(h))$. Then apply
\autoref{thm:rat-denom-unit-contextual-action-empty-pair-commutes} to the two
outer empty-endpoint actions $(l_2,r_2)$ and $(l_3,r_3)$. This gives the
classifier row between $H_{0123}$ and $H_{0132}$.

The strict-support rows are obtained in parallel. For the first and middle
generators, read the support part of the adjacent-swap row through the same
contextual action pair transport support row
\autoref{thm:field-rat-denominator-contextual-action-pair-transport-carrier-support}.
For the last generator, apply
\autoref{thm:rat-denom-unit-contextual-action-empty-pair-strict-support-commutes}
over the carried two-step base. These three applications give the displayed
strict-support equivalences.
\end{proof}

\begin{theorem}[RatUp FieldUp four context-pair full reverse-order coherence]
\label{thm:ratup-fieldup-four-context-pair-full-reverse-order-coherence}
For the concrete denominator tuple
$(C_D^u,R_D^u,\mu_D^u,e_D^u,N_D^u)$ and the contextual action $\Gamma$ of
\autoref{def:ratup-fieldup-denominator-contextual-action-instance}, let
$h,l_0,r_0,l_1,r_1,l_2,r_2,l_3,r_3:\mathsf{BHist}$. Suppose $C_D^u(h)$ and
suppose that $(l_0,r_0)$, $(l_1,r_1)$, $(l_2,r_2)$, and $(l_3,r_3)$ are
carried denominator context pairs. Let
$$
H_{0123}:=
\Gamma_{\emp,\emp,l_3,r_3}
(\Gamma_{\emp,\emp,l_2,r_2}
(\Gamma_{\emp,\emp,l_1,r_1}
(\Gamma_{\emp,\emp,l_0,r_0}(h)))).
$$
Let
$$
H_{3210}:=
\Gamma_{\emp,\emp,l_0,r_0}
(\Gamma_{\emp,\emp,l_1,r_1}
(\Gamma_{\emp,\emp,l_2,r_2}
(\Gamma_{\emp,\emp,l_3,r_3}(h)))).
$$
Then the forward and reverse four-context histories are denominator-classified:
$$
R_D^u(H_{0123},H_{3210}).
$$
Their strict-support readbacks agree:
$$
N_D^u(H_{0123})
\Longleftrightarrow
N_D^u(H_{3210}).
$$
\end{theorem}
\begin{proof}
Introduce the six intermediate four-context histories with application orders
$1023$, $1203$, $1230$, $2130$, $2310$, and $3210$. Each is obtained from the
preceding order by one adjacent exchange in the carried context-pair ledger.

For the classifier spine, instantiate
\autoref{thm:ratup-fieldup-four-context-pair-adjacent-swap-coverage} at the
current ordering for the six adjacent exchanges
$0123 \to 1023 \to 1203 \to 1230 \to 2130 \to 2310 \to 3210$.

The carrier premises at every stage are supplied by
\autoref{thm:field-rat-denominator-contextual-action-composition-support-laws},
since each current base is an iterated $\Gamma$-action on the same carried
history and the same four carried denominator context pairs.

Chain the six displayed $R_D^u$ rows by transitivity of the concrete
denominator classifier. The resulting composite row has endpoints
$H_{0123}$ and $H_{3210}$, giving the first conclusion.

For strict support, read the support part of each adjacent-swap coverage row
at the same six current orderings. Compose the six equivalences in sequence,
with the inverse direction of each equivalence used exactly when the selected
adjacent row is traversed opposite to its displayed orientation. The composite
equivalence has endpoints $N_D^u(H_{0123})$ and $N_D^u(H_{3210})$.
\end{proof}
