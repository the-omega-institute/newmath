\chapter{A Concrete Naming Certificate for $\FieldUp$}
\label{ch:concrete-instances-field-namecert}

The field interface $\FieldUp$ extends \autoref{ch:concrete-instances-commring-namecert} with the multiplicative inverse field. The inverse domain is a unary certificate $\mathsf{NonZero}_F(a)$ derived from constructive apartness of $a$ from $0$; inverse formation is licensed only by an explicit witness $p:\mathsf{NonZero}_F(a)$. This proof-relevant discipline replaces the classical "$a \ne 0$ implies $a$ invertible" formulation, which would otherwise require LEM. The rational carrier from \autoref{ch:concrete-instances-rat-namecert} supplies the candidate carrier and classifier source; a $\FieldUp$ certificate is present only when commutative-ring structure, constructive nonzero evidence, inverse closure, inverse laws indexed by nonzero witnesses, and the ledger are supplied over that source.

\section{Carrier definition}

\begin{definition}[Field carrier]
\label{def:field-carrier}
Use a commutative ring setup extended with a constructive apartness predicate and define $\mathsf{NonZero}_F(a)$ as the unary certificate obtained from apartness of $a$ from $0$. The inverse datum has type $\mathsf{inv}(a,p)$ for $p:\mathsf{NonZero}_F(a)$, so no inverse is formed without the nonzero witness.
\end{definition}

\section{Concrete singleton empty-history field schema instance}
\label{sec:field-concrete-singleton-instance}

\begin{definition}[Singleton empty-history field schema instance]
\label{def:singleton-empty-history-field-schema-instance}
Let $A_{\emp}$ be the singleton history source of
Definition~\ref{def:list-singleton-history-source} at $a_0=\emp$. Define the
classifier
$$
\rho_{\emp}(h,k)\Longleftrightarrow
A_{\emp}(h)\land A_{\emp}(k)\land\hsame(h,k).
$$
Take $\mathsf{add}$, $\mathsf{neg}$, $0$, $\mathsf{mul}$, and $1$ to be the
constant empty-history operations of
Theorem~\ref{thm:singleton-empty-history-commring-laws}. Define
$\mathsf{FieldSingletonNonZero}(h)$ by
$$
\mathsf{FieldSingletonNonZero}(h)\Longleftrightarrow
A_{\emp}(h)\land \hsame(h,\mathsf{e0}(\emp)).
$$
The proof-indexed inverse operation is total on explicit witnesses:
$\mathsf{inv}(h,p):=\emp$ for
$p:\mathsf{FieldSingletonNonZero}(h)$. Theorem~\ref{thm:field-singleton-nonzero-absurd}
shows that this witness domain is empty on the singleton carrier.
\end{definition}

\begin{theorem}[Field singleton classifier empty endpoints]
\label{thm:field-singleton-classifier-empty-endpoints}
For histories $h,k:\mathsf{BHist}$, the singleton field classifier satisfies
$$
\rho_{\emp}(h,k)
\quad\Longleftrightarrow\quad
h\hsame\emp\land k\hsame\emp.
$$
\end{theorem}
\begin{proof}
The forward direction unpacks the two carrier fields of
$\rho_{\emp}(h,k)$; these are precisely the endpoint comparisons
$h\hsame\emp$ and $k\hsame\emp$.

Conversely, the two endpoint comparisons supply the singleton carrier fields.
Composing $h\hsame\emp$ with the symmetry of $k\hsame\emp$ gives
$h\hsame k$, completing the classifier package.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.FieldSingletonClassifier\_empty\_endpoints\_iff}

\begin{theorem}[Field singleton classifier visible-left exclusion]
\label{thm:field-singleton-classifier-append-visible-left-absurd}
For histories $p,q,h:\mathsf{BHist}$, neither
$\rho_{\emp}(\mathsf{append}(p,\mathsf{e0}(q)),h)$ nor
$\rho_{\emp}(\mathsf{append}(p,\mathsf{e1}(q)),h)$ can hold.
\end{theorem}
\begin{proof}
Unfold the singleton field classifier. In either case, the left carrier field
identifies the displayed append endpoint with $\emp$. Empty-result inversion
for append then forces the visible right component, either $\mathsf{e0}(q)$ or
$\mathsf{e1}(q)$, to be $\emp$. The constructors of $\mathsf{BHist}$ separate
visible endpoints from the empty endpoint, so each case is impossible.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.FieldSingletonClassifier\_append\_visible\_left\_absurd}

\begin{theorem}[Field singleton classifier visible-prefix exclusion]
\label{thm:field-singleton-classifier-append-visible-prefix-absurd}
For histories $p,q,h:\mathsf{BHist}$, neither
$\rho_{\emp}(\mathsf{append}(\mathsf{e0}(p),q),h)$ nor
$\rho_{\emp}(\mathsf{append}(\mathsf{e1}(p),q),h)$ can hold.
\end{theorem}
\begin{proof}
Unfold the singleton field classifier. In either case, the left carrier field
identifies the displayed append endpoint with $\emp$. Empty-result inversion
for append then forces the visible prefix, either $\mathsf{e0}(p)$ or
$\mathsf{e1}(p)$, to be $\emp$. Constructor separation for
$\mathsf{BHist}$ rules out both alternatives.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.FieldSingletonClassifier\_append\_visible\_prefix\_absurd}

\begin{definition}[Singleton empty-classifier nonzero domain]
\label{def:singleton-empty-classifier-nonzero-domain}
Reuse $A_{\emp}$ and $\rho_{\emp}$ from
Definition~\ref{def:singleton-empty-history-field-schema-instance}. Define
the classifier-level nonzero witness domain by
$$
\mathsf{FieldSingletonEmptyNonZero}(h)\Longleftrightarrow
\rho_{\emp}(h,\emp)\to\mathsf{False}.
$$
The proof-indexed inverse on this domain is the constant empty-history
operation $\mathsf{inv}_{\emp}(h,p):=\emp$ for
$p:\mathsf{FieldSingletonEmptyNonZero}(h)$.
\end{definition}

\begin{theorem}[Singleton empty-classifier carrier obstruction]
\label{thm:singleton-empty-classifier-carrier-absurd}
For every $h:\mathsf{BHist}$,
$A_{\emp}(h)\to\mathsf{FieldSingletonEmptyNonZero}(h)\to\mathsf{False}$.
\end{theorem}
\begin{proof}
Assume $A_{\emp}(h)$. The singleton source is generated at $\emp$, so this
carrier witness supplies the endpoint comparison $h\hsame\emp$.

The same singleton source gives the reflexive carrier witness at $\emp$.
Together with the endpoint comparison above, these data assemble the classifier
package $\rho_{\emp}(h,\emp)$.

Now assume $p:\mathsf{FieldSingletonEmptyNonZero}(h)$. By
Definition~\ref{def:singleton-empty-classifier-nonzero-domain}, $p$ maps the
classifier package $\rho_{\emp}(h,\emp)$ to $\mathsf{False}$, which closes the
claim.
\end{proof}

This gives the concrete $\mathsf{BHist}$ $\FieldUp$ schema a direct
nonzero-domain law target for the singleton empty-classifier instance.

\begin{theorem}[Singleton nonzero endpoint obstruction]
\label{thm:field-singleton-nonzero-absurd}
For every $h:\mathsf{BHist}$, a witness
$p:\mathsf{FieldSingletonNonZero}(h)$ leads to contradiction.
\end{theorem}
\begin{proof}
The witness contains both the singleton-carrier endpoint $h\hsame\emp$ and
the zero-head endpoint $h\hsame\mathsf{e0}(\emp)$. Compose the first endpoint
in reverse with the second endpoint to obtain
$\emp\hsame\mathsf{e0}(\emp)$. The constructors $\emp$ and
$\mathsf{e0}(\emp)$ are disjoint in the closed inductive $\mathsf{BHist}$
kernel, so no such equality exists.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.FieldSingletonNonZero\_absurd}

\begin{theorem}[Singleton empty nonzero endpoint obstruction]
\label{thm:field-singleton-nonzero-empty-endpoint-absurd}
For every $h:\mathsf{BHist}$, if $h\hsame\emp$ and
$p:\mathsf{fieldSingletonEmptyNonZero}(h)$, then contradiction follows.
\end{theorem}
\begin{proof}
The endpoint $h\hsame\emp$ supplies the source side of the empty singleton
classifier. The empty target is classified by reflexivity, and the same
endpoint supplies the classifier equality from $h$ to $\emp$. This builds the
forbidden classifier input to $p$.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.fieldSingletonEmptyNonZero\_empty\_endpoint\_absurd}

\begin{theorem}[Singleton empty-history field nonzero domain is empty]
\label{thm:singleton-empty-history-field-nonzero-domain-empty}
For every $h:\mathsf{BHist}$,
$\mathsf{FieldSingletonNonZero}(h)\Longrightarrow\mathsf{False}$.
\end{theorem}
\begin{proof}
Unfold
Definition~\ref{def:singleton-empty-history-field-schema-instance}. A witness of
$\mathsf{FieldSingletonNonZero}(h)$ supplies $A_{\emp}(h)$ and
$\hsame(h,\mathsf{e0}(\emp))$.

The singleton source component $A_{\emp}(h)$ gives $\hsame(h,\emp)$. Apply
Theorem~\ref{thm:field-singleton-nonzero-absurd} to the two displayed
endpoints, and the contradiction follows.
\end{proof}

This block supplies the per-object certificate: traditional target empty
inverse domain of the terminal proof-indexed field; BEDC source the concrete
$\mathsf{BHist}$ singleton field schema; carrier $A_{\emp}$; classifier
$\rho_{\emp}$; stability obligation nonzero contradiction; ledger no nonzero
witness; core theorem Theorem~\ref{thm:singleton-empty-history-field-nonzero-domain-empty};
exactness predicate witness-domain emptiness; standard bridge terminal field;
status ProofTarget.

\begin{theorem}[Singleton empty nonzero empty endpoint obstruction]
\label{thm:field-singleton-empty-nonzero-empty-endpoint-absurd}
For every $h:\mathsf{BHist}$, if $h\hsame\emp$, then the proof-defined
predicate $\mathsf{fieldSingletonEmptyNonZero}(h)$ is impossible.
\end{theorem}
\begin{proof}
The endpoint $h\hsame\emp$, together with reflexivity at $\emp$, gives the
classifier witness $\rho_{\emp}(h,\emp)$. The predicate
$\mathsf{fieldSingletonEmptyNonZero}(h)$ is defined as the exclusion of exactly
that classifier witness, so it yields a contradiction.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.fieldSingletonEmptyNonZero\_empty\_endpoint\_absurd}

\begin{theorem}[Singleton nonzero forces the zero-headed empty endpoint]
\label{thm:field-singleton-nonzero-empty-e0-forced}
For every $a:\mathsf{BHist}$, a witness of
$\mathsf{FieldSingletonNonZero}(a)$ forces
$\emp\hsame\mathsf{e0}(\emp)$.
\end{theorem}
\begin{proof}
Unpack the witness into the singleton-carrier endpoint $a\hsame\emp$ and the
zero-headed endpoint $a\hsame\mathsf{e0}(\emp)$. Symmetry of the first endpoint
and transitivity with the second endpoint give the displayed classifier.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.FieldSingletonNonZero\_empty\_e0\_forced}

\begin{theorem}[Singleton empty-history field schema laws]
\label{thm:singleton-empty-history-field-schema-laws}
The tuple
$$
(A_{\emp},\rho_{\emp},\mathsf{add},0,\mathsf{neg},\mathsf{mul},1,
\mathsf{FieldSingletonNonZero},\mathsf{inv})
$$
satisfies the $\FieldUp$ schema laws over the concrete $\mathsf{BHist}$
singleton carrier.
\end{theorem}
\begin{proof}
The commutative-ring fields are exactly the fields supplied by
Theorem~\ref{thm:singleton-empty-history-commring-laws} for the displayed
carrier, classifier, and constant empty-history operations.

The nonzero exclusion at $0$ follows from
Theorem~\ref{thm:singleton-empty-history-field-nonzero-domain-empty} by taking
$h=0$.

Nonzero transport along $\rho_{\emp}$ is vacuous. Given
$\rho_{\emp}(h,k)$ and $p:\mathsf{FieldSingletonNonZero}(h)$,
Theorem~\ref{thm:singleton-empty-history-field-nonzero-domain-empty} gives
contradiction; from contradiction one obtains the required witness at $k$.

Inverse closure is direct because $\mathsf{inv}(h,p)$ is definitionally
$\emp$, whose carrier endpoint is supplied by the singleton source. Inverse
congruence is likewise direct because both displayed inverse endpoints are
$\emp$, hence are related by $\rho_{\emp}$.

The left and right inverse laws have the same proof shape. Each law is indexed
by a witness $p:\mathsf{FieldSingletonNonZero}(h)$, but both products are the
constant empty-history product and $1=\emp$. The classifier witness is therefore
the reflexive singleton endpoint, and no classical inverse is formed outside
the proof-indexed domain.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.singleton\_empty\_history\_field\_schema\_laws}

\begin{definition}[Classifier-complement singleton nonzero domain]
\label{def:field-singleton-empty-nonzero}
For the same singleton carrier and classifier, define
$\mathsf{fieldSingletonEmptyNonZero}(h)$ by
$$
\mathsf{fieldSingletonEmptyNonZero}(h)\Longleftrightarrow
(\rho_{\emp}(h,\emp)\to\bot).
$$
The multiplication, unit, and proof-indexed inverse are the constant
empty-history operations
$\mathsf{fieldSingletonEmptyMul}(x,y):=\emp$,
$\mathsf{fieldSingletonEmptyOne}:=\emp$, and
$\mathsf{fieldSingletonEmptyInv}(h,p):=\emp$.
\end{definition}
\leandef{BEDC.Derived.FieldUp.fieldSingletonEmptyNonZero}

\begin{theorem}[Singleton empty-classifier append split]
\label{thm:field-singleton-empty-classifier-append-split-empty}
For histories $p,q,h:\mathsf{BHist}$, the singleton empty classifier
$\rho_{\emp}(\mathsf{append}(p,q),h)$ holds exactly when
$p\hsame\emp$, $q\hsame\emp$, and $h$ lies in the singleton carrier
$A_{\emp}$.
\end{theorem}
\begin{proof}
Unfold the classifier. In the forward direction,
$\rho_{\emp}(\mathsf{append}(p,q),h)$ supplies
$\mathsf{append}(p,q)\hsame\emp$ and the carrier witness for $h$.
Empty-result inversion for append splits the first endpoint into
$p\hsame\emp$ and $q\hsame\emp$.

Conversely, from $p\hsame\emp$ and $q\hsame\emp$, append empty-result
inversion reconstructs $\mathsf{append}(p,q)\hsame\emp$. Together with the
given carrier witness for $h$, transitivity through $\emp$ supplies the
classifier equality from $\mathsf{append}(p,q)$ to $h$.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.fieldSingletonEmptyClassifier\_append\_split\_empty\_iff}

\begin{theorem}[Classifier-complement singleton field schema laws]
\label{thm:field-singleton-empty-schema-laws}
For the tuple
$$
(A_{\emp},\rho_{\emp},\mathsf{fieldSingletonEmptyNonZero},
\mathsf{fieldSingletonEmptyMul},\mathsf{fieldSingletonEmptyOne},
\mathsf{fieldSingletonEmptyInv})
$$
the singleton carrier is inhabited by $\emp$; the predicate
$\mathsf{fieldSingletonEmptyNonZero}(\emp)$ is contradictory; the predicate is
transported along $\rho_{\emp}$; the inverse of every proof-indexed element is
again in $A_{\emp}$; and the left and right products with the proof-indexed
inverse are classified as the empty-history unit.
\end{theorem}
\begin{proof}
The carrier inhabitance clause is the reflexive endpoint
$\hsame(\emp,\emp)$, so $\emp$ lies in $A_{\emp}$.

For contradiction at the unit, suppose
$p:\mathsf{fieldSingletonEmptyNonZero}(\emp)$. The displayed classifier
witness $\rho_{\emp}(\emp,\emp)$ is built from the two reflexive carrier
endpoints and reflexivity of $\hsame(\emp,\emp)$. Applying $p$ to this witness
closes the contradiction.

For transport, assume $\rho_{\emp}(h,k)$ and
$p:\mathsf{fieldSingletonEmptyNonZero}(h)$. Given a classifier witness
$\rho_{\emp}(k,\emp)$, compose the classifier equality from
$\rho_{\emp}(h,k)$ with the equality from $\rho_{\emp}(k,\emp)$ to obtain
$\rho_{\emp}(h,\emp)$. The predicate $p$ then gives contradiction, hence the
transported predicate holds at $k$.

Inverse closure is immediate because every inverse endpoint is definitionally
$\emp$, which is in the singleton carrier. The two inverse laws are immediate
for the same reason: both products are the constant empty-history product, the
unit is $\emp$, and the required classifier witnesses are built from
reflexivity of $\hsame(\emp,\emp)$.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.field\_singleton\_empty\_schema\_laws}

\begin{theorem}[Singleton empty-history FieldUp schema certificate]
\label{thm:singleton-empty-history-field-schema-certificate}
The tuple from
Definition~\ref{def:singleton-empty-history-field-schema-instance}, together
with the law package from
Theorem~\ref{thm:singleton-empty-history-field-schema-laws}, supplies the
carrier, source, pattern, classifier, stability, and ledger fields required by
Definition~\ref{def:field-certificate-obligations} over the concrete
$\mathsf{BHist}$ singleton source. Its ten-field certificate template is:
\begin{enumerate}
  \item Traditional target: the singleton proof-indexed $\FieldUp$ schema.
  \item BEDC source: $\mathsf{BHist}$ with $A_{\emp}$.
  \item Carrier choice: $A_{\emp}$.
  \item Classifier choice: $\rho_{\emp}$.
  \item Stability obligations: commutative-ring laws plus vacuous nonzero
  transport, inverse closure, and inverse laws.
  \item Ledger: the singleton endpoint and the empty nonzero witness domain.
  \item Core theorem list:
  Theorem~\ref{thm:singleton-empty-history-field-schema-laws} and the
  semantic certificate theorem for the same singleton carrier.
  \item Exactness predicate: singleton endpoint classification together with
  the $\mathsf{SemanticNameCert}$ source, pattern, and ledger fields.
  \item Standard bridge: the terminal proof-indexed schema.
  \item Status: CheckedCert.
\end{enumerate}
\end{theorem}
\begin{proof}
The source and carrier fields are the singleton history source
$A_{\emp}$ from Definition~\ref{def:singleton-empty-history-field-schema-instance}.
The classifier field is $\rho_{\emp}$, whose only endpoints are the empty
histories admitted by $A_{\emp}$.

For the algebraic stability fields, use
Theorem~\ref{thm:singleton-empty-history-commring-laws}. It supplies the
commutative-ring part for the constant empty-history operations on the same
carrier and classifier.

For the field-specific clauses, apply
Theorem~\ref{thm:singleton-empty-history-field-schema-laws}. Nonzero transport,
inverse closure, inverse congruence, and the left and right inverse laws all
begin with a witness $p:\mathsf{FieldSingletonNonZero}(h)$.

Theorem~\ref{thm:singleton-empty-history-field-nonzero-domain-empty} shows that
no such witness exists, while the inverse operation itself still has the total
proof-indexed form $\mathsf{inv}(h,p):=\emp$. The ledger therefore records only
the singleton endpoint and the empty witness domain, giving the required
concrete $\mathsf{BHist}$ FieldUp naming certificate template.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.fieldSingletonEmptyCarrier\_semanticNameCert}
