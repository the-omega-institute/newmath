\begin{theorem}[Singleton empty nonzero append right cancellation]
\label{thm:field-singleton-empty-nonzero-append-right-cancel}
For histories $P,Q:\mathsf{BHist}$, if $P$ lies in the singleton empty
carrier, then appending $P$ on the right does not change the
classifier-complement singleton nonzero predicate:
$$
\mathsf{fieldSingletonEmptyCarrier}(P)\Rightarrow
\bigl(
\mathsf{fieldSingletonEmptyNonZero}(\mathsf{append}(Q,P))
\Longleftrightarrow
\mathsf{fieldSingletonEmptyNonZero}(Q)
\bigr).
$$
\end{theorem}
\begin{proof}
In the forward direction, suppose $Q$ were classified against $\emp$. Combining
that endpoint witness with the displayed carrier witness for $P$ rebuilds a
singleton empty classifier witness for $\mathsf{append}(Q,P)$, contradicting
the appended complement predicate.

Conversely, any classifier witness for $\mathsf{append}(Q,P)$ supplies
$\mathsf{append}(Q,P)\hsame\emp$. Empty-result inversion for append extracts
$Q\hsame\emp$, and this reconstructs the forbidden classifier witness for
$Q$ against $\emp$.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.fieldSingletonEmptyNonZero\_append\_right\_cancel\_iff}

\label{thm:field-singleton-empty-nonzero-continuation-left-endpoint-cancel-iff}
For histories $P,Q,R:\mathsf{BHist}$, if $Q$ lies in the singleton empty
carrier and $\ContR(P,Q,R)$, then the classifier-complement singleton nonzero
predicate is unchanged from $P$ to $R$:
$$
\mathsf{fieldSingletonEmptyCarrier}(Q)\Rightarrow
\ContR(P,Q,R)\Rightarrow
\bigl(
\mathsf{fieldSingletonEmptyNonZero}(R)
\Longleftrightarrow
\mathsf{fieldSingletonEmptyNonZero}(P)
\bigr).
$$
\end{theorem}
\begin{proof}
For the forward implication, a forbidden classifier on $P$ combines with the
singleton empty carrier of $Q$ and the continuation ledger to rebuild a
forbidden classifier on $R$. Conversely, a forbidden classifier on $R$ is
transported back through the displayed continuation; append-empty inversion
then exposes the forbidden classifier on $P$.
\end{proof}

\begin{theorem}[Singleton empty nonzero continuation commutativity]
\label{thm:fieldup-singleton-empty-nonzero-continuation-comm}
For histories $P,Q,left,right:\mathsf{BHist}$, if
$$
\ContR(P,Q,left)
\quad\text{and}\quad
\ContR(Q,P,right),
$$
then the classifier-complement singleton nonzero predicate is invariant under
exchanging the two continuation inputs:
$$
\mathsf{fieldSingletonEmptyNonZero}(left)
\Longleftrightarrow
\mathsf{fieldSingletonEmptyNonZero}(right).
$$
\end{theorem}
\begin{proof}
For one direction, suppose $left$ is singleton nonzero and $right$ were empty.
Transport the displayed continuation for $right$ to an empty result and invert
the empty continuation. Both inputs are empty, so the opposite continuation has
empty result, contradicting singleton nonzero at $left$. The converse is the
same argument with the two continuations exchanged.
\end{proof}

$$

\mathsf{fieldSingletonEmptyNonZero}(\mathsf{append}(L,\mathsf{append}(Q,R)))

\mathsf{fieldSingletonEmptyNonZero}(Q)

the enclosed nonzero predicate.

$\mathsf{append}(L,\mathsf{append}(Q,R))$ against $\emp$ splits by
empty-result inversion for append. The inner split yields
$\mathsf{fieldSingletonEmptyCarrier}(Q)$, which rebuilds the forbidden
classifier witness for $Q$ against $\emp$.

\begin{theorem}[Classifier-complement singleton field schema laws]
\label{thm:field-singleton-empty-schema-laws}
For the tuple
$$
(A_{\emp},\rho_{\emp},\mathsf{fieldSingletonEmptyNonZero},
\mathsf{fieldSingletonEmptyMul},\mathsf{fieldSingletonEmptyOne},
\mathsf{fieldSingletonEmptyInv})
$$
the singleton carrier is inhabited by $\emp$; the predicate
$\mathsf{fieldSingletonEmptyNonZero}(\emp)$ is contradictory; the predicate is
transported along $\rho_{\emp}$; the inverse of every proof-indexed element is
again in $A_{\emp}$; and the left and right products with the proof-indexed
inverse are classified as the empty-history unit.
\end{theorem}
\begin{proof}
The carrier inhabitance clause is the reflexive endpoint
$\hsame(\emp,\emp)$, so $\emp$ lies in $A_{\emp}$.

For contradiction at the unit, suppose
$p:\mathsf{fieldSingletonEmptyNonZero}(\emp)$. The displayed classifier
witness $\rho_{\emp}(\emp,\emp)$ is built from the two reflexive carrier
endpoints and reflexivity of $\hsame(\emp,\emp)$. Applying $p$ to this witness
closes the contradiction.

For transport, assume $\rho_{\emp}(h,k)$ and
$p:\mathsf{fieldSingletonEmptyNonZero}(h)$. Given a classifier witness
$\rho_{\emp}(k,\emp)$, compose the classifier equality from
$\rho_{\emp}(h,k)$ with the equality from $\rho_{\emp}(k,\emp)$ to obtain
$\rho_{\emp}(h,\emp)$. The predicate $p$ then gives contradiction, hence the
transported predicate holds at $k$.

Inverse closure is immediate because every inverse endpoint is definitionally
$\emp$, which is in the singleton carrier. The two inverse laws are immediate
for the same reason: both products are the constant empty-history product, the
unit is $\emp$, and the required classifier witnesses are built from
reflexivity of $\hsame(\emp,\emp)$.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.field\_singleton\_empty\_schema\_laws}
