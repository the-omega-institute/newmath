\begin{theorem}[Singleton empty nonzero append split]
\label{thm:singleton-empty-nonzero-append-split}
For histories $L,Q:\mathsf{BHist}$, the classifier-complement singleton empty
nonzero predicate on an append splits exactly over the two factors:
$$
\mathsf{fieldSingletonEmptyNonZero}(\mathsf{append}(L,Q))
\Longleftrightarrow
\mathsf{fieldSingletonEmptyNonZero}(L)\lor
\mathsf{fieldSingletonEmptyNonZero}(Q).
$$
\end{theorem}
\begin{proof}
For the forward implication, split on the left factor. If $L=\emp$, a
classifier witness for $Q$ against $\emp$ rebuilds the forbidden classifier
witness for $\mathsf{append}(\emp,Q)$. If $L$ is visibly headed, constructor
separation already makes $L$ a singleton-empty nonzero witness.

For the reverse implication, any classifier witness for
\(\mathsf{append}(L,Q)\) against \(\emp\) splits by empty-result inversion for
append into carrier witnesses for both $L$ and $Q$. The chosen left or right
nonzero witness rejects the corresponding carrier witness.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.fieldSingletonEmptyNonZero\_append\_split\_iff}

\begin{theorem}[Singleton empty nonzero continuation endpoint split]
\label{thm:field-singleton-empty-nonzero-continuation-endpoint-split}
For histories $P,Q,R:\mathsf{BHist}$ with $\ContR(P,Q,R)$, the
classifier-complement singleton empty nonzero predicate on the continuation
result is exactly the disjunction of the corresponding endpoint predicates:
$$
\mathsf{fieldSingletonEmptyNonZero}(R)
\Longleftrightarrow
\mathsf{fieldSingletonEmptyNonZero}(P)\lor
\mathsf{fieldSingletonEmptyNonZero}(Q).
$$
\end{theorem}
\begin{proof}
Unfold the continuation ledger as $R=\mathsf{append}(P,Q)$. In the forward
direction, split on the concrete shape of $P$: the empty case transfers the
forbidden classifier to $Q$, while either visible head gives constructor
separation at the left endpoint. In the reverse direction, a classifier of
$R$ against $\emp$ splits by empty append inversion, and the selected endpoint
nonzero witness rejects the matching empty component.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.fieldSingletonEmptyNonZero\_continuation\_endpoint\_split\_iff}
