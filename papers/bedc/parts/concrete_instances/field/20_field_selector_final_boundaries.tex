\begin{theorem}[RatUp FieldUp selector-independent final boundary]
\label{thm:ratup-fieldup-selector-independent-final-boundary}
For Definition~\ref{def:ratup-fieldup-concrete-exit-object}, the concrete
denominator tuple $(C_D^u,R_D^u,\mu_D^u,e_D^u,N_D^u)$, histories
$h,k,\ell:\mathsf{BHist}$ with $C_D^u(h)$, $C_D^u(k)$, and $C_D^u(\ell)$,
selectors $s,t,u,v:\mathsf{BMark}$, and contexts
$L,R,r:\mathsf{BHist}$, let $E_q(h,k,\ell)$ be the bracketing endpoint
selector of Definition~\ref{def:ratup-fieldup-bracketing-endpoint-selector}
for each $q:\mathsf{BMark}$. If $\mathsf{ThreeSupp}_D^u(h,k,\ell)$ holds,
then the same concrete $\FieldUp(\RatUp)$ final exit boundary contains the
following rows:
\begin{enumerate}
  \item every row of
  \autoref{thm:ratup-fieldup-final-certificate-two-selector-noncollapse-boundary}
  for the selector pair $(s,t)$;
  \item every row of
  \autoref{thm:ratup-fieldup-final-certificate-two-selector-noncollapse-boundary}
  for the selector pair $(u,v)$;
  \item
  $$
  R_D^u(E_s(h,k,\ell),E_u(h,k,\ell))
  \land
  R_D^u(E_t(h,k,\ell),E_v(h,k,\ell));
  $$
  \item
  $$
  \mathsf{RatHistoryClassifier}(E_s(h,k,\ell),E_u(h,k,\ell))
  \land
  \mathsf{RatHistoryClassifier}(E_t(h,k,\ell),E_v(h,k,\ell)).
  $$
\end{enumerate}
\end{theorem}
\begin{proof}
First apply
\autoref{thm:ratup-fieldup-selector-pair-strict-exit-determinacy} to the
histories, carrier witnesses, support witness, selectors $s,t,u,v$, and
contexts $L,R:\mathsf{BHist}$. Its componentwise rows give
$R_D^u(E_s(h,k,\ell),E_u(h,k,\ell))$,
$R_D^u(E_t(h,k,\ell),E_v(h,k,\ell))$, and the two displayed
$\mathsf{RatHistoryClassifier}$ rows. The same application also supplies the
strict exit rows for both selector pairs $(s,t)$ and $(u,v)$.

Next apply
\autoref{thm:ratup-fieldup-bracketing-selector-complete-obstruction-package} to
the pair $(s,t)$ and then to the pair $(u,v)$ with the same contexts
$L,R:\mathsf{BHist}$. These two obstruction packages provide the contextual
singleton-classifier exclusions and contextual nonzero rows consumed by the
final certificate boundary for each selected pair.

Finally apply
\autoref{thm:ratup-fieldup-final-certificate-two-selector-noncollapse-boundary}
once with selectors $s,t$ and once with selectors $u,v$, keeping the same
concrete exit object, denominator tuple, histories, support witness, and context
$r:\mathsf{BHist}$. Combining those two final-boundary packages with the
componentwise comparison rows from the strict determinacy theorem yields one
selector-independent final boundary for the concrete
$\FieldUp(\RatUp)$ certificate.
\end{proof}

Per-object template: traditional target selector-independent final boundary
inside $\FieldUp(\mathsf{RatUp})$; BEDC source $\mathsf{BHist}$,
$\mathsf{BMark}$, $\mathsf{RatUp}$, and
Definition~\ref{def:ratup-fieldup-concrete-exit-object}; denominator tuple
$(C_D^u,R_D^u,\mu_D^u,e_D^u,N_D^u)$; selected endpoints
$E_s(h,k,\ell)$, $E_t(h,k,\ell)$, $E_u(h,k,\ell)$, and
$E_v(h,k,\ell)$; carrier rows the two final certificate boundaries plus the
componentwise denominator comparisons; classifier rows
$\mathsf{RatHistoryClassifier}$ for the two componentwise selector comparisons;
nonzero rows the contextual witnesses consumed by both final certificate
boundaries; stability obligations selector-pair strict exit determinacy,
bracketing selector complete obstruction, and final certificate two-selector
noncollapse; ledger selector pairs $(s,t)$ and $(u,v)$ plus contexts $L,R,r$;
exactness predicate one final FieldUp exit boundary independent of certified
selector-pair replacement; downstream consumer stable algebra-hierarchy exit
certificate; status $\mathsf{ProofTarget}$.

\begin{theorem}[RatUp FieldUp selector-chain final boundary]
\label{thm:ratup-fieldup-selector-chain-final-boundary}
For Definition~\ref{def:ratup-fieldup-concrete-exit-object}, the concrete
denominator tuple $(C_D^u,R_D^u,\mu_D^u,e_D^u,N_D^u)$, histories
$h,k,\ell:\mathsf{BHist}$ with $C_D^u(h)$, $C_D^u(k)$, and $C_D^u(\ell)$,
selectors $s,t,u,v,w,z:\mathsf{BMark}$, and contexts
$L,R,r:\mathsf{BHist}$, let $E_q(h,k,\ell)$ be the bracketing endpoint
selector of Definition~\ref{def:ratup-fieldup-bracketing-endpoint-selector}
for each $q:\mathsf{BMark}$. If $\mathsf{ThreeSupp}_D^u(h,k,\ell)$ holds,
then the same concrete $\FieldUp(\RatUp)$ final exit boundary contains the
following rows:
\begin{enumerate}
  \item every row of
  \autoref{thm:ratup-fieldup-final-certificate-two-selector-noncollapse-boundary}
  for the selector pair $(s,t)$;
  \item every row of
  \autoref{thm:ratup-fieldup-final-certificate-two-selector-noncollapse-boundary}
  for the selector pair $(u,v)$;
  \item every row of
  \autoref{thm:ratup-fieldup-final-certificate-two-selector-noncollapse-boundary}
  for the selector pair $(w,z)$;
  \item
  $$
  R_D^u(E_s(h,k,\ell),E_w(h,k,\ell))
  \land
  R_D^u(E_t(h,k,\ell),E_z(h,k,\ell));
  $$
  \item
  $$
  \mathsf{RatHistoryClassifier}(E_s(h,k,\ell),E_w(h,k,\ell))
  \land
  \mathsf{RatHistoryClassifier}(E_t(h,k,\ell),E_z(h,k,\ell)).
  $$
\end{enumerate}
\end{theorem}
\begin{proof}
First apply
\autoref{thm:ratup-fieldup-selector-independent-final-boundary} to the
selector comparison from $(s,t)$ to $(u,v)$ with the same concrete exit object,
denominator tuple, histories, carrier witnesses, support witness, and contexts
$L,R,r:\mathsf{BHist}$. This gives the final-boundary rows for $(s,t)$ and
$(u,v)$ together with the componentwise denominator and
$\mathsf{RatHistoryClassifier}$ rows from $E_s$ to $E_u$ and from $E_t$ to
$E_v$.

Next apply
\autoref{thm:ratup-fieldup-selector-independent-final-boundary} to the
selector comparison from $(u,v)$ to $(w,z)$ over the same data. This gives the
final-boundary rows for $(u,v)$ and $(w,z)$ together with the componentwise
denominator and $\mathsf{RatHistoryClassifier}$ rows from $E_u$ to $E_w$ and
from $E_v$ to $E_z$.

It remains to collapse the two adjacent replacements into endpoint rows. Use
the transitivity field of the concrete denominator classifier $R_D^u$ to
compose
$$
R_D^u(E_s(h,k,\ell),E_u(h,k,\ell))
\quad\text{with}\quad
R_D^u(E_u(h,k,\ell),E_w(h,k,\ell))
$$

and likewise compose the $E_t$ to $E_v$ row with the $E_v$ to $E_z$ row.

Finally use the transitivity field of $\mathsf{RatHistoryClassifier}$ in the
same two components. Conjunction introduction repacks the three
final-boundary row families and the two chained componentwise classifier
families into one stable final boundary for the concrete
$\FieldUp(\RatUp)$ exit certificate.
\end{proof}
