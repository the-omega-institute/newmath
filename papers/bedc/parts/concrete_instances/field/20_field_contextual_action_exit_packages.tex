\begin{theorem}[RatUp FieldUp contextual action composition support laws]
\label{thm:field-rat-denominator-contextual-action-composition-support-laws}

$(C_D^u,R_D^u,\mu_D^u,e_D^u,N_D^u)$ and the contextual action $\Gamma$ of
\autoref{def:ratup-fieldup-denominator-contextual-action-instance}, let

$p\hsame\emp$, $q\hsame\emp$, $p'\hsame\emp$, and $q'\hsame\emp$, and suppose
$C_D^u(h)$, $C_D^u(l)$, $C_D^u(r)$, $C_D^u(l')$, and $C_D^u(r')$. Then the
nested contextual action is carried:
$$
C_D^u(\Gamma_{p',q',l',r'}(\Gamma_{p,q,l,r}(h))).
$$
It is denominator-classified by the single contextual action with collected
append contexts:
$$
R_D^u(\Gamma_{p',q',l',r'}(\Gamma_{p,q,l,r}(h)),
\Gamma_{p',q',\mathsf{append}(l',l),\mathsf{append}(r,r')}(h)).
$$
Its strict support is exactly the five visible support sites:
$$
N_D^u(\Gamma_{p',q',l',r'}(\Gamma_{p,q,l,r}(h)))
\Longleftrightarrow
N_D^u(l')\lor N_D^u(l)\lor N_D^u(h)\lor N_D^u(r)\lor N_D^u(r').
$$
\end{theorem}
\begin{proof}
Apply
\autoref{thm:field-rat-denominator-contextual-action-composition} to the
empty-endpoint nested actions. Its classifier and support clauses identify the
two-layer action with the corresponding collected action once the inner and
outer carried contexts are available.

The carrier inputs are supplied by denominator-unit carrier closure under
$\mathsf{append}$. First close $l$ with $h$ and then with $r$; transport the
result along the two endpoint equalities $p\hsame\emp$ and $q\hsame\emp$ to get
the carried inner action. Close $l'$ with that carried inner action and then
with $r'$; transport along $p'\hsame\emp$ and $q'\hsame\emp$ to get the carried
nested action.

The same append-closure rows also carry
$\mathsf{append}(l',l)$ and $\mathsf{append}(r,r')$, so the collected single
action is a legitimate denominator action. Carrier transport along $\hsame$
keeps the concrete action endpoints inside $C_D^u$ throughout the comparison.

For strict support, apply bilateral support exactness first to the inner action:
its support is read as $N_D^u(l)\lor N_D^u(h)\lor N_D^u(r)$. Apply the same
exactness theorem to the outer action and substitute the inner readback. The
finite disjunction reassociates to
$N_D^u(l')\lor N_D^u(l)\lor N_D^u(h)\lor N_D^u(r)\lor N_D^u(r')$, which is the
displayed support law.
\end{proof}

\begin{theorem}[RatUp FieldUp canonical exit singleton context boundary]
\label{thm:ratup-fieldup-canonical-exit-singleton-context-boundary}
For Definition~\ref{def:ratup-fieldup-concrete-exit-object}, the source tuple
$\mathsf{RatUpFieldSource}_F$, and histories $L,R,h,k:\mathsf{BHist}$ with
$\mathsf{FieldSingletonCarrier}(L)$ and $\mathsf{FieldSingletonCarrier}(R)$, the
same canonical $\FieldUp(\RatUp)$ exit certificate includes the following
singleton-context boundary rows over $\mathsf{BHist}$.

$$
\mathsf{SemanticNameCert}
\bigl(h\mapsto
  \mathsf{FieldSingletonCarrier}(\mathsf{append}(L,\mathsf{append}(h,R)))\bigr)
\bigl(h\mapsto
  \mathsf{FieldSingletonCarrier}(\mathsf{append}(L,\mathsf{append}(h,R)))\bigr)
\bigl(h\mapsto

\bigl(h,k\mapsto
  \mathsf{FieldSingletonClassifier}
    (\mathsf{append}(L,\mathsf{append}(h,R)),
     \mathsf{append}(L,\mathsf{append}(k,R)))\bigr).
$$

\item The contextual carrier is exact:
$$
\mathsf{FieldSingletonCarrier}(\mathsf{append}(L,\mathsf{append}(h,R)))
\Longleftrightarrow
\mathsf{FieldSingletonCarrier}(h).
$$

\item The contextual classifier is exact:
$$
\mathsf{FieldSingletonClassifier}
  (\mathsf{append}(L,\mathsf{append}(h,R)),
   \mathsf{append}(L,\mathsf{append}(k,R)))
\Longleftrightarrow
\mathsf{FieldSingletonClassifier}(h,k).
$$
\end{enumerate}
\end{theorem}
\begin{proof}
Apply \autoref{thm:ratup-fieldup-canonical-machine-checkable-exit} to fix
Definition~\ref{def:ratup-fieldup-concrete-exit-object} as the concrete
$\mathsf{BHist}$-anchored $\FieldUp(\RatUp)$ exit object over

a row of the same canonical certificate, not as a separate schematic parameter.

For the semantic certificate row, apply
\autoref{thm:field-singleton-carrier-two-sided-context-semantic-namecert} to the
two carrier witnesses for $L$ and $R$. Its displayed source, pattern, ledger, and
classifier are exactly the contextual
$\mathsf{FieldSingletonCarrier}$ and $\mathsf{FieldSingletonClassifier}$ rows

For the two exactness rows, apply
\autoref{thm:field-singleton-two-sided-context-exactness} to the same witnesses
and the histories $h$ and $k$. The carrier projection gives the displayed
equivalence for $h$, and the classifier projection gives the displayed
equivalence for the pair $(h,k)$.

Thus the canonical exit consumes concrete $\mathsf{BHist}$, $\mathsf{append}$,
$\mathsf{FieldSingletonCarrier}$, and $\mathsf{FieldSingletonClassifier}$ rows

\end{proof}
