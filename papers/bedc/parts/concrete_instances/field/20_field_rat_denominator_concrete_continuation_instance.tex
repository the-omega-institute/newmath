\section{Concrete Rat denominator continuation instance}
\label{sec:field-rat-denominator-concrete-continuation-instance}

\begin{definition}[Concrete Rat denominator continuation instance]
\label{def:field-rat-denominator-concrete-continuation-instance}
The concrete denominator-side continuation instance for
$\FieldUp(\mathsf{RatUp})$ has carrier, classifier, operation, continuation
witness, and ledger predicate
$$
C_D(h)\Longleftrightarrow\mathsf{RatHistoryCarrier}(h),
$$
$$
R_D(h,k)\Longleftrightarrow\mathsf{RatHistoryClassifier}(h,k),
$$
$$
\mu_D(h,k):=\mathsf{append}(h,k),
$$
with the canonical continuation witness
$\Cont(h,k,\mu_D(h,k))$ and ledger predicate
$\mathsf{RatHistoryLedgerPolicy}$. The empty history $\emp$ is an external
classifier-unit endpoint for the operation and is not a carried denominator.
\end{definition}

\begin{theorem}[Rat denominator append concrete instance laws]
\label{thm:field-rat-denominator-append-concrete-instance-laws}
For histories $d,d',e,e',f,\rho,v:\mathsf{BHist}$, the concrete function data
of \autoref{def:field-rat-denominator-concrete-continuation-instance}, with
$\mu_D(h,k):=\mathsf{append}(h,k)$, satisfy the following append laws.
\begin{enumerate}
  \item If $C_D(d)$ and $C_D(e)$, then $C_D(\mu_D(d,e))$.
  \item If $C_D(d)$ and $C_D(e)$, then $R_D(\mu_D(d,e),\mu_D(e,d))$.
  \item If $C_D(d)$, $C_D(e)$, and $C_D(f)$, then
  $R_D(\mu_D(\mu_D(d,e),f),\mu_D(d,\mu_D(e,f)))$.
  \item If $R_D(d,d')$ and $R_D(e,e')$, then
  $R_D(\mu_D(d,e),\mu_D(d',e'))$.
  \item If $\mathsf{RatHistoryLedgerPolicy}(\rho,v)$, $C_D(e)$, and
  $R_D(e,e')$, then
  $\mathsf{RatHistoryLedgerPolicy}(\mu_D(\rho,e),\mu_D(v,e'))$.
  \item No carried history $u:\mathsf{BHist}$ can satisfy either one-sided
  classifier-unit law

  $$
  \forall x:\mathsf{BHist}.\;
  C_D(x)\to
  R_D(\mu_D(x,u),x)
  $$

  or

  $$
  \forall x:\mathsf{BHist}.\;
  C_D(x)\to
  R_D(\mu_D(u,x),x).
  $$
\end{enumerate}
\end{theorem}
\begin{proof}
Unfold $C_D$, $R_D$, and $\mu_D$. The operation is ordinary history append,
so the first five clauses are the concrete append package proved for
$\mathsf{RatHistoryCarrier}$, $\mathsf{RatHistoryClassifier}$, and
$\mathsf{RatHistoryLedgerPolicy}$.

Carrier closure uses the carried denominator closure of append. The
commutative classifier clause uses the unary append commutativity theorem
inside the two carried histories.

Associativity compares the two append bracketings by append associativity and
then repacks the result into $R_D$. Binary classifier congruence and ledger
closure are the corresponding Rat denominator append-closure clauses for the
classifier and ledger predicates.

For a one-sided carried unit, pass from the displayed append law to its
canonical continuation instance using $\Cont(x,u,\mu_D(x,u))$ or
$\Cont(u,x,\mu_D(u,x))$. Then
\autoref{thm:field-rat-denominator-continuation-unit-endpoint-empty} forces
$u\hsame\emp$, while \autoref{thm:rat-history-carrier-not-empty} rules out a
carried empty denominator.
\end{proof}

\begin{theorem}[Rat denominator continuation concrete instance laws]
\label{thm:field-rat-denominator-concrete-continuation-instance-laws}
For histories
$d,d',e,e',f,\rho,v,de,ed,ef,left,right,dep,rhoe,vep:\mathsf{BHist}$, the
relational continuation data of
\autoref{def:field-rat-denominator-concrete-continuation-instance} satisfy the
following laws.
\begin{enumerate}
  \item If $C_D(d)$, $C_D(e)$, and $\Cont(d,e,de)$, then $C_D(de)$.
  \item If $C_D(d)$, $C_D(e)$, $\Cont(d,e,de)$, and $\Cont(e,d,ed)$, then
  $R_D(de,ed)$.
  \item If $C_D(d)$, $C_D(e)$, $C_D(f)$, $\Cont(d,e,de)$,
  $\Cont(e,f,ef)$, $\Cont(de,f,left)$, and $\Cont(d,ef,right)$, then
  $R_D(left,right)$.
  \item If $R_D(d,d')$, $R_D(e,e')$, $\Cont(d,e,de)$, and
  $\Cont(d',e',dep)$, then $R_D(de,dep)$.
  \item If $\mathsf{RatHistoryLedgerPolicy}(\rho,v)$, $C_D(e)$,
  $R_D(e,e')$, $\Cont(\rho,e,rhoe)$, and $\Cont(v,e',vep)$, then
  $\mathsf{RatHistoryLedgerPolicy}(rhoe,vep)$.
  \item No carried history $u:\mathsf{BHist}$ can satisfy either one-sided
  classifier-unit law

  $$
  \forall x,r:\mathsf{BHist}.\;
  C_D(x)\to
  \Cont(x,u,r)\to
  R_D(r,x)
  $$

  or

  $$
  \forall x,r:\mathsf{BHist}.\;
  C_D(x)\to
  \Cont(u,x,r)\to
  R_D(r,x).
  $$
\end{enumerate}
\end{theorem}
\begin{proof}
Unfold $C_D$ and $R_D$. The relational presentation leaves each output
history named by its $\Cont$ witness instead of replacing it by append at the
statement level.

Carrier closure is
\autoref{thm:field-rat-denominator-continuation-carrier-closure}. The
commutative classifier clause is
\autoref{thm:field-rat-denominator-continuation-commutative-classifier}
applied to the two displayed continuations.

Associativity uses
\autoref{thm:field-rat-denominator-continuation-semigroup-laws} on the four
continuation witnesses. The theorem supplies the carried intermediate results
and the history-sameness comparison needed to read the two bracketings as
classifier-equivalent.

For binary classifier congruence, unpack $R_D(d,d')$ and $R_D(e,e')$. The
right input classifier supplies the matching denominator comparison required
by \autoref{thm:field-rat-denominator-continuation-classifier-closure}, and
the two displayed continuations give $R_D(de,dep)$.

The ledger clause applies
\autoref{thm:field-rat-denominator-continuation-ledger-closure} to
$\mathsf{RatHistoryLedgerPolicy}(\rho,v)$, the carried and classified right
tail, and the two displayed continuations.

Finally,
\autoref{thm:field-rat-denominator-continuation-unit-endpoint-empty} says that
any one-sided classifier-unit endpoint is $\hsame$ to $\emp$, and its carried
endpoint clause rules out such a history inside $\mathsf{RatHistoryCarrier}$.
\end{proof}

\begin{theorem}[Rat denominator append-continuation endpoint bridge]
\label{thm:field-rat-denominator-functional-relational-instance-equivalence}
For the concrete data of
\autoref{def:field-rat-denominator-concrete-continuation-instance}, if
$C_D(h)$ and $C_D(k)$, then every history $r:\mathsf{BHist}$ satisfies the
endpoint classifier bridge

$$
\Cont(h,k,r)\Longleftrightarrow R_D(r,\mu_D(h,k)).
$$

Thus the canonical append endpoint $\mu_D(h,k)$ and any displayed continuation
endpoint $r$ are identified exactly at the denominator classifier surface.
\end{theorem}
\begin{proof}
Unfold $C_D$, $R_D$, and $\mu_D$ only to read them as
$\mathsf{RatHistoryCarrier}$, $\mathsf{RatHistoryClassifier}$, and
$\mathsf{append}$. With the two carrier hypotheses in place,
\autoref{thm:rat-history-classifier-continuation-tail-iff} gives, for every
history $r$, exactly
$\Cont(h,k,r)\Longleftrightarrow R_D(r,\mu_D(h,k))$.

At the canonical endpoint, the same bridge reads the canonical continuation
witness as the reflexive denominator classifier comparison
$R_D(\mu_D(h,k),\mu_D(h,k))$.
\end{proof}

\begin{theorem}[Rat denominator append-relational unit-law equivalence]
\label{thm:field-rat-denominator-append-relational-unit-law-equivalence}
For the concrete data of
\autoref{def:field-rat-denominator-concrete-continuation-instance}, every
history $u:\mathsf{BHist}$ satisfying $C_D(u)$ has the following two one-sided
unit-law equivalences:

$$
\left(\forall x:\mathsf{BHist}.\;
C_D(x)\to R_D(\mu_D(x,u),x)\right)
\Longleftrightarrow
\left(\forall x,r:\mathsf{BHist}.\;
C_D(x)\to \Cont(x,u,r)\to R_D(r,x)\right)
$$

and

$$
\left(\forall x:\mathsf{BHist}.\;
C_D(x)\to R_D(\mu_D(u,x),x)\right)
\Longleftrightarrow
\left(\forall x,r:\mathsf{BHist}.\;
C_D(x)\to \Cont(u,x,r)\to R_D(r,x)\right).
$$

Thus the functional append presentation and the relational continuation
presentation carry exactly the same one-sided internal-unit obstruction for a
carried denominator endpoint.
\end{theorem}
\begin{proof}
Fix $u:\mathsf{BHist}$ with $C_D(u)$.

For the right-unit forward direction, assume the append family
$\forall x.\;C_D(x)\to R_D(\mu_D(x,u),x)$. Given $x$, $r$, $C_D(x)$, and
$\Cont(x,u,r)$, apply
\autoref{thm:field-rat-denominator-functional-relational-instance-equivalence}
to the carried pair $(x,u)$. This classifies the displayed endpoint $r$ with
the canonical append endpoint $\mu_D(x,u)$. The assumed append family
classifies $\mu_D(x,u)$ with $x$, and classifier transitivity gives $R_D(r,x)$.

For the right-unit reverse direction, assume the relational family
$\forall x,r.\;C_D(x)\to\Cont(x,u,r)\to R_D(r,x)$. At each carried $x$, use the
canonical continuation witness $\Cont(x,u,\mu_D(x,u))$ from
\autoref{def:field-rat-denominator-concrete-continuation-instance}. Applying
the relational family to this witness gives $R_D(\mu_D(x,u),x)$.

The left-unit equivalence is identical with the inputs reversed. The forward
direction compares any displayed $\Cont(u,x,r)$ endpoint to the canonical
append endpoint $\mu_D(u,x)$ by
\autoref{thm:field-rat-denominator-functional-relational-instance-equivalence},
then composes with the assumed append left-unit row by classifier transitivity.
The reverse direction applies the relational left-unit family to the canonical
witness $\Cont(u,x,\mu_D(u,x))$.
\end{proof}

This theorem fits the ten-field per-object template as follows: traditional
target append/$\Cont$ unit-law equivalence for strict denominator multiplication
support; BEDC source $\mathsf{BHist}$ together with $\Cont$,
$\mathsf{RatHistoryCarrier}$, and $\mathsf{RatHistoryClassifier}$; carrier
$C_D$; classifier $R_D$; stability obligation endpoint bridge plus classifier
transitivity; ledger $\mathsf{RatHistoryLedgerPolicy}$; core theorem
\autoref{thm:field-rat-denominator-append-relational-unit-law-equivalence};
exactness the displayed iff pair; standard bridge strict denominator semigroup
interface; status $\mathsf{ProofTarget}$.

For the ten-field exit certificate of
\autoref{def:ratup-fieldup-ten-field-exit-instance}, this theorem occupies only
the denominator endpoint-identification part of the classifier and stability
rows. The traditional target, BEDC source, carrier, ledger, core theorem list,
exactness predicate, standard bridge, and status rows are supplied by their
named concrete certificates, while the append and relational denominator laws
remain the checked packages of
\autoref{thm:field-rat-denominator-append-concrete-instance-laws} and
\autoref{thm:field-rat-denominator-concrete-continuation-instance-laws}.

\begin{definition}[Concrete Rat denominator strict semigroup interface]
\label{def:field-rat-denominator-strict-semigroup-interface}
The denominator multiplication interface carried by the concrete
$\FieldUp(\mathsf{RatUp})$ continuation data is the triple

$$
C_D(h):=\mathsf{RatHistoryCarrier}(h),
$$

$$
R_D(h,k):=\mathsf{RatHistoryClassifier}(h,k),
$$

and

$$
\mu_D(h,k):=\mathsf{append}(h,k).
$$

It is a strict denominator semigroup interface: the operation is the canonical
append endpoint, the classifier is the Rat denominator history classifier, and
the carried endpoints are exactly the Rat denominator histories. The interface
has no carried unit endpoint. The empty history is only an external classifier
endpoint, and no $u:\mathsf{BHist}$ with $C_D(u)$ can act as a one-sided unit
for the carried denominator multiplication.
\end{definition}

\begin{theorem}[Rat denominator carried unit obstruction package]
\label{thm:field-rat-denominator-carried-unit-obstruction-package}
For the concrete strict denominator semigroup interface of
\autoref{def:field-rat-denominator-strict-semigroup-interface}, every
$u:\mathsf{BHist}$ with $C_D(u)$ satisfies the following four impossibility
rows:
\begin{enumerate}
  \item the functional right-unit family

  $$
  \forall x:\mathsf{BHist}.\;
  C_D(x)\to R_D(\mu_D(x,u),x)
  $$

  implies $\mathsf{False}$;

  \item the functional left-unit family

  $$
  \forall x:\mathsf{BHist}.\;
  C_D(x)\to R_D(\mu_D(u,x),x)
  $$

  implies $\mathsf{False}$;

  \item the relational right-unit family

  $$
  \forall x,r:\mathsf{BHist}.\;
  C_D(x)\to \Cont(x,u,r)\to R_D(r,x)
  $$

  implies $\mathsf{False}$;

  \item the relational left-unit family

  $$
  \forall x,r:\mathsf{BHist}.\;
  C_D(x)\to \Cont(u,x,r)\to R_D(r,x)
  $$

  implies $\mathsf{False}$.
\end{enumerate}

The ten-field template represented by this certificate is: traditional target
strict denominator semigroup support; BEDC source $\mathsf{BHist}$ with
$\Cont$, $\mathsf{RatHistoryCarrier}$, and
$\mathsf{RatHistoryClassifier}$; carrier $C_D$; classifier $R_D$; stability
append/$\Cont$ endpoint bridge plus classifier transitivity; ledger
$\mathsf{RatHistoryLedgerPolicy}$; core theorem this package; exactness the
four impossibility rows; standard bridge strict non-unital denominator
multiplication; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Fix $u:\mathsf{BHist}$ with $C_D(u)$.

For the two functional rows, clause 6 of
\autoref{thm:field-rat-denominator-append-concrete-instance-laws} states that a
carried history cannot satisfy either one-sided append classifier-unit law.
Applying that clause to $u$ and the displayed functional right-unit or left-unit
family yields $\mathsf{False}$.

For the two relational rows, clause 6 of
\autoref{thm:field-rat-denominator-concrete-continuation-instance-laws} gives
the same contradiction for a carried history satisfying either one-sided
$\Cont$-based classifier-unit law. Applying it to the displayed relational
right-unit or left-unit family yields $\mathsf{False}$.

Finally,
\autoref{thm:field-rat-denominator-append-relational-unit-law-equivalence}
records that the append presentation and the relational continuation
presentation express the same one-sided carried-unit boundary. The right-unit
equivalence links the first and third rows, and the left-unit equivalence links
the second and fourth rows, so the four contradictions are one certificate-level
obstruction for the concrete denominator interface.
\end{proof}
