\section{Concrete Rat denominator continuation instance}
\label{sec:field-rat-denominator-concrete-continuation-instance}

\begin{definition}[Concrete Rat denominator continuation instance]
\label{def:field-rat-denominator-concrete-continuation-instance}
The concrete denominator-side continuation instance for
$\FieldUp(\mathsf{RatUp})$ has carrier, classifier, operation, continuation
witness, and ledger predicate
$$
C_D(h)\Longleftrightarrow\mathsf{RatHistoryCarrier}(h),
$$
$$
R_D(h,k)\Longleftrightarrow\mathsf{RatHistoryClassifier}(h,k),
$$
$$
\mu_D(h,k):=\mathsf{append}(h,k),
$$
with the canonical continuation witness
$\Cont(h,k,\mu_D(h,k))$ and ledger predicate
$\mathsf{RatHistoryLedgerPolicy}$. The empty history $\emp$ is an external
classifier-unit endpoint for the operation and is not a carried denominator.
\end{definition}

\begin{theorem}[Rat denominator append concrete instance laws]
\label{thm:field-rat-denominator-append-concrete-instance-laws}
For histories $d,d',e,e',f,\rho,v:\mathsf{BHist}$, the concrete function data
of \autoref{def:field-rat-denominator-concrete-continuation-instance}, with
$\mu_D(h,k):=\mathsf{append}(h,k)$, satisfy the following append laws.
\begin{enumerate}
  \item If $C_D(d)$ and $C_D(e)$, then $C_D(\mu_D(d,e))$.
  \item If $C_D(d)$ and $C_D(e)$, then $R_D(\mu_D(d,e),\mu_D(e,d))$.
  \item If $C_D(d)$, $C_D(e)$, and $C_D(f)$, then
  $R_D(\mu_D(\mu_D(d,e),f),\mu_D(d,\mu_D(e,f)))$.
  \item If $R_D(d,d')$ and $R_D(e,e')$, then
  $R_D(\mu_D(d,e),\mu_D(d',e'))$.
  \item If $\mathsf{RatHistoryLedgerPolicy}(\rho,v)$, $C_D(e)$, and
  $R_D(e,e')$, then
  $\mathsf{RatHistoryLedgerPolicy}(\mu_D(\rho,e),\mu_D(v,e'))$.
  \item No carried history $u:\mathsf{BHist}$ can satisfy either one-sided
  classifier-unit law

  $$
  \forall x:\mathsf{BHist}.\;
  C_D(x)\to
  R_D(\mu_D(x,u),x)
  $$

  or

  $$
  \forall x:\mathsf{BHist}.\;
  C_D(x)\to
  R_D(\mu_D(u,x),x).
  $$
\end{enumerate}
\end{theorem}
\begin{proof}
Unfold $C_D$, $R_D$, and $\mu_D$. The operation is ordinary history append,
so the first five clauses are the concrete append package proved for
$\mathsf{RatHistoryCarrier}$, $\mathsf{RatHistoryClassifier}$, and
$\mathsf{RatHistoryLedgerPolicy}$.

Carrier closure uses the carried denominator closure of append. The
commutative classifier clause uses the unary append commutativity theorem
inside the two carried histories.

Associativity compares the two append bracketings by append associativity and
then repacks the result into $R_D$. Binary classifier congruence and ledger
closure are the corresponding Rat denominator append-closure clauses for the
classifier and ledger predicates.

For a one-sided carried unit, pass from the displayed append law to its
canonical continuation instance using $\Cont(x,u,\mu_D(x,u))$ or
$\Cont(u,x,\mu_D(u,x))$. Then
\autoref{thm:field-rat-denominator-continuation-unit-endpoint-empty} forces
$u\hsame\emp$, while \autoref{thm:rat-history-carrier-not-empty} rules out a
carried empty denominator.
\end{proof}

\begin{theorem}[Rat denominator continuation concrete instance laws]
\label{thm:field-rat-denominator-concrete-continuation-instance-laws}
For histories
$d,d',e,e',f,\rho,v,de,ed,ef,left,right,dep,rhoe,vep:\mathsf{BHist}$, the
relational continuation data of
\autoref{def:field-rat-denominator-concrete-continuation-instance} satisfy the
following laws.
\begin{enumerate}
  \item If $C_D(d)$, $C_D(e)$, and $\Cont(d,e,de)$, then $C_D(de)$.
  \item If $C_D(d)$, $C_D(e)$, $\Cont(d,e,de)$, and $\Cont(e,d,ed)$, then
  $R_D(de,ed)$.
  \item If $C_D(d)$, $C_D(e)$, $C_D(f)$, $\Cont(d,e,de)$,
  $\Cont(e,f,ef)$, $\Cont(de,f,left)$, and $\Cont(d,ef,right)$, then
  $R_D(left,right)$.
  \item If $R_D(d,d')$, $R_D(e,e')$, $\Cont(d,e,de)$, and
  $\Cont(d',e',dep)$, then $R_D(de,dep)$.
  \item If $\mathsf{RatHistoryLedgerPolicy}(\rho,v)$, $C_D(e)$,
  $R_D(e,e')$, $\Cont(\rho,e,rhoe)$, and $\Cont(v,e',vep)$, then
  $\mathsf{RatHistoryLedgerPolicy}(rhoe,vep)$.
  \item No carried history $u:\mathsf{BHist}$ can satisfy either one-sided
  classifier-unit law

  $$
  \forall x,r:\mathsf{BHist}.\;
  C_D(x)\to
  \Cont(x,u,r)\to
  R_D(r,x)
  $$

  or

  $$
  \forall x,r:\mathsf{BHist}.\;
  C_D(x)\to
  \Cont(u,x,r)\to
  R_D(r,x).
  $$
\end{enumerate}
\end{theorem}
\begin{proof}
Unfold $C_D$ and $R_D$. The relational presentation leaves each output
history named by its $\Cont$ witness instead of replacing it by append at the
statement level.

Carrier closure is
\autoref{thm:field-rat-denominator-continuation-carrier-closure}. The
commutative classifier clause is
\autoref{thm:field-rat-denominator-continuation-commutative-classifier}
applied to the two displayed continuations.

Associativity uses
\autoref{thm:field-rat-denominator-continuation-semigroup-laws} on the four
continuation witnesses. The theorem supplies the carried intermediate results
and the history-sameness comparison needed to read the two bracketings as
classifier-equivalent.

For binary classifier congruence, unpack $R_D(d,d')$ and $R_D(e,e')$. The
right input classifier supplies the matching denominator comparison required
by \autoref{thm:field-rat-denominator-continuation-classifier-closure}, and
the two displayed continuations give $R_D(de,dep)$.

The ledger clause applies
\autoref{thm:field-rat-denominator-continuation-ledger-closure} to
$\mathsf{RatHistoryLedgerPolicy}(\rho,v)$, the carried and classified right
tail, and the two displayed continuations.

Finally,
\autoref{thm:field-rat-denominator-continuation-unit-endpoint-empty} says that
any one-sided classifier-unit endpoint is $\hsame$ to $\emp$, and its carried
endpoint clause rules out such a history inside $\mathsf{RatHistoryCarrier}$.
\end{proof}

\begin{theorem}[Rat denominator append-continuation law equivalence]
\label{thm:field-rat-denominator-functional-relational-instance-equivalence}
For the concrete data of
\autoref{def:field-rat-denominator-concrete-continuation-instance}, the append
law package of
\autoref{thm:field-rat-denominator-append-concrete-instance-laws} and the
relational $\Cont$ law package of
\autoref{thm:field-rat-denominator-concrete-continuation-instance-laws} are the
same concrete law package in the following sense.
\begin{enumerate}
  \item For all histories $h,k,r:\mathsf{BHist}$, $\Cont(h,k,r)$ identifies
  $r$ with $\mu_D(h,k)$:

  $$
  \Cont(h,k,r)\quad\text{if and only if}\quad r=\mu_D(h,k).
  $$

  In particular, the canonical output satisfies $\Cont(h,k,\mu_D(h,k))$.
  \item Transporting each displayed relational output along this identification
  converts the six append clauses into the six relational clauses.
  \item Instantiating each relational output by the canonical witness
  $\Cont(h,k,\mu_D(h,k))$ converts the six relational clauses back into the six
  append clauses.
\end{enumerate}
\end{theorem}
\begin{proof}
Unfold the concrete operation in
\autoref{def:field-rat-denominator-concrete-continuation-instance}: by
$\mathsf{BEDC.FKernel.Cont.cont\_iff\_append}$, a continuation witness
$\Cont(h,k,r)$ is exactly the assertion $r=\mathsf{append}(h,k)$, and
$\mu_D(h,k)$ is defined to be $\mathsf{append}(h,k)$. This gives the displayed
identification and the canonical witness at $r=\mu_D(h,k)$.

To pass from append laws to relational laws, use the checked append package
$\mathsf{BEDC.Derived.FieldUp.field\_rat\_denominator\_append\_concrete\_instance\_laws}$
and replace every displayed output by the corresponding $\Cont$ output through
the equality above. Carrier closure, commutative classifier compatibility,
associativity, binary classifier congruence, ledger closure, and the one-sided
unit obstruction are therefore the six relational clauses of
\autoref{thm:field-rat-denominator-concrete-continuation-instance-laws}.

Conversely, start from the checked relational package
$\mathsf{BEDC.Derived.FieldUp.field\_rat\_denominator\_continuation\_concrete\_instance\_laws}$.
Instantiate each continuation output by the canonical witness
$\Cont(h,k,\mu_D(h,k))$. The same identification rewrites the six relational
outputs as append outputs, giving exactly the six clauses of
\autoref{thm:field-rat-denominator-append-concrete-instance-laws}. Thus the
functional append presentation and the relational continuation presentation are
one concrete $\FieldUp(\mathsf{RatUp})$ denominator law package.
\end{proof}
