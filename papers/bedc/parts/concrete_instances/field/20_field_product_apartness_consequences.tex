\begin{theorem}[Field product nonzero excludes zero factors]
\label{thm:field-product-nonzero-excludes-zero-factors}
For a BEDC history-level $\FieldUp$ candidate with ring zero absorption,
multiplication congruence, classifier transport of $\mathsf{NonZero}_F$,
and contradiction of $\mathsf{NonZero}_F(0)$, every $a,b:\mathsf{BHist}$
with
$$
p_{ab}:\mathsf{NonZero}_F(\mathsf{mul}(a,b))
$$
satisfies both $a\hsame0\to\bot$ and $b\hsame0\to\bot$.
\end{theorem}
\begin{proof}
Assume $a\hsame0$. Multiplication congruence sends
$\mathsf{mul}(a,b)$ to $\mathsf{mul}(0,b)$, and
Theorem~\ref{thm:ring-mul-zero-absorption} identifies
$\mathsf{mul}(0,b)$ with $0$. Transitivity gives
$\mathsf{mul}(a,b)\hsame0$.

Transport $p_{ab}:\mathsf{NonZero}_F(\mathsf{mul}(a,b))$ along this
classifier equality. The transport clause supplies
$\mathsf{NonZero}_F(0)$, and the field zero-contradiction clause closes
the left-factor obstruction.

The right-factor obstruction is symmetric. From $b\hsame0$, multiplication
congruence sends $\mathsf{mul}(a,b)$ to $\mathsf{mul}(a,0)$, and
Theorem~\ref{thm:ring-mul-zero-absorption} identifies that product with
$0$. Transporting the same witness $p_{ab}$ to $\mathsf{NonZero}_F(0)$
again contradicts the field zero clause.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.field\_product\_nonzero\_excludes\_zero\_factors}

\begin{corollary}[Field product nonzero empty-factor contradiction]
\label{thm:field-product-nonzero-empty-factors-absurd}
Under the same $\FieldUp$ and $\RingUp$ hypotheses as
\autoref{thm:field-product-nonzero-excludes-zero-factors}, a proof
$p_{ab}:\mathsf{NonZero}_F(\mathsf{mul}(a,b))$ determines the pair
$\bigl(a\hsame0\to\bot\bigr)\land\bigl(b\hsame0\to\bot\bigr)$.
\end{corollary}
\begin{proof}
Apply \autoref{thm:field-product-nonzero-excludes-zero-factors} to
$p_{ab}$. Its two components are exactly the left-factor and right-factor
contradictions against zero.
\end{proof}
\leanchecked{BEDC.Derived.FieldUp.field\_product\_nonzero\_empty\_factors\_absurd}

\begin{theorem}[Field product nonzero of nonzero factors]
\label{thm:field-product-nonzero-of-nonzero-factors}
For a BEDC history-level $\FieldUp$ candidate with the hypotheses of
\autoref{thm:field-nonzero-factors-exclude-zero-product}, assume also a
constructive bridge
$$
\alpha:\bigl(h\hsame0\to\bot\bigr)\to\mathsf{NonZero}_F(h)
$$
from apartness from zero into the nonzero classifier. Then witnesses
$p_a:\mathsf{NonZero}_F(a)$ and $p_b:\mathsf{NonZero}_F(b)$ determine
$$
\mathsf{NonZero}_F(\mathsf{mul}(a,b)).
$$
\end{theorem}
\begin{proof}
Theorem~\ref{thm:field-nonzero-factors-exclude-zero-product} turns
$p_a$ and $p_b$ into a constructive refutation of
$\mathsf{mul}(a,b)\hsame0$. Applying the bridge $\alpha$ to this product
apartness gives the displayed product nonzero witness.
\end{proof}

$$
(a\hsame0\to\bot)\wedge(x\hsame0\to\bot)\wedge(b\hsame0\to\bot).

$\mathsf{mul}(0,x)$, and \autoref{thm:ring-mul-zero-absorption} classifies
$\mathsf{mul}(0,x)$ as $0$. By transitivity, $\mathsf{mul}(a,x)\hsame0$,

\begin{theorem}[Field two-sided product nonzero iff middle factor nonzero]
\label{thm:field-two-sided-product-nonzero-middle-factor-iff}
For a BEDC history-level $\FieldUp$ candidate over $\mathsf{BHist}$ with
the hypotheses of
\autoref{thm:field-product-nonzero-of-nonzero-factors} and
\autoref{thm:field-two-sided-product-nonzero-excludes-middle-zero}, assume
certified outer witnesses $p_a:\mathsf{NonZero}_F(a)$ and
$p_b:\mathsf{NonZero}_F(b)$. Then every $x:\mathsf{BHist}$ satisfies
$$
\mathsf{NonZero}_F(\mathsf{mul}(\mathsf{mul}(a,x),b))
\Longleftrightarrow
\mathsf{NonZero}_F(x).
$$
\end{theorem}
\begin{proof}
For the forward direction, assume
$p_{axb}:\mathsf{NonZero}_F(\mathsf{mul}(\mathsf{mul}(a,x),b))$.
Theorem~\ref{thm:field-two-sided-product-nonzero-excludes-middle-zero}
turns this witness into a refutation of $x\hsame0$. The apartness-to-nonzero
bridge required by
\autoref{thm:field-product-nonzero-of-nonzero-factors} then gives
$\mathsf{NonZero}_F(x)$.

For the reverse direction, assume $p_x:\mathsf{NonZero}_F(x)$. Apply
\autoref{thm:field-product-nonzero-of-nonzero-factors} to $p_a$ and $p_x$
to obtain $p_{ax}:\mathsf{NonZero}_F(\mathsf{mul}(a,x))$.

Apply the same product-closure theorem to $p_{ax}$ and $p_b$. The resulting
witness is exactly
$\mathsf{NonZero}_F(\mathsf{mul}(\mathsf{mul}(a,x),b))$, completing the
two-sided equivalence.
\end{proof}

Per-object certificate: Traditional target: exact middle-factor nonzero
transport through certified two-sided products. BEDC source: $\FieldUp$ plus
$\RingUp$ zero absorption. Carrier choice: $\mathsf{BHist}$. Classifier
choice: $\hsame$. Stability obligations: nonzero transport, product closure,
product zero-factor exclusion, and ring zero absorption. Ledger: outer
certificates $p_a,p_b$, middle certificate $p_x$, intermediate product
witness $p_{ax}$, and two-sided product witness $p_{axb}$. Core theorem list:
Theorem~\ref{thm:field-two-sided-product-nonzero-middle-factor-iff}.
Exactness predicate: nonzero of the middle factor is equivalent to nonzero of
the certified outer product. Downstream use: FieldUp exact nonzero propagation
for affine and denominator-product reasoning. Status: ProofTarget.
