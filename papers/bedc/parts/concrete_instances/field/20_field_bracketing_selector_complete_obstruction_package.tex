\begin{theorem}[RatUp FieldUp bracketing selector complete obstruction package]
\label{thm:ratup-fieldup-bracketing-selector-complete-obstruction-package}
For the concrete denominator tuple $(C_D^u,R_D^u,\mu_D^u,e_D^u,N_D^u)$ and
histories $h,k,\ell:\mathsf{BHist}$ with $C_D^u(h)$, $C_D^u(k)$, and
$C_D^u(\ell)$, let $E_s(h,k,\ell)$ be the endpoint selector of
\autoref{def:ratup-fieldup-bracketing-endpoint-selector}. If
$\mathsf{ThreeSupp}_D^u(h,k,\ell)$ holds, then for all selectors
$s,t:\mathsf{BMark}$ and contexts $L,R:\mathsf{BHist}$, setting
$a:=E_s(h,k,\ell)$ and $b:=E_t(h,k,\ell)$ gives one concrete selector
obstruction package with the following rows:
\begin{enumerate}
  \item $a\hsame b$;
  \item $R_D^u(a,b)$;
  \item $\mathsf{RatHistoryClassifier}(a,b)$;
  \item
  $$
  \mathsf{fieldSingletonEmptyClassifier}
  (\mathsf{append}(L,a),\mathsf{append}(R,b))\to\bot;
  $$
  \item
  $$
  \mathsf{fieldSingletonEmptyCarrier}(\mathsf{append}(L,a))\to\bot;
  $$
  \item
  $$
  \mathsf{fieldSingletonEmptyNonZero}(\mathsf{append}(L,a));
  $$
  \item
  $$
  \mathsf{fieldSingletonEmptyCarrier}(\mathsf{append}(a,R))\to\bot;
  $$
  \item
  $$
  \mathsf{fieldSingletonEmptyNonZero}(\mathsf{append}(a,R));
  $$
  \item
  $$
  \mathsf{fieldSingletonEmptyCarrier}(\mathsf{append}(L,b))\to\bot;
  $$
  \item
  $$
  \mathsf{fieldSingletonEmptyNonZero}(\mathsf{append}(L,b));
  $$
  \item
  $$
  \mathsf{fieldSingletonEmptyCarrier}(\mathsf{append}(b,R))\to\bot;
  $$
  \item
  $$
  \mathsf{fieldSingletonEmptyNonZero}(\mathsf{append}(b,R)).
  $$
\end{enumerate}
\end{theorem}
\begin{proof}
First apply
\autoref{thm:ratup-fieldup-bracketing-selector-independence-package} to the
same denominator tuple, carrier witnesses, support witness, selector pair, and
the context $L:\mathsf{BHist}$. This supplies $a\hsame b$ and
$\mathsf{RatHistoryClassifier}(a,b)$ as the common selector-independent rows.

Next apply \autoref{thm:ratup-fieldup-bracketing-selector-exit-subpackage} to
the selector pair $s,t:\mathsf{BMark}$ and the contexts $L,R:\mathsf{BHist}$.
Its exit row gives $R_D^u(a,b)$ and the singleton-classifier exclusion for
$(\mathsf{append}(L,a),\mathsf{append}(R,b))$.

We now build the contextual nonzero rows for the left endpoint. Apply
\autoref{thm:ratup-fieldup-selector-contextual-nonzero-package} to the selector
$s$ with context $L$ to obtain the carrier exclusion and
$\mathsf{fieldSingletonEmptyNonZero}$ witness for $\mathsf{append}(L,a)$.
Apply the same theorem to $s$ with context $R$ to obtain the corresponding
right-context rows for $\mathsf{append}(a,R)$.

Finally apply
\autoref{thm:ratup-fieldup-selector-contextual-nonzero-package} to the selector
$t$ with context $L$, and then with context $R$. These two applications give the
carrier exclusions and $\mathsf{fieldSingletonEmptyNonZero}$ witnesses for
$\mathsf{append}(L,b)$ and $\mathsf{append}(b,R)$. Replacing $a$ and $b$ by the
displayed selector endpoints assembles the concrete obstruction package.
\end{proof}
