\begin{theorem}[Field multiplication cancellation from apartness-certified factors]
\label{thm:field-nonzero-mul-cancel-from-apartness}

right and left multiplication by apartness-certified nonzero factors are
cancellative. Any witness $p_b:\mathsf{NonZero}_F(b)$ makes
$$
xb\hsame yb
\quad\Longrightarrow\quad
x\hsame y,
$$
and any witness $p_a:\mathsf{NonZero}_F(a)$ makes
$$
ax\hsame ay
\quad\Longrightarrow\quad
x\hsame y.
$$
\end{theorem}
\begin{proof}
For the right clause, apply
Theorem~\ref{thm:field-right-mul-equation-solution-from-apartness} to
$xb\hsame yb$. This gives
$x\hsame (yb)\mathsf{inv}(b,p_b)$. Theorem~\ref{thm:field-mul-inverse-right-cancel-from-apartness}
identifies $(yb)\mathsf{inv}(b,p_b)$ with $y$, so transitivity of
$\hsame$ gives $x\hsame y$.

For the left clause, apply
Theorem~\ref{thm:field-left-mul-equation-solution-from-apartness} to
$ax\hsame ay$. This gives
$x\hsame \mathsf{inv}(a,p_a)(ay)$. Theorem~\ref{thm:field-mul-inverse-left-cancel-from-apartness}
identifies $\mathsf{inv}(a,p_a)(ay)$ with $y$, and transitivity of
$\hsame$ finishes.
\end{proof}

\begin{theorem}[Field equal-squares signed factor exclusion]
\label{thm:field-equal-squares-signed-factor-exclusion}

constructive clauses hold.
\begin{enumerate}
\item If $p_+:\mathsf{NonZero}_F(a+b)$, then $a+(-b)\hsame0$.
\item If $p_-:\mathsf{NonZero}_F(a+(-b))$, then $a+b\hsame0$.
\end{enumerate}
\end{theorem}
\begin{proof}

Theorem~\ref{thm:commring-difference-of-squares-zero-exact} to the square
equality. The signed factor product is therefore classified as zero:
$$

$$

For the first clause, use the witness $p_+:\mathsf{NonZero}_F(a+b)$ and
apply Theorem~\ref{thm:field-one-sided-zero-product-cancel-from-apartness} to
the signed factor product with left factor $a+b$ and right factor $a+(-b)$.
The left-factor clause gives
$a+(-b)\hsame0$.

For the second clause, use the witness $p_-:\mathsf{NonZero}_F(a+(-b))$ and
apply Theorem~\ref{thm:field-one-sided-zero-product-cancel-from-apartness} to
the same signed factor product. The right-factor clause gives $a+b\hsame0$.
\end{proof}
