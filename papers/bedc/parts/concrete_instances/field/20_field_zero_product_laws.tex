\begin{theorem}[Field multiplication cancellation from apartness-certified factors]
\label{thm:field-nonzero-mul-cancel-from-apartness}
For a BEDC history-level $\FieldUp$ candidate with proof-indexed inverses,
right and left multiplication by apartness-certified nonzero factors are
cancellative. Any witness $p_b:\mathsf{NonZero}_F(b)$ makes
$$
xb\hsame yb
\quad\Longrightarrow\quad
x\hsame y,
$$
and any witness $p_a:\mathsf{NonZero}_F(a)$ makes
$$
ax\hsame ay
\quad\Longrightarrow\quad
x\hsame y.
$$
\end{theorem}
\begin{proof}
For the right clause, apply
Theorem~\ref{thm:field-right-mul-equation-solution-from-apartness} to
$xb\hsame yb$. This gives
$x\hsame (yb)\mathsf{inv}(b,p_b)$. Theorem~\ref{thm:field-mul-inverse-right-cancel-from-apartness}
identifies $(yb)\mathsf{inv}(b,p_b)$ with $y$, so transitivity of
$\hsame$ gives $x\hsame y$.

For the left clause, apply
Theorem~\ref{thm:field-left-mul-equation-solution-from-apartness} to
$ax\hsame ay$. This gives
$x\hsame \mathsf{inv}(a,p_a)(ay)$. Theorem~\ref{thm:field-mul-inverse-left-cancel-from-apartness}
identifies $\mathsf{inv}(a,p_a)(ay)$ with $y$, and transitivity of
$\hsame$ finishes.
\end{proof}

\begin{theorem}[Field two-sided multiplication exactness from apartness]
\label{thm:field-two-sided-mul-exact-from-apartness}
For a BEDC history-level $\FieldUp$ candidate with proof-indexed inverses,
multiplication congruence, and witnesses $p_a:\mathsf{NonZero}_F(a)$ and
$p_b:\mathsf{NonZero}_F(b)$, every $x,y:\mathsf{BHist}$ satisfy
$$
(a x)b\hsame(a y)b
\Longleftrightarrow
x\hsame y.
$$
\end{theorem}
\begin{proof}
For the forward direction, apply
Theorem~\ref{thm:field-nonzero-mul-cancel-from-apartness} with the
right-factor witness $p_b$. This cancels the certified factor $b$ and gives
$a x\hsame a y$.

Apply Theorem~\ref{thm:field-nonzero-mul-cancel-from-apartness} again with
the left-factor witness $p_a$. This cancels the certified factor $a$ and
gives $x\hsame y$.

For the reverse direction, start from $x\hsame y$. Multiplication congruence
transports this classifier equality through left multiplication by $a$, giving
$a x\hsame a y$.

Apply multiplication congruence once more through right multiplication by $b$.
The result is exactly
$$
(a x)b\hsame(a y)b.
$$
\end{proof}

\begin{theorem}[Field two-sided zero product exactness from apartness]
\label{thm:field-two-sided-zero-product-cancel-from-apartness}
For a BEDC history-level $\FieldUp$ candidate with proof-indexed inverses,
ring zero absorption, multiplication associativity, left and right
multiplicative identity laws, multiplication congruence, indexed left and
right inverse laws, and witnesses $p_a:\mathsf{NonZero}_F(a)$ and
$p_b:\mathsf{NonZero}_F(b)$, every $x:\mathsf{BHist}$ satisfies
$$
(a x)b\hsame0
\Longleftrightarrow
x\hsame0.
$$
\end{theorem}
\begin{proof}
For the forward direction, apply
Theorem~\ref{thm:field-right-mul-equation-solution-from-apartness} with the
right-factor witness $p_b$ to the hypothesis $(a x)b\hsame0$. This solves the
right multiplication equation and gives
$a x\hsame0\mathsf{inv}(b,p_b)$.

Use Theorem~\ref{thm:ring-mul-zero-absorption} to identify
$0\mathsf{inv}(b,p_b)$ with $0$. Transitivity of $\hsame$ therefore gives
$a x\hsame0$.

Now apply the checked left inverse cancellation chain with the left-factor
witness $p_a$ to $a x\hsame0$. The left solver gives
$x\hsame\mathsf{inv}(a,p_a)0$.

Use Theorem~\ref{thm:ring-mul-zero-absorption} again to identify
$\mathsf{inv}(a,p_a)0$ with $0$. Transitivity of $\hsame$ yields
$x\hsame0$.

For the reverse direction, start from $x\hsame0$. Multiplication congruence
transports this equality through left multiplication by $a$, giving
$a x\hsame a0$.

Apply Theorem~\ref{thm:ring-mul-zero-absorption} to collapse $a0\hsame0$.
Transitivity gives $a x\hsame0$.

Multiplication congruence now transports through right multiplication by $b$,
giving $(a x)b\hsame0b$. A second application of
Theorem~\ref{thm:ring-mul-zero-absorption} collapses $0b\hsame0$, and
transitivity yields $(a x)b\hsame0$.
\end{proof}

\label{thm:field-nonzero-factors-exclude-zero-product}

$\mathsf{NonZero}_F(0)$.

The field stability certificate includes contradiction of
$\mathsf{NonZero}_F(0)$, so the transported evidence closes the goal with
$\bot$.

The symmetric route uses $p_b$ in the right-factor clause and transports
$p_a$ after deriving $a\hsame0$.
\end{proof}

\begin{theorem}[Field apart-zero append factor closure]
\label{thm:field-apartzero-append-factor-closed}
For histories $p,q:\mathsf{BHist}$, if either $p$ or $q$ is apart from zero,
then the appended history $\mathsf{append}(p,q)$ is apart from zero:
$$
\mathsf{ApartZero}_F(p)\lor\mathsf{ApartZero}_F(q)
\Longrightarrow
\mathsf{ApartZero}_F(\mathsf{append}(p,q)).
$$
\end{theorem}
\begin{proof}
Assume $\mathsf{append}(p,q)\hsame0$. Empty-append inversion gives both
$p\hsame0$ and $q\hsame0$. If the apart-zero witness is on the left factor,
it rejects the first equality; if it is on the right factor, it rejects the
second equality. Thus either factor witness prevents the appended history
from classifying as zero.
\end{proof}

\begin{theorem}[Field equal-squares signed factor exclusion]
\label{thm:field-equal-squares-signed-factor-exclusion}
For a BEDC history-level $\FieldUp$ candidate with proof-indexed inverses,
ring zero absorption, multiplication associativity, left and right
multiplicative identity laws, multiplication congruence, indexed inverse laws,
the $\CommRingUp$ hypotheses required by
Theorem~\ref{thm:commring-difference-of-squares-zero-exact}, and
every $a,b:\mathsf{BHist}$ with $a\cdot a\hsame b\cdot b$, the following
constructive clauses hold.
\begin{enumerate}
\item If $p_+:\mathsf{NonZero}_F(a+b)$, then $a+(-b)\hsame0$.
\item If $p_-:\mathsf{NonZero}_F(a+(-b))$, then $a+b\hsame0$.
\end{enumerate}
\end{theorem}
\begin{proof}
Apply the reverse direction of
Theorem~\ref{thm:commring-difference-of-squares-zero-exact} to the square
equality. The signed factor product is therefore classified as zero:
$$
(a+b)\cdot(a+(-b))\hsame0.
$$

For the first clause, use the witness $p_+:\mathsf{NonZero}_F(a+b)$ and
apply Theorem~\ref{thm:field-one-sided-zero-product-cancel-from-apartness} to
the signed factor product with left factor $a+b$ and right factor $a+(-b)$.
The left-factor clause gives
$a+(-b)\hsame0$.

For the second clause, use the witness $p_-:\mathsf{NonZero}_F(a+(-b))$ and
apply Theorem~\ref{thm:field-one-sided-zero-product-cancel-from-apartness} to
the same signed factor product. The right-factor clause gives $a+b\hsame0$.
\end{proof}
