\chapter{A Concrete Naming Certificate for $\TensorProductUp$}
\label{ch:concrete-instances-tensorproduct-namecert}

The tensor-product interface $\TensorProductUp$ packages the universal bilinear factorization of two module carriers. It depends on $\NameCert_{\ModuleUp}$ (\autoref{ch:concrete-instances-module-namecert}) and supplies the multilinear data on which higher tensor algebras and exterior powers are built.

\section{Carrier definition}
\label{sec:tensorproduct-carrier}

\begin{definition}[Tensor-product singleton carrier]\label{def:tensorproduct-singleton-carrier}
For a generated history $h$, the singleton tensor-product carrier holds when there are
histories $l$ and $r$ such that $l$ and $r$ are both module-singleton histories and
$\Cont(l,r,h)$ holds. Thus the tensor-product singleton object is represented by a
concrete continuation ledger from its two singleton module endpoints.
\end{definition}
\leandef{BEDC.Derived.TensorProductUp.TensorProductSingletonCarrier}

\begin{theorem}[Tensor-product singleton carrier empty iff]
\label{thm:tensorproduct-singleton-carrier-empty-iff}
For a generated history $h$, the tensor-product singleton carrier holds exactly
when $h$ is the empty history:
$$
\mathsf{TensorProductSingletonCarrier}(h)\Longleftrightarrow \hsame(h,\emp).
$$
\end{theorem}
\begin{proof}
For the forward direction, unpack the tensor carrier into singleton module
endpoints and a continuation ledger. The endpoint carriers identify both
factors with $\emp$, so continuation respect compares the ledger output with
the empty right-unit continuation. Conversely, if $h\hsame\emp$, use two
empty singleton module endpoints and transport the empty right-unit continuation
to the result $h$.
\end{proof}
\leanchecked{BEDC.Derived.TensorProductUp.TensorProductSingletonCarrier\_empty\_iff}

\begin{theorem}[Tensor-product singleton carrier empty endpoint iff]
\label{thm:tensorproduct-singleton-carrier-empty-endpoint-iff}
For a generated history $h$, the tensor-product singleton carrier holds exactly
when $h$ is $\hsame$ to the empty history.
\end{theorem}
\begin{proof}
Unpack the carrier into two module-singleton endpoints and a continuation
ledger. The two module endpoints are empty, so the continuation endpoint is the
empty append. Conversely, the empty endpoint is obtained from the two empty
module factors and the left-unit continuation law.
\end{proof}
\leanchecked{BEDC.Derived.TensorProductUp.TensorProductSingletonCarrier\_empty\_endpoint\_iff}

\begin{theorem}[Tensor-product singleton result-empty continuation]
\label{thm:tensorproduct-singleton-carrier-result-empty-continuation}
For a history $t:\mathsf{BHist}$, if $t$ satisfies the tensor-product
singleton carrier, then $t$ continues with the empty suffix to the empty
history:
$$
\mathsf{TensorProductSingletonCarrier}(t)\Rightarrow
\mathsf{Cont}(t,\emp,\emp).
$$
\end{theorem}
\begin{proof}
Unpack the carrier into module-singleton endpoints $l,r$ and a continuation
$\mathsf{Cont}(l,r,t)$. The two endpoint carrier witnesses identify both
$l$ and $r$ with $\emp$, and the displayed continuation therefore reduces to
the empty result. The right-unit continuation gives
$\mathsf{Cont}(t,\emp,\emp)$.
\end{proof}
\leanchecked{BEDC.Derived.TensorProductUp.TensorProductSingletonCarrier\_result\_empty\_continuation}

\begin{theorem}[Tensor-product singleton suffix carrier]\label{thm:tensorproduct-singleton-carrier-continuation-suffix-carrier}
Let $p$, $s$, and $o$ be generated histories. If $p$ and $o$ satisfy the
tensor-product singleton carrier and $\Cont(p,s,o)$ holds, then $s$ satisfies the
module-singleton carrier.
\end{theorem}
\begin{proof}
Unpack the two tensor-product singleton carriers into module-singleton endpoints
and continuation ledgers. The endpoint carriers identify both tensor endpoints with
the empty history. The displayed continuation $\Cont(p,s,o)$ therefore has empty
source and empty result, so the append-empty inversion for continuation identifies
the suffix $s$ with the empty history.
\end{proof}
\leanchecked{BEDC.Derived.TensorProductUp.TensorProductSingletonCarrier\_continuation\_suffix\_carrier}

\begin{theorem}[Tensor-product singleton carrier continuation result closure]
\label{thm:tensorproduct-singleton-carrier-continuation-result-closed}
Let $p$, $s$, and $o$ be generated histories. If $p$ satisfies the
tensor-product singleton carrier, $s$ satisfies the module-singleton carrier,
and $\Cont(p,s,o)$ holds, then $o$ satisfies the tensor-product singleton carrier.
\end{theorem}
\begin{proof}
The tensor-product singleton carrier identifies $p$ with the empty history,
and the module-singleton carrier identifies $s$ with the empty history.
Transporting the continuation along these endpoint identifications gives
$\Cont(\emp,\emp,o)$. The left-unit result law identifies $o$ with $\emp$,
and the tensor-product singleton empty characterization rebuilds the carrier.
\end{proof}
\leanchecked{BEDC.Derived.TensorProductUp.TensorProductSingletonCarrier\_continuation\_result\_closed}

\begin{theorem}[Tensor-product singleton continuation prefix carrier]
\label{thm:tensorproduct-singleton-carrier-continuation-prefix-carrier}
Let $pre$, $pair$, and $out$ be generated histories. If $pair$ and $out$
satisfy the tensor-product singleton carrier and $\Cont(pre,pair,out)$ holds,
then $pre$ satisfies the module-singleton carrier.
\end{theorem}
\begin{proof}
The tensor-product singleton carrier identifies both $pair$ and $out$ with
the empty history. After transporting the displayed continuation to an empty
result, continuation empty-result inversion exposes the prefix endpoint as
empty, which is exactly the module-singleton carrier.
\end{proof}
\leanchecked{BEDC.Derived.TensorProductUp.TensorProductSingletonCarrier\_continuation\_prefix\_carrier}

\begin{definition}[Tensor product singleton factor]
\label{def:tensorproduct-singleton-factor}
For histories $l,r,t:\mathsf{BHist}$, the tensor-product singleton factor
predicate holds when $l$, $r$, and $t$ all lie in the module singleton carrier
and the continuation relation composes $l$ with $r$ to produce $t$:
$$
\mathsf{TensorProductSingletonFactor}(l,r,t)
:=
A_{\emp}(l)\land A_{\emp}(r)\land A_{\emp}(t)\land
\mathsf{Cont}(l,r,t).
$$
\end{definition}
\leandef{BEDC.Derived.TensorProductUp.TensorProductSingletonFactor}

\begin{theorem}[Tensor product singleton carrier factor witness]
\label{thm:tensorproduct-singleton-carrier-factor-witness}
If a tensor history satisfies the tensor-product singleton carrier, then it has
left and right singleton module factors whose tensor-product singleton factor
predicate holds, and the original continuation witness is retained.
\end{theorem}
\begin{proof}
Eliminate the carrier witness. The two module-singleton factors force the left
and right endpoints to be empty, and the continuation ledger then forces the
tensor endpoint to be empty. These three singleton carrier facts together with
the continuation ledger form the concrete factor witness.
\end{proof}
\leanchecked{BEDC.Derived.TensorProductUp.TensorProductSingletonCarrier\_factor\_witness}

\begin{theorem}[Tensor product singleton factor hsame transport]
\label{thm:tensorproduct-singleton-factor-hsame-transport}
If $\mathsf{TensorProductSingletonFactor}(l,r,t)$ holds and
$l\hsame l'$, $r\hsame r'$, and $t\hsame t'$, then
$\mathsf{TensorProductSingletonFactor}(l',r',t')$ holds, and the transported
continuation witness $\mathsf{Cont}(l',r',t')$ is retained.
\end{theorem}
\begin{proof}
Unpack the singleton factor. Transport each singleton carrier endpoint by
symmetry and transitivity of $\hsame$. Transport the continuation witness
across the three endpoint comparisons by continuation $\hsame$ transport.
\end{proof}
\leanchecked{BEDC.Derived.TensorProductUp.TensorProductSingletonFactor\_hsame\_transport}

\section{The certificate}
\label{sec:tensorproduct-certificate}

\begin{theorem}[Tensor product singleton factor tensor semantic name certificate]
\label{thm:tensorproduct-singleton-factor-tensor-semantic-namecert}
Given a concrete singleton factor
$\mathsf{TensorProductSingletonFactor}(l,r,t)$, the tensor endpoint carrier
$u\mapsto\mathsf{TensorProductSingletonFactor}(l,r,u)$ forms a semantic name
certificate when source, pattern, and ledger policy are all this carrier and
the classifier is $\hsame$.
\end{theorem}
\begin{proof}
The displayed factor supplies the inhabited carrier. Reflexivity, symmetry,
and transitivity are those of $\hsame$. Carrier transport is exactly
\autoref{thm:tensorproduct-singleton-factor-hsame-transport} with the left and
right endpoints fixed by reflexivity.
\end{proof}
\leanchecked{BEDC.Derived.TensorProductUp.TensorProductSingletonFactor\_tensor\_semanticNameCert}

\begin{theorem}[Tensor product singleton factor source-target swap]
\label{thm:tensorproduct-singleton-factor-source-target-swap}
For histories $l,r,t:\mathsf{BHist}$, if
$$
\mathsf{TensorProductSingletonFactor}(l,r,t)
$$
holds, then
$$
\mathsf{TensorProductSingletonFactor}(r,l,t)
$$
holds. The resulting factor contains the swapped continuation witness
$\Cont(r,l,t)$.
\end{theorem}
\begin{proof}
Unpack \autoref{def:tensorproduct-singleton-factor}. The factor supplies
$A_{\emp}(l)$, $A_{\emp}(r)$, $A_{\emp}(t)$, and
$\Cont(l,r,t)$. In the concrete singleton module of
\autoref{def:singleton-empty-history-module-instance}, the first two endpoint
witnesses are the comparisons $l\hsame\emp$ and $r\hsame\emp$. Hence
$l\hsame r$ follows by composing $l\hsame\emp$ with the symmetry of
$r\hsame\emp$, and $r\hsame l$ follows by the same argument with the roles
reversed.

Apply \autoref{thm:tensorproduct-singleton-factor-hsame-transport} to the
original factor with the comparisons $l\hsame r$, $r\hsame l$, and
$t\hsame t$. Its conclusion is exactly the swapped predicate
$\mathsf{TensorProductSingletonFactor}(r,l,t)$, and its continuation component
is $\Cont(r,l,t)$.
\end{proof}
\leanchecked{BEDC.Derived.TensorProductUp.TensorProductSingletonFactor\_source\_target\_swap}

\begin{theorem}[Tensor product singleton factor swap symmetry]
\label{thm:tensorproduct-singleton-factor-swap-symmetry}
For histories $l,r,t:\mathsf{BHist}$, if
$$
\mathsf{TensorProductSingletonFactor}(l,r,t)
$$
holds, then
$$
\mathsf{TensorProductSingletonFactor}(r,l,t)
$$
holds. The resulting factor includes the continuation witness $\Cont(r,l,t)$.
\end{theorem}
\begin{proof}
Unpack \autoref{def:tensorproduct-singleton-factor}. The factor consists of
$A_{\emp}(l)$, $A_{\emp}(r)$, $A_{\emp}(t)$, and $\Cont(l,r,t)$. By
\autoref{def:singleton-empty-history-module-instance} and
\autoref{def:list-singleton-history-source}, the two factor carrier witnesses
are endpoint comparisons $l\hsame\emp$ and $r\hsame\emp$. Transitivity of
$\hsame$ gives $l\hsame r$ by composing $l\hsame\emp$ with the symmetry of
$r\hsame\emp$, and similarly gives $r\hsame l$.

Apply \autoref{thm:tensorproduct-singleton-factor-hsame-transport} to the
original factor with endpoint comparisons $l\hsame r$, $r\hsame l$, and the
reflexive comparison $t\hsame t$. The transported conclusion is
$\mathsf{TensorProductSingletonFactor}(r,l,t)$, and the retained continuation
component is $\Cont(r,l,t)$.
\end{proof}

\begin{theorem}[Tensor product singleton factor associator]
\label{thm:tensorproduct-singleton-factor-associator}
For histories $l,r,s,t,u:\mathsf{BHist}$, if
$$
\mathsf{TensorProductSingletonFactor}(l,r,t)
\quad\text{and}\quad
\mathsf{TensorProductSingletonFactor}(t,s,u)
$$
hold, then there is a history $m:\mathsf{BHist}$ such that
$$
\mathsf{TensorProductSingletonFactor}(r,s,m)
\quad\text{and}\quad
\mathsf{TensorProductSingletonFactor}(l,m,u)
$$
hold. The two resulting factors retain the rebracketed continuation ledgers
$\Cont(r,s,m)$ and $\Cont(l,m,u)$.
\end{theorem}
\begin{proof}
Unpack \autoref{def:tensorproduct-singleton-factor}. The first factor supplies
$A_{\emp}(l)$, $A_{\emp}(r)$, $A_{\emp}(t)$, and $\Cont(l,r,t)$; the second
supplies $A_{\emp}(t)$, $A_{\emp}(s)$, $A_{\emp}(u)$, and $\Cont(t,s,u)$.
Apply the associativity existence form of the continuation spine to the two
continuation ledgers $\Cont(l,r,t)$ and $\Cont(t,s,u)$. This gives a history
$m$ with
$$
\Cont(r,s,m)
\quad\text{and}\quad
\Cont(l,m,u).
$$

It remains only to show that $m$ lies in the singleton module carrier. By
\autoref{def:singleton-empty-history-module-instance} and
\autoref{def:list-singleton-history-source}, the carrier witnesses
$A_{\emp}(r)$ and $A_{\emp}(s)$ identify $r$ and $s$ with $\emp$. Transporting
$\Cont(r,s,m)$ along these two endpoint comparisons gives
$\Cont(\emp,\emp,m)$. The continuation unit laws of
\autoref{thm:continuation-unit-laws-spine} compare this ledger with the empty
unit continuation and identify $m$ with $\emp$, which is exactly
$A_{\emp}(m)$.

Now package the first rebracketed factor from $A_{\emp}(r)$, $A_{\emp}(s)$,
$A_{\emp}(m)$, and $\Cont(r,s,m)$. Package the second from $A_{\emp}(l)$,
$A_{\emp}(m)$, $A_{\emp}(u)$, and $\Cont(l,m,u)$.
\end{proof}
\leanchecked{BEDC.Derived.TensorProductUp.TensorProductSingletonFactor\_associator}

\begin{theorem}[Tensor product singleton factor classifier uniqueness]
\label{thm:tensorproduct-singleton-factor-classifier-uniqueness}
For histories $l,r,t,t':\mathsf{BHist}$, if
$$
\mathsf{TensorProductSingletonFactor}(l,r,t)
\quad\text{and}\quad
\mathsf{TensorProductSingletonFactor}(l,r,t')
$$
hold, then
$$
t\hsame t'.
$$
Thus any two tensor endpoints obtained from the same singleton module factor pair are classifier-equal in the tensor endpoint certificate of \autoref{thm:tensorproduct-singleton-factor-tensor-semantic-namecert}.
\end{theorem}
\begin{proof}
Unpack \autoref{def:tensorproduct-singleton-factor} for both factor witnesses. The first witness supplies $\Cont(l,r,t)$; the second supplies $\Cont(l,r,t')$. These two continuation ledgers have the same source history $l$ and the same right history $r$. Applying \autoref{thm:continuation-determinacy-up-to-history-sameness} to the two ledgers identifies their outputs, giving $t\hsame t'$. The classifier in \autoref{thm:tensorproduct-singleton-factor-tensor-semantic-namecert} is $\hsame$ on this endpoint carrier, so this output identification is exactly tensor classifier equality.
\end{proof}
