\chapter{Naming Certificate for RealCompletenessUp}
\label{ch:concrete-instances-real-completeness}

This chapter names the certificate boundary for Bishop-style completeness of $\RealUp$ as read through regular rational sequences. The object is a bridge certificate: it records the finite source rows that feed the diagonal-limit surface, the classifier that identifies equal completion witnesses, and the ledger needed to trace each named limit back to RatUp, StreamNameUp, RegSeqRat, and RealUp data.

\section{Source spec for RealCompletenessUp}
\label{sec:concrete-instances-real-completeness-source-spec}

\begin{definition}[Source rows of RealCompletenessUp]
\label{def:concrete-instances-real-completeness-source-rows}
A source row for $\RealCompletenessUp$ is a finite tuple
$$
(\rho,\sigma,r,\mu,\delta,w,p,\tau,c,u).
$$
Here $\rho$ is the RatUp seed surface, $\sigma$ is the StreamNameUp source, $r$ is the supplied RealUp name, $\mu$ is the regular-Cauchy modulus trace, $\delta$ is the diagonal extraction trace, $w$ is the stationary finite-window bridge ledger, $p$ is the finite partition spine, $\tau$ is the selected-tail trace, $c$ is the completion certificate, and $u$ is the uniqueness certificate. The tuple is only a certificate boundary; the chapter does not identify real numbers with host quotients.
\end{definition}

\section{Pattern spec}
\label{sec:concrete-instances-real-completeness-pattern}

\begin{definition}[Pattern witness for RealCompletenessUp]
\label{def:concrete-instances-real-completeness-pattern}
The pattern predicate accepts exactly those source rows whose histories display the regular-completion route
$$
\rho \leadsto \sigma \leadsto r,\qquad
r \leadsto \mu \leadsto \delta,\qquad
\delta \leadsto w \leadsto \tau,\qquad
\tau \leadsto p \leadsto c,\qquad
c \leadsto \delta \leadsto u.
$$
In paper terms, this says that a regular Cauchy input over RegSeqRat is read through its explicit modulus, the diagonal representative is exposed to the finite RealUp window ledgers, the selected endpoints remain tied to the stationary rational seeds, and the completion and uniqueness rows are recorded as finite continuation data.
\end{definition}

\section{Stability certificate}
\label{sec:concrete-instances-real-completeness-stability}

\begin{theorem}[Stability of RealCompletenessUp]
\label{thm:concrete-instances-real-completeness-stability}
If a source row satisfies the $\RealCompletenessUp$ pattern and another source row is classified with it componentwise, then the second source row satisfies the same pattern. Equivalently, the named completion surface is closed under replacement of every RatUp, StreamNameUp, RegSeqRat, RealUp, modulus, diagonal, window, partition, selected-tail, completion, and uniqueness component by an $\hsame$ component.
\end{theorem}
\begin{proof}
The classifier supplies $\hsame$ for each component of the source row. Since $\hsame$ is the finite history equality relation in the kernel, each continuation row in the pattern transports by cases on those componentwise equalities. The resulting continuation chain is the same finite chain, now read on the classified row.
\end{proof}
