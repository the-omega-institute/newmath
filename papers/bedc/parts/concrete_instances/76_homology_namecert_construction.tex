\chapter{A Concrete Naming Certificate for $\HomologyUp$}
\label{ch:concrete-instances-homology-namecert}

The homology interface $\HomologyUp$ packages a chain complex over an abelian-group carrier together with the kernel/image quotients computing the homology groups. It depends on $\NameCert_{\AbGroupUp}$ (\autoref{ch:concrete-instances-abgroup-namecert}) and $\NameCert_{\QuotientGroupUp}$ (\autoref{ch:concrete-instances-quotientgroup-namecert}), and is the constructive substrate for both singular and simplicial homology.

\section{Carrier definition}
\label{sec:homology-carrier}

The cycle carrier of a BEDC chain differential $d:\mathsf{BHist}\to
\mathsf{BHist}$ is the kernel predicate
$$
Z_d(h) :\Longleftrightarrow \hsame(d(h),\mathsf{Empty}).
$$
The boundary carrier is the image predicate
$$
B_d(h) :\Longleftrightarrow \exists u:\mathsf{BHist},\ \hsame(h,d(u)).
$$
\leandef{BEDC.Derived.HomologyUp.HomologyCycleCarrier}
\leandef{BEDC.Derived.HomologyUp.HomologyBoundaryCarrier}

\begin{theorem}[Homology boundary carrier nonempty preimage]
\label{thm:homology-boundary-carrier-nonempty-preimage}
If $B_d(h)$ and $h$ is not history-same to $\mathsf{Empty}$, then some
boundary preimage $u$ has a nonempty differential endpoint:
$$
\exists u:\mathsf{BHist},\
\hsame(h,d(u))\land \neg\hsame(d(u),\mathsf{Empty}).
$$
\end{theorem}
\begin{proof}
Choose the boundary witness $u$ with $\hsame(h,d(u))$. If $d(u)$ were
history-same to $\mathsf{Empty}$, transitivity of $\hsame$ would make $h$
history-same to $\mathsf{Empty}$, contradicting the nonempty hypothesis.
\end{proof}
\leanchecked{BEDC.Derived.HomologyUp.HomologyBoundaryCarrier\_nonempty\_preimage}

\begin{theorem}[Homology boundary carrier empty preimage]
\label{thm:homology-boundary-carrier-empty-preimage}
If $B_d(h)$ and $h\hsame\mathsf{Empty}$, then some boundary preimage $u$
has an empty differential endpoint:
$$
\exists u:\mathsf{BHist},\
\hsame(h,d(u))\land \hsame(d(u),\mathsf{Empty}).
$$
\end{theorem}
\begin{proof}
Choose the boundary witness $u$ with $\hsame(h,d(u))$. Symmetry of that
witness followed by the displayed empty endpoint comparison gives
$\hsame(d(u),\mathsf{Empty})$.
\end{proof}
\leanchecked{BEDC.Derived.HomologyUp.HomologyBoundaryCarrier\_empty\_preimage}

\begin{theorem}[Homology boundary carrier cycle closure]
\label{thm:homology-boundary-carrier-cycle-closed}
Assume the BEDC chain differential is square-zero:
$$
\hsame(d(d(u)),\mathsf{Empty}).
$$
Every boundary history is a cycle history: if $B_d(h)$, then $Z_d(h)$.
\end{theorem}
\begin{proof}
Choose a boundary witness $u$ with $\hsame(h,d(u))$. Transporting the cycle
test along this witness reduces the claim to the displayed square-zero law at
$u$.
\end{proof}
\leanchecked{BEDC.Derived.HomologyUp.HomologyBoundaryCarrier\_cycle\_closed}

\begin{theorem}[Homology cycle carrier append closure]
\label{thm:homology-cycle-carrier-append-closed}
Assume the differential respects BEDC continuation append:
$$
\hsame(d(\mathsf{append}(u,v)),\mathsf{append}(d(u),d(v))).
$$
If $Z_d(h)$ and $Z_d(k)$, then
$Z_d(\mathsf{append}(h,k))$.
\end{theorem}
\begin{proof}
The append law transports the displayed differential to
$\mathsf{append}(d(h),d(k))$. The two cycle hypotheses identify both
endpoints with $\mathsf{Empty}$, and the BEDC append kernel has
$\mathsf{append}(\mathsf{Empty},\mathsf{Empty})=\mathsf{Empty}$.
\end{proof}
\leanchecked{BEDC.Derived.HomologyUp.HomologyCycleCarrier\_append\_closed}

\begin{theorem}[Homology cycle carrier append components]
\label{thm:homology-cycle-carrier-append-components}
Assume the differential respects BEDC continuation append:
$$
\hsame(d(\mathsf{append}(u,v)),\mathsf{append}(d(u),d(v))).
$$
Then an appended history is a cycle exactly when both appended components are
cycles:
$$
Z_d(\mathsf{append}(h,k))
\quad\Longleftrightarrow\quad
Z_d(h)\wedge Z_d(k).
$$
\end{theorem}
\begin{proof}
The forward direction transports $d(\mathsf{append}(h,k))$ to
$\mathsf{append}(d(h),d(k))$. If this endpoint is empty, BEDC append-empty
inversion gives $d(h)\hsame\mathsf{Empty}$ and
$d(k)\hsame\mathsf{Empty}$. The reverse direction is the homology cycle
append-closure theorem.
\end{proof}
\leanchecked{BEDC.Derived.HomologyUp.HomologyCycleCarrier\_append\_components\_iff}

\begin{theorem}[Homology boundary carrier append closure]
\label{thm:homology-boundary-carrier-append-closed}
Assume the differential respects BEDC continuation append:
$$
\hsame(d(\mathsf{append}(u,v)),\mathsf{append}(d(u),d(v))).
$$
If $B_d(h)$ and $B_d(k)$, then
$B_d(\mathsf{append}(h,k))$.
\end{theorem}
\begin{proof}
Choose boundary witnesses $u$ and $v$ with $\hsame(h,d(u))$ and
$\hsame(k,d(v))$. The BEDC witness for the appended boundary is
$\mathsf{append}(u,v)$. Endpoint congruence carries
$\mathsf{append}(h,k)$ to $\mathsf{append}(d(u),d(v))$, and the differential
append law identifies that history with $d(\mathsf{append}(u,v))$.
\end{proof}
\leanchecked{BEDC.Derived.HomologyUp.HomologyBoundaryCarrier\_append\_closed}

\begin{theorem}[Homology boundary carrier append hsame transport]
\label{thm:homology-boundary-carrier-append-hsame-transport}
Assume the differential respects BEDC continuation append. If $B_d(h)$ and
$B_d(k)$ and $\hsame(\mathsf{append}(h,k),r)$, then $B_d(r)$.
\end{theorem}
\begin{proof}
Choose boundary witnesses $u$ and $v$ for $h$ and $k$. Their appended BEDC
preimage $\mathsf{append}(u,v)$ maps, by the differential append law, to the
same endpoint as $\mathsf{append}(d(u),d(v))$. The two boundary endpoint
comparisons identify $\mathsf{append}(h,k)$ with this value, and the final
$\hsame$ comparison transports the image witness to $r$.
\end{proof}
\leanchecked{BEDC.Derived.HomologyUp.HomologyBoundaryCarrier\_append\_hsame\_transport}

\begin{theorem}[Homology boundary carrier continuation cycle closure]
\label{thm:homology-boundary-cont-cycle-closed}
Assume the differential respects BEDC continuation append and is square-zero.
If $B_d(h)$, $B_d(k)$, and $\ContR(h,k,r)$, then $Z_d(r)$.
\end{theorem}
\begin{proof}
Boundary append closure with endpoint transport first makes the continuation
result $r$ a boundary carrier. The square-zero differential condition turns any
boundary carrier into a cycle carrier, so $Z_d(r)$ follows.
\end{proof}
\leanchecked{BEDC.Derived.HomologyUp.HomologyBoundaryCarrier\_cont\_cycle\_closed}

\begin{theorem}[Homology boundary carrier continuation preimage]
\label{thm:homology-boundary-carrier-cont-preimage}
Assume the differential respects BEDC continuation append. If $B_d(h)$,
$B_d(k)$, and $\Cont(h,k,r)$, then the continuation result has compatible
boundary preimages:
$$
\exists u,v,uv:\mathsf{BHist},\
\Cont(u,v,uv)\wedge \hsame(r,d(uv))\wedge
\hsame(h,d(u))\wedge \hsame(k,d(v)).
$$
\end{theorem}
\begin{proof}
Choose boundary witnesses $u$ and $v$ for $h$ and $k$. Set
$uv=\mathsf{append}(u,v)$, giving $\Cont(u,v,uv)$ by the BEDC continuation
constructor. The input continuation reads $r$ as
$\mathsf{append}(h,k)$; the two boundary witnesses identify this with
$\mathsf{append}(d(u),d(v))$, and the differential append law identifies the
endpoint with $d(uv)$.
\end{proof}
\leanchecked{BEDC.Derived.HomologyUp.HomologyBoundaryCarrier\_cont\_preimage}

\begin{theorem}[Homology boundary carrier continuation preimage append]
\label{thm:homology-boundary-carrier-cont-preimage-append}
Assume the differential respects BEDC continuation append. If $B_d(h)$,
$B_d(k)$, and $\ContR(h,k,r)$, then there are BEDC preimages $u$ and $v$
such that
$$
\hsame(h,d(u))\land
\hsame(k,d(v))\land
\hsame(r,d(\mathsf{append}(u,v))).
$$
\end{theorem}
\begin{proof}
Choose boundary witnesses $u$ and $v$ for $h$ and $k$. The continuation ledger
identifies $r$ with $\mathsf{append}(h,k)$. The witness endpoints identify
that appended history with $\mathsf{append}(d(u),d(v))$, and the differential
append law identifies this with $d(\mathsf{append}(u,v))$.
\end{proof}
\leanchecked{BEDC.Derived.HomologyUp.HomologyBoundaryCarrier\_cont\_preimage\_append}

\begin{theorem}[Homology boundary carrier continuation nonempty preimage append]
\label{thm:homology-boundary-carrier-cont-nonempty-preimage-append}
Assume the differential respects BEDC continuation append. If $B_d(h)$,
$B_d(k)$, $\ContR(h,k,r)$, and $r$ is not history-same to $\mathsf{Empty}$,
then there are BEDC preimages $u$ and $v$ such that
$$
\hsame(h,d(u))\land
\hsame(k,d(v))\land
\neg\hsame(d(\mathsf{append}(u,v)),\mathsf{Empty}).
$$
\end{theorem}
\begin{proof}
The preimage-append theorem gives witnesses $u$ and $v$ with
$\hsame(r,d(\mathsf{append}(u,v)))$. If the displayed differential endpoint
were history-same to $\mathsf{Empty}$, transitivity would make $r$ history-same
to $\mathsf{Empty}$, contradicting the nonempty hypothesis.
\end{proof}
\leanchecked{BEDC.Derived.HomologyUp.HomologyBoundaryCarrier\_cont\_nonempty\_preimage\_append}

\begin{theorem}[Homology boundary carrier append empty preimages]
\label{thm:homology-boundary-carrier-append-empty-preimages}
If $B_d(h)$ and $B_d(k)$ and
$\hsame(\mathsf{append}(h,k),\mathsf{Empty})$, then the two boundary
witnesses can be chosen with empty differential endpoints:
$$
\exists u,v:\mathsf{BHist},\
\hsame(h,d(u))\land \hsame(k,d(v))\land
\hsame(d(u),\mathsf{Empty})\land\hsame(d(v),\mathsf{Empty}).
$$
\end{theorem}
\begin{proof}
BEDC append-empty inversion splits the appended endpoint comparison into
$h\hsame\mathsf{Empty}$ and $k\hsame\mathsf{Empty}$. Applying empty-preimage
inversion to the two boundary carrier hypotheses produces witnesses $u$ and
$v$ with the stated endpoint comparisons, which are then paired.
\end{proof}
\leanchecked{BEDC.Derived.HomologyUp.HomologyBoundaryCarrier\_append\_empty\_preimages}

\begin{theorem}[Homology boundary carrier append nonempty preimage]
\label{thm:homology-boundary-carrier-append-nonempty-preimage}
If $B_d(h)$ and $B_d(k)$ and
$\mathsf{append}(h,k)$ is not history-same to $\mathsf{Empty}$, then at least
one side has a boundary preimage with a nonempty differential endpoint:
$$
\begin{aligned}
&\left(\exists u:\mathsf{BHist},\
\hsame(h,d(u))\land\neg\hsame(d(u),\mathsf{Empty})\right)\\
&\quad{}\lor
\left(\exists v:\mathsf{BHist},\
\hsame(k,d(v))\land\neg\hsame(d(v),\mathsf{Empty})\right).
\end{aligned}
$$
\end{theorem}
\begin{proof}
BEDC append nonemptiness splits into a nonempty left component or a nonempty
right component. In the left case, boundary nonempty-preimage inversion
applied to $B_d(h)$ gives the left disjunct. In the right case, the same
argument applied to $B_d(k)$ gives the right disjunct.
\end{proof}
\leanchecked{BEDC.Derived.HomologyUp.HomologyBoundaryCarrier\_append\_nonempty\_preimage}

\begin{theorem}[Homology boundary carrier continuation cycle components]
\label{thm:homology-boundary-cont-cycle-components}
Assume the differential respects BEDC continuation append and is square-zero.
If $B_d(r)$ and $\ContR(h,k,r)$, then both continuation components are cycles:
$$
Z_d(h)\wedge Z_d(k).
$$
\end{theorem}
\begin{proof}
The square-zero law turns the boundary carrier on $r$ into a cycle carrier on
the continuation result. Since $\ContR(h,k,r)$ identifies $r$ with
$\mathsf{append}(h,k)$, the append-components theorem splits that cycle
predicate into the two component cycle predicates.
\end{proof}
\leanchecked{BEDC.Derived.HomologyUp.HomologyBoundaryCarrier\_cont\_cycle\_components}

\begin{theorem}[Homology boundary carrier cycle from square-zero differential]
\label{thm:homology-boundary-carrier-cycle-of-d-squared-zero}
Assume the differential respects $\hsame$ and satisfies the square-zero
condition
$$
\hsame(d(d(u)),\mathsf{Empty})
$$
for every $u:\mathsf{BHist}$. If $B_d(h)$, then $Z_d(h)$.
\end{theorem}
\begin{proof}
Choose a boundary witness $u$ with $\hsame(h,d(u))$. Congruence for the
differential transports this witness to
$\hsame(d(h),d(d(u)))$. The square-zero row identifies $d(d(u))$ with
$\mathsf{Empty}$, and transitivity of $\hsame$ gives the cycle predicate.
\end{proof}
\leanchecked{BEDC.Derived.HomologyUp.HomologyBoundaryCarrier\_cycle\_of\_d\_squared\_zero}

\begin{theorem}[Homology cycle carrier append hsame transport]
\label{thm:homology-cycle-carrier-append-hsame-transport}
Assume the differential respects BEDC continuation append and is congruent
for $\hsame$. If $Z_d(h)$, $Z_d(k)$, and
$\hsame(\mathsf{append}(h,k),r)$, then $Z_d(r)$.
\end{theorem}
\begin{proof}
The append-closure theorem gives
$Z_d(\mathsf{append}(h,k))$. Congruence for the differential transports the
endpoint comparison to an $\hsame$ comparison between
$d(\mathsf{append}(h,k))$ and $d(r)$. Symmetry of that comparison, followed
by the cycle row for the append, proves that $d(r)$ is $\hsame$ to
$\mathsf{Empty}$.
\end{proof}
\leanchecked{BEDC.Derived.HomologyUp.HomologyCycleCarrier\_append\_hsame\_transport}

\section{The certificate}
\label{sec:homology-certificate}

\begin{theorem}[Homology singleton cycle semantic name certificate]
\label{thm:homology-singleton-cycle-semantic-namecert}
Let the singleton cycle carrier contain exactly the BEDC histories
$h\hsame\mathsf{BHist.Empty}$. Let its classifier require both endpoints to
lie in that carrier and require $h\hsame k$. This classifier is a semantic
naming certificate for the carrier used simultaneously as source, pattern,
and ledger policy.
\end{theorem}
\begin{proof}
The carrier is inhabited by $\mathsf{BHist.Empty}$ through reflexivity of
$\hsame$. Classifier reflexivity, symmetry, and transitivity are obtained by
unpacking the two endpoint-carrier witnesses and composing the underlying
$\hsame$ proofs. Carrier transport along a classifier edge is the right
endpoint carrier witness already stored in the classifier. The semantic
pattern and ledger fields are the source witness itself.
\end{proof}
\leanchecked{BEDC.Derived.HomologyUp.HomologySingletonCycle\_semanticNameCert}
