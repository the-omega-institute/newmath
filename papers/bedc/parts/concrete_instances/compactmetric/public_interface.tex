\paragraph{Public export and downstream consumption}

\begin{definition}[Compact metric public interface]
\label{def:compactmetric-public-interface}
For a certified metric carrier $X$, the public compact-metric interface
$$
\mathsf{CompactMetricPublic}(X)
$$
is the exported package consisting exactly of the three visible fields of
\autoref{def:compactmetric-certificate-carrier}: the $\MetricSpaceUp$
carrier, classifier, and distance rows; the $\TotallyBoundedUp$
positive-tolerance $\mathsf{ProbeBundle}$-net selector
$\varepsilon \mapsto (\mathcal{P}_{\varepsilon},net_{\varepsilon})$; and the
$\CompleteMetricUp$ regular-Cauchy limit-witness selector
$s \mapsto (\mu_s,\ell_s,L_s)$. No open-cover compactness witness, subsequence
selector, or host-level compact-object predicate is part of
$\mathsf{CompactMetricPublic}(X)$.
\end{definition}
\leanchecked{BEDC.Derived.CompactMetricUp.CompactMetricTasteGate\_single\_carrier\_alignment}

\begin{theorem}[Compact metric public interface stability]
\label{thm:compactmetric-public-interface-stability}
For a certified metric carrier $X$, the package
$\mathsf{CompactMetricPublic}(X)$ is transported by the inherited metric
classifier. The metric carrier and distance rows are carried by the
$\MetricSpaceUp$ classifier data; each positive-tolerance finite-net row is
transported by
\autoref{thm:totallybounded-probebundle-net-coverage-stability}; and each
regular-Cauchy limit-witness row is transported by
\autoref{thm:completemetric-limit-witness-stability}.
\end{theorem}
\leanchecked{BEDC.Derived.CompactMetricUp.CompactMetricPublicInterface\_stability}
\begin{proof}
Apply \autoref{thm:compactmetric-certificate-package} to view
$\mathsf{CompactMetricPublic}(X)$ as the visible export of
$\mathsf{CompactMetricCertificate}(X)$. The inherited metric classifier
transports the carrier histories and distance rows, so the public metric field
has the same source classifier as the certificate package.

For the totally-bounded selector, fix a transported positive $\RatUp$
tolerance history. The certificate supplies the matching closed
$\mathsf{ProbeBundle}$ net and its distance and comparison ledgers, and
\autoref{thm:totallybounded-probebundle-net-coverage-stability} transports
that whole row.

For the complete-metric selector, fix a transported regular Cauchy stream name.
The certificate supplies the modulus, limit history, threshold rows, and
late-observation distance bounds, and
\autoref{thm:completemetric-limit-witness-stability} transports the displayed
limit witness. These three transported fields are precisely
$\mathsf{CompactMetricPublic}(X)$.
\end{proof}

\begin{theorem}[Compact metric downstream consumption bridge]
\label{thm:compactmetric-downstream-consumption-bridge}
Any downstream $\ContinuousMapUp$ or Heine--Cantor argument whose compact input
requirements are a metric carrier, explicit finite nets at positive rational
tolerances, and regular-Cauchy limit witnesses may consume exactly the three
fields of $\mathsf{CompactMetricPublic}(X)$. Such an argument gains no
additional open-cover compactness principle and no subsequence principle from
the compact-metric input.
\end{theorem}
\leanchecked{BEDC.Derived.CompactMetricUp.CompactMetricPublicInterface\_downstream\_consumption\_bridge}
\begin{proof}
Unpack \autoref{def:compactmetric-public-interface}. A request for the carrier,
classifier, or distance rows is routed to the $\MetricSpaceUp$ field. This is
the same metric source used by the total-boundedness and completeness clauses,
so no extra carrier equivalence is introduced.

A request for a finite net at a positive rational tolerance is routed to the
$\TotallyBoundedUp$ selector
$\varepsilon \mapsto (\mathcal{P}_{\varepsilon},net_{\varepsilon})$. Its
centers, membership rows, distance witnesses, and rational comparison ledgers
are the rows supplied by the certificate carrier.

A request for a limit witness for a regular Cauchy stream is routed to the
$\CompleteMetricUp$ selector $s \mapsto (\mu_s,\ell_s,L_s)$. The modulus,
limit history, threshold ledger, and late-distance bounds are exactly the rows
available to the downstream proof.

The public interface has no fourth field. Therefore a downstream
$\ContinuousMapUp$ or Heine--Cantor proof that uses
$\mathsf{CompactMetricPublic}(X)$ consumes only metric rows, positive-tolerance
ProbeBundle nets, and regular-Cauchy limit witnesses.
\end{proof}

\begin{theorem}[CompactMetricUp metric source scope]
\label{thm:compactmetric-metric-source-scope}
For every certified metric carrier $X$ equipped with
$\mathsf{CompactMetricCertificate}(X)$, the public $\CompactMetricUp$
certificate exposes the same $\MetricSpaceUp$ source carrier, point
classifier, and $\mathsf{MetricDistanceWitness}$ rows used by the
totally-bounded and complete-metric fields. The finite-net family and the
complete-limit family are therefore both read over one metric classifier
scope, with no independent carrier or distance package.
\end{theorem}
\begin{proof}
Project the first field of
\autoref{def:compactmetric-certificate-carrier}. It is already the
$\MetricSpaceUp$ carrier and distance package for $X$, so it supplies the
source history object, point classifier, and distance-witness ledger rows.

The remaining certificate fields are indexed by the same carrier $X$ and use
the same point histories when they mention net centers, stream observations,
and limit histories. Repack these rows under the projected classifier scope.
This gives the displayed $\CompactMetricUp$ source data consumed by both the
totally-bounded and complete-metric obligations.
\end{proof}

\begin{theorem}[CompactMetricUp finite-net family coverage]
\label{thm:compactmetric-net-family-coverage}
For every certified metric carrier $X$, every
$\mathsf{CompactMetricCertificate}(X)$, and every positive $\RatUp$ tolerance
history $\varepsilon$, the certificate projects a
$\mathsf{TotallyBoundedProbeBundleNet}(X,\varepsilon,\mathcal{P}_{\varepsilon},net_{\varepsilon})$
package. This package is stable under the metric carrier classifier and
rational tolerance classifier transport of
\autoref{thm:totallybounded-probebundle-net-namecert-target} and
\autoref{thm:totallybounded-probebundle-net-coverage-stability}.
\end{theorem}
\begin{proof}
Fix the positive tolerance row $\varepsilon$. Project the second field of
\autoref{def:compactmetric-certificate-carrier}; it supplies the displayed
$\mathsf{TotallyBoundedProbeBundleNet}$ package at that tolerance, including
the ProbeBundle centers, membership rows, distance witnesses, and rational
comparison ledgers.

The package is a $\NameCert_{\TotallyBoundedUp}$ target by
\autoref{thm:totallybounded-probebundle-net-namecert-target}. Apply
\autoref{thm:totallybounded-probebundle-net-coverage-stability} to transport
the net along the shared metric classifier and the transported positive
$\RatUp$ tolerance. The resulting row is the finite-net family coverage
exported by the $\CompactMetricUp$ certificate.
\end{proof}

\begin{theorem}[CompactMetricUp complete-limit family coverage]
\label{thm:compactmetric-limit-family-completeness}
For every certified metric carrier $X$, every
$\mathsf{CompactMetricCertificate}(X)$, and every regular
$\StreamNameUp$ Cauchy stream name $s$ whose observations are carried by $X$,
the certificate projects a
$\mathsf{CompleteMetricLimitWitness}(X,s,\mu_s,\ell_s,L_s)$ as in
\autoref{def:completemetric-limit-witness}. This witness is stable under
classifier transport by
\autoref{thm:completemetric-limit-witness-stability}.
\end{theorem}
\begin{proof}
Fix the regular Cauchy stream name $s$ and its carrier row over $X$. Project
the third field of \autoref{def:compactmetric-certificate-carrier}; it gives
the modulus history, limit history, threshold ledgers, and late-observation
distance bounds that form the displayed
$\mathsf{CompleteMetricLimitWitness}$.

Apply \autoref{thm:completemetric-limit-witness-stability} pointwise to the
stream observations, the modulus, the proposed limit, and the late-bound
ledger rows. The transported rows again satisfy
\autoref{def:completemetric-limit-witness}, so the complete-limit family is
available as a separate $\CompactMetricUp$ obligation site.
\end{proof}
