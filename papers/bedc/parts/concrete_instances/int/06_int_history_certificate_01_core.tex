\section{The certificate}

\begin{definition}[Integer history carrier]
\label{def:intup-history-carrier}
The public BEDC-history carrier for $\IntUp$ packages the signed unary
representative into the visible history tag:
$$
\mathsf{IntHistoryCarrier}(h)
:=
\bigl(\exists m:\mathsf{BHist}.\;
\mathsf{Unary}(m)\land\hsame(h,\Ezero(m))\bigr)
\lor
\bigl(\exists m:\mathsf{BHist}.\;
\mathsf{Unary}(m)\land\hsame(h,\Eone(m))\bigr).
$$
The $\Ezero$ branch and the $\Eone$ branch encode the two signed unary
presentations as actual BEDC histories.
\end{definition}
\leandef{BEDC.Derived.IntUp.IntHistoryCarrier}

\begin{theorem}[Integer history carrier signed branch determinacy]
\label{thm:int-history-carrier-signed-branch-determinacy}
For every history $h:\mathsf{BHist}$, if $\mathsf{IntHistoryCarrier}(h)$,
then exactly one visible signed carrier branch supplies the witness:
$$
\begin{gathered}
\Bigl(\exists m:\mathsf{BHist}.\;
\mathsf{Unary}(m)\land\hsame(h,\Ezero(m))\land
\forall p:\mathsf{BHist}.\;
\mathsf{Unary}(p)\to\hsame(h,\Eone(p))\to\mathsf{False}
\Bigr)\\
\lor\\
\Bigl(\exists m:\mathsf{BHist}.\;
\mathsf{Unary}(m)\land\hsame(h,\Eone(m))\land
\forall p:\mathsf{BHist}.\;
\mathsf{Unary}(p)\to\hsame(h,\Ezero(p))\to\mathsf{False}
\Bigr).
\end{gathered}
$$
The ten-field fit is: traditional target signed-integer carrier readback; BEDC
source $\mathsf{BHist}$, $\hsame$, $\Ezero$, and $\Eone$; carrier
$\mathsf{IntHistoryCarrier}$; classifier not needed; stability from constructor
inversion and $\hsame$ laws; ledger visible sign tag plus unary payload; core
theorem this carrier branch determinacy block; exactness exclusive branch
determinacy; standard bridge ordinary signed unary integer tag readback; status
$\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Split the defining disjunction of $\mathsf{IntHistoryCarrier}(h)$.

In the zero-tag branch, keep the unary payload $m$ and the endpoint witness
$\hsame(h,\Ezero(m))$. If a one-tag carrier witness for the same $h$ were also
available, its endpoint witness $\hsame(h,\Eone(p))$ would compose with
$\hsame(h,\Ezero(m))$, using symmetry on one side, to produce
$\hsame(\Ezero(m),\Eone(p))$. BEDC history constructor no-confusion rules out
that mixed visible-tag equality.

In the one-tag branch, keep the unary payload and endpoint witness in the same
way. A hypothetical zero-tag witness composes into
$\hsame(\Eone(m),\Ezero(p))$, again contradicting constructor no-confusion.
Thus the carrier witness determines the visible signed branch before any
classifier evidence is used.
\end{proof}
\leanchecked{BEDC.Derived.IntUp.IntHistoryCarrier\_signed\_branch\_determinacy}

\begin{definition}[Integer history classifier]
\label{def:intup-history-classifier}
The public classifier relates exactly same-tag signed unary histories:
$$
\mathsf{IntHistoryClassifier}(h,k)
:=
\bigl(\exists m,n:\mathsf{BHist}.\;
\mathsf{Unary}(m)\land\mathsf{Unary}(n)\land
\hsame(h,\Ezero(m))\land\hsame(k,\Ezero(n))\land\hsame(m,n)\bigr)
\lor
\bigl(\exists m,n:\mathsf{BHist}.\;
\mathsf{Unary}(m)\land\mathsf{Unary}(n)\land
\hsame(h,\Eone(m))\land\hsame(k,\Eone(n))\land\hsame(m,n)\bigr).
$$
\end{definition}
\leandef{BEDC.Derived.IntUp.IntHistoryClassifier}

\begin{theorem}[Integer history classifier carrier endpoints]
\label{thm:int-history-classifier-carrier-endpoints}
For all histories $h,k:\mathsf{BHist}$, classifier evidence exposes both
endpoints as public signed-history carriers:
$$
\forall h,k:\mathsf{BHist},\quad
\mathsf{IntHistoryClassifier}(h,k)\to
\mathsf{IntHistoryCarrier}(h)\land\mathsf{IntHistoryCarrier}(k).
$$
The ten-field fit is: traditional target domain coverage for signed-unary
integer equality; BEDC source $\mathsf{IntHistoryCarrier}$,
$\mathsf{IntHistoryClassifier}$, $\mathsf{BHist}$, $\hsame$, $\Ezero$, and
$\Eone$; carrier the public signed-history carrier; classifier the two-branch
public classifier; stability extraction of admitted endpoints from classifier
evidence; ledger visible sign tag plus unary payload; core theorem this
carrier-endpoint block; exactness classifier domain coverage; standard bridge
equality only relates admitted signed representatives; status
$\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Split the classifier witness into its two visible signed branches.

In the zero branch, expose payloads $m,n:\mathsf{BHist}$, the unary witnesses
for $m$ and $n$, and the endpoint witnesses $\hsame(h,\Ezero(m))$ and
$\hsame(k,\Ezero(n))$. The carrier witness for $h$ is the zero-tag
$\mathsf{IntHistoryCarrier}$ branch with payload $m$, its unary witness, and
the displayed endpoint witness. The carrier witness for $k$ is packed in the
same way using payload $n$.

In the one branch, expose the corresponding payloads, unary witnesses, and
endpoint witnesses $\hsame(h,\Eone(m))$ and $\hsame(k,\Eone(n))$. Packing the
one-tag carrier branch for $h$ and for $k$ gives the two conjuncts required by
the statement.
\end{proof}

\begin{theorem}[Integer history classifier mixed visible-tag exclusion]
\label{thm:int-history-classifier-mixed-visible-tag-exclusion}
For all $m,n:\mathsf{BHist}$, the public integer-history classifier excludes
opposite visible sign tags without requiring the displayed payloads themselves
to be externally unary:
$$
\forall m,n:\mathsf{BHist},\quad
\bigl(\mathsf{IntHistoryClassifier}(\Ezero(m),\Eone(n))\to\mathsf{False}\bigr)
\land
\bigl(\mathsf{IntHistoryClassifier}(\Eone(m),\Ezero(n))\to\mathsf{False}\bigr).
$$
The ten-field fit is: traditional target signed-integer mixed-tag separation;
BEDC source $\mathsf{BHist}$, $\hsame$, $\Ezero$, $\Eone$, and
$\mathsf{IntHistoryClassifier}$; carrier $\mathsf{IntHistoryCarrier}$;
classifier the public two-branch classifier; stability from constructor
no-confusion and $\hsame$ composition; ledger visible sign tag plus hidden unary
payload; core theorem this mixed visible-tag exclusion block; exactness
mixed-tag exclusion; standard bridge ordinary signed-unary sign disjointness;
status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Fix $m,n:\mathsf{BHist}$.

For the endpoint pair $\Ezero(m),\Eone(n)$, split the defining disjunction of
$\mathsf{IntHistoryClassifier}$. In the zero-branch witness, the second endpoint
is also exposed as some $\Ezero(q)$, while the displayed endpoint supplies
$\hsame(\Eone(n),\Ezero(q))$ up to symmetry. Composing these endpoint witnesses
gives a mixed visible-tag history sameness judgment, contradicting BEDC history
constructor no-confusion. In the one-branch witness, the first endpoint is also
exposed as some $\Eone(p)$, and the same composition gives
$\hsame(\Ezero(m),\Eone(p))$, again impossible by constructor no-confusion.

For the endpoint pair $\Eone(m),\Ezero(n)$, use the symmetric case split. A
zero-branch witness exposes the first endpoint as $\Ezero(p)$ and composes into
$\hsame(\Eone(m),\Ezero(p))$; a one-branch witness exposes the second endpoint
as $\Eone(q)$ and composes into $\hsame(\Ezero(n),\Eone(q))$. Both conclusions
are mixed visible-tag history sameness judgments, so constructor no-confusion
closes the two cases. The unary witnesses hidden inside the classifier branches
are used only to recognize the public branch witnesses; no unary hypothesis on
the displayed payloads $m,n$ is needed.
\end{proof}
\leanchecked{BEDC.Derived.IntUp.IntHistoryClassifier\_mixed\_visible\_tag\_exclusion}

\begin{theorem}[Integer history classifier visible branch exactness]
\label{thm:int-history-classifier-visible-branch-exactness}
For any $m,n:\mathsf{BHist}$ with $\mathsf{Unary}(m)$ and
$\mathsf{Unary}(n)$, the public integer-history classifier reads back the
same visible tag precisely as payload history sameness:
$$
\bigl(
\mathsf{IntHistoryClassifier}(\Ezero(m),\Ezero(n))
\Longleftrightarrow
\hsame(m,n)
\bigr)
\land
\bigl(
\mathsf{IntHistoryClassifier}(\Eone(m),\Eone(n))
\Longleftrightarrow
\hsame(m,n)
\bigr).
$$
The ten-field fit is: traditional target signed-integer visible-tag readback;
BEDC source $\mathsf{BHist}$; admissible payloads $\mathsf{Unary}$; payload
relation $\hsame$; public classifier $\mathsf{IntHistoryClassifier}$; carrier
the public integer history carrier; classifier same-tag payload sameness;
stability by constructor inversion and $\hsame$ laws; ledger visible tag plus
unary payload; exactness the two iff clauses; standard bridge ordinary
signed-integer tag readback; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
For the zero-tag forward direction, split a witness of
$\mathsf{IntHistoryClassifier}(\Ezero(m),\Ezero(n))$ by the two displayed
classifier branches. In the zero branch, constructor inversion on the endpoint
witnesses reduces the visible histories to payload histories, and transitivity
of $\hsame$ composes the two endpoint equalities with the stored payload
comparison. In the one branch, the endpoint evidence would identify a zero tag
with a one tag, contradicting history no-confusion.

The one-tag forward direction is symmetric. The matching one branch gives the
payload comparison by constructor inversion and transitivity of $\hsame$; the
zero branch is ruled out by the same zero/one history no-confusion.

For the reverse directions, assume $\hsame(m,n)$ together with the displayed
unary witnesses for $m$ and $n$. The zero-tag classifier witness is obtained by
choosing the zero branch, using the two unary witnesses, reflexive endpoint
witnesses for $\Ezero(m)$ and $\Ezero(n)$, and the given payload relation. The
one-tag case is identical with $\Eone$ in place of $\Ezero$.
\end{proof}
\leanchecked{BEDC.Derived.IntUp.IntHistoryClassifier\_visible\_branch\_exactness\_iff}

\begin{theorem}[Integer history classifier signed branch determinacy]
\label{thm:int-history-classifier-signed-branch-determinacy}
For all histories $h,k:\mathsf{BHist}$, if
$\mathsf{IntHistoryClassifier}(h,k)$, then exactly one visible signed branch
supplies the classifier witnesses:
$$
\begin{gathered}
\Bigl(\exists m,n:\mathsf{BHist}.\;
\mathsf{Unary}(m)\land\mathsf{Unary}(n)\land
\hsame(h,\Ezero(m))\land\hsame(k,\Ezero(n))\land\hsame(m,n)\\
\land\;
\forall p,q:\mathsf{BHist}.\;
\mathsf{Unary}(p)\to\mathsf{Unary}(q)\to
\hsame(h,\Eone(p))\to\hsame(k,\Eone(q))\to\hsame(p,q)\to\mathsf{False}
\Bigr)\\
\lor\\
\Bigl(\exists m,n:\mathsf{BHist}.\;
\mathsf{Unary}(m)\land\mathsf{Unary}(n)\land
\hsame(h,\Eone(m))\land\hsame(k,\Eone(n))\land\hsame(m,n)\\
\land\;
\forall p,q:\mathsf{BHist}.\;
\mathsf{Unary}(p)\to\mathsf{Unary}(q)\to
\hsame(h,\Ezero(p))\to\hsame(k,\Ezero(q))\to\hsame(p,q)\to\mathsf{False}
\Bigr).
\end{gathered}
$$
The ten-field fit is: traditional target signed integer branch readback; BEDC
source $\mathsf{BHist}$, $\hsame$, $\Ezero$, $\Eone$, and
$\mathsf{IntHistoryClassifier}$; carrier $\mathsf{IntHistoryCarrier}$;
classifier $\mathsf{IntHistoryClassifier}$; stability inherited from
\autoref{thm:concrete-int-history-semantic-namecert}; ledger the visible sign
tag and unary payload; core theorem this signed branch determinacy block;
exactness signed branch determinacy; standard bridge ordinary signed-integer
sign readback; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Split the defining disjunction of $\mathsf{IntHistoryClassifier}(h,k)$.

In the zero-tag case, keep the exposed unary payloads $m,n$, the two visible
endpoint witnesses $h\hsame\Ezero(m)$ and $k\hsame\Ezero(n)$, and the payload
comparison $m\hsame n$. To rule out a one-tag witness for the same endpoints,
compose $h\hsame\Ezero(m)$ with the hypothetical $h\hsame\Eone(p)$, using
symmetry on one side as needed. The result is a visible contradiction
$\hsame(\Ezero(m),\Eone(p))$ by BEDC constructor no-confusion.

In the one-tag case, keep the exposed unary payloads and endpoint witnesses in
the same way. A hypothetical zero-tag witness for the same endpoints composes
the endpoint comparisons into $\hsame(\Eone(m),\Ezero(p))$, again contradicting
BEDC constructor no-confusion. Thus the classifier witness determines the
visible signed branch and leaves only the payload equality inside that branch.
\end{proof}
\leanchecked{BEDC.Derived.IntUp.IntHistoryClassifier\_signed\_branch\_determinacy}

\begin{theorem}[Concrete IntUp history semantic certificate]
\label{thm:concrete-int-history-semantic-namecert}
The integer history carrier and classifier form the public semantic naming
certificate for the concrete signed unary history presentation:
$$
\mathsf{SemanticNameCert}(
\mathsf{IntHistoryCarrier},
\mathsf{IntHistoryCarrier},
\mathsf{IntHistoryCarrier},
\mathsf{IntHistoryClassifier}).
$$
Moreover the classifier excludes mixed visible tags:
$$
\forall m,n:\mathsf{BHist},\quad
\mathsf{Unary}(m)\to\mathsf{Unary}(n)\to
\bigl(\mathsf{IntHistoryClassifier}(\Ezero(m),\Eone(n))\to\mathsf{False}\bigr)
\land
\bigl(\mathsf{IntHistoryClassifier}(\Eone(m),\Ezero(n))\to\mathsf{False}\bigr).
$$
\end{theorem}
\begin{proof}
For inhabitance, choose the zero-tagged empty magnitude
$\Ezero(\emp)$. The unary base witness gives $\mathsf{Unary}(\emp)$, and
reflexivity of $\hsame$ gives the visible endpoint witness.

For reflexivity, split the carrier witness for $h$. In the $\Ezero$ branch,
reuse the same magnitude on both endpoints and use reflexivity of $\hsame$ for
the payload comparison. The $\Eone$ branch is identical with the one-tagged
constructor.

For symmetry, unpack the classifier branch. The two endpoint carrier witnesses
are kept in the same visible tag, and symmetry of $\hsame$ reverses the payload
comparison. Transitivity composes two same-tag classifier witnesses by composing
their payload $\hsame$ links through the shared middle payload.

For carrier transport, suppose $\mathsf{IntHistoryCarrier}(h)$ and
$\mathsf{IntHistoryClassifier}(h,k)$. In a same-tag branch, compose the
classifier endpoint witness for $k$ with the stored unary payload witness, then
repack the corresponding $\Ezero$ or $\Eone$ carrier branch.

The semantic pattern and ledger fields are the carrier itself, so both
soundness obligations are identity implications from a transported carrier
witness. Mixed visible tags are impossible because any mixed classifier branch
would compose the visible endpoint witnesses into an
$\hsame(\Ezero(m'),\Eone(n'))$ or
$\hsame(\Eone(m'),\Ezero(n'))$ judgment, contradicting BEDC constructor
no-confusion.
\end{proof}
\leanchecked{BEDC.Derived.IntUp.IntHistory\_semanticNameCert}

\begin{theorem}[Integer history classifier same-tag readback]
\label{thm:int-history-classifier-same-tag-readback}
For all $m,n:\mathsf{BHist}$, the public integer history classifier reads back
exactly payload equality on each visible signed tag:
$$
\mathsf{IntHistoryClassifier}(\Ezero(m),\Ezero(n))
\Longleftrightarrow
\mathsf{Unary}(m)\land\mathsf{Unary}(n)\land\hsame(m,n).
$$
The one-tag branch has the same payload criterion:
$$
\mathsf{IntHistoryClassifier}(\Eone(m),\Eone(n))
\Longleftrightarrow
\mathsf{Unary}(m)\land\mathsf{Unary}(n)\land\hsame(m,n).
$$
The ten-field fit is the following. The traditional target is same-sign
equality of signed integers. The BEDC source consists of
$\mathsf{IntHistoryCarrier}$, $\mathsf{IntHistoryClassifier}$, $\hsame$, and
constructor no-confusion. The carrier is the same-tag unary-history carrier.
The classifier is the displayed two-branch definition. Stability is given by
transport along the payload $\hsame$. The ledger records the sign tag together
with the unary magnitude. The core theorem is this same-tag readback statement.
Exactness is the two displayed iff rows. The downstream consumer is the
$\RatUp$ integer numerator readback. The status is $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Unfold $\mathsf{IntHistoryClassifier}$ at same-tag endpoints.

For the zero-tag forward direction, keep the matching $\Ezero$ branch. The
endpoint witnesses have the form $\hsame(\Ezero(m),\Ezero(m'))$ and
$\hsame(\Ezero(n),\Ezero(n'))$, so constructor inversion reads back
$\hsame(m,m')$ and $\hsame(n,n')$. Transitivity with the stored payload
$\hsame(m',n')$ yields $\hsame(m,n)$, and the same branch supplies
$\mathsf{Unary}(m)$ and $\mathsf{Unary}(n)$ after the endpoint readback. The
opposite $\Eone$ branch would force an $\Ezero$ endpoint to be $\hsame$ to an
$\Eone$ endpoint, contradicting constructor no-confusion.

For the zero-tag reverse direction, assume
$\mathsf{Unary}(m)\land\mathsf{Unary}(n)\land\hsame(m,n)$. Choose the
$\Ezero$ branch with payload witnesses $m$ and $n$, use reflexivity for the
two visible endpoint witnesses, and use the supplied payload $\hsame(m,n)$.

The one-tag forward direction is identical after replacing $\Ezero$ by
$\Eone$. The matching branch gives endpoint inversion and payload transport;
the mixed branch contradicts $\Eone$/$\Ezero$ no-confusion.

For the one-tag reverse direction, choose the $\Eone$ branch with payload
witnesses $m$ and $n$, reflexive visible endpoints, the two unary witnesses,
and the supplied payload $\hsame(m,n)$.
\end{proof}
\leanchecked{BEDC.Derived.IntUp.IntHistoryClassifier\_same\_tag\_readback\_iff}
