\chapter{A Concrete Naming Certificate for $\FieldExtUp$}
\label{ch:concrete-instances-fieldext-namecert}

The field-extension interface $\FieldExtUp$ packages an embedding of one field carrier into another, recording the larger field as a vector space over the smaller. It depends on $\NameCert_{\FieldUp}$ (\autoref{ch:concrete-instances-field-namecert}) and $\NameCert_{\VecSpaceUp}$ (\autoref{ch:concrete-instances-vecspace-namecert}), and is the substrate for Galois, separable, and algebraic-closure certificates.

\section{Carrier definition}
\label{sec:fieldext-carrier}

\begin{definition}[Concrete singleton empty-history field extension instance]
\label{def:fieldext-singleton-empty-history-instance}
Let the base field carrier and extension field carrier both be
$\mathsf{FieldSingletonCarrier}$ over $\mathsf{BHist}$, with field classifier
$\mathsf{FieldSingletonClassifier}$. Equivalently, this is the singleton
empty-history carrier $A_{\emp}$ with shared classifier
$$
\rho_{\emp}(h,k):=
A_{\emp}(h)\land A_{\emp}(k)\land\hsame(h,k).
$$
Both field endpoints use the singleton $\FieldUp$ operations of
Definition~\ref{def:singleton-empty-history-field-schema-instance} for both
the base and extension fields:
$\mathsf{FieldSingletonAdd}$, $\mathsf{FieldSingletonMul}$,
$\mathsf{FieldSingletonZero}$, $\mathsf{FieldSingletonOne}$,
$\mathsf{FieldSingletonNeg}$, and the proof-indexed inverse
$\mathsf{FieldSingletonInv}$. Define the embedding by
$\iota_{\emp}(h):=h$ on $\mathsf{BHist}$.

Read the extension as the singleton $\VecSpaceUp$ carrier
$\mathsf{VecSpaceSingletonCarrier}$ over the singleton scalar field. Its
vector classifier is $\mathsf{VecSpaceSingletonClassifier}$ and its scalar
action is $\mathsf{VecSpaceSingletonSmul}$. The ledger is inherited from the
singleton $\FieldUp$ certificate on the two field sides and from the singleton
$\VecSpaceUp$ certificate on the vector-space-over-base side.

The certificate fit has ten fields:
\begin{enumerate}
  \item Traditional target: the terminal field extension.
  \item BEDC source: $\mathsf{BHist}$.
  \item Carrier choice: $A_{\emp}$ for the base field, extension field, and
  vector carrier.
  \item Classifier choice: $\rho_{\emp}$ on all displayed endpoints.
  \item Stability obligations: the singleton field laws together with the
  singleton vector-space laws.
  \item Ledger: the identity embedding and vector-space provenance over the
  same empty-history endpoint.
  \item Core theorem list:
  Theorems~\ref{thm:fieldext-singleton-embedding-obligations},
  \ref{thm:fieldext-singleton-embedding-exactness},
  \ref{thm:fieldext-singleton-vector-space-over-base},
  \ref{thm:fieldext-singleton-embedded-scalar-action-collapse},
  \ref{thm:fieldext-singleton-carrier-coincidence},
  \ref{thm:fieldext-singleton-identity-tower-composition},
  \ref{thm:fieldext-singleton-exact-endpoint-ledger},
  \ref{thm:fieldext-singleton-exact-endpoint-classification},
  \ref{thm:fieldext-singleton-ledger-provenance}, and
  \ref{thm:fieldext-singleton-certificate-obligation-package}.
  \item Exactness predicate: singleton endpoint collapse.
  \item Standard bridge: the terminal field extension.
  \item Status: $\mathsf{ProofTarget}$.
\end{enumerate}
\end{definition}

\begin{definition}[Singleton FieldExt obligation scope]
\label{def:fieldext-singleton-obligation-scope}
The terminal $\FieldExtUp$ instance over $A_{\emp}$ is the following ten-field
paper template. The traditional target is the one-point field extension. The BEDC
source is $\mathsf{BHist}$. The base field carrier, extension field carrier, and
extension vector carrier are all $A_{\emp}$. The classifier is the shared
singleton relation $\rho_{\emp}$ from
Definition~\ref{def:fieldext-singleton-empty-history-instance}. The embedding is the
identity map $\iota_{\emp}(h):=h$. The extension-field vector-space-over-base
action is the singleton vector-space action over the singleton $\FieldUp$ source.

The stability obligations are the singleton field rows of
Theorem~\ref{thm:singleton-empty-history-field-schema-laws} for the base and
extension fields, together with the singleton vector-space rows of
Theorem~\ref{thm:singleton-empty-history-vecspace-laws}. The ledger consists of
identity-embedding provenance and singleton scalar-action provenance over the
same empty-history endpoint. The downstream consumers are the Galois,
number-field, and algebraic-closure certificate families, each of which reads the
same carrier, classifier, exactness, scalar-action, and ledger fields.
\end{definition}

\begin{definition}[RatUp reflexive field-extension carrier]

Let the base field carrier and the ambient field carrier both be the concrete

\autoref{def:ratup-fieldup-concrete-source-tuple}:
$$
C_{\mathsf{base}}=C_{\mathsf{amb}}=\mathsf{RatHistoryCarrier},
\qquad
R_{\mathsf{base}}=R_{\mathsf{amb}}=\mathsf{RatHistoryClassifier}.
$$
The classifier for the extension package is the RatUp classifier on the same
$\mathsf{BHist}$ endpoints. The embedding is the identity map
$\iota_{\mathsf{Rat}}(h):=h$ on $\mathsf{BHist}$.

The ambient field is also used as its own vector carrier over the base field.

\autoref{thm:ratup-fieldup-assembled-certificate-boundary}, and vector

displayed object is the reflexive field extension
$$
\RatUp \hookrightarrow \RatUp
$$
with no change of BEDC carrier, classifier, ledger source, or operation table.
\end{definition}

\begin{theorem}[Singleton field-extension embedding obligations]
\label{thm:fieldext-singleton-embedding-obligations}
For the concrete data of
Definition~\ref{def:fieldext-singleton-empty-history-instance}, let
$h,k:\mathsf{BHist}$ be carried by $\mathsf{FieldSingletonCarrier}$. Then
$\iota_{\emp}(h)$ is carried by the extension field,
$$
\mathsf{FieldSingletonClassifier}(h,k)\quad\text{if and only if}\quad
\mathsf{FieldSingletonClassifier}(\iota_{\emp}(h),\iota_{\emp}(k)),
$$
and the embedding preserves the concrete singleton zero, one, addition,
negation, multiplication, and proof-indexed inverse up to $\rho_{\emp}$.
Thus the base endpoint and its embedded extension endpoint have the same
singleton classification for each displayed field operation.
\end{theorem}
\leanchecked{BEDC.Derived.FieldExtUp.FieldExtSingletonEmbedding\_laws}
\begin{proof}
Unfold $\iota_{\emp}$. The carrier statement is immediate from the assumption
$A_{\emp}(h)$, because the base and extension field carriers are the same
singleton carrier $A_{\emp}$.

The classifier comparison has the same two directions. After unfolding the
identity embedding, both sides are the identical classifier assertion
$\rho_{\emp}(h,k)$.

For zero, one, addition, negation, and multiplication, apply
Theorem~\ref{thm:singleton-empty-history-field-schema-laws} to obtain the
field-law rows over $\mathsf{FieldSingletonCarrier}$ and
$\mathsf{FieldSingletonClassifier}$. Then use
Theorem~\ref{thm:field-singleton-concrete-operation-table-collapse} to read
each concrete operation endpoint as the singleton empty-history endpoint.

For the inverse rows, fix a proof-indexed witness in the singleton nonzero
domain. The same operation-table collapse theorem gives the inverse endpoint
row for $\mathsf{FieldSingletonInv}(h,p)$, and the law package uses that same
proof-indexed domain. Hence base inverse and embedded extension inverse are
classified by $\rho_{\emp}$ at the same singleton endpoint.
\end{proof}

\begin{theorem}[Singleton FieldExt embedding exactness]
\label{thm:fieldext-singleton-embedding-exactness}
For the concrete data of
Definition~\ref{def:fieldext-singleton-empty-history-instance}, let
$h,k:\mathsf{BHist}$ be carried by $A_{\emp}$. Then
$$
\rho_{\emp}(h,k)\quad\text{if and only if}\quad
\rho_{\emp}(\iota_{\emp}(h),\iota_{\emp}(k)).
$$
Moreover, for every displayed singleton field operation in the base package, its
embedded extension endpoint is classified with the corresponding source endpoint
by $\rho_{\emp}$.
\end{theorem}
\begin{proof}
Unfold $\iota_{\emp}$. The displayed equivalence is the classifier comparison
recorded in Theorem~\ref{thm:fieldext-singleton-embedding-obligations}; after the
identity embedding is unfolded, both sides are the same singleton classifier
assertion.

For the operation endpoints, apply
Theorem~\ref{thm:fieldext-singleton-embedding-obligations} to the carried inputs. It
gives preservation of zero, one, addition, negation, multiplication, and the
proof-indexed inverse up to $\rho_{\emp}$.

The concrete endpoint identification comes from the singleton operation-table
collapse in Theorem~\ref{thm:field-singleton-concrete-operation-table-collapse}.
Thus each embedded operation row lands in the same $\rho_{\emp}$ class as its
source row.
\end{proof}

\begin{theorem}[RatUp reflexive embedding transport]
\label{thm:fieldext-reflexive-embedding-transport}
For the concrete data of
Definition~\ref{def:fieldext-reflexive-extension-carrier}, the identity
embedding $\iota_{\mathsf{Rat}}$ transports carrier membership and classifier
comparisons exactly. If $h,k:\mathsf{BHist}$, then
$\mathsf{RatHistoryCarrier}(h)$ gives
$\mathsf{RatHistoryCarrier}(\iota_{\mathsf{Rat}}(h))$, and
$\mathsf{RatHistoryClassifier}(h,k)$ gives and is given by
$\mathsf{RatHistoryClassifier}(\iota_{\mathsf{Rat}}(h),\iota_{\mathsf{Rat}}(k))$. The same
identity transport preserves the zero, one, addition, negation,
multiplication, apartness, and proof-indexed inverse rows of the
$\FieldUp(\RatUp)$ certificate.
\end{theorem}
\begin{proof}
Unfold $\iota_{\mathsf{Rat}}$. Carrier preservation is immediate because the base and
ambient carriers are both $\mathsf{RatHistoryCarrier}$.

The classifier transport has identical source and target assertions after
unfolding the identity map. Reflexivity, symmetry, transitivity, and endpoint
carrier extraction are the RatUp classifier fields recorded in
\autoref{thm:rat-history-semantic-name-certificate}.

For the operation rows, read the base and ambient field packages from
\autoref{thm:ratup-fieldup-assembled-certificate-boundary}. They use the same
carrier, classifier, denominator tuple, contextual action, and affine source.
Since the embedding is identity, every displayed FieldUp operation endpoint is
compared with itself under $\mathsf{RatHistoryClassifier}$, including the
proof-indexed inverse endpoint whenever the apartness witness is supplied.
\end{proof}

\begin{theorem}[RatUp reflexive vector-space package]
\label{thm:fieldext-reflexive-vector-space-package}
For the concrete data of
Definition~\ref{def:fieldext-reflexive-extension-carrier}, the ambient
$\RatUp$ carrier is a vector space over the base $\RatUp$ carrier. Its vector
addition, zero, and negation are the FieldUp additive rows, and its scalar
action is FieldUp multiplication. The vector carrier classifier is
$\mathsf{RatHistoryClassifier}$, the scalar classifier is the same RatUp
classifier, and scalar-action congruence is the multiplication congruence row
of the concrete $\FieldUp(\RatUp)$ certificate.
\end{theorem}
\begin{proof}
Start from the concrete field certificate
\autoref{thm:ratup-fieldup-assembled-certificate-boundary}. Its carrier and
classifier are $\mathsf{RatHistoryCarrier}$ and
$\mathsf{RatHistoryClassifier}$, and its algebraic rows include additive
abelian-group laws, multiplication closure, distributivity, unit laws, and
classifier congruence for multiplication.

Use the field multiplication row as scalar action. Associativity of scalar
action is exactly associativity of FieldUp multiplication. Left and right
distributivity of scalar action are the two distributivity rows of the same
FieldUp certificate, read with the first factor as scalar and the second
factor as vector.

The vector addition, zero, and negation rows are the additive rows already
present in the field certificate. The projection principles
\autoref{prop:vecspace-module-fragment-projection} and
\autoref{cor:vecspace-additive-carrier-projection} identify these as the
module-shaped and additive carrier rows expected by $\VecSpaceUp$ once the
scalar field is fixed. Therefore the self-action package supplies the
paper-side vector-space obligations over the unchanged RatUp carrier.
\end{proof}

Embedding stability is
\autoref{thm:fieldext-reflexive-embedding-transport}. It records carrier

The vector-space component is
\autoref{thm:fieldext-reflexive-vector-space-package}. It uses FieldUp

The ledger clauses do not introduce a second provenance source. Every scalar,
vector, embedding, and operation endpoint is read from the same
$\mathsf{RatUpFieldSource}_F$ tuple of
\autoref{def:ratup-fieldup-concrete-source-tuple}. The exactness clauses are

\end{proof}

\label{thm:fieldext-ratup-reflexive-operation-table-coverage}

\begin{theorem}[Singleton field-extension carrier coincidence]
\label{thm:fieldext-singleton-carrier-coincidence}

Theorem~\ref{thm:fieldext-singleton-embedding-obligations} and
Theorem~\ref{thm:fieldext-singleton-vector-space-over-base}. Each row is
displayed with the same concrete predicate $\rho_{\emp}$, so no transport

\end{proof}

composition introduces no field endpoint beyond the empty-history endpoint.

\begin{theorem}[Singleton field-extension exact endpoint classification]
\label{thm:fieldext-singleton-exact-endpoint-classification}
For the concrete data of
Definition~\ref{def:fieldext-singleton-empty-history-instance}, every displayed
base-field operation endpoint, every displayed extension-field operation
endpoint, the vector scalar-action endpoint, and the identity embedding
endpoint are carried by $A_{\emp}$ and classified with $\emp$ by
$\rho_{\emp}$. In particular, the operation table of the base field, the

\begin{proof}

Theorem~\ref{thm:field-singleton-concrete-operation-table-collapse}. Both
field structures in

singleton $\FieldUp$ operation table.

For the vector scalar-action endpoint, apply
Theorem~\ref{thm:singleton-empty-history-vecspace-endpoint-collapse}. Its
scalar-action clause reads the endpoint as $\emp$ over the same
$A_{\emp}$ and $\rho_{\emp}$.

The embedding endpoint is immediate from $\iota_{\emp}(h):=h$. If $h$ is
carried by $A_{\emp}$, then the singleton carrier identifies $h$ with

\end{proof}

carrier, classifier, and singleton operation endpoints.

The vector-space provenance row is supplied by
Theorem~\ref{thm:singleton-empty-history-vecspace-semantic-certificate}. It
uses the same scalar source, vector source, classifier, and empty-history
scalar-action endpoint as the displayed $\FieldExtUp$ data.

The embedding row is supplied by
Theorem~\ref{thm:fieldext-singleton-embedding-obligations}. Since

\begin{theorem}[Singleton field-extension certificate projections]
\label{thm:fieldext-singleton-certificate-projections}

  \item its vector-space projection is the singleton $\VecSpaceUp$-over-base

The embedding row connecting the first two projections is the identity map
$\iota_{\emp}(h):=h$.
\end{theorem}
\begin{proof}
The source and extension projections use the same displayed singleton field
data. Apply
Theorem~\ref{thm:singleton-empty-history-field-schema-certificate} once for

The vector-space projection is the semantic singleton vector-space package.
Apply Theorem~\ref{thm:singleton-empty-history-vecspace-semantic-certificate}
with the scalar carrier, vector carrier, and classifier specified in
Definition~\ref{def:fieldext-singleton-empty-history-instance}.

The projection map from the source field into the extension field is
$\iota_{\emp}$. By its definition, it is the identity on histories, so it
preserves the displayed carrier and classifier rows without introducing a
second endpoint choice.
\end{proof}

\begin{theorem}[Singleton field-extension operation readback exactness]
\label{thm:fieldext-singleton-operation-readback-exactness}
For the concrete data of
Definition~\ref{def:fieldext-singleton-empty-history-instance}, every zero, one,
addition, negation, multiplication, proof-indexed inverse, and scalar-action
endpoint in the singleton field-extension package reads back to the singleton

against the corresponding embedded source-field endpoint or extension-field

\end{theorem}
\begin{proof}
For the field operations, apply
Theorem~\ref{thm:field-singleton-concrete-operation-table-collapse}. It
collapses zero, one, addition, negation, multiplication, and the
proof-indexed inverse endpoint to the singleton empty-history endpoint.

The embedding comparison is supplied by
Theorem~\ref{thm:fieldext-singleton-embedding-obligations}. Since the embedding is
the identity on histories, the embedded source endpoint and the corresponding
extension-field endpoint are compared by the same classifier
$\rho_{\emp}$.

For scalar action, apply
Theorem~\ref{thm:singleton-empty-history-vecspace-endpoint-collapse} to read
the vector-space action endpoint as the singleton empty-history endpoint.
Then apply
Theorem~\ref{thm:fieldext-singleton-vector-space-over-base} to identify
that scalar-action endpoint with the matching extension-field multiplication

\end{proof}

\begin{theorem}[Singleton field-extension exactness and ledger coverage]
\label{thm:fieldext-singleton-exactness-ledger-coverage}
For the concrete data of
Definition~\ref{def:fieldext-singleton-empty-history-instance}, every
$\FieldExtUp$ source row, pattern row, classifier row, stability row,
exactness row, and ledger row is covered by exactly one of the following
provenance classes: the identity embedding row, the singleton field operation
table row, or the singleton vector-space scalar-action row.
\end{theorem}
\begin{proof}
First take the projected certificate rows from
Theorem~\ref{thm:fieldext-singleton-certificate-projections}. These account
for the source-field, extension-field, and vector-space certificate surfaces
over the common $A_{\emp}$ and $\rho_{\emp}$.

Next use
Theorem~\ref{thm:fieldext-singleton-operation-readback-exactness}. It assigns
the field operation endpoints to the singleton field operation table row and
the scalar-action endpoints to the singleton vector-space scalar-action row,
with all endpoint comparisons made by $\rho_{\emp}$.

It remains to account for the cross-carrier rows. The identity embedding row
from Definition~\ref{def:fieldext-singleton-empty-history-instance} and
Theorem~\ref{thm:fieldext-singleton-embedding-obligations} covers carrier
preservation, classifier preservation, and embedded source-operation
stability.

Finally apply
Theorem~\ref{thm:fieldext-singleton-certificate-obligation-package}. Its assembled
$\FieldExtUp$ $\NameCert$ obligation list is precisely the union of these
three provenance classes, so the source, pattern, classifier, stability,
exactness, and ledger rows are all covered.
\end{proof}
