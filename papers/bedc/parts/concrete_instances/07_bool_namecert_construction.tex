\chapter{A Concrete Naming Certificate for \(\BoolUp\)}
\label{ch:concrete-instances-bool-namecert}

The boolean interface \(\BoolUp\) is licensed by raw mark sameness on the two-element basis: \(\bzero\) read as false, \(\bone\) read as true. It is not a primitive proposition or truth-value type; it is licensed only via the underlying \(\Mark\) structure together with a classifier specification that distinguishes the two marks under \(\msame\).

\section{Roadmap template status}

\begin{enumerate}

  \item \textbf{BEDC source}: the finite mark kernel $\Mark$, with $\bzero$ and $\bone$ as the displayed points.
  \item \textbf{Carrier choice}: the mark-level carrier is $\Mark$ itself; the history carrier is $\mathsf{BoolHistoryCarrier}$.

  for $\BoolUp$, with status $\mathsf{ProofTarget}$. At this site the public-object scope is the displayed paper-side $\mathsf{SemanticNameCert}$ theorem, while the checked Lean target is only the core history $\NameCert$, Theorem~\ref{thm:concrete-bool-namecert}, via the $BEDC.Derived.BoolUp$ Lean target $\mathsf{bool\_history\_name\_certificate}$. \section{The certificate}

\begin{definition}[Bool source specification]
\label{def:bool-source-specification}
The source specification for \(\BoolUp\) is the mark-level classifier generated by internal sameness to one of the two finite-kernel marks:
$$
  \mathsf{BoolSourceSpec}(v)
  :=
  \msame(v,\bzero)\lor\msame(v,\bone).
$$
\end{definition}
\leandef{BEDC.Derived.BoolUp.BoolSourceSpec}
\leandef{BEDC.Derived.BoolUp.BoolHistoryCarrier}

\begin{definition}[Bool endpoint bridge]
\label{def:bool-endpoint-bridge}
The endpoint map $\mathsf{BoolEndpoint}:\mathsf{BMark}\to\mathsf{BHist}$ sends the two traditional boolean points from the BEDC mark source into the displayed history endpoints:
$$
  \mathsf{BoolEndpoint}(\bzero)=\emp,\qquad
  \mathsf{BoolEndpoint}(\bone)=\Eone(\emp).
$$
The carrier endpoint choice is therefore $\mathsf{BoolHistoryCarrier}$, and the classifier transport target is $\hsame$ on the two displayed history endpoints.
\end{definition}

\begin{theorem}[Bool endpoint bridge exactness]
\label{thm:bool-endpoint-bridge-exactness}
For marks $v,w:\mathsf{BMark}$, mark sameness holds exactly when the displayed history endpoints are internally the same:
$$
  \msame(v,w)
  \quad\Longleftrightarrow\quad
  \hsame(\mathsf{BoolEndpoint}(v),\mathsf{BoolEndpoint}(w)).
$$
\end{theorem}
\begin{proof}
Proceed by cases on $v$ and $w$. The matching $\bzero/\bzero$ and $\bone/\bone$ cases reduce both directions to reflexivity. In the mixed cases, the mark-to-history direction has no $\msame$ witness by mark no-confusion, and the history-to-mark direction has no $\hsame$ witness by the no-confusion between $\emp$ and $\Eone(\emp)$.
\end{proof}

\begin{theorem}[Bool history carrier endpoint coverage]
\label{thm:bool-history-carrier-endpoint-coverage}
For every history $h:\mathsf{BHist}$, the checked boolean history carrier is exactly the image of the endpoint bridge:
$$
  \mathsf{BoolHistoryCarrier}(h)
  \quad\Longleftrightarrow\quad
  \exists v:\mathsf{BMark},\ \hsame(h,\mathsf{BoolEndpoint}(v)).
$$
\end{theorem}
\begin{proof}
In the forward direction, split the carrier witness into its empty and one-endpoint branches. In the empty branch choose $v=\bzero$ and use the defining endpoint equality $\mathsf{BoolEndpoint}(\bzero)=\emp$; in the one-endpoint branch choose $v=\bone$ and use $\mathsf{BoolEndpoint}(\bone)=\Eone(\emp)$.

In the reverse direction, case-analyze the displayed mark $v$. The $\bzero$ case rewrites the endpoint witness to an $\hsame$ witness against $\emp$ and gives the empty carrier branch. The $\bone$ case rewrites it to an $\hsame$ witness against $\Eone(\emp)$ and gives the one-endpoint carrier branch.
\end{proof}

\begin{theorem}[Bool endpoint readback determinism]
\label{thm:bool-endpoint-readback-determinism}
For marks $v,w:\mathsf{BMark}$ and history $h:\mathsf{BHist}$, if
$\hsame(h,\mathsf{BoolEndpoint}(v))$ and
$\hsame(h,\mathsf{BoolEndpoint}(w))$, then $\msame(v,w)$.
\end{theorem}
\begin{proof}
Compose symmetry of $\hsame(h,\mathsf{BoolEndpoint}(v))$ with
$\hsame(h,\mathsf{BoolEndpoint}(w))$ to obtain
$\hsame(\mathsf{BoolEndpoint}(v),\mathsf{BoolEndpoint}(w))$. The reverse
direction of Theorem~\ref{thm:bool-endpoint-bridge-exactness} converts this
displayed endpoint sameness into $\msame(v,w)$.
\end{proof}

\begin{theorem}[Bool endpoint readback total uniqueness]
\label{thm:bool-endpoint-readback-total-unique}
For every history $h:\mathsf{BHist}$, if $\mathsf{BoolHistoryCarrier}(h)$
holds, then there exists a mark $v:\mathsf{BMark}$ such that
$$
  \hsame(h,\mathsf{BoolEndpoint}(v))
  \quad\text{and}\quad
  \mathsf{BoolSourceSpec}(v)
  \quad\text{and}\quad
  \forall w:\mathsf{BMark},\
    \hsame(h,\mathsf{BoolEndpoint}(w)) \to \msame(v,w).
$$
\end{theorem}
\begin{proof}
Use Theorem~\ref{thm:bool-history-carrier-endpoint-coverage} on the carrier
witness for $h$ to obtain a displayed endpoint $v$ and a readback witness
$\hsame(h,\mathsf{BoolEndpoint}(v))$. Case analysis on $v$ gives
$\mathsf{BoolSourceSpec}(v)$ for both boolean marks. For any other displayed
readback $w$, Theorem~\ref{thm:bool-endpoint-readback-determinism} applied to
the two endpoint witnesses gives $\msame(v,w)$.
\end{proof}

\begin{theorem}[Bool classifier readback total uniqueness]
\label{thm:bool-classifier-readback-total-unique}
For every history $h:\mathsf{BHist}$, if $\mathsf{BoolHistoryCarrier}(h)$
holds, then there exists a mark $v:\mathsf{BMark}$ such that
$$
  \mathsf{BoolSourceSpec}(v)
  \quad\land\quad
  \mathsf{BoolHistoryClassifier}(h,\mathsf{BoolEndpoint}(v))
  \quad\land\quad
  \forall w:\mathsf{BMark},\
    \mathsf{BoolHistoryClassifier}(h,\mathsf{BoolEndpoint}(w)) \to \msame(v,w).
$$
The ten-field fit is: traditional target public boolean readback; BEDC source
$\mathsf{BHist}$ with $\hsame$; carrier $\mathsf{BoolHistoryCarrier}$;
classifier $\mathsf{BoolHistoryClassifier}$; stability obligations endpoint
coverage and classifier determinacy; ledger displayed
$\mathsf{BoolEndpoint}(v)$; core theorem this total uniqueness statement;
exactness classifier-visible existence and uniqueness up to $\msame$; standard
bridge ordinary booleans through $\bzero,\bone$ endpoints; status
$\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:bool-endpoint-bridge-classifier-completeness} to the
carrier witness for $h$. This gives a mark $v$ with
$\mathsf{BoolSourceSpec}(v)$ and
$\mathsf{BoolHistoryClassifier}(h,\mathsf{BoolEndpoint}(v))$. For uniqueness,
unpack this classifier witness and any displayed classifier readback at
$\mathsf{BoolEndpoint}(w)$ to expose endpoint witnesses
$\hsame(h,\mathsf{BoolEndpoint}(v))$ and
$\hsame(h,\mathsf{BoolEndpoint}(w))$. Theorem~\ref{thm:bool-endpoint-readback-determinism}
then gives $\msame(v,w)$.
\end{proof}

\begin{theorem}[Bool endpoint bridge classifier completeness]
\label{thm:bool-endpoint-bridge-classifier-completeness}
The endpoint bridge preserves the checked boolean history classifier in both finite certificate directions. For marks $v,w:\mathsf{BMark}$,
$$
  \mathsf{BoolHistoryClassifier}(\mathsf{BoolEndpoint}(v),\mathsf{BoolEndpoint}(w))
  \quad\Longleftrightarrow\quad
  \msame(v,w).
$$
Moreover, for every history $h:\mathsf{BHist}$ satisfying $\mathsf{BoolHistoryCarrier}(h)$, there exists a mark $v:\mathsf{BMark}$ such that both $\mathsf{BoolSourceSpec}(v)$ and $\mathsf{BoolHistoryClassifier}(h,\mathsf{BoolEndpoint}(v))$ hold.
\end{theorem}
\begin{proof}
For the first clause, unfold the classifier at the two displayed endpoints. The endpoint carrier fields are supplied by the two endpoint cases, so the classifier is equivalent to history sameness between the two endpoint images. The endpoint bridge exactness theorem then identifies this condition with $\msame(v,w)$.

For the second clause, apply carrier endpoint coverage to the carrier witness for $h$ and obtain an endpoint $v$ with $\hsame(h,\mathsf{BoolEndpoint}(v))$. Case analysis on $v$ supplies $\mathsf{BoolSourceSpec}(v)$, since the two mark constructors satisfy the source specification by reflexivity. The same history-sameness witness, together with the carrier witness for $h$ and the endpoint carrier witness, is exactly the data required by the definition of the displayed classifier relation.
\end{proof}

\begin{theorem}[Bool history classifier endpoint readback]
\label{thm:bool-history-classifier-endpoint-readback}
For histories $h,k:\mathsf{BHist}$, if
$\mathsf{BoolHistoryClassifier}(h,k)$ holds, then there exist marks
$v,w:\mathsf{BMark}$ such that
$$
  \hsame(h,\mathsf{BoolEndpoint}(v))
  \quad\land\quad
  \hsame(k,\mathsf{BoolEndpoint}(w))
  \quad\land\quad
  \msame(v,w).
$$
\end{theorem}
\begin{proof}
Unpack the classifier witness into the two $\mathsf{BoolHistoryCarrier}$
endpoint witnesses and the stored $\hsame(h,k)$ link. Apply
Theorem~\ref{thm:bool-history-carrier-endpoint-coverage} to both carrier
witnesses, obtaining marks $v$ and $w$ with endpoint readbacks from $h$ and
$k$. Compose the readback for $h$, the stored $\hsame(h,k)$ witness, and the
readback for $k$ to obtain
$\hsame(\mathsf{BoolEndpoint}(v),\mathsf{BoolEndpoint}(w))$. The reverse
direction of Theorem~\ref{thm:bool-endpoint-bridge-exactness} converts this
endpoint sameness into $\msame(v,w)$.
\end{proof}

\begin{theorem}[Bool history classifier mark-readback exactness]
\label{thm:bool-history-classifier-mark-readback-exactness}
For histories $h,k:\mathsf{BHist}$,
$\mathsf{BoolHistoryClassifier}(h,k)$ holds if and only if there exist marks
$v,w:\mathsf{BMark}$ satisfying
$$
  \hsame(h,\mathsf{BoolEndpoint}(v))
  \land
  \hsame(k,\mathsf{BoolEndpoint}(w))
  \land
  \msame(v,w).
$$
\end{theorem}
\begin{proof}
The forward direction is exactly
Theorem~\ref{thm:bool-history-classifier-endpoint-readback}. For the reverse
direction, take displayed marks $v,w$ with the two endpoint readbacks and
$\msame(v,w)$. Endpoint carrier coverage gives
$\mathsf{BoolHistoryCarrier}(\mathsf{BoolEndpoint}(v))$ and
$\mathsf{BoolHistoryCarrier}(\mathsf{BoolEndpoint}(w))$. The forward direction
of Theorem~\ref{thm:bool-endpoint-bridge-exactness} reads $\msame(v,w)$ as
$\hsame(\mathsf{BoolEndpoint}(v),\mathsf{BoolEndpoint}(w))$, hence the displayed
endpoints satisfy $\mathsf{BoolHistoryClassifier}$. Finally apply endpoint
transport for $\mathsf{BoolHistoryClassifier}$ along the two given $\hsame$
readbacks to obtain $\mathsf{BoolHistoryClassifier}(h,k)$.
\end{proof}

\begin{theorem}[Bool source-qualified classifier readback exactness]
\label{thm:bool-source-qualified-classifier-readback-exactness}
For histories $h,k:\mathsf{BHist}$,
$\mathsf{BoolHistoryClassifier}(h,k)$ holds if and only if there exist marks
$v,w:\mathsf{BMark}$ satisfying
$$
  \mathsf{BoolSourceSpec}(v)
  \land
  \mathsf{BoolSourceSpec}(w)
  \land
  \hsame(h,\mathsf{BoolEndpoint}(v))
  \land
  \hsame(k,\mathsf{BoolEndpoint}(w))
  \land
  \msame(v,w).
$$
The traditional target is the two-point boolean readback; the BEDC source is
$\mathsf{BMark}$ and $\mathsf{BHist}$, the carrier is
$\mathsf{BoolHistoryCarrier}$, the classifier is
$\mathsf{BoolHistoryClassifier}$, the exactness clause is source-qualified
endpoint readback, and the status is $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
For the forward direction, apply
Theorem~\ref{thm:bool-history-classifier-mark-readback-exactness} to obtain
endpoint marks $v,w$ with the two $\hsame$ readbacks and $\msame(v,w)$. Case
analysis on the two displayed marks supplies $\mathsf{BoolSourceSpec}(v)$ and
$\mathsf{BoolSourceSpec}(w)$, since the two constructors are exactly the
allowed source marks. For the reverse direction, discard the two
$\mathsf{BoolSourceSpec}$ witnesses and apply the reverse direction of
Theorem~\ref{thm:bool-history-classifier-mark-readback-exactness}.
\end{proof}

\begin{theorem}[Bool source respects mark sameness]
\label{thm:bool-source-spec-respects-msame}
If \(v\) satisfies \(\mathsf{BoolSourceSpec}\) and \(\msame(v,w)\), then \(w\) also satisfies \(\mathsf{BoolSourceSpec}\).
\end{theorem}
\begin{proof}
The source condition is a disjunction of \(\msame(v,\bzero)\) and \(\msame(v,\bone)\). In either case, symmetry gives \(\msame(w,v)\), and transitivity transports the corresponding constructor witness from \(v\) to \(w\).
\end{proof}

the right-endpoint classifier witness.
