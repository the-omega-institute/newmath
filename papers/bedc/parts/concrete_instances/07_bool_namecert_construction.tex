\chapter{A Concrete Naming Certificate for \(\BoolUp\)}
\label{ch:concrete-instances-bool-namecert}

The boolean interface \(\BoolUp\) is licensed by raw mark sameness on the two-element basis: \(\bzero\) read as false, \(\bone\) read as true. It is not a primitive proposition or truth-value type; it is licensed only via the underlying \(\Mark\) structure together with a classifier specification that distinguishes the two marks under \(\msame\).

\section{Roadmap template status}

\begin{enumerate}
  \item \textbf{Traditional target}: the two-point boolean object with false and true.
  \item \textbf{BEDC source}: the finite mark kernel $\Mark$, with $\bzero$ and $\bone$ as the displayed points.
  \item \textbf{Carrier choice}: the mark-level carrier is $\Mark$ itself; the history carrier is $\mathsf{BoolHistoryCarrier}$.
  \item \textbf{Classifier choice}: the classifier is $\msame$ on marks and $\mathsf{BoolHistoryClassifier}$ on the two licensed histories.
  \item \textbf{Stability obligations}: mark transport, history-carrier transport, unary closure, endpoint exclusion, classifier symmetry, and classifier transitivity.
  \item \textbf{Ledger}: no quotient is used; the hidden data are only the generated history witnesses for the two boolean endpoints.
  \item \textbf{Core theorem list}: source transport, history-carrier transport, history classifier equivalence, constructor exclusions, and the history core $\NameCert$.
  \item \textbf{Exactness predicate}: the two endpoints are exactly the $\bzero$ and $\bone$ branches licensed by $\msame$.
  \item \textbf{Standard bridge}: false and true are read back as the two finite-kernel marks.
  \item \textbf{Status}: $\mathsf{CheckedCert}$ for the boolean history core $\NameCert$ checked in Lean; the carrier-as-pattern semantic certificate target remains $\mathsf{ProofTarget}$.
\end{enumerate}

Use the existing two-constructor mark carrier \(\Mark\) with \(\bzero\) interpreted as false and \(\bone\) interpreted as true.
\end{definition}

\section{The certificate}

\begin{definition}[Bool source specification]
\label{def:bool-source-specification}
The source specification for \(\BoolUp\) is the mark-level classifier generated by internal sameness to one of the two finite-kernel marks:
\[
  \mathsf{BoolSourceSpec}(v)
  :=
  \msame(v,\bzero)\lor\msame(v,\bone).
\]
\end{definition}
\leandef{BEDC.Derived.BoolUp.BoolSourceSpec}
\leandef{BEDC.Derived.BoolUp.BoolHistoryCarrier}

\begin{theorem}[Bool source respects mark sameness]
\label{thm:bool-source-spec-respects-msame}
If \(v\) satisfies \(\mathsf{BoolSourceSpec}\) and \(\msame(v,w)\), then \(w\) also satisfies \(\mathsf{BoolSourceSpec}\).
\end{theorem}
\begin{proof}
The source condition is a disjunction of \(\msame(v,\bzero)\) and \(\msame(v,\bone)\). In either case, symmetry gives \(\msame(w,v)\), and transitivity transports the corresponding constructor witness from \(v\) to \(w\).
\end{proof}
\leanchecked{BEDC.Derived.BoolUp.BoolSourceSpec\_respects\_msame}

\begin{theorem}[Bool source transport along mark sameness]
\label{prop:bool-source-msame-transport}
If \(v\) and \(w\) are internally the same mark, then the boolean source predicate transports from \(v\) to \(w\).
\end{theorem}
\leanchecked{BEDC.Derived.BoolUp.BoolSourceSpec\_msame\_transport}
\begin{proof}
The proof eliminates the \(\msame(v,w)\) witness, reducing \(w\) definitionally to \(v\), and reuses the original source-classifier witness.
\end{proof}

\begin{theorem}[Bool history carrier transport]
\label{thm:bool-history-carrier-hsame-transport}
If \(h\) is internally history-same as \(k\) and \(h\) lies in the boolean
history carrier, then \(k\) also lies in the boolean history carrier.
\end{theorem}
\begin{proof}
The carrier witness is either sameness to \(\emp\) or sameness to
\(\Eone(\emp)\). In both cases, symmetry of the link from \(h\) to \(k\)
followed by transitivity transports the endpoint equality to \(k\).
\end{proof}
\leanchecked{BEDC.Derived.BoolUp.BoolHistoryCarrier\_hsame\_transport}

\begin{theorem}[Bool history carrier is unary]
\label{thm:bool-history-carrier-unary}
Every boolean history carrier witness determines a unary history.
\end{theorem}
\begin{proof}
The carrier has two endpoint cases. In the empty endpoint case, unary closure
of \(\emp\) transports back along history sameness. In the one-endpoint case,
unary closure of \(\Eone(\emp)\) transports back along the corresponding
history-sameness witness.
\end{proof}
\leanchecked{BEDC.Derived.BoolUp.BoolHistoryCarrier\_unary}

\begin{theorem}[Bool history carrier excludes zero-extension]
\label{thm:bool-history-carrier-e0-absurd}
No zero-extension history \(\Ezero(h)\) lies in the boolean history carrier.
\end{theorem}
\begin{proof}
The carrier witness is either sameness to \(\emp\) or sameness to
\(\Eone(\emp)\). The first case contradicts generated-history separation
between \(\Ezero(h)\) and \(\emp\), and the second contradicts constructor
separation between zero- and one-extensions.
\end{proof}
\leanchecked{BEDC.Derived.BoolUp.BoolHistoryCarrier\_e0\_absurd}

\begin{theorem}[Bool history one-extension tail]
\label{thm:bool-history-carrier-e1-tail-empty}
If \(\Eone(h)\) lies in the boolean history carrier, then \(h\) is internally
the same as \(\emp\).
\end{theorem}
\begin{proof}
The carrier witness is either sameness to \(\emp\) or sameness to
\(\Eone(\emp)\). The empty endpoint contradicts generated-history
no-confusion for \(\Eone(h)\). The \(\Eone(\emp)\) endpoint reduces by
one-constructor inversion to \(\hsame(h,\emp)\).
\end{proof}
\leanchecked{BEDC.Derived.BoolUp.BoolHistoryCarrier\_e1\_tail\_empty}

\begin{theorem}[Bool history double one-extension exclusion]
\label{thm:bool-history-carrier-double-one-absurd}
No history of the form \(\Eone(\Eone(h))\) lies in the boolean history carrier.
\end{theorem}
\begin{proof}
A carrier witness for \(\Eone(\Eone(h))\) forces its tail \(\Eone(h)\) to be
internally the same as \(\emp\). Generated-history separation between
\(\Eone(h)\) and \(\emp\) eliminates that possibility.
\end{proof}
\leanchecked{BEDC.Derived.BoolUp.BoolHistoryCarrier\_e1\_e1\_absurd}

\begin{theorem}[Bool history one-extension carrier iff]
\label{thm:bool-history-carrier-e1-iff-tail-empty}
A one-extension history \(\Eone(h)\) lies in the boolean history carrier exactly
when its tail is internally the same as \(\emp\).
\end{theorem}
\begin{proof}
The forward direction is the tail inversion result above. Conversely, if
\(\hsame(h,\emp)\), one-constructor congruence gives
\(\hsame(\Eone(h),\Eone(\emp))\), which is the nonempty endpoint case of the
boolean history carrier.
\end{proof}
\leanchecked{BEDC.Derived.BoolUp.BoolHistoryCarrier\_e1\_iff\_tail\_empty}

\begin{theorem}[Bool history classifier excludes zero-growth endpoints]
\label{thm:bool-history-classifier-e0-endpoint-absurd}
The boolean history classifier cannot relate a history of the form \(\Ezero(h)\)
from either endpoint position.
\end{theorem}
\begin{proof}
Unpack the classifier witness. On the left endpoint, its first carrier field is
a boolean history carrier witness for \(\Ezero(h)\); on the right endpoint, its
second carrier field is such a witness. The zero-growth carrier exclusion
eliminates both cases.
\end{proof}
\leanchecked{BEDC.Derived.BoolUp.BoolHistoryClassifier\_e0\_endpoint\_absurd\_pair}

\begin{theorem}[Bool history classifier one-tail sameness]
\label{thm:bool-history-classifier-e1-tail-hsame}
If the boolean history classifier relates $\Eone(h)$ to $\Eone(k)$, then the
tails $h$ and $k$ are internally history-same.
\end{theorem}
\begin{proof}
Unpack the classifier witness to expose its underlying
$\hsame(\Eone(h),\Eone(k))$ field. One-constructor inversion for generated
histories then yields $\hsame(h,k)$.
\end{proof}
\leanchecked{BEDC.Derived.BoolUp.BoolHistoryClassifier\_e1\_tail\_hsame}

\begin{theorem}[Bool history classifier one-tail empty equivalence]
\label{thm:bool-history-classifier-e1-iff-tails-empty}
The boolean history classifier relates \(\Eone(h)\) to \(\Eone(k)\) exactly
when both tails \(h\) and \(k\) are internally history-same as \(\emp\).
\end{theorem}
\begin{proof}
For the forward direction, unpack the classifier into its endpoint carrier
witnesses and apply the one-tail carrier inversion at both endpoints. For the
reverse direction, the tail witnesses rebuild both endpoint carriers, while
their composite \(\hsame(h,k)\) link lifts through the \(\Eone\) constructor.
\end{proof}
\leanchecked{BEDC.Derived.BoolUp.BoolHistoryClassifier\_e1\_iff\_tails\_empty}

\begin{theorem}[Bool history classifier one-left endpoints]
\label{thm:bool-history-classifier-eone-left-endpoints}
The boolean history classifier relates \(\Eone(h)\) to \(k\) exactly when
\(h\) is internally the same as \(\emp\) and \(k\) is internally the same as
\(\Eone(\emp)\).
\end{theorem}
\begin{proof}
Forward, the left carrier forces the one-extension tail to be empty, and the
exposed classifier sameness transports the right endpoint to
\(\Eone(\emp)\). Reverse, the endpoint sameness witnesses provide both carrier
witnesses, and one-extension congruence composed with the right endpoint
symmetry gives the classifier link.
\end{proof}
\leanchecked{BEDC.Derived.BoolUp.BoolHistoryClassifier\_e1\_left\_iff}

\begin{theorem}[Bool history classifier cases]
\label{thm:bool-history-classifier-cases}
Every boolean history classifier witness relates two histories in the same
displayed endpoint class: either both are internally the same as \(\emp\), or
both are internally the same as \(\Eone(\emp)\).
\end{theorem}
\begin{proof}
Unpack the classifier witness into endpoint carrier data and the internal
\(\HSame\) link. The left endpoint carrier has exactly two cases. In either
case, symmetry of the classifier link followed by transitivity transports the
left endpoint class to the right endpoint.
\end{proof}
\leanchecked{BEDC.Derived.BoolUp.BoolHistoryClassifier\_cases}

\begin{theorem}[Bool history classifier constructor separation]
\label{thm:bool-history-classifier-constructor-separation}
The boolean history classifier cannot relate \(\emp\) to \(\Eone(\emp)\).
\end{theorem}
\begin{proof}
Eliminating the classifier witness exposes the underlying \(\HSame\) witness
between \(\emp\) and \(\Eone(\emp)\). Generated-history constructor
no-confusion rules out such a witness.
\end{proof}
\leanchecked{BEDC.Derived.BoolUp.BoolHistoryClassifier\_constructor\_separation}

\begin{theorem}[Bool history classifier transitivity]
\label{thm:bool-history-classifier-transitivity}
If a boolean history classifier relates \(h\) to \(k\), and relates \(k\) to
\(r\), then it relates \(h\) to \(r\).
\end{theorem}
\begin{proof}
Eliminate the two classifier witnesses. The endpoint carrier witnesses for
\(h\) and \(r\) are retained, while the exposed \(\hsame(h,k)\) and
\(\hsame(k,r)\) links compose by generated-history transitivity.
\end{proof}
\leanchecked{BEDC.Derived.BoolUp.BoolHistoryClassifier\_trans}

\begin{theorem}[Bool history classifier symmetry]
\label{thm:bool-history-classifier-symmetry}
If the boolean history classifier relates \(h\) to \(k\), then it relates
\(k\) to \(h\).
\end{theorem}
\begin{proof}
Eliminate the classifier witness. The two endpoint carrier witnesses exchange
places, and the exposed \(\hsame(h,k)\) link reverses by generated-history
symmetry.
\end{proof}
\leanchecked{BEDC.Derived.BoolUp.BoolHistoryClassifier\_symm}

\begin{theorem}[Bool history classifier endpoint transport]
\label{thm:bool-history-classifier-hsame-transport}
If $h$ is internally the same as $h'$ and $k$ is internally the same as $k'$,
then the boolean history classifier transports from $(h,k)$ to $(h',k')$.
\end{theorem}
\begin{proof}
Eliminate the classifier witness into its two endpoint carrier proofs and the
exposed $\hsame(h,k)$ link. Transport each endpoint carrier along its
corresponding $\hsame$ proof, then compose symmetry of the left endpoint with
the stored classifier link and the right endpoint transport.
\end{proof}
\leanchecked{BEDC.Derived.BoolUp.BoolHistoryClassifier\_hsame\_transport}

\begin{theorem}[Bool history classifier empty-left equivalence]
\label{thm:bool-history-classifier-empty-left-iff}
The boolean history classifier relates \(\emp\) to \(k\) exactly when \(k\)
is internally the same history as \(\emp\).
\end{theorem}
\begin{proof}
For the forward direction, eliminate the classifier witness and symmetrize the
exposed \(\hsame(\emp,k)\) link. For the reverse direction, the empty carrier
witness supplies the left endpoint, the assumed \(\hsame(k,\emp)\) supplies the
right endpoint, and symmetry supplies the classifier link.
\end{proof}
\leanchecked{BEDC.Derived.BoolUp.BoolHistoryClassifier\_empty\_left\_iff}

\begin{definition}[Bool history semantic fields]
\label{def:bool-history-semantic-fields}
The history-level semantic fields used by the boolean certificate are all the
same carrier predicate.  For a BEDC history $h$,
$$
  \mathsf{PatternSpec}(h)
  \quad\text{and}\quad
  \mathsf{LedgerPolicy}(h)
$$
are both read as $\mathsf{BoolHistoryCarrier}(h)$, namely that $h$ is
internally the same as one of the two visible boolean endpoints:
$$
  \hsame(h,\emp)
  \quad\text{or}\quad
  \hsame(h,\Eone(\emp)).
$$
\end{definition}

\begin{theorem}[BoolUp semantic certificate target]
\label{thm:bool-semantic-namecert-target}
For $\mathsf{BoolHistoryCarrier}$, the checked core certificate assembles with
the carrier-as-pattern fields into the paper-side target
$$
\mathsf{SemanticNameCert}(\mathsf{BoolHistoryCarrier},\mathsf{BoolHistoryCarrier},
\mathsf{BoolHistoryCarrier},\mathsf{BoolHistoryClassifier}).
$$
\end{theorem}
\begin{proof}
Use the checked core certificate $\mathsf{bool\_history\_name\_certificate}$ as
the $\NameCert$ component.  A carrier witness for a licensed boolean history is
already one of the two endpoint cases $\hsame(h,\emp)$ or
$\hsame(h,\Eone(\emp))$, so the carrier predicate itself supplies the required
history-level pattern and ledger fields.  These three fields are exactly the
constructor data for the corresponding semantic naming certificate.
\end{proof}

\begin{corollary}[\(\BoolUp\) is licensed, not primitive]
\label{cor:bool-licensed-not-primitive}
\(\BoolUp\) is licensed by mark sameness and is not a primitive truth-value type.
\end{corollary}
