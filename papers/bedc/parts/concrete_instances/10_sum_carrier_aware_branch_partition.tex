\begin{theorem}[Sum history classifier branch partition]
\label{thm:sum-history-classifier-branch-partition}
For branch carriers $Left,Right:\mathsf{BHist}\to\mathsf{Prop}$, branch
classifiers $LeftEq,RightEq:\mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$,
and histories $h,k$, the history-level sum classifier is exactly the visible
left-or-right constructor analysis:
$$
\mathsf{SumHistoryClassifier}(Left,Right,LeftEq,RightEq,h,k)
\Longleftrightarrow
\bigl(\exists l,l'.\;
\hsame(h,\Ezero(l))\land
\hsame(k,\Ezero(l'))\land
LeftEq(l,l')\bigr)
\lor
\bigl(\exists r,r'.\;
\hsame(h,\Eone(r))\land
\hsame(k,\Eone(r'))\land
RightEq(r,r')\bigr).
$$
In the ten-field fit, the traditional target is disjoint-sum branch readback,
the BEDC source is $\mathsf{BHist}$, the classifier is
$\mathsf{SumHistoryClassifier}$, the ledger records the visible tag plus
payload, and the status is $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
For the forward direction, case-analyze the classifier witness. The left
constructor supplies the two $\Ezero$ endpoint links and the stored $LeftEq$
payload relation, yielding the left disjunct. The right constructor supplies
the two $\Eone$ endpoint links and the stored $RightEq$ payload relation,
yielding the right disjunct.

For the reverse direction, split the displayed disjunction. In the left branch,
unpack $l,l'$ together with the two visible left endpoint witnesses and the
stored $LeftEq(l,l')$ witness, then rebuild the left classifier constructor.
In the right branch, unpack $r,r'$ together with the two visible right endpoint
witnesses and the stored $RightEq(r,r')$ witness, then rebuild the right
classifier constructor.
\end{proof}

\begin{theorem}[Sum history carrier visible branch cases]
\label{thm:sum-history-carrier-visible-branch-cases}
Let $Left,Right:\mathsf{BHist}\to\mathsf{Prop}$ be source carriers. If
$\mathsf{SumHistoryCarrier}(Left,Right,h)$ holds, then the visible history has
one of the two displayed branch witnesses:
$$
\bigl(\exists l.\; Left(l)\land \hsame(h,\Ezero(l))\bigr)
\lor
\bigl(\exists r.\; Right(r)\land \hsame(h,\Eone(r))\bigr).
$$
In the ten-field fit, the traditional target is disjoint-sum carrier readback,
the BEDC source is $\mathsf{BHist}$, the carrier is
$\mathsf{SumHistoryCarrier}$, no classifier is needed, the stability obligation
is constructor inversion, the ledger records the visible tag plus payload, the
core theorem is branch cases, exactness coverage is supplied by the visible
branch witnesses, the standard bridge is $\mathsf{inl}$ and $\mathsf{inr}$, and
the status is $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Case-analyze the generated $\mathsf{SumHistoryCarrier}$ witness. The left
constructor returns a payload $l$, the source-carrier witness $Left(l)$, and
the displayed endpoint link $\hsame(h,\Ezero(l))$. The right constructor
returns the corresponding payload $r$, the source-carrier witness $Right(r)$,
and the displayed endpoint link $\hsame(h,\Eone(r))$.
\end{proof}

\begin{theorem}[Sum history classifier carrier-aware branch partition]
\label{thm:sum-history-classifier-carrier-aware-branch-partition}
Let $Left,Right:\mathsf{BHist}\to\mathsf{Prop}$ be source carriers, let
$LeftEq,RightEq:\mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$ be source
classifiers, and assume the two carriers transport along $\hsame$:
$$
\forall x\,y.\; Left(x)\to \hsame(x,y)\to Left(y)
$$
and
$$
\forall x\,y.\; Right(x)\to \hsame(x,y)\to Right(y).
$$
If $\mathsf{SumHistoryCarrier}(Left,Right,h)$,
$\mathsf{SumHistoryCarrier}(Left,Right,k)$, and
$\mathsf{SumHistoryClassifier}(Left,Right,LeftEq,RightEq,h,k)$ hold, then a
visible classifier side supplies the branch data used by the carrier-backed
certificate analysis:
$$
\begin{array}{l}
\exists a,b.\; Left(a)\land Left(b)\land
\hsame(h,\Ezero(a))\land \hsame(k,\Ezero(b))\land LeftEq(a,b)
\end{array}
$$
or
$$
\begin{array}{l}
\exists a,b.\; Right(a)\land Right(b)\land
\hsame(h,\Eone(a))\land \hsame(k,\Eone(b))\land RightEq(a,b).
\end{array}
$$
\end{theorem}
\begin{proof}
Split the displayed $\mathsf{SumHistoryClassifier}$ witness into its visible
left or right constructor branch. In the left branch, combine the two endpoint
carrier witnesses with the visible $\Ezero$ endpoints stored by the classifier.
The checked left carrier inversion for $\mathsf{SumHistoryCarrier}$ reads the
endpoint carrier data back to left payload carriers, and the displayed carrier
transport assumption aligns those source-membership witnesses with the
classifier payloads. The stored $LeftEq$ witness then supplies the left
disjunct. The right branch is identical, using the checked right carrier
inversion, the $\Eone$ endpoints, the right carrier-transport assumption, and
the stored $RightEq$ witness.
\end{proof}

\begin{theorem}[Sum history source weakening]
\label{thm:sum-history-source-weakening}
Let $Left,Right,Left',Right':\mathsf{BHist}\to\mathsf{Prop}$ be source
carriers, and let
$LeftEq,RightEq,LeftEq',RightEq':\mathsf{BHist}\to\mathsf{BHist}\to
\mathsf{Prop}$ be source classifiers. Assume source-carrier inclusion
$$
\forall h.\; Left(h)\to Left'(h)
$$
and
$$
\forall h.\; Right(h)\to Right'(h),
$$
and assume component-relation weakening
$$
\forall a\,b.\; LeftEq(a,b)\to LeftEq'(a,b)
$$
and
$$
\forall a\,b.\; RightEq(a,b)\to RightEq'(a,b).
$$
Then carrier and classifier witnesses refine over the same visible endpoints:
$$
\mathsf{SumHistoryCarrier}(Left,Right,h)\to
\mathsf{SumHistoryCarrier}(Left',Right',h)
$$
and
$$
\mathsf{SumHistoryClassifier}(Left,Right,LeftEq,RightEq,h,k)\to
\mathsf{SumHistoryClassifier}(Left',Right',LeftEq',RightEq',h,k).
$$
In the ten-field fit, the traditional target is disjoint-sum stability under
source refinement, the BEDC source is the two $\mathsf{BHist}$ carriers, the
carrier is $\mathsf{SumHistoryCarrier}$, the component classifier is
$\mathsf{SumHistoryClassifier}$, the stability datum is source inclusion plus
component-relation weakening, the ledger keeps the visible tag unchanged, the
core theorem is carrier and classifier weakening, exactness is preserved by
the same endpoint witnesses, the standard bridge is the $\mathsf{inl}$ and
$\mathsf{inr}$ restriction, and the status is $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
For the carrier claim, split the $\mathsf{SumHistoryCarrier}$ witness into its
left and right constructor cases. The endpoint history witness is reused
unchanged, while the displayed source-carrier inclusion maps the payload
membership proof into the refined source carrier. For the classifier claim,
split the $\mathsf{SumHistoryClassifier}$ witness into its left and right
constructor cases. Reuse the same two visible endpoints and apply the matching
component-relation weakening hypothesis to the stored $LeftEq$ or $RightEq$
witness; the carrier-side endpoint data is transported by the corresponding
source inclusion in the same branch.
\end{proof}

\begin{theorem}[Sum history classifier relation weakening]
\label{thm:sum-history-classifier-relation-weakening}
Let $Left,Right,LeftAlt,RightAlt:\mathsf{BHist}\to\mathsf{Prop}$ be arbitrary
source carriers, and let
$LeftEq,RightEq,LeftEq',RightEq':\mathsf{BHist}\to\mathsf{BHist}\to
\mathsf{Prop}$ be component classifiers. Assume component-relation weakening
$$
\forall a\,b.\; LeftEq(a,b)\to LeftEq'(a,b)
$$
and
$$
\forall a\,b.\; RightEq(a,b)\to RightEq'(a,b).
$$
Then the sum history classifier relation weakens independently of source-carrier
inclusion:
$$
\mathsf{SumHistoryClassifier}(Left,Right,LeftEq,RightEq,h,k)\to
\mathsf{SumHistoryClassifier}(LeftAlt,RightAlt,LeftEq',RightEq',h,k).
$$
In the ten-field fit, the traditional target is disjoint-sum classifier
monotonicity, the BEDC source is $\mathsf{BHist}$, the classifier is
$\mathsf{SumHistoryClassifier}$, the stability datum is component-relation
weakening, the ledger keeps the visible tag unchanged, exactness is branch
partition preservation, and the status is $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Split the displayed $\mathsf{SumHistoryClassifier}$ witness into its visible
left or right constructor branch. In each branch, reuse the same visible endpoint
witnesses for $h$ and $k$. The left branch applies the left component-relation
weakening hypothesis to the stored $LeftEq$ witness and rebuilds the left tagged
classifier witness; the right branch applies the right component-relation
weakening hypothesis to the stored $RightEq$ witness and rebuilds the right
tagged classifier witness.
\end{proof}

\begin{theorem}[Sum ledger raw-visible branch partition]
\label{thm:sum-ledger-raw-visible-branch-partition}
Let $Left,Right:\mathsf{BHist}\to\mathsf{Prop}$ be source carriers and let
$\rho,v:\mathsf{BHist}$ be histories. If
$\mathsf{SumHistoryLedgerPolicy}(Left,Right,\rho,v)$ holds, then the same
visible branch and payload appear on both ends of the ledger link:
$$
\bigl(\exists l.\; Left(l)\land\hsame(\rho,\Ezero(l))\land\hsame(v,\Ezero(l))\bigr)
\lor
\bigl(\exists r.\; Right(r)\land\hsame(\rho,\Eone(r))\land\hsame(v,\Eone(r))\bigr).
$$
In the ten-field fit, the traditional target is disjoint-sum ledger branch
readback; the BEDC source is $\mathsf{BHist}$ with $\hsame$; the carrier is
$\mathsf{SumHistoryCarrier}$; no classifier is needed for the direct branch
exposure; the ledger is $\mathsf{SumHistoryLedgerPolicy}$; exactness is the
same raw-visible branch and payload on both endpoints; and the status is
$\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Unpack the ledger into the raw $\mathsf{SumHistoryCarrier}$ witness and the
stored raw-visible $\hsame(\rho,v)$ link. Split the raw carrier. In the left
branch, keep the raw payload $l$ and its $Left(l)$ witness, retain the stored
raw tag witness $\hsame(\rho,\Ezero(l))$, and compose symmetry of
$\hsame(\rho,v)$ with that raw tag witness to obtain
$\hsame(v,\Ezero(l))$. The right branch uses the same argument with the raw
payload $r$, the $Right(r)$ witness, and the $\Eone$ tag.
\end{proof}
