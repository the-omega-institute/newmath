\chapter{A Concrete Naming Certificate for \(\IntUp\)}
\label{ch:concrete-instances-int-namecert}

The integer interface \(\IntUp\) is licensed by signed unary histories: a sign component together with a unary magnitude under continuation discipline. It is not a primitive operation of the finite kernel. The construction follows the 5-field NameCert template established by \autoref{ch:concrete-instances-nat-namecert} and \autoref{ch:concrete-instances-add-namecert}, with the carrier extended by a sign indicator and the stability fields adjusted for sign-aware continuation.

\section{Carrier definition}

\begin{definition}[Signed unary integer carrier]
\label{def:intup-carrier}
The carrier predicate for \(\IntUp\) uses an existing mark as the sign component and an existing unary history as the magnitude component:
$$
  \mathsf{IntCarrier}(s,h)
  :=
  (s=\bzero \lor s=\bone)\land \mathsf{Unary}(h).
$$
\end{definition}
\leandef{BEDC.Derived.IntUp.IntCarrier}
\leanchecked{BEDC.Derived.IntUp.IntCarrier\_sign\_cases}
\leanvariant{BEDC.Derived.IntUp.IntCarrier\_continuation\_closed\_same\_sign}
\begin{proof}
This is a definitional clause over the finite-kernel mark constructors and the unary-history predicate. With the sign fixed, continuation of two carrier magnitudes remains inside the same signed carrier.
\end{proof}

\begin{theorem}[Signed carrier transports along mark sameness]
\label{thm:int-carrier-sign-msame-transport}
If \(\mathsf{IntCarrier}(s,h)\) and \(\msame(s,s')\), then
\(\mathsf{IntCarrier}(s',h)\).
\end{theorem}
\begin{proof}
The sign-sameness witness identifies the two finite-kernel sign marks. After
that identification, the original signed unary carrier witness has the required
transported sign.
\end{proof}
\leanchecked{BEDC.Derived.IntUp.IntCarrier\_sign\_msame\_transport}

\begin{theorem}[Signed carrier transports along history sameness]
\label{thm:int-carrier-transport-hsame-magnitude}
For a fixed sign \(s\), if \(\mathsf{IntCarrier}(s,h)\) and \(\hsame(h,k)\), then \(\mathsf{IntCarrier}(s,k)\).
\end{theorem}
\begin{proof}
The carrier splits into the sign clause and the unary magnitude clause. The sign clause is unchanged, while the unary-history witness is transported along \(\hsame(h,k)\).
\end{proof}
\leanchecked{BEDC.Derived.IntUp.IntCarrier\_transport\_hsame\_magnitude}

\begin{theorem}[Integer magnitude transport along history sameness]
\label{prop:intup-magnitude-hsame-transport}
For a fixed sign \(s\), if \(\mathsf{IntCarrier}(s,h)\) holds and \(h\) is internally the same history as \(k\), then \(\mathsf{IntCarrier}(s,k)\) holds.
\end{theorem}
\leanchecked{BEDC.Derived.IntUp.IntCarrier\_magnitude\_hsame\_transport}
\begin{proof}
The sign disjunction is preserved unchanged, while the unary-history component is transported across \(\hsame(h,k)\) by the finite-kernel unary transport theorem.
\end{proof}

\begin{theorem}[Integer carrier magnitude induction]
\label{thm:int-carrier-magnitude-induction}
For a fixed sign \(s:\mathsf{BMark}\), any predicate \(P\) on histories holds
for every integer-carried magnitude if it holds at \(\emp\) and is preserved
by one-step unary extension along \(\mathsf{IntCarrier}(s,-)\).
\end{theorem}
\begin{proof}
Unpack \(\mathsf{IntCarrier}(s,h)\) into its sign clause and unary-history
witness. Apply unary-history induction to the magnitude; the step rebuilds the
integer carrier from the fixed sign clause and the current unary witness.
\end{proof}
\leanchecked{BEDC.Derived.IntUp.IntCarrier\_magnitude\_induction}

\begin{definition}[Integer classifier specification]
\label{def:intup-classifier-specification}
The classifier for two signed unary representatives requires both representatives to be in the integer carrier, their signs to be mark-same, and their magnitudes to be history-same:
$$
  \mathsf{IntClassifierSpec}((s,h),(t,k))
  :=
  \mathsf{IntCarrier}(s,h)\land
  \mathsf{IntCarrier}(t,k)\land
  \msame(s,t)\land
  \hsame(h,k).
$$
\end{definition}

\begin{definition}[Integer difference-pair carrier]
\label{def:intup-difference-pair-carrier}
The signed-pair presentation of $\IntUp$ uses two unary histories, read as a
positive counter and a negative counter:
$$
  \mathsf{IntPairCarrier}(p,n)
  :=
  \mathsf{Unary}(p)\land\mathsf{Unary}(n).
$$
\end{definition}
\leandef{BEDC.Derived.IntUp.IntPairCarrier}

\begin{definition}[Integer difference-pair classifier]
\label{def:intup-difference-pair-classifier}
Two carried unary pairs classify together when their cross-additions are
history-same:
$$
  \mathsf{IntPairClassifier}((p,n),(q,m))
  :=
  \mathsf{IntPairCarrier}(p,n)\land
  \mathsf{IntPairCarrier}(q,m)\land
  \hsame(\mathsf{append}(p,m),\mathsf{append}(q,n)).
$$
\end{definition}
\leandef{BEDC.Derived.IntUp.IntPairClassifier}

\begin{theorem}[Integer difference-pair classifier equivalence fields]
\label{thm:intup-difference-pair-classifier-equivalence-fields}
On carried unary pairs, $\mathsf{IntPairClassifier}$ supplies the expected
integer quotient fields:
\begin{enumerate}
\item reflexivity:
for every $\mathsf{IntPairCarrier}(p,n)$,
$\mathsf{IntPairClassifier}((p,n),(p,n))$;
\item symmetry:
from $\mathsf{IntPairClassifier}((p,n),(q,m))$ one obtains
$\mathsf{IntPairClassifier}((q,m),(p,n))$;
\item transitivity:
from $\mathsf{IntPairClassifier}((p,n),(q,m))$ and
$\mathsf{IntPairClassifier}((q,m),(r,o))$ one obtains
$\mathsf{IntPairClassifier}((p,n),(r,o))$;
\item componentwise transport:
if $p\hsame p'$, $n\hsame n'$, $q\hsame q'$, and $m\hsame m'$, then a
classifier witness for $((p,n),(q,m))$ transports to one for
$( (p',n'),(q',m') )$ whenever all four pairs are carried.
\end{enumerate}
The ten-field fit is: traditional target signed integers as unary difference
pairs; BEDC source $\mathsf{Unary}$, $\hsame$, and $\mathsf{append}$;
carrier $\mathsf{IntPairCarrier}$; classifier
$\mathsf{IntPairClassifier}$; stability componentwise $\hsame$ transport;
ledger the two unary counters; core theorem this block; exactness
cross-addition classification; standard bridge the ordinary presentation of
integers as $(p,n)$ modulo $p+m=q+n$; status $\mathsf{ProofTarget}$.
\end{theorem}
\leanchecked{BEDC.Derived.IntUp.IntPairClassifier\_equivalence\_fields}
\begin{proof}
For reflexivity, unary append closure shows both cross-additions
$\mathsf{append}(p,n)$ are unary, and reflexivity of $\hsame$ supplies the
cross-addition equality. The two carrier witnesses are reused unchanged.

For symmetry, unpack the classifier witness. The carrier witnesses swap
positions, and symmetry of $\hsame$ reverses the cross-addition comparison
between $\mathsf{append}(p,m)$ and $\mathsf{append}(q,n)$.

For transitivity, suppose the two displayed classifier witnesses give
$\mathsf{append}(p,m)\hsame\mathsf{append}(q,n)$ and
$\mathsf{append}(q,o)\hsame\mathsf{append}(r,m)$. Append the missing
contexts to both comparisons so that the shared unary factors are visible.

Using $\mathsf{append\_assoc}$, unary append commutativity, and transitivity
of $\hsame$, rewrite the composed comparison to the form
$$
\mathsf{append}(\mathsf{append}(p,o),m)
\hsame
\mathsf{append}(\mathsf{append}(r,n),m).
$$

The common suffix $m$ is carried by unary closure, so
$\mathsf{append\_right\_cancel}$, equivalently common-context cancellation,
removes it and yields
$\mathsf{append}(p,o)\hsame\mathsf{append}(r,n)$. Together with the endpoint
carrier witnesses, this is exactly
$\mathsf{IntPairClassifier}((p,n),(r,o))$.

For componentwise transport, transport unary carrier evidence along the four
$\hsame$ links. Compatibility of $\mathsf{append}$ with $\hsame$ transports
the cross-addition comparison from
$\mathsf{append}(p,m)\hsame\mathsf{append}(q,n)$ to
$\mathsf{append}(p',m')\hsame\mathsf{append}(q',n')$.
\end{proof}

\begin{theorem}[Integer classifier endpoint sign cases]
\label{thm:int-classifier-endpoint-sign-cases}
If $\mathsf{IntClassifierSpec}(x,y)$, then the two endpoint signs are either
both $\bzero$ or both $\bone$, and both endpoint magnitudes are unary:
$$
\bigl((x_1=\bzero\wedge y_1=\bzero)\vee

$$
\end{theorem}
\begin{proof}

and magnitude sameness. The left carrier splits the left sign into the two

sameness transports the same sign case to the right endpoint, while the right
carrier supplies the unary witness for the right magnitude.
\end{proof}

\begin{theorem}[Integer classifier cross-sign exclusion]
\label{thm:int-classifier-cross-sign-exclusion}
For all $h,k:\mathsf{BHist}$, cross-signed endpoints cannot be classified:
$$
\forall h,k:\mathsf{BHist},\quad
\bigl(\mathsf{IntClassifierSpec}((\bzero,h),(\bone,k))\to\mathsf{False}\bigr)
\land
\bigl(\mathsf{IntClassifierSpec}((\bone,h),(\bzero,k))\to\mathsf{False}\bigr).
$$
The ten-field fit is: traditional target signed-integer sign separation; BEDC
source $\mathsf{IntCarrier}$, $\mathsf{IntClassifierSpec}$, and raw mark
no-confusion; carrier signed unary representatives; classifier
$\mathsf{IntClassifierSpec}$; stability from existing sign and magnitude
transport; ledger the sign mark and unary magnitude; core theorem this block;
exactness cross-sign exclusion; standard bridge ordinary signed-integer sign
disjointness; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Project the sign-sameness component from
\autoref{def:intup-classifier-specification}. In the first branch, a classifier
from $(\bzero,h)$ to $(\bone,k)$ supplies $\msame(\bzero,\bone)$, which
contradicts raw mark no-confusion in \autoref{thm:mark-no-confusion}.

The second branch is the opposite orientation. A classifier from $(\bone,h)$ to
$(\bzero,k)$ supplies $\msame(\bone,\bzero)$, and the exchanged half of
\autoref{thm:mark-no-confusion} excludes that mark-sameness judgment.
\end{proof}
\leanchecked{BEDC.Derived.IntUp.IntClassifierSpec\_cross\_sign\_exclusion}

\section{The certificate}

\begin{definition}[Integer history carrier]
\label{def:intup-history-carrier}
The public BEDC-history carrier for $\IntUp$ packages the signed unary
representative into the visible history tag:
$$
\mathsf{IntHistoryCarrier}(h)
:=
\bigl(\exists m:\mathsf{BHist}.\;
\mathsf{Unary}(m)\land\hsame(h,\Ezero(m))\bigr)
\lor
\bigl(\exists m:\mathsf{BHist}.\;
\mathsf{Unary}(m)\land\hsame(h,\Eone(m))\bigr).
$$
The $\Ezero$ branch and the $\Eone$ branch encode the two signed unary
presentations as actual BEDC histories.
\end{definition}
\leandef{BEDC.Derived.IntUp.IntHistoryCarrier}

\begin{definition}[Integer history classifier]
\label{def:intup-history-classifier}
The public classifier relates exactly same-tag signed unary histories:
$$
\mathsf{IntHistoryClassifier}(h,k)
:=
\bigl(\exists m,n:\mathsf{BHist}.\;
\mathsf{Unary}(m)\land\mathsf{Unary}(n)\land
\hsame(h,\Ezero(m))\land\hsame(k,\Ezero(n))\land\hsame(m,n)\bigr)
\lor
\bigl(\exists m,n:\mathsf{BHist}.\;
\mathsf{Unary}(m)\land\mathsf{Unary}(n)\land
\hsame(h,\Eone(m))\land\hsame(k,\Eone(n))\land\hsame(m,n)\bigr).
$$
\end{definition}
\leandef{BEDC.Derived.IntUp.IntHistoryClassifier}

\begin{theorem}[Integer history classifier visible branch exactness]
\label{thm:int-history-classifier-visible-branch-exactness}
For any $m,n:\mathsf{BHist}$ with $\mathsf{Unary}(m)$ and
$\mathsf{Unary}(n)$, the public integer-history classifier reads back the
same visible tag precisely as payload history sameness:
$$
\bigl(
\mathsf{IntHistoryClassifier}(\Ezero(m),\Ezero(n))
\Longleftrightarrow
\hsame(m,n)
\bigr)
\land
\bigl(
\mathsf{IntHistoryClassifier}(\Eone(m),\Eone(n))
\Longleftrightarrow
\hsame(m,n)
\bigr).
$$
The ten-field fit is: traditional target signed-integer visible-tag readback;
BEDC source $\mathsf{BHist}$; admissible payloads $\mathsf{Unary}$; payload
relation $\hsame$; public classifier $\mathsf{IntHistoryClassifier}$; carrier
the public integer history carrier; classifier same-tag payload sameness;
stability by constructor inversion and $\hsame$ laws; ledger visible tag plus
unary payload; exactness the two iff clauses; standard bridge ordinary
signed-integer tag readback; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
For the zero-tag forward direction, split a witness of
$\mathsf{IntHistoryClassifier}(\Ezero(m),\Ezero(n))$ by the two displayed
classifier branches. In the zero branch, constructor inversion on the endpoint
witnesses reduces the visible histories to payload histories, and transitivity
of $\hsame$ composes the two endpoint equalities with the stored payload
comparison. In the one branch, the endpoint evidence would identify a zero tag
with a one tag, contradicting history no-confusion.

The one-tag forward direction is symmetric. The matching one branch gives the
payload comparison by constructor inversion and transitivity of $\hsame$; the
zero branch is ruled out by the same zero/one history no-confusion.

For the reverse directions, assume $\hsame(m,n)$ together with the displayed
unary witnesses for $m$ and $n$. The zero-tag classifier witness is obtained by
choosing the zero branch, using the two unary witnesses, reflexive endpoint
witnesses for $\Ezero(m)$ and $\Ezero(n)$, and the given payload relation. The
one-tag case is identical with $\Eone$ in place of $\Ezero$.
\end{proof}

\begin{theorem}[Concrete IntUp history semantic certificate]
\label{thm:concrete-int-history-semantic-namecert}
The integer history carrier and classifier form the public semantic naming
certificate for the concrete signed unary history presentation:
$$
\mathsf{SemanticNameCert}(
\mathsf{IntHistoryCarrier},
\mathsf{IntHistoryCarrier},
\mathsf{IntHistoryCarrier},
\mathsf{IntHistoryClassifier}).
$$
Moreover the classifier excludes mixed visible tags:
$$
\forall m,n:\mathsf{BHist},\quad
\mathsf{Unary}(m)\to\mathsf{Unary}(n)\to
\bigl(\mathsf{IntHistoryClassifier}(\Ezero(m),\Eone(n))\to\mathsf{False}\bigr)
\land
\bigl(\mathsf{IntHistoryClassifier}(\Eone(m),\Ezero(n))\to\mathsf{False}\bigr).
$$
\end{theorem}
\begin{proof}
For inhabitance, choose the zero-tagged empty magnitude
$\Ezero(\emp)$. The unary base witness gives $\mathsf{Unary}(\emp)$, and
reflexivity of $\hsame$ gives the visible endpoint witness.

For reflexivity, split the carrier witness for $h$. In the $\Ezero$ branch,
reuse the same magnitude on both endpoints and use reflexivity of $\hsame$ for
the payload comparison. The $\Eone$ branch is identical with the one-tagged
constructor.

For symmetry, unpack the classifier branch. The two endpoint carrier witnesses
are kept in the same visible tag, and symmetry of $\hsame$ reverses the payload
comparison. Transitivity composes two same-tag classifier witnesses by composing
their payload $\hsame$ links through the shared middle payload.

For carrier transport, suppose $\mathsf{IntHistoryCarrier}(h)$ and
$\mathsf{IntHistoryClassifier}(h,k)$. In a same-tag branch, compose the
classifier endpoint witness for $k$ with the stored unary payload witness, then
repack the corresponding $\Ezero$ or $\Eone$ carrier branch.

The semantic pattern and ledger fields are the carrier itself, so both
soundness obligations are identity implications from a transported carrier
witness. Mixed visible tags are impossible because any mixed classifier branch
would compose the visible endpoint witnesses into an
$\hsame(\Ezero(m'),\Eone(n'))$ or
$\hsame(\Eone(m'),\Ezero(n'))$ judgment, contradicting BEDC constructor
