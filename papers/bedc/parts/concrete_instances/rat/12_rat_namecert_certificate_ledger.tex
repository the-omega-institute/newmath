\begin{definition}[Rat ledger policy]
\label{def:rat-ledger-policy}
Record the raw rational-history carrier and an $\hsame$ link to the visible denominator history.
\end{definition}
\leandef{BEDC.Derived.RatUp.RatHistoryLedgerPolicy}

\begin{theorem}[Rat ledger visible carrier]
\label{thm:rat-ledger-visible-carrier}
If a raw rational history satisfies the ledger policy for a visible history, then the visible history is itself a rational-history carrier.
\end{theorem}
\leanchecked{BEDC.Derived.RatUp.RatHistoryLedgerPolicy\_visible\_carrier}
\begin{proof}
Unpack the ledger policy into a raw rational-history carrier and an $\hsame$ link from the raw history to the visible history. Transport the raw carrier across that link.
\end{proof}

\begin{theorem}[Rat ledger visible positive denominator readback]
\label{thm:rat-ledger-visible-positive-denominator-readback}
If $\mathsf{RatHistoryLedgerPolicy}(\rho,v)$ holds and
$\hsame(v,\mathsf{e1}(t))$ holds, then $\mathsf{UnaryHistory}(t)$ holds.
\end{theorem}
\leanchecked{BEDC.Derived.RatUp.RatHistoryLedgerPolicy\_visible\_positive\_denominator\_readback}
\begin{proof}
The ledger visible-carrier theorem transports the raw rational-history carrier
to $v$. Transport that carrier once more across $\hsame(v,\mathsf{e1}(t))$ and
apply the one-extension rational-history carrier characterization.
\end{proof}

\begin{theorem}[Rat ledger append unary denominator closure]
\label{thm:rat-ledger-append-unary-denominator-closed}
If $\mathsf{RatHistoryLedgerPolicy}(\rho,v)$ holds, $\mathsf{UnaryHistory}(t)$
holds, and $\hsame(t,t')$ holds, then appending those unary denominator tails
preserves the rational ledger:
$$
\mathsf{RatHistoryLedgerPolicy}(\mathsf{append}(\rho,t),\mathsf{append}(v,t')).
$$
\end{theorem}
\leanchecked{BEDC.Derived.RatUp.RatHistoryLedgerPolicy\_append\_unary\_denominator\_closed}
\begin{proof}
Unpack the ledger into a raw rational-history carrier and an $\hsame(\rho,v)$
link. The denominator-append carrier theorem closes the raw endpoint under the
unary tail, and the two $\hsame$ links identify the appended raw and visible
histories.
\end{proof}

\begin{theorem}[Rat ledger prepend unary denominator closure]
\label{thm:rat-ledger-prepend-unary-denominator-closed}
If $\mathsf{RatHistoryLedgerPolicy}(\rho,v)$ holds,
$\mathsf{UnaryHistory}(p)$ holds, and $\hsame(p,p')$ holds, then prefixing
those unary denominator histories preserves the rational ledger:
$$
\mathsf{RatHistoryLedgerPolicy}(\mathsf{append}(p,\rho),\mathsf{append}(p',v)).
$$
\end{theorem}
\leanchecked{BEDC.Derived.RatUp.RatHistoryLedgerPolicy\_prepend\_unary\_denominator\_closed}
\begin{proof}
Unpack the ledger into a raw rational-history carrier and an $\hsame(\rho,v)$
link. The denominator-prefix carrier theorem closes the raw endpoint under the
unary prefix, and the two $\hsame$ links identify the prefixed raw and visible
histories.
\end{proof}

\begin{theorem}[Rat ledger unary denominator context closure]
\label{thm:rat-ledger-unary-denominator-context-closed}
If $\mathsf{RatHistoryLedgerPolicy}(\rho,v)$ holds, unary prefixes $p,p'$
satisfy $\hsame(p,p')$, and unary tails $t,t'$ satisfy $\hsame(t,t')$, then
placing the matching prefix and tail around the raw and visible denominator
histories preserves the rational ledger:
$$
\mathsf{RatHistoryLedgerPolicy}(\mathsf{append}(p,\mathsf{append}(\rho,t)),
\mathsf{append}(p',\mathsf{append}(v,t'))).
$$
\end{theorem}
\leanchecked{BEDC.Derived.RatUp.RatHistoryLedgerPolicy\_unary\_denominator\_context\_closed}
\begin{proof}
First apply rat ledger append unary denominator closure to the displayed ledger
and the matching tails. Then apply rat ledger prepend unary denominator closure
to the resulting ledger and the matching prefixes.
\end{proof}

\begin{theorem}[Rat ledger unary context zero-extension endpoint exclusion]
\label{thm:rat-ledger-unary-context-zero-extension-endpoint-exclusion}
Let $\rho,v,p,p',t,t',z,z':\mathsf{BHist}$. If
$\mathsf{RatHistoryLedgerPolicy}(\rho,v)$ holds, $p,p'$ are matched unary
prefixes with $\hsame(p,p')$, and $t,t'$ are matched unary tails with
$\hsame(t,t')$, then the contextual raw and visible ledger endpoints cannot be
zero-extension endpoints:
$$
(\hsame(\mathsf{append}(p,\mathsf{append}(\rho,t)),\Ezero(z))\to
\mathsf{False})\land
(\hsame(\mathsf{append}(p',\mathsf{append}(v,t')),\Ezero(z'))\to
\mathsf{False}).
$$
\end{theorem}
\begin{proof}
Apply rat ledger unary denominator context closure to obtain the contextual
ledger. The raw-visible classifier theorem turns that ledger into a rational
history classifier between the contextual raw and visible endpoints. Transport
this classifier across a displayed zero-extension equality on either endpoint,
then use rational history classifier zero-extension endpoint exclusion.
\end{proof}
\leanchecked{BEDC.Derived.RatUp.RatHistoryLedgerPolicy\_unary\_context\_zero\_extension\_endpoint\_exclusion}

\begin{theorem}[Rat ledger history-sameness transport]
\label{thm:rat-ledger-hsame-transport}
If a rational ledger relates raw and visible histories, then transporting the
raw endpoint and the visible endpoint along \(\hsame\) yields another rational
ledger relation.
\end{theorem}
\leanchecked{BEDC.Derived.RatUp.RatHistoryLedgerPolicy\_hsame\_transport}
\begin{proof}
Unpack the ledger into a raw carrier and a raw-visible link. Transport the raw
carrier across the raw-endpoint link, then compose the reversed raw transport
link, the stored ledger link, and the visible-endpoint link.
\end{proof}

\begin{theorem}[Rat ledger raw-visible classifier]
\label{thm:rat-ledger-raw-visible-classifier}
If $\mathsf{RatHistoryLedgerPolicy}(\rho,v)$ holds, then
$\mathsf{RatHistoryClassifier}(\rho,v)$ holds.
\end{theorem}
\leanchecked{BEDC.Derived.RatUp.RatHistoryLedgerPolicy\_raw\_visible\_classifier}
\begin{proof}
Unpack the ledger into the raw rational-history carrier for $\rho$ and the
stored $\hsame(\rho,v)$ link. The visible-carrier theorem supplies the
rational-history carrier for $v$, and these witnesses form the raw-visible
rational history classifier.
\end{proof}

\begin{theorem}[Rat ledger shared-raw visible classifier]
\label{thm:rat-history-ledger-policy-shared-raw-visible-classifier}
If both $\mathsf{RatHistoryLedgerPolicy}(\rho,v)$ and
$\mathsf{RatHistoryLedgerPolicy}(\rho,v')$ hold, then
$\mathsf{RatHistoryClassifier}(v,v')$ holds.
\end{theorem}
\begin{proof}
Convert both rational ledger witnesses to raw-visible classifiers. Reverse the
first classifier by rational classifier symmetry, then compose it with the
second raw-visible classifier by rational classifier transitivity.
\end{proof}
\leanchecked{BEDC.Derived.RatUp.RatHistoryLedgerPolicy\_shared\_raw\_visible\_classifier}

\begin{theorem}[Rat ledger shared visible zero-extension exclusion]
\label{thm:rat-ledger-shared-visible-zero-extension-exclusion}
If both $\mathsf{RatHistoryLedgerPolicy}(\rho,v)$ and
$\mathsf{RatHistoryLedgerPolicy}(\rho,v')$ hold, then neither visible
denominator endpoint can be displayed as a zero-extension endpoint:
$$
(\hsame(v,\Ezero(z))\to\mathsf{False})\land
(\hsame(v',\Ezero(z'))\to\mathsf{False}).
$$
\end{theorem}
\begin{proof}
Use the shared-raw visible classifier to obtain
$\mathsf{RatHistoryClassifier}(v,v')$. If $v$ is displayed as $\Ezero(z)$,
transport the left endpoint of this classifier to obtain a rational classifier
with zero-extension source, which contradicts rational classifier
zero-extension endpoint exclusion. The visible endpoint $v'$ is identical:
transport the right endpoint to $\Ezero(z')$ and apply the right branch of the
same exclusion theorem.
\end{proof}
\leanchecked{BEDC.Derived.RatUp.RatHistoryLedgerPolicy\_shared\_visible\_e0\_endpoint\_absurd}

\begin{theorem}[Rat ledger shared-raw contextual visible classifier]
\label{thm:rat-history-ledger-shared-raw-context-visible-classifier}
If both $\mathsf{RatHistoryLedgerPolicy}(\rho,v)$ and
$\mathsf{RatHistoryLedgerPolicy}(\rho,v')$ hold, unary prefixes $p,p_v,p_{v'}$
satisfy $\hsame(p,p_v)$ and $\hsame(p,p_{v'})$, and unary tails
$t,t_v,t_{v'}$ satisfy $\hsame(t,t_v)$ and $\hsame(t,t_{v'})$, then the
contextual visible denominator endpoints are classified:
$$
\mathsf{RatHistoryClassifier}(\mathsf{append}(p_v,\mathsf{append}(v,t_v)),
\mathsf{append}(p_{v'},\mathsf{append}(v',t_{v'}))).
$$
\end{theorem}
\begin{proof}
Apply \autoref{thm:rat-ledger-unary-denominator-context-closed} to the first
ledger with the raw context $p,t$ and the visible context $p_v,t_v$. This
produces a ledger from the shared raw contextual endpoint
$\mathsf{append}(p,\mathsf{append}(\rho,t))$ to the first contextual visible
endpoint.

Apply the same context-closure theorem to the second ledger with the same raw
context $p,t$ and the visible context $p_{v'},t_{v'}$. The two contextual
ledger witnesses now share the raw endpoint
$\mathsf{append}(p,\mathsf{append}(\rho,t))$.

Finally apply
\autoref{thm:rat-history-ledger-policy-shared-raw-visible-classifier} to those
two contextual ledger witnesses. Its conclusion is exactly the displayed
classifier between the two contextual visible denominator endpoints.
\end{proof}
\leanchecked{BEDC.Derived.RatUp.RatHistoryLedgerPolicy\_shared\_raw\_contextual\_visible\_classifier}
\leanchecked{BEDC.Derived.RatUp.RatHistoryLedgerPolicy\_shared\_raw\_context\_visible\_classifier}

\begin{theorem}[Rat ledger shared-raw contextual visible endpoint equivalence]
\label{thm:rat-history-ledger-shared-raw-context-visible-endpoint-equivalence}
If both $\mathsf{RatHistoryLedgerPolicy}(\rho,v)$ and
$\mathsf{RatHistoryLedgerPolicy}(\rho,v')$ hold, unary prefixes $p,p_v,p_{v'}$
satisfy $\hsame(p,p_v)$ and $\hsame(p,p_{v'})$, and unary tails
$t,t_v,t_{v'}$ satisfy $\hsame(t,t_v)$ and $\hsame(t,t_{v'})$, then every
target history $w$ sees the same classifier fiber over the two contextual
visible denominator endpoints:
$$
\mathsf{RatHistoryClassifier}(\mathsf{append}(p_v,\mathsf{append}(v,t_v)),w)
\Longleftrightarrow
\mathsf{RatHistoryClassifier}(\mathsf{append}(p_{v'},\mathsf{append}(v',t_{v'})),w).
$$
\end{theorem}
\leanchecked{BEDC.Derived.RatUp.RatHistoryLedgerPolicy\_shared\_raw\_context\_visible\_endpoint\_equivalence}
\begin{proof}
Apply
\autoref{thm:rat-history-ledger-shared-raw-context-visible-classifier} to the
two ledger witnesses and the matching unary prefix and tail hypotheses. This
gives the classifier edge from
$\mathsf{append}(p_v,\mathsf{append}(v,t_v))$ to
$\mathsf{append}(p_{v'},\mathsf{append}(v',t_{v'}))$.

For the forward direction, assume the left contextual visible endpoint is
classified with $w$. Reverse the contextual endpoint edge by rational classifier
symmetry, then compose it with the left-endpoint-to-$w$ classifier by
transitivity. The result is the right-endpoint-to-$w$ classifier.

For the reverse direction, assume the right contextual visible endpoint is
classified with $w$. Compose the contextual endpoint edge itself with the
right-endpoint-to-$w$ classifier by transitivity. The result is the
left-endpoint-to-$w$ classifier.
\end{proof}

\begin{theorem}[Rat ledger shared-raw context one-tail readback]
\label{thm:rat-history-ledger-shared-raw-context-e1-pair-readback}
If both $\mathsf{RatHistoryLedgerPolicy}(\rho,v)$ and
$\mathsf{RatHistoryLedgerPolicy}(\rho,v')$ hold, unary prefixes
$p,p_v,p_{v'}$ satisfy $\hsame(p,p_v)$ and $\hsame(p,p_{v'})$, unary tails
$t,t_v,t_{v'}$ satisfy $\hsame(t,t_v)$ and $\hsame(t,t_{v'})$, and the two
contextual visible endpoints display as one-extension histories,
$$
\hsame(\mathsf{append}(p_v,\mathsf{append}(v,t_v)),\Eone(a))
\quad\text{and}\quad
\hsame(\mathsf{append}(p_{v'},\mathsf{append}(v',t_{v'})),\Eone(b)),
$$
then the displayed tails are unary and internally the same:
$$
\mathsf{Unary}(a)\land \mathsf{Unary}(b)\land\hsame(a,b).
$$

Its ten-field fit is: traditional target shared-denominator visible-tail
determinacy; BEDC source $\RatUp$ ledger over $\mathsf{BHist}$; carrier positive
unary denominators; classifier $\mathsf{RatHistoryClassifier}$; stability from
unary context closure and classifier transport; ledger the shared raw
denominator plus two visible contexts; core theorem this block; exactness the
displayed tail readback; standard bridge normalized denominator readback; status
$\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
First invoke
\autoref{thm:rat-history-ledger-shared-raw-context-visible-classifier} with the
two ledger witnesses and the matching raw-to-visible unary contexts. This gives
the rational history classifier between the two contextual visible endpoints.

Then apply
\autoref{thm:rat-history-classifier-unary-context-e1-pair-readback} to that
classifier and to the two displayed one-extension witnesses. The readback
conclusion is exactly $\mathsf{Unary}(a)\land \mathsf{Unary}(b)\land
\hsame(a,b)$.
\end{proof}
\leanchecked{BEDC.Derived.RatUp.RatHistoryLedgerPolicy\_shared\_raw\_context\_e1\_pair\_readback}

\begin{theorem}[Rat history ledger classifier endpoint equivalence]
\label{thm:rat-history-ledger-classifier-endpoint-equivalence}
If $\mathsf{RatHistoryLedgerPolicy}(\rho,v)$ holds, then every target history
$w$ is classified from $\rho$ exactly when it is classified from $v$:
$$
\mathsf{RatHistoryClassifier}(\rho,w)\leftrightarrow
\mathsf{RatHistoryClassifier}(v,w).
$$
\end{theorem}
\leanchecked{BEDC.Derived.RatUp.RatHistoryLedgerPolicy\_classifier\_endpoint\_equivalence}
\begin{proof}
The ledger supplies the raw-visible rational classifier. For the raw-to-visible
direction, reverse that classifier by symmetry and compose it with the
raw-target classifier by transitivity. The visible-to-raw direction is the
ledger classifier extension theorem.
\end{proof}

\begin{theorem}[Rat history ledger displayed context classifier endpoint equivalence]
\label{thm:rat-history-ledger-cont-unary-context-classifier-endpoint-equivalence}
If $\mathsf{RatHistoryLedgerPolicy}(\rho,v)$ holds, unary prefixes
$p_\rho,p_v$ satisfy $\hsame(p_\rho,p_v)$, unary tails $t_\rho,t_v$ satisfy
$\hsame(t_\rho,t_v)$, and the displayed continuation witnesses
$\mathsf{Cont}(p_\rho,\rho,m_\rho)$,
$\mathsf{Cont}(m_\rho,t_\rho,out_\rho)$,
$\mathsf{Cont}(p_v,v,m_v)$, and
$\mathsf{Cont}(m_v,t_v,out_v)$ expose the two contextual endpoints, then
every target history $w$ is classified from the displayed raw endpoint exactly
when it is classified from the displayed visible endpoint:
$$
\mathsf{RatHistoryClassifier}(out_\rho,w)\leftrightarrow
\mathsf{RatHistoryClassifier}(out_v,w).
$$
\end{theorem}
\leanchecked{BEDC.Derived.RatUp.RatHistoryLedgerPolicy\_cont\_unary\_context\_classifier\_endpoint\_equivalence}
\begin{proof}
First close the ledger under the matching unary denominator context. Apply
\autoref{thm:rat-ledger-unary-denominator-context-closed} to the original
ledger, the prefix match, and the tail match. This gives a ledger between
$\mathsf{append}(p_\rho,\mathsf{append}(\rho,t_\rho))$ and
$\mathsf{append}(p_v,\mathsf{append}(v,t_v))$.

The four continuation witnesses identify the displayed endpoints with those
normalized append endpoints. Associativity of $\mathsf{append}$ turns the
two-step displays through $m_\rho$ and $m_v$ into $\hsame$ links from
$out_\rho$ and $out_v$ to the corresponding normalized contextual histories.

Transport the contextual ledger across these two endpoint identifications by
\autoref{thm:rat-ledger-hsame-transport}. The transported ledger now relates
$out_\rho$ and $out_v$ directly.

Finally apply
\autoref{thm:rat-history-ledger-classifier-endpoint-equivalence} to the
transported ledger. Its conclusion is the displayed classifier endpoint
equivalence for every target history $w$.
\end{proof}

\begin{theorem}[Rat history ledger displayed context positive denominators]
\label{thm:rat-history-ledger-cont-unary-context-positive-denominators}
If $\mathsf{RatHistoryLedgerPolicy}(\rho,v)$ holds, unary prefixes
$p_\rho,p_v$ satisfy $\hsame(p_\rho,p_v)$, unary tails $t_\rho,t_v$ satisfy
$\hsame(t_\rho,t_v)$, and the displayed continuations
$\mathsf{Cont}(p_\rho,\rho,m_\rho)$,
$\mathsf{Cont}(m_\rho,t_\rho,out_\rho)$,
$\mathsf{Cont}(p_v,v,m_v)$, and
$\mathsf{Cont}(m_v,t_v,out_v)$ expose the contextual endpoints, then
$$
\mathsf{PositiveUnaryDenominator}(out_\rho)\land
\mathsf{PositiveUnaryDenominator}(out_v).
$$
\end{theorem}
\begin{proof}
Close the ledger under matching unary denominator context by applying
\autoref{thm:rat-ledger-unary-denominator-context-closed} to the original
ledger, the prefix match, and the tail match. This gives a ledger between the
two normalized histories
$\mathsf{append}(p_\rho,\mathsf{append}(\rho,t_\rho))$ and
$\mathsf{append}(p_v,\mathsf{append}(v,t_v))$.

The four continuation witnesses identify $out_\rho$ and $out_v$ with the
two displayed two-step continuations through $m_\rho$ and $m_v$. Associativity
of $\mathsf{append}$ turns those two displays into $\hsame$ links from the
normalized contextual histories to the displayed outputs.

Transport the contextual ledger across these two endpoint identifications by
\autoref{thm:rat-ledger-hsame-transport}. The resulting ledger relates
$out_\rho$ and $out_v$ directly.

Finally read the raw and visible carrier witnesses from the displayed ledger.
The carrier equivalence for rational histories gives
$\mathsf{PositiveUnaryDenominator}(out_\rho)$, and the visible-carrier
projection gives $\mathsf{PositiveUnaryDenominator}(out_v)$.
\end{proof}
\leanchecked{BEDC.Derived.RatUp.RatHistoryLedgerPolicy\_cont\_unary\_context\_positive\_denominators}

\begin{theorem}[Rat ledger displayed context zero-endpoint exclusion]
\label{thm:rat-history-ledger-cont-unary-context-e0-endpoint-absurd}
If $\mathsf{RatHistoryLedgerPolicy}(\rho,v)$ holds, unary prefixes
$p_\rho,p_v$ satisfy $\hsame(p_\rho,p_v)$, unary tails $t_\rho,t_v$ satisfy
$\hsame(t_\rho,t_v)$, and the displayed continuations
$\mathsf{Cont}(p_\rho,\rho,m_\rho)$,
$\mathsf{Cont}(m_\rho,t_\rho,out_\rho)$,
$\mathsf{Cont}(p_v,v,m_v)$, and
$\mathsf{Cont}(m_v,t_v,out_v)$ expose the contextual endpoints, then neither
displayed endpoint can be a zero-extension endpoint:
$$
(\hsame(out_\rho,\Ezero(z_\rho))\to\mathsf{False})\land
(\hsame(out_v,\Ezero(z_v))\to\mathsf{False}).
$$
\end{theorem}
\begin{proof}
Apply
\autoref{thm:rat-history-ledger-cont-unary-context-positive-denominators} to
the displayed contextual ledger data. This gives positive unary denominator
witnesses for both $out_\rho$ and $out_v$.

If $out_\rho$ is displayed as $\Ezero(z_\rho)$, transport the positive
denominator witness along that $\hsame$ link and contradict
\autoref{thm:rat-positive-denominator-e0-absurd}. The visible endpoint
$out_v$ is identical: transport its positive denominator witness along
$\hsame(out_v,\Ezero(z_v))$ and use the same zero-extension exclusion.
\end{proof}
\leanchecked{BEDC.Derived.RatUp.RatHistoryLedgerPolicy\_cont\_unary\_context\_e0\_endpoint\_absurd}
