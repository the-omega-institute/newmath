\begin{definition}[Rat ledger policy]
\label{def:rat-ledger-policy}
Record the raw rational-history carrier and an $\hsame$ link to the visible denominator history.
\end{definition}
\leandef{BEDC.Derived.RatUp.RatHistoryLedgerPolicy}

\begin{theorem}[Rat ledger visible carrier]
\label{thm:rat-ledger-visible-carrier}
If a raw rational history satisfies the ledger policy for a visible history, then the visible history is itself a rational-history carrier.
\end{theorem}
\leanchecked{BEDC.Derived.RatUp.RatHistoryLedgerPolicy\_visible\_carrier}
\begin{proof}
Unpack the ledger policy into a raw rational-history carrier and an $\hsame$ link from the raw history to the visible history. Transport the raw carrier across that link.
\end{proof}

\begin{theorem}[Rat ledger visible positive denominator readback]
\label{thm:rat-ledger-visible-positive-denominator-readback}
If $\mathsf{RatHistoryLedgerPolicy}(\rho,v)$ holds and
$\hsame(v,\mathsf{e1}(t))$ holds, then $\mathsf{UnaryHistory}(t)$ holds.
\end{theorem}
\leanchecked{BEDC.Derived.RatUp.RatHistoryLedgerPolicy\_visible\_positive\_denominator\_readback}
\begin{proof}
The ledger visible-carrier theorem transports the raw rational-history carrier
to $v$. Transport that carrier once more across $\hsame(v,\mathsf{e1}(t))$ and
apply the one-extension rational-history carrier characterization.
\end{proof}

\begin{theorem}[Rat ledger append unary denominator closure]
\label{thm:rat-ledger-append-unary-denominator-closed}
If $\mathsf{RatHistoryLedgerPolicy}(\rho,v)$ holds, $\mathsf{UnaryHistory}(t)$
holds, and $\hsame(t,t')$ holds, then appending those unary denominator tails
preserves the rational ledger:
$$
\mathsf{RatHistoryLedgerPolicy}(\mathsf{append}(\rho,t),\mathsf{append}(v,t')).
$$
\end{theorem}
\leanchecked{BEDC.Derived.RatUp.RatHistoryLedgerPolicy\_append\_unary\_denominator\_closed}
\begin{proof}
Unpack the ledger into a raw rational-history carrier and an $\hsame(\rho,v)$
link. The denominator-append carrier theorem closes the raw endpoint under the
unary tail, and the two $\hsame$ links identify the appended raw and visible
histories.
\end{proof}

\begin{theorem}[Rat ledger prepend unary denominator closure]
\label{thm:rat-ledger-prepend-unary-denominator-closed}
If $\mathsf{RatHistoryLedgerPolicy}(\rho,v)$ holds,
$\mathsf{UnaryHistory}(p)$ holds, and $\hsame(p,p')$ holds, then prefixing
those unary denominator histories preserves the rational ledger:
$$
\mathsf{RatHistoryLedgerPolicy}(\mathsf{append}(p,\rho),\mathsf{append}(p',v)).
$$
\end{theorem}
\leanchecked{BEDC.Derived.RatUp.RatHistoryLedgerPolicy\_prepend\_unary\_denominator\_closed}
\begin{proof}
Unpack the ledger into a raw rational-history carrier and an $\hsame(\rho,v)$
link. The denominator-prefix carrier theorem closes the raw endpoint under the
unary prefix, and the two $\hsame$ links identify the prefixed raw and visible
histories.
\end{proof}

\begin{theorem}[Rat ledger unary denominator context closure]
\label{thm:rat-ledger-unary-denominator-context-closed}
If $\mathsf{RatHistoryLedgerPolicy}(\rho,v)$ holds, unary prefixes $p,p'$
satisfy $\hsame(p,p')$, and unary tails $t,t'$ satisfy $\hsame(t,t')$, then
placing the matching prefix and tail around the raw and visible denominator
histories preserves the rational ledger:
$$
\mathsf{RatHistoryLedgerPolicy}(\mathsf{append}(p,\mathsf{append}(\rho,t)),
\mathsf{append}(p',\mathsf{append}(v,t'))).
$$
\end{theorem}
\leanchecked{BEDC.Derived.RatUp.RatHistoryLedgerPolicy\_unary\_denominator\_context\_closed}
\begin{proof}
First apply rat ledger append unary denominator closure to the displayed ledger
and the matching tails. Then apply rat ledger prepend unary denominator closure
to the resulting ledger and the matching prefixes.
\end{proof}

\begin{theorem}[Rat ledger history-sameness transport]
\label{thm:rat-ledger-hsame-transport}
If a rational ledger relates raw and visible histories, then transporting the
raw endpoint and the visible endpoint along \(\hsame\) yields another rational
ledger relation.
\end{theorem}
\leanchecked{BEDC.Derived.RatUp.RatHistoryLedgerPolicy\_hsame\_transport}
\begin{proof}
Unpack the ledger into a raw carrier and a raw-visible link. Transport the raw
carrier across the raw-endpoint link, then compose the reversed raw transport
link, the stored ledger link, and the visible-endpoint link.
\end{proof}

\begin{theorem}[Rat ledger raw-visible classifier]
\label{thm:rat-ledger-raw-visible-classifier}
If $\mathsf{RatHistoryLedgerPolicy}(\rho,v)$ holds, then
$\mathsf{RatHistoryClassifier}(\rho,v)$ holds.
\end{theorem}
\leanchecked{BEDC.Derived.RatUp.RatHistoryLedgerPolicy\_raw\_visible\_classifier}
\begin{proof}
Unpack the ledger into the raw rational-history carrier for $\rho$ and the
stored $\hsame(\rho,v)$ link. The visible-carrier theorem supplies the
rational-history carrier for $v$, and these witnesses form the raw-visible
rational history classifier.
\end{proof}

\begin{theorem}[Rat ledger shared-raw visible classifier]
\label{thm:rat-history-ledger-policy-shared-raw-visible-classifier}
If both $\mathsf{RatHistoryLedgerPolicy}(\rho,v)$ and
$\mathsf{RatHistoryLedgerPolicy}(\rho,v')$ hold, then
$\mathsf{RatHistoryClassifier}(v,v')$ holds.
\end{theorem}
\begin{proof}
Convert both rational ledger witnesses to raw-visible classifiers. Reverse the
first classifier by rational classifier symmetry, then compose it with the
second raw-visible classifier by rational classifier transitivity.
\end{proof}
\leanchecked{BEDC.Derived.RatUp.RatHistoryLedgerPolicy\_shared\_raw\_visible\_classifier}

\begin{theorem}[Rat ledger classifier extension]
\label{thm:rat-ledger-classifier-extension}
If $\mathsf{RatHistoryLedgerPolicy}(\rho,v)$ holds and
$\mathsf{RatHistoryClassifier}(v,t)$ holds, then
$\mathsf{RatHistoryClassifier}(\rho,t)$ holds.
\end{theorem}
\leanchecked{BEDC.Derived.RatUp.RatHistoryLedgerPolicy\_classifier\_extension}
\begin{proof}
Convert the raw-visible ledger into the raw-visible classifier, then compose it
with the visible-target classifier by transitivity.
\end{proof}

\begin{theorem}[Rat history ledger classifier endpoint equivalence]
\label{thm:rat-history-ledger-classifier-endpoint-equivalence}
If $\mathsf{RatHistoryLedgerPolicy}(\rho,v)$ holds, then every target history
$w$ is classified from $\rho$ exactly when it is classified from $v$:
$$
\mathsf{RatHistoryClassifier}(\rho,w)\leftrightarrow
\mathsf{RatHistoryClassifier}(v,w).
$$
\end{theorem}
\leanchecked{BEDC.Derived.RatUp.RatHistoryLedgerPolicy\_classifier\_endpoint\_equivalence}
\begin{proof}
The ledger supplies the raw-visible rational classifier. For the raw-to-visible
direction, reverse that classifier by symmetry and compose it with the
raw-target classifier by transitivity. The visible-to-raw direction is the
ledger classifier extension theorem.
\end{proof}
