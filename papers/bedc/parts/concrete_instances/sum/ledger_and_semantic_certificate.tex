\begin{definition}[Sum ledger policy]
\label{def:sum-ledger-policy}
Record a raw sum history carrying the selected injection tag and source ledger
together with an $\hsame$ link to the visible history exposed by the ledger.
\end{definition}
\leandef{BEDC.Derived.SumUp.SumHistoryLedgerPolicy}

\begin{theorem}[Sum ledger visible carrier]
\label{thm:sum-ledger-visible-carrier}
If a visible history is linked by $\hsame$ to a raw sum-ledger history, then
the visible history is itself a sum carrier.
\end{theorem}
\begin{proof}
Unpack the raw sum carrier and transport it across the stored $\hsame$ link
by sum-carrier transport.
\end{proof}
\leanchecked{BEDC.Derived.SumUp.SumHistoryLedgerPolicy\_visible\_carrier}

\begin{theorem}[Sum ledger raw-visible classifier]
\label{thm:sum-ledger-raw-visible-classifier}
If $\mathsf{SumHistoryLedgerPolicy}(L,R,\rho,v)$ holds, then
$\mathsf{SumHistoryClassifier}(L,R,\hsame,\hsame,\rho,v)$ holds.
\end{theorem}
\begin{proof}
Unpack the ledger into its raw sum carrier and the stored $\hsame(\rho,v)$
link. The raw carrier determines whether $\rho$ is left-tagged or right-tagged;
transport the tag witness across the stored link and use reflexivity of
$\hsame$ on the selected component to build the matching classifier branch.
\end{proof}
\leanchecked{BEDC.Derived.SumUp.SumHistoryLedgerPolicy\_raw\_visible\_classifier}

\begin{theorem}[Sum ledger history-same transport]
\label{thm:sum-ledger-history-same-transport}
Let $Left,Right:\mathsf{BHist}\to\mathsf{Prop}$ be source carriers and let
$raw,raw',visible,visible':\mathsf{BHist}$ be histories. If
$\mathsf{SumHistoryLedgerPolicy}(Left,Right,raw,visible)$,
$\hsame(raw,raw')$, and $\hsame(visible,visible')$ hold, then
$$
  \mathsf{SumHistoryLedgerPolicy}(Left,Right,raw',visible')
$$
holds.
\end{theorem}
\begin{proof}
Unpack the ledger into the raw $\mathsf{SumHistoryCarrier}$ witness and the
stored link $\hsame(raw,visible)$. Transport the raw carrier along
$\hsame(raw,raw')$ by
Theorem~\ref{thm:sum-history-carrier-hsame-transport}. For the raw-visible
link of the transported ledger, compose symmetry of $\hsame(raw,raw')$, the
stored $\hsame(raw,visible)$ link, and $\hsame(visible,visible')$ to obtain
$\hsame(raw',visible')$.
\end{proof}

\begin{theorem}[Sum ledger two-step branch exactness]
\label{thm:sum-ledger-two-step-branch-exactness}
Let $Left,Right:\mathsf{BHist}\to\mathsf{Prop}$ be source carriers and let
$\rho,v,w:\mathsf{BHist}$ be histories. If
$\mathsf{SumHistoryLedgerPolicy}(Left,Right,\rho,v)$ and
$\mathsf{SumHistoryLedgerPolicy}(Left,Right,v,w)$ hold, then the whole
two-step chain has a visible same-branch payload readback:
$$
\bigl(\exists l\,l':\mathsf{BHist}.\; Left(l)\land Left(l')\land\hsame(\rho,\Ezero(l))\land\hsame(w,\Ezero(l'))\land\hsame(l,l')\bigr)
\lor
\bigl(\exists r\,r':\mathsf{BHist}.\; Right(r)\land Right(r')\land\hsame(\rho,\Eone(r))\land\hsame(w,\Eone(r'))\land\hsame(r,r')\bigr).
$$
The ten-field fit is: traditional target disjoint-sum ledger-chain readback;
BEDC source $\mathsf{BHist}$ with $\hsame$; carrier
$\mathsf{SumHistoryCarrier}$; classifier unused in the final exactness statement
but supplied by Theorem~\ref{thm:sum-ledger-two-step-classifier-composition};
ledger two composable $\mathsf{SumHistoryLedgerPolicy}$ witnesses; core theorem
this branch exactness statement; exactness raw-to-visible same-branch payload
readback; standard bridge tagged sum ledger chains; status
$\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:sum-ledger-raw-visible-branch-exactness} to each ledger
witness. The first application exposes either a left payload visible at
$\rho$ and $v$, or a right payload visible at $\rho$ and $v$; the second exposes
the corresponding alternatives at $v$ and $w$. Split the four branch cases. In
the left-left case, use Theorem~\ref{thm:sum-visible-branch-payload-determinism}
at the shared middle endpoint $v$ to identify the two left payloads and assemble
the left disjunct. The right-right case is identical with the right branch.
The mixed left-right and right-left cases contradict generated-history
no-confusion between $\Ezero$ and $\Eone$ at the shared middle endpoint $v$.
\end{proof}
\leanchecked{BEDC.Derived.SumUp.SumHistoryLedgerPolicy\_two\_step\_branch\_exactness}

\begin{theorem}[Sum ledger two-step endpoint exactness]
\label{thm:sum-ledger-two-step-endpoint-exactness}
Let $Left,Right:\mathsf{BHist}\to\mathsf{Prop}$ be source carriers and let
$\rho,v,w,l,l',r,r':\mathsf{BHist}$ be histories. If
$\mathsf{SumHistoryLedgerPolicy}(Left,Right,\rho,v)$ and
$\mathsf{SumHistoryLedgerPolicy}(Left,Right,v,w)$ hold, then the two-ledger
chain has endpoint exactness from $\rho$ to $w$. First, both mixed endpoint
pairs are impossible:
$$
(\hsame(\rho,\Ezero(l))\land \hsame(w,\Eone(r')))\to\bot
$$
and
$$
(\hsame(\rho,\Eone(r))\land \hsame(w,\Ezero(l')))\to\bot.
$$
Second, same-tag endpoint pairs determine their payloads:
$$
(\hsame(\rho,\Ezero(l))\land\hsame(w,\Ezero(l')))\to\hsame(l,l')
$$
and
$$
(\hsame(\rho,\Eone(r))\land\hsame(w,\Eone(r')))\to\hsame(r,r').
$$
Third, final visible endpoints of $w$ read back source payload evidence:
$$
(\hsame(w,\Ezero(l'))\to\exists a:\mathsf{BHist}.\;Left(a)\land\hsame(l',a))
$$
and
$$
(\hsame(w,\Eone(r'))\to\exists b:\mathsf{BHist}.\;Right(b)\land\hsame(r',b)).
$$
\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:sum-ledger-two-step-classifier-composition} to the two
ledger witnesses and obtain
$\mathsf{SumHistoryClassifier}(Left,Right,\hsame,\hsame,\rho,w)$. The two mixed
endpoint conclusions follow from
Theorem~\ref{thm:sum-history-classifier-tag-separation}; for the reversed
displayed order, use classifier symmetry from
Theorem~\ref{thm:sum-history-classifier-hsame-symmetry} before applying the same
tag-separation theorem. The left same-tag conclusion is exactly the left
endpoint inversion supplied by
Theorem~\ref{thm:sum-history-classifier-left-hsame-inversion}, and the right
same-tag conclusion is the corresponding right inversion from
Theorem~\ref{thm:sum-history-classifier-right-hsame-inversion}. For the final
visible readback, apply Theorem~\ref{thm:sum-ledger-visible-carrier} to the
second ledger witness, obtaining a visible carrier for $w$; then apply
Theorem~\ref{thm:sum-history-carrier-visible-tag-exactness} to the displayed
$\Ezero$ or $\Eone$ endpoint of $w$ to extract the matching source payload
evidence.
\end{proof}
\leanchecked{BEDC.Derived.SumUp.SumHistoryLedgerPolicy\_two\_step\_endpoint\_exactness}

\begin{definition}[Sum ledger finite provenance chain]
\label{def:sum-history-ledger-chain}
For source carriers $Left,Right:\mathsf{BHist}\to\mathsf{Prop}$, define
$\mathsf{SumHistoryLedgerChain}_{Left,Right}(\rho,z)$ as the finite provenance
relation generated by the following two clauses. A displayed one-ledger witness
$\mathsf{SumHistoryLedgerPolicy}(Left,Right,\rho,z)$ gives
$\mathsf{SumHistoryLedgerChain}_{Left,Right}(\rho,z)$. If there is a history
$v:\mathsf{BHist}$ such that
$\mathsf{SumHistoryLedgerPolicy}(Left,Right,\rho,v)$ and
$\mathsf{SumHistoryLedgerChain}_{Left,Right}(v,z)$ hold, then
$\mathsf{SumHistoryLedgerChain}_{Left,Right}(\rho,z)$ also holds.
\end{definition}
\leandef{BEDC.Derived.SumUp.SumHistoryLedgerChain}

\begin{theorem}[Sum ledger chain classifier composition]
\label{thm:sum-ledger-chain-classifier-composition}
\label{thm:sum-history-ledger-chain-classifier-composition}
Let $Left,Right:\mathsf{BHist}\to\mathsf{Prop}$ be source carriers and let
$raw,visible:\mathsf{BHist}$ be histories. If

\mathsf{SumHistoryClassifier}(Left,Right,\hsame,\hsame,raw,visible)

\end{proof}

$$
(\hsame(\rho,\Ezero(l))\land\hsame(z,\Eone(r')))\to\bot
$$
and
$$
(\hsame(\rho,\Eone(r))\land\hsame(z,\Ezero(l')))\to\bot.

Second, same-tag endpoint pairs determine their payloads:
$$
(\hsame(\rho,\Ezero(l))\land\hsame(z,\Ezero(l')))\to\hsame(l,l')
$$
and
$$
(\hsame(\rho,\Eone(r))\land\hsame(z,\Eone(r')))\to\hsame(r,r').

evidence:
$$
(\hsame(z,\Ezero(l'))\to\exists a:\mathsf{BHist}.\;Left(a)\land\hsame(l',a))

and
$$
(\hsame(z,\Eone(r'))\to\exists b:\mathsf{BHist}.\;Right(b)\land\hsame(r',b)).
$$
The ten-field fit is: traditional target disjoint-sum finite ledger provenance;

$\mathsf{SumHistoryClassifier}(Left,Right,\hsame,\hsame)$ supplied one ledger at

$\mathsf{SumHistoryLedgerChain}_{Left,Right}$ generated by
$\mathsf{SumHistoryLedgerPolicy}$; core theorem this chain endpoint exactness

\begin{proof}

\begin{theorem}[Concrete $\SumUp$ semantic name certificate]
\label{thm:concrete-sum-namecert}
Let $Left,Right:\mathsf{BHist}\to\mathsf{Prop}$ be source carriers with classifiers
$LeftEq$ and $RightEq$. If $\NameCert(Left,LeftEq)$ and
$\NameCert(Right,RightEq)$ are supplied, then the tagged sum carrier, the tagged
sum classifier, and the carrier-as-pattern and carrier-as-ledger tuple form
$$
\begin{array}{l}
\mathsf{SemanticNameCert}( \\
\quad \mathsf{SumHistoryCarrier}(Left,Right), \\
\quad \mathsf{SumHistoryCarrier}(Left,Right), \\
\quad \mathsf{SumHistoryCarrier}(Left,Right), \\
\quad \mathsf{SumHistoryClassifier}(Left,Right,LeftEq,RightEq)).
\end{array}
$$
\end{theorem}
\begin{proof}
The carrier is represented by $e0$-tagged left histories and $e1$-tagged right histories. Theorem~\ref{thm:sum-classifier-no-source-membership} marks the split: the classifier delegates within matching tags to the corresponding source certificate, while source-carrier evidence is taken from $\mathsf{SumHistoryCarrier}$ when certificate fields require it. Reflexivity is built by tagging the source certificate reflexivity field on the selected component. Symmetry and transitivity invert the tagged classifier witnesses, use the matching source certificate fields inside a left--left or right--right branch, and discharge mixed tags by constructor no-confusion.

The core certificate fields are assembled from
Theorem~\ref{thm:sum-history-classifier-source-certified-stability-fields}.
That theorem supplies carrier-relative reflexivity, source-certified symmetry,
source-certified transitivity, and classifier-induced carrier transport for the
tagged carrier and tagged classifier. The inhabited field is separate: take the
inhabited witness $l$ from the left source certificate, choose the visible
endpoint $\Ezero(l)$, and rebuild
$\mathsf{SumHistoryCarrier}(Left,Right,\Ezero(l))$ by the left tagged
constructor with the reflexive endpoint witness and the stored $Left(l)$
evidence.

Inside the carrier-transport field supplied by the cited theorem, source-carrier
membership is not transported along an arbitrary source $\NameCert$ merely from
$\hsame(l,a)$. In the left branch, constructor inversion first aligns the tagged
payloads and obtains $\hsame(l,a)$; the checked finite-kernel identity
$\mathsf{BEDC.FKernel.Hist.hsame\_iff\_eq}$ converts that history-sameness
witness to kernel equality before the stored $Left(l)$ evidence is rewritten as
$Left(a)$. Only after this equality rewrite does
$\NameCert.carrier\_respects\_equiv$ from the left source certificate apply
along the displayed source classifier witness $LeftEq(a,b)$, yielding
$Left(b)$ and hence the target left tagged carrier. The right branch is the same
with $RightEq(a,b)$ and the right source certificate. Pattern soundness and
ledger soundness are identity implications, since both semantic fields are the
same predicate $\mathsf{SumHistoryCarrier}(Left,Right)$.
\end{proof}
\leanchecked{BEDC.Derived.SumUp.sum\_history\_semantic\_name\_certificate}
