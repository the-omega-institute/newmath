\chapter{A Concrete Naming Certificate for $\MagmaUp$}
\label{ch:concrete-instances-magma-namecert}

The magma interface $\MagmaUp$ packages a carrier endowed with a single binary operation, with no associativity, identity, or inverse demands. It is the smallest algebraic naming certificate above the inductive carriers of \autoref{ch:concrete-instances-bool-namecert}--\autoref{ch:concrete-instances-list-namecert} and the entry point of the algebraic ladder culminating in $\NameCert_{\FieldUp}$ (\autoref{ch:concrete-instances-field-namecert}).

\section{Carrier definition}
\label{sec:magma-carrier}

\begin{definition}[Concrete unary-history magma instance]
\label{def:concrete-unary-history-magma-instance}
Let $U(h)$ abbreviate $\mathsf{UnaryHistory}(h)$. The concrete unary-history
magma instance has carrier $U$, binary operation
$\mathsf{mul}_U(h,k):=\mathsf{append}(h,k)$, equivalently witnessed by
$\Cont(h,k,\mathsf{append}(h,k))$, and carrier-aware classifier
$$
R_U(h,k):=U(h)\land U(k)\land\hsame(h,k).
$$
The classifier only compares unary endpoints, so every public equality claim
retains both endpoint carrier witnesses.
\end{definition}

\begin{theorem}[Concrete unary-history magma laws]
\label{thm:concrete-unary-history-magma-laws}
For the instance of \autoref{def:concrete-unary-history-magma-instance},
$\mathsf{mul}_U$ is closed on unary histories. Moreover, if $R_U(h,h')$ and
$R_U(k,k')$, then
$$
R_U(\mathsf{mul}_U(h,k),\mathsf{mul}_U(h',k')).
$$
\end{theorem}
\begin{proof}
Closure is exactly
\texttt{BEDC.FKernel.Unary.unary\_append\_closed}: two unary endpoint
witnesses for $h$ and $k$ produce a unary endpoint witness for
$\mathsf{append}(h,k)$.

For congruence, unpack $R_U(h,h')$ and $R_U(k,k')$ into unary carrier
witnesses and endpoint $\hsame$ proofs. The two products are witnessed by
their canonical continuation facts
$\Cont(h,k,\mathsf{append}(h,k))$ and
$\Cont(h',k',\mathsf{append}(h',k'))$. Transport the continuation result
along the endpoint $\hsame$ data using
\texttt{BEDC.FKernel.Cont.cont\_result\_hsame\_transport}; the closure clause
supplies the carrier witnesses for both products, and the transported result
is the classifier equality required by $R_U$.
\end{proof}
\leanchecked{BEDC.Derived.MagmaUp.concrete\_unary\_history\_magma\_cont\_laws}

\begin{theorem}[Concrete unary-history magma continuation result unary exactness]
\label{thm:concrete-unary-history-magma-cont-result-unary}
For histories $h,k,r:\mathsf{BHist}$, if $\ContR(h,k,r)$, then the
continuation result is unary exactly when both continuation inputs are unary:
$$
\Unary(r)\Longleftrightarrow \Unary(h)\land\Unary(k).
$$
\end{theorem}
\leanchecked{BEDC.Derived.MagmaUp.concrete\_unary\_history\_magma\_cont\_result\_unary\_iff}
\begin{proof}
The continuation ledger identifies $r$ with $\mathsf{append}(h,k)$. The
unary append factor theorem says precisely that
$\Unary(h)\land\Unary(k)$ is equivalent to
$\Unary(\mathsf{append}(h,k))$. Transporting along the continuation equality
gives the displayed result-side exactness.
\end{proof}

\begin{theorem}[Concrete unary-history magma classifier append factors]
\label{thm:concrete-unary-history-magma-classifier-append-factors}
For histories $h,h',k,k':\mathsf{BHist}$, the concrete unary-history magma
classifier relates the two appended endpoints exactly when all four factors
are unary histories and the two appended endpoints are history-same:
$$
R_U(\mathsf{append}(h,k),\mathsf{append}(h',k'))
\Longleftrightarrow
U(h)\land U(k)\land U(h')\land U(k')
\land
\hsame(\mathsf{append}(h,k),\mathsf{append}(h',k')).
$$
\end{theorem}
\begin{proof}
Unfold $R_U$. The forward direction gives unary carrier witnesses for the two
appended endpoints. Apply unary append factor exactness to each carrier
witness, obtaining the unary witnesses for $h,k,h',k'$, and retain the
classifier's endpoint $\hsame$ component. Conversely, the four unary factor
witnesses close both appended endpoints under unary append closure, and the
displayed endpoint $\hsame$ proof repackages the concrete classifier.
\end{proof}
\leanchecked{BEDC.Derived.MagmaUp.concrete\_unary\_history\_magma\_classifier\_append\_factors\_iff}

\begin{theorem}[Concrete unary-history magma classifier append-empty endpoints]
\label{thm:concrete-unary-history-magma-classifier-append-empty-endpoints}
Let $U(h)$ abbreviate $\mathsf{UnaryHistory}(h)$ and let
$$
R_U(h,k):=U(h)\land U(k)\land\hsame(h,k).
$$
For histories $h,k:\mathsf{BHist}$, the concrete unary-history magma
classifier relates $\mathsf{append}(h,k)$ to $\emp$ exactly when both append
endpoints are empty:
$$
R_U(\mathsf{append}(h,k),\emp)
\Longleftrightarrow
h\hsame\emp\land k\hsame\emp.
$$
\end{theorem}
\begin{proof}
Unfold $R_U$. The forward direction reads
$\mathsf{append}(h,k)\hsame\emp$ from the classifier and applies append-empty
exactness to recover $h\hsame\emp$ and $k\hsame\emp$. Conversely, the two empty
endpoints make the append endpoint unary and history-same to $\emp$, while
$\emp$ itself is unary.
\end{proof}
\leanchecked{BEDC.Derived.MagmaUp.concrete\_unary\_history\_magma\_classifier\_append\_empty\_endpoints\_iff}

\begin{theorem}[Concrete unary-history magma continuation result classifier factors]
\label{thm:concrete-unary-history-magma-cont-result-classifier-factors}
For histories $h,h',k,k',r,r':\mathsf{BHist}$, if
$\Cont(h,k,r)$ and $\Cont(h',k',r')$, then the concrete unary-history magma
classifier relates the two continuation results exactly when all four source
factors are unary and the two results are history-same:
$$
R_U(r,r')
\Longleftrightarrow
U(h)\land U(k)\land U(h')\land U(k')\land\hsame(r,r').
$$
\end{theorem}
\begin{proof}
Eliminate the two continuation witnesses. The results become the displayed
append endpoints $\mathsf{append}(h,k)$ and $\mathsf{append}(h',k')$. The
append-factor classifier exactness theorem then supplies the four unary factor
witnesses in the forward direction and rebuilds the result classifier from the
four witnesses and the endpoint $\hsame$ proof in the reverse direction.
\end{proof}
\leanchecked{BEDC.Derived.MagmaUp.concrete\_unary\_history\_magma\_cont\_result\_classifier\_factors\_iff}

\begin{theorem}[Concrete unary-history magma continuation left-context classifier]
\label{thm:concrete-unary-history-magma-cont-left-context-classifier}
Let $U(h)$ abbreviate $\mathsf{UnaryHistory}(h)$ and let
$$
R_U(h,k):=U(h)\land U(k)\land\hsame(h,k).
$$
If $R_U(l,l')$, $\ContR(l,r,a)$, and $\ContR(l',r',a')$, then the concrete
magma classifier relates the two continuation results exactly when the two
right inputs are unary and history-same:
$$
R_U(a,a')
\Longleftrightarrow
U(r)\land U(r')\land\hsame(r,r').
$$
\end{theorem}
\begin{proof}
In the forward direction, unary exactness for each continuation recovers the
two right-input carriers from the result carriers. The classifier equality
between results cancels the shared classified left context, giving
$r\hsame r'$. Conversely, unary closure for the two continuations supplies the
result carriers, and continuation respect for $\hsame$ transports the left
classifier and the right-input comparison to the result classifier.
\end{proof}
\leanchecked{BEDC.Derived.MagmaUp.concrete\_unary\_history\_magma\_cont\_left\_context\_classifier\_iff}

\begin{theorem}[Concrete unary-history magma classifier middle-context cancellation]
\label{thm:concrete-unary-history-magma-classifier-append-middle-cancel}
Let $U(h)$ abbreviate $\mathsf{UnaryHistory}(h)$ and let
$$
R_U(h,k):=U(h)\land U(k)\land\hsame(h,k).
$$
If $U(l)$ and $U(r)$, then the common append context cancels from the concrete
magma classifier:
$$
R_U(\mathsf{append}(l,\mathsf{append}(m,r)),
    \mathsf{append}(l,\mathsf{append}(m',r)))
\Longleftrightarrow
U(m)\land U(m')\land \hsame(m,m').
$$
\end{theorem}
\begin{proof}
The forward direction first applies unary append-factor exactness to the two
classified contextual endpoints, exposing unary witnesses for the two middle
histories. The classifier equality then cancels the common left append context
and the common right append context, yielding $\hsame(m,m')$. Conversely, unary
closure rebuilds both contextual endpoints, and the middle equality transports
through the two fixed append contexts.
\end{proof}
\leanchecked{BEDC.Derived.MagmaUp.concrete\_unary\_history\_magma\_classifier\_append\_middle\_cancel\_iff}

\begin{theorem}[Concrete unary-history magma classifier rotated middle cancellation]
\label{thm:concrete-unary-history-magma-classifier-append-rotated-middle-cancel}
Let $U(h)$ abbreviate $\mathsf{UnaryHistory}(h)$ and let
$$
R_U(h,k):=U(h)\land U(k)\land\hsame(h,k).
$$
If $U(l)$ and $U(r)$, then the concrete magma classifier also cancels a common
context when the right endpoint is bracketed as a left append followed by the
right context:
$$
R_U(\mathsf{append}(l,\mathsf{append}(m,r)),
    \mathsf{append}(\mathsf{append}(l,m'),r))
\Longleftrightarrow
U(m)\land U(m')\land \hsame(m,m').
$$
\end{theorem}
\begin{proof}
Append associativity identifies the rotated right endpoint with the nested
right endpoint used by middle-context cancellation. The forward direction
transports the classifier to that nested bracketing and applies the
middle-cancellation theorem. Conversely, the nested classifier supplied by
middle cancellation is transported back across associativity.
\end{proof}
\leanchecked{BEDC.Derived.MagmaUp.concrete\_unary\_history\_magma\_classifier\_append\_rotated\_middle\_cancel\_iff}

\begin{theorem}[Concrete unary-history magma continuation common-context classifier]
\label{thm:concrete-unary-history-magma-cont-common-context-classifier}
Let $U(h)$ abbreviate $\mathsf{UnaryHistory}(h)$ and let
$$
R_U(h,k):=U(h)\land U(k)\land\hsame(h,k).
$$
For histories $l,l',m,m',r,r',a,a':\mathsf{BHist}$, if
$R_U(l,l')$ and $R_U(r,r')$, then continuations through the two classified
contexts expose exactly the middle classifier:
$$
\ContR(l,\mathsf{append}(m,r),a)\land
\ContR(l',\mathsf{append}(m',r'),a')
\Longrightarrow
\bigl(R_U(a,a')\Longleftrightarrow
  U(m)\land U(m')\land\hsame(m,m')\bigr).
$$
\end{theorem}
\begin{proof}
The context classifiers identify the left contexts and the right contexts
while carrying all four endpoints as unary histories. Eliminating the two
continuation witnesses turns the results into the two displayed append
contexts. The middle-context cancellation theorem then cancels the common
unary left context and common unary right context, giving exactly the two
middle carrier witnesses and the middle $\hsame$ comparison; the reverse
direction rebuilds the contextual classifier by unary append closure.
\end{proof}
\leanchecked{BEDC.Derived.MagmaUp.concrete\_unary\_history\_magma\_cont\_common\_context\_classifier\_iff}

\begin{theorem}[Concrete unary-history magma nested continuation common-context classifier]
\label{thm:concrete-unary-history-magma-cont-nested-common-context-classifier}
Let $U(h)$ abbreviate $\mathsf{UnaryHistory}(h)$ and let
$$
R_U(h,k):=U(h)\land U(k)\land\hsame(h,k).
$$
For histories $l,l',m,m',r,r',lm,lm',a,a':\mathsf{BHist}$, if
$R_U(l,l')$ and $R_U(r,r')$, and if the two results are obtained through
nested continuations
$$
\ContR(l,m,lm),\quad \ContR(lm,r,a),\quad
\ContR(l',m',lm'),\quad \ContR(lm',r',a'),
$$
then the result classifier exposes exactly the middle classifier:
$$
R_U(a,a')
\Longleftrightarrow
U(m)\land U(m')\land\hsame(m,m').
$$
\end{theorem}
\begin{proof}
The context classifiers identify the two left contexts and the two right
contexts while retaining their unary carrier witnesses. Eliminating the nested
continuation witnesses presents the two results as left-associated append
contexts. Associativity transports those endpoints to the common middle-context
form, where the concrete magma middle-cancellation theorem cancels the fixed
unary left and right contexts. The reverse direction rebuilds the two
left-associated results by transporting the same classifier back across
associativity.
\end{proof}
\leanchecked{BEDC.Derived.MagmaUp.concrete\_unary\_history\_magma\_cont\_nested\_common\_context\_classifier\_iff}

\section{The certificate}

\begin{lemma}[Concrete unary-history magma classifier stability]
\label{lem:concrete-unary-history-magma-classifier-stability}
Let $U(h)$ abbreviate $\mathsf{UnaryHistory}(h)$ and define
$$
R_U(h,k):=U(h)\land U(k)\land\hsame(h,k).
$$
Then $R_U$ has carrier reflexivity on every $U(h)$, symmetry,
transitivity, and carrier transport in both endpoints. Concretely:
\begin{enumerate}
\item if $U(h)$, then $R_U(h,h)$;
\item if $R_U(h,k)$, then $R_U(k,h)$;
\item if $R_U(h,k)$ and $R_U(k,l)$, then $R_U(h,l)$;
\item if $R_U(h,k)$, then both carrier witnesses $U(h)$ and $U(k)$ are
available for the certificate endpoint fields.
\end{enumerate}
\end{lemma}
\begin{proof}
Unpack $R_U$ as the triple consisting of the left carrier witness, the right
carrier witness, and the endpoint $\hsame$ proof.

For carrier reflexivity, a witness $U(h)$ supplies both endpoint carrier
fields, and \texttt{BEDC.FKernel.Hist.hsame\_refl} supplies the endpoint
classifier proof.

For symmetry, swap the two carrier witnesses and apply
\texttt{BEDC.FKernel.Hist.hsame\_symm} to the endpoint proof.

For transitivity, retain the outer carrier witnesses $U(h)$ and $U(l)$ from
the two packed classifiers. Compose the endpoint equalities with
\texttt{BEDC.FKernel.Hist.hsame\_trans}, then repack the outer carrier
witnesses with the composed endpoint proof.

The carrier-transport fields are the two carrier projections already stored in
each packed $R_U(h,k)$ witness.
\end{proof}
\leanchecked{BEDC.Derived.MagmaUp.concrete\_unary\_history\_magma\_classifier\_stability}

\begin{theorem}[Concrete unary-history MagmaUp semantic certificate]
\label{thm:concrete-unary-history-magma-semantic-certificate}
Let $U(h)$ abbreviate $\mathsf{UnaryHistory}(h)$, let
$\mathsf{mul}_U(h,k):=\mathsf{append}(h,k)$, and let
$$
R_U(h,k):=U(h)\land U(k)\land\hsame(h,k).
$$
The tuple with source, pattern, and ledger all equal to $U$, classifier
$R_U$, and operation $\mathsf{mul}_U$ is the concrete $\MagmaUp$ semantic
naming certificate over $\mathsf{BHist}$. Its certificate fields are: carrier
inhabitation by $\emp$; classifier reflexivity, symmetry, transitivity, and
carrier transport; operation closure; operation congruence; ledger append
provenance through $\Cont$; exactness
$$
R_U(h,k)\Longleftrightarrow U(h)\land U(k)\land\hsame(h,k);
$$
and status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
The carrier field is supplied by the empty unary history. The classifier
fields are exactly
\autoref{lem:concrete-unary-history-magma-classifier-stability}: carrier
reflexivity, symmetry, transitivity, and the two endpoint carrier projections
for $R_U$.

The operation fields are the concrete laws of
Theorem~\ref{thm:concrete-unary-history-magma-laws}, whose checked Lean target
is
\texttt{BEDC.Derived.MagmaUp.concrete\_unary\_history\_magma\_cont\_laws}.
Closure is the unary append-closure component, congruence is the append
classifier-transport component, and the ledger row is the canonical
continuation witness $\Cont(h,k,\mathsf{append}(h,k))$. Exactness is the
definition of $R_U$, so the displayed predicate is both the public classifier
and the certificate's endpoint criterion.
\end{proof}
