\chapter{A Concrete Naming Certificate for $\MagmaUp$}
\label{ch:concrete-instances-magma-namecert}

The magma interface $\MagmaUp$ packages a carrier endowed with a single binary operation, with no associativity, identity, or inverse demands. It is the smallest algebraic naming certificate above the inductive carriers of \autoref{ch:concrete-instances-bool-namecert}--\autoref{ch:concrete-instances-list-namecert} and the entry point of the algebraic ladder culminating in $\NameCert_{\FieldUp}$ (\autoref{ch:concrete-instances-field-namecert}).

\section{Carrier definition}
\label{sec:magma-carrier}

\begin{definition}[Concrete unary-history magma instance]
\label{def:concrete-unary-history-magma-instance}
Let $U(h)$ abbreviate $\mathsf{UnaryHistory}(h)$. The concrete unary-history
magma instance has carrier $U$, binary operation
$\mathsf{mul}_U(h,k):=\mathsf{append}(h,k)$, equivalently witnessed by
$\Cont(h,k,\mathsf{append}(h,k))$, and carrier-aware classifier
$$
R_U(h,k):=U(h)\land U(k)\land\hsame(h,k).
$$
The classifier only compares unary endpoints, so every public equality claim
retains both endpoint carrier witnesses.
\end{definition}

\begin{theorem}[Concrete unary-history magma laws]
\label{thm:concrete-unary-history-magma-laws}
For the instance of \autoref{def:concrete-unary-history-magma-instance},
$\mathsf{mul}_U$ is closed on unary histories. Moreover, if $R_U(h,h')$ and
$R_U(k,k')$, then
$$
R_U(\mathsf{mul}_U(h,k),\mathsf{mul}_U(h',k')).
$$
\end{theorem}
\begin{proof}
Closure is exactly
\texttt{BEDC.FKernel.Unary.unary\_append\_closed}: two unary endpoint
witnesses for $h$ and $k$ produce a unary endpoint witness for
$\mathsf{append}(h,k)$.

For congruence, unpack $R_U(h,h')$ and $R_U(k,k')$ into unary carrier
witnesses and endpoint $\hsame$ proofs. The two products are witnessed by
their canonical continuation facts
$\Cont(h,k,\mathsf{append}(h,k))$ and
$\Cont(h',k',\mathsf{append}(h',k'))$. Transport the continuation result
along the endpoint $\hsame$ data using
\texttt{BEDC.FKernel.Cont.cont\_result\_hsame\_transport}; the closure clause
supplies the carrier witnesses for both products, and the transported result
is the classifier equality required by $R_U$.
\end{proof}
\leanchecked{BEDC.Derived.MagmaUp.concrete\_unary\_history\_magma\_cont\_laws}

\begin{theorem}[Concrete unary-history magma continuation result unary exactness]
\label{thm:concrete-unary-history-magma-cont-result-unary}
For histories $h,k,r:\mathsf{BHist}$, if $\ContR(h,k,r)$, then the
continuation result is unary exactly when both continuation inputs are unary:
$$
\Unary(r)\Longleftrightarrow \Unary(h)\land\Unary(k).
$$
\end{theorem}
\leanchecked{BEDC.Derived.MagmaUp.concrete\_unary\_history\_magma\_cont\_result\_unary\_iff}
\begin{proof}
The continuation ledger identifies $r$ with $\mathsf{append}(h,k)$. The
unary append factor theorem says precisely that
$\Unary(h)\land\Unary(k)$ is equivalent to
$\Unary(\mathsf{append}(h,k))$. Transporting along the continuation equality
gives the displayed result-side exactness.
\end{proof}

\(\Ezero(t)\) or \(\Eone(t)\) cannot have an empty result:

\quad\wedge\quad

\begin{proof}

continuation inversion would then force the visible right input itself to be
\(\emp\). The zero- and one-headed cases are both excluded by constructor
separation for \(\mathsf{BHist}\).

\begin{theorem}[Concrete unary-history magma classifier append factors]
\label{thm:concrete-unary-history-magma-classifier-append-factors}
For histories $h,h',k,k':\mathsf{BHist}$, the concrete unary-history magma
classifier relates the two appended endpoints exactly when all four factors
are unary histories and the two appended endpoints are history-same:
$$
R_U(\mathsf{append}(h,k),\mathsf{append}(h',k'))
\Longleftrightarrow
U(h)\land U(k)\land U(h')\land U(k')
\land
\hsame(\mathsf{append}(h,k),\mathsf{append}(h',k')).
$$
\end{theorem}
\begin{proof}
Unfold $R_U$. The forward direction gives unary carrier witnesses for the two
appended endpoints. Apply unary append factor exactness to each carrier
witness, obtaining the unary witnesses for $h,k,h',k'$, and retain the
classifier's endpoint $\hsame$ component. Conversely, the four unary factor
witnesses close both appended endpoints under unary append closure, and the
displayed endpoint $\hsame$ proof repackages the concrete classifier.
\end{proof}

\begin{lemma}[Concrete unary-history magma classifier stability]
\label{lem:concrete-unary-history-magma-classifier-stability}
Let $U(h)$ abbreviate $\mathsf{UnaryHistory}(h)$ and define
$$
R_U(h,k):=U(h)\land U(k)\land\hsame(h,k).
$$
Then $R_U$ has carrier reflexivity on every $U(h)$, symmetry,
transitivity, and carrier transport in both endpoints. Concretely:
\begin{enumerate}
\item if $U(h)$, then $R_U(h,h)$;
\item if $R_U(h,k)$, then $R_U(k,h)$;
\item if $R_U(h,k)$ and $R_U(k,l)$, then $R_U(h,l)$;
\item if $R_U(h,k)$, then both carrier witnesses $U(h)$ and $U(k)$ are
available for the certificate endpoint fields.
\end{enumerate}
\end{lemma}
\begin{proof}
Unpack $R_U$ as the triple consisting of the left carrier witness, the right
carrier witness, and the endpoint $\hsame$ proof.

For carrier reflexivity, a witness $U(h)$ supplies both endpoint carrier
fields, and \texttt{BEDC.FKernel.Hist.hsame\_refl} supplies the endpoint
classifier proof.

For symmetry, swap the two carrier witnesses and apply
\texttt{BEDC.FKernel.Hist.hsame\_symm} to the endpoint proof.

For transitivity, retain the outer carrier witnesses $U(h)$ and $U(l)$ from
the two packed classifiers. Compose the endpoint equalities with
\texttt{BEDC.FKernel.Hist.hsame\_trans}, then repack the outer carrier
witnesses with the composed endpoint proof.

The carrier-transport fields are the two carrier projections already stored in
each packed $R_U(h,k)$ witness.
\end{proof}
\leanchecked{BEDC.Derived.MagmaUp.concrete\_unary\_history\_magma\_classifier\_stability}

\begin{theorem}[Concrete unary-history MagmaUp semantic certificate]
\label{thm:concrete-unary-history-magma-semantic-certificate}
Let $U(h)$ abbreviate $\mathsf{UnaryHistory}(h)$, let
$\mathsf{mul}_U(h,k):=\mathsf{append}(h,k)$, and let
$$
R_U(h,k):=U(h)\land U(k)\land\hsame(h,k).
$$

$R_U$, and operation $\mathsf{mul}_U$ is the concrete $\MagmaUp$ semantic
naming certificate over $\mathsf{BHist}$. Its certificate fields are: carrier
inhabitation by $\emp$; classifier reflexivity, symmetry, transitivity, and
carrier transport; operation closure; operation congruence; ledger append
provenance through $\Cont$; exactness
$$
R_U(h,k)\Longleftrightarrow U(h)\land U(k)\land\hsame(h,k);
$$
and status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
The carrier field is supplied by the empty unary history. The classifier
fields are exactly
\autoref{lem:concrete-unary-history-magma-classifier-stability}: carrier
reflexivity, symmetry, transitivity, and the two endpoint carrier projections
for $R_U$.

The operation fields are the concrete laws of
Theorem~\ref{thm:concrete-unary-history-magma-laws}, whose checked Lean target
is
\texttt{BEDC.Derived.MagmaUp.concrete\_unary\_history\_magma\_cont\_laws}.
Closure is the unary append-closure component, congruence is the append
classifier-transport component, and the ledger row is the canonical
continuation witness $\Cont(h,k,\mathsf{append}(h,k))$. Exactness is the
definition of $R_U$, so the displayed predicate is both the public classifier
and the certificate's endpoint criterion.
\end{proof}
