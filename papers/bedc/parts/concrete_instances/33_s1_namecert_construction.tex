\chapter{A Concrete Naming Certificate for \(\SOneUp\)}
\label{ch:concrete-instances-s1-namecert}

The circle interface $\SOneUp$ is a certificate-obligation schema over the product carrier $\RealUp \times \RealUp$ from \autoref{ch:concrete-instances-real-namecert}. Its carrier is the unit-circle subspace: pairs $(x,y)$ equipped with a certified equation witness that $x^2 + y^2$ is classified as the real unit. The interface is not a primitive topological space; it is admitted only when the product real witnesses, equation witness, componentwise classifier, and equation transport data are supplied.

\section{Carrier definition}

\begin{definition}[SOne carrier]
\label{def:s1-carrier}
Use pairs $(x,y)$ of Bishop real witnesses with a certified unit equation witness $x^2 + y^2 \approx 1$ under $\RealClassifier$.
\end{definition}
\leandef{BEDC.Derived.S1Up.SOneHistoryCarrier}

\begin{theorem}[SOne history carrier rational components]
\label{thm:s1-history-carrier-rational-components}
For every history $h$, the concrete $\SOneUp$ history carrier exposes product
components $x,y$ and rational denominator tails $d_x,d_y$ such that both
components are displayed as real constants and the BEDC continuation ledger
composes the two components into $h$:
$$
\mathsf{SOneHistoryCarrier}(h)
\Longrightarrow
\exists x,y,d_x,d_y.\;
\hsame(x,\Eone(d_x))\land
\mathsf{RatHistoryCarrier}(d_x)\land
\hsame(y,\Eone(d_y))\land
\mathsf{RatHistoryCarrier}(d_y)\land
\ContR(x,y,h).
$$
\end{theorem}
\begin{proof}
Unpack the $\SOneUp$ carrier as the $\ProdUp$ carrier over two real-constant
history carriers. The product carrier supplies the two component histories and
their continuation ledger into $h$. Each real-constant carrier then exposes a
rational denominator tail together with the displayed $\Eone$ presentation of
that component.
\end{proof}
\leanchecked{BEDC.Derived.S1Up.SOneHistoryCarrier\_rational\_components}

\section{The certificate}

\begin{definition}[SOne source specification]
\label{def:s1-source-specification}
The source is the $\ProdUp$ pair of two $\RealUp$ witnesses together with the real-algebra data needed to form $x^2 + y^2$ and the displayed unit witness.
\end{definition}

\begin{definition}[SOne pattern specification]
\label{def:s1-pattern-specification}
The pattern is unit-circle formation from a real pair and a certified proof object for the equation $x^2 + y^2 \approx 1$.
\end{definition}

\begin{definition}[SOne classifier specification]
\label{def:s1-classifier-specification}
The classifier relates two carrier witnesses $(x,y,p)$ and $(x',y',p')$ when $\RealClassifier(x,x')$ and $\RealClassifier(y,y')$ hold componentwise and the unit equation witness transports from $p$ to $p'$ along those two real-classifier witnesses.
\end{definition}

\begin{definition}[SOne stability certificate]
\label{def:s1-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item reflexivity from reflexivity of the two real components and identity transport of the equation witness
  \item symmetry by symmetry of the two real components and reverse transport of the equation witness
  \item transitivity by transitivity of the two real components and composition of equation transport
  \item compatibility with the $\ProdUp$ classifier over $\RealUp \times \RealUp$
  \item preservation of the certified equation $x^2 + y^2 \approx 1$
\end{enumerate}
\end{definition}

\begin{definition}[SOne ledger policy]
\label{def:s1-ledger-policy}
Record both real witnesses, the componentwise real-classifier witnesses used in classifier derivations, and the transported unit equation proof.
\end{definition}

\begin{definition}[$\SOneUp$ certificate obligations]
\label{def:s1-certificate-obligations}
The carrier, source, pattern, classifier, stability, and ledger data listed above are the obligations required for the interface name $\SOneUp$.
\end{definition}

\paragraph{Primitive boundary.}
$\SOneUp$ is not a BEDC primitive. It is admitted only when those obligations are supplied.

\begin{remark}[Zero-axiom encoding note]
\label{rem:s1-zero-axiom-note}
The interface records explicit component witnesses and equation transport, with no quotient-type representative and no use of $\mathsf{Quot.sound}$.
\end{remark}
