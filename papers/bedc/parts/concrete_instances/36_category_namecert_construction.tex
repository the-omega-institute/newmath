\chapter{A Concrete Naming Certificate for \(\CategoryUp\)}
\label{ch:concrete-instances-category-namecert}

The category interface \(\CategoryUp\) has certificate obligations over an object carrier together with, for each ordered pair of objects, a morphism carrier, identity morphisms at each object, a binary composition operation respecting source and target, and the associativity and identity laws expressed as morphism-classifier sameness. The size discipline is BEDC-internal: object and morphism carriers are each covered by their own NameCert obligations, avoiding any global "set of all objects" assumption. It is not a primitive structure; it is admitted only as a multi-sorted carrier with relational composition.

\section{Carrier definition}

\begin{definition}[Category carrier]
\label{def:category-carrier}
Use a multi-sorted setup with object carrier, morphism carrier for each ordered object pair, identities, composition, and law fields.
\end{definition}
\leandef{BEDC.Derived.CategoryUp.CategoryHomCarrier}

\begin{theorem}[Category empty identity hom carrier]
\label{thm:category-hom-carrier-empty-identity}
For the unary-continuation category, if \(h\) is unary, then the empty history
is a hom-carrier from \(h\) to \(h\).
\end{theorem}
\begin{proof}
The source and target carrier witnesses are the given unary witness for \(h\).
The morphism carrier is the unary empty history, and the continuation field is
the right-unit continuation \(\ContR(h,\emp,h)\).
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.CategoryHomCarrier\_empty\_identity}

\begin{theorem}[Category empty identity hom carrier exactness]
\label{thm:category-hom-carrier-empty-identity-iff}
For histories \(a,b\), the empty history is a hom-carrier from \(a\) to \(b\)
exactly when \(a\) and \(b\) are unary and internally the same:
$$
\mathsf{CategoryHomCarrier}(a,b,\emp)
\quad\Longleftrightarrow\quad
\mathsf{UnaryHistory}(a)\wedge \mathsf{UnaryHistory}(b)\wedge \hsame(a,b).
$$
\end{theorem}
\begin{proof}
Unpack the hom-carrier witness. Its continuation field is
\(\ContR(a,\emp,b)\); determinism against the right-unit continuation
\(\ContR(a,\emp,a)\) identifies \(a\) and \(b\). Conversely, unary source and
target witnesses together with \(\hsame(a,b)\) transport the right-unit
continuation to a continuation ending at \(b\), while \(\emp\) supplies the
empty morphism carrier.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.CategoryHomCarrier\_empty\_identity\_iff}

\begin{theorem}[Category hom carrier composition closure]
\label{thm:category-hom-carrier-comp-closed}
For the unary-continuation category, if \(f\) is a hom-carrier from \(a\) to \(b\), \(g\) is a hom-carrier from \(b\) to \(c\), and \(\ContR(f,g,fg)\), then \(fg\) is a hom-carrier from \(a\) to \(c\).
\end{theorem}
\begin{proof}
Unpack the two hom-carrier witnesses. The source, middle, target, and morphism histories are unary. Unary continuation closure gives unary membership for the composite morphism \(fg\), and associativity of the recursive append representative underlying \(\ContR\) supplies the continuation from \(a\) through \(fg\) to \(c\).
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.CategoryHomCarrier\_comp\_closed}

\begin{theorem}[Category hom carrier composition public readback]
\label{thm:category-hom-carrier-comp-public-readback}
For the unary-continuation category, if \(f\) is a hom-carrier from \(a\) to
\(b\), \(g\) is a hom-carrier from \(b\) to \(c\), and \(\ContR(f,g,fg)\),
then \(fg\) is a hom-carrier from \(a\) to \(c\), and every displayed
hom-carrier from \(a\) to \(c\) has morphism history \(\hsame\)-equal to
\(fg\).
\end{theorem}
\begin{proof}
Composition closure supplies the displayed carrier for \(fg\). The
hom-carrier morphism determinacy theorem then compares that carrier with any
other displayed carrier over the same source and target.
\end{proof}

\begin{theorem}[Category hom carrier composite target determinacy]
\label{thm:category-hom-carrier-comp-target-deterministic}
For the unary-continuation category, if \(f\) is a hom-carrier from \(a\) to
\(b\), \(g\) is a hom-carrier from \(b\) to \(c\), and \(fg\) is their
continuation composite, then any displayed hom-carrier from \(a\) to \(d\) with
the same composite history forces \(\HSame(c,d)\).
\end{theorem}
\begin{proof}
Hom-carrier composition closure makes \(fg\) a carrier from \(a\) to \(c\).
Target determinacy for the fixed morphism history \(fg\) compares this carrier
with the displayed carrier from \(a\) to \(d\).
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.CategoryHomCarrier\_comp\_target\_deterministic}

\begin{theorem}[Category hom carrier composite middle-object determinacy]
\label{thm:category-hom-carrier-comp-middle-object-deterministic}
For the unary-continuation category, if \(f\) is a hom-carrier from \(a\) to
\(b\), \(g\) is a hom-carrier from \(b'\) to \(c\),
\(\ContR(f,g,fg)\), and \(fg\) is displayed as a hom-carrier from \(a\) to
\(c\), then \(\HSame(b,b')\).
\end{theorem}
\begin{proof}
Unpack the hom-carrier witnesses. The left leg gives \(\ContR(a,f,b)\), the
composite gives \(\ContR(f,g,fg)\), and the displayed composite gives
\(\ContR(a,fg,c)\). Associativity of append turns these three continuation
facts into \(\ContR(b,g,c)\). Comparing this continuation with the right leg
\(\ContR(b',g,c)\) by right-context cancellation identifies the two middle
objects.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.CategoryHomCarrier\_comp\_middle\_object\_deterministic}

\begin{theorem}[Category hom carrier tail commutativity]
\label{thm:category-hom-carrier-tail-comm-hsame}
For the unary-continuation category, if \(f\) is a hom-carrier from \(a\) to
\(b\), \(g\) is a hom-carrier from \(b\) to \(c\),
\(\ContR(f,g,fg)\), and \(\ContR(g,f,gf)\), then
\(\HSame(fg,gf)\).
\end{theorem}
\begin{proof}
Unpack the two hom-carrier witnesses to obtain unary membership for \(f\) and
\(g\). Unary continuation commutativity applied to the two continuation
ledgers identifies the two tail composites.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.CategoryHomCarrier\_tail\_comm\_hsame}

\begin{theorem}[Category hom carrier identity square closure]
\label{thm:category-hom-carrier-identity-square-closed}
For the unary-continuation category, if \(f\) is a hom-carrier from \(a\)
to \(b\), \(left\) is obtained by composing the empty continuation before
\(f\), and \(right\) is obtained by composing the empty continuation after
\(f\), then \(left\) and \(right\) are both hom-carriers from \(a\) to \(b\),
and \(\HSame(left,right)\).
\end{theorem}
\begin{proof}
The left unit continuation identifies \(left\) with \(f\). Determinism of
\(\ContR\), compared against the right unit continuation for \(f\), identifies
\(right\) with \(f\). Transporting the hom-carrier witness for \(f\) across
these two history equalities gives the two carrier witnesses, and transitivity
of \(\HSame\) identifies \(left\) and \(right\).
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.CategoryHomCarrier\_identity\_square\_closed}

\begin{theorem}[Category hom carrier identity semantic name certificate]

\begin{theorem}[Category hom carrier associativity witness]
\label{thm:category-hom-carrier-assoc-witness}
For the unary-continuation category, let $f$ be a hom-carrier from $a$ to $b$, let $g$ be a hom-carrier from $b$ to $c$, and let $h$ be a hom-carrier from $c$ to $d$. If $fg$, $gh$, $left$, and $right$ satisfy
$$
\ContR(f,g,fg),\quad \ContR(g,h,gh),\quad \ContR(fg,h,left),\quad \ContR(f,gh,right),
$$
then $left$ and $right$ are both hom-carriers from $a$ to $d$, and $\HSame(left,right)$.
\end{theorem}
\begin{proof}
First build the left-associated carrier. Apply Theorem~\ref{thm:category-hom-carrier-comp-closed} to the carriers $f:a\to b$ and $g:b\to c$ with $\ContR(f,g,fg)$, obtaining $fg:a\to c$. Apply the same theorem to $fg:a\to c$ and $h:c\to d$ with $\ContR(fg,h,left)$, obtaining $left:a\to d$.

Next build the right-associated carrier. Apply Theorem~\ref{thm:category-hom-carrier-comp-closed} to the carriers $g:b\to c$ and $h:c\to d$ with $\ContR(g,h,gh)$, obtaining $gh:b\to d$. Apply the same theorem to $f:a\to b$ and $gh:b\to d$ with $\ContR(f,gh,right)$, obtaining $right:a\to d$.

Finally apply continuation associativity, Theorem~\ref{thm:continuation-associativity}, to the displayed continuation square. The two composition trees therefore have the same morphism classifier, namely $\HSame(left,right)$.
\end{proof}

Per-object certificate: traditional target categorical associativity; BEDC source $\mathsf{CategoryHomCarrier}$ plus $\Cont$; carrier $\mathsf{BHist}$ hom witnesses; classifier $\hsame$; stability obligation associativity under morphism classifier; ledger the two composition trees; core theorem list entry this associativity witness; exactness predicate equality of the two composite trees; standard bridge deferred; status $\mathsf{ProofTarget}$.

\begin{theorem}[Category hom carrier fourfold associativity witness]
\label{thm:category-hom-carrier-fourfold-assoc-witness}
For the unary-continuation category, let $f$ be a hom-carrier from $a$ to $b$, let $g$ be a hom-carrier from $b$ to $c$, let $h$ be a hom-carrier from $c$ to $d$, and let $i$ be a hom-carrier from $d$ to $e$. If $fg$, $fgh$, $hi$, $ghi$, $left$, and $right$ satisfy
$$
\ContR(f,g,fg),\quad \ContR(fg,h,fgh),\quad \ContR(h,i,hi),\quad \ContR(g,hi,ghi),\quad \ContR(fgh,i,left),\quad \ContR(f,ghi,right),
$$
then $left$ and $right$ are both hom-carriers from $a$ to $e$, and $\HSame(left,right)$.
\end{theorem}
\begin{proof}
First build the left fourfold carrier. Apply Theorem~\ref{thm:category-hom-carrier-comp-closed} to $f:a\to b$ and $g:b\to c$ with $\ContR(f,g,fg)$, obtaining $fg:a\to c$. Apply the same theorem to $fg:a\to c$ and $h:c\to d$ with $\ContR(fg,h,fgh)$, obtaining $fgh:a\to d$. Apply it once more to $fgh:a\to d$ and $i:d\to e$ with $\ContR(fgh,i,left)$, obtaining $left:a\to e$.

Next build the right fourfold carrier. Apply Theorem~\ref{thm:category-hom-carrier-comp-closed} to $h:c\to d$ and $i:d\to e$ with $\ContR(h,i,hi)$, obtaining $hi:c\to e$. Apply the same theorem to $g:b\to c$ and $hi:c\to e$ with $\ContR(g,hi,ghi)$, obtaining $ghi:b\to e$. Apply it once more to $f:a\to b$ and $ghi:b\to e$ with $\ContR(f,ghi,right)$, obtaining $right:a\to e$.

Finally apply the kernel fourfold continuation associativity witness, $cont\_assoc\_four$, to the six displayed continuation relations. The two fourfold composition ledgers therefore generate the same composite history classifier, namely $\HSame(left,right)$.
\end{proof}

Per-object certificate: traditional target fourfold categorical associativity coherence; BEDC source $\mathsf{CategoryHomCarrier}$ plus $\Cont$; carrier $\mathsf{BHist}$ hom witnesses; classifier $\hsame$; stability obligation rebracketing under morphism classifier; ledger the two fourfold composition trees; core theorem list entry fourfold coherence; exactness predicate equality of generated composite histories; standard bridge deferred; status $\mathsf{ProofTarget}$.

\begin{theorem}[Category hom carrier associative composition closure]
\label{thm:category-hom-carrier-comp-assoc-closed}
For the unary-continuation category, composable hom-carriers \(f:a\to b\),
\(g:b\to c\), and \(h:c\to d\) remain hom-carriers after either
association of composition. If \(\ContR(f,g,fg)\), \(\ContR(g,h,gh)\),
\(\ContR(fg,h,left)\), and \(\ContR(f,gh,right)\), then both \(left\) and
\(right\) are hom-carriers from \(a\) to \(d\), and
\(\HSame(left,right)\).
\end{theorem}
\begin{proof}
Apply hom-carrier composition closure first to \(f,g\) and then to \(fg,h\)
for the left association. Apply the same closure first to \(g,h\) and then
to \(f,gh\) for the right association. The continuation associativity law
identifies the two composite histories.
\end{proof}

\begin{theorem}[Continuation morphism tail determinacy]
\label{thm:continuation-morphism-tail-deterministic}
For histories $a,b$, any two continuation morphisms from $a$ to $b$ have
history-same tails:
$$
left,right:\mathsf{ContinuationMorphism}(a,b)
\Longrightarrow
\hsame(\mathsf{tail}(left),\mathsf{tail}(right)).
$$
\end{theorem}
\begin{proof}
Unpack both continuation morphism records. Their continuation witnesses have
the same source $a$ and target $b$, so continuation left-context cancellation
identifies the two displayed tails.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.ContinuationMorphism\_tail\_deterministic}

\begin{definition}[Continuation morphism composition closure]
\label{def:continuation-morphism-comp-closed}
Given continuation morphisms from \(a\) to \(b\) and from \(b\) to \(c\), their
closed composite is the continuation morphism from \(a\) to \(c\) whose tail is
the append of the two displayed tails. The continuation witness is obtained by
unpacking the first relation and reassociating the append spine.
\end{definition}
\leandef{BEDC.Derived.CategoryUp.ContinuationMorphism\_comp\_closed}

\begin{theorem}[Continuation morphism associative composition closure]
\label{thm:continuation-morphism-comp-assoc-closed}
For histories \(a,b,c,d\), any continuation morphisms
\(a\to b\), \(b\to c\), and \(c\to d\) admit both associated composite
morphisms \(a\to d\), and the two composite tails are internally the same.
\end{theorem}
\begin{proof}
Unpack the three morphism records into their tails and continuation witnesses.
The left-associated composite tail is \(\mathsf{append}(\mathsf{append}(f,g),h)\);
the right-associated composite tail is
\(\mathsf{append}(f,\mathsf{append}(g,h))\). The continuation witnesses reduce
to associativity of BEDC history append, and the tail comparison is exactly the
same associativity equality.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.ContinuationMorphism\_comp\_assoc\_closed}

\begin{theorem}[Continuation morphism associative composite tail]
\label{thm:continuation-morphism-comp-assoc-tail-hsame}
For histories $a,b,c,d$, any continuation morphisms $a\to b$, $b\to c$, and
$c\to d$ have left-associated and right-associated canonical composite tails
that are history-same:
$$
\hsame
\left(
\mathsf{tail}\bigl((first\circ second)\circ third\bigr),
\mathsf{tail}\bigl(first\circ(second\circ third)\bigr)
\right).
$$
\end{theorem}
\begin{proof}
Unpack the three continuation morphisms into their displayed tails. The two
canonical composite tails reduce to
$\mathsf{append}(\mathsf{append}(f,g),h)$ and
$\mathsf{append}(f,\mathsf{append}(g,h))$, which are identified by BEDC history
append associativity.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.ContinuationMorphism\_comp\_assoc\_tail\_hsame}

\section{The certificate}

\begin{definition}[Category source specification]
\label{def:category-source-specification}
The source is the licensed object carrier and each licensed hom-carrier.
\end{definition}

\begin{definition}[Category pattern specification]
\label{def:category-pattern-specification}
The pattern is morphism expression formation from identities and composable pairs.
\end{definition}

\begin{definition}[Category classifier specification]
\label{def:category-classifier-specification}
The classifier is object sameness for objects and hom-carrier sameness for morphisms, indexed by source and target.
\end{definition}

\begin{definition}[Category stability certificate]
\label{def:category-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item object classifier reflexivity and transitivity
  \item morphism classifier reflexivity and transitivity
  \item identity morphism stability
  \item composition closure
  \item associativity under morphism classifier
  \item left and right identity laws
  \item composition congruence
\end{enumerate}
\end{definition}
\leanchecked{BEDC.Derived.CategoryUp.ContinuationMorphism\_identity\_comp\_closure}

\begin{definition}[Category ledger policy]
\label{def:category-ledger-policy}
Record object ledgers, hom-carrier ledgers, and composition trees.
\end{definition}

\begin{definition}[\(\CategoryUp\) certificate obligations]
\label{def:category-certificate-obligations}
The carrier, source, pattern, classifier, stability, and ledger data listed above are the obligations required for the interface name \(\CategoryUp\).
\end{definition}

\begin{theorem}[Category left zero-headed continuation result cases]
\label{thm:category-cont-left-e0-result-cases}
If $\ContR(\Ezero(h),k,\Ezero(r))$, then either $k=\emp$ and
$\HSame(h,r)$, or $k=\Ezero(k_0)$ for some $k_0$ and
$\ContR(\Ezero(h),k_0,r)$.
\end{theorem}
\leanchecked{BEDC.Derived.CategoryUp.category\_cont\_left\_e0\_result\_cases}
\begin{proof}
Classify the continuation history $k$. The empty branch identifies the inner
histories. The zero branch exposes the predecessor of $k$ and removes the
matching outer constructor from the continuation result. The one branch would
identify a zero-headed history with a one-headed history, which is impossible
by generated-history no-confusion.
\end{proof}

\paragraph{Primitive boundary.}
\(\CategoryUp\) is not a BEDC primitive. It is admitted only when those obligations are supplied.

\label{rem:category-zero-axiom-note}
Avoids a global universe of all objects by using explicit licensed carriers.
\end{remark}
