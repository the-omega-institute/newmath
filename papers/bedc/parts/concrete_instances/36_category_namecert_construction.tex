\chapter{A Concrete Naming Certificate for \(\CategoryUp\)}
\label{ch:concrete-instances-category-namecert}

The category interface \(\CategoryUp\) has certificate obligations over an object carrier together with, for each ordered pair of objects, a morphism carrier, identity morphisms at each object, a binary composition operation respecting source and target, and the associativity and identity laws expressed as morphism-classifier sameness. The size discipline is BEDC-internal: object and morphism carriers are each covered by their own NameCert obligations, avoiding any global "set of all objects" assumption. It is not a primitive structure; it is admitted only as a multi-sorted carrier with relational composition.

\section{Carrier definition}

\begin{definition}[Category carrier]
\label{def:category-carrier}
Use a multi-sorted setup with object carrier, morphism carrier for each ordered object pair, identities, composition, and law fields.
\end{definition}
\leandef{BEDC.Derived.CategoryUp.CategoryHomCarrier}

\begin{theorem}[Category empty identity hom carrier]
\label{thm:category-hom-carrier-empty-identity}
For the unary-continuation category, if \(h\) is unary, then the empty history
is a hom-carrier from \(h\) to \(h\).
\end{theorem}
\begin{proof}
The source and target carrier witnesses are the given unary witness for \(h\).
The morphism carrier is the unary empty history, and the continuation field is
the right-unit continuation \(\ContR(h,\emp,h)\).
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.CategoryHomCarrier\_empty\_identity}

\begin{theorem}[Category hom carrier composition closure]
\label{thm:category-hom-carrier-comp-closed}
For the unary-continuation category, if \(f\) is a hom-carrier from \(a\) to \(b\), \(g\) is a hom-carrier from \(b\) to \(c\), and \(\ContR(f,g,fg)\), then \(fg\) is a hom-carrier from \(a\) to \(c\).
\end{theorem}
\begin{proof}
Unpack the two hom-carrier witnesses. The source, middle, target, and morphism histories are unary. Unary continuation closure gives unary membership for the composite morphism \(fg\), and associativity of the recursive append representative underlying \(\ContR\) supplies the continuation from \(a\) through \(fg\) to \(c\).
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.CategoryHomCarrier\_comp\_closed}

\begin{theorem}[Category hom carrier associativity witness]
\label{thm:category-hom-carrier-assoc-witness}
For the unary-continuation category, let $f$ be a hom-carrier from $a$ to $b$, let $g$ be a hom-carrier from $b$ to $c$, and let $h$ be a hom-carrier from $c$ to $d$. If $fg$, $gh$, $left$, and $right$ satisfy
$$
\ContR(f,g,fg),\quad \ContR(g,h,gh),\quad \ContR(fg,h,left),\quad \ContR(f,gh,right),
$$
then $left$ and $right$ are both hom-carriers from $a$ to $d$, and $\HSame(left,right)$.
\end{theorem}
\begin{proof}
First build the left-associated carrier. Apply Theorem~\ref{thm:category-hom-carrier-comp-closed} to the carriers $f:a\to b$ and $g:b\to c$ with $\ContR(f,g,fg)$, obtaining $fg:a\to c$. Apply the same theorem to $fg:a\to c$ and $h:c\to d$ with $\ContR(fg,h,left)$, obtaining $left:a\to d$.

Next build the right-associated carrier. Apply Theorem~\ref{thm:category-hom-carrier-comp-closed} to the carriers $g:b\to c$ and $h:c\to d$ with $\ContR(g,h,gh)$, obtaining $gh:b\to d$. Apply the same theorem to $f:a\to b$ and $gh:b\to d$ with $\ContR(f,gh,right)$, obtaining $right:a\to d$.

Finally apply continuation associativity, Theorem~\ref{thm:continuation-associativity}, to the displayed continuation square. The two composition trees therefore have the same morphism classifier, namely $\HSame(left,right)$.
\end{proof}

Per-object certificate: traditional target categorical associativity; BEDC source $\mathsf{CategoryHomCarrier}$ plus $\Cont$; carrier $\mathsf{BHist}$ hom witnesses; classifier $\hsame$; stability obligation associativity under morphism classifier; ledger the two composition trees; core theorem list entry this associativity witness; exactness predicate equality of the two composite trees; standard bridge deferred; status $\mathsf{ProofTarget}$.

\begin{theorem}[Category hom carrier associative composition closure]
\label{thm:category-hom-carrier-comp-assoc-closed}
For the unary-continuation category, composable hom-carriers \(f:a\to b\),
\(g:b\to c\), and \(h:c\to d\) remain hom-carriers after either
association of composition. If \(\ContR(f,g,fg)\), \(\ContR(g,h,gh)\),
\(\ContR(fg,h,left)\), and \(\ContR(f,gh,right)\), then both \(left\) and
\(right\) are hom-carriers from \(a\) to \(d\), and
\(\HSame(left,right)\).
\end{theorem}
\begin{proof}
Apply hom-carrier composition closure first to \(f,g\) and then to \(fg,h\)
for the left association. Apply the same closure first to \(g,h\) and then
to \(f,gh\) for the right association. The continuation associativity law
identifies the two composite histories.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.CategoryHomCarrier\_comp\_assoc\_closed}

\section{The certificate}

\begin{definition}[Category source specification]
\label{def:category-source-specification}
The source is the licensed object carrier and each licensed hom-carrier.
\end{definition}

\begin{definition}[Category pattern specification]
\label{def:category-pattern-specification}
The pattern is morphism expression formation from identities and composable pairs.
\end{definition}

\begin{definition}[Category classifier specification]
\label{def:category-classifier-specification}
The classifier is object sameness for objects and hom-carrier sameness for morphisms, indexed by source and target.
\end{definition}

\begin{definition}[Category stability certificate]
\label{def:category-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item object classifier reflexivity and transitivity
  \item morphism classifier reflexivity and transitivity
  \item identity morphism stability
  \item composition closure
  \item associativity under morphism classifier
  \item left and right identity laws
  \item composition congruence
\end{enumerate}
\end{definition}
\leanchecked{BEDC.Derived.CategoryUp.ContinuationMorphism\_identity\_comp\_closure}

\begin{definition}[Category ledger policy]
\label{def:category-ledger-policy}
Record object ledgers, hom-carrier ledgers, and composition trees.
\end{definition}

\begin{definition}[\(\CategoryUp\) certificate obligations]
\label{def:category-certificate-obligations}
The carrier, source, pattern, classifier, stability, and ledger data listed above are the obligations required for the interface name \(\CategoryUp\).
\end{definition}

\begin{theorem}[Category left zero-headed continuation result cases]
\label{thm:category-cont-left-e0-result-cases}
If $\ContR(\Ezero(h),k,\Ezero(r))$, then either $k=\emp$ and
$\HSame(h,r)$, or $k=\Ezero(k_0)$ for some $k_0$ and
$\ContR(\Ezero(h),k_0,r)$.
\end{theorem}
\leanchecked{BEDC.Derived.CategoryUp.category\_cont\_left\_e0\_result\_cases}
\begin{proof}
Classify the continuation history $k$. The empty branch identifies the inner
histories. The zero branch exposes the predecessor of $k$ and removes the
matching outer constructor from the continuation result. The one branch would
identify a zero-headed history with a one-headed history, which is impossible
by generated-history no-confusion.
\end{proof}

\paragraph{Primitive boundary.}
\(\CategoryUp\) is not a BEDC primitive. It is admitted only when those obligations are supplied.

\begin{remark}[Zero-axiom encoding note]
\label{rem:category-zero-axiom-note}
Avoids a global universe of all objects by using explicit licensed carriers.
\end{remark}
