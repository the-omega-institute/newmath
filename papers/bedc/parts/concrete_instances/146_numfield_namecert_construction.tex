\chapter{A Concrete Naming Certificate for $\NumFieldUp$}
\label{ch:concrete-instances-numfield-namecert}

The number-field interface $\NumFieldUp$ packages a finite field extension of the rational-number certificate. It depends on $\NameCert_{\FieldExtUp}$ (\autoref{ch:concrete-instances-fieldext-namecert}) and $\NameCert_{\RatUp}$ (\autoref{ch:concrete-instances-rat-namecert}).

\section{Carrier definition}
\label{sec:numfield-carrier}

\begin{definition}[RatUp reflexive number-field source]
\label{def:numfield-reflexive-ratup-source}
The concrete $\NumFieldUp$ carrier is the RatUp history carrier
$\mathsf{RatHistoryCarrier}$ on $\mathsf{BHist}$. Its classifier is
$\mathsf{RatHistoryClassifier}$. The finite-extension witness is the
identity $\FieldExtUp$ package of
\autoref{def:fieldext-reflexive-extension-carrier}, read through the public
interface of \autoref{thm:fieldext-reflexive-public-namecert-interface}.

Thus the $\NumFieldUp$ source has no separate number-field history sort:
its base field, ambient field, vector carrier, embedding row, exactness
row, and ledger row are all the RatUp reflexive $\FieldExtUp$ rows over
$\mathsf{RatHistoryCarrier}$ and $\mathsf{RatHistoryClassifier}$.
\end{definition}

\begin{theorem}[RatUp reflexive NumField carrier and classifier]
\label{thm:numfield-reflexive-carrier-classifier}
For the source of
\autoref{def:numfield-reflexive-ratup-source}, the $\NumFieldUp$ carrier is
exactly $\mathsf{RatHistoryCarrier}$ and the $\NumFieldUp$ endpoint
classifier is exactly $\mathsf{RatHistoryClassifier}$. Every displayed
number-field endpoint is therefore a RatUp history endpoint, and every
endpoint comparison is a RatUp classifier comparison.
\end{theorem}
\begin{proof}
Unfold \autoref{def:numfield-reflexive-ratup-source}. The carrier and
classifier fields are specified directly as
$\mathsf{RatHistoryCarrier}$ and $\mathsf{RatHistoryClassifier}$.

The RatUp semantic name certificate supplies the carried-history
interpretation for these fields. No quotient, completion, polynomial
carrier, or new endpoint classifier is introduced by the $\NumFieldUp$
reading, so each number-field endpoint is read through the same RatUp
history row.
\end{proof}

\begin{theorem}[RatUp reflexive FieldExt consumption for NumFieldUp]
\label{thm:numfield-reflexive-fieldext-consumption}
For the source of
\autoref{def:numfield-reflexive-ratup-source}, the finite-extension
component consumed by $\NumFieldUp$ is the identity $\FieldExtUp$ row of
\autoref{def:fieldext-reflexive-extension-carrier}. Its base field and
ambient field are both the concrete $\FieldUp(\RatUp)$ certificate, its
embedding is $\iota_{\mathsf{Rat}}(h):=h$, and its scalar action is the
RatUp multiplication row.
\end{theorem}
\begin{proof}
Read the finite-extension witness from
\autoref{thm:fieldext-reflexive-public-namecert-interface}. That theorem
exposes the public $\FieldExtUp$ interface whose carrier, classifier, base
field, ambient field, identity embedding, self-vector-space rows, exactness,
and ledger all live over the RatUp operation table.

The $\NumFieldUp$ source only consumes that interface. Because the consumed
row is the identity extension, the base and ambient field certificates are
the same RatUp field certificate, and the scalar-action row is the same
multiplication row already present in $\FieldUp(\RatUp)$.
\end{proof}

\begin{theorem}[RatUp reflexive NumField ledger exactness]
\label{thm:numfield-reflexive-ledger-exactness}
For the source of
\autoref{def:numfield-reflexive-ratup-source}, the $\NumFieldUp$ exactness
and ledger rows are inherited from the RatUp reflexive $\FieldExtUp$
public interface. In particular, every carrier display, classifier display,
field-operation endpoint, identity-embedding endpoint, vector-space
endpoint, scalar-action endpoint, and ledger provenance row is one of the
rows listed by \autoref{thm:fieldext-reflexive-public-namecert-interface}.
\end{theorem}
\begin{proof}
First apply
\autoref{thm:numfield-reflexive-fieldext-consumption}. It identifies the
finite-extension component used by $\NumFieldUp$ with the public RatUp
reflexive $\FieldExtUp$ interface.

Next read the exactness and ledger clauses from
\autoref{thm:fieldext-reflexive-public-namecert-interface}. Its public
surface includes exact endpoint classification and ledger provenance over
the RatUp operation table. Since the carrier and classifier are unchanged
by \autoref{thm:numfield-reflexive-carrier-classifier}, those clauses
serve as the displayed $\NumFieldUp$ exactness and ledger rows without any
additional transport.
\end{proof}

\begin{theorem}[RatUp reflexive NumField NameCert obligation surface]
\label{thm:numfield-reflexive-namecert-obligation-surface}
The source of \autoref{def:numfield-reflexive-ratup-source} assembles the
paper-side $\NameCert_{\NumFieldUp}$ obligation surface. The surface
records the RatUp history carrier, the RatUp history classifier, the
identity finite-extension witness, the inherited FieldExt stability rows,
the inherited exact endpoint classification, and the inherited ledger
provenance from the RatUp reflexive $\FieldExtUp$ public interface.
\end{theorem}
\begin{proof}
Use \autoref{thm:numfield-reflexive-carrier-classifier} for the carrier and
classifier slots. These slots are the RatUp history carrier and classifier,
so the $\NumFieldUp$ source is rooted in $\mathsf{BHist}$.

Use \autoref{thm:numfield-reflexive-fieldext-consumption} for the finite
field-extension slot. It supplies the identity embedding, the shared
base-and-ambient RatUp field certificates, and the self-vector-space row
required by the number-field reading.

Use \autoref{thm:numfield-reflexive-ledger-exactness} for the exactness and
ledger slots. Together these three rows cover the carrier, classifier,
finite-extension, stability, exactness, and ledger obligations named by the
paper-side $\NameCert_{\NumFieldUp}$ surface.
\end{proof}
