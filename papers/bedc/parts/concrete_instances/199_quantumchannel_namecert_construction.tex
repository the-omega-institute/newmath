\chapter{A Concrete Naming Certificate for $\QuantumChannelUp$}
\label{ch:concrete-instances-quantumchannel-namecert}

The quantum-channel interface $\QuantumChannelUp$ packages a completely positive trace-preserving (CPTP) linear map between density-matrix spaces. It depends on $\NameCert_{\DensityMatrixUp}$ (\autoref{ch:concrete-instances-densitymatrix-namecert}) and $\NameCert_{\LinearMapUp}$ (\autoref{ch:concrete-instances-linearmap-namecert}).

\begin{definition}[DensityMatrix trace-class carrier]
\label{def:densitymatrix-trace-class-carrier}
Let $H$ be a $\HilbertUp$ carrier, and let $\mathsf{TC}(H)$ be the trace-class operator-ideal carrier supplied for $H$ by the $\OperatorIdealUp$ dependency of \autoref{ch:concrete-instances-densitymatrix-namecert}. Write $\mathsf{Pos}_H(T)$ for positivity of a trace-class operator on $H$ and $\operatorname{Tr}_H(T)$ for its trace. The concrete $\DensityMatrixUp(H)$ carrier is
$$
\mathsf{Dens}_H(\rho)
:\Longleftrightarrow
\rho\in\mathsf{TC}(H)\land \mathsf{Pos}_H(\rho)\land \operatorname{Tr}_H(\rho)=1.
$$
Thus a carried density matrix is exactly positive trace-class trace-one data on the Hilbert carrier named by \autoref{sec:densitymatrix-carrier}.
\end{definition}

\begin{definition}[QuantumChannel CPTP carrier]
\label{def:quantumchannel-cptp-carrier}
Let $H$ and $K$ be $\HilbertUp$ carriers. A map $\Phi:\mathsf{TC}(H)\to\mathsf{TC}(K)$ is carried by $\QuantumChannelUp(H,K)$, written $\mathsf{QChan}_{H,K}(\Phi)$, when it carries the $\LinearMapUp(\mathsf{TC}(H),\mathsf{TC}(K))$ map-code of \autoref{def:linearmap-carrier} and satisfies the following two operator rows:
\begin{enumerate}
  \item complete positivity: for every finite ancilla Hilbert carrier $A$ and every positive trace-class operator $X$ on $A\otimes H$, the amplified operator $(\operatorname{id}_A\otimes\Phi)(X)$ is positive trace-class on $A\otimes K$;
  \item trace preservation: for every $T\in\mathsf{TC}(H)$,
$$
\operatorname{Tr}_K(\Phi(T))=\operatorname{Tr}_H(T).
$$
\end{enumerate}
The carrier is the CPTP linear-map interface announced in \autoref{ch:concrete-instances-quantumchannel-namecert}; its middle space for composition is the displayed trace-class carrier $\mathsf{TC}(K)$.
\end{definition}

\begin{lemma}[QuantumChannel density-matrix image]
\label{lem:quantumchannel-densitymatrix-image}
Let $H$ and $K$ be $\HilbertUp$ carriers. If
$$
\mathsf{QChan}_{H,K}(\Phi)
\qquad\text{and}\qquad
\mathsf{Dens}_H(\rho),
$$
then
$$
\mathsf{Dens}_K(\Phi(\rho)).
$$
\end{lemma}
\begin{proof}
Unpack $\mathsf{Dens}_H(\rho)$ by \autoref{def:densitymatrix-trace-class-carrier}. It gives that $\rho$ is trace-class on $H$, positive, and has trace one. Unpack $\mathsf{QChan}_{H,K}(\Phi)$ by \autoref{def:quantumchannel-cptp-carrier}. The carried $\LinearMapUp$ row supplies output closure, so $\Phi(\rho)$ is trace-class on $K$.

Apply complete positivity with the one-dimensional ancilla. The amplification by the one-dimensional identity is the original map on trace-class operators, hence positivity of $\rho$ gives positivity of $\Phi(\rho)$ on $K$. Trace preservation gives
$$
\operatorname{Tr}_K(\Phi(\rho))=\operatorname{Tr}_H(\rho)=1.
$$
The three rows are exactly the $\mathsf{Dens}_K$ carrier clauses in \autoref{def:densitymatrix-trace-class-carrier}.
\end{proof}

\begin{theorem}[QuantumChannel composition closure]
\label{thm:quantumchannel-composition-closure}
Let $H$, $K$, and $L$ be $\HilbertUp$ carriers with trace-class spaces $\mathsf{TC}(H)$, $\mathsf{TC}(K)$, and $\mathsf{TC}(L)$. If
$$
\mathsf{QChan}_{H,K}(\Phi)
\qquad\text{and}\qquad
\mathsf{QChan}_{K,L}(\Psi),
$$
then the composite map $\Psi\circ\Phi:\mathsf{TC}(H)\to\mathsf{TC}(L)$ is carried by $\QuantumChannelUp(H,L)$:
$$
\mathsf{QChan}_{H,L}(\Psi\circ\Phi).
$$
Moreover every $\DensityMatrixUp(H)$ input is sent to a $\DensityMatrixUp(L)$ output:
$$
\mathsf{Dens}_H(\rho)\Longrightarrow\mathsf{Dens}_L((\Psi\circ\Phi)(\rho)).
$$
\end{theorem}
\begin{proof}
Unpack the two channel carriers by \autoref{def:quantumchannel-cptp-carrier}. Their $\LinearMapUp$ rows are maps
$$
\Phi:\mathsf{TC}(H)\to\mathsf{TC}(K)
\qquad\text{and}\qquad
\Psi:\mathsf{TC}(K)\to\mathsf{TC}(L)
$$
with the matching middle trace-class carrier $\mathsf{TC}(K)$. The trace-class carriers are module and vector-space carriers inherited from the operator-ideal and Hilbert surfaces cited in \autoref{ch:concrete-instances-densitymatrix-namecert}. Therefore the composition row of \autoref{def:linearmap-stability-certificate}, concretely supplied by \autoref{thm:module-linearmap-composition-certificate}, gives the carried $\LinearMapUp(\mathsf{TC}(H),\mathsf{TC}(L))$ map-code for $\Psi\circ\Phi$.

For complete positivity, let $A$ be a finite ancilla Hilbert carrier and let $X$ be a positive trace-class operator on $A\otimes H$. Complete positivity of $\Phi$ gives that
$$
Y:=(\operatorname{id}_A\otimes\Phi)(X)
$$
is positive trace-class on $A\otimes K$. Complete positivity of $\Psi$ applied to $Y$ gives that
$$
(\operatorname{id}_A\otimes\Psi)(Y)
$$
is positive trace-class on $A\otimes L$. The amplified-map ledger is functorial in the operator argument, so
$$
(\operatorname{id}_A\otimes\Psi)((\operatorname{id}_A\otimes\Phi)(X))
=
(\operatorname{id}_A\otimes(\Psi\circ\Phi))(X).
$$
Thus $\Psi\circ\Phi$ is completely positive.

For trace preservation, let $T\in\mathsf{TC}(H)$. Output closure for $\Phi$ gives $\Phi(T)\in\mathsf{TC}(K)$. Trace preservation for $\Psi$, followed by trace preservation for $\Phi$, gives
$$
\operatorname{Tr}_L((\Psi\circ\Phi)(T))
=
\operatorname{Tr}_L(\Psi(\Phi(T)))
=
\operatorname{Tr}_K(\Phi(T))
=
\operatorname{Tr}_H(T).
$$
The linear-map row, complete positivity row, and trace-preservation row are precisely the three clauses of \autoref{def:quantumchannel-cptp-carrier}; hence $\mathsf{QChan}_{H,L}(\Psi\circ\Phi)$ holds.

Finally assume $\mathsf{Dens}_H(\rho)$. Applying \autoref{lem:quantumchannel-densitymatrix-image} to the composite channel just constructed gives
$$
\mathsf{Dens}_L((\Psi\circ\Phi)(\rho)),
$$
which is the asserted $\DensityMatrixUp$ output closure.
\end{proof}

\begin{definition}[QuantumChannel affine mixture spine]
\label{def:quantumchannel-affine-mixture-spine}
Let $H$ and $K$ be $\HilbertUp$ carriers, and write
$$
\mathsf{Map}_{H,K}:=\mathsf{TC}(H)\to\mathsf{TC}(K).
$$
Use the pointwise real scalar action and pointwise trace-class addition on maps,
$$
(a\odot_{H,K}\Phi)(T):=a\odot_K\Phi(T),
\qquad
(\Phi\oplus_{H,K}\Psi)(T):=\Phi(T)+_{\mathsf{TC}(K)}\Psi(T).
$$
For a finite map spine $\Phi_s:\mathsf{ListCarrier}(\mathsf{Map}_{H,K})$, a finite real-weight spine $\lambda:\mathsf{ListCarrier}(\RealUp)$, and an endpoint $\Phi_{\mathrm{mix}}:\mathsf{Map}_{H,K}$, define
$$
\mathsf{QChanAffSpine}_{H,K}(\Phi_s,\lambda,\Phi_{\mathrm{mix}})
:\Longleftrightarrow
\mathsf{ConvAffSpine}_{\mathsf{QChan}_{H,K}}(\Phi_s,\lambda,\Phi_{\mathrm{mix}}),
$$
where the right side is the finite affine-spine datum of \autoref{def:convexset-finite-affine-spine-data} read with carrier $C=\mathsf{QChan}_{H,K}$, scalar action $\odot_{H,K}$, and addition $\oplus_{H,K}$. Thus the datum records carried channel entries, nonnegative real weights with total weight classified as one, and a recursive pointwise affine endpoint.
\end{definition}

\begin{lemma}[QuantumChannel binary convex mixture closure]
\label{lem:quantumchannel-binary-convex-mixture-closure}
Let $H$ and $K$ be $\HilbertUp$ carriers. Let $\Phi,\Psi:\mathsf{TC}(H)\to\mathsf{TC}(K)$ and let $a,b$ be real scalar coefficients. Define $\Phi_{\mathrm{mix}}:\mathsf{TC}(H)\to\mathsf{TC}(K)$ pointwise by
$$
\Phi_{\mathrm{mix}}(T)
=
(a\odot_K\Phi(T))+_{\mathsf{TC}(K)}(b\odot_K\Psi(T)).
$$
If
$$
\mathsf{QChan}_{H,K}(\Phi)
\land
\mathsf{QChan}_{H,K}(\Psi)
\land
\mathsf{NonNeg}_{\RealUp}(a)
\land
\mathsf{NonNeg}_{\RealUp}(b)
\land
 a+b=1,
$$
then
$$
\mathsf{QChan}_{H,K}(\Phi_{\mathrm{mix}}).
$$
\end{lemma}
\begin{proof}
Unpack the two channel hypotheses by \autoref{def:quantumchannel-cptp-carrier}. Their first rows carry the two $\LinearMapUp(\mathsf{TC}(H),\mathsf{TC}(K))$ map-codes of \autoref{def:linearmap-carrier}. The trace-class carriers are real vector-space and module carriers inherited from the $\OperatorIdealUp$ and $\HilbertUp$ dependency surfaces cited in \autoref{ch:concrete-instances-densitymatrix-namecert}.

The pointwise formula for $\Phi_{\mathrm{mix}}$ gives a carried linear-map code. Indeed, output closure follows from output closure for $\Phi$ and $\Psi$ together with scalar-action and addition closure in $\mathsf{TC}(K)$. Source-classifier descent follows by applying the descent rows for $\Phi$ and $\Psi$, then scalar-action congruence and additive congruence in the target trace-class module. Additivity follows from the additivity rows for $\Phi$ and $\Psi$, scalar distributivity over target addition, and the middle-four additive law in the target module. Scalar compatibility follows from the scalar rows for $\Phi$ and $\Psi$, scalar associativity and commutativity in the real field, and target scalar distributivity over addition. Zero preservation follows from the zero rows for $\Phi$ and $\Psi$, the target zero-action laws, and the target additive unit law. These are exactly the rows of \autoref{def:module-linearmap-certificate-package}; equivalently, they are the pointwise-sum mechanism of \autoref{thm:module-linearmap-pointwise-sum-certificate} applied after checking the two pointwise scalar multiples. Hence $\Phi_{\mathrm{mix}}$ carries the required $\LinearMapUp(\mathsf{TC}(H),\mathsf{TC}(K))$ map-code.

For complete positivity, let $A$ be a finite ancilla Hilbert carrier and let $X$ be a positive trace-class operator on $A\otimes H$. Complete positivity of $\Phi$ and $\Psi$ gives positive trace-class operators
$$
Y_{\Phi}:=(\operatorname{id}_A\otimes\Phi)(X),
\qquad
Y_{\Psi}:=(\operatorname{id}_A\otimes\Psi)(X)
$$
on $A\otimes K$. The amplified-map ledger is pointwise affine in the channel argument, so
$$
(\operatorname{id}_A\otimes\Phi_{\mathrm{mix}})(X)
=
(a\odot_{A\otimes K}Y_{\Phi})+_{\mathsf{TC}(A\otimes K)}(b\odot_{A\otimes K}Y_{\Psi}).
$$
The positivity cone on trace-class operators is closed under nonnegative real scalar action and addition, and the hypotheses supply $\mathsf{NonNeg}_{\RealUp}(a)$ and $\mathsf{NonNeg}_{\RealUp}(b)$. Therefore $(\operatorname{id}_A\otimes\Phi_{\mathrm{mix}})(X)$ is positive trace-class on $A\otimes K$. Since $A$ and $X$ were arbitrary, $\Phi_{\mathrm{mix}}$ is completely positive.

For trace preservation, let $T\in\mathsf{TC}(H)$. Trace linearity over the trace-class vector carrier and trace preservation for $\Phi$ and $\Psi$ give
$$
\begin{aligned}
\operatorname{Tr}_K(\Phi_{\mathrm{mix}}(T))
&=
 a\operatorname{Tr}_K(\Phi(T))+b\operatorname{Tr}_K(\Psi(T))\\
&=
 a\operatorname{Tr}_H(T)+b\operatorname{Tr}_H(T)\\
&=
 (a+b)\operatorname{Tr}_H(T)\\
&=
 \operatorname{Tr}_H(T).
\end{aligned}
$$
Thus the linear-map row, complete-positivity row, and trace-preservation row required by \autoref{def:quantumchannel-cptp-carrier} all hold for $\Phi_{\mathrm{mix}}$.
\end{proof}

\begin{theorem}[QuantumChannel finite convex mixture closure]
\label{thm:quantumchannel-finite-convex-mixture-closure}
Let $H$ and $K$ be $\HilbertUp$ carriers. For every finite map spine $\Phi_s:\mathsf{ListCarrier}(\mathsf{Map}_{H,K})$, finite real-weight spine $\lambda:\mathsf{ListCarrier}(\RealUp)$, and endpoint $\Phi_{\mathrm{mix}}:\mathsf{Map}_{H,K}$, if
$$
\mathsf{QChanAffSpine}_{H,K}(\Phi_s,\lambda,\Phi_{\mathrm{mix}}),
$$
then the pointwise affine mixture is carried by $\QuantumChannelUp(H,K)$:
$$
\mathsf{QChan}_{H,K}(\Phi_{\mathrm{mix}}).
$$
Equivalently, any finite pointwise affine mixture of carried $\mathsf{QChan}_{H,K}$ entries with nonnegative real weights whose finite sum is classified as one is again a carried quantum channel from $H$ to $K$.
\end{theorem}
\leanchecked{BEDC.Derived.QuantumChannelUp.QuantumChannelAffineMixtureSpine\_finite\_closure}
\begin{proof}
Unpack \autoref{def:quantumchannel-affine-mixture-spine}. The hypothesis is the finite affine-spine datum of \autoref{def:convexset-finite-affine-spine-data} specialized to the concrete carrier $C=\mathsf{QChan}_{H,K}$ on the pointwise map carrier $\mathsf{Map}_{H,K}$. It contains the visible finite readback of carried channel entries and nonnegative unit-sum weights, together with a local bracketing witness for the endpoint $\Phi_{\mathrm{mix}}$.

Induct on the local bracketing witness. In the singleton clause, the endpoint is the displayed map $\Phi$, and the clause itself supplies $\mathsf{QChan}_{H,K}(\Phi)$.

In the binary-extension clause, the induction hypothesis gives that the tail endpoint $\Theta$ is carried by $\mathsf{QChan}_{H,K}$. The same clause supplies a carried head map $\Phi$, nonnegative real coefficients $a,b$, and the local unit-sum identity $a+b=1$. Its endpoint is the pointwise map
$$
T\longmapsto (a\odot_K\Phi(T))+_{\mathsf{TC}(K)}(b\odot_K\Theta(T)).
$$
Applying \autoref{lem:quantumchannel-binary-convex-mixture-closure} to $\Phi$, $\Theta$, $a$, and $b$ gives that this endpoint is carried by $\mathsf{QChan}_{H,K}$.

The induction covers every recursive affine bracketing packaged by $\mathsf{QChanAffSpine}_{H,K}(\Phi_s,\lambda,\Phi_{\mathrm{mix}})$. Hence $\mathsf{QChan}_{H,K}(\Phi_{\mathrm{mix}})$ holds.
\end{proof}

\begin{theorem}[QuantumChannel finite density-mixture preservation]
\label{thm:quantumchannel-density-finite-mixture-preservation}
Let $H$ and $K$ be $\HilbertUp$ carriers, let $\Phi:\mathsf{TC}(H)\to\mathsf{TC}(K)$, let $xs:\mathsf{ListCarrier}(\mathsf{TC}(H))$ be a finite trace-class spine, let $\lambda:\mathsf{ListCarrier}(\RealUp)$ be a finite real-weight spine, and let $\rho_{\mathrm{mix}}\in\mathsf{TC}(H)$. Write $\Phi_{\ast}xs$ for the entrywise image spine defined by
$$
\Phi_{\ast}(\mathsf{nil}) := \mathsf{nil},
\qquad
\Phi_{\ast}(\mathsf{cons}(\rho,xs')) :=
\mathsf{cons}(\Phi(\rho),\Phi_{\ast}(xs')).
$$
If
$$
\mathsf{QChan}_{H,K}(\Phi)
\qquad\text{and}\qquad
\mathsf{DensAffSpine}_H(xs,\lambda,\rho_{\mathrm{mix}}),
$$
then
$$
\mathsf{DensAffSpine}_K(\Phi_{\ast}xs,\lambda,\Phi(\rho_{\mathrm{mix}}))
\qquad\text{and}\qquad
\mathsf{Dens}_K(\Phi(\rho_{\mathrm{mix}})).
$$
\end{theorem}
\begin{proof}
Unpack $\mathsf{QChan}_{H,K}(\Phi)$ by \autoref{def:quantumchannel-cptp-carrier}. Its carried $\LinearMapUp(\mathsf{TC}(H),\mathsf{TC}(K))$ row, read through \autoref{def:linearmap-carrier}, supplies output closure, additivity, and scalar compatibility on the trace-class vector carriers. Hence for carried trace-class endpoints $T,U$ and real scalars $a,b$,
$$
\Phi\bigl((a\odot_H T)+_{\mathsf{TC}(H)}(b\odot_H U)\bigr)
=
(a\odot_K\Phi(T))+_{\mathsf{TC}(K)}(b\odot_K\Phi(U)).
$$

Unpack $\mathsf{DensAffSpine}_H(xs,\lambda,\rho_{\mathrm{mix}})$ by \autoref{def:densitymatrix-affine-mixture-spine}. This is the affine-spine datum of \autoref{def:convexset-finite-affine-spine-data} for the concrete carrier $\mathsf{Dens}_H$. The global readback gives a position match between $xs$ and $\lambda$, carried source entries, nonnegative weights, and unit finite weight sum. The recursive definition of $\Phi_{\ast}xs$ has the same finite positions as $xs$, so the same position match applies to $\Phi_{\ast}xs$ and $\lambda$. For each displayed source entry $\rho$, the readback gives $\mathsf{Dens}_H(\rho)$; applying \autoref{lem:quantumchannel-densitymatrix-image} gives $\mathsf{Dens}_K(\Phi(\rho))$. The weight nonnegativity clauses and the finite unit-sum clause are unchanged, since the weight spine is still $\lambda$.

It remains to rebuild the local bracketing witness over $K$. Induct on the source bracketing witness packaged by \autoref{def:convexset-finite-affine-spine-data}. In the singleton clause, the source clause supplies $\mathsf{Dens}_H(\rho)$, nonnegativity of the displayed weight, and classification of that weight as the scalar unit. The density-image lemma gives $\mathsf{Dens}_K(\Phi(\rho))$, and the singleton bracketing clause over $K$ gives the target bracket for $\mathsf{cons}(\Phi(\rho),\mathsf{nil})$ with endpoint $\Phi(\rho)$.

In the binary-extension clause, write the source endpoint as
$$
z=(a\odot_H\rho)+_{\mathsf{TC}(H)}(b\odot_H\tau).
$$
The induction hypothesis gives the target tail bracket with endpoint $\Phi(\tau)$. The source clause supplies $\mathsf{Dens}_H(\rho)$, nonnegativity of $a$ and $b$, and the same local unit-sum comparison. By \autoref{lem:quantumchannel-densitymatrix-image}, $\mathsf{Dens}_K(\Phi(\rho))$. The binary bracketing clause over $K$ therefore builds the endpoint
$$
(a\odot_K\Phi(\rho))+_{\mathsf{TC}(K)}(b\odot_K\Phi(\tau)).
$$
The displayed linearity identity rewrites this target endpoint as $\Phi(z)$. Thus the binary target bracket matches the entrywise image spine and has endpoint $\Phi(z)$.

The induction supplies the target bracketing witness for the whole source witness, and the global readback was already rebuilt. Repacking these two rows by \autoref{def:densitymatrix-affine-mixture-spine} gives
$$
\mathsf{DensAffSpine}_K(\Phi_{\ast}xs,\lambda,\Phi(\rho_{\mathrm{mix}})).
$$
Applying \autoref{thm:densitymatrix-finite-affine-mixture-closure} to this target density affine spine gives $\mathsf{Dens}_K(\Phi(\rho_{\mathrm{mix}}))$.
\end{proof}

\begin{theorem}[QuantumChannel identity channel is CPTP]
\label{thm:quantumchannel-identity-channel-cptp}
Let $H$ be a $\HilbertUp$ carrier, and let $\operatorname{id}_{\mathsf{TC}(H)}:\mathsf{TC}(H)\to\mathsf{TC}(H)$ be the identity function on the trace-class carrier supplied by the $\OperatorIdealUp$ dependency of \autoref{ch:concrete-instances-densitymatrix-namecert}. Then
$$
\mathsf{QChan}_{H,H}(\operatorname{id}_{\mathsf{TC}(H)}).
$$
Moreover every carried density matrix is sent to the same carried density-matrix endpoint:
$$
\mathsf{Dens}_H(\rho)
\Longrightarrow
\operatorname{id}_{\mathsf{TC}(H)}(\rho)=\rho
\land
\mathsf{Dens}_H(\operatorname{id}_{\mathsf{TC}(H)}(\rho)).
$$
\end{theorem}
\leanchecked{BEDC.Derived.QuantumChannelUp.QuantumChannelSingleton\_identity\_channel\_cptp}
\begin{proof}
The trace-class carrier $\mathsf{TC}(H)$ is a module carrier over the scalar certificate used by the trace-class operator-ideal surface. Applying \autoref{thm:module-linearmap-identity-certificate} with $M=\mathsf{TC}(H)$ gives the module identity certificate for $\operatorname{id}_{\mathsf{TC}(H)}$. Read through \autoref{def:linearmap-carrier}, this is exactly the carried $\LinearMapUp(\mathsf{TC}(H),\mathsf{TC}(H))$ map-code required as the first row of \autoref{def:quantumchannel-cptp-carrier}.

For complete positivity, let $A$ be a finite ancilla Hilbert carrier and let $X$ be a positive trace-class operator on $A\otimes H$. The amplified identity map is the identity on the trace-class carrier over $A\otimes H$:
$$
(\operatorname{id}_A\otimes\operatorname{id}_{\mathsf{TC}(H)})(X)=X.
$$
The input $X$ is already positive trace-class on $A\otimes H$, and the target channel also has Hilbert carrier $H$. Hence the complete-positivity row of \autoref{def:quantumchannel-cptp-carrier} holds.

For trace preservation, let $T\in\mathsf{TC}(H)$. Since the map is the identity on $\mathsf{TC}(H)$,
$$
\operatorname{Tr}_H(\operatorname{id}_{\mathsf{TC}(H)}(T))
=
\operatorname{Tr}_H(T).
$$
Thus the linear-map row, complete-positivity row, and trace-preservation row are precisely the three clauses of \autoref{def:quantumchannel-cptp-carrier}, so
$$
\mathsf{QChan}_{H,H}(\operatorname{id}_{\mathsf{TC}(H)})
$$
holds.

Now assume $\mathsf{Dens}_H(\rho)$. The identity equation gives $\operatorname{id}_{\mathsf{TC}(H)}(\rho)=\rho$. Applying \autoref{lem:quantumchannel-densitymatrix-image} to the channel just constructed gives
$$
\mathsf{Dens}_H(\operatorname{id}_{\mathsf{TC}(H)}(\rho)),
$$
which is the asserted density-matrix output clause.
\end{proof}

\begin{theorem}[QuantumChannel composition associativity law]
\label{thm:quantumchannel-composition-associativity-law}
Let $H$, $K$, $L$, and $J$ be $\HilbertUp$ carriers with trace-class spaces $\mathsf{TC}(H)$, $\mathsf{TC}(K)$, $\mathsf{TC}(L)$, and $\mathsf{TC}(J)$. Write $\sim_{\mathsf{TC}(J)}$ for the classifier on the final trace-class module inherited from the operator-ideal certificate. If
$$
\mathsf{QChan}_{H,K}(\Phi),
\qquad
\mathsf{QChan}_{K,L}(\Psi),
\qquad
\mathsf{QChan}_{L,J}(\Theta),
$$
then both bracketings of the triple composite are carried by $\QuantumChannelUp(H,J)$:
$$
\mathsf{QChan}_{H,J}((\Theta\circ\Psi)\circ\Phi)
\qquad\text{and}\qquad
\mathsf{QChan}_{H,J}(\Theta\circ(\Psi\circ\Phi)).
$$
Their underlying map-code endpoints are related by the pointwise $\LinearMapUp(\mathsf{TC}(H),\mathsf{TC}(J))$ classifier:
$$
(\Theta\circ\Psi)\circ\Phi
\sim_{\LinearMapUp(\mathsf{TC}(H),\mathsf{TC}(J))}
\Theta\circ(\Psi\circ\Phi).
$$
Equivalently, every carried trace-class endpoint $T:\mathsf{TC}(H)$ satisfies
$$
((\Theta\circ\Psi)\circ\Phi)(T)
\sim_{\mathsf{TC}(J)}
(\Theta\circ(\Psi\circ\Phi))(T).
$$
Moreover, every carried density-matrix input is sent to carried density-matrix outputs by both bracketings:
$$
\mathsf{Dens}_H(\rho)
\Longrightarrow
\mathsf{Dens}_J(((\Theta\circ\Psi)\circ\Phi)(\rho))
\land
\mathsf{Dens}_J((\Theta\circ(\Psi\circ\Phi))(\rho)).
$$
\end{theorem}
\begin{proof}
Apply \autoref{thm:quantumchannel-composition-closure} to $\Phi$ and $\Psi$. This gives
$$
\mathsf{QChan}_{H,L}(\Psi\circ\Phi).
$$
Applying the same theorem to $\Psi\circ\Phi$ and $\Theta$ gives
$$
\mathsf{QChan}_{H,J}(\Theta\circ(\Psi\circ\Phi)).
$$
In the other bracketing, first apply \autoref{thm:quantumchannel-composition-closure} to $\Psi$ and $\Theta$, obtaining
$$
\mathsf{QChan}_{K,J}(\Theta\circ\Psi).
$$
Applying the same theorem to $\Phi$ and $\Theta\circ\Psi$ gives
$$
\mathsf{QChan}_{H,J}((\Theta\circ\Psi)\circ\Phi).
$$
By \autoref{def:quantumchannel-cptp-carrier}, these two carrier conclusions include the carried $\LinearMapUp$ map-code row, complete positivity for every finite ancilla, and trace preservation on $\mathsf{TC}(H)$.

It remains to compare the two carried map-code endpoints. Unpack the three channel hypotheses by \autoref{def:quantumchannel-cptp-carrier}. Their first rows are carried $\LinearMapUp$ map-codes, read concretely as the module-linear packages of \autoref{def:module-linearmap-certificate-package} through \autoref{def:linearmap-carrier}. The trace-class carriers are module carriers over the common scalar source inherited from the $\OperatorIdealUp$ and $\HilbertUp$ dependency surfaces cited in \autoref{ch:concrete-instances-densitymatrix-namecert}. Applying \autoref{thm:module-linearmap-composition-associativity-certificate} with
$$
M=\mathsf{TC}(H),\qquad
N=\mathsf{TC}(K),\qquad
P=\mathsf{TC}(L),\qquad
Q=\mathsf{TC}(J),
$$
and with $f=\Phi$, $g=\Psi$, and $k=\Theta$, gives the pointwise $\LinearMapUp(\mathsf{TC}(H),\mathsf{TC}(J))$ classifier comparison
$$
(\Theta\circ\Psi)\circ\Phi
\sim_{\LinearMapUp(\mathsf{TC}(H),\mathsf{TC}(J))}
\Theta\circ(\Psi\circ\Phi).
$$
This is exactly the classifier specified in \autoref{def:linearmap-classifier-specification}: it is pointwise comparison in the final trace-class classifier, together with the two carried map-code endpoints already supplied by the two channel-carrier conclusions above.

Concretely, let $T$ be a carried trace-class endpoint of $\mathsf{TC}(H)$. The output-closure rows of the three input channel map-codes make $\Phi(T)$ carried in $\mathsf{TC}(K)$, $\Psi(\Phi(T))$ carried in $\mathsf{TC}(L)$, and $\Theta(\Psi(\Phi(T)))$ carried in $\mathsf{TC}(J)$. Both bracketed composites evaluate to this same endpoint:
$$
((\Theta\circ\Psi)\circ\Phi)(T)
=
\Theta(\Psi(\Phi(T)))
=
(\Theta\circ(\Psi\circ\Phi))(T).
$$
Classifier reflexivity for the final trace-class module therefore gives
$$
((\Theta\circ\Psi)\circ\Phi)(T)
\sim_{\mathsf{TC}(J)}
(\Theta\circ(\Psi\circ\Phi))(T),
$$
which is the displayed pointwise form.

Finally assume $\mathsf{Dens}_H(\rho)$. Applying \autoref{lem:quantumchannel-densitymatrix-image} to the channel $\mathsf{QChan}_{H,J}((\Theta\circ\Psi)\circ\Phi)$ gives
$$
\mathsf{Dens}_J(((\Theta\circ\Psi)\circ\Phi)(\rho)).
$$
Applying the same lemma to $\mathsf{QChan}_{H,J}(\Theta\circ(\Psi\circ\Phi))$ gives
$$
\mathsf{Dens}_J((\Theta\circ(\Psi\circ\Phi))(\rho)).
$$
These are the two asserted density-output clauses.
\end{proof}

\begin{definition}[Unitary trace-class conjugation map]
\label{def:quantumchannel-unitary-conjugation-map}
Let $H$ be a $\HilbertUp$ carrier and let $U:H\to H$ be an endpoint carried by the $\UnitaryGroupUp(H)$ carrier of \autoref{sec:unitarygroup-carrier}. Write $U^{-1}:H\to H$ for the inverse endpoint supplied by the same unitary automorphism row. For $T\in\mathsf{TC}(H)$ define the unitary trace-class conjugation expression by
$$
\operatorname{Ad}^{\mathsf{TC}}_U(T):=U\circ T\circ U^{-1}.
$$
The expression is read in the bounded-operator composition ring over $H$ from \autoref{def:operatorideal-hilbert-source-surface}; its trace-class output closure is the first carrier row proved in \autoref{thm:unitary-conjugation-quantumchannel}.
\end{definition}

\begin{theorem}[Unitary conjugation is a QuantumChannel]
\label{thm:unitary-conjugation-quantumchannel}
Let $H$ be a $\HilbertUp$ carrier. If $U:H\to H$ is carried by the $\UnitaryGroupUp(H)$ automorphism carrier and $U^{-1}$ is its carried inverse, then the conjugation map of \autoref{def:quantumchannel-unitary-conjugation-map} is carried by $\QuantumChannelUp(H,H)$:
$$
\mathsf{QChan}_{H,H}(\operatorname{Ad}^{\mathsf{TC}}_U).
$$
Consequently every carried density matrix is sent to a carried density matrix:
$$
\mathsf{Dens}_H(\rho)\Longrightarrow \mathsf{Dens}_H(\operatorname{Ad}^{\mathsf{TC}}_U(\rho)).
$$
\end{theorem}
\begin{proof}
Unpack the $\UnitaryGroupUp$ carrier cited in \autoref{sec:unitarygroup-carrier}. The unitary witness supplies bounded $\LinearMapUp(H,H)$ endpoints $U$ and $U^{-1}$, the two inverse equations against the Hilbert identity, and the inner-product preservation row identifying $U^{-1}$ with the adjoint-side action of $U$. Hence both endpoints are carried bounded operators in the ambient operator surface $\mathsf{Op}_H$ of \autoref{def:operatorideal-hilbert-source-surface}.

First prove trace-class output closure. Let $T\in\mathsf{TC}(H)$. Apply the right absorption half of \autoref{thm:operatorideal-two-sided-absorption-row} to the carried bounded operator $U^{-1}$ and the trace-class operator $T$. This gives $T\circ U^{-1}\in\mathsf{TC}(H)$. Apply the left absorption half of the same theorem to the carried bounded operator $U$ and this trace-class operator. The result is
$$
U\circ T\circ U^{-1}\in\mathsf{TC}(H),
$$
which is output closure for $\operatorname{Ad}^{\mathsf{TC}}_U$.

The same bounded-composition rows give the linear-map package. Source-classifier descent follows from the operator-ideal classifier transport row, since composing on the left by the fixed carried bounded operator $U$ and on the right by the fixed carried bounded operator $U^{-1}$ preserves the pointwise bounded-operator classifier. For carried $T,S\in\mathsf{TC}(H)$ and a carried real scalar $a$, bilinearity of bounded-operator composition over the trace-class module gives
$$
\operatorname{Ad}^{\mathsf{TC}}_U(T+_{\mathsf{TC}(H)}S)
=
\operatorname{Ad}^{\mathsf{TC}}_U(T)+_{\mathsf{TC}(H)}\operatorname{Ad}^{\mathsf{TC}}_U(S)
$$
and
$$
\operatorname{Ad}^{\mathsf{TC}}_U(a\odot_H T)
=
a\odot_H\operatorname{Ad}^{\mathsf{TC}}_U(T).
$$
The zero row is the same calculation at the trace-class zero operator. These are the output-closure, descent, additive, scalar, and zero rows of \autoref{def:module-linearmap-certificate-package}, read through the concrete $\LinearMapUp$ carrier in \autoref{def:linearmap-carrier}. Thus $\operatorname{Ad}^{\mathsf{TC}}_U$ carries the required linear-map row in \autoref{def:quantumchannel-cptp-carrier}.

For complete positivity, let $A$ be a finite ancilla Hilbert carrier and let $X$ be a positive trace-class operator on $A\otimes H$. Put
$$
V_A:=\operatorname{id}_A\otimes U,
\qquad
V_A^{-1}:=\operatorname{id}_A\otimes U^{-1}.
$$
The tensor row for Hilbert carriers makes $V_A$ a carried unitary automorphism of $A\otimes H$ with inverse $V_A^{-1}$. The amplified conjugation map is the corresponding bounded conjugation:
$$
(\operatorname{id}_A\otimes\operatorname{Ad}^{\mathsf{TC}}_U)(X)
=
V_A\circ X\circ V_A^{-1}.
$$
Trace-class membership of this amplified output follows from \autoref{thm:operatorideal-two-sided-absorption-row} on $A\otimes H$. For positivity, evaluate the quadratic-form predicate at an arbitrary carried vector $z$ of $A\otimes H$. The unitary inverse-adjoint row gives
$$
\langle V_A\circ X\circ V_A^{-1}z,z\rangle
=
\langle X(V_A^{-1}z),V_A^{-1}z\rangle.
$$
Since $X$ is positive and $V_A^{-1}z$ is again a carried Hilbert endpoint, the right side is nonnegative. Therefore the amplified output is positive trace-class on $A\otimes H$. As $A$ and $X$ were arbitrary, $\operatorname{Ad}^{\mathsf{TC}}_U$ is completely positive.

For trace preservation, let $T\in\mathsf{TC}(H)$. The cyclic trace row for one trace-class factor and one carried bounded factor gives
$$
\begin{aligned}
\operatorname{Tr}_H(\operatorname{Ad}^{\mathsf{TC}}_U(T))
&=\operatorname{Tr}_H(U\circ (T\circ U^{-1}))\\
&=\operatorname{Tr}_H((T\circ U^{-1})\circ U)\\
&=\operatorname{Tr}_H(T\circ (U^{-1}\circ U))\\
&=\operatorname{Tr}_H(T\circ \operatorname{id}_H)\\
&=\operatorname{Tr}_H(T).
\end{aligned}
$$
The last two equalities use the inverse equation supplied by the unitary automorphism row and the bounded-operator identity law.

The carried linear-map row, complete-positivity row, and trace-preservation row are precisely the clauses of \autoref{def:quantumchannel-cptp-carrier}; hence
$$
\mathsf{QChan}_{H,H}(\operatorname{Ad}^{\mathsf{TC}}_U)
$$
holds. If $\mathsf{Dens}_H(\rho)$, applying \autoref{lem:quantumchannel-densitymatrix-image} to this channel gives
$$
\mathsf{Dens}_H(\operatorname{Ad}^{\mathsf{TC}}_U(\rho)),
$$
which is the asserted density-matrix endpoint preservation.
\end{proof}

\begin{theorem}[QuantumChannel identity composition unit laws]
\label{thm:quantumchannel-identity-composition-unit-laws}
Let $H$ and $K$ be $\HilbertUp$ carriers with trace-class spaces $\mathsf{TC}(H)$ and $\mathsf{TC}(K)$. Let $\operatorname{id}_{\mathsf{TC}(H)}$ and $\operatorname{id}_{\mathsf{TC}(K)}$ be the identity functions on these trace-class carriers, and write $\sim_{\mathsf{TC}(K)}$ for the target trace-class classifier inherited from the $\OperatorIdealUp$ dependency surface. If
$$
\mathsf{QChan}_{H,K}(\Phi),
$$
then both unit composites are carried by $\QuantumChannelUp(H,K)$:
$$
\mathsf{QChan}_{H,K}(\Phi\circ\operatorname{id}_{\mathsf{TC}(H)})
\qquad\text{and}\qquad
\mathsf{QChan}_{H,K}(\operatorname{id}_{\mathsf{TC}(K)}\circ\Phi).
$$
Moreover both unit composites are related to $\Phi$ by the pointwise $\LinearMapUp(\mathsf{TC}(H),\mathsf{TC}(K))$ classifier:
$$
\Phi\circ\operatorname{id}_{\mathsf{TC}(H)}
\sim_{\LinearMapUp(\mathsf{TC}(H),\mathsf{TC}(K))}
\Phi
\qquad\text{and}\qquad
\operatorname{id}_{\mathsf{TC}(K)}\circ\Phi
\sim_{\LinearMapUp(\mathsf{TC}(H),\mathsf{TC}(K))}
\Phi.
$$
Equivalently, every carried trace-class endpoint $T\in\mathsf{TC}(H)$ satisfies
$$
(\Phi\circ\operatorname{id}_{\mathsf{TC}(H)})(T)
\sim_{\mathsf{TC}(K)}
\Phi(T)
\qquad\text{and}\qquad
(\operatorname{id}_{\mathsf{TC}(K)}\circ\Phi)(T)
\sim_{\mathsf{TC}(K)}
\Phi(T).
$$
\end{theorem}
\begin{proof}
Apply \autoref{thm:quantumchannel-identity-channel-cptp} to $H$ and to $K$. This gives
$$
\mathsf{QChan}_{H,H}(\operatorname{id}_{\mathsf{TC}(H)})
\qquad\text{and}\qquad
\mathsf{QChan}_{K,K}(\operatorname{id}_{\mathsf{TC}(K)}).
$$
Applying \autoref{thm:quantumchannel-composition-closure} first to $\operatorname{id}_{\mathsf{TC}(H)}$ and $\Phi$ gives
$$
\mathsf{QChan}_{H,K}(\Phi\circ\operatorname{id}_{\mathsf{TC}(H)}).
$$
Applying the same theorem to $\Phi$ and $\operatorname{id}_{\mathsf{TC}(K)}$ gives
$$
\mathsf{QChan}_{H,K}(\operatorname{id}_{\mathsf{TC}(K)}\circ\Phi).
$$
These are the two carried quantum-channel clauses.

It remains to compare the underlying map-code endpoints. Unpack $\mathsf{QChan}_{H,K}(\Phi)$ by \autoref{def:quantumchannel-cptp-carrier}. Its first row carries the $\LinearMapUp(\mathsf{TC}(H),\mathsf{TC}(K))$ map-code of \autoref{def:linearmap-carrier}. The trace-class carriers are module carriers over the scalar certificate inherited from the $\OperatorIdealUp$ and $\HilbertUp$ dependency surfaces cited in \autoref{ch:concrete-instances-densitymatrix-namecert}. Applying \autoref{thm:module-linearmap-identity-unit-laws} with
$$
M=\mathsf{TC}(H),
\qquad
N=\mathsf{TC}(K),
\qquad
f=\Phi,
$$
gives the two pointwise $\LinearMapUp(\mathsf{TC}(H),\mathsf{TC}(K))$ classifier comparisons
$$
\operatorname{id}_{\mathsf{TC}(K)}\circ\Phi
\sim_{\LinearMapUp(\mathsf{TC}(H),\mathsf{TC}(K))}
\Phi
\qquad\text{and}\qquad
\Phi\circ\operatorname{id}_{\mathsf{TC}(H)}
\sim_{\LinearMapUp(\mathsf{TC}(H),\mathsf{TC}(K))}
\Phi.
$$

For the displayed pointwise form, let $T$ be a carried trace-class endpoint of $\mathsf{TC}(H)$. The output-closure row of the carried $\LinearMapUp$ map-code for $\Phi$ gives that $\Phi(T)$ is carried in $\mathsf{TC}(K)$. The identity equations give
$$
(\Phi\circ\operatorname{id}_{\mathsf{TC}(H)})(T)=\Phi(T)
\qquad\text{and}\qquad
(\operatorname{id}_{\mathsf{TC}(K)}\circ\Phi)(T)=\Phi(T).
$$
Classifier reflexivity for the target trace-class module therefore gives
$$
(\Phi\circ\operatorname{id}_{\mathsf{TC}(H)})(T)
\sim_{\mathsf{TC}(K)}
\Phi(T)
\qquad\text{and}\qquad
(\operatorname{id}_{\mathsf{TC}(K)}\circ\Phi)(T)
\sim_{\mathsf{TC}(K)}
\Phi(T).
$$
By \autoref{def:linearmap-classifier-specification}, these pointwise target-classifier comparisons are exactly the underlying $\LinearMapUp(\mathsf{TC}(H),\mathsf{TC}(K))$ classifier relations.
\end{proof}

\begin{theorem}[Unitary conjugation channel composition law]
\label{thm:unitary-conjugation-channel-composition-law}
Let $H$ be a $\HilbertUp$ carrier. Let $U,V:H\to H$ be endpoints carried by the $\UnitaryGroupUp(H)$ automorphism carrier, with carried inverses $U^{-1}$ and $V^{-1}$. Let $W:=U\circ V$ be their unitary-group product endpoint, and let $W^{-1}:=V^{-1}\circ U^{-1}$ be the inverse endpoint supplied by the product row. Then the composite conjugation map and the product-unitary conjugation map are both carried quantum channels:
$$
\mathsf{QChan}_{H,H}(\operatorname{Ad}^{\mathsf{TC}}_U\circ\operatorname{Ad}^{\mathsf{TC}}_V)
\qquad\text{and}\qquad
\mathsf{QChan}_{H,H}(\operatorname{Ad}^{\mathsf{TC}}_{W}).
$$
Moreover the composite channel endpoint is classified by the product-unitary endpoint:
$$
\operatorname{Ad}^{\mathsf{TC}}_U\circ\operatorname{Ad}^{\mathsf{TC}}_V
\sim_{\LinearMapUp(\mathsf{TC}(H),\mathsf{TC}(H))}
\operatorname{Ad}^{\mathsf{TC}}_{W}.
$$
Equivalently, writing $\sim_{\mathsf{TC}(H)}$ for the trace-class classifier inherited from the operator-ideal surface, every carried trace-class endpoint $T\in\mathsf{TC}(H)$ satisfies
$$
(\operatorname{Ad}^{\mathsf{TC}}_U\circ\operatorname{Ad}^{\mathsf{TC}}_V)(T)
\sim_{\mathsf{TC}(H)}
\operatorname{Ad}^{\mathsf{TC}}_{U\circ V}(T).
$$
Consequently every carried density matrix is sent to carried density-matrix endpoints by both representatives:
$$
\mathsf{Dens}_H(\rho)
\Longrightarrow
\mathsf{Dens}_H((\operatorname{Ad}^{\mathsf{TC}}_U\circ\operatorname{Ad}^{\mathsf{TC}}_V)(\rho))
\land
\mathsf{Dens}_H(\operatorname{Ad}^{\mathsf{TC}}_{W}(\rho)).
$$
\end{theorem}
\begin{proof}
The $\UnitaryGroupUp(H)$ automorphism carrier cited in \autoref{sec:unitarygroup-carrier} is a group of Hilbert automorphisms. Thus its product row carries $W=U\circ V$, and the inverse row identifies the carried inverse of $W$ with $V^{-1}\circ U^{-1}$. Applying \autoref{thm:unitary-conjugation-quantumchannel} to $U$, to $V$, and to $W$ gives
$$
\mathsf{QChan}_{H,H}(\operatorname{Ad}^{\mathsf{TC}}_U),
\qquad
\mathsf{QChan}_{H,H}(\operatorname{Ad}^{\mathsf{TC}}_V),
\qquad
\mathsf{QChan}_{H,H}(\operatorname{Ad}^{\mathsf{TC}}_{W}).
$$
Apply \autoref{thm:quantumchannel-composition-closure} with $\Phi=\operatorname{Ad}^{\mathsf{TC}}_V$ and $\Psi=\operatorname{Ad}^{\mathsf{TC}}_U$. This gives the carried composite channel
$$
\mathsf{QChan}_{H,H}(\operatorname{Ad}^{\mathsf{TC}}_U\circ\operatorname{Ad}^{\mathsf{TC}}_V).
$$
Together with the displayed $W$ clause, this proves the two quantum-channel carrier conclusions.

It remains to identify the endpoints. Let $T$ be a carried endpoint of $\mathsf{TC}(H)$. By \autoref{def:quantumchannel-unitary-conjugation-map},
$$
(\operatorname{Ad}^{\mathsf{TC}}_U\circ\operatorname{Ad}^{\mathsf{TC}}_V)(T)
=
U\circ (V\circ T\circ V^{-1})\circ U^{-1}.
$$
The bounded-operator composition ring over $H$ from \autoref{def:operatorideal-hilbert-source-surface} is associative on carried endpoints, and \autoref{thm:operatorideal-two-sided-absorption-row} supplies the trace-class carrier rows for each intermediate conjugation. Hence the displayed endpoint is classified in the trace-class carrier with
$$
(U\circ V)\circ T\circ (V^{-1}\circ U^{-1}).
$$
The product inverse row gives $W^{-1}=V^{-1}\circ U^{-1}$, while $W=U\circ V$ by definition. Therefore
$$
(U\circ V)\circ T\circ (V^{-1}\circ U^{-1})
=
W\circ T\circ W^{-1}
=
\operatorname{Ad}^{\mathsf{TC}}_{W}(T).
$$
Classifier reflexivity and classifier transport in the trace-class operator-ideal carrier give
$$
(\operatorname{Ad}^{\mathsf{TC}}_U\circ\operatorname{Ad}^{\mathsf{TC}}_V)(T)
\sim_{\mathsf{TC}(H)}
\operatorname{Ad}^{\mathsf{TC}}_{W}(T).
$$
Since $T$ was arbitrary, \autoref{def:linearmap-classifier-specification} packages these pointwise trace-class comparisons, together with the two carried linear-map rows included in the two channel carrier conclusions by \autoref{def:quantumchannel-cptp-carrier}, as
$$
\operatorname{Ad}^{\mathsf{TC}}_U\circ\operatorname{Ad}^{\mathsf{TC}}_V
\sim_{\LinearMapUp(\mathsf{TC}(H),\mathsf{TC}(H))}
\operatorname{Ad}^{\mathsf{TC}}_{W}.
$$

Finally assume $\mathsf{Dens}_H(\rho)$. Applying \autoref{lem:quantumchannel-densitymatrix-image} to the composite channel gives
$$
\mathsf{Dens}_H((\operatorname{Ad}^{\mathsf{TC}}_U\circ\operatorname{Ad}^{\mathsf{TC}}_V)(\rho)).
$$
Applying the same lemma to $\mathsf{QChan}_{H,H}(\operatorname{Ad}^{\mathsf{TC}}_{W})$ gives
$$
\mathsf{Dens}_H(\operatorname{Ad}^{\mathsf{TC}}_{W}(\rho)).
$$
These are the asserted density-matrix endpoint clauses.
\end{proof}
