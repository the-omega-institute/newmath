\begin{theorem}[Constant complex sequence has constant complex limit]
\label{thm:cplx-constant-sequence-constant-limit}
Let $z$ be a complex history. Define the constant sequence $s:\Unary\to\ComplexUp$ by $s(n):=z$, and define the two trivial moduli by
$$
N_{\mathrm{const}}(k):=\emp,
\qquad
M_{\mathrm{const}}(k):=\emp.
$$
If $\mathsf{CplxRegSeq}(s,N_{\mathrm{const}})$ holds by \autoref{lem:cplx-constant-regular}, then
$$
\mathsf{CplxLim}(s,N_{\mathrm{const}},z,M_{\mathrm{const}})
$$
holds.
\end{theorem}
\begin{proof}
Fix a unary precision $k$ and a unary index $n$ satisfying the threshold clause in \autoref{def:cplx-lim} for $M_{\mathrm{const}}(k)$. Since $M_{\mathrm{const}}(k)=N_{\mathrm{const}}(k)=\emp$, the same threshold clause makes $n$ acceptable for the regularity condition of \autoref{def:cplx-regseq}. Take the second regularity index to be $m:=\emp$; it is acceptable because $\hsame(\emp,N_{\mathrm{const}}(k))$ is reflexivity.

Apply $\mathsf{CplxRegSeq}(s,N_{\mathrm{const}})$ at the indices $n$ and $m$. This gives a distance witness $D$ such that $\mathsf{CplxDist}(s(n),s(m),D)$ satisfies the unary representation of $D\leq 2^{-k}$. By the definition of the constant sequence, $s(m)=s(\emp)=z$. Therefore the same witness $D$ is exactly the distance estimate from $s(n)$ to $z$ required in \autoref{def:cplx-lim}. Since $k$ and $n$ were arbitrary, $M_{\mathrm{const}}$ is a valid complex-limit modulus for the constant sequence.
\end{proof}
\leanchecked{BEDC.Derived.ComplexLimitUp.ComplexLimit\_constant\_of\_regular}

\begin{theorem}[Eventually constant complex sequence has constant limit]
\label{thm:cplx-eventually-constant-sequence-constant-limit}
Let $s:\Unary\to\ComplexUp$ be a complex sequence, let $N,M:\Unary\to\Unary$ be unary moduli, and let $z$ be a complex history. Suppose $\mathsf{CplxRegSeq}(s,N)$ and suppose that $s$ is pointwise history-same to $z$ above $M$:
$$
\forall k,n:\Unary,\
\bigl(\mathsf{NatUnaryStrictPrefix}(M(k),n)\lor\hsame(n,M(k))\bigr)
\Longrightarrow
\hsame(s(n),z).
$$
Then
$$
\mathsf{CplxLim}(s,N,z,M).
$$
\end{theorem}
\begin{proof}
Fix a unary precision $k$ and a unary index $n$ satisfying the threshold clause in \autoref{def:cplx-lim} for $M(k)$. Put $c(u):=z$ for the constant sequence. The empty threshold used in \autoref{thm:cplx-constant-sequence-constant-limit} accepts every unary index: if $n\hsame\emp$ this is the non-strict branch, and otherwise the tail $n$ with the left-unit continuation $\ContR(\emp,n,n)$ gives $\mathsf{NatUnaryStrictPrefix}(\emp,n)$ by \autoref{def:nat-unary-strict-prefix}. Hence the constant-limit theorem supplies a distance witness $D$ from $c(n)=z$ to $z$ satisfying the unary estimate $D\leq 2^{-k}$.

The above-$M$ hypothesis gives $\hsame(s(n),z)$, hence $\hsame(s(n),c(n))$. Apply \autoref{thm:complex-distance-hsame-transport-with-relation} to the distance witness for $c(n)$ and $z$, using reflexivity of $\hsame$ on the limit endpoint and on $D$. The transported witness has endpoints $s(n)$ and $z$, and the same distance history $D$, so the estimate $D\leq 2^{-k}$ is unchanged.

Since $k$ and $n$ were arbitrary among the accepted indices, these witnesses are exactly the convergence pattern required by \autoref{def:cplx-lim}. The hypothesis $\mathsf{CplxRegSeq}(s,N)$ supplies the regular-sequence side of the predicate, so $\mathsf{CplxLim}(s,N,z,M)$ holds.
\end{proof}
\leanchecked{BEDC.Derived.ComplexLimitUp.ComplexLimit\_eventually\_constant\_closed}

\begin{definition}[Pointwise difference of complex sequences]
\label{def:cplx-pointwise-difference}
For complex sequences $s,t:\Unary\to\ComplexUp$, the pointwise difference $s\ominus t$ is the sequence
$$
(s\ominus t)(n):=s(n)-t(n),
$$
where subtraction is the componentwise rational-history subtraction continuation on the two rational components of the complex carrier.
\end{definition}
\leandef{BEDC.Derived.ComplexLimitUp.ComplexPointwiseDifference}

\begin{lemma}[Difference of regular complex sequences is regular with explicit modulus pasting]
\label{lem:cplx-difference-regular}
If $\mathsf{CplxRegSeq}(s,N_s)$ and $\mathsf{CplxRegSeq}(t,N_t)$ hold, then $\mathsf{CplxRegSeq}(s\ominus t,N_{\ominus})$ holds for the explicit pasted modulus
$$
N_{\ominus}(k):=\max(N_s(k+1),N_t(k+1)).
$$
\end{lemma}
\leanchecked{BEDC.Derived.ComplexLimitUp.ComplexRegularSequence\_pointwise\_difference\_closed}
\begin{proof}
Fix a unary precision $k$ and accepted unary indices $n,m$ above $N_{\ominus}(k)$. The maximum clause makes the same indices accepted above $N_s(k+1)$ and above $N_t(k+1)$. The two regularity hypotheses therefore give distance witnesses $D_s$ and $D_t$ for $s(n)$ to $s(m)$ and for $t(n)$ to $t(m)$, each bounded by the unary dyadic budget $2^{-(k+1)}$.

By the componentwise subtraction continuation,
$$
(s(n)-t(n))-(s(m)-t(m))=(s(n)-s(m))-(t(n)-t(m))
$$
component by component. The distance predicate of \autoref{def:cplx-distance} reads this as the modulus of a componentwise difference. Applying the symmetry of complex distance from \autoref{lem:cplx-dist-symmetric} to the $t$-term if needed, and then applying the triangle inequality of \autoref{lem:cplx-dist-triangle}, gives a distance witness for $(s\ominus t)(n)$ to $(s\ominus t)(m)$ bounded by $D_s+D_t$. Since both summands are bounded by $2^{-(k+1)}$, unary dyadic addition gives $D_s+D_t\leq 2^{-k}$. This is exactly the regularity clause of \autoref{def:cplx-regseq} for $s\ominus t$ with modulus $N_{\ominus}$.
\end{proof}

\begin{theorem}[Limit respects pointwise difference]
\label{thm:cplx-lim-difference}
If $\mathsf{CplxLim}(s,N_s,z_s,M_s)$ and $\mathsf{CplxLim}(t,N_t,z_t,M_t)$ hold, then
$$
\mathsf{CplxLim}(s\ominus t,N_{\ominus},z_s-z_t,M_{\ominus})
$$
holds with explicit pasted moduli
$$
N_{\ominus}(k):=\max(N_s(k+1),N_t(k+1)),
\qquad
M_{\ominus}(k):=\max(M_s(k+1),M_t(k+1)).
$$
\end{theorem}
\begin{proof}
The regular-sequence part is \autoref{lem:cplx-difference-regular}, applied to the regularity fields contained in the two limit hypotheses of \autoref{def:cplx-lim}.

It remains to verify the limit estimate. Fix a unary precision $k$ and a unary index $n$ accepted above $M_{\ominus}(k)$. By the maximum clause, $n$ is accepted above both $M_s(k+1)$ and $M_t(k+1)$. The two limit hypotheses provide distance witnesses $E_s$ and $E_t$ from $s(n)$ to $z_s$ and from $t(n)$ to $z_t$, each bounded by $2^{-(k+1)}$.

The componentwise subtraction ledgers give the identity
$$
(s\ominus t)(n)-(z_s-z_t)=(s(n)-z_s)-(t(n)-z_t).
$$
Using \autoref{def:cplx-distance}, the desired distance is the complex modulus of the right-hand side. Symmetry of complex distance from \autoref{lem:cplx-dist-symmetric} handles the signed second summand, and the triangle inequality of \autoref{lem:cplx-dist-triangle} bounds the displayed distance by $E_s+E_t$. The dyadic half-budget calculation gives $E_s+E_t\leq 2^{-k}$. Since $k$ and $n$ were arbitrary among the accepted indices, $M_{\ominus}$ satisfies the convergence clause of \autoref{def:cplx-lim} for the pointwise difference sequence and limit point $z_s-z_t$.
\end{proof}
\leanchecked{BEDC.Derived.ComplexLimitUp.ComplexLimit\_pointwise\_difference}
\leanchecked{BEDC.Derived.ComplexLimitUp.ComplexLimit\_pointwise\_difference\_closed}
