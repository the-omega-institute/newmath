\begin{theorem}[Product componentwise history classifier endpoint transport]
\label{thm:prod-component-history-classifier-hsame-transport}
For source-history carriers $L,R:\mathsf{BHist}\to\mathsf{Prop}$, source
classifiers $\sim_L,\sim_R:\mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$,
and histories $h,k,h',k'$, if
$\mathsf{ProdComponentHistoryClassifier}_{L,R,\sim_L,\sim_R}(h,k)$,
$\hsame(h,h')$, and $\hsame(k,k')$ hold, then
$\mathsf{ProdComponentHistoryClassifier}_{L,R,\sim_L,\sim_R}(h',k')$ holds.
\end{theorem}
\begin{proof}
The per-object fit is: traditional target ordered-pair classifier endpoint
stability; BEDC source $\mathsf{BHist}$ with $\Cont$ and $\hsame$; carrier
$\mathsf{ProdHistoryCarrier}$ through $\mathsf{ProdPairRep}$; classifier
$\mathsf{ProdComponentHistoryClassifier}$; stability obligation endpoint
transport; ledger displayed pair representations; core theorem this transport
law; exactness classifier preservation under endpoint $\hsame$; standard bridge
stable ordered-pair endpoints; status $\mathsf{ProofTarget}$. Unpack the
componentwise classifier witness into pair representations for $h$ and $k$ and
component classifier witnesses $l_h\sim_L l_k$ and $r_h\sim_R r_k$. Apply
Theorem~\ref{thm:prod-pair-rep-hsame-transport} to move the displayed
representation of $h$ along $\hsame(h,h')$, and apply the same theorem to move
the displayed representation of $k$ along $\hsame(k,k')$. The component
classifier witnesses are unchanged, so the transported pair representations
and the original component witnesses rebuild
$\mathsf{ProdComponentHistoryClassifier}_{L,R,\sim_L,\sim_R}(h',k')$.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdComponentHistoryClassifier\_hsame\_transport}

\begin{theorem}[Product pair representation hsame coherence]
\label{thm:prod-pair-rep-hsame-coherence}
For source-history carriers $L,R:\mathsf{BHist}\to\mathsf{Prop}$, source
classifiers $\sim_L,\sim_R:\mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$,
and histories $h,k,l,r,l',r'$, if
$\mathsf{ProdPairRepCoherent}_{L,R,\sim_L,\sim_R}$,
$\mathsf{ProdPairRep}_{L,R}(h,l,r)$,
$\mathsf{ProdPairRep}_{L,R}(k,l',r')$, and $\hsame(h,k)$ hold, then
$(l\sim_L l')\wedge(r\sim_R r')$ holds.

The per-object fit is: traditional target ordered-pair component comparison;
BEDC source $\mathsf{BHist}$ with $\Cont$ and $\hsame$; carrier
$\mathsf{ProdPairRep}$; classifier pair $\sim_L,\sim_R$; stability obligation
endpoint-transported coherence; ledger displayed pair representations at
internally same product endpoints; core theorem this hsame-coherence bridge;
exactness component witnesses compared after endpoint transport; standard bridge
ordered-pair readback over stable product endpoints; status
$\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Transport the second displayed pair representation from $k$ to $h$. Concretely,
symmetry of $\hsame(h,k)$ gives $\hsame(k,h)$, and
Theorem~\ref{thm:prod-pair-rep-hsame-transport} sends
$\mathsf{ProdPairRep}_{L,R}(k,l',r')$ to
$\mathsf{ProdPairRep}_{L,R}(h,l',r')$ without changing the displayed components.
Now apply $\mathsf{ProdPairRepCoherent}_{L,R,\sim_L,\sim_R}$ at the single
endpoint $h$ to the two displayed representations
$\mathsf{ProdPairRep}_{L,R}(h,l,r)$ and
$\mathsf{ProdPairRep}_{L,R}(h,l',r')$. The coherence datum returns exactly
$(l\sim_L l')\wedge(r\sim_R r')$.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdPairRep\_hsame\_coherence}

\begin{theorem}[Product componentwise history classifier endpoint carriers]
\label{thm:prod-component-history-classifier-endpoint-carriers}
Let $L,R:\mathsf{BHist}\to\mathsf{Prop}$ be source-history carriers and let
$\sim_L,\sim_R:\mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$ be source
classifiers. If
$\mathsf{ProdComponentHistoryClassifier}_{L,R,\sim_L,\sim_R}(h,k)$ holds, then
both $\mathsf{ProdHistoryCarrier}(L,R,h)$ and
$\mathsf{ProdHistoryCarrier}(L,R,k)$ hold. The per-object template role is:
traditional target ordered-pair classifier endpoint coverage; BEDC source
$\mathsf{BHist}$ with $\Cont$; carrier $\mathsf{ProdHistoryCarrier}$;
classifier $\mathsf{ProdComponentHistoryClassifier}$; ledger displayed pair
representation; core theorem endpoint carrier exposure; exactness classifier
endpoints are generated product endpoints; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Unpack the componentwise classifier witness to obtain component histories
$l_h,r_h,l_k,r_k$ and displayed pair representations
$\mathsf{ProdPairRep}_{L,R}(h,l_h,r_h)$ and
$\mathsf{ProdPairRep}_{L,R}(k,l_k,r_k)$. Apply pair-representation coverage to
the first displayed pair representation to rebuild
$\mathsf{ProdHistoryCarrier}(L,R,h)$ and to the second displayed pair
representation to rebuild $\mathsf{ProdHistoryCarrier}(L,R,k)$.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdComponentHistoryClassifier\_endpoint\_carriers}

\begin{theorem}[Product history classifier hsame component cancellation]
\label{thm:prod-history-classifier-hsame-component-cancellation}
For source predicates $L,R:\mathsf{BHist}\to\mathsf{Prop}$ and histories
$h,k,l,l',r,r':\mathsf{BHist}$, suppose
$$
\mathsf{ProdHistoryClassifier}(L,R,h,k),
\qquad
\Cont(l,r,h),
\qquad
\Cont(l',r',k).
$$
Then the displayed component comparisons determine each other:
$$
(\hsame(l,l')\to\hsame(r,r'))
\land
(\hsame(r,r')\to\hsame(l,l')).
$$

The ten-field fit is: traditional target ordered-pair component cancellation;
BEDC source $\mathsf{BHist}$ with $\Cont$ and $\hsame$; carrier
$\mathsf{ProdHistoryCarrier}$; classifier $\mathsf{ProdHistoryClassifier}$;
stability from endpoint $\hsame$ plus continuation cancellation; ledger
displayed pair continuations; core theorem this component-cancellation block;
exactness component readback under one known component comparison; standard
bridge ordered-pair cancellation; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Unfold the product-history classifier only to expose the stored endpoint
comparison $\hsame(h,k)$. The two displayed continuation witnesses already show
that $h$ is presented by $(l,r)$ and $k$ is presented by $(l',r')$.

For the first implication, assume $\hsame(l,l')$. Use
\autoref{thm:cont-cancel-hsame-left-context} on the two displayed
continuations, with the left contexts aligned by $\hsame(l,l')$ and the results
aligned by the classifier endpoint comparison $\hsame(h,k)$. The left-context
cancellation clause yields $\hsame(r,r')$.

For the second implication, assume $\hsame(r,r')$. Transport the right suffix
of one displayed continuation along this comparison so the two continuations
have a common suffix. Then apply the common-suffix cancellation clause recorded
in \autoref{thm:continuation-right-cancellation}, using the same endpoint
comparison $\hsame(h,k)$, to obtain $\hsame(l,l')$.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdHistoryClassifier\_component\_hsame\_cancellation}

\begin{theorem}[Product history classifier empty-left component inversion]
\label{thm:prod-history-classifier-empty-left-component-inversion}
For source predicates $L,R:\mathsf{BHist}\to\mathsf{Prop}$ and a history
$k:\mathsf{BHist}$, if
$\mathsf{ProdHistoryClassifier}(L,R,\mathsf{Empty},k)$ holds, then there are
component histories $l,r,l',r'$ such that $L(l)$, $R(r)$,
$l\hsame\mathsf{Empty}$, $r\hsame\mathsf{Empty}$, $L(l')$, $R(r')$,
$\Cont(l',r',k)$, and $\mathsf{Empty}\hsame k$ all hold.
\end{theorem}
\begin{proof}
Invert the product-history classifier into displayed product carrier witnesses
for the empty left endpoint and for $k$. The empty endpoint carrier supplies a
continuation $\Cont(l,r,\mathsf{Empty})$, and empty-result inversion for BEDC
continuations forces both displayed components to be $\mathsf{Empty}$. Keep the
right endpoint continuation witness and the stored classifier endpoint equality
to obtain the stated exposure.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdHistoryClassifier\_empty\_left\_component\_inversion}
