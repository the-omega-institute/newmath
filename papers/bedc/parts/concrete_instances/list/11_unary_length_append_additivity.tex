\begin{definition}[Unary NatAdd relation]
\label{def:unary-natadd-relation}
For histories $u,v,w:\mathsf{BHist}$, write
$$
\mathsf{NatAdd}(u,v,w)
\quad\Longleftrightarrow\quad
\UnaryCont(u,v,w).
$$
This is the unary-continuation additive relation of \autoref{def:unary-continuation-result} and \autoref{def:additive-pattern-specification}.
\end{definition}

\begin{definition}[List unary length token]
\label{def:list-unary-length-token}
For finite BEDC-history spines, define the unary length token by recursion:
$$
\begin{array}{rcl}
\mathsf{ListUnaryLength}(\mathsf{nil}) &=& \emp,\\
\mathsf{ListUnaryLength}(\mathsf{cons}(a,xs)) &=& \Eone(\mathsf{ListUnaryLength}(xs)).
\end{array}
$$
This is the unary-history counterpart of the finite-spine value $\ell(xs)$ used in \autoref{def:list-public-length-readback}; it records the same constructor count as a $\NatUp$ token.
\end{definition}
\leandef{BEDC.Derived.ListUp.ListUnaryLength}

\begin{definition}[List public unary length token]
\label{def:list-public-unary-length-token}
For a BEDC history source carrier $A:\mathsf{BHist}\to\mathsf{Prop}$, define
the public unary length token relation by
$$
\mathsf{ListPublicUnaryLength}_A(h,u)
\quad\Longleftrightarrow\quad
\exists xs:\mathsf{HistSpine}.\;
\mathsf{ListSpineRep}_A(h,xs)\land
\hsame(\mathsf{ListUnaryLength}(xs),u).
$$
Thus the certificate-level endpoint exposes the same unary $\NatUp$ token that
the finite-spine presentation computes internally.
\end{definition}

\begin{theorem}[List classifier preserves unary length token]
\label{thm:list-classifier-unary-length-token-hsame}
Let $\sim_A:\mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$ be a source classifier and let $xs,ys:\mathsf{HistSpine}$ be finite BEDC-history spines. If
$$
\mathsf{ListClassifierSpec}(\sim_A,xs,ys)
$$
holds, then
$$
\hsame(\mathsf{ListUnaryLength}(xs),\mathsf{ListUnaryLength}(ys)).
$$
\end{theorem}
\leanchecked{BEDC.Derived.ListUp.ListClassifierSpec\_unary\_length\_token\_hsame}
\begin{proof}
Induct on the displayed finite-spine classifier witness.

In the nil-nil case, both spines are $\mathsf{nil}$, so both unary length tokens reduce to $\emp$. Reflexivity of $\hsame$ closes the empty-token comparison.

In the cons-cons case, write the spines as $\mathsf{cons}(a,xs_0)$ and $\mathsf{cons}(b,ys_0)$. The classifier witness contains the source head comparison $a\sim_A b$ and the tail classifier $\mathsf{ListClassifierSpec}(\sim_A,xs_0,ys_0)$; the induction hypothesis gives
$$
\hsame(\mathsf{ListUnaryLength}(xs_0),\mathsf{ListUnaryLength}(ys_0)).
$$
The recursive clause of \autoref{def:list-unary-length-token} turns the two outer length tokens into one-extensions of those tail tokens. Applying the $\Eone$-congruence direction of \autoref{cor:eone-constructor-inversion} gives the required $\hsame$ comparison for the two cons spines.
\end{proof}

\begin{theorem}[List public unary length classifier transport]
\label{thm:list-public-unary-length-classifier-transport}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier, let
$\sim_A$ be a source classifier, and let
$\mathsf{cert}:\NameCert(A,\sim_A)$. Assume
$\mathsf{ListSpineRepCoherent}_{A,\sim_A}$. Then for all histories $h,k$ and
all unary length tokens $u:\mathsf{BHist}$, a classifier witness
$\mathsf{ListSpineHistoryClassifier}_{A,\sim_A}(h,k)$ transports public unary
length readback in both directions:
\begin{enumerate}
  \item if $\mathsf{ListPublicUnaryLength}_A(h,u)$ holds, then
  $\mathsf{ListPublicUnaryLength}_A(k,u)$ holds;
  \item if $\mathsf{ListPublicUnaryLength}_A(k,u)$ holds, then
  $\mathsf{ListPublicUnaryLength}_A(h,u)$ holds.
\end{enumerate}
The ten-field fit is: traditional target finite-list length as a $\NatUp$ unary
token; BEDC source $\mathsf{BHist}$ with $\NameCert$; carrier
$\mathsf{ListHistoryCarrier}_A$; classifier
$\mathsf{ListSpineHistoryClassifier}_{A,\sim_A}$; stability from represented
spine coherence and source-certificate transitivity; ledger finite spine plus
unary token; core theorem public unary length classifier transport; exactness by
public readback $\mathsf{ListPublicUnaryLength}_A$; standard bridge ordinary
finite-list length; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Fix $h,k,u$ and a bridge classifier
$C:\mathsf{ListSpineHistoryClassifier}_{A,\sim_A}(h,k)$. Unpack $C$ into finite
spines $p$ and $q$, endpoint presentations
$\mathsf{ListSpineRep}_A(h,p)$ and $\mathsf{ListSpineRep}_A(k,q)$, and a
finite-spine classifier $\mathsf{ListClassifierSpec}(\sim_A,p,q)$.

For transport from $h$ to $k$, unpack
$\mathsf{ListPublicUnaryLength}_A(h,u)$ as a finite spine $xs$, a representation
$\mathsf{ListSpineRep}_A(h,xs)$, and
$\hsame(\mathsf{ListUnaryLength}(xs),u)$. Coherence at the shared endpoint $h$
compares the two presentations of $h$, giving
$\mathsf{ListClassifierSpec}(\sim_A,xs,p)$.

Compose that finite-spine classifier with the bridge classifier
$\mathsf{ListClassifierSpec}(\sim_A,p,q)$, using the transitivity clause of
$\mathsf{cert}$ at each source head. Apply
Theorem~\ref{thm:list-classifier-unary-length-token-hsame} to obtain
$$
\hsame(\mathsf{ListUnaryLength}(xs),\mathsf{ListUnaryLength}(q)).
$$
Symmetry and transitivity of $\hsame$, together with the original comparison
with $u$, give $\hsame(\mathsf{ListUnaryLength}(q),u)$. Repack $q$ and its
representation at $k$ as $\mathsf{ListPublicUnaryLength}_A(k,u)$.

For transport from $k$ to $h$, unpack
$\mathsf{ListPublicUnaryLength}_A(k,u)$ as a finite spine $ys$, a representation
$\mathsf{ListSpineRep}_A(k,ys)$, and
$\hsame(\mathsf{ListUnaryLength}(ys),u)$. Coherence at the shared endpoint $k$
compares the bridge spine $q$ with $ys$, giving
$\mathsf{ListClassifierSpec}(\sim_A,q,ys)$.

Compose the bridge classifier
$\mathsf{ListClassifierSpec}(\sim_A,p,q)$ with this endpoint-coherence
classifier, again using source-certificate transitivity headwise. Theorem~\ref{thm:list-classifier-unary-length-token-hsame}
gives
$$
\hsame(\mathsf{ListUnaryLength}(p),\mathsf{ListUnaryLength}(ys)).
$$
Transitivity with the original comparison to $u$ gives
$\hsame(\mathsf{ListUnaryLength}(p),u)$. Repack $p$ and its representation at
$h$ as $\mathsf{ListPublicUnaryLength}_A(h,u)$.
\end{proof}

\begin{theorem}[List public append unary-length classifier congruence]
\label{thm:list-public-append-unary-length-classifier-congruence}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier, let
$\sim_A$ be a source classifier, let $\mathsf{cert}:\NameCert(A,\sim_A)$, and
assume $\mathsf{ListSpineRepCoherent}_{A,\sim_A}$. For histories
$h,h',k,k',r,r',u,v,w:\mathsf{BHist}$, assume
$$
\mathsf{ListSpineHistoryClassifier}_{A,\sim_A}(h,h'),\qquad
\mathsf{ListSpineHistoryClassifier}_{A,\sim_A}(k,k'),
$$
$$
\mathsf{ListPublicAppend}_A(h,k,r),\qquad
\mathsf{ListPublicAppend}_A(h',k',r'),
$$
and
$$
\mathsf{ListPublicUnaryLength}_A(h,u),\qquad
\mathsf{ListPublicUnaryLength}_A(k,v),\qquad
\mathsf{ListPublicUnaryLength}_A(r,w)
$$
hold. Then
$$
\mathsf{ListPublicUnaryLength}_A(h',u)\land
\mathsf{ListPublicUnaryLength}_A(k',v)\land
\mathsf{ListPublicUnaryLength}_A(r',w).
$$
The ten-field fit is: traditional target finite-list append stability for unary
length tokens; BEDC source $\mathsf{BHist}$ through $\mathsf{ListSpineRep}$;
carrier $\mathsf{ListHistoryCarrier}_A$; classifier
$\mathsf{ListSpineHistoryClassifier}_{A,\sim_A}$; stability from append
congruence and unary-length classifier transport; ledger public unary length
tokens; core theorem this composite; exactness via public append and public
unary length readback; standard bridge ordinary finite-list append length
invariance; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:list-public-unary-length-classifier-transport} to the
classifier witness from $h$ to $h'$ and the public unary length token at $h$.
Its forward transport clause gives $\mathsf{ListPublicUnaryLength}_A(h',u)$.

Apply the same theorem to the classifier witness from $k$ to $k'$ and the
public unary length token at $k$. The forward transport clause gives
$\mathsf{ListPublicUnaryLength}_A(k',v)$.

Next apply the second clause of
Theorem~\ref{thm:list-public-append-readback-total-congruent} to the two
endpoint classifier witnesses and the two public append readbacks. This gives
$$
\mathsf{ListSpineHistoryClassifier}_{A,\sim_A}(r,r').
$$

Apply Theorem~\ref{thm:list-public-unary-length-classifier-transport} once more
to this output classifier and the public unary length token at $r$. Its forward
transport clause gives $\mathsf{ListPublicUnaryLength}_A(r',w)$, and the three
transported witnesses form the displayed conjunction.
\end{proof}

\begin{theorem}[Public list unary length additivity under append]
\label{thm:list-unary-length-append-additivity}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier. For histories $h,k,r:\mathsf{BHist}$ and finite spines $xs,ys:\mathsf{HistSpine}$, assume
$$
\mathsf{ListSpineRep}_A(h,xs),\qquad
\mathsf{ListSpineRep}_A(k,ys),\qquad
\mathsf{ListSpineRep}_A(r,xs{++}ys).
$$
Set
$$
L_x=\mathsf{ListUnaryLength}(xs),\qquad
L_y=\mathsf{ListUnaryLength}(ys),\qquad
L_{xy}=\mathsf{ListUnaryLength}(xs{++}ys).
$$
Then
$$
\mathsf{NatAdd}(L_y,L_x,L_{xy})
$$
and for every $w:\mathsf{BHist}$,
$$
\mathsf{NatAdd}(L_x,L_y,w)\Longrightarrow \hsame(L_{xy},w).
$$
\end{theorem}
\begin{proof}
By \autoref{def:list-spine-representation}, the three representation witnesses say that the public endpoints use the displayed finite spines and that the append endpoint uses the spine append $xs{++}ys$. The proof of the additive length claim depends on the spines alone.

A preliminary induction on a finite spine $zs$ gives $\Unary(\mathsf{ListUnaryLength}(zs))$. The nil case is the base unary generator from \autoref{def:unary-history-judgment}; the cons case applies \autoref{lem:unary-closure-under-one-extension} to the induction hypothesis.

Now induct on $xs$ to prove $\mathsf{NatAdd}(L_y,L_x,L_{xy})$. In the nil case, $L_x=\emp$ and $L_{xy}=L_y$. The preliminary unary induction gives $\Unary(L_y)$, and the right-unit clause of \autoref{thm:continuation-unit-laws-spine} gives $\ContR(L_y,\emp,L_y)$. Unfolding \autoref{def:unary-natadd-relation} gives $\mathsf{NatAdd}(L_y,\emp,L_y)$.

In the cons case, write $xs=\mathsf{cons}(a,xs_0)$ and let $L_0=\mathsf{ListUnaryLength}(xs_0)$ and $T=\mathsf{ListUnaryLength}(xs_0{++}ys)$. The induction hypothesis gives $\mathsf{NatAdd}(L_y,L_0,T)$, hence $\ContR(L_y,L_0,T)$ after unfolding \autoref{def:unary-natadd-relation}. The one-step rule in \autoref{def:continuation-relation} gives $\ContR(L_y,\Eone(L_0),\Eone(T))$. The preliminary unary induction and \autoref{lem:unary-closure-under-one-extension} supply the two unary input witnesses, so this is exactly $\mathsf{NatAdd}(L_y,L_x,L_{xy})$ by the recursive clauses of \autoref{def:list-unary-length-token}.

Finally assume $\mathsf{NatAdd}(L_x,L_y,w)$. Together with the reversed append-aligned witness just proved, \autoref{thm:unary-cont-comm} compares the two unary continuation results and gives $\hsame(w,L_{xy})$. Symmetry of $\hsame$ gives $\hsame(L_{xy},w)$.
\end{proof}
\leanchecked{BEDC.Derived.ListUp.ListUnaryLength\_append\_cont}

\begin{theorem}[Public list unary length is associative under append]
\label{thm:list-unary-length-append-associativity}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier. For histories
$h_x,h_y,h_z,h_{xy},h_{(xy)z},h_{yz},h_{x(yz)}:\mathsf{BHist}$ and finite spines
$xs,ys,zs:\mathsf{HistSpine}$, assume
$$
\mathsf{ListSpineRep}_A(h_x,xs),\quad
\mathsf{ListSpineRep}_A(h_y,ys),\quad
\mathsf{ListSpineRep}_A(h_z,zs),
$$
$$
\mathsf{ListSpineRep}_A(h_{xy},xs{++}ys),\quad
\mathsf{ListSpineRep}_A(h_{(xy)z},(xs{++}ys){++}zs),
$$
and
$$
\mathsf{ListSpineRep}_A(h_{yz},ys{++}zs),\quad
\mathsf{ListSpineRep}_A(h_{x(yz)},xs{++}(ys{++}zs)).
$$
Set
$$
L_x=\mathsf{ListUnaryLength}(xs),\quad
L_y=\mathsf{ListUnaryLength}(ys),\quad
L_z=\mathsf{ListUnaryLength}(zs),
$$
$$
L_{xy}=\mathsf{ListUnaryLength}(xs{++}ys),\quad
L_{yz}=\mathsf{ListUnaryLength}(ys{++}zs),
$$
and
$$
L_{(xy)z}=\mathsf{ListUnaryLength}((xs{++}ys){++}zs),\quad
L_{x(yz)}=\mathsf{ListUnaryLength}(xs{++}(ys{++}zs)).
$$
Then
$$
\hsame(L_{(xy)z},L_{x(yz)}).
$$
\end{theorem}
\begin{proof}
Apply \autoref{thm:list-unary-length-append-additivity} to the witnesses for $xs$, $ys$, and $xs{++}ys$, obtaining
$\mathsf{NatAdd}(L_y,L_x,L_{xy})$. Apply the same theorem to the witnesses for $xs{++}ys$, $zs$, and $(xs{++}ys){++}zs$, obtaining
$\mathsf{NatAdd}(L_z,L_{xy},L_{(xy)z})$.

Apply \autoref{thm:list-unary-length-append-additivity} to the witnesses for $ys$, $zs$, and $ys{++}zs$, obtaining
$\mathsf{NatAdd}(L_z,L_y,L_{yz})$. Apply it once more to the witnesses for $xs$, $ys{++}zs$, and $xs{++}(ys{++}zs)$, obtaining
$\mathsf{NatAdd}(L_{yz},L_x,L_{x(yz)})$.

Unfolding \autoref{def:unary-natadd-relation}, these four witnesses are unary continuations for the two parenthesized length composites. The associativity theorem for unary continuation, \autoref{lem:unary-continuation-associativity}, compares the continuation first through $L_{xy}$ with the continuation first through $L_{yz}$.

The displayed additivity witnesses identify the left parenthesized endpoint with $L_{(xy)z}$ and the right parenthesized endpoint with $L_{x(yz)}$. Symmetry and transitivity of $\hsame$ compose those endpoint comparisons with the associativity comparison, yielding $\hsame(L_{(xy)z},L_{x(yz)})$.
\end{proof}
