\begin{theorem}[Framed list bridge empty-left source refinement]
\label{thm:framed-list-bridge-empty-left-source-refinement}
Let $A,B:\mathsf{BHist}\to\mathsf{Prop}$ be BEDC history source carriers and
let $\sim_A,\sim_B$ be source classifiers. Assume source refinement preserves
carrier membership and head classification:
$$
\forall x:\mathsf{BHist}.\; A(x)\to B(x),
\qquad
\forall x\,y:\mathsf{BHist}.\; x\sim_A y\to x\sim_B y.
$$
Then every framed bridge from the displayed empty-left endpoint
$\Ezero(\emp)$ to $k$ over $(A,\sim_A)$ refines to a framed bridge over
$(B,\sim_B)$:
$$
\mathsf{FramedListBridgeClassifier}_{A,\sim_A}(\Ezero(\emp),k)
\to
\mathsf{FramedListBridgeClassifier}_{B,\sim_B}(\Ezero(\emp),k).
$$
\end{theorem}
\begin{proof}
Unpack the framed bridge into finite spines $xs,ys$, two
$\mathsf{FramedListSpineRep}$ witnesses, and a
$\mathsf{ListClassifierSpec}(\sim_A,xs,ys)$ witness. The carrier inclusion
transports every entry field in both spine representations while preserving the
endpoint equalities. Induction on the finite-spine classifier transports each
head relation through the classifier inclusion and rebuilds the tail witness.
Repack the same two spines with these transported witnesses.
\end{proof}
\leanchecked{BEDC.Derived.ListUp.FramedListBridgeClassifier\_empty\_left\_source\_refinement}

\begin{theorem}[Framed list bridge classifier constructor exactness]
\label{thm:framed-list-bridge-classifier-constructor-inversion}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier, let
$\sim_A$ be a source classifier, let $\mathsf{cert}:\NameCert(A,\sim_A)$, and
assume $\mathsf{ListSourceHsameCompatible}(A,\sim_A)$. For histories
$h,k:\mathsf{BHist}$, finite spines $xs,ys$, and source histories
$a,b:\mathsf{BHist}$, the framed bridge classifier has the following constructor
exactness behavior:
The roadmap certificate reads: traditional target finite-list framed bridge
constructor exactness; BEDC source $\mathsf{BHist}$ with $\mathsf{PairFrame}$
and $\mathsf{FramedListEndpoint}$; carrier
$\mathsf{FramedListHistoryCarrier}_A$; classifier
$\mathsf{FramedListBridgeClassifier}$; stability from the source $\NameCert$
plus $\mathsf{ListSourceHsameCompatible}$; ledger displayed framed spines; core
theorem nil-nil introduction, mixed nil/cons separation, and cons-cons iff;
exactness finite-spine constructor readback; standard bridge finite spines;
status $\mathsf{CheckedCert}$.
\begin{enumerate}
  \item if $\mathsf{FramedListSpineRep}_A(h,\mathsf{nil})$ and
  $\mathsf{FramedListSpineRep}_A(k,\mathsf{nil})$ hold, then
  $\mathsf{FramedListBridgeClassifier}_{A,\sim_A}(h,k)$ holds;
  \item if $\mathsf{FramedListSpineRep}_A(h,\mathsf{nil})$,
  $\mathsf{FramedListSpineRep}_A(k,\mathsf{cons}(b,ys))$, and
  $\mathsf{FramedListBridgeClassifier}_{A,\sim_A}(h,k)$ hold, then $\bot$;
  \item if $\mathsf{FramedListSpineRep}_A(h,\mathsf{cons}(a,xs))$,
  $\mathsf{FramedListSpineRep}_A(k,\mathsf{nil})$, and
  $\mathsf{FramedListBridgeClassifier}_{A,\sim_A}(h,k)$ hold, then $\bot$;
  \item if $\mathsf{FramedListSpineRep}_A(h,\mathsf{cons}(a,xs))$ and
  $\mathsf{FramedListSpineRep}_A(k,\mathsf{cons}(b,ys))$ hold, then

  $$
  \mathsf{FramedListBridgeClassifier}_{A,\sim_A}(h,k)
  \Longleftrightarrow
  (a\sim_A b)\land \mathsf{ListClassifierSpec}(\sim_A,xs,ys).
  $$

\end{enumerate}
\end{theorem}
\begin{proof}
In the nil-nil case, the nil branch of
Theorem~\ref{thm:list-classifier-case-exactness} gives

$$
\mathsf{ListClassifierSpec}(\sim_A,\mathsf{nil},\mathsf{nil}).
$$

The reverse direction of
Theorem~\ref{thm:framed-list-bridge-classifier-displayed-spine-exactness}
repackages this finite-spine classifier with the two displayed nil endpoint
representatives.

In the nil-cons case, apply
Theorem~\ref{thm:framed-list-bridge-classifier-displayed-spine-exactness} to the
two displayed endpoint representatives and the bridge witness. This yields

$$
\mathsf{ListClassifierSpec}(\sim_A,\mathsf{nil},\mathsf{cons}(b,ys)).
$$

The mixed nil-cons clause of Theorem~\ref{thm:list-classifier-case-exactness}
eliminates that finite-spine classifier.

The cons-nil case is symmetric. Displayed-spine exactness gives

$$
\mathsf{ListClassifierSpec}(\sim_A,\mathsf{cons}(a,xs),\mathsf{nil}),
$$

and the mixed cons-nil clause of
Theorem~\ref{thm:list-classifier-case-exactness} eliminates this shape.

In the cons-cons case, displayed-spine exactness turns the bridge witness into

$$
\mathsf{ListClassifierSpec}(\sim_A,\mathsf{cons}(a,xs),\mathsf{cons}(b,ys)).
$$

The cons-cons clause of Theorem~\ref{thm:list-classifier-case-exactness}
projects the source-head classifier $a\sim_A b$ and the tail classifier
$\mathsf{ListClassifierSpec}(\sim_A,xs,ys)$.

Conversely, suppose the source-head classifier $a\sim_A b$ and the tail
classifier $\mathsf{ListClassifierSpec}(\sim_A,xs,ys)$ are given. The cons-cons
clause of Theorem~\ref{thm:list-classifier-case-exactness} rebuilds

$$
\mathsf{ListClassifierSpec}(\sim_A,\mathsf{cons}(a,xs),\mathsf{cons}(b,ys)).
$$

The reverse direction of
Theorem~\ref{thm:framed-list-bridge-classifier-displayed-spine-exactness}
repackages this finite-spine classifier as
$\mathsf{FramedListBridgeClassifier}_{A,\sim_A}(h,k)$.
\end{proof}
\leanchecked{BEDC.Derived.ListUp.FramedListBridgeClassifier\_constructor\_exactness}

\begin{theorem}[Framed list bridge classifier equivalence fields]
\label{thm:framed-list-bridge-classifier-equivalence-fields}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier, let
$\sim_A$ be a source classifier, let $\mathsf{cert}:\NameCert(A,\sim_A)$, and
assume $\mathsf{ListSourceHsameCompatible}(A,\sim_A)$. Define the framed carrier
by
$$
\mathsf{FramedListHistoryCarrier}_A(h)
\quad\Longleftrightarrow\quad
\exists xs:\mathsf{HistSpine}.\;\mathsf{FramedListSpineRep}_A(h,xs).
$$
\leandef{BEDC.Derived.ListUp.FramedListHistoryCarrier}
Then $\mathsf{FramedListBridgeClassifier}_{A,\sim_A}$ satisfies the five
classifier fields below. Carrier-relative reflexivity:
$$
\mathsf{FramedListHistoryCarrier}_A(h)\to
\mathsf{FramedListBridgeClassifier}_{A,\sim_A}(h,h).
$$
Endpoint exposure for any bridge witness:
$$
\mathsf{FramedListBridgeClassifier}_{A,\sim_A}(h,k)\to
\mathsf{FramedListHistoryCarrier}_A(h)\land
\mathsf{FramedListHistoryCarrier}_A(k).
$$
Symmetry:
$$
\mathsf{FramedListBridgeClassifier}_{A,\sim_A}(h,k)\to
\mathsf{FramedListBridgeClassifier}_{A,\sim_A}(k,h).
$$
Transitivity:
$$
\mathsf{FramedListBridgeClassifier}_{A,\sim_A}(h,k)\to
\mathsf{FramedListBridgeClassifier}_{A,\sim_A}(k,u)\to
\mathsf{FramedListBridgeClassifier}_{A,\sim_A}(h,u).
$$
Carrier transport:
$$
\mathsf{FramedListHistoryCarrier}_A(h)\to
\mathsf{FramedListBridgeClassifier}_{A,\sim_A}(h,k)\to
\mathsf{FramedListHistoryCarrier}_A(k).
$$
The ten-field fit is: traditional target finite-list framed bridge equivalence
fields; BEDC source $\mathsf{BHist}$; carrier framed spine representation;
classifier $\mathsf{FramedListBridgeClassifier}_{A,\sim_A}$; stability
obligations the five displayed clauses; ledger finite displayed spines; core
theorem this fields theorem; exactness by displayed-spine exactness and framed
coherence; standard bridge finite spines; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Endpoint exposure unpacks the bridge witness into finite spines $xs,ys$ with
$\mathsf{FramedListSpineRep}_A(h,xs)$,
$\mathsf{FramedListSpineRep}_A(k,ys)$, and
$\mathsf{ListClassifierSpec}(\sim_A,xs,ys)$; the two representation witnesses
are exactly the two framed-carrier witnesses.

For carrier-relative reflexivity, choose the displayed spine supplied by the
framed-carrier witness at $h$ on both endpoints. Induct over that finite spine.
The nil case is the nil constructor of $\mathsf{ListClassifierSpec}$. In the
cons case, the entry predicate stored in $\mathsf{FramedListSpineRep}$ supplies
$A$ for the displayed head, so $\mathsf{cert.equiv\_refl}$ gives the head
classifier witness and the induction hypothesis supplies the tail witness.

For symmetry, unpack the bridge witness, apply
Theorem~\ref{thm:list-classifier-symm-from-namecert} to the finite-spine
classifier witness, and rebuild the bridge with the two displayed endpoints
swapped. For transitivity, unpack bridge witnesses for $h,k$ and $k,u$. The two
middle spines both represent $k$, so Theorem~\ref{thm:framed-list-spine-coherence}
compares them. Compose the first bridge classifier, the middle comparison, and
the second bridge classifier by two applications of
Theorem~\ref{thm:list-classifier-trans-from-namecert}, then rebuild the bridge
from the exposed first and last spines. Carrier transport is the right endpoint
projection of endpoint exposure, with the displayed left-carrier premise matching
the $\NameCert$ field shape.
\end{proof}
\leanchecked{BEDC.Derived.ListUp.FramedListBridgeClassifier\_equivalence\_fields}

\begin{corollary}[Framed finite-spine paper certificate]
\label{cor:framed-list-carrier-certificate}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier, let
$\sim_A$ be a source classifier, and let
$\mathsf{cert}:\NameCert(A,\sim_A)$. Assume
$\mathsf{ListSourceHsameCompatible}(A,\sim_A)$. Define the framed carrier by
$$
\mathsf{FramedListHistoryCarrier}_A(h)
\quad\Longleftrightarrow\quad
\exists xs:\mathsf{HistSpine}.\;\mathsf{FramedListSpineRep}_A(h,xs),
$$
and define the framed bridge classifier by
$$
\mathsf{FramedListBridgeClassifier}_{A,\sim_A}(h,k)
\quad\Longleftrightarrow\quad
\exists xs\,ys.\;
\mathsf{FramedListSpineRep}_A(h,xs)\land
\mathsf{FramedListSpineRep}_A(k,ys)\land
\mathsf{ListClassifierSpec}(\sim_A,xs,ys).
$$
\leandef{BEDC.Derived.ListUp.FramedListBridgeClassifier}
With pattern and ledger predicates both given by
$\mathsf{FramedListHistoryCarrier}_A$, these fields form the paper certificate
target
$$
\mathsf{SemanticNameCert}(\mathsf{FramedListHistoryCarrier}_A,
\mathsf{FramedListHistoryCarrier}_A,
\mathsf{FramedListHistoryCarrier}_A,
\mathsf{FramedListBridgeClassifier}_{A,\sim_A}).
$$
This certificate is for the framed carrier and framed bridge classifier above.
It does not identify $\mathsf{FramedListHistoryCarrier}_A$ with the unframed
$\mathsf{ListHistoryCarrier}_A$ used by the public $\ListUp$ target.
\end{corollary}
\begin{proof}
The source certificate supplies the entrywise reflexivity, symmetry,
transitivity, and carrier transport facts for $\sim_A$. The framed endpoint
definition supplies a separated finite-spine readback, and
Theorem~\ref{thm:framed-list-spine-coherence} supplies the comparison datum for
two presentations of the same history. Reflexivity chooses the displayed spine
for a framed carrier witness and classifies it entry by entry using
$\mathsf{cert}$. Symmetry and transitivity unpack the two existential framed
bridge witnesses and apply the structural finite-spine symmetry and
transitivity theorems for $\mathsf{ListClassifierSpec}$; at the shared middle
history, framed coherence identifies the two exposed middle spines. Carrier
transport follows because a framed bridge witness contains a framed
representation of the target history. Since the pattern and ledger predicates
are the carrier itself, their soundness fields are identity implications.
\end{proof}
\leanchecked{BEDC.Derived.ListUp.FramedListBridgeClassifier\_semantic\_name\_certificate}
