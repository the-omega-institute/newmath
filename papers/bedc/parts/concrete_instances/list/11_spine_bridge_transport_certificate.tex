\begin{theorem}[List spine bridge source refinement]
\label{thm:list-spine-bridge-source-refinement}
Let $A,B:\mathsf{BHist}\to\mathsf{Prop}$ be BEDC history source carriers, and
let $\sim_A$ and $\sim_B$ be source classifiers. The ten-field fit is:
traditional target finite-list source refinement; BEDC source $\mathsf{BHist}$
with $\mathsf{ListSpineRep}$; carrier $\mathsf{ListHistoryCarrier}$; classifier
$\mathsf{ListSpineBridgeClassifier}$; stability obligation carrier and
classifier preservation under source refinement; ledger finite spine
representation; core theorem the three displayed implications below; exactness
inherited from $\mathsf{ListSpineRep}$; standard bridge finite-spine readback;
status $\mathsf{ProofTarget}$. Assume the source carrier and source classifier
of $A$ are included in those of $B$:
$$
\forall h:\mathsf{BHist}.\; A(h)\to B(h)
$$
and
$$
\forall x\,y:\mathsf{BHist}.\; x\sim_A y\to x\sim_B y.
$$
Then the displayed list-spine data transport from $A$ to $B$:
$$
\mathsf{ListSpineRep}_A(h,xs)
\to
\mathsf{ListSpineRep}_B(h,xs),
$$
$$
\mathsf{ListSpineBridgeClassifier}_{A,\sim_A}(h,k)
\to
\mathsf{ListSpineBridgeClassifier}_{B,\sim_B}(h,k),
$$
and
$$
\mathsf{ListSpineHistoryClassifier}_{A,\sim_A}(h,k)
\to
\mathsf{ListSpineHistoryClassifier}_{B,\sim_B}(h,k).
$$
\end{theorem}
\begin{proof}
Proceed by induction on the displayed $\mathsf{ListSpineRep}_A(h,xs)$ witness.
The nil representation keeps the same endpoint equality. In the cons case, the
carrier inclusion turns the recorded source datum $A(a)$ into $B(a)$, while the
induction hypothesis transports the tail representation and the continuation
and endpoint witnesses are reused unchanged. For the finite-spine classifier
component, induct on the displayed $\mathsf{ListClassifierSpec}(\sim_A,xs,ys)$
witness and apply the classifier inclusion at each cons head. A bridge witness is
then repacked with the transported left and right spines and the transported
finite-spine classifier. The certificate classifier clause follows by unfolding
that it is the bridge classifier and rebuilding the same endpoints.
\end{proof}

\begin{lemma}[List spine representation endpoint transport]
\label{lem:list-spine-representation-endpoint-transport}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier. For all
histories $h$ and $h'$ and every finite list spine $xs$,
$\mathsf{ListSpineRep}_A(h,xs)$ and $\hsame(h,h')$ imply
$\mathsf{ListSpineRep}_A(h',xs)$.
\end{lemma}
\begin{proof}
Proceed by induction on the displayed finite spine. The empty spine is transported
by composing the endpoint empty witness with $\hsame(h,h')$. For a cons spine,
keep the recorded head payload and tail representation, and compose the outer
history endpoint witness with $\hsame(h,h')$; the payload and tail witnesses are
unchanged.
\end{proof}

\begin{theorem}[List spine bridge endpoint transport]
\label{thm:list-spine-bridge-endpoint-transport}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier and let
$\sim_A$ be a source classifier. The ten-field per-object fit is: traditional
target finite-list classifier stability; BEDC source $A$ over $\mathsf{BHist}$;
source classifier $\sim_A$; carrier $\mathsf{ListHistoryCarrier}_A$; classifier
$\mathsf{ListSpineBridgeClassifier}_{A,\sim_A}$; stability endpoint transport;
ledger finite-spine presentation; exactness via $\mathsf{ListSpineRep}$; standard
bridge finite spines; status $\mathsf{ProofTarget}$. For all histories $h$, $k$,
$h'$, and $k'$,
$$
\mathsf{ListSpineBridgeClassifier}_{A,\sim_A}(h,k)
\to
\hsame(h,h')
\to
\hsame(k,k')
\to
\mathsf{ListSpineBridgeClassifier}_{A,\sim_A}(h',k').
$$
\end{theorem}
\begin{proof}
Unpack the bridge witness into finite spines $xs$ and $ys$, the represented
endpoints $\mathsf{ListSpineRep}_A(h,xs)$ and
$\mathsf{ListSpineRep}_A(k,ys)$, and the finite-spine classifier witness. Apply
Lemma~\ref{lem:list-spine-representation-endpoint-transport} to transport the
two displayed endpoint representations along $\hsame(h,h')$ and $\hsame(k,k')$.
The finite-spine classifier witness is independent of the represented endpoints,
so the bridge witness is rebuilt over $h'$ and $k'$ with the same spines.
\end{proof}

\begin{theorem}[List recursive history classifier endpoint transport]
\label{thm:list-recursive-history-classifier-endpoint-transport}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier and let
$\sim_A$ be a source classifier. The ten-field fit is: traditional target
finite-list generated-classifier endpoint stability; BEDC source
$\mathsf{BHist}$; carrier $\mathsf{ListHistoryCarrier}_A$; classifier
$\mathsf{ListHistoryClassifier}^{\mathrm{rec}}_{A,\sim_A}$; stability obligation
endpoint transport; ledger finite spine; core theorem this transport law;
exactness through Theorem~\ref{thm:list-spine-representation-exactness};
standard bridge finite-spine readback; status $\mathsf{ProofTarget}$. For all
histories $h$, $k$, $h'$, and $k'$,
$$
\mathsf{ListHistoryClassifier}^{\mathrm{rec}}_{A,\sim_A}(h,k)
\to
\hsame(h,h')
\to
\hsame(k,k')
\to
\mathsf{ListHistoryClassifier}^{\mathrm{rec}}_{A,\sim_A}(h',k').
$$
\end{theorem}
\begin{proof}
Convert the displayed recursive classifier witness to
$\mathsf{ListSpineBridgeClassifier}_{A,\sim_A}(h,k)$ using the classifier
equivalence in Theorem~\ref{thm:list-spine-representation-exactness}. Apply
Theorem~\ref{thm:list-spine-bridge-endpoint-transport} to move the two
endpoints along $\hsame(h,h')$ and $\hsame(k,k')$. Convert the resulting bridge
witness back through the reverse direction of the same exactness theorem to
obtain $\mathsf{ListHistoryClassifier}^{\mathrm{rec}}_{A,\sim_A}(h',k')$.
\end{proof}

\begin{theorem}[List spine bridge nil-cons separation under coherence]
\label{thm:list-spine-bridge-nil-cons-separation-under-coherence}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier, let
$\sim_A$ be a source classifier, and assume
$\mathsf{ListSpineRepCoherent}_{A,\sim_A}$. For all histories $h$ and $k$, all
source histories $a$, and all finite spines $xs$, the bridge classifier separates
the nil and cons displayed endpoints:
$$
\mathsf{ListSpineRep}_A(h,\mathsf{nil})
\to
\mathsf{ListSpineRep}_A(k,\mathsf{cons}(a,xs))
\to
\mathsf{ListSpineBridgeClassifier}_{A,\sim_A}(h,k)
\to
\bot.
$$
It also separates the symmetric cons-vs-nil orientation:
$$
\mathsf{ListSpineRep}_A(h,\mathsf{cons}(a,xs))
\to
\mathsf{ListSpineRep}_A(k,\mathsf{nil})
\to
\mathsf{ListSpineBridgeClassifier}_{A,\sim_A}(h,k)
\to
\bot.
$$
\end{theorem}
\begin{proof}
Unpack the bridge witness into finite spines $u$ and $v$, endpoint presentations
$\mathsf{ListSpineRep}_A(h,u)$ and $\mathsf{ListSpineRep}_A(k,v)$, and a
finite-spine classifier $\mathsf{ListClassifierSpec}(\sim_A,u,v)$. In the
nil-vs-cons orientation, apply the coherence datum at endpoint $h$ to relate the
displayed $\mathsf{nil}$ spine to $u$, and at endpoint $k$ to relate $v$ to the
displayed $\mathsf{cons}(a,xs)$ spine. By
Theorem~\ref{thm:list-classifier-case-exactness}, the comparison from
$\mathsf{nil}$ to $u$ forces $u$ to have nil outer shape, and the bridge
classifier from $u$ to $v$ then forces $v$ to have nil outer shape. The
comparison from $v$ to the displayed $\mathsf{cons}(a,xs)$ spine is therefore a
mixed nil-cons classifier witness, contrary to
Lemma~\ref{lem:list-classifier-nil-cons-no-confusion}. In the cons-vs-nil
orientation, the first coherence comparison forces $u$ to have cons outer shape,
the bridge classifier forces $v$ to have cons outer shape, and the endpoint
coherence comparison against the displayed nil spine is the symmetric impossible
mixed classifier witness.
\end{proof}

\begin{theorem}[List public classifier cons endpoint inversion]
\label{thm:list-public-classifier-cons-endpoint-inversion}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier, let
$\mathsf{Rel}:\mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$ be a source
classifier, let $\mathsf{cert}:\NameCert(A,\mathsf{Rel})$, and assume
$\mathsf{ListSpineRepCoherent}_{A,\mathsf{Rel}}$. The ten-field fit is:
traditional target finite-list cons inversion; BEDC source $\mathsf{BHist}$
with $\ContR$; carrier $\mathsf{ListHistoryCarrier}_A$; classifier
$\mathsf{ListSpineHistoryClassifier}_{A,\mathsf{Rel}}$; stability from
$\mathsf{cert}$ and spine coherence; ledger finite-spine representatives for
the two cons endpoints and the two tails; core theorem public cons inversion;
exactness by
Theorem~\ref{thm:list-represented-spine-classifier-constructor-exactness};
standard bridge finite spines; status $\mathsf{ProofTarget}$. For histories
$h,k,a,b,t,u,p,q:\mathsf{BHist}$, if
$$
\begin{array}{l}
A(a)\land A(b)\land
\mathsf{ListHistoryCarrier}_A(t)\land
\mathsf{ListHistoryCarrier}_A(u)\land\\
\ContR(a,t,p)\land \ContR(b,u,q)\land
\hsame(h,\Eone(p))\land \hsame(k,\Eone(q))
\end{array}
$$
and
$$
\mathsf{ListSpineHistoryClassifier}_{A,\mathsf{Rel}}(h,k)
$$
hold, then
$$
\mathsf{Rel}(a,b)\land
\mathsf{ListSpineHistoryClassifier}_{A,\mathsf{Rel}}(t,u).
$$
\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:list-spine-representation-coverage} to the two tail
carrier witnesses, choosing finite spines $xs$ and $ys$ with
$\mathsf{ListSpineRep}_A(t,xs)$ and $\mathsf{ListSpineRep}_A(u,ys)$. The cons
constructor for $\mathsf{ListSpineRep}_A$ combines these tail representations
with $A(a)$, $A(b)$, the two continuation witnesses, and the two endpoint
witnesses to build displayed representations of $h$ by
$\mathsf{cons}(a,xs)$ and $k$ by $\mathsf{cons}(b,ys)$. Apply
Theorem~\ref{thm:list-represented-spine-classifier-constructor-exactness} to
those represented cons endpoints and the public classifier witness. Its
cons-cons clause yields $\mathsf{Rel}(a,b)$ together with
$\mathsf{ListClassifierSpec}(\mathsf{Rel},xs,ys)$. Repack the two tail
representations and this finite-spine classifier witness through the bridge
definition of $\mathsf{ListSpineHistoryClassifier}_{A,\mathsf{Rel}}$ to obtain
the tail classifier $\mathsf{ListSpineHistoryClassifier}_{A,\mathsf{Rel}}(t,u)$.
\end{proof}

\begin{theorem}[List public classifier nil endpoint inversion]
\label{thm:list-public-classifier-nil-endpoint-inversion}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier, let
$\mathsf{Rel}:\mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$ be a source
classifier, let $\mathsf{cert}:\NameCert(A,\mathsf{Rel})$, and assume
$\mathsf{ListSpineRepCoherent}_{A,\mathsf{Rel}}$. The ten-field fit is:
traditional target finite-list nil readback; BEDC source $\mathsf{BHist}$;
carrier $\mathsf{ListHistoryCarrier}_A$; classifier
$\mathsf{ListSpineHistoryClassifier}_{A,\mathsf{Rel}}$; stability from
$\mathsf{cert}$ and spine coherence; ledger nil finite-spine endpoint; core
theorem public nil inversion; exactness by
Theorem~\ref{thm:list-represented-spine-nil-classifier-inversion}; standard
bridge finite spines; status $\mathsf{ProofTarget}$. For all histories
$h,k:\mathsf{BHist}$, if
$$
\hsame(h,\emp)
$$
and
$$
\mathsf{ListSpineHistoryClassifier}_{A,\mathsf{Rel}}(h,k)
$$
hold, then
$$
\hsame(k,\emp).
$$
Symmetrically, if
$$
\hsame(k,\emp)
$$
and
$$
\mathsf{ListSpineHistoryClassifier}_{A,\mathsf{Rel}}(h,k)
$$
hold, then
$$
\hsame(h,\emp).
$$
\end{theorem}
\begin{proof}
From the displayed nil endpoint witness $\hsame(h,\emp)$, use the nil
constructor of $\mathsf{ListSpineRep}_A$ to present $h$ by
$\mathsf{nil}$. Unfold the bridge definition of
$\mathsf{ListSpineHistoryClassifier}_{A,\mathsf{Rel}}(h,k)$, obtaining finite
spines $xs$ and $ys$, endpoint representations
$\mathsf{ListSpineRep}_A(h,xs)$ and $\mathsf{ListSpineRep}_A(k,ys)$, and the
finite-spine classifier witness. Apply
Theorem~\ref{thm:list-represented-spine-nil-classifier-inversion} to the
displayed nil representation of $h$, the represented endpoint for $k$, and the
public classifier witness. The theorem forces $ys=\mathsf{nil}$; unfolding the
nil case of $\mathsf{ListSpineRep}_A(k,ys)$ recovers $\hsame(k,\emp)$. The
symmetric clause starts from $\hsame(k,\emp)$, builds the displayed nil
representation of $k$, applies the symmetric half of the same represented-spine
nil inversion to force $xs=\mathsf{nil}$, and unfolds the resulting nil
representation of $h$ to recover $\hsame(h,\emp)$.
\end{proof}

\mathsf{SemanticNameCert}(\mathsf{ListHistoryCarrier}_A,

\mathsf{ListHistoryLedger}_A,

$$

\end{proof}

\begin{theorem}[List public-object certificate exposes endpoint spine ledgers]
\label{thm:list-public-object-endpoint-spine-ledgers}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier, let
$\sim_A$ be a source classifier, and let
$\mathsf{cert}:\NameCert(A,\sim_A)$. If
$\mathsf{ListConsBoundaryCoherent}_{A,\sim_A}$ holds, then the displayed
public-object certificate target
$$
\mathsf{SemanticNameCert}(\mathsf{ListHistoryCarrier}_A,
\mathsf{ListHistoryPattern}_A,
\mathsf{ListHistoryLedger}_A,
\mathsf{ListSpineHistoryClassifier}_{A,\sim_A})
$$
holds. Moreover, for all histories $h$ and $k$, every certificate-classifier
witness
$$
\mathsf{ListSpineHistoryClassifier}_{A,\sim_A}(h,k)
$$
exposes finite-spine endpoint ledgers:
$$
\exists xs:\mathsf{HistSpine}.\;\mathsf{ListSpineRep}_A(h,xs)
$$
and
$$
\exists ys:\mathsf{HistSpine}.\;\mathsf{ListSpineRep}_A(k,ys).
$$
Consequently,
$$
\mathsf{ListHistoryCarrier}_A(h)\land\mathsf{ListHistoryCarrier}_A(k)
$$
by Theorem~\ref{thm:list-spine-representation-coverage}. The per-object fit is:
traditional target finite lists; BEDC source $A$ with
$\NameCert(A,\sim_A)$; carrier $\mathsf{ListHistoryCarrier}_A$; classifier
$\mathsf{ListSpineHistoryClassifier}_{A,\sim_A}$; ledger and exactness
finite-spine endpoint exposure; core certificate theorem
Theorem~\ref{thm:list-public-object-certificate-from-cons-boundary-coherence};
standard bridge finite spines; status Phase Finite Data.
\end{theorem}
\begin{proof}
First apply
Theorem~\ref{thm:list-public-object-certificate-from-cons-boundary-coherence}
to the source certificate and the cons-boundary coherence witness; this gives
the displayed semantic certificate target. For the endpoint ledger claim, unfold
$\mathsf{ListSpineHistoryClassifier}_{A,\sim_A}$ as the bridge classifier. A
bridge witness consists of finite spines $xs$ and $ys$, endpoint presentations
$\mathsf{ListSpineRep}_A(h,xs)$ and $\mathsf{ListSpineRep}_A(k,ys)$, and the
finite-spine classifier witness $\mathsf{ListClassifierSpec}(\sim_A,xs,ys)$.
The two endpoint presentations are the required ledgers. Applying the reverse
direction of Theorem~\ref{thm:list-spine-representation-coverage} to these
representation witnesses recovers
$\mathsf{ListHistoryCarrier}_A(h)$ and $\mathsf{ListHistoryCarrier}_A(k)$.
\end{proof}
