\begin{theorem}[List spine bridge source refinement]
\label{thm:list-spine-bridge-source-refinement}
Let $A,B:\mathsf{BHist}\to\mathsf{Prop}$ be BEDC history source carriers, and
let $\sim_A$ and $\sim_B$ be source classifiers. The ten-field fit is:
traditional target finite-list source refinement; BEDC source $\mathsf{BHist}$
with $\mathsf{ListSpineRep}$; carrier $\mathsf{ListHistoryCarrier}$; classifier
$\mathsf{ListSpineBridgeClassifier}$; stability obligation carrier and
classifier preservation under source refinement; ledger finite spine
representation; core theorem the three displayed implications below; exactness
inherited from $\mathsf{ListSpineRep}$; standard bridge finite-spine readback;
status $\mathsf{ProofTarget}$. Assume the source carrier and source classifier
of $A$ are included in those of $B$:
$$
\forall h:\mathsf{BHist}.\; A(h)\to B(h)
$$
and
$$
\forall x\,y:\mathsf{BHist}.\; x\sim_A y\to x\sim_B y.
$$
Then the displayed list-spine data transport from $A$ to $B$:
$$
\mathsf{ListSpineRep}_A(h,xs)
\to
\mathsf{ListSpineRep}_B(h,xs),
$$
$$
\mathsf{ListSpineBridgeClassifier}_{A,\sim_A}(h,k)
\to
\mathsf{ListSpineBridgeClassifier}_{B,\sim_B}(h,k),
$$
and
$$
\mathsf{ListSpineHistoryClassifier}_{A,\sim_A}(h,k)
\to
\mathsf{ListSpineHistoryClassifier}_{B,\sim_B}(h,k).
$$
\end{theorem}
\begin{proof}
Proceed by induction on the displayed $\mathsf{ListSpineRep}_A(h,xs)$ witness.
The nil representation keeps the same endpoint equality. In the cons case, the
carrier inclusion turns the recorded source datum $A(a)$ into $B(a)$, while the
induction hypothesis transports the tail representation and the continuation
and endpoint witnesses are reused unchanged. For the finite-spine classifier
component, induct on the displayed $\mathsf{ListClassifierSpec}(\sim_A,xs,ys)$
witness and apply the classifier inclusion at each cons head. A bridge witness is
then repacked with the transported left and right spines and the transported
finite-spine classifier. The certificate classifier clause follows by unfolding
that it is the bridge classifier and rebuilding the same endpoints.
\end{proof}

\begin{lemma}[List spine representation endpoint transport]
\label{lem:list-spine-representation-endpoint-transport}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier. For all
histories $h$ and $h'$ and every finite list spine $xs$,
$\mathsf{ListSpineRep}_A(h,xs)$ and $\hsame(h,h')$ imply
$\mathsf{ListSpineRep}_A(h',xs)$.
\end{lemma}
\begin{proof}
Proceed by induction on the displayed finite spine. The empty spine is transported
by composing the endpoint empty witness with $\hsame(h,h')$. For a cons spine,
keep the recorded head payload and tail representation, and compose the outer
history endpoint witness with $\hsame(h,h')$; the payload and tail witnesses are
unchanged.
\end{proof}

\begin{theorem}[List spine bridge endpoint transport]
\label{thm:list-spine-bridge-endpoint-transport}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier and let
$\sim_A$ be a source classifier. The ten-field per-object fit is: traditional
target finite-list classifier stability; BEDC source $A$ over $\mathsf{BHist}$;
source classifier $\sim_A$; carrier $\mathsf{ListHistoryCarrier}_A$; classifier
$\mathsf{ListSpineBridgeClassifier}_{A,\sim_A}$; stability endpoint transport;
ledger finite-spine presentation; exactness via $\mathsf{ListSpineRep}$; standard
bridge finite spines; status $\mathsf{ProofTarget}$. For all histories $h$, $k$,
$h'$, and $k'$,
$$
\mathsf{ListSpineBridgeClassifier}_{A,\sim_A}(h,k)
\to
\hsame(h,h')
\to
\hsame(k,k')
\to
\mathsf{ListSpineBridgeClassifier}_{A,\sim_A}(h',k').
$$
\end{theorem}
\begin{proof}
Unpack the bridge witness into finite spines $xs$ and $ys$, the represented
endpoints $\mathsf{ListSpineRep}_A(h,xs)$ and
$\mathsf{ListSpineRep}_A(k,ys)$, and the finite-spine classifier witness. Apply
Lemma~\ref{lem:list-spine-representation-endpoint-transport} to transport the
two displayed endpoint representations along $\hsame(h,h')$ and $\hsame(k,k')$.
The finite-spine classifier witness is independent of the represented endpoints,
so the bridge witness is rebuilt over $h'$ and $k'$ with the same spines.
\end{proof}

\begin{theorem}[List spine bridge nil-cons separation under coherence]
\label{thm:list-spine-bridge-nil-cons-separation-under-coherence}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier, let
$\sim_A$ be a source classifier, let $\mathsf{cert}:\NameCert(A,\sim_A)$, and
assume $\mathsf{ListSpineRepCoherent}_{A,\sim_A}$. For all histories $h$ and $k$,
all source histories $a$, and all finite spines $xs$, the bridge classifier
separates the nil and cons displayed endpoints:
$$
\mathsf{ListSpineRep}_A(h,\mathsf{nil})
\to
\mathsf{ListSpineRep}_A(k,\mathsf{cons}(a,xs))
\to
\mathsf{ListSpineBridgeClassifier}_{A,\sim_A}(h,k)
\to
\bot.
$$
It also separates the symmetric cons-vs-nil orientation:
$$
\mathsf{ListSpineRep}_A(h,\mathsf{cons}(a,xs))
\to
\mathsf{ListSpineRep}_A(k,\mathsf{nil})
\to
\mathsf{ListSpineBridgeClassifier}_{A,\sim_A}(h,k)
\to
\bot.
$$
\end{theorem}
\begin{proof}
Unpack the bridge witness into finite spines $u$ and $v$, endpoint presentations
$\mathsf{ListSpineRep}_A(h,u)$ and $\mathsf{ListSpineRep}_A(k,v)$, and a
finite-spine classifier $\mathsf{ListClassifierSpec}(\sim_A,u,v)$. In the
nil-vs-cons orientation, apply the coherence datum at endpoint $h$ to relate the
displayed $\mathsf{nil}$ spine to $u$, and at endpoint $k$ to relate $v$ to the
displayed $\mathsf{cons}(a,xs)$ spine. Compose these two finite-spine classifier
witnesses with the bridge classifier witness using transitivity supplied by
$\mathsf{cert}:\NameCert(A,\sim_A)$. The resulting
$\mathsf{ListClassifierSpec}(\sim_A,\mathsf{nil},\mathsf{cons}(a,xs))$ contradicts
Lemma~\ref{lem:list-classifier-nil-cons-no-confusion}. The cons-vs-nil
orientation is identical after swapping the two endpoint coherence comparisons
and using the symmetric nil-cons no-confusion clause.
\end{proof}

\begin{theorem}[List public-object certificate from cons-boundary coherence]
\label{thm:list-public-object-certificate-from-cons-boundary-coherence}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier, let
$\sim_A$ be a source classifier, and let
$\mathsf{cert}:\NameCert(A,\sim_A)$. The ten-field per-object fit is:
traditional target finite lists; BEDC source $A$ with $\NameCert(A,\sim_A)$;
carrier $\mathsf{ListHistoryCarrier}_A$; classifier
$\mathsf{ListSpineHistoryClassifier}_{A,\sim_A}$; stability supplied by the
source certificate and $\mathsf{ListConsBoundaryCoherent}_{A,\sim_A}$; ledger
$\mathsf{ListHistoryLedger}_A$; core theorem list
Theorem~\ref{thm:list-spine-coherence-from-cons-boundary-coherence} and
Theorem~\ref{thm:concrete-list-namecert}; exactness through the spine bridge
$\mathsf{ListSpineRep}_A$; standard bridge finite spines; status
$\mathsf{ProofTarget}$ under
$\mathsf{ListConsBoundaryCoherent}_{A,\sim_A}$. The semantic certificate target
is
$$
\mathsf{SemanticNameCert}(\mathsf{ListHistoryCarrier}_A,
\mathsf{ListHistoryPattern}_A,
\mathsf{ListHistoryLedger}_A,
\mathsf{ListSpineHistoryClassifier}_{A,\sim_A}).
$$
If $\mathsf{ListConsBoundaryCoherent}_{A,\sim_A}$ holds, then this semantic
certificate target holds.
\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:list-spine-coherence-from-cons-boundary-coherence} to
the cons-boundary coherence witness, obtaining
$\mathsf{ListSpineRepCoherent}_{A,\sim_A}$. Then apply
Theorem~\ref{thm:concrete-list-namecert} with the source certificate
$\mathsf{cert}$ and this spine-coherence witness. The resulting certificate is
exactly the displayed semantic target for the list public object.
\end{proof}
