\begin{theorem}[List spine bridge source refinement]
\label{thm:list-spine-bridge-source-refinement}
Let $A,B:\mathsf{BHist}\to\mathsf{Prop}$ be BEDC history source carriers, and
let $\sim_A$ and $\sim_B$ be source classifiers. The ten-field fit is:
traditional target finite-list source refinement; BEDC source $\mathsf{BHist}$
with $\mathsf{ListSpineRep}$; carrier $\mathsf{ListHistoryCarrier}$; classifier
$\mathsf{ListSpineBridgeClassifier}$; stability obligation carrier and
classifier preservation under source refinement; ledger finite spine
representation; core theorem the three displayed implications below; exactness
inherited from $\mathsf{ListSpineRep}$; standard bridge finite-spine readback;
status $\mathsf{ProofTarget}$. Assume the source carrier and source classifier
of $A$ are included in those of $B$:
$$
\forall h:\mathsf{BHist}.\; A(h)\to B(h)
$$
and
$$
\forall x\,y:\mathsf{BHist}.\; x\sim_A y\to x\sim_B y.
$$
Then the displayed list-spine data transport from $A$ to $B$:
$$
\mathsf{ListSpineRep}_A(h,xs)
\to
\mathsf{ListSpineRep}_B(h,xs),
$$
$$
\mathsf{ListSpineBridgeClassifier}_{A,\sim_A}(h,k)
\to
\mathsf{ListSpineBridgeClassifier}_{B,\sim_B}(h,k),
$$
and
$$
\mathsf{ListSpineHistoryClassifier}_{A,\sim_A}(h,k)
\to
\mathsf{ListSpineHistoryClassifier}_{B,\sim_B}(h,k).
$$
\end{theorem}
\begin{proof}
Proceed by induction on the displayed $\mathsf{ListSpineRep}_A(h,xs)$ witness.
The nil representation keeps the same endpoint equality. In the cons case, the
carrier inclusion turns the recorded source datum $A(a)$ into $B(a)$, while the
induction hypothesis transports the tail representation and the continuation
and endpoint witnesses are reused unchanged. For the finite-spine classifier
component, induct on the displayed $\mathsf{ListClassifierSpec}(\sim_A,xs,ys)$
witness and apply the classifier inclusion at each cons head. A bridge witness is
then repacked with the transported left and right spines and the transported
finite-spine classifier. The certificate classifier clause follows by unfolding
that it is the bridge classifier and rebuilding the same endpoints.
\end{proof}

\begin{theorem}[List public semantic fields source equivalence]
\label{thm:list-public-semantic-fields-source-equivalence}
Let $A,B:\mathsf{BHist}\to\mathsf{Prop}$ be BEDC history source carriers, and
let $\sim_A$ and $\sim_B$ be source classifiers. The ten-field fit is:
traditional target finite-list public semantic-field source-presentation
invariance; BEDC source $\mathsf{BHist}$ with presentations $A$ and $B$; public
semantic fields $\mathsf{ListHistoryCarrier}$, $\mathsf{ListHistoryPattern}$,
and $\mathsf{ListHistoryLedger}$; public bridge classifier
$\mathsf{ListSpineBridgeClassifier}$ and certificate classifier
$\mathsf{ListSpineHistoryClassifier}$; stability obligation source-carrier and
source-classifier equivalence; ledger finite spine; core theorem the five
displayed equivalences below; exactness inherited from the list semantic-field
and spine-bridge transport theorems; standard bridge finite-spine readback;
status $\mathsf{ProofTarget}$ for the finite-list public certificate interface.
Assume the source carriers are extensionally equivalent,
$$
\forall h:\mathsf{BHist}.\; A(h)\Longleftrightarrow B(h),
$$
and the source classifiers are extensionally equivalent,
$$
\forall x\,y:\mathsf{BHist}.\; x\sim_A y\Longleftrightarrow x\sim_B y.
$$
Then, for every history $h$,
$$
\mathsf{ListHistoryCarrier}_A(h)
\Longleftrightarrow
\mathsf{ListHistoryCarrier}_B(h),
$$
$$
\mathsf{ListHistoryPattern}_A(h)
\Longleftrightarrow
\mathsf{ListHistoryPattern}_B(h),
$$
and
$$
\mathsf{ListHistoryLedger}_A(h)
\Longleftrightarrow
\mathsf{ListHistoryLedger}_B(h).
$$
For every pair of histories $h,k$,
$$
\mathsf{ListSpineBridgeClassifier}_{A,\sim_A}(h,k)
\Longleftrightarrow
\mathsf{ListSpineBridgeClassifier}_{B,\sim_B}(h,k),
$$
and
$$
\mathsf{ListSpineHistoryClassifier}_{A,\sim_A}(h,k)
\Longleftrightarrow
\mathsf{ListSpineHistoryClassifier}_{B,\sim_B}(h,k).
$$
\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:list-history-semantic-fields-source-refinement} in the
$A\Rightarrow B$ direction using the forward half of the carrier equivalence,
and in the $B\Rightarrow A$ direction using the reverse half. Pairing these two
applications gives the carrier, pattern, and ledger equivalences. For the
bridge classifier, apply Theorem~\ref{thm:list-spine-bridge-source-refinement}
from $A$ to $B$ using the forward halves of the carrier and classifier
equivalences, then apply it from $B$ to $A$ using the reverse halves. The
certificate classifier clause is the same two-direction application because
$\mathsf{ListSpineHistoryClassifier}$ is the public certificate presentation of
the same bridge classifier endpoints.
\end{proof}

\begin{theorem}[List finite-spine classifier source equivalence]
\label{thm:list-finite-spine-classifier-source-equivalence}
Let $\sim_A$ and $\sim_B$ be source classifiers on $\mathsf{BHist}$. Assume
$$
\forall x\,y:\mathsf{BHist}.\; x\sim_A y\Longleftrightarrow x\sim_B y.
$$
Then, for all finite spines $xs$ and $ys$,
$$
\mathsf{ListClassifierSpec}(\sim_A,xs,ys)
\Longleftrightarrow
\mathsf{ListClassifierSpec}(\sim_B,xs,ys).
$$
\end{theorem}
\begin{proof}
For the forward implication, induct on the displayed
$\mathsf{ListClassifierSpec}(\sim_A,xs,ys)$ witness. The nil branch is rebuilt
by the nil constructor. In the cons branch, transport the head classifier witness
through the forward half of the classifier equivalence and apply the induction
hypothesis to the tail witness; these two data rebuild the cons witness for
$\sim_B$. The reverse implication is the same induction on the displayed
$\mathsf{ListClassifierSpec}(\sim_B,xs,ys)$ witness, using the reverse half of
the classifier equivalence at each head.
\end{proof}

\begin{theorem}[List spine-bridge coherence source equivalence]
\label{thm:list-spine-bridge-coherence-source-equivalence}
Let $A,B:\mathsf{BHist}\to\mathsf{Prop}$ be BEDC history source carriers, and
let $\sim_A$ and $\sim_B$ be source classifiers. Assume
$$
\forall h:\mathsf{BHist}.\; A(h)\Longleftrightarrow B(h)
$$
and
$$
\forall x\,y:\mathsf{BHist}.\; x\sim_A y\Longleftrightarrow x\sim_B y.
$$
Then
$$
\mathsf{ListSpineRepCoherent}_{A,\sim_A}
\Longleftrightarrow
\mathsf{ListSpineRepCoherent}_{B,\sim_B}.
$$
\end{theorem}
\begin{proof}
Assume $\mathsf{ListSpineRepCoherent}_{A,\sim_A}$. Given two
$B$-presentations of the same list endpoint, use
Theorem~\ref{thm:list-spine-bridge-source-refinement} in the reverse carrier
direction to view the displayed $\mathsf{ListSpineRep}_B$ witnesses as
$\mathsf{ListSpineRep}_A$ witnesses over the same finite spines. Apply the
$A$-coherence datum to obtain
$\mathsf{ListClassifierSpec}(\sim_A,xs,ys)$, then transport that finite-spine
classifier through
Theorem~\ref{thm:list-finite-spine-classifier-source-equivalence} to obtain the
required $\mathsf{ListClassifierSpec}(\sim_B,xs,ys)$ witness. This proves the
forward implication. The converse repeats the same argument with the two source
presentations interchanged, using the forward carrier direction and the reverse
classifier direction.
\end{proof}

coherence transport, both semantic certificates, and the five public field
equivalences displayed below; exactness inherited from the list semantic-field

source carriers are extensionally equivalent,
$$
\forall h:\mathsf{BHist}.\; A(h)\Longleftrightarrow B(h),
$$
and the source classifiers are extensionally equivalent,
$$
\forall x\,y:\mathsf{BHist}.\; x\sim_A y\Longleftrightarrow x\sim_B y.
$$
If the $A$-presentation has coherent list-spine representatives,

\mathsf{ListSpineRepCoherent}_{A,\sim_A},
$$
then the $B$-presentation has coherent list-spine representatives,

\mathsf{ListSpineRepCoherent}_{B,\sim_B},

the two source presentations carry semantic public certificates,

\mathsf{SemanticNameCert}(\mathsf{ListHistoryCarrier}_A,

\mathsf{ListHistoryLedger}_A,

and

and, for every history $h$,

Theorem~\ref{thm:concrete-list-namecert} with $\mathsf{cert}_A$ and the
original coherence witness to obtain the semantic certificate for the
$A$-presentation, and apply the same theorem with $\mathsf{cert}_B$ and the
transported coherence witness to obtain the semantic certificate for the
$B$-presentation. Finally apply
Theorem~\ref{thm:list-public-semantic-fields-source-equivalence} to the carrier
and classifier equivalences, yielding the carrier, pattern, ledger,
bridge-classifier, and public certificate-classifier equivalences.
\end{proof}

\begin{lemma}[List spine representation endpoint transport]
\label{lem:list-spine-representation-endpoint-transport}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier. For all
histories $h$ and $h'$ and every finite list spine $xs$,
$\mathsf{ListSpineRep}_A(h,xs)$ and $\hsame(h,h')$ imply
$\mathsf{ListSpineRep}_A(h',xs)$.
\end{lemma}
\begin{proof}
Proceed by induction on the displayed finite spine. The empty spine is transported
by composing the endpoint empty witness with $\hsame(h,h')$. For a cons spine,
keep the recorded head payload and tail representation, and compose the outer
history endpoint witness with $\hsame(h,h')$; the payload and tail witnesses are
unchanged.
\end{proof}

\begin{theorem}[List spine bridge endpoint transport]
\label{thm:list-spine-bridge-endpoint-transport}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier and let
$\sim_A$ be a source classifier. The ten-field per-object fit is: traditional
target finite-list classifier stability; BEDC source $A$ over $\mathsf{BHist}$;
source classifier $\sim_A$; carrier $\mathsf{ListHistoryCarrier}_A$; classifier
$\mathsf{ListSpineBridgeClassifier}_{A,\sim_A}$; stability endpoint transport;
ledger finite-spine presentation; exactness via $\mathsf{ListSpineRep}$; standard
bridge finite spines; status $\mathsf{ProofTarget}$. For all histories $h$, $k$,
$h'$, and $k'$,
$$
\mathsf{ListSpineBridgeClassifier}_{A,\sim_A}(h,k)
\to
\hsame(h,h')
\to
\hsame(k,k')
\to
\mathsf{ListSpineBridgeClassifier}_{A,\sim_A}(h',k').
$$
\end{theorem}
\begin{proof}
Unpack the bridge witness into finite spines $xs$ and $ys$, the represented
endpoints $\mathsf{ListSpineRep}_A(h,xs)$ and
$\mathsf{ListSpineRep}_A(k,ys)$, and the finite-spine classifier witness. Apply
Lemma~\ref{lem:list-spine-representation-endpoint-transport} to transport the
two displayed endpoint representations along $\hsame(h,h')$ and $\hsame(k,k')$.
The finite-spine classifier witness is independent of the represented endpoints,
so the bridge witness is rebuilt over $h'$ and $k'$ with the same spines.
\end{proof}

\begin{theorem}[List spine bridge nil-cons separation under coherence]
\label{thm:list-spine-bridge-nil-cons-separation-under-coherence}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier, let
$\sim_A$ be a source classifier, let $\mathsf{cert}:\NameCert(A,\sim_A)$, and
assume $\mathsf{ListSpineRepCoherent}_{A,\sim_A}$. For all histories $h$ and $k$,
all source histories $a$, and all finite spines $xs$, the bridge classifier
separates the nil and cons displayed endpoints:
$$
\mathsf{ListSpineRep}_A(h,\mathsf{nil})
\to
\mathsf{ListSpineRep}_A(k,\mathsf{cons}(a,xs))
\to
\mathsf{ListSpineBridgeClassifier}_{A,\sim_A}(h,k)
\to
\bot.
$$
It also separates the symmetric cons-vs-nil orientation:
$$
\mathsf{ListSpineRep}_A(h,\mathsf{cons}(a,xs))
\to
\mathsf{ListSpineRep}_A(k,\mathsf{nil})
\to
\mathsf{ListSpineBridgeClassifier}_{A,\sim_A}(h,k)
\to
\bot.
$$
\end{theorem}
\begin{proof}
Unpack the bridge witness into finite spines $u$ and $v$, endpoint presentations
$\mathsf{ListSpineRep}_A(h,u)$ and $\mathsf{ListSpineRep}_A(k,v)$, and a
finite-spine classifier $\mathsf{ListClassifierSpec}(\sim_A,u,v)$. In the
nil-vs-cons orientation, apply the coherence datum at endpoint $h$ to relate the
displayed $\mathsf{nil}$ spine to $u$, and at endpoint $k$ to relate $v$ to the
displayed $\mathsf{cons}(a,xs)$ spine. Compose these two finite-spine classifier
witnesses with the bridge classifier witness using transitivity supplied by
$\mathsf{cert}:\NameCert(A,\sim_A)$. The resulting
$\mathsf{ListClassifierSpec}(\sim_A,\mathsf{nil},\mathsf{cons}(a,xs))$ contradicts
Lemma~\ref{lem:list-classifier-nil-cons-no-confusion}. The cons-vs-nil
orientation is identical after swapping the two endpoint coherence comparisons
and using the symmetric nil-cons no-confusion clause.
\end{proof}

Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier, let

\mathsf{SemanticNameCert}(\mathsf{ListHistoryCarrier}_A,
\mathsf{ListHistoryPattern}_A,
\mathsf{ListHistoryLedger}_A,

\end{proof}

\begin{theorem}[List public-object certificate source equivalence]
\label{thm:list-public-object-certificate-source-equivalence-with-fields}
\label{thm:list-public-object-certificate-source-equivalence-ten-field}
\label{thm:list-public-object-certificate-source-equivalence-fit}
Let $A,B:\mathsf{BHist}\to\mathsf{Prop}$ be BEDC history source carriers, and
let $\sim_A$ and $\sim_B$ be source classifiers. Let
$\mathsf{cert}_A:\NameCert(A,\sim_A)$ and
$\mathsf{cert}_B:\NameCert(B,\sim_B)$. The ten-field certificate fit is:
traditional target finite-list public certificate equivalence; BEDC source
$\mathsf{BHist}$ with equivalent source presentations $A$ and $B$; carriers
$\mathsf{ListHistoryCarrier}_A$ and $\mathsf{ListHistoryCarrier}_B$; classifiers
$\mathsf{ListSpineHistoryClassifier}_{A,\sim_A}$ and
$\mathsf{ListSpineHistoryClassifier}_{B,\sim_B}$; stability source-carrier and
source-classifier equivalence; ledger finite generated history spine; core
theorem certificate transport plus semantic-field equivalence; exactness through
$\mathsf{ListSpineRep}$ and $\mathsf{ListClassifierSpec}$; standard bridge
finite spines; status $\mathsf{ProofTarget}$ for the finite-list public
certificate interface. Assume the source carriers are extensionally equivalent,
$$
\forall h:\mathsf{BHist}.\; A(h)\Longleftrightarrow B(h),
$$
and the source classifiers are extensionally equivalent,
$$
\forall x\,y:\mathsf{BHist}.\; x\sim_A y\Longleftrightarrow x\sim_B y.
$$
If the $A$-presentation has coherent list-spine representatives,
$$
\mathsf{ListSpineRepCoherent}_{A,\sim_A},
$$
then the $B$-presentation also has coherent list-spine representatives,
$$
\mathsf{ListSpineRepCoherent}_{B,\sim_B}.
$$
Moreover, both public finite-list semantic certificates hold:
$$
\mathsf{SemanticNameCert}(\mathsf{ListHistoryCarrier}_A,
\mathsf{ListHistoryPattern}_A,
\mathsf{ListHistoryLedger}_A,
\mathsf{ListSpineHistoryClassifier}_{A,\sim_A})
$$
and
$$
\mathsf{SemanticNameCert}(\mathsf{ListHistoryCarrier}_B,
\mathsf{ListHistoryPattern}_B,
\mathsf{ListHistoryLedger}_B,
\mathsf{ListSpineHistoryClassifier}_{B,\sim_B}).
$$
Finally, for every history $h$,
$$
\mathsf{ListHistoryCarrier}_A(h)
\Longleftrightarrow
\mathsf{ListHistoryCarrier}_B(h),
$$
$$
\mathsf{ListHistoryPattern}_A(h)
\Longleftrightarrow
\mathsf{ListHistoryPattern}_B(h),
$$
and
$$
\mathsf{ListHistoryLedger}_A(h)
\Longleftrightarrow
\mathsf{ListHistoryLedger}_B(h).
$$
For every pair of histories $h,k$,
$$
\mathsf{ListSpineBridgeClassifier}_{A,\sim_A}(h,k)
\Longleftrightarrow
\mathsf{ListSpineBridgeClassifier}_{B,\sim_B}(h,k),
$$
and
$$
\mathsf{ListSpineHistoryClassifier}_{A,\sim_A}(h,k)
\Longleftrightarrow
\mathsf{ListSpineHistoryClassifier}_{B,\sim_B}(h,k).
$$
\end{theorem}
\begin{proof}
To obtain $\mathsf{ListSpineRepCoherent}_{B,\sim_B}$, take two displayed
$B$-spine presentations of the same endpoint. Use
Theorem~\ref{thm:list-spine-bridge-source-refinement} with the reverse halves
of the carrier and classifier equivalences to transport those presentations
back to $A$. Apply the assumed $\mathsf{ListSpineRepCoherent}_{A,\sim_A}$ to
get the finite-spine classifier comparison over $\sim_A$, then transport that
comparison forward entry by entry through the classifier equivalence to obtain
the required $\mathsf{ListClassifierSpec}(\sim_B)$ comparison. Apply
Theorem~\ref{thm:concrete-list-namecert} once with $\mathsf{cert}_A$ and the
assumed $A$-coherence, and once with $\mathsf{cert}_B$ and the transported
$B$-coherence, giving the two displayed semantic certificates. The five
semantic-field and classifier equivalences are exactly
Theorem~\ref{thm:list-public-semantic-fields-source-equivalence} applied to the
same source-carrier and source-classifier equivalences.
\end{proof}
