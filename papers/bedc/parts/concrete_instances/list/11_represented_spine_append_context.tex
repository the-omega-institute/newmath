\section{Public append readback}

\begin{definition}[List public append readback]
\label{def:list-public-append-readback}
For a BEDC history source carrier $A:\mathsf{BHist}\to\mathsf{Prop}$, define
$\mathsf{ListPublicAppend}_A(h,k,r)$ to mean that there exist finite
BEDC-history spines $xs,ys:\mathsf{HistSpine}$ such that
$$
\mathsf{ListSpineRep}_A(h,xs),
\quad
\mathsf{ListSpineRep}_A(k,ys),
$$
and
$$
\mathsf{ListSpineRep}_A(r,xs{++}ys)
$$
hold.
\end{definition}

\begin{lemma}[List spine append representation construction]
\label{lem:list-spine-append-representation-construction}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier. For
finite BEDC-history spines $xs,ys:\mathsf{HistSpine}$ and histories $h,k$, if
$$
\mathsf{ListSpineRep}_A(h,xs)
\quad\hbox{and}\quad
\mathsf{ListSpineRep}_A(k,ys)
$$
hold, then there exists a history $r:\mathsf{BHist}$ such that
$$
\mathsf{ListSpineRep}_A(r,xs{++}ys).
$$
\end{lemma}
\begin{proof}
Proceed by induction on the displayed representation
$\mathsf{ListSpineRep}_A(h,xs)$.

In the nil case, the displayed spine $xs$ is $\mathsf{nil}$, so
$xs{++}ys$ is $ys$. Reuse the endpoint $k$ together with the given
representation $\mathsf{ListSpineRep}_A(k,ys)$.

In the cons case, write $xs=\mathsf{cons}(a,xs_0)$ and let the head
constructor provide $A(a)$, a tail endpoint $t$, a tail representation
$\mathsf{ListSpineRep}_A(t,xs_0)$, and a constructor ledger $\ContR(a,t,h)$.
The induction hypothesis applied to the tail representation and the endpoint
representation of $k$ gives a history $r_0$ represented by $xs_0{++}ys$.

Use the same head payload $a$ and the constructed tail endpoint $r_0$ to form a
BEDC constructor endpoint $r$ with $\ContR(a,r_0,r)$. The cons constructor for
$\mathsf{ListSpineRep}_A$ then represents $r$ by
$\mathsf{cons}(a,xs_0{++}ys)$, which is $xs{++}ys$.
\end{proof}
\leanchecked{BEDC.Derived.ListUp.ListSpineRep\_append\_representation}

\begin{theorem}[List represented-spine same-tag context exactness]
\label{thm:list-represented-spine-same-tag-context-exactness}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier, let
$\mathsf{cert}:\NameCert(A,\hsame)$, and assume
$\mathsf{ListSpineRepCoherent}_{A,\hsame}$. For finite BEDC-history spines
$p,p',x,y,s,s':\mathsf{HistSpine}$, suppose
$$
\mathsf{ListClassifierSpec}(\hsame,p,p')
$$
and
$$
\mathsf{ListClassifierSpec}(\hsame,s,s')
$$
hold. Then same visible constructor tags expose exactly the untagged payload
spine classifier:
\begin{enumerate}
  \item for histories $h_0,k_0$, if
  $$
  \mathsf{ListSpineRep}_A(h_0,p{++}(\mathsf{map}(e0,x){++}s)),
  \quad
  \mathsf{ListSpineRep}_A(k_0,p'{++}(\mathsf{map}(e0,y){++}s'))
  $$
  hold, then
  $$
  \mathsf{ListSpineHistoryClassifier}_{A,\hsame}(h_0,k_0)
  \Longleftrightarrow
  \mathsf{ListClassifierSpec}(\hsame,x,y);
  $$
  \item for histories $h_1,k_1$, if
  $$
  \mathsf{ListSpineRep}_A(h_1,p{++}(\mathsf{map}(e1,x){++}s)),
  \quad
  \mathsf{ListSpineRep}_A(k_1,p'{++}(\mathsf{map}(e1,y){++}s'))
  $$
  hold, then
  $$
  \mathsf{ListSpineHistoryClassifier}_{A,\hsame}(h_1,k_1)
  \Longleftrightarrow
  \mathsf{ListClassifierSpec}(\hsame,x,y).
  $$
\end{enumerate}

The ten-field fit is: traditional target finite-list same-tag payload readback;
BEDC source $A$ with $\NameCert(A,\hsame)$; carrier
$\mathsf{ListHistoryCarrier}_A$; classifier
$\mathsf{ListSpineHistoryClassifier}_{A,\hsame}$; stability from spine
coherence, constructor-map stability, and append-context rebuilding; ledger
visible $e0/e1$ branch tags in finite spines; core theorem same-tag context
exactness; exactness by displayed-spine classifier readback and constructor-map
inversion; standard bridge finite-spine readback; status
$\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
For the $e0$ forward implication, apply
Theorem~\ref{thm:list-represented-spine-classifier-displayed-exactness} to the
two represented contextual endpoints. This reads the public classifier as
$$
\mathsf{ListClassifierSpec}(\hsame,
  p{++}(\mathsf{map}(e0,x){++}s),
  p'{++}(\mathsf{map}(e0,y){++}s')).
$$

Apply Theorem~\ref{thm:list-classifier-BHist-append-context-iff} with the
classified prefixes and suffixes to cancel the displayed context. The result is
$\mathsf{ListClassifierSpec}(\hsame,\mathsf{map}(e0,x),\mathsf{map}(e0,y))$,
and Theorem~\ref{thm:list-classifier-hsame-map-e0-inversion} reflects it to
$\mathsf{ListClassifierSpec}(\hsame,x,y)$.

For the $e0$ reverse implication, start with
$\mathsf{ListClassifierSpec}(\hsame,x,y)$. Theorem~\ref{thm:list-classifier-hsame-map-e0}
transports it to the mapped middle spines, and the reverse direction of
Theorem~\ref{thm:list-classifier-BHist-append-context-iff} rebuilds the
classified prefix and suffix context.

The reverse direction of
Theorem~\ref{thm:list-represented-spine-classifier-displayed-exactness} then
repacks the displayed contextual classifier as
$\mathsf{ListSpineHistoryClassifier}_{A,\hsame}(h_0,k_0)$.

The $e1$ clause is identical with $e1$ in place of $e0$. Its forward direction
uses displayed exactness and append-context cancellation to obtain
$\mathsf{ListClassifierSpec}(\hsame,\mathsf{map}(e1,x),\mathsf{map}(e1,y))$,
then applies Theorem~\ref{thm:list-classifier-hsame-map-e1-inversion}. Its
reverse direction applies Theorem~\ref{thm:list-classifier-hsame-map-e1},
rebuilds the classified context, and repacks through displayed exactness for
the represented endpoints $h_1,k_1$.
\end{proof}

\begin{theorem}[List represented-spine same-tag payload determinacy]
\label{thm:list-represented-spine-same-tag-payload-determinacy}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier, let
$\mathsf{cert}:\NameCert(A,\hsame)$, and assume
$\mathsf{ListSpineRepCoherent}_{A,\hsame}$. For finite BEDC-history spines
$p,p',x,y,s,s':\mathsf{HistSpine}$ and a history $h$, suppose
$$
\mathsf{ListClassifierSpec}(\hsame,p,p')
$$
and
$$
\mathsf{ListClassifierSpec}(\hsame,s,s')
$$
hold. Then same visible constructor tags around two represented-spine readbacks
of $h$ determine the untagged payload classifier:
\begin{enumerate}
  \item if
  $$
  \mathsf{ListSpineRep}_A(h,p{++}(\mathsf{map}(e0,x){++}s)),
  \quad
  \mathsf{ListSpineRep}_A(h,p'{++}(\mathsf{map}(e0,y){++}s'))
  $$
  hold, then
  $$
  \mathsf{ListClassifierSpec}(\hsame,x,y);
  $$
  \item if
  $$
  \mathsf{ListSpineRep}_A(h,p{++}(\mathsf{map}(e1,x){++}s)),
  \quad
  \mathsf{ListSpineRep}_A(h,p'{++}(\mathsf{map}(e1,y){++}s'))
  $$
  hold, then
  $$
  \mathsf{ListClassifierSpec}(\hsame,x,y).
  $$
\end{enumerate}

The ten-field fit is: traditional target finite-list same-history readback
determinacy; BEDC source $A$ with $\NameCert(A,\hsame)$; carrier
$\mathsf{ListHistoryCarrier}_A$; classifier
$\mathsf{ListSpineHistoryClassifier}_{A,\hsame}$; stability from spine
coherence and certificate reflexivity; ledger visible $e0/e1$ branch tags in
finite spines; core theorem same-tag context exactness at identical endpoints;
exactness by represented-spine carrier coverage and public classifier
reflexivity; standard bridge finite-spine readback; status
$\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Start with either displayed represented-spine presentation of $h$. By
Theorem~\ref{thm:list-spine-representation-exactness}, the presentation exposes
$\mathsf{ListHistoryCarrier}_A(h)$.

Apply Theorem~\ref{thm:concrete-list-namecert} to $\mathsf{cert}$ and the
coherence witness. The resulting semantic certificate supplies classifier
reflexivity for the carrier witness, hence
$$
\mathsf{ListSpineHistoryClassifier}_{A,\hsame}(h,h).
$$

For the $e0$ clause, specialize
Theorem~\ref{thm:list-represented-spine-same-tag-context-exactness} to the two
endpoints $h$ and $h$, using the displayed $e0$ presentations, the classified
prefix and suffix hypotheses, and the reflexive public classifier above. Project
the forward implication of the $e0$ branch to obtain
$\mathsf{ListClassifierSpec}(\hsame,x,y)$.

The $e1$ clause is the same specialization with $e1$ in place of $e0$. Project
the forward implication of the $e1$ branch, again using the public reflexivity of
$h$, to obtain $\mathsf{ListClassifierSpec}(\hsame,x,y)$.
\end{proof}
