\begin{theorem}[List represented-spine append-context cancellation]
\label{thm:list-represented-spine-append-context-cancellation}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier, let
$\sim_A$ be a source classifier, let $\mathsf{cert}:\NameCert(A,\sim_A)$, and
assume $\mathsf{ListSpineRepCoherent}_{A,\sim_A}$. For finite BEDC-history
spines $p,p',x,y,s,s':\mathsf{HistSpine}$ and histories $h,k,m,n$, suppose
$$
\mathsf{ListClassifierSpec}(\sim_A,p,p')
$$
and
$$
\mathsf{ListClassifierSpec}(\sim_A,s,s')
$$
hold. If
$$
\mathsf{ListSpineRep}_A(h,p{++}(x{++}s)),
\quad
\mathsf{ListSpineRep}_A(k,p'{++}(y{++}s')),
$$
and
$$
\mathsf{ListSpineRep}_A(m,x),
\quad
\mathsf{ListSpineRep}_A(n,y)
$$
hold, then
$$
\mathsf{ListSpineHistoryClassifier}_{A,\sim_A}(h,k)
\Longleftrightarrow
\mathsf{ListSpineHistoryClassifier}_{A,\sim_A}(m,n).
$$
\end{theorem}
\begin{proof}
For the forward implication, apply
Theorem~\ref{thm:list-represented-spine-classifier-displayed-exactness} to the
represented contextual endpoints $h$ and $k$. This turns the public classifier
witness into
$$
\mathsf{ListClassifierSpec}(\sim_A,p{++}(x{++}s),p'{++}(y{++}s')).
$$

The classified prefix and suffix hypotheses allow
Theorem~\ref{thm:list-classifier-BHist-append-context-iff} to cancel the displayed
finite-spine context, giving $\mathsf{ListClassifierSpec}(\sim_A,x,y)$. Applying
the reverse direction of
Theorem~\ref{thm:list-represented-spine-classifier-displayed-exactness} to the
represented middle endpoints $m$ and $n$ repacks that witness as
$\mathsf{ListSpineHistoryClassifier}_{A,\sim_A}(m,n)$.

For the reverse implication, apply
Theorem~\ref{thm:list-represented-spine-classifier-displayed-exactness} to the
represented middle endpoints $m$ and $n$. This reads the public middle
classifier as $\mathsf{ListClassifierSpec}(\sim_A,x,y)$.

Use Theorem~\ref{thm:list-classifier-BHist-append-context-iff} in the opposite
direction with the same classified prefixes and suffixes to obtain
$$
\mathsf{ListClassifierSpec}(\sim_A,p{++}(x{++}s),p'{++}(y{++}s')).
$$
The reverse direction of
Theorem~\ref{thm:list-represented-spine-classifier-displayed-exactness} for the
represented contextual endpoints $h$ and $k$ repacks the displayed contextual
classifier as $\mathsf{ListSpineHistoryClassifier}_{A,\sim_A}(h,k)$.
\end{proof}

\begin{theorem}[List represented-spine mixed-tag context separation]
\label{thm:list-represented-spine-mixed-tag-context-separation}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier, let
$\mathsf{cert}:\NameCert(A,\hsame)$, and assume
$\mathsf{ListSpineRepCoherent}_{A,\hsame}$. For finite BEDC-history spines
$p,p',x,y,s,s':\mathsf{HistSpine}$ and histories $h,k,m,n$, suppose
$$
\mathsf{ListClassifierSpec}(\hsame,p,p')
$$
and
$$
\mathsf{ListClassifierSpec}(\hsame,s,s')
$$
hold. If
$$
\mathsf{ListSpineRep}_A(h,p{++}(\mathsf{map}(e0,x){++}s)),
\quad
\mathsf{ListSpineRep}_A(k,p'{++}(\mathsf{map}(e1,y){++}s')),
$$
and
$$
\mathsf{ListSpineRep}_A(m,\mathsf{map}(e0,x)),
\quad
\mathsf{ListSpineRep}_A(n,\mathsf{map}(e1,y))
$$
hold, and if
$$
\mathsf{ListSpineHistoryClassifier}_{A,\hsame}(h,k)
$$
holds, then $x=\mathsf{nil}$ and $y=\mathsf{nil}$.

The ten-field fit is: traditional target finite-list tagged-branch separation;
BEDC source $A$ with $\NameCert(A,\hsame)$; carrier
$\mathsf{ListHistoryCarrier}_A$; classifier
$\mathsf{ListSpineHistoryClassifier}_{A,\hsame}$; stability from spine
coherence, append-context cancellation, and history constructor no-confusion;
ledger visible $e0/e1$ branch tags in finite spines; core theorem mixed-tag
context separation; exactness by displayed-spine classifier exactness; standard
bridge finite-spine readback; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:list-represented-spine-append-context-cancellation} with
middle spines $\mathsf{map}(e0,x)$ and $\mathsf{map}(e1,y)$. The classified
prefix and suffix hypotheses, the four represented-spine witnesses, and the
public classifier on $h,k$ transfer the classifier to the represented middle
histories $m,n$.

Next apply
Theorem~\ref{thm:list-represented-spine-classifier-displayed-exactness} to the
middle representations of $m$ and $n$. This reads the public middle classifier as
$$
\mathsf{ListClassifierSpec}(\hsame,\mathsf{map}(e0,x),\mathsf{map}(e1,y)).
$$

Finally apply Theorem~\ref{thm:list-classifier-mixed-tag-empty}. The displayed
classifier can relate an $e0$-mapped spine to an $e1$-mapped spine only in the
empty-empty case, so $x=\mathsf{nil}$ and $y=\mathsf{nil}$.
\end{proof}
