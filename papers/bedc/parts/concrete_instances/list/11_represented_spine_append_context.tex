\begin{theorem}[List represented-spine append-context cancellation]
\label{thm:list-represented-spine-append-context-cancellation}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier, let
$\sim_A$ be a source classifier, let $\mathsf{cert}:\NameCert(A,\sim_A)$, and
assume $\mathsf{ListSpineRepCoherent}_{A,\sim_A}$. For finite BEDC-history
spines $p,p',x,y,s,s':\mathsf{HistSpine}$ and histories $h,k,m,n$, suppose
$$
\mathsf{ListClassifierSpec}(\sim_A,p,p')
$$
and
$$
\mathsf{ListClassifierSpec}(\sim_A,s,s')
$$
hold. If
$$
\mathsf{ListSpineRep}_A(h,p{++}(x{++}s)),
\quad
\mathsf{ListSpineRep}_A(k,p'{++}(y{++}s')),
$$
and
$$
\mathsf{ListSpineRep}_A(m,x),
\quad
\mathsf{ListSpineRep}_A(n,y)
$$
hold, then
$$
\mathsf{ListSpineHistoryClassifier}_{A,\sim_A}(h,k)
\Longleftrightarrow
\mathsf{ListSpineHistoryClassifier}_{A,\sim_A}(m,n).
$$
\end{theorem}
\begin{proof}
For the forward implication, apply
Theorem~\ref{thm:list-represented-spine-classifier-displayed-exactness} to the
represented contextual endpoints $h$ and $k$. This turns the public classifier
witness into
$$
\mathsf{ListClassifierSpec}(\sim_A,p{++}(x{++}s),p'{++}(y{++}s')).
$$

The classified prefix and suffix hypotheses allow
Theorem~\ref{thm:list-classifier-BHist-append-context-iff} to cancel the displayed
finite-spine context, giving $\mathsf{ListClassifierSpec}(\sim_A,x,y)$. Applying
the reverse direction of
Theorem~\ref{thm:list-represented-spine-classifier-displayed-exactness} to the
represented middle endpoints $m$ and $n$ repacks that witness as
$\mathsf{ListSpineHistoryClassifier}_{A,\sim_A}(m,n)$.

For the reverse implication, apply
Theorem~\ref{thm:list-represented-spine-classifier-displayed-exactness} to the
represented middle endpoints $m$ and $n$. This reads the public middle
classifier as $\mathsf{ListClassifierSpec}(\sim_A,x,y)$.

Use Theorem~\ref{thm:list-classifier-BHist-append-context-iff} in the opposite
direction with the same classified prefixes and suffixes to obtain
$$
\mathsf{ListClassifierSpec}(\sim_A,p{++}(x{++}s),p'{++}(y{++}s')).
$$
The reverse direction of
Theorem~\ref{thm:list-represented-spine-classifier-displayed-exactness} for the
represented contextual endpoints $h$ and $k$ repacks the displayed contextual
classifier as $\mathsf{ListSpineHistoryClassifier}_{A,\sim_A}(h,k)$.
\end{proof}

\begin{theorem}[List classifier mixed-tag append-context exactness]
\label{thm:list-classifier-mixed-tag-append-context-exactness}
For finite BEDC-history spines $p,p',x,y,s,s'$, suppose
$$
\mathsf{ListClassifierSpec}(\hsame,p,p')
$$
and
$$
\mathsf{ListClassifierSpec}(\hsame,s,s')
$$
hold. Then mixed visible constructor tags are classified by the finite-spine
classifier exactly in the empty middle case:
\begin{enumerate}
  \item
  $$
  \mathsf{ListClassifierSpec}(\hsame,
    p{++}(\mathsf{map}(e0,x){++}s),
    p'{++}(\mathsf{map}(e1,y){++}s'))
  \Longleftrightarrow
  x=\mathsf{nil}\land y=\mathsf{nil};
  $$
  \item
  $$
  \mathsf{ListClassifierSpec}(\hsame,
    p{++}(\mathsf{map}(e1,x){++}s),
    p'{++}(\mathsf{map}(e0,y){++}s'))
  \Longleftrightarrow
  x=\mathsf{nil}\land y=\mathsf{nil}.
  $$
\end{enumerate}
\end{theorem}
\begin{proof}
For the $e0/e1$ forward implication, apply
Theorem~\ref{thm:list-classifier-BHist-append-context-iff} with the classified
prefixes and suffixes. This cancels the displayed context and leaves
$$
\mathsf{ListClassifierSpec}(\hsame,\mathsf{map}(e0,x),\mathsf{map}(e1,y)).
$$
Theorem~\ref{thm:list-classifier-mixed-tag-empty} then forces
$x=\mathsf{nil}$ and $y=\mathsf{nil}$.

For the $e0/e1$ reverse implication, assume
$x=\mathsf{nil}$ and $y=\mathsf{nil}$. The mapped middle spines are both
$\mathsf{nil}$, so the nil clause of
Definition~\ref{def:list-classifier-specification} gives the displayed nil
classifier. The reverse direction of
Theorem~\ref{thm:list-classifier-BHist-append-context-iff} rebuilds the
classified prefix and suffix context.

For the $e1/e0$ forward implication, first cancel the displayed prefix and
suffix context by Theorem~\ref{thm:list-classifier-BHist-append-context-iff}.
Apply Theorem~\ref{thm:list-classifier-hsame-symm} to reverse the resulting
middle classifier, obtaining
$$
\mathsf{ListClassifierSpec}(\hsame,\mathsf{map}(e0,y),\mathsf{map}(e1,x)).
$$
Theorem~\ref{thm:list-classifier-mixed-tag-empty} gives
$y=\mathsf{nil}$ and $x=\mathsf{nil}$, hence the required conjunction.

For the $e1/e0$ reverse implication, the assumptions again make both mapped
middle spines $\mathsf{nil}$. Use the nil classifier and rebuild the displayed
context by the reverse direction of
Theorem~\ref{thm:list-classifier-BHist-append-context-iff}.
\end{proof}
