\chapter{A Concrete Naming Certificate for \(\ListUp\)}
\label{ch:concrete-instances-list-namecert}

The list interface $\ListUp$ is licensed over a source carrier endowed with a $\NameCert$ and a spine-representation coherence witness. The carrier is the inductive sequence built from a nil constructor and a cons constructor consuming one source witness and one shorter list. The classifier is structural sameness lifted from the source classifier through finite spines: nil-with-nil and cons-with-cons under componentwise sameness. It is not a primitive sequence type; it is licensed only as a derived inductive over the source under this coherence datum.

\section{Carrier definition}

\begin{definition}[List finite-spine carrier]
\label{def:list-carrier}
For a source type $A$, $\mathsf{ListCarrier}(A)$ denotes the finite-spine helper with entries of type $A$; $\mathsf{ListClassifierSpec}$ uses this helper for structural nil/cons reasoning. The exported BEDC public history carrier is $\mathsf{ListHistoryCarrier}_S$ in Definition~\ref{def:list-history-carrier}.
\end{definition}

\section{The certificate}

\begin{definition}[List source specification]
\label{def:list-source-specification}
The source is finite generated sequencing over the source certificate for \(A\).
\end{definition}

\begin{definition}[List history carrier]
\label{def:list-history-carrier}
For a source carrier $S:\mathsf{BHist}\to\mathsf{Prop}$, define $\mathsf{ListHistoryCarrier}_S(h)$ as the least history predicate generated by two clauses:
\begin{enumerate}
  \item nil: if $\hsame(h,\emp)$, then $\mathsf{ListHistoryCarrier}_S(h)$;
  \item cons: if there are histories $a,t,p$ with $S(a)$, $\mathsf{ListHistoryCarrier}_S(t)$, $\ContR(a,t,p)$, and $\hsame(h,\Eone(p))$, then $\mathsf{ListHistoryCarrier}_S(h)$.
\end{enumerate}
\end{definition}

\begin{definition}[List recursive history classifier]
\label{def:list-history-classifier}
For a source carrier $S:\mathsf{BHist}\to\mathsf{Prop}$ and source classifier
$\sim_S:\mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$, define
$\mathsf{ListHistoryClassifier}^{\mathrm{rec}}_{S,\sim_S}(h,k)$ as the least relation
generated by two clauses:
\begin{enumerate}
  \item nil-nil: if $\hsame(h,\emp)$ and $\hsame(k,\emp)$, then
  $\mathsf{ListHistoryClassifier}^{\mathrm{rec}}_{S,\sim_S}(h,k)$;
  \item cons-cons: if there are histories $a,b,t,u,p,q$ such that $S(a)$,
  $S(b)$, $a\sim_S b$, $\mathsf{ListHistoryClassifier}^{\mathrm{rec}}_{S,\sim_S}(t,u)$,
  $\ContR(a,t,p)$, $\ContR(b,u,q)$, $\hsame(h,\Eone(p))$, and
  $\hsame(k,\Eone(q))$, then
  $\mathsf{ListHistoryClassifier}^{\mathrm{rec}}_{S,\sim_S}(h,k)$.
\end{enumerate}
\end{definition}

\begin{theorem}[List recursive history classifier carrier endpoints]
\label{thm:list-history-classifier-carrier-endpoints}
For any source carrier $S:\mathsf{BHist}\to\mathsf{Prop}$ and source classifier
$\sim_S:\mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$, if
$\mathsf{ListHistoryClassifier}^{\mathrm{rec}}_{S,\sim_S}(h,k)$ holds, then both
$\mathsf{ListHistoryCarrier}_S(h)$ and $\mathsf{ListHistoryCarrier}_S(k)$
hold.
\end{theorem}
\begin{proof}
Induct on the displayed classifier witness. In the nil-nil clause, the two
endpoint witnesses rebuild the nil clause of $\mathsf{ListHistoryCarrier}_S$
on both sides. In the cons-cons clause, the induction hypothesis gives carrier
witnesses for the two tails $t$ and $u$; combine those with the source carrier
witnesses $S(a)$ and $S(b)$, the continuation witnesses $\ContR(a,t,p)$ and
$\ContR(b,u,q)$, and the endpoint witnesses $\hsame(h,\Eone(p))$ and
$\hsame(k,\Eone(q))$ to rebuild the two cons carrier clauses.
\end{proof}

\begin{theorem}[List recursive history classifier constructor exactness]
\label{thm:list-history-classifier-constructor-exactness}
For any source carrier $S:\mathsf{BHist}\to\mathsf{Prop}$, source classifier
$\sim_S:\mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$, and histories $h,k$,
the recursive classifier is exactly the constructor disjunction
$$
\mathsf{ListHistoryClassifier}^{\mathrm{rec}}_{S,\sim_S}(h,k)
\Longleftrightarrow
\bigl(\hsame(h,\emp)\wedge\hsame(k,\emp)\bigr)\vee
\exists a\,b\,t\,u\,p\,q.\;S(a)\wedge S(b)\wedge a\sim_S b
\wedge \mathsf{ListHistoryClassifier}^{\mathrm{rec}}_{S,\sim_S}(t,u)
\wedge \ContR(a,t,p)\wedge \ContR(b,u,q)\wedge \hsame(h,\Eone(p))\wedge \hsame(k,\Eone(q)).
$$
\end{theorem}
\begin{proof}
For the forward implication, induct on the generated classifier witness. The
nil-nil constructor gives the left disjunct, and the cons-cons constructor
gives the right disjunct with its source head witnesses, recursive tail-classifier
witness, continuation witnesses, and visible $e1$ endpoint witnesses. For the
reverse implication, the left disjunct is exactly the nil-nil generation clause,
and the right disjunct is exactly the cons-cons generation clause. Mixed nil/cons
endpoint claims reduce to BEDC history no-confusion:
$\mathsf{not\_hsame\_emp\_e1}$ eliminates nil against cons, and
$\mathsf{not\_hsame\_e1\_empty}$ eliminates cons against nil.
\end{proof}

\begin{theorem}[List history carrier generated cases]
\label{thm:list-history-carrier-generated-cases}
For any source carrier $S:\mathsf{BHist}\to\mathsf{Prop}$ and any history $h$, if $\mathsf{ListHistoryCarrier}_S(h)$, then either $\hsame(h,\emp)$, or there exist histories $a,t,p$ such that $S(a)$, $\mathsf{ListHistoryCarrier}_S(t)$, $\ContR(a,t,p)$, and $\hsame(h,\Eone(p))$.
\end{theorem}
\begin{proof}
Apply the induction principle for the least predicate displayed in the preceding definition. The nil constructor gives the left disjunct directly. The cons constructor exposes its witnesses $a,t,p$ together with $S(a)$, the recursive carrier witness for $t$, the continuation witness $\ContR(a,t,p)$, and the endpoint witness $\hsame(h,\Eone(p))$, which are exactly the data required for the right disjunct.
\end{proof}

\begin{theorem}[List history carrier transport]
\label{thm:list-history-carrier-hsame-transport}
For any source carrier $S:\mathsf{BHist}\to\mathsf{Prop}$, if $\hsame(h,k)$ and $\mathsf{ListHistoryCarrier}_S(h)$, then $\mathsf{ListHistoryCarrier}_S(k)$.
\end{theorem}
\begin{proof}
Proceed by cases on the displayed generation of $\mathsf{ListHistoryCarrier}_S(h)$. In the nil case, compose the symmetry of $\hsame(h,k)$ with the endpoint witness $\hsame(h,\emp)$ to obtain $\hsame(k,\emp)$, then use the nil clause. In the cons case, keep the witnesses $a,t,p$, the source witness $S(a)$, the recursive carrier witness for $t$, and the continuation witness $\ContR(a,t,p)$. Compose the symmetry of $\hsame(h,k)$ with the endpoint witness $\hsame(h,\Eone(p))$ to obtain $\hsame(k,\Eone(p))$, then rebuild the cons clause for $k$.
\end{proof}

\begin{theorem}[List recursive history classifier reflexivity, symmetry, and endpoint exposure from a source certificate]\label{thm:list-recursive-history-classifier-equivalence-fields}
Let $A : \mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier, let $\sim_A : \mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$ be a source classifier, and let $\mathsf{cert} : \NameCert(A,\sim_A)$. The generated classifier $\mathsf{ListHistoryClassifier}^{\mathrm{rec}}_{A,\sim_A}$ records the following recursive classifier fields here, rather than the entire certificate stability block:
\begin{enumerate}
  \item if $\mathsf{ListHistoryCarrier}_A(h)$ holds, then $\mathsf{ListHistoryClassifier}^{\mathrm{rec}}_{A,\sim_A}(h,h)$ holds;
  \item if $\mathsf{ListHistoryClassifier}^{\mathrm{rec}}_{A,\sim_A}(h,k)$ holds, then $\mathsf{ListHistoryClassifier}^{\mathrm{rec}}_{A,\sim_A}(k,h)$ holds;
  \item if $\mathsf{ListHistoryClassifier}^{\mathrm{rec}}_{A,\sim_A}(h,k)$ holds, then both $\mathsf{ListHistoryCarrier}_A(h)$ and $\mathsf{ListHistoryCarrier}_A(k)$ hold.
\end{enumerate}
\end{theorem}
\begin{proof}
For reflexivity, induct on the displayed generation of $\mathsf{ListHistoryCarrier}_A(h)$. The nil clause is rebuilt from $\hsame$ reflexivity and the given endpoint link. In the cons clause, the source certificate supplies head reflexivity from the head carrier witness, the induction hypothesis supplies tail reflexivity, and the stored $\ContR$ witness and endpoint $\hsame$ link rebuild the cons-cons classifier clause.
For endpoint membership, induct on the displayed generation of $\mathsf{ListHistoryClassifier}^{\mathrm{rec}}_{A,\sim_A}(h,k)$. The nil-nil clause rebuilds two nil carrier witnesses. The cons-cons clause combines the two source carrier witnesses, the induction hypotheses for the two tails, the two $\ContR$ witnesses, and the two endpoint $\hsame$ links to rebuild both cons carrier witnesses.
For symmetry, induct on the classifier generation. The nil-nil case swaps the two endpoint $\hsame$ witnesses. The cons-cons case uses the symmetry field of $\mathsf{cert}$ on the head classifier witness, applies the induction hypothesis to the tail classifier witness, swaps the two $\ContR$ witnesses and endpoint links, and rebuilds the cons-cons clause.
\end{proof}
Transitivity for the certificate classifier is the separate spine-coherent result Theorem~\ref{thm:list-history-classifier-trans-from-spine-coherence}.

\begin{definition}[List pattern specification]
\label{def:list-pattern-specification}
The pattern is nil-or-cons generation with structurally smaller tail.
\end{definition}

\begin{definition}[List history pattern]
\label{def:list-history-pattern}
For a source carrier $S:\mathsf{BHist}\to\mathsf{Prop}$, the predicate
$\mathsf{ListHistoryPattern}_S(h)$ holds when $h$ has the visible BEDC shape of
a list endpoint: either $\hsame(h,\emp)$, or there are histories $a,t,p$ with
$S(a)$, $\mathsf{ListHistoryCarrier}_S(t)$, $\ContR(a,t,p)$, and
$\hsame(h,\Eone(p))$.
\end{definition}
