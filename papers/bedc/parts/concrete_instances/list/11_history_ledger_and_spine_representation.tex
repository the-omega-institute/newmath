\begin{definition}[List ledger policy]
\label{def:list-ledger-policy}
Record the finite constructor spine and source-carrier evidence for each visible element; the paper-side $\mathsf{ListHistoryLedger}$ used in Theorem~\ref{thm:concrete-list-namecert} does not carry a separate source ledger.
\end{definition}

\begin{definition}[List history ledger]
\label{def:list-history-ledger}
For a source carrier $S:\mathsf{BHist}\to\mathsf{Prop}$,
$\mathsf{ListHistoryLedger}_S(h)$ records the visible history endpoint as a
finite nil/cons spine: the empty endpoint is recorded by $\hsame(h,\emp)$, and
a cons endpoint is recorded by histories $a,t,p$ with $S(a)$,
$\mathsf{ListHistoryLedger}_S(t)$, $\ContR(a,t,p)$, and
$\hsame(h,\Eone(p))$.
\end{definition}

\begin{theorem}[List history carrier, pattern, and ledger equivalence]
\label{thm:list-history-carrier-pattern-ledger-equivalence}
For every source carrier $S:\mathsf{BHist}\to\mathsf{Prop}$ and history $h$,
the three displayed list-history fields determine the same endpoints:
$$
  \mathsf{ListHistoryCarrier}_S(h)
  \Longleftrightarrow
  \mathsf{ListHistoryPattern}_S(h),
$$
and
$$
  \mathsf{ListHistoryCarrier}_S(h)
  \Longleftrightarrow
  \mathsf{ListHistoryLedger}_S(h).
$$
\end{theorem}
\begin{proof}
For the carrier-pattern equivalence, use the displayed generated cases for
$\mathsf{ListHistoryCarrier}_S$. The nil case gives $\hsame(h,\emp)$, which is
the nil pattern branch. The cons case exposes histories $a,t,p$ with $S(a)$,
$\mathsf{ListHistoryCarrier}_S(t)$, $\ContR(a,t,p)$, and
$\hsame(h,\Eone(p))$, which is exactly the cons pattern branch. Conversely,
each pattern branch is one of the two carrier constructors, using the same
$\hsame$, $S(a)$, $\ContR(a,t,p)$, and tail carrier witnesses.

For the carrier-ledger equivalence, induct over the displayed nil/cons
generation in each direction. From carrier to ledger, the nil constructor
becomes the nil ledger clause, and the cons constructor reuses $S(a)$,
$\ContR(a,t,p)$, the endpoint witness $\hsame(h,\Eone(p))$, and the induction
hypothesis on the tail. From ledger to carrier, the nil ledger clause rebuilds
the nil carrier clause, and the cons ledger clause reuses the same source,
continuation, endpoint, and tail witnesses, with the induction hypothesis
turning the tail ledger witness into the required tail carrier witness.
\end{proof}

\begin{definition}[List spine representation]\label{def:list-spine-representation}
Let $\mathsf{HistSpine}$ be the finite-spine carrier whose entries are BEDC histories. For a BEDC history source carrier $A:\mathsf{BHist}\to\mathsf{Prop}$, the relation $\mathsf{ListSpineRep}_A(h,xs)$ has $xs:\mathsf{HistSpine}$ and says that $h$ represents $xs$ by the same nil and cons clauses used by $\mathsf{ListHistoryCarrier}_A$; the source predicate is carried by the clauses:
\begin{enumerate}
  \item nil: $xs=\mathsf{nil}$ and $\hsame(h,\emp)$;
  \item cons: $xs=\mathsf{cons}(a,xs')$ and there are histories $t,p$ with $A(a)$, $\mathsf{ListSpineRep}_A(t,xs')$, $\ContR(a,t,p)$, and $\hsame(h,\Eone(p))$.
\end{enumerate}
The spine-bridge classifier used by the certificate is
$$
  \mathsf{ListSpineBridgeClassifier}_{A,\sim_A}(h,k)
  \quad\Longleftrightarrow\quad
  \exists xs,ys:\mathsf{HistSpine}.\;
  \mathsf{ListSpineRep}_A(h,xs)\land
  \mathsf{ListSpineRep}_A(k,ys)\land
  \mathsf{ListClassifierSpec}(\sim_A,xs,ys).
$$
The certificate classifier is this bridge classifier:
$$
  \mathsf{ListSpineHistoryClassifier}_{A,\sim_A}(h,k)
  \quad\Longleftrightarrow\quad
  \mathsf{ListSpineBridgeClassifier}_{A,\sim_A}(h,k).
$$
Thus $\mathsf{ListClassifierSpec}$ remains the finite-spine classifier, $\mathsf{ListSpineBridgeClassifier}_{A,\sim_A}$ names the BEDC-history bridge through finite spines, and $\mathsf{ListSpineHistoryClassifier}_{A,\sim_A}$ is the certificate classifier whose endpoints are BEDC histories. The recursive classifier $\mathsf{ListHistoryClassifier}^{\mathrm{rec}}_{S,\sim_S}$ from Definition~\ref{def:list-history-classifier} is a paper-side generator for carrier and endpoint reasoning.
\end{definition}

\begin{theorem}[List spine nil-cons no-confusion]\label{thm:list-spine-rep-nil-cons-no-confusion}
For every BEDC history source carrier $A:\mathsf{BHist}\to\mathsf{Prop}$, history $h$, source history $a$, and tail spine $xs:\mathsf{HistSpine}$, $\mathsf{ListSpineRep}_A(h,\mathsf{nil})$ and $\mathsf{ListSpineRep}_A(h,\mathsf{cons}(a,xs))$ are contradictory.
\end{theorem}
\begin{proof}
Unpack the nil representation to obtain $\hsame(h,\emp)$. Unpack the cons representation to obtain an endpoint witness $\hsame(h,\Eone(p))$ for some continuation package $p$. Compose the symmetry of the nil endpoint with the cons endpoint to get $\hsame(\emp,\Eone(p))$, then close by BEDC history constructor no-confusion between the empty history and a one-step endpoint.
\end{proof}

\begin{theorem}[List spine representation exactness]
\label{thm:list-spine-representation-exactness}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier and let $\sim_A$ be a source classifier. The spine representation of Definition~\ref{def:list-spine-representation} is exact for the generated list carrier and classifier:
\begin{enumerate}
  \item if $\mathsf{ListSpineRep}_A(h,xs)$ holds, then $\mathsf{ListHistoryCarrier}_A(h)$ holds;
  \item if $\mathsf{ListHistoryCarrier}_A(h)$ holds, then there is a finite spine $xs$ such that $\mathsf{ListSpineRep}_A(h,xs)$ holds;
  \item for all histories $h,k$, the spine-induced classifier $\mathsf{ListSpineBridgeClassifier}_{A,\sim_A}(h,k)$ holds exactly when the recursive generated classifier $\mathsf{ListHistoryClassifier}^{\mathrm{rec}}_{A,\sim_A}(h,k)$ holds.
\end{enumerate}
\end{theorem}
\begin{proof}
For soundness, induct on the displayed derivation of $\mathsf{ListSpineRep}_A(h,xs)$. The nil representation is exactly the nil generator of $\mathsf{ListHistoryCarrier}_A$, and the cons representation carries the head witness $A(a)$, the recursive representation of the tail, the continuation witness $\ContR(a,t,p)$, and the endpoint witness $\hsame(h,\Eone(p))$, so the cons generator rebuilds the carrier.

For completeness, induct on the displayed generation of $\mathsf{ListHistoryCarrier}_A(h)$. The nil carrier chooses $xs=\mathsf{nil}$ and the stored empty endpoint witness. The cons carrier first applies the induction hypothesis to the tail carrier, then chooses $xs=\mathsf{cons}(a,xs')$ and reuses the source, continuation, and endpoint witnesses.

For classifier equivalence, unfold both classifier presentations. A bridge witness exposes finite spines and a $\mathsf{ListClassifierSpec}$ witness; applying carrier soundness to both representation endpoints gives the generated list histories, and induction on the finite-spine classifier derivation rebuilds the nil-nil and cons-cons classifier clauses. Conversely, a generated classifier derivation is handled by induction on its nil/cons clauses: the nil case chooses two nil spines, and the cons case uses the induction hypotheses for the tail classifier together with the stored source head classification and continuation witnesses to construct the finite-spine classifier witness.
\end{proof}

\begin{definition}[List spine representation coherence]
\label{def:list-spine-rep-coherence}
For a BEDC history source carrier $A:\mathsf{BHist}\to\mathsf{Prop}$ and
source classifier $\sim_A$, define
$\mathsf{ListSpineRepCoherent}_{A,\sim_A}$ as the proposition that every two
finite-spine presentations of the same history are classified by the finite
spine classifier. Concretely, for every history $m$ and spines $ys,ys':\mathsf{HistSpine}$,
if $\mathsf{ListSpineRep}_A(m,ys)$ and $\mathsf{ListSpineRep}_A(m,ys')$, then
$\mathsf{ListClassifierSpec}(\sim_A,ys,ys')$.
\end{definition}
