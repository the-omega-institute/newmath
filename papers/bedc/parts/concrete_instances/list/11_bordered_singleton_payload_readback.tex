\section{Bordered singleton payload readback}
\label{sec:list-bordered-singleton-payload-readback}

\begin{theorem}[List classifier bordered singleton payload readback]
\label{thm:list-classifier-bordered-singleton-payload-readback}
For finite BEDC-history lists $p$, $p'$, $s$, and $s'$ and histories $x$ and
$y$, if
$$
\mathsf{ListClassifierSpec}(\hsame,p,p'),
\quad
\mathsf{ListClassifierSpec}(\hsame,s,s'),
\quad
\mathsf{ListClassifierSpec}(\hsame,p{++}(x::\mathsf{nil}){++}s,
p'{++}(y::\mathsf{nil}){++}s')
$$
hold, then $\hsame(x,y)$.
\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:list-classifier-append-two-sided-cancel-classified} to
the displayed prefix classifiers, suffix classifiers, and bordered singleton
classification. This cancels the classified boundaries and gives
$\mathsf{ListClassifierSpec}(\hsame,x::\mathsf{nil},y::\mathsf{nil})$.

Apply Theorem~\ref{thm:list-classifier-hsame-singleton-iff} in the forward
direction. The singleton classifier witness reads back exactly the payload
relation $\hsame(x,y)$.
\end{proof}

\begin{theorem}[List classifier bordered one-headed singleton payload readback]
\label{thm:list-classifier-bordered-e1-singleton-payload-readback}
For finite BEDC-history lists $p$, $p'$, $s$, and $s'$ and histories $x$ and
$y$, if
$$
\mathsf{ListClassifierSpec}(\hsame,p,p'),
\quad
\mathsf{ListClassifierSpec}(\hsame,s,s'),
\quad
\mathsf{ListClassifierSpec}(\hsame,p{++}(\Eone(x)::\mathsf{nil}){++}s,
p'{++}(\Eone(y)::\mathsf{nil}){++}s')
$$
hold, then $\hsame(x,y)$.
\end{theorem}
\begin{proof}
Cancel the classified suffixes from the displayed bordered classifier, then
cancel the classified prefixes. The remaining singleton classifier identifies
$\Eone(x)$ with $\Eone(y)$. One-headed constructor exactness reads back
the payload relation $\hsame(x,y)$.
\end{proof}
