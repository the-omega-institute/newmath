
\begin{theorem}[Framed list-spine coherence]
\label{thm:framed-list-spine-coherence}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier and let
$\sim_A$ be a source classifier. If
$\mathsf{ListSourceHsameCompatible}(A,\sim_A)$ holds, then any two framed
presentations of the same history classify their finite spines:
$$
\forall h\,xs\,ys.\;
\mathsf{FramedListSpineRep}_A(h,xs)\to
\mathsf{FramedListSpineRep}_A(h,ys)\to
\mathsf{ListClassifierSpec}(\sim_A,xs,ys).
$$
\end{theorem}
\begin{proof}
Lemma~\ref{lem:list-pair-frame-injectivity} supplies the required injectivity
of $\mathsf{PairFrame}$. The framed list endpoint separates nil and cons by
the outer $\Ezero/\Eone$ tag, so mixed nil/cons presentations contradict BEDC
history no-confusion. In the nil-nil case the finite-spine classifier is the nil
constructor. In the cons-cons case, injectivity of the outer $\Eone$ endpoint
and then injectivity of $\mathsf{PairFrame}$ expose $\hsame$ between the two
heads and $\hsame$ between the framed tails. The representation hypotheses give
$A$ for both heads, so $\mathsf{ListSourceHsameCompatible}(A,\sim_A)$ supplies
the head classifier. Apply the induction hypothesis to the two framed tail
presentations, and combine the head classifier with the tail classifier by the
cons constructor of $\mathsf{ListClassifierSpec}$.
\end{proof}
\leanchecked{BEDC.Derived.ListUp.FramedListSpineRep\_coherence}

\begin{theorem}[Framed list-spine length determinism]
\label{thm:framed-list-spine-length-determinism}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier and let
$\sim_A$ be a source classifier. If
$\mathsf{ListSourceHsameCompatible}(A,\sim_A)$,
$\mathsf{FramedListSpineRep}_A(h,xs)$, and
$\mathsf{FramedListSpineRep}_A(h,ys)$ hold, then
$$
\ell(xs)=\ell(ys).
$$
The roadmap ten-field fit is: traditional target finite-list framed readback; BEDC source $\mathsf{BHist}$; carrier $\mathsf{FramedListHistoryCarrier}_A$; classifier $\mathsf{ListClassifierSpec}$; stability framed endpoint coherence; ledger finite spine; core theorem framed length determinism; exactness through framed spine representation; standard bridge finite spines; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:framed-list-spine-coherence} to the compatibility
witness and the two displayed framed presentations of $h$. This gives
$\mathsf{ListClassifierSpec}(\sim_A,xs,ys)$. Then apply
Theorem~\ref{thm:list-classifier-length-eq} to read back equality of the two
finite spine lengths.
\end{proof}
\leanchecked{BEDC.Derived.ListUp.FramedListSpineRep\_length\_determinism}

\begin{theorem}[Framed list bridge classifier preserves represented length]
\label{thm:framed-list-bridge-classifier-represented-length}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier and let
$\sim_A$ be a source classifier. For histories $h,k:\mathsf{BHist}$ and finite
spines $xs_0,ys_0$, if $\mathsf{ListSourceHsameCompatible}(A,\sim_A)$,
$\mathsf{FramedListSpineRep}_A(h,xs_0)$,
$\mathsf{FramedListSpineRep}_A(k,ys_0)$, and
$\mathsf{FramedListBridgeClassifier}_{A,\sim_A}(h,k)$ hold, then
$$
\ell(xs_0)=\ell(ys_0).
$$
The roadmap ten-field fit is: traditional target finite-list framed readback along a bridge classifier; BEDC source $\mathsf{BHist}$; carrier $\mathsf{FramedListHistoryCarrier}_A$; classifier $\mathsf{FramedListBridgeClassifier}_{A,\sim_A}$; stability framed endpoint coherence and source classifier length soundness; ledger finite spine representatives at both endpoints; core theorem bridge length preservation; exactness by framed spine representation; standard bridge finite spines; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Unpack the framed bridge classifier witness to finite spines $xs,ys$ with
$\mathsf{FramedListSpineRep}_A(h,xs)$,
$\mathsf{FramedListSpineRep}_A(k,ys)$, and
$\mathsf{ListClassifierSpec}(\sim_A,xs,ys)$. Apply
Theorem~\ref{thm:framed-list-spine-coherence} at $h$ to classify $xs_0$ with
$xs$, and at $k$ to classify $ys$ with $ys_0$. Apply
Theorem~\ref{thm:list-classifier-length-eq} to the three finite-spine classifier
witnesses $xs_0\sim_A xs$, $xs\sim_A ys$, and $ys\sim_A ys_0$, then chain the
three finite-length equalities.
\end{proof}
\leanchecked{BEDC.Derived.ListUp.FramedListBridgeClassifier\_represented\_length}

\begin{definition}[Framed list public length readback]
\label{def:framed-list-public-length-readback}
For a BEDC history source carrier $A:\mathsf{BHist}\to\mathsf{Prop}$, define
$\mathsf{FramedListPublicLength}_A(h,n)$ by
$$
\mathsf{FramedListPublicLength}_A(h,n)
\quad\Longleftrightarrow\quad
\exists xs:\mathsf{HistSpine}.\;
\mathsf{FramedListSpineRep}_A(h,xs)\land \ell(xs)=n.
$$
Here $\ell$ is the finite-spine presentation length read from the framed spine
representation. It is not a separate $\NatUp$ operation and is not an additional
field of the framed semantic certificate.
\end{definition}
\leandef{BEDC.Derived.ListUp.FramedListPublicLength}

\begin{theorem}[Framed list public length readback is well-defined]
\label{thm:framed-list-public-length-readback-well-defined}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier and let
$\sim_A$ be a source classifier. Assume
$\mathsf{ListSourceHsameCompatible}(A,\sim_A)$. Then the framed public length
readback has the following two properties.
\begin{enumerate}
  \item If $\mathsf{FramedListPublicLength}_A(h,n)$ and
  $\mathsf{FramedListPublicLength}_A(h,m)$, then $n=m$.
  \item If $\mathsf{FramedListBridgeClassifier}_{A,\sim_A}(h,k)$ and
  $\mathsf{FramedListPublicLength}_A(h,n)$, then
  $\mathsf{FramedListPublicLength}_A(k,n)$.
\end{enumerate}
The roadmap ten-field fit is: traditional target finite-list framed length
readback; BEDC source $\mathsf{BHist}$; carrier
$\mathsf{FramedListHistoryCarrier}_A$; classifier
$\mathsf{FramedListBridgeClassifier}_{A,\sim_A}$; stability obligation framed
endpoint coherence; ledger finite-spine length witness; core theorem
well-definedness and classifier preservation; exactness via
$\mathsf{FramedListSpineRep}$; standard bridge finite spines; status
$\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
For uniqueness at one endpoint, unpack the two
$\mathsf{FramedListPublicLength}_A$ witnesses. They give finite spines $xs$ and
$ys$, representations $\mathsf{FramedListSpineRep}_A(h,xs)$ and
$\mathsf{FramedListSpineRep}_A(h,ys)$, and displayed equalities
$\ell(xs)=n$ and $\ell(ys)=m$. Apply
Theorem~\ref{thm:framed-list-spine-length-determinism} to the two framed
representations of $h$, obtaining $\ell(xs)=\ell(ys)$. Chain the inverse of
$\ell(xs)=n$, this equality, and $\ell(ys)=m$ to get $n=m$.

For preservation along the framed bridge classifier, unpack the public length
witness for $h$ as a finite spine $xs_0$ with
$\mathsf{FramedListSpineRep}_A(h,xs_0)$ and $\ell(xs_0)=n$. Unpack the bridge
classifier into finite spines $xs,ys$ with
$\mathsf{FramedListSpineRep}_A(h,xs)$,
$\mathsf{FramedListSpineRep}_A(k,ys)$, and
$\mathsf{ListClassifierSpec}(\sim_A,xs,ys)$.

Use Theorem~\ref{thm:framed-list-bridge-classifier-represented-length} with the
displayed representative $xs_0$ at $h$, the bridge representative $ys$ at $k$,
and the bridge classifier witness to obtain $\ell(xs_0)=\ell(ys)$. Chain this
length equality with $\ell(xs_0)=n$, then choose $ys$ as the framed public length
witness for $k$.
\end{proof}
\leanchecked{BEDC.Derived.ListUp.FramedListPublicLength\_well\_defined}

\begin{theorem}[Framed list bridge classifier has common public length]
\label{thm:list-framed-bridge-classifier-public-length-total}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier and let
$\sim_A$ be a source classifier. If $\mathsf{cert}:\NameCert(A,\sim_A)$,
$\mathsf{ListSourceHsameCompatible}(A,\sim_A)$, and
$\mathsf{FramedListBridgeClassifier}_{A,\sim_A}(h,k)$ hold, then $h$ and $k$
are framed list carrier histories and share a framed public length:
$$
\mathsf{FramedListHistoryCarrier}_A(h)\land
\mathsf{FramedListHistoryCarrier}_A(k)
\land
\exists n.\;
\mathsf{FramedListPublicLength}_A(h,n)\land
\mathsf{FramedListPublicLength}_A(k,n).
$$
\end{theorem}
\begin{proof}
Use the equivalence-field theorem for the framed bridge classifier to obtain
the two carrier endpoints. Unpack the bridge witness into displayed finite
spines $xs,ys$ and a finite-spine classifier
$\mathsf{ListClassifierSpec}(\sim_A,xs,ys)$. The list classifier length theorem
gives $\ell(xs)=\ell(ys)$, so $\ell(xs)$ is a shared public length for the two
displayed framed spine representatives.
\end{proof}
\leanchecked{BEDC.Derived.ListUp.FramedListBridgeClassifier\_public\_length\_total}

\begin{theorem}[Framed list public length constructor readback]
\label{thm:framed-list-public-length-constructor-readback}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier. For
every history $h$ and every finite length value $n$, the framed public length
readback exposes the finite-spine constructors:
\begin{enumerate}
  \item $\mathsf{FramedListPublicLength}_A(h,0)$ holds if and only if
  $\mathsf{FramedListSpineRep}_A(h,\mathsf{nil})$ holds.
  \item If $\mathsf{FramedListPublicLength}_A(h,n+1)$ holds, then
  $$
  \exists a:\mathsf{BHist},\,xs:\mathsf{HistSpine}.\;
  A(a)\land
  \mathsf{FramedListSpineRep}_A(h,\mathsf{cons}(a,xs))\land
  \mathsf{FramedListPublicLength}_A(\mathsf{FramedListEndpoint}(xs),n).
  $$
\end{enumerate}
The roadmap ten-field fit is: traditional target framed finite-list constructor
readback; BEDC source $\mathsf{BHist}$; carrier
$\mathsf{FramedListHistoryCarrier}_A$; classifier-independent spine exposure;
ledger $\mathsf{FramedListPublicLength}_A$; core theorem nil and successor
public-length inversion; exactness by finite framed-spine constructor analysis;
standard bridge finite spines; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
For the zero forward direction, unfold
Definition~\ref{def:framed-list-public-length-readback}. A public length witness
with value $0$ supplies a finite spine $xs$ with
$\mathsf{FramedListSpineRep}_A(h,xs)$ and $\ell(xs)=0$. Case-analyze $xs$. The
cons case has successor length and cannot have value $0$, while the nil case is
the required framed nil representation.

For the zero reverse direction, choose the nil finite spine in
Definition~\ref{def:framed-list-public-length-readback}. The displayed
$\mathsf{FramedListSpineRep}_A(h,\mathsf{nil})$ witness and
$\ell(\mathsf{nil})=0$ repack as
$\mathsf{FramedListPublicLength}_A(h,0)$.

For the successor clause, unfold the same definition. A witness for
$\mathsf{FramedListPublicLength}_A(h,n+1)$ gives a finite spine $xs_0$ with
$\mathsf{FramedListSpineRep}_A(h,xs_0)$ and $\ell(xs_0)=n+1$. The nil case would
force $\ell(\mathsf{nil})=n+1$, impossible for finite lengths, so the exposed
spine has the form $\mathsf{cons}(a,xs)$.

The cons representation already supplies $A(a)$ and
$\mathsf{FramedListSpineRep}_A(h,\mathsf{cons}(a,xs))$. Its entry predicate
restricts to the tail entries of $xs$, and reflexivity gives
$\hsame(\mathsf{FramedListEndpoint}(xs),\mathsf{FramedListEndpoint}(xs))$.
Together these build
$\mathsf{FramedListSpineRep}_A(\mathsf{FramedListEndpoint}(xs),xs)$.

The displayed successor length equality identifies $\ell(xs)$ with $n$.
Repacking the tail representation and this predecessor length through
Definition~\ref{def:framed-list-public-length-readback} gives
$\mathsf{FramedListPublicLength}_A(\mathsf{FramedListEndpoint}(xs),n)$, so $a$
and $xs$ are the required constructor witnesses.
\end{proof}
\leanchecked{BEDC.Derived.ListUp.FramedListPublicLength\_constructor\_endpoint\_readback}

\begin{theorem}[Framed list public length successor empty absurd]
\label{thm:framed-list-public-length-successor-empty-absurd}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier. If a
history $h$ has successor framed public length, then $h$ cannot also be the
empty framed endpoint:
$$
\mathsf{FramedListPublicLength}_A(h,n+1)
\Longrightarrow
\hsame(h,\mathsf{FramedListEndpoint}(\mathsf{nil}))
\Longrightarrow
\mathsf{False}.
$$
\end{theorem}
\begin{proof}
Apply constructor readback to the successor public-length witness. It exposes
a head $a$ and tail spine $xs$ with a displayed representation of $h$ by the
cons endpoint $\mathsf{FramedListEndpoint}(\mathsf{cons}(a,xs))$. Composing
this endpoint display with the assumed empty-frame display gives an internal
sameness between a cons framed endpoint and the nil framed endpoint. Framed
endpoint no-confusion for nil versus cons eliminates this case.
\end{proof}
\leanchecked{BEDC.Derived.ListUp.FramedListPublicLength\_successor\_empty\_absurd}
