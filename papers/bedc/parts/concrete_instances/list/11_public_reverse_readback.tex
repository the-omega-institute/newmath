\section{Public reverse readback}
\label{sec:list-public-reverse-readback}

\begin{definition}[List public reverse readback]
\label{def:list-public-reverse-readback}
For a BEDC history source carrier $A:\mathsf{BHist}\to\mathsf{Prop}$, define
$\mathsf{ListPublicReverse}_A(h,k)$ to mean that there exists a finite
BEDC-history spine $xs:\mathsf{HistSpine}$ such that
$$
\mathsf{ListSpineRep}_A(h,xs)
$$
and
$$
\mathsf{ListSpineRep}_A(k,\mathsf{reverse}(xs))
$$
hold.
\end{definition}

\begin{theorem}[List public reverse readback is classifier-functional and involutive]
\label{thm:list-public-reverse-readback-functional-involutive}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier, let
$\sim_A$ be a source classifier, let $\mathsf{cert}:\NameCert(A,\sim_A)$, and
assume $\mathsf{ListSpineRepCoherent}_{A,\sim_A}$. For histories
$h,h',k,k':\mathsf{BHist}$, the public reverse readback relation satisfies:
\begin{enumerate}
  \item if $\mathsf{ListPublicReverse}_A(h,k)$ holds, then
  $\mathsf{ListHistoryCarrier}_A(h)\land\mathsf{ListHistoryCarrier}_A(k)$;
  \item if $\mathsf{ListPublicReverse}_A(h,k)$ holds, then
  $\mathsf{ListPublicReverse}_A(k,h)$;
  \item if $\mathsf{ListPublicReverse}_A(h,k)$ and
  $\mathsf{ListPublicReverse}_A(h,k')$ hold, then
  $\mathsf{ListSpineHistoryClassifier}_{A,\sim_A}(k,k')$;
  \item if $\mathsf{ListSpineHistoryClassifier}_{A,\sim_A}(h,h')$,
  $\mathsf{ListPublicReverse}_A(h,k)$, and
  $\mathsf{ListPublicReverse}_A(h',k')$ hold, then
  $\mathsf{ListSpineHistoryClassifier}_{A,\sim_A}(k,k')$.
\end{enumerate}
\end{theorem}
\begin{proof}
For carrier soundness, unpack the reverse witness as a spine $xs$ with
$\mathsf{ListSpineRep}_A(h,xs)$ and
$\mathsf{ListSpineRep}_A(k,\mathsf{reverse}(xs))$. Apply
Theorem~\ref{thm:list-spine-representation-exactness} to each displayed
representation to obtain the two public carrier witnesses.

For symmetry, keep the displayed representation of $k$ by
$\mathsf{reverse}(xs)$. Finite-spine reverse involutivity identifies
$\mathsf{reverse}(\mathsf{reverse}(xs))$ with $xs$, so the displayed
representation of $h$ becomes the required reversed-spine representation from
$k$ back to $h$.

For classifier functionality over a fixed source endpoint, unpack the two
reverse witnesses as spines $xs$ and $ys$. Coherence for the two presentations
of $h$ gives $\mathsf{ListClassifierSpec}(\sim_A,xs,ys)$. Apply
Theorem~\ref{thm:list-classifier-reverse-iff} to classify
$\mathsf{reverse}(xs)$ with $\mathsf{reverse}(ys)$, then repack this
finite-spine classifier with the two displayed representations of $k$ and $k'$
by the reverse direction of
Theorem~\ref{thm:list-represented-spine-classifier-displayed-exactness}.

For classifier stability along classified source endpoints, unpack the two
reverse witnesses as spines $xs$ and $ys$. The forward direction of
Theorem~\ref{thm:list-represented-spine-classifier-displayed-exactness} reads
the public classifier between $h$ and $h'$ as
$$
\mathsf{ListClassifierSpec}(\sim_A,xs,ys).
$$

Apply Theorem~\ref{thm:list-classifier-reverse-iff} to obtain a classifier
between $\mathsf{reverse}(xs)$ and $\mathsf{reverse}(ys)$. Repack that
finite-spine classifier with the reverse-readback representations of $k$ and
$k'$ by the reverse direction of
Theorem~\ref{thm:list-represented-spine-classifier-displayed-exactness}.
\end{proof}

\begin{theorem}[List public reverse preserves public length]
\label{thm:list-public-reverse-length-preservation}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier, let
$\sim_A$ be a source classifier, let $\mathsf{cert}:\NameCert(A,\sim_A)$, and
assume $\mathsf{ListSpineRepCoherent}_{A,\sim_A}$. For histories
$h,k:\mathsf{BHist}$ and a finite length value $n$, the public reverse readback
relation preserves the public length ledger in both directions:
\begin{enumerate}
  \item if $\mathsf{ListPublicReverse}_A(h,k)$ and
  $\mathsf{ListPublicLength}_A(h,n)$ hold, then
  $\mathsf{ListPublicLength}_A(k,n)$;
  \item if $\mathsf{ListPublicReverse}_A(h,k)$ and
  $\mathsf{ListPublicLength}_A(k,n)$ hold, then
  $\mathsf{ListPublicLength}_A(h,n)$.
\end{enumerate}
\end{theorem}
\begin{proof}
For the forward direction, unpack the reverse witness as a spine $xs$ with
$\mathsf{ListSpineRep}_A(h,xs)$ and
$\mathsf{ListSpineRep}_A(k,\mathsf{reverse}(xs))$. Unpack the public length
witness for $h$ as a spine $ys$ with $\mathsf{ListSpineRep}_A(h,ys)$ and
$\ell(ys)=n$.

Spine coherence compares the two displayed representations of $h$, giving
$\mathsf{ListClassifierSpec}(\sim_A,xs,ys)$. The length exactness theorem for
classified finite spines identifies $\ell(xs)$ with $\ell(ys)$, and finite-spine
reverse length preservation identifies $\ell(\mathsf{reverse}(xs))$ with
$\ell(xs)$. Chaining these equalities with $\ell(ys)=n$ gives
$\ell(\mathsf{reverse}(xs))=n$.

Repack the displayed representation
$\mathsf{ListSpineRep}_A(k,\mathsf{reverse}(xs))$ together with the resulting
length equality through Definition~\ref{def:list-public-length-readback}. This
is $\mathsf{ListPublicLength}_A(k,n)$.

For the reverse direction, use the involutive clause of
Theorem~\ref{thm:list-public-reverse-readback-functional-involutive} to turn
$\mathsf{ListPublicReverse}_A(h,k)$ into $\mathsf{ListPublicReverse}_A(k,h)$.
Apply the forward direction just proved to this reversed public-reverse witness
and the displayed public length witness for $k$, obtaining
$\mathsf{ListPublicLength}_A(h,n)$.
\end{proof}

\begin{theorem}[List public reverse fixed-length classifier automorphism]
\label{thm:list-public-reverse-fixed-length-classifier-automorphism}
The roadmap ten-field fit is: traditional target finite-list reverse as a
fixed-length classifier automorphism; BEDC source $\mathsf{BHist}$ finite
spines; carrier $\mathsf{ListHistoryCarrier}_A$; classifier
$\mathsf{ListSpineHistoryClassifier}_{A,\sim_A}$; stability from spine
coherence and the source $\NameCert$; ledger finite spine reversal plus public
length; core theorem this fixed-length automorphism package; exactness through
reverse readback and length transport; standard bridge finite spines; status
$\mathsf{ProofTarget}$.

Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier, let
$\sim_A$ be a source classifier, let $\mathsf{cert}:\NameCert(A,\sim_A)$, and
assume $\mathsf{ListSpineRepCoherent}_{A,\sim_A}$. For histories
$h,h',k,k':\mathsf{BHist}$ and a finite length value $n$, if
$$
\mathsf{ListSpineHistoryClassifier}_{A,\sim_A}(h,h')\land
\mathsf{ListPublicReverse}_A(h,k)\land
\mathsf{ListPublicReverse}_A(h',k')\land
\mathsf{ListPublicLength}_A(h,n)
$$
holds, then
$$
\mathsf{ListPublicLength}_A(k,n)\land
\mathsf{ListPublicLength}_A(k',n)\land
\mathsf{ListSpineHistoryClassifier}_{A,\sim_A}(k,k')\land
\mathsf{ListPublicReverse}_A(k,h)\land
\mathsf{ListPublicReverse}_A(k',h')
$$
holds.
\end{theorem}
\begin{proof}
First put both classified source endpoints on the displayed public-length
fiber. Apply Theorem~\ref{thm:list-public-length-classifier-transport} to the
classifier witness between $h$ and $h'$ and the public length witness for $h$.
This gives $\mathsf{ListPublicLength}_A(h',n)$.

Next transport the length through the two public reverse witnesses. Apply
Theorem~\ref{thm:list-public-reverse-length-preservation} to
$\mathsf{ListPublicReverse}_A(h,k)$ and $\mathsf{ListPublicLength}_A(h,n)$ to
obtain $\mathsf{ListPublicLength}_A(k,n)$. Apply the same theorem to
$\mathsf{ListPublicReverse}_A(h',k')$ and the transported length witness for
$h'$ to obtain $\mathsf{ListPublicLength}_A(k',n)$.

Now classify the reversed endpoints. The classifier-stability clause of
Theorem~\ref{thm:list-public-reverse-readback-functional-involutive}, applied
to the source classifier witness and the two reverse readback witnesses,
produces $\mathsf{ListSpineHistoryClassifier}_{A,\sim_A}(k,k')$.

Finally, reverse back along each public reverse witness. The involutive clause
of Theorem~\ref{thm:list-public-reverse-readback-functional-involutive} sends
$\mathsf{ListPublicReverse}_A(h,k)$ to $\mathsf{ListPublicReverse}_A(k,h)$ and
sends $\mathsf{ListPublicReverse}_A(h',k')$ to
$\mathsf{ListPublicReverse}_A(k',h')$. Combining the two length witnesses, the
reversed-endpoint classifier, and these two reverse-back witnesses gives the
claimed conjunction.
\end{proof}
