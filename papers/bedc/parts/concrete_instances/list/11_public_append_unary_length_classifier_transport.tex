\begin{theorem}[List public append unary length classifier transport]
\label{thm:list-public-append-unary-length-classifier-transport}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier, let
$\sim_A$ be a source classifier, and let
$\mathsf{cert}:\NameCert(A,\sim_A)$. Assume
$\mathsf{ListSpineRepCoherent}_{A,\sim_A}$. For histories
$h,h',k,k',r,r':\mathsf{BHist}$, suppose
$$
\mathsf{ListPublicAppend}_A(h,k,r)
$$
holds, and suppose also that
$$
\mathsf{ListSpineHistoryClassifier}_{A,\sim_A}(h,h'),\qquad
\mathsf{ListSpineHistoryClassifier}_{A,\sim_A}(k,k'),\qquad
\mathsf{ListSpineHistoryClassifier}_{A,\sim_A}(r,r')
$$
hold. Then there exist unary length histories $u,v,w:\mathsf{BHist}$ such that
\begin{enumerate}
  \item $\mathsf{ListPublicUnaryLength}_A(h,u)$ and
  $\mathsf{ListPublicUnaryLength}_A(h',u)$ hold;
  \item $\mathsf{ListPublicUnaryLength}_A(k,v)$ and
  $\mathsf{ListPublicUnaryLength}_A(k',v)$ hold;
  \item $\mathsf{ListPublicUnaryLength}_A(r,w)$ and
  $\mathsf{ListPublicUnaryLength}_A(r',w)$ hold;
  \item
  $$
  \mathsf{NatAdd}(v,u,w);
  $$
  \item for every $z:\mathsf{BHist}$,
  $$
  \mathsf{NatAdd}(u,v,z)\to\hsame(w,z).
  $$
\end{enumerate}
\end{theorem}
\leanchecked{BEDC.Derived.ListUp.FramedListPublicAppend\_public\_length\_classifier\_transport}
\begin{proof}
Apply Theorem~\ref{thm:list-public-append-unary-length-additivity} to the
displayed public append witness. This gives histories $u,v,w$ with public unary
length witnesses for $h,k,r$, together with
$\mathsf{NatAdd}(v,u,w)$ and the commuted uniqueness row
$\mathsf{NatAdd}(u,v,z)\to\hsame(w,z)$ for arbitrary $z$.

Next apply Theorem~\ref{thm:list-public-unary-length-classifier-transport}
three times, each time using $\mathsf{cert}$ and
$\mathsf{ListSpineRepCoherent}_{A,\sim_A}$. The classifier from $h$ to $h'$
transports the public unary length witness at $h$ to one at $h'$ with the same
token $u$; the classifier from $k$ to $k'$ does the same for $v$; and the
classifier from $r$ to $r'$ does the same for $w$.

Finally, repack the original witnesses for $h,k,r$, the transported witnesses
for $h',k',r'$, and the two Nat-addition rows obtained from append additivity.
No additional computation is needed, since classifier transport preserves the
unary token rather than replacing it.
\end{proof}
