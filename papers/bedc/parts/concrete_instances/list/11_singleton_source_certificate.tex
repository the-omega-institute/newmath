\begin{definition}[Singleton history source for lists]
\label{def:list-singleton-history-source}
For a fixed BEDC history $a_0:\mathsf{BHist}$, define the singleton history
source carrier $A_{a_0}:\mathsf{BHist}\to\mathsf{Prop}$ by
$$
A_{a_0}(x)\Longleftrightarrow\hsame(x,a_0).
$$
\end{definition}

\begin{theorem}[List singleton source framed carrier certificate]
\label{thm:list-singleton-source-framed-carrier-certificate}
Fix a BEDC history $a_0:\mathsf{BHist}$ and let $A_{a_0}$ be the singleton
history source of Definition~\ref{def:list-singleton-history-source}. The
ten-field per-object fit is: traditional target framed carrier certificate for
finite lists over one BEDC history class; BEDC source $\mathsf{BHist}$ with
$\hsame$; carrier
$\mathsf{FramedListHistoryCarrier}_{A_{a_0}}$; classifier
$\mathsf{FramedListBridgeClassifier}_{A_{a_0},\hsame}$; stability from
$\mathsf{ListSourceHsameCompatible}(A_{a_0},\hsame)$ and framed endpoint
coherence; ledger finite framed spine; core theorem
Corollary~\ref{cor:framed-list-carrier-certificate}; exactness
through framed spine representation; standard bridge finite spines; status
$\mathsf{ProofTarget}$. The framed semantic certificate
$$
\mathsf{SemanticNameCert}(\mathsf{FramedListHistoryCarrier}_{A_{a_0}},
\mathsf{FramedListHistoryCarrier}_{A_{a_0}},
\mathsf{FramedListHistoryCarrier}_{A_{a_0}},
\mathsf{FramedListBridgeClassifier}_{A_{a_0},\hsame})
$$
hold.
\end{theorem}
\begin{proof}
Theorem~\ref{thm:singleton-history-source-namecert} supplies the source
certificate $\NameCert(A_{a_0},\hsame)$. The compatibility predicate
$\mathsf{ListSourceHsameCompatible}(A_{a_0},\hsame)$ is immediate: the final displayed input is already the required $\hsame(a,b)$ witness. Apply
Corollary~\ref{cor:framed-list-carrier-certificate} with the singleton source
certificate and this compatibility witness.
\end{proof}

\begin{theorem}[List singleton source spine coherence]
\label{thm:list-singleton-source-spine-coherence}
For any BEDC history $a_0$, let $A_{a_0}$ be the singleton history source of
Definition~\ref{def:list-singleton-history-source}. Then
$\mathsf{ListSpineRepCoherent}_{A_{a_0},\hsame}$ holds.
\end{theorem}
\begin{proof}
Compare two singleton-source spine presentations of the same endpoint by cases
on the displayed spines. The mixed nil/cons cases contradict
Theorem~\ref{thm:list-spine-rep-nil-cons-no-confusion}; the nil/nil case is the nil finite-spine classifier. In the cons/cons case, write the two displayed
outer endpoints as $\Eone(p)$ and $\Eone(p')$. The common endpoint witness gives
$\hsame(\Eone(p),\Eone(p'))$; Theorem~\ref{thm:history-constructor-characterization}
inverts it to $\hsame(p,p')$. The source facts $A_{a_0}(a)$ and $A_{a_0}(a')$
give $\hsame(a,a_0)$ and $\hsame(a',a_0)$, hence $\hsame(a,a')$ by symmetry and
transitivity. Apply Theorem~\ref{thm:cont-cancel-hsame-left-context} to
$\ContR(a,t,p)$, $\ContR(a',t',p')$, the head sameness, and the result sameness;
this yields $\hsame(t,t')$. Transport the second tail representation along
Lemma~\ref{lem:list-spine-representation-endpoint-transport}, apply the
induction hypothesis to the two tail presentations at the aligned endpoint, and
rebuild the $\mathsf{ListClassifierSpec}(\hsame,-,-)$ cons witness.
\end{proof}

\begin{definition}[Hsame-bounded list source]
\label{def:list-hsame-bounded-source}
For a source predicate $A:\mathsf{BHist}\to\mathsf{Prop}$ and a center
$a_0:\mathsf{BHist}$, define
$\mathsf{ListHsameBoundedSource}(A,a_0)$ by
$$
\mathsf{ListHsameBoundedSource}(A,a_0)
\Longleftrightarrow
\forall a.\; A(a)\to\hsame(a,a_0).
$$
\end{definition}

\begin{theorem}[Hsame-bounded list source spine coherence]
\label{thm:list-hsame-bounded-source-spine-coherence}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ and $a_0:\mathsf{BHist}$. If
$\mathsf{ListHsameBoundedSource}(A,a_0)$ holds, then
$\mathsf{ListSpineRepCoherent}_{A,\hsame}$ holds.
\end{theorem}
\begin{proof}
Compare two spine presentations of a common endpoint by cases on the two
spines. The nil/nil case is classified by the nil finite-spine witness. The
mixed nil/cons and cons/nil cases contradict
Theorem~\ref{thm:list-spine-rep-nil-cons-no-confusion}. In the cons/cons case,
write the two displayed outer endpoints as $\Eone(p)$ and $\Eone(p')$. The
common endpoint witness gives $\hsame(\Eone(p),\Eone(p'))$, and
Theorem~\ref{thm:history-constructor-characterization} inverts it to
$\hsame(p,p')$. The source facts $A(a)$ and $A(a')$, together with
$\mathsf{ListHsameBoundedSource}(A,a_0)$, give $\hsame(a,a_0)$ and
$\hsame(a',a_0)$, hence $\hsame(a,a')$ by symmetry and transitivity. Apply
Theorem~\ref{thm:cont-cancel-hsame-left-context} to $\ContR(a,t,p)$,
$\ContR(a',t',p')$, the head sameness, and the result sameness to obtain
$\hsame(t,t')$. Transport the second tail representation along
Lemma~\ref{lem:list-spine-representation-endpoint-transport}, apply the
induction hypothesis to the aligned tail presentations, and rebuild the cons
witness for $\mathsf{ListClassifierSpec}(\hsame,-,-)$.
\end{proof}

\begin{corollary}[Hsame-bounded list source unframed certificate]
\label{thm:list-hsame-bounded-source-public-object-certificate}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ and $a_0:\mathsf{BHist}$. If
$\mathsf{cert}:\NameCert(A,\hsame)$ and
$\mathsf{ListHsameBoundedSource}(A,a_0)$, then the ten-field per-object fit is:
traditional target finite lists over one internal history class; BEDC source
$\mathsf{BHist}$ with $\hsame$; carrier $\mathsf{ListHistoryCarrier}_A$;
classifier $\mathsf{ListSpineHistoryClassifier}_{A,\hsame}$; stability from the
source $\NameCert(A,\hsame)$ and
Theorem~\ref{thm:list-hsame-bounded-source-spine-coherence}; ledger
$\mathsf{ListHistoryLedger}_A$; core theorem list
Theorem~\ref{thm:list-hsame-bounded-source-spine-coherence} and
Theorem~\ref{thm:concrete-list-namecert}; exactness through the spine bridge;
standard bridge finite spines; status $\mathsf{ProofTarget}$. Hence
$$
\mathsf{SemanticNameCert}(\mathsf{ListHistoryCarrier}_A,
\mathsf{ListHistoryPattern}_A,
\mathsf{ListHistoryLedger}_A,
\mathsf{ListSpineHistoryClassifier}_{A,\hsame})
$$
holds.
\end{corollary}
\begin{proof}
The displayed certificate $\mathsf{cert}$ supplies the source
$\NameCert(A,\hsame)$ input. The bounded-source hypothesis supplies
$\mathsf{ListSpineRepCoherent}_{A,\hsame}$ by
Theorem~\ref{thm:list-hsame-bounded-source-spine-coherence}. Apply
Theorem~\ref{thm:concrete-list-namecert} with these two inputs.
\end{proof}

\begin{theorem}[Hsame-bounded list source endpoint spine ledgers]
\label{thm:list-hsame-bounded-source-public-object-endpoint-spine-ledgers}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ and $a_0:\mathsf{BHist}$. If
$\mathsf{cert}:\NameCert(A,\hsame)$ and
$\mathsf{ListHsameBoundedSource}(A,a_0)$, then the displayed public-object
semantic certificate
$$
\mathsf{SemanticNameCert}(\mathsf{ListHistoryCarrier}_A,
\mathsf{ListHistoryPattern}_A,
\mathsf{ListHistoryLedger}_A,
\mathsf{ListSpineHistoryClassifier}_{A,\hsame})
$$
holds. Moreover, for all histories $h$ and $k$, every classifier witness
$$
\mathsf{ListSpineHistoryClassifier}_{A,\hsame}(h,k)
$$
yields finite spines $xs,ys:\mathsf{HistSpine}$ with
$$
\mathsf{ListSpineRep}_A(h,xs)
$$
and
$$
\mathsf{ListSpineRep}_A(k,ys).
$$
Consequently,
$$
\mathsf{ListHistoryCarrier}_A(h)\land\mathsf{ListHistoryCarrier}_A(k)
$$
holds. The per-object fit is: traditional target finite lists over one
hsame-bounded BEDC history source; BEDC source $\mathsf{BHist}$ with $\hsame$;
carrier $\mathsf{ListHistoryCarrier}_A$; classifier
$\mathsf{ListSpineHistoryClassifier}_{A,\hsame}$; ledger finite-spine endpoint
exposure; core route Theorem~\ref{thm:list-hsame-bounded-source-spine-coherence},
Theorem~\ref{thm:list-spine-coherence-iff-cons-boundary-coherence}, and
Theorem~\ref{thm:list-public-object-endpoint-spine-ledgers}; standard bridge
finite spines; status Phase Finite Data.
\end{theorem}
\begin{proof}
Theorem~\ref{thm:list-hsame-bounded-source-spine-coherence} turns the
hsame-bounded source hypothesis into
$\mathsf{ListSpineRepCoherent}_{A,\hsame}$. The forward implication of
Theorem~\ref{thm:list-spine-coherence-iff-cons-boundary-coherence} converts this
spine coherence datum into
$\mathsf{ListConsBoundaryCoherent}_{A,\hsame}$. Apply
Theorem~\ref{thm:list-public-object-endpoint-spine-ledgers} to the source
certificate $\mathsf{cert}$ and this cons-boundary witness. Its certificate
component is exactly the displayed
$\mathsf{SemanticNameCert}(\mathsf{ListHistoryCarrier}_A,
\mathsf{ListHistoryPattern}_A,\mathsf{ListHistoryLedger}_A,
\mathsf{ListSpineHistoryClassifier}_{A,\hsame})$, and its endpoint-ledger
component supplies the finite spines $xs$ and $ys$ with the two displayed
$\mathsf{ListSpineRep}_A$ facts. The final carrier conjunction follows from
Theorem~\ref{thm:list-spine-representation-coverage}.
\end{proof}
