\section{Singleton source length shape exhaustion}
\label{sec:list-singleton-source-length-shape-exhaustion}

\begin{theorem}[List singleton source public length shape exhaustion]
\label{thm:list-singleton-source-public-length-shape-exhaustion}
Fix a BEDC history $a_0:\mathsf{BHist}$ and let $A_{a_0}$ be the singleton
history source of Definition~\ref{def:list-singleton-history-source}. For every
history $h$, if $\mathsf{ListHistoryCarrier}_{A_{a_0}}(h)$ holds, then one of
the following two disjoint public length shapes is available:
\begin{enumerate}
  \item $\mathsf{ListPublicLength}_{A_{a_0}}(h,0)\land \hsame(h,\emp)$.
  \item There exist a finite length value $n$ and histories
  $a,t,p:\mathsf{BHist}$ such that
  $$
  \left(
  \begin{array}{l}
  \mathsf{ListPublicLength}_{A_{a_0}}(h,n+1)\land
  A_{a_0}(a)\land\\
  \mathsf{ListHistoryCarrier}_{A_{a_0}}(t)\land
  \mathsf{ListPublicLength}_{A_{a_0}}(t,n)\land\\
  \ContR(a,t,p)\land \hsame(h,\Eone(p))
  \end{array}
  \right).
  $$
\end{enumerate}
The nil alternative and every displayed cons alternative are disjoint by BEDC
history no-confusion between $\emp$ and $\Eone(p)$.
\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:list-singleton-source-finite-data-certificate-package}
at the fixed singleton source. Its totality clause gives a finite length value
$m$ with $\mathsf{ListPublicLength}_{A_{a_0}}(h,m)$.

Case-analyze the finite length value $m$.

In the zero case, the zero readback clause of
Theorem~\ref{thm:list-singleton-source-finite-data-certificate-package}
identifies $\mathsf{ListPublicLength}_{A_{a_0}}(h,0)$ with
$\hsame(h,\emp)$. Pair the given zero public length witness with this endpoint
witness to obtain the nil alternative.

In the successor case, write $m=n+1$. The successor exactness clause of
Theorem~\ref{thm:list-singleton-source-finite-data-certificate-package}
turns $\mathsf{ListPublicLength}_{A_{a_0}}(h,n+1)$ into histories
$a,t,p:\mathsf{BHist}$ with $A_{a_0}(a)$,
$\mathsf{ListHistoryCarrier}_{A_{a_0}}(t)$,
$\mathsf{ListPublicLength}_{A_{a_0}}(t,n)$, $\ContR(a,t,p)$, and
$\hsame(h,\Eone(p))$. These data are exactly the displayed cons alternative.

For disjointness, suppose the nil alternative and one displayed cons
alternative both hold. Compose symmetry of the nil endpoint witness
$\hsame(h,\emp)$ with the cons endpoint witness $\hsame(h,\Eone(p))$ to obtain
$\hsame(\emp,\Eone(p))$. The BEDC history no-confusion theorem,
Theorem~\ref{thm:history-no-confusion}, excludes this empty-versus-one
identification.
\end{proof}

\begin{theorem}[List singleton source classified length-shape synchronization]
\label{thm:list-singleton-source-classified-length-shape-synchronization}
Fix a BEDC history $a_0:\mathsf{BHist}$ and let $A_{a_0}$ be the singleton
history source of Definition~\ref{def:list-singleton-history-source}. For all
histories $h,k:\mathsf{BHist}$, if
$\mathsf{ListHistoryCarrier}_{A_{a_0}}(h)$ and
$\mathsf{ListSpineHistoryClassifier}_{A_{a_0},\hsame}(h,k)$ hold, then one of
the following two disjoint synchronized public endpoint shapes is available:
\begin{enumerate}
  \item
  $$
  \mathsf{ListPublicLength}_{A_{a_0}}(h,0)\land
  \mathsf{ListPublicLength}_{A_{a_0}}(k,0)\land
  \hsame(h,\emp)\land \hsame(k,\emp).
  $$
  \item There exist a finite length value $n$ and histories
  $a,b,t,u,p,q:\mathsf{BHist}$ such that
  $$
  \left(
  \begin{array}{l}
  \mathsf{ListPublicLength}_{A_{a_0}}(h,n+1)\land
  \mathsf{ListPublicLength}_{A_{a_0}}(k,n+1)\land\\
  A_{a_0}(a)\land A_{a_0}(b)\land
  \mathsf{ListHistoryCarrier}_{A_{a_0}}(t)\land
  \mathsf{ListHistoryCarrier}_{A_{a_0}}(u)\land\\
  \mathsf{ListPublicLength}_{A_{a_0}}(t,n)\land
  \mathsf{ListPublicLength}_{A_{a_0}}(u,n)\land\\
  \ContR(a,t,p)\land \ContR(b,u,q)\land
  \hsame(h,\Eone(p))\land \hsame(k,\Eone(q))
  \end{array}
  \right).
  $$
\end{enumerate}
The nil endpoint shape and every displayed cons endpoint shape are disjoint by
BEDC history no-confusion between $\emp$ and $\Eone(p)$.
\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:list-singleton-source-public-length-shape-exhaustion} to
the carried endpoint $h$. This gives either the nil public shape of $h$ or a
successor public shape of $h$ with histories $a,t,p:\mathsf{BHist}$ and tail
length $n$.

In the nil case, apply
Theorem~\ref{thm:list-singleton-source-finite-data-certificate-package} to the
same singleton source. Its classified endpoint length clause transports the
public length $0$ along
$\mathsf{ListSpineHistoryClassifier}_{A_{a_0},\hsame}(h,k)$, using
public-length uniqueness to identify the transported value with $0$.

The package zero readback clause then converts
$\mathsf{ListPublicLength}_{A_{a_0}}(k,0)$ into $\hsame(k,\emp)$. Pair this
with the nil data already obtained for $h$ to get the first synchronized
alternative.

In the successor case for $h$, again use the classified endpoint length clause
of Theorem~\ref{thm:list-singleton-source-finite-data-certificate-package} to
transport the public endpoint length $n+1$ across the classifier witness.
Public-length uniqueness identifies the transported public length of $k$ with
$n+1$.

Apply the package successor constructor exactness clause to
$\mathsf{ListPublicLength}_{A_{a_0}}(k,n+1)$. It supplies histories
$b,u,q:\mathsf{BHist}$ with $A_{a_0}(b)$,
$\mathsf{ListHistoryCarrier}_{A_{a_0}}(u)$,
$\mathsf{ListPublicLength}_{A_{a_0}}(u,n)$, $\ContR(b,u,q)$, and
$\hsame(k,\Eone(q))$.

Collect $n$ with the histories $a,b,t,u,p,q$. The successor data for $h$ and
the transported successor data for $k$ are exactly the second synchronized
alternative.

For disjointness, suppose the nil alternative and a displayed cons alternative
both hold. Compose the nil endpoint witness $\hsame(h,\emp)$ with the cons
endpoint witness $\hsame(h,\Eone(p))$ by symmetry and transitivity of
$\hsame$. This gives $\hsame(\emp,\Eone(p))$, contradicting
Theorem~\ref{thm:history-no-confusion}. The same argument applies from the
$k$ endpoint using $\Eone(q)$.
\end{proof}
