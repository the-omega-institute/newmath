\begin{definition}[Framed list-spine representation]
\label{def:framed-list-spine-representation}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier. Define
$\mathsf{FramedListSpineRep}_A(h,xs)$ to mean that every entry of the finite
spine $xs$ satisfies $A$ and that
$$
\hsame(h,\mathsf{FramedListEndpoint}(xs)).
$$
\end{definition}
\leandef{BEDC.Derived.ListUp.FramedListSpineRep}

\begin{theorem}[Framed list-spine representation hsame transport]
\label{thm:framed-list-spine-rep-hsame-transport}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier. If
$\mathsf{FramedListSpineRep}_A(h,xs)$ holds and $\hsame(h,h')$ holds, then
$\mathsf{FramedListSpineRep}_A(h',xs)$ holds.
\end{theorem}
\begin{proof}
The entrywise source predicate over $xs$ is unchanged. The endpoint witness for
$h'$ is obtained by composing $\hsame(h',h)$, the symmetry of the displayed
$\hsame(h,h')$, with the stored witness
$\hsame(h,\mathsf{FramedListEndpoint}(xs))$.
\end{proof}
\leanchecked{BEDC.Derived.ListUp.FramedListSpineRep\_hsame\_transport}

\begin{theorem}[Framed list-spine cons endpoint injectivity]
\label{thm:list-spine-rep-cons-endpoint-injective}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier. If
$\mathsf{FramedListSpineRep}_A(h,\mathsf{cons}(a,xs))$ and
$\mathsf{FramedListSpineRep}_A(k,\mathsf{cons}(b,ys))$ hold, and
$\hsame(h,k)$ holds, then the two framed cons endpoints are $\hsame$, and their
head payloads and tail endpoints are separately $\hsame$:
$$
\hsame(\Eone(\mathsf{PairFrame}(a,\mathsf{FramedListEndpoint}(xs))),
\Eone(\mathsf{PairFrame}(b,\mathsf{FramedListEndpoint}(ys))))
$$
and
$$
\hsame(a,b)\land
\hsame(\mathsf{FramedListEndpoint}(xs),\mathsf{FramedListEndpoint}(ys)).
$$
\end{theorem}
\begin{proof}
Compose the stored endpoint witness for the first representation, the displayed
$\hsame(h,k)$ witness, and the stored endpoint witness for the second
representation. This gives $\hsame$ between the two cons endpoints. Applying
\autoref{thm:framed-list-endpoint-no-confusion-cons-inversion} to that cons-cons
endpoint equality exposes both the head equality $\hsame(a,b)$ and the tail
endpoint equality.
\end{proof}
\leanchecked{BEDC.Derived.ListUp.FramedListSpineRep\_cons\_endpoint\_injective}

\begin{theorem}[Framed list-spine nil-cons no-confusion]
\label{thm:framed-list-spine-rep-nil-cons-no-confusion}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier. No
single history can be represented both by the framed nil spine and by a framed
cons spine:
$$
\mathsf{FramedListSpineRep}_A(h,\mathsf{nil})\to
\mathsf{FramedListSpineRep}_A(h,\mathsf{cons}(a,xs))\to
\mathsf{False}.
$$
\end{theorem}
\begin{proof}
Unpack the two framed representations and compose their endpoint classifiers
through symmetry and transitivity of $\hsame$. This gives
$\hsame(\mathsf{FramedListEndpoint}(\mathsf{nil}),
\mathsf{FramedListEndpoint}(\mathsf{cons}(a,xs)))$, which contradicts the
framed endpoint nil-cons no-confusion theorem.
\end{proof}
\leanchecked{BEDC.Derived.ListUp.FramedListSpineRep\_nil\_cons\_no\_confusion}

\begin{theorem}[Framed list-spine coherence]
\label{thm:framed-list-spine-coherence}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier and let
$\sim_A$ be a source classifier. If
$\mathsf{ListSourceHsameCompatible}(A,\sim_A)$ holds, then any two framed
presentations of the same history classify their finite spines:
$$
\forall h\,xs\,ys.\;
\mathsf{FramedListSpineRep}_A(h,xs)\to
\mathsf{FramedListSpineRep}_A(h,ys)\to
\mathsf{ListClassifierSpec}(\sim_A,xs,ys).
$$
\end{theorem}
\begin{proof}
Lemma~\ref{lem:list-pair-frame-injectivity} supplies the required injectivity
of $\mathsf{PairFrame}$. The framed list endpoint separates nil and cons by
the outer $\Ezero/\Eone$ tag, so mixed nil/cons presentations contradict BEDC
history no-confusion. In the nil-nil case the finite-spine classifier is the nil
constructor. In the cons-cons case, injectivity of the outer $\Eone$ endpoint
and then injectivity of $\mathsf{PairFrame}$ expose $\hsame$ between the two
heads and $\hsame$ between the framed tails. The representation hypotheses give
$A$ for both heads, so $\mathsf{ListSourceHsameCompatible}(A,\sim_A)$ supplies
the head classifier. Apply the induction hypothesis to the two framed tail
presentations, and combine the head classifier with the tail classifier by the
cons constructor of $\mathsf{ListClassifierSpec}$.
\end{proof}
\leanchecked{BEDC.Derived.ListUp.FramedListSpineRep\_coherence}

\begin{theorem}[Framed list-spine length determinism]
\label{thm:framed-list-spine-length-determinism}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier and let
$\sim_A$ be a source classifier. If
$\mathsf{ListSourceHsameCompatible}(A,\sim_A)$,
$\mathsf{FramedListSpineRep}_A(h,xs)$, and
$\mathsf{FramedListSpineRep}_A(h,ys)$ hold, then
$$
\ell(xs)=\ell(ys).
$$
The roadmap ten-field fit is: traditional target finite-list framed readback; BEDC source $\mathsf{BHist}$; carrier $\mathsf{FramedListHistoryCarrier}_A$; classifier $\mathsf{ListClassifierSpec}$; stability framed endpoint coherence; ledger finite spine; core theorem framed length determinism; exactness through framed spine representation; standard bridge finite spines; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:framed-list-spine-coherence} to the compatibility
witness and the two displayed framed presentations of $h$. This gives
$\mathsf{ListClassifierSpec}(\sim_A,xs,ys)$. Then apply
Theorem~\ref{thm:list-classifier-length-eq} to read back equality of the two
finite spine lengths.
\end{proof}
\leanchecked{BEDC.Derived.ListUp.FramedListSpineRep\_length\_determinism}

\begin{theorem}[Framed list bridge classifier preserves represented length]
\label{thm:framed-list-bridge-classifier-represented-length}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier and let
$\sim_A$ be a source classifier. For histories $h,k:\mathsf{BHist}$ and finite
spines $xs_0,ys_0$, if $\mathsf{ListSourceHsameCompatible}(A,\sim_A)$,
$\mathsf{FramedListSpineRep}_A(h,xs_0)$,
$\mathsf{FramedListSpineRep}_A(k,ys_0)$, and
$\mathsf{FramedListBridgeClassifier}_{A,\sim_A}(h,k)$ hold, then
$$
\ell(xs_0)=\ell(ys_0).
$$
The roadmap ten-field fit is: traditional target finite-list framed readback along a bridge classifier; BEDC source $\mathsf{BHist}$; carrier $\mathsf{FramedListHistoryCarrier}_A$; classifier $\mathsf{FramedListBridgeClassifier}_{A,\sim_A}$; stability framed endpoint coherence and source classifier length soundness; ledger finite spine representatives at both endpoints; core theorem bridge length preservation; exactness by framed spine representation; standard bridge finite spines; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Unpack the framed bridge classifier witness to finite spines $xs,ys$ with
$\mathsf{FramedListSpineRep}_A(h,xs)$,
$\mathsf{FramedListSpineRep}_A(k,ys)$, and
$\mathsf{ListClassifierSpec}(\sim_A,xs,ys)$. Apply
Theorem~\ref{thm:framed-list-spine-coherence} at $h$ to classify $xs_0$ with
$xs$, and at $k$ to classify $ys$ with $ys_0$. Apply
Theorem~\ref{thm:list-classifier-length-eq} to the three finite-spine classifier
witnesses $xs_0\sim_A xs$, $xs\sim_A ys$, and $ys\sim_A ys_0$, then chain the
three finite-length equalities.
\end{proof}
\leanchecked{BEDC.Derived.ListUp.FramedListBridgeClassifier\_represented\_length}
