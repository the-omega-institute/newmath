\chapter{A Concrete Naming Certificate for \(\CommRingUp\)}
\label{ch:concrete-instances-commring-namecert}

The commutative ring interface \(\CommRingUp\) extends \autoref{ch:concrete-instances-ring-namecert} with the commutativity field \(a \cdot b = b \cdot a\) for the multiplicative operation under the carrier's classifier. The integer ring from \autoref{ch:concrete-instances-int-namecert} is the canonical concrete instance.

\section{Carrier definition}

\begin{definition}[CommRing carrier]
\label{def:commring-carrier}
Use a ring setup extended with commutativity of multiplication.
\end{definition}

\section{Concrete singleton empty-history commutative ring instance}
\label{sec:commring-concrete-singleton-instance}

\begin{definition}[Singleton empty-history commutative ring instance]
\label{def:singleton-empty-history-commring-instance}
Let $A_{\emp}$ be the singleton history source of
Definition~\ref{def:list-singleton-history-source} at $a_0=\emp$. Define the
classifier
$$
\rho_{\emp}(h,k)\Longleftrightarrow
A_{\emp}(h)\land A_{\emp}(k)\land\hsame(h,k).
$$
Define the concrete commutative ring operations by
$$
\mathsf{add}(h,k):=\emp,\qquad
\mathsf{neg}(h):=\emp,\qquad
0:=\emp,\qquad
\mathsf{mul}(h,k):=\emp,\qquad
1:=\emp.
$$
Equip this singleton ring tuple with the multiplicative commutativity field
$$
\rho_{\emp}(\mathsf{mul}(x,y),\mathsf{mul}(y,x)).
$$
The concrete commutative-ring certificate has the following ten fields:
\begin{enumerate}
  \item traditional target: the one-element commutative ring;
  \item BEDC source: $\mathsf{BHist}$ with the singleton empty-history ring instance;
  \item carrier: $A_{\emp}$;
  \item classifier: $\rho_{\emp}$;
  \item stability obligations: the singleton ring laws, multiplicative commutativity, and multiplication-commutativity compatibility;
  \item ledger: constant-operation provenance for $0$, $1$, addition, negation, and multiplication;
  \item core theorem: Theorem~\ref{thm:singleton-empty-history-commring-laws};
  \item exactness: endpoint classification by membership in the singleton carrier and $\hsame$;
  \item standard bridge: the ordinary one-element commutative ring;
  \item status: $\mathsf{ProofTarget}$.
\end{enumerate}
\end{definition}

\begin{theorem}[Singleton empty-history commutative ring laws]
\label{thm:singleton-empty-history-commring-laws}
The tuple
$(A_{\emp},\rho_{\emp},\mathsf{add},0,\mathsf{neg},\mathsf{mul},1)$ satisfies
all singleton empty-history ring laws from
Theorem~\ref{thm:singleton-empty-history-ring-laws}, together with
multiplicative commutativity and the compatibility fields for multiplication
commutativity under $\rho_{\emp}$.
\end{theorem}
\begin{proof}
The ring-law part is exactly Theorem~\ref{thm:singleton-empty-history-ring-laws}
applied to the displayed singleton empty-history operation tuple. This supplies
carrier inhabitance, additive abelian-group laws, multiplicative monoid laws,
operation congruence, both distributive laws, and zero multiplication laws.

For multiplicative commutativity, both $\mathsf{mul}(x,y)$ and
$\mathsf{mul}(y,x)$ are definitionally $\emp$. The two endpoint witnesses for
$\rho_{\emp}$ are the singleton carrier witnesses for $\emp$, and the relation
component is $\hsame(\emp,\emp)$.

For compatibility, the multiplication operation is constant on the singleton
carrier. Replacing inputs by $\rho_{\emp}$-related inputs does not change either
product endpoint: every source and target product is still $\emp$. Hence the
same endpoint witnesses and reflexive $\hsame(\emp,\emp)$ package the
commutativity field after transport.
\end{proof}
\leanchecked{BEDC.Derived.CommRingUp.singleton\_empty\_history\_commring\_laws}

The $\CommRingUp$ schema is instantiated by this concrete $\mathsf{BHist}$
carrier and operations, enabling later $\mathsf{BHist}$-anchored Lean proofs
instead of abstract operator fields.

\section{The certificate}

\begin{definition}[CommRing source specification]
\label{def:commring-source-specification}
The source is a certified ring carrier plus multiplicative commutativity.
\end{definition}

\begin{definition}[CommRing pattern specification]
\label{def:commring-pattern-specification}
The pattern is ring expression formation where multiplication may be exchanged by the commutativity field.
\end{definition}

\begin{definition}[CommRing classifier specification]
\label{def:commring-classifier-specification}
The classifier is the ring carrier classifier.
\end{definition}

\begin{definition}[CommRing stability certificate]
\label{def:commring-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item all ring stability fields
  \item multiplicative commutativity
  \item compatibility of multiplication commutativity with ring congruence
\end{enumerate}
\end{definition}

\begin{theorem}[CommRing right distributivity from left distributivity]
\label{thm:commring-right-distrib-from-left}
For BEDC history addition and multiplication, multiplicative commutativity,
additive congruence, and left distributivity imply right distributivity:
$$
(x+y)\cdot z \hsame x\cdot z + y\cdot z .
$$
\end{theorem}
\begin{proof}
Commute \((x+y)\cdot z\) to \(z\cdot(x+y)\). Apply left distributivity, then
use additive congruence with multiplicative commutativity on both summands.
\end{proof}
\leanchecked{BEDC.Derived.CommRingUp.commring\_right\_distrib\_from\_left}

\begin{theorem}[CommRing left distributivity from right distributivity]
\label{thm:commring-left-distrib-from-right}
For BEDC history addition and multiplication, multiplicative commutativity,
additive congruence, and right distributivity imply left distributivity:
$$
x\cdot(y+z)\hsame x\cdot y + x\cdot z .
$$
\end{theorem}
\begin{proof}
Commute $x\cdot(y+z)$ to $(y+z)\cdot x$. Apply right distributivity, then
use additive congruence with multiplicative commutativity on both summands.
\end{proof}
\leanchecked{BEDC.Derived.CommRingUp.commring\_left\_distrib\_from\_right}

\begin{theorem}[CommRing square of a sum expansion]
\label{thm:commring-square-add-expand}
For BEDC history addition and multiplication, multiplicative commutativity,
additive congruence, and both distributive laws imply
$$
(a+b)\cdot(a+b)\hsame (a\cdot a+a\cdot b)+(a\cdot b+b\cdot b).
$$
\end{theorem}
\begin{proof}
Apply right distributivity to the outer product. Apply left distributivity to
both exposed factors, then use additive congruence and multiplicative
commutativity to replace \(b\cdot a\) by \(a\cdot b\).
\end{proof}
\leanchecked{BEDC.Derived.CommRingUp.commring\_square\_add\_expand}

\begin{theorem}[CommRing square of a signed difference]
\label{thm:commring-square-signed-difference-expand}
For BEDC history operations $add$, $mul$, and $neg$ satisfying the $\RingUp$
additive abelian-group laws, addition and multiplication congruence, both
distributive laws, multiplicative commutativity, and the $\RingUp$
negative-product laws, every $a,b : \mathsf{BHist}$ satisfies
$$
(a+(-b))\cdot(a+(-b))\hsame
(a\cdot a+(-(a\cdot b)))+((-(a\cdot b))+(b\cdot b)).
$$
\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:commring-square-add-expand} with second input $-b$.
This gives the ledger expression
$$
(a\cdot a+a\cdot(-b))+(a\cdot(-b)+(-b)\cdot(-b)).
$$

Use Theorem~\ref{thm:ring-mul-neg-right-eq-neg-mul} on each exposed
$a\cdot(-b)$ term, and use Theorem~\ref{thm:ring-mul-neg-neg-eq-mul} on
$(-b)\cdot(-b)$.

Nested additive congruence transports these three product rewrites through the
two additions. Transitivity of $\hsame$ composes the square expansion with the
transported ledger, yielding the stated signed-difference square form.
\end{proof}

For the $\CommRingUp$ per-object template, this theorem is a Core theorem list
item for traditional target commutative-ring polynomial algebra. Its BEDC source
is $\RingUp$ plus $\CommRingUp$ over $\mathsf{BHist}$, its carrier is
$\mathsf{BHist}$, its classifier is $\hsame$, its stability obligations are
distributivity, congruence, commutativity, and negative-product laws, and its
ledger records polynomial expression provenance with status ProofTarget.

\begin{theorem}[CommRing product of binary sums]
\label{thm:commring-mul-add-add-expand}
For BEDC history addition and multiplication, multiplicative commutativity,
additive congruence, and left distributivity imply the expansion
$$
(a+b)\cdot(c+d)\hsame
(a\cdot c+a\cdot d)+(b\cdot c+b\cdot d).
$$
\end{theorem}
\begin{proof}
First derive right distributivity from multiplicative commutativity and left
distributivity, then apply it to the outer product. Apply left distributivity
to the two inner products $a\cdot(c+d)$ and $b\cdot(c+d)$ and transport both
equalities through additive congruence.
\end{proof}
\leanchecked{BEDC.Derived.CommRingUp.commring\_mul\_add\_add\_expand}

\begin{theorem}[CommRing column expansion of binary sums]
\label{thm:commring-mul-add-add-column-expand}
For BEDC history addition and multiplication, additive associativity,
additive commutativity, additive congruence, multiplicative commutativity,
and left distributivity imply, for every $a,b,c,d : \mathsf{BHist}$,
$$
(a+b)\cdot(c+d)\hsame
(a\cdot c+b\cdot c)+(a\cdot d+b\cdot d).
$$
\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:commring-mul-add-add-expand} to obtain the row-major
expansion of $(a+b)\cdot(c+d)$ as
$$
(a\cdot c+a\cdot d)+(b\cdot c+b\cdot d).
$$

On the additive operation, apply
Theorem~\ref{thm:abgroup-middle-four-interchange} with the four summands
$a\cdot c$, $a\cdot d$, $b\cdot c$, and $b\cdot d$. Additive congruence then
composes this interchange with the row-major expansion and yields the
column-major normal form.
\end{proof}

\begin{theorem}[CommRing difference of squares]
\label{thm:commring-difference-of-squares}
For BEDC history operations $add$, $mul$, $neg$, and $zero$ satisfying the
$\RingUp$ additive abelian-group laws, addition and multiplication
congruence, left distributivity, multiplicative commutativity, and the
$\RingUp$ negative-product laws, every $a,b : \mathsf{BHist}$ satisfies
$$
(a+b)\cdot(a+(-b))\hsame (a\cdot a)+(-(b\cdot b)).
$$
\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:commring-mul-add-add-expand} to expand
$(a+b)\cdot(a+(-b))$ into the four-term ledger expression
$(a\cdot a+a\cdot(-b))+(b\cdot a+b\cdot(-b))$.

Use Theorem~\ref{thm:ring-mul-neg-right-eq-neg-mul} to rewrite
$a\cdot(-b)$ as $-(a\cdot b)$. Commute $b\cdot a$ to $a\cdot b$ by
multiplicative commutativity, and use the same right-negative product theorem
to rewrite $b\cdot(-b)$ as $-(b\cdot b)$.

The expression is therefore transported by additive congruence to
$(a\cdot a+(-(a\cdot b)))+(a\cdot b+(-(b\cdot b)))$. Additive associativity
and commutativity move the middle pair next to each other without changing the
outer square terms.

Finally, additive inverse collapse gives $(-(a\cdot b))+(a\cdot b)\hsame0$.
The zero laws remove this neutral summand, leaving
$(a\cdot a)+(-(b\cdot b))$ under $\hsame$.
\end{proof}

\begin{theorem}[CommRing equal squares give a signed zero factor]
\label{thm:commring-equal-squares-signed-factor-zero}
For BEDC history operations $add$, $mul$, $neg$, and $zero$ satisfying the
$\RingUp$ additive abelian-group laws, addition and multiplication
congruence, left distributivity, multiplicative commutativity, and the
$\RingUp$ negative-product laws, every $a,b : \mathsf{BHist}$ with
$a\cdot a\hsame b\cdot b$ satisfies
$$
(a+b)\cdot(a+(-b))\hsame0 .
$$
\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:commring-difference-of-squares} to rewrite the signed
factor product as $(a\cdot a)+(-(b\cdot b))$. This expansion already includes
the negative-product normalizations and the commutation of the middle product.

Transport the assumed square equality $a\cdot a\hsame b\cdot b$ through
additive congruence while keeping the right summand $-(b\cdot b)$ fixed. The
right side is therefore identified with $(b\cdot b)+(-(b\cdot b))$.

Theorem~\ref{thm:ring-add-right-inverse} collapses this final additive inverse
pair to $0$. Transitivity of $\hsame$ composes the difference-of-squares
normal form, the transported square equality, and the inverse collapse.
\end{proof}

\begin{theorem}[CommRing signed binary product normal form]
\label{thm:commring-signed-binary-product-normal-form}
For BEDC history operations $add$, $mul$, $neg$, and $zero$ satisfying the
$\RingUp$ additive abelian-group laws, addition and multiplication
congruence, left distributivity, multiplicative commutativity, and the
$\RingUp$ negative-product laws, every $a,b,c,d : \mathsf{BHist}$ satisfies
$$
(a+(-b))\cdot(c+(-d))\hsame
(a\cdot c+(-(a\cdot d)))+((-(b\cdot c))+(b\cdot d)).
$$
\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:commring-mul-add-add-expand} with the four inputs
$a$, $-b$, $c$, and $-d$. The left side expands to
$$
(a\cdot c+a\cdot(-d))+((-b)\cdot c+(-b)\cdot(-d)).
$$

Next normalize the signed products. Theorem~\ref{thm:ring-mul-neg-right-eq-neg-mul}
rewrites $a\cdot(-d)$ as $-(a\cdot d)$,
Theorem~\ref{thm:ring-mul-neg-left-eq-neg-mul} rewrites $(-b)\cdot c$ as
$-(b\cdot c)$, and Theorem~\ref{thm:ring-mul-neg-neg-eq-mul} rewrites
$(-b)\cdot(-d)$ as $b\cdot d$.

Use nested additive congruence to transport the three rewritten summands
through the outer two additions. Transitivity of $\hsame$ composes the
expansion step with this transport, giving the stated four-term normal form.
\end{proof}

\begin{theorem}[CommRing signed column expansion of binary sums]

For BEDC history operations $add$, $mul$, $neg$, and $zero$ satisfying the

$\RingUp$ negative-product laws, every $a,b,c,d : \mathsf{BHist}$ satisfies

(a+(-b))\cdot(c+(-d))\hsame
(a\cdot c+(-(b\cdot c)))+((-(a\cdot d))+(b\cdot d)).

\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:commring-signed-binary-product-normal-form} to obtain

(a\cdot c+(-(a\cdot d)))+((-(b\cdot c))+(b\cdot d)).
$$

On the additive operation, apply

\begin{theorem}[CommRing signed column cancellations force zero product]
\label{thm:commring-signed-column-cancellations-zero}
For BEDC history operations $add$, $mul$, $neg$, and $zero$ satisfying the
$\RingUp$ additive abelian-group laws, addition and multiplication
congruence, left distributivity, multiplicative commutativity, and the
$\RingUp$ negative-product laws, every $a,b,c,d : \mathsf{BHist}$ with
$a\cdot c\hsame b\cdot c$ and $a\cdot d\hsame b\cdot d$ satisfies
$$
(a+(-b))\cdot(c+(-d))\hsame0.
$$
\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:commring-signed-column-binary-product-normal-form} to
rewrite the signed product as
$$
(a\cdot c+(-(b\cdot c)))+((-(a\cdot d))+(b\cdot d)).
$$

The first hypothesis is transported by additive congruence through the left
signed column. It identifies $a\cdot c+(-(b\cdot c))$ with
$(b\cdot c)+(-(b\cdot c))$, and Theorem~\ref{thm:ring-add-right-inverse}
collapses this right-inverse pair to $0$.

For the second signed column, apply additive inverse congruence to
$a\cdot d\hsame b\cdot d$ and identify $-(a\cdot d)$ with $-(b\cdot d)$.
Additive congruence transports $(-(a\cdot d))+(b\cdot d)$ to
$(-(b\cdot d))+(b\cdot d)$, and the additive left-inverse law collapses this
left-inverse pair to $0$.

The column normal form is therefore transported to $0+0$. The additive
left-zero law collapses $0+0$ to $0$, and transitivity of $\hsame$ composes
the normal-form step, the two signed-column collapses, and the final zero
collapse.
\end{proof}
