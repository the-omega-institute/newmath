\begin{theorem}[CommRing zero linear factor annihilator branch]
\label{thm:commring-zero-linear-factor-annihilator-branch}
For BEDC history operations $add$, $mul$, $neg$, and $zero$ satisfying the
$\RingUp$ additive abelian-group laws, addition and multiplication congruence,
left distributivity, the $\RingUp$ negative-product laws, and the
$\CommRingUp$ multiplicative commutativity field used in
\autoref{thm:commring-difference-of-squares-zero-exact}, every
$a,b : \mathsf{BHist}$ satisfies
$$
\bigl((a+b)\hsame0 \lor (a+(-b))\hsame0\bigr)
\Longrightarrow
\bigl((a+b)\hsame0 \land
\forall c : \mathsf{BHist},\ ((a+b)\cdot c\hsame0 \land c\cdot(a+b)\hsame0)\bigr)
\lor
\bigl((a+(-b))\hsame0 \land
\forall c : \mathsf{BHist},\ ((a+(-b))\cdot c\hsame0 \land c\cdot(a+(-b))\hsame0)\bigr).
$$
\end{theorem}
\begin{proof}
First derive the right distributive law from the supplied left distributive law
by \autoref{thm:commring-right-distrib-from-left}. The zero-classifier factor
absorption theorem is then available with both distributive inputs.

If the displayed disjunction chooses $(a+b)\hsame0$, retain that branch witness.
For an arbitrary $c:\mathsf{BHist}$, apply
\autoref{thm:ring-zero-classifier-factor-absorption} with $x=a+b$ and $y=c$.
Its two components give $(a+b)\cdot c\hsame0$ and $c\cdot(a+b)\hsame0$, so the
left disjunct packages the original witness with the universal annihilator row.

If the displayed disjunction chooses $(a+(-b))\hsame0$, retain that branch
witness instead. For an arbitrary $c:\mathsf{BHist}$, apply
\autoref{thm:ring-zero-classifier-factor-absorption} with $x=a+(-b)$ and $y=c$.
The two components give $(a+(-b))\cdot c\hsame0$ and
$c\cdot(a+(-b))\hsame0$, yielding the right disjunct with the same
branch-preserving form.
\end{proof}
\leanchecked{BEDC.Derived.CommRingUp.commring\_zero\_linear\_factor\_annihilator\_branch}

\begin{theorem}[CommRing zero linear factor signed product zero]
\label{thm:commring-zero-linear-factor-signed-product-zero}
For BEDC history operations $add$, $mul$, $neg$, and $zero$ satisfying the
$\RingUp$ additive abelian-group laws, addition and multiplication congruence,
left distributivity, the $\RingUp$ negative-product laws, and the
$\CommRingUp$ multiplicative commutativity field used in
\autoref{thm:commring-difference-of-squares-zero-exact}, every
$a,b : \mathsf{BHist}$ satisfies
$$
\bigl((a+b)\hsame0 \lor (a+(-b))\hsame0\bigr)
\Longrightarrow
(a+b)\cdot(a+(-b))\hsame0 .
$$
\end{theorem}
\begin{proof}
Apply \autoref{thm:commring-zero-linear-factor-annihilator-branch} to the chosen
zero linear factor. The result keeps the selected branch and supplies its
two-sided annihilator row for every multiplier.

In the $(a+b)\hsame0$ branch, specialize the universal row at
$c=a+(-b)$ and take its left component. This is exactly
$(a+b)\cdot(a+(-b))\hsame0$.

In the $(a+(-b))\hsame0$ branch, specialize the universal row at $c=a+b$ and
take its right component. The oriented product is
$(a+b)\cdot(a+(-b))$, giving the required zero classifier.
\end{proof}

\begin{theorem}[CommRing zero linear factor forces equal squares]
\label{thm:commring-zero-linear-factor-equal-squares}
For BEDC history operations $add$, $mul$, $neg$, and $zero$ satisfying the
$\RingUp$ additive abelian-group laws, addition and multiplication congruence,
left distributivity, the $\RingUp$ negative-product laws, and the
$\CommRingUp$ multiplicative commutativity field used in
\autoref{thm:commring-difference-of-squares-zero-exact}, every
$a,b : \mathsf{BHist}$ satisfies
$$
\bigl((a+b)\hsame0 \lor (a+(-b))\hsame0\bigr)
\Longrightarrow
a\cdot a\hsame b\cdot b .
$$
\end{theorem}
\begin{proof}
Apply \autoref{thm:commring-zero-linear-factor-signed-product-zero} to the
chosen zero linear factor. This produces
$(a+b)\cdot(a+(-b))\hsame0$.

The forward direction of
\autoref{thm:commring-difference-of-squares-zero-exact} sends this signed
product-zero witness to $a\cdot a\hsame b\cdot b$.
\end{proof}

\begin{theorem}[CommRing zero linear factor exact package]
\label{thm:commring-zero-linear-factor-exact-package}

$\RingUp$ additive abelian-group laws, addition and multiplication congruence,
left distributivity, the $\RingUp$ negative-product laws, and the
$\CommRingUp$ multiplicative commutativity field used in
\autoref{thm:commring-difference-of-squares-zero-exact}, every
$a,b : \mathsf{BHist}$ satisfies
$$
\bigl((a+b)\hsame0 \lor (a+(-b))\hsame0\bigr)
\Longrightarrow
\bigl((a+b)\cdot(a+(-b))\hsame0 \land a\cdot a\hsame b\cdot b\bigr).
$$
\end{theorem}
\begin{proof}
Apply \autoref{thm:commring-zero-linear-factor-signed-product-zero} to the
chosen zero linear factor. This gives the signed product-zero witness
$(a+b)\cdot(a+(-b))\hsame0$.

Feed that witness through the forward direction of
\autoref{thm:commring-difference-of-squares-zero-exact}. The resulting
square equality is $a\cdot a\hsame b\cdot b$, and pairing it with the
signed product-zero witness gives the displayed conjunction.
\end{proof}
