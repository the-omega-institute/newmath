\chapter{A Concrete Naming Certificate for \(\CommRingUp\)}
\label{ch:concrete-instances-commring-namecert}

The commutative ring interface \(\CommRingUp\) extends \autoref{ch:concrete-instances-ring-namecert} with the commutativity field \(a \cdot b = b \cdot a\) for the multiplicative operation under the carrier's classifier. The integer ring from \autoref{ch:concrete-instances-int-namecert} is the canonical concrete instance.

\section{Carrier definition}

\begin{definition}[CommRing carrier]
\label{def:commring-carrier}
Use a ring setup extended with commutativity of multiplication.
\end{definition}

\section{Concrete singleton history commutative ring instance}
\label{sec:commring-concrete-singleton-history-instance}

\begin{theorem}[Concrete singleton history commutative ring laws]
\label{thm:concrete-singleton-history-commring-laws}
Let $A_{\emp}(h)$ abbreviate $\hsame(h,\emp)$ and let
$R_{\emp}(h,k)$ abbreviate
$A_{\emp}(h)\land A_{\emp}(k)\land\hsame(h,k)$. With
$$
\mathsf{add}(h,k):=\mathsf{append}(h,k),\qquad
\mathsf{neg}(h):=\emp,\qquad
0:=\emp,
$$
and
$$
\mathsf{mul}(h,k):=\mathsf{append}(h,k),\qquad
1:=\emp,
$$
the carrier contains $0$, is closed under addition, negation, and
multiplication, has left and right additive zero laws under $R_{\emp}$, and
satisfies multiplicative commutativity under $R_{\emp}$.
\end{theorem}
\begin{proof}
The carrier witness $A_{\emp}(0)$ is reflexive $\hsame(\emp,\emp)$. For
addition and multiplication closure, if $A_{\emp}(x)$ and $A_{\emp}(y)$ hold,
then both inputs are the empty history, so
$\mathsf{append}(x,y)$ reduces to $\emp$. The negation operation is constantly
$\emp$, so its closure is immediate from the same reflexive endpoint witness.

For the two additive zero laws, the singleton input witness identifies $x$
with $\emp$; after that identification, both
$\mathsf{append}(\emp,x)$ and $\mathsf{append}(x,\emp)$ are the empty history,
and the classifier packages the two carrier endpoints with reflexive
$\hsame$. Multiplicative commutativity is identical: two singleton carrier
witnesses identify $x$ and $y$ with $\emp$, so the two append products are
both $\emp$ and are related by $R_{\emp}$.
\end{proof}
\leanchecked{BEDC.Derived.CommRingUp.concrete\_singleton\_history\_commring\_laws}

$$
\mathsf{add}(h,k):=\emp,\qquad
\mathsf{neg}(h):=\emp,\qquad

\mathsf{mul}(h,k):=\emp,\qquad

$$

\begin{theorem}[CommRing right distributivity from left distributivity]
\label{thm:commring-right-distrib-from-left}
For BEDC history addition and multiplication, multiplicative commutativity,
additive congruence, and left distributivity imply right distributivity:
$$
(x+y)\cdot z \hsame x\cdot z + y\cdot z .
$$
\end{theorem}
\begin{proof}
Commute \((x+y)\cdot z\) to \(z\cdot(x+y)\). Apply left distributivity, then
use additive congruence with multiplicative commutativity on both summands.
\end{proof}
\leanchecked{BEDC.Derived.CommRingUp.commring\_right\_distrib\_from\_left}

x\cdot(y+z)\hsame x\cdot y + x\cdot z .

\end{theorem}

\begin{theorem}[CommRing one-sided distributivity equivalence]
\label{thm:commring-one-sided-distributivity-equivalence}
For BEDC history operations $add$ and $mul$, multiplicative commutativity and
additive congruence imply that left distributivity is equivalent to right
distributivity:
$$
\left(\forall x\,y\,z : \mathsf{BHist},\ x\cdot(y+z)\hsame x\cdot y+x\cdot z\right)
\leftrightarrow
\left(\forall x\,y\,z : \mathsf{BHist},\ (x+y)\cdot z\hsame x\cdot z+y\cdot z\right).
$$
\end{theorem}
\begin{proof}
For the forward direction, assume the displayed left distributivity law. Applying
Theorem~\ref{thm:commring-right-distrib-from-left} with the shared
multiplicative commutativity and additive congruence fields gives the displayed
right distributivity law for all $x,y,z : \mathsf{BHist}$.

For the reverse direction, assume the displayed right distributivity law.
Applying Theorem~\ref{thm:commring-left-distrib-from-right} with the same
commutativity and congruence fields gives the displayed left distributivity law
for all $x,y,z : \mathsf{BHist}$.
\end{proof}

\begin{proof}

\end{proof}

\begin{theorem}[CommRing difference of squares]
\label{thm:commring-difference-of-squares}
For BEDC history operations $add$, $mul$, $neg$, and $zero$ satisfying the
$\RingUp$ additive abelian-group laws, addition and multiplication
congruence, left distributivity, multiplicative commutativity, and the
$\RingUp$ negative-product laws, every $a,b : \mathsf{BHist}$ satisfies
$$
(a+b)\cdot(a+(-b))\hsame (a\cdot a)+(-(b\cdot b)).
$$
\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:commring-mul-add-add-expand} to expand
$(a+b)\cdot(a+(-b))$ into the four-term ledger expression
$(a\cdot a+a\cdot(-b))+(b\cdot a+b\cdot(-b))$.

Use Theorem~\ref{thm:ring-mul-neg-right-eq-neg-mul} to rewrite
$a\cdot(-b)$ as $-(a\cdot b)$. Commute $b\cdot a$ to $a\cdot b$ by
multiplicative commutativity, and use the same right-negative product theorem
to rewrite $b\cdot(-b)$ as $-(b\cdot b)$.

The expression is therefore transported by additive congruence to
$(a\cdot a+(-(a\cdot b)))+(a\cdot b+(-(b\cdot b)))$. Additive associativity
and commutativity move the middle pair next to each other without changing the
outer square terms.

Finally, additive inverse collapse gives $(-(a\cdot b))+(a\cdot b)\hsame0$.
The zero laws remove this neutral summand, leaving
$(a\cdot a)+(-(b\cdot b))$ under $\hsame$.
\end{proof}
\leanchecked{BEDC.Derived.CommRingUp.commring\_difference\_of\_squares}

\begin{theorem}[CommRing difference-of-squares zero exactness]
\label{thm:commring-difference-of-squares-zero-exact}
For BEDC history operations $add$, $mul$, $neg$, and $zero$ satisfying the
$\RingUp$ additive abelian-group laws, addition and multiplication
congruence, left distributivity, multiplicative commutativity, and the
$\RingUp$ negative-product laws, every $a,b : \mathsf{BHist}$ satisfies
$$
(a+b)\cdot(a+(-b))\hsame0
\Longleftrightarrow
a\cdot a\hsame b\cdot b .
$$
\end{theorem}
\begin{proof}
For the forward direction, apply Theorem~\ref{thm:commring-difference-of-squares}
to identify the signed product with $(a\cdot a)+(-(b\cdot b))$. Compose this
identification, using symmetry where needed, with the zero-product hypothesis.
The difference term is therefore classified as $0$.

Add $b\cdot b$ on the right by additive congruence. On the left endpoint,
additive associativity and commutativity move the inverse pair into the form
$(b\cdot b)+(-(b\cdot b))$; Theorem~\ref{thm:ring-add-right-inverse}
collapses that pair to $0$, and the zero laws leave $a\cdot a$. On the right
endpoint, the zero law collapses $0+(b\cdot b)$ to $b\cdot b$. Transitivity
gives $a\cdot a\hsame b\cdot b$.

For the reverse direction, apply Theorem~\ref{thm:commring-difference-of-squares}
to rewrite the signed factor product as $(a\cdot a)+(-(b\cdot b))$. Transport
the assumed square equality $a\cdot a\hsame b\cdot b$ through additive
congruence while keeping the right summand $-(b\cdot b)$ fixed. The right side
is therefore identified with $(b\cdot b)+(-(b\cdot b))$.

Theorem~\ref{thm:ring-add-right-inverse} collapses this final additive inverse
pair to $0$. Transitivity of $\hsame$ composes the difference-of-squares
normal form, the transported square equality, and the inverse collapse.
\end{proof}
\leanchecked{BEDC.Derived.CommRingUp.commring\_equal\_squares\_signed\_factor\_zero}
\leanchecked{BEDC.Derived.CommRingUp.commring\_difference\_of\_squares\_zero\_exact}
