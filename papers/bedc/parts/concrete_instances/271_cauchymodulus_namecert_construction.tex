\chapter{A Concrete Naming Certificate for $\CauchyModulusUp$}
\label{ch:concrete-instances-cauchymodulus-namecert}

The $\CauchyModulusUp$ interface records the explicit rate data used by the
Bishop-style regular-sequence construction of $\RegSeqRat$ and $\RealUp$. It
sits over $\NatUp$, $\RatUp$, and $\StreamNameUp$: natural-number rows index
observations, rational tolerance rows state requested bounds, stream-name rows
identify the scheduled observations, and a finite ledger witnesses that the
displayed modulus bound has been consumed by the downstream regular-sequence
and real-number carriers.

\section{Carrier and classifier}
\label{sec:cauchymodulus-carrier}

\begin{definition}[Cauchy-modulus carrier]
\label{def:cauchymodulus-carrier}
A $\CauchyModulusUp$ carrier packet is a finite $\mathsf{BHist}$ ledger with
four visible groups of rows.
\begin{enumerate}
\item Unary index rows supplied by $\NatUp$, including the requested precision
index and the threshold index returned by the modulus.
\item Dyadic or rational tolerance rows supplied by $\RatUp$, including the
positive tolerance budget and the displayed comparison to the tolerance read by
the regular-sequence carrier.
\item Scheduled observation rows supplied by $\StreamNameUp$, naming the
stream entries whose late pairwise distance is being bounded.
\item A finite consumption ledger stating that every displayed observation
pair past the threshold uses the same modulus row, the same tolerance row, and
the same $\mathsf{Pkg}$ provenance before it is passed to $\RegSeqRat$ or
$\RealUp$.
\end{enumerate}
The carrier accepts only packets in which the index, tolerance, schedule, and
consumption rows are present in the same finite history and are connected by
displayed $\Cont$ routes. It has no quotient of streams, no ambient limit
object, and no hidden choice of a threshold outside the ledger.
\end{definition}

\begin{definition}[Cauchy-modulus classifier]
\label{def:cauchymodulus-classifier}
The $\CauchyModulusUp$ classifier compares two accepted packets by componentwise
$\hsame$ on their precision-index rows, threshold-index rows, tolerance rows,
scheduled observation rows, consumption-ledger rows, $\Cont$ routes, and
$\mathsf{Pkg}$ provenance. Two packets classify together only when the same
visible modulus schedule can be replayed through the transported finite ledger.
\end{definition}
\leandef{BEDC.Derived.CauchyModulusUp.CauchyModulusClassifier}

\begin{theorem}[Cauchy-modulus $\hsame$ stability]
\label{thm:cauchymodulus-hsame-stability}
Let $P$ and $Q$ be accepted $\CauchyModulusUp$ packets. If every index,
tolerance, scheduled-observation, consumption-ledger, $\Cont$, and
$\mathsf{Pkg}$ row of $P$ is transported to the corresponding row of $Q$ by
displayed $\hsame$ comparisons, then $Q$ is accepted by the carrier of
\autoref{def:cauchymodulus-carrier} and the classifier of
\autoref{def:cauchymodulus-classifier} identifies the two packets.
\end{theorem}
\leanchecked{BEDC.Derived.CauchyModulusUp.CauchyModulusPacket\_hsame\_stability}
\leanchecked{BEDC.Derived.CauchyModulusUp.CauchyModulusCarrier\_hsame\_stability}
\begin{proof}
Project $P$ to the four carrier groups. The supplied $\hsame$ rows transport
the natural indices and rational tolerance rows first, so the threshold still
answers the same displayed precision request.

Transport the scheduled observation rows through the $\StreamNameUp$ surface.
The finite consumption ledger then replays the same $\Cont$ routes from the
transported threshold and tolerance rows to the transported late-observation
claims.

Finally repack the transported rows with the transported $\mathsf{Pkg}$
provenance. These are exactly the fields required by the carrier and by the
componentwise classifier, so $Q$ is accepted and the two packets classify
together.
\end{proof}
\leanchecked{BEDC.Derived.CauchyModulusUp.CauchyModulusLedgerPacket\_hsame\_stability}

\begin{theorem}[Cauchy-modulus finite-window closure]
\label{thm:cauchymodulus-cont-window-closure}
For every accepted $\CauchyModulusUp$ packet and every displayed precision
request in its $\NatUp$ index rows, the packet exposes a finite $\Cont$ window
containing the threshold row, the rational tolerance row, the scheduled late
observation rows, and the consumption ledger read by $\RegSeqRat$ and
$\RealUp$. No consumer may read a convergence rate except through that finite
window.
\end{theorem}
\leanchecked{BEDC.Derived.CauchyModulusUp.CauchyModulusPacket\_cont\_window\_closure}
\begin{proof}
Inspect the accepted carrier packet at the requested precision row. The
threshold is present as a visible $\NatUp$ row, and the tolerance budget is
present as a visible $\RatUp$ row in the same finite history.

The scheduled observation rows are linked to those two rows by the displayed
$\Cont$ routes. Restrict the packet to the rows mentioned by those routes and
to the corresponding consumption-ledger entries.

This restricted packet is finite and contains every row used by the downstream
regular-sequence and real-number readbacks. Since the carrier has no other
constructor for rate data, every convergence-rate read factors through the
finite window just described.
\end{proof}

\begin{theorem}[Cauchy-modulus finite-window composition handoff]
\label{thm:cauchymodulus-window-composition-handoff}
Suppose two accepted $\CauchyModulusUp$ precision windows are composable
through the displayed $\Cont$ routes of their carrier packets. Then their
composition exposes a single finite handoff packet whose rows are exactly the
$\NatUp$ precision and threshold rows, the $\RatUp$ tolerance rows, the
$\StreamNameUp$ scheduled observation rows, the $\RegSeqRat$ late-pair evidence
rows, the $\RealUp$ classifier transport rows, the displayed $\hsame$
comparisons, and the $\mathsf{Pkg}$ provenance rows already present in the two
windows. The handoff packet introduces no selected limit, quotient stream, or
hidden threshold selector outside those rows.
\end{theorem}
\begin{proof}
Project the two accepted windows by
\autoref{thm:cauchymodulus-cont-window-closure}. Each projection is finite and
contains its displayed precision request, threshold row, tolerance row,
scheduled observations, and consumption ledger.

Compose the two $\Cont$ ledgers along their shared carrier rows. The resulting
ledger has only the rows named by the projections: the natural-number
precision and threshold rows, rational tolerance rows, scheduled observation
rows, regular-sequence late-pair evidence, real-classifier transport rows,
$\hsame$ comparisons, and $\mathsf{Pkg}$ provenance.

Finally apply \autoref{thm:cauchymodulus-hsame-stability} to transport the
shared classifier comparisons across the composed packet. Since
\autoref{thm:cauchymodulus-cont-window-closure} already exhausts every
consumer-visible rate read, the composed packet has no constructor for a
selected limit, quotient stream, or threshold selector beyond the displayed
finite ledger.
\end{proof}

\begin{theorem}[Monotone tail obligation]
\label{thm:cauchymodulus-monotone-tail-obligation}
Let $M$ be an accepted $\CauchyModulusUp$ carrier packet. If a tolerance row is
transported to a smaller positive dyadic tolerance through the displayed
comparison ledger, then the threshold selected by the modulus row can be
transported to a threshold that works for the smaller tolerance, and the
late-pair bound rows remain accepted after $\hsame$ transport of the
observation indices.
\end{theorem}
\leanchecked{BEDC.Derived.CauchyModulusUp.CauchyModulusTailWindow\_smaller\_tolerance\_transport}
\leanchecked{BEDC.Derived.CauchyModulusUp.CauchyModulus\_monotone\_tail\_refinement\_window}
\begin{proof}
Unpack the carrier packet of \autoref{def:cauchymodulus-carrier}. The selected
threshold and every late-pair bound are visible rows of the same finite
$\mathsf{BHist}$ ledger, so decreasing the tolerance asks only for a refined
dyadic or rational comparison row, not for a new stream object.

Transport the observation endpoints by the displayed $\hsame$ rows and compose
the old bound with the smaller-tolerance comparison. The provenance ledger
supplies the $\Cont$ and $\Pkg$ rows needed to repack the same late-pair
evidence under the transported threshold.
\end{proof}

\begin{theorem}[RegSeqRat regularity obligation]
\label{thm:cauchymodulus-regseqrat-regularity-obligation}
Every accepted $\CauchyModulusUp$ carrier packet whose source stream is a
$\RegSeqRat$ regular rational sequence supplies the explicit regularity data
required by the $\RegSeqRat$ phase: for displayed indices $m,n$ past the
threshold selected by the modulus row, the bound ledger gives the estimate
required by the regular-sequence classifier, and this estimate is stable under
componentwise $\hsame$ transport of $m$, $n$, and the scheduled observations.
\end{theorem}
\leanchecked{BEDC.Derived.CauchyModulusUp.CauchyModulusRegularityWindow\_hsame\_transport}
\leanchecked{BEDC.Derived.CauchyModulusUp.CauchyModulusPacket\_regseqrat\_regularity\_handoff}
\begin{proof}
Read the $\RegSeqRat$ regularity row as a request for a visible late-pair
estimate. The carrier already stores exactly that estimate in the consumption
ledger, indexed by the threshold and by the two displayed stream observations.

For stability, transport the two indices and observations through the
classifier of \autoref{def:cauchymodulus-classifier}. The comparison row is
part of the ledger, so the transported packet again contains the regularity
estimate demanded by $\RegSeqRat$.
\end{proof}
\leanchecked{BEDC.Derived.CauchyModulusUp.CauchyModulusLedgerPacket\_regseqrat\_regularity\_rows}

\begin{theorem}[Real classifier transport obligation]
\label{thm:cauchymodulus-real-classifier-transport-obligation}
Assume the $\RealUp$ classifier is read from $\RegSeqRat$ representatives by
rational tolerance tests. A $\CauchyModulusUp$ carrier packet transports across
the $\RealUp$ classifier by sending its stream observations through the
$\RegSeqRat$ classifier, transporting the modulus threshold rows by $\hsame$,
and reusing the tail-bound ledger for the transported representatives.
\end{theorem}
\begin{proof}
The only analytic datum used by the real classifier is an explicit tolerance
test on sufficiently late representatives. The $\CauchyModulusUp$ packet
exposes this datum as a threshold row together with the late-pair ledger.

Apply the $\RegSeqRat$ transport to the representative stream and then repack
the threshold, comparison, and provenance rows. Since no equivalence class of
real numbers is formed, the transport is a row-by-row certificate operation
inside the displayed $\mathsf{BHist}$ packet.
\end{proof}

\begin{theorem}[Cauchy-modulus scoped dependency certificate]
\label{thm:cauchymodulus-scoped-dependency-certificate}
The carrier, classifier, $\hsame$ stability, finite $\Cont$ window closure,
monotone tail obligation, $\RegSeqRat$ regularity handoff, and $\RealUp$
classifier transport theorem bind every exported $\CauchyModulusUp$ row to the
following dependency stack: $\NatUp$ index rows, $\RatUp$ tolerance rows,
$\StreamNameUp$ scheduled observations, $\RegSeqRat$ regularity rows, $\RealUp$
classifier transport, $\hsame$ transport, finite $\Cont$ windows, and
$\mathsf{Pkg}$ provenance. No exported row uses a selected limit, a stream
quotient, or an unrecorded threshold selector.
\end{theorem}
\begin{proof}
Start with the carrier and classifier of
\autoref{def:cauchymodulus-carrier} and
\autoref{def:cauchymodulus-classifier}. They list the visible precision,
tolerance, scheduled-observation, comparison, $\Cont$, and $\mathsf{Pkg}$ rows,
so every consumer-visible coordinate begins as a displayed finite
$\mathsf{BHist}$ entry.

Next project the transport rows in order. The $\hsame$ stability theorem keeps
the classifier componentwise, the finite $\Cont$ window theorem bounds the
readback window, the monotone tail theorem records stronger-tolerance reuse,
and the $\RegSeqRat$ handoff reads the regularity estimate from the same late
pair ledger.

Finally apply the $\RealUp$ classifier transport theorem. It reuses the
representative-stream, threshold, comparison, and provenance rows already named
above, so the scoped dependency certificate closes inside the displayed
dependency stack and introduces no selected limit, stream quotient, or hidden
threshold choice.
\end{proof}

\label{thm:cauchymodulus-namecert-obligation-surface}
