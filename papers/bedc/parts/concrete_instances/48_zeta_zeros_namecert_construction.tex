\chapter{A Concrete Naming Certificate for \(\ZetaZerosUp\)}
\label{ch:concrete-instances-zeta-zeros-namecert}

This chapter gives a concrete naming-certificate construction for the predicate "\(s\) is a zero of \(\zeta\)" without yet constructing any non-trivial zeros. The trivial zeros at negative even integers \(-2, -4, -6, \ldots\) are constructed explicitly; the non-trivial zero predicate is characterized as elements of the critical strip whose \(\zeta\)-value (under the analytic continuation \(\ZetaContUp\), \autoref{ch:concrete-instances-zeta-continuation}) is the zero complex history. The chapter explicitly notes that whether non-trivial zeros all lie on the critical line is the content of the Riemann Hypothesis (\autoref{ch:capstones-riemann-hypothesis}); this chapter only sets up the predicate, the classifier, and the closure laws compatible with conjugate symmetry. Builds on \(\ZetaBasicUp\) (\autoref{ch:concrete-instances-zeta-basic-namecert}), \(\ZetaContUp\) (\autoref{ch:concrete-instances-zeta-continuation}), and \(\CritStripUp\) (\autoref{ch:concrete-instances-critical-strip-namecert}). The construction is fully constructive at the predicate level and uses no \(\mathsf{Classical.choice}\), no \(\mathsf{Quot.sound}\), and no \(\mathsf{propext}\), so the certificate's Lean targets fall under the project's axiom-purity gate.

\section{\(\zeta\)-evaluation and zero predicate}

\begin{definition}[\(\zeta\)-value evaluation]
\label{def:zeta-val}
For complex history \(s\) with \(\neg \hsame(s, \Eone(\emp))\) (i.e.\ \(s \neq 1\)), the predicate \(\mathsf{ZetaVal}(s, z)\) holds when \(z\) is the unique complex history obtained by evaluating the analytically continued \(\zeta\) at \(s\) (cite \autoref{ch:concrete-instances-zeta-continuation}).
\end{definition}

\begin{lemma}[\(\zeta\)-value well-defined]
\label{lem:zeta-val-well-defined}
For every \(s\) with \(s \neq 1\), \(\mathsf{ZetaVal}(s, z)\) determines a unique complex history \(z\) up to \(\hsame\) on both real components.
\end{lemma}
\begin{proof}
Direct uniqueness from the analytic continuation operation of \(\ZetaContUp\).
\end{proof}

\begin{definition}[Zero predicate]
\label{def:zeta-zero}
The predicate \(\mathsf{ZetaZero}(s)\) holds when there exists \(z\) with \(\mathsf{ZetaVal}(s, z)\) and \(\hsame\) holds on both real components of \(z\) and the zero complex history.
\end{definition}

\begin{theorem}[Zero complex history carrier]
\label{thm:zeta-zero-complex-history-carrier}
The complex zero history represented as the product of two unit rational denominators is carried by \(\ComplexUp\).
\end{theorem}
\begin{proof}
Each unit denominator satisfies the positive-denominator predicate because its tail is \(\emp\). The product history is then carried by the product carrier for the two rational components.
\end{proof}
\leanchecked{BEDC.Derived.ZetaZerosUp.ZetaZeroComplexHistory\_carrier}

\begin{remark}[Decidability gap]
\label{rem:zero-decidability-gap}
The predicate \(\mathsf{ZetaVal}(s, z)\) for fixed \(s\) is decidable to within any unary precision \(k\), since the analytic continuation is computable to any precision. However, deciding \(\mathsf{ZetaZero}(s)\) requires comparing the value to the exact zero complex history, which is not decidable from a finite-precision approximation in general --- this is the constructive obstruction at the heart of why RH is open. The predicate is *defined* constructively, but its *exhaustive enumeration* over the critical strip is not.
\end{remark}

\section{Trivial zeros}

\begin{theorem}[Trivial zero witness]
\label{thm:trivial-zero-witness}
For every unary \(n \geq \Eone(\emp)\), the negative even integer \(s = -2n\) (encoded as a complex history with rational real part \(-2n\) and imaginary part empty) satisfies \(\mathsf{ZetaZero}(s)\). The witness comes from the functional equation \(\zeta(s) = 2^s \pi^{s-1} \sin(\pi s / 2) \Gamma(1 - s) \zeta(1 - s)\), where \(\sin(\pi s / 2) = 0\) at even \(s \leq -2\) (\autoref{ch:concrete-instances-zeta-continuation}).
\end{theorem}
\begin{proof}[Sketch]
At \(s = -2n\) with \(n \geq 1\), \(\sin(\pi s / 2) = \sin(-n \pi) = 0\) constructively (rational multiple of \(\pi\) with even integer multiplier). The other functional-equation factors are non-zero and finite at \(s = -2n\) (Gamma function value at a positive integer). Hence \(\zeta(-2n) = 0\) constructively.
\end{proof}

\section{Non-trivial zeros}

\begin{definition}[Non-trivial zero predicate]
\label{def:nontrivial-zero}
The predicate \(\mathsf{NontrivialZero}(s)\) holds when both \(\mathsf{ZetaZero}(s)\) and \(\mathsf{InCritStrip}(s)\) (\autoref{ch:concrete-instances-critical-strip-namecert}).
\end{definition}

\begin{theorem}[Non-trivial zeros lie in the strip]
\label{thm:nontrivial-zero-in-strip}
For every \(s\) with \(\mathsf{NontrivialZero}(s)\), \(\mathsf{InCritStrip}(s)\) holds by definition. Conversely, the non-vanishing of \(\zeta\) for \(\Re(s) \geq 1\) (\autoref{ch:concrete-instances-zeta-basic-namecert}) and the trivial-zero classification at \(\Re(s) \leq 0\) bracket the location of any non-trivial zero strictly inside \(0 < \Re(s) < 1\).
\end{theorem}
\begin{proof}[Sketch]
\(\zeta\)'s non-vanishing on \(\Re(s) > 1\) gives the right boundary; the functional equation reduces \(\Re(s) \leq 0\) to \(\Re(s) \geq 1\); the explicit trivial-zero witnesses cover the negative even integers exactly.
\end{proof}

\section{Closure laws}

\begin{theorem}[Conjugate symmetry of zeros]
\label{thm:conjugate-symmetric}
If \(\mathsf{ZetaZero}(s)\), then \(\mathsf{ZetaZero}(\overline{s})\), where \(\overline{s}\) is the complex history obtained by negating the imaginary component.
\end{theorem}
\begin{proof}[Sketch]
\(\zeta\) takes complex conjugates to complex conjugates, since the Dirichlet sum and analytic continuation both respect conjugation; if \(\zeta(s) = 0\), then \(\overline{\zeta(s)} = \zeta(\overline{s}) = 0\).
\end{proof}

\begin{theorem}[No zero on \(\Re(s) = 1\)]
\label{thm:no-zero-real-part-one}
For every \(s\) with \(\hsame(\sigma, \Eone(\emp))\) (i.e.\ \(\Re(s) = 1\)), \(\neg \mathsf{ZetaZero}(s)\). This is the classical non-vanishing of \(\zeta\) on the line \(\Re(s) = 1\), proved by an elementary trigonometric inequality (de la Vall\'ee-Poussin / Mertens) made constructive by explicit unary-history bounds.
\end{theorem}
\begin{proof}[Sketch]
The classical inequality \(3 + 4 \cos\theta + \cos 2\theta \geq 0\) translates to a constructive product-bound on \(|\zeta(\sigma)^3 \zeta(\sigma + it)^4 \zeta(\sigma + 2it)|\), which forces \(\zeta(1 + it) \neq 0\) as \(\sigma \to 1^+\) with explicit modulus.
\end{proof}

\section{The certificate}

\begin{definition}[Zeta zero source specification]
\label{def:zeta-zero-source-spec}
The source specification carries the constraint \(\mathsf{ZetaZero}(s)\) plus an optional \(\mathsf{InCritStrip}(s)\) flag distinguishing trivial from non-trivial zeros.
\end{definition}
\leandef{BEDC.Derived.ZetaZerosUp.ZetaZeroSourceSpec}

\begin{definition}[Zeta zero pattern specification]
\label{def:zeta-zero-pattern-spec}
The pattern is the joint predicate \(\mathsf{ZetaVal}(s, z) \wedge \hsame(z, \emp_{\ComplexUp})\) on complex-history pairs.
\end{definition}

\begin{definition}[Zeta zero classifier specification]
\label{def:zeta-zero-classifier-spec}
Two zero witnesses at the same \(s\) are classified together when their underlying \(\mathsf{ZetaVal}\) witnesses agree under \(\hsame\) on the zero complex history.
\end{definition}

\begin{theorem}[Zeta zero source classifies against zero]
\label{thm:zeta-zero-source-spec-zero-classifier}
Every source history carried by \(\ComplexUp\) and \(\hsame\)-identified with the zero complex history is classified with that zero history by the \(\ComplexUp\) classifier.
\end{theorem}
\begin{proof}
The source predicate supplies the left carrier and the \(\hsame\) comparison. The zero-history carrier is \autoref{thm:zeta-zero-complex-history-carrier}; these three fields are exactly the \(\ComplexClassifier\) record.
\end{proof}
\leanchecked{BEDC.Derived.ZetaZerosUp.ZetaZeroSourceSpec\_zero\_classifier}

\begin{definition}[Zeta zero stability certificate]
\label{def:zeta-zero-stability-cert}
The stability certificate consists of \(\hsame\)-respect of \(\mathsf{ZetaZero}\), conjugate symmetry of zeros, and the boundary exclusion witnesses (no zeros on \(\Re(s) = 1\), trivial zeros at negative even integers, no zeros on \(\Re(s) \geq 1\)).
\end{definition}

\begin{definition}[Zeta zero ledger policy]
\label{def:zeta-zero-ledger-policy}
The ledger policy records: (i) trivial zeros at \(s = -2n\) for unary \(n \geq 1\), (ii) non-trivial zeros all lie strictly inside \(\mathsf{InCritStrip}\), (iii) conjugate symmetry, (iv) the explicit zero classifier under \(\hsame\). The ledger explicitly notes that exhaustive enumeration of non-trivial zeros is not constructively decidable from finite-precision data --- that gap is what RH (\autoref{ch:capstones-riemann-hypothesis}) addresses.
\end{definition}

\begin{theorem}[Concrete \(\ZetaZerosUp\) naming certificate]
\label{thm:zeta-zeros-namecert}
The five fields above form a naming certificate over \(\Hist\) for the \(\zeta\)-zero interface \(\ZetaZerosUp\).
\end{theorem}
\begin{proof}[Sketch]
The five fields satisfy the closure laws of the naming-certificate framework (\autoref{ch:core-typed-naming-certificates}). The certificate uses no \(\mathsf{Classical.choice}\), no \(\mathsf{Quot.sound}\), and no \(\mathsf{propext}\), so the certificate's Lean targets fall under the project's axiom-purity gate.
\end{proof}

\begin{theorem}[Semantic \(\ZetaZerosUp\) naming certificate]
\label{thm:zeta-zeros-semantic-namecert}
The concrete certificate of \autoref{thm:zeta-zeros-namecert} satisfies the semantic naming-certificate schema of \(\NameCert\) (\autoref{ch:core-typed-naming-certificates}).
\end{theorem}
\begin{proof}[Sketch]
Direct verification of the four \(\NameCert\) field types using the constructions of this chapter.
\end{proof}
