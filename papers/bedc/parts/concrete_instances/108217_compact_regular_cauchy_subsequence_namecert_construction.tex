\begin{theorem}[Compact regular Cauchy subsequence NameCert obligations]
\label{thm:compact-regular-cauchy-subsequence-namecert-obligations}

\begin{theorem}[Compact regular Cauchy subsequence route]
\label{thm:compact-regular-cauchy-subsequence-route}

\begin{proof}
