\section{The certificate}

\begin{definition}[Automorphic adele graph carrier]
\label{def:automorphic-adele-graph-carrier}
For a history $g$, the predicate
$\mathsf{AutomorphicAdeleGraphCarrier}(g)$ holds exactly when there are
histories $domain,value:\mathsf{BHist}$ such that
$$
\mathsf{AdeleHistoryCarrier}(domain)\land
\mathsf{AdeleHistoryCarrier}(value)\land
\Cont(domain,value,g).
$$
\end{definition}
\leandef{BEDC.Derived.AutomorphicUp.AutomorphicAdeleGraphCarrier}

\begin{theorem}[Automorphic adele graph carrier hsame transport]
\label{thm:automorphic-adele-graph-carrier-hsame-transport}
If $\mathsf{AutomorphicAdeleGraphCarrier}(g)$ and $g\hsame g'$, then
$\mathsf{AutomorphicAdeleGraphCarrier}(g')$.
\end{theorem}
\begin{proof}
Unpack the graph-carrier witness for $g$ into histories
$domain,value:\mathsf{BHist}$, adele endpoint witnesses for both histories, and
a continuation witness $\Cont(domain,value,g)$.

Apply continuation-result $\hsame$ transport along $g\hsame g'$. This keeps the
hidden domain and value histories fixed and yields a transported continuation
witness $\Cont(domain,value,g')$.

Repack the same adele endpoint witnesses together with the transported
continuation ledger. The resulting witness is exactly
$\mathsf{AutomorphicAdeleGraphCarrier}(g')$.
\end{proof}
\leanchecked{BEDC.Derived.AutomorphicUp.AutomorphicAdeleGraphCarrier\_hsame\_transport}

  \item Traditional target: automorphic graph carrier transport.
  \item BEDC source: $\AdeleUp$ together with $\Cont$ over histories.

\label{thm:automorphic-adele-graph-semantic-namecert}

predicate, and ledger policy, forms a $\mathsf{SemanticNameCert}$ with
classifier $\hsame$.

the certificate uses it as its shared source, pattern, and ledger transport.
