The automorphic-form interface $\AutomorphicUp$ packages a function on $\NameCert_{\AdeleUp}$ (\autoref{ch:concrete-instances-adele-namecert}) invariant under a discrete subgroup action and satisfying the constructive growth conditions. It supplies the substrate connecting representations of the adelic group to L-functions in the Langlands programme.

\section{Carrier definition}
\label{sec:automorphic-carrier}

\begin{theorem}[Automorphic adele graph continuation is non-empty]
\label{thm:automorphic-adele-graph-cont-nonempty}
For adelic domain and value histories, any continuation graph produced by
\(\Cont\) is not \(\hsame\)-equal to \(\emp\).
\end{theorem}
\begin{proof}
Transport the continuation result to \(\emp\). Empty-result inversion forces the
domain endpoint to be \(\emp\), contradicting the non-empty property of the
\(\AdeleUp\) carrier.
\end{proof}
\leanchecked{BEDC.Derived.AutomorphicUp.AutomorphicAdeleGraph\_cont\_nonempty}

\begin{theorem}[Automorphic adele graph visible-context non-empty]
\label{thm:automorphic-adele-graph-visible-context-nonempty}
For adelic domain and value histories, a displayed graph inside a visible
left and right continuation context cannot be \(\hsame\)-equal to \(\emp\).
\end{theorem}
\begin{proof}
Peel the right visible context and then the left visible context using
continuation cancellation. The recovered central graph continuation is
non-empty by the automorphic adele graph continuation theorem.
\end{proof}
\leanchecked{BEDC.Derived.AutomorphicUp.AutomorphicAdeleGraph\_visible\_context\_nonempty}
