\begin{theorem}[Automorphic adele graph visible-context core is non-empty]
\label{thm:automorphic-adele-graph-visible-context-core-nonempty}
For adelic domain and value histories, if a visible two-sided context displays
\(\Cont(p \cdot d, v \cdot q, (p \cdot c) \cdot q)\), then the exposed core
\(c\) is not \(\hsame\)-equal to \(\emp\).
\end{theorem}
\begin{proof}
Suffix cancellation removes the visible right context \(q\), yielding
\(\Cont(p \cdot d, v, p \cdot c)\). Prefix cancellation removes the visible
left context \(p\), yielding \(\Cont(d,v,c)\). The core non-emptiness theorem
for automorphic adele graphs then rules out \(c\hsame\emp\).
\end{proof}

\label{thm:automorphic-adele-graph-visible-context-core-deterministic-nonempty}

\begin{proof}

two recovered cores, and automorphic adele graph non-emptiness rules out
emptiness for each core.

\begin{definition}[Automorphic adele graph carrier]
\label{def:automorphic-adele-graph-carrier}
For a history $g$, the predicate
$\mathsf{AutomorphicAdeleGraphCarrier}(g)$ holds exactly when there are
histories $domain,value:\mathsf{BHist}$ such that
$$
\mathsf{AdeleHistoryCarrier}(domain)\land
\mathsf{AdeleHistoryCarrier}(value)\land
\Cont(domain,value,g).
$$
\end{definition}
\leandef{BEDC.Derived.AutomorphicUp.AutomorphicAdeleGraphCarrier}

\begin{theorem}[Automorphic adele graph carrier hsame transport]
\label{thm:automorphic-adele-graph-carrier-hsame-transport}
If $\mathsf{AutomorphicAdeleGraphCarrier}(g)$ and $g\hsame g'$, then
$\mathsf{AutomorphicAdeleGraphCarrier}(g')$.
\end{theorem}
\begin{proof}
Unpack the graph-carrier witness for $g$ into histories
$domain,value:\mathsf{BHist}$, adele endpoint witnesses for both histories, and
a continuation witness $\Cont(domain,value,g)$.

Apply continuation-result $\hsame$ transport along $g\hsame g'$. This keeps the
hidden domain and value histories fixed and yields a transported continuation
witness $\Cont(domain,value,g')$.

Repack the same adele endpoint witnesses together with the transported
continuation ledger. The resulting witness is exactly
$\mathsf{AutomorphicAdeleGraphCarrier}(g')$.
\end{proof}
\leanchecked{BEDC.Derived.AutomorphicUp.AutomorphicAdeleGraphCarrier\_hsame\_transport}

  \item BEDC source: $\AdeleUp$ together with $\Cont$ over histories.

\label{thm:automorphic-adele-graph-semantic-namecert}

\begin{closurestatus}{\AutomorphicUp}
  \constructivestory{An automorphic graph starts with two admitted $\AdeleUp$ histories, one serving as the domain endpoint and one as the value endpoint, and the kernel continuation displays their graph history; without both adele carriers and the continuation witness, the history is outside the carrier. The classifier is ordinary history-sameness on these carried graph endpoints, with transport theorems peeling or replacing visible left and right contexts so that the underlying domain, value, and graph core remain determined. The source, pattern, and ledger are the same graph-carrier predicate: the source accepts exactly these adele-domain, adele-value, and continuation packages, the pattern records non-emptiness, context determinacy, endpoint readback, and result factorization, and the ledger keeps the hidden domain, value, continuation, and context witnesses available. Since the adele carrier is inhabited and nonempty, a displayed adele endpoint can be used on both sides to inhabit the graph carrier, while $\hsame$ supplies reflexivity, symmetry, and transitivity and continuation-result transport closes the carrier under classified graph endpoints.
