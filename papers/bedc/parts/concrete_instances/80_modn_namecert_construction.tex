\chapter{A Concrete Naming Certificate for $\ModNUp$}
\label{ch:concrete-instances-modn-namecert}

The modular-arithmetic interface $\ModNUp$ packages the carrier of integers modulo a positive natural $n$. It depends on $\NameCert_{\IntUp}$ (\autoref{ch:concrete-instances-int-namecert}) and is the canonical instance of $\NameCert_{\QuotientRingUp}$ (\autoref{ch:concrete-instances-quotientring-namecert}) at the ideal $n\mathbb{Z}$.

\begin{definition}[ModN quotient presentation]
\label{def:modn-nz-quotient-presentation}
Let $R_{\mathbb{Z}}$ be a displayed $\RingUp$ certificate for $\IntUp$ with carrier $C_{\mathbb{Z}}$, classifier $\sim_{\mathbb{Z}}$, addition $+_{\mathbb{Z}}$, additive inverse $-_{\mathbb{Z}}$, zero $0_{\mathbb{Z}}$, and multiplication $\cdot_{\mathbb{Z}}$. Let $n:\NatUp$ be positive, and let $I_n(x)$ be a certified $\IdealUp$ predicate over $R_{\mathbb{Z}}$ whose intended membership is $x\in n\mathbb{Z}$. The quotient presentation of $\ModNUp(n)$ is the specialization
$$
\ModNUp(n):=\QuotientRingUp(R_{\mathbb{Z}},I_n).
$$
Its public classifier is the ideal-coset classifier
$$
\mathsf{ModNClassifier}_n(a,b)\Longleftrightarrow a\equiv_{I_n}b,
$$
where $\equiv_{I_n}$ is the relation of \autoref{def:quotientring-ideal-coset-relation}. Thus, after unfolding the quotient-ring relation,
$$
\mathsf{ModNClassifier}_n(a,b)\Longleftrightarrow
C_{\mathbb{Z}}(a)\land C_{\mathbb{Z}}(b)\land I_n(a-_{\mathbb{Z}}b).
$$
\end{definition}

\begin{theorem}[ModN as the nZ quotient certificate]
\label{thm:modn-nz-quotient-certificate}
Under the hypotheses of \autoref{def:modn-nz-quotient-presentation}, the quotient-ring certificate at $(R_{\mathbb{Z}},I_n)$ supplies a $\NameCert$ for $\ModNUp(n)$ with exactly the quotient-ring classifier. More precisely,
$$
\NameCert_{\ModNUp(n)}=
\NameCert_{\QuotientRingUp(R_{\mathbb{Z}},I_n)}
$$
as a five-field certificate in the sense of \autoref{def:typed-naming-certificate}, and for all carried endpoints $a,b$,
$$
\mathsf{Classifier}_{\ModNUp(n)}(a,b)
\Longleftrightarrow
\mathsf{Classifier}_{\QuotientRingUp(R_{\mathbb{Z}},I_n)}(a,b)
\Longleftrightarrow
C_{\mathbb{Z}}(a)\land C_{\mathbb{Z}}(b)\land I_n(a-_{\mathbb{Z}}b).
$$
The additive, inverse, and multiplicative operations on $\ModNUp(n)$ are representative-independent by the same quotient-ring descent rows.
\end{theorem}
\begin{proof}
The positive $\NatUp$ witness selects the modulus parameter $n$ and the predicate $I_n$; it does not add a second quotient relation. The certified $\IdealUp$ hypothesis puts $I_n$ in exactly the scope required by \autoref{sec:ideal-certificate} and by the ideal-coset relation of \autoref{def:quotientring-ideal-coset-relation} over the displayed integer ring $R_{\mathbb{Z}}$.

Apply \autoref{thm:quotientring-ideal-coset-equivalence} with $R=R_{\mathbb{Z}}$ and $I=I_n$. This gives reflexivity, symmetry, and transitivity for the quotient classifier on carried integer endpoints. Carrier transport for the core $\NameCert$ of \autoref{def:lean-carrier-naming-certificate} is obtained by unfolding $a\equiv_{I_n}b$: the relation itself stores the two endpoint carrier witnesses $C_{\mathbb{Z}}(a)$ and $C_{\mathbb{Z}}(b)$.

The additive and inverse representative-independence rows are \autoref{thm:quotientring-additive-descent} specialized to $R_{\mathbb{Z}}$ and $I_n$. The multiplicative representative-independence row is \autoref{thm:quotientring-multiplication-descends-to-cosets} with the same specialization. Hence the quotient-ring source, pattern, classifier, stability, and ledger fields form the five fields demanded by \autoref{def:typed-naming-certificate}.

By \autoref{def:modn-nz-quotient-presentation}, $\ModNUp(n)$ is this quotient-ring presentation with no classifier-changing map inserted. Therefore every field of $\NameCert_{\ModNUp(n)}$ is the corresponding field of $\NameCert_{\QuotientRingUp(R_{\mathbb{Z}},I_n)}$. In particular, the classifier field is copied. The first displayed equivalence is this field identity, and the second is exactly \autoref{def:quotientring-ideal-coset-relation} unfolded at $R_{\mathbb{Z}}$ and $I_n$.
\end{proof}
\leanchecked{BEDC.Derived.ModNUp.ModN\_nz\_quotient\_certificate}

\begin{definition}[Concrete ModN BHist carrier and classifier]
\label{def:modn-concrete-bhist-carrier-classifier}
For a unary modulus history $m:\mathsf{BHist}$, define the concrete ModN
carrier and classifier by
$$
\mathsf{ModNCarrier}_m(h):=\mathsf{UnaryHistory}(m)\land
\mathsf{RingSingletonCarrier}(h)
$$
and
$$
\mathsf{ModNClassifier}_m(h,k):=
\mathsf{ModNCarrier}_m(h)\land
\mathsf{ModNCarrier}_m(k)\land
\mathsf{RingSingletonClassifier}(h,k).
$$
Equivalently, these are the paper names for the Lean objects
$\mathsf{ModNQuotientCarrier}(m,h)$ and
$\mathsf{ModNQuotientClassifier}(m,h,k)$.
\end{definition}

\begin{theorem}[Concrete ModN classifier rows]
\label{thm:modn-concrete-classifier-rows}
For each unary modulus history $m:\mathsf{BHist}$, the classifier
$\mathsf{ModNClassifier}_m$ satisfies the following concrete rows:
\begin{enumerate}
  \item if $\mathsf{ModNCarrier}_m(h)$, then
  $\mathsf{ModNClassifier}_m(h,h)$;
  \item if $\mathsf{ModNClassifier}_m(h,k)$, then
  $\mathsf{ModNClassifier}_m(k,h)$;
  \item if $\mathsf{ModNClassifier}_m(h,k)$ and
  $\mathsf{ModNClassifier}_m(k,l)$, then
  $\mathsf{ModNClassifier}_m(h,l)$;
  \item if $\mathsf{ModNClassifier}_m(h,k)$ and
  $\mathsf{ModNCarrier}_m(h)$, then $\mathsf{ModNCarrier}_m(k)$.
\end{enumerate}
\end{theorem}
\leanchecked{BEDC.Derived.ModNUp.ModNQuotientClassifier\_concrete\_rows}
\begin{proof}
Unfold \autoref{def:modn-concrete-bhist-carrier-classifier}. The modulus
component of every carrier witness is the same
$\mathsf{UnaryHistory}(m)$ proof, so no modulus data is transformed by the
classifier rows.

For reflexivity, keep the carrier witness for $h$ on both endpoints and use
the reflexivity row of the singleton $\RingUp$ classifier on the
$\mathsf{RingSingletonCarrier}(h)$ component.

For symmetry and transitivity, project the two endpoint carriers from the
given $\mathsf{ModNClassifier}_m$ hypotheses. The remaining relation fields
are exactly the symmetry and transitivity rows for
$\mathsf{RingSingletonClassifier}$, then the original unary modulus witness is
repacked into each endpoint carrier.

Carrier transport is a projection: the right endpoint carrier is already a
field of $\mathsf{ModNClassifier}_m(h,k)$.
\end{proof}

\begin{theorem}[Concrete ModN singleton operation descent rows]
\label{thm:modn-singleton-operation-descent-rows}
For each unary modulus history $m:\mathsf{BHist}$, the singleton ring
operations preserve $\mathsf{ModNCarrier}_m$ and descend through
$\mathsf{ModNClassifier}_m$:
\begin{enumerate}
  \item if $\mathsf{ModNCarrier}_m(h)$ and
  $\mathsf{ModNCarrier}_m(k)$, then
  $\mathsf{ModNCarrier}_m(\mathsf{RingSingletonAdd}(h,k))$ and
  $\mathsf{ModNCarrier}_m(\mathsf{RingSingletonMul}(h,k))$;
  \item if $\mathsf{ModNCarrier}_m(h)$, then
  $\mathsf{ModNCarrier}_m(\mathsf{RingSingletonNeg}(h))$;
  \item if $\mathsf{ModNClassifier}_m(h,h')$ and
  $\mathsf{ModNClassifier}_m(k,k')$, then
  $\mathsf{ModNClassifier}_m(\mathsf{RingSingletonAdd}(h,k),
  \mathsf{RingSingletonAdd}(h',k'))$;
  \item under the same classifier hypotheses,
  $\mathsf{ModNClassifier}_m(\mathsf{RingSingletonMul}(h,k),
  \mathsf{RingSingletonMul}(h',k'))$;
  \item if $\mathsf{ModNClassifier}_m(h,h')$, then
  $\mathsf{ModNClassifier}_m(\mathsf{RingSingletonNeg}(h),
  \mathsf{RingSingletonNeg}(h'))$.
\end{enumerate}
\end{theorem}
\leanchecked{BEDC.Derived.ModNUp.ModNQuotientCarrier\_singleton\_operation\_descent\_rows}
\begin{proof}
Unfold the ModN carrier. Preservation of the modulus component is immediate:
each output carrier keeps the same $\mathsf{UnaryHistory}(m)$ witness carried by
the input endpoints.

The $\RingUp$ singleton operation laws supply the history component. Addition,
negation, and multiplication all return the singleton ring endpoint, so the
$\mathsf{RingSingletonCarrier}$ field is obtained from the singleton closure
rows of \autoref{thm:singleton-empty-history-ring-laws}.

For descent, project the $\mathsf{RingSingletonClassifier}$ fields from the two
ModN classifier hypotheses. The singleton $\RingUp$ congruence rows identify
the add, multiply, and negated outputs under
$\mathsf{RingSingletonClassifier}$.

Finally, pair those singleton classifier witnesses with the output carriers
just constructed. This repacks the result in the definition of
$\mathsf{ModNClassifier}_m$.
\end{proof}

\begin{theorem}[Concrete ModN NameCert scope]
\label{thm:modn-concrete-namecert-scope}
For every unary modulus history $m:\mathsf{BHist}$, the carrier
$\mathsf{ModNCarrier}_m$, classifier $\mathsf{ModNClassifier}_m$, and singleton
ring operations $\mathsf{RingSingletonAdd}$, $\mathsf{RingSingletonNeg}$, and
$\mathsf{RingSingletonMul}$ form the concrete $\mathsf{BHist}$ scope consumed by
\autoref{thm:modn-nz-quotient-certificate}. In that scope the quotient
certificate reads exactly the carrier rows of
\autoref{thm:modn-concrete-classifier-rows} and the operation rows of
\autoref{thm:modn-singleton-operation-descent-rows}.
\end{theorem}
\begin{proof}
Begin with a unary modulus witness for $m$ and unfold
\autoref{def:modn-concrete-bhist-carrier-classifier}. The source, pattern, and
ledger carriers are all the same predicate $\mathsf{ModNCarrier}_m$, so the
core certificate fields are supplied by
\autoref{thm:modn-concrete-classifier-rows}.

Next use \autoref{thm:modn-singleton-operation-descent-rows} to fix the
additive, inverse, and multiplicative output obligations. These rows ensure
that operations remain inside the concrete carrier and that replacing
classified representatives does not change the classified output.

Only after those rows are in place, apply
\autoref{thm:modn-nz-quotient-certificate}. The checked quotient certificate
then packages the already fixed carrier, classifier, stability, and ledger
fields as the displayed $\mathsf{NameCert}$ scope for $\ModNUp(m)$.
\end{proof}
