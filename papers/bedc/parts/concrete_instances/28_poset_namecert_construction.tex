\chapter{A Concrete Naming Certificate for \(\POSetUp\)}
\label{ch:concrete-instances-poset-namecert}

The partially ordered set interface \(\POSetUp\) extends \autoref{ch:concrete-instances-preorder-namecert} with the antisymmetry field: if \(a \le b\) and \(b \le a\) then \(a\) and \(b\) are classified the same by the carrier. The integer order from \autoref{ch:concrete-instances-int-namecert} is the canonical concrete instance.

\begin{remark}[\(\POSetUp\) interface scope]
\label{rem:concrete-instances-poset-namecert-scope}
This chapter records the \(\POSetUp\) licensing target. The concrete prefix
antisymmetry theorem is linked to the unary preorder classifier, and the
certificate fields are specified below.
\end{remark}

\section{Carrier definition}

\begin{definition}[POSet carrier]
\label{def:poset-carrier}
Use a preorder setup extended with antisymmetry under the carrier classifier.
\end{definition}

\section{Concrete unary prefix POSet instance}
\label{sec:poset-concrete-unary-prefix-instance}

\begin{definition}[Concrete unary prefix POSet instance]
\label{def:concrete-unary-prefix-poset-instance}
The concrete unary prefix POSet instance is the following ten-field tuple.
\begin{enumerate}
  \item The traditional target is the finite unary-word prefix partial order.
  \item The BEDC source is $\mathsf{BHist}$ restricted to unary histories.
  \item The carrier is $\mathsf{PreorderCarrier}=\mathsf{UnaryHistory}$.
  \item The classifier is the carrier-aware specification $\mathsf{PreorderCarrierClassifierSpec}(\mathsf{PreorderCarrier})$, whose equality content is $\hsame$.
  \item The relation is $\mathsf{PreorderPrefixLE}$.
  \item Stability consists of classifier reflexivity, classifier transitivity, relation reflexivity, relation transitivity, relation congruence, and antisymmetry under the classifier.
  \item The ledger records both comparison-tail witnesses used for $h\le k$ and $k\le h$.
  \item The core theorem list is the concrete prefix stability certificate together with prefix antisymmetry.
  \item Exactness is the bidirectional prefix collapse: two opposing prefix comparisons collapse exactly to the classifier relation.
  \item The standard bridge identifies $\mathsf{PreorderPrefixLE}$ with the initial-segment order on finite unary words, and the status is $\mathsf{ProofTarget}$.
\end{enumerate}
\end{definition}

\begin{theorem}[Concrete unary prefix POSet laws]
\label{thm:concrete-unary-prefix-poset-laws}
The tuple in \autoref{def:concrete-unary-prefix-poset-instance} satisfies the $\POSetUp$ carrier field, the preorder fields for $\mathsf{PreorderPrefixLE}$, classifier congruence for $\hsame$, and antisymmetry under the carrier classifier.
\end{theorem}
\begin{proof}
The carrier and classifier fields are the concrete preorder fields: the carrier is $\mathsf{PreorderCarrier}=\mathsf{UnaryHistory}$, and the classifier is $\mathsf{PreorderCarrierClassifierSpec}(\mathsf{PreorderCarrier})$.

The preorder and classifier-congruence fields are supplied by the concrete prefix stability certificate in \autoref{def:preorder-prefix-stability-certificate}. Equivalently, \autoref{thm:preorder-prefix-order-laws-hsame} records the reflexive and transitive prefix laws together with the history-sameness classifier boundary.

It remains to supply the partial-order field. Apply \autoref{thm:poset-prefix-antisymmetry}: opposing $\mathsf{PreorderPrefixLE}$ witnesses provide the two comparison tails, and the theorem collapses them to $\hsame(h,k)$, which is exactly antisymmetry under the carrier classifier.
\end{proof}

\section{The certificate}

\begin{definition}[POSet source specification]
\label{def:poset-source-specification}
The source is a certified preorder carrier plus an antisymmetry field.
\end{definition}

\begin{definition}[POSet pattern specification]
\label{def:poset-pattern-specification}
The pattern is preorder relation formation with classifier collapse from bidirectional comparison.
\end{definition}

\begin{definition}[POSet classifier specification]
\label{def:poset-classifier-specification}
The classifier is the preorder carrier classifier.
\end{definition}

\begin{theorem}[Prefix antisymmetry]
\label{thm:poset-prefix-antisymmetry}
For the concrete unary history prefix preorder, if \(h\le k\) and \(k\le h\),
then \(h\) and \(k\) are classified the same by the preorder classifier.
\end{theorem}
\leanvariant{BEDC.Derived.PreorderUp.PreorderPrefixLE\_antisymm\_hsame}
\begin{proof}
Unpack the two prefix witnesses. Substituting the first continuation into the
second gives a loop from \(h\) through the appended forward and backward tails.
Right-unit uniqueness and append-empty inversion force the forward tail to be
\(\emp\), so the first continuation reduces to \(\hsame(h,k)\), which is the
classifier specification.
\end{proof}

\begin{theorem}[Prefix classifier exactness for concrete POSet]
\label{thm:poset-prefix-classifier-exactness}
For carrier histories $h,k$ in $\mathsf{PreorderCarrier}$, the carrier-aware
classifier relation
$\mathsf{PreorderCarrierClassifierSpec}(\mathsf{PreorderCarrier})(h,k)$ holds
exactly when $\mathsf{PreorderPrefixLE}(h,k)$ and
$\mathsf{PreorderPrefixLE}(k,h)$ both hold.
\end{theorem}
\begin{proof}
Assume first that the carrier-aware classifier holds. Its equality content is
$\hsame(h,k)$, and symmetry gives $\hsame(k,h)$. Applying
\autoref{thm:preorder-prefix-le-of-hsame} in the two orientations gives
$\mathsf{PreorderPrefixLE}(h,k)$ and $\mathsf{PreorderPrefixLE}(k,h)$.

Conversely, assume both prefix comparisons and keep the carrier hypotheses for
$h$ and $k$. By \autoref{thm:poset-prefix-antisymmetry}, the opposing
comparisons collapse to $\hsame(h,k)$. Combining this equality content with the
two carrier witnesses reassembles
$\mathsf{PreorderCarrierClassifierSpec}(\mathsf{PreorderCarrier})(h,k)$.
\end{proof}

Thus \autoref{thm:poset-prefix-classifier-exactness} supplies the exactness
predicate named in item 9 of
\autoref{def:concrete-unary-prefix-poset-instance}.

\begin{definition}[POSet stability certificate]
\label{def:poset-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item all preorder stability fields
  \item antisymmetry: \(a\le b\) and \(b\le a\) imply \(a\sim_C b\)
  \item compatibility of antisymmetry with relation congruence
\end{enumerate}
\end{definition}

\begin{definition}[POSet ledger policy]
\label{def:poset-ledger-policy}
Record both comparison derivations when antisymmetry is used.
\end{definition}

\begin{definition}[\(\POSetUp\) certificate obligations]
\label{def:poset-certificate-obligations}
The carrier, source, pattern, classifier, stability, and ledger data listed above are the obligations required for the interface name \(\POSetUp\).
\end{definition}

\paragraph{Primitive boundary.}
\(\POSetUp\) is not a BEDC primitive. It is admitted only when those obligations are supplied.
