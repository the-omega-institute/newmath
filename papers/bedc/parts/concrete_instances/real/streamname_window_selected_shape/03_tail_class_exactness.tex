\begin{theorem}[Real unary StreamNameUp finite-window selected canonical e-one tail full package]
\label{thm:real-unary-streamname-finite-window-selected-canonical-e1-tail-full-package}

$\alpha,\omega,k,z_s,z_t,d_s,d_t:\Hist$, assume

\mathsf{RatStreamNameCarrier}(s),\quad
\mathsf{RatStreamNameCarrier}(t),\quad
\mathsf{RatStreamNameClassifier}(s,t),

$$
\mathsf{UnaryHistory}(\alpha),\quad
\mathsf{UnaryHistory}(\omega),\quad
\mathsf{UnaryOffsetLe}(k,\omega).
$$
Then there exist histories $u,v:\Hist$ such that the selected endpoints carry
the complete displayed e-one certificate and every other selected e-one display

$$
\begin{array}{l}
\exists u,v:\Hist,
\hsame(s(\mathsf{append}(\alpha,k)),\Eone(u))\land
\hsame(t(\mathsf{append}(\alpha,k)),\Eone(v))\land\\
\mathsf{RealUnaryStreamWindowClassifier}(s,t,\alpha,\omega)\land\\
\mathsf{RatHistoryClassifier}(s(\mathsf{append}(\alpha,k)),t(\mathsf{append}(\alpha,k)))\land\\
\mathsf{RatHistoryClassifier}(\Eone(u),\Eone(v))\land\\
\mathsf{UnaryHistory}(u)\land\mathsf{UnaryHistory}(v)\land\hsame(u,v)\land\\
\mathsf{PositiveUnaryDenominator}(s(\mathsf{append}(\alpha,k)))\land
\mathsf{PositiveUnaryDenominator}(t(\mathsf{append}(\alpha,k)))\land\\
\mathsf{UnaryHistory}(s(\mathsf{append}(\alpha,k)))\land
\mathsf{UnaryHistory}(t(\mathsf{append}(\alpha,k)))\land\\

(\hsame(t(\mathsf{append}(\alpha,k)),\emp)\to\mathsf{False})\land\\
(\hsame(s(\mathsf{append}(\alpha,k)),\Ezero(z_s))\to\mathsf{False})\land
(\hsame(t(\mathsf{append}(\alpha,k)),\Ezero(z_t))\to\mathsf{False})\land\\
(\hsame(s(\mathsf{append}(\alpha,k)),d_s\cdot\Ezero(z_s))\to
\mathsf{False})\land\\

\forall u',v':\Hist,\,
\bigl(\hsame(s(\mathsf{append}(\alpha,k)),\Eone(u'))\land

\qquad
\mathsf{UnaryHistory}(u')\land
\mathsf{UnaryHistory}(v')\land
\hsame(u,u')\land
\hsame(v,v')\land\\
\qquad
\hsame(u',v)\land
\hsame(u,v').
\end{array}

\autoref{thm:real-unary-streamname-finite-window-selected-e1-tail-coverage-package}
with the displayed histories $z_s,z_t,d_s,d_t$. The resulting witnesses $u$ and

\autoref{thm:real-unary-streamname-finite-window-selected-e1-tail-pairwise-coherence}
with $a=u$, $a'=u'$, $b=v$, and $b'=v'$. The required displayed endpoint
comparisons are exactly the two chosen displays and the two arbitrary displays.

$\mathsf{UnaryHistory}(u')$, $\mathsf{UnaryHistory}(v')$, $\hsame(u,u')$,
$\hsame(v,v')$, $\hsame(u',v)$, and $\hsame(u,v')$. Repack this universal
coherence clause after the full coverage rows and package $u,v$ as the displayed
canonical full witnesses.
\end{proof}
