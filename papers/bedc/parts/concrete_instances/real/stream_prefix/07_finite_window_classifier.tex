\begin{definition}[Real unary offset bound]
\label{def:real-unary-offset-bound}
For histories $k,w:\Hist$, the unary offset bound is the BEDC continuation
witness that $k$ is a certified unary observation lying below the certified
unary bound $w$:
$$
\mathsf{UnaryOffsetLe}(k,w)
\Longleftrightarrow
\mathsf{UnaryHistory}(k)\land
\exists \tau:\Hist,\ \mathsf{UnaryHistory}(\tau)\land \ContR(k,\tau,w).
$$
\end{definition}

\begin{definition}[Real unary stream finite window classifier]
\label{def:real-unary-stream-window-classifier}
For observation maps $s,t:\Hist\to\Hist$ and histories $a,w:\Hist$, read $a$
as the anchor and $w$ as the bound. The finite window classifier is the
pointwise RatUp classifier over every certified unary offset below $w$:
$$
\mathsf{RealUnaryStreamWindowClassifier}(s,t,a,w)
\Longleftrightarrow
\mathsf{UnaryHistory}(a)\land \mathsf{UnaryHistory}(w)\land
\forall k,\ \mathsf{UnaryOffsetLe}(k,w)\to
\mathsf{RatHistoryClassifier}(
s(\mathsf{append}(a,k)),t(\mathsf{append}(a,k))).
$$
\end{definition}

\begin{theorem}[Real unary stream finite-window coverage]
\label{thm:real-unary-stream-window-coverage}
For observation maps $s,t:\Hist\to\Hist$ and histories $a,w:\Hist$,
$$
\mathsf{RealUnaryStreamClassifier}(s,t)\land
\mathsf{UnaryHistory}(a)\land \mathsf{UnaryHistory}(w)
\Longrightarrow
\mathsf{RealUnaryStreamWindowClassifier}(s,t,a,w).
$$
\end{theorem}
\begin{proof}
Unfold Definition~\ref{def:real-unary-stream-window-classifier}. The anchor and
bound fields are exactly the displayed assumptions $\mathsf{UnaryHistory}(a)$
and $\mathsf{UnaryHistory}(w)$.

Fix an offset $k$ and assume $\mathsf{UnaryOffsetLe}(k,w)$. Unfold
Definition~\ref{def:real-unary-offset-bound}; the offset witness projects
$\mathsf{UnaryHistory}(k)$ together with a unary tail whose continuation reaches
the bound.

Apply unary append closure to $\mathsf{UnaryHistory}(a)$ and
$\mathsf{UnaryHistory}(k)$ to obtain
$\mathsf{UnaryHistory}(\mathsf{append}(a,k))$. Evaluating
$\mathsf{RealUnaryStreamClassifier}(s,t)$ at this certified observation gives
$$
\mathsf{RatHistoryClassifier}(
s(\mathsf{append}(a,k)),t(\mathsf{append}(a,k))).
$$

Since the offset was arbitrary under $\mathsf{UnaryOffsetLe}(k,w)$, these
pointwise endpoint classifiers repack into
$\mathsf{RealUnaryStreamWindowClassifier}(s,t,a,w)$.
\end{proof}

\label{thm:real-stream-window-selected-full-shape-readback-package}
For rational-observation streams $x,y:\mathbb{N}\to\Hist$, an anchor index

\begin{enumerate}
\item $\mathsf{RatHistoryClassifier}(\Eone(a),\Eone(b))$;
\item $\mathsf{UnaryHistory}(a)\land \mathsf{UnaryHistory}(b)\land

\item $\mathsf{PositiveUnaryDenominator}(x(n+k))\land
\mathsf{PositiveUnaryDenominator}(y(n+k))$;

\item $(\hsame(x(n+k),\Ezero(z_X))\to\mathsf{False})\land

\begin{proof}

$\mathsf{RatHistoryClassifier}(\Eone(a),\Eone(b))$, and
\autoref{thm:rat-history-classifier-e1-tail-unary-iff} reads back the two
unary tails together with $\hsame(a,b)$.

Apply \autoref{thm:rat-history-classifier-positive-denominators} to the
original endpoint classifier. This gives positive unary denominator witnesses
for $x(n+k)$ and $y(n+k)$, and
\autoref{thm:positive-unary-denominator-unary-and-nonempty} converts those
witnesses into unary and nonempty endpoint rows.

For the left zero-extension test, transport
$\mathsf{PositiveUnaryDenominator}(x(n+k))$ along the proposed comparison
$\hsame(x(n+k),\Ezero(z_X))$ by
\autoref{thm:rat-positive-unary-denominator-hsame-transport}. The resulting
positive denominator on $\Ezero(z_X)$ contradicts
\autoref{thm:rat-positive-denominator-e0-absurd}.

For the right zero-extension test, repeat the same transport from $y(n+k)$ to
$\Ezero(z_Y)$. The positive-denominator zero-head absurdity theorem gives the

\begin{lemma}[Unary offset bound transitivity]
\label{lem:real-unary-offset-bound-transitivity}
For histories $k,w,u:\Hist$,
$$
\mathsf{UnaryOffsetLe}(k,w)\land \mathsf{UnaryOffsetLe}(w,u)
\Longrightarrow
\mathsf{UnaryOffsetLe}(k,u).
$$
\end{lemma}
\begin{proof}
Unfold Definition~\ref{def:real-unary-offset-bound} in the first offset row.
It gives $\mathsf{UnaryHistory}(k)$ and a unary tail $\tau:\Hist$ with
$\ContR(k,\tau,w)$.

Unfold the second offset row. It gives a unary tail $\sigma:\Hist$ with
$\ContR(w,\sigma,u)$, and it also certifies the intermediate bound $w$ as a
unary history.

Apply unary append closure to the two tails to obtain
$\mathsf{UnaryHistory}(\mathsf{append}(\tau,\sigma))$. Continuation
associativity composes $\ContR(k,\tau,w)$ with $\ContR(w,\sigma,u)$, and
determinacy identifies the composed endpoint with $u$.

Thus $k$ has the unary tail $\mathsf{append}(\tau,\sigma)$ whose continuation
reaches $u$, so Definition~\ref{def:real-unary-offset-bound} repacks this data
as $\mathsf{UnaryOffsetLe}(k,u)$.
\end{proof}

\mathsf{RealUnaryStreamWindowClassifier}(s,t,a,w).

\begin{proof}

Definition~\ref{def:real-unary-offset-bound}.

\autoref{lem:real-unary-offset-bound-transitivity} to
$\mathsf{UnaryOffsetLe}(k,w)$ and $\mathsf{UnaryOffsetLe}(w,u)$, obtaining

$$
\mathsf{RatHistoryClassifier}(

\label{def:real-stream-finite-window-classifier}
For rational-observation streams $x,y:\mathbb{N}\to\Hist$, an anchor index $n$,

pointwise rational-history classifier
$$
\mathsf{RealStreamWindowClassifier}(x,y,n,m)

\label{thm:real-stream-classifier-finite-window-coverage}

\begin{theorem}[Real stream finite-window stability package]
\label{thm:real-stream-window-classifier-stability-package}
For rational-observation streams $x,x',y,y',z:\mathbb{N}\to\Hist$ and indices
$n,m$, the finite window classifier supplies the following stability rows:
\begin{enumerate}
  \item if
  $\forall k,\ k\le m\Longrightarrow\mathsf{RatHistoryCarrier}(x(n+k))$,
  then $\mathsf{RealStreamWindowClassifier}(x,x,n,m)$;
  \item if $\mathsf{RealStreamWindowClassifier}(x,y,n,m)$, then
  $\mathsf{RealStreamWindowClassifier}(y,x,n,m)$;
  \item if $\mathsf{RealStreamWindowClassifier}(x,y,n,m)$ and
  $\mathsf{RealStreamWindowClassifier}(y,z,n,m)$, then
  $\mathsf{RealStreamWindowClassifier}(x,z,n,m)$;
  \item if $\mathsf{RealStreamWindowClassifier}(x,y,n,m)$ and, for every
  $k\le m$, both $\hsame(x(n+k),x'(n+k))$ and
  $\hsame(y(n+k),y'(n+k))$ hold, then
  $\mathsf{RealStreamWindowClassifier}(x',y',n,m)$.
\end{enumerate}
\end{theorem}
\begin{proof}
Unfold the finite window classifier in each row and fix an offset $k\le m$.

\begin{enumerate}
  \item The reflexivity row reads the carrier hypothesis at this offset. Pair
  the carrier with itself, and use $\hsame$ reflexivity to build
  $\mathsf{RatHistoryClassifier}(x(n+k),x(n+k))$.

  \item The symmetry row evaluates the source window at the offset and applies
  \autoref{thm:rat-history-classifier-symmetry} to reverse the rational-history
  classifier.

  \item The transitivity row evaluates the two source windows at the same
  offset and applies \autoref{thm:rat-history-classifier-trans} to compose the
  two rational-history classifiers through the shared middle stream $y$.

  \item The transport row evaluates the source window at the offset and reads
  the two pointwise $\hsame$ hypotheses at the same offset. Apply
  \autoref{thm:rat-history-classifier-hsame-transport} to transport the
  rational-history classifier to the endpoints $x'(n+k)$ and $y'(n+k)$.
\end{enumerate}

Since the offset was arbitrary under the bound, each row repacks into the
corresponding $\mathsf{RealStreamWindowClassifier}$ statement.
\end{proof}

The ten-field fit is: traditional target finite observation-window stability;
BEDC source RealUp rational-observation streams plus RatUp classifier; carrier
the bounded window predicate; classifier
$\mathsf{RealStreamWindowClassifier}$; stability the four displayed rows;
ledger anchor $n$, radius $m$, and offset witnesses; core theorem this block;
exactness bounded pointwise rational classification; standard bridge finite
