\begin{definition}[Real stream finite window classifier]
\label{def:real-stream-finite-window-classifier}
For rational-observation streams $x,y:\mathbb{N}\to\Hist$, an anchor index $n$,
and a zero-based window radius $m$, the finite window classifier is the bounded
pointwise rational-history classifier
$$
\mathsf{RealStreamWindowClassifier}(x,y,n,m)
\Longleftrightarrow
\forall k,\ k\le m\Longrightarrow
\mathsf{RatHistoryClassifier}(x(n+k),y(n+k)).
$$
\end{definition}

\begin{theorem}[Real stream window classifier hsame transport]
\label{thm:real-stream-window-classifier-hsame-transport}
For rational-observation streams $x,y,x',y':\mathbb{N}\to\Hist$, an anchor
index $n$, and a radius $m$, assume
$\mathsf{RealStreamWindowClassifier}(x,y,n,m)$. If every bounded offset
$k\le m$ carries both endpoint rows
$x(n+k)\hsame x'(n+k)$ and $y(n+k)\hsame y'(n+k)$, then
$\mathsf{RealStreamWindowClassifier}(x',y',n,m)$.
\end{theorem}
\begin{proof}
Unfold the finite window classifier and fix an arbitrary offset $k\le m$. The
source window hypothesis supplies the endpoint classifier
$\mathsf{RatHistoryClassifier}(x(n+k),y(n+k))$.

The offset transport hypothesis gives the two rows
$x(n+k)\hsame x'(n+k)$ and $y(n+k)\hsame y'(n+k)$ at the same bounded
observation. Apply
\autoref{thm:rat-history-classifier-hsame-transport} to the source rational
classifier and these endpoint rows.

The result is
$\mathsf{RatHistoryClassifier}(x'(n+k),y'(n+k))$. Since the offset was
arbitrary under the bound $k\le m$, these transported endpoint classifiers
repack into $\mathsf{RealStreamWindowClassifier}(x',y',n,m)$.
\end{proof}

\begin{theorem}[Real stream prefix-window exactness at zero]
\label{thm:real-stream-prefix-window-exactness-at-zero}
For rational-observation streams $x,y:\mathbb{N}\to\Hist$ and every finite
offset $m$,
$$
\mathsf{RealStreamPrefixClassifier}(x,y,m)
\Longleftrightarrow
\mathsf{RealStreamWindowClassifier}(x,y,0,m).
$$
\end{theorem}
\begin{proof}
Induct on $m$. At $m=0$, the recursive prefix classifier is exactly the
rational-history classifier at observation $0$, and the window condition has
only the offset $k=0$.

For the successor step, use
\autoref{lem:real-stream-prefix-successor-inversion} to split a prefix through
$m+1$ into the prefix through $m$ and the rational-history classifier at the new
endpoint. The induction hypothesis converts the shorter prefix into the window
condition through offset $m$.

Given any $k\le m+1$, split constructively between $k\le m$ and $k=m+1$. The
first case is handled by the shorter window; the endpoint case is exactly the
successor rational-history classifier.

Conversely, assume the window condition through $m+1$. Its restriction to
offsets $k\le m$ gives the zero-based window through $m$, hence by induction the
prefix through $m$. Taking $k=m+1$ gives the endpoint classifier, and the reverse
direction of successor inversion pairs these two witnesses into the prefix
through $m+1$.
\end{proof}

\begin{theorem}[Real stream classifier finite-window coverage]
\label{thm:real-stream-classifier-finite-window-coverage}
For rational-observation streams $x,y:\mathbb{N}\to\Hist$ and indices $n,m$,
every pointwise stream classifier covers the finite observation window anchored
at $n$ with radius $m$:
$$
\mathsf{RealStreamClassifier}(x,y)
\Longrightarrow
\mathsf{RealStreamWindowClassifier}(x,y,n,m).
$$
\end{theorem}
\begin{proof}
Assume $\mathsf{RealStreamClassifier}(x,y)$ and fix an offset $k\le m$. The
stream classifier is pointwise, so evaluating it at the observation index $n+k$
gives $\mathsf{RatHistoryClassifier}(x(n+k),y(n+k))$.

Since the chosen offset was arbitrary under the window bound, these endpoint
classifiers assemble exactly into
$\mathsf{RealStreamWindowClassifier}(x,y,n,m)$.
\end{proof}

\begin{theorem}[Real stream window selected full shape readback package]
\label{thm:real-stream-window-selected-full-shape-readback-package}
For rational-observation streams $x,y:\mathbb{N}\to\Hist$, an anchor index
$n$, a radius $m$, a bounded offset $k$ with $k\le m$, and histories
$a,b,z_X,z_Y:\Hist$, assume
$\mathsf{RealStreamWindowClassifier}(x,y,n,m)$,
$\hsame(x(n+k),\Eone(a))$, and $\hsame(y(n+k),\Eone(b))$. Then the selected
endpoints at $n+k$ carry the following package:
\begin{enumerate}
\item $\mathsf{RatHistoryClassifier}(\Eone(a),\Eone(b))$;
\item $\mathsf{UnaryHistory}(a)\land \mathsf{UnaryHistory}(b)\land
\hsame(a,b)$;
\item $\mathsf{PositiveUnaryDenominator}(x(n+k))\land
\mathsf{PositiveUnaryDenominator}(y(n+k))$;
\item $\mathsf{UnaryHistory}(x(n+k))\land
\mathsf{UnaryHistory}(y(n+k))\land
(\hsame(x(n+k),\emp)\to\mathsf{False})\land
(\hsame(y(n+k),\emp)\to\mathsf{False})$;
\item $(\hsame(x(n+k),\Ezero(z_X))\to\mathsf{False})\land
(\hsame(y(n+k),\Ezero(z_Y))\to\mathsf{False})$.
\end{enumerate}
\end{theorem}
\begin{proof}
Unfold the finite window classifier at the selected offset. The bound $k\le m$
turns the window hypothesis into
$\mathsf{RatHistoryClassifier}(x(n+k),y(n+k))$.

Transport this rational classifier across the two displayed one-seal endpoint
comparisons by \autoref{thm:rat-history-classifier-hsame-transport}. The
transported classifier is
$\mathsf{RatHistoryClassifier}(\Eone(a),\Eone(b))$, and
\autoref{thm:rat-history-classifier-e1-tail-unary-iff} reads back the two
unary tails together with $\hsame(a,b)$.

Apply \autoref{thm:rat-history-classifier-positive-denominators} to the
original endpoint classifier. This gives positive unary denominator witnesses
for $x(n+k)$ and $y(n+k)$, and
\autoref{thm:positive-unary-denominator-unary-and-nonempty} converts those
witnesses into unary and nonempty endpoint rows.

For the left zero-extension test, transport
$\mathsf{PositiveUnaryDenominator}(x(n+k))$ along the proposed comparison
$\hsame(x(n+k),\Ezero(z_X))$ by
\autoref{thm:rat-positive-unary-denominator-hsame-transport}. The resulting
positive denominator on $\Ezero(z_X)$ contradicts
\autoref{thm:rat-positive-denominator-e0-absurd}.

For the right zero-extension test, repeat the same transport from $y(n+k)$ to
$\Ezero(z_Y)$. The positive-denominator zero-head absurdity theorem gives the
second contradiction, completing the listed package.
\end{proof}
