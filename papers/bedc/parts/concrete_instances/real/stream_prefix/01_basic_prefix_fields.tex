\begin{theorem}[Real stream classifier prefix closure]
\label{thm:real-stream-classifier-prefix}
If rational-observation streams are pointwise related by the rational history classifier, then every finite observation prefix is related by the recursively accumulated prefix classifier.
\end{theorem}
\begin{proof}
Induct on the selected prefix length. The base case is the pointwise classifier at zero. The successor case pairs the induction hypothesis for the shorter prefix with the pointwise classifier at the successor observation.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealStreamClassifier\_prefix}

\begin{theorem}[Real stream prefix classifier hsame transport]
\label{thm:real-stream-prefix-classifier-hsame-transport}
If two rational-observation streams are related pointwise by $\hsame$ on both
sides, then every classified finite prefix transports to the pointwise-related
streams.
\end{theorem}
\begin{proof}
Induct on the prefix length. The base observation is rational classifier
transport at index $0$. At a successor index, the induction hypothesis
transports the shorter prefix and rational classifier transport handles the
successor observation.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealStreamPrefixClassifier\_hsame\_transport}

\begin{theorem}[Real stream prefix unary context closure]
\label{thm:real-stream-prefix-unary-context-closed}
If every prefix-side context and every tail-side context is unary pointwise, and
the two displayed context streams are pointwise $\hsame$ on each side, then any
classified finite real-stream prefix remains classified after each endpoint is
wrapped by the corresponding unary prefix and tail contexts.
\end{theorem}
\begin{proof}
Induct on the finite prefix length. At the base observation, apply rational
history classifier unary-context closure to the endpoint classifier and the
four pointwise context hypotheses at index $0$. At a successor index, keep the
inductively transported shorter prefix and apply the same rational
unary-context closure theorem to the successor endpoint classifier.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealStreamPrefixClassifier\_unary\_context\_closed}

\begin{theorem}[Real stream prefix classifier reflexivity]
\label{thm:real-stream-prefix-classifier-refl}
For every rational-observation stream $x:\mathbb{N}\to\Hist$, if
$\mathsf{RatHistoryCarrier}(x(i))$ for every observation index $i$, then
$\mathsf{RealStreamPrefixClassifier}(x,x,n)$ for every prefix index $n$.
\end{theorem}
\begin{proof}
Induct on $n$. At $0$, the carrier witness for $x(0)$ supplies the rational
history classifier for the endpoint, with $\hsame$ reflexivity providing the
identity transport on both sides.

At a successor index, pair the induction hypothesis for the shorter prefix with
the same rational classifier construction at the new endpoint $n+1$.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealStreamPrefixClassifier\_refl}

\begin{theorem}[Real stream prefix classifier symmetry]
\label{thm:real-stream-prefix-classifier-symm}
For rational-observation streams $x,y:\mathbb{N}\to\Hist$, if
$\mathsf{RealStreamPrefixClassifier}(x,y,n)$, then
$\mathsf{RealStreamPrefixClassifier}(y,x,n)$.
\end{theorem}
\begin{proof}
Induct on $n$. At $0$, apply rational-history classifier symmetry to the base
endpoint classifier.

At a successor index, reverse the shorter prefix by the induction hypothesis and
reverse the displayed successor endpoint by rational-history classifier
symmetry.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealStreamPrefixClassifier\_symm}

\begin{theorem}[Real stream prefix classifier endpoint projection]
\label{thm:real-stream-prefix-classifier-endpoint}
For rational-observation streams $x,y:\mathbb{N}\to\Hist$, if
$\mathsf{RealStreamPrefixClassifier}(x,y,n)$, then
$\mathsf{RatHistoryClassifier}(x(n),y(n))$.
\end{theorem}
\begin{proof}
Split on $n$. At $0$, the prefix witness is exactly the selected rational-history
classifier.

At a successor index, the recursive prefix witness is a pair; its right
component is the rational-history classifier at the selected endpoint.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealStreamPrefixClassifier\_endpoint}

\begin{theorem}[Real stream prefix classifier base of successor]
\label{thm:real-stream-prefix-classifier-base-of-successor}
For rational-observation streams $x,y:\mathbb{N}\to\Hist$, every nonempty
classified finite prefix contains the base observation classifier:
$$
\mathsf{RealStreamPrefixClassifier}(x,y,n+1)
\Longrightarrow
\mathsf{RealStreamPrefixClassifier}(x,y,0).
$$
\end{theorem}
\begin{proof}
Induct on $n$. At $0$, the successor prefix is the pair of the base classifier
and the first successor endpoint classifier, so project its left component. At
a further successor, recursively descend through the left prefix component until
the base observation classifier is exposed.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealStreamPrefixClassifier\_base\_of\_successor}
