\begin{theorem}[Real stream finite-window overlap partition]
\label{thm:real-stream-window-overlap-partition}
For rational-observation streams $x,y:\mathbb{N}\to\Hist$ and indices
$n,m,r$,
$$
\mathsf{RealStreamWindowClassifier}(x,y,n,m+r)
\Longleftrightarrow
\mathsf{RealStreamWindowClassifier}(x,y,n,m)\land
\mathsf{RealStreamWindowClassifier}(x,y,n+m,r).
$$
\end{theorem}
\begin{proof}
For the forward direction, assume the wide window
$\mathsf{RealStreamWindowClassifier}(x,y,n,m+r)$. Apply
\autoref{thm:real-stream-window-subwindow-extraction} with shift $0$ and
radius $m$; the bound $0+m\le m+r$ extracts the left window
$\mathsf{RealStreamWindowClassifier}(x,y,n,m)$.

Apply \autoref{thm:real-stream-window-subwindow-extraction} again with shift
$m$ and radius $r$. The bound $m+r\le m+r$ extracts the right overlapping
window $\mathsf{RealStreamWindowClassifier}(x,y,n+m,r)$. These two extracted
windows form the displayed conjunction.

For the reverse direction, assume the two overlapping windows. Unfold
Definition~\ref{def:real-stream-finite-window-classifier} and fix a selected
offset $k\le m+r$. It remains to produce the RatUp endpoint classifier at
$n+k$.

If $k\le m$, the left window applies directly at offset $k$ and gives
$$
\mathsf{RatHistoryClassifier}(x(n+k),y(n+k)).
$$

In the remaining bounded-offset case, write $k=m+j$ with $j\le r$. The right
window applies at offset $j$ and gives
$$
\mathsf{RatHistoryClassifier}(x((n+m)+j),y((n+m)+j)).
$$
The arithmetic identity $(n+m)+j=n+(m+j)=n+k$ identifies this endpoint with
the selected endpoint of the wide window.

The two offset cases cover every $k\le m+r$, with the boundary endpoint
$k=m$ shared by both windows. Hence the collected endpoint classifiers repack
as $\mathsf{RealStreamWindowClassifier}(x,y,n,m+r)$.
\end{proof}

\begin{theorem}[Real stream finite-window three-piece partition]
\label{thm:real-stream-window-three-piece-partition}
For rational-observation streams $x,y:\mathbb{N}\to\Hist$ and indices
$n,m,r,s$,
$$
\mathsf{RealStreamWindowClassifier}(x,y,n,m+1+r+1+s)
\Longleftrightarrow
\mathsf{RealStreamWindowClassifier}(x,y,n,m)\land
\mathsf{RealStreamWindowClassifier}(x,y,n+(m+1),r)\land
\mathsf{RealStreamWindowClassifier}(x,y,n+(m+1)+(r+1),s).
$$
\end{theorem}
\begin{proof}
For the forward direction, assume the wide window
$\mathsf{RealStreamWindowClassifier}(x,y,n,m+1+r+1+s)$. Apply
\autoref{thm:real-stream-window-adjacent-partition} with right radius
$r+1+s$ to split it into the left window at $(n,m)$ and the remaining right
block at $(n+(m+1),r+1+s)$.

Apply \autoref{thm:real-stream-window-adjacent-partition} again to that
remaining block, now with left radius $r$ and right radius $s$. This gives the
middle window at $(n+(m+1),r)$ and the right window at
$(n+(m+1)+(r+1),s)$, and the three extracted windows form the displayed
conjunction.

For the reverse direction, assume the three displayed windows. Apply the
reverse direction of \autoref{thm:real-stream-window-adjacent-partition} to
the middle and right windows, producing the combined right block
$\mathsf{RealStreamWindowClassifier}(x,y,n+(m+1),r+1+s)$.

Apply the reverse direction of
\autoref{thm:real-stream-window-adjacent-partition} once more to the left
window and this combined right block. The result is the wide window
$\mathsf{RealStreamWindowClassifier}(x,y,n,m+1+r+1+s)$.
\end{proof}
