\begin{theorem}[Real stream finite-window overlap partition]
\label{thm:real-stream-window-overlap-partition}
For rational-observation streams $x,y:\mathbb{N}\to\Hist$ and indices
$n,m,r$,
$$
\mathsf{RealStreamWindowClassifier}(x,y,n,m+r)
\Longleftrightarrow
\mathsf{RealStreamWindowClassifier}(x,y,n,m)\land
\mathsf{RealStreamWindowClassifier}(x,y,n+m,r).
$$
\end{theorem}
\begin{proof}
For the forward direction, assume the wide window
$\mathsf{RealStreamWindowClassifier}(x,y,n,m+r)$. Apply
\autoref{thm:real-stream-window-subwindow-extraction} with shift $0$ and
radius $m$; the bound $0+m\le m+r$ extracts the left window
$\mathsf{RealStreamWindowClassifier}(x,y,n,m)$.

Apply \autoref{thm:real-stream-window-subwindow-extraction} again with shift
$m$ and radius $r$. The bound $m+r\le m+r$ extracts the right overlapping
window $\mathsf{RealStreamWindowClassifier}(x,y,n+m,r)$. These two extracted
windows form the displayed conjunction.

For the reverse direction, assume the two overlapping windows. Unfold
Definition~\ref{def:real-stream-finite-window-classifier} and fix a selected
offset $k\le m+r$. It remains to produce the RatUp endpoint classifier at
$n+k$.

If $k\le m$, the left window applies directly at offset $k$ and gives
$$
\mathsf{RatHistoryClassifier}(x(n+k),y(n+k)).
$$

In the remaining bounded-offset case, write $k=m+j$ with $j\le r$. The right
window applies at offset $j$ and gives
$$
\mathsf{RatHistoryClassifier}(x((n+m)+j),y((n+m)+j)).
$$
The arithmetic identity $(n+m)+j=n+(m+j)=n+k$ identifies this endpoint with
the selected endpoint of the wide window.

The two offset cases cover every $k\le m+r$, with the boundary endpoint
$k=m$ shared by both windows. Hence the collected endpoint classifiers repack
as $\mathsf{RealStreamWindowClassifier}(x,y,n,m+r)$.
\end{proof}

\label{thm:real-stream-window-overlap-boundary-readback-package}
For rational-observation streams $x,y:\mathbb{N}\to\Hist$, indices $n,m,r$,
and histories $a,b,z_X,z_Y:\Hist$, assume
$\mathsf{RealStreamWindowClassifier}(x,y,n,m+r)$,
$\hsame(x(n+m),\Eone(a))$, and $\hsame(y(n+m),\Eone(b))$. Then the shared
boundary endpoint at $n+m$ carries the following package:
\begin{enumerate}
\item $\mathsf{RealStreamWindowClassifier}(x,y,n,m)$;
\item $\mathsf{RealStreamWindowClassifier}(x,y,n+m,r)$;
\item $\mathsf{RatHistoryClassifier}(\Eone(a),\Eone(b))$;
\item $\mathsf{UnaryHistory}(a)\land \mathsf{UnaryHistory}(b)\land
\hsame(a,b)$;
\item $\mathsf{PositiveUnaryDenominator}(x(n+m))\land
\mathsf{PositiveUnaryDenominator}(y(n+m))$;
\item $\mathsf{UnaryHistory}(x(n+m))\land
\mathsf{UnaryHistory}(y(n+m))\land
(\hsame(x(n+m),\emp)\to\mathsf{False})\land
(\hsame(y(n+m),\emp)\to\mathsf{False})$;
\item $(\hsame(x(n+m),\Ezero(z_X))\to\mathsf{False})\land
(\hsame(y(n+m),\Ezero(z_Y))\to\mathsf{False})$.

\end{theorem}
\begin{proof}
Apply the forward direction of
\autoref{thm:real-stream-window-overlap-partition} to the wide window
$\mathsf{RealStreamWindowClassifier}(x,y,n,m+r)$. This produces the left
window over $(n,m)$ and the right overlapping window over $(n+m,r)$.

The shared boundary is the selected offset $m$ inside the wide window. Since

\end{proof}

\begin{theorem}[Real stream finite-window three-piece partition]
\label{thm:real-stream-window-three-piece-partition}
For rational-observation streams $x,y:\mathbb{N}\to\Hist$ and indices
$n,m,r,s$,
$$
\mathsf{RealStreamWindowClassifier}(x,y,n,m+1+r+1+s)
\Longleftrightarrow
\mathsf{RealStreamWindowClassifier}(x,y,n,m)\land
\mathsf{RealStreamWindowClassifier}(x,y,n+(m+1),r)\land
\mathsf{RealStreamWindowClassifier}(x,y,n+(m+1)+(r+1),s).
$$
\end{theorem}
\begin{proof}
For the forward direction, assume the wide window
$\mathsf{RealStreamWindowClassifier}(x,y,n,m+1+r+1+s)$. Apply
\autoref{thm:real-stream-window-adjacent-partition} with right radius
$r+1+s$ to split it into the left window at $(n,m)$ and the remaining right
block at $(n+(m+1),r+1+s)$.

Apply \autoref{thm:real-stream-window-adjacent-partition} again to that
remaining block, now with left radius $r$ and right radius $s$. This gives the
middle window at $(n+(m+1),r)$ and the right window at
$(n+(m+1)+(r+1),s)$, and the three extracted windows form the displayed
conjunction.

For the reverse direction, assume the three displayed windows. Apply the
reverse direction of \autoref{thm:real-stream-window-adjacent-partition} to
the middle and right windows, producing the combined right block
$\mathsf{RealStreamWindowClassifier}(x,y,n+(m+1),r+1+s)$.

Apply the reverse direction of
\autoref{thm:real-stream-window-adjacent-partition} once more to the left
window and this combined right block. The result is the wide window
$\mathsf{RealStreamWindowClassifier}(x,y,n,m+1+r+1+s)$.
\end{proof}

\begin{definition}[Real stream finite-window partition spine]
\label{def:real-stream-window-partition-spine}
For a rational-observation anchor $n$ and a total finite bound $w$, a nonempty
$\mathsf{BEDC}$ finite partition spine is a finite $\mathsf{BHist}$-coded chain
$P$ of unary radius histories $a_0,\ldots,a_{q-1}:\Hist$ together with endpoint
continuation rows. Its numeric radii $r_i$ are the unary lengths displayed by
the $a_i$, its segment anchors are
$$
n_0=n,\qquad n_{i+1}=n_i+(r_i+1),
$$
and its accumulated displayed bound is
$$
w=r_0+1+r_1+1+\cdots+1+r_{q-1}.
$$
The singleton spine has accumulated bound $r_0$ and one segment at the original
anchor. The successor spine appends one unary radius $a_i$ by a native unary
$\mathsf{append}$ display and a $\mathsf{Cont}$ row from the previous endpoint
anchor to the next endpoint anchor. Thus every segment named by $P$ is a
concrete RealUp finite window
$\mathsf{RealStreamWindowClassifier}(x,y,n_i,r_i)$, and the final accumulated
anchor is obtained only by repeated unary $\mathsf{append}$ and $\mathsf{Cont}$
composition.
\end{definition}

\begin{theorem}[Real stream finite-window partition fold]
\label{thm:real-stream-window-finite-partition-fold}
For rational-observation streams $x,y:\mathbb{N}\to\Hist$, an anchor $n$, a
total bound $w$, and a certified finite partition spine $P$ over $(n,w)$ with
segments $(n_i,r_i)$, the total window classifier is equivalent to the
conjunction of the segment classifiers named by $P$:
$$
\mathsf{RealStreamWindowClassifier}(x,y,n,w)
\Longleftrightarrow
\bigwedge_{(n_i,r_i)\in P}
\mathsf{RealStreamWindowClassifier}(x,y,n_i,r_i).
$$
For the singleton spine, the right side is the single segment classifier.
\end{theorem}
\begin{proof}
Proceed by induction on the certified partition spine $P$.

In the singleton case, the spine definition identifies the total bound with the
single segment radius. The right side is the same
$\mathsf{RealStreamWindowClassifier}(x,y,n,r_0)$ row, so the displayed
equivalence is reflexive.

For the successor step, write the predecessor spine as $P_0$ with accumulated
bound $u$ and final anchor $n+u+1$ in the displayed unary continuation ledger.
The new certified unary radius is $r$, so the successor spine displays the
total bound as $u+1+r$ and names the new final segment
$\mathsf{RealStreamWindowClassifier}(x,y,n+(u+1),r)$.

For the forward direction of the successor step, apply
\autoref{thm:real-stream-window-adjacent-partition} to the total window
$\mathsf{RealStreamWindowClassifier}(x,y,n,u+1+r)$. This gives the predecessor
wide window $\mathsf{RealStreamWindowClassifier}(x,y,n,u)$ and the new final
segment. Apply the induction hypothesis to the predecessor wide window to
obtain every segment classifier named by $P_0$, then conjoin that list with the
new final segment.

For the reverse direction, assume the conjunction over the successor spine.
Project the predecessor segment conjunction and the new final segment. The
induction hypothesis repacks the predecessor segment conjunction into
$\mathsf{RealStreamWindowClassifier}(x,y,n,u)$. Apply the reverse direction of
\autoref{thm:real-stream-window-adjacent-partition} to this predecessor window
and the new final segment, producing
$\mathsf{RealStreamWindowClassifier}(x,y,n,u+1+r)$.

The three-segment case is the displayed consistency check: taking a spine with
three radii $m,r,s$ reduces the induction twice and gives exactly
\autoref{thm:real-stream-window-three-piece-partition}. Its two displayed cut
points are the first two successor anchors in the finite spine ledger.
\end{proof}
