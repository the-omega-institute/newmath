\begin{definition}[Real stability certificate]
\label{def:real-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item pointwise stream stability inherited from $\StreamNameUp$ through
  \autoref{thm:real-unary-stream-classifier-stability-package}
  \item transport of the $\RegSeqRat$ regularity witness along stream classification
  \item classifier reflexivity, symmetry, and transitivity by dyadic tolerance comparison
  \item compatibility of constant dyadic embedding with the closeness classifier
  \item closure of real operations only when supplied as certified $\RegSeqRat$ transformations
\end{enumerate}
\end{definition}

\begin{definition}[RegSeqRat stream certificate spine]
\label{def:regseqrat-stream-certificate-spine}
A $\RegSeqRat$ stream certificate spine for a rational-observation stream
$s:\Hist\to\Hist$ is a rational stream-name carrier
$\mathsf{RatStreamNameCarrier}(s)$ together with a regularity witness stable
under $\mathsf{RealUnaryStreamClassifier}$. Its ledger records the unary
observation indices, the endpoint $\RatUp$ carriers exposed at those indices,
the pointwise $\RatUp$ classifiers read from the stream classifier, and the
dyadic-tolerance requests used by the regularity witness.

The ten-field fit is: traditional target Bishop regular rational sequence;
BEDC source $\mathsf{BHist}$, $\mathsf{UnaryHistory}$, $\RatUp$, and
$\StreamNameUp$; carrier $\mathsf{RatStreamNameCarrier}$ plus the regularity
witness; classifier $\mathsf{RealUnaryStreamClassifier}$; stability pointwise
RatUp reflexivity, symmetry, transitivity, and $\hsame$ transport; ledger
finite observations and tolerance witnesses; core theorem list
\autoref{thm:regseqrat-stream-classifier-stability-spine} and
\autoref{thm:regseqrat-realup-seal-consumes-stream-spine}; exactness pointwise
stream classification; standard bridge RealUp seal; status
$\mathsf{ProofTarget}$.
\end{definition}

\begin{theorem}[RegSeqRat carrier and regularity obligations]
\label{thm:regseqrat-obligation-carrier}
A $\RegSeqRat$ stream certificate spine consists of a
$\mathsf{RatStreamNameCarrier}$ row, pointwise
$\mathsf{RatHistoryCarrier}$ rows over certified
$\mathsf{UnaryHistory}$ observations, a regularity ledger for
dyadic-tolerance requests, and constant-stream inhabitation supplied by
\autoref{thm:regseqrat-constant-stream-spine}.
\end{theorem}
\begin{proof}
Unfold Definition~\ref{def:regseqrat-stream-certificate-spine}. The carrier
field is the displayed $\mathsf{RatStreamNameCarrier}$ row, and each certified
unary observation exposes an endpoint $\RatUp$ carrier row in the ledger.

The same definition records the dyadic-tolerance requests used by the
regularity witness. For inhabitation, apply
\autoref{thm:regseqrat-constant-stream-spine} to any supplied
$\mathsf{RatHistoryCarrier}(d)$. The theorem builds the constant rational
stream, its self classifier, and the constant regularity ledger, which are
exactly the listed carrier and regularity obligations.
\end{proof}

\begin{theorem}[RegSeqRat classifier and transport obligations]
\label{thm:regseqrat-obligation-classifier}
The $\RegSeqRat$ classifier obligation supplies reflexivity, symmetry,
transitivity, endpoint-carrier exposure, and pointwise $\hsame$ transport for
$\mathsf{RealUnaryStreamClassifier}$ and
$\mathsf{RatStreamNameClassifier}$.
\end{theorem}
\begin{proof}
Apply \autoref{thm:regseqrat-stream-classifier-stability-spine}. Its stability
package supplies the reflexive, symmetric, and transitive
$\mathsf{RealUnaryStreamClassifier}$ rows, and it exposes the endpoint
$\mathsf{RatHistoryCarrier}$ rows at each certified unary observation.

For the transport row, use
\autoref{thm:real-unary-streamname-transported-classifier}. The pointwise
$\hsame$ witnesses over unary observations transport the RatUp point
classifiers and repack them as the
$\mathsf{RatStreamNameClassifier}$ row.
\end{proof}

\begin{theorem}[RegSeqRat seal-interface obligations]
\label{thm:regseqrat-obligation-seal-interface}
Sealed $\RealUp$ candidates exposing $\RegSeqRat$ stream certificate spines
consume the recorded stream comparison as the RealUp classifier row. The
ledger records only finite observations, $\RatUp$ classifiers,
dyadic-tolerance witnesses, and seal rows.
\end{theorem}
\begin{proof}
Apply \autoref{thm:regseqrat-realup-seal-consumes-stream-spine}. A sealed
$\RealUp$ candidate exposing two rational stream names reads the recorded
$\mathsf{RealUnaryStreamClassifier}$ comparison and exports that same row as
the classifier required by the RealUp seal.

The allowed ledger entries are those named in
Definition~\ref{def:real-ledger-policy}: finite stream observations, dyadic
comparison witnesses, regularity certificates, and classifier comparison
certificates. In the present seal interface these are precisely the finite
observation rows, RatUp classifier rows, dyadic-tolerance witnesses, and seal
rows used by the consumer theorem.
\end{proof}

\begin{theorem}[RegSeqRat constant stream spine]
\label{thm:regseqrat-constant-stream-spine}
For every history $d:\Hist$, a carrier
$\mathsf{RatHistoryCarrier}(d)$ supplies a $\RegSeqRat$ stream certificate spine
for $\mathsf{RatConstStream}_d$. The spine consists of the
$\mathsf{RatStreamNameCarrier}(\mathsf{RatConstStream}_d)$ row, the self
$\mathsf{RealUnaryStreamClassifier}(\mathsf{RatConstStream}_d,
\mathsf{RatConstStream}_d)$ row, and a constant regularity ledger: at any two
certified unary observation indices, both exposed endpoints are the same
$\RatUp$ point $d$, and every dyadic-tolerance request is discharged by RatUp
classifier reflexivity.
\end{theorem}
\begin{proof}
Apply the carrier half of
\autoref{thm:rat-stream-name-constant-point-exactness} from right to left to
the displayed $\mathsf{RatHistoryCarrier}(d)$. This gives the stream-name
carrier row for $\mathsf{RatConstStream}_d$.

For the classifier row, unfold
Definition~\ref{def:real-unary-observation-stream-classifier}. A certified
unary observation $n$ reads
$\mathsf{RatConstStream}_d(n)=d$, so the required pointwise
$\mathsf{RatHistoryClassifier}(\mathsf{RatConstStream}_d(n),
\mathsf{RatConstStream}_d(n))$ is RatUp classifier reflexivity at the supplied
carrier of $d$.

The regularity ledger is constant. Given any two certified unary observation
indices, the point-exactness theorem identifies both exposed endpoints with
the same RatUp point $d$. Thus the two observations have the same carrier, and
each dyadic tolerance request compares $d$ with itself, again by RatUp
reflexivity. These rows are exactly the data named in
Definition~\ref{def:regseqrat-stream-certificate-spine}.
\end{proof}

\begin{theorem}[RegSeqRat reindexed constant stream spine]
\label{thm:regseqrat-reindexed-constant-stream-spine}
For histories $d,e:\Hist$ and observation schedules $r,q:\Hist\to\Hist$,
$\mathsf{RatHistoryCarrier}(d)$ supplies $\RegSeqRat$ stream certificate
spines for both $\mathsf{RatConstStream}_d\circ r$ and
$\mathsf{RatConstStream}_d$. If $\mathsf{RatHistoryClassifier}(d,e)$ is
supplied, then the two independently reindexed constant streams
$\mathsf{RatConstStream}_d\circ r$ and
$\mathsf{RatConstStream}_e\circ q$ carry compatible $\RegSeqRat$ stream
certificate spines and the classifier row
$\mathsf{RealUnaryStreamClassifier}(\mathsf{RatConstStream}_d\circ r,
\mathsf{RatConstStream}_e\circ q)$.
\end{theorem}
\begin{proof}
First build the carried stream rows. Apply
\autoref{thm:rat-stream-name-independent-reindexed-constant-classifier} to the
carrier row of $d$ to obtain the stream-name carrier for
$\mathsf{RatConstStream}_d\circ r$, and apply
\autoref{thm:regseqrat-constant-stream-spine} to the same carrier row to obtain
the unreindexed constant-stream spine.

The regularity ledger for $\mathsf{RatConstStream}_d\circ r$ is the same
constant ledger used in \autoref{thm:regseqrat-constant-stream-spine}. At any
certified unary observation index, the schedule $r$ is read before the constant
stream is evaluated, and the exposed endpoint is still $d$. Thus every dyadic
tolerance request again compares the RatUp point $d$ with itself and is
discharged by RatUp classifier reflexivity.

Now suppose $\mathsf{RatHistoryClassifier}(d,e)$ is supplied. Its endpoint
carrier rows provide the corresponding constant regularity ledgers for the
$d$-stream and the $e$-stream. The StreamNameUp classifier row follows from
\autoref{thm:rat-stream-name-independent-reindexed-constant-classifier}, while
\autoref{thm:real-constant-independent-reindexed-stream-bridge} reads the same
point classifier as the RealUp unary-observation stream classifier for the two
independently reindexed constant streams.

Combining these rows gives the two compatible $\RegSeqRat$ spines together with
the displayed $\mathsf{RealUnaryStreamClassifier}$ row. No quotient,
completion, or host real structure is used.
\end{proof}

\begin{theorem}[RegSeqRat constant stream seal classifier package]
\label{thm:regseqrat-constant-stream-seal-classifier-package}
Let $X$ and $Y$ be sealed $\RealUp$ candidates exposing
$\mathsf{RatConstStream}_d\circ r$ and
$\mathsf{RatConstStream}_e\circ q$, respectively. If the two exposed streams
carry the compatible $\RegSeqRat$ stream certificate spines supplied by
\autoref{thm:regseqrat-reindexed-constant-stream-spine}, and if the seal
records the displayed row
$\mathsf{RealUnaryStreamClassifier}(\mathsf{RatConstStream}_d\circ r,
\mathsf{RatConstStream}_e\circ q)$, then the seal exports the RealUp
observation classifier required by
\autoref{def:real-classifier-specification}.
\end{theorem}
\begin{proof}
Project the two exposed rational-observation streams from the $\RealUp$ seals.
By hypothesis, the two streams already carry the $\RegSeqRat$ spines produced
for the independently reindexed constant streams.

Use the classifier row supplied by
\autoref{thm:regseqrat-reindexed-constant-stream-spine}. It is exactly the
$\mathsf{RealUnaryStreamClassifier}(\mathsf{RatConstStream}_d\circ r,
\mathsf{RatConstStream}_e\circ q)$ row recorded by the seal.

Finally apply \autoref{thm:regseqrat-realup-seal-consumes-stream-spine} to the
two spines and the displayed stream classifier. The conclusion is precisely
the RealUp observation classifier named in
\autoref{def:real-classifier-specification}.
\end{proof}

\begin{theorem}[RegSeqRat stream classifier stability spine]
\label{thm:regseqrat-stream-classifier-stability-spine}
Let $s,t,u,s',t':\Hist\to\Hist$ be rational-observation maps. Suppose
$$
\mathsf{RatStreamNameCarrier}(s),\quad
\mathsf{RatStreamNameCarrier}(t),\quad
\mathsf{RatStreamNameCarrier}(u),
$$
$$
\mathsf{RealUnaryStreamClassifier}(s,t),\quad
\mathsf{RealUnaryStreamClassifier}(t,u),
$$
and suppose the displayed endpoints transport pointwise over unary
observations:
$$
\forall n,\ \mathsf{UnaryHistory}(n)\to\hsame(s(n),s'(n)),
\quad
\forall n,\ \mathsf{UnaryHistory}(n)\to\hsame(t(n),t'(n)).
$$
Then the $\RegSeqRat$ stream certificate spine supplies
$\mathsf{RealUnaryStreamClassifier}(s,s)$,
$\mathsf{RealUnaryStreamClassifier}(t,s)$,
$\mathsf{RealUnaryStreamClassifier}(s,u)$, the endpoint rows
$\mathsf{RatHistoryCarrier}(s(n))$ and
$\mathsf{RatHistoryCarrier}(t(n))$ at every certified unary observation $n$,
and the transported stream-name classifier
$\mathsf{RatStreamNameClassifier}(s',t')$.
\end{theorem}
\begin{proof}
Unfold Definition~\ref{def:regseqrat-stream-certificate-spine},
Definition~\ref{def:rat-stream-name-carrier},
Definition~\ref{def:rat-stream-name-classifier}, and
Definition~\ref{def:real-unary-observation-stream-classifier}. The spine is
therefore the existing BEDC unary stream row together with the regularity
ledger; no separate classifier is being postulated.

Apply \autoref{thm:real-unary-stream-classifier-stability-package} to the
three carried streams, the two displayed RealUp unary stream classifiers, and
the two pointwise $\hsame$ transport rows. Its reflexivity, symmetry, and
transitivity clauses give the three
$\mathsf{RealUnaryStreamClassifier}$ conclusions, and its endpoint clause gives
the pointwise $\RatUp$ carrier rows.

The same package returns
$\mathsf{RatStreamNameClassifier}(s,t)$ and
$\mathsf{RealUnaryStreamClassifier}(s',t')$. Combining the transported point
classifier with RatUp classifier $\hsame$ transport and the transported
endpoint carrier rows repacks exactly
$\mathsf{RatStreamNameClassifier}(s',t')$.
\end{proof}

\begin{theorem}[RegSeqRat seal supplies RealUp observation classifier]
\label{thm:regseqrat-realup-seal-consumes-stream-spine}
Let $X$ and $Y$ be sealed $\RealUp$ candidates exposing rational-observation
streams $s,t:\Hist\to\Hist$. If the exposed streams satisfy the
$\RegSeqRat$ stream certificate spine and the seal records
$\mathsf{RealUnaryStreamClassifier}(s,t)$ as its stream comparison row, then
the sealed candidates export exactly the
$\mathsf{RealUnaryStreamClassifier}(s,t)$ row required by
\autoref{def:real-classifier-specification}.
\end{theorem}
\begin{proof}
Project the exposed stream data from the two $\RealUp$ seals. The
$\RegSeqRat$ spine gives the $\mathsf{RatStreamNameCarrier}$ rows, the
regularity ledger, and the pointwise RatUp classifier row over every certified
unary observation.

Now read the seal's stored stream comparison row. By
Definition~\ref{def:real-unary-observation-stream-classifier}, this is exactly
the pointwise rational-history classifier over the same certified unary
observations.

Repack that row under the RealUp seal. The result is the classifier row named
by \autoref{def:real-classifier-specification}; no quotient or completion
principle is used.
\end{proof}

\begin{theorem}[RegSeqRat stream spine scoped package]
\label{thm:regseqrat-stream-spine-scoped-package}
Let $s,t,u,s',t':\Hist\to\Hist$ be rational-observation maps carrying
$\mathsf{RatStreamNameCarrier}(s)$, $\mathsf{RatStreamNameCarrier}(t)$, and
$\mathsf{RatStreamNameCarrier}(u)$. Suppose the stream comparison rows
$\mathsf{RealUnaryStreamClassifier}(s,t)$ and
$\mathsf{RealUnaryStreamClassifier}(t,u)$ are recorded, and suppose the endpoint
transports are pointwise over unary observations:
$$
\forall n,\ \mathsf{UnaryHistory}(n)\to\hsame(s(n),s'(n)),
\quad
\forall n,\ \mathsf{UnaryHistory}(n)\to\hsame(t(n),t'(n)).
$$
Then the $\RegSeqRat$ stream certificate spine supplies the following scoped
package.
\begin{enumerate}
  \item For any $d:\Hist$ with $\mathsf{RatHistoryCarrier}(d)$, the constant
  stream $\mathsf{RatConstStream}_d$ inhabits the $\RegSeqRat$ stream spine.
  \item The classifier rows
  $\mathsf{RealUnaryStreamClassifier}(s,s)$,
  $\mathsf{RealUnaryStreamClassifier}(t,s)$, and
  $\mathsf{RealUnaryStreamClassifier}(s,u)$ are available.
  \item At every certified unary observation $n$, the endpoint carrier rows
  $\mathsf{RatHistoryCarrier}(s(n))$ and
  $\mathsf{RatHistoryCarrier}(t(n))$ are available.
  \item The transported stream-name classifier
  $\mathsf{RatStreamNameClassifier}(s',t')$ is available.
  \item For sealed $\RealUp$ candidates exposing $s$ and $t$, whose seal records
  $\mathsf{RealUnaryStreamClassifier}(s,t)$ as the stream comparison row, the
  RealUp seal exports the row required by
  \autoref{def:real-classifier-specification}.
\end{enumerate}
\end{theorem}
\begin{proof}
The constant-stream clause is exactly
\autoref{thm:regseqrat-constant-stream-spine}. It turns any
$\mathsf{RatHistoryCarrier}(d)$ into the carried constant stream, its reflexive
unary stream classifier, and the constant regularity ledger.

For the stability clauses, apply
\autoref{thm:regseqrat-stream-classifier-stability-spine} to $s,t,u,s',t'$,
the three carried-stream rows, the two displayed classifier rows, and the two
pointwise $\hsame$ transports. This gives the reflexive, symmetric, and
transitive $\mathsf{RealUnaryStreamClassifier}$ rows, the endpoint
$\RatUp$ carrier rows at every certified unary observation, and the transported
$\mathsf{RatStreamNameClassifier}(s',t')$ row.

Finally, apply
\autoref{thm:regseqrat-realup-seal-consumes-stream-spine} to any sealed
$\RealUp$ candidates exposing $s$ and $t$ with the recorded stream comparison.
The seal consumer repacks that same comparison as the RealUp classifier export.

All rows are obtained by projection, pointwise $\hsame$ transport, and repacking
of the listed certificate data. No quotient, completion, or host equality
principle is used.
\end{proof}

\begin{theorem}[Real unary stream classifier stability package]
\label{thm:real-unary-stream-classifier-stability-package}
For rational-observation maps $s,t,u,s',t':\Hist\to\Hist$, suppose
$$
\mathsf{RatStreamNameCarrier}(s),\quad
\mathsf{RatStreamNameCarrier}(t),\quad
\mathsf{RatStreamNameCarrier}(u),
$$
$$
\mathsf{RealUnaryStreamClassifier}(s,t),\quad
\mathsf{RealUnaryStreamClassifier}(t,u),
$$
and suppose the endpoint transports are pointwise over unary observations:
$$
\forall n,\ \mathsf{UnaryHistory}(n)\to\hsame(s(n),s'(n)),
\quad
\forall n,\ \mathsf{UnaryHistory}(n)\to\hsame(t(n),t'(n)).
$$
Then the same BEDC unary observation rows supply the following stability
package:
\begin{enumerate}
  \item $\mathsf{RatStreamNameClassifier}(s,t)$.
  \item $\mathsf{RealUnaryStreamClassifier}(s,s)$,
  $\mathsf{RealUnaryStreamClassifier}(t,s)$, and
  $\mathsf{RealUnaryStreamClassifier}(s,u)$.
  \item For every certified unary observation $n$, both endpoint RatUp carrier
  rows $\mathsf{RatHistoryCarrier}(s(n))$ and
  $\mathsf{RatHistoryCarrier}(t(n))$.
  \item $\mathsf{RealUnaryStreamClassifier}(s',t')$.
\end{enumerate}
\end{theorem}
\begin{proof}
Unfold Definition~\ref{def:rat-stream-name-carrier},
Definition~\ref{def:rat-stream-name-classifier}, and
Definition~\ref{def:real-unary-observation-stream-classifier}. After this
unfolding, both StreamNameUp and RealUp classifier rows are the same pointwise
RatUp classifier over histories $n$ equipped with
$\mathsf{UnaryHistory}(n)$.

The first conclusion is therefore just the displayed
$\mathsf{RealUnaryStreamClassifier}(s,t)$ read under the StreamNameUp
classifier name. For $\mathsf{RealUnaryStreamClassifier}(s,s)$, fix a
certified observation $n$; the carrier row for $s$ gives
$\mathsf{RatHistoryCarrier}(s(n))$, and RatUp classifier reflexivity gives the
required point classifier at that observation.

For the reversed classifier $\mathsf{RealUnaryStreamClassifier}(t,s)$, read the
point classifier from $\mathsf{RealUnaryStreamClassifier}(s,t)$ and apply RatUp
classifier symmetry. For $\mathsf{RealUnaryStreamClassifier}(s,u)$, read the
two point classifiers from the assumptions at the same certified observation
and compose them by RatUp classifier transitivity.

For endpoint carriers, fix a certified unary observation $n$ and unpack the
RatUp classifier obtained from $\mathsf{RealUnaryStreamClassifier}(s,t)$ at
that observation. Its endpoint data are precisely
$\mathsf{RatHistoryCarrier}(s(n))$ and $\mathsf{RatHistoryCarrier}(t(n))$.

For transport, fix the same certified observation $n$. Apply
$\mathsf{RatHistoryClassifier\_hsame\_transport}$ to the point classifier from
$\mathsf{RealUnaryStreamClassifier}(s,t)$ and to the two pointwise $\hsame$
witnesses from $s,t$ to $s',t'$. Repacking the transported point classifier
over all certified unary observations gives
$\mathsf{RealUnaryStreamClassifier}(s',t')$.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealUnaryStreamClassifier\_streamName\_stability\_package}

\begin{theorem}[Real unary stream-name transported classifier]
\label{thm:real-unary-streamname-transported-classifier}
For rational-observation maps $s,t,s',t':\Hist\to\Hist$, suppose
$$
\mathsf{RatStreamNameCarrier}(s),\quad
\mathsf{RatStreamNameCarrier}(t),\quad
\mathsf{RealUnaryStreamClassifier}(s,t),
$$
and suppose the endpoint transports are pointwise over unary observations:
$$
\forall n,\ \mathsf{UnaryHistory}(n)\to\hsame(s(n),s'(n)),
\quad
\forall n,\ \mathsf{UnaryHistory}(n)\to\hsame(t(n),t'(n)).
$$
Then $\mathsf{RatStreamNameClassifier}(s',t')$.
\end{theorem}
\begin{proof}
Unfold Definition~\ref{def:rat-stream-name-carrier},
Definition~\ref{def:rat-stream-name-classifier}, and
Definition~\ref{def:real-unary-observation-stream-classifier}. The
$\mathsf{RealUnaryStreamClassifier}(s,t)$ row supplies the RatUp point
classifier at each certified unary observation, while the two displayed
carrier hypotheses supply the endpoint stream carriers before transport.

Fix a certified unary observation $n$ with witness
$\nu:\mathsf{UnaryHistory}(n)$. Apply
\autoref{thm:rat-history-classifier-hsame-transport} to the RatUp point
classifier read from $\mathsf{RealUnaryStreamClassifier}(s,t)$ at $(n,\nu)$
and to the two pointwise $\hsame$ witnesses from $s,t$ to $s',t'$. This gives
$\mathsf{RatHistoryClassifier}(s'(n),t'(n))$.

Read the endpoint carrier fields of this transported RatUp classifier. The
left field gives $\mathsf{RatHistoryCarrier}(s'(n))$ and the right field gives
$\mathsf{RatHistoryCarrier}(t'(n))$. Since $(n,\nu)$ was arbitrary, these
fields build $\mathsf{RatStreamNameCarrier}(s')$ and
$\mathsf{RatStreamNameCarrier}(t')$.

The same transported RatUp point classifier, again for arbitrary $(n,\nu)$,
supplies the pointwise classifier row of
$\mathsf{RatStreamNameClassifier}(s',t')$. Repack that pointwise row with the
two transported stream carriers to obtain the claimed StreamNameUp
classifier.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealUnaryStreamClassifier\_streamName\_transport}

The ten-field fit is: traditional target transport of a unary real stream-name
classifier; BEDC source StreamNameUp read through the RealUp unary classifier;
carrier the transported RatUp endpoint carrier rows for $s'$ and $t'$;
classifier the transported RatUp point classifier; stability pointwise
$\hsame$ transport along unary observations; ledger the observation witness
and the two endpoint transport witnesses; core theorem this block; exactness
$\mathsf{RatStreamNameClassifier}(s',t')$ rather than only
$\mathsf{RealUnaryStreamClassifier}(s',t')$; standard bridge the downstream
RealUp certificate boundary; status $\mathsf{ProofTarget}$.

\begin{definition}[Real ledger policy]
\label{def:real-ledger-policy}
Record finite stream observations, dyadic comparison witnesses, regularity certificates, and classifier comparison certificates when sameness is used.
\end{definition}

\begin{definition}[RealUp certificate obligations]
\label{def:real-certificate-obligations}
The carrier, source, pattern, classifier, stability, and ledger data listed above are the required fields for using the interface name $\RealUp$.
\end{definition}

\paragraph{Primitive boundary.}
$\RealUp$ is not a BEDC primitive. It is admitted only when those obligations are supplied.

\begin{remark}[Zero-axiom encoding note]
\label{rem:real-zero-axiom-note}
The interface avoids quotient machinery. It does not use $\mathsf{Quot.sound}$; sameness is carried by dyadic comparison certificates and the $\RegSeqRat$ classifier.
\end{remark}
