\begin{theorem}[Real stream finite-window subwindow extraction]
\label{thm:real-stream-window-subwindow-extraction}
For rational-observation streams $x,y:\mathbb{N}\to\Hist$ and indices
$n,j,r,m$, if
$$
\mathsf{RealStreamWindowClassifier}(x,y,n,m)
$$
and $j+r\le m$, then
$$
\mathsf{RealStreamWindowClassifier}(x,y,n+j,r).
$$
\end{theorem}
\begin{proof}
Unfold Definition~\ref{def:real-stream-finite-window-classifier} in the source
window and fix a shifted offset $k\le r$. From $k\le r$ and $j+r\le m$,
bounded-offset monotonicity gives $j+k\le j+r\le m$.

Apply the source window at offset $j+k$. This yields the endpoint classifier
$$
\mathsf{RatHistoryClassifier}(x(n+(j+k)),y(n+(j+k))).
$$

The arithmetic identity $(n+j)+k=n+(j+k)$ identifies this endpoint with the
corresponding endpoint of the shifted window. Since the chosen $k$ was
arbitrary under the radius $r$, these endpoint classifiers repack as
$\mathsf{RealStreamWindowClassifier}(x,y,n+j,r)$.
\end{proof}

\begin{theorem}[Real stream finite-window selected zero-subwindow package]
\label{thm:real-stream-window-selected-zero-subwindow-package}
For rational-observation streams $x,y:\mathbb{N}\to\Hist$ and indices $n,j,m$,
if
$$
\mathsf{RealStreamWindowClassifier}(x,y,n,m)
$$
and $j\le m$, then both
$$
\mathsf{RealStreamWindowClassifier}(x,y,n+j,0)
$$
and
$$
\mathsf{RatHistoryClassifier}(x(n+j),y(n+j))
$$
hold.
\end{theorem}
\begin{proof}
Apply \autoref{thm:real-stream-window-subwindow-extraction} to the source
window with shift $j$ and radius $0$. The displayed bound $j\le m$ is exactly
the required inequality $j+0\le m$, so this gives the zero-radius shifted
window
$$
\mathsf{RealStreamWindowClassifier}(x,y,n+j,0).
$$

It remains to read the selected endpoint. Unfold
Definition~\ref{def:real-stream-finite-window-classifier} for the zero-radius
window and take offset $0\le 0$. The endpoint index is
$(n+j)+0=n+j$, so the window row returns
$\mathsf{RatHistoryClassifier}(x(n+j),y(n+j))$, as in the zero-window
exactness calculation of
\autoref{thm:real-stream-classifier-zero-window-exactness}.
\end{proof}
