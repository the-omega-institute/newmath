\begin{theorem}[Real stream transport with endpoint readback]
\label{thm:real-stream-classifier-transport-with-endpoint-readback}
For rational-observation streams $x,x',y,y':\mathbb{N}\to\Hist$, assume
$\forall n,\ \hsame(x(n),x'(n))$, assume
$\forall n,\ \hsame(y(n),y'(n))$, and assume
$\mathsf{RealStreamClassifier}(x,y)$. Then the transported streams carry the
following readback package:
\begin{enumerate}
  \item $\mathsf{RealStreamClassifier}(x',y')$;
  \item for every $n$,
  $\mathsf{RealStreamPrefixClassifier}(x',y',n)$;
  \item for every $n$,
  $\mathsf{RatHistoryClassifier}(x'(n),y'(n))$.
\end{enumerate}
\end{theorem}
\begin{proof}
First take the forward implication from
\autoref{thm:real-stream-classifier-hsame-transport} and apply it to the two
pointwise $\hsame$ hypotheses and the original stream classifier. This gives
$\mathsf{RealStreamClassifier}(x',y')$.

Next use the forward direction of
\autoref{thm:real-stream-finite-prefix-exactness} on the transported stream
classifier. The result is
$\mathsf{RealStreamPrefixClassifier}(x',y',n)$ for every finite prefix index
$n$.

Finally fix an index $n$ and apply
\autoref{thm:real-stream-prefix-classifier-endpoint} to the transported prefix
witness at that index. This recovers the endpoint
$\mathsf{RatHistoryClassifier}(x'(n),y'(n))$, completing the three displayed
fields of the package.
\end{proof}

\begin{theorem}[Real stream transported selected zero-endpoint exclusion]
\label{thm:real-stream-classifier-transport-selected-e0-endpoint-absurd}

$$

\forall i,\ \hsame(y(i),y'(i)),\quad
\mathsf{RealStreamClassifier}(x,y),
$$
then neither transported selected endpoint can be displayed as a
zero-extension endpoint:
$$
(\hsame(x'(n),\Ezero(z_x))\to\mathsf{False})\land
(\hsame(y'(n),\Ezero(z_y))\to\mathsf{False}).
$$
\end{theorem}
\begin{proof}
Transport the pointwise rational-history classifier at index $n$ along the
two streamwise $\hsame$ hypotheses. The transported classifier exposes
positive unary denominator witnesses at $x'(n)$ and $y'(n)$. If either
endpoint is displayed as $\Ezero$, transport the corresponding positive
denominator witness to that zero-extension endpoint and apply the rational
positive-denominator zero-endpoint contradiction.
\end{proof}
