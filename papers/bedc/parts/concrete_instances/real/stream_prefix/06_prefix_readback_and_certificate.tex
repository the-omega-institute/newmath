\begin{theorem}[Real stream prefix classifier transitivity]
\label{thm:real-stream-prefix-classifier-trans}
For rational-observation streams $x,y,z:\mathbb{N}\to\Hist$, if the finite
prefix classifier relates $x$ to $y$ through index $n$ and relates $y$
to $z$ through the same index, then it relates $x$ to $z$ through
index $n$.
\end{theorem}
\begin{proof}
Induct on the prefix index. At zero, use rational history classifier
transitivity. At a successor index, combine the induction hypothesis for the
shorter prefix with rational history classifier transitivity at the new
endpoint.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealStreamPrefixClassifier\_trans}

\begin{theorem}[Real stream prefix one-tail pair readback]
\label{thm:real-stream-prefix-e1-pair-readback}
If a classified finite rational-observation prefix displays its selected left and right endpoints as one-extension histories, then the displayed tails are unary and internally the same.
\end{theorem}
\begin{proof}
Split on the selected observation index. At zero, use the base rational-history classifier; at a successor, use the right endpoint of the prefix classifier. Transport the selected classifier to the displayed one-extension endpoints, then apply rational one-tail classifier exactness.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealStreamPrefixClassifier\_e1\_pair\_readback}

\begin{theorem}[Real stream prefix classifier hsame one-tail pair readback]
\label{thm:real-stream-prefix-classifier-hsame-e1-pair-readback}
For rational-observation streams $x,x',y,y':\mathbb{N}\to\Hist$, assume
$\forall n,\ \hsame(x(n),x'(n))$ and
$\forall n,\ \hsame(y(n),y'(n))$. If
$\mathsf{RealStreamPrefixClassifier}(x,y,n)$ and the transported endpoints
satisfy $\hsame(x'(n),\Eone(a))$ and $\hsame(y'(n),\Eone(b))$, then
$$
\mathsf{UnaryHistory}(a)\land \mathsf{UnaryHistory}(b)\land \hsame(a,b).
$$
\end{theorem}
\begin{proof}
First transport the finite prefix classifier from $(x,y)$ to $(x',y')$ using
the pointwise $\hsame$ hypotheses. The transported prefix has displayed
one-extension endpoints at index $n$, so the one-tail pair readback theorem
returns the two unary tail witnesses and $\hsame(a,b)$.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealStreamPrefixClassifier\_hsame\_e1\_pair\_readback}

\begin{theorem}[Real stream prefix certificate boundary]
\label{thm:real-stream-prefix-certificate-boundary}
For rational-observation streams transported by pointwise $\hsame$, the finite
prefix classifier supplies a closed BEDC certificate boundary. The same prefix
surface gives transitive prefix comparison, one-extension endpoint unary
tails, and transported tail $\hsame$ readback.

This boundary uses only the displayed finite prefix surface. It does not read
host index equality, quotient structure, completion data, $\RealUp$
completeness, metric structure, order structure, or any standard bridge row.
\end{theorem}
\begin{proof}
Apply \autoref{thm:real-stream-prefix-classifier-trans} to close the comparison
surface under transitive prefix composition at the displayed observation
index.

Apply \autoref{thm:real-stream-prefix-e1-pair-readback} to read the selected
left and right one-extension endpoints back to unary tails and their internal
$\hsame$ comparison.

Finally apply
\autoref{thm:real-stream-prefix-classifier-hsame-e1-pair-readback}. The
pointwise $\hsame$ hypotheses transport the same finite prefix certificate to
the displayed endpoint streams, so the certificate boundary survives the
transport without adding any host-side equality, analytic completion, metric,
order, or bridge datum.
\end{proof}
