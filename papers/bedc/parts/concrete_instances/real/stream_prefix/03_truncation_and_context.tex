\begin{theorem}[Real stream prefix tail extension with unary anchor]
\label{thm:real-stream-prefix-tail-extension-with-unary-anchor}
For rational-observation streams $x,y:\mathbb{N}\to\Hist$, a classified prefix
through $n$ extends across a classified finite tail of length $m$, and the
left endpoint at the new boundary retains its unary anchor:
$$
\mathsf{RealStreamPrefixClassifier}(x,y,n)
\Longrightarrow
\Bigl(\forall k<m,
\mathsf{RatHistoryClassifier}(x(n+k+1),y(n+k+1))\Bigr)
\Longrightarrow
\Bigl(\forall i,\ \mathsf{UnaryHistory}(x(i))\Bigr)
\Longrightarrow
\mathsf{RealStreamPrefixClassifier}(x,y,n+m)\land
\mathsf{UnaryHistory}(x(n+m)).
$$
\end{theorem}
\begin{proof}
Induct on $m$. The empty tail is the given prefix through $n$, with the unary
side condition supplying the endpoint anchor at $x(n)$. For a successor tail,
restrict the pointwise tail classifier to the shorter tail, apply the induction
hypothesis to obtain the prefix through $n+m$, and append the final rational
endpoint classifier at $x(n+m+1),y(n+m+1)$. The unary side condition supplies
the new left endpoint anchor.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealStreamPrefixClassifier\_tail\_extension\_with\_unary\_anchor}

\begin{theorem}[Real stream prefix classifier truncation]
\label{thm:real-stream-prefix-classifier-truncation}
For rational-observation streams $x,y:\mathbb{N}\to\Hist$ and indices $m,n$,
any prefix classifier through $m+n$ restricts to the prefix classifier through
$n$:
$$
\mathsf{RealStreamPrefixClassifier}(x,y,m+n)
\Longrightarrow
\mathsf{RealStreamPrefixClassifier}(x,y,n).
$$
\end{theorem}
\begin{proof}
Induct on $m$. At $m=0$, the source prefix is already the target prefix, since
$0+n=n$.

At a successor $m+1$, the recursive definition of
$\mathsf{RealStreamPrefixClassifier}(x,y,(m+1)+n)$ exposes its left component,
which is the classified prefix through $m+n$. Apply the induction hypothesis to
this exposed prefix to obtain the classified prefix through $n$.
\end{proof}

\begin{theorem}[Real stream prefix truncated unary context closure]
\label{thm:real-stream-prefix-truncated-unary-context-closed}
For rational-observation streams $x,y:\mathbb{N}\to\Hist$, if a prefix through
$m+n$ is classified, then after descending to the prefix through $n$, matching
unary prefix and tail contexts may be placed around every displayed endpoint:
$$
\mathsf{RealStreamPrefixClassifier}(x,y,m+n)
\Longrightarrow
\mathsf{RealStreamPrefixClassifier}(x',y',n),
$$
where $x'(i)=\mathsf{append}(p_X(i),\mathsf{append}(x(i),t_X(i)))$ and
$y'(i)=\mathsf{append}(p_Y(i),\mathsf{append}(y(i),t_Y(i)))$, with
$p_X(i)$ and $t_X(i)$ unary and with $p_X(i)\hsame p_Y(i)$ and
$t_X(i)\hsame t_Y(i)$ for every index $i$.
\end{theorem}
\begin{proof}
Induct on $m$ to restrict the source prefix from $m+n$ to $n$. The base case is
definitional, and the successor case projects the left component of the
recursive prefix witness. Apply the stream-prefix unary-context closure theorem
to the truncated prefix and the pointwise matching unary contexts.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealStreamPrefixClassifier\_truncated\_unary\_context\_closed}

\begin{theorem}[Real stream prefix truncated e1 pair readback]
\label{thm:real-stream-prefix-truncated-e1-pair-readback}
For rational-observation streams $x,y:\mathbb{N}\to\Hist$, if the finite prefix
through $m+n$ is classified and the selected endpoint at $n$ is displayed as
one-headed on both sides,
$$
\hsame(x(n),\Eone(d))
\quad\text{and}\quad
\hsame(y(n),\Eone(e)),
$$
then both tails are unary and internally same:
$$
\mathsf{UnaryHistory}(d)\wedge \mathsf{UnaryHistory}(e)\wedge \hsame(d,e).
$$
\end{theorem}
\begin{proof}
First descend the recursive prefix classifier from $m+n$ to $n$ by induction on
$m$. The base case is definitional. In the successor case, the prefix witness
at $(m+1)+n$ exposes the witness at $m+n$ as its left component. After the
classified prefix at $n$ is obtained, apply the e1 pair readback theorem at the
selected endpoint.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealStreamPrefixClassifier\_truncated\_e1\_pair\_readback}

\begin{theorem}[Real stream prefix truncated continuation endpoint context closure]
\label{thm:real-stream-prefix-truncated-cont-endpoint-context-closed}
For rational-observation streams $x,y:\mathbb{N}\to\Hist$, a classified prefix
through $m+n$ supplies the rational classifier at endpoint $n$. If the selected
left and right endpoints are then continued through unary prefix and tail
contexts, with matching $\hsame$ contexts on the two sides, the two displayed
continuation results remain rational-history classified.
\end{theorem}
\begin{proof}
Induct on $m$ to descend the recursive prefix classifier from $m+n$ to $n$.
Project the rational-history classifier at endpoint $n$. Apply rational-history
classifier unary-denominator context closure to the selected endpoint and the
two unary context pairs. The continuation witnesses identify the displayed
outputs with the corresponding appended contexts, and associativity of
$\mathsf{append}$ supplies the final endpoint transports.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealStreamPrefixClassifier\_truncated\_cont\_endpoint\_context\_closed}

\begin{lemma}[Real stream prefix previous classifier with unary anchor]
\label{lem:real-stream-prefix-previous-with-unary}
For rational-observation streams \(x,y:\mathbb{N}\to\Hist\), if every left observation \(x(i)\) is unary, then every successor finite-prefix classifier contains both the previous prefix classifier and the unary anchor at the previous left endpoint:
$$
\mathsf{RealStreamPrefixClassifier}(x,y,n+1)
\Longrightarrow
\mathsf{RealStreamPrefixClassifier}(x,y,n)\wedge \mathsf{UnaryHistory}(x(n)).
$$
\end{lemma}
\begin{proof}
The recursive successor prefix is a pair whose left component is the previous prefix classifier. The unary side condition supplies \(\mathsf{UnaryHistory}(x(n))\) at the same previous endpoint.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealStreamPrefixClassifier\_previous\_with\_unary}

\begin{theorem}[Real stream prefix add-left previous classifier with unary anchor]
\label{thm:real-stream-prefix-add-left-previous-with-unary}
For rational-observation streams \(x,y:\mathbb{N}\to\Hist\), if every left observation \(x(i)\) is unary, then any prefix classified through \(m+n\) descends to the prefix classified through \(n\), while retaining the unary anchor at \(x(n)\):
$$
\mathsf{RealStreamPrefixClassifier}(x,y,m+n)
\Longrightarrow
\mathsf{RealStreamPrefixClassifier}(x,y,n)\wedge \mathsf{UnaryHistory}(x(n)).
$$
\end{theorem}
\begin{proof}
Induct on \(m\). At \(m=0\), the displayed prefix is the target prefix and the unary side condition supplies the endpoint anchor. At a successor \(m+1\), apply \autoref{lem:real-stream-prefix-previous-with-unary} to peel one recursive prefix layer, then apply the induction hypothesis to the exposed prefix through \(m+n\).
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealStreamPrefixClassifier\_add\_left\_previous\_with\_unary}

\begin{lemma}[Real stream prefix add-left previous classifier with right unary anchor]
\label{lem:real-stream-prefix-add-left-previous-with-right-unary}
For rational-observation streams $x,y:\mathbb{N}\to\Hist$, if every right observation $y(i)$ is unary, then any prefix classified through $m+n$ descends to the prefix classified through $n$, while retaining the unary anchor at $y(n)$:
$$
\mathsf{RealStreamPrefixClassifier}(x,y,m+n)
\Longrightarrow
\mathsf{RealStreamPrefixClassifier}(x,y,n)\wedge \mathsf{UnaryHistory}(y(n)).
$$
\end{lemma}
\begin{proof}
Apply symmetry of the finite-prefix classifier to read the prefix as a classifier from $y$ to $x$. Then apply the add-left previous theorem with unary anchors on the left stream $y$. The recovered prefix is finally symmetrized back to a classifier from $x$ to $y$, while the unary endpoint anchor remains $y(n)$.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealStreamPrefixClassifier\_add\_left\_previous\_with\_right\_unary}
