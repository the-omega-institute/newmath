\begin{theorem}[Real stream finite-window adjacent partition]
\label{thm:real-stream-window-adjacent-partition}
For rational-observation streams $x,y:\mathbb{N}\to\Hist$ and indices
$n,m,r$,
$$
\mathsf{RealStreamWindowClassifier}(x,y,n,m+1+r)
\Longleftrightarrow
\mathsf{RealStreamWindowClassifier}(x,y,n,m)\land
\mathsf{RealStreamWindowClassifier}(x,y,n+(m+1),r).
$$
\end{theorem}
\begin{proof}
For the forward direction, assume the joined window
$\mathsf{RealStreamWindowClassifier}(x,y,n,m+1+r)$. Apply
\autoref{thm:real-stream-window-subwindow-extraction} with shift $0$ and
radius $m$; the bound $0+m\le m+1+r$ is immediate, so the left piece is
$\mathsf{RealStreamWindowClassifier}(x,y,n,m)$.

Apply \autoref{thm:real-stream-window-subwindow-extraction} again with shift
$m+1$ and radius $r$. The bound $(m+1)+r\le m+1+r$ identifies exactly the
right adjacent piece
$\mathsf{RealStreamWindowClassifier}(x,y,n+(m+1),r)$. These two extracted
windows form the conjunction.

For the reverse direction, assume the two adjacent pieces. Apply
\autoref{thm:real-stream-window-adjacent-concatenation} to the left piece and
the right piece at anchor $n+(m+1)$. The concatenation theorem repacks the
bounded endpoint witnesses into
$\mathsf{RealStreamWindowClassifier}(x,y,n,m+1+r)$.
\end{proof}

\begin{theorem}[Real unary stream finite-window adjacent partition]
\label{thm:real-unary-stream-window-adjacent-partition}
For observation maps $s,t:\Hist\to\Hist$ and unary anchor histories
$a,w,r:\Hist$,
$$
\begin{aligned}
&
\mathsf{RealUnaryStreamWindowClassifier}
(s,t,a,\mathsf{append}(w,\mathsf{BHist.e1}(r)))
\\
&\Longleftrightarrow
\mathsf{RealUnaryStreamWindowClassifier}(s,t,a,w)\land
\mathsf{RealUnaryStreamWindowClassifier}
(s,t,\mathsf{append}(a,\mathsf{BHist.e1}(w)),r).
\end{aligned}
$$
\end{theorem}
\begin{proof}
For the forward direction, unfold
Definition~\ref{def:real-unary-stream-window-classifier} in the joined unary
window. The displayed append ledger below
$\mathsf{append}(w,\mathsf{BHist.e1}(r))$ separates offsets that remain below
$w$ from offsets that pass the visible successor
$\mathsf{BHist.e1}(w)$.

The first ledger projection supplies every offset required for
$\mathsf{RealUnaryStreamWindowClassifier}(s,t,a,w)$. The second projection
supplies an offset below $r$ from the shifted anchor
$\mathsf{append}(a,\mathsf{BHist.e1}(w))$, and the carried $\hsame$ row
identifies that shifted endpoint with the corresponding endpoint of the joined
window. The RatUp endpoint classifier transports along this row.

For the reverse direction, assume the left and right unary windows. Apply
\autoref{thm:real-unary-stream-window-adjacent-concatenation}. It repacks the
two windows, their $\mathsf{UnaryHistory}$ rows, and the finite $\Cont$ append
ledger into the joined classifier over
$\mathsf{append}(w,\mathsf{BHist.e1}(r))$.
\end{proof}
