\begin{theorem}[Real selected transport constant real classifier package]
\label{thm:real-selected-transport-constant-real-classifier-package}
For rational-observation streams
$x,x',y,y',p_X,p_Y,t_X,t_Y,m_X,m_Y,o_X,o_Y:\mathbb{N}\to\Hist$,
indices $m,n$, and histories $a,a',b,b',z_X,z_Y:\Hist$, assume the pointwise
transport rows and the source stream classifier row
$$
\forall i,\ \hsame(x(i),x'(i)),\quad
\forall i,\ \hsame(y(i),y'(i)),\quad
\mathsf{RealStreamClassifier}(x,y).
$$
Assume also the selected transported context rows
$$
\mathsf{UnaryHistory}(p_X(n)),\quad
\mathsf{UnaryHistory}(t_X(n)),\quad
\hsame(p_X(n),p_Y(n)),\quad
\hsame(t_X(n),t_Y(n)),
$$
the selected continuation rows
$$
\mathsf{Cont}(p_X(n),x'(n),m_X(n)),\quad
\mathsf{Cont}(m_X(n),t_X(n),o_X(n)),
\quad\mathsf{Cont}(p_Y(n),y'(n),m_Y(n)),\quad
\mathsf{Cont}(m_Y(n),t_Y(n),o_Y(n)),
$$
and the selected e-one displays
$$
\hsame(o_X(n),\Eone(a)),\quad
\hsame(o_X(n),\Eone(a')),\quad
\hsame(o_Y(n),\Eone(b)),\quad
\hsame(o_Y(n),\Eone(b')),\quad
\hsame(x'(n),\Eone(a)),\quad
\hsame(y'(n),\Eone(b)).
$$
Then the selected transport carries the constant-real displayed classifier rows
and the synchronized tail rows
$$
\begin{array}{l}
\mathsf{RealConstantHistoryClassifier}(\Eone(\Eone(a)),\Eone(\Eone(b)))\land\\
\mathsf{RealConstantHistoryClassifier}(\Eone(\Eone(a')),\Eone(\Eone(b')))\land\\
\hsame(a,a')\land
\hsame(b,b')\land
\hsame(a,b)\land
\hsame(a',b')\land
\hsame(a,b')\land
\hsame(a',b).
\end{array}
$$
\end{theorem}
\begin{proof}
Invoke
\autoref{thm:real-selected-transport-synchronized-package-from-stream-classifier}
with the two pointwise transport rows, the source stream classifier row, and the
displayed context, continuation, and e-one hypotheses. The synchronized
displayed package gives
$\mathsf{RatHistoryClassifier}(\Eone(a),\Eone(b))$,
$\mathsf{RatHistoryClassifier}(\Eone(a'),\Eone(b'))$, and the six pairwise tail
rows.

Apply the reverse direction of
\autoref{thm:real-constant-history-classifier-e1-iff-rat} with
$(d,e)=(\Eone(a),\Eone(b))$. This turns the first rational classifier row into
$\mathsf{RealConstantHistoryClassifier}(\Eone(\Eone(a)),\Eone(\Eone(b)))$.

Apply the same reverse direction with
$(d,e)=(\Eone(a'),\Eone(b'))$. This turns the second rational classifier row
into
$\mathsf{RealConstantHistoryClassifier}(\Eone(\Eone(a')),\Eone(\Eone(b')))$.
Conjoin these two constant-real rows with the six synchronized tail rows.
\end{proof}
