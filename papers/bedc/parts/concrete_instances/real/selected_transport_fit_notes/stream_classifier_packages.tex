\begin{theorem}[Real selected transport fit from stream classifier]
\label{thm:real-selected-transport-fit-from-stream-classifier}
For rational-observation streams
$x,x',y,y',p_X,p_Y,t_X,t_Y,m_X,m_Y,o_X,o_Y:\mathbb{N}\to\Hist$,
indices $m,n$, and histories $a,a',b,b',z_X,z_Y:\Hist$, assume the pointwise
transport rows and the source stream classifier row
$$
\forall i,\ \hsame(x(i),x'(i)),\quad
\forall i,\ \hsame(y(i),y'(i)),\quad
\mathsf{RealStreamClassifier}(x,y).
$$
Assume also the selected transported context rows
$$
\mathsf{UnaryHistory}(p_X(n)),\quad
\mathsf{UnaryHistory}(t_X(n)),\quad
\hsame(p_X(n),p_Y(n)),\quad
\hsame(t_X(n),t_Y(n)),
$$
the selected continuation rows
$$
\mathsf{Cont}(p_X(n),x'(n),m_X(n)),\quad
\mathsf{Cont}(m_X(n),t_X(n),o_X(n)),
\quad\mathsf{Cont}(p_Y(n),y'(n),m_Y(n)),\quad
\mathsf{Cont}(m_Y(n),t_Y(n),o_Y(n)),
$$
and the selected e-one displays
$$
\hsame(o_X(n),\Eone(a)),\quad
\hsame(o_X(n),\Eone(a')),\quad
\hsame(o_Y(n),\Eone(b)),\quad
\hsame(o_Y(n),\Eone(b')),\quad
\hsame(x'(n),\Eone(a)),\quad
\hsame(y'(n),\Eone(b)).
$$
Then the selected-transport fit carries all four certificate packages:
\begin{enumerate}
\item the contextual endpoint package of
\autoref{thm:real-stream-classifier-transported-contextual-full-endpoint-package}
for $o_X(n)$ and $o_Y(n)$ with tails $a$ and $b$;
\item the contextual displayed e-one classifier package of
\autoref{thm:real-stream-classifier-transported-contextual-displayed-e1-classifier-package}
for the displayed endpoints $\Eone(a)$ and $\Eone(b)$;
\item the contextual e-one tail determinacy package of
\autoref{thm:real-stream-classifier-transported-contextual-e1-tail-determinacy}
for the four tails $a,a',b,b'$;
\item the finite-prefix selected full-shape readback package of
\autoref{thm:real-stream-prefix-transported-selected-full-shape-readback-package}
for $x'(n)$ and $y'(n)$ with tails $a$ and $b$.
\end{enumerate}
\end{theorem}
\begin{proof}
Apply \autoref{thm:real-stream-classifier-prefix} to
$\mathsf{RealStreamClassifier}(x,y)$ at the finite length $m+n$. This gives the
derived row $\mathsf{RealStreamPrefixClassifier}(x,y,m+n)$ needed by the
finite-prefix package.

Apply
\autoref{thm:real-stream-classifier-transported-contextual-full-endpoint-package}
to the two pointwise transport rows, the source stream classifier, the selected
context rows, the four continuation rows, and the two contextual displays
$\hsame(o_X(n),\Eone(a))$ and $\hsame(o_Y(n),\Eone(b))$. This gives the
contextual endpoint package.

Apply
\autoref{thm:real-stream-classifier-transported-contextual-displayed-e1-classifier-package}
to the same transported contextual data and the same two contextual displays.
This supplies the displayed e-one classifier row together with the endpoint
readback rows carried by that package.

Apply
\autoref{thm:real-stream-classifier-transported-contextual-e1-tail-determinacy}
to the same transported contextual data and all four contextual e-one displays.
This yields the tail-determinacy package for $a,a',b,b'$.

Finally apply
\autoref{thm:real-stream-prefix-transported-selected-full-shape-readback-package}
to the derived prefix classifier row, the two pointwise transport rows, and the
selected endpoint displays $\hsame(x'(n),\Eone(a))$ and
$\hsame(y'(n),\Eone(b))$. Conjoining the finite-prefix readback package with the
three contextual packages gives the four-package selected-transport fit.
\end{proof}

rational-observation streams; classifier $\mathsf{RealStreamClassifier}$ with
the derived $\mathsf{RealStreamPrefixClassifier}$ row; stability obligations

exactness predicate the four-package selected endpoint and contextual readback
certificate; standard bridge selected RegSeqRat and dyadic contextual endpoint
readback; status $\mathsf{ProofTarget}$.

\begin{theorem}[Real selected transport synchronized e-one endpoints]
\label{thm:real-selected-transport-synchronized-e1-endpoints}

$x,x',y,y',p_X,p_Y,t_X,t_Y,m_X,m_Y,o_X,o_Y:\mathbb{N}\to\Hist$,
indices $m,n$, and histories $a,a',b,b',z_X,z_Y:\Hist$, assume the pointwise
transport rows and the source stream classifier row
$$
\forall i,\ \hsame(x(i),x'(i)),\quad
\forall i,\ \hsame(y(i),y'(i)),\quad

$$
\mathsf{UnaryHistory}(p_X(n)),\quad
\mathsf{UnaryHistory}(t_X(n)),\quad
\hsame(p_X(n),p_Y(n)),\quad
\hsame(t_X(n),t_Y(n)),
$$
the selected continuation rows
$$
\mathsf{Cont}(p_X(n),x'(n),m_X(n)),\quad
\mathsf{Cont}(m_X(n),t_X(n),o_X(n)),
\quad\mathsf{Cont}(p_Y(n),y'(n),m_Y(n)),\quad
\mathsf{Cont}(m_Y(n),t_Y(n),o_Y(n)),
$$
and the selected e-one displays
$$
\hsame(o_X(n),\Eone(a)),\quad
\hsame(o_X(n),\Eone(a')),\quad

\hsame(o_Y(n),\Eone(b')),\quad

Then the selected transported endpoints satisfy the synchronized classifier and

$$
\begin{array}{l}
\mathsf{RatHistoryClassifier}(\Eone(a),\Eone(b))\land
\mathsf{RatHistoryClassifier}(\Eone(a'),\Eone(b'))\land\\
\mathsf{UnaryHistory}(a)\land
\mathsf{UnaryHistory}(a')\land
\mathsf{UnaryHistory}(b)\land
\mathsf{UnaryHistory}(b')\land\\
\hsame(a,a')\land \hsame(b,b')\land \hsame(a,b)\land\\
\hsame(a',b')\land \hsame(a,b')\land \hsame(a',b).
\end{array}
$$
\end{theorem}
\begin{proof}
First apply
\autoref{thm:real-stream-classifier-transported-contextual-full-endpoint-package}
to the displayed pointwise transport rows, the source stream classifier, the
selected context rows, the continuation rows, and the contextual displays for
$a$ and $b$.

This supplies the contextual endpoint package and, in particular, the tail row
$\mathsf{UnaryHistory}(a)\land\mathsf{UnaryHistory}(b)\land
\hsame(a,b)$.

Next apply
\autoref{thm:real-stream-classifier-transported-contextual-e1-tail-determinacy}
to the same transported contextual data and all four contextual e-one displays.
This supplies $\mathsf{UnaryHistory}(a')$, $\mathsf{UnaryHistory}(b')$,
$\hsame(a,a')$, $\hsame(b,b')$, and $\hsame(a',b')$, while retaining the
$a,b$ tail rows already identified above.

The off-diagonal tail rows are constructive consequences of these rows. Compose
$\hsame(a,b)$ with $\hsame(b,b')$ to obtain $\hsame(a,b')$; compose the
symmetry of $\hsame(a,a')$ with $\hsame(a,b)$ to obtain $\hsame(a',b)$.

Apply \autoref{thm:real-stream-classifier-prefix} to
$\mathsf{RealStreamClassifier}(x,y)$ at the finite length $m+n$. This gives the
derived row $\mathsf{RealStreamPrefixClassifier}(x,y,m+n)$ used by the selected
finite-prefix package.

Apply
\autoref{thm:real-stream-prefix-transported-selected-full-shape-readback-package}
to the derived prefix classifier row, the two pointwise transport rows, and the
selected prefix endpoint displays $\hsame(x'(n),\Eone(a))$ and
$\hsame(y'(n),\Eone(b))$. The returned finite-prefix readback package includes
$\mathsf{RatHistoryClassifier}(\Eone(a),\Eone(b))$.

Finally use the reverse direction of
\autoref{thm:rat-history-classifier-e1-tail-unary-iff} on
$\mathsf{UnaryHistory}(a')$, $\mathsf{UnaryHistory}(b')$, and
$\hsame(a',b')$. It gives
$\mathsf{RatHistoryClassifier}(\Eone(a'),\Eone(b'))$, which combines with the
first displayed classifier, the four unary tail rows, and the six pairwise
synchronized tail rows.
\end{proof}
