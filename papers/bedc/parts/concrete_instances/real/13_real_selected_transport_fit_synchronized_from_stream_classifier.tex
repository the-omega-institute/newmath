\begin{theorem}[Real selected transport synchronized fit from stream classifier]
\label{thm:real-selected-transport-fit-synchronized-from-stream-classifier}
For rational-observation streams
$x,x',y,y',p_X,p_Y,t_X,t_Y,m_X,m_Y,o_X,o_Y:\mathbb{N}\to\Hist$,
indices $m,n$, and histories $a,a',b,b',z_X,z_Y:\Hist$, assume the pointwise
transport rows and the source stream classifier row
$$
\forall i,\ \hsame(x(i),x'(i)),\quad
\forall i,\ \hsame(y(i),y'(i)),\quad
\mathsf{RealStreamClassifier}(x,y).
$$
Assume also the selected transported context rows
$$
\mathsf{UnaryHistory}(p_X(n)),\quad
\mathsf{UnaryHistory}(t_X(n)),\quad
\hsame(p_X(n),p_Y(n)),\quad
\hsame(t_X(n),t_Y(n)),
$$
the selected continuation rows
$$
\mathsf{Cont}(p_X(n),x'(n),m_X(n)),\quad
\mathsf{Cont}(m_X(n),t_X(n),o_X(n)),
\quad\mathsf{Cont}(p_Y(n),y'(n),m_Y(n)),\quad
\mathsf{Cont}(m_Y(n),t_Y(n),o_Y(n)),
$$
and the selected e-one displays
$$
\hsame(o_X(n),\Eone(a)),\quad
\hsame(o_X(n),\Eone(a')),\quad
\hsame(o_Y(n),\Eone(b)),\quad
\hsame(o_Y(n),\Eone(b')),\quad
\hsame(x'(n),\Eone(a)),\quad
\hsame(y'(n),\Eone(b)).
$$
Then the selected transport carries the following packages simultaneously.
\begin{enumerate}
\item The contextual endpoint package of
\autoref{thm:real-stream-classifier-transported-contextual-full-endpoint-package}:
$$
\begin{array}{l}
\mathsf{RatHistoryClassifier}(o_X(n),o_Y(n)),\\
\mathsf{PositiveUnaryDenominator}(o_X(n))\land
\mathsf{PositiveUnaryDenominator}(o_Y(n)),\\
\mathsf{UnaryHistory}(o_X(n))\land
\mathsf{UnaryHistory}(o_Y(n)),\\
(\hsame(o_X(n),\emp)\to\mathsf{False})\land
(\hsame(o_Y(n),\emp)\to\mathsf{False}),\\
(\hsame(o_X(n),\Ezero(z_X))\to\mathsf{False})\land
(\hsame(o_Y(n),\Ezero(z_Y))\to\mathsf{False}),\\
\mathsf{UnaryHistory}(a)\land\mathsf{UnaryHistory}(b)\land\hsame(a,b).
\end{array}
$$
\item The contextual e-one tail determinacy package of
\autoref{thm:real-stream-classifier-transported-contextual-e1-tail-determinacy}:
$$
\begin{array}{l}
\mathsf{UnaryHistory}(a)\land
\mathsf{UnaryHistory}(a')\land
\mathsf{UnaryHistory}(b)\land
\mathsf{UnaryHistory}(b')\land\\
\hsame(a,a')\land
\hsame(b,b')\land
\hsame(a,b)\land
\hsame(a',b').
\end{array}
$$
\item The finite-prefix selected full-shape readback package of
\autoref{thm:real-stream-prefix-transported-selected-full-shape-readback-package}:
$$
\begin{array}{l}
\mathsf{RatHistoryClassifier}(\Eone(a),\Eone(b))\land
\mathsf{UnaryHistory}(a)\land \mathsf{UnaryHistory}(b)\land \hsame(a,b)\land\\
\mathsf{PositiveUnaryDenominator}(x'(n))\land
\mathsf{PositiveUnaryDenominator}(y'(n))\land
\mathsf{UnaryHistory}(x'(n))\land
\mathsf{UnaryHistory}(y'(n))\land\\
(\hsame(x'(n),\emp)\to\mathsf{False})\land
(\hsame(y'(n),\emp)\to\mathsf{False})\land\\
(\hsame(x'(n),\Ezero(z_X))\to\mathsf{False})\land
(\hsame(y'(n),\Ezero(z_Y))\to\mathsf{False}).
\end{array}
$$
\item The synchronized displayed rows
$$
\begin{array}{l}
\mathsf{RatHistoryClassifier}(\Eone(a),\Eone(b))\land
\mathsf{RatHistoryClassifier}(\Eone(a'),\Eone(b'))\land\\
\mathsf{UnaryHistory}(a)\land
\mathsf{UnaryHistory}(a')\land
\mathsf{UnaryHistory}(b)\land
\mathsf{UnaryHistory}(b')\land\\
\hsame(a,a')\land
\hsame(b,b')\land
\hsame(a,b)\land
\hsame(a',b').
\end{array}
$$
\end{enumerate}
\end{theorem}
\begin{proof}
Apply \autoref{thm:real-stream-classifier-prefix} to
$\mathsf{RealStreamClassifier}(x,y)$ at the finite length $m+n$. This gives the
prefix row $\mathsf{RealStreamPrefixClassifier}(x,y,m+n)$.

Apply \autoref{thm:real-selected-transport-fit-synchronized-package} to the two
pointwise transport rows, the source stream classifier row, the derived prefix
row, the selected context rows, the selected continuation rows, and the six
selected e-one display rows. Its conclusion is exactly the four packages listed
above.
\end{proof}
