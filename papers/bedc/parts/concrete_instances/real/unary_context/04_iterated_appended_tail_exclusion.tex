\begin{theorem}[Real stream classifier iterated unary denominator context appended e-zero tail exclusion]
\label{thm:real-stream-classifier-iterated-unary-denominator-context-appended-e0-tail-absurd}
For rational-observation streams
$x,y,p_X,p_Y,t_X,t_Y,m_X,m_Y,u_X,u_Y,q_X,q_Y,s_X,s_Y,v_X,v_Y,o_X,o_Y:
\mathbb{N}\to\Hist$, a selected index $n$, and histories
$d_X,d_Y,z_X,z_Y:\Hist$, assume
$$
\mathsf{RealStreamClassifier}(x,y).
$$
Assume also that for every observation index $k$ the first context layer
satisfies
$$
\mathsf{UnaryHistory}(p_X(k)),\quad \mathsf{UnaryHistory}(t_X(k)),\quad
\hsame(p_X(k),p_Y(k)),\quad \hsame(t_X(k),t_Y(k)),
$$
$$
\mathsf{Cont}(p_X(k),x(k),m_X(k)),\quad
\mathsf{Cont}(m_X(k),t_X(k),u_X(k)),
$$
$$
\mathsf{Cont}(p_Y(k),y(k),m_Y(k)),\quad
\mathsf{Cont}(m_Y(k),t_Y(k),u_Y(k)),
$$
and that for every observation index $k$ the second context layer satisfies
$$
\mathsf{UnaryHistory}(q_X(k)),\quad \mathsf{UnaryHistory}(s_X(k)),\quad
\hsame(q_X(k),q_Y(k)),\quad \hsame(s_X(k),s_Y(k)),
$$
$$
\mathsf{Cont}(q_X(k),u_X(k),v_X(k)),\quad
\mathsf{Cont}(v_X(k),s_X(k),o_X(k)),
$$
$$
\mathsf{Cont}(q_Y(k),u_Y(k),v_Y(k)),\quad
\mathsf{Cont}(v_Y(k),s_Y(k),o_Y(k)).
$$
Then the selected second-layer outputs cannot be displayed as denominators with
an appended e-zero tail:
$$
(\hsame(o_X(n),d_X\cdot\Ezero(z_X))\to\mathsf{False})\land
(\hsame(o_Y(n),d_Y\cdot\Ezero(z_Y))\to\mathsf{False}).
$$
\end{theorem}
\begin{proof}
First apply
\autoref{thm:real-stream-classifier-unary-denominator-context-closed} to the
source RealUp stream classifier and the first-layer unary context and
continuation rows. This produces the intermediate stream classifier
$\mathsf{RealStreamClassifier}(u_X,u_Y)$.

Next apply
\autoref{thm:real-stream-classifier-unary-denominator-context-appended-e0-tail-absurd}
to this intermediate stream classifier, using the second-layer unary context and
continuation rows at the same selected index. The resulting two contradictions
are exactly the displayed exclusions for $o_X(n)$ and $o_Y(n)$.
\end{proof}
