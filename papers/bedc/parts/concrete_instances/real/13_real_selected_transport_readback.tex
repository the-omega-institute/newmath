\mathsf{Cont}(p_X(n),x'(n),m_X(n)),\quad
\mathsf{Cont}(m_X(n),t_X(n),o_X(n)),
\quad
\mathsf{Cont}(p_Y(n),y'(n),m_Y(n)),\quad
\mathsf{Cont}(m_Y(n),t_Y(n),o_Y(n)),
$$
and the two one-headed endpoint seals
$$
\hsame(o_X(n),\Eone(a))\quad\text{and}\quad \hsame(o_Y(n),\Eone(b)).
$$
Then the transported selected contextual endpoints and their displayed e-one
readbacks carry all of the following data:
\begin{enumerate}
\item $\mathsf{RatHistoryClassifier}(o_X(n),o_Y(n))$.
\item $\mathsf{RatHistoryClassifier}(\Eone(a),\Eone(b))$.
\item $\mathsf{UnaryHistory}(a)$ and $\mathsf{UnaryHistory}(b)$.
\item $\hsame(a,b)$.
\item $\mathsf{PositiveUnaryDenominator}(o_X(n))$ and
$\mathsf{PositiveUnaryDenominator}(o_Y(n))$.
\item $\mathsf{UnaryHistory}(o_X(n))$ and $\mathsf{UnaryHistory}(o_Y(n))$.
\item $\hsame(o_X(n),\emp)\to\mathsf{False}$ and
$\hsame(o_Y(n),\emp)\to\mathsf{False}$.
\item $\hsame(o_X(n),\Ezero(z_X))\to\mathsf{False}$ and
$\hsame(o_Y(n),\Ezero(z_Y))\to\mathsf{False}$.
\end{enumerate}
\end{theorem}
\begin{proof}
Apply
\autoref{thm:real-stream-classifier-transported-contextual-full-endpoint-package}
under the displayed pointwise transport rows, selected context rows,
continuation rows, and one-headed endpoint seals. This returns the classifier
for $o_X(n)$ and $o_Y(n)$, the positive-denominator rows, the endpoint unary
rows, the two nonempty exclusions, the two zero-headed exclusions, and the tail
readback
$\mathsf{UnaryHistory}(a)\land\mathsf{UnaryHistory}(b)\land\hsame(a,b)$.

Now transport the first classifier row across the two displayed e-one seals by
\autoref{thm:rat-history-classifier-hsame-transport}. The source classifier is
$\mathsf{RatHistoryClassifier}(o_X(n),o_Y(n))$, and the endpoint comparisons are
$\hsame(o_X(n),\Eone(a))$ and $\hsame(o_Y(n),\Eone(b))$; the transported row is
$\mathsf{RatHistoryClassifier}(\Eone(a),\Eone(b))$.

Finally project the three tail readback components from the returned
conjunction and repack them with the transported displayed e-one classifier and
the remaining endpoint rows already supplied by
\autoref{thm:real-stream-classifier-transported-contextual-full-endpoint-package}.
\end{proof}

\begin{theorem}[Real stream classifier transported contextual e-one tail determinacy]
\label{thm:real-stream-classifier-transported-contextual-e1-tail-determinacy}
For rational-observation streams
$x,x',y,y',p_X,p_Y,t_X,t_Y,m_X,m_Y,o_X,o_Y:\mathbb{N}\to\Hist$, a selected
index $n$, and histories $a,a',b,b':\Hist$, assume
$$
\forall i,\ \hsame(x(i),x'(i)),\quad
\forall i,\ \hsame(y(i),y'(i)),\quad
\mathsf{RealStreamClassifier}(x,y).
$$
Assume also the selected transported context rows
$$
\mathsf{UnaryHistory}(p_X(n)),\quad
\mathsf{UnaryHistory}(t_X(n)),\quad
\hsame(p_X(n),p_Y(n)),\quad
\hsame(t_X(n),t_Y(n)),
$$
the four selected continuation rows
$$
\mathsf{Cont}(p_X(n),x'(n),m_X(n)),\quad
\mathsf{Cont}(m_X(n),t_X(n),o_X(n)),
\quad
\mathsf{Cont}(p_Y(n),y'(n),m_Y(n)),\quad
\mathsf{Cont}(m_Y(n),t_Y(n),o_Y(n)),
$$
and the four e-one displays
$$
\hsame(o_X(n),\Eone(a)),\quad
\hsame(o_X(n),\Eone(a')),\quad
\hsame(o_Y(n),\Eone(b)),\quad
\hsame(o_Y(n),\Eone(b')).
$$
Then all of the following data hold:
\begin{enumerate}
\item $\mathsf{UnaryHistory}(a)$;
\item $\mathsf{UnaryHistory}(a')$;
\item $\mathsf{UnaryHistory}(b)$;
\item $\mathsf{UnaryHistory}(b')$;
\item $\hsame(a,a')$;
\item $\hsame(b,b')$;
\item $\hsame(a,b)$;
\item $\hsame(a',b')$.
\end{enumerate}
\end{theorem}
\begin{proof}
Apply
\autoref{thm:real-stream-classifier-transported-contextual-full-endpoint-package}
to the displays $\hsame(o_X(n),\Eone(a))$ and
$\hsame(o_Y(n),\Eone(b))$. Its tail row supplies
$\mathsf{UnaryHistory}(a)$, $\mathsf{UnaryHistory}(b)$, and $\hsame(a,b)$.

Apply the same transported contextual endpoint package to the displays
$\hsame(o_X(n),\Eone(a'))$ and $\hsame(o_Y(n),\Eone(b'))$. This gives
$\mathsf{UnaryHistory}(a')$, $\mathsf{UnaryHistory}(b')$, and
$\hsame(a',b')$.

For the $X$-tail comparison, compose the first display of $o_X(n)$ with the
symmetry of the second display. The resulting
$\hsame(\Eone(a),\Eone(a'))$ is inverted by
$\mathsf{BEDC.FKernel.Hist.hsame\_e1\_inversion\_iff}$ to obtain
$\hsame(a,a')$.

The same composition and e-one inversion applied to the two displays of
$o_Y(n)$ gives $\hsame(b,b')$. Package these two internal tail comparisons
with the two endpoint-package tail rows.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealStreamClassifier\_transported\_contextual\_e1\_tail\_determinacy}

\begin{theorem}[Real stream prefix transported selected full shape readback package]
\label{thm:real-stream-prefix-transported-selected-full-shape-readback-package}
For rational-observation streams $x,x',y,y':\mathbb{N}\to\Hist$, indices
$m,n$, and histories $a,b,z_X,z_Y:\Hist$, if
$$
\mathsf{RealStreamPrefixClassifier}(x,y,m+n),
$$
$$
\forall i,\ \hsame(x(i),x'(i)),\quad
\forall i,\ \hsame(y(i),y'(i)),\quad
\hsame(x'(n),\Eone(a)),\quad
\hsame(y'(n),\Eone(b)),
$$
then the transported selected finite-prefix endpoints carry the full readback
package
$$
\begin{array}{l}
\mathsf{RatHistoryClassifier}(\Eone(a),\Eone(b))\land
\mathsf{UnaryHistory}(a)\land \mathsf{UnaryHistory}(b)\land \hsame(a,b)\land\\
\mathsf{PositiveUnaryDenominator}(x'(n))\land
\mathsf{PositiveUnaryDenominator}(y'(n))\land
\mathsf{UnaryHistory}(x'(n))\land
\mathsf{UnaryHistory}(y'(n))\land\\
(\hsame(x'(n),\emp)\to\mathsf{False})\land
(\hsame(y'(n),\emp)\to\mathsf{False})\land\\
(\hsame(x'(n),\Ezero(z_X))\to\mathsf{False})\land
(\hsame(y'(n),\Ezero(z_Y))\to\mathsf{False}).
\end{array}
$$
\end{theorem}
\begin{proof}
First truncate the displayed prefix classifier from length $m+n$ to the selected
prefix through $n$ by
\autoref{thm:real-stream-prefix-classifier-truncation}. Transport this shorter
prefix along the two pointwise $\hsame$ rows by
\autoref{thm:real-stream-prefix-classifier-hsame-transport}, obtaining a
classified prefix for $x'$ and $y'$ at index $n$.

Project the selected endpoint by
\autoref{thm:real-stream-prefix-classifier-endpoint}. This gives
$\mathsf{RatHistoryClassifier}(x'(n),y'(n))$. Transport that endpoint classifier
across the displayed e-one seals with
\autoref{thm:rat-history-classifier-hsame-transport}, giving
$\mathsf{RatHistoryClassifier}(\Eone(a),\Eone(b))$.

Apply \autoref{thm:rat-history-classifier-positive-denominators} to the
transported endpoint classifier before the e-one seals. The two projected rows
are $\mathsf{PositiveUnaryDenominator}(x'(n))$ and
$\mathsf{PositiveUnaryDenominator}(y'(n))$.

Apply \autoref{thm:positive-unary-denominator-unary-and-nonempty} separately to
the two positive-denominator rows. This gives
$\mathsf{UnaryHistory}(x'(n))$, $\mathsf{UnaryHistory}(y'(n))$, and the two
empty-endpoint exclusions.

For the e-one tails, use
\autoref{thm:rat-history-classifier-e1-tail-unary-iff} in the forward direction
on $\mathsf{RatHistoryClassifier}(\Eone(a),\Eone(b))$. It supplies
$\mathsf{UnaryHistory}(a)$, $\mathsf{UnaryHistory}(b)$, and $\hsame(a,b)$.

It remains to exclude displayed zero-headed endpoints. If
$\hsame(x'(n),\Ezero(z_X))$ holds, transport
$\mathsf{PositiveUnaryDenominator}(x'(n))$ along this comparison with
\autoref{thm:rat-positive-unary-denominator-hsame-transport}, then contradict
\autoref{thm:rat-positive-denominator-e0-absurd}. The same argument for
$y'(n)$ and $z_Y$ gives the right zero-headed exclusion.

Collect the transported e-one classifier, the e-one tail rows, the positive
denominator witnesses, the unary and nonempty endpoint rows, and the two
zero-headed exclusions. These are exactly the displayed finite-prefix readback
package.
\end{proof}

\leanchecked{BEDC.Derived.RealUp.RealStreamPrefixClassifier\_transported\_selected\_full\_shape\_readback\_package}
