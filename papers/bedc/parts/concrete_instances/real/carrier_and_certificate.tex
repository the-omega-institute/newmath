\chapter{Certificate Obligations for $\RealUp$}
\label{ch:concrete-instances-real-namecert}

The real number interface $\RealUp$ is specified here as a certificate-obligation schema over sealed $\RegSeqRat$ stream names. These names are built from $\DyadicRatCore$ observations through $\StreamNameUp$, then sealed with regular-sequence data and a closeness classifier. The classical Dedekind / Cauchy completion is outside this schema because quotient and completion infrastructure is not exported by the mathlib-free BEDC kernel; the interface name is available only through the dyadic rational core, certified stream names, regular-sequence certificates, and their classifier.

\section{Carrier definition}

\begin{definition}[Real carrier]
\label{def:real-carrier}
Use a sealed $\RegSeqRat$ name. Its observations form a $\StreamNameUp$ stream over $\DyadicRatCore$, and its regularity certificate is exposed only through the $\RealUp$ seal.
\end{definition}

\begin{definition}[Constant real history carrier]
\label{def:real-constant-history-carrier}
A history is in the constant real history carrier when it is internally the same as an $\Eone$-sealed rational history:
$$
\mathsf{RealConstantHistoryCarrier}(h)
\quad\Longleftrightarrow\quad
\exists d,\ \hsame(h,\Eone(d))\ \wedge\ \mathsf{RatHistoryCarrier}(d).
$$
\end{definition}
\leandef{BEDC.Derived.RealUp.RealConstantHistoryCarrier}

\begin{theorem}[Constant real carrier at an e-one seal]
\label{thm:real-constant-history-carrier-e1-iff-rat}
For every history $d$,
$$
\mathsf{RealConstantHistoryCarrier}(\Eone(d))
\quad\Longleftrightarrow\quad
\mathsf{RatHistoryCarrier}(d).
$$
\end{theorem}
\begin{proof}
Unpack the constant carrier witness. One-constructor inversion turns
$\hsame(\Eone(d),\Eone(d'))$ into $\hsame(d,d')$, and rational history carrier
transport moves the witness back along that equality. The converse uses $d$
itself as the exposed rational history and reflexivity of $\hsame$.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealConstantHistoryCarrier\_e1\_iff\_rat}

\begin{theorem}[Constant real carrier e-zero exclusion]
\label{thm:real-constant-history-carrier-e0-absurd}
For every history tail $t$,
$$
\mathsf{RealConstantHistoryCarrier}(\Ezero(t))
\Longrightarrow
\bot.
$$
\end{theorem}
\begin{proof}
Unpack the constant real carrier witness. It displays $\Ezero(t)$ as
$\Eone(d)$ for some rational history $d$. BEDC history no-confusion between
zero-extension and one-extension constructors eliminates that display.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealConstantHistoryCarrier\_e0\_absurd}

\begin{theorem}[Constant real carrier empty exclusion]
\label{thm:real-constant-history-carrier-empty-absurd}
The empty history is not a constant real history carrier:
$$
\mathsf{RealConstantHistoryCarrier}(\emp)
\Longrightarrow
\bot.
$$
\end{theorem}
\begin{proof}
Unpack the constant real carrier witness. It displays $\emp$ as an
$\Eone$-sealed rational history, and BEDC history no-confusion between the
empty constructor and the one-extension constructor eliminates that display.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealConstantHistoryCarrier\_empty\_absurd}

\section{The certificate-obligation schema}

\begin{definition}[Real source specification]
\label{def:real-source-specification}
The source has three supplied layers: $\DyadicRatCore$ observations, $\StreamNameUp$ stream naming, and the $\RegSeqRat$ certificate over those observations.
\end{definition}

\begin{definition}[Real pattern specification]
\label{def:real-pattern-specification}
The pattern is sealed regular-sequence generation: every finite observation is a certified dyadic rational, stream observations are stable pointwise, and the $\RegSeqRat$ regularity witness supplies the dyadic Cauchy bound.
\end{definition}

\begin{definition}[Real classifier specification]
\label{def:real-classifier-specification}
The classifier is $\RegSeqRat$ closeness: two sealed real names classify together when their dyadic regular-sequence witnesses supply comparison certificates at every requested dyadic tolerance.
\end{definition}

\begin{definition}[Constant real history classifier]
\label{def:real-constant-history-classifier}
Two constant real histories classify together when each exposes an $\Eone$-sealed rational history and the exposed rational histories classify:
$$
\mathsf{RealConstantHistoryClassifier}(h,k)
\quad\Longleftrightarrow\quad
\exists d,e,\ \hsame(h,\Eone(d))\ \wedge\ \hsame(k,\Eone(e))
\ \wedge\ \mathsf{RatHistoryClassifier}(d,e).
$$
\end{definition}
\leandef{BEDC.Derived.RealUp.RealConstantHistoryClassifier}

\begin{theorem}[Constant real classifier at e-one seals]
\label{thm:real-constant-history-classifier-e1-iff-rat}
For all histories $d$ and $e$,
$$
\mathsf{RealConstantHistoryClassifier}(\Eone(d),\Eone(e))
\quad\Longleftrightarrow\quad
\mathsf{RatHistoryClassifier}(d,e).
$$
\end{theorem}
\begin{proof}
Eliminate the two seal witnesses. One-constructor inversion gives the two tail
$\hsame$ witnesses, and rational classifier transport carries the exposed
classifier back to the displayed tails. The reverse direction chooses the same
two rational histories and uses reflexivity at both seals.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealConstantHistoryClassifier\_e1\_iff\_rat}

\begin{theorem}[Constant real classifier endpoint carriers]
\label{thm:real-constant-history-classifier-endpoint-carriers}
If two constant real histories classify together, then both endpoints are in the
constant real history carrier:
$$
\mathsf{RealConstantHistoryClassifier}(h,k)
\Longrightarrow
\mathsf{RealConstantHistoryCarrier}(h)\land
\mathsf{RealConstantHistoryCarrier}(k).
$$
\end{theorem}
\begin{proof}
Unpack the classifier witness into rational histories $d,e$, endpoint displays
$\hsame(h,\Eone(d))$ and $\hsame(k,\Eone(e))$, and a rational history
classifier between $d$ and $e$. The rational classifier stores rational carrier
witnesses at both exposed endpoints, so pairing those carrier witnesses with the
two endpoint displays gives the two constant real history carriers.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealConstantHistoryClassifier\_endpoint\_carriers}

\begin{theorem}[Constant real classifier endpoint transport]
\label{thm:real-constant-history-classifier-endpoint-transport}
If constant real histories $h,k$ classify together, and $h\hsame h'$ and
$k\hsame k'$, then $h'$ and $k'$ classify together.
\end{theorem}
\begin{proof}
Unpack the two sealed rational witnesses. Symmetry and transitivity of
$\hsame$ move the displayed endpoint equalities from $h,k$ to $h',k'$, while
the rational classifier witness remains attached to the same sealed tails.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealConstantHistoryClassifier\_endpoint\_transport}

\begin{theorem}[Constant real classifier invalid endpoint exclusions]
\label{thm:real-constant-history-classifier-invalid-endpoint-absurd}
For every history $h$ and history tail $t$, the constant real-history classifier
excludes every endpoint shape that is not an $\Eone$-sealed rational history:
\begin{enumerate}
  \item $\mathsf{RealConstantHistoryClassifier}(\Ezero(t),h)\Longrightarrow\bot$;
  \item $\mathsf{RealConstantHistoryClassifier}(h,\Ezero(t))\Longrightarrow\bot$;
  \item $\mathsf{RealConstantHistoryClassifier}(\emp,h)\Longrightarrow\bot$;
  \item $\mathsf{RealConstantHistoryClassifier}(h,\emp)\Longrightarrow\bot$.
\end{enumerate}
In the ten-field fit, the traditional target is sealed constant-real endpoint
no-confusion; the BEDC source is $\mathsf{BHist}$ rational seals; the carrier is
$\mathsf{RealConstantHistoryCarrier}$; the classifier is
$\mathsf{RealConstantHistoryClassifier}$; the stability obligations are endpoint
carrier projection and carrier exclusions; the ledger is invalid endpoint shape;
the core theorem is this block; the exactness predicate is impossible
non-$\Eone$ endpoint; the standard bridge is constant rational real names; and
the status is $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
\begin{enumerate}
  \item Apply
  \autoref{thm:real-constant-history-classifier-endpoint-carriers} to the
  classifier witness. Its left projection gives
  $\mathsf{RealConstantHistoryCarrier}(\Ezero(t))$. Apply
  \autoref{thm:real-constant-history-carrier-e0-absurd}.

  \item Apply
  \autoref{thm:real-constant-history-classifier-endpoint-carriers} to the
  classifier witness. Its right projection gives
  $\mathsf{RealConstantHistoryCarrier}(\Ezero(t))$. Apply
  \autoref{thm:real-constant-history-carrier-e0-absurd}.

  \item Apply
  \autoref{thm:real-constant-history-classifier-endpoint-carriers} to the
  classifier witness. Its left projection gives
  $\mathsf{RealConstantHistoryCarrier}(\emp)$. Apply
  \autoref{thm:real-constant-history-carrier-empty-absurd}.

  \item Apply
  \autoref{thm:real-constant-history-classifier-endpoint-carriers} to the
  classifier witness. Its right projection gives
  $\mathsf{RealConstantHistoryCarrier}(\emp)$. Apply
  \autoref{thm:real-constant-history-carrier-empty-absurd}.
\end{enumerate}
\end{proof}

\begin{theorem}[Constant real classifier equivalence fields]
\label{thm:real-constant-history-classifier-equivalence-fields}
The constant real history classifier supplies the equivalence fields carried by
the constant-real certificate package:
\begin{enumerate}
  \item if $\mathsf{RealConstantHistoryCarrier}(h)$, then
  $\mathsf{RealConstantHistoryClassifier}(h,h)$;
  \item if $\mathsf{RealConstantHistoryClassifier}(h,k)$, then
  $\mathsf{RealConstantHistoryClassifier}(k,h)$;
  \item if $\mathsf{RealConstantHistoryClassifier}(h,k)$ and
  $\mathsf{RealConstantHistoryClassifier}(k,\ell)$, then
  $\mathsf{RealConstantHistoryClassifier}(h,\ell)$.
\end{enumerate}
In the ten-field fit, the traditional target is constant rational embedding
into $\RealUp$; the BEDC source is $BHist$ rational histories; the carrier is
$\mathsf{RealConstantHistoryCarrier}$; the classifier is
$\mathsf{RealConstantHistoryClassifier}$; the stability obligation is the
equivalence fields above; the ledger is the sealed rational tail; the core
theorem is this displayed package; exactness is inherited from the rational
classifier; the standard bridge is constant rational real names; and the status
is $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
For reflexivity, unpack $\mathsf{RealConstantHistoryCarrier}(h)$ as a sealed
rational tail $d$ with $\hsame(h,\Eone(d))$ and
$\mathsf{RatHistoryCarrier}(d)$. The rational carrier supplies the reflexive
rational classifier on $d$, using $\hsame$ reflexivity on the exposed tail.
Packing the same sealed tail twice and using the displayed endpoint witness on
both sides gives $\mathsf{RealConstantHistoryClassifier}(h,h)$.

For symmetry, unpack
$\mathsf{RealConstantHistoryClassifier}(h,k)$ into sealed rational tails $d,e$,
endpoint displays for $h$ and $k$, and
$\mathsf{RatHistoryClassifier}(d,e)$. Keep the same endpoint displays in the
opposite order and apply $\mathsf{RatHistoryClassifier\_symm}$ to the rational
classifier.

For transitivity, unpack the two classifiers into tails $d,e$ and $e',f$. The
two middle displays for $k$ compare $\Eone(e)$ and $\Eone(e')$ by $\hsame$
symmetry and transitivity; one-constructor inversion turns that comparison into
$\hsame(e,e')$. Transport the right rational classifier from $e'$ to $e$, then
compose it with the left rational classifier by
$\mathsf{RatHistoryClassifier\_trans}$. Repacking the outer endpoint displays
for $h$ and $\ell$ gives the required classifier.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealConstantHistoryClassifier\_equivalence\_fields}
