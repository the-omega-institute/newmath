\begin{theorem}[Real constant classifier appended e-one tail unary readback]
\label{thm:real-constant-history-classifier-append-e1-tail-unary-readback}
For histories $d,e,tail_D,tail_E$, a constant RealUp classifier whose two
rational denominators are displayed as appended e-one tails reads back both
unary factors and the rational endpoint classifier:
$$
\mathsf{RealConstantHistoryClassifier}(\Eone(d\cdot\Eone(tail_D)),
  \Eone(e\cdot\Eone(tail_E)))
\Longrightarrow
\mathsf{UnaryHistory}(d)\land\mathsf{UnaryHistory}(tail_D)\land
\mathsf{UnaryHistory}(e)\land\mathsf{UnaryHistory}(tail_E)\land
\hsame(d\cdot\Eone(tail_D),e\cdot\Eone(tail_E)).
$$
\end{theorem}
\begin{proof}
Use the e-one RealUp/RatUp classifier equivalence to expose the underlying
RatUp history classifier. Its positive-denominator fields apply to both
displayed denominators. The appended e-one denominator decomposition reads
each positive denominator as unary evidence for its head and tail, and the
remaining classifier field is the displayed endpoint $\hsame$ relation.
\end{proof}

\begin{theorem}[Real constant appended e-one tail rational-stream bridge readback]
\label{thm:real-constant-appended-tail-rational-stream-bridge-readback}
For histories $d,e,tail_D,tail_E:\Hist$ and observation maps
$r,q:\Hist\to\Hist$, set
$$
D=d\cdot\Eone(tail_D)
\qquad\text{and}\qquad
E=e\cdot\Eone(tail_E).
$$
Write the four bridge classifier rows as
$$
A=\mathsf{RatHistoryClassifier}(D,E),
\qquad
B=\mathsf{RatStreamNameClassifier}(\mathsf{RatConstStream}_D\circ r,
\mathsf{RatConstStream}_E\circ q),
$$
$$
U=\mathsf{RealUnaryStreamClassifier}(\mathsf{RatConstStream}_D\circ r,
\mathsf{RatConstStream}_E\circ q),
\qquad
C=\mathsf{RealConstantHistoryClassifier}(\Eone(D),\Eone(E)).
$$

$$
R=\mathsf{UnaryHistory}(d)\land\mathsf{UnaryHistory}(tail_D)\land
\mathsf{UnaryHistory}(e)\land\mathsf{UnaryHistory}(tail_E)\land
\hsame(D,E).
$$
Then the bridge rows are linked by
$$
(A\Longleftrightarrow B)\land(B\Longleftrightarrow U)\land
(U\Longleftrightarrow C),
$$
and every row reads back the same appended-tail denominator package:
$$
(A\Longrightarrow R)\land(B\Longrightarrow R)\land
(U\Longrightarrow R)\land(C\Longrightarrow R).
$$
\end{theorem}
\begin{proof}
Instantiate \autoref{thm:real-constant-independent-reindexed-stream-bridge}
with the histories $D$ and $E$ and the observation maps $r$ and $q$. The three
resulting equivalences are exactly the displayed links $A\Longleftrightarrow B$,
$B\Longleftrightarrow U$, and $U\Longleftrightarrow C$.

The direct readback from $C$ is
\autoref{thm:real-constant-history-classifier-append-e1-tail-unary-readback},
applied to the decompositions $D=d\cdot\Eone(tail_D)$ and
$E=e\cdot\Eone(tail_E)$.

For the remaining rows, transport along the instantiated bridge until the
constant real-history row $C$ is reached. The row $U$ uses the final bridge
equivalence, the row $B$ uses the second and final equivalences, and the row
$A$ uses all three equivalences. The same readback theorem then yields the
single conclusion $R$ in each case.
\end{proof}
