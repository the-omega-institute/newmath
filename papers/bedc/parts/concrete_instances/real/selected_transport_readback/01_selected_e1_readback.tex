\begin{theorem}[Real stream classifier selected e-one pair readback]
\label{thm:real-stream-classifier-selected-e1-pair-readback}
For rational-observation streams $x,y:\mathbb{N}\to\Hist$, every selected
observation index $n$, and histories $a,b:\Hist$, if
$$
\mathsf{RealStreamClassifier}(x,y),\quad
\hsame(x(n),\Eone(a)),\quad
\hsame(y(n),\Eone(b)),
$$
then
$$
\mathsf{UnaryHistory}(a)\land \mathsf{UnaryHistory}(b)\land \hsame(a,b).
$$
\end{theorem}
\begin{proof}
Apply \autoref{thm:real-stream-classifier-prefix} to the stream classifier at
the selected index $n$. This produces the finite-prefix classifier witness
needed at that index.

Apply \autoref{thm:real-stream-prefix-e1-pair-readback} to that prefix witness
and to the two displayed e-one endpoint equalities. The readback theorem returns
the two unary tail witnesses and the internal sameness relation
$\hsame(a,b)$.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealStreamClassifier\_selected\_e1\_pair\_readback}

\begin{theorem}[Real stream classifier transported selected e-one tail determinacy]
\label{thm:real-stream-classifier-transported-selected-e1-tail-determinacy}
For rational-observation streams $x,x',y,y':\mathbb{N}\to\Hist$, every selected
index $n$, and histories $a,a',b,b':\Hist$, assume
$$
\forall i,\ \hsame(x(i),x'(i)),\quad
\forall i,\ \hsame(y(i),y'(i)),\quad
\mathsf{RealStreamClassifier}(x,y).
$$
If the transported selected endpoints satisfy the four e-one displays
$$
\hsame(x'(n),\Eone(a)),\quad
\hsame(x'(n),\Eone(a')),\quad
\hsame(y'(n),\Eone(b)),\quad
\hsame(y'(n),\Eone(b')),
$$
then all of the following data hold:
\begin{enumerate}
\item $\mathsf{UnaryHistory}(a)$;
\item $\mathsf{UnaryHistory}(a')$;
\item $\mathsf{UnaryHistory}(b)$;
\item $\mathsf{UnaryHistory}(b')$;
\item $\hsame(a,a')$;
\item $\hsame(b,b')$;
\item $\hsame(a,b)$;
\item $\hsame(a',b')$.
\end{enumerate}
\end{theorem}
\begin{proof}
Apply \autoref{thm:real-stream-classifier-transported-selected-e1-pair-readback}
to the displays $\hsame(x'(n),\Eone(a))$ and
$\hsame(y'(n),\Eone(b))$. This supplies $\mathsf{UnaryHistory}(a)$,
$\mathsf{UnaryHistory}(b)$, and $\hsame(a,b)$.

Apply the same transported pair readback theorem again to the displays
$\hsame(x'(n),\Eone(a'))$ and $\hsame(y'(n),\Eone(b'))$. This supplies
$\mathsf{UnaryHistory}(a')$, $\mathsf{UnaryHistory}(b')$, and
$\hsame(a',b')$.

For the $x'$-tail comparison, compose the first selected endpoint display with
the symmetry of the second display. The resulting
$\hsame(\Eone(a),\Eone(a'))$ is inverted by
$\mathsf{BEDC.FKernel.Hist.hsame\_e1\_iff}$, giving $\hsame(a,a')$.

The same composition and e-one inversion applied to the two $y'$ endpoint
displays gives $\hsame(b,b')$. Package these same-side comparisons with the two
cross-tail rows from transported pair readback.
\end{proof}

Fit: traditional target canonical selected transported rational observation
tails; BEDC source RealUp streams, RatUp e-one readback, and FKernel e-one
no-confusion; carrier rational-observation streams; classifier
$\mathsf{RealStreamClassifier}$; stability pointwise $\hsame$ transport plus
e-one inversion; ledger selected index and four e-one seals; core theorem
\autoref{thm:real-stream-classifier-transported-selected-e1-tail-determinacy};
exactness unique tails for transported selected endpoints; standard bridge
RegSeqRat selected denominator tails; status $\mathsf{ProofTarget}$.
