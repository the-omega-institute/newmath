$$
\forall i,\ \hsame(x(i),x'(i)),\quad
\forall i,\ \hsame(y(i),y'(i)),\quad
\mathsf{RealStreamClassifier}(x,y),
$$
then
$$
\mathsf{RatHistoryClassifier}(x'(n),y'(n))\land
\mathsf{PositiveUnaryDenominator}(x'(n))\land
\mathsf{PositiveUnaryDenominator}(y'(n)).
$$
\end{theorem}
\begin{proof}
Apply \autoref{thm:real-stream-classifier-transport-with-endpoint-readback}
to the two pointwise $\hsame$ hypotheses and the source stream classifier. The
endpoint field of that package gives
$\mathsf{RatHistoryClassifier}(x'(n),y'(n))$ at the selected index $n$.

Apply \autoref{thm:rat-history-classifier-positive-denominators} to this
selected endpoint classifier. It exposes
$\mathsf{PositiveUnaryDenominator}(x'(n))$ and
$\mathsf{PositiveUnaryDenominator}(y'(n))$.

Package the endpoint classifier with the two positive-denominator witnesses.
The result is exactly the displayed transported selected denominator readback
conjunction.
\end{proof}
