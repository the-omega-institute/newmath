\begin{theorem}[Independent reindexed constant rational embedding certificate]
\label{thm:real-independent-reindexed-constant-rational-embedding-certificate}
For histories $d,e:\Hist$ and observation maps $r,q:\Hist\to\Hist$, the two
concrete endpoint assignments
$$
d\longmapsto (\mathsf{RatConstStream}_d\circ r,\Eone(d)),
\qquad
e\longmapsto (\mathsf{RatConstStream}_e\circ q,\Eone(e))
$$
supply the independently scheduled constant rational embedding certificate into
RealUp. The first endpoint has carrier exactness
$$
\mathsf{RatHistoryCarrier}(d)
\quad\Longleftrightarrow\quad
\mathsf{RatStreamNameCarrier}(\mathsf{RatConstStream}_d\circ r)
\quad\Longleftrightarrow\quad
\mathsf{RealConstantHistoryCarrier}(\Eone(d)),
$$
and the second endpoint has carrier exactness
$$
\mathsf{RatHistoryCarrier}(e)
\quad\Longleftrightarrow\quad
\mathsf{RatStreamNameCarrier}(\mathsf{RatConstStream}_e\circ q)
\quad\Longleftrightarrow\quad
\mathsf{RealConstantHistoryCarrier}(\Eone(e)).
$$
Its classifier exactness is the three-row chain displayed in
\autoref{thm:real-constant-independent-reindexed-stream-bridge}: the RatUp
classifier of $d$ and $e$, the StreamNameUp classifier of
$\mathsf{RatConstStream}_d\circ r$ and $\mathsf{RatConstStream}_e\circ q$, the
RealUp unary-observation stream classifier of those same streams, and the
RealUp constant-history classifier of $\Eone(d)$ and $\Eone(e)$ are equivalent.
\end{theorem}
\begin{proof}
Apply \autoref{thm:real-constant-stream-carrier-reindexed-streamname-bridge}
to $d$ with observation map $r$. This gives the first carrier row, identifying
the RatUp carrier of $d$, the StreamNameUp carrier of
$\mathsf{RatConstStream}_d\circ r$, and the RealUp constant carrier of
$\Eone(d)$.

Apply the same theorem separately to $e$ with observation map $q$. This gives
the second carrier row, identifying the RatUp carrier of $e$, the StreamNameUp
carrier of $\mathsf{RatConstStream}_e\circ q$, and the RealUp constant carrier
of $\Eone(e)$.

For classifier exactness, apply
\autoref{thm:real-constant-independent-reindexed-stream-bridge}. Its three
equivalence rows are precisely the classifier rows required for independently
scheduled constant streams and their paired e-one seals.
\end{proof}

\begin{theorem}[Real stream classifier finite-prefix stability package]
\label{thm:real-stream-classifier-finite-prefix-stability-package}

$n$, if
$$
\forall i,\ \mathsf{RatHistoryCarrier}(x(i)),\quad
\mathsf{RealStreamClassifier}(x,y),\quad
\mathsf{RealStreamClassifier}(y,z),
$$
then
$$
\mathsf{RealStreamPrefixClassifier}(x,x,n)
\land
\mathsf{RealStreamPrefixClassifier}(y,x,n)
\land
\mathsf{RealStreamPrefixClassifier}(x,z,n)
\land
\mathsf{RatHistoryClassifier}(x(n),z(n)).
$$
\end{theorem}
\begin{proof}
Apply \autoref{thm:real-stream-classifier-prefix} to
$\mathsf{RealStreamClassifier}(x,y)$ at the selected prefix index $n$, and
apply the same theorem to $\mathsf{RealStreamClassifier}(y,z)$ at $n$. These
give the two finite-prefix classifier witnesses for $x$ with $y$ and for $y$
with $z$.

Use \autoref{thm:real-stream-prefix-classifier-refl} with the displayed
rational-history carrier row for $x$ to obtain
$\mathsf{RealStreamPrefixClassifier}(x,x,n)$. Apply
\autoref{thm:real-stream-prefix-classifier-symm} to the $x,y$ prefix witness
to obtain $\mathsf{RealStreamPrefixClassifier}(y,x,n)$.

Compose the $x,y$ and $y,z$ prefix witnesses by
\autoref{thm:real-stream-prefix-classifier-trans}, yielding
$\mathsf{RealStreamPrefixClassifier}(x,z,n)$. Finally apply
\autoref{thm:real-stream-prefix-classifier-endpoint} to this composite prefix
to recover $\mathsf{RatHistoryClassifier}(x(n),z(n))$.
\end{proof}

$\mathsf{RealStreamPrefixClassifier}$. Stability: prefix reflexivity, symmetry,
transitivity, and endpoint projection. Ledger: the selected prefix index $n$.
Core theorem: \autoref{thm:real-stream-classifier-finite-prefix-stability-package}.
Exactness: the displayed four-part conjunction. Standard bridge: finite
rational prefix agreement. Status: $\mathsf{ProofTarget}$.

\begin{theorem}[Real unary StreamNameUp stability package]
\label{thm:real-unary-streamname-stability-package}
For certified unary-observation maps $s,t,u,s',t':\Hist\to\Hist$, suppose
$$
\mathsf{RatStreamNameCarrier}(s),\quad
\mathsf{RatStreamNameCarrier}(t),\quad
\mathsf{RatStreamNameCarrier}(u),
$$
$$
\mathsf{RealUnaryStreamClassifier}(s,t),\quad
\mathsf{RealUnaryStreamClassifier}(t,u),
$$
and suppose the transported endpoints satisfy
$$
\forall n,\ \mathsf{UnaryHistory}(n)\to\hsame(s(n),s'(n)),
\quad
\forall n,\ \mathsf{UnaryHistory}(n)\to\hsame(t(n),t'(n)).
$$
Then the same BEDC unary observation rows supply the following public
certificate fields:
\begin{enumerate}
  \item $\mathsf{RatStreamNameClassifier}(s,t)$.
  \item $\mathsf{RealUnaryStreamClassifier}(s,s)$,
  $\mathsf{RealUnaryStreamClassifier}(t,s)$, and
  $\mathsf{RealUnaryStreamClassifier}(s,u)$.
  \item For every certified unary observation $n$, both
  $\mathsf{RatHistoryCarrier}(s(n))$ and $\mathsf{RatHistoryCarrier}(t(n))$.
  \item $\mathsf{RealUnaryStreamClassifier}(s',t')$.
\end{enumerate}
\end{theorem}
\begin{proof}
Unfold Definition~\ref{def:rat-stream-name-carrier},
Definition~\ref{def:rat-stream-name-classifier}, and
Definition~\ref{def:real-unary-observation-stream-classifier}. After this
unfolding, both StreamNameUp and RealUp classifier rows are the same pointwise
RatUp classifier over histories $n$ equipped with
$\mathsf{UnaryHistory}(n)$.

The first conclusion is therefore just the displayed
$\mathsf{RealUnaryStreamClassifier}(s,t)$ read under the StreamNameUp
classifier name. For $\mathsf{RealUnaryStreamClassifier}(s,s)$, fix a
certified observation $n$; the carrier row for $s$ gives
$\mathsf{RatHistoryCarrier}(s(n))$, and RatUp classifier reflexivity gives the
required point classifier at that observation.

For the reversed classifier $\mathsf{RealUnaryStreamClassifier}(t,s)$, read the
point classifier from $\mathsf{RealUnaryStreamClassifier}(s,t)$ and apply RatUp
classifier symmetry. For $\mathsf{RealUnaryStreamClassifier}(s,u)$, read the
two point classifiers from the assumptions at the same certified observation
and compose them by RatUp classifier transitivity.

For endpoint carriers, fix a certified unary observation $n$ and unpack the
RatUp classifier obtained from $\mathsf{RealUnaryStreamClassifier}(s,t)$ at
that observation. Its endpoint data are precisely
$\mathsf{RatHistoryCarrier}(s(n))$ and $\mathsf{RatHistoryCarrier}(t(n))$.

For transport, fix the same certified observation $n$. Apply
$\mathsf{RatHistoryClassifier\_hsame\_transport}$ to the point classifier from
$\mathsf{RealUnaryStreamClassifier}(s,t)$ and to the two pointwise $\hsame$
witnesses from $s,t$ to $s',t'$. Repacking the transported point classifier
over all certified unary observations gives
$\mathsf{RealUnaryStreamClassifier}(s',t')$.
\end{proof}
