\begin{theorem}[Real constant classifier inner e-zero endpoint absurd]
\label{thm:real-constant-history-classifier-e1-e0-endpoint-absurd}
For histories $tail,d:\mathsf{BHist}$, the constant real-history classifier
excludes an $\Ezero$ rational endpoint even when it is sealed by the outer
constant-real $\Eone$ constructor:
$$
\begin{array}{l}
\mathsf{RealConstantHistoryClassifier}(\Eone(\Ezero(tail)),\Eone(d))
\Longrightarrow\mathsf{False},\\
\mathsf{RealConstantHistoryClassifier}(\Eone(d),\Eone(\Ezero(tail)))
\Longrightarrow\mathsf{False}.
\end{array}
$$
\end{theorem}
\begin{proof}
Expose the sealed constant-real classifier with
\autoref{thm:real-constant-history-classifier-e1-iff-rat}. The resulting
rational classifier has an $\Ezero$ endpoint on one side, so the RatUp endpoint
exclusion applies in the corresponding orientation.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealConstantHistoryClassifier\_e1\_e0\_endpoint\_absurd}

\begin{theorem}[Real constant classifier appended e-zero tail absurd]
\label{thm:real-constant-history-classifier-e1-append-e0-tail-absurd}
For histories $d,e,tail:\mathsf{BHist}$, the constant real-history classifier
excludes a sealed rational denominator whose visible denominator ends in an
appended zero-headed tail:
$$
\begin{array}{l}
\mathsf{RealConstantHistoryClassifier}(\Eone(d\cdot\Ezero(tail)),\Eone(e))
\Longrightarrow\mathsf{False},\\
\mathsf{RealConstantHistoryClassifier}(\Eone(e),\Eone(d\cdot\Ezero(tail)))
\Longrightarrow\mathsf{False}.
\end{array}
$$
\end{theorem}
\begin{proof}
For the left orientation, expose the constant RealUp classifier with
\autoref{thm:real-constant-history-classifier-e1-iff-rat}. The resulting
RatUp classifier relates $d\cdot\Ezero(tail)$ to $e$, so
\autoref{thm:rat-history-classifier-positive-denominators} gives the left
positive-denominator row. This contradicts
\autoref{thm:positive-unary-denominator-append-e0-tail-absurd}.

For the right orientation, expose the same constant RealUp classifier with
\autoref{thm:real-constant-history-classifier-e1-iff-rat}. The resulting
RatUp classifier relates $e$ to $d\cdot\Ezero(tail)$, so
\autoref{thm:rat-history-classifier-positive-denominators} gives the right
positive-denominator row. This again contradicts
\autoref{thm:positive-unary-denominator-append-e0-tail-absurd}.
\end{proof}
