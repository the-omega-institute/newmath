\begin{theorem}[Real constant classifier inner e-zero endpoint absurd]
\label{thm:real-constant-history-classifier-e1-e0-endpoint-absurd}
For histories $tail,d:\mathsf{BHist}$, the constant real-history classifier
excludes an $\Ezero$ rational endpoint even when it is sealed by the outer
constant-real $\Eone$ constructor:
$$
\begin{array}{l}
\mathsf{RealConstantHistoryClassifier}(\Eone(\Ezero(tail)),\Eone(d))
\Longrightarrow\mathsf{False},\\
\mathsf{RealConstantHistoryClassifier}(\Eone(d),\Eone(\Ezero(tail)))
\Longrightarrow\mathsf{False}.
\end{array}
$$
\end{theorem}
\begin{proof}
Expose the sealed constant-real classifier with
\autoref{thm:real-constant-history-classifier-e1-iff-rat}. The resulting
rational classifier has an $\Ezero$ endpoint on one side, so the RatUp endpoint
exclusion applies in the corresponding orientation.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealConstantHistoryClassifier\_e1\_e0\_endpoint\_absurd}

\begin{theorem}[Real constant classifier malformed inner denominator exclusions]
\label{thm:real-constant-history-classifier-malformed-inner-denominator-absurd}
For histories $d,p,z:\mathsf{BHist}$, the constant real-history classifier
excludes every exposed rational denominator that is internally empty or ends
with an appended e-zero tail:
$$
\begin{array}{l}
\mathsf{RealConstantHistoryClassifier}(\Eone(\emp),\Eone(d))
\Longrightarrow\mathsf{False},\\
\mathsf{RealConstantHistoryClassifier}(\Eone(d),\Eone(\emp))
\Longrightarrow\mathsf{False},\\
\mathsf{RealConstantHistoryClassifier}(\Eone(p\cdot\Ezero(z)),\Eone(d))
\Longrightarrow\mathsf{False},\\
\mathsf{RealConstantHistoryClassifier}(\Eone(d),\Eone(p\cdot\Ezero(z)))
\Longrightarrow\mathsf{False}.
\end{array}
$$
In the ten-field fit, the traditional target is constant-real malformed
denominator exclusion; the BEDC source is $\mathsf{BHist}$ e-one rational seals
and RatUp positive-denominator shape; the carrier is
$\mathsf{RealConstantHistoryCarrier}$; the classifier is
$\mathsf{RealConstantHistoryClassifier}$; the stability obligations are e-one
exposure, $\hsame$ reflexivity, and RatUp denominator projection; the ledger is
empty and appended e-zero denominator shapes; the core theorem is this four-way
exclusion; the exactness predicate is impossibility of nonpositive exposed
denominator tails; the standard bridge is constant rational real names; and the
status is $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
For each clause, first expose the sealed constant-real classifier with
\autoref{thm:real-constant-history-classifier-e1-iff-rat}. The displayed
assumption becomes a RatUp classifier on the two inner histories.

For the two inner-empty clauses, apply
\autoref{thm:rat-history-classifier-endpoints-not-empty} to the exposed RatUp
classifier. In the left orientation the first nonemptiness row contradicts
$\hsame(\emp,\emp)$ by reflexivity; in the right orientation the second
nonemptiness row gives the same contradiction.

For the left appended e-zero-tail clause, apply
\autoref{thm:rat-history-classifier-positive-denominators} to the exposed RatUp
classifier. The left projection gives a positive unary denominator witness for
$p\cdot\Ezero(z)$, contradicting
\autoref{thm:positive-unary-denominator-append-e0-tail-absurd}.

For the right appended e-zero-tail clause, the same positive-denominator
projection gives the witness on the right endpoint $p\cdot\Ezero(z)$. This is
again impossible by
\autoref{thm:positive-unary-denominator-append-e0-tail-absurd}.
\end{proof}
