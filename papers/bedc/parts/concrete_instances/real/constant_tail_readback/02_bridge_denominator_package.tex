\begin{theorem}[Real constant appended-tail bridge denominator package]
\label{thm:real-constant-appended-tail-bridge-denominator-package}
For histories $d,e,tail_D,tail_E,z_D,z_E:\Hist$ and observation maps
$r,q:\Hist\to\Hist$, set
$$
D=d\cdot\Eone(tail_D)
\qquad\text{and}\qquad
E=e\cdot\Eone(tail_E).
$$
Write the bridge classifier rows as
$$
A=\mathsf{RatHistoryClassifier}(D,E),
\qquad
B=\mathsf{RatStreamNameClassifier}(\mathsf{RatConstStream}_D\circ r,
\mathsf{RatConstStream}_E\circ q),
$$
$$
U=\mathsf{RealUnaryStreamClassifier}(\mathsf{RatConstStream}_D\circ r,
\mathsf{RatConstStream}_E\circ q),
\qquad
C=\mathsf{RealConstantHistoryClassifier}(\Eone(D),\Eone(E)).
$$
Let $P(D,E,z_D,z_E)$ abbreviate
$\mathsf{RealConstantAppendedTailDenominatorPackage}
(d,e,tail_D,tail_E,z_D,z_E)$ from
\autoref{def:real-constant-appended-tail-denominator-package}.
Then every bridge row entails this package:
$$
(A\Longrightarrow P(D,E,z_D,z_E))\land
(B\Longrightarrow P(D,E,z_D,z_E))\land
(U\Longrightarrow P(D,E,z_D,z_E))\land
(C\Longrightarrow P(D,E,z_D,z_E)).
$$
\end{theorem}
\begin{proof}
The $C$ row gives the package directly. Apply
\autoref{thm:real-constant-history-classifier-append-e1-tail-denominator-package}
to the displayed decompositions $D=d\cdot\Eone(tail_D)$ and
$E=e\cdot\Eone(tail_E)$, with the endpoint zero witnesses $z_D$ and $z_E$.
The rows are the named projections of
\autoref{def:real-constant-appended-tail-denominator-package}.

The bridge equivalences are those of
\autoref{thm:real-constant-appended-tail-rational-stream-bridge-readback}.
From $U$, use the final bridge equivalence to obtain $C$, then apply the
preceding package readback.

From $B$, use the second bridge equivalence and then the final bridge
equivalence. This reaches the constant real-history row, so the same package
readback supplies $P(D,E,z_D,z_E)$.

From $A$, use all three bridge equivalences in order. The resulting $C$ row
again yields the complete appended-tail denominator package.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealConstant\_appended\_tail\_bridge\_denominator\_package}

\begin{theorem}[Real constant appended-tail bridge denominator package exactness]
\label{thm:real-constant-appended-tail-bridge-denominator-package-exactness}
For histories $d,e,tail_D,tail_E,z_D,z_E:\Hist$ and observation maps
$r,q:\Hist\to\Hist$, set
$$
D=d\cdot\Eone(tail_D)
\qquad\text{and}\qquad

$$
Write the bridge classifier rows as
$$
A=\mathsf{RatHistoryClassifier}(D,E),
\qquad
B=\mathsf{RatStreamNameClassifier}(\mathsf{RatConstStream}_D\circ r,
\mathsf{RatConstStream}_E\circ q),
$$
$$
U=\mathsf{RealUnaryStreamClassifier}(\mathsf{RatConstStream}_D\circ r,
\mathsf{RatConstStream}_E\circ q),
\qquad
C=\mathsf{RealConstantHistoryClassifier}(\Eone(D),\Eone(E)).
$$
Let $P(D,E,z_D,z_E)$ abbreviate
$\mathsf{RealConstantAppendedTailDenominatorPackage}

Then the denominator package is exact for the bridge rows:

P(D,E,z_D,z_E)\Longleftrightarrow A\land B\land U\land C.

\begin{proof}

carry $A$ to $B$, then to $U$, and then to $C$, so the four bridge rows hold

Conversely assume $A\land B\land U\land C$. Project the final row $C$ and
apply
\autoref{thm:real-constant-history-classifier-append-e1-tail-denominator-package}
to the decompositions $D=d\cdot\Eone(tail_D)$ and
$E=e\cdot\Eone(tail_E)$, with the zero witnesses $z_D$ and $z_E$. This
recovers every row of $P(D,E,z_D,z_E)$ as named in
\autoref{def:real-constant-appended-tail-denominator-package}.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealConstant\_appended\_tail\_bridge\_denominator\_package\_exactness}

\begin{theorem}[Real constant appended-tail bridge single-row denominator package exactness]
\label{thm:real-constant-appended-tail-bridge-single-row-denominator-package-exactness}
For histories $d,e,tail_D,tail_E,z_D,z_E:\Hist$ and observation maps
$r,q:\Hist\to\Hist$, use the same definitions of $D$, $E$, the bridge rows
$A,B,U,C$, and the appended-tail denominator package
$P(D,E,z_D,z_E)$ as in
\autoref{thm:real-constant-appended-tail-bridge-denominator-package-exactness}.
If $D=d\cdot\Eone(tail_D)$ and $E=e\cdot\Eone(tail_E)$, then each concrete
bridge row is exact for the denominator package:
$$
(A\Longleftrightarrow P(D,E,z_D,z_E))\land
(B\Longleftrightarrow P(D,E,z_D,z_E))\land
(U\Longleftrightarrow P(D,E,z_D,z_E))\land
(C\Longleftrightarrow P(D,E,z_D,z_E)).
$$
\end{theorem}
\begin{proof}
The row-to-package directions are the four projections of
\autoref{thm:real-constant-appended-tail-bridge-denominator-package} under the
displayed appended-tail decompositions.

For the package-to-row directions, assume $P(D,E,z_D,z_E)$. Apply
\autoref{thm:real-constant-appended-tail-bridge-denominator-package-exactness}
under the same decompositions. The resulting conjunction
$A\land B\land U\land C$ supplies the corresponding row by projection.

Combining the four row-to-package projections with these four projections from
the exactness conjunction gives the stated rowwise equivalences for
$A,B,U,C$.
\end{proof}

\begin{theorem}[Real constant appended-tail bridge appended e-zero exclusion without tail agreement]
\label{thm:real-constant-appended-tail-bridge-appended-ezero-exclusion}
For histories $d,e,tail_D,tail_E,u_D,u_E,z_D,z_E:\Hist$ and observation maps
$r,q:\Hist\to\Hist$, set
$$
D=d\cdot\Eone(tail_D)
\qquad\text{and}\qquad
E=e\cdot\Eone(tail_E).
$$
Write the bridge classifier rows as
$$
A=\mathsf{RatHistoryClassifier}(D,E),
\qquad
B=\mathsf{RatStreamNameClassifier}(\mathsf{RatConstStream}_D\circ r,
\mathsf{RatConstStream}_E\circ q),
$$
$$
U=\mathsf{RealUnaryStreamClassifier}(\mathsf{RatConstStream}_D\circ r,
\mathsf{RatConstStream}_E\circ q),
\qquad
C=\mathsf{RealConstantHistoryClassifier}(\Eone(D),\Eone(E)).
$$
Let $R(D,E,u_D,u_E,z_D,z_E)$ denote the paired exclusions
$$
(\hsame(D,u_D\cdot\Ezero(z_D))\Longrightarrow\bot)\land
(\hsame(E,u_E\cdot\Ezero(z_E))\Longrightarrow\bot).
$$
Then every bridge row entails $R(D,E,u_D,u_E,z_D,z_E)$:
$$
(A\Longrightarrow R(D,E,u_D,u_E,z_D,z_E))\land
(B\Longrightarrow R(D,E,u_D,u_E,z_D,z_E))\land
(U\Longrightarrow R(D,E,u_D,u_E,z_D,z_E))\land
(C\Longrightarrow R(D,E,u_D,u_E,z_D,z_E)).
$$
\end{theorem}
\begin{proof}
Apply
\autoref{thm:real-constant-appended-tail-bridge-denominator-package}. For each
of the four bridge rows $A,B,U,C$, this supplies the package
$P(D,E,z_D,z_E)$.

Project the two positive-denominator rows of $P$. If
$\hsame(D,u_D\cdot\Ezero(z_D))$, transport the
$\mathsf{PositiveUnaryDenominator}(D)$ witness across the displayed appended
e-zero-tail sameness and contradict
\autoref{thm:positive-unary-denominator-append-e0-tail-absurd}.

The same argument applied to $E$ transports the
$\mathsf{PositiveUnaryDenominator}(E)$ witness across
$\hsame(E,u_E\cdot\Ezero(z_E))$ and gives the second contradiction. These two
contradictions are exactly $R(D,E,u_D,u_E,z_D,z_E)$ for each bridge row.
\end{proof}
\leanchecked{BEDC.Derived.RealUp.RealConstant\_appended\_tail\_bridge\_appended\_e0\_tail\_absurd}
