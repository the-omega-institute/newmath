\begin{theorem}[Real constant appended-tail bridge head-tail agreement equivalence]
\label{thm:real-constant-appended-tail-bridge-head-tail-agreement-equivalence}
For histories $d,e,tail_D,tail_E,z_D,z_E:\Hist$ and observation maps
$r,q:\Hist\to\Hist$, set
$$
D=d\cdot\Eone(tail_D)
\qquad\text{and}\qquad
E=e\cdot\Eone(tail_E).
$$
Write the bridge classifier rows as
$$
A=\mathsf{RatHistoryClassifier}(D,E),
\qquad
B=\mathsf{RatStreamNameClassifier}(\mathsf{RatConstStream}_D\circ r,
\mathsf{RatConstStream}_E\circ q),
$$
$$
U=\mathsf{RealUnaryStreamClassifier}(\mathsf{RatConstStream}_D\circ r,
\mathsf{RatConstStream}_E\circ q),
\qquad
C=\mathsf{RealConstantHistoryClassifier}(\Eone(D),\Eone(E)).
$$
Then each bridge row identifies denominator-head agreement with displayed
e-one-tail agreement:
$$
(A\Longrightarrow(\hsame(d,e)\Longleftrightarrow\hsame(tail_D,tail_E)))\land
(B\Longrightarrow(\hsame(d,e)\Longleftrightarrow\hsame(tail_D,tail_E)))\land
(U\Longrightarrow(\hsame(d,e)\Longleftrightarrow\hsame(tail_D,tail_E)))\land
(C\Longrightarrow(\hsame(d,e)\Longleftrightarrow\hsame(tail_D,tail_E))).
$$
\end{theorem}
\begin{proof}
Fix one of the bridge rows $A,B,U,C$. Apply
\autoref{thm:real-constant-appended-tail-bridge-denominator-package} to the
same displayed data and the chosen row. This recovers the denominator package
$P(D,E,z_D,z_E)$, whose final component is $D\hsame E$.

For the direction from displayed tail agreement to head agreement, assume
$\hsame(tail_D,tail_E)$. Transport this equality through the $\Eone$
constructor and the right append context $e\cdot(-)$, so the endpoint
comparison $D\hsame E$ is read against the common suffix $\Eone(tail_D)$.
BEDC right append cancellation then yields $\hsame(d,e)$.

For the direction from head agreement to displayed tail agreement, assume
$\hsame(d,e)$. Transport this equality through the left append context
$(-)\cdot\Eone(tail_E)$, so the same endpoint comparison is read with common
left head $d$. BEDC left append cancellation gives
$\hsame(\Eone(tail_D),\Eone(tail_E))$, and e-one constructor inversion gives
$\hsame(tail_D,tail_E)$.

The argument uses only the selected row to recover $D\hsame E$, so repeating it
for $A$, $B$, $U$, and $C$ gives the four displayed implications.
\end{proof}

\begin{theorem}[Real constant appended-tail bridge malformed sealed denominator exclusion]
\label{thm:real-constant-appended-tail-bridge-malformed-sealed-denominator-exclusion}
For histories $d,e,tail_D,tail_E,z_D,z_E:\Hist$ and observation maps
$r,q:\Hist\to\Hist$, set
$$
D=d\cdot\Eone(tail_D)
\qquad\text{and}\qquad
E=e\cdot\Eone(tail_E).
$$
Write the bridge classifier rows as
$$
A=\mathsf{RatHistoryClassifier}(D,E),
\qquad
B=\mathsf{RatStreamNameClassifier}(\mathsf{RatConstStream}_D\circ r,
\mathsf{RatConstStream}_E\circ q),
$$
$$
U=\mathsf{RealUnaryStreamClassifier}(\mathsf{RatConstStream}_D\circ r,
\mathsf{RatConstStream}_E\circ q),
\qquad
C=\mathsf{RealConstantHistoryClassifier}(\Eone(D),\Eone(E)).
$$
Let $M(D,E)$ denote the paired malformed sealed denominator exclusion
$$
\bigl(\mathsf{MalformedRealSealedDenom}(\Eone(D))
\Longrightarrow\mathsf{False}\bigr)
\land
\bigl(\mathsf{MalformedRealSealedDenom}(\Eone(E))
\Longrightarrow\mathsf{False}\bigr).
$$
Then every bridge row entails the same exclusion predicate:
$$
(A\Longrightarrow M(D,E))\land
(B\Longrightarrow M(D,E))\land
(U\Longrightarrow M(D,E))\land
(C\Longrightarrow M(D,E)).
$$
\end{theorem}
\begin{proof}
Apply
\autoref{thm:real-constant-appended-tail-bridge-denominator-package} to the
displayed data. For any chosen bridge row among $A,B,U,C$, this recovers the
package $P(D,E,z_D,z_E)$, including the positive-denominator rows and the
nonempty rows for $D$ and $E$.

Unfold
\autoref{def:real-constant-malformed-sealed-denominator-display} at
$\Eone(D)$. The endpoint $D$ has three branches.
\begin{enumerate}
  \item In the sealed-empty branch, e-one constructor inversion turns
  $\hsame(\Eone(D),\Eone(\emp))$ into $D\hsame\emp$, contradicting the
  nonempty row for $D$ from the package.

  \item In the sealed inner e-zero branch, use the displayed witness $t$.
  E-one constructor inversion turns
  $\hsame(\Eone(D),\Eone(\Ezero(t)))$ into $D\hsame\Ezero(t)$. Transport the
  positive-denominator witness for $D$ across this sameness by
  \autoref{thm:rat-positive-unary-denominator-hsame-transport}; the
  transported witness contradicts
  \autoref{thm:rat-positive-denominator-e0-absurd}.

  \item In the sealed appended e-zero branch, use the displayed witnesses
  $p,z$. E-one constructor inversion turns
  $\hsame(\Eone(D),\Eone(p\cdot\Ezero(z)))$ into
  $D\hsame p\cdot\Ezero(z)$. Transport the positive-denominator witness for
  $D$ across this sameness by
  \autoref{thm:rat-positive-unary-denominator-hsame-transport}; the
  transported witness contradicts
  \autoref{thm:positive-unary-denominator-append-e0-tail-absurd}.
\end{enumerate}

Unfold
\autoref{def:real-constant-malformed-sealed-denominator-display} at
$\Eone(E)$. The endpoint $E$ has the same three branches.
\begin{enumerate}
  \item In the sealed-empty branch, e-one constructor inversion turns
  $\hsame(\Eone(E),\Eone(\emp))$ into $E\hsame\emp$, contradicting the
  nonempty row for $E$ from the package.

  \item In the sealed inner e-zero branch, use the displayed witness $t$.
  E-one constructor inversion turns
  $\hsame(\Eone(E),\Eone(\Ezero(t)))$ into $E\hsame\Ezero(t)$. Transport the
  positive-denominator witness for $E$ across this sameness by
  \autoref{thm:rat-positive-unary-denominator-hsame-transport}; the
  transported witness contradicts
  \autoref{thm:rat-positive-denominator-e0-absurd}.

  \item In the sealed appended e-zero branch, use the displayed witnesses
  $p,z$. E-one constructor inversion turns
  $\hsame(\Eone(E),\Eone(p\cdot\Ezero(z)))$ into
  $E\hsame p\cdot\Ezero(z)$. Transport the positive-denominator witness for
  $E$ across this sameness by
  \autoref{thm:rat-positive-unary-denominator-hsame-transport}; the
  transported witness contradicts
  \autoref{thm:positive-unary-denominator-append-e0-tail-absurd}.
\end{enumerate}

These two endpoint exclusions are exactly $M(D,E)$ for each of the four
bridge-row implications.
\end{proof}

\begin{theorem}[Real constant appended-tail bridge full denominator sanity exactness]
\label{thm:real-constant-appended-tail-bridge-full-denominator-sanity-exactness}
For histories $d,e,tail_D,tail_E,u_D,u_E,z_D,z_E:\Hist$ and observation maps
$r,q:\Hist\to\Hist$, use the same definitions of $D$, $E$, the bridge rows
$A,B,U,C$, the full denominator-tail package $Q(D,E,u_D,u_E,z_D,z_E)$, and
the malformed sealed denominator exclusion $M(D,E)$ as in
\autoref{thm:real-constant-appended-tail-bridge-full-denominator-tail-package-exactness}
and
\autoref{thm:real-constant-appended-tail-bridge-malformed-sealed-denominator-exclusion}.
Let $S(D,E,u_D,u_E,z_D,z_E)$ denote the combined denominator-sanity package
$$
S(D,E,u_D,u_E,z_D,z_E)\mathrel{:=}Q(D,E,u_D,u_E,z_D,z_E)\land M(D,E).
$$
If $\hsame(tail_D,tail_E)$, then the combined package is exact for the bridge
rows:
$$
S(D,E,u_D,u_E,z_D,z_E)\Longleftrightarrow A\land B\land U\land C.
$$
\end{theorem}
\begin{proof}
Assume $\hsame(tail_D,tail_E)$.

For the forward direction, assume $S(D,E,u_D,u_E,z_D,z_E)$. Project its
$Q(D,E,u_D,u_E,z_D,z_E)$ component and apply
\autoref{thm:real-constant-appended-tail-bridge-full-denominator-tail-package-exactness}
under the tail agreement. This gives $A\land B\land U\land C$.

For the reverse direction, assume $A\land B\land U\land C$. Apply
\autoref{thm:real-constant-appended-tail-bridge-full-denominator-tail-package-exactness}
under the same tail agreement to recover
$Q(D,E,u_D,u_E,z_D,z_E)$.

It remains to build $M(D,E)$. Apply
\autoref{thm:real-constant-appended-tail-bridge-malformed-sealed-denominator-exclusion}
to the displayed data. The first projected implication has type
$A\Longrightarrow M(D,E)$, and the $A$ row supplied by the row conjunction gives
$M(D,E)$.

Conjoining the recovered $Q(D,E,u_D,u_E,z_D,z_E)$ with $M(D,E)$ is exactly
$S(D,E,u_D,u_E,z_D,z_E)$.
\end{proof}
