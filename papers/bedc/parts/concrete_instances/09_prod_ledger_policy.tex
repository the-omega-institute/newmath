\begin{definition}[Prod ledger policy]
\label{def:prod-ledger-policy}
Record a raw product history carrying both component ledgers together with an
\(\hsame\) link to the visible history exposed by the ledger.
\end{definition}
\leandef{BEDC.Derived.ProdUp.ProdHistoryLedgerPolicy}

\begin{theorem}[Prod ledger visible carrier]
\label{thm:prod-ledger-visible-carrier}
If a visible history is linked by \(\hsame\) to a raw product-ledger history,
then the visible history is itself a product carrier.
\end{theorem}
\begin{proof}
Unpack the raw product carrier and transport it across the stored \(\hsame\)
link by product-carrier transport.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdHistoryLedgerPolicy\_visible\_carrier}

\begin{theorem}[Prod ledger raw-visible classifier]
\label{thm:prod-ledger-raw-visible-classifier}
If $\mathsf{ProdHistoryLedgerPolicy}(L,R,\rho,v)$ holds, then
$\mathsf{ProdHistoryClassifier}(L,R,\rho,v)$ holds.
\end{theorem}
\begin{proof}
Unpack the ledger into the raw product carrier for $\rho$ and the stored
$\hsame(\rho,v)$ link. The visible-carrier theorem supplies the product
carrier for $v$. These three witnesses are exactly the product classifier
fields for the raw ledger history and the visible history.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdHistoryLedgerPolicy\_raw\_visible\_classifier}

\begin{theorem}[Prod ledger history-same transport]
\label{thm:prod-ledger-hsame-transport}
Let $L,R:\mathsf{BHist}\to\mathsf{Prop}$ be source carriers and let
$\rho,\rho',v,v':\mathsf{BHist}$ be histories. If $\hsame(\rho,\rho')$,
$\hsame(v,v')$, and $\mathsf{ProdHistoryLedgerPolicy}(L,R,\rho,v)$ hold, then
$\mathsf{ProdHistoryLedgerPolicy}(L,R,\rho',v')$ holds. The object fit is:
traditional target ordered-product ledger stability; BEDC source
$\mathsf{BHist}$ with $\Cont$ and $\hsame$; carrier
$\mathsf{ProdHistoryCarrier}$; classifier $\mathsf{ProdHistoryClassifier}$;
ledger $\mathsf{ProdHistoryLedgerPolicy}$; exactness raw-visible transport;
standard bridge displayed ordered-pair ledger; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Unpack the ledger into a raw product carrier for $\rho$ and the stored
raw-visible link $\hsame(\rho,v)$. Transport the raw carrier across
$\hsame(\rho,\rho')$ by Theorem~\ref{thm:prod-history-carrier-hsame-transport}.
For the visible link, compose symmetry of $\hsame(\rho,\rho')$, the stored
raw-visible link, and $\hsame(v,v')$ to obtain $\hsame(\rho',v')$. The
transported raw carrier and this composed link rebuild
$\mathsf{ProdHistoryLedgerPolicy}(L,R,\rho',v')$.
\end{proof}

\begin{theorem}[Prod ledger classifier composition]
\label{thm:prod-ledger-classifier-extension}
Let $L,R:\mathsf{BHist}\to\mathsf{Prop}$ be source carriers and let
$\rho,v,w:\mathsf{BHist}$ be histories. If
$\mathsf{ProdHistoryLedgerPolicy}(L,R,\rho,v)$ holds and
$\mathsf{ProdHistoryClassifier}(L,R,v,w)$ holds, then
$\mathsf{ProdHistoryClassifier}(L,R,\rho,w)$ holds. The object fit is:
traditional target ordered-product ledger propagation; BEDC source
$\mathsf{BHist}$ with $\Cont$; carrier $\mathsf{ProdHistoryCarrier}$;
classifier $\mathsf{ProdHistoryClassifier}$; ledger evidence the raw-visible
pair; exactness raw-to-visible-to-target composition; status
$\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:prod-ledger-raw-visible-classifier} to the ledger
policy to obtain $\mathsf{ProdHistoryClassifier}(L,R,\rho,v)$. Compose that
raw-visible classifier with the visible-target classifier by product history
classifier transitivity, yielding
$\mathsf{ProdHistoryClassifier}(L,R,\rho,w)$.
\end{proof}

\begin{theorem}[Prod ledger two-step classifier composition]
\label{thm:prod-ledger-two-step-classifier-composition}
Let $L,R:\mathsf{BHist}\to\mathsf{Prop}$ be source carriers and let
$\rho,v,w:\mathsf{BHist}$ be histories. If
$\mathsf{ProdHistoryLedgerPolicy}(L,R,\rho,v)$ and
$\mathsf{ProdHistoryLedgerPolicy}(L,R,v,w)$ hold, then
$\mathsf{ProdHistoryClassifier}(L,R,\rho,w)$ holds. The object fit is:
ordered-product ledger chain; BEDC source $\mathsf{BHist}$ with $\Cont$;
carrier $\mathsf{ProdHistoryCarrier}$; classifier
$\mathsf{ProdHistoryClassifier}$; ledger evidence raw-visible-visible;
exactness by classifier composition.
\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:prod-ledger-raw-visible-classifier} to both ledger
policies to obtain classifiers from $\rho$ to $v$ and from $v$ to $w$. Compose
them by Theorem~\ref{thm:prod-history-classifier-transitivity}.
\end{proof}

\begin{theorem}[Prod ledger component exposure]
\label{thm:prod-ledger-component-exposure}
If $\mathsf{ProdHistoryLedgerPolicy}(L,R,\rho,v)$ holds, then there exist histories $l$ and $r$ such that $L(l)$, $R(r)$, $\Cont(l,r,\rho)$, $\hsame(\rho,v)$, $\mathsf{ProdHistoryCarrier}(L,R,v)$, and $\mathsf{ProdHistoryClassifier}(L,R,\rho,v)$ all hold.
\end{theorem}
\begin{proof}
Unfold the ledger into the raw product carrier for $\rho$ and the stored raw-visible witness $\hsame(\rho,v)$. Unfold the raw carrier to expose component histories $l$ and $r$ with $L(l)$, $R(r)$, and $\Cont(l,r,\rho)$. Theorem~\ref{thm:prod-ledger-visible-carrier} supplies $\mathsf{ProdHistoryCarrier}(L,R,v)$, and Theorem~\ref{thm:prod-ledger-raw-visible-classifier} supplies $\mathsf{ProdHistoryClassifier}(L,R,\rho,v)$. These witnesses give the stated component exposure.
\end{proof}
