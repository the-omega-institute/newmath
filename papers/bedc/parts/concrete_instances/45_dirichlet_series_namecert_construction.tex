\chapter{A Concrete Naming Certificate for \(\DirichletSeriesUp\)}
\label{ch:concrete-instances-dirichlet-series-namecert}

This chapter gives a concrete naming-certificate construction for Dirichlet series \(\sum_{n=1}^{\infty} a_n n^{-s}\) with complex argument \(s\). The construction sits above \(\ComplexSeriesUp\) (\autoref{ch:concrete-instances-complex-series-namecert}) and \(\NatUp\) (\autoref{ch:concrete-instances-nat-namecert}). The abscissa of absolute convergence is given as an explicit \(\RealUp\)-history witness rather than a real supremum; convergence in the open half-plane \(\Re(s) > \sigma_a\) is proved by majorant comparison; holomorphicity in the half-plane is established with a uniform modulus over compact sub-half-planes; the Euler product over primes (\autoref{ch:concrete-instances-prime-namecert}) is supplied for completely multiplicative coefficients. The construction is fully constructive and uses no \(\mathsf{Classical.choice}\), no \(\mathsf{Quot.sound}\), and no \(\mathsf{propext}\), so the certificate's Lean targets fall under the project's axiom-purity gate.

\section{Dirichlet term and partial sum}

\begin{definition}[Dirichlet term]
\label{def:dir-term}
For coefficient sequence \(a : \Unary \to \ComplexUp\)-history (with \(a_0\) ignored, indexing starts at \(\Eone(\emp)\)), positive unary \(n\), complex history \(s\), the Dirichlet term \(\mathsf{DirTerm}(a, n, s, t_n)\) is the explicit complex history \(t_n = a_n \cdot n^{-s}\), where \(n^{-s}\) is computed via \(\exp(-s \log n)\) on the rational/real-history representation of \(\log n\) (defer constructive log to the \(\RealUp\)-namecert machinery, \autoref{ch:concrete-instances-real-namecert}).
\end{definition}
\leandef{BEDC.Derived.DirichletSeriesUp.DirTerm}

\begin{lemma}[Dirichlet term well-defined]
\label{lem:dir-term-well-defined}
For every \((a, n, s)\) with positive unary \(n\) and complex history \(s\), \(\mathsf{DirTerm}(a, n, s, t_n)\) determines a unique complex history \(t_n\) up to \(\hsame\).
\end{lemma}
\leanstmt{BEDC.Derived.DirichletSeriesUp.dir\_term\_well\_defined}
\begin{proof}
Constructive existence and uniqueness of \(\log n\) on positive unary \(n\); composition with constructive complex \(\exp\) and complex multiplication.
\end{proof}

\begin{definition}[Dirichlet partial sum]
\label{def:dir-part-sum}
For \(a, s\) and unary \(N\), the Dirichlet partial sum \(\mathsf{DirPartSum}(a, s, N, S_N)\) is the complex-history value of \(\sum_{n=1}^{N} a_n n^{-s}\), defined by an explicit unary recursion.
\end{definition}
\leandef{BEDC.Derived.DirichletSeriesUp.DirPartSum}

\begin{lemma}[Dirichlet partial sums are complex histories]
\label{lem:dir-part-sum-complex}
For every \((a, s, N)\), there is a unique complex history \(S_N\) with \(\mathsf{DirPartSum}(a, s, N, S_N)\), up to \(\hsame\).
\end{lemma}
\leanstmt{BEDC.Derived.DirichletSeriesUp.dir\_part\_sum\_complex}
\begin{proof}
By unary induction on \(N\), using totality of \(\mathsf{DirTerm}\) and complex addition.
\end{proof}

\section{Abscissa of absolute convergence}

\begin{definition}[Abscissa of absolute convergence]
\label{def:abs-conv-abscissa}
For coefficient sequence \(a\), the abscissa of absolute convergence \(\mathsf{AbsConvAbscissa}(a, \sigma_a)\) is the predicate that \(\sigma_a\) is a real-history with the property: for every unary precision \(k\) and every complex \(s\) with the rational representation of \(\Re(s) - \sigma_a \geq 2^{-k}\), the modulus partial sums of the Dirichlet series at \(s\) are uniformly bounded above by an explicit constant.
\end{definition}
\leandef{BEDC.Derived.DirichletSeriesUp.AbsConvAbscissa}

\begin{theorem}[Abscissa is unique]
\label{thm:abs-conv-abscissa-unique}
If \(\mathsf{AbsConvAbscissa}(a, \sigma_a)\) and \(\mathsf{AbsConvAbscissa}(a, \sigma_a')\), then \(\hsame(\sigma_a, \sigma_a')\) on the rational representation of the real-history.
\end{theorem}
\leanstmt{BEDC.Derived.DirichletSeriesUp.abs\_conv\_abscissa\_unique}
\begin{proof}[Sketch]
Suppose distinct. Pick a rational \(\sigma\) strictly between them. The bound holds from one side and fails from the other; contradiction.
\end{proof}

\section{Convergence and holomorphicity in the half-plane}

\begin{theorem}[Half-plane convergence]
\label{thm:half-plane-convergence}
If \(\mathsf{AbsConvAbscissa}(a, \sigma_a)\) and \(s\) satisfies \(\Re(s) > \sigma_a\) (decidable strict inequality with explicit witness gap \(\epsilon\)), then \(\sum a_n n^{-s}\) converges absolutely in \(\ComplexSeriesUp\) with explicit modulus given by majorant comparison to \(\sum n^{-(\sigma_a + \epsilon/2)}\).
\end{theorem}
\leanstmt{BEDC.Derived.DirichletSeriesUp.half\_plane\_convergence}
\begin{proof}[Sketch]
The bound \(|a_n n^{-s}| \leq C n^{-(\sigma_a + \epsilon/2)}\) is given by \(\mathsf{AbsConvAbscissa}\) at precision \(k\) chosen by \(\epsilon\); the resulting series converges with explicit geometric-like modulus.
\end{proof}

\begin{theorem}[Half-plane holomorphicity]
\label{thm:half-plane-holomorphic}
The Dirichlet sum function \(s \mapsto \sum_{n=1}^{\infty} a_n n^{-s}\) is holomorphic in the open half-plane \(\Re(s) > \sigma_a\), with uniform modulus over each compact sub-half-plane \(\{s : \Re(s) \geq \sigma_a + \epsilon, |\Im(s)| \leq T\}\) for explicit unary \(\epsilon, T\) (cite \autoref{ch:concrete-instances-holomorphic-namecert}).
\end{theorem}
\leanstmt{BEDC.Derived.DirichletSeriesUp.half\_plane\_holomorphic}
\begin{proof}[Sketch]
Term-wise complex differentiability of \(n^{-s}\) gives derivative \(-(\log n) n^{-s}\); the differentiated series converges absolutely on the compact sub-half-plane by the same majorant.
\end{proof}

\section{Euler product for multiplicative coefficients}

\begin{definition}[Completely multiplicative coefficient]
\label{def:completely-multiplicative}
A coefficient sequence \(a\) is completely multiplicative when \(a_1 = \Eone(\emp)\) and for all positive unary \(m, n\), \(a_{m \cdot n} = a_m \cdot a_n\) (product in \(\ComplexUp\)).
\end{definition}
\leandef{BEDC.Derived.DirichletSeriesUp.CompletelyMultiplicative}

\begin{theorem}[Euler product]
\label{thm:euler-product}
If \(a\) is completely multiplicative and \(\Re(s) > \sigma_a\), the Dirichlet series factors as the explicit Euler product \(\sum a_n n^{-s} = \prod_p (1 - a_p p^{-s})^{-1}\) over primes \(p\) (in the sense of \(\PrimeUp\), \autoref{ch:concrete-instances-prime-namecert}). The product converges absolutely with explicit modulus over compact sub-half-planes.
\end{theorem}
\leanstmt{BEDC.Derived.DirichletSeriesUp.euler\_product}
\begin{proof}[Sketch]
Group terms by prime factorization (\(\mathsf{factorization\_exists}\) of \autoref{ch:concrete-instances-prime-namecert}); the geometric series \(\sum_{k} (a_p p^{-s})^k\) gives the local factor at each prime.
\end{proof}

\section{The certificate}

\begin{definition}[Dirichlet source specification]
\label{def:dir-source-spec}
The source specification on \((a, s)\) carries the constraint \(\mathsf{AbsConvAbscissa}(a, \sigma_a)\) and \(\Re(s) > \sigma_a\).
\end{definition}
\leandef{BEDC.Derived.DirichletSeriesUp.DirSourceSpec}

\begin{definition}[Dirichlet pattern specification]
\label{def:dir-pattern-spec}
The pattern is the regular Cauchy partial-sum predicate of \autoref{ch:concrete-instances-complex-series-namecert} applied to the Dirichlet term sequence.
\end{definition}
\leandef{BEDC.Derived.DirichletSeriesUp.DirPatternSpec}

\begin{definition}[Dirichlet classifier specification]
\label{def:dir-classifier-spec}
Two convergent Dirichlet series at the same \(s\) are classified together when their limits agree under \(\hsame\) on both real components.
\end{definition}
\leandef{BEDC.Derived.DirichletSeriesUp.DirClassifierSpec}

\begin{definition}[Dirichlet stability certificate]
\label{def:dir-stability-cert}
The stability certificate consists of \(\hsame\)-respect on coefficients and on \(s\), transport of the abscissa under \(\hsame\), and compatibility with both partial-sum and limit predicates.
\end{definition}
\leandef{BEDC.Derived.DirichletSeriesUp.DirStabilityCert}

\begin{definition}[Dirichlet ledger policy]
\label{def:dir-ledger-policy}
The ledger policy records: (i) absolute convergence in the open half-plane, (ii) holomorphicity over compact sub-half-planes, (iii) Euler product for completely multiplicative coefficients, (iv) coefficient linearity in the half-plane.
\end{definition}
\leandef{BEDC.Derived.DirichletSeriesUp.DirLedgerPolicy}

\begin{theorem}[Concrete \(\DirichletSeriesUp\) naming certificate]
\label{thm:dirichlet-namecert}
The five fields above form a naming certificate over \(\Hist\) for the Dirichlet series interface \(\DirichletSeriesUp\).
\end{theorem}
\leanstmt{BEDC.Derived.DirichletSeriesUp.dirichlet\_namecert}
\begin{proof}[Sketch]
The five fields satisfy the closure laws of the naming-certificate framework (\autoref{ch:core-typed-naming-certificates}). The certificate uses no \(\mathsf{Classical.choice}\), no \(\mathsf{Quot.sound}\), and no \(\mathsf{propext}\), so the certificate's Lean targets fall under the project's axiom-purity gate.
\end{proof}
