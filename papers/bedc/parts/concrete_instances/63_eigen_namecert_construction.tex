\chapter{A Concrete Naming Certificate for $\EigenUp$}
\label{ch:concrete-instances-eigen-namecert}
\origin{human}

The eigen-pair interface $\EigenUp$ packages an eigenvalue / eigenvector pair $(\lambda, v)$ for a linear endomorphism. It depends on $\NameCert_{\LinearMapUp}$ (\autoref{ch:concrete-instances-linearmap-namecert}) and $\NameCert_{\DeterminantUp}$ (\autoref{ch:concrete-instances-determinant-namecert}), with $\lambda$ certified as a root of the characteristic polynomial.

\section{Carrier definition}
\label{sec:eigen-carrier}

\begin{definition}[Eigen component singleton carrier]
\label{def:eigen-component-singleton-carrier}
For histories $f,\lambda,v,pair:\mathsf{BHist}$, the component singleton
eigen carrier requires $f$ and $v$ to be singleton linear-map histories,
$\lambda$ to be a singleton commutative-ring history, and $pair$ to be the
BEDC continuation endpoint of $\lambda$ followed by $v$.
\end{definition}
\leandef{BEDC.Derived.EigenUp.EigenComponentSingletonCarrier}

\begin{theorem}[Eigen component singleton carrier singleton append]
\label{thm:eigen-component-singleton-carrier-singleton-append}
For histories $m,\lambda,v,pair:\mathsf{BHist}$, if the component singleton
eigen carrier holds for $(m,\lambda,v,pair)$ and $\lambda$ also carries the
determinant singleton scalar certificate, then
$\mathsf{append}(m,pair)$ lies in the eigen singleton carrier.
\end{theorem}
\begin{proof}
Unpack the component carrier into the linear-map carrier for $m$, the
linear-map carrier for $v$, and the continuation endpoint
$\Cont(\lambda,v,pair)$. The determinant singleton carrier for $\lambda$
supplies the scalar witness required by the eigen singleton carrier. The
stored continuation rewrites $\mathsf{append}(m,pair)$ as
$\mathsf{append}(m,\mathsf{append}(\lambda,v))$, giving the required BEDC
continuation endpoint.
\end{proof}
\leanchecked{BEDC.Derived.EigenUp.EigenComponentSingletonCarrier\_singleton\_append}

\begin{theorem}[Eigen component singleton carrier continuation result transport]
\label{thm:eigen-component-singleton-carrier-cont-result-transport}
For histories \(m,\lambda,v,pair,full,full':\mathsf{BHist}\), suppose the
component singleton eigen carrier holds for \((m,\lambda,v,pair)\),
\(\lambda\) carries the determinant singleton scalar certificate,
\(\Cont(m,pair,full)\), and \(full\hsame full'\). Then \(full'\) lies in the
eigen singleton carrier.
\end{theorem}
\begin{proof}
Unpack the component carrier to expose the linear-map carriers for \(m\) and
\(v\) and the scalar-vector continuation \(\Cont(\lambda,v,pair)\). Compose
this continuation with \(\Cont(m,pair,full)\) to rewrite the endpoint as
\(\Cont(m,\mathsf{append}(\lambda,v),full)\). The determinant singleton carrier
for \(\lambda\) and the final \(\hsame\)-transport place \(full'\) in the
eigen singleton carrier.
\end{proof}
\leanchecked{BEDC.Derived.EigenUp.EigenComponentSingletonCarrier\_cont\_result\_transport}

\begin{theorem}[Eigen component singleton carrier pair determinacy]
\label{thm:eigen-component-singleton-carrier-pair-deterministic}
For histories $f,\lambda,v,pair,pair':\mathsf{BHist}$, two component
singleton eigen carriers over the same $(f,\lambda,v)$ determine the same
scalar-vector endpoint $\hsame(pair,pair')$. Both stored continuation ledgers
$\Cont(\lambda,v,pair)$ and $\Cont(\lambda,v,pair')$ remain available.
\end{theorem}
\begin{proof}
Unpack both component singleton carriers. They contain the two BEDC
continuation ledgers with the same source $\lambda$ and tail $v$. The
kernel determinacy theorem for continuation endpoints identifies their
results, and the two ledgers are retained directly from the carriers.
\end{proof}
\leanchecked{BEDC.Derived.EigenUp.EigenComponentSingletonCarrier\_pair\_deterministic}

\begin{theorem}[Eigen component singleton carrier append classifier]
\label{thm:eigen-component-singleton-carrier-append-classifier}
For histories $m,m',\lambda,\lambda',v,v',pair,pair':\mathsf{BHist}$, suppose
the component singleton eigen carriers hold for $(m,\lambda,v,pair)$ and
$(m',\lambda',v',pair')$, both scalar components carry determinant singleton
certificates, and the three displayed components satisfy
$m\hsame m'$, $\lambda\hsame\lambda'$, and $v\hsame v'$. Then the append
endpoints $\mathsf{append}(m,pair)$ and $\mathsf{append}(m',pair')$ are
classified by the eigen singleton classifier.
\end{theorem}
\begin{proof}
The component carriers and determinant singleton scalar certificates give the
two eigen singleton carrier endpoints by
\autoref{thm:eigen-component-singleton-carrier-singleton-append}. Unpacking the
component carriers exposes the two BEDC continuations
$\Cont(\lambda,v,pair)$ and $\Cont(\lambda',v',pair')$. Continuation respects
the scalar and vector history equalities, so $pair\hsame pair'$. Applying the
same continuation-respect principle to the map equality and these pair
endpoints identifies $\mathsf{append}(m,pair)$ with
$\mathsf{append}(m',pair')$, closing the restricted classifier.
\end{proof}
\leanchecked{BEDC.Derived.EigenUp.EigenComponentSingletonCarrier\_append\_classifier}

\begin{theorem}[Eigen singleton carrier pair emptiness]
\label{thm:eigen-singleton-carrier-pair-empty}
For histories $f,\lambda,v,pair:\mathsf{BHist}$, the component singleton
eigen carrier is equivalent to the three singleton component carriers
together with $\hsame(pair,\emp)$.
\end{theorem}
\begin{proof}
Unfold the component singleton eigen carrier. In the forward direction, the
scalar and vector component carriers identify $\lambda$ and $v$ with $\emp$,
so $\mathsf{append}(\lambda,v)$ is empty by the BEDC append-empty inversion,
and the stored continuation transports this emptiness to $pair$. In the
reverse direction, the same append-empty fact reconstructs the continuation
endpoint from the given $\hsame(pair,\emp)$ witness.
\end{proof}
\leanchecked{BEDC.Derived.EigenUp.EigenSingletonCarrier\_pair\_empty\_iff}

\begin{definition}[Eigen singleton carrier]
\label{def:eigen-singleton-carrier}
A history $p$ lies in the singleton eigen carrier when there are histories
$m,\lambda,v$ such that $m$ and $v$ are singleton linear-map carriers,
$\lambda$ is a singleton determinant carrier, and $p$ is the continuation
result
$$
\Cont(m,\mathsf{append}(\lambda,v),p).
$$
\end{definition}
\leandef{BEDC.Derived.EigenUp.EigenSingletonCarrier}

\begin{theorem}[Eigen singleton carrier continuation result transport]
\label{thm:eigen-singleton-carrier-cont-result-transport}
For histories $m,\lambda,v,p,p':\mathsf{BHist}$, if $m$ and $v$ are singleton
linear-map carriers, $\lambda$ is a singleton determinant carrier,
$\Cont(m,\mathsf{append}(\lambda,v),p)$, and $p\hsame p'$, then $p'$ lies in
the eigen singleton carrier.
\end{theorem}
\begin{proof}
Reuse the same displayed witnesses $m,\lambda,v$. The carrier fields are
unchanged, and the continuation result transports along $p\hsame p'$ by the
kernel result-transport theorem for $\Cont$.
\end{proof}
\leanchecked{BEDC.Derived.EigenUp.EigenSingletonCarrier\_cont\_result\_transport}

\begin{theorem}[Eigen singleton carrier empty iff]
\label{thm:eigen-singleton-carrier-empty-iff}
For every history $p:\mathsf{BHist}$, the singleton eigen carrier holds for
$p$ exactly when $p\hsame\emp$.
\end{theorem}
\begin{proof}
For the forward direction, unpack the map, scalar, and vector witnesses. The
linear-map and determinant singleton carriers identify each component with
$\emp$, so the nested append endpoint is empty; the stored continuation
transports this emptiness to $p$. For the reverse direction, choose all three
witness components to be $\emp$ and use the displayed $p\hsame\emp$ witness
as the empty continuation result.
\end{proof}
\leanchecked{BEDC.Derived.EigenUp.EigenSingletonCarrier\_empty\_iff}

\begin{theorem}[Eigen singleton continuation sources empty]
\label{thm:eigen-singleton-carrier-cont-sources-empty}
For BEDC histories $map$, $payload$, and $pair$, if $pair$ lies in the eigen
singleton carrier and $\Cont(map,payload,pair)$, then both continuation sources
are empty:
$$
\mathsf{EigenSingletonCarrier}(pair)\land \Cont(map,payload,pair)
\Longrightarrow map\hsame\emp\ \wedge\ payload\hsame\emp.
$$
\end{theorem}
\begin{proof}
The carrier-empty equivalence identifies $pair$ with $\emp$. The continuation
payload identifies $pair$ with $\mathsf{append}(map,payload)$, hence the
append endpoint is empty. BEDC append-empty inversion splits this endpoint into
the two empty source comparisons.
\end{proof}
\leanchecked{BEDC.Derived.EigenUp.EigenSingletonCarrier\_cont\_sources\_empty}

\begin{theorem}[Eigen singleton carrier append-context empty iff]
\label{thm:eigen-singleton-carrier-append-context-empty-iff}
For histories $L,R,h:\mathsf{BHist}$, the contextual eigen singleton carrier
holds for $\mathsf{append}(L,\mathsf{append}(h,R))$ exactly when the two
context histories are empty and the middle history lies in the eigen singleton
carrier:
$$
\mathsf{EigenSingletonCarrier}(\mathsf{append}(L,\mathsf{append}(h,R)))
\Longleftrightarrow
L\hsame\emp\ \wedge\ \mathsf{EigenSingletonCarrier}(h)\ \wedge\ R\hsame\emp.
$$
\end{theorem}
\begin{proof}
Use the eigen singleton carrier empty equivalence to convert the contextual
carrier into emptiness of the nested append. BEDC append-empty inversion splits
the outer and inner append endpoints, yielding empty left and right contexts and
empty middle endpoint. The reverse direction rebuilds the nested append-empty
witness from the three displayed fields and applies the same carrier-empty
equivalence backwards.
\end{proof}
\leanchecked{BEDC.Derived.EigenUp.EigenSingletonCarrier\_append\_context\_empty\_iff}

\begin{theorem}[Eigen singleton carrier continuation left-unit result]
\label{thm:eigen-singleton-carrier-cont-left-unit-result}
For histories $pair,q,result:\mathsf{BHist}$, if $pair$ lies in the eigen
singleton carrier and $\Cont(pair,q,result)$, then $result\hsame q$.
\end{theorem}
\begin{proof}
The eigen singleton carrier empty equivalence identifies $pair$ with $\emp$.
After this source readback, the displayed continuation has empty source, so
the BEDC left-unit result theorem for $\Cont$ reads the endpoint back to the
tail $q$.
\end{proof}
\leanchecked{BEDC.Derived.EigenUp.EigenSingletonCarrier\_cont\_left\_unit\_result}

\begin{theorem}[Eigen singleton carrier continuation right-unit result]
\label{thm:eigen-singleton-carrier-cont-right-unit-result}
For histories $q,pair,result:\mathsf{BHist}$, if $pair$ lies in the eigen
singleton carrier and $\Cont(q,pair,result)$, then $result\hsame q$.
\end{theorem}
\begin{proof}
The carrier-empty equivalence identifies $pair$ with $\emp$. Transport the
displayed continuation across that endpoint comparison and compare it with the
right-unit continuation at $q$; continuation determinacy gives the result.
\end{proof}
\leanchecked{BEDC.Derived.EigenUp.EigenSingletonCarrier\_cont\_right\_unit\_result}

\section{The certificate}
\label{sec:eigen-certificate}

\begin{definition}[Eigen singleton classifier]
\label{def:eigen-singleton-classifier}
Two histories $p,q:\mathsf{BHist}$ are classified by the singleton eigen
interface when both lie in the singleton eigen carrier and $p\hsame q$.
\end{definition}
\leandef{BEDC.Derived.EigenUp.EigenSingletonClassifier}

\begin{theorem}[Eigen singleton classifier append-context empty iff]
\label{thm:eigen-singleton-classifier-append-context-empty-iff}
For histories $L,R,h,k:\mathsf{BHist}$, the contextual singleton eigen
classifier between $\mathsf{append}(L,h)$ and $\mathsf{append}(k,R)$ is
equivalent to empty endpoint contexts and the core classifier between $h$ and
$k$:
$$
\mathsf{EigenSingletonClassifier}(\mathsf{append}(L,h),\mathsf{append}(k,R))
\Longleftrightarrow
L\hsame\emp\ \wedge\ \mathsf{EigenSingletonClassifier}(h,k)\ \wedge\ R\hsame\emp.
$$
\end{theorem}
\begin{proof}
The forward direction sends both contextual carrier fields through the eigen
carrier empty equivalence and splits the two append-empty witnesses. These
middle empty witnesses give the core carriers and identify $h$ with $k$. In the
reverse direction, the empty endpoint contexts and the two core carrier fields
rebuild the left and right contextual carriers, and both contextual endpoints
are history-same because each is empty.
\end{proof}
\leanchecked{BEDC.Derived.EigenUp.EigenSingletonClassifier\_append\_context\_empty\_iff}

\begin{theorem}[Eigen singleton classifier continuation results hsame]
\label{thm:eigen-singleton-classifier-continuation-results-hsame}
For histories $h,k,q,q',r,r':\mathsf{BHist}$, if
$\mathsf{EigenSingletonClassifier}(h,k)$, $q\hsame q'$,
$\Cont(h,q,r)$, and $\Cont(k,q',r')$, then $r\hsame r'$.
\end{theorem}
\begin{proof}
The classifier exposes eigen singleton carrier witnesses for both sources.
The carrier continuation left-unit result reads $r$ back to $q$ and $r'$ back
to $q'$. Composing these two readbacks with the supplied tail equality
$q\hsame q'$ gives the endpoint equality $r\hsame r'$.
\end{proof}
\leanchecked{BEDC.Derived.EigenUp.EigenSingletonClassifier\_continuation\_results\_hsame}

\begin{theorem}[Eigen singleton semantic name certificate]
\label{thm:eigen-singleton-semantic-namecert}
The singleton eigen carrier, equipped with the carrier-restricted
$\hsame$ classifier above, satisfies the semantic naming-certificate schema.
\end{theorem}
\begin{proof}
The carrier is inhabited by the displayed empty eigen pair: empty linear-map
history, empty determinant history, empty vector history, and the empty
continuation endpoint. Reflexivity, symmetry, and transitivity are inherited
from $\hsame$, with carrier membership retained in the classifier fields.
Pattern and ledger soundness are the identity inclusions on the same carrier.
\end{proof}
\leanchecked{BEDC.Derived.EigenUp.eigen\_singleton\_semantic\_name\_certificate}
