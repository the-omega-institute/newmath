\chapter{A Concrete Naming Certificate for $\EigenUp$}
\label{ch:concrete-instances-eigen-namecert}

The eigen-pair interface $\EigenUp$ packages an eigenvalue / eigenvector pair $(\lambda, v)$ for a linear endomorphism. It depends on $\NameCert_{\LinearMapUp}$ (\autoref{ch:concrete-instances-linearmap-namecert}) and $\NameCert_{\DeterminantUp}$ (\autoref{ch:concrete-instances-determinant-namecert}), with $\lambda$ certified as a root of the characteristic polynomial.

\section{Carrier definition}
\label{sec:eigen-carrier}

\begin{definition}[Eigen component singleton carrier]
\label{def:eigen-component-singleton-carrier}
For histories $f,\lambda,v,pair:\mathsf{BHist}$, the component singleton
eigen carrier requires $f$ and $v$ to be singleton linear-map histories,
$\lambda$ to be a singleton commutative-ring history, and $pair$ to be the
BEDC continuation endpoint of $\lambda$ followed by $v$.
\end{definition}
\leandef{BEDC.Derived.EigenUp.EigenComponentSingletonCarrier}

\begin{theorem}[Eigen singleton carrier pair emptiness]
\label{thm:eigen-singleton-carrier-pair-empty}
For histories $f,\lambda,v,pair:\mathsf{BHist}$, the component singleton
eigen carrier is equivalent to the three singleton component carriers
together with $\hsame(pair,\emp)$.
\end{theorem}
\begin{proof}
Unfold the component singleton eigen carrier. In the forward direction, the
scalar and vector component carriers identify $\lambda$ and $v$ with $\emp$,
so $\mathsf{append}(\lambda,v)$ is empty by the BEDC append-empty inversion,
and the stored continuation transports this emptiness to $pair$. In the
reverse direction, the same append-empty fact reconstructs the continuation
endpoint from the given $\hsame(pair,\emp)$ witness.
\end{proof}
\leanchecked{BEDC.Derived.EigenUp.EigenSingletonCarrier\_pair\_empty\_iff}

\begin{definition}[Eigen singleton carrier]
\label{def:eigen-singleton-carrier}
A history $p$ lies in the singleton eigen carrier when there are histories
$m,\lambda,v$ such that $m$ and $v$ are singleton linear-map carriers,
$\lambda$ is a singleton determinant carrier, and $p$ is the continuation
result
$$
\Cont(m,\mathsf{append}(\lambda,v),p).
$$
\end{definition}
\leandef{BEDC.Derived.EigenUp.EigenSingletonCarrier}

\begin{theorem}[Eigen singleton carrier continuation result transport]
\label{thm:eigen-singleton-carrier-cont-result-transport}
For histories $m,\lambda,v,p,p':\mathsf{BHist}$, if $m$ and $v$ are singleton
linear-map carriers, $\lambda$ is a singleton determinant carrier,
$\Cont(m,\mathsf{append}(\lambda,v),p)$, and $p\hsame p'$, then $p'$ lies in
the eigen singleton carrier.
\end{theorem}
\begin{proof}
Reuse the same displayed witnesses $m,\lambda,v$. The carrier fields are
unchanged, and the continuation result transports along $p\hsame p'$ by the
kernel result-transport theorem for $\Cont$.
\end{proof}
\leanchecked{BEDC.Derived.EigenUp.EigenSingletonCarrier\_cont\_result\_transport}

\section{The certificate}
\label{sec:eigen-certificate}

\begin{definition}[Eigen singleton classifier]
\label{def:eigen-singleton-classifier}
Two histories $p,q:\mathsf{BHist}$ are classified by the singleton eigen
interface when both lie in the singleton eigen carrier and $p\hsame q$.
\end{definition}
\leandef{BEDC.Derived.EigenUp.EigenSingletonClassifier}

\begin{theorem}[Eigen singleton semantic name certificate]
\label{thm:eigen-singleton-semantic-namecert}
The singleton eigen carrier, equipped with the carrier-restricted
$\hsame$ classifier above, satisfies the semantic naming-certificate schema.
\end{theorem}
\begin{proof}
The carrier is inhabited by the displayed empty eigen pair: empty linear-map
history, empty determinant history, empty vector history, and the empty
continuation endpoint. Reflexivity, symmetry, and transitivity are inherited
from $\hsame$, with carrier membership retained in the classifier fields.
Pattern and ledger soundness are the identity inclusions on the same carrier.
\end{proof}
\leanchecked{BEDC.Derived.EigenUp.eigen\_singleton\_semantic\_name\_certificate}
