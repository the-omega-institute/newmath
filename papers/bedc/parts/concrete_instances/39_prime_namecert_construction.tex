\chapter{A Concrete Naming Certificate for \(\PrimeUp\)}
\label{ch:concrete-instances-prime-namecert}

This chapter gives a concrete naming-certificate construction for the prime number interface \(\PrimeUp\) at the paper-proof level. The construction sits above the unary natural-number certificate \(\NatUp\) (\autoref{ch:concrete-instances-nat-namecert}) and the additive certificate \(\AddUp\) (\autoref{ch:concrete-instances-add-namecert}), and is the first arithmetic chapter to exhibit a non-trivial property of unary histories beyond pure additive structure. The certificate is fully constructive: every theorem produces an explicit witness, primality is decidable by trial division on the underlying unary segment, and Euclid's infinitude of primes is supplied as an effective bound function rather than as a non-constructive existence claim. The standard limit results --- the fundamental theorem of arithmetic in both existence and uniqueness directions --- are reduced, in the same chapter, to closure laws on the prime classifier.

\section{Multiplicative structure on unary histories}

\begin{definition}[Unary multiplication relation]
\label{def:nat-multiplication}
For unary histories \(d, q, n\) with \(\mathsf{Unary}(d)\), \(\mathsf{Unary}(q)\), and \(\mathsf{Unary}(n)\), the multiplicative relation \(\mathsf{NatMul}(d, q, n)\) holds when \(n\) is the unary continuation of \(d\) repeated \(q\) times. Concretely, \(\mathsf{NatMul}\) is the closed inductive predicate generated by:
\begin{enumerate}
  \item \(\mathsf{NatMul}(d, \emp, \emp)\) for every unary \(d\) (zero multiplied by anything is zero);
  \item if \(\mathsf{NatMul}(d, q, n)\) and \(\Cont(n, d, n')\), then \(\mathsf{NatMul}(d, \Eone(q), n')\) (one more copy of \(d\) appended).
\end{enumerate}
\end{definition}
\leandef{BEDC.Derived.PrimeUp.NatMul}

\begin{lemma}[Unary multiplication successor inversion]
\label{lem:nat-mul-succ-inversion}
If $\mathsf{NatMul}(d,\Eone(q),n')$, then there is a predecessor product
$n$ such that $\mathsf{NatMul}(d,q,n)$ and $\Cont(n,d,n')$.
\end{lemma}
\begin{proof}
Invert the closed successor clause in the inductive definition of
$\mathsf{NatMul}$. The displayed predecessor product and continuation
witness are exactly the constructor payloads.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatMul\_succ\_inversion}

\begin{lemma}[Unary multiplication is functional]
\label{lem:nat-mul-functional}
For each pair of unary histories \((d, q)\), there is at most one unary \(n\) with \(\mathsf{NatMul}(d, q, n)\), up to \(\hsame\).
\end{lemma}
\begin{proof}
By induction on the second argument \(q\), using the determinacy of \(\Cont\) at each appending step.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatMul\_functional}

\begin{lemma}[Unary multiplication is total on unary inputs]
\label{lem:nat-mul-total}
For every pair of unary histories \((d, q)\) there exists a unary \(n\) with \(\mathsf{NatMul}(d, q, n)\).
\end{lemma}
\begin{proof}
By unary induction on \(q\). The base case yields \(n = \emp\); the step appends one further copy of \(d\) using \(\Cont\). Closure of unary histories under \(\Cont\) (\autoref{ch:concrete-instances-nat-namecert}) keeps \(n\) unary.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatMul\_total}

\begin{lemma}[Unary multiplication results are unary]
\label{lem:nat-mul-result-unary}
For every unary multiplicand $d$, every derivation
$\mathsf{NatMul}(d,q,n)$ has a unary result $n$.
\end{lemma}
\begin{proof}
Induct on the $\mathsf{NatMul}$ derivation. The zero constructor returns
$\emp$, and the successor constructor appends the unary multiplicand to the
unary inductive result, so unary closure under $\Cont$ applies.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatMul\_result\_unary}

\begin{lemma}[Unit left multiplication reads back the multiplier]
\label{lem:nat-mul-unit-left-hsame}
For every unary history $q$, if $\mathsf{NatMul}(\Eone(\emp), q, n)$, then $\hsame(n,q)$.
\end{lemma}
\begin{proof}
Induct on the concrete multiplication derivation for $q$. The zero case is immediate. In the successor case, unary inversion exposes the tail multiplier, the induction hypothesis identifies the previous product with that tail, and determinacy of $\Cont$ identifies the final appended unit with the canonical successor.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatMul\_unit\_left\_hsame}

\begin{lemma}[\(\Ezero\)-multiplier exclusion]
\label{thm:natmul-e0-multiplier-absurd}
No unary multiplication witness has an \(\Ezero\)-headed multiplier:
for all histories \(d,q,n\), \(\mathsf{NatMul}(d,\Ezero(q),n)\) is impossible.
\end{lemma}
\begin{proof}
By constructor inversion on the closed relation \(\mathsf{NatMul}\): its
multiplier argument is generated only by \(\emp\) and \(\Eone\)-successor cases.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatMul\_e0\_multiplier\_absurd}

\begin{definition}[Divisibility relation]
\label{def:nat-divides}
For unary histories \(d, n\), the divisibility relation \(\mathsf{NatDivides}(d, n)\) holds when there exists a unary \(q\) with \(\mathsf{NatMul}(d, q, n)\). The witness \(q\) is the explicit quotient.
\end{definition}
\leandef{BEDC.Derived.PrimeUp.NatDivides}

\begin{lemma}[Divisibility is reflexive on the unit and on every unary history]
\label{lem:divides-reflexive}
For every unary \(n\), \(\mathsf{NatDivides}(\Eone(\emp), n)\) and \(\mathsf{NatDivides}(n, n)\).
\end{lemma}
\begin{proof}
The unit case takes \(q = n\) in \(\mathsf{NatMul}(\Eone(\emp), n, n)\) by induction. The self-divisibility case takes \(q = \Eone(\emp)\) and uses the continuation unit at the empty quotient.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatDivides\_reflexive}
\leanchecked{BEDC.Derived.PrimeUp.NatDivides\_unit\_self\_reflexive}
\leanchecked{BEDC.Derived.PrimeUp.NatDivides\_reflexive\_pair}

\begin{lemma}[Divisibility is transitive]
\label{lem:divides-transitive}
If \(\mathsf{NatDivides}(d, e)\) and \(\mathsf{NatDivides}(e, n)\), then \(\mathsf{NatDivides}(d, n)\).
\end{lemma}
\begin{proof}
From the witnessing quotients \(q_1\) with \(\mathsf{NatMul}(d, q_1, e)\) and \(q_2\) with \(\mathsf{NatMul}(e, q_2, n)\), the composite quotient is \(q_1 \cdot q_2\), itself constructed as a unary history by \autoref{lem:nat-mul-total}.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatDivides\_transitive}

\section{The prime predicate}

\begin{definition}[Strict greater-than on unary histories]
\label{def:nat-strict-greater}
The relation \(\mathsf{NatGT}(n, m)\) holds when \(\mathsf{NatUnaryStrictPrefix}(m, n)\) holds (\autoref{ch:concrete-instances-nat-namecert}). Equivalently, \(n\) extends \(m\) by at least one further \(\Eone\) step.
\end{definition}

\begin{definition}[Prime predicate]
\label{def:nat-prime}
A unary history \(p\) is prime, written \(\mathsf{NatPrime}(p)\), when:
\begin{enumerate}
  \item \(\mathsf{Unary}(p)\),
  \item \(\mathsf{NatGT}(p, \Eone(\emp))\) (i.e., \(p\) strictly exceeds the unit), and
  \item for every unary \(d\) with \(\mathsf{NatDivides}(d, p)\), either \(\hsame(d, \Eone(\emp))\) or \(\hsame(d, p)\).
\end{enumerate}
\end{definition}
\leandef{BEDC.Derived.PrimeUp.NatPrime}

\begin{lemma}[The first two primes]
\label{lem:first-primes}
\(\mathsf{NatPrime}(\Eone(\Eone(\emp)))\) and \(\mathsf{NatPrime}(\Eone(\Eone(\Eone(\emp))))\). That is, the unary representations of \(2\) and \(3\) are prime.
\end{lemma}
\begin{proof}
For \(p = 2\), the only divisors are the unit and \(p\) itself by \autoref{def:nat-divides} and the case analysis on the quotient \(q\). The same case analysis applies to \(p = 3\), with the additional sub-case ruling out \(q = 2\) by the failure of \(\mathsf{NatMul}(2, 2, 3)\).
\end{proof}

\begin{lemma}[Non-primality of the unit]
\label{lem:unit-not-prime}
\(\neg \mathsf{NatPrime}(\Eone(\emp))\).
\end{lemma}
\begin{proof}
The unit fails the strict greater-than condition \(\mathsf{NatGT}(\Eone(\emp), \Eone(\emp))\) by the irreflexivity of \(\mathsf{NatUnaryStrictPrefix}\) (\autoref{ch:concrete-instances-nat-namecert}).
\end{proof}

\section{Effective primality testing}

\begin{definition}[Trial-division witness up to \(b\)]
\label{def:trial-division-witness}
For unary histories \(b, n\), the trial-division witness \(\mathsf{TrialDiv}(b, n)\) is the closed inductive predicate generated by:
\begin{enumerate}
  \item \(\mathsf{TrialDiv}(\Eone(\emp), n)\) for every unary \(n\) (the unit is examined first);
  \item if \(\mathsf{TrialDiv}(b, n)\), and either \(\neg \mathsf{NatDivides}(b, n)\) or \(\hsame(b, \Eone(\emp))\) or \(\hsame(b, n)\), and \(\Cont(b, \Eone(\emp), b')\), then \(\mathsf{TrialDiv}(b', n)\) (advance the test to \(b'\)).
\end{enumerate}
\end{definition}
\leandef{BEDC.Derived.PrimeUp.TrialDiv}

\begin{theorem}[Trial-division step preserves unary closure]
\label{thm:trialdiv-step-unary-closed}
If \(\mathsf{TrialDiv}(b,n)\) has been reached and the admissible step
condition for \(b\) holds, then every continuation
\(\Cont(b,\Eone(\emp),b')\) produces a unary \(b'\) and the next witness
\(\mathsf{TrialDiv}(b',n)\).
\end{theorem}
\begin{proof}
Induct on the trial-division witness. The base witness starts at
\(\Eone(\emp)\), which is unary; the step case uses unary closure of
\(\Cont\) with the unit continuation and rebuilds the trial-division step.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.TrialDiv\_step\_unary\_closed}

\begin{theorem}[Trial-division bounds are unary]
\label{thm:trial-div-bound-unary}
If \(\mathsf{TrialDiv}(b,n)\), then \(b\) is a unary history.
\end{theorem}
\begin{proof}
Induct on the trial-division witness. The unit constructor gives \(\mathsf{Unary}(\Eone(\emp))\). The step constructor extends the previous unary bound by the unary unit through \(\Cont\), so unary closure under continuation gives the next bound.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.TrialDiv\_bound\_unary}

\begin{theorem}[Decidability of primality]
\label{thm:prime-decidable}
For every unary history \(n\), the proposition \(\mathsf{NatPrime}(n)\) is decidable: there is an explicit unary algorithm that returns either a proof of \(\mathsf{NatPrime}(n)\) or an explicit non-trivial divisor of \(n\).
\end{theorem}
\begin{proof}[Sketch]
The algorithm constructs \(\mathsf{TrialDiv}(b, n)\) by unary induction on \(b\) up to \(n\) itself. At each step, it checks divisibility of \(n\) by \(b\) using \autoref{lem:nat-mul-total} and \autoref{lem:nat-mul-functional}: this is decidable because the search for a quotient \(q\) of \(\mathsf{NatMul}(b, q, n)\) is bounded above by \(n\). If a non-trivial divisor is found (i.e.\ \(b\) divides \(n\) and is neither the unit nor \(n\)), the algorithm returns it and refutes primality. Otherwise, \(\mathsf{TrialDiv}\) reaches \(b = n\) and primality holds by \autoref{def:nat-prime}.
\end{proof}

\section{Euclid's theorem on the infinitude of primes}

\begin{definition}[Unary factorial]
\label{def:unary-factorial}
The unary factorial \(\mathsf{NatFact}(n, n!)\) is the closed inductive predicate generated by:
\begin{enumerate}
  \item \(\mathsf{NatFact}(\emp, \Eone(\emp))\) (zero factorial is one);
  \item if \(\mathsf{NatFact}(n, m)\) and \(\mathsf{NatMul}(\Eone(n), m, m')\), then \(\mathsf{NatFact}(\Eone(n), m')\).
\end{enumerate}
\end{definition}
\leandef{BEDC.Derived.PrimeUp.NatFact}

\begin{lemma}[Factorial is total and functional on unary histories]
\label{lem:fact-total}
For every unary \(n\), there is exactly one unary \(m\) with \(\mathsf{NatFact}(n, m)\).
\end{lemma}
\begin{proof}
By unary induction on \(n\), using the totality and functionality of \(\mathsf{NatMul}\) (\autoref{lem:nat-mul-total}, \autoref{lem:nat-mul-functional}) at each step.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatFact\_total\_functional}

\begin{lemma}[Successor of factorial is coprime to all bounded predecessors]
\label{lem:fact-plus-one-coprime}
Let \(m\) be the unary factorial of \(n\), and let \(m'\) be the successor \(\Eone(m)\) (i.e., \(n! + 1\)). Then for every unary \(d\) with \(\Eone(\emp) \neq d\) and \(\mathsf{NatGT}(\Eone(n), d) \vee \hsame(d, \Eone(n)) \vee \mathsf{NatGT}(d, \Eone(n)) \wedge \mathsf{NatDivides}(d, n)\), we have \(\neg \mathsf{NatDivides}(d, m')\).
\end{lemma}
\begin{proof}[Sketch]
If \(d\) divides both \(n!\) and \(n! + 1\), then \(d\) divides their difference, which is the unit; hence \(d\) is the unit, contradicting \(d \neq \Eone(\emp)\).
\end{proof}

\begin{theorem}[Euclid's theorem, constructive form]
\label{thm:euclid-infinite}
There exists an explicit total unary function \(\mathsf{NatNextPrime} : \mathsf{Unary} \to \mathsf{Unary}\) such that for every unary \(n\), \(\mathsf{NatPrime}(\mathsf{NatNextPrime}(n))\) and \(\mathsf{NatGT}(\mathsf{NatNextPrime}(n), n)\).
\end{theorem}
\begin{proof}[Sketch]
Given \(n\), compute the unary factorial \(m = n!\) by \autoref{lem:fact-total}, then form \(m' = m + 1\). By \autoref{lem:fact-plus-one-coprime}, every prime divisor of \(m'\) exceeds \(n\). The primality test of \autoref{thm:prime-decidable} either finds \(m'\) itself prime (return it) or returns an explicit non-trivial divisor; in the latter case, recursing on the divisor's prime factor (also exceeding \(n\)) terminates in a prime by well-foundedness of \(\mathsf{NatUnaryStrictPrefix}\). The function is total because the search is bounded.
\end{proof}

\section{The fundamental theorem of arithmetic}

\begin{definition}[Prime factorization list]
\label{def:prime-factorization-list}
A prime factorization of a unary history \(n\) is a finite list of unary primes \([p_1, \dots, p_k]\) (in non-decreasing unary order) such that the iterated multiplicative continuation \(p_1 \cdot p_2 \cdots p_k\) yields a unary history \(\hsame\)-equal to \(n\).
\end{definition}

\begin{theorem}[Existence of prime factorization]
\label{thm:ftoa-existence}
For every unary history \(n\) with \(\mathsf{NatGT}(n, \emp)\), there exists a prime factorization list of \(n\). The list is constructed explicitly by repeated trial division using \autoref{thm:prime-decidable}.
\end{theorem}
\begin{proof}[Sketch]
Strong induction on the unary history \(n\) along \(\mathsf{NatUnaryStrictPrefix}\). If \(n = \Eone(\emp)\), the empty list factors it. Otherwise apply \autoref{thm:prime-decidable}: either \(n\) is prime (return \([n]\)) or a non-trivial divisor \(d\) is exhibited; recurse on \(d\) and on the quotient \(q = n / d\), both strictly smaller than \(n\), and concatenate the resulting lists.
\end{proof}

\begin{theorem}[Uniqueness of prime factorization]
\label{thm:ftoa-uniqueness}
If \([p_1, \dots, p_j]\) and \([q_1, \dots, q_k]\) are two prime factorization lists of the same unary history \(n\), then \(j = k\) and the lists agree pointwise up to \(\hsame\).
\end{theorem}
\begin{proof}[Sketch]
Standard Euclid's-lemma argument: if a prime \(p\) divides a product \(a \cdot b\), then \(p\) divides \(a\) or \(p\) divides \(b\), proved by induction on the unary representation of \(a\). Iterating Euclid's lemma over the two lists shows pointwise agreement up to \(\hsame\) and length agreement \(j = k\).
\end{proof}

\section{The certificate}

\begin{definition}[Prime source specification]
\label{def:prime-source-specification}
The source specification on \(\mathsf{Unary}\) carries the additional constraint \(\mathsf{NatPrime}(p)\). Its generator is the prime predicate of \autoref{def:nat-prime}.
\end{definition}

\begin{definition}[Prime pattern specification]
\label{def:prime-pattern-specification}
The pattern of \(\PrimeUp\) is trial-division-bounded primality: for an input unary \(n\), the pattern is the trial-division predicate \(\mathsf{TrialDiv}(n, n)\) restricted to inputs satisfying \(\mathsf{NatPrime}\) at the upper bound.
\end{definition}

\begin{definition}[Prime classifier specification]
\label{def:prime-classifier-specification}
The classifier on prime carriers is \(\hsame\) restricted to unary primes. Two unary primes classify together iff they are \(\hsame\)-equal as histories.
\end{definition}

\begin{definition}[Prime stability certificate]
\label{def:prime-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item closure of the prime predicate under \(\hsame\): if \(\mathsf{NatPrime}(p)\) and \(\hsame(p, p')\), then \(\mathsf{NatPrime}(p')\);
  \item transport of divisibility under \(\hsame\): if \(\mathsf{NatDivides}(d, n)\), \(\hsame(d, d')\), and \(\hsame(n, n')\), then \(\mathsf{NatDivides}(d', n')\);
  \item compatibility of the trial-division witness with \(\hsame\) on bounds.
\end{enumerate}
\end{definition}

\begin{definition}[Prime ledger policy]
\label{def:prime-ledger-policy}
The ledger policy records:
\begin{enumerate}
  \item composition of factorizations: concatenation of prime factorization lists for products;
  \item Euclid's lemma: prime divisibility distributes over products;
  \item infinitude closure: \(\mathsf{NatNextPrime}\) of \autoref{thm:euclid-infinite} produces a new prime exceeding any given bound.
\end{enumerate}
\end{definition}

\begin{theorem}[Concrete \(\PrimeUp\) naming certificate]
\label{thm:prime-namecert}
The data \((\mathsf{NatPrime}, \mathsf{TrialDiv}, \hsame, \mathsf{PrimeStabilityCert}, \mathsf{PrimeLedgerPolicy})\) of \autoref{def:prime-source-specification}--\autoref{def:prime-ledger-policy} forms a naming certificate over \(\Hist\) for the prime number interface \(\PrimeUp\). The certificate is decidable (\autoref{thm:prime-decidable}), unbounded (\autoref{thm:euclid-infinite}), and exhaustive (\autoref{thm:ftoa-existence}, \autoref{thm:ftoa-uniqueness}).
\end{theorem}
\begin{proof}[Sketch]
The five fields satisfy the closure laws of the naming-certificate framework (\autoref{ch:core-typed-naming-certificates}). Decidability is \autoref{thm:prime-decidable}. Unboundedness is \autoref{thm:euclid-infinite}. Exhaustiveness follows from existence (\autoref{thm:ftoa-existence}) together with uniqueness (\autoref{thm:ftoa-uniqueness}). The certificate uses no \(\mathsf{Classical.choice}\), no \(\mathsf{Quot.sound}\), and no \(\mathsf{propext}\), so its Lean targets fall under the project's axiom-purity gate.
\end{proof}

\begin{theorem}[Semantic \(\PrimeUp\) naming certificate]
\label{thm:prime-semantic-namecert}
The concrete certificate of \autoref{thm:prime-namecert} satisfies the semantic naming-certificate schema of \(\NameCert\) (\autoref{ch:core-typed-naming-certificates}): the carrier is a \(\Hist\)-predicate, the classifier is a restriction of \(\hsame\), and every closure law is given by an explicit constructive proof.
\end{theorem}
\begin{proof}[Sketch]
Direct verification of the four \(\NameCert\) field types using the constructions of this chapter.
\end{proof}
