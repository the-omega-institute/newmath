\begin{theorem}[ProbSpace complement mass additivity]
\label{thm:probspace-complement-mass-additivity}
For a $\ProbSpaceUp$ carrier as in \autoref{sec:probspace-carrier}, write $\Omega$ for its total event, $\mathcal{A}$ for the designated $\sigma$-algebra inherited from the underlying $\MeasureUp$ carrier, $\mu$ for its $\RealUp$-valued measure map, $1_R$ for the probability unit selected by the $\ProbSpaceUp$ total-mass row, and $x\sim_R y$ for the $\RealUp$ classifier of \autoref{def:real-classifier-specification}. Let $A,A^\complement\in\mathcal{A}$ be disjoint measurable events whose measurable binary union is the total event $\Omega$. Suppose $\varnothing\in\mathcal{A}$, the underlying $\MeasureUp$ certificate supplies $\sigma$-additivity for every pairwise-disjoint measurable sequence, classifier-stable head-tail readback and pointwise classifier stability for the resulting countable $\RealUp$ sums, and the target $\RealUp$ addition is carried by its additive $\AbGroupUp$ row with zero $0_R$. If the $\ProbSpaceUp$ total-mass row gives
$$
\mu(\Omega)\sim_R1_R,
$$
then
$$
\mu(A)+\mu(A^\complement)\sim_R1_R.
$$
\end{theorem}
\leanchecked{BEDC.Derived.ProbSpaceUp.ProbSpaceComplementMass\_additive\_readback}
\begin{proof}
Let $U:=A\cup A^\complement$ be the measurable binary union supplied by the $\sigma$-algebra row. By hypothesis, $U=\Omega$. Applying \autoref{thm:measure-finite-disjoint-union-additivity} to the disjoint pair $A,A^\complement$ gives
$$
\mu(U)\sim_R\mu(A)+\mu(A^\complement).
$$
Reading $U$ as $\Omega$ gives
$$
\mu(\Omega)\sim_R\mu(A)+\mu(A^\complement).
$$
The probability-space total-mass row gives
$$
\mu(\Omega)\sim_R1_R.
$$
The classifier $\sim_R$ is the $\RealUp$ classifier, so its symmetry and transitivity fields compose these two comparisons. Symmetry of the finite-additivity comparison gives
$$
\mu(A)+\mu(A^\complement)\sim_R\mu(\Omega),
$$
and transitivity with the total-mass comparison yields
$$
\mu(A)+\mu(A^\complement)\sim_R1_R.
$$
\end{proof}

\begin{theorem}[ProbSpace complement mass as one minus event mass]
\label{thm:probspace-complement-mass-one-minus}
For a $\ProbSpaceUp$ carrier as in \autoref{sec:probspace-carrier}, write $\Omega$ for its total event, $\mathcal{A}$ for the designated $\sigma$-algebra inherited from the underlying $\MeasureUp$ carrier, $\mu$ for its $\RealUp$-valued measure map, $1_R$ for the probability unit selected by the $\ProbSpaceUp$ total-mass row, and $x\sim_R y$ for the $\RealUp$ classifier of \autoref{def:real-classifier-specification}. Let $A,A^\complement\in\mathcal{A}$ be disjoint measurable events whose measurable binary union is the total event $\Omega$. Suppose $\varnothing\in\mathcal{A}$, the underlying $\MeasureUp$ certificate supplies $\sigma$-additivity for every pairwise-disjoint measurable sequence, classifier-stable head-tail readback and pointwise classifier stability for the resulting countable $\RealUp$ sums, and the target $\RealUp$ addition is carried by its additive $\AbGroupUp$ row with zero $0_R$ and inverse operation $-$. If the $\ProbSpaceUp$ total-mass row gives
$$
\mu(\Omega)\sim_R1_R,
$$
then
$$
\mu(A^\complement)\sim_R1_R+(-\mu(A)).
$$
\end{theorem}
\leanchecked{BEDC.Derived.ProbSpaceUp.ProbSpaceComplementMass\_right\_solution}
\leanvariant{BEDC.Derived.ProbSpaceUp.ProbSpaceComplementMass\_one\_minus\_singleton}
\begin{proof}
By \autoref{thm:probspace-complement-mass-additivity} applied to the displayed pair and these hypotheses,
$$
\mu(A)+\mu(A^\complement)\sim_R1_R.
$$
The additive $\AbGroupUp$ row of \autoref{def:abgroup-carrier} contains the same carrier classifier $\sim_R$, the same addition $+$, the zero $0_R$, the inverse operation $-$, addition congruence, and commutativity. By \autoref{prop:abgroup-forgets-group-certificate}, omitting the commutativity field leaves a $\GroupUp$ row for the same data.

Use the commutativity field to exchange the two summands:
$$
\mu(A^\complement)+\mu(A)\sim_R\mu(A)+\mu(A^\complement).
$$
Transitivity with the complement-additivity comparison gives the right-multiplication equation
$$
\mu(A^\complement)+\mu(A)\sim_R1_R.
$$
Now read \autoref{thm:group-right-mul-equation-solution} in the retained additive group, with multiplication written as $+$, inverse written as $-$, $x=\mu(A^\complement)$, $a=\mu(A)$, and $b=1_R$. It solves the displayed equation as
$$
\mu(A^\complement)\sim_R1_R+(-\mu(A)).
$$
\end{proof}

\begin{theorem}[ProbSpace monotone event bounds]
\label{thm:probspace-monotone-event-bounds}
For a $\ProbSpaceUp$ carrier obtained from the probability substrate of \autoref{thm:measure-probability-substrate-obligations}, write $\Omega$ for its total event, $\mathcal{A}$ for the displayed measurable-event family inherited from the underlying $\MeasureUp$ carrier, $\mu$ for its $\RealUp$-valued measure map, $1_R$ for the endpoint selected by the total-mass row, and $x\sim_R y$ for the $\RealUp$ classifier of \autoref{def:real-classifier-specification}. Let $A,B\in\mathcal{A}$ be displayed events in the same carried fiber, and suppose the public event readback displays
$$
A\subseteq B\subseteq\Omega.
$$
Assume the inherited complement/relative-difference rows give the two measurable decompositions required by \autoref{thm:measure-monotone-under-measurable-inclusion}, and that the target RealAlgOrder order row is stable under $\sim_R$ and transitive. If the total-mass row gives
$$
\mu(\Omega)\sim_R1_R,
$$
then
$$
\mu(A)\le_R\mu(B),\qquad \mu(B)\le_R1_R,\qquad \mu(A)\le_R1_R.
$$
\end{theorem}
\leanchecked{BEDC.Derived.ProbSpaceUp.ProbSpaceMonotoneEvent\_bounds}
\begin{proof}
By \autoref{thm:measure-probability-substrate-obligations}, the $\ProbSpaceUp$ consumer reads the same displayed measurable events, relative-difference rows, finite-additivity rows, nonnegative value rows, and RealAlgOrder additive order rows as the underlying $\MeasureUp$ package, with the total-event row and unit endpoint added by the probability surface. The total-mass comparison is the normalization row isolated in \autoref{thm:measure-probability-normalization-row}.

Apply the inherited complement/relative-difference row of \autoref{def:measure-complement-difference-row} to the displayed inclusion $A\subseteq B$. It supplies a measurable event $B\setminus A$, disjointness from $A$, and a binary-union readback classifying $A\cup(B\setminus A)$ as $B$. These are exactly the decomposition inputs for \autoref{thm:measure-monotone-under-measurable-inclusion}, so the inherited MeasureUp monotonicity row gives
$$
\mu(A)\le_R\mu(B).
$$

Apply the same row to the displayed inclusion $B\subseteq\Omega$. It supplies $\Omega\setminus B$ and a disjoint decomposition of $\Omega$ with first summand $B$. A second application of \autoref{thm:measure-monotone-under-measurable-inclusion} gives
$$
\mu(B)\le_R\mu(\Omega).
$$

The total-mass row gives $\mu(\Omega)\sim_R1_R$. Stability of the RealAlgOrder order row under the $\RealUp$ classifier transports the right endpoint of the last comparison across this classifier equality, yielding
$$
\mu(B)\le_R1_R.
$$
Finally, transitivity of the same order row composes $\mu(A)\le_R\mu(B)$ with $\mu(B)\le_R1_R$, and gives
$$
\mu(A)\le_R1_R.
$$
\end{proof}

\begin{definition}[ProbSpace public event packet]
\label{def:probspace-public-event-packet}
For a $\ProbSpaceUp$ carrier obtained from the probability substrate of \autoref{thm:measure-probability-substrate-obligations}, the public event packet is the tuple whose carrier entries are the $\MeasureUp$ measurable-event family $\mathcal{A}$, the total event $\Omega$, the $\RealUp$-valued measure map $\mu$, and the selected endpoint $1_R$. Its classifier entries are the displayed measurable-event readback classifier inherited from $\MeasureUp$, the $\RealUp$ classifier $\sim_R$, the RealAlgOrder comparison row $\le_R$, and the classifier-stability fields for addition, order, complement, and relative difference. Its ledger entries are the underlying $\mathsf{BHist}$ histories, the carried package row $\mathsf{Pkg}$, the $\NameCert_{\MeasureUp}$ and $\NameCert_{\RealUp}$ rows read by the probability substrate, and the $\NameCert_{\ProbSpaceUp}$ row that records the total-mass comparison
$$
\mu(\Omega)\sim_R1_R.
$$
The packet is public when these entries are exposed as named BEDC rows, with no appeal to a host probability-space axiom outside the displayed $\mathsf{BHist}$, $\mathsf{Pkg}$, and $\NameCert$ ledger.
\end{definition}
\leandef{BEDC.Derived.ProbSpaceUp.ProbSpacePublicEventPacket}

\begin{theorem}[ProbSpace total event normalization row]
\label{thm:probspace-total-event-normalization-row}
For the public event packet of \autoref{def:probspace-public-event-packet}, the displayed total-mass comparison
$$
\mu(\Omega)\sim_R1_R
$$
is exported as a $\ProbSpaceUp$ BEDC row. It is a row of the carried certificate surface, not an assertion that the ambient host theory supplies probability normalization for arbitrary outside measures.
\end{theorem}
\begin{proof}
Project the total-mass entry from the $\NameCert_{\ProbSpaceUp}$ ledger in \autoref{def:probspace-public-event-packet}. The projected row has left endpoint $\mu(\Omega)$, right endpoint $1_R$, and classifier $\sim_R$.

The classifier is the same $\RealUp$ classifier used in \autoref{def:real-classifier-specification}. Hence the row is read as the BEDC comparison $\mu(\Omega)\sim_R1_R$ inside the carried package, and no host normalization principle is introduced.
\end{proof}

\begin{theorem}[ProbSpace event classifier transport row]
\label{thm:probspace-event-classifier-transport-row}
For the public event packet of \autoref{def:probspace-public-event-packet}, measurable-event readback transport together with $\RealUp$ classifier transport preserves the measure endpoint, the complement packet, and the relative-difference rows used by \autoref{thm:probspace-monotone-event-bounds}. In particular, if displayed events $A$ and $B$ are transported along the inherited event readback classifier and their measured endpoints are transported along $\sim_R$, then the rows presenting $A^\complement$, $B\setminus A$, $\mu(A)$, $\mu(B)$, $\mu(A^\complement)$, and $\mu(B\setminus A)$ remain rows of the same public packet.
\end{theorem}
\begin{proof}
Start with the MeasureUp complement and relative-difference row of \autoref{def:measure-complement-difference-row}. It is one of the inherited entries of the public event packet, so event readback transport keeps the displayed complement and relative-difference events in the same measurable-event family $\mathcal{A}$.

The measured endpoints are carried by the $\RealUp$-valued measure map $\mu$. Apply the same $\RealUp$ classifier transport used in \autoref{thm:probspace-monotone-event-bounds}; it transports endpoint comparisons without changing the named packet entries.

Combining these two transports preserves the complement packet, the relative-difference packet, and the RealAlgOrder rows used for monotonicity. Thus the transported rows remain available on the public $\ProbSpaceUp$ surface.
\end{proof}
\leanchecked{BEDC.Derived.ProbSpaceUp.ProbSpacePublicEventPacket\_transport\_rows}

\begin{theorem}[ProbSpace public certificate export]
\label{thm:probspace-public-certificate-export}
The public event packet of \autoref{def:probspace-public-event-packet} exports the probability-space certificate surface $\NameCert_{\ProbSpaceUp}$. The exported surface contains the complement-mass additivity row of \autoref{thm:probspace-complement-mass-additivity}, the one-minus complement row of \autoref{thm:probspace-complement-mass-one-minus}, the monotone event-bounds row of \autoref{thm:probspace-monotone-event-bounds}, the total event normalization row of \autoref{thm:probspace-total-event-normalization-row}, and the event classifier transport row of \autoref{thm:probspace-event-classifier-transport-row}. Consequently, downstream $\RandomVarUp$, $\DistributionUp$, $\IndependenceUp$, and $\CondExpUp$ objects may cite this single $\NameCert_{\ProbSpaceUp}$ surface for probability-space rows rather than reconstructing those rows from separate consequences.
\end{theorem}
\leanchecked{BEDC.Derived.ProbSpaceUp.ProbSpacePublicEventPacket\_public\_certificate\_export}
\begin{proof}
The packet already carries the $\MeasureUp$ measurable-event family, complement and relative-difference rows, the $\RealUp$ classifier and order rows, the total event, the measure map, and the unit endpoint. By \autoref{thm:probspace-total-event-normalization-row}, its total-mass comparison is a named certificate row.

Add the three scoped consequences already proved in this section. The complement-mass additivity row comes from \autoref{thm:probspace-complement-mass-additivity}; the one-minus row comes from \autoref{thm:probspace-complement-mass-one-minus}; and the monotone event-bounds row comes from \autoref{thm:probspace-monotone-event-bounds}.

It remains to make these rows stable under the event and endpoint classifiers read by downstream objects. This is exactly \autoref{thm:probspace-event-classifier-transport-row}, so the rows assemble into one public $\NameCert_{\ProbSpaceUp}$ surface.
\end{proof}

\begin{definition}[Binary cover-difference ledger]
\label{def:probspace-binary-cover-difference-ledger}
Let $P$ be a $\mathrm{ProbSpace}^{\uparrow}$ carrier obtained from the probability substrate of a $\mathrm{Measure}^{\uparrow}$ carrier. Write $\mathcal A$ for the displayed measurable-event family inherited from the underlying $\mathrm{Measure}^{\uparrow}$ carrier, write $\mu$ for the $\RealUp$-valued measure map, and write $E \sim_{\mathcal A} F$ for the public measurable-event classifier. For displayed events $A,B \in \mathcal A$, a binary cover-difference ledger for $(A,B)$ consists of displayed events
$$
U,I,D \in \mathcal A
$$
and public rows saying that $U$ is the displayed binary union of $A$ and $B$, $I$ is the displayed binary intersection of $A$ and $B$, and $D$ is the displayed outside-$A$ part of $B$, together with the two disjoint cover readbacks
$$
A \perp D,\qquad A \cup D \sim_{\mathcal A} U,
$$
and
$$
I \perp D,\qquad I \cup D \sim_{\mathcal A} B.
$$
The ledger is a $\BEDC$ event-surface object: it records measurable closure, disjointness, and classifier readback rows, and it does not add host set subtraction or quotient equality of events.
\end{definition}

\begin{theorem}[ProbSpace binary inclusion-exclusion identity]
\label{thm:probspace-binary-inclusion-exclusion-identity}
Let $P$ be a $\mathrm{ProbSpace}^{\uparrow}$ carrier obtained from the probability substrate of a $\mathrm{Measure}^{\uparrow}$ carrier. Write $\mathcal A$ for its displayed measurable-event family, write $\mu$ for its $\RealUp$-valued measure map, and write $x \sim_R y$ for the $\RealUp$ classifier. Let $A,B \in \mathcal A$ be displayed events in one carried fiber. Suppose that $(A,B)$ has a binary cover-difference ledger with displayed events $U,I,D$ in the sense of \autoref{def:probspace-binary-cover-difference-ledger}. Suppose also that the finite disjoint-union additivity row applies to the disjoint pairs $(A,D)$ and $(I,D)$, and that the target $\RealUp$ addition is carried by its additive $\AbGroupUp$ row. Then
$$
\mu(U)+\mu(I)\sim_R\mu(A)+\mu(B).
$$
Consequently, when the public notation $A\cup B$ names the displayed event $U$ and the public notation $A\cap B$ names the displayed event $I$, one obtains
$$
\mu(A\cup B)+\mu(A\cap B)\sim_R\mu(A)+\mu(B).
$$
\end{theorem}
\begin{proof}
The first disjoint cover row in the ledger gives $A \perp D$ and
$$
A\cup D \sim_{\mathcal A} U.
$$
Apply the finite disjoint-union additivity row to the displayed disjoint pair $(A,D)$. This gives
$$
\mu(A\cup D)\sim_R\mu(A)+\mu(D).
$$
The event-value transport row carries the event classifier comparison $A\cup D \sim_{\mathcal A} U$ to the $\RealUp$ endpoint classifier. Composing that transport with the displayed finite-additivity comparison gives
$$
\mu(U)\sim_R\mu(A)+\mu(D).
$$
The second disjoint cover row in the ledger gives $I \perp D$ and
$$
I\cup D \sim_{\mathcal A} B.
$$
Apply the same finite disjoint-union additivity row to the displayed disjoint pair $(I,D)$. This gives
$$
\mu(I\cup D)\sim_R\mu(I)+\mu(D).
$$
Transporting the left endpoint through $I\cup D \sim_{\mathcal A} B$ and composing in the $\RealUp$ classifier gives
$$
\mu(B)\sim_R\mu(I)+\mu(D).
$$
Now compute in the additive $\AbGroupUp$ row carried by the target $\RealUp$ endpoint surface. Addition congruence applied to the comparison
$$
\mu(U)\sim_R\mu(A)+\mu(D)
$$
gives
$$
\mu(U)+\mu(I)\sim_R(\mu(A)+\mu(D))+\mu(I).
$$
Associativity gives
$$
(\mu(A)+\mu(D))+\mu(I)\sim_R\mu(A)+(\mu(D)+\mu(I)).
$$
Commutativity gives
$$
\mu(D)+\mu(I)\sim_R\mu(I)+\mu(D).
$$
The earlier comparison
$$
\mu(B)\sim_R\mu(I)+\mu(D)
$$
therefore gives, by symmetry, transitivity, and addition congruence with the fixed left summand $\mu(A)$,
$$
\mu(A)+(\mu(D)+\mu(I))\sim_R\mu(A)+\mu(B).
$$
Transitivity of the $\RealUp$ classifier composes the displayed chain:
$$
\mu(U)+\mu(I)\sim_R\mu(A)+\mu(B).
$$
Substituting the public displayed representatives $U=A\cup B$ and $I=A\cap B$ gives
$$
\mu(A\cup B)+\mu(A\cap B)\sim_R\mu(A)+\mu(B).
$$
\end{proof}
