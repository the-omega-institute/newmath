\begin{definition}[Metric ContinuousMapUp source specification]
\label{def:continuousmapup-metric-source-specification}
For histories $source,map,target,modulus,cert,distance:\mathsf{BHist}$, the
metric $\ContinuousMapUp$ source row is the conjunction
$$
\begin{array}{l}
\mathsf{CategoryHomCarrier}(source,target,map)\land
\mathsf{ContinuousModulusWitness}(target,modulus,cert)\land
\mathsf{MetricDistanceWitness}(source,target,distance).
\end{array}
$$
It names the categorical graph carrier, the target-side modulus witness, and
the endpoint distance witness carried by the six-history certificate object.
\end{definition}

\begin{definition}[Metric ContinuousMapUp pattern specification]
\label{def:continuousmapup-metric-pattern-specification}
The metric $\ContinuousMapUp$ pattern is the six-field carrier formation
$$
\mathsf{ContinuousMapCarrier}(source,map,target,modulus,cert,distance).
$$
The displayed carrier is the public pattern consumed by compact-continuous and
Heine--Cantor arguments: it packages the source, map, target, modulus,
certificate, and distance histories in one BEDC object.
\end{definition}

\leanchecked{BEDC.Derived.ContinuousMapUp.ContinuousMapMetricPatternSpec\_carrier\_readback}

\begin{definition}[Metric ContinuousMapUp classifier specification]
\label{def:continuousmapup-metric-classifier-specification}
For two metric $\ContinuousMapUp$ carriers with histories
$source,map,target,modulus,cert,distance$ and
$source',map',target',modulus',cert',distance'$, the classifier is
componentwise $\hsame$ on the six displayed histories:
$$
\begin{array}{l}
\hsame(source,source')\land
\hsame(map,map')\land
\hsame(target,target')\land
\hsame(modulus,modulus')\land
\hsame(cert,cert')\land
\hsame(distance,distance').
\end{array}
$$
The same classifier row records the hidden canonical-distance exactness
$$
\hsame(distance,\mathsf{append}(source,target))
$$
for each admitted carrier, and similarly for the primed carrier.
\end{definition}

\begin{definition}[Metric ContinuousMapUp stability certificate]
\label{def:continuousmapup-metric-stability-certificate}
The metric $\ContinuousMapUp$ stability certificate consists of:
\begin{enumerate}
  \item field transport of the six histories under the componentwise
  $\hsame$ classifier;
  \item canonical-distance exactness for each admitted carrier;
  \item two-sided identity closure by empty maps and empty distance ledgers;
  \item composition closure by categorical graph composition, modulus
  composition, and canonical endpoint-distance normalization;
  \item associativity of the composed canonical package.
\end{enumerate}
\end{definition}

\begin{definition}[Metric ContinuousMapUp ledger policy]
\label{def:continuousmapup-metric-ledger-policy}
The metric $\ContinuousMapUp$ ledger policy records the six histories
$source,map,target,modulus,cert,distance$, the visible source row of
\autoref{def:continuousmapup-metric-source-specification}, and the hidden
canonical-distance ledger
$\hsame(distance,\mathsf{append}(source,target))$. Under composition it keeps
the graph and modulus continuation ledgers visible and rewrites the displayed
distance to the canonical endpoint append.
\end{definition}

\begin{theorem}[Metric ContinuousMapUp certificate bridge]
\label{thm:continuousmapup-metric-certificate-bridge}
The five rows of
\autoref{def:continuousmapup-metric-source-specification}--\autoref{def:continuousmapup-metric-ledger-policy}
form the paper-side $\NameCert$ obligations for $\ContinuousMapUp$ with carrier
$$
\mathsf{ContinuousMapCarrier}(source,map,target,modulus,cert,distance).
$$
The bridge identifies the source row with the
$\mathsf{CategoryUp}$, $\mathsf{ContinuousUp}$, and $\mathsf{MetricUp}$
components; exposes componentwise $\hsame$ on the six displayed histories; and
uses the hidden canonical-distance exactness
$\hsame(distance,\mathsf{append}(source,target))$ as the distance certificate
consumed by identity and composition.
\end{theorem}
\begin{proof}
First unpack the carrier through
\autoref{thm:continuousmap-carrier-categorical-canonical-distance-exactness}.
This gives the categorical hom row, the target-side modulus row, and the
canonical endpoint-distance classifier. Together with
\autoref{def:continuousmapup-metric-source-specification}, these are exactly
the visible source obligations.

Next use
\autoref{thm:continuousmap-carrier-hsame-field-transport-depth} to transport
the six displayed histories along the componentwise classifier. The transport
step supplies the classifier and stability fields without changing the public
carrier shape.

Identity closure is supplied by
\autoref{thm:continuousmap-empty-identities-two-sided-composition-units}. It
shows that empty identity maps and empty distance ledgers serve as the
two-sided units required by the certificate policy.

Finally apply
\autoref{thm:continuousmap-carrier-composition-associative-canonical-package}.
It composes the graph, modulus, certificate, and distance rows while
normalizing the distance back to the canonical endpoint append. This is the
ledger behavior stated in
\autoref{def:continuousmapup-metric-ledger-policy}, so the five displayed rows
give the advertised $\NameCert$ obligations for $\ContinuousMapUp$.
\end{proof}

\begin{theorem}[ContinuousMap composite graph depth addition]
\label{thm:continuousmap-comp-graph-depth-add}
For histories $source,mid,target,mapF,mapG,mapFG,modF,modG,modFG,certF,certG,certFG$,
assume the two continuous-function carriers
$$
\begin{array}{l}
\mathsf{ContinuousFunctionCarrier}(source,mapF,mid,modF,certF),\\
\mathsf{ContinuousFunctionCarrier}(mid,mapG,target,modG,certG),
\end{array}
$$
and the continuation ledgers
$$
\ContR(mapF,mapG,mapFG),\qquad
\ContR(modF,modG,modFG),\qquad
\ContR(target,modFG,certFG),
$$
then the composite graph has metric depth
$$
\mathsf{MetricDistanceDepth}(target)
=
\mathsf{MetricDistanceDepth}(source)+
\mathsf{MetricDistanceDepth}(mapFG).
$$
\end{theorem}
\begin{proof}
Composition closure for continuous-function carriers builds the composite
carrier from the two displayed carriers and the continuation ledgers. Its graph
fields are exactly a metric distance witness with endpoints $source$, $mapFG$,
and $target$.

Metric-distance depth addition for that witness gives the displayed equality.
\end{proof}

$\mathsf{append}(source,target)$.

$p,source,map,target,modulus,cert,distance,displayed:\mathsf{BHist}$, assume
$\mathsf{UnaryHistory}(p)$, the categorical and continuity rows
$$
\begin{array}{l}
\mathsf{CategoryHomCarrier}(source,target,map),\\
\mathsf{ContinuousModulusWitness}(target,modulus,cert),
\end{array}
$$
and the canonical source-target distance classifier
$$
\hsame(distance,\mathsf{append}(source,target)).
$$
Then the prefixed continuous-map carrier is exactly the prefixed canonical
distance classifier:
$$
\begin{array}{l}
\mathsf{ContinuousMapCarrier}
(\mathsf{append}(p,source),map,\mathsf{append}(p,target),modulus,
\mathsf{append}(p,cert),displayed)\\
\Longleftrightarrow
\hsame(displayed,
\mathsf{append}(\mathsf{append}(p,source),\mathsf{append}(p,target))).
\end{array}
$$
\end{theorem}
\begin{proof}
First assemble the unprefixed central carrier. Apply
Theorem~\ref{thm:continuousmap-carrier-categorical-canonical-distance-exactness}
in the reverse direction to the categorical hom row, the
target-side modulus witness, and
$\hsame(distance,\mathsf{append}(source,target))$. This gives
$\mathsf{ContinuousMapCarrier}(source,map,target,modulus,cert,distance)$.

With that carrier in hand, apply
Theorem~\ref{thm:continuousmap-prefix-canonical-distance-exactness} to the
unary prefix $p$ and the central carrier. Its conclusion is exactly the
displayed equivalence between the prefixed carrier and the canonical prefixed
source-target distance classifier.

The ten-field fit is: traditional target continuous maps read as categorical
arrows with visible modulus data under a unary prefix; BEDC source
$\mathsf{CategoryHomCarrier}$, $\mathsf{ContinuousModulusWitness}$, $\hsame$,
and $\mathsf{UnaryHistory}$; carrier $\mathsf{ContinuousMapCarrier}$;
classifier $\hsame$ on the displayed prefixed distance; stability from
categorical canonical-distance exactness and prefix canonical-distance
exactness; ledger prefix, source, map, target, modulus, certificate, and
displayed distance histories; core theorem this iff; exactness the displayed
prefixed classifier; downstream use compact-continuous and category-bridge
normalization; status $\mathsf{ProofTarget}$.
\end{proof}
