\begin{theorem}[ContinuousMap composite graph depth addition]
\label{thm:continuousmap-comp-graph-depth-add}
For histories $source,mid,target,mapF,mapG,mapFG,modF,modG,modFG,certF,certG,certFG$,
assume the two continuous-function carriers
$$
\begin{array}{l}
\mathsf{ContinuousFunctionCarrier}(source,mapF,mid,modF,certF),\\
\mathsf{ContinuousFunctionCarrier}(mid,mapG,target,modG,certG),
\end{array}
$$
and the continuation ledgers
$$
\ContR(mapF,mapG,mapFG),\qquad
\ContR(modF,modG,modFG),\qquad
\ContR(target,modFG,certFG),
$$
then the composite graph has metric depth
$$
\mathsf{MetricDistanceDepth}(target)
=
\mathsf{MetricDistanceDepth}(source)+
\mathsf{MetricDistanceDepth}(mapFG).
$$
\end{theorem}
\begin{proof}
Composition closure for continuous-function carriers builds the composite
carrier from the two displayed carriers and the continuation ledgers. Its graph
fields are exactly a metric distance witness with endpoints $source$, $mapFG$,
and $target$.

Metric-distance depth addition for that witness gives the displayed equality.
\end{proof}
\leanchecked{BEDC.Derived.ContinuousMapUp.ContinuousMap\_comp\_graph\_depth\_add}

\begin{theorem}[ContinuousMap carrier graph depth addition]
\label{thm:continuousmap-carrier-graph-cont-depth-add}
For histories $source,map,target,modulus,cert,distance:\mathsf{BHist}$, every
continuous-map carrier
$$
\mathsf{ContinuousMapCarrier}(source,map,target,modulus,cert,distance)
$$
exposes its graph continuation and the corresponding metric depth identity:
$$
\ContR(source,map,target)
\quad\land\quad
\mathsf{MetricDistanceDepth}(target)
=
\mathsf{MetricDistanceDepth}(source)+\mathsf{MetricDistanceDepth}(map).
$$
\end{theorem}
\begin{proof}
Project the continuous-function carrier from the displayed
$\mathsf{ContinuousMapCarrier}$. Its graph field is exactly the continuation
$\ContR(source,map,target)$. Repackage the source, map, and target unary rows
with this graph continuation as a metric-distance witness from $source$ and
$map$ to $target$, then apply metric-distance depth addition.
\end{proof}
\leanchecked{BEDC.Derived.ContinuousMapUp.ContinuousMapCarrier\_graph\_cont\_depth\_add}

\begin{theorem}[ContinuousMap carrier distance empty iff]
\label{thm:continuousmap-carrier-distance-empty-iff}
For histories $source,map,target,modulus,cert,distance:\mathsf{BHist}$, every
continuous-map carrier satisfies
$$
\mathsf{ContinuousMapCarrier}(source,map,target,modulus,cert,distance)
\Longrightarrow
\bigl(\hsame(distance,\emp)\Longleftrightarrow
\hsame(source,\emp)\land\hsame(target,\emp)\bigr).
$$
\end{theorem}
\begin{proof}
Project the metric-distance witness from the displayed
$\mathsf{ContinuousMapCarrier}$. If the displayed distance is $\hsame$ to
$\emp$, transport it to the empty distance ledger and apply metric
empty-distance readback to identify both endpoints with $\emp$. Conversely,
empty endpoint classifiers build the metric-distance witness with empty
distance ledger; metric result determinacy compares this witness with the
carrier's displayed distance.
\end{proof}
\leanchecked{BEDC.Derived.ContinuousMapUp.ContinuousMapCarrier\_distance\_empty\_iff}

\begin{theorem}[ContinuousMap carrier composition closure]
\label{thm:continuousmap-carrier-comp-closed}
For histories
$source,mid,target,mapF,mapG,mapFG,modF,modG,modFG,certF,certG,certFG,distF,distG:\mathsf{BHist}$,
assume the two continuous-map carriers
$$
\begin{array}{l}
\mathsf{ContinuousMapCarrier}(source,mapF,mid,modF,certF,distF),\\
\mathsf{ContinuousMapCarrier}(mid,mapG,target,modG,certG,distG),
\end{array}
$$
and the continuation ledgers
$$
\ContR(mapF,mapG,mapFG),\qquad
\ContR(modF,modG,modFG),\qquad
\ContR(target,modFG,certFG).
$$
Then the composite is again a continuous-map carrier with the canonical
source-target distance ledger:
$$
\mathsf{ContinuousMapCarrier}
(source,mapFG,target,modFG,certFG,\mathsf{append}(source,target)).
$$
\end{theorem}
\begin{proof}
Project the continuous-function halves from the two
$\mathsf{ContinuousMapCarrier}$ hypotheses. Apply
Theorem~\ref{thm:continuous-function-carrier-comp-closed} to those halves and
the three continuation ledgers. This produces the composite
$\mathsf{ContinuousFunctionCarrier}(source,mapFG,target,modFG,certFG)$.

Now package the canonical distance ledger. Apply
Theorem~\ref{thm:continuousmap-carrier-canonical-distance-iff} in the reverse
direction to the composite continuous-function carrier, yielding the displayed
$\mathsf{ContinuousMapCarrier}$ with distance
$\mathsf{append}(source,target)$.

The ten-field fit is: traditional target composition of uniformly continuous
maps, BEDC source $\mathsf{ContinuousMapCarrier}$ plus $\ContR$, carrier
$\mathsf{ContinuousMapCarrier}$, classifier $\hsame$, stability from
continuous-function composition and canonical metric ledger packaging, ledger
the hidden canonical distance, core theorem this closure, exactness
canonical-distance carrier iff, standard bridge deferred, and status
$\mathsf{ProofTarget}$.
\end{proof}
\leanchecked{BEDC.Derived.ContinuousMapUp.ContinuousMapCarrier\_comp\_closed}

\label{thm:continuousmap-empty-distance-comp-closed}
For histories
$source,mid,target,mapF,mapG,mapFG,modF,modG,modFG,certF,certG,certFG:\mathsf{BHist}$,
assume the two continuous-map carriers have empty distance ledgers:
$$

\mathsf{ContinuousMapCarrier}(source,mapF,mid,modF,certF,\emp),\\
\mathsf{ContinuousMapCarrier}(mid,mapG,target,modG,certG,\emp),
\end{array}
$$
and assume the continuation ledgers for graph composition, modulus
composition, and final certificate formation. Then $source$, $mid$, and
$target$ are all $\hsame$ to $\emp$, and the composite carrier also has empty
distance ledger:
$$
\mathsf{ContinuousMapCarrier}(source,mapFG,target,modFG,certFG,\emp).
$$
\end{theorem}
\begin{proof}
Apply the empty-distance metric readback to the two input carriers. The first
carrier identifies $source$ and $mid$ with $\emp$, while the second identifies
$mid$ and $target$ with $\emp$. Composition closure supplies the composite
continuous-function carrier. The empty endpoint comparisons repackage the
metric-distance witness for the composite as the empty distance ledger.
\end{proof}

\begin{theorem}[ContinuousMap carrier composition distance determinacy]
\label{thm:continuousmap-carrier-comp-distance-deterministic}
For histories
$source,mid,target,mapF,mapG,mapFG,modF,modG,modFG,certF,certG,certFG,distF,distG,displayed:\mathsf{BHist}$,
assume the two continuous-map carriers
$$
\begin{array}{l}
\mathsf{ContinuousMapCarrier}(source,mapF,mid,modF,certF,distF),\\
\mathsf{ContinuousMapCarrier}(mid,mapG,target,modG,certG,distG),
\end{array}
$$
the continuation ledgers
$$
\ContR(mapF,mapG,mapFG),\qquad
\ContR(modF,modG,modFG),\qquad
\ContR(target,modFG,certFG),
$$
and a displayed composite carrier
$$
\mathsf{ContinuousMapCarrier}(source,mapFG,target,modFG,certFG,displayed).
$$
Then the displayed composite distance is the canonical source-target ledger:
$$
\hsame(displayed,\mathsf{append}(source,target)).
$$
\end{theorem}
\begin{proof}
Theorem~\ref{thm:continuousmap-carrier-comp-closed} applied to the two input
carriers and the three continuation ledgers builds the canonical composite
carrier
$$
\mathsf{ContinuousMapCarrier}
(source,mapFG,target,modFG,certFG,\mathsf{append}(source,target)).
$$

Compare the displayed composite carrier with this canonical carrier by
Theorem~\ref{thm:continuousmap-target-cert-distance-deterministic}, with the
same source, map, target, modulus, and certificate histories on both sides.
The distance component of that determinacy conclusion gives
$\hsame(displayed,\mathsf{append}(source,target))$.
\end{proof}
\leanchecked{BEDC.Derived.ContinuousMapUp.ContinuousMapCarrier\_comp\_distance\_deterministic}

\begin{theorem}[ContinuousMap composite canonical distance exactness]
\label{thm:continuousmap-composite-canonical-distance-exactness}
For histories
$source,mid,target,mapF,mapG,mapFG,modF,modG,modFG,certF,certG,certFG,distF,distG,displayed:\mathsf{BHist}$,
assume the two continuous-map carriers
$$
\begin{array}{l}
\mathsf{ContinuousMapCarrier}(source,mapF,mid,modF,certF,distF),\\
\mathsf{ContinuousMapCarrier}(mid,mapG,target,modG,certG,distG),
\end{array}
$$
and the continuation ledgers
$$
\ContR(mapF,mapG,mapFG),\qquad
\ContR(modF,modG,modFG),\qquad
\ContR(target,modFG,certFG).
$$
Then the displayed composite carrier predicate is exactly the canonical
source-target distance classifier:
$$
\mathsf{ContinuousMapCarrier}(source,mapFG,target,modFG,certFG,displayed)
\Longleftrightarrow
\hsame(displayed,\mathsf{append}(source,target)).
$$
\end{theorem}
\begin{proof}
In the forward direction, suppose the displayed composite carrier is given.
Apply
Theorem~\ref{thm:continuousmap-carrier-comp-distance-deterministic} to the two
input carriers, the three continuation ledgers, and the displayed composite
carrier. This gives $\hsame(displayed,\mathsf{append}(source,target))$.

Conversely, suppose the displayed distance satisfies
$\hsame(displayed,\mathsf{append}(source,target))$.
Theorem~\ref{thm:continuousmap-carrier-comp-closed} applied to the two input
carriers and the three continuation ledgers builds the canonical composite
carrier with distance $\mathsf{append}(source,target)$. Project its
continuous-function half.

Now apply
Theorem~\ref{thm:continuousmap-carrier-canonical-distance-exactness} in the
reverse direction to that continuous-function half and the displayed
$\hsame$ row. The result is the composite
$\mathsf{ContinuousMapCarrier}(source,mapFG,target,modFG,certFG,displayed)$.

The ten-field fit is: traditional target composition of uniformly continuous
maps with exact hidden distance ledger; BEDC source two
$\mathsf{ContinuousMapCarrier}$ hypotheses plus three continuation ledgers;
carrier $\mathsf{ContinuousMapCarrier}$; classifier $\hsame$ on the displayed
distance; stability from composition closure and canonical-distance exactness;
ledger the hidden source-target distance; core theorem this iff; exactness the
displayed equivalence; standard bridge ordinary composition distance
bookkeeping; status $\mathsf{ProofTarget}$.
\end{proof}
\leanchecked{BEDC.Derived.ContinuousMapUp.ContinuousMapCarrier\_comp\_displayed\_distance\_canonical\_iff}

\begin{theorem}[ContinuousMap composition canonical certificate package]
\label{thm:continuousmap-composition-canonical-certificate-package}
For histories
$source,mid,target,mapF,mapG,mapFG,modF,modG,modFG,certF,certG,certFG,distF,distG,displayed:\mathsf{BHist}$,
assume the two continuous-map carriers
$$
\begin{array}{l}
\mathsf{ContinuousMapCarrier}(source,mapF,mid,modF,certF,distF),\\
\mathsf{ContinuousMapCarrier}(mid,mapG,target,modG,certG,distG),
\end{array}
$$
and the continuation ledgers
$$
\ContR(mapF,mapG,mapFG),\qquad
\ContR(modF,modG,modFG),\qquad
\ContR(target,modFG,certFG).
$$
Then the composition data determine the canonical certificate package:
\begin{enumerate}
\item the composite carrier closes over the canonical source-target distance
ledger,
$$
\mathsf{ContinuousMapCarrier}
(source,mapFG,target,modFG,certFG,\mathsf{append}(source,target));
$$

\item every displayed composite carrier with the same visible fields has that
canonical distance ledger, namely if
$$
\mathsf{ContinuousMapCarrier}(source,mapFG,target,modFG,certFG,displayed),
$$
then
$$
\hsame(displayed,\mathsf{append}(source,target));
$$

\item the composite graph obeys metric-depth accounting,
$$
\mathsf{MetricDistanceDepth}(target)
=
\mathsf{MetricDistanceDepth}(source)+
\mathsf{MetricDistanceDepth}(mapFG).
$$
\end{enumerate}
\end{theorem}
\begin{proof}
For the carrier row, apply
Theorem~\ref{thm:continuousmap-carrier-comp-closed} to the two input carriers
and the three continuation ledgers. This gives the displayed
$\mathsf{ContinuousMapCarrier}$ with distance
$\mathsf{append}(source,target)$.

For distance uniqueness, suppose a displayed composite carrier with the same
source, map, target, modulus, and certificate histories is given. Compare the
displayed carrier with the canonical carrier just obtained by
Theorem~\ref{thm:continuousmap-target-cert-distance-deterministic}. The
distance component gives $\hsame(displayed,\mathsf{append}(source,target))$.

For metric-depth accounting, project the continuous-function halves from the
two input $\mathsf{ContinuousMapCarrier}$ hypotheses. Apply
Theorem~\ref{thm:continuousmap-comp-graph-depth-add} to those halves and the
same three continuation ledgers, producing the stated equality for
$\mathsf{MetricDistanceDepth}$.
\end{proof}

\begin{theorem}[ContinuousMap composition certificate determinacy]
\label{thm:continuousmap-comp-certificate-deterministic}
For histories
$source,mid,target,mapF,mapG,mapFG,modF,modG,modFG,certF,certG,certFG,certFG_2,distF,distG,displayed:\mathsf{BHist}$,
assume the two continuous-map carriers
$$
\mathsf{ContinuousMapCarrier}(source,mapF,mid,modF,certF,distF),
\qquad
\mathsf{ContinuousMapCarrier}(mid,mapG,target,modG,certG,distG),
$$
and the composition ledgers
$$
\ContR(mapF,mapG,mapFG),\qquad
\ContR(modF,modG,modFG),\qquad
\ContR(target,modFG,certFG).
$$
If another displayed composite carrier has the same source, map, target,
modulus, and a certificate $certFG_2$,
$$
\mathsf{ContinuousMapCarrier}(source,mapFG,target,modFG,certFG_2,displayed),
$$
then the certificate histories agree:
$$
\hsame(certFG,certFG_2).
$$
\end{theorem}
\begin{proof}
First close the canonical composite carrier from the two input carriers and the
three continuation ledgers by
Theorem~\ref{thm:continuousmap-carrier-comp-closed}. Compare this canonical
carrier with the displayed carrier by
Theorem~\ref{thm:continuousmap-target-cert-distance-deterministic}. The
certificate component of that comparison is the required
$\hsame(certFG,certFG_2)$.
\end{proof}
\leanchecked{BEDC.Derived.ContinuousMapUp.ContinuousMapCarrier\_comp\_certificate\_deterministic}

\begin{theorem}[ContinuousMap composite target, certificate, and distance determinacy]
\label{thm:continuousmap-carrier-comp-target-cert-distance-deterministic}
For histories
$source,mid,target,target_2,mapF,mapG,mapFG,modF,modG,modFG,certF,certG,certFG,certFG_2,distF,distG,displayed:\mathsf{BHist}$,
assume the two continuous-map carriers
$$
\mathsf{ContinuousMapCarrier}(source,mapF,mid,modF,certF,distF),
\qquad
\mathsf{ContinuousMapCarrier}(mid,mapG,target,modG,certG,distG),
$$
and the composition ledgers
$$
\ContR(mapF,mapG,mapFG),\qquad
\ContR(modF,modG,modFG),\qquad
\ContR(target,modFG,certFG).
$$
If a displayed composite carrier uses the same source, composite map, and
modulus but displays target $target_2$, certificate $certFG_2$, and distance
$displayed$, then the displayed rows agree with the canonical composite:
$$
\begin{array}{l}
\hsame(target,target_2)\land \hsame(certFG,certFG_2)\land
\hsame(displayed,\mathsf{append}(source,target))\\
\qquad{}\land\ContR(source,target,\mathsf{append}(source,target)).
\end{array}
$$
\end{theorem}
\begin{proof}
First close the canonical composite carrier from the two input carriers and
the three continuation ledgers. Its target is $target$, its certificate is
$certFG$, and its distance ledger is $\mathsf{append}(source,target)$. Compare
this canonical carrier with the displayed composite carrier by
\autoref{thm:continuousmap-target-cert-distance-deterministic}. The comparison
gives target and certificate determinacy, identifies the canonical distance
with $displayed$, and exposes the canonical endpoint continuation. Symmetry of
$\hsame$ orients the displayed distance as
$\hsame(displayed,\mathsf{append}(source,target))$.
\end{proof}
\leanchecked{BEDC.Derived.ContinuousMapUp.ContinuousMapCarrier\_comp\_target\_cert\_distance\_deterministic}
