\chapter{A Concrete Naming Certificate for \(\PreorderUp\)}
\label{ch:concrete-instances-preorder-namecert}

The preorder interface \(\PreorderUp\) is licensed by a carrier with a binary relation \(\le : C \times C \to \mathsf{Prop}\) satisfying reflexivity and transitivity under the carrier's classifier. It is the simplest order-theoretic NameCert and self-contained: no other licensed object is required as a dependency.

\begin{remark}[Preorder certificate scope]
\label{rem:concrete-instances-preorder-namecert-roadmap}
This chapter records the preorder licensing target through a history
carrier, restricted classifier, and reflexive-transitive relation data.
\end{remark}

\section{Carrier definition}

\begin{definition}[Preorder carrier]
\label{def:preorder-carrier}
Use an abstract carrier \(C\), classifier \(\sim_C\), and relation \(\le_C\) with preorder law fields.
\end{definition}

\section{Concrete unary prefix preorder instance}
\label{sec:preorder-concrete-unary-prefix-instance}

\begin{definition}[Concrete unary prefix preorder instance]
\label{def:concrete-unary-prefix-preorder-instance}
The concrete unary prefix preorder instance is the following specialization of
the $\PreorderUp$ source data:
$$
\mathsf{Carrier}:=\mathsf{PreorderCarrier}=\mathsf{UnaryHistory}.
$$
$$
\mathsf{classifier}:=\mathsf{PreorderCarrierClassifierSpec}(\mathsf{PreorderCarrier}).
$$
$$
\mathsf{le}:=\mathsf{PreorderPrefixLE}.
$$
\end{definition}

\begin{theorem}[Concrete unary prefix preorder laws]
\label{thm:concrete-unary-prefix-preorder-laws}
For the concrete unary prefix preorder instance, the carrier is inhabited by
$\emp$, carrier membership transports along $\hsame$, the relation
$\mathsf{PreorderPrefixLE}$ is reflexive on carrier histories, transitive, and
congruent under the classifier in both arguments.
\end{theorem}
\begin{proof}
The carrier and transport fields are supplied by the checked semantic naming
certificate named $\mathsf{BEDC.Derived.PreorderUp.}$
$\mathsf{preorder\_carrier\_semantic\_name\_certificate}$.
This identifies the concrete carrier with unary histories and gives the
$\hsame$ transport operation used by the restricted classifier.

The relational fields come from the checked concrete prefix stability
certificate named $\mathsf{BEDC.Derived.PreorderUp.}$
$\mathsf{preorder\_prefix\_stability\_certificate\_fields}$.
Its reflexivity component is the carrier-relative preorder reflexivity field,
its congruence component is the classifier-respecting transport of the prefix
relation in both endpoints, and its transitivity component is the concrete
prefix composition law.

At the theorem level, reflexivity is obtained by applying
$\mathsf{BEDC.Derived.PreorderUp.PreorderPrefixLE\_of\_hsame}$ to the
reflexive $\hsame$ witness, while transitivity is
$\mathsf{BEDC.Derived.PreorderUp.PreorderPrefixLE\_trans}$. Repacking these
fields gives the concrete unary prefix preorder laws.
\end{proof}

Thus the $\PreorderUp$ schema is instantiated by this concrete $\mathsf{BHist}$
carrier and relation. The corresponding formal target can therefore address
the concrete prefix preorder directly, rather than only the relation fields of
an arbitrary carrier.

\section{Concrete singleton empty preorder instance}
\label{sec:preorder-concrete-singleton-empty-instance}

\begin{definition}[Concrete singleton empty preorder instance]
\label{def:concrete-singleton-empty-preorder-instance}
Let $A_{\emp}$ be the singleton history source of
Definition~\ref{def:list-singleton-history-source} at $a_0=\emp$. Define the
restricted classifier by
$$
\rho_{\emp}(h,k):=
A_{\emp}(h)\land A_{\emp}(k)\land\hsame(h,k).
$$
Define the carrier-guarded restricted order by
$$
h\le_{\emp}^{\mathrm{sg}}k:=
A_{\emp}(h)\land A_{\emp}(k)\land \mathsf{PreorderPrefixLE}(h,k).
$$
The concrete preorder package is the fixed $\mathsf{BHist}$ tuple
$(A_{\emp},\rho_{\emp},\le_{\emp}^{\mathrm{sg}})$.
\end{definition}

\begin{theorem}[Singleton empty preorder laws]
\label{thm:singleton-empty-preorder-laws}
For the concrete package of
\autoref{def:concrete-singleton-empty-preorder-instance}, the carrier is
inhabited by $\emp$, carrier membership transports along $\hsame$,
$\le_{\emp}^{\mathrm{sg}}$ is reflexive on carried histories and transitive,
and $\le_{\emp}^{\mathrm{sg}}$ is congruent under $\rho_{\emp}$ in both
endpoints. Moreover,
\autoref{thm:singleton-preorder-guarded-prefix-exactness} gives the exact
classifier row for all histories.
\end{theorem}
\begin{proof}
The singleton source supplies the inhabitance witness $A_{\emp}(\emp)$.
Unfolding the source shows that any carried endpoint is identified with
$\emp$ by $\hsame$, and transport along $\hsame$ is exactly the transport law
recorded for the singleton history source.

Reflexivity follows by applying
$\mathsf{PreorderPrefixLE\_of\_hsame}$ to the reflexive witness
$\hsame(h,h)$ for each carried history $h$, then pairing the prefix witness
with the two copies of the carrier witness for $h$. Transitivity projects the
two guarded comparisons to their prefix witnesses and applies
$\mathsf{PreorderPrefixLE\_trans}$; the endpoint carrier witnesses are carried
by the first and second guarded comparisons.

For congruence, unfold $\rho_{\emp}$ at the source and target endpoints. The
classifier fields provide the carrier witnesses and the $\hsame$ witnesses
needed to move each endpoint. The prefix stability certificate for
$\mathsf{PreorderPrefixLE}$ then transports the comparison in the left endpoint
and in the right endpoint, after which the transported endpoint carriers are
repacked into the guarded comparison.

The exact classifier row is
\autoref{thm:singleton-preorder-guarded-prefix-exactness}.
\end{proof}

\begin{theorem}[Singleton guarded prefix preorder exactness]
\label{thm:singleton-preorder-guarded-prefix-exactness}
For all histories $h,k:\mathsf{BHist}$ in the concrete singleton empty
preorder instance,
$$
\rho_{\emp}(h,k)
\Longleftrightarrow
(h\le_{\emp}^{\mathrm{sg}}k\land k\le_{\emp}^{\mathrm{sg}}h).
$$
\end{theorem}
\begin{proof}
For the forward direction, unfold $\rho_{\emp}(h,k)$. The classifier witness
contains the two carrier witnesses and the history-sameness witness
$\hsame(h,k)$. Apply \autoref{thm:preorder-prefix-le-of-hsame} to
$\hsame(h,k)$ to obtain the prefix comparison from $h$ to $k$.

Apply the same theorem to the symmetric history-sameness witness to obtain the
prefix comparison from $k$ to $h$. Pair each prefix comparison with the two
carrier witnesses in the appropriate order; this gives the two guarded
comparisons.

For the reverse direction, project the carrier witnesses from either guarded
comparison. Project the prefix witnesses from both guarded comparisons and
apply \autoref{thm:preorder-prefix-le-antisymm-hsame} to recover
$\hsame(h,k)$.

Finally, combine the carrier witnesses with this history-sameness witness.
This is exactly the restricted classifier $\rho_{\emp}(h,k)$.
\end{proof}

\begin{theorem}[Singleton empty preorder terminal comparison package]
\label{thm:preorder-singleton-empty-terminal-comparison-package}
For histories $h,k:\mathsf{BHist}$ carried by $A_{\emp}$ in the concrete
singleton empty preorder instance, the restricted comparison package is pinned
to the terminal endpoint:
$$
A_{\emp}(h)\land A_{\emp}(k)
\Longrightarrow
h\le_{\emp}^{\mathrm{sg}}k\land k\le_{\emp}^{\mathrm{sg}}h\land
\rho_{\emp}(h,k).
$$
Its ten-field fit is the following: the traditional target is the one-point
preorder; the BEDC source is $\mathsf{BHist}$ with singleton carrier
$A_{\emp}$; the carrier is $A_{\emp}$; the classifier is $\rho_{\emp}$;
stability is supplied by singleton endpoint transport and
\autoref{thm:preorder-prefix-le-of-hsame}; the ledger records the two endpoint
witnesses; the core theorem is the displayed terminal comparison package;
exactness is guarded prefix exactness; the standard bridge is the one-point
preorder; and the status is $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Assume $A_{\emp}(h)$ and $A_{\emp}(k)$. Unfolding the singleton carrier
witnesses identifies both histories with the endpoint $\emp$, giving
$h\hsame\emp$ and $k\hsame\emp$.

Compose the first witness with the symmetry of the second to obtain
$h\hsame k$, and compose the second witness with the symmetry of the first to
obtain $k\hsame h$.

Apply \autoref{thm:preorder-prefix-le-of-hsame} to these two $\hsame$ witnesses.
The first application gives the prefix part of
$h\le_{\emp}^{\mathrm{sg}}k$, and the second gives the prefix part of
$k\le_{\emp}^{\mathrm{sg}}h$.

Finally, package $\rho_{\emp}(h,k)$ from the two carrier witnesses together
with the composed witness $h\hsame k$. Repack the two carrier witnesses with
the two prefix witnesses to obtain the guarded comparisons; these comparisons
and the classifier witness are exactly the displayed terminal comparison
package.
\end{proof}

\begin{definition}[Singleton empty preorder public comparison readback]
\label{def:preorder-singleton-empty-public-comparison-readback}
For the concrete singleton empty preorder instance, the public readback of
the guarded comparison $h\le_{\emp}^{\mathrm{sg}}k$ is the endpoint-carrier
pair
$$
A_{\emp}(h)\land A_{\emp}(k).
$$
\end{definition}

\begin{theorem}[Singleton empty preorder single-comparison exactness]
\label{thm:preorder-singleton-empty-single-comparison-exactness}
For all histories $h,k:\mathsf{BHist}$ in the concrete singleton empty
preorder instance, one guarded comparison is exactly endpoint carriage:
$$
h\le_{\emp}^{\mathrm{sg}}k
\Longleftrightarrow
A_{\emp}(h)\land A_{\emp}(k).
$$
Its ten-field fit is the following: the traditional target is one-point
preorder comparison readback; the BEDC source is $\mathsf{BHist}$; the carrier
is $A_{\emp}$; the classifier is $\rho_{\emp}$; the relation is
$\le_{\emp}^{\mathrm{sg}}$; stability is supplied by singleton transport and
\autoref{thm:preorder-prefix-le-of-hsame}; the ledger records the two endpoint
witnesses and the single comparison witness; the core theorem is this
single-comparison exactness block; exactness is the displayed if-and-only-if;
the standard bridge is the one-point preorder; and the status is
$\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
For the forward direction, unfold the guarded relation
$h\le_{\emp}^{\mathrm{sg}}k$. Its witness is a package consisting of
$A_{\emp}(h)$, $A_{\emp}(k)$, and
$\mathsf{PreorderPrefixLE}(h,k)$. Project the first two fields to obtain the
endpoint-carrier pair.

For the reverse direction, assume $A_{\emp}(h)\land A_{\emp}(k)$. Apply
\autoref{thm:preorder-singleton-empty-terminal-comparison-package} to these
two carrier witnesses. The resulting terminal comparison package contains
$h\le_{\emp}^{\mathrm{sg}}k$ as its first component, which is the required
single guarded comparison.
\end{proof}

\begin{theorem}[Singleton empty $\PreorderUp$ naming certificate]
\label{thm:singleton-empty-preorder-namecert}
The tuple $(A_{\emp},\rho_{\emp},\le_{\emp}^{\mathrm{sg}})$ of
\autoref{def:concrete-singleton-empty-preorder-instance} supplies the source,
pattern, classifier, stability, ledger, and singleton exactness rows for a
concrete $\PreorderUp$ certificate over $A_{\emp}$, $\rho_{\emp}$, and
$\le_{\emp}^{\mathrm{sg}}$. The exactness row is
$$
\rho_{\emp}(h,k)\Longleftrightarrow
(h\le_{\emp}^{\mathrm{sg}}k\land k\le_{\emp}^{\mathrm{sg}}h).
$$
\end{theorem}
\begin{proof}
The carrier row is the singleton $\mathsf{BHist}$ source $A_{\emp}$, and
\autoref{thm:singleton-empty-preorder-laws} supplies its inhabitance by $\emp$
together with transport along $\hsame$. The pattern row is the restricted
comparison judgment $h\le_{\emp}^{\mathrm{sg}}k$, and the classifier row is
$\rho_{\emp}$.

The same law package gives the stability rows: reflexivity of
$\le_{\emp}^{\mathrm{sg}}$ on carried histories, transitivity, and endpoint
congruence for the classifier. The ledger row is
\autoref{def:preorder-ledger-policy}, which packages the two carried endpoints
and the restricted comparison witness.

\autoref{thm:singleton-preorder-guarded-prefix-exactness} gives the displayed
two-sided exactness statement for all histories. That theorem is exactly the
singleton exactness row of the concrete $\PreorderUp$ certificate.
\end{proof}

This singleton package gives a concrete $\mathsf{BHist}$ source for the
$\PreorderUp$ relation fields, so order formalization can target this carrier
rather than arbitrary relation parameters.

\begin{definition}[Preorder source specification]

\begin{definition}[Preorder pattern specification]

\begin{theorem}[Prefix from history sameness]
\label{thm:preorder-prefix-le-of-hsame}
If $h$ and $k$ are internally history-same, then $h\le k$ in the concrete unary
history prefix relation.
\end{theorem}
\begin{proof}
Eliminate the $\hsame(h,k)$ witness. The prefix relation is then witnessed by
the empty unary tail together with the right-unit continuation law for $h$.
\end{proof}
\leanchecked{BEDC.Derived.PreorderUp.PreorderPrefixLE\_of\_hsame}

\begin{theorem}[Prefix append by unary tail]
\label{thm:preorder-prefix-append-unary-tail}
For every history \(h\), appending a unary tail to \(h\) gives a concrete
prefix relation from \(h\) to the append result.
\end{theorem}
\begin{proof}
Use the unary tail itself as the prefix witness. The continuation component is
the definitional append equation, introduced by the BEDC continuation
constructor.
\end{proof}
\leanchecked{BEDC.Derived.PreorderUp.PreorderPrefixLE\_append\_unary\_tail}

\begin{theorem}[Prefix from empty is unary carrier]
\label{thm:preorder-prefix-empty-left-iff-unary}
For the concrete unary history preorder, $\emp\le h$ holds exactly when $h$ is in the preorder carrier.
\end{theorem}
\begin{proof}
Eliminate the prefix witness from $\emp$ to expose its unary tail and continuation result. The left-unit result identifies the target with the tail, so unary membership transports to $h$. Conversely, use $h$ itself as the unary tail and the continuation left unit.
\end{proof}
\leanchecked{BEDC.Derived.PreorderUp.PreorderPrefixLE\_empty\_left\_iff\_unary}

\begin{theorem}[Prefix to empty identifies the source]
\label{thm:preorder-prefix-empty-target-iff}
For the concrete unary history preorder, $h\le \emp$ holds exactly when
$h\hsame \emp$.
\end{theorem}
\begin{proof}
Eliminate the prefix witness into a unary tail and a continuation into
$\emp$. The empty-result inversion theorem forces the source history to be
$\emp$. Conversely, after the history-sameness proof identifies the source
with $\emp$, use the empty unary tail and the continuation right unit.
\end{proof}
\leanchecked{BEDC.Derived.PreorderUp.PreorderPrefixLE\_empty\_target\_iff}

\begin{theorem}[Prefix relation preserves the carrier]
\label{thm:preorder-prefix-preserves-carrier}
For the concrete unary history carrier, if $h$ is a carrier history and $h$ is related to $k$ by appending a unary prefix tail, then $k$ is also in the carrier.
\end{theorem}
\begin{proof}
Unpack the prefix witness into its unary tail and continuation proof. Unary continuation closure sends the carrier witness for $h$, the unary witness for the tail, and the continuation proof to a carrier witness for $k$.
\end{proof}
\leanchecked{BEDC.Derived.PreorderUp.PreorderPrefixLE\_preserves\_carrier}

\begin{theorem}[Prefix target carrier reflects source carrier]
\label{thm:preorder-prefix-source-carrier-of-target-carrier}
For the concrete unary history carrier, if \(h\le k\) is witnessed by a unary
prefix tail and \(k\) is in the carrier, then \(h\) is also in the carrier.
\end{theorem}
\begin{proof}
Unpack the prefix witness into its unary tail and continuation proof. The
target carrier is unary, so unary continuation left-factor closure recovers
unary membership of the source history \(h\).
\end{proof}
\leanchecked{BEDC.Derived.PreorderUp.PreorderPrefixLE\_source\_carrier\_of\_target\_carrier}

\begin{theorem}[Prefix carrier excludes zero target]
\label{thm:preorder-prefix-carrier-no-e0-target}
If $h$ is in the concrete preorder carrier, then a prefix comparison from

\end{theorem}

The prefix relation preserves the concrete carrier from source to target.
Thus $\mathsf{e0}(k)$ would be a unary history, contradicting the BEDC unary

\begin{definition}[Preorder classifier specification]
\label{def:preorder-classifier-specification}
The classifier is the supplied carrier sameness, and the relation is stable under classifier replacement.
\end{definition}

\begin{theorem}[Prefix right extension]
\label{thm:preorder-prefix-right-extension}
If \(h\le k\) is witnessed by a unary continuation tail, and \(k\) continues by a unary tail to \(r\), then \(h\le r\).
\end{theorem}
\leanchecked{BEDC.Derived.PreorderUp.PreorderPrefixLE\_right\_extension}
\begin{proof}
Unpack the first prefix witness as a unary tail from \(h\) to \(k\). Compose this continuation with the continuation from \(k\) to \(r\); associativity of continuation supplies the resulting path from \(h\), and unary closure under append supplies the combined unary tail.
\end{proof}

\begin{theorem}[Prefix preserves strict right extension]
\label{thm:preorder-prefix-le-strict-right-extension}
If $h\le k$ is witnessed by a unary continuation tail and $k$ is a strict
unary prefix of $r$, then $h$ is a strict unary prefix of $r$.
\end{theorem}
\begin{proof}
Unpack the weak prefix tail from $h$ to $k$ and the nonempty strict tail from
$k$ to $r$. Appending the two tails gives a unary tail from $h$ to $r$ by
unary append closure and continuation associativity. If the appended tail were
$\emp$, append-empty inversion would force the strict right tail to be
$\emp$, contradicting strictness.
\end{proof}
\leanchecked{BEDC.Derived.PreorderUp.PreorderPrefixLE\_strict\_right\_extension}

\begin{theorem}[Prefix transitivity]
\label{thm:preorder-prefix-transitivity}
If \(h\le k\) and \(k\le r\) are witnessed by unary continuation tails, then
\(h\le r\).
\end{theorem}
\leanchecked{BEDC.Derived.PreorderUp.PreorderPrefixLE\_trans}
\begin{proof}
Unpack the two prefix witnesses. Append the two unary tails, use unary closure
under append for the combined tail, and use continuation associativity to
identify the direct continuation from \(h\) to \(r\).
\end{proof}

\begin{theorem}[Strict unary prefix preserves right extension]
\label{thm:nat-unary-strict-prefix-right-extension}
If $h$ is a strict unary prefix of $k$ and $k\le r$ is witnessed by a unary
continuation tail, then $h$ is a strict unary prefix of $r$.
\end{theorem}
\begin{proof}
Unpack the nonempty strict tail from $h$ to $k$ and the weak prefix tail from
$k$ to $r$. Their append is unary and composes the two continuations from
$h$ to $r$. If the appended tail were $\emp$, append-empty inversion would
force the original strict tail to be $\emp$, contradicting strictness.
\end{proof}
\leanchecked{BEDC.Derived.PreorderUp.NatUnaryStrictPrefix\_right\_extension}

\begin{theorem}[Prefix antisymmetry collapses to history sameness]
\label{thm:preorder-prefix-le-antisymm-hsame}
If \(h\le k\) and \(k\le h\) are both witnessed by unary continuation tails,
then \(\hsame(h,k)\).
\end{theorem}
\leanchecked{BEDC.Derived.PreorderUp.PreorderPrefixLE\_antisymm\_hsame}
\begin{proof}
Unpack the two prefix witnesses. Substituting the first continuation equation
into the second gives a round trip from \(h\) through the two tails back to
\(h\). Associativity rewrites this as equality between the empty right tail and
the combined tail under the same left prefix, and left cancellation forces the
forward tail to be empty. The first continuation equation then reduces to
\(\hsame(h,k)\).
\end{proof}

\begin{theorem}[Concrete prefix order laws up to history-sameness]
\label{thm:preorder-prefix-order-laws-hsame}
The concrete unary prefix relation $\mathsf{PreorderPrefixLE}$ is reflexive,
transitive, and antisymmetric up to $\hsame$: for every history $h$,
$h\le h$; for all histories $h,k,r$, $h\le k$ and $k\le r$ imply
$h\le r$; and for all histories $h,k$, $h\le k$ and $k\le h$ imply
$\hsame(h,k)$.
\end{theorem}
\begin{proof}
Reflexivity follows from $\hsame$ reflexivity together with
$\mathsf{PreorderPrefixLE\_of\_hsame}$. Transitivity is exactly
$\mathsf{PreorderPrefixLE\_trans}$. Antisymmetry up to history-sameness is
exactly $\mathsf{PreorderPrefixLE\_antisymm\_hsame}$.
\end{proof}

\begin{theorem}[Prefix left extension]
\label{thm:preorder-prefix-left-extension}
If \(x\) continues by a unary tail to \(h\), and \(h\le k\) is witnessed by a
unary continuation tail, then \(x\le k\).
\end{theorem}
\leanchecked{BEDC.Derived.PreorderUp.PreorderPrefixLE\_left\_extension}
\begin{proof}
Unpack the prefix witness from \(h\) to \(k\). Append the incoming unary tail to
the prefix tail, use unary closure under append for the combined tail, and use
continuation associativity to identify the resulting continuation from \(x\) to
\(k\).
\end{proof}

\begin{theorem}[Prefix append tail]
\label{thm:preorder-prefix-append-tail}
For every history \(h\), appending a unary history \(t\) gives a concrete
prefix comparison \(h \le \mathsf{append}(h,t)\).
\end{theorem}
\begin{proof}
Use \(t\) itself as the prefix witness. Its unary membership is the supplied
carrier for the tail, and the continuation component is the defining
append-result equality.
\end{proof}
\leanchecked{BEDC.Derived.PreorderUp.PreorderPrefixLE\_append\_tail}

\begin{theorem}[Prefix append tail backforces empty]
\label{thm:preorder-prefix-append-tail-backforces-empty}
If \(t\) is unary and \(\mathsf{append}(h,t)\le h\), then \(t\) is internally
the same as \(\emp\).
\end{theorem}
\begin{proof}
The unary tail gives the forward prefix comparison
\(h\le\mathsf{append}(h,t)\). Together with the assumed backward comparison,
prefix antisymmetry gives \(\hsame(h,\mathsf{append}(h,t))\). Rewriting the
left endpoint as \(\mathsf{append}(h,\emp)\) and cancelling the common left
context forces \(\hsame(t,\emp)\).
\end{proof}
\leanchecked{BEDC.Derived.PreorderUp.PreorderPrefixLE\_append\_tail\_backforces\_empty}

\begin{theorem}[Prefix append tail backforces empty equivalence]
\label{thm:preorder-prefix-append-tail-backforces-empty-iff}
If \(t\) is unary, then \(\mathsf{append}(h,t)\le h\) holds exactly when
\(t\hsame\emp\).
\end{theorem}
\begin{proof}
The forward direction is the backforces-empty implication for unary tails. For
the reverse direction, \(t\hsame\emp\) identifies
\(\mathsf{append}(h,t)\) with the right-unit endpoint \(h\); the prefix
relation is then obtained from endpoint sameness.
\end{proof}
\leanchecked{BEDC.Derived.PreorderUp.PreorderPrefixLE\_append\_tail\_backforces\_empty\_iff}

\begin{theorem}[Prefix left context monotonicity]
\label{thm:preorder-prefix-left-context-monotonicity}
If \(h\le k\) is witnessed by a unary continuation tail, then appending the
same left context to both endpoints preserves the prefix comparison:
\(\mathsf{append}(x,h)\le\mathsf{append}(x,k)\).
\end{theorem}
\begin{proof}
Unpack the prefix witness as a unary tail and a continuation equation from
\(h\) to \(k\). Reuse the same unary tail after the common left context.
Associativity of append identifies the resulting continuation endpoint with
\(\mathsf{append}(x,k)\).
\end{proof}
\leanchecked{BEDC.Derived.PreorderUp.PreorderPrefixLE\_append\_left\_context}

\begin{theorem}[Prefix right context monotonicity]
\label{thm:preorder-prefix-right-context-monotonicity}
If \(x\) is unary and \(h\le k\) is witnessed by a unary continuation tail, then appending \(x\) to both endpoints preserves the prefix comparison:
\(\mathsf{append}(h,x)\le\mathsf{append}(k,x)\).
\end{theorem}
\leanchecked{BEDC.Derived.PreorderUp.PreorderPrefixLE\_append\_right\_context}
\begin{proof}
Unpack the prefix witness as a unary tail and a continuation equation from \(h\) to \(k\). Reuse the same tail. Associativity moves the right context next to the tail, unary commutativity swaps the two unary histories, and associativity moves the tail back to the continuation endpoint.
\end{proof}

\begin{theorem}[Prefix two-sided context monotonicity]
\label{thm:preorder-prefix-append-both-context}
If \(r\) is unary and \(h\le k\) is witnessed by a unary continuation tail, then
placing the same left context outside both right-contexted endpoints preserves
the prefix comparison:
\(\mathsf{append}(\ell,\mathsf{append}(h,r))\le
\mathsf{append}(\ell,\mathsf{append}(k,r))\).
\end{theorem}
\begin{proof}
First apply right-context monotonicity using the unary carrier of \(r\). Then
apply left-context monotonicity to the resulting comparison.
\end{proof}
\leanchecked{BEDC.Derived.PreorderUp.PreorderPrefixLE\_append\_both\_context}

\begin{theorem}[Prefix right context cancellation]
\label{thm:preorder-prefix-right-context-cancellation}
If \(x\) is unary and
\(\mathsf{append}(h,x)\le\mathsf{append}(k,x)\), then \(h\le k\).
\end{theorem}
\begin{proof}
Unpack the prefix witness as a unary tail and a continuation equation. Append
associativity brings the right context next to that tail, unary commutativity
exchanges the two unary histories, and associativity restores a common right
context. Right cancellation removes the shared \(x\), leaving the continuation
witness from \(h\) to \(k\).
\end{proof}
\leanchecked{BEDC.Derived.PreorderUp.PreorderPrefixLE\_cancel\_right\_context}

\begin{theorem}[Prefix left context cancellation]
\label{thm:preorder-prefix-left-context-cancellation}
If appending the same left context to \(h\) and \(k\) makes
\(\mathsf{append}(x,h)\le\mathsf{append}(x,k)\), then \(h\le k\).
\end{theorem}
\begin{proof}
Unpack the prefix witness as a unary tail and a continuation equation from
\(\mathsf{append}(x,h)\) to \(\mathsf{append}(x,k)\). Continuation associativity rewrites the
right side as a common left append by \(x\); left cancellation removes that
context and gives the continuation equation from \(h\) to \(k\) with the same
unary tail.
\end{proof}
\leanchecked{BEDC.Derived.PreorderUp.PreorderPrefixLE\_cancel\_left\_context}

\begin{theorem}[Prefix two-sided context cancellation]
\label{thm:preorder-prefix-cancel-both-context}
If \(r\) is unary and the two endpoints become prefix-comparable after adding a
common left context outside a common right context,
$$
\mathsf{append}(\ell,\mathsf{append}(h,r))\le
\mathsf{append}(\ell,\mathsf{append}(k,r)),
$$
then \(h\le k\).
\end{theorem}
\begin{proof}
First cancel the common left context, giving
\(\mathsf{append}(h,r)\le\mathsf{append}(k,r)\). Then use right-context
cancellation with the unary witness for \(r\), leaving the prefix comparison
from \(h\) to \(k\).
\end{proof}
\leanchecked{BEDC.Derived.PreorderUp.PreorderPrefixLE\_cancel\_both\_context}

\begin{theorem}[Prefix two-sided context equivalence]
\label{thm:preorder-prefix-append-both-context-iff}
If \(r\) is unary, then placing the same left context outside both
right-contexted endpoints preserves and reflects the concrete prefix
comparison:
$$
  \mathsf{append}(\ell,\mathsf{append}(h,r))\le
  \mathsf{append}(\ell,\mathsf{append}(k,r))
  \quad\Longleftrightarrow\quad
  h\le k.
$$
\end{theorem}
\begin{proof}
The forward direction first cancels the common left context, then cancels the
common unary right context. The reverse direction is exactly two-sided context
monotonicity.
\end{proof}
\leanchecked{BEDC.Derived.PreorderUp.PreorderPrefixLE\_append\_both\_context\_iff}

\begin{definition}[Opposite preorder certificate data]
\label{def:opposite-preorder-certificate-data}
Let a $\PreorderUp$ certificate be presented by a carrier predicate $\mathcal C$, classifier $\sim_C$, and relation $\le_C$. Define the opposite relation by
$$
a\le_C^{\mathrm{op}} b:=b\le_C a.
$$
The opposite preorder data $C^{\mathrm{op}}$ have the same carrier predicate $\mathcal C$, the same classifier $\sim_C$, and relation $\le_C^{\mathrm{op}}$.
\end{definition}

\begin{theorem}[Opposite preorder certificate]
\label{thm:opposite-preorder-certificate}
If $(\mathcal C,\sim_C,\le_C)$ satisfies the $\PreorderUp$ certificate obligations of \autoref{def:preorder-certificate-obligations}, then the data of \autoref{def:opposite-preorder-certificate-data} also satisfy the $\PreorderUp$ certificate obligations. Thus the same carrier and classifier carry a preorder certificate with relation $\le_C^{\mathrm{op}}$.
\end{theorem}
\begin{proof}
Unpack the preorder obligations through \autoref{def:preorder-certificate-obligations}. The carrier row of \autoref{def:preorder-carrier} is unchanged, and the carrier-and-classifier core is the same $\NameCert$ data. Hence classifier reflexivity, symmetry, transitivity, and carrier transport are copied from \autoref{def:lean-carrier-naming-certificate}. The source row of \autoref{def:preorder-source-specification} is the same carrier and classifier equipped with the reversed relation symbol.

The reflexivity field for the opposite relation is immediate. For every carried endpoint $a$,
$$
a\le_C^{\mathrm{op}}a
$$
unfolds to $a\le_C a$, which is the original preorder reflexivity field.

For transitivity, assume $a\le_C^{\mathrm{op}} b$ and $b\le_C^{\mathrm{op}} c$. By definition these are $b\le_C a$ and $c\le_C b$. Applying the original transitivity field of \autoref{def:preorder-stability-certificate} to the composable pair $c\le_C b$ and $b\le_C a$ gives $c\le_C a$, which is exactly $a\le_C^{\mathrm{op}} c$.

Relation congruence is inherited with the two endpoints exchanged. Given $a\sim_C a'$, $b\sim_C b'$, and $a\le_C^{\mathrm{op}}b$, unfold the last witness to $b\le_C a$. The original relation-congruence field, applied with endpoint comparisons $b\sim_C b'$ and $a\sim_C a'$, gives
$$
b'\le_C a'.
$$
This is precisely $a'\le_C^{\mathrm{op}} b'$.

The pattern row of \autoref{def:preorder-pattern-specification} remains binary comparison expression formation over carried endpoints: a node $a\le_C^{\mathrm{op}}b$ is read as the original comparison node $b\le_C a$ with endpoint order exchanged. The classifier row of \autoref{def:preorder-classifier-specification} is unchanged because endpoint classification is still $\sim_C$. The ledger row of \autoref{def:preorder-ledger-policy} records the same two carrier witnesses and the same relation derivation, with source and target slots exchanged for the displayed comparison.

These source, pattern, classifier, stability, and ledger rows are exactly the five fields of the paper naming-certificate template in \autoref{def:typed-naming-certificate}, specialized to the preorder obligations of \autoref{def:preorder-certificate-obligations}. Therefore $C^{\mathrm{op}}$ carries a $\PreorderUp$ certificate.
\end{proof}
