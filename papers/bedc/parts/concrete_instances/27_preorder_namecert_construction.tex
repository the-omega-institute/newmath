\chapter{A Concrete Naming Certificate for \(\PreorderUp\)}
\label{ch:concrete-instances-preorder-namecert}

The preorder interface \(\PreorderUp\) is licensed by a carrier with a binary relation \(\le : C \times C \to \mathsf{Prop}\) satisfying reflexivity and transitivity under the carrier's classifier. It is the simplest order-theoretic NameCert and self-contained: no other licensed object is required as a dependency.

\begin{remark}[Preorder certificate scope]
\label{rem:concrete-instances-preorder-namecert-roadmap}
This chapter records the preorder licensing target through a history
carrier, restricted classifier, and reflexive-transitive relation data.
\end{remark}

\section{Carrier definition}

\begin{definition}[Preorder carrier]
\label{def:preorder-carrier}
Use an abstract carrier \(C\), classifier \(\sim_C\), and relation \(\le_C\) with preorder law fields.
\end{definition}

\section{The certificate}

\begin{definition}[Preorder source specification]
\label{def:preorder-source-specification}
The source is the carrier \(C\) with its classifier and relation.
\end{definition}

\begin{definition}[Preorder pattern specification]
\label{def:preorder-pattern-specification}
The pattern is relational judgment formation \(a\le_C b\) over carrier witnesses.
\end{definition}

\begin{theorem}[Prefix from history sameness]
\label{thm:preorder-prefix-le-of-hsame}
If $h$ and $k$ are internally history-same, then $h\le k$ in the concrete unary
history prefix relation.
\end{theorem}
\begin{proof}
Eliminate the $\hsame(h,k)$ witness. The prefix relation is then witnessed by
the empty unary tail together with the right-unit continuation law for $h$.
\end{proof}
\leanchecked{BEDC.Derived.PreorderUp.PreorderPrefixLE\_of\_hsame}

\begin{theorem}[Prefix from empty is unary carrier]
\label{thm:preorder-prefix-empty-left-iff-unary}
For the concrete unary history preorder, $\emp\le h$ holds exactly when $h$ is in the preorder carrier.
\end{theorem}
\begin{proof}
Eliminate the prefix witness from $\emp$ to expose its unary tail and continuation result. The left-unit result identifies the target with the tail, so unary membership transports to $h$. Conversely, use $h$ itself as the unary tail and the continuation left unit.
\end{proof}
\leanchecked{BEDC.Derived.PreorderUp.PreorderPrefixLE\_empty\_left\_iff\_unary}

\begin{theorem}[Prefix to empty identifies the source]
\label{thm:preorder-prefix-empty-target-iff}
For the concrete unary history preorder, $h\le \emp$ holds exactly when
$h\hsame \emp$.
\end{theorem}
\begin{proof}
Eliminate the prefix witness into a unary tail and a continuation into
$\emp$. The empty-result inversion theorem forces the source history to be
$\emp$. Conversely, after the history-sameness proof identifies the source
with $\emp$, use the empty unary tail and the continuation right unit.
\end{proof}
\leanchecked{BEDC.Derived.PreorderUp.PreorderPrefixLE\_empty\_target\_iff}

\begin{theorem}[Preorder prefix results remain in the carrier]
\label{thm:preorder-prefix-result-carrier}
For the unary history preorder prefix relation, if \(h\) is in the preorder
carrier and \(h\le k\) is witnessed by a unary continuation tail, then \(k\)
is also in the preorder carrier.
\end{theorem}
\leanchecked{BEDC.Derived.PreorderUp.PreorderPrefixLE\_result\_carrier}
\begin{proof}
Unpack the prefix witness into a tail \(t\), its unary carrier proof, and a
continuation proof \(\Cont(h,t,k)\). The finite-kernel unary continuation
closure theorem gives unary membership of the result \(k\).
\end{proof}

\begin{theorem}[Prefix relation preserves the carrier]
\label{thm:preorder-prefix-preserves-carrier}
For the concrete unary history carrier, if $h$ is a carrier history and $h$ is related to $k$ by appending a unary prefix tail, then $k$ is also in the carrier.
\end{theorem}
\begin{proof}
Unpack the prefix witness into its unary tail and continuation proof. Unary continuation closure sends the carrier witness for $h$, the unary witness for the tail, and the continuation proof to a carrier witness for $k$.
\end{proof}
\leanchecked{BEDC.Derived.PreorderUp.PreorderPrefixLE\_preserves\_carrier}

\begin{definition}[Preorder classifier specification]
\label{def:preorder-classifier-specification}
The classifier is the supplied carrier sameness, and the relation is stable under classifier replacement.
\end{definition}

\begin{theorem}[Prefix right extension]
\label{thm:preorder-prefix-right-extension}
If \(h\le k\) is witnessed by a unary continuation tail, and \(k\) continues by a unary tail to \(r\), then \(h\le r\).
\end{theorem}
\leanchecked{BEDC.Derived.PreorderUp.PreorderPrefixLE\_right\_extension}
\begin{proof}
Unpack the first prefix witness as a unary tail from \(h\) to \(k\). Compose this continuation with the continuation from \(k\) to \(r\); associativity of continuation supplies the resulting path from \(h\), and unary closure under append supplies the combined unary tail.
\end{proof}

\begin{theorem}[Prefix target remains in the preorder carrier]
\label{thm:preorder-prefix-target-carrier}
For the history preorder carrier, if \(h\) is unary and \(k\) is reached from \(h\) by a unary continuation tail, then \(k\) is also unary.
\end{theorem}
\begin{proof}
Unpack the prefix relation into a unary tail and a continuation witness. Unary closure under continuation sends the source carrier witness and the tail witness to the target history.
\end{proof}
\leanchecked{BEDC.Derived.PreorderUp.PreorderPrefixLE\_target\_carrier}

\begin{theorem}[Prefix antisymmetry collapses to history sameness]
\label{thm:preorder-prefix-le-antisymm-hsame}
If \(h\le k\) and \(k\le h\) are both witnessed by unary continuation tails,
then \(\hsame(h,k)\).
\end{theorem}
\leanchecked{BEDC.Derived.PreorderUp.PreorderPrefixLE\_antisymm\_hsame}
\begin{proof}
Unpack the two prefix witnesses. Substituting the first continuation equation
into the second gives a round trip from \(h\) through the two tails back to
\(h\). Associativity rewrites this as equality between the empty right tail and
the combined tail under the same left prefix, and left cancellation forces the
forward tail to be empty. The first continuation equation then reduces to
\(\hsame(h,k)\).
\end{proof}

\begin{theorem}[Prefix left extension]
\label{thm:preorder-prefix-left-extension}
If \(x\) continues by a unary tail to \(h\), and \(h\le k\) is witnessed by a
unary continuation tail, then \(x\le k\).
\end{theorem}
\leanchecked{BEDC.Derived.PreorderUp.PreorderPrefixLE\_left\_extension}
\begin{proof}
Unpack the prefix witness from \(h\) to \(k\). Append the incoming unary tail to
the prefix tail, use unary closure under append for the combined tail, and use
continuation associativity to identify the resulting continuation from \(x\) to
\(k\).
\end{proof}

\begin{definition}[Preorder stability certificate]
\label{def:preorder-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item classifier reflexivity
  \item classifier transitivity
  \item preorder reflexivity
  \item preorder transitivity
  \item relation congruence in both arguments
\end{enumerate}
\end{definition}

\begin{definition}[Preorder ledger policy]
\label{def:preorder-ledger-policy}
Record the compared carrier witnesses and the relation derivation.
\end{definition}

\begin{definition}[\(\PreorderUp\) certificate obligations]
\label{def:preorder-certificate-obligations}
The carrier, source, pattern, classifier, stability, and ledger data listed above are the obligations required for the interface name \(\PreorderUp\).
\end{definition}

\begin{theorem}[Concrete \(\PreorderUp\) naming certificate]
\label{thm:concrete-preorder-namecert}
Any carrier relation satisfying reflexivity and transitivity forms a valid NameCert for \(\PreorderUp(C)\).
\end{theorem}
