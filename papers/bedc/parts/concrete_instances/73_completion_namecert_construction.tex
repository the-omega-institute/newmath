\chapter{A Concrete Naming Certificate for $\CompletionUp$}
\label{ch:concrete-instances-completion-namecert}

The completion interface $\CompletionUp$ packages the universal regular-Cauchy completion of a metric naming certificate. It depends on $\NameCert_{\MetricSpaceUp}$ (\autoref{ch:concrete-instances-metric-namecert}) and the real-completeness capstone (\autoref{ch:capstones-real-completeness}), generalising the $\RatUp \to \RealUp$ construction to arbitrary located metric carriers.

\section{Carrier definition}
\label{sec:completion-carrier}

\begin{theorem}[Completion metric distance visible-tail readback]\label{thm:completion-metric-distance-e1-tail-real-prefix-readback}
Let $x$ and $y$ be real stream representatives and let $n$ be a prefix index. If the
real stream prefix classifier holds at $n$ and the metric distance witness between
$x_n$ and $y_n$ has visible result $\Eone(t)$, then the visible tail $t$ is unary and
the rational endpoint classifier holds between $x_n$ and $y_n$.
\end{theorem}
\begin{proof}
The metric distance witness contains a unary witness for its visible result; inversion
of unary histories through $\Eone$ gives unary membership of $t$. The real stream
prefix classifier supplies the rational endpoint classifier at the selected index.
\end{proof}
\leanchecked{BEDC.Derived.CompletionUp.CompletionMetricDistanceWitness\_e1\_tail\_real\_prefix\_readback}

\begin{theorem}[Completion empty-distance real-prefix readback]
\label{thm:completion-empty-distance-real-prefix-readback}
Let $x$ and $y$ be real stream representatives and let $n$ be a prefix index. If the
real stream prefix classifier holds at $n$ and the metric distance witness between
$x_n$ and $y_n$ has empty result, then both selected endpoints classify as $\emp$
and the rational endpoint classifier holds between $x_n$ and $y_n$.
\end{theorem}
\begin{proof}
The empty-distance exactness theorem for metric distance witnesses reads back
the two endpoint classifiers against $\emp$. The real stream prefix classifier
supplies the rational endpoint classifier at the selected index.
\end{proof}
\leanchecked{BEDC.Derived.CompletionUp.CompletionMetricDistanceWitness\_empty\_distance\_real\_prefix\_readback}

\begin{theorem}[Completion visible-tail excludes empty distance]
\label{thm:completion-metric-distance-e1-tail-empty-distance-absurd}
Let $x$ and $y$ be real stream representatives and let $n$ be a prefix index.
If the real stream prefix classifier holds at $n$ and the metric distance
witness between $x_n$ and $y_n$ has visible result $\Eone(t)$, then the
rational endpoint classifier holds between $x_n$ and $y_n$, and no empty
metric distance witness can classify the same endpoints.
\end{theorem}
\begin{proof}
Visible-tail readback supplies the rational endpoint classifier. If an empty
distance witness also classified the same endpoints, metric result
determinacy would identify $\Eone(t)$ with $\emp$, contradicting the BEDC
no-confusion theorem for visible constructors.
\end{proof}
\leanchecked{BEDC.Derived.CompletionUp.CompletionMetricDistanceWitness\_e1\_tail\_empty\_distance\_absurd}

\begin{theorem}[Completion visible-tail transport]
\label{thm:completion-metric-distance-witness-e1-tail-hsame-transport}
Let $x,x',y,y'$ be real stream representatives and let $n$ be a prefix
index. If every endpoint of $x$ is $\hsame$-equal to the corresponding
endpoint of $x'$ and every endpoint of $y$ is $\hsame$-equal to the
corresponding endpoint of $y'$, then a real prefix classifier and a metric
distance witness from $x_n$ to $y_n$ with visible result $\Eone(t)$ transport
to the primed streams. The transported witness still reads back unary
membership of $t$ and the rational endpoint classifier between $x'_n$ and
$y'_n$.
\end{theorem}
\begin{proof}
Transport the finite real prefix classifier endpointwise. Transport the metric
distance continuation ledger along the endpoint equalities while keeping the
visible result fixed, and transport the unary endpoint carriers contained in
the witness. The visible-tail readback theorem applied to the transported
prefix and witness gives the unary tail and rational endpoint classifier.
\end{proof}
\leanchecked{BEDC.Derived.CompletionUp.CompletionMetricDistanceWitness\_e1\_tail\_hsame\_transport}

\begin{theorem}[Completion metric distance visible-tail determinacy]
\label{thm:completion-metric-distance-witness-e1-result-tail-deterministic}
Let $x$ and $y$ be real stream representatives and let $n$ be a prefix index. If the
real stream prefix classifier holds at $n$ and two metric distance witnesses between
$x_n$ and $y_n$ have visible results $\Eone(d)$ and $\Eone(e)$, then $d$ and $e$ are
unary, their tails are $\hsame$-equal, and the rational endpoint classifier holds
between $x_n$ and $y_n$.
\end{theorem}
\begin{proof}
Apply the visible-tail readback to each distance witness to obtain unary evidence
for both tails and the rational endpoint classifier. The metric distance result
determinacy theorem compares the two visible results over the same endpoints; e1
inversion for $\hsame$ descends this comparison to $d\hsame e$.
\end{proof}
\leanchecked{BEDC.Derived.CompletionUp.CompletionMetricDistanceWitness\_e1\_result\_tail\_deterministic}

\begin{theorem}[Completion symmetric visible-tail readback]
\label{thm:completion-metric-distance-witness-e1-symmetric-tail-real-prefix-readback}
Let $x$ and $y$ be real stream representatives and let $n$ be a prefix index. If the
real stream prefix classifier holds at $n$, a metric distance witness from $x_n$ to
$y_n$ has visible result $\Eone(d)$, and a metric distance witness from $y_n$ to
$x_n$ has visible result $\Eone(e)$, then $d$ and $e$ are unary, $d\hsame e$, and the
rational endpoint classifier holds between $y_n$ and $x_n$.
\end{theorem}
\begin{proof}
Symmetry of the real stream prefix classifier gives the rational endpoint classifier
in the reversed direction. Visible-tail readback applies to both witnesses. The
symmetric metric distance classifier compares the two visible result histories, and
e1 inversion for $\hsame$ descends the comparison to the tails.
\end{proof}
\leanchecked{BEDC.Derived.CompletionUp.CompletionMetricDistanceWitness\_e1\_symmetric\_tail\_real\_prefix\_readback}

\begin{theorem}[Completion metric distance visible-context tail readback]
\label{thm:completion-metric-distance-visible-context-e1-tail-real-prefix-readback}

context displays the metric distance witness with central result $\Eone(t)$,
then $t$ is unary and the rational endpoint classifier holds between
$x_n$ and $y_n$.
\end{theorem}
\begin{proof}
Apply visible-context cancellation for metric distance witnesses to recover
the central witness. Its $\Eone(t)$ distance carrier inverts to unary
membership of $t$, and the real stream prefix classifier supplies the
rational endpoint classifier at $n$.
\end{proof}
\leanchecked{BEDC.Derived.CompletionUp.CompletionMetricDistanceWitness\_visible\_context\_e1\_tail\_real\_prefix\_readback}

\begin{theorem}[Completion visible-context tail transport]
\label{thm:completion-metric-distance-visible-context-e1-tail-hsame-transport}
Let $x,x'$ and $y,y'$ be pointwise history-same real stream representatives.
If the real stream prefix classifier holds for $x,y$ at $n$ and a visible
context displays a metric distance witness with central result $\Eone(t)$,
then the transported representatives $x',y'$ have the same prefix classifier,
the same visible-context metric witness, unary tail evidence for $t$, and the
rational endpoint classifier at $n$.
\end{theorem}
\begin{proof}
Cancel the visible prefix and suffix to recover the central metric witness.
Pointwise $\hsame$ transports the real prefix classifier and the central
continuation ledger. Rebuild the visible-context witness from the transported
central witness, then apply visible-context tail readback to recover the unary
tail and the transported rational endpoint classifier.
\end{proof}
\leanchecked{BEDC.Derived.CompletionUp.CompletionMetricDistanceWitness\_visible\_context\_e1\_tail\_hsame\_transport}

\begin{theorem}[Completion visible-context empty-distance transport]
\label{thm:completion-visible-context-empty-distance-hsame-transport}
Let $x,x',y,y'$ be real stream representatives such that
$x_m\hsame x'_m$ and $y_m\hsame y'_m$ for every index $m$. If the real stream
prefix classifier holds for $x,y$ at $n$ and a visible-context metric distance
witness from $x_n$ to $y_n$ has central empty distance, then the transported
prefix classifier holds for $x',y'$ at $n$, the corresponding visible-context
empty-distance witness holds for $x'_n,y'_n$, the visible prefix and suffix are
unary, both transported central endpoints classify as $\emp$, and the rational
endpoint classifier holds between $x'_n$ and $y'_n$.
\end{theorem}
\begin{proof}
Apply empty-distance visible-context exactness to the original witness. This
recovers unary evidence for the visible prefix and suffix and empty endpoint
classifiers for $x_n$ and $y_n$. Transport the real stream prefix classifier
along the pointwise $\hsame$ hypotheses, transport the two endpoint empty
classifiers by symmetry and transitivity of $\hsame$, and rebuild the
visible-context empty-distance witness from the transported data. The
transported real prefix classifier supplies the rational endpoint classifier at
the selected index.
\end{proof}
\leanchecked{BEDC.Derived.CompletionUp.CompletionMetricDistanceWitness\_visible\_context\_empty\_distance\_hsame\_transport}

\begin{theorem}[Concrete CompletionUp naming certificate]\label{thm:concrete-completion-namecert}
The $\CompletionUp$ chapter carries a concrete naming-certificate boundary for the universal
regular-Cauchy completion of a metric naming certificate. The certificate has the following
ten fields.
\begin{enumerate}
\item The traditional target is the universal regular-Cauchy completion associated to
$\NameCert_{\MetricSpaceUp}$.
\item The BEDC source is $\mathsf{BHist}$ equipped with $\mathsf{RealStreamPrefixClassifier}$,
$\mathsf{MetricDistanceWitness}$, and the supplied real-completeness capstone data.
\item The carrier consists of completion candidates, namely regular Cauchy point-name
streams modulo the pointwise $\hsame$ transport already used by the readback rows.
\item The classifier reads pointwise metric-distance tolerance through the real-prefix
classifier at a selected prefix index.
\item Stability is supplied by the checked visible-tail, empty-distance, transport,
symmetry, determinacy, and prefix-independent readback theorems listed in this chapter.
\item The ledger records the selected prefix index, the visible left and right contexts,
the metric-distance witness, the Cauchy modulus, and the limit candidate.
\item The core theorem is this certificate boundary, which repacks the carrier,
classifier, stability, ledger, and exactness clauses for $\CompletionUp$.
\item Exactness is finite-prefix distance readback together with limit uniqueness under
the classifier.
\item The standard bridge slot is carried by the conservative completion bridge, so this
certificate exposes the paper-side boundary consumed by $\CompleteMetricUp$ and
$\CompactMetricUp$ without asserting that bridge here.
\item The certificate status is $\mathsf{ProofTarget}$.
\end{enumerate}
\end{theorem}
\leanchecked{BEDC.Derived.CompletionUp.CompletionMetricDistanceWitness\_semanticNameCert}
\begin{proof}
Endpoint coverage comes from the two direct real-prefix readback rows. Apply
\autoref{thm:completion-metric-distance-e1-tail-real-prefix-readback} for visible
$\Eone$ distance results, equivalently the checked
\texttt{CompletionMetricDistanceWitness\_e1\_tail\_real\_prefix\_readback} row, and
apply \autoref{thm:completion-empty-distance-real-prefix-readback} for empty results,
equivalently the checked
\texttt{CompletionMetricDistanceWitness\_empty\_distance\_real\_prefix\_readback} row.
These two cases cover the finite-prefix distance readback used by the classifier.

Carrier and classifier transport are the two $\hsame$ transport rows. Use
\autoref{thm:completion-metric-distance-witness-e1-tail-hsame-transport} for ordinary
visible tails and \autoref{thm:completion-metric-distance-visible-context-e1-tail-hsame-transport}
for contextualized tails. The empty contextual case is carried by
\autoref{thm:completion-visible-context-empty-distance-hsame-transport}, so the selected
prefix index and its visible contexts can be moved along pointwise representative equality.

Symmetry and uniqueness are read from the symmetric and deterministic rows. Apply
\autoref{thm:completion-metric-distance-witness-e1-result-tail-deterministic},
\autoref{thm:completion-metric-distance-witness-e1-symmetric-tail-real-prefix-readback},
\autoref{thm:completion-visible-context-symmetric-tail-readback}, and
\autoref{thm:completion-metric-distance-visible-context-e1-tail-deterministic} to compare
all visible distance tails attached to the same finite-prefix classifier data.

Context-insensitive readback is supplied by the prefix-independent visible-context rows.
Apply \autoref{thm:completion-metric-distance-visible-context-e1-prefix-independent-tail-deterministic}
and \autoref{thm:completion-metric-distance-visible-context-e1-prefix-independent-symmetric-tail-readback}
to remove dependence on the chosen visible left and right contexts. The Cauchy modulus and
limit candidate are then ledger fields, while limit uniqueness is the exactness condition
read through the same classifier.

This is not a one-theorem variant of any checked row: its conclusion packages the source,
carrier, classifier, stability obligations, ledger, exactness predicate, and downstream
bridge slots into one $\CompletionUp$ certificate boundary.
\end{proof}

\begin{theorem}[Completion carrier obligation]\label{thm:completion-carrier-obligation}
The $\CompletionUp$ carrier obligation is the BEDC finite-prefix regular-Cauchy point-name packet over $\mathsf{BHist}$, $\mathsf{RealStreamPrefixClassifier}$, $\mathsf{MetricDistanceWitness}$, Cauchy modulus data, visible-context transport, and $\mathsf{Pkg}$ provenance. Its carrier row consists only of displayed point-name streams, selected prefix indices, finite distance-readback rows, visible left and right contexts, modulus rows, limit-candidate ledger rows, and package provenance.

The obligation contains no quotient real, no host real equality, no choice-selected limit, and no representative selected outside the displayed finite-prefix packet.
\end{theorem}
\begin{proof}
Read the carrier and ledger fields of \autoref{thm:concrete-completion-namecert}. The source is $\mathsf{BHist}$ equipped with the real-prefix classifier, metric-distance witness rows, Cauchy modulus data, visible contexts, and package provenance. These are exactly the finite coordinates named in the carrier statement.

The readback rows are supplied by \autoref{thm:completion-metric-distance-e1-tail-real-prefix-readback} and \autoref{thm:completion-empty-distance-real-prefix-readback}. The visible-context coordinates are transported by \autoref{thm:completion-metric-distance-witness-e1-tail-hsame-transport}, \autoref{thm:completion-metric-distance-visible-context-e1-tail-hsame-transport}, and \autoref{thm:completion-visible-context-empty-distance-hsame-transport}.

The deterministic and prefix-independent rows of \autoref{thm:concrete-completion-namecert} show that the carrier is exhausted by those finite-prefix coordinates. Since no cited row introduces a quotient type, host real equality, or choice-selected limit, none is part of the carrier obligation.
\end{proof}

\begin{theorem}[Completion classifier stability obligation]\label{thm:completion-classifier-stability-obligation}
The $\CompletionUp$ classifier is stable under displayed $\hsame$ and $\Cont$ transport of the selected prefix, visible left and right contexts, metric-distance witness, Cauchy modulus, and limit-candidate ledger. After transport, the classifier still reads only the transported real-prefix distance result, the transported visible-context rows, the transported modulus row, the transported limit-candidate row, and the transported $\mathsf{Pkg}$ provenance.

The stability obligation introduces no quotient real equality, no host real equality, no completion point outside the packet, and no choice-selected representative.
\end{theorem}
\begin{proof}
Begin with the carrier rows named in \autoref{thm:completion-carrier-obligation}. A displayed $\hsame$ route moves the selected prefix and the visible contexts, while the displayed $\Cont$ routes preserve the finite readback boundary consumed by the classifier.

Apply \autoref{thm:completion-metric-distance-witness-e1-tail-hsame-transport} to the ordinary visible tail and \autoref{thm:completion-metric-distance-visible-context-e1-tail-hsame-transport} to the contextualized visible tail. For the empty contextual distance row, apply \autoref{thm:completion-visible-context-empty-distance-hsame-transport}.

The transported classifier remains deterministic by \autoref{thm:completion-metric-distance-witness-e1-result-tail-deterministic} and \autoref{thm:completion-metric-distance-visible-context-e1-tail-deterministic}. Prefix-independent visible-context readback is supplied by \autoref{thm:completion-metric-distance-visible-context-e1-prefix-independent-tail-deterministic} and \autoref{thm:completion-metric-distance-visible-context-e1-prefix-independent-symmetric-tail-readback}.

Thus the transported classifier has the same finite-prefix inventory as the original packet. The proof never passes through a quotient type, host real equality, an external completion point, or a choice-selected representative.
\end{proof}

\begin{theorem}[Completion ledger exactness obligation]\label{thm:completion-ledger-exactness-obligation}
Every public $\CompletionUp$ consumer read factors through the selected prefix index, real-prefix readback, metric-distance witness, visible-context transport, Cauchy modulus, limit-candidate row, and $\mathsf{Pkg}$ provenance displayed in the $\CompletionUp$ packet. The exactness ledger has no further public coordinate.

In particular, a consumer cannot read a quotient real, host real equality, choice-selected limit, or representative outside the finite-prefix regular-Cauchy point-name packet.
\end{theorem}
\begin{proof}
Project a public consumer read through the ten-field boundary of \autoref{thm:concrete-completion-namecert}. The carrier field supplies the selected prefix and point-name packet, while the classifier field supplies the real-prefix distance readback through $\mathsf{MetricDistanceWitness}$.

The visible-context rows are exactly the rows transported by \autoref{thm:completion-metric-distance-visible-context-e1-tail-hsame-transport} and \autoref{thm:completion-visible-context-empty-distance-hsame-transport}. The ordinary and empty readback cases are exactly \autoref{thm:completion-metric-distance-e1-tail-real-prefix-readback} and \autoref{thm:completion-empty-distance-real-prefix-readback}.

Use the determinacy and prefix-independent rows cited in \autoref{thm:concrete-completion-namecert} to rule out any second public read path for the same displayed prefix data. The Cauchy modulus, limit-candidate row, and package provenance are ledger coordinates, so they are readable only as the displayed rows of that packet.

Therefore every public read factors through the listed finite-prefix coordinates. No step creates a quotient real, host real equality, choice-selected limit, or hidden representative.
\end{proof}
