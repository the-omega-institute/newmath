\chapter{A Concrete Naming Certificate for $\CompletionUp$}
\label{ch:concrete-instances-completion-namecert}

The completion interface $\CompletionUp$ packages the universal regular-Cauchy completion of a metric naming certificate. It depends on $\NameCert_{\MetricSpaceUp}$ (\autoref{ch:concrete-instances-metric-namecert}) and the real-completeness capstone (\autoref{ch:capstones-real-completeness}), generalising the $\RatUp \to \RealUp$ construction to arbitrary located metric carriers.

\section{Carrier definition}
\label{sec:completion-carrier}

\begin{theorem}[Completion metric distance visible-tail readback]\label{thm:completion-metric-distance-e1-tail-real-prefix-readback}
Let $x$ and $y$ be real stream representatives and let $n$ be a prefix index. If the
real stream prefix classifier holds at $n$ and the metric distance witness between
$x_n$ and $y_n$ has visible result $\Eone(t)$, then the visible tail $t$ is unary and
the rational endpoint classifier holds between $x_n$ and $y_n$.
\end{theorem}
\begin{proof}
The metric distance witness contains a unary witness for its visible result; inversion
of unary histories through $\Eone$ gives unary membership of $t$. The real stream
prefix classifier supplies the rational endpoint classifier at the selected index.
\end{proof}
\leanchecked{BEDC.Derived.CompletionUp.CompletionMetricDistanceWitness\_e1\_tail\_real\_prefix\_readback}

\begin{theorem}[Completion metric distance visible-tail determinacy]
\label{thm:completion-metric-distance-witness-e1-result-tail-deterministic}
Let $x$ and $y$ be real stream representatives and let $n$ be a prefix index. If the
real stream prefix classifier holds at $n$ and two metric distance witnesses between
$x_n$ and $y_n$ have visible results $\Eone(d)$ and $\Eone(e)$, then $d$ and $e$ are
unary, their tails are $\hsame$-equal, and the rational endpoint classifier holds
between $x_n$ and $y_n$.
\end{theorem}
\begin{proof}
Apply the visible-tail readback to each distance witness to obtain unary evidence
for both tails and the rational endpoint classifier. The metric distance result
determinacy theorem compares the two visible results over the same endpoints; e1
inversion for $\hsame$ descends this comparison to $d\hsame e$.
\end{proof}
\leanchecked{BEDC.Derived.CompletionUp.CompletionMetricDistanceWitness\_e1\_result\_tail\_deterministic}

\begin{theorem}[Completion symmetric visible-tail readback]
\label{thm:completion-metric-distance-witness-e1-symmetric-tail-real-prefix-readback}
Let $x$ and $y$ be real stream representatives and let $n$ be a prefix index. If the
real stream prefix classifier holds at $n$, a metric distance witness from $x_n$ to
$y_n$ has visible result $\Eone(d)$, and a metric distance witness from $y_n$ to
$x_n$ has visible result $\Eone(e)$, then $d$ and $e$ are unary, $d\hsame e$, and the
rational endpoint classifier holds between $y_n$ and $x_n$.
\end{theorem}
\begin{proof}
Symmetry of the real stream prefix classifier gives the rational endpoint classifier
in the reversed direction. Visible-tail readback applies to both witnesses. The
symmetric metric distance classifier compares the two visible result histories, and
e1 inversion for $\hsame$ descends the comparison to the tails.
\end{proof}
\leanchecked{BEDC.Derived.CompletionUp.CompletionMetricDistanceWitness\_e1\_symmetric\_tail\_real\_prefix\_readback}

\begin{theorem}[Completion metric distance visible-context tail readback]
\label{thm:completion-metric-distance-visible-context-e1-tail-real-prefix-readback}
Let $x$ and $y$ be real stream representatives and let $n$ be a prefix
index. If the real stream prefix classifier holds at $n$ and a visible
context displays the metric distance witness with central result $\Eone(t)$,
then $t$ is unary and the rational endpoint classifier holds between
$x_n$ and $y_n$.
\end{theorem}
\begin{proof}
Apply visible-context cancellation for metric distance witnesses to recover
the central witness. Its $\Eone(t)$ distance carrier inverts to unary
membership of $t$, and the real stream prefix classifier supplies the
rational endpoint classifier at $n$.
\end{proof}
\leanchecked{BEDC.Derived.CompletionUp.CompletionMetricDistanceWitness\_visible\_context\_e1\_tail\_real\_prefix\_readback}

\begin{theorem}[Completion visible-context symmetric tail readback]
\label{thm:completion-visible-context-symmetric-tail-readback}
Let $x$ and $y$ be real stream representatives and let $n$ be a prefix
index. If the real stream prefix classifier holds at $n$, a visible-context
metric distance witness from $x_n$ to $y_n$ has central result $\Eone(d)$,
and a visible-context metric distance witness from $y_n$ to $x_n$ has central
result $\Eone(e)$, then $d$ and $e$ are unary, $d\hsame e$, and the rational
endpoint classifier holds between $y_n$ and $x_n$.
\end{theorem}
\begin{proof}
Apply visible-context exactness to both displayed distance witnesses. This
recovers the two central metric distance witnesses with visible results
$\Eone(d)$ and $\Eone(e)$. The symmetric visible-tail readback theorem then
extracts both unary tails, compares them by $\hsame$, and reads back the
reversed rational endpoint classifier.
\end{proof}
\leanchecked{BEDC.Derived.CompletionUp.CompletionMetricDistanceWitness\_visible\_context\_e1\_symmetric\_tail\_real\_prefix\_readback}

\begin{theorem}[Completion metric distance visible-context tail determinacy]
\label{thm:completion-metric-distance-visible-context-e1-tail-deterministic}
Let $x$ and $y$ be real stream representatives and let $n$ be a prefix

endpoint classifier holds between $x_n$ and $y_n$.
\end{theorem}
\begin{proof}

\end{proof}
