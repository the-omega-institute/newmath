\chapter{A Concrete Naming Certificate for $\ContactUp$}
\label{ch:concrete-instances-contact-namecert}

The contact-structure interface $\ContactUp$ packages a maximally nonintegrable hyperplane field on an odd-dimensional manifold carrier, recorded as a $1$-form whose top-degree wedge is nondegenerate. It depends on $\NameCert_{\DiffFormUp}$ (\autoref{ch:concrete-instances-diffform-namecert}) and $\NameCert_{\ManifoldUp}$ (\autoref{ch:concrete-instances-manifold-namecert}).

\section{Carrier definition}
\label{sec:contact-carrier}

\begin{definition}[ContactUp carrier-classifier surface]
\label{def:contactup-carrier-classifier-surface}
Fix a $\NameCert_{\ManifoldUp}$ source $M$ and a $\NameCert_{\DiffFormUp}$ source $\Omega$. A $\ContactUp$ carrier-classifier surface over $BHist$ is the displayed BEDC package
$$
\mathsf{ContactCarrierClassifier}_{M,\Omega}.
$$
Its source rows are the $BHist$ carrier rows inherited from $M$ and $\Omega$, the odd-dimensional manifold witness on the shared manifold endpoint, a carried $\DiffFormUp$ degree-one form row $\alpha$, a contact hyperplane row $\ker(\alpha)$ on the same manifold carrier, an exterior-derivative input row $d\alpha$, the wedge ladder rows for $\alpha\wedge(d\alpha)^k$, and the top wedge row $\alpha\wedge(d\alpha)^n$ when the displayed dimension row is $2n+1$.

Its classifier is the restriction of $\hsame$ to rows that preserve the shared $\ManifoldUp$ carrier, the $\DiffFormUp$ one-form row, the hyperplane row, the exterior-derivative input row, and every displayed wedge-ladder row. The surface contains no nonintegrability primitive and no standard bridge row.
\end{definition}
\leandef{BEDC.Derived.ContactUp.ContactCarrierClassifierSurface}

\begin{theorem}[ContactUp form-row obligation surface]
\label{thm:contactup-form-row-obligation}
Every $\ContactUp$ carrier-classifier surface of \autoref{def:contactup-carrier-classifier-surface} exposes exactly the $\ManifoldUp$ source, the $\DiffFormUp$ contact consumer face, the degree-one form row $\alpha$, the exterior-derivative input row $d\alpha$, and the wedge ladder rows $\alpha\wedge(d\alpha)^k$ routed by the $\DiffFormUp$ root threshold. It exposes no nilpotence row, no symplectic closedness row, no contact nonintegrability witness, and no standard bridge as part of the $\DiffFormUp$ root face.
\end{theorem}
\leanchecked{BEDC.Derived.ContactUp.ContactCarrierClassifierSurface\_form\_row\_obligation}
\begin{proof}
Unpack \autoref{def:contactup-carrier-classifier-surface}. The source fields are the shared $\ManifoldUp$ carrier and the $\DiffFormUp$ carried rows for the one-form, exterior-input, and wedge ladder.

Apply \autoref{thm:diffform-root-consumer-face-coverage} and \autoref{thm:diffform-root-consumer-face-disjointness}. They place the contact reads in the $\ContactUp$ consumer face and separate them from the de Rham, symplectic, and shared root laws.

Finally apply \autoref{thm:diffform-root-export-wedge-threshold}. The threshold exports carried forms, classifier data, wedge ledgers, and exterior-input routing, so the $\ContactUp$ surface receives exactly the form row, derivative input row, and wedge ladder rows stated above.
\end{proof}

\section{The certificate}
\label{sec:contact-certificate}

\begin{theorem}[ContactUp nondegeneracy obligation]
\label{thm:contactup-nondegeneracy-obligation}
For a $\ContactUp$ carrier-classifier surface over a displayed dimension row $2n+1$, top-degree wedge nondegeneracy is a $\ContactUp$-local obligation transported by the $\ContactUp$ classifier:
$$
\mathsf{Nondegenerate}_{M}(\alpha\wedge(d\alpha)^n).
$$
The $\DiffFormUp$ root supplies the carried form row, exterior-derivative input row, and wedge routing needed to form the top row, but it does not supply this nondegeneracy predicate as a primitive.
\end{theorem}
\leanchecked{BEDC.Derived.ContactUp.ContactCarrierClassifierSurface\_top\_wedge\_transport}
\leanchecked{BEDC.Derived.ContactUp.ContactCarrierClassifierSurface\_nondegeneracy\_obligation}
\begin{proof}
Begin with the top wedge row named in \autoref{def:contactup-carrier-classifier-surface}. It is assembled from the degree-one row, the exterior-derivative input row, and the wedge ladder exported by \autoref{thm:diffform-root-export-wedge-threshold}.

Now use the classifier clause of the same definition. It transports the shared manifold carrier, the one-form row, the derivative input row, and the wedge ladder rows, so the nondegeneracy predicate is stated on the transported $\ContactUp$ surface.

The exclusion clause follows from \autoref{thm:diffform-root-consumer-face-disjointness}: $\DiffFormUp$ exports routing rows to the contact consumer face, while contact nonintegrability remains outside the shared root primitive surface.
\end{proof}
\leanvariant{BEDC.Derived.ContactUp.ContactCarrierClassifierSurface\_nondegeneracy\_obligation}

\begin{theorem}[ContactUp ledger exactness obligation]
\label{thm:contactup-ledger-exactness-obligation}
The $\ContactUp$ ledger and exactness obligation records exactly the hidden manifold chart rows supplied by the $\ManifoldUp$ source, the one-form source row $\alpha$, the exterior-derivative input row $d\alpha$, the wedge ladder rows $\alpha\wedge(d\alpha)^k$, the top wedge row $\alpha\wedge(d\alpha)^n$, the nonintegrability witness of \autoref{thm:contactup-nondegeneracy-obligation}, and a coverage, soundness, and completeness predicate for the carrier-classifier surface of \autoref{def:contactup-carrier-classifier-surface}.

The obligation is exact only when every certified contact read factors through that carrier-classifier surface and every listed ledger row is classified by the restricted $\hsame$ transport named there.
\end{theorem}
\leanchecked{BEDC.Derived.ContactUp.ContactUpLedgerExactness\_obligation}
\begin{proof}
Unpack \autoref{def:contactup-carrier-classifier-surface}. Its source and classifier clauses list the $\ManifoldUp$ chart carrier, the $\DiffFormUp$ one-form row, the exterior-input row, the wedge ladder, the top wedge row, and the hyperplane row.

Apply \autoref{thm:contactup-form-row-obligation} to identify the contact consumer face and the routed form rows, then apply \autoref{thm:contactup-nondegeneracy-obligation} for the local nonintegrability witness.

The coverage predicate says every certified contact read lands in this displayed package. Soundness says each such read is one of the listed rows. Completeness says each listed row is available to the carrier-classifier. These three clauses are precisely the ledger exactness obligation.
\end{proof}

\begin{theorem}[ContactUp public NameCert export]
\label{thm:contactup-public-namecert-export}
The carrier-classifier surface of
\autoref{def:contactup-carrier-classifier-surface}, the form-row obligation of
\autoref{thm:contactup-form-row-obligation}, the nondegeneracy obligation of
\autoref{thm:contactup-nondegeneracy-obligation}, and the ledger exactness
obligation of \autoref{thm:contactup-ledger-exactness-obligation} assemble the
public $\NameCert_{\ContactUp}$ surface over $\mathsf{BHist}$, the
$\ManifoldUp$ carrier, the $\DiffFormUp$ one-form row, the exterior-input row,
the wedge ladder, the top wedge, the restricted $\hsame$ classifier, and the
exact ledger.

This public surface exports the carried contact rows and their exact ledger. It
does not export a separate nonintegrability primitive and does not include a
standard bridge.
\end{theorem}
\leanchecked{BEDC.Derived.ContactUp.ContactCarrierClassifierSurface\_public\_namecert\_export}
\begin{proof}
Project the carrier field from
\autoref{def:contactup-carrier-classifier-surface}. The carrier consists of
the shared $\ManifoldUp$ endpoint, the $\DiffFormUp$ degree-one form row, the
exterior-input row, the wedge ladder, and the top wedge row over
$\mathsf{BHist}$.

Project the classifier field from the same definition. It is the restriction
of $\hsame$ to transports preserving the manifold carrier, one-form row,
hyperplane row, exterior-input row, and wedge-ladder rows.

Project the routed form face from
\autoref{thm:contactup-form-row-obligation} and the local top-wedge condition
from \autoref{thm:contactup-nondegeneracy-obligation}. These two rows provide
the form and nondegeneracy fields without adding a nonintegrability primitive.

Finally repack the coverage, soundness, completeness, and row list of
\autoref{thm:contactup-ledger-exactness-obligation} as the exact ledger field.
Because every projected row is already part of the displayed
carrier-classifier surface, the public certificate contains no standard bridge
row.
\end{proof}

\begin{definition}[ContactUp standard bridge surface]
\label{def:contact-standard-bridge-surface}
The $\ContactUp$ standard bridge surface is the public $\ContactUp$
carrier-classifier rows, the form-row obligation, the nondegeneracy row, the
ledger exactness row, and the public $\NameCert$ export, read as the source
side of the standard contact-structure interpretation. Its source side is
exactly the surface displayed by
\autoref{thm:contactup-public-namecert-export}.
\end{definition}

\begin{theorem}[ContactUp public bridge compatibility]
\label{thm:contact-public-bridge-compatibility}
Every row interpreted by the bridge surface of
\autoref{def:contact-standard-bridge-surface} is already one of the public
$\ContactUp$ rows, and every public $\ContactUp$ row has a corresponding
bridge-side row. The bridge adds no host manifold, no open-cover compactness
condition, and no external equality beyond the public carrier, classifier,
obligation, ledger, and $\NameCert$ rows.
\end{theorem}
\leanchecked{BEDC.Derived.ContactUp.ContactCarrierClassifierSurface\_public\_bridge\_compatibility}
\begin{proof}
Use \autoref{thm:contactup-form-row-obligation} for the routed one-form,
exterior-input, and wedge-ladder rows, then
\autoref{thm:contactup-nondegeneracy-obligation} for the contact-local
top-wedge row.

Apply \autoref{thm:contactup-ledger-exactness-obligation}. Its coverage,
soundness, and completeness clauses say that the listed bridge reads are
exactly the certified public contact reads.

Finally use \autoref{thm:contactup-public-namecert-export}. It repacks those
rows as the public $\NameCert_{\ContactUp}$ surface, so the correspondence is
row-for-row and introduces no host manifold, compactness, or external equality
datum.
\end{proof}
