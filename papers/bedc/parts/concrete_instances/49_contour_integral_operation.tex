\chapter{Contour Integration as an Operation on \(\HolomorphicUp\)}
\label{ch:concrete-instances-contour-integral}

This chapter licenses contour integration \(\int_\gamma f \, dz\) along a piecewise-rational-linear contour \(\gamma\) inside the holomorphic domain of \(f\). Unlike the naming-certificate chapters that pin down a closed object (a number system, an algebraic carrier, a topological structure), this chapter licenses an \emph{operation}: a constructive procedure that turns a (contour, holomorphic-function) pair into a complex-history value. The chapter sits above \(\HolomorphicUp\) (\autoref{ch:concrete-instances-holomorphic-namecert}) and \(\ContinuousFunctionUp\) (\autoref{ch:concrete-instances-continuous-namecert}). Following the Bishop-style discipline of the rest of the paper, the carrier of contours is restricted to piecewise-linear paths over \(\RatUp\) endpoints with explicit Riemann-sum modulus, and the integral is the unique complex-history value approached by these sums up to an effective rate of convergence. The chapter ends with an \emph{operation certificate} --- an analogue of the naming certificate adapted to operations, with input/output classifier-aware predicate plus operation laws (linearity, concatenation, reversal, reparameterization) in place of the standard 5-field NameCert tuple.

\section{Piecewise-linear contour carrier}

\begin{definition}[Piecewise-linear contour]
\label{def:pl-contour}
A piecewise-linear contour \(\gamma\) is a finite list \([z_0, z_1, \ldots, z_k]\) of complex-history endpoints with \(k \geq 1\), each \(z_i\) classified by \(\ComplexUp\). The closed inductive predicate \(\mathsf{PLContour}(\gamma)\) is generated by:
\begin{enumerate}
  \item \(\mathsf{PLContour}([z_0, z_1])\) for any pair of complex histories (the singleton segment);
  \item if \(\mathsf{PLContour}([z_0, \ldots, z_k])\) and \(z_{k+1}\) is a complex history, then \(\mathsf{PLContour}([z_0, \ldots, z_k, z_{k+1}])\) (extension by one further segment).
\end{enumerate}
The single-segment BEDC carrier records two complex-history endpoints \(z_0,z_1\) and the continuation witness \(\Cont(z_0,z_1,\gamma)\). The total length is the sum of the segment lengths \(|z_{i+1} - z_i|\), each computed via the metric carrier of \(\ComplexUp\) (\autoref{ch:concrete-instances-complex-namecert}).
\end{definition}
\leandef{BEDC.Derived.ContourIntegralUp.PLContourSegment}

\begin{lemma}[Piecewise-linear contour well-formedness]
\label{lem:pl-contour-well-formed}
For every BEDC single-segment \(\mathsf{PLContourSegment}(\gamma)\), there are complex-history endpoints \(z_0,z_1\) with \(\Cont(z_0,z_1,\gamma)\), and the contour history \(\gamma\) is unary.
\end{lemma}
\begin{proof}
Destructure the segment witness. The endpoint carrier fields give \(z_0\) and \(z_1\) as complex histories. Each complex history is unary, and the continuation witness closes unary histories under \(\Cont\), so \(\gamma\) is unary.
\end{proof}
\leanchecked{BEDC.Derived.ContourIntegralUp.PLContourSegment\_unary\_result}

\begin{lemma}[Piecewise-linear segment \(\hsame\)-transport]
\label{lem:pl-contour-segment-hsame-transport}
If \(\mathsf{PLContourSegment}(\gamma)\) and \(\hsame(\gamma,\gamma')\), then \(\mathsf{PLContourSegment}(\gamma')\), and \(\gamma'\) is unary.
\end{lemma}
\begin{proof}
Keep the same endpoints \(z_0,z_1\) and transport the continuation result from \(\gamma\) to \(\gamma'\) along \(\hsame(\gamma,\gamma')\). Unary closure for \(\gamma'\) follows from the same endpoint carriers and the transported continuation.
\end{proof}
\leanchecked{BEDC.Derived.ContourIntegralUp.PLContourSegment\_hsame\_transport}

\begin{lemma}[Concatenation of piecewise-linear contours]
\label{lem:pl-contour-concat}
If \(\mathsf{PLContour}([z_0, \ldots, z_j])\) and \(\mathsf{PLContour}([w_0, \ldots, w_k])\) and \(\hsame(z_j, w_0)\), then \(\mathsf{PLContour}([z_0, \ldots, z_j, w_1, \ldots, w_k])\) and the total length of the concatenation equals the sum of the two component lengths.
\end{lemma}
\begin{proof}
By induction on the second contour, using the extension constructor of \(\mathsf{PLContour}\) at each step. Length additivity follows by direct computation.
\end{proof}

\section{Riemann sum approximation}

\begin{definition}[Subdivided Riemann sum]
\label{def:riemann-sum}
For a function \(f\) classified by \(\ContinuousFunctionUp\) on a domain containing \(\gamma\), a piecewise-linear contour \(\gamma = [z_0, \ldots, z_k]\), a unary subdivision count \(n\), and an explicit choice of sampling points \(S_n = \{\xi_{i,j}\}\) on each subsegment, the Riemann sum \(\mathsf{RiemannSum}(f, \gamma, n, S_n)\) is the complex-history value
$$
\sum_{i=0}^{k-1} \sum_{j=0}^{n-1} f(\xi_{i,j}) \cdot \frac{z_{i+1} - z_i}{n}.
$$
The closed inductive carrier indexes both the segment and the subdivision step.
\end{definition}

\begin{lemma}[Riemann sum is a well-defined complex history]
\label{lem:riemann-sum-complex}
For every \(\mathsf{PLContour}(\gamma)\), continuous \(f\) on a domain containing \(\gamma\), unary \(n\), and sampling choice \(S_n\), the value \(\mathsf{RiemannSum}(f, \gamma, n, S_n)\) is classified by \(\ComplexUp\).
\end{lemma}
\begin{proof}
By induction on the length of the contour list. The base case is a single sum of \(n\) products of complex histories, each well-defined by the multiplicative closure of \(\ComplexUp\) (\autoref{ch:concrete-instances-complex-namecert}). The step adds one further segment's contribution to a previously well-defined complex history; closure under finite sums of \(\ComplexUp\) preserves the classification.
\end{proof}

\section{The contour integral}

\begin{definition}[Contour integral predicate]
\label{def:cont-int}
For a function \(f\) classified by \(\HolomorphicUp\) on a neighbourhood containing \(\gamma\), a piecewise-linear contour \(\gamma\), and a complex history \(I\), the predicate \(\mathsf{ContInt}(f, \gamma, I)\) holds when there is an explicit modulus function \(N : \mathsf{Unary} \to \mathsf{Unary}\) such that for every unary precision \(k\) and every \(n\) with \(\mathsf{NatGT}(n, N(k))\) and every sampling choice \(S_n\),
$$
\bigl| \mathsf{RiemannSum}(f, \gamma, n, S_n) - I \bigr| \leq 2^{-k}.
$$
\end{definition}

\begin{lemma}[Existence of the contour integral on holomorphic inputs]
\label{lem:cont-int-exists}
For every \(\mathsf{PLContour}(\gamma)\) and every \(f\) classified by \(\HolomorphicUp\) on an open neighbourhood of (the image of) \(\gamma\), there exists a complex history \(I\) with \(\mathsf{ContInt}(f, \gamma, I)\).
\end{lemma}
\begin{proof}[Sketch]
The holomorphicity classifier of \(f\) on the neighbourhood provides an explicit local modulus of continuity. Combining this with the bounded total length of \(\gamma\) (\autoref{lem:pl-contour-well-formed}) yields a uniform Cauchy modulus on the Riemann-sum sequence indexed by \(n\); its limit \(I\) is constructed as a complex history by the Cauchy completeness of \(\ComplexUp\) (\autoref{ch:concrete-instances-complex-namecert}).
\end{proof}

\begin{theorem}[Uniqueness of the contour integral up to \(\hsame\)]
\label{thm:cont-int-unique}
If \(\mathsf{ContInt}(f, \gamma, I_1)\) and \(\mathsf{ContInt}(f, \gamma, I_2)\), then \(\hsame(I_1, I_2)\).
\end{theorem}
\begin{proof}
For each unary \(k\), the Riemann sum at precision \(\max(N_1(k+1), N_2(k+1))\) is within \(2^{-(k+1)}\) of both \(I_1\) and \(I_2\), so \(|I_1 - I_2| \leq 2^{-k}\) for every \(k\). Hence \(|I_1 - I_2|\) is the zero history, and \(\hsame(I_1, I_2)\) follows from the metric reflection of \(\ComplexUp\).
\end{proof}

\section{Operation laws}

\begin{lemma}[Linearity in the integrand]
\label{lem:cont-int-linear}
If \(\mathsf{ContInt}(f, \gamma, I_f)\) and \(\mathsf{ContInt}(g, \gamma, I_g)\) and \(a, b\) are complex histories, then \(\mathsf{ContInt}(a f + b g, \gamma, a I_f + b I_g)\).
\end{lemma}
\begin{proof}
Riemann sums are linear in the integrand, and the bound on \(|aI_f + bI_g - \mathsf{RiemannSum}(af + bg, \gamma, n, S_n)|\) follows by the triangle inequality from the bounds on the two component Riemann sums.
\end{proof}

\begin{lemma}[Concatenation in the contour]
\label{lem:cont-int-concat}
If \(\gamma\) is the concatenation of \(\gamma_1\) and \(\gamma_2\) (with shared endpoint via \autoref{lem:pl-contour-concat}) and \(\mathsf{ContInt}(f, \gamma_1, I_1)\) and \(\mathsf{ContInt}(f, \gamma_2, I_2)\), then \(\mathsf{ContInt}(f, \gamma, I_1 + I_2)\).
\end{lemma}
\begin{proof}
A Riemann sum on the concatenated contour is the sum of the Riemann sums on the two component contours; pass to the joint modulus \(\max(N_1(k+1), N_2(k+1))\).
\end{proof}

\begin{lemma}[Reversal negates the integral]
\label{lem:cont-int-reversal}
Let \(\gamma^- = [z_k, z_{k-1}, \ldots, z_0]\) be the reverse of \(\gamma = [z_0, \ldots, z_k]\). If \(\mathsf{ContInt}(f, \gamma, I)\), then \(\mathsf{ContInt}(f, \gamma^-, -I)\).
\end{lemma}
\begin{proof}
Each Riemann-sum segment \(f(\xi_{i,j}) \cdot (z_{i+1} - z_i)/n\) flips sign under reversal because \(z_{i+1} - z_i\) is replaced by \(z_i - z_{i+1}\). The modulus is unchanged.
\end{proof}

\begin{lemma}[Reparameterization invariance for piecewise-linear reparameterizations]
\label{lem:cont-int-reparameterization}
If \(\gamma'\) is obtained from \(\gamma\) by inserting one additional intermediate endpoint along an existing segment (a piecewise-linear refinement), then \(\mathsf{ContInt}(f, \gamma, I)\) iff \(\mathsf{ContInt}(f, \gamma', I)\).
\end{lemma}
\begin{proof}
Refinement decomposes a single segment's Riemann contribution into a sum over the two refined sub-segments; this leaves both the limit \(I\) and the modulus unchanged up to an additive constant in \(k\).
\end{proof}

\section{Cauchy's theorem and the integral formula}

\begin{theorem}[Cauchy's theorem on triangulated regions]
\label{thm:cauchy-triangle}
If \(\gamma\) is a closed piecewise-linear contour bounding a triangle inside the holomorphic domain of \(f\), then \(\mathsf{ContInt}(f, \gamma, 0)\), where \(0\) denotes the zero complex history.
\end{theorem}
\begin{proof}[Sketch]
The standard Goursat-style nested-triangle subdivision argument: bisect the triangle into four congruent sub-triangles, observe that the sum of the four sub-integrals equals the original integral (cancellations along shared edges by \autoref{lem:cont-int-reversal}), pick the largest sub-integral and iterate. Holomorphicity provides the bound that the integral over the \(n\)-th nested triangle decays like \(4^{-n} \cdot \text{(perimeter)}^2\), forcing the limit to be zero.
\end{proof}

\begin{lemma}[Cauchy on triangulated polygons]
\label{lem:cauchy-polygon}
For any closed piecewise-linear \(\gamma\) bounding a region triangulated by finitely many triangles inside the holomorphic domain of \(f\), \(\mathsf{ContInt}(f, \gamma, 0)\).
\end{lemma}
\begin{proof}
Apply \autoref{thm:cauchy-triangle} to each triangle in the triangulation. The sum of the triangle integrals equals the integral over the boundary of the polygon by \autoref{lem:cont-int-concat} and \autoref{lem:cont-int-reversal}; the interior edges cancel pairwise.
\end{proof}

\begin{theorem}[Cauchy integral formula, constructive form]
\label{thm:cauchy-integral-formula}
Let \(z_0\) be a complex history, \(r\) a positive rational history, and \(\gamma\) a closed piecewise-linear contour approximating the circle of radius \(r\) about \(z_0\) inside the open disk \(\mathsf{OpenDisk}(z_0, r)\). For \(f\) classified by \(\HolomorphicUp\) on the disk,
$$
\mathsf{ContInt}\bigl(\lambda z. f(z) / (z - z_0), \gamma, 2\pi i \cdot f(z_0)\bigr)
$$
holds, with explicit modulus computed from the rational approximation of \(\pi\) provided by \(\RealUp\) (\autoref{ch:concrete-instances-real-namecert}).
\end{theorem}
\begin{proof}[Sketch]
By Cauchy's theorem on triangulated regions, the integral over \(\gamma\) equals the integral over a small piecewise-linear circle approximation around \(z_0\) (the difference contour bounds a triangulated annulus). On the small approximation, the integrand is approximately \(f(z_0)/(z - z_0)\), whose contour integral is \(2\pi i \cdot f(z_0)\) by direct computation on the rational endpoints. The explicit modulus combines the holomorphicity modulus with the rational-\(\pi\) approximation modulus.
\end{proof}

\section{Operation laws and ledger}

\begin{definition}[Contour-integral input specification]
\label{def:cont-int-input-spec}
The input specification of the contour-integral operation is the pair \((f, \gamma)\) where \(f\) is classified by \(\HolomorphicUp\) on an open neighbourhood and \(\gamma\) is classified by \(\mathsf{PLContour}\) inside that neighbourhood.
\end{definition}

\begin{definition}[Contour-integral output specification]
\label{def:cont-int-output-spec}
The output specification is a single complex history \(I\) classified by \(\ComplexUp\), uniquely determined up to \(\hsame\) by \autoref{thm:cont-int-unique}.
\end{definition}

\begin{definition}[Contour-integral classifier]
\label{def:cont-int-classifier}
The classifier of the operation is \(\hsame\) on the output complex history. Two outputs are classified together iff they are \(\hsame\)-equal.
\end{definition}

\begin{definition}[Contour-integral operation laws]
\label{def:cont-int-operation-laws}
The operation laws of the contour-integral operation collect:
\begin{enumerate}
  \item linearity in the integrand (\autoref{lem:cont-int-linear});
  \item concatenation in the contour (\autoref{lem:cont-int-concat});
  \item reversal of the contour negates the integral (\autoref{lem:cont-int-reversal});
  \item invariance under piecewise-linear reparameterization (\autoref{lem:cont-int-reparameterization}).
\end{enumerate}
\end{definition}

\begin{definition}[Contour-integral ledger policy]
\label{def:cont-int-ledger-policy}
The ledger policy records:
\begin{enumerate}
  \item Cauchy closure: closed contours bounding triangulated regions inside the holomorphic domain produce zero (\autoref{thm:cauchy-triangle}, \autoref{lem:cauchy-polygon});
  \item Cauchy integral formula composition (\autoref{thm:cauchy-integral-formula}): point evaluation factors through a rational-approximated circle integral.
\end{enumerate}
\end{definition}

\begin{theorem}[Concrete contour-integral operation certificate]
\label{thm:contour-integral-operation}
The data \((\mathsf{ContIntInputSpec}, \mathsf{ContIntOutputSpec}, \hsame, \mathsf{ContIntOperationLaws}, \mathsf{ContIntLedgerPolicy})\) of \autoref{def:cont-int-input-spec}--\autoref{def:cont-int-ledger-policy} forms an operation certificate over \(\Hist\) for contour integration on \(\HolomorphicUp\). The operation is total on its input specification (\autoref{lem:cont-int-exists}), uniquely determined up to \(\hsame\) (\autoref{thm:cont-int-unique}), and closed under the four operation laws.
\end{theorem}
\begin{proof}[Sketch]
Totality is \autoref{lem:cont-int-exists}. Uniqueness is \autoref{thm:cont-int-unique}. The operation laws are \autoref{lem:cont-int-linear}, \autoref{lem:cont-int-concat}, \autoref{lem:cont-int-reversal}, and \autoref{lem:cont-int-reparameterization}. The ledger policy fields are \autoref{thm:cauchy-triangle}, \autoref{lem:cauchy-polygon}, and \autoref{thm:cauchy-integral-formula}. The certificate uses no \(\mathsf{Classical.choice}\), no \(\mathsf{Quot.sound}\), and no \(\mathsf{propext}\), so the operation's Lean targets fall under the project's axiom-purity gate.
\end{proof}

\begin{theorem}[Semantic contour-integral operation certificate]
\label{thm:contour-integral-semantic}
The concrete operation certificate of \autoref{thm:contour-integral-operation} satisfies the semantic operation-certificate schema: the input and output specifications are \(\Hist\)-predicates, the classifier is a restriction of \(\hsame\), and every operation law is given by an explicit constructive proof. The operation is functorial in its input pair under \(\hsame\): if \(\hsame(f, f')\) (in the sense of pointwise classifier on the holomorphic carrier) and \(\hsame(\gamma, \gamma')\) (in the sense of endpoint-wise classifier on \(\mathsf{PLContour}\)), then the resulting integrals are \(\hsame\)-equal.
\end{theorem}
\begin{proof}[Sketch]
Direct verification of the schema fields using the constructions of this chapter and the \(\hsame\)-respect of the underlying \(\HolomorphicUp\) and \(\ComplexUp\) classifiers.
\end{proof}
