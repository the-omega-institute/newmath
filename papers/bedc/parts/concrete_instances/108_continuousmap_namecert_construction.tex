\chapter{A Concrete Naming Certificate for $\ContinuousMapUp$}
\label{ch:concrete-instances-continuousmap-namecert}

The continuous-map interface $\ContinuousMapUp$ packages a function between metric carriers together with a pointwise modulus of continuity. It depends on $\NameCert_{\MetricSpaceUp}$ (\autoref{ch:concrete-instances-metric-namecert}) and lifts $\NameCert_{\ContinuousFunctionUp}$ (\autoref{ch:concrete-instances-continuous-namecert}) to general L10 metric carriers.

\section{Carrier definition}
\label{sec:continuousmap-carrier}

\begin{definition}[ContinuousMap carrier]
\label{def:continuousmap-carrier}
For histories $source,map,target,modulus,cert,dist:\mathsf{BHist}$,
$\mathsf{ContinuousMapCarrier}(source,map,target,modulus,cert,dist)$ is the
conjunction of the continuous-function carrier
$\mathsf{ContinuousFunctionCarrier}(source,map,target,modulus,cert)$ and the
metric distance witness $\mathsf{MetricDistanceWitness}(source,target,dist)$.
\end{definition}
\leandef{BEDC.Derived.ContinuousMapUp.ContinuousMapCarrier}

\begin{theorem}[ContinuousMap canonical distance carrier]
\label{thm:continuousmap-carrier-canonical-distance-iff}
For histories $source,map,target,modulus,cert:\mathsf{BHist}$, the carrier
with canonical distance ledger $\mathsf{append}(source,target)$ is exactly
the underlying continuous-function carrier:
$$
\mathsf{ContinuousMapCarrier}
(source,map,target,modulus,cert,\mathsf{append}(source,target))
\Longleftrightarrow
\mathsf{ContinuousFunctionCarrier}(source,map,target,modulus,cert).
$$
\end{theorem}
\leanchecked{BEDC.Derived.ContinuousMapUp.ContinuousMapCarrier\_canonical\_distance\_iff}
\begin{proof}
The forward direction projects the function-carrier component. Conversely, the
function carrier supplies unary source and target histories; unary append
closure and the reflexive continuation equation build the canonical metric
distance witness.
\end{proof}

\begin{theorem}[ContinuousMap carrier empty distance exactness]
\label{thm:continuousmap-carrier-empty-distance-exact}
For histories $source,map,modulus,cert:\mathsf{BHist}$,
$$
\begin{array}{l}
\mathsf{ContinuousMapCarrier}(source,map,\emp,modulus,cert,\emp)\\
\quad\Longleftrightarrow\quad
\mathsf{UnaryHistory}(source)\land
\mathsf{UnaryHistory}(map)\land
\mathsf{UnaryHistory}(modulus)\land
source\sim\emp\land map\sim\emp\land cert\sim modulus.
\end{array}
$$
\end{theorem}
\leanchecked{BEDC.Derived.ContinuousMapUp.ContinuousMapCarrier\_empty\_distance\_exact}
\begin{proof}
Unpack the carrier into the continuous-function row and the metric-distance
row. Empty distance forces the source endpoint to be $\emp$, while the
continuous-function graph row into $\emp$ forces the map component to be
$\emp$. The remaining certificate row is the left-unit continuation from
$\emp$ to $modulus$, hence $cert\sim modulus$. Conversely, the displayed
unary and boundary data build the two continuation rows and the empty-distance
metric witness directly.
\end{proof}

\begin{theorem}[ContinuousMap empty certificate exactness]
\label{thm:continuousmap-carrier-empty-cert-exactness}
For histories $source,map,target,modulus,dist:\mathsf{BHist}$,
$$
\mathsf{ContinuousMapCarrier}(source,map,target,modulus,\emp,dist)
$$
holds exactly when
$$
source \hsame \emp,\quad map \hsame \emp,\quad target \hsame \emp,\quad
modulus \hsame \emp,\quad dist \hsame \emp .
$$
\end{theorem}
\begin{proof}
The function-carrier half gives empty source, map, target, and modulus by the
empty-certificate theorem for $\mathsf{ContinuousFunctionCarrier}$. With source
and target reduced to $\emp$, the metric-distance continuation is compared with
the right unit continuation, forcing the displayed distance to be empty. The
reverse direction reassembles the function witness and the empty distance
witness after the five endpoint equalities are exposed.
\end{proof}
\leanchecked{BEDC.Derived.ContinuousMapUp.ContinuousMapCarrier\_empty\_cert\_iff}

\begin{theorem}[Empty identity maps reflect metric witnesses]
\label{thm:continuousmap-empty-identity-metric-witness-reflects}
For histories $x,y,x',y',d,m_x,m_y,c_x,c_y:\mathsf{BHist}$, if
$\mathsf{ContinuousFunctionCarrier}(x,\emp,x',m_x,c_x)$ and
$\mathsf{ContinuousFunctionCarrier}(y,\emp,y',m_y,c_y)$ are empty-map
continuous carriers, then every displayed metric witness
$\mathsf{MetricDistanceWitness}(x',y',d)$ reflects to a source-side witness
$\mathsf{MetricDistanceWitness}(x,y,d)$.
\end{theorem}
\begin{proof}
The empty-map carrier readback identifies $x'$ with $x$ and $y'$ with $y$.
Unary endpoint membership and the continuation field of the metric witness
transport along these two $\hsame$ witnesses, while the distance history is
held fixed.
\end{proof}
\leanchecked{BEDC.Derived.ContinuousMapUp.ContinuousMap\_empty\_identity\_metric\_witness\_reflects}

\begin{theorem}[Empty identity maps preserve metric result determinacy]
\label{thm:continuousmap-empty-identity-metric-result-deterministic}
For histories $x,y,x',y',d,d',m_x,m_y,c_x,c_y:\mathsf{BHist}$, if the two

\end{theorem}
\begin{proof}
First reflect the image-side witness through the empty continuous maps to get

\end{proof}

\begin{theorem}[ContinuousMap empty-target metric rebased carrier]
\label{thm:continuousmap-empty-target-metric-rebased-carrier}
For histories $source,modulus,cert,dist,cert'$, suppose
$$
\mathsf{ContinuousFunctionCarrier}(source,\emp,source,modulus,cert),
\qquad
\mathsf{MetricDistanceWitness}(source,\emp,dist),
\qquad
\ContR(dist,modulus,cert').
$$
Then the metric distance ledger itself supports the empty-map continuous
function carrier
$$
\mathsf{ContinuousFunctionCarrier}(dist,\emp,dist,modulus,cert').
$$
\end{theorem}
\begin{proof}
Unpack the continuous-function carrier and the metric distance witness. The
metric witness supplies unary membership of $dist$, while the continuous
carrier supplies unary membership of $modulus$. The empty map is unary, the
right-unit continuation gives $\ContR(dist,\emp,dist)$, and the displayed
continuation gives the terminal certificate row.
\end{proof}
\leanchecked{BEDC.Derived.ContinuousMapUp.ContinuousMap\_empty\_target\_metric\_rebased\_carrier}

\begin{theorem}[ContinuousMap empty-target metric certificate determinacy]
\label{thm:continuousmap-empty-target-metric-cert-deterministic}
For histories $source,modulus,cert,dist,cert'$, suppose
$$
\mathsf{ContinuousFunctionCarrier}(source,\emp,source,modulus,cert),
\qquad
\mathsf{MetricDistanceWitness}(source,\emp,dist),
\qquad
\ContR(dist,modulus,cert').
$$
Then the original continuous-function certificate and the metric-rebased

$$
\hsame(cert,cert').
$$
\end{theorem}
\begin{proof}
The empty-right metric witness identifies $dist$ with $source$ while preserving
the unary distance ledger. The original carrier supplies the continuation
$\ContR(source,modulus,cert)$; transporting the source side across the metric

\end{proof}

\begin{theorem}[ContinuousMap target, certificate, and distance determinacy]
\label{thm:continuousmap-target-cert-distance-deterministic}
For histories $source$, $map$, $target$, $target'$, $modulus$, $cert$, $cert'$,
$distance$, and $distance'$, two concrete continuous-map carriers with the same
source, map, and modulus determine the target, certificate, and distance
ledgers up to $\hsame$:
$$
\begin{array}{l}
\mathsf{ContinuousMapCarrier}(source,map,target,modulus,cert,distance)
\Longrightarrow
\mathsf{ContinuousMapCarrier}(source,map,target',modulus,cert',distance')
\\
\Longrightarrow
\hsame(target,target')\land\hsame(cert,cert')\land
\hsame(distance,distance')\land\ContR(source,target,distance).
\end{array}
$$
\end{theorem}
\begin{proof}
The continuous-function components have the same source, map, and modulus, so
continuous-function target-certificate determinacy gives
$\hsame(target,target')$ and $\hsame(cert,cert')$. The two metric distance
witnesses then share source histories and target histories up to $\hsame$;
metric result determinacy identifies $distance$ and $distance'$. The
continuation ledger $\ContR(source,target,distance)$ is the final field of the
left metric distance witness.
\end{proof}
\leanchecked{BEDC.Derived.ContinuousMapUp.ContinuousMapCarrier\_target\_cert\_distance\_deterministic}

\begin{theorem}[ContinuousMap carrier canonical distance exactness]
\label{thm:continuousmap-carrier-canonical-distance-exactness}
For histories $source,map,target,modulus,cert,distance:\mathsf{BHist}$,
the concrete continuous-map carrier is exactly the continuous-function carrier
together with the canonical source-target distance classifier:
$$
\mathsf{ContinuousMapCarrier}(source,map,target,modulus,cert,distance)
\Longleftrightarrow
\mathsf{ContinuousFunctionCarrier}(source,map,target,modulus,cert)\land
\hsame(distance,\mathsf{append}(source,target)).
$$
\end{theorem}
\begin{proof}
In the forward direction, project the continuous-function carrier from the
given continuous-map carrier. Theorem~\ref{thm:continuousmap-carrier-canonical-distance-iff}
then supplies the carrier with distance ledger
$\mathsf{append}(source,target)$, and
Theorem~\ref{thm:continuousmap-target-cert-distance-deterministic} compares
the two carriers with the same source, map, target, modulus, and certificate,
identifying $distance$ with $\mathsf{append}(source,target)$.

Conversely, start from the displayed continuous-function carrier and the
classifier $\hsame(distance,\mathsf{append}(source,target))$.
Theorem~\ref{thm:continuousmap-carrier-canonical-distance-iff} builds the
canonical-distance carrier. Its metric component contains the continuation
$\ContR(source,target,\mathsf{append}(source,target))$ and unary membership of
the canonical distance ledger. Transport the continuation by
$\mathsf{BEDC.Derived.MetricUp.MetricDistanceWitness\_cont\_hsame\_transport}$,
transport unary membership of the distance ledger along $\hsame$, and repack
these fields as the required $\mathsf{MetricDistanceWitness}(source,target,distance)$.
Together with the original continuous-function carrier this is precisely
$\mathsf{ContinuousMapCarrier}(source,map,target,modulus,cert,distance)$.

The ten-field fit is: traditional target normalized continuous-map distance
ledger; BEDC source $\mathsf{ContinuousFunctionCarrier}$,
$\mathsf{MetricDistanceWitness}$, and $\ContR$; carrier
$\mathsf{ContinuousMapCarrier}$; classifier $\hsame$; stability obligations
distance determinacy and metric hsame transport; ledger the hidden distance
history; core theorem this iff; exactness the displayed iff; standard bridge
deferred; status $\mathsf{ProofTarget}$.
\end{proof}
\leanchecked{BEDC.Derived.ContinuousMapUp.ContinuousMapCarrier\_canonical\_distance\_exactness}

\begin{theorem}[ContinuousMap empty map with empty distance boundaries]
\label{thm:continuousmap-empty-map-empty-distance-boundaries}
For histories $source$, $target$, $modulus$, and $cert$, a concrete
continuous-map carrier with empty graph map and empty distance ledger is
equivalent to empty source and target boundaries together with the empty-source
modulus witness:
$$
\mathsf{ContinuousMapCarrier}(source,\emp,target,modulus,cert,\emp)
\Longleftrightarrow
\hsame(source,\emp)\land\hsame(target,\emp)\land
\mathsf{ContinuousModulusWitness}(\emp,modulus,cert).
$$
\end{theorem}
\begin{proof}
In the forward direction, empty-map exactness for the continuous-function
component gives a modulus witness over the target, while empty-distance
exactness for the metric witness collapses both endpoint histories to
$\emp$. Transporting across the target collapse presents the modulus witness
over $\emp$.

Conversely, the empty endpoint data presents the graph carrier as the
empty-map exactness case, and metric empty-distance exactness builds the
distance witness over the two empty endpoints. Their conjunction is exactly the
concrete continuous-map carrier.
\end{proof}
\leanchecked{BEDC.Derived.ContinuousMapUp.ContinuousMapCarrier\_empty\_map\_empty\_distance\_boundaries\_iff}

\begin{theorem}[ContinuousMap prefix canonical distance closure]
\label{thm:continuousmap-carrier-prefix-canonical-distance-closed}

\begin{array}{l}
\mathsf{ContinuousMapCarrier}
(\mathsf{append}(p,source),map,\mathsf{append}(p,target),modulus,

\end{theorem}
\leanchecked{BEDC.Derived.ContinuousMapUp.ContinuousMapCarrier\_prefix\_canonical\_distance\_closed}
\begin{proof}

unary by unary append closure, their appended distance is unary by another

\end{proof}

\begin{theorem}[ContinuousMap empty identities are two-sided composition units]
\label{thm:continuousmap-empty-identities-two-sided-composition-units}
For histories
$source,target,map,modulus,cert,distance,displayed_L,displayed_R:\mathsf{BHist}$,
assume
$$
\mathsf{ContinuousMapCarrier}(source,map,target,modulus,cert,distance).
$$
Then the empty-map identities at the source and target are continuous-map
carriers, and they are two-sided units for composition:
\begin{enumerate}
\item the source identity carrier is
$$
\mathsf{ContinuousMapCarrier}
(source,\emp,source,\emp,source,\mathsf{append}(source,source));
$$

\item the target identity carrier is
$$
\mathsf{ContinuousMapCarrier}
(target,\emp,target,\emp,target,\mathsf{append}(target,target));
$$

\item composing the source identity on the left with the displayed carrier,
using
$$
\ContR(\emp,map,map),\qquad
\ContR(\emp,modulus,modulus),\qquad
\ContR(target,modulus,cert),
$$
closes as the canonical carrier
$$
\mathsf{ContinuousMapCarrier}
(source,map,target,modulus,cert,\mathsf{append}(source,target));
$$

\item composing the displayed carrier with the target identity on the right,
using
$$
\ContR(map,\emp,map),\qquad
\ContR(modulus,\emp,modulus),\qquad
\ContR(target,modulus,cert),
$$
closes as the same canonical carrier
$$
\mathsf{ContinuousMapCarrier}
(source,map,target,modulus,cert,\mathsf{append}(source,target));
$$

\item if the left and right unit composites are displayed with distances
$displayed_L$ and $displayed_R$, respectively, then
$$
\hsame(displayed_L,\mathsf{append}(source,target))\land
\hsame(displayed_R,\mathsf{append}(source,target)).
$$
\end{enumerate}
\end{theorem}
\begin{proof}
Project the continuous-function carrier from the given
$\mathsf{ContinuousMapCarrier}$. Its fields supply unary endpoint data and the
certificate continuation $\ContR(target,modulus,cert)$.

For the source identity, use the unary source field and the left-unit
continuation $\ContR(source,\emp,source)$ to form
$\mathsf{ContinuousModulusWitness}(source,\emp,source)$. Apply
Theorem~\ref{thm:continuous-function-carrier-empty-map-identity}, then package
the canonical distance by
Theorem~\ref{thm:continuousmap-carrier-canonical-distance-iff}.

For the target identity, the same construction uses the unary target field and
the left-unit continuation $\ContR(target,\emp,target)$. The empty-map identity
theorem builds the continuous-function carrier at $target$, and canonical
distance packaging gives the displayed target identity carrier.

Now apply Theorem~\ref{thm:continuousmap-carrier-comp-closed} to the source
identity carrier and the displayed carrier, with the BEDC left-unit laws
$\ContR(\emp,map,map)$ and $\ContR(\emp,modulus,modulus)$ and the projected
certificate ledger $\ContR(target,modulus,cert)$. This gives the left unit
composite over $\mathsf{append}(source,target)$.

Apply the same composition-closure theorem to the displayed carrier and the
target identity carrier, using the right-unit laws $\ContR(map,\emp,map)$ and
$\ContR(modulus,\emp,modulus)$ together with the same certificate ledger. This
gives the right unit composite over the same canonical distance.

Finally, apply
Theorem~\ref{thm:continuousmap-carrier-comp-distance-deterministic} to each unit
composite presentation. Comparing any displayed left or right unit-composite
distance with the corresponding canonical composite identifies it by $\hsame$
with $\mathsf{append}(source,target)$.
\end{proof}

\begin{theorem}[ContinuousMap carrier composition distance determinacy]
\label{thm:continuousmap-carrier-comp-distance-deterministic}
For histories
$source,mid,target,mapF,mapG,mapFG,modF,modG,modFG,certF,certG,certFG,distF,distG,displayed:\mathsf{BHist}$,
assume the two continuous-map carriers
$$
\begin{array}{l}
\mathsf{ContinuousMapCarrier}(source,mapF,mid,modF,certF,distF),\\
\mathsf{ContinuousMapCarrier}(mid,mapG,target,modG,certG,distG),
\end{array}
$$
the continuation ledgers
$$
\ContR(mapF,mapG,mapFG),\qquad
\ContR(modF,modG,modFG),\qquad
\ContR(target,modFG,certFG),
$$
and a displayed composite carrier
$$
\mathsf{ContinuousMapCarrier}(source,mapFG,target,modFG,certFG,displayed).
$$
Then the displayed composite distance is the canonical source-target ledger:
$$
\hsame(displayed,\mathsf{append}(source,target)).
$$
\end{theorem}
\begin{proof}
Theorem~\ref{thm:continuousmap-carrier-comp-closed} applied to the two input
carriers and the three continuation ledgers builds the canonical composite
carrier
$$
\mathsf{ContinuousMapCarrier}
(source,mapFG,target,modFG,certFG,\mathsf{append}(source,target)).
$$

Compare the displayed composite carrier with this canonical carrier by
Theorem~\ref{thm:continuousmap-target-cert-distance-deterministic}, with the
same source, map, target, modulus, and certificate histories on both sides.
The distance component of that determinacy conclusion gives
$\hsame(displayed,\mathsf{append}(source,target))$.
\end{proof}

\begin{theorem}[ContinuousMap composition canonical certificate package]
\label{thm:continuousmap-composition-canonical-certificate-package}
For histories
$source,mid,target,mapF,mapG,mapFG,modF,modG,modFG,certF,certG,certFG,distF,distG,displayed:\mathsf{BHist}$,
assume the two continuous-map carriers
$$
\begin{array}{l}
\mathsf{ContinuousMapCarrier}(source,mapF,mid,modF,certF,distF),\\
\mathsf{ContinuousMapCarrier}(mid,mapG,target,modG,certG,distG),
\end{array}
$$
and the continuation ledgers
$$
\ContR(mapF,mapG,mapFG),\qquad
\ContR(modF,modG,modFG),\qquad
\ContR(target,modFG,certFG).
$$
Then the composition data determine the canonical certificate package:
\begin{enumerate}
\item the composite carrier closes over the canonical source-target distance
ledger,
$$
\mathsf{ContinuousMapCarrier}
(source,mapFG,target,modFG,certFG,\mathsf{append}(source,target));
$$

\item every displayed composite carrier with the same visible fields has that
canonical distance ledger, namely if
$$
\mathsf{ContinuousMapCarrier}(source,mapFG,target,modFG,certFG,displayed),
$$
then
$$
\hsame(displayed,\mathsf{append}(source,target));
$$

\item the composite graph obeys metric-depth accounting,
$$
\mathsf{MetricDistanceDepth}(target)
=
\mathsf{MetricDistanceDepth}(source)+
\mathsf{MetricDistanceDepth}(mapFG).
$$
\end{enumerate}
\end{theorem}
\begin{proof}
For the carrier row, apply
Theorem~\ref{thm:continuousmap-carrier-comp-closed} to the two input carriers
and the three continuation ledgers. This gives the displayed
$\mathsf{ContinuousMapCarrier}$ with distance
$\mathsf{append}(source,target)$.

For distance uniqueness, suppose a displayed composite carrier with the same
source, map, target, modulus, and certificate histories is given. Compare the
displayed carrier with the canonical carrier just obtained by
Theorem~\ref{thm:continuousmap-target-cert-distance-deterministic}. The
distance component gives $\hsame(displayed,\mathsf{append}(source,target))$.

For metric-depth accounting, project the continuous-function halves from the
two input $\mathsf{ContinuousMapCarrier}$ hypotheses. Apply
Theorem~\ref{thm:continuousmap-comp-graph-depth-add} to those halves and the
same three continuation ledgers, producing the stated equality for
$\mathsf{MetricDistanceDepth}$.
\end{proof}

\begin{theorem}[ContinuousMap associative composition canonical package]
\label{thm:continuousmap-carrier-composition-associative-canonical-package}
For histories
$source,mid_1,mid_2,target,mapF,mapG,mapH,mapFG,mapGH,mapL,mapR,modF,modG,modH,modFG,modGH,modL,modR,certF,certG,certH,certFG,certGH,certL,certR,distF,distG,distH,displayedL,displayedR:\mathsf{BHist}$,
assume the three concrete continuous-map carriers
$$
\begin{array}{l}
\mathsf{ContinuousMapCarrier}(source,mapF,mid_1,modF,certF,distF),\\
\mathsf{ContinuousMapCarrier}(mid_1,mapG,mid_2,modG,certG,distG),\\
\mathsf{ContinuousMapCarrier}(mid_2,mapH,target,modH,certH,distH),
\end{array}
$$
the left-bracketed continuation ledgers
$$
\begin{array}{lll}
\ContR(mapF,mapG,mapFG),&
\ContR(modF,modG,modFG),&
\ContR(mid_2,modFG,certFG),\\
\ContR(mapFG,mapH,mapL),&
\ContR(modFG,modH,modL),&
\ContR(target,modL,certL),
\end{array}
$$
and the right-bracketed continuation ledgers
$$
\begin{array}{lll}
\ContR(mapG,mapH,mapGH),&
\ContR(modG,modH,modGH),&
\ContR(target,modGH,certGH),\\
\ContR(mapF,mapGH,mapR),&
\ContR(modF,modGH,modR),&
\ContR(target,modR,certR).
\end{array}
$$
Then the threefold composition data determine the associative canonical
certificate package:
\begin{enumerate}
\item the left-bracketed composite carrier closes over the canonical
source-target distance ledger,
$$
\mathsf{ContinuousMapCarrier}
(source,mapL,target,modL,certL,\mathsf{append}(source,target));
$$

\item the right-bracketed composite carrier closes over the same canonical
source-target distance ledger,
$$
\mathsf{ContinuousMapCarrier}
(source,mapR,target,modR,certR,\mathsf{append}(source,target));
$$

\item the two composite graph-map histories are classified together,
$$
\hsame(mapL,mapR);
$$

\item the two composite modulus histories are classified together,
$$
\hsame(modL,modR);
$$

\item every displayed carrier with either bracketed visible fields has the
canonical distance ledger:
$$
\begin{array}{l}
\mathsf{ContinuousMapCarrier}(source,mapL,target,modL,certL,displayedL)
\Rightarrow
\hsame(displayedL,\mathsf{append}(source,target)),\\
\mathsf{ContinuousMapCarrier}(source,mapR,target,modR,certR,displayedR)
\Rightarrow
\hsame(displayedR,\mathsf{append}(source,target)).
\end{array}
$$
\end{enumerate}
\end{theorem}
\begin{proof}
Build the left bracketing in two closure steps. First apply
Theorem~\ref{thm:continuousmap-carrier-comp-closed} to the
$source \to mid_1$ and $mid_1 \to mid_2$ carriers, using the graph, modulus,
and certificate ledgers ending at $certFG$. Apply the same theorem again to
that intermediate carrier and the $mid_2 \to target$ carrier, using the
ledgers ending at $certL$. This gives the left composite carrier with distance
$\mathsf{append}(source,target)$.

Build the right bracketing symmetrically. Apply
Theorem~\ref{thm:continuousmap-carrier-comp-closed} to the
$mid_1 \to mid_2$ and $mid_2 \to target$ carriers, using the ledgers ending at
$certGH$. Then compose the $source \to mid_1$ carrier with that intermediate
carrier using the ledgers ending at $certR$. This gives the right composite
carrier with the same displayed distance ledger
$\mathsf{append}(source,target)$.

For the graph-map row, apply continuation associativity up to history
sameness, Theorem~\ref{thm:continuation-associativity-up-to-hsame}, to
$\ContR(mapF,mapG,mapFG)$, $\ContR(mapFG,mapH,mapL)$,
$\ContR(mapG,mapH,mapGH)$, and $\ContR(mapF,mapGH,mapR)$. This is the paper
form of the checked kernel theorem
\texttt{continuation\_associativity\_up\_to\_hsame}; uniqueness of the
associative continuation endpoint is also recorded by
\texttt{cont\_assoc\_unique}.

For the modulus row, use the same continuation-associativity argument on
$\ContR(modF,modG,modFG)$, $\ContR(modFG,modH,modL)$,
$\ContR(modG,modH,modGH)$, and $\ContR(modF,modGH,modR)$. The resulting
classifier row is $\hsame(modL,modR)$.

Finally apply
Theorem~\ref{thm:continuousmap-carrier-comp-distance-deterministic} to the
left two-step composition data for any displayed left-bracketed carrier, and
apply the same theorem to the right two-step composition data for any
displayed right-bracketed carrier. The two applications give the two displayed
distance-determinacy clauses.
\end{proof}
\leanchecked{BEDC.Derived.ContinuousMapUp.ContinuousMapCarrier\_composition\_associative\_canonical\_package}
