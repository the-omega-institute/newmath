\chapter{A Concrete Naming Certificate for $\MeasureUp$}
\label{ch:concrete-instances-measure-namecert}

The measure interface $\MeasureUp$ packages a carrier with a constructive $\sigma$-additive non-negative real-valued set function on a designated $\sigma$-algebra of measurable subsets. It depends on $\NameCert_{\RealUp}$ (\autoref{ch:concrete-instances-real-namecert}) and supplies the substrate for the integral certificate $\NameCert_{\IntegralUp}$ (\autoref{ch:concrete-instances-integral-namecert}).

\section{Carrier definition}
\label{sec:measure-carrier}

\begin{definition}[MeasureUp carrier surface]
\label{def:measureup-carrier-surface}
\label{def:measure-carrier-specification}
A $\MeasureUp$ carrier surface is a tuple
$$
(\mathsf{MCarrier},\sim_M,\mathcal{A},\mu,\sim_R)
$$
with the following rows. The carrier row is a predicate
$\mathsf{MCarrier}:\mathsf{BHist}\to\mathsf{Prop}$. The classifier row
$h\sim_M k$ is a NameCert-style classifier on carried histories in the sense of
\autoref{def:lean-carrier-naming-certificate}. The measurable-subset row
$\mathcal{A}(h,E)$ assigns to each carried history $h$ a designated family of
measurable subsets $E$ of its event object. The measure row
$\mu_h(E)$ assigns to every $\mathcal{A}(h,E)$ a $\RealUp$ endpoint classified
by the $\RealUp$ classifier $\sim_R$ of
\autoref{def:real-classifier-specification}.

Carrier transport is part of the surface: if $\mathsf{MCarrier}(h)$,
$h\sim_M k$, and $\mathcal{A}(h,E)$, then the transported event row is read in
the $k$-fiber, and the corresponding measure values compare under $\sim_R$.
\end{definition}

\begin{definition}[MeasureUp countable-additivity row]
\label{def:measureup-countable-additivity-row}
For a $\MeasureUp$ carrier surface as in
\autoref{def:measureup-carrier-surface}, the countable-additivity row fixes,
for every carried history $h$, a sequence $E:\mathbb{N}\to\mathcal{P}(h)$
satisfying:
\begin{enumerate}
\item each $E_n$ is measurable, $\mathcal{A}(h,E_n)$;
\item the sequence is pairwise disjoint in the $h$-fiber;
\item the union event $\bigcup_{n:\mathbb{N}}E_n$ is supplied as a measurable
row of $\mathcal{A}(h,-)$;
\item there is a countable $\RealUp$ sum endpoint
$\sum_{n=0}^{\infty}\mu_h(E_n)$;
\item the head-tail readback compares this endpoint with
$\mu_h(E_0)+\sum_{n=0}^{\infty}\mu_h(E_{n+1})$ under $\sim_R$;
\item pointwise $\sim_R$-classification of the summands transports to
$\sim_R$-classification of the two countable sum endpoints.
\end{enumerate}
The row asserts the sigma-additivity comparison
$$
\mu_h\left(\bigcup_{n:\mathbb{N}}E_n\right)\sim_R
\sum_{n=0}^{\infty}\mu_h(E_n).
$$
\end{definition}

\begin{theorem}[Measure carrier classifier obligations]
\label{thm:measure-carrier-classifier-obligations}
Every $\MeasureUp$ carrier surface supplies the NameCert-style reflexivity,
symmetry, and transitivity rows for $\sim_M$ on carried histories, and it
transports $\mathsf{MCarrier}$, measurable rows, and measure comparisons along
that classifier.
\end{theorem}
\begin{proof}
Unpack the carrier surface of \autoref{def:measureup-carrier-surface}. Its
classifier row is required to be a carrier naming certificate in the sense of
\autoref{def:lean-carrier-naming-certificate}, so reflexivity, symmetry,
transitivity, and carrier transport are projections of that row.

The measurable-subset and measure rows are indexed by carried histories. The
surface states that a classifier comparison between carried histories re-reads
the event row in the target fiber and compares the corresponding $\RealUp$
measure values. Applying that transport clause gives exactly the displayed
carrier, classifier, and measure-stability obligations.
\end{proof}

\begin{theorem}[Measure sigma-algebra obligations]
\label{thm:measure-sigma-algebra-obligations}
For every $\MeasureUp$ carrier surface, the designated measurable family
$\mathcal{A}(h,-)$ is closed under the empty event, binary union, countable
union, and complement rows used by the measure certificate whenever
$\mathsf{MCarrier}(h)$ holds.
\end{theorem}
\begin{proof}
Fix a carried history $h$. The measurable-subset row in
\autoref{def:measureup-carrier-surface} is the designated sigma-algebra family
for the $h$-fiber, so its certificate clauses give the empty event and
complement rows.

The same sigma-algebra row supplies finite binary union as the two-term case of
its closure data, and it supplies countable union for every measurable sequence.
These are the exact rows consumed by the sigma-additivity comparison in
\autoref{def:measureup-countable-additivity-row}.
\end{proof}

\begin{theorem}[Measure RealUp sum scope]
\label{thm:measure-realup-sum-scope}
The countable-additivity comparison of
\autoref{def:measureup-countable-additivity-row} is scoped only to the
$\RealUp$ classifier stability of \autoref{def:real-classifier-specification}
and to the additive $\AbGroupUp$ zero, unit, and congruence rows cited from
\autoref{def:abgroup-carrier}. It does not assert an independent construction
of real analysis outside those rows.
\end{theorem}
\begin{proof}
Unpack the countable-additivity row. Its endpoint is a $\RealUp$ countable sum,
and every comparison involving that endpoint is stated under the classifier
$\sim_R$ from \autoref{def:real-classifier-specification}.

The two measure consequences below use this endpoint only through head-tail
readback, pointwise classifier stability, sigma-additivity, and the additive
$\AbGroupUp$ rows of \autoref{def:abgroup-carrier}. Thus the scope of the row is
exactly the named classifier and additive certificate surface, with no extra
analytic closure claim.
\end{proof}

\begin{definition}[MeasureUp carrier obligation inventory]
\label{def:measureup-carrier-obligation-inventory}
A $\MeasureUp$ carrier is a $\mathsf{BHist}$ package whose rows expose the
constructive measure-space data needed by the certificate. Its ten-field
inventory is:
\begin{enumerate}
  \item traditional target: a constructive measure space;
  \item BEDC source: $\RealUp$, the additive $\AbGroupUp$ row for $\RealUp$
  addition, and countable StreamName-style measurable families;
  \item carrier choice: a $\mathsf{BHist}$ package containing a sample/event
  carrier, a $\sigma$-algebra predicate $\mathcal{A}$, a $\RealUp$-valued map
  $\mu$, and rows for the empty event, binary union, and countable union;
  \item classifier choice: componentwise $\hsame$ on the carrier package,
  together with the $\RealUp$ classifier on measure values;
  \item stability obligations: carrier transport, measurable-set transport,
  $\mu$-congruence, $\sigma$-additivity, and head-tail readback for countable
  sums;
  \item ledger: event rows, measurable sequence rows, and countable-sum rows;
  \item core theorem list: empty set zero, finite disjoint-union additivity,
  and countable disjoint additivity;
  \item exactness predicate: measurable package coverage, soundness, and
  completeness;
  \item standard bridge: absent at this paper site;
  \item status: paper obligation surface.
\end{enumerate}
\end{definition}

\begin{definition}[MeasureUp sigma-algebra carrier surface]
\label{def:measure-sigma-algebra-carrier-surface}
The $\MeasureUp$ sigma-algebra carrier surface is the following ten-field
BEDC package.
\begin{enumerate}
  \item traditional target: a constructive measure space whose measurable
  predicates form a $\sigma$-algebra and whose measure map takes values in
  $\RealUp$;
  \item BEDC source: $\mathsf{BHist}$ presentations, the certificate
  $\NameCert_{\RealUp}$, and the additive $\AbGroupUp$ row on real histories;
  \item carrier: a $\mathsf{BHist}$ package exposing measurable-event rows
  for the empty event, binary union, countable union, disjoint sequence
  readback, and $\RealUp$ measure values;
  \item classifier: componentwise $\hsame$ on the package rows, with the
  $\RealUp$ classifier applied to the visible measure endpoints;
  \item stability rows: transport for the empty event, binary union,
  countable union, disjoint sequence readback, and $\RealUp$ measure values;
  \item ledger rows: hidden measurable-set membership witnesses and
  countable-sum witnesses attached to the visible rows they justify;
  \item core theorem list:
  \autoref{thm:measure-empty-set-zero-from-sigma-additivity} and
  \autoref{thm:measure-finite-disjoint-union-additivity};
  \item exactness: visible row exhaustion, meaning every measurable-set,
  disjointness, value, and countable-sum witness consumed by the certificate is
  attached to one of the displayed rows;
  \item standard bridge: no $\StdBridge$ is asserted at this paper site;
  \item status: paper-side target for the $\MeasureUp$ NameCert obligation
  package.
\end{enumerate}
\end{definition}

\begin{definition}[MeasureUp certificate surface]
\label{def:measure-certificate-surface}
A $\MeasureUp$ certificate surface over the BEDC history substrate consists of the following data.
\begin{enumerate}
  \item a carrier predicate $\mathsf{MeasCode}:\mathsf{BHist}\to\mathsf{Prop}$ for histories that present measure codes;
  \item a classifier $\sim_M:\mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$ and a core $\NameCert(\mathsf{MeasCode},\sim_M)$, so membership in $\mathsf{MeasCode}$ transports along $\sim_M$ by the $\NameCert$ transport field;
  \item a designated measurable-family predicate $\mathsf{MeasFam}$ on predicates $E:\mathsf{BHist}\to\mathsf{Prop}$, stable under the displayed classifier and under $\hsame$ on the endpoint histories that present the same event;
  \item a $\RealUp$-valued endpoint map $\mu$ from measurable predicates to their measure endpoints, with equality of measurable presentations transported through the $\RealUp$ classifier $\sim_R$ of \autoref{def:real-classifier-specification};
  \item a countable-sum endpoint $\Sigma_\mu(E_\bullet)$ for each measurable sequence $E_\bullet:\mathbb{N}\to(\mathsf{BHist}\to\mathsf{Prop})$;
  \item $\sigma$-algebra rows asserting $\mathsf{MeasFam}(\varnothing)$, binary-union closure for disjoint measurable pairs, and countable-union closure for measurable sequences;
  \item a $\sigma$-additivity row: whenever $E_\bullet$ is a pairwise-disjoint measurable sequence with countable union $U$, the comparison $\mu(U)\sim_R\Sigma_\mu(E_\bullet)$ holds;
  \item a head-tail readback row: for every measurable sequence $E_\bullet$, the endpoint $\Sigma_\mu(E_\bullet)$ is classified with $\mu(E_0)+\Sigma_\mu(E_{\bullet+1})$ by the $\RealUp$ additive operation;
  \item a countable-sum classifier row: pointwise classifier equality of measurable sequences transports their countable endpoints through $\sim_R$.
\end{enumerate}
The dependencies of the surface are exactly the BEDC objects named above: $\NameCert$ supplies carrier and classifier transport; $\mathsf{BHist}$ supplies the visible code substrate; $\hsame$ and $\Cont$ supply history sameness and continuation discipline for event presentations; $\RealUp$ supplies the codomain classifier; and the additive $\AbGroupUp$ row of \autoref{def:abgroup-carrier} supplies the addition, zero, congruence, and cancellation laws used by the displayed measure consequences.
\end{definition}

\begin{definition}[Measure classifier specification]
\label{def:measure-classifier-specification}
For two $\MeasureUp$ carriers $M$ and $N$ as in \autoref{def:measure-carrier-specification}, the public classifier $M\sim_{\MeasureUp}N$ consists of shared event-carrier sameness and endpoint agreement. Shared event-carrier sameness says that an event witness of $M$ and an event witness of $N$ classify the same displayed event precisely when their BHist presentations are related by the displayed $\hsame$ transport rows and their measurable witnesses transport between the two displayed measurable families. Endpoint agreement says that, for every such shared measurable event, the endpoints $\mu_M(E)$ and $\mu_N(E')$ compare through $\mathsf{RealConstantHistoryClassifier}$ and hence through the $\RealUp$ classifier of \autoref{def:real-classifier-specification}. The classifier does not assert quotient equality or host set equality; it only records the public event witness, measurable witness, and $\RealUp$ endpoint rows visible in the carrier ledger.
\end{definition}

\begin{theorem}[Measure carrier transport obligation]
\label{thm:measure-obligation-carrier-transport}
Let $\mathcal{M}$ be a $\MeasureUp$ certificate surface as in \autoref{def:measure-certificate-surface}. If $h,k:\mathsf{BHist}$ satisfy $\mathsf{MeasCode}(h)$ and $h\sim_M k$, then $\mathsf{MeasCode}(k)$. Moreover, every displayed row attached to the carried measure code, namely the measurable-family row, $\sigma$-algebra rows, $\RealUp$ endpoint map, $\sigma$-additivity row, head-tail row, and countable-sum classifier row, may be read at $k$ after transporting its carrier proof along $h\sim_M k$.
\end{theorem}
\begin{proof}
The carrier part is the transport field of the core $\NameCert(\mathsf{MeasCode},\sim_M)$ in the surface. Apply it to the carried proof at $h$ and the classifier comparison $h\sim_M k$.

The remaining rows are fields of the same certificate surface indexed by the carried measure code. Once the carrier proof has been moved from $h$ to $k$, the row projections are read with that transported proof. No quotient or host equality of histories is used; the only movement is along the displayed $\NameCert$ classifier.
\end{proof}

\begin{theorem}[MeasureUp carrier-classifier obligation]
\label{thm:measure-carrier-classifier-obligation}
For the $\MeasureUp$ sigma-algebra carrier surface of
\autoref{def:measure-sigma-algebra-carrier-surface}, the carrier slots are
exactly the displayed $\mathsf{BHist}$ rows for measurable events, empty event,
binary union, countable union, disjoint sequence readback, $\RealUp$ measure
values, hidden measurable-set membership, and countable-sum witnesses. The
classifier slots are exactly componentwise $\hsame$ on those rows, together
with the $\RealUp$ classifier on measure endpoints. Transport along the
classifier carries each carrier slot to the corresponding slot of the target
presentation.
\end{theorem}
\begin{proof}
Read the ten fields of
\autoref{def:measure-sigma-algebra-carrier-surface}. The carrier entries are
the rows named in the carrier, stability, and ledger clauses, and no additional
carrier row is listed by the exactness clause.

For transport, project the componentwise $\hsame$ witness at each visible row.
The event rows move by their own history-sameness components, and the
measure-value rows move by first projecting the endpoint comparison and then
using $\NameCert_{\RealUp}$ to read it as a $\RealUp$ classifier comparison.

The ledger rows are repacked after the visible rows move. Since exactness says
that every hidden membership or countable-sum witness is attached to one of the
displayed rows, transporting the visible row also transports the only ledger
entry that can justify it.
\end{proof}

\begin{theorem}[Measure constant-empty sigma-additive row]
\label{thm:measure-obligation-sigma-additive-row}
Let $\mathcal{M}$ be a $\MeasureUp$ certificate surface as in \autoref{def:measure-certificate-surface}. Define $E_n:=\varnothing$ for every $n:\mathbb{N}$. Then $E_\bullet$ is a measurable sequence, it is pairwise disjoint, its countable union is $\varnothing$, and the $\sigma$-additivity row gives the comparison
$$
\mu(\varnothing)\sim_R\Sigma_\mu(E_\bullet).
$$
This is the sigma-additive input used by \autoref{thm:measure-empty-set-zero-from-sigma-additivity}.
\end{theorem}
\begin{proof}
The $\sigma$-algebra rows of the surface include the empty-event row, so each $E_n$ is measurable. Pairwise disjointness is immediate because both sides of every pairwise intersection are the empty predicate.

The countable union of the constant-empty sequence is again the empty predicate. The countable-union row therefore identifies the union endpoint required by the $\sigma$-additivity row with $\varnothing$.

Apply the $\sigma$-additivity row to this measurable pairwise-disjoint sequence and its displayed union. The resulting comparison is precisely $\mu(\varnothing)\sim_R\Sigma_\mu(E_\bullet)$.
\end{proof}

\begin{theorem}[MeasureUp sigma-additivity obligation]
\label{thm:measure-sigma-additivity-obligation}
For the $\MeasureUp$ sigma-algebra carrier surface of
\autoref{def:measure-sigma-algebra-carrier-surface}, the sigma-additivity row
classifies the value of a countable disjoint union by the displayed countable
$\RealUp$ sum. Concretely, for every pairwise-disjoint measurable sequence
$E_\bullet$ with countable union $U$, the certificate supplies
$$
\mu(U)\sim_R\Sigma_\mu(E_\bullet).
$$
The same row supplies the measurable-sequence, disjointness, union, head-tail,
and $\RealUp$ endpoint hypotheses consumed by
\autoref{thm:measure-empty-set-zero-from-sigma-additivity} and
\autoref{thm:measure-finite-disjoint-union-additivity}.
\end{theorem}
\begin{proof}
Project the sigma-additivity row from the carrier surface and apply it to the
displayed measurable sequence $E_\bullet$, its pairwise-disjointness witness,
and its countable-union row $U$. The projected conclusion is exactly the
$\RealUp$ classifier comparison between $\mu(U)$ and
$\Sigma_\mu(E_\bullet)$.

For the empty-zero theorem, specialize $E_\bullet$ to the constant empty
sequence. The empty-event, countable-union, disjointness, and head-tail rows
are precisely the hypotheses required there.

For the finite disjoint-union theorem, specialize $E_\bullet$ to the sequence
with first terms $A$ and $B$ and empty tail. Binary-union closure supplies the
visible union row, while countable-union and head-tail readback supply the
countable sum whose first two summands are $\mu(A)$ and $\mu(B)$.
\end{proof}

\begin{theorem}[Measure countable-sum classifier stability obligation]
\label{thm:measure-obligation-countable-sum-classifier-stability}
Let $\mathcal{M}$ be a $\MeasureUp$ certificate surface as in \autoref{def:measure-certificate-surface}. Let $E_\bullet$ and $F_\bullet$ be measurable sequences. If every index satisfies $\mu(E_n)\sim_R\mu(F_n)$ through the $\RealUp$ classifier, then their countable endpoints satisfy
$$
\Sigma_\mu(E_\bullet)\sim_R\Sigma_\mu(F_\bullet).
$$
In particular, the all-empty tail used in \autoref{thm:measure-empty-set-zero-from-sigma-additivity} and the shifted empty tail used in \autoref{thm:measure-finite-disjoint-union-additivity} transport to the same countable $\RealUp$ sum.
\end{theorem}
\begin{proof}
The countable-sum classifier row is part of the certificate surface. Feed it the two measurable sequences and the pointwise $\RealUp$ classifier comparison of their measure endpoints.

For the all-empty tail, the pointwise comparison is reflexivity of $\sim_R$ at each endpoint $\mu(\varnothing)$. For the shifted finite-union tail, every shifted term is the empty event, so the same reflexive pointwise comparison applies after reading the sequence definition at each index.

Thus both downstream arguments may replace a syntactically shifted empty countable sum by the canonical all-empty countable sum using only classifier transport, not host-level equality of sums.
\end{proof}

\begin{theorem}[MeasureUp ledger exactness obligation]
\label{thm:measure-ledger-exactness-obligation}
For the $\MeasureUp$ sigma-algebra carrier surface of
\autoref{def:measure-sigma-algebra-carrier-surface}, the ledger contains
exactly the empty-event, binary-union, countable-union, disjointness,
measure-value, and countable-sum rows. Every hidden measurable-set membership
witness and every countable-sum witness is attached to one of those displayed
rows, and no external choice object or quotient object is part of the ledger.
\end{theorem}
\begin{proof}
Unpack the ledger and exactness clauses of
\autoref{def:measure-sigma-algebra-carrier-surface}. The ledger clause lists
only hidden measurable-set membership witnesses and countable-sum witnesses,
and the surrounding carrier and stability clauses list the visible rows to
which those witnesses may attach.

Exactness is visible row exhaustion. Therefore every ledger witness must be
read through the empty-event, binary-union, countable-union, disjointness,
measure-value, or countable-sum row that exposes it.

Since the ten-field surface names only $\mathsf{BHist}$ rows, $\hsame$,
$\NameCert_{\RealUp}$, and the additive $\AbGroupUp$ row, there is no remaining
slot in which an external choice object or quotient object could be stored.
\end{proof}

\begin{theorem}[Measure carrier and classifier stability]
\label{thm:measure-carrier-and-classifier-stability}
Let $M$ and $N$ be $\MeasureUp$ carriers as in \autoref{def:measure-carrier-specification}, related by the classifier of \autoref{def:measure-classifier-specification}. If $E$ is a displayed event of $M$, $E'$ is a displayed event of $N$, and the event presentations satisfy $\hsame(E,E')$, then the measurable witness for $E$ in $\mathcal A_M$ transports to a measurable witness for $E'$ in $\mathcal A_N$. Moreover the visible measure endpoints satisfy
$$
\mu_M(E)\sim_R\mu_N(E')
$$
through $\mathsf{RealConstantHistoryClassifier}$ and the $\RealUp$ classifier of \autoref{def:real-classifier-specification}.
\end{theorem}
\begin{proof}
The shared event-carrier component of \autoref{def:measure-classifier-specification} is exactly the transport rule for displayed event witnesses. Apply it to the $\hsame(E,E')$ comparison to move the event witness and its measurable row from $M$ to $N$.

The endpoint component of the classifier reads the two ledger endpoints attached to the transported measurable event. Since the carrier specification records the event witness, measurable witness, and $\RealUp$ endpoint in the same visible ledger, the endpoint comparison is the displayed $\mathsf{RealConstantHistoryClassifier}$ row.

Finally, \autoref{def:real-classifier-specification} identifies that row as a valid $\RealUp$ classifier comparison. No quotient of event codes and no host equality of subsets is introduced.
\end{proof}

\begin{theorem}[MeasureUp classifier stability obligations]
\label{thm:measureup-classifier-stability-obligations}
For a $\MeasureUp$ carrier with inventory
\autoref{def:measureup-carrier-obligation-inventory}, componentwise $\hsame$
of carrier packages transports every row named by the inventory. In particular,
if two packages are componentwise $\hsame$, then measurable membership in
$\mathcal{A}$, the $\RealUp$-valued measure rows for $\mu$, the
$\sigma$-algebra closure rows, and the $\RealUp$ classifier on $\mu$-values are
preserved across the transport.
\end{theorem}
\begin{proof}
Unpack the carrier inventory. The event carrier, the measurable predicate, the
measure map, and the empty, binary-union, and countable-union rows are separate
components of the same $\mathsf{BHist}$ package.

Componentwise $\hsame$ supplies a transport witness for each component. Applying
the measurable-set component transports membership in $\mathcal{A}$, and
applying the closure-row components transports the empty, binary-union, and
countable-union witnesses.

It remains to identify the value rows. The $\mu$-congruence component sends
transported measurable events to transported $\RealUp$ values, and the
$\RealUp$ classifier of \autoref{def:real-classifier-specification} composes
those comparisons inside the codomain. Hence the classifier on $\mu$-values is
stable under the same componentwise carrier transport.
\end{proof}

\begin{theorem}[MeasureUp sigma-additive certificate obligations]
\label{thm:measureup-sigma-additive-certificate-obligations}
For a $\MeasureUp$ carrier with inventory
\autoref{def:measureup-carrier-obligation-inventory}, a $\MeasureUp$
certificate consists of the following rows:
\begin{enumerate}
  \item empty-event membership $\varnothing\in\mathcal{A}$;
  \item binary disjoint-union closure in $\mathcal{A}$;
  \item countable disjoint-union closure in $\mathcal{A}$ for
  StreamName-style measurable families;
  \item $\sigma$-additivity into the $\RealUp$ countable-sum classifier;
  \item classifier-stable head-tail readback for those countable sums.
\end{enumerate}
With these rows, \autoref{thm:measure-empty-set-zero-from-sigma-additivity} and
\autoref{thm:measure-finite-disjoint-union-additivity} are consequences of the
certificate package.
\end{theorem}
\begin{proof}
The first row gives the all-empty measurable sequence used for the zero law.
The countable disjoint-union row identifies its union as the empty event, and
the $\sigma$-additivity row compares $\mu(\varnothing)$ with the associated
$\RealUp$ countable sum.

Head-tail readback splits that sum into its head and tail. The classifier
stability theorem transports the tail back to the original all-empty sum, and
the additive $\AbGroupUp$ row then cancels the repeated summand exactly as in
\autoref{thm:measure-empty-set-zero-from-sigma-additivity}.

For the finite disjoint-union law, use the measurable sequence whose first two
terms are the given disjoint events and whose remaining terms are empty. Binary
closure supplies measurability of $A\cup B$, countable closure supplies the same
event as the countable union, and $\sigma$-additivity compares its measure with
the corresponding countable sum.

Two applications of head-tail readback expose the first two summands. The empty
tail is classified as $0_R$ by the zero law, and the additive $\AbGroupUp$ row
removes that unit. The resulting comparison is the statement of
\autoref{thm:measure-finite-disjoint-union-additivity}.
\end{proof}

\begin{theorem}[Measure finite-support countable readback]
\label{thm:measure-finite-support-countable-readback}
Let $M$ be a $\MeasureUp$ carrier as in \autoref{def:measure-carrier-specification}. Let $E_\bullet$ be a pairwise-disjoint measurable sequence such that $E_n=\varnothing$ for every $n\geq2$. Write
$$
S:=\Sigma_\mu(E_\bullet),\qquad
T:=\Sigma_\mu(E_{\bullet+1}),\qquad
U:=\Sigma_\mu(E_{\bullet+2}).
$$
If the carrier supplies classifier-stable head-tail readback, pointwise countable-sum stability, and the additive $\AbGroupUp$ row of \autoref{def:abgroup-carrier} with zero $0_R$, then
$$
S\sim_R\mu(E_0)+\mu(E_1).
$$
\end{theorem}
\begin{proof}
Apply the head-tail row to the original sequence. It gives $S\sim_R\mu(E_0)+T$. Apply the same row to the shifted sequence. It gives $T\sim_R\mu(E_1)+U$.

It remains to read back the empty tail. Let $F_n:=\varnothing$ for all $n$. Since $E_{n+2}=\varnothing$ by hypothesis, pointwise countable-sum stability gives $U\sim_R\Sigma_\mu(F_\bullet)$. The constant-empty $\sigma$-additivity row identifies $\mu(\varnothing)$ with $\Sigma_\mu(F_\bullet)$, and \autoref{thm:measure-empty-set-zero-from-sigma-additivity} identifies $\mu(\varnothing)$ with $0_R$.

Transport this comparison through the additive congruence supplied by \autoref{def:abgroup-carrier}. The right-unit law then reads $\mu(E_1)+U$ as $\mu(E_1)$, and a final congruence step reads $\mu(E_0)+T$ as $\mu(E_0)+\mu(E_1)$.
\end{proof}

\begin{theorem}[Measure certificate obligation threshold]
\label{thm:measure-certificate-obligation-threshold}
A $\MeasureUp$ NameCert obligation threshold is reached by the following paper-side data: a carrier satisfying \autoref{def:measure-carrier-specification}, a classifier satisfying \autoref{def:measure-classifier-specification}, the stability row of \autoref{thm:measure-carrier-and-classifier-stability}, the finite-support countable readback row of \autoref{thm:measure-finite-support-countable-readback}, the empty-set-zero theorem \autoref{thm:measure-empty-set-zero-from-sigma-additivity}, and the finite disjoint-union additivity theorem \autoref{thm:measure-finite-disjoint-union-additivity}. These data provide the carrier transport, classifier transport, $\sigma$-additive readback, zero endpoint, binary additivity, and visible ledger rows required by the public $\MeasureUp$ certificate surface.
\end{theorem}
\begin{proof}
The carrier and classifier definitions supply the public surface: event witnesses, measurable witnesses, endpoint rows, and classifier movement are all visible BHist and $\RealUp$ rows. The stability theorem is the transport field of the surface, with endpoint movement discharged by \autoref{def:real-classifier-specification}.

The finite-support theorem turns the countable-sum row into the binary readback needed by finite disjoint union. The empty-set-zero theorem supplies the additive zero endpoint, and the additive $\AbGroupUp$ row of \autoref{def:abgroup-carrier} supplies the congruence and unit laws used in that readback.

Finally, \autoref{thm:measure-finite-disjoint-union-additivity} packages these rows into the displayed binary additivity clause. The cancellation step used by the zero endpoint is exactly the group row cited in \autoref{thm:group-left-absorb-right-factor-unit}, so the threshold is assembled from visible certificate fields rather than from external set equality.
\end{proof}

\begin{theorem}[Measure empty set is zero]
\label{thm:measure-empty-set-zero-from-sigma-additivity}
For a $\MeasureUp$ carrier as in \autoref{sec:measure-carrier}, let $\mathcal{A}$ be its designated $\sigma$-algebra, let $\mu$ be its $\RealUp$-valued measure map, and write $x\sim_R y$ for the $\RealUp$ classifier of \autoref{def:real-classifier-specification}. Suppose $\varnothing\in\mathcal{A}$, and suppose the $\MeasureUp$ certificate supplies $\sigma$-additivity for every pairwise-disjoint measurable sequence, together with classifier-stable head-tail readback for the resulting countable $\RealUp$ sum. If the target $\RealUp$ addition is carried by its additive $\AbGroupUp$ row with zero $0_R$, then
$$
\mu(\varnothing)\sim_R 0_R.
$$
\end{theorem}
\begin{proof}
Let $E_n:=\varnothing$ for every $n:\mathbb{N}$. The hypothesis $\varnothing\in\mathcal{A}$ makes every $E_n$ measurable. The sequence is pairwise disjoint, and its union is again $\varnothing$.

Let $S$ be the countable $\RealUp$ sum $\sum_{n=0}^{\infty}\mu(E_n)$. The $\sigma$-additivity row gives
$$
\mu(\varnothing)\sim_R S.
$$
The head-tail row for this sum gives
$$
S\sim_R\mu(E_0)+\sum_{n=1}^{\infty}\mu(E_n).
$$
Since $E_0=\varnothing$, the first summand is $\mu(\varnothing)$. The tail sequence $(E_{n+1})_{n\geq0}$ is pointwise the same empty sequence as $(E_n)_{n\geq0}$, so countable-sum classifier stability gives
$$
\sum_{n=1}^{\infty}\mu(E_n)\sim_R S.
$$
Composing this with the $\sigma$-additivity comparison and transporting the right summand through addition yields
$$
S\sim_R\mu(\varnothing)+\mu(\varnothing).
$$
Composing once more with $\mu(\varnothing)\sim_R S$ gives
$$
\mu(\varnothing)\sim_R\mu(\varnothing)+\mu(\varnothing).
$$
By symmetry, $\mu(\varnothing)+\mu(\varnothing)\sim_R\mu(\varnothing)$.

The additive $\AbGroupUp$ row described in \autoref{def:abgroup-carrier} gives the group hypotheses for $+$ and $0_R$. Applying \autoref{thm:group-left-absorb-right-factor-unit} with $a=\mu(\varnothing)$ and $b=\mu(\varnothing)$ forces the right summand to the additive unit, yielding $\mu(\varnothing)\sim_R0_R$.
\end{proof}

\begin{theorem}[Measure finite disjoint-union additivity]
\label{thm:measure-finite-disjoint-union-additivity}
For a $\MeasureUp$ carrier as in \autoref{sec:measure-carrier}, let $\mathcal{A}$ be its designated $\sigma$-algebra, let $\mu$ be its $\RealUp$-valued measure map, and write $x\sim_R y$ for the $\RealUp$ classifier of \autoref{def:real-classifier-specification}. Let $A,B\in\mathcal{A}$ be disjoint, and let $A\cup B$ be the measurable binary union supplied by the $\sigma$-algebra row. Suppose $\varnothing\in\mathcal{A}$, and suppose the $\MeasureUp$ certificate supplies $\sigma$-additivity for every pairwise-disjoint measurable sequence, together with classifier-stable head-tail readback and pointwise classifier stability for the resulting countable $\RealUp$ sums. If the target $\RealUp$ addition is carried by its additive $\AbGroupUp$ row with zero $0_R$, then
$$
\mu(A\cup B)\sim_R\mu(A)+\mu(B).
$$
\end{theorem}
\begin{proof}
Define a measurable sequence by $E_0:=A$, $E_1:=B$, and $E_n:=\varnothing$ for every $n\geq2$. Each term is measurable because $A,B,\varnothing\in\mathcal{A}$. The sequence is pairwise disjoint: the only nonempty pair to check is $A$ against $B$, and this is the displayed disjointness hypothesis. Its countable union is exactly $A\cup B$.

Let
$$
S:=\sum_{n=0}^{\infty}\mu(E_n),\qquad
T:=\sum_{n=0}^{\infty}\mu(E_{n+1}),\qquad
U:=\sum_{n=0}^{\infty}\mu(E_{n+2}).
$$
The $\sigma$-additivity row gives
$$
\mu(A\cup B)\sim_R S.
$$
The first head-tail readback gives
$$
S\sim_R\mu(E_0)+T,
$$
hence $S\sim_R\mu(A)+T$. The second head-tail readback gives
$$
T\sim_R\mu(E_1)+U,
$$
hence $T\sim_R\mu(B)+U$.

It remains to identify the empty tail. Let $F_n:=\varnothing$ for every $n$, and let $S_{\varnothing}:=\sum_{n=0}^{\infty}\mu(F_n)$. Since $E_{n+2}=F_n$ for every $n$, pointwise classifier stability of the countable sum gives
$$
U\sim_R S_{\varnothing}.
$$
Applying $\sigma$-additivity to the all-empty sequence gives $\mu(\varnothing)\sim_R S_{\varnothing}$, while \autoref{thm:measure-empty-set-zero-from-sigma-additivity} gives $\mu(\varnothing)\sim_R0_R$. By symmetry and transitivity of the $\RealUp$ classifier, $S_{\varnothing}\sim_R0_R$, and therefore
$$
U\sim_R0_R.
$$

The additive $\AbGroupUp$ row of \autoref{def:abgroup-carrier} supplies addition congruence and the right-unit law. Transporting $U\sim_R0_R$ through addition gives
$$
\mu(B)+U\sim_R\mu(B)+0_R\sim_R\mu(B),
$$
so $T\sim_R\mu(B)$. Transporting this comparison through addition with the fixed left summand $\mu(A)$ gives
$$
\mu(A)+T\sim_R\mu(A)+\mu(B).
$$
Composing this with $S\sim_R\mu(A)+T$ and with $\mu(A\cup B)\sim_R S$ yields the displayed classification.
\end{proof}

\begin{theorem}[MeasureUp NameCert obligation package]
\label{thm:measure-namecert-obligation-package}
The rows displayed in
\autoref{def:measure-sigma-algebra-carrier-surface}, together with the
classifier transport of
\autoref{thm:measure-carrier-classifier-obligation}, the sigma-additivity
classification of \autoref{thm:measure-sigma-additivity-obligation}, the
ledger exactness of \autoref{thm:measure-ledger-exactness-obligation}, and the
two consequence theorems
\autoref{thm:measure-empty-set-zero-from-sigma-additivity} and
\autoref{thm:measure-finite-disjoint-union-additivity}, form the
$\MeasureUp$ NameCert obligation package. This package is the measure substrate
consumed by $\IntegralUp$, $\ProbSpaceUp$, and $\FunctionalAnalysisUp$.
\end{theorem}
\begin{proof}
Start with the carrier surface and project its visible event, value, stability,
and ledger rows. Repack those projections as the carrier and classifier side of
the $\MeasureUp$ certificate.

The carrier-classifier obligation supplies componentwise transport for every
displayed row. The sigma-additivity obligation supplies the countable-union
classification needed by the measure consequences, while the ledger exactness
obligation identifies all hidden witnesses with their visible rows.

Finally include the empty-zero and finite disjoint-union consequence theorems
as the core theorem rows of the package. The downstream chapters named in the
statement consume only this package surface: $\IntegralUp$ uses the measurable
event and countable-sum rows, $\ProbSpaceUp$ uses the empty and disjoint-union
rows, and $\FunctionalAnalysisUp$ uses the $\RealUp$ value classifier and
additive compatibility rows.
\end{proof}

\begin{definition}[Measure event row coverage]
\label{def:measure-event-row-coverage}
For the $\MeasureUp$ sigma-algebra carrier surface of
\autoref{def:measure-sigma-algebra-carrier-surface}, the event row coverage
predicate is the assertion that every row consumed by a downstream certificate
is attached to a visible $\mathsf{BHist}$ ledger row of that carrier surface.
Concretely, the covered rows are:
\begin{enumerate}
  \item every measurable event read by the measurable-family predicate
  $\mathcal{A}$;
  \item the empty-event row;
  \item every binary-union row, including its disjointness witness when the
  union is read as a disjoint union;
  \item every countable-union row for a StreamName-style measurable family,
  together with its pairwise-disjointness witness when sigma-additivity is
  invoked;
  \item every $\RealUp$ endpoint row $\mu(E)$ classified by the measure-value
  classifier;
  \item every countable-sum endpoint $\Sigma_\mu(E_\bullet)$ and every
  head-tail readback row attached to that endpoint.
\end{enumerate}
Coverage requires the attachment to be exact: the downstream consumer reads the
displayed $\mathsf{BHist}$ row, its $\hsame$ transport row, and the associated
$\RealUp$ endpoint classifier row, and it reads no additional event carrier,
choice object, quotient object, or analytic completion object from the
$\MeasureUp$ package.
\end{definition}

\begin{theorem}[Measure measurable family exactness]
\label{thm:measure-measurable-family-exactness}
For the $\MeasureUp$ sigma-algebra carrier surface, the event row coverage
predicate of \autoref{def:measure-event-row-coverage} gives coverage,
soundness, and completeness for the measurable-family classifier. Coverage
means that every measurable event, closure event, disjointness witness,
$\RealUp$ endpoint, and countable-sum endpoint used by the classifier is read
from a displayed ledger row. Soundness means that transporting a covered row
along the classifier gives the corresponding covered row in the target
presentation. Completeness means that every row available to the
measurable-family classifier is one of the rows listed in the coverage
predicate.
\end{theorem}
\begin{proof}
Unpack \autoref{def:measure-sigma-algebra-carrier-surface}. Its carrier,
stability, ledger, and exactness fields list exactly the measurable-event,
empty-event, binary-union, countable-union, disjointness, endpoint, and
countable-sum rows named in \autoref{def:measure-event-row-coverage}.

Coverage is therefore projection from the displayed carrier and ledger fields.
Each consumed event or endpoint row is attached to one of those fields by the
exactness clause of the surface.

Soundness is the transport statement of
\autoref{thm:measure-carrier-classifier-obligation}. Componentwise $\hsame$
moves the event rows, and the $\RealUp$ classifier moves the endpoint rows.
The ledger witness attached to a visible row is repacked after that visible row
has been transported.

Completeness is the exhaustion statement of
\autoref{thm:measure-ledger-exactness-obligation}. Since the ledger contains
only the listed rows and every hidden measurable-set or countable-sum witness
is attached to one of them, the measurable-family classifier has no further
row to read.
\end{proof}

\begin{theorem}[Measure probability normalization row]
\label{thm:measure-probability-normalization-row}
Let a covered $\MeasureUp$ package carry a total event row $\Omega$ whose
endpoint is classified as the probability unit:
$$
\mu(\Omega)\sim_R 1_R.
$$
Together with the empty-zero theorem
\autoref{thm:measure-empty-set-zero-from-sigma-additivity} and the finite
disjoint-union theorem
\autoref{thm:measure-finite-disjoint-union-additivity}, this row supplies the
normalization and finite-additivity rows consumed by $\ProbSpaceUp$. The claim
is only the projection of the total-event endpoint and the repacking of the
two named $\MeasureUp$ consequence theorems into the probability-space
certificate surface.
\end{theorem}
\begin{proof}
Project the total-event ledger row from the covered measure package. Its
endpoint comparison is already the normalization row required by
$\ProbSpaceUp$, namely $\mu(\Omega)\sim_R1_R$.

Next project the empty event, binary disjoint-union, endpoint, and additive
$\AbGroupUp$ rows exposed by \autoref{def:measure-event-row-coverage}. With
those rows in hand, \autoref{thm:measure-empty-set-zero-from-sigma-additivity}
gives the zero endpoint for the empty event.

Finally apply \autoref{thm:measure-finite-disjoint-union-additivity} to a
covered disjoint measurable pair. The result is exactly the finite-additivity
comparison used by the probability-space carrier; no sigma-additivity theorem
beyond the named $\MeasureUp$ package rows is imported into $\ProbSpaceUp$.
\end{proof}

\begin{theorem}[Measure integral premeasure consumption]
\label{thm:measure-integral-premeasure-consumption}
The $\IntegralUp$ premeasure interface may consume exactly the following rows
from the $\MeasureUp$ package: covered measurable-event rows, $\RealUp$
endpoint classifier rows, countable-sum endpoint rows, classifier-stable
head-tail readback rows, and the finite disjoint-union theorem
\autoref{thm:measure-finite-disjoint-union-additivity}. The same exposed
endpoint classifier and additive compatibility rows are the MeasureUp rows
available to $\FunctionalAnalysisUp$. These rows are precisely the public
surface supplied by \autoref{def:measure-certificate-surface},
\autoref{thm:measureup-sigma-additive-certificate-obligations}, and
\autoref{thm:measure-namecert-obligation-package}.
\end{theorem}
\begin{proof}
Begin with the public certificate surface of
\autoref{def:measure-certificate-surface}. Its projections provide the
measurable-family predicate, endpoint map, countable-sum endpoint, head-tail
readback row, and countable-sum classifier row.

The certificate obligation theorem
\autoref{thm:measureup-sigma-additive-certificate-obligations} identifies the
sigma-algebra rows and the sigma-additivity rows that make those projections
available to premeasure constructions. In particular it names the empty,
binary disjoint-union, countable disjoint-union, sigma-additivity, and
head-tail rows as the complete certificate rows used by the measure
consequences.

The NameCert package theorem
\autoref{thm:measure-namecert-obligation-package} repacks those rows as the
surface consumed by downstream analysis chapters. Reading that package through
\autoref{def:measure-event-row-coverage} leaves exactly the measurable-event,
endpoint, countable-sum, head-tail, and finite disjoint-union rows named in
the statement, with no additional carrier row available for $\IntegralUp$ or
$\FunctionalAnalysisUp$.
\end{proof}
