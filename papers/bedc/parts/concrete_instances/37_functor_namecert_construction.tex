\chapter{A Concrete Naming Certificate for \(\FunctorUp\)}
\label{ch:concrete-instances-functor-namecert}

The functor interface \(\FunctorUp\) has certificate obligations between two categories from \autoref{ch:concrete-instances-category-namecert} as a pair of mappings (one on objects, one on morphisms) preserving identity and composition under the morphism classifier of the codomain. It is not a primitive function space between categories; it is admitted only as a structure-preserving pair of carrier mappings.

\section{Carrier definition}

\begin{definition}[Functor carrier]
\label{def:functor-carrier}
Use a structure with object mapping, morphism mapping, and proofs preserving identity and composition.
\end{definition}

\begin{theorem}[Prefix functor preserves hom carriers]
\label{thm:functor-prefix-hom-carrier-preserves}
For the unary-continuation category, if \(p\) is unary and \(f\) is a hom-carrier from \(a\) to \(b\), then the same morphism \(f\) is a hom-carrier from \(\mathsf{append}(p,a)\) to \(\mathsf{append}(p,b)\).
\end{theorem}
\begin{proof}
Unpack the hom-carrier witness into unary source, unary target, unary morphism, and a continuation from \(a\) through \(f\) to \(b\). Unary append closure gives the prefixed source and target carriers. Continuation associativity identifies appending \(f\) after \(\mathsf{append}(p,a)\) with \(\mathsf{append}(p,b)\).
\end{proof}
\leanchecked{BEDC.Derived.FunctorUp.FunctorPrefixHomCarrier\_preserves}

\begin{theorem}[Prefix functor reflects hom carriers]
\label{thm:functor-prefix-hom-carrier-reflects}

\(\mathsf{append}(p,a)\) to \(\mathsf{append}(p,b)\), then \(f\) is a
hom-carrier from \(a\) to \(b\).
\end{theorem}
\begin{proof}
Unpack the prefixed hom-carrier witness. Unary append factorization recovers
unary membership for \(a\) and \(b\), while the morphism carrier is unchanged.

\end{proof}

$$
\mathsf{CategoryHomCarrier}(\mathsf{append}(p,a),\mathsf{append}(p,b),\emp)

\end{theorem}
\begin{proof}
Apply empty identity hom-carrier exactness in the category chapter to obtain
sameness of the two prefixed endpoints. Left cancellation for generated-history
append then reflects this sameness to the tails \(a,b\). Conversely, tail
sameness lifts to sameness of the prefixed endpoints, and category empty
identity exactness reconstructs the hom-carrier.
\end{proof}

\label{thm:functor-prefix-hom-carrier-zero-headed-component-absurd}

$$
\mathsf{CategoryHomCarrier}(\mathsf{append}(p,a),\mathsf{append}(p,b),f)
\Longrightarrow
\left(
\exists z,\ p=\Ezero(z)
\right)
\lor
\left(
\exists z,\ a=\Ezero(z)
\right)

witness contradicts the unary zero-extension separation theorem.

\section{The certificate}

\begin{definition}[Functor source specification]
\label{def:functor-source-specification}
The source is a pair of certified categories.
\end{definition}

\begin{definition}[Functor pattern specification]
\label{def:functor-pattern-specification}
The pattern is structure-preserving mapping formation over object and morphism carriers.
\end{definition}

\begin{definition}[Functor classifier specification]
\label{def:functor-classifier-specification}
The classifier relates functors pointwise on object images and morphism images using the codomain category classifiers.
\end{definition}

\begin{definition}[Functor stability certificate]
\label{def:functor-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item pointwise object-map classifier reflexivity
  \item pointwise morphism-map classifier reflexivity
  \item pointwise transitivity
  \item identity preservation
  \item composition preservation
  \item stability under source and target classifier replacement
\end{enumerate}
\end{definition}

\begin{definition}[Functor ledger policy]
\label{def:functor-ledger-policy}
Record the two category ledgers and the preservation witnesses.
\end{definition}

\begin{definition}[\(\FunctorUp\) certificate obligations]
\label{def:functor-certificate-obligations}
The carrier, source, pattern, classifier, stability, and ledger data listed above are the obligations required for the interface name \(\FunctorUp\).
\end{definition}

\paragraph{Primitive boundary.}
\(\FunctorUp\) is not a BEDC primitive. It is admitted only when those obligations are supplied.

\begin{remark}[Zero-axiom encoding note]
\label{rem:functor-zero-axiom-note}
Avoids funext by pointwise classifier sameness for object and morphism mappings.
\end{remark}

\begin{theorem}[Functor hom-carrier codomain evaluation]
\label{thm:functor-hom-carrier-codomain-evaluation}
Let \(C\) and \(E\) be certified categories, and let \(H:C\to E\) be a certified functor with object map \(H_0\). Write \(\mathsf{FunctorHomCarrier}_{H}(x,y)\) for the predicate on codomain morphism candidates \(m\) abbreviated by
\[
\mathsf{FunctorHomCarrier}_{H}(x,y,m)
\equiv
\mathsf{HomCarrier}_{E}(H_0(x),H_0(y),m).
\]
Then for every pair of source objects \(x,y\) and every codomain morphism candidate \(m\), the functor-indexed hom-carrier classifier at \(x,y\) evaluates to the codomain category hom-carrier classifier at the image endpoints:
\[
\mathsf{FunctorHomCarrier}_{H}(x,y,m)
\quad\Longleftrightarrow\quad
\mathsf{HomCarrier}_{E}(H_0(x),H_0(y),m).
\]
\end{theorem}
\begin{proof}
The functor carrier of \autoref{def:functor-carrier} supplies the object map \(H_0\). The category carrier of \autoref{def:category-carrier} supplies, in the codomain category \(E\), a hom-carrier classifier for each ordered pair of codomain objects. Applying that category classifier to the image pair \(H_0(x),H_0(y)\) is exactly the displayed abbreviation for \(\mathsf{FunctorHomCarrier}_{H}(x,y,-)\). The pointwise classifier discipline in \autoref{def:functor-classifier-specification} and \autoref{rem:functor-zero-axiom-note} reads classifier equality at the morphism argument \(m\), so unfolding the abbreviation gives the stated biconditional.
\end{proof}

\begin{theorem}[Functor composite preservation witnesses]
\label{thm:functor-composite-preservation-witnesses}
Let \(C,D,E\) be certified categories. Let \(F:C\to D\) and \(G:D\to E\) be certified \(\FunctorUp\) carriers with object maps \(F_0,G_0\) and morphism maps \(F_1,G_1\). Write \(H\) for the composite pair with
\[
H_0(x)=G_0(F_0(x)),
\qquad
H_1(u)=G_1(F_1(u)).
\]
Then \(H\) has identity and composition preservation witnesses under the morphism classifier of \(E\): for every object \(x\) of \(C\), \(H_1(\mathsf{id}_x)\) is classified with \(\mathsf{id}_{H_0(x)}\), and for every composable pair \(u:x\to y\), \(v:y\to z\) in \(C\), \(H_1(v\circ_C u)\) is classified with \(H_1(v)\circ_E H_1(u)\).
\end{theorem}
\begin{proof}
The functor carrier of \autoref{def:functor-carrier} supplies the object and morphism maps for \(F\) and \(G\). The stability certificate in \autoref{def:functor-stability-certificate} supplies identity preservation, composition preservation, and classifier-replacement stability, all read through the codomain morphism classifier. By \autoref{thm:functor-hom-carrier-codomain-evaluation}, the hom-carrier classifier for the composite pair \(H\) is the hom-carrier classifier of \(E\) at the image endpoints \(H_0(x)\) and \(H_0(y)\).

Fix an object \(x\) of \(C\). The identity field of \(F\) classifies \(F_1(\mathsf{id}_x)\) with \(\mathsf{id}_{F_0(x)}\) in \(D\). The classifier-replacement stability field of \(G\) transports that comparison through \(G_1\), giving a classifier witness in \(E\) between \(G_1(F_1(\mathsf{id}_x))\) and \(G_1(\mathsf{id}_{F_0(x)})\). The identity field of \(G\), applied at \(F_0(x)\), classifies \(G_1(\mathsf{id}_{F_0(x)})\) with \(\mathsf{id}_{G_0(F_0(x))}\). Morphism-classifier transitivity from the category stability certificate in \autoref{def:category-stability-certificate} composes the two comparisons, giving the identity witness for \(H\).

Fix composable morphisms \(u:x\to y\) and \(v:y\to z\) in \(C\). The composition field of \(F\) classifies \(F_1(v\circ_C u)\) with \(F_1(v)\circ_D F_1(u)\) in \(D\). The classifier-replacement stability field of \(G\) transports this comparison through \(G_1\), giving a classifier witness in \(E\) between \(G_1(F_1(v\circ_C u))\) and \(G_1(F_1(v)\circ_D F_1(u))\). The composition field of \(G\), applied to the composable pair \(F_1(u),F_1(v)\), classifies \(G_1(F_1(v)\circ_D F_1(u))\) with \(G_1(F_1(v))\circ_E G_1(F_1(u))\). Transitivity in \(E\) gives the composition witness for \(H\), and \autoref{def:functor-ledger-policy} records these two preservation ledgers as the composite functor ledger data.
\end{proof}

\begin{theorem}[Functor composition carries the hom-carrier classifier]
\label{thm:functor-composite-hom-carrier-classifier}
Let $C,D,E$ be certified categories. Let $F:C\to D$ and $G:D\to E$ be certified $\FunctorUp$ carriers with object maps $F_0,G_0$ and morphism maps $F_1,G_1$. Let $H=G\circ F$ be the composite carrier whose object and morphism maps are
$$
H_0(x)=G_0(F_0(x)),
\qquad
H_1(u)=G_1(F_1(u)).
$$
For every pair of source objects $x,y$ and every codomain morphism candidate $m$, the hom-carrier classifier carried by $H$ is pointwise the codomain hom-carrier classifier at the composed image endpoints:
$$
\mathsf{FunctorHomCarrier}_{G\circ F}(x,y,m)
\quad\Longleftrightarrow\quad
\mathsf{HomCarrier}_{E}(G_0(F_0(x)),G_0(F_0(y)),m).
$$
\end{theorem}
\begin{proof}
The carrier fields in \autoref{def:functor-carrier} give the maps $F_0,F_1,G_0,G_1$, and \autoref{thm:functor-composite-preservation-witnesses} supplies the identity and composition preservation witnesses for the composite pair $H$ under the morphism classifier of $E$. Thus $H$ is read as the certified functor $C\to E$ with object map $H_0$.

Apply \autoref{thm:functor-hom-carrier-codomain-evaluation} to this functor $H$. It gives
$$
\mathsf{FunctorHomCarrier}_{H}(x,y,m)
\quad\Longleftrightarrow\quad
\mathsf{HomCarrier}_{E}(H_0(x),H_0(y),m).
$$
By the displayed definition of the composite object map, the right-hand side is exactly $\mathsf{HomCarrier}_{E}(G_0(F_0(x)),G_0(F_0(y)),m)$. The classifier discipline of \autoref{def:functor-classifier-specification} and the pointwise reading in \autoref{rem:functor-zero-axiom-note} identify classifier sameness by this morphism-argument equivalence, so the composite functor carries precisely the composed hom-carrier classifier.
\end{proof}

\begin{proposition}[Functor composition identity needs morphism-sameness respect]
\label{prop:functor-composition-identity-sameness-respect-required}
Let $C,D,E$ be finite certified categories, and let $F:C\to D$ and $G:D\to E$ be carrier pairs with object maps $F_0,G_0$ and morphism maps $F_1,G_1$. Write $m\sim_D n$ and $r\sim_E s$ for the endpoint-indexed morphism classifiers, and suppose the $E$-classifier is symmetric and transitive. Assume the raw identity and raw composition preservation ledgers for $F$ and $G$ are present in the sense of \autoref{def:functor-carrier}. Fix an object $x$ of $C$ and set $H=G\circ F$ with $H_0(x)=G_0(F_0(x))$ and $H_1(u)=G_1(F_1(u))$. If
$$
F_1(\mathsf{id}_x)\sim_D \mathsf{id}_{F_0(x)}
$$
and
$$
G_1(\mathsf{id}_{F_0(x)})\sim_E \mathsf{id}_{G_0(F_0(x))}
$$
but
$$
\neg\bigl(G_1(F_1(\mathsf{id}_x))\sim_E G_1(\mathsf{id}_{F_0(x)})\bigr),
$$
then $H$ has no identity preservation witness at $x$:
$$
\neg\bigl(H_1(\mathsf{id}_x)\sim_E \mathsf{id}_{H_0(x)}\bigr).
$$
\end{proposition}
\begin{proof}
Assume that the displayed composite identity witness exists. Unfolding the composite carrier gives
$$
G_1(F_1(\mathsf{id}_x))\sim_E \mathsf{id}_{G_0(F_0(x))}.
$$
The raw identity ledger for $G$ gives
$$
G_1(\mathsf{id}_{F_0(x)})\sim_E \mathsf{id}_{G_0(F_0(x))}.
$$
Symmetry reverses the second comparison, and transitivity composes it with the first. Hence
$$
G_1(F_1(\mathsf{id}_x))\sim_E G_1(\mathsf{id}_{F_0(x)}),
$$
contradicting the displayed non-respect premise. Thus the composite identity witness cannot exist.

The comparison supplied by $F$ is only a $D$-morphism classifier witness between $F_1(\mathsf{id}_x)$ and $\mathsf{id}_{F_0(x)}$. To use the raw identity ledger for $G$, that witness must be transported through $G_1$. This is exactly the classifier-replacement stability field in \autoref{def:functor-stability-certificate} and the step used in \autoref{thm:functor-composite-preservation-witnesses}. Raw composition preservation does not supply this transport, so it cannot restore identity preservation for $G\circ F$.
\end{proof}
