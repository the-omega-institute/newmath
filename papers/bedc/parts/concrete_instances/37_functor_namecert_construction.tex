\chapter{A Concrete Naming Certificate for \(\FunctorUp\)}
\label{ch:concrete-instances-functor-namecert}

The functor interface \(\FunctorUp\) has certificate obligations between two categories from \autoref{ch:concrete-instances-category-namecert} as a pair of mappings (one on objects, one on morphisms) preserving identity and composition under the morphism classifier of the codomain. It is not a primitive function space between categories; it is admitted only as a structure-preserving pair of carrier mappings.

\section{Carrier definition}

\begin{definition}[Functor carrier]
\label{def:functor-carrier}
Use a structure with object mapping, morphism mapping, and proofs preserving identity and composition.
\end{definition}

\begin{theorem}[Prefix functor preserves hom carriers]
\label{thm:functor-prefix-hom-carrier-preserves}
For the unary-continuation category, if \(p\) is unary and \(f\) is a hom-carrier from \(a\) to \(b\), then the same morphism \(f\) is a hom-carrier from \(\mathsf{append}(p,a)\) to \(\mathsf{append}(p,b)\).
\end{theorem}
\begin{proof}
Unpack the hom-carrier witness into unary source, unary target, unary morphism, and a continuation from \(a\) through \(f\) to \(b\). Unary append closure gives the prefixed source and target carriers. Continuation associativity identifies appending \(f\) after \(\mathsf{append}(p,a)\) with \(\mathsf{append}(p,b)\).
\end{proof}
\leanchecked{BEDC.Derived.FunctorUp.FunctorPrefixHomCarrier\_preserves}

\begin{theorem}[Prefix functor reflects hom carriers]
\label{thm:functor-prefix-hom-carrier-reflects}
For the unary-continuation category, if \(f\) is a hom-carrier from
\(\mathsf{append}(p,a)\) to \(\mathsf{append}(p,b)\), then \(f\) is a
hom-carrier from \(a\) to \(b\).
\end{theorem}
\begin{proof}
Unpack the prefixed hom-carrier witness. Unary append factorization recovers
unary membership for \(a\) and \(b\), while the morphism carrier is unchanged.
The continuation field reflects across the shared prefix by continuation prefix
cancellation.
\end{proof}
\leanchecked{BEDC.Derived.FunctorUp.FunctorPrefixHomCarrier\_reflects}

\begin{theorem}[Prefix functor preserves identity hom carrier]
\label{thm:functor-prefix-hom-carrier-identity-closed}
For the unary-continuation category, if \(p\) and \(a\) are unary and
\(\ContR(\emp,\emp,id)\), then \(id\) is a hom-carrier from
\(\mathsf{append}(p,a)\) to \(\mathsf{append}(p,a)\).
\end{theorem}
\begin{proof}
Determinism of \(\ContR\), compared against the empty right unit, identifies
\(id\) with \(\emp\). Unary append closure gives the prefixed source and
target carriers. The empty morphism is unary, and the right unit continuation
at \(\mathsf{append}(p,a)\) supplies the hom-continuation witness.
\end{proof}
\leanchecked{BEDC.Derived.FunctorUp.FunctorPrefixHomCarrier\_identity\_closed}

\begin{theorem}[Prefix functor preserves hom composition]
\label{thm:functor-prefix-hom-carrier-comp-preserves}
For the unary-continuation category, if \(p\) is unary, \(f\) is a hom-carrier from \(a\) to \(b\), \(g\) is a hom-carrier from \(b\) to \(c\), and \(\ContR(f,g,fg)\), then \(fg\) is a hom-carrier from \(\mathsf{append}(p,a)\) to \(\mathsf{append}(p,c)\).
\end{theorem}
\begin{proof}
First use category hom-carrier composition closure to place \(fg\) from \(a\) to \(c\). Then apply prefix preservation to that composite hom-carrier with the unary witness for \(p\).
\end{proof}
\leanchecked{BEDC.Derived.FunctorUp.FunctorPrefixHomCarrier\_comp\_preserves}

\begin{theorem}[Prefix functor preserves associative hom composition]
\label{thm:functor-prefix-hom-carrier-comp-assoc-preserves}
For the unary-continuation category, if \(p\) is unary and \(f,g,h\) are
composable hom-carriers with composites \(fg\), \(gh\), \(left\), and
\(right\), then \(left\) and \(right\) are hom-carriers from
\(\mathsf{append}(p,a)\) to \(\mathsf{append}(p,d)\), and
\(\HSame(left,right)\).
\end{theorem}
\begin{proof}
Use category-level associative composition closure to obtain the two
unprefixed composite hom-carriers and their classifier sameness. Prefix
preservation transports each composite hom-carrier across the unary prefix
\(p\), while the classifier sameness is the same continuation associativity
witness.
\end{proof}
\leanchecked{BEDC.Derived.FunctorUp.FunctorPrefixHomCarrier\_comp\_assoc\_preserves}

\begin{theorem}[Prefix functor preserves empty identity hom carriers]
\label{thm:functor-prefix-hom-carrier-empty-identity-preserves}
For the unary-continuation category, if \(p\) and \(a\) are unary, then the
empty history is a hom-carrier from \(\mathsf{append}(p,a)\) to
\(\mathsf{append}(p,a)\).
\end{theorem}
\begin{proof}
First use the empty identity hom-carrier theorem at \(a\). Prefix preservation
then transports that hom-carrier along the unary prefix \(p\).
\end{proof}
\leanchecked{BEDC.Derived.FunctorUp.FunctorPrefixHomCarrier\_empty\_identity\_preserves}

\begin{theorem}[Prefix functor empty identity hom carrier exactness]
\label{thm:functor-prefix-hom-carrier-empty-identity-iff}
For histories \(p,a,b\), the empty history is a hom-carrier from
\(\mathsf{append}(p,a)\) to \(\mathsf{append}(p,b)\) exactly when the two
prefixed endpoints are unary and the unprefixed tails \(a,b\) are internally
the same:
$$
\mathsf{CategoryHomCarrier}(\mathsf{append}(p,a),\mathsf{append}(p,b),\emp)
\quad\Longleftrightarrow\quad
\mathsf{UnaryHistory}(\mathsf{append}(p,a))\wedge
\mathsf{UnaryHistory}(\mathsf{append}(p,b))\wedge \hsame(a,b).
$$
\end{theorem}
\begin{proof}
Apply empty identity hom-carrier exactness in the category chapter to obtain
sameness of the two prefixed endpoints. Left cancellation for generated-history
append then reflects this sameness to the tails \(a,b\). Conversely, tail
sameness lifts to sameness of the prefixed endpoints, and category empty
identity exactness reconstructs the hom-carrier.
\end{proof}
\leanchecked{BEDC.Derived.FunctorUp.FunctorPrefixHomCarrier\_empty\_identity\_iff}

\section{The certificate}

\begin{definition}[Functor source specification]
\label{def:functor-source-specification}
The source is a pair of certified categories.
\end{definition}

\begin{definition}[Functor pattern specification]
\label{def:functor-pattern-specification}
The pattern is structure-preserving mapping formation over object and morphism carriers.
\end{definition}

\begin{definition}[Functor classifier specification]
\label{def:functor-classifier-specification}
The classifier relates functors pointwise on object images and morphism images using the codomain category classifiers.
\end{definition}

\begin{definition}[Functor stability certificate]
\label{def:functor-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item pointwise object-map classifier reflexivity
  \item pointwise morphism-map classifier reflexivity
  \item pointwise transitivity
  \item identity preservation
  \item composition preservation
  \item stability under source and target classifier replacement
\end{enumerate}
\end{definition}

\begin{definition}[Functor ledger policy]
\label{def:functor-ledger-policy}
Record the two category ledgers and the preservation witnesses.
\end{definition}

\begin{definition}[\(\FunctorUp\) certificate obligations]
\label{def:functor-certificate-obligations}
The carrier, source, pattern, classifier, stability, and ledger data listed above are the obligations required for the interface name \(\FunctorUp\).
\end{definition}

\paragraph{Primitive boundary.}
\(\FunctorUp\) is not a BEDC primitive. It is admitted only when those obligations are supplied.

\begin{remark}[Zero-axiom encoding note]
\label{rem:functor-zero-axiom-note}
Avoids funext by pointwise classifier sameness for object and morphism mappings.
\end{remark}
