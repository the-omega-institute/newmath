\chapter{Naming Certificate for HaltingDistinctionLimitUp}
\label{ch:concrete-instances-halting-as-form-of-distinction-limit}

HaltingDistinctionLimitUp names the bridge object that reads the halting problem as the certificate-level limit of the form of distinction. Its rows do not supply a halting oracle; they record the finite continuation ledgers and diagonal-obstruction ledgers by which a proposed classifier is tested against the open meta loop.

\section{Source spec for HaltingDistinctionLimitUp}
\label{sec:concrete-instances-halting-as-form-of-distinction-limit-source-spec}

\begin{definition}[Source rows of HaltingDistinctionLimitUp]
\label{def:concrete-instances-halting-as-form-of-distinction-limit-source-rows}
A source row is a seven-entry continuation package
$$
(q, f, r, \ell_T, d, r_D, \ell_D)
$$
where $q$ is the encoded certificate-history query, $f$ is the finite fuel or witness surface, $r$ is the observed result, $\ell_T$ is the termination ledger, $d$ is the diagonal branch, $r_D$ is the diagonal result, and $\ell_D$ is the public obstruction ledger. The row is bridge data: it connects an ordinary $\Cont$ trace with the diagonal trace that tests the would-be halting classifier.
\end{definition}
\leandef{BEDC.Derived.HaltingDistinctionLimitUp.SourceRows}

\section{Pattern spec}
\label{sec:concrete-instances-halting-as-form-of-distinction-limit-pattern}

\begin{definition}[Pattern witness for HaltingDistinctionLimitUp]
\label{def:concrete-instances-halting-as-form-of-distinction-limit-pattern}
A source row has the HaltingDistinctionLimitUp pattern when its entries form the five displayed continuation links
$$
\begin{aligned}
\Cont(q,f,r),\quad
\Cont(q,r,\ell_T),\quad
\Cont(q,\ell_T,d),\quad
\Cont(d,q,r_D),\quad
\Cont(r_D,\ell_T,\ell_D).
\end{aligned}
$$
The first two links record a finite positive termination witness; the last three links record the diagonal branch that turns a total certificate-level classifier into an obstruction ledger. This pattern is structural and does not assert that every query has such a witness.
\end{definition}
\leandef{BEDC.Derived.HaltingDistinctionLimitUp.Pattern}

\section{Classifier spec}
\label{sec:concrete-instances-halting-as-form-of-distinction-limit-classifier}

\begin{definition}[Classifier for HaltingDistinctionLimitUp]
\label{def:concrete-instances-halting-as-form-of-distinction-limit-classifier}
Two source rows classify the same HaltingDistinctionLimitUp object when their query, fuel, observed result, termination ledger, diagonal branch, diagonal result, and obstruction ledger are respectively $\hsame$-equivalent. The classifier is intentionally componentwise: a row is not identified merely by a reported truth value, because that would smuggle in the global halting decision the chapter is locating as a limit.
\end{definition}
\leandef{BEDC.Derived.HaltingDistinctionLimitUp.Classifier}

\section{Stability certificate}
\label{sec:concrete-instances-halting-as-form-of-distinction-limit-stability}

\begin{theorem}[Stability of HaltingDistinctionLimitUp]
\label{thm:concrete-instances-halting-as-form-of-distinction-limit-stability}
The HaltingDistinctionLimitUp pattern is stable under the componentwise classifier: if two rows are classifier-equivalent and the first row carries the five-link continuation pattern, then the second row carries the same pattern. Thus the bridge is closed under replacement of a row by the same encoded query, same termination ledger, same diagonal branch, and same obstruction ledger.
\end{theorem}
\leanchecked{BEDC.Derived.HaltingDistinctionLimitUp.stability}

\section{Ledger policy}
\label{sec:concrete-instances-halting-as-form-of-distinction-limit-ledger}

\begin{definition}[Ledger for HaltingDistinctionLimitUp]
\label{def:concrete-instances-halting-as-form-of-distinction-limit-ledger}
The ledger retains the source row together with three readback surfaces: the termination trace, the diagonal trace, and the public endpoint. These are the raw histories needed to audit the bridge from a claimed halting witness to the diagonal obstruction. The ledger records evidence already present in finite $\Cont$ rows; it stores no global predicate deciding halting for all naming certificates.
\end{definition}
\leandef{BEDC.Derived.HaltingDistinctionLimitUp.Ledger}
