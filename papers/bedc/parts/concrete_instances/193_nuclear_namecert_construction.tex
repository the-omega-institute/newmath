\chapter{A Concrete Naming Certificate for $\NuclearUp$}
\label{ch:concrete-instances-nuclear-namecert}

The nuclear-operator interface $\NuclearUp$ packages compact operators between Banach spaces representable as absolutely convergent rank-one sums. It depends on $\NameCert_{\BanachUp}$ (\autoref{ch:concrete-instances-banach-namecert}) and $\NameCert_{\OperatorIdealUp}$ (\autoref{ch:concrete-instances-operatorideal-namecert}).

\section{Carrier definition}
\label{sec:nuclear-carrier}

\begin{theorem}[NuclearUp carrier obligation]\label{thm:nuclear-carrier-obligation}
The carrier for $\NuclearUp$ is a $\mathsf{BHist}$ predicate packaging a $\BanachUp$ source endpoint, an $\OperatorIdealUp$ compact-operator surface, and a unary rank-index ledger. The carried endpoint records the visible $\mathsf{BHist}$ rows, their $\Pkg$ provenance, and the rank-index observations used by the nuclear expansion; hidden representatives of the Banach spaces, compact operators, and rank-one terms are not carrier equality.
\end{theorem}
\leanchecked{BEDC.Derived.NuclearUp.NuclearCarrier\_obligation}
\begin{proof}
Unfold the carrier field of the ten-field certificate template. The named source object is $\NameCert_{\BanachUp}$, the operator surface is $\NameCert_{\OperatorIdealUp}$, and the rank index is represented as a unary $\mathsf{BHist}$ ledger row.

Packing these three rows through $\mathsf{BHist}$ and $\Pkg$ gives the visible carrier predicate. Since all remaining data are placed behind the ledger row, the carrier equality is exactly the displayed BEDC package equality and not equality of host representatives.
\end{proof}

\begin{theorem}[NuclearUp compact-operator obligation]\label{thm:nuclear-compact-operator-obligation}
The compactness row for $\NuclearUp$ is read through certified $\OperatorIdealUp$ approximation data and $\BanachUp$ classifier transport. It is not an appeal to external norm compactness: compactness is visible only when the $\OperatorIdealUp$ surface supplies the approximation row and the $\BanachUp$ endpoints transport it along the displayed $\hsame$ classifier.
\end{theorem}
\leanchecked{BEDC.Derived.NuclearUp.NuclearCompactOperator\_banach\_transport\_surface}
\begin{proof}
Unfold the compact-operator field of the certificate template after \autoref{thm:nuclear-carrier-obligation}. The available objects are the $\NameCert_{\OperatorIdealUp}$ approximation row and the $\NameCert_{\BanachUp}$ classifier on source and target histories.

The construction transports approximation observations along the $\BanachUp$ classifier and then repacks them as the compact row of the carried $\mathsf{BHist}$ endpoint. No norm-compactness principle is introduced, because the only compactness witness consumed by the template is the certified operator-ideal approximation row.
\end{proof}

\section{The certificate}
\label{sec:nuclear-certificate}

\begin{theorem}[NuclearUp rank-one-sum obligation]\label{thm:nuclear-rank-one-sum-obligation}
Rank-one expansion data for $\NuclearUp$ supplies unary $\mathsf{BHist}$ indices, coefficient and vector observations, a tail modulus, and $\Cont$-compatible partial sums. The partial-sum rows are stable under $\hsame$, so changing the displayed Banach histories by the certified classifier preserves the carried nuclear expansion package.
\end{theorem}
\leanchecked{BEDC.Derived.NuclearUp.NuclearRankOnePrefixLedger\_semantic\_name\_certificate}
\begin{proof}
Unfold the expansion field of the ten-field template. Each unary index names one rank-one row, with its scalar coefficient, source vector observation, target vector observation, and tail-modulus entry recorded as $\mathsf{BHist}$ data.

The $\Cont$ row is then applied to the finite partial sums generated by those indices. Stability follows by transporting the coefficient and vector observations along $\hsame$ before repacking the same tail-modulus ledger, so the expansion remains a carried nuclear package.
\end{proof}

\begin{theorem}[NuclearUp ledger exactness obligation]\label{thm:nuclear-ledger-exactness-obligation}
The $\NuclearUp$ ledger hides exactly rank indices, rank-one projections, and tail-modulus evidence. Its exactness row must supply coverage for every visible nuclear package, soundness for every ledger entry projected to the visible carrier, and completeness saying that no visible compact rank-one-sum package is lost by the hidden ledger representation.
\end{theorem}
\leanchecked{BEDC.Derived.NuclearUp.NuclearLedgerExactness\_obligation}
\begin{proof}
Unfold the ledger and exactness fields of the certificate template. The hidden rows are precisely the unary rank index, the rank-one projection data named by that index, and the tail-modulus witness used by the expansion row.

Coverage reads a visible nuclear package back into such ledger rows. Soundness projects any ledger row to the $\mathsf{BHist}$ carrier of \autoref{thm:nuclear-carrier-obligation}, and completeness states that the projection captures all visible compact rank-one-sum packages admitted by \autoref{thm:nuclear-compact-operator-obligation} and \autoref{thm:nuclear-rank-one-sum-obligation}.
\end{proof}

\begin{theorem}[NuclearUp dependency boundary]\label{thm:nuclear-dependency-boundary-obligation}
Downstream consumers of $\NuclearUp$ may use only the $\BanachUp$ interface, the $\OperatorIdealUp$ interface, and the rank-one ledger specified in this chapter. They may not use a trace-class principle, a spectral principle, choice, or host compactness as part of the $\NuclearUp$ certificate.
\end{theorem}
\leanchecked{BEDC.Derived.NuclearUp.NuclearDependencyBoundary\_obligation}
\begin{proof}
Unfold the dependency field of the ten-field template. The declared dependencies are exactly $\NameCert_{\BanachUp}$, $\NameCert_{\OperatorIdealUp}$, and the unary rank-one ledger rows fixed by \autoref{thm:nuclear-rank-one-sum-obligation}.

Every public observation is obtained by projecting those named dependencies through $\mathsf{BHist}$, $\Pkg$, $\hsame$, and $\Cont$. Since trace-class, spectral, choice, and host compactness principles are not dependency rows, they cannot be consumed by the certified $\NuclearUp$ surface.
\end{proof}

\begin{theorem}[NuclearUp carrier compact-operator obligation]
\label{thm:nuclear-carrier-compact-operator-obligation}
A $\NuclearUp$ carrier must expose a $\BanachUp$ source pair, an $\OperatorIdealUp$ compact-operator row, and a carrier predicate over $\mathsf{BHist}$ for operators represented by certificate-visible finite prefixes of compact maps. The source pair names the domain and codomain Banach carriers, the compact-operator row records that each visible prefix endpoint is compact in the cited operator-ideal surface, and the carrier predicate admits only histories whose operator evidence factors through those finite prefixes.
\end{theorem}
\leanchecked{BEDC.Derived.NuclearUp.NuclearCompactPrefixCarrier\_banach\_operator\_rows}

\begin{proof}
Project the $\NameCert_{\BanachUp}$ row named in the introduction to obtain the source and target Banach carriers and their classifier transport fields.

Next project the $\NameCert_{\OperatorIdealUp}$ dependency row. Its compact-operator entry is the only operator-ideal evidence exposed to the nuclear carrier at this level, so every admitted operator history must be witnessed by a finite prefix of compact maps visible in that row.

The carrier predicate is then the BEDC history predicate that stores exactly these three projections: Banach source, Banach target, and compact-prefix evidence. No host operator or infinite expansion is admitted without an accompanying finite prefix row.
\end{proof}

\begin{theorem}[NuclearUp rank-one-sum ledger obligation]
\label{thm:nuclear-rank-one-sum-ledger-obligation}
The $\NuclearUp$ pattern row records each rank-one summand, coefficient-bound witness, finite-prefix certificate, and convergence ledger entry. Consequently no unrecorded infinite sum or host sequence is used by the certificate surface.
\end{theorem}
\leanchecked{BEDC.Derived.NuclearUp.NuclearRankOnePrefixLedger\_tail\_readback}

\begin{proof}
Read the pattern row as a finite BEDC ledger. The empty prefix carries no summand and no endpoint beyond the zero prefix certified by the dependency rows.

For a nonempty prefix, project its head row. It contains the rank-one summand, its coefficient-bound witness, the finite-prefix endpoint, and the ledger entry that relates the endpoint to the previous prefix.

Apply induction over the finite prefix spine. The induction hypothesis accounts for every earlier summand and bound, while the head projection accounts for the current row. Since every accepted prefix is built by these constructors, an infinite host sequence can be referenced only through the displayed convergence ledger and never as unrecorded data.
\end{proof}

\begin{theorem}[NuclearUp OperatorIdeal and Banach dependency obligation]
\label{thm:nuclear-operatorideal-banach-dependency-obligation}
Stability and exactness for $\NuclearUp$ are scoped only to the cited $\NameCert_{\BanachUp}$ and $\NameCert_{\OperatorIdealUp}$ rows: carrier transport uses Banach classifier transport, ideal closure uses the $\OperatorIdealUp$ compactness and ideal-law rows, and exactness is the visible rank-one ledger iff the nuclear classifier.
\end{theorem}
\leanchecked{BEDC.Derived.NuclearUp.NuclearOperatorIdealBanachDependency\_rank\_rows}

\begin{proof}
For carrier transport, use only the two Banach classifiers named by the source pair. Transport of a carried nuclear operator changes the visible source or target history only along those $\BanachUp$ classifier witnesses.

For ideal closure, project the compactness and ideal-law rows from $\NameCert_{\OperatorIdealUp}$. Composition or finite linear recombination is accepted exactly when the cited operator-ideal laws produce the next compact-prefix witness.

For exactness, unfold the nuclear classifier as equality of the visible rank-one ledger modulo the Banach and operator-ideal transports just named. The forward direction projects the ledger equivalence from the classifier witness. The reverse direction repacks the same ledger equivalence, together with the Banach transport and compact-prefix witnesses, as the nuclear classifier.
\end{proof}

\begin{theorem}[NuclearUp scoped BEDC boundary]
\label{thm:nuclear-scoped-bhist-boundary}
The carrier, compact-prefix, rank-one-ledger, and dependency rows of \autoref{thm:nuclear-carrier-compact-operator-obligation}, \autoref{thm:nuclear-rank-one-sum-ledger-obligation}, and \autoref{thm:nuclear-operatorideal-banach-dependency-obligation} are scoped exactly to $\mathsf{BHist}$, $\mathsf{Pkg}$, $\hsame$, $\Cont$, $\NameCert_{\BanachUp}$, $\NameCert_{\OperatorIdealUp}$, unary rank-index rows, rank-one projection rows, compact-prefix witnesses, coefficient-bound witnesses, and tail-modulus ledger rows. No trace-class row, spectral row, choice principle, host compactness datum, or host infinite-sequence datum is part of this boundary.
\end{theorem}

\begin{proof}
Project the carrier and compact-prefix rows of \autoref{thm:nuclear-carrier-compact-operator-obligation}. The accepted data are the $\mathsf{BHist}$ carrier predicate, its $\mathsf{Pkg}$ provenance, Banach classifier transport through $\hsame$, the $\OperatorIdealUp$ compact row, and the compact-prefix witnesses exposed by that operator-ideal surface.

Project the pattern row of \autoref{thm:nuclear-rank-one-sum-ledger-obligation}. Each admitted prefix contributes only a unary rank-index row, a rank-one projection row, a coefficient-bound witness, a compact-prefix endpoint, and a tail-modulus ledger row, with $\Cont$ used only for the finite partial-sum observation named by those rows.

Finally project the dependency rows of \autoref{thm:nuclear-operatorideal-banach-dependency-obligation}. They name exactly $\NameCert_{\BanachUp}$ and $\NameCert_{\OperatorIdealUp}$ as external certificate inputs, so every accepted dependency is already one of the displayed BEDC rows. Since trace-class, spectral, choice, host compactness, and host infinite-sequence data are never produced by these projections, they are outside the scoped boundary.
\end{proof}
