\begin{lemma}[Directed meet associativity comparison from bounds]
\label{lem:lattice-directed-meet-associativity-comparison-from-bounds}
Let $A$ be a $\LatticeUp$ carrier with order $\le$ and classifier $\sim_C$. Assume the inherited $\PreorderUp$ transitivity field from \autoref{def:preorder-stability-certificate} and the directional meet lower-bound and greatest-lower-bound implication fields from \autoref{def:lattice-stability-certificate}. Then for all $x,y,z\in A$,
$$
(x\wedge y)\wedge z\le x\wedge(y\wedge z).
$$
\end{lemma}
\begin{proof}
Put
$$
w=(x\wedge y)\wedge z.
$$
The meet lower-left field gives
$$
w\le x\wedge y,
$$
and the meet lower-left field applied to $x\wedge y$ gives
$$
x\wedge y\le x.
$$
By the inherited preorder transitivity field,
$$
w\le x.
$$
The meet lower-right field applied to $x\wedge y$ gives
$$
x\wedge y\le y,
$$
so transitivity with $w\le x\wedge y$ gives
$$
w\le y.
$$
The meet lower-right field applied to $w=(x\wedge y)\wedge z$ gives
$$
w\le z.
$$
Applying meet greatestness to the two bounds $w\le y$ and $w\le z$ gives
$$
w\le y\wedge z.
$$
A second application of meet greatestness, now to $w\le x$ and $w\le y\wedge z$, gives
$$
w\le x\wedge(y\wedge z).
$$
Substituting the displayed definition of $w$ is the required comparison.
\end{proof}

\begin{lemma}[Directed join associativity comparison from bounds]\label{lem:lattice-directed-join-associativity-comparison-from-bounds} Let $A$ be a $\LatticeUp$ carrier with order $\le_C$ and classifier $\sim_C$. Assume the inherited $\PreorderUp$ transitivity field from \autoref{def:preorder-stability-certificate} and the directional join upper-bound and least-upper-bound implication fields from \autoref{def:lattice-stability-certificate}. Then for all $x,y,z\in A$, $(x\vee y)\vee z\le_C x\vee(y\vee z)$.
\end{lemma}
\begin{proof} Put $w=x\vee(y\vee z)$. The join upper-left field gives $x\le_C w$. The join upper-left field applied to $y\vee z$ gives $y\le_C y\vee z$, and the join upper-right field applied to $w=x\vee(y\vee z)$ gives $y\vee z\le_C w$. By the inherited preorder transitivity field, $y\le_C w$. Applying join leastness to the two bounds $x\le_C w$ and $y\le_C w$ gives $x\vee y\le_C w$. The join upper-right field applied to $y\vee z$ gives $z\le_C y\vee z$, and transitivity with $y\vee z\le_C w$ gives $z\le_C w$. A second application of join leastness, now to $x\vee y\le_C w$ and $z\le_C w$, gives $(x\vee y)\vee z\le_C w$. Substituting the definition of $w$ gives $(x\vee y)\vee z\le_C x\vee(y\vee z)$.
\end{proof}
