\begin{lemma}[Directed meet associativity comparison from bounds]
\label{lem:lattice-directed-meet-associativity-comparison-from-bounds}
Let $A$ be a $\LatticeUp$ carrier with order $\le$ and classifier $\sim_C$. Assume the inherited $\PreorderUp$ transitivity field from \autoref{def:preorder-stability-certificate} and the directional meet lower-bound and greatest-lower-bound implication fields from \autoref{def:lattice-stability-certificate}. Then for all $x,y,z\in A$,
$$
(x\wedge y)\wedge z\le x\wedge(y\wedge z).
$$
\end{lemma}
\begin{proof}
Put
$$
w=(x\wedge y)\wedge z.
$$
The meet lower-left field gives
$$
w\le x\wedge y,
$$
and the meet lower-left field applied to $x\wedge y$ gives
$$
x\wedge y\le x.
$$
By the inherited preorder transitivity field,
$$
w\le x.
$$
The meet lower-right field applied to $x\wedge y$ gives
$$
x\wedge y\le y,
$$
so transitivity with $w\le x\wedge y$ gives
$$
w\le y.
$$
The meet lower-right field applied to $w=(x\wedge y)\wedge z$ gives
$$
w\le z.
$$
Applying meet greatestness to the two bounds $w\le y$ and $w\le z$ gives
$$
w\le y\wedge z.
$$
A second application of meet greatestness, now to $w\le x$ and $w\le y\wedge z$, gives
$$
w\le x\wedge(y\wedge z).
$$
Substituting the displayed definition of $w$ is the required comparison.
\end{proof}

\begin{lemma}[Directed join associativity comparison from bounds]\label{lem:lattice-directed-join-associativity-comparison-from-bounds} Let $A$ be a $\LatticeUp$ carrier with order $\le_C$ and classifier $\sim_C$. Assume the inherited $\PreorderUp$ transitivity field from \autoref{def:preorder-stability-certificate} and the directional join upper-bound and least-upper-bound implication fields from \autoref{def:lattice-stability-certificate}. Then for all $x,y,z\in A$, $(x\vee y)\vee z\le_C x\vee(y\vee z)$.
\end{lemma}
\begin{proof} Put $w=x\vee(y\vee z)$. The join upper-left field gives $x\le_C w$. The join upper-left field applied to $y\vee z$ gives $y\le_C y\vee z$, and the join upper-right field applied to $w=x\vee(y\vee z)$ gives $y\vee z\le_C w$. By the inherited preorder transitivity field, $y\le_C w$. Applying join leastness to the two bounds $x\le_C w$ and $y\le_C w$ gives $x\vee y\le_C w$. The join upper-right field applied to $y\vee z$ gives $z\le_C y\vee z$, and transitivity with $y\vee z\le_C w$ gives $z\le_C w$. A second application of join leastness, now to $x\vee y\le_C w$ and $z\le_C w$, gives $(x\vee y)\vee z\le_C w$. Substituting the definition of $w$ gives $(x\vee y)\vee z\le_C x\vee(y\vee z)$.
\end{proof}

\begin{lemma}[Meet absorbs a smaller ordered endpoint]
\label{lem:lattice-meet-absorbs-smaller-ordered-endpoint}
Let $A$ be a $\LatticeUp$ carrier with order $\le_C$ and classifier $\sim_C$. Assume the inherited $\PreorderUp$ reflexivity field from \autoref{def:preorder-stability-certificate}, the inherited $\POSetUp$ antisymmetry field from \autoref{def:poset-stability-certificate}, and the directional meet lower-bound and greatest-lower-bound fields from \autoref{def:lattice-stability-certificate}. If $x,z\in A$ and $x\le_C z$, then
$$
z\wedge x\sim_C x.
$$
\end{lemma}
\begin{proof}
The meet lower-right field gives
$$
z\wedge x\le_C x.
$$
The inherited preorder reflexivity field gives
$$
x\le_C x.
$$
Together with the assumed comparison $x\le_C z$, meet greatestness applied with witness endpoint $x$ and operands $z$ and $x$ gives
$$
x\le_C z\wedge x.
$$
The antisymmetry field from \autoref{def:poset-stability-certificate} applied to these two comparisons yields
$$
z\wedge x\sim_C x.
$$
\end{proof}

\begin{lemma}[Modular lower comparison from directional bounds]
\label{lem:lattice-modular-lower-comparison-from-directional-bounds}
Let $A$ be a $\LatticeUp$ carrier with order $\le_C$. Assume the inherited $\PreorderUp$ transitivity field from \autoref{def:preorder-stability-certificate} and the directional meet lower-bound, meet greatestness, join upper-bound, and join leastness fields from \autoref{def:lattice-stability-certificate}. If $x,y,z\in A$ and $x\le_C z$, then
$$
x\vee(y\wedge z)\le_C (x\vee y)\wedge z.
$$
\end{lemma}
\begin{proof}
Put
$$
n=x\vee(y\wedge z),\qquad m=(x\vee y)\wedge z.
$$
The join upper-left field gives
$$
x\le_C x\vee y.
$$
The meet lower-left field gives
$$
y\wedge z\le_C y,
$$
and the join upper-right field gives
$$
y\le_C x\vee y.
$$
By transitivity,
$$
y\wedge z\le_C x\vee y.
$$
Join leastness with upper target $x\vee y$ gives
$$
n\le_C x\vee y.
$$
The assumption gives $x\le_C z$, and the meet lower-right field gives
$$
y\wedge z\le_C z.
$$
Join leastness with upper target $z$ gives
$$
n\le_C z.
$$
Meet greatestness applied to the two bounds $n\le_C x\vee y$ and $n\le_C z$ gives
$$
n\le_C m.
$$
Substituting the displayed definitions of $m$ and $n$ gives the claimed comparison.
\end{proof}

\begin{theorem}[Distributivity gives the modular law]
\label{thm:lattice-distributivity-implies-modular-law}
Let $A$ be a $\LatticeUp$ carrier with order $\le_C$ and classifier $\sim_C$. Assume the classifier symmetry and transitivity fields from \autoref{def:lean-carrier-naming-certificate}, the inherited $\PreorderUp$ reflexivity and transitivity fields from \autoref{def:preorder-stability-certificate}, the inherited $\POSetUp$ antisymmetry field from \autoref{def:poset-stability-certificate}, and the directional meet and join bound-characterization fields from \autoref{def:lattice-stability-certificate}. Assume also the join classifier-congruence field from \autoref{def:lattice-classifier-specification}, and suppose the distributive law
$$
a\wedge(b\vee c)\sim_C (a\wedge b)\vee(a\wedge c)
$$
holds for all $a,b,c\in A$. If $x,y,z\in A$ and $x\le_C z$, then
$$
x\vee(y\wedge z)\le_C (x\vee y)\wedge z
\qquad\text{and}\qquad
x\vee(y\wedge z)\sim_C (x\vee y)\wedge z.
$$
\end{theorem}
\begin{proof}
The first conclusion is \autoref{lem:lattice-modular-lower-comparison-from-directional-bounds}.

Put
$$
m=(x\vee y)\wedge z,
\qquad
n=x\vee(y\wedge z).
$$
By \autoref{thm:lattice-commutativity-from-directional-bounds}, applied to the meet operands $x\vee y$ and $z$,
$$
m\sim_C z\wedge(x\vee y).
$$
The assumed distributive law with $a=z$, $b=x$, and $c=y$ gives
$$
z\wedge(x\vee y)\sim_C (z\wedge x)\vee(z\wedge y).
$$
By \autoref{lem:lattice-meet-absorbs-smaller-ordered-endpoint} and the assumption $x\le_C z$,
$$
z\wedge x\sim_C x.
$$
By \autoref{thm:lattice-commutativity-from-directional-bounds}, applied to the meet operands $z$ and $y$,
$$
z\wedge y\sim_C y\wedge z.
$$
Join classifier congruence applied to the last two classifier comparisons gives
$$
(z\wedge x)\vee(z\wedge y)\sim_C x\vee(y\wedge z).
$$
Classifier transitivity composes the three displayed classifier comparisons into
$$
m\sim_C n.
$$
Classifier symmetry reverses this comparison, giving
$$
n\sim_C m.
$$
Substituting the displayed definitions of $m$ and $n$ gives
$$
x\vee(y\wedge z)\sim_C (x\vee y)\wedge z.
$$
\end{proof}
