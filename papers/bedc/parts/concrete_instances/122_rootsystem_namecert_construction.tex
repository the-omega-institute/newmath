\chapter{A Concrete Naming Certificate for $\RootSystemUp$}
\label{ch:concrete-instances-rootsystem-namecert}
\origin{human}

The root-system interface $\RootSystemUp$ packages a finite set of nonzero vectors in a finite-dimensional inner-product space closed under the reflections they generate, with integer Cartan pairings. It depends on $\NameCert_{\InnerProductUp}$ (\autoref{ch:concrete-instances-innerproduct-namecert}), $\NameCert_{\FinSetUp}$ (\autoref{ch:concrete-instances-finset-namecert}), and $\NameCert_{\IntUp}$ (\autoref{ch:concrete-instances-int-namecert}), and is the combinatorial backbone of semisimple Lie-algebra classification.

\section{Carrier definition}
\label{sec:rootsystem-carrier}

\begin{definition}[RootSystem BHist certificate surface]
\label{def:rootsystem-bhist-certificate-surface}
A $\RootSystemUp$ BHist certificate surface is a displayed package with the
following ten fields:
\begin{enumerate}
  \item the traditional target is a finite root system: a finite set of
  nonzero vectors in an inner-product vector carrier, closed under the
  displayed root reflections, with integer Cartan pairings;
  \item the BEDC source is the product of the visible
  $\InnerProductUp$, $\FinSetUp$, and $\IntUp$ certificate surfaces;
  \item the vector carrier $V:\mathsf{BHist}\to\mathsf{Prop}$ and vector
  classifier $\sim_V$ are inherited from the $\InnerProductUp$ source;
  \item the root carrier is a finite root-row predicate
  $R:\mathsf{BHist}\to\mathsf{Prop}$ over $V$, presented by a
  $\FinSetUp$ enumeration whose entries are carried by $V$;
  \item the public root classifier combines the inner-product vector
  classifier $\sim_V$ with the finite-set membership classifier, so two
  root rows match only when their displayed vector representatives and
  finite-support rows match;
  \item each root row records a nonzero witness, expressed as failure of the
  vector-zero classification under $\sim_V$;
  \item for carried roots $\alpha$ and $\beta$, the reflection row displays a
  carried vector $s_\alpha(\beta)$ together with a $\Cont$ route witnessing
  that the row is computed from the inner-product and vector-space
  operations;
  \item for carried roots $\alpha$ and $\beta$, the Cartan endpoint row
  displays an $\IntUp$-carried history
  $c_{\alpha,\beta}$ for the pairing
  $2\langle\beta,\alpha\rangle/\langle\alpha,\alpha\rangle$;
  \item the stability rows transport nonzero witnesses, reflection closure,
  Cartan pairing endpoints, and ledger descent through the root classifier
  and $\hsame$;
  \item the ledger records finite support, nonzero witnesses, reflection
  routes, and integer Cartan endpoints, with every row descending to the
  displayed sources named in the second field.
\end{enumerate}
The surface therefore names a finite root-row certificate inside BEDC
histories. It does not name an ambient Lie algebra, a Weyl group, or a
classification theorem.
\end{definition}
\leanchecked{BEDC.Derived.RootSystemUp.RootSystemTasteGate\_single\_carrier\_alignment}

\section{The certificate}
\label{sec:rootsystem-certificate}

\begin{theorem}[RootSystem nonzero finite carrier obligation]
\label{thm:rootsystem-nonzero-finiteness-obligation}
For a $\RootSystemUp$ BHist certificate surface, the public root carrier
exposes only the $\FinSetUp$ finite row $R$ of nonzero
$\InnerProductUp$ vectors. Every displayed root history is an entry of the
finite-support row, is carried by the inner-product vector carrier $V$, and
has a recorded witness that it is not classified as the vector zero by
$\sim_V$.
\end{theorem}
\begin{proof}
Project the finite root-row predicate from
\autoref{def:rootsystem-bhist-certificate-surface}. The fourth field gives
the $\FinSetUp$ enumeration, and its entries are explicitly carried by the
vector carrier inherited from $\InnerProductUp$.

Project the nonzero row from the sixth field. It is attached to each displayed
entry of the finite-support row and is stated through the vector classifier
$\sim_V$. Thus a public root has no access path except through the finite
enumeration and the nonzero witness recorded beside that entry.
\end{proof}
\leanchecked{BEDC.Derived.RootSystemUp.RootSystemFiniteSupportCarrier\_nonzero\_rows}
\leanchecked{BEDC.Derived.RootSystemUp.RootSystemFiniteSupportCarrier\_project\_rows}

\begin{theorem}[RootSystem carrier classifier obligation]
\label{thm:rootsystem-carrier-classifier-obligation}
For a $\RootSystemUp$ BHist certificate surface, root-carrier transport is
exactly transport along the inherited vector classifier and the finite-set
membership classifier. If two root rows have $\hsame$ finite-support entries
and their vector representatives are classified by $\sim_V$, then all visible
carrier data of the two root rows agree by the public root classifier.
\end{theorem}
\begin{proof}
Unpack the classifier field of
\autoref{def:rootsystem-bhist-certificate-surface}. It has two visible
components: the $\InnerProductUp$ vector classifier on representatives and
the $\FinSetUp$ classifier on the finite membership presentation.

Use the $\hsame$ witness on the finite-support entries to identify the
displayed rows selected from the enumeration. Use the vector classifier
witness to transport the inherited vector-carrier facts and the attached
nonzero witness. Combining these two projections is precisely the public
root classifier described by the fifth field.
\end{proof}
\leanchecked{BEDC.Derived.RootSystemUp.RootSystemFiniteSupportCarrier\_classifier\_transport}

\begin{theorem}[RootSystem reflection closure obligation]
\label{thm:rootsystem-reflection-closure-obligation}
For a $\RootSystemUp$ BHist certificate surface and carried roots
$\alpha$ and $\beta$, each displayed reflection row sends $\beta$ to a
carried root $s_\alpha(\beta)$. The closure witness uses the recorded
$\Cont$ route for the reflected vector and the inherited inner-product vector
classifier transport; no external reflection action is added.
\end{theorem}
\begin{proof}
Read the reflection row from the seventh field of
\autoref{def:rootsystem-bhist-certificate-surface}. It supplies both the
reflected vector history $s_\alpha(\beta)$ and the $\Cont$ route expressing
how that history is built from the displayed inner-product and vector-space
operations.

Apply the vector-carrier and classifier transport rows inherited from
$\InnerProductUp$ along that route. The reflection-closure component of the
surface then identifies the reflected vector with an entry of the finite root
row. Since membership is classified by the public root classifier, the result
is a carried root row rather than a host-level vector equality.
\end{proof}
\leanchecked{BEDC.Derived.RootSystemUp.RootSystemReflectionClosure\_result\_unary}

\begin{theorem}[RootSystem Cartan ledger obligation]
\label{thm:rootsystem-cartan-ledger-obligation}
For a $\RootSystemUp$ BHist certificate surface and carried roots
$\alpha$ and $\beta$, each Cartan endpoint $c_{\alpha,\beta}$ is an
$\IntUp$-carried history. If the input root rows are transported by the
public root classifier, the corresponding Cartan endpoints transport through
the $\IntUp$ classifier and the ledger records the same integer endpoint row.
\end{theorem}
\begin{proof}
Project the Cartan endpoint row from the eighth field of
\autoref{def:rootsystem-bhist-certificate-surface}. Its codomain is the
integer-history carrier supplied by $\NameCert_{\IntUp}$, so the endpoint is
not merely a scalar quotient in the inner-product source.

Transport the two input roots through the public root classifier. The
stability field carries the numerator, denominator, and displayed integer
endpoint through the inherited vector and scalar classifiers and then through
the $\IntUp$ classifier. The ledger component records this transported
endpoint beside the same pair of root rows, which gives stable integer Cartan
data.
\end{proof}
\leanchecked{BEDC.Derived.RootSystemUp.RootSystemCartanLedger\_cont\_transport}

\begin{theorem}[RootSystem NameCert obligation surface]
\label{thm:rootsystem-namecert-obligation-surface}
The nonzero finite carrier obligation of
\autoref{thm:rootsystem-nonzero-finiteness-obligation}, the carrier
classifier obligation of
\autoref{thm:rootsystem-carrier-classifier-obligation}, the reflection
closure obligation of
\autoref{thm:rootsystem-reflection-closure-obligation}, and the Cartan ledger
obligation of \autoref{thm:rootsystem-cartan-ledger-obligation} assemble the
paper-side $\NameCert_{\RootSystemUp}$ obligation surface. This surface does
not claim Weyl-group closure, Lie-algebra classification, or standard bridge
closure.
\end{theorem}
\begin{proof}
Start from the certificate surface of
\autoref{def:rootsystem-bhist-certificate-surface}. The finite root-row
carrier and its nonzero witnesses are supplied by
\autoref{thm:rootsystem-nonzero-finiteness-obligation}.

The public classifier and its $\hsame$ transport behavior are supplied by
\autoref{thm:rootsystem-carrier-classifier-obligation}. The reflection rows
and their closure under carried roots are supplied by
\autoref{thm:rootsystem-reflection-closure-obligation}.

Finally, the integer Cartan endpoints and their ledger stability are supplied
by \autoref{thm:rootsystem-cartan-ledger-obligation}. These four projections
cover carrier, classifier, stability, and ledger data for the naming
certificate. The listed exclusions are separate downstream interfaces, so
they are not packaged into this $\RootSystemUp$ surface.
\end{proof}

\label{thm:rootsystem-mature-bridge-consumption}

reflection closure, and Cartan ledger rows of

closure, and integer Cartan ledger boundary for the $\RootSystemUp$ source.

The $\WeylGroupUp$ standard bridge proof rows then consume those same rows:
\autoref{thm:weylgroup-rootsystem-group-carrier-row} reads the carrier and

Finally, \autoref{thm:weylgroup-standard-reflection-group-bridge} and
\autoref{thm:weylgroup-root-system-terminal-standard-bridge} package exactly
those consumed rows as the terminal standard bridge surface. The two excluded
classification claims are not fields of those bridge rows, so they remain
outside the $\RootSystemUp$ closure claim.

\begin{theorem}[RootSystem simple-reflection involution row]
\label{thm:rootsystem-simple-reflection-involution-row}
For a $\RootSystemUp$ BHist certificate surface, let $\alpha$ and $\beta$ be carried root rows, and let the displayed reflection row along $\alpha$ be read through its Cartan-normalized $\Cont$ route
$$
s_\alpha(\gamma)
\sim_V
\gamma-c_{\alpha,\gamma}\cdot_V\alpha
$$
for every carried input $\gamma$, where $c_{\alpha,\gamma}$ is the integer Cartan endpoint supplied by the same surface. Then the iterated displayed row is classified with the original input root:
$$
s_\alpha(s_\alpha(\beta))\sim_V\beta.
$$
The witness uses only the inherited $\InnerProductUp$ vector classifier, the displayed vector-space rows, the reflection $\Cont$ route, and the Cartan integer endpoint ledger.
\end{theorem}
\leanchecked{BEDC.Derived.RootSystemUp.RootSystemReflectionClosure\_involution\_row}
\begin{proof}
By \autoref{thm:rootsystem-reflection-closure-obligation}, the first reflected row
$s_\alpha(\beta)$ is a carried root, so the reflection row may be applied to it again. The same theorem keeps both displayed reflection endpoints inside the public root carrier, and \autoref{thm:rootsystem-carrier-classifier-obligation} says that their comparison is the inherited vector classifier $\sim_V$ together with the finite-support classifier.

Write $c=c_{\alpha,\beta}$. The Cartan-normalized reflection route for the first step gives the vector-classifier row
$$
s_\alpha(\beta)
\sim_V
\beta-c\cdot_V\alpha.
$$
The Cartan endpoint is the displayed integer history for
$$
c=2\langle\beta,\alpha\rangle/\langle\alpha,\alpha\rangle
$$
by \autoref{thm:rootsystem-cartan-ledger-obligation}. Apply the scalar projection and form-law rows of the $\InnerProductUp$ root basis, namely \autoref{thm:innerproduct-root-scalar-projection-transport} and \autoref{thm:innerproduct-root-unblock-form-laws}, to the carried vectors $\alpha$ and $\beta-c\cdot_V\alpha$. Sesquilinearity gives the retained scalar-classifier computation
$$
\begin{aligned}
2\langle s_\alpha(\beta),\alpha\rangle
&\sim_{\mathsf{scal}}
2\langle\beta-c\cdot_V\alpha,\alpha\rangle \\
&\sim_{\mathsf{scal}}
2\langle\beta,\alpha\rangle-2c\langle\alpha,\alpha\rangle \\
&\sim_{\mathsf{scal}}
-c\langle\alpha,\alpha\rangle.
\end{aligned}
$$
The last comparison uses the defining Cartan endpoint row for $c$, transported through the retained scalar classifier. Since $\alpha$ is a root, \autoref{thm:rootsystem-nonzero-finiteness-obligation} supplies the nonzero witness for $\alpha$, and the positive-definite exactness in \autoref{thm:innerproduct-root-diagonal-zero-exactness} prevents the diagonal denominator $\langle\alpha,\alpha\rangle$ from being scalar-zero. Thus the displayed Cartan endpoint for the reflected input is classified as
$$
c_{\alpha,s_\alpha(\beta)}\sim_{\IntUp}-c.
$$
This is an integer-ledger comparison, not a host scalar equality, because \autoref{thm:rootsystem-cartan-ledger-obligation} keeps Cartan endpoints in the $\IntUp$ carrier and transports them through its classifier.

Now apply the reflection route a second time:
$$
s_\alpha(s_\alpha(\beta))
\sim_V
s_\alpha(\beta)-c_{\alpha,s_\alpha(\beta)}\cdot_V\alpha.
$$
Substitute the first reflection row and the Cartan comparison just obtained, using vector addition, scalar multiplication, additive inverse, and classifier congruence from the inherited $\VecSpaceUp$ rows exposed by \autoref{thm:innerproduct-root-unblock-carrier-classifier}. The right endpoint is classified as
$$
\begin{aligned}
(\beta-c\cdot_V\alpha)-(-c)\cdot_V\alpha
&\sim_V
\beta+((-c)+c)\cdot_V\alpha \\
&\sim_V
\beta+0\cdot_V\alpha \\
&\sim_V
\beta.
\end{aligned}
$$
Composing these vector-classifier witnesses by the public root classifier gives
$s_\alpha(s_\alpha(\beta))\sim_V\beta$. The proof has used the two displayed reflection $\Cont$ routes, the Cartan endpoint ledger, the inner-product form laws, and the inherited vector-space classifier congruence; no ambient Weyl group, host-level equality of vectors, or quotient of reflection words is introduced.
\end{proof}

\begin{theorem}[RootSystem simple reflection negates its axis]
\label{thm:rootsystem-simple-reflection-axis-negation}
For a $\RootSystemUp$ BHist certificate surface, let $\alpha$ be a carried root row. Let the displayed reflection row along $\alpha$ be read through its Cartan-normalized $\Cont$ route
$$
s_\alpha(\gamma)
\sim_V
\gamma-c_{\alpha,\gamma}\cdot_V\alpha
$$
for every carried input $\gamma$, where $c_{\alpha,\gamma}$ is the integer Cartan endpoint supplied by the same surface. Then the reflection of $\alpha$ through its own displayed root axis is classified with the vector-space additive inverse of $\alpha$:
$$
s_\alpha(\alpha)\sim_V -_V\alpha.
$$
The witness is internal to the $\RootSystemUp$ surface: it uses the inherited $\InnerProductUp$ vector classifier, the displayed vector-space additive rows, the reflection $\Cont$ route, and the Cartan endpoint ledger.
\end{theorem}
\leanchecked{BEDC.Derived.RootSystemUp.RootSystemReflectionClosure\_axis\_negation}
\begin{proof}
By \autoref{thm:rootsystem-nonzero-finiteness-obligation}, the carried root row $\alpha$ is a carried vector endpoint in the inherited $\InnerProductUp$ vector carrier and carries a witness that $\alpha$ is not classified with the vector zero $0_V$. Therefore the vector-space reduct carries $-_V\alpha$ and all displayed endpoints formed from $\alpha$ by addition and scalar action.

Read the reflection row from the seventh field of \autoref{def:rootsystem-bhist-certificate-surface} and specialize its displayed route to the carried input $\gamma=\alpha$. This gives
$$
s_\alpha(\alpha)
\sim_V
\alpha-c_{\alpha,\alpha}\cdot_V\alpha.
$$
It remains to identify the diagonal Cartan endpoint $c_{\alpha,\alpha}$ with the carried integer $2$ and then simplify inside the vector-space reduct.

The eighth field of \autoref{def:rootsystem-bhist-certificate-surface}, read through \autoref{thm:rootsystem-cartan-ledger-obligation}, says that $c_{\alpha,\alpha}$ is the $\IntUp$ endpoint for
$$
2\langle\alpha,\alpha\rangle/\langle\alpha,\alpha\rangle.
$$
The root form-law surface of \autoref{thm:innerproduct-root-unblock-form-laws} places the diagonal endpoint $\langle\alpha,\alpha\rangle$ in the retained scalar source. Since $\alpha$ is not vector-zero, the exact diagonal zero test of \autoref{thm:innerproduct-root-diagonal-zero-exactness} prevents $\langle\alpha,\alpha\rangle$ from being scalar-zero. Hence the scalar field cancellation row applies to the displayed nonzero diagonal denominator and classifies
$$
2\langle\alpha,\alpha\rangle/\langle\alpha,\alpha\rangle
\sim_{\mathsf{scal}}
2.
$$
Transporting this scalar comparison through the integer endpoint ledger, by the Cartan endpoint transport component of \autoref{thm:rootsystem-cartan-ledger-obligation}, gives the Cartan-classifier comparison
$$
c_{\alpha,\alpha}\sim_{\IntUp}2.
$$
This is a comparison of displayed Cartan endpoint histories, not a host-level scalar equality.

Use the inherited $\VecSpaceUp$ rows exposed by \autoref{thm:innerproduct-root-vecspace-scalar-exposure}. The scalar-action congruence field listed in \autoref{def:module-stability-certificate}, inherited by \autoref{def:vecspace-stability-certificate}, transports the Cartan comparison to
$$
c_{\alpha,\alpha}\cdot_V\alpha
\sim_V
2\cdot_V\alpha.
$$
The scalar-addition distributivity and unit-action rows listed in \autoref{def:module-stability-certificate}, inherited by \autoref{def:vecspace-stability-certificate}, give the vector-space doubling comparison between $2\cdot_V\alpha$ and $\alpha+_V\alpha$, and subtraction is addition by the additive inverse. Thus
$$
\begin{aligned}
\alpha-c_{\alpha,\alpha}\cdot_V\alpha
&\sim_V \alpha-2\cdot_V\alpha \\
&\sim_V \alpha+_V(-_V(\alpha+_V\alpha)) \\
&\sim_V \alpha+_V((-_V\alpha)+_V(-_V\alpha)) \\
&\sim_V (\alpha+_V(-_V\alpha))+_V(-_V\alpha) \\
&\sim_V 0_V+_V(-_V\alpha) \\
&\sim_V -_V\alpha.
\end{aligned}
$$
The displayed steps use only additive inverse congruence, associativity, commutativity where needed to expose the inverse pair, the additive inverse law, and the additive unit law in the inherited vector-space reduct.

Composing the specialized reflection route with this vector-space simplification by transitivity of the public vector classifier gives
$$
s_\alpha(\alpha)\sim_V -_V\alpha.
$$
By \autoref{thm:rootsystem-reflection-closure-obligation}, the left endpoint is also a carried root row, while the comparison itself remains the inherited vector classifier comparison required by the statement.
\end{proof}
