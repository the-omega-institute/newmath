\chapter{A Concrete Naming Certificate for $\LieGroupUp$}
\label{ch:concrete-instances-liegroup-namecert}

The Lie-group interface $\LieGroupUp$ packages a group carrier whose underlying set is a manifold and whose multiplication and inverse are constructively smooth. It depends on $\NameCert_{\GroupUp}$ (\autoref{ch:concrete-instances-group-namecert}) and $\NameCert_{\ManifoldUp}$ (\autoref{ch:concrete-instances-manifold-namecert}), and is the substrate for representation, exponential, and root-system certificates.

\section{Carrier definition}
\label{sec:liegroup-carrier}

\begin{definition}[Singleton smooth group carrier]
\label{def:liegroup-singleton-smooth-group-carrier}
The concrete $\LieGroupUp$ carrier is the shared singleton history carrier
$$
L_{\emp}(h):\Longleftrightarrow \hsame(h,\emp)
$$
on $\mathsf{BHist}$. Its group rows are the singleton $\GroupUp$ rows
$$
\mathsf{mul}(h,k):=\mathsf{append}(h,k),\qquad
\mathsf{inv}(h):=\emp,\qquad
e:=\emp,
$$
with classifier
$$
R_{\emp}(h,k):\Longleftrightarrow
L_{\emp}(h)\wedge L_{\emp}(k)\wedge \hsame(h,k).
$$
Its smooth-manifold rows are the singleton $\ManifoldUp$ chart rows with the
one chart $\chi_{\emp}$, top domain $L_{\emp}$, constant value $0_{\RealUp}$,
and the self-overlap constant transition. Thus the algebraic carrier and the
manifold carrier are the same predicate on histories, not two transported
copies.
\end{definition}

\section{The certificate}
\label{sec:liegroup-certificate}

\begin{theorem}[LieGroup singleton carrier obligation]
\label{thm:liegroup-singleton-carrier-obligation}
The carrier $L_{\emp}$ of
\autoref{def:liegroup-singleton-smooth-group-carrier} simultaneously supplies
the $\GroupUp$ singleton carrier and classifier rows and the $\ManifoldUp$
singleton chart rows. Concretely, every accepted history is classified with
$\emp$ by $R_{\emp}$, multiplication and inverse stay inside $L_{\emp}$, and
the sole chart domain is exactly the same carried predicate $L_{\emp}$.
\end{theorem}
\begin{proof}
For the algebraic side, unfold
\autoref{def:liegroup-singleton-smooth-group-carrier}. The carrier,
classifier, multiplication, unit, and inverse rows are exactly the rows used by
\autoref{thm:group-singleton-history-semantic-namecert}; the checked Lean row
is
$\mathsf{BEDC.Derived.GroupUp.GroupSingletonHistory\_laws}$.

For the manifold side, the chart-domain row is the singleton empty-chart row
of \autoref{thm:manifold-singleton-chart-coverage}. Its carried predicate is
$\hsame(h,\emp)$, which is definitionally the same predicate as $L_{\emp}$.

The two sides therefore share the same history carrier and the same restricted
$\hsame$ classifier. Projection from the group row gives the algebraic carrier
and classifier fields, while projection from the manifold row gives the chart
domain, chart value, and singleton atlas fields.
\end{proof}

\begin{theorem}[LieGroup singleton operation smoothness obligation]
\label{thm:liegroup-singleton-operation-smoothness-obligation}
For histories carried by $L_{\emp}$, both $\mathsf{mul}$ and $\mathsf{inv}$ of
\autoref{def:liegroup-singleton-smooth-group-carrier} are smooth operations in
the singleton atlas. Every accepted multiplication result and every accepted
inverse result is classifier-equal to $\emp$, so each operation is read by the
sole chart as a constant transition on the singleton overlap.
\end{theorem}
\begin{proof}
Start with multiplication. If $L_{\emp}(h)$ and $L_{\emp}(k)$ hold, the group
singleton laws identify $\mathsf{append}(h,k)$ with $\emp$ and keep the result
inside the singleton carrier. This is the multiplication-closure row of
$\mathsf{BEDC.Derived.GroupUp.GroupSingletonHistory\_laws}$.

For inverse, $\mathsf{inv}(h)$ is definitionally $\emp$. The inverse-closure
row of the same group theorem gives the carrier witness, and the classifier
row identifies the result with the unit endpoint.

Now read both operations through the manifold atlas. By
\autoref{thm:manifold-singleton-chart-coverage}, every carried input and every
accepted output lies in the only chart domain and has the constant chart value
$0_{\RealUp}$. The overlap transition is therefore the singleton constant
transition of \autoref{thm:manifold-singleton-transition-smoothness}, whose
checked Lean row is
$\mathsf{BEDC.Derived.ManifoldUp.ManifoldSingleton\_transition\_smoothness}$.
\end{proof}

\begin{definition}[LieGroup singleton public smooth group package]
\label{def:liegroup-singleton-public-smooth-group-package}
The public smooth group package underlying the singleton
$\NameCert_{\LieGroupUp}$ has BEDC source $\mathsf{BHist}$, carrier
$L_{\emp}$, classifier $R_{\emp}$, multiplication graph
$$
\mathsf{mul}(h,k)=\mathsf{append}(h,k),
$$
inverse
$$
\mathsf{inv}(h)=\emp,
$$
and unit $\emp$. The smooth rows use the singleton chart $\chi_{\emp}$ on the
same carrier predicate, with constant chart value $0_{\RealUp}$ and self-overlap
transition certified by $\Cont$. Thus every public algebraic operation is read
through the displayed BEDC operation rows rather than through an external
standard-space representative.
\end{definition}

\begin{theorem}[LieGroup singleton classifier transport]
\label{thm:liegroup-singleton-classifier-transport}
For the package of
\autoref{def:liegroup-singleton-public-smooth-group-package}, classifier
equivalence is congruent for multiplication and inverse. If $R_{\emp}(h,h')$
and $R_{\emp}(k,k')$, then
$$
R_{\emp}(\mathsf{mul}(h,k),\mathsf{mul}(h',k')).
$$
If $R_{\emp}(h,h')$, then
$$
R_{\emp}(\mathsf{inv}(h),\mathsf{inv}(h')).
$$
\end{theorem}
\begin{proof}
Project the endpoint carrier witnesses from $R_{\emp}(h,h')$ and
$R_{\emp}(k,k')$. Each projection gives membership in $L_{\emp}$ together with
an $\hsame$ path to the empty endpoint.

For multiplication, the singleton group laws identify both
$\mathsf{append}(h,k)$ and $\mathsf{append}(h',k')$ with $\emp$ while keeping
both results in $L_{\emp}$. The classifier row is then the pair of carrier
witnesses plus the $\hsame$ comparison obtained by transitivity through
$\emp$.

For inverse, both operation outputs are definitionally $\emp$. Reflexivity gives
the carrier witnesses and the endpoint comparison, so the restricted classifier
$R_{\emp}$ holds for the two inverse outputs.
\end{proof}

\begin{theorem}[LieGroup singleton chart operation readback]
\label{thm:liegroup-singleton-chart-operation-readback}
The multiplication and inverse outputs covered by
\autoref{thm:liegroup-singleton-classifier-transport} are read by the singleton
chart as the constant endpoint $0_{\RealUp}$. Consequently their chart
self-overlap row is the same singleton transition row certified for
$\ManifoldUp$.
\end{theorem}
\begin{proof}
Apply \autoref{thm:liegroup-singleton-classifier-transport} to obtain the
classifier witnesses for the operation outputs. In each case the classifier
contains the carrier predicate $L_{\emp}$ for the output history.

Apply \autoref{thm:manifold-singleton-chart-coverage} to those carrier
witnesses. The unique chart $\chi_{\emp}$ reads every carried endpoint as the
constant value $0_{\RealUp}$, so multiplication and inverse have the same chart
readback.

Finally apply \autoref{thm:manifold-singleton-transition-smoothness}. Since the
only overlap is the singleton self-overlap and both chart values are constant,
the operation readback has exactly the displayed transition row.
\end{proof}

\begin{theorem}[LieGroup singleton public NameCert export]
\label{thm:liegroup-singleton-public-namecert-export}
The public $\NameCert_{\LieGroupUp}$ surface for the singleton smooth group
exports the following ten fields.
\begin{description}
\item[Traditional target.] The one-point Lie group.
\item[BEDC source.] $\mathsf{BHist}$ with distinguished empty history $\emp$.
\item[Carrier.] The predicate $L_{\emp}(h)$.
\item[Classifier.] The restricted classifier $R_{\emp}(h,k)$.
\item[Algebraic rows.] Multiplication $\mathsf{append}$, inverse $\emp$, unit
$\emp$, and the singleton $\GroupUp$ laws.
\item[Smooth rows.] The singleton $\ManifoldUp$ chart $\chi_{\emp}$, constant
value $0_{\RealUp}$, and the singleton transition row.
\item[Classifier transport.] Multiplication and inverse preserve $R_{\emp}$ as
in \autoref{thm:liegroup-singleton-classifier-transport}.
\item[Smooth operation congruence.] Operation outputs related by the classifier
have the same chart readback and transition row as in
\autoref{thm:liegroup-singleton-chart-operation-readback}.
\item[Exactness.] Every accepted public row reduces to $\mathsf{BHist}$,
$\hsame$, $\Cont$, $\mathsf{append}$, $\emp$, the displayed singleton
$\GroupUp$ laws, and the displayed singleton $\ManifoldUp$ chart and transition.
\item[Bridge and status.] No $\StdBridge$ row is claimed; the paper-side target
is under $\mathsf{BEDC.Derived.LieGroupUp}$.
\end{description}
\end{theorem}
\begin{proof}
The carrier and classifier fields are supplied by
\autoref{def:liegroup-singleton-smooth-group-carrier}, and the public operation
symbols are fixed in
\autoref{def:liegroup-singleton-public-smooth-group-package}.

The algebraic rows are the singleton $\GroupUp$ laws. They give the endpoint
carrier facts and identify multiplication and inverse outputs with $\emp$.

The smooth rows are the singleton $\ManifoldUp$ rows. Chart coverage reads every
accepted endpoint as $0_{\RealUp}$, and transition smoothness gives the constant
self-overlap row.

The transport and smooth-operation congruence rows are exactly
\autoref{thm:liegroup-singleton-classifier-transport} and
\autoref{thm:liegroup-singleton-chart-operation-readback}. Repacking these rows
gives the ten-field public certificate surface, with exactness reduced to the
listed singleton BEDC data and with no standard bridge component.
\end{proof}

\begin{definition}[LieGroup singleton translation and conjugation rows]
\label{def:liegroup-singleton-translation-and-conjugation-rows}
For a carried singleton parameter $s$ and carried singleton input $x$, the
action rows attached to
\autoref{def:liegroup-singleton-smooth-group-carrier} are
$$
\mathsf{Left}_s(x):=\mathsf{mul}(s,x),\qquad
\mathsf{Right}_s(x):=\mathsf{mul}(x,s),
$$
$$
\mathsf{Conj}_s(x):=\mathsf{mul}(\mathsf{mul}(s,x),\mathsf{inv}(s)).
$$
The adjoint readback row is the differential of $\mathsf{Conj}_s$ at $\emp$,
read on the singleton tangent carrier as the constant zero endomap of the
singleton $\LieAlgebraUp$ carrier. Thus translation, conjugation, and adjoint
readback all use the same carried predicate $L_{\emp}$ and the same singleton
chart endpoint.
\end{definition}

\begin{theorem}[LieGroup singleton left-translation smoothness]
\label{thm:liegroup-singleton-left-translation-smooth}
If $L_{\emp}(s)$ and $L_{\emp}(x)$ hold, then
$\mathsf{Left}_s(x)$ from
\autoref{def:liegroup-singleton-translation-and-conjugation-rows} is carried by
$L_{\emp}$, is classified by the singleton classifier with endpoint $\emp$, and
is read in the singleton atlas as the constant smooth transition on the sole
overlap.
\end{theorem}
\begin{proof}
The group row of
$\mathsf{BEDC.Derived.GroupUp.GroupSingletonHistory\_laws}$ gives closure of
$\mathsf{mul}(s,x)$ under the singleton carrier whenever both inputs are carried
by $L_{\emp}$. This is exactly the left-translation carrier row after unfolding
\autoref{def:liegroup-singleton-translation-and-conjugation-rows}.

By \autoref{thm:liegroup-singleton-carrier-obligation}, the carried group result
is the same object as the singleton manifold point. Therefore the sole chart
domain accepts the result and reads it at the constant endpoint $0_{\RealUp}$.

The overlap transition is the singleton constant transition supplied by
\autoref{thm:manifold-singleton-transition-smoothness}. Hence the left
translation is smooth as a singleton-atlas self-map.
\end{proof}

\begin{theorem}[LieGroup singleton right-translation smoothness]
\label{thm:liegroup-singleton-right-translation-smooth}
If $L_{\emp}(s)$ and $L_{\emp}(x)$ hold, then
$\mathsf{Right}_s(x)$ from
\autoref{def:liegroup-singleton-translation-and-conjugation-rows} is carried by
$L_{\emp}$, is classified by the singleton classifier with endpoint $\emp$, and
is read in the singleton atlas as the constant smooth transition on the sole
overlap.
\end{theorem}
\begin{proof}
The multiplication-closure row of
$\mathsf{BEDC.Derived.GroupUp.GroupSingletonHistory\_laws}$ applies symmetrically
to $\mathsf{mul}(x,s)$. Since $L_{\emp}(x)$ and $L_{\emp}(s)$ hold, the
right-translation value remains inside the singleton carrier.

The carrier identification in
\autoref{thm:liegroup-singleton-carrier-obligation} then moves this accepted
group value into the singleton chart domain without changing the predicate being
used. The classifier endpoint is still $\emp$.

Finally, \autoref{thm:manifold-singleton-transition-smoothness} reads the only
self-overlap as a constant smooth transition. Thus the right translation is a
smooth singleton-atlas self-map.
\end{proof}

\begin{theorem}[LieGroup singleton conjugation smoothness]
\label{thm:liegroup-singleton-conjugation-smooth}
If $L_{\emp}(s)$ and $L_{\emp}(x)$ hold, then
$\mathsf{Conj}_s(x)$ from
\autoref{def:liegroup-singleton-translation-and-conjugation-rows} is carried by
$L_{\emp}$, classified with endpoint $\emp$, and read by the singleton atlas as a
constant smooth self-map.
\end{theorem}
\begin{proof}
First apply
\autoref{thm:liegroup-singleton-left-translation-smooth} to $s$ and $x$. The
result $\mathsf{mul}(s,x)$ is carried by $L_{\emp}$ and is read in the singleton
chart as the constant endpoint.

Next use the inverse-closure row from
$\mathsf{BEDC.Derived.GroupUp.GroupSingletonHistory\_laws}$ on the carried
parameter $s$. Since $\mathsf{inv}(s)$ is definitionally $\emp$, it is also
accepted by the singleton carrier and classifier.

Apply the right-translation row
\autoref{thm:liegroup-singleton-right-translation-smooth} to the carried value
$\mathsf{mul}(s,x)$ and the carried inverse $\mathsf{inv}(s)$. The resulting
history is precisely $\mathsf{Conj}_s(x)$ after unfolding the definition, so it
is carried, classified with $\emp$, and read as the singleton constant smooth
self-map.
\end{proof}

\begin{theorem}[LieGroup singleton adjoint readback]
\label{thm:liegroup-singleton-adjoint-readback}
For every carried singleton parameter $s$, the differential at $\emp$ of the
singleton conjugation row $\mathsf{Conj}_s$ is the constant zero endomap on the
singleton tangent carrier, equivalently on the singleton $\LieAlgebraUp$ carrier.
\end{theorem}
\begin{proof}
By \autoref{thm:liegroup-singleton-conjugation-smooth}, conjugation by any
carried singleton parameter is read in the unique chart as a constant self-map
at endpoint $0_{\RealUp}$. A constant singleton chart map has the zero tangent
readback on the sole tangent point.

The $\LieAlgebraUp$ singleton rows identify the Lie-algebra carrier with the same
singleton tangent carrier at $\emp$. Therefore the conjugation differential is
the constant zero endomap on that carrier, which is exactly the adjoint readback
row.
\end{proof}

\begin{theorem}[LieGroup singleton NameCert obligation surface]
\label{thm:liegroup-singleton-namecert-obligation}
The rows of
\autoref{def:liegroup-singleton-smooth-group-carrier},
\autoref{def:liegroup-singleton-public-smooth-group-package},
\autoref{thm:liegroup-singleton-carrier-obligation}, and
\autoref{thm:liegroup-singleton-operation-smoothness-obligation},
\autoref{thm:liegroup-singleton-classifier-transport},
\autoref{thm:liegroup-singleton-chart-operation-readback}, and
\autoref{thm:liegroup-singleton-public-namecert-export},
\autoref{def:liegroup-singleton-translation-and-conjugation-rows},
\autoref{thm:liegroup-singleton-left-translation-smooth},
\autoref{thm:liegroup-singleton-right-translation-smooth},
\autoref{thm:liegroup-singleton-conjugation-smooth}, and
\autoref{thm:liegroup-singleton-adjoint-readback} assemble the
paper-side $\NameCert_{\LieGroupUp}$ obligation surface for the singleton
smooth group carrier.
\begin{description}
\item[Traditional target.] The one-point Lie group.
\item[BEDC source.] $\mathsf{BHist}$ with distinguished empty history $\emp$.
\item[Carrier.] The predicate $L_{\emp}(h)$, equivalently $\hsame(h,\emp)$.
\item[Classifier.] The restricted singleton classifier $R_{\emp}$.
\item[Stability.] Carrier transport, group operation closure, classifier
transport for multiplication and inverse, chart-domain transport, chart-value
constancy, smooth multiplication congruence, smooth inverse congruence,
transition-row constancy, left and right translation smoothness, conjugation
smoothness, and adjoint readback constancy.
\item[Ledger.] The rows $\mathsf{BHist}$, $\hsame$, $\Cont$,
$\mathsf{append}$, $\emp$, the singleton $\GroupUp$ laws, the singleton
$\ManifoldUp$ chart, the singleton constant transition, translation rows,
conjugation rows, and the zero singleton adjoint readback.
\item[Core theorem list.]
\autoref{thm:group-singleton-history-semantic-namecert},
\autoref{thm:manifold-singleton-namecert-obligation-package},
\autoref{thm:liegroup-singleton-carrier-obligation}, and
\autoref{thm:liegroup-singleton-operation-smoothness-obligation},
\autoref{thm:liegroup-singleton-classifier-transport},
\autoref{thm:liegroup-singleton-chart-operation-readback},
\autoref{thm:liegroup-singleton-public-namecert-export},
\autoref{thm:liegroup-singleton-left-translation-smooth},
\autoref{thm:liegroup-singleton-right-translation-smooth},
\autoref{thm:liegroup-singleton-conjugation-smooth}, and
\autoref{thm:liegroup-singleton-adjoint-readback}.
\item[Exactness.] A Lie-group operation row is accepted only when its carrier,
classifier, chart-domain, and transition witnesses all reduce to the displayed
singleton rows.
\item[No standard bridge.] The surface contains no $\StdBridge$ row.
\item[Status.] A paper-side $\NameCert_{\LieGroupUp}$ obligation target under
$\mathsf{BEDC.Derived.LieGroupUp}$.
\end{description}
\end{theorem}
\begin{proof}
The carrier theorem supplies the shared object boundary: one predicate
$L_{\emp}$ is both the group carrier and the manifold point carrier. This
prevents any separate transport problem between algebraic and smooth rows.

The group rows are inherited from
\autoref{thm:group-singleton-history-semantic-namecert}, backed by
$\mathsf{BEDC.Derived.GroupUp.GroupSingletonHistory\_laws}$. They give
inhabitance, multiplication closure, inverse closure, unit laws, and inverse
laws for the append operation with constant inverse.

The manifold rows are inherited from
\autoref{thm:manifold-singleton-namecert-obligation-package}, backed by
$\mathsf{BEDC.Derived.ManifoldUp.ManifoldSingleton\_semanticNameCert}$ and
the singleton transition smoothness row. They give the sole chart domain,
constant chart value, and constant self-overlap transition.

The operation-smoothness and classifier-transport theorems tie these
projections together. Once a group operation result is accepted, it is again
classifier-equal to $\emp$, hence it is read by the only chart as the same
constant endpoint. The translation, conjugation, and adjoint rows are the same
singleton readback applied to the displayed public operation rows. Repacking the
listed fields gives the obligation surface in the statement, with no standard
bridge component claimed.
\end{proof}
