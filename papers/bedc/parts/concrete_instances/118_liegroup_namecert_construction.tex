\chapter{A Concrete Naming Certificate for $\LieGroupUp$}
\label{ch:concrete-instances-liegroup-namecert}

The Lie-group interface $\LieGroupUp$ packages a group carrier whose underlying set is a manifold and whose multiplication and inverse are constructively smooth. It depends on $\NameCert_{\GroupUp}$ (\autoref{ch:concrete-instances-group-namecert}) and $\NameCert_{\ManifoldUp}$ (\autoref{ch:concrete-instances-manifold-namecert}), and is the substrate for representation, exponential, and root-system certificates.

\section{Carrier definition}
\label{sec:liegroup-carrier}

\begin{definition}[Singleton smooth group carrier]
\label{def:liegroup-singleton-smooth-group-carrier}
The concrete $\LieGroupUp$ carrier is the shared singleton history carrier
$$
L_{\emp}(h):\Longleftrightarrow \hsame(h,\emp)
$$
on $\mathsf{BHist}$. Its group rows are the singleton $\GroupUp$ rows
$$
\mathsf{mul}(h,k):=\mathsf{append}(h,k),\qquad
\mathsf{inv}(h):=\emp,\qquad
e:=\emp,
$$
with classifier
$$
R_{\emp}(h,k):\Longleftrightarrow
L_{\emp}(h)\wedge L_{\emp}(k)\wedge \hsame(h,k).
$$
Its smooth-manifold rows are the singleton $\ManifoldUp$ chart rows with the
one chart $\chi_{\emp}$, top domain $L_{\emp}$, constant value $0_{\RealUp}$,
and the self-overlap constant transition. Thus the algebraic carrier and the
manifold carrier are the same predicate on histories, not two transported
copies.
\end{definition}

\section{The certificate}
\label{sec:liegroup-certificate}

\begin{theorem}[LieGroup singleton carrier obligation]
\label{thm:liegroup-singleton-carrier-obligation}
The carrier $L_{\emp}$ of
\autoref{def:liegroup-singleton-smooth-group-carrier} simultaneously supplies
the $\GroupUp$ singleton carrier and classifier rows and the $\ManifoldUp$
singleton chart rows. Concretely, every accepted history is classified with
$\emp$ by $R_{\emp}$, multiplication and inverse stay inside $L_{\emp}$, and
the sole chart domain is exactly the same carried predicate $L_{\emp}$.
\end{theorem}
\leanchecked{BEDC.Derived.LieGroupUp.LieGroupSingleton\_carrier\_obligation}
\begin{proof}
For the algebraic side, unfold
\autoref{def:liegroup-singleton-smooth-group-carrier}. The carrier,
classifier, multiplication, unit, and inverse rows are exactly the rows used by
\autoref{thm:group-singleton-history-semantic-namecert}; the checked Lean row
is
$\mathsf{BEDC.Derived.GroupUp.GroupSingletonHistory\_laws}$.

For the manifold side, the chart-domain row is the singleton empty-chart row
of \autoref{thm:manifold-singleton-chart-coverage}. Its carried predicate is
$\hsame(h,\emp)$, which is definitionally the same predicate as $L_{\emp}$.

The two sides therefore share the same history carrier and the same restricted
$\hsame$ classifier. Projection from the group row gives the algebraic carrier
and classifier fields, while projection from the manifold row gives the chart
domain, chart value, and singleton atlas fields.
\end{proof}

\begin{theorem}[LieGroup singleton operation smoothness obligation]
\label{thm:liegroup-singleton-operation-smoothness-obligation}
For histories carried by $L_{\emp}$, both $\mathsf{mul}$ and $\mathsf{inv}$ of
\autoref{def:liegroup-singleton-smooth-group-carrier} are smooth operations in
the singleton atlas. Every accepted multiplication result and every accepted
inverse result is classifier-equal to $\emp$, so each operation is read by the
sole chart as a constant transition on the singleton overlap.
\end{theorem}
\begin{proof}
Start with multiplication. If $L_{\emp}(h)$ and $L_{\emp}(k)$ hold, the group
singleton laws identify $\mathsf{append}(h,k)$ with $\emp$ and keep the result
inside the singleton carrier. This is the multiplication-closure row of
$\mathsf{BEDC.Derived.GroupUp.GroupSingletonHistory\_laws}$.

For inverse, $\mathsf{inv}(h)$ is definitionally $\emp$. The inverse-closure
row of the same group theorem gives the carrier witness, and the classifier
row identifies the result with the unit endpoint.

Now read both operations through the manifold atlas. By
\autoref{thm:manifold-singleton-chart-coverage}, every carried input and every
accepted output lies in the only chart domain and has the constant chart value
$0_{\RealUp}$. The overlap transition is therefore the singleton constant
transition of \autoref{thm:manifold-singleton-transition-smoothness}, whose
checked Lean row is
$\mathsf{BEDC.Derived.ManifoldUp.ManifoldSingleton\_transition\_smoothness}$.
\end{proof}
\leanchecked{BEDC.Derived.LieGroupUp.LieGroupSingletonOperation\_smoothness}

\begin{theorem}[LieGroup singleton classifier transport]
\label{thm:liegroup-singleton-classifier-transport}
For the package of
\autoref{def:liegroup-singleton-public-smooth-group-package}, classifier
equivalence is congruent for multiplication and inverse. If $R_{\emp}(h,h')$
and $R_{\emp}(k,k')$, then
$$
R_{\emp}(\mathsf{mul}(h,k),\mathsf{mul}(h',k')).
$$
If $R_{\emp}(h,h')$, then
$$
R_{\emp}(\mathsf{inv}(h),\mathsf{inv}(h')).
$$
\end{theorem}
\begin{proof}
Project the endpoint carrier witnesses from $R_{\emp}(h,h')$ and
$R_{\emp}(k,k')$. Each projection gives membership in $L_{\emp}$ together with
an $\hsame$ path to the empty endpoint.

For multiplication, the singleton group laws identify both
$\mathsf{append}(h,k)$ and $\mathsf{append}(h',k')$ with $\emp$ while keeping
both results in $L_{\emp}$. The classifier row is then the pair of carrier
witnesses plus the $\hsame$ comparison obtained by transitivity through
$\emp$.

For inverse, both operation outputs are definitionally $\emp$. Reflexivity gives
the carrier witnesses and the endpoint comparison, so the restricted classifier
$R_{\emp}$ holds for the two inverse outputs.
\end{proof}

\begin{theorem}[LieGroup singleton chart operation readback]
\label{thm:liegroup-singleton-chart-operation-readback}
The multiplication and inverse outputs covered by
\autoref{thm:liegroup-singleton-classifier-transport} are read by the singleton
chart as the constant endpoint $0_{\RealUp}$. Consequently their chart
self-overlap row is the same singleton transition row certified for
$\ManifoldUp$.
\end{theorem}
\begin{proof}
Apply \autoref{thm:liegroup-singleton-classifier-transport} to obtain the
classifier witnesses for the operation outputs. In each case the classifier
contains the carrier predicate $L_{\emp}$ for the output history.

Apply \autoref{thm:manifold-singleton-chart-coverage} to those carrier
witnesses. The unique chart $\chi_{\emp}$ reads every carried endpoint as the
constant value $0_{\RealUp}$, so multiplication and inverse have the same chart
readback.

Finally apply \autoref{thm:manifold-singleton-transition-smoothness}. Since the
only overlap is the singleton self-overlap and both chart values are constant,
the operation readback has exactly the displayed transition row.
\end{proof}

\begin{definition}[LieGroup singleton translation and conjugation rows]
\label{def:liegroup-singleton-translation-and-conjugation-rows}
For a carried singleton parameter $s$ and carried singleton input $x$, the
action rows attached to
\autoref{def:liegroup-singleton-smooth-group-carrier} are
$$
\mathsf{Left}_s(x):=\mathsf{mul}(s,x),\qquad
\mathsf{Right}_s(x):=\mathsf{mul}(x,s),
$$
$$
\mathsf{Conj}_s(x):=\mathsf{mul}(\mathsf{mul}(s,x),\mathsf{inv}(s)).
$$
The adjoint readback row is the differential of $\mathsf{Conj}_s$ at $\emp$,
read on the singleton tangent carrier as the constant zero endomap of the
singleton $\LieAlgebraUp$ carrier. Thus translation, conjugation, and adjoint
readback all use the same carried predicate $L_{\emp}$ and the same singleton
chart endpoint.
\end{definition}

\label{thm:liegroup-singleton-left-translation-smooth}

\end{theorem}
\begin{proof}

$\mathsf{BEDC.Derived.GroupUp.GroupSingletonHistory\_laws}$ gives closure of

\autoref{def:liegroup-singleton-translation-and-conjugation-rows}.

By \autoref{thm:liegroup-singleton-carrier-obligation}, the carried group result
is the same object as the singleton manifold point. Therefore the sole chart
domain accepts the result and reads it at the constant endpoint $0_{\RealUp}$.

The overlap transition is the singleton constant transition supplied by
\autoref{thm:manifold-singleton-transition-smoothness}. Hence the left
translation is smooth as a singleton-atlas self-map.
\end{proof}

\begin{definition}[Singleton LieGroup conjugation action]
\label{def:liegroup-singleton-conjugation-action}
On the singleton smooth group carrier

$\mathsf{BHist}$-valued conjugation action by
$$
\mathsf{Ad}_{\emp}(s,x):=
\mathsf{append}(\mathsf{append}(s,x),\emp).
$$
The inputs are carried histories $L_{\emp}(s)$ and $L_{\emp}(x)$, and the
terminal factor $\emp$ is the singleton inverse row of the group carrier.
\end{definition}

\begin{theorem}[LieGroup singleton conjugation carrier and smoothness]
\label{thm:liegroup-singleton-conjugation-carrier-and-smoothness}
If $L_{\emp}(s)$ and $L_{\emp}(x)$ hold, then
$\mathsf{Ad}_{\emp}(s,x)$ is again carried by $L_{\emp}$. Moreover the singleton
manifold chart reads this conjugation endpoint as the same constant smooth
transition that reads multiplication and inverse in
\autoref{thm:liegroup-singleton-operation-smoothness-obligation}.
\end{theorem}
\begin{proof}
Unfold \autoref{def:liegroup-singleton-conjugation-action}. The endpoint is the
append-composite $(s*x)*\emp$ in the singleton group rows. Since $s$ and $x$ are
carried, \autoref{thm:group-singleton-history-semantic-namecert} gives closure
of $s*x$, and the right multiplication by $\emp$ stays in the same carrier. The
Lean-side row
$\mathsf{BEDC.Derived.GroupUp.GroupSingletonClassifier\_conjugation\_terminal\_collapse\_iff}$
records the same terminal collapse for the singleton classifier.

For smoothness, the carried input pair and the carried output all lie in the
sole singleton chart domain. Applying
\autoref{thm:manifold-singleton-transition-smoothness} reads the source and
target charts as the constant value $0_{\RealUp}$, so the displayed action has
the constant smooth transition row already used by the multiplication and
inverse obligations. The corresponding Lean-side row is
$\mathsf{BEDC.Derived.ManifoldUp.ManifoldSingleton\_transition\_smoothness}$.
\end{proof}

\begin{theorem}[LieGroup singleton conjugation action law]
\label{thm:liegroup-singleton-conjugation-action-law}
For carried histories $L_{\emp}(s)$, $L_{\emp}(t)$, and $L_{\emp}(x)$, the two
iterated endpoints
$$
\mathsf{Ad}_{\emp}(s,\mathsf{Ad}_{\emp}(t,x))
$$
and
$$
\mathsf{Ad}_{\emp}(\mathsf{append}(s,t),x)
$$
are both classified by the singleton classifier $R_{\emp}$, hence represent the
same carried singleton endpoint.
\end{theorem}
\begin{proof}
First unfold the action on the left. The inner endpoint is
$\mathsf{append}(\mathsf{append}(t,x),\emp)$, and
\autoref{thm:liegroup-singleton-conjugation-carrier-and-smoothness} carries it
back into $L_{\emp}$. Applying the same theorem to $s$ and this inner endpoint
therefore carries the whole left-hand endpoint.

Next unfold the action on the right. Group singleton closure carries
$\mathsf{append}(s,t)$, and another application of
\autoref{thm:liegroup-singleton-conjugation-carrier-and-smoothness} carries
$\mathsf{Ad}_{\emp}(\mathsf{append}(s,t),x)$.

It remains to compare the classifier rows. By the singleton group classifier,
every carried endpoint is $\hsame$ to $\emp$. The $\Cont$ associativity and
readback rows allow the nested append words on both sides to be read as
singleton terminal words, and the group singleton closure removes the terminal
factor. Thus both displayed endpoints satisfy $R_{\emp}$ against $\emp$, and
therefore against each other by the restricted singleton classifier.
\end{proof}

shared history predicate $L_{\emp}$ and the restricted $\hsame$ classifier.
These are the only source rows a dependent certificate can name.

By \autoref{thm:liegroup-singleton-exact-ledger}, the remaining visible rows
are precisely the group operation rows, singleton chart rows, and singleton
transition row. The NameCert surface in
\autoref{thm:liegroup-singleton-namecert-obligation} repacks those rows as the
public obligation surface for $\LieGroupUp$.

Therefore any $\ExpMapUp$, $\AdjointRepUp$, $\RootSystemUp$, or
representation certificate built over this singleton source can project only
those exported rows. Since the ledger contains no connectedness, completeness,
nontrivial Lie-algebra, or $\StdBridge$ row, no downstream consumer receives
such a claim from this package.
\end{proof}
