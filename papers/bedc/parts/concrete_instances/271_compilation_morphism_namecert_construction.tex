\chapter{Naming Certificate for $\mathsf{CompilationMorphismUp}$}
\label{ch:concrete-instances-compilation-as-namecert-morphism}

The compilation-morphism interface $\mathsf{CompilationMorphismUp}$ names a classifier-preserving translation from a source naming certificate to a target naming certificate. Its public object is the finite BEDC row that records source and target observations, the translated endpoint, and the correctness ledger tying the two presentations together.

\section{Source spec for $\mathsf{CompilationMorphismUp}$}
\label{sec:concrete-instances-compilation-morphism-source-spec}

\begin{definition}[Source rows of $\mathsf{CompilationMorphismUp}$]
\label{def:concrete-instances-compilation-morphism-source-rows}
A source row for $\mathsf{CompilationMorphismUp}$ is a finite tuple
$$
(\mathsf{sourceHistory},\mathsf{targetHistory},
\mathsf{sourceEndpoint},\mathsf{targetEndpoint},
\mathsf{sourceMark},\mathsf{targetMark},
\mathsf{morphismLedger},\mathsf{correctnessLedger})
$$
where the source and target histories are the carried presentations in the two naming certificates, the two endpoint histories expose the observed source and target classifier rows, the two marks record the visible extension steps, and the two ledgers record the morphism transport and the correctness comparison. This source row is a BEDC history packet, not a host-level compiler function.
\end{definition}
\leandef{BEDC.Derived.CompilationMorphismUp.SourceRows}

\section{Pattern spec}
\label{sec:concrete-instances-compilation-morphism-pattern}

\begin{definition}[Pattern witness for $\mathsf{CompilationMorphismUp}$]
\label{def:concrete-instances-compilation-morphism-pattern}
The pattern of a compilation-morphism row is the conjunction of four visible relations:
$$
\begin{aligned}
&\Ext(\mathsf{sourceHistory},\mathsf{sourceMark},\mathsf{sourceEndpoint}),\\
&\Ext(\mathsf{targetHistory},\mathsf{targetMark},\mathsf{targetEndpoint}),\\
&\Cont(\mathsf{sourceEndpoint},\mathsf{morphismLedger},\mathsf{targetEndpoint}),\\
&\Cont(\mathsf{sourceHistory},\mathsf{correctnessLedger},\mathsf{targetHistory}).
\end{aligned}
$$
The first two rows expose the source and target certificate observations. The two continuation rows say that the target observation is reached by a named morphism ledger and that the source presentation is tied to the target presentation by a correctness ledger.
\end{definition}
\leandef{BEDC.Derived.CompilationMorphismUp.Pattern}

\section{Classifier spec}
\label{sec:concrete-instances-compilation-morphism-classifier}

\begin{definition}[Classifier for $\mathsf{CompilationMorphismUp}$]
\label{def:concrete-instances-compilation-morphism-classifier}
Two source rows classify as the same $\mathsf{CompilationMorphismUp}$ object when their source histories, target histories, source endpoints, target endpoints, morphism ledgers, and correctness ledgers agree by $\hsame$, and their source and target marks agree by $\msame$. Thus the classifier identifies different presentations only when they expose the same BEDC morphism packet componentwise; it does not quotient compilers by an external semantic equality.
\end{definition}
\leandef{BEDC.Derived.CompilationMorphismUp.Classifier}

\section{Stability certificate}
\label{sec:concrete-instances-compilation-morphism-stability}

\begin{theorem}[Stability of $\mathsf{CompilationMorphismUp}$]
\label{thm:concrete-instances-compilation-morphism-stability}
If a compilation-morphism source row satisfies the pattern, then every classified presentation of the same row also satisfies the pattern. Equivalently, the extension witnesses for the source and target observations, the morphism continuation ledger, and the correctness continuation ledger are stable under the componentwise $\hsame$ and $\msame$ classifier.
\end{theorem}
\leanchecked{BEDC.Derived.CompilationMorphismUp.stability}

\begin{proof}
Unfold the classifier. Each classifier component is an internal BEDC sameness row, so eliminating these rows identifies the two presentations component by component. The four pattern witnesses for the first presentation are then the four pattern witnesses for the classified presentation.
\end{proof}

\section{Ledger policy}
\label{sec:concrete-instances-compilation-morphism-ledger}

\begin{definition}[Ledger for $\mathsf{CompilationMorphismUp}$]
\label{def:concrete-instances-compilation-morphism-ledger}
The ledger keeps the full source row: both certificate-side histories, both exposed endpoints, both marks, the morphism ledger, and the correctness ledger. This is enough to replay the classifier-preserving transport and the $\hsame$-preservation claim at the BEDC boundary while keeping optimisation strategies, intermediate-representation schedules, decompilation choices, and host-language semantics outside the public carrier.
\end{definition}
\leandef{BEDC.Derived.CompilationMorphismUp.Ledger}
