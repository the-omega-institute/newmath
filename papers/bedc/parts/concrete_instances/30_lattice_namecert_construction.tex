\chapter{A Concrete Naming Certificate for \(\LatticeUp\)}
\label{ch:concrete-instances-lattice-namecert}

The lattice interface \(\LatticeUp\) extends \autoref{ch:concrete-instances-poset-namecert} with binary meet \(\wedge\) and join \(\vee\) operations, each classified as the greatest lower bound and least upper bound of two elements under the partial order. Idempotence, commutativity, associativity, and absorption laws are derived from the bound characterizations under the carrier's classifier.

\section{Carrier definition}

\begin{definition}[Lattice carrier]
\label{def:lattice-carrier}
Use a poset setup extended with meet and join operations carrying bound-characterization proofs.
\end{definition}

\section{Concrete singleton prefix lattice instance}
\label{sec:lattice-concrete-singleton-prefix-instance}

\begin{definition}[Concrete singleton prefix lattice instance]
\label{def:concrete-singleton-prefix-lattice-instance}
Let $A_{\emp}$ be the singleton history source of
Definition~\ref{def:list-singleton-history-source} at $a_0=\emp$. Define
$\rho_{\emp}$ by
$$
\rho_{\emp}(h,k)\Longleftrightarrow
A_{\emp}(h)\land A_{\emp}(k)\land\hsame(h,k).
$$
Let $h\le_{\emp} k$ mean that $A_{\emp}(h)$ and $A_{\emp}(k)$ hold and that
$\mathsf{PreorderPrefixLE}(h,k)$ witnesses the prefix comparison between the
carried endpoints. Define the concrete lattice operations by
$$
\mathsf{meet}(h,k):=\emp,\qquad
\mathsf{join}(h,k):=\emp.
$$
The certificate has ten fields: traditional target the one-point lattice; BEDC
source $\mathsf{BHist}$ plus the singleton source and concrete prefix order;
carrier $A_{\emp}$; classifier $\rho_{\emp}$; stability obligations for poset
laws, meet lower bounds, meet greatestness, join upper bounds, join leastness,
and meet and join congruence; ledger singleton endpoint plus prefix witnesses;
core theorem Theorem~\ref{thm:concrete-singleton-prefix-lattice-laws};
exactness singleton endpoint classification; standard bridge terminal
one-point lattice; and status $\mathsf{ProofTarget}$.
\end{definition}

\begin{theorem}[Concrete singleton prefix lattice laws]
\label{thm:concrete-singleton-prefix-lattice-laws}
For all carried histories $h,k,z$ in $A_{\emp}$, the endpoints
$\mathsf{meet}(h,k)$ and $\mathsf{join}(h,k)$ are carried by $A_{\emp}$ and
satisfy the meet lower-bound laws, meet greatest-lower-bound law, join
upper-bound laws, join least-upper-bound law, and meet and join congruence laws
for the restricted prefix order $\le_{\emp}$ under the classifier
$\rho_{\emp}$.
\end{theorem}
\begin{proof}
The singleton carrier witnesses identify each carried endpoint with $\emp$.
Thus $A_{\emp}(h)$, $A_{\emp}(k)$, and $A_{\emp}(z)$ reduce every endpoint
comparison needed by the order, meet, and join clauses to the empty-history
endpoint.

The restricted prefix comparisons are supplied by
$\mathsf{PreorderPrefixLE\_of\_hsame}$ after the singleton witnesses produce
the relevant $\hsame$ facts. The inherited order boundary is the concrete unary
prefix poset law package of Theorem~\ref{thm:concrete-unary-prefix-poset-laws},
with antisymmetry supplied by Theorem~\ref{thm:poset-prefix-antisymmetry}.

The meet and join outputs are both definitionally $\emp$. The lower-bound and
upper-bound clauses therefore use the singleton endpoint witness for $\emp$
and the prefix witnesses from the previous paragraph; the greatest-lower-bound
and least-upper-bound clauses use the same reduction for the third carried
history $z$.

Congruence is immediate because both operations ignore their operands and
return $\emp$. After unfolding, every source and target output is classified by
$\rho_{\emp}(\emp,\emp)$, assembled from two copies of $A_{\emp}(\emp)$ and the
reflexive witness $\hsame(\emp,\emp)$.
\end{proof}
\leanchecked{BEDC.Derived.LatticeUp.LatticeSingletonPrefix\_laws}

\begin{theorem}[Singleton lattice order has empty endpoints]
\label{thm:lattice-singleton-le-empty-endpoints}
For $h,k:\mathsf{BHist}$, the restricted singleton prefix order satisfies
$$
h\le_{\emp} k
\quad\Longleftrightarrow\quad
h\hsame\emp\land k\hsame\emp.
$$
\end{theorem}
\begin{proof}
The forward direction reads the two singleton carrier fields from the
definition of $h\le_{\emp} k$. These fields are exactly $h\hsame\emp$ and
$k\hsame\emp$.

For the reverse direction, the two endpoint witnesses supply the singleton
carrier fields. Their transitive composite through $\emp$ gives $h\hsame k$,
and the prefix comparison follows from the concrete
$\mathsf{PreorderPrefixLE\_of\_hsame}$ constructor.
\end{proof}

\begin{theorem}[Singleton lattice carrier order collapse]
\label{thm:lattice-singleton-carrier-order-collapse}
If $h$ and $k$ are carried by $A_{\emp}$, then both restricted order
comparisons hold, the singleton classifier relates $h$ and $k$, and both
endpoints are internally the same as $\emp$:
$$
h\le_{\emp} k\land k\le_{\emp} h\land \rho_{\emp}(h,k)
\land h\hsame\emp\land k\hsame\emp.
$$
\end{theorem}
\begin{proof}
The two carrier witnesses are the endpoint comparisons
$h\hsame\emp$ and $k\hsame\emp$. Their composite through $\emp$ gives
$h\hsame k$, and symmetry gives $k\hsame h$.

The two restricted order comparisons are obtained from these two sameness
witnesses by $\mathsf{PreorderPrefixLE\_of\_hsame}$, together with the same
carrier fields. The classifier package uses the carrier fields and
$h\hsame k$, and the endpoint clauses are the original carrier witnesses.
\end{proof}

$$
\mathsf{LatticeSingletonClassifier}(P,Q)

\mathsf{LatticeSingletonClassifier}(R,P).

\begin{theorem}[Singleton meet--join absorption gives order and classifier data]
\label{thm:lattice-singleton-meet-join-absorption-order-classifier}
If $h,k:\mathsf{BHist}$ are singleton-carried, then the two singleton lattice
absorption endpoints
$$
h\wedge(h\vee k)
\qquad\text{and}\qquad
h\vee(h\wedge k)
$$
are both definitionally the empty history. They satisfy the restricted prefix
order comparisons
$$
h\wedge(h\vee k)\le_{\emp} h
\qquad\text{and}\qquad
h\le_{\emp} h\vee(h\wedge k),
$$
are both classified with $h$, and both read back to $\emp$.
\end{theorem}
\begin{proof}
In the singleton lattice, meet and join ignore their inputs and return $\emp$.
The carrier witness for $h$ gives $h\hsame\emp$, while reflexivity gives
$\emp\hsame\emp$.

The first order comparison uses symmetry of $h\hsame\emp$ to obtain
$\emp\hsame h$ and then applies the prefix comparison constructor. The second
uses $h\hsame\emp$ directly. The same two $\hsame$ witnesses assemble the
classifier tuples with $h$, and the endpoint readbacks are both reflexivity at
$\emp$.
\end{proof}

$t$, to be empty. Conversely, if $t\hsame\emp$, the right continuation unit
identifies $\mathsf{append}(h,t)$ with $h$, so the given singleton carrier
for $h$ also carries the append endpoint. The prefix witness is the concrete
append-tail continuation.
\end{proof}
\leanchecked{BEDC.Derived.LatticeUp.LatticeSingletonLE\_append\_tail\_upper\_empty\_iff}

\begin{theorem}[Singleton lattice meet greatest lower bound empty exactness]
\label{thm:lattice-singleton-meet-glb-empty-exactness}
For carried singleton endpoints $h,k,z$, the statement that $z$ is below the
concrete meet of $h$ and $k$ is equivalent to the two endpoint lower-bound
clauses together with the fact that the concrete meet is empty:
$$
z\le_{\emp}\mathsf{meet}(h,k)
\quad\Longleftrightarrow\quad
z\le_{\emp}h\wedge z\le_{\emp}k\wedge \mathsf{meet}(h,k)\hsame\emp.
$$
\end{theorem}
\begin{proof}
The singleton carrier identifies $h,k,z$ with $\emp$, and the concrete meet
operation is definitionally $\emp$. Thus the forward direction builds both
lower-bound comparisons from the endpoint sameness witnesses using the
singleton prefix comparison. Conversely, the displayed empty-meet endpoint
and the carrier witness for $z$ build the singleton comparison
$z\le_{\emp}\mathsf{meet}(h,k)$.
\end{proof}
\leanchecked{BEDC.Derived.LatticeUp.LatticeSingletonMeet\_greatest\_lower\_bound\_empty\_iff}

The $\LatticeUp$ schema is instantiated by a concrete $\mathsf{BHist}$ carrier
and operations, so Lean can target $\mathsf{BEDC.Derived.LatticeUp}$ as a
$\mathsf{BHist}$-anchored certificate rather than an abstract operator schema.

\section{The certificate}

\begin{definition}[Lattice source specification]
\label{def:lattice-source-specification}
The source is a certified poset carrier plus binary meet and join data.
\end{definition}

\begin{definition}[Lattice pattern specification]
\label{def:lattice-pattern-specification}
The pattern is expression formation by \(\wedge\) and \(\vee\) over carrier witnesses.
\end{definition}

\begin{definition}[Lattice classifier specification]
\label{def:lattice-classifier-specification}
The classifier is the poset carrier classifier, with meet and join congruent in both arguments.
\end{definition}

\begin{definition}[Lattice stability certificate]
\label{def:lattice-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item all poset stability fields
  \item meet lower-bound laws
  \item meet greatestness law
  \item join upper-bound laws
  \item join leastness law
  \item meet and join congruence
\end{enumerate}
\end{definition}

\begin{definition}[Lattice ledger policy]
\label{def:lattice-ledger-policy}
Record operands and the bound-characterization certificate for each meet or join.
\end{definition}

\begin{definition}[\(\LatticeUp\) certificate obligations]
\label{def:lattice-certificate-obligations}
The carrier, source, pattern, classifier, stability, and ledger data listed above are the obligations required for the interface name \(\LatticeUp\).
\end{definition}

\paragraph{Primitive boundary.}
\(\LatticeUp\) is not a BEDC primitive. It is admitted only when those obligations are supplied.

\begin{lemma}[Meet idempotence from bound characterization]
\label{lem:lattice-meet-idempotence-from-bounds}
Let \(A\) be a \(\LatticeUp\) carrier with classifier \(\sim_C\).
For every \(x\in A\),
\[
  x\wedge x \sim_C x .
\]
\end{lemma}
\begin{proof}
The meet lower-left field gives
\[
  x\wedge x \le x .
\]
The inherited preorder reflexivity field gives \(x\le x\). Applying meet greatestness with \(z=x\), \(a=x\), and \(b=x\), using the two reflexivity witnesses \(x\le x\) and \(x\le x\), gives
\[
  x\le x\wedge x .
\]
The inherited poset antisymmetry field applied to
\[
  x\wedge x\le x  \qquad\text{and}\qquad  x\le x\wedge x
\]
yields
\[
  x\wedge x\sim_C x .
\]
\end{proof}

\begin{theorem}[Lattice idempotence and absorption from bound characterization]
\label{thm:lattice-idempotence-absorption-from-bound-characterization}
Let \(A\) be a \(\LatticeUp\) carrier with classifier \(\sim_C\). Assume the inherited \(\POSetUp\) reflexivity and antisymmetry fields, and assume the directional meet and join bound-characterization fields:
\[
\begin{gathered}
  \forall a\,b,\ a\wedge b\le a,\qquad
  \forall a\,b,\ a\wedge b\le b,\\
  \forall z\,a\,b,\ z\le a\to z\le b\to z\le a\wedge b,\\
  \forall a\,b,\ a\le a\vee b,\qquad
  \forall a\,b,\ b\le a\vee b,\\
  \forall z\,a\,b,\ a\le z\to b\le z\to a\vee b\le z.
\end{gathered}
\]
Then for all \(x,y\in A\),
\[
  x\wedge x\sim_C x,\qquad
  x\vee x\sim_C x,\qquad
  x\wedge(x\vee y)\sim_C x,\qquad
  x\vee(x\wedge y)\sim_C x .
\]
\end{theorem}
\begin{proof}
The first assertion is Lemma~\ref{lem:lattice-meet-idempotence-from-bounds}. The second assertion is Lemma~\ref{lem:lattice-join-idempotence-from-bounds}. The third assertion is Lemma~\ref{lem:lattice-meet-absorption-from-bounds}. The fourth assertion is Lemma~\ref{lem:lattice-join-absorption-from-bounds}.
\end{proof}

\begin{theorem}[Lattice commutativity from directional bounds]
\label{thm:lattice-commutativity-from-directional-bounds}
Let \(A\) be a \(\LatticeUp\) carrier with classifier \(\sim_C\). Assume the inherited \(\POSetUp\) antisymmetry field and the directional meet and join bound-characterization fields from \autoref{def:lattice-stability-certificate}. Then for all \(x,y\in A\),
\[
  x\wedge y\sim_C y\wedge x,
  \qquad
  x\vee y\sim_C y\vee x .
\]
\end{theorem}
\begin{proof}
For the meet comparison, the meet lower-bound laws in \autoref{def:lattice-stability-certificate} give
\[
  x\wedge y\le x
  \qquad\text{and}\qquad
  x\wedge y\le y .
\]
Applying meet greatestness with \(z=x\wedge y\), \(a=y\), and \(b=x\) gives
\[
  x\wedge y\le y\wedge x .
\]
The same lower-bound laws applied to \(y\wedge x\) give
\[
  y\wedge x\le y
  \qquad\text{and}\qquad
  y\wedge x\le x .
\]
Applying meet greatestness with \(z=y\wedge x\), \(a=x\), and \(b=y\) gives
\[
  y\wedge x\le x\wedge y .
\]
Since \(\LatticeUp\) extends the \(\POSetUp\) layer, the antisymmetry field recorded in \autoref{def:poset-stability-certificate} applies to these two comparisons and yields
\[
  x\wedge y\sim_C y\wedge x .
\]

For the join comparison, the join upper-bound laws in \autoref{def:lattice-stability-certificate} give
\[
  y\le x\vee y
  \qquad\text{and}\qquad
  x\le x\vee y .
\]
Applying join leastness with \(z=x\vee y\), \(a=y\), and \(b=x\) gives
\[
  y\vee x\le x\vee y .
\]
The same upper-bound laws applied to \(y\vee x\) give
\[
  x\le y\vee x
  \qquad\text{and}\qquad
  y\le y\vee x .
\]
Applying join leastness with \(z=y\vee x\), \(a=x\), and \(b=y\) gives
\[
  x\vee y\le y\vee x .
\]
The inherited antisymmetry field from \autoref{def:poset-stability-certificate} applied to these two comparisons yields
\[
  x\vee y\sim_C y\vee x .
\]
\end{proof}

\begin{theorem}[Lattice opposite absorption from directional bounds]
\label{thm:lattice-opposite-absorption-from-directional-bounds}
Let \(A\) be a \(\LatticeUp\) carrier with classifier \(\sim_C\). Assume the inherited preorder reflexivity field, the inherited \(\POSetUp\) antisymmetry field, and the directional meet and join bound-characterization fields from \autoref{def:lattice-stability-certificate}. Then for all \(x,y\in A\),
\[
  x\wedge(y\vee x)\sim_C x,\qquad
  (x\vee y)\wedge x\sim_C x,\qquad
  x\vee(y\wedge x)\sim_C x,\qquad
  (x\wedge y)\vee x\sim_C x .
\]
\end{theorem}
\begin{proof}
The inherited reflexivity field is the preorder reflexivity component of \autoref{def:preorder-stability-certificate}; the inherited classifier collapse is the antisymmetry component of \autoref{def:poset-stability-certificate}. The meet and join directional fields are those listed in \autoref{def:lattice-stability-certificate}.

For \(x\wedge(y\vee x)\), the meet lower-left field gives
\[
  x\wedge(y\vee x)\le x .
\]
The inherited preorder reflexivity field gives
\[
  x\le x .
\]
The join upper-right field applied to \(y\vee x\) gives
\[
  x\le y\vee x .
\]
Applying meet greatestness with \(z=x\), \(a=x\), and \(b=y\vee x\), using the two comparisons above, gives
\[
  x\le x\wedge(y\vee x) .
\]
Antisymmetry from \autoref{def:poset-stability-certificate} applied to
\[
  x\wedge(y\vee x)\le x
  \qquad\text{and}\qquad
  x\le x\wedge(y\vee x)
\]
yields
\[
  x\wedge(y\vee x)\sim_C x .
\]

For \((x\vee y)\wedge x\), the meet lower-right field gives
\[
  (x\vee y)\wedge x\le x .
\]
The join upper-left field gives
\[
  x\le x\vee y .
\]
The inherited preorder reflexivity field gives
\[
  x\le x .
\]
Applying meet greatestness with \(z=x\), \(a=x\vee y\), and \(b=x\), using these two comparisons, gives
\[
  x\le (x\vee y)\wedge x .
\]
Antisymmetry applied to
\[
  (x\vee y)\wedge x\le x
  \qquad\text{and}\qquad
  x\le (x\vee y)\wedge x
\]
yields
\[
  (x\vee y)\wedge x\sim_C x .
\]

For \(x\vee(y\wedge x)\), the join upper-left field gives
\[
  x\le x\vee(y\wedge x) .
\]
The inherited preorder reflexivity field gives
\[
  x\le x .
\]
The meet lower-right field applied to \(y\wedge x\) gives
\[
  y\wedge x\le x .
\]
Applying join leastness with \(z=x\), \(a=x\), and \(b=y\wedge x\), using the two comparisons above, gives
\[
  x\vee(y\wedge x)\le x .
\]
Antisymmetry applied to
\[
  x\vee(y\wedge x)\le x
  \qquad\text{and}\qquad
  x\le x\vee(y\wedge x)
\]
yields
\[
  x\vee(y\wedge x)\sim_C x .
\]

For \((x\wedge y)\vee x\), the join upper-right field gives
\[
  x\le (x\wedge y)\vee x .
\]
The meet lower-left field gives
\[
  x\wedge y\le x .
\]
The inherited preorder reflexivity field gives
\[
  x\le x .
\]
Applying join leastness with \(z=x\), \(a=x\wedge y\), and \(b=x\), using these two comparisons, gives
\[
  (x\wedge y)\vee x\le x .
\]
Antisymmetry applied to
\[
  (x\wedge y)\vee x\le x
  \qquad\text{and}\qquad
  x\le (x\wedge y)\vee x
\]
yields
\[
  (x\wedge y)\vee x\sim_C x .
\]
\end{proof}

\begin{theorem}[Lattice bound uniqueness from directional certificates]
\label{thm:lattice-bound-uniqueness-from-directional-certificates}
Let $A$ be a $\LatticeUp$ carrier with classifier $\sim_C$. Assume the inherited $\POSetUp$ antisymmetry field from \autoref{def:poset-stability-certificate} and the directional meet and join bound-characterization fields from \autoref{def:lattice-stability-certificate}. For all $x,y,m,j\in A$, suppose $m$ satisfies the lower-bound and greatestness clauses
$$
\begin{gathered}
m\le x,\qquad m\le y,\\
\forall z\in A,\ z\le x\to z\le y\to z\le m,
\end{gathered}
$$
and suppose $j$ satisfies the upper-bound and leastness clauses
$$
\begin{gathered}
x\le j,\qquad y\le j,\\
\forall z\in A,\ x\le z\to y\le z\to j\le z.
\end{gathered}
$$
Then
$$
m\sim_C x\wedge y,\qquad j\sim_C x\vee y .
$$
\end{theorem}
\begin{proof}
The certified meet fields give
$$
x\wedge y\le x
\qquad\text{and}\qquad
x\wedge y\le y .
$$
Applying the assumed greatestness clause for $m$ with $z=x\wedge y$ gives
$$
x\wedge y\le m .
$$
The two assumed lower-bound clauses for $m$, together with the certified meet greatestness field applied with $z=m$, give
$$
m\le x\wedge y .
$$
The antisymmetry field from \autoref{def:poset-stability-certificate} applied to
$$
m\le x\wedge y
\qquad\text{and}\qquad
x\wedge y\le m
$$
yields
$$
m\sim_C x\wedge y .
$$

The certified join fields give
$$
x\le x\vee y
\qquad\text{and}\qquad
y\le x\vee y .
$$
Applying the assumed leastness clause for $j$ with $z=x\vee y$ gives
$$
j\le x\vee y .
$$
The two assumed upper-bound clauses for $j$, together with the certified join leastness field applied with $z=j$, give
$$
x\vee y\le j .
$$
The antisymmetry field from \autoref{def:poset-stability-certificate} applied to
$$
j\le x\vee y
\qquad\text{and}\qquad
x\vee y\le j
$$
yields
$$
j\sim_C x\vee y .
$$
\end{proof}
