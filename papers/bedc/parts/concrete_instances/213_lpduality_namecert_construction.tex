\begin{definition}[Finite $\LPDualityUp$ ordered-field feasibility row]
\label{def:lpduality-finite-ordered-field-feasibility-row}
An $\LPDualityUp$ row over an ordered field consists of finite row-position and column-position carriers $I$ and $J$, a scalar carrier $F$ equipped with the $\FieldUp$ operations and classifier of \autoref{ch:concrete-instances-field-namecert}, and a $\TotalOrderUp$ comparison on the same scalar carrier and classifier as in \autoref{ch:concrete-instances-totalorder-namecert}. Write $+_F$, $\cdot_F$, $0_F$, $\sim_F$, and $\preceq_F$ for these data, with $u\preceq_F v$ meaning that $u$ is less than or classifier-equal to $v$. The nonnegative scalar predicate is
$$
\mathsf{NonNeg}_F(a)\Longleftrightarrow 0_F\preceq_F a.
$$
The scalar order rows used by the certificate are classifier stability of $\preceq_F$,
$$
u\sim_Fu'\land v\sim_Fv'\land u\preceq_Fv
\Longrightarrow u'\preceq_Fv',
$$
left and right multiplication monotonicity by nonnegative scalars,
$$
\mathsf{NonNeg}_F(a)\land u\preceq_Fv
\Longrightarrow
(a\cdot_Fu\preceq_Fa\cdot_Fv)
\land
(u\cdot_Fa\preceq_Fv\cdot_Fa),
$$
and finite-sum monotonicity over every finite position carrier $K$,
$$
\Bigl(\forall k\in K.\; r_k\preceq_Fs_k\Bigr)
\Longrightarrow
\sum_{k\in K}r_k\preceq_F\sum_{k\in K}s_k.
$$

For coefficients $A:I\times J\to F$, right-hand side $b:I\to F$, and cost vector $c:J\to F$, define
$$
\alpha_i(x):=\sum_{j\in J} A_{ij}\cdot_F x_j,
\qquad
\beta_j(y):=\sum_{i\in I} y_i\cdot_F A_{ij}.
$$
The finite distributivity and interchange row records the classifier comparison
$$
\sum_{j\in J}\beta_j(y)\cdot_Fx_j
\sim_F
\sum_{i\in I}y_i\cdot_F\alpha_i(x).
$$
The primal and dual feasible carriers, read as the row descriptions of the corresponding $\PolytopeUp$ feasible sets over the convex substrate of \autoref{ch:concrete-instances-convexset-namecert} and \autoref{ch:concrete-instances-polytope-namecert}, are
$$
\mathsf{PrFeas}_{A,b}(x)\Longleftrightarrow
\Bigl(\forall j\in J.\;\mathsf{NonNeg}_F(x_j)\Bigr)
\land
\Bigl(\forall i\in I.\;\alpha_i(x)\preceq_F b_i\Bigr)
$$
and
$$
\mathsf{DuFeas}_{A,b,c}(y)\Longleftrightarrow
\Bigl(\forall i\in I.\;\mathsf{NonNeg}_F(y_i)\Bigr)
\land
\Bigl(\forall j\in J.\;c_j\preceq_F\beta_j(y)\Bigr).
$$
The objective endpoints are
$$
\mathsf{PrObj}_{A,c}(x):=\sum_{j\in J} c_j\cdot_Fx_j,
\qquad
\mathsf{DuObj}_{b}(y):=\sum_{i\in I} y_i\cdot_Fb_i.
$$
\end{definition}

\begin{theorem}[LPDuality feasible weak duality]
\label{thm:lpduality-feasible-weak-duality}
Let an $\LPDualityUp$ row over an ordered field be given as in \autoref{def:lpduality-finite-ordered-field-feasibility-row}. If $x$ is primal feasible and $y$ is dual feasible,
$$
\mathsf{PrFeas}_{A,b}(x)\land\mathsf{DuFeas}_{A,b,c}(y),
$$
then the primal objective value at $x$ is less than or classifier-equal to the dual objective value at $y$:
$$
\mathsf{PrObj}_{A,c}(x)\preceq_F\mathsf{DuObj}_{b}(y).
$$
Equivalently, the $\TotalOrderUp$ comparison between the two objective endpoints cannot select the strict-right branch.
\end{theorem}
\leanchecked{BEDC.Derived.LPDualityUp.LPDualityFeasibleWeakDuality\_prefix\_order}
\begin{proof}
Unpack the feasibility predicates of \autoref{def:lpduality-finite-ordered-field-feasibility-row}. From primal feasibility, for each row position $i\in I$ there is a constraint comparison
$$
\alpha_i(x)\preceq_F b_i,
$$
and from dual feasibility there is a nonnegativity witness $\mathsf{NonNeg}_F(y_i)$. Multiplication monotonicity by the nonnegative scalar $y_i$ gives
$$
y_i\cdot_F\alpha_i(x)\preceq_F y_i\cdot_Fb_i.
$$
Finite-sum monotonicity over $I$ yields
$$
\sum_{i\in I}y_i\cdot_F\alpha_i(x)
\preceq_F
\sum_{i\in I}y_i\cdot_Fb_i.
$$
The right-hand side is $\mathsf{DuObj}_{b}(y)$ by definition.

From dual feasibility, for each column position $j\in J$ there is a comparison
$$
c_j\preceq_F\beta_j(y),
$$
and from primal feasibility there is a nonnegativity witness $\mathsf{NonNeg}_F(x_j)$. Right multiplication monotonicity by the nonnegative scalar $x_j$ gives
$$
c_j\cdot_Fx_j\preceq_F\beta_j(y)\cdot_Fx_j.
$$
Finite-sum monotonicity over $J$ yields
$$
\sum_{j\in J}c_j\cdot_Fx_j
\preceq_F
\sum_{j\in J}\beta_j(y)\cdot_Fx_j.
$$
The left-hand side is $\mathsf{PrObj}_{A,c}(x)$ by definition.

The finite distributivity and interchange row gives the classifier comparison
$$
\sum_{j\in J}\beta_j(y)\cdot_Fx_j
\sim_F
\sum_{i\in I}y_i\cdot_F\alpha_i(x).
$$
Classifier stability of $\preceq_F$, applied to reflexivity at the left endpoint of this comparison, turns it into the order comparison
$$
\sum_{j\in J}\beta_j(y)\cdot_Fx_j
\preceq_F
\sum_{i\in I}y_i\cdot_F\alpha_i(x).
$$
Composing the three displayed comparisons by transitivity of the $\TotalOrderUp$ order gives
$$
\mathsf{PrObj}_{A,c}(x)
\preceq_F
\mathsf{DuObj}_{b}(y).
$$
This is exactly the weak-duality comparison for the feasible primal and dual witnesses.
\end{proof}

\begin{corollary}[LPDuality weak-duality equality gives optimality]
\label{cor:lpduality-weak-duality-equality-optimality}
Let an $\LPDualityUp$ row over an ordered field be given as in \autoref{def:lpduality-finite-ordered-field-feasibility-row}. Suppose $x$ is primal feasible and $y$ is dual feasible, and suppose their objective endpoints are equal under the scalar classifier:
$$
\mathsf{PrFeas}_{A,b}(x)\land\mathsf{DuFeas}_{A,b,c}(y),
\qquad
\mathsf{PrObj}_{A,c}(x)\sim_F\mathsf{DuObj}_{b}(y).
$$
Then $x$ is primal optimal and $y$ is dual optimal in the row sense:
$$
\forall x'.\;\mathsf{PrFeas}_{A,b}(x')
\Longrightarrow
\mathsf{PrObj}_{A,c}(x')\preceq_F\mathsf{PrObj}_{A,c}(x),
$$
and
$$
\forall y'.\;\mathsf{DuFeas}_{A,b,c}(y')
\Longrightarrow
\mathsf{DuObj}_{b}(y)\preceq_F\mathsf{DuObj}_{b}(y').
$$
\end{corollary}
\begin{proof}
For the primal optimality clause, fix $x'$ with $\mathsf{PrFeas}_{A,b}(x')$. Applying \autoref{thm:lpduality-feasible-weak-duality} to $x'$ and the given dual feasible witness $y$ gives
$$
\mathsf{PrObj}_{A,c}(x')\preceq_F\mathsf{DuObj}_{b}(y).
$$
The scalar classifier is reflexive at $\mathsf{PrObj}_{A,c}(x')$, and symmetry of the displayed classifier equality gives
$$
\mathsf{DuObj}_{b}(y)\sim_F\mathsf{PrObj}_{A,c}(x).
$$
The classifier-stability row for $\preceq_F$ in \autoref{def:lpduality-finite-ordered-field-feasibility-row} transports the right endpoint of the weak-duality comparison, yielding
$$
\mathsf{PrObj}_{A,c}(x')\preceq_F\mathsf{PrObj}_{A,c}(x).
$$
Since $x'$ was arbitrary, the first universal clause follows.

For the dual optimality clause, fix $y'$ with $\mathsf{DuFeas}_{A,b,c}(y')$. Applying \autoref{thm:lpduality-feasible-weak-duality} to the given primal feasible witness $x$ and $y'$ gives
$$
\mathsf{PrObj}_{A,c}(x)\preceq_F\mathsf{DuObj}_{b}(y').
$$
The displayed classifier equality gives
$$
\mathsf{PrObj}_{A,c}(x)\sim_F\mathsf{DuObj}_{b}(y),
$$
and reflexivity of the scalar classifier holds at $\mathsf{DuObj}_{b}(y')$. The same classifier-stability row transports the left endpoint of the weak-duality comparison, yielding
$$
\mathsf{DuObj}_{b}(y)\preceq_F\mathsf{DuObj}_{b}(y').
$$
Since $y'$ was arbitrary, the second universal clause follows.
\end{proof}
\leanchecked{BEDC.Derived.LPDualityUp.LPDualityWeakDualityEquality\_optimality}

\begin{theorem}[LPDuality complementary slackness gives objective equality]
\label{thm:lpduality-complementary-slackness-objective-equality}
Let an $\LPDualityUp$ row over an ordered field be given as in \autoref{def:lpduality-finite-ordered-field-feasibility-row}. Suppose $x$ is primal feasible and $y$ is dual feasible,
$$
\mathsf{PrFeas}_{A,b}(x)\land\mathsf{DuFeas}_{A,b,c}(y),
$$
and suppose the two complementary-slackness product families are classifier-equal termwise:
$$
\forall j\in J.\; c_j\cdot_Fx_j\sim_F\beta_j(y)\cdot_Fx_j,
\qquad
\forall i\in I.\; y_i\cdot_F\alpha_i(x)\sim_Fy_i\cdot_Fb_i.
$$
Then the primal and dual objective endpoints are classifier-equal,
$$
\mathsf{PrObj}_{A,c}(x)\sim_F\mathsf{DuObj}_{b}(y).
$$
Moreover, $x$ is primal optimal and $y$ is dual optimal in the row sense:
$$
\forall x'.\;\mathsf{PrFeas}_{A,b}(x')
\Longrightarrow
\mathsf{PrObj}_{A,c}(x')\preceq_F\mathsf{PrObj}_{A,c}(x),
$$
and
$$
\forall y'.\;\mathsf{DuFeas}_{A,b,c}(y')
\Longrightarrow
\mathsf{DuObj}_{b}(y)\preceq_F\mathsf{DuObj}_{b}(y').
$$
\end{theorem}
\leanchecked{BEDC.Derived.LPDualityUp.LPDualityComplementarySlackness\_objective\_equality}
\begin{proof}
For each column position $j\in J$, order reflexivity gives
$$
c_j\cdot_Fx_j\preceq_Fc_j\cdot_Fx_j
\qquad\text{and}\qquad
\beta_j(y)\cdot_Fx_j\preceq_F\beta_j(y)\cdot_Fx_j.
$$
Transporting these reflexive comparisons by the classifier-stability row for $\preceq_F$ and by the displayed column slack equality, in both directions, gives
$$
c_j\cdot_Fx_j\preceq_F\beta_j(y)\cdot_Fx_j
\qquad\text{and}\qquad
\beta_j(y)\cdot_Fx_j\preceq_Fc_j\cdot_Fx_j.
$$
Finite-sum monotonicity over $J$ therefore yields the opposing comparisons
$$
\sum_{j\in J}c_j\cdot_Fx_j
\preceq_F
\sum_{j\in J}\beta_j(y)\cdot_Fx_j,
\qquad
\sum_{j\in J}\beta_j(y)\cdot_Fx_j
\preceq_F
\sum_{j\in J}c_j\cdot_Fx_j.
$$
Antisymmetry of the ordered-field order gives the classifier equality
$$
\mathsf{PrObj}_{A,c}(x)
\sim_F
\sum_{j\in J}\beta_j(y)\cdot_Fx_j.
$$

The same argument over the row positions uses the displayed row slack equality. For each $i\in I$, classifier stability transports the two reflexive order comparisons into
$$
y_i\cdot_F\alpha_i(x)\preceq_Fy_i\cdot_Fb_i
\qquad\text{and}\qquad
y_i\cdot_Fb_i\preceq_Fy_i\cdot_F\alpha_i(x).
$$
Finite-sum monotonicity over $I$, followed by antisymmetry, gives
$$
\sum_{i\in I}y_i\cdot_F\alpha_i(x)
\sim_F
\mathsf{DuObj}_{b}(y).
$$

The finite distributivity and interchange row in \autoref{def:lpduality-finite-ordered-field-feasibility-row} supplies
$$
\sum_{j\in J}\beta_j(y)\cdot_Fx_j
\sim_F
\sum_{i\in I}y_i\cdot_F\alpha_i(x).
$$
Transitivity of the scalar classifier composes the three displayed classifier equalities into
$$
\mathsf{PrObj}_{A,c}(x)\sim_F\mathsf{DuObj}_{b}(y).
$$

The feasibility hypotheses and this objective equality are precisely the hypotheses of \autoref{cor:lpduality-weak-duality-equality-optimality}. Applying that corollary gives the stated primal and dual optimality clauses.
\end{proof}

\begin{theorem}[LPDuality dual feasibility binary convex closure]
\label{thm:lpduality-dual-feasibility-binary-convex-closure}
Let an $\LPDualityUp$ row over an ordered field be given as in \autoref{def:lpduality-finite-ordered-field-feasibility-row}. Let $y,y':I\to F$ be dual feasible witnesses, and let $\lambda,\mu:F$ be scalar coefficients such that
$$
\mathsf{DuFeas}_{A,b,c}(y)
\land
\mathsf{DuFeas}_{A,b,c}(y')
\land
\mathsf{NonNeg}_F(\lambda)
\land
\mathsf{NonNeg}_F(\mu)
\land
(\lambda+_F\mu)\sim_F1_F.
$$
Define $\eta:I\to F$ componentwise by
$$
\eta_i:=\lambda\cdot_F y_i+_F\mu\cdot_F y'_i
\qquad (i\in I).
$$
Then
$$
\mathsf{DuFeas}_{A,b,c}(\eta).
$$
\end{theorem}
\begin{proof}
Unpack the two dual-feasibility hypotheses from \autoref{def:lpduality-finite-ordered-field-feasibility-row}. They give
$$
\forall i\in I.\; \mathsf{NonNeg}_F(y_i),
\qquad
\forall i\in I.\; \mathsf{NonNeg}_F(y'_i),
$$
and
$$
\forall j\in J.\; c_j\preceq_F\beta_j(y),
\qquad
\forall j\in J.\; c_j\preceq_F\beta_j(y').
$$
We verify the two clauses of $\mathsf{DuFeas}_{A,b,c}(\eta)$.

First fix $i\in I$. Since $\mathsf{NonNeg}_F(y_i)$ and $\mathsf{NonNeg}_F(y'_i)$ mean $0_F\preceq_F y_i$ and $0_F\preceq_F y'_i$, left multiplication monotonicity by the nonnegative scalars $\lambda$ and $\mu$ gives
$$
\lambda\cdot_F0_F\preceq_F\lambda\cdot_F y_i,
\qquad
\mu\cdot_F0_F\preceq_F\mu\cdot_F y'_i.
$$
Apply finite-sum monotonicity over a two-point carrier to these two comparisons. This yields
$$
(\lambda\cdot_F0_F)+_F(\mu\cdot_F0_F)
\preceq_F
(\lambda\cdot_F y_i)+_F(\mu\cdot_F y'_i).
$$
The field rows from \autoref{ch:concrete-instances-field-namecert} identify the left endpoint with $0_F$, using zero multiplication and the additive unit law. By the definition of $\eta_i$, the right endpoint is $\eta_i$. Classifier stability of $\preceq_F$ therefore gives
$$
0_F\preceq_F\eta_i.
$$
Thus $\mathsf{NonNeg}_F(\eta_i)$ holds for every $i\in I$.

Next fix $j\in J$. Dual feasibility of $y$ and $y'$ gives
$$
c_j\preceq_F\beta_j(y),
\qquad
c_j\preceq_F\beta_j(y').
$$
Left multiplication monotonicity by the nonnegative scalars $\lambda$ and $\mu$ yields
$$
\lambda\cdot_F c_j\preceq_F\lambda\cdot_F\beta_j(y),
\qquad
\mu\cdot_F c_j\preceq_F\mu\cdot_F\beta_j(y').
$$
Finite-sum monotonicity over a two-point carrier gives
$$
(\lambda\cdot_F c_j)+_F(\mu\cdot_F c_j)
\preceq_F
(\lambda\cdot_F\beta_j(y))+_F(\mu\cdot_F\beta_j(y')).
$$
On the left, distributivity and the coefficient-sum hypothesis give the classifier chain
$$
(\lambda\cdot_F c_j)+_F(\mu\cdot_F c_j)
\sim_F
(\lambda+_F\mu)\cdot_F c_j
\sim_F
1_F\cdot_F c_j
\sim_F
c_j.
$$
On the right, expand the definition of $\beta_j$ and use distributivity, associativity, and finite-sum compatibility with the scalar classifier from \autoref{ch:concrete-instances-field-namecert}:
$$
(\lambda\cdot_F\beta_j(y))+_F(\mu\cdot_F\beta_j(y'))
\sim_F
\sum_{i\in I}(\lambda\cdot_F y_i)\cdot_F A_{ij}
+_F
\sum_{i\in I}(\mu\cdot_F y'_i)\cdot_F A_{ij}
$$
and hence
$$
(\lambda\cdot_F\beta_j(y))+_F(\mu\cdot_F\beta_j(y'))
\sim_F
\sum_{i\in I}(\lambda\cdot_F y_i+_F\mu\cdot_F y'_i)\cdot_F A_{ij}
=
\beta_j(\eta).
$$
Classifier stability of $\preceq_F$ transports the displayed order comparison along these left and right classifier equalities, yielding
$$
c_j\preceq_F\beta_j(\eta).
$$
Thus the dual constraint clause holds for every $j\in J$.

We have shown both
$$
\forall i\in I.\; \mathsf{NonNeg}_F(\eta_i)
$$
and
$$
\forall j\in J.\; c_j\preceq_F\beta_j(\eta).
$$
By the definition of dual feasibility in \autoref{def:lpduality-finite-ordered-field-feasibility-row}, this is exactly
$$
\mathsf{DuFeas}_{A,b,c}(\eta).
$$
\end{proof}

\begin{theorem}[B-516: $\LPDualityUp$ primal feasibility binary convex closure]
\label{thm:lpduality-primal-feasibility-binary-convex-closure}
Let $F$ be the ordered-field scalar interface carried by a finite $\LPDualityUp$ row certificate as in \autoref{def:lpduality-finite-ordered-field-feasibility-row}. Let $J$ be the finite column index object, let $I_0$ be any finite family of certified primal rows, let
$$
A:I_0\times J\to F,
\qquad
b:I_0\to F
$$
be the row-coefficient and right-hand-side data, and let
$$
x,x':J\to F
$$
be two primal vectors. Write
$$
\mathsf{Row}_{i}(z):=\sum_{j\in J} A_{ij}\cdot_F z_j
$$
for the finite row evaluation, and define
$$
\mathsf{PrFeas}_{A,b}^{I_0}(z)
\quad:\Longleftrightarrow\quad
\forall i\in I_0,\; \mathsf{Row}_{i}(z)\preceq_F b_i .
$$
Assume
$$
\mathsf{PrFeas}_{A,b}^{I_0}(x),
\qquad
\mathsf{PrFeas}_{A,b}^{I_0}(x'),
$$
and let $\alpha,\beta:F$ satisfy
$$
\mathsf{NonNeg}_F(\alpha),
\qquad
\mathsf{NonNeg}_F(\beta),
\qquad
\alpha+_F\beta\sim_F 1_F .
$$
Define the componentwise binary affine mixture
$$
\lambda_j:=\alpha\cdot_F x_j+_F\beta\cdot_F x'_j,
\qquad
j\in J .
$$
The finite row certificate records the row-affine-linearity comparison for this mixture:
$$
\mathsf{Row}_{i}(\lambda)
\sim_F
\alpha\cdot_F \mathsf{Row}_{i}(x)+_F\beta\cdot_F \mathsf{Row}_{i}(x')
\qquad
(i\in I_0).
$$
The scalar order portion of the same certificate records binary order-addition monotonicity, the two-point finite-sum monotonicity instance of \autoref{def:lpduality-finite-ordered-field-feasibility-row}:
$$
p\preceq_F q
\land
r\preceq_F s
\Longrightarrow
p+_F r\preceq_F q+_F s .
$$
Then
$$
\mathsf{PrFeas}_{A,b}^{I_0}(\lambda).
$$
Moreover, the closure is row-local and coefficient-sign-free: no hypothesis of the form $\mathsf{NonNeg}_F(A_{ij})$ is required for any row coefficient.
\end{theorem}
\begin{proof}
Fix a row $i\in I_0$. By the definition of $\lambda$, field distributivity, commutativity of scalar multiplication, and the finite-sum linearity certificate over the finite column object $J$ recorded by the row-affine-linearity field displayed in the theorem statement,
$$
\begin{aligned}
\mathsf{Row}_{i}(\lambda)
&=\sum_{j\in J} A_{ij}\cdot_F(\alpha\cdot_F x_j+_F\beta\cdot_F x'_j)\\
&\sim_F
\sum_{j\in J}\bigl(\alpha\cdot_F(A_{ij}\cdot_F x_j)+_F
\beta\cdot_F(A_{ij}\cdot_F x'_j)\bigr)\\
&\sim_F
\alpha\cdot_F\sum_{j\in J}A_{ij}\cdot_F x_j
+_F
\beta\cdot_F\sum_{j\in J}A_{ij}\cdot_F x'_j\\
&=
\alpha\cdot_F \mathsf{Row}_{i}(x)+_F\beta\cdot_F \mathsf{Row}_{i}(x').
\end{aligned}
$$
The two feasibility assumptions give
$$
\mathsf{Row}_{i}(x)\preceq_F b_i,
\qquad
\mathsf{Row}_{i}(x')\preceq_F b_i .
$$
Since $\alpha$ and $\beta$ are nonnegative, the multiplication-monotonicity row of \autoref{def:lpduality-finite-ordered-field-feasibility-row} yields
$$
\alpha\cdot_F\mathsf{Row}_{i}(x)\preceq_F \alpha\cdot_F b_i,
\qquad
\beta\cdot_F\mathsf{Row}_{i}(x')\preceq_F \beta\cdot_F b_i .
$$
The binary order-addition monotonicity row displayed in the theorem statement gives
$$
\alpha\cdot_F\mathsf{Row}_{i}(x)+_F\beta\cdot_F\mathsf{Row}_{i}(x')
\preceq_F
\alpha\cdot_F b_i+_F\beta\cdot_F b_i .
$$
By right distributivity and the convex-weight certificate,
$$
\alpha\cdot_F b_i+_F\beta\cdot_F b_i
\sim_F
(\alpha+_F\beta)\cdot_F b_i
\sim_F
1_F\cdot_F b_i
\sim_F
b_i .
$$
The classifier-stability row for $\preceq_F$ recorded in \autoref{def:lpduality-finite-ordered-field-feasibility-row}, applied to the row-evaluation classifier comparison above and to the final scalar classifier comparison, yields
$$
\mathsf{Row}_{i}(\lambda)\preceq_F b_i .
$$
Because $i\in I_0$ was arbitrary, $\mathsf{PrFeas}_{A,b}^{I_0}(\lambda)$ follows.
The proof used only finite-sum linearity, commutative-field distributivity, ordered-field monotonicity under nonnegative scalar multiplication, additive monotonicity of the order through the displayed two-point finite-sum row, and classifier stability for $\preceq_F$ from \autoref{def:lpduality-finite-ordered-field-feasibility-row}. The entries $A_{ij}$ were never compared with $0_F$. Hence the closure is local in the chosen row ledger $I_0$ and does not require any coefficient-sign side condition.
\end{proof}

\begin{lemma}[LPDuality primal objective binary affine readback]
\label{lem:lpduality-primal-objective-binary-affine-readback}
Let an $\LPDualityUp$ row over an ordered field be given as in \autoref{def:lpduality-finite-ordered-field-feasibility-row}. Let $x,x':J\to F$ be primal vectors, let $\alpha,\beta:F$ be scalar coefficients, and define $\lambda:J\to F$ componentwise by
$$
\lambda_j:=\alpha\cdot_F x_j+_F\beta\cdot_F x'_j
\qquad (j\in J).
$$
Then the primal objective is affine over this binary mixture:
$$
\mathsf{PrObj}_{A,c}(\lambda)
\sim_F
\alpha\cdot_F\mathsf{PrObj}_{A,c}(x)+_F
\beta\cdot_F\mathsf{PrObj}_{A,c}(x').
$$
\end{lemma}
\begin{proof}
Expand the objective endpoint from \autoref{def:lpduality-finite-ordered-field-feasibility-row}. By the definition of $\lambda$, distributivity in the carried $\FieldUp$ row, commutativity and associativity of multiplication, and finite-sum compatibility with the scalar classifier, one has
$$
\begin{aligned}
\mathsf{PrObj}_{A,c}(\lambda)
&=\sum_{j\in J}c_j\cdot_F(\alpha\cdot_F x_j+_F\beta\cdot_F x'_j)\\
&\sim_F
\sum_{j\in J}\bigl(\alpha\cdot_F(c_j\cdot_Fx_j)+_F
\beta\cdot_F(c_j\cdot_Fx'_j)\bigr)\\
&\sim_F
\alpha\cdot_F\sum_{j\in J}c_j\cdot_Fx_j
+_F
\beta\cdot_F\sum_{j\in J}c_j\cdot_Fx'_j\\
&=
\alpha\cdot_F\mathsf{PrObj}_{A,c}(x)+_F
\beta\cdot_F\mathsf{PrObj}_{A,c}(x').
\end{aligned}
$$
This is the asserted classifier equality.
\end{proof}

\begin{theorem}[LPDuality optimal primal face binary convex closure]
\label{thm:lpduality-optimal-primal-face-binary-convex-closure}
Let an $\LPDualityUp$ row over an ordered field be given as in \autoref{def:lpduality-finite-ordered-field-feasibility-row}. Let $x,x':J\to F$ be primal feasible witnesses, let $\tau:F$ be a scalar endpoint, and suppose both witnesses lie in the same primal objective fiber:
$$
\mathsf{PrFeas}_{A,b}(x),
\qquad
\mathsf{PrFeas}_{A,b}(x'),
$$
$$
\mathsf{PrObj}_{A,c}(x)\sim_F\tau,
\qquad
\mathsf{PrObj}_{A,c}(x')\sim_F\tau.
$$
Let $\alpha,\beta:F$ satisfy
$$
\mathsf{NonNeg}_F(\alpha),
\qquad
\mathsf{NonNeg}_F(\beta),
\qquad
\alpha+_F\beta\sim_F1_F,
$$
and define the componentwise binary affine mixture
$$
\lambda_j:=\alpha\cdot_F x_j+_F\beta\cdot_F x'_j
\qquad (j\in J).
$$
Then $\lambda$ is primal feasible and remains in the same primal objective fiber:
$$
\mathsf{PrFeas}_{A,b}(\lambda),
\qquad
\mathsf{PrObj}_{A,c}(\lambda)\sim_F\tau.
$$
Consequently, when $\tau$ is the scalar endpoint certified as the primal optimum of the row, the certified primal optimal face at $\tau$ is closed under binary affine combinations with nonnegative coefficients whose sum is classifier-equal to $1_F$.
\end{theorem}
\begin{proof}
Unpack the two primal-feasibility hypotheses using \autoref{def:lpduality-finite-ordered-field-feasibility-row}. They give
$$
\forall j\in J.\;\mathsf{NonNeg}_F(x_j),
\qquad
\forall j\in J.\;\mathsf{NonNeg}_F(x'_j),
$$
and
$$
\forall i\in I.\;\alpha_i(x)\preceq_F b_i,
\qquad
\forall i\in I.\;\alpha_i(x')\preceq_F b_i.
$$
First fix $j\in J$. Since $\mathsf{NonNeg}_F(x_j)$ and $\mathsf{NonNeg}_F(x'_j)$ mean $0_F\preceq_Fx_j$ and $0_F\preceq_Fx'_j$, multiplication monotonicity by the nonnegative scalars $\alpha$ and $\beta$ gives
$$
\alpha\cdot_F0_F\preceq_F\alpha\cdot_Fx_j,
\qquad
\beta\cdot_F0_F\preceq_F\beta\cdot_Fx'_j.
$$
The two-point finite-sum monotonicity instance in \autoref{def:lpduality-finite-ordered-field-feasibility-row} yields
$$
\alpha\cdot_F0_F+_F\beta\cdot_F0_F
\preceq_F
\alpha\cdot_Fx_j+_F\beta\cdot_Fx'_j.
$$
The $\FieldUp$ zero laws identify the left endpoint with $0_F$, and the definition of $\lambda_j$ identifies the right endpoint with $\lambda_j$. Classifier stability of $\preceq_F$ therefore gives
$$
0_F\preceq_F\lambda_j.
$$
Thus $\mathsf{NonNeg}_F(\lambda_j)$ holds for every $j\in J$.

For the row inequalities, apply \autoref{thm:lpduality-primal-feasibility-binary-convex-closure} to the finite row family $I$, with $\mathsf{Row}_i(z)=\alpha_i(z)$. The displayed row-constraint clauses for $x$ and $x'$, together with the same nonnegative coefficients and the classifier equality $\alpha+_F\beta\sim_F1_F$, give
$$
\forall i\in I.\;\alpha_i(\lambda)\preceq_F b_i.
$$
Combining this row-constraint clause with the nonnegativity clause just proved is exactly
$$
\mathsf{PrFeas}_{A,b}(\lambda)
$$
by \autoref{def:lpduality-finite-ordered-field-feasibility-row}.

It remains to read back the objective endpoint. By \autoref{lem:lpduality-primal-objective-binary-affine-readback},
$$
\mathsf{PrObj}_{A,c}(\lambda)
\sim_F
\alpha\cdot_F\mathsf{PrObj}_{A,c}(x)+_F
\beta\cdot_F\mathsf{PrObj}_{A,c}(x').
$$
Scalar classifier congruence transports the two objective-fiber hypotheses through multiplication and addition, giving
$$
\alpha\cdot_F\mathsf{PrObj}_{A,c}(x)+_F
\beta\cdot_F\mathsf{PrObj}_{A,c}(x')
\sim_F
\alpha\cdot_F\tau+_F\beta\cdot_F\tau.
$$
Distributivity, the coefficient-sum classifier equality, and the multiplicative unit law give
$$
\alpha\cdot_F\tau+_F\beta\cdot_F\tau
\sim_F
(\alpha+_F\beta)\cdot_F\tau
\sim_F
1_F\cdot_F\tau
\sim_F
\tau.
$$
Transitivity of the scalar classifier composes the displayed classifier equalities into
$$
\mathsf{PrObj}_{A,c}(\lambda)\sim_F\tau.
$$
The final sentence is the specialization of the same statement to a scalar endpoint already certified as optimal for the row.
\end{proof}
