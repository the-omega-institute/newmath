The linear-programming-duality interface $\LPDualityUp$ packages a primal/dual pair of linear programs together with strong-duality and complementary-slackness certificates over an ordered field. It depends on $\NameCert_{\ConvexSetUp}$ (\autoref{ch:concrete-instances-convexset-namecert}) and $\NameCert_{\PolytopeUp}$ (\autoref{ch:concrete-instances-polytope-namecert}).

\begin{definition}[Finite $\LPDualityUp$ ordered-field feasibility row]
\label{def:lpduality-finite-ordered-field-feasibility-row}
An $\LPDualityUp$ row over an ordered field consists of finite row-position and column-position carriers $I$ and $J$, a scalar carrier $F$ equipped with the $\FieldUp$ operations and classifier of \autoref{ch:concrete-instances-field-namecert}, and a $\TotalOrderUp$ comparison on the same scalar carrier and classifier as in \autoref{ch:concrete-instances-totalorder-namecert}. Write $+_F$, $\cdot_F$, $0_F$, $\sim_F$, and $\preceq_F$ for these data, with $u\preceq_F v$ meaning that $u$ is less than or classifier-equal to $v$. The nonnegative scalar predicate is
$$
\mathsf{NonNeg}_F(a)\Longleftrightarrow 0_F\preceq_F a.
$$
The scalar order rows used by the certificate are classifier stability of $\preceq_F$,
$$
u\sim_Fu'\land v\sim_Fv'\land u\preceq_Fv
\Longrightarrow u'\preceq_Fv',
$$
left and right multiplication monotonicity by nonnegative scalars,
$$
\mathsf{NonNeg}_F(a)\land u\preceq_Fv
\Longrightarrow
(a\cdot_Fu\preceq_Fa\cdot_Fv)
\land
(u\cdot_Fa\preceq_Fv\cdot_Fa),
$$
and finite-sum monotonicity over every finite position carrier $K$,
$$
\Bigl(\forall k\in K.\; r_k\preceq_Fs_k\Bigr)
\Longrightarrow
\sum_{k\in K}r_k\preceq_F\sum_{k\in K}s_k.
$$

For coefficients $A:I\times J\to F$, right-hand side $b:I\to F$, and cost vector $c:J\to F$, define
$$
\alpha_i(x):=\sum_{j\in J} A_{ij}\cdot_F x_j,
\qquad
\beta_j(y):=\sum_{i\in I} y_i\cdot_F A_{ij}.
$$
The finite distributivity and interchange row records the classifier comparison
$$
\sum_{j\in J}\beta_j(y)\cdot_Fx_j
\sim_F
\sum_{i\in I}y_i\cdot_F\alpha_i(x).
$$
The primal and dual feasible carriers, read as the row descriptions of the corresponding $\PolytopeUp$ feasible sets over the convex substrate of \autoref{ch:concrete-instances-convexset-namecert} and \autoref{ch:concrete-instances-polytope-namecert}, are
$$
\mathsf{PrFeas}_{A,b}(x)\Longleftrightarrow
\Bigl(\forall j\in J.\;\mathsf{NonNeg}_F(x_j)\Bigr)
\land
\Bigl(\forall i\in I.\;\alpha_i(x)\preceq_F b_i\Bigr)
$$
and
$$
\mathsf{DuFeas}_{A,b,c}(y)\Longleftrightarrow
\Bigl(\forall i\in I.\;\mathsf{NonNeg}_F(y_i)\Bigr)
\land
\Bigl(\forall j\in J.\;c_j\preceq_F\beta_j(y)\Bigr).
$$
The objective endpoints are
$$
\mathsf{PrObj}_{A,c}(x):=\sum_{j\in J} c_j\cdot_Fx_j,
\qquad
\mathsf{DuObj}_{b}(y):=\sum_{i\in I} y_i\cdot_Fb_i.
$$
\end{definition}

\begin{theorem}[LPDuality feasible weak duality]
\label{thm:lpduality-feasible-weak-duality}
Let an $\LPDualityUp$ row over an ordered field be given as in \autoref{def:lpduality-finite-ordered-field-feasibility-row}. If $x$ is primal feasible and $y$ is dual feasible,
$$
\mathsf{PrFeas}_{A,b}(x)\land\mathsf{DuFeas}_{A,b,c}(y),
$$
then the primal objective value at $x$ is less than or classifier-equal to the dual objective value at $y$:
$$
\mathsf{PrObj}_{A,c}(x)\preceq_F\mathsf{DuObj}_{b}(y).
$$
Equivalently, the $\TotalOrderUp$ comparison between the two objective endpoints cannot select the strict-right branch.
\end{theorem}
\leanchecked{BEDC.Derived.LPDualityUp.LPDualityFeasibleWeakDuality\_prefix\_order}
\begin{proof}
Unpack the feasibility predicates of \autoref{def:lpduality-finite-ordered-field-feasibility-row}. From primal feasibility, for each row position $i\in I$ there is a constraint comparison
$$
\alpha_i(x)\preceq_F b_i,
$$
and from dual feasibility there is a nonnegativity witness $\mathsf{NonNeg}_F(y_i)$. Multiplication monotonicity by the nonnegative scalar $y_i$ gives
$$
y_i\cdot_F\alpha_i(x)\preceq_F y_i\cdot_Fb_i.
$$
Finite-sum monotonicity over $I$ yields
$$
\sum_{i\in I}y_i\cdot_F\alpha_i(x)
\preceq_F
\sum_{i\in I}y_i\cdot_Fb_i.
$$
The right-hand side is $\mathsf{DuObj}_{b}(y)$ by definition.

From dual feasibility, for each column position $j\in J$ there is a comparison
$$
c_j\preceq_F\beta_j(y),
$$
and from primal feasibility there is a nonnegativity witness $\mathsf{NonNeg}_F(x_j)$. Right multiplication monotonicity by the nonnegative scalar $x_j$ gives
$$
c_j\cdot_Fx_j\preceq_F\beta_j(y)\cdot_Fx_j.
$$
Finite-sum monotonicity over $J$ yields
$$
\sum_{j\in J}c_j\cdot_Fx_j
\preceq_F
\sum_{j\in J}\beta_j(y)\cdot_Fx_j.
$$
The left-hand side is $\mathsf{PrObj}_{A,c}(x)$ by definition.

The finite distributivity and interchange row gives the classifier comparison
$$
\sum_{j\in J}\beta_j(y)\cdot_Fx_j
\sim_F
\sum_{i\in I}y_i\cdot_F\alpha_i(x).
$$
Classifier stability of $\preceq_F$, applied to reflexivity at the left endpoint of this comparison, turns it into the order comparison
$$
\sum_{j\in J}\beta_j(y)\cdot_Fx_j
\preceq_F
\sum_{i\in I}y_i\cdot_F\alpha_i(x).
$$
Composing the three displayed comparisons by transitivity of the $\TotalOrderUp$ order gives
$$
\mathsf{PrObj}_{A,c}(x)
\preceq_F
\mathsf{DuObj}_{b}(y).
$$
This is exactly the weak-duality comparison for the feasible primal and dual witnesses.
\end{proof}

\begin{corollary}[LPDuality weak-duality equality gives optimality]
\label{cor:lpduality-weak-duality-equality-optimality}
Let an $\LPDualityUp$ row over an ordered field be given as in \autoref{def:lpduality-finite-ordered-field-feasibility-row}. Suppose $x$ is primal feasible and $y$ is dual feasible, and suppose their objective endpoints are equal under the scalar classifier:
$$
\mathsf{PrFeas}_{A,b}(x)\land\mathsf{DuFeas}_{A,b,c}(y),
\qquad
\mathsf{PrObj}_{A,c}(x)\sim_F\mathsf{DuObj}_{b}(y).
$$
Then $x$ is primal optimal and $y$ is dual optimal in the row sense:
$$
\forall x'.\;\mathsf{PrFeas}_{A,b}(x')
\Longrightarrow
\mathsf{PrObj}_{A,c}(x')\preceq_F\mathsf{PrObj}_{A,c}(x),
$$
and
$$
\forall y'.\;\mathsf{DuFeas}_{A,b,c}(y')
\Longrightarrow
\mathsf{DuObj}_{b}(y)\preceq_F\mathsf{DuObj}_{b}(y').
$$
\end{corollary}
\begin{proof}
For the primal optimality clause, fix $x'$ with $\mathsf{PrFeas}_{A,b}(x')$. Applying \autoref{thm:lpduality-feasible-weak-duality} to $x'$ and the given dual feasible witness $y$ gives
$$
\mathsf{PrObj}_{A,c}(x')\preceq_F\mathsf{DuObj}_{b}(y).
$$
The scalar classifier is reflexive at $\mathsf{PrObj}_{A,c}(x')$, and symmetry of the displayed classifier equality gives
$$
\mathsf{DuObj}_{b}(y)\sim_F\mathsf{PrObj}_{A,c}(x).
$$
The classifier-stability row for $\preceq_F$ in \autoref{def:lpduality-finite-ordered-field-feasibility-row} transports the right endpoint of the weak-duality comparison, yielding
$$
\mathsf{PrObj}_{A,c}(x')\preceq_F\mathsf{PrObj}_{A,c}(x).
$$
Since $x'$ was arbitrary, the first universal clause follows.

For the dual optimality clause, fix $y'$ with $\mathsf{DuFeas}_{A,b,c}(y')$. Applying \autoref{thm:lpduality-feasible-weak-duality} to the given primal feasible witness $x$ and $y'$ gives
$$
\mathsf{PrObj}_{A,c}(x)\preceq_F\mathsf{DuObj}_{b}(y').
$$
The displayed classifier equality gives
$$
\mathsf{PrObj}_{A,c}(x)\sim_F\mathsf{DuObj}_{b}(y),
$$
and reflexivity of the scalar classifier holds at $\mathsf{DuObj}_{b}(y')$. The same classifier-stability row transports the left endpoint of the weak-duality comparison, yielding
$$
\mathsf{DuObj}_{b}(y)\preceq_F\mathsf{DuObj}_{b}(y').
$$
Since $y'$ was arbitrary, the second universal clause follows.
\end{proof}
\leanchecked{BEDC.Derived.LPDualityUp.LPDualityWeakDualityEquality\_optimality}

\begin{theorem}[LPDuality complementary slackness gives objective equality]
\label{thm:lpduality-complementary-slackness-objective-equality}
Let an $\LPDualityUp$ row over an ordered field be given as in \autoref{def:lpduality-finite-ordered-field-feasibility-row}. Suppose $x$ is primal feasible and $y$ is dual feasible,
$$
\mathsf{PrFeas}_{A,b}(x)\land\mathsf{DuFeas}_{A,b,c}(y),
$$
and suppose the two complementary-slackness product families are classifier-equal termwise:
$$
\forall j\in J.\; c_j\cdot_Fx_j\sim_F\beta_j(y)\cdot_Fx_j,
\qquad
\forall i\in I.\; y_i\cdot_F\alpha_i(x)\sim_Fy_i\cdot_Fb_i.
$$
Then the primal and dual objective endpoints are classifier-equal,
$$
\mathsf{PrObj}_{A,c}(x)\sim_F\mathsf{DuObj}_{b}(y).
$$
Moreover, $x$ is primal optimal and $y$ is dual optimal in the row sense:
$$
\forall x'.\;\mathsf{PrFeas}_{A,b}(x')
\Longrightarrow
\mathsf{PrObj}_{A,c}(x')\preceq_F\mathsf{PrObj}_{A,c}(x),
$$
and
$$
\forall y'.\;\mathsf{DuFeas}_{A,b,c}(y')
\Longrightarrow
\mathsf{DuObj}_{b}(y)\preceq_F\mathsf{DuObj}_{b}(y').
$$
\end{theorem}
\leanchecked{BEDC.Derived.LPDualityUp.LPDualityComplementarySlackness\_objective\_hsame}
\begin{proof}
For each column position $j\in J$, order reflexivity gives
$$
c_j\cdot_Fx_j\preceq_Fc_j\cdot_Fx_j
\qquad\text{and}\qquad
\beta_j(y)\cdot_Fx_j\preceq_F\beta_j(y)\cdot_Fx_j.
$$
Transporting these reflexive comparisons by the classifier-stability row for $\preceq_F$ and by the displayed column slack equality, in both directions, gives
$$
c_j\cdot_Fx_j\preceq_F\beta_j(y)\cdot_Fx_j
\qquad\text{and}\qquad
\beta_j(y)\cdot_Fx_j\preceq_Fc_j\cdot_Fx_j.
$$
Finite-sum monotonicity over $J$ therefore yields the opposing comparisons
$$
\sum_{j\in J}c_j\cdot_Fx_j
\preceq_F
\sum_{j\in J}\beta_j(y)\cdot_Fx_j,
\qquad
\sum_{j\in J}\beta_j(y)\cdot_Fx_j
\preceq_F
\sum_{j\in J}c_j\cdot_Fx_j.
$$
Antisymmetry of the ordered-field order gives the classifier equality
$$
\mathsf{PrObj}_{A,c}(x)
\sim_F
\sum_{j\in J}\beta_j(y)\cdot_Fx_j.
$$

The same argument over the row positions uses the displayed row slack equality. For each $i\in I$, classifier stability transports the two reflexive order comparisons into
$$
y_i\cdot_F\alpha_i(x)\preceq_Fy_i\cdot_Fb_i
\qquad\text{and}\qquad
y_i\cdot_Fb_i\preceq_Fy_i\cdot_F\alpha_i(x).
$$
Finite-sum monotonicity over $I$, followed by antisymmetry, gives
$$
\sum_{i\in I}y_i\cdot_F\alpha_i(x)
\sim_F
\mathsf{DuObj}_{b}(y).
$$

The finite distributivity and interchange row in \autoref{def:lpduality-finite-ordered-field-feasibility-row} supplies
$$
\sum_{j\in J}\beta_j(y)\cdot_Fx_j
\sim_F
\sum_{i\in I}y_i\cdot_F\alpha_i(x).
$$
Transitivity of the scalar classifier composes the three displayed classifier equalities into
$$
\mathsf{PrObj}_{A,c}(x)\sim_F\mathsf{DuObj}_{b}(y).
$$

The feasibility hypotheses and this objective equality are precisely the hypotheses of \autoref{cor:lpduality-weak-duality-equality-optimality}. Applying that corollary gives the stated primal and dual optimality clauses.
\end{proof}
