\chapter{A Concrete Naming Certificate for $\ResidueUp$}
\label{ch:concrete-instances-residue-namecert}

The residue interface $\ResidueUp$ packages the residue at an isolated singularity of a holomorphic function and the residue-theorem witness relating contour integrals to residue sums. It depends on $\NameCert_{\HolomorphicUp}$ (\autoref{ch:concrete-instances-holomorphic-namecert}) and $\NameCert_{\ContourUp}$ (\autoref{ch:concrete-instances-contour-integral}).

\section{Carrier definition}
\label{sec:residue-carrier}

\begin{definition}[Residue pole data]
\label{def:residue-pole-data}
The predicate \(\mathsf{ResiduePoleData}(f,z_0,r,p,g,I,\rho)\) records an open-disk witness \(\mathsf{HolomorphicOpenDisk}(z_0,r,p,g)\), complex-history carriers for the integral \(I\) and residue \(\rho\), a complex-pair carrier for \(f\), and the continuation witness \(\Cont(I,\rho,f)\).
\end{definition}
\leandef{BEDC.Derived.ResidueUp.ResiduePoleData}

\begin{theorem}[Residue pole data under \(\hsame\)]
\label{thm:residue-pole-data-hsame-transport}
If all seven histories in \(\mathsf{ResiduePoleData}(f,z_0,r,p,g,I,\rho)\) are transported along \(\hsame\), then the transported histories again satisfy \(\mathsf{ResiduePoleData}\), and the transported integral and residue continue to the transported function history.
\end{theorem}
\begin{proof}
Transport the open-disk witness through holomorphic-disk \(\hsame\)-stability. Transport the complex carriers for \(I\), \(\rho\), and the product carrier for \(f\) through the product-history carrier stability, then transport \(\Cont(I,\rho,f)\) componentwise along the same \(\hsame\) witnesses.
\end{proof}
\leanchecked{BEDC.Derived.ResidueUp.ResiduePoleData\_hsame\_transport}

\begin{theorem}[Residue pole data empty function endpoints]
\label{thm:residue-pole-data-empty-function-endpoints}
If $\mathsf{ResiduePoleData}(f,z_0,r,p,g,I,\rho)$ holds and
$f \hsame \emp$, then both residue endpoints are empty:
$$
I \hsame \emp \land \rho \hsame \emp.
$$
\end{theorem}
\begin{proof}
Destructure the residue pole data to expose $\mathsf{Cont}(I,\rho,f)$.
Transport the continuation result along $f \hsame \emp$, then apply empty-result
inversion for BEDC continuations to recover the two empty endpoints.
\end{proof}
\leanchecked{BEDC.Derived.ResidueUp.ResiduePoleData\_empty\_function\_endpoints}

\begin{theorem}[Residue pole data empty function endpoints iff]
\label{thm:residue-pole-data-empty-function-endpoints-iff}
If $\mathsf{ResiduePoleData}(f,z_0,r,p,g,I,\rho)$ holds, then the function
history is empty exactly when both residue endpoints are empty:
$$
f \hsame \emp \Longleftrightarrow I \hsame \emp \land \rho \hsame \emp.
$$
\end{theorem}
\begin{proof}
The forward implication is the empty-function endpoint theorem. Conversely,
unpack the residue pole data to expose $\Cont(I,\rho,f)$. Compare this
continuation with $\Cont(\emp,\emp,\emp)$ using the supplied endpoint
equalities; continuation result respect for $\hsame$ gives $f\hsame\emp$.
\end{proof}
\leanchecked{BEDC.Derived.ResidueUp.ResiduePoleData\_empty\_function\_endpoints\_iff}

\begin{theorem}[Residue pole data residue suffix closure]
\label{thm:residue-pole-data-residue-suffix-closure}
If \(\mathsf{ResiduePoleData}(f,z_0,r,p,g,I,\rho)\) holds, \(q\) is unary,
and \(\Cont(f,q,f_q)\), then there is a residue history \(\rho_q\) such that
\(\mathsf{ResiduePoleData}(f_q,z_0,r,p,g,I,\rho_q)\), with
\(\Cont(\rho,q,\rho_q)\) and \(\Cont(I,\rho_q,f_q)\).
\end{theorem}
\begin{proof}
Unpack the residue data to obtain \(\Cont(I,\rho,f)\). Continuation
associativity against \(\Cont(f,q,f_q)\) supplies \(\rho_q\) with
\(\Cont(\rho,q,\rho_q)\) and \(\Cont(I,\rho_q,f_q)\). The residue carrier is
closed under the unary suffix \(q\), and the product carrier for \(f_q\) is
rebuilt from the integral carrier, the shifted residue carrier, and the final
continuation.
\end{proof}
\leanchecked{BEDC.Derived.ResidueUp.ResiduePoleData\_residue\_suffix\_closure}

\begin{theorem}[Residue pole data suffix closure with empty function endpoints]
\label{thm:residue-pole-data-suffix-closure-empty-function-endpoints}
If $\mathsf{ResiduePoleData}(f,z_0,r,p,g,I,\rho)$ holds, $q$ is unary,
$\Cont(f,q,f_q)$, and $f_q\hsame\emp$, then there is a residue history
$\rho_q$ such that $\mathsf{ResiduePoleData}(f_q,z_0,r,p,g,I,\rho_q)$,
$\Cont(\rho,q,\rho_q)$, and
$$
I\hsame\emp\land \rho_q\hsame\emp.
$$
\end{theorem}
\begin{proof}
Apply suffix closure to obtain the transported residue witness $\rho_q$ and
the closed residue data over $f_q$. The empty-function endpoint theorem then
applies to that closed data and $f_q\hsame\emp$, identifying both the integral
and the transported residue with $\emp$.
\end{proof}
\leanchecked{BEDC.Derived.ResidueUp.ResiduePoleData\_suffix\_closure\_empty\_function\_endpoints}

\begin{theorem}[Residue pole data empty suffix reflects endpoints]
\label{thm:residue-pole-data-empty-suffix-reflects-endpoints}
If $\mathsf{ResiduePoleData}(f,z_0,r,p,g,I,\rho)$ holds, $\Cont(f,q,f_q)$,
and $f_q\hsame\emp$, then the integral endpoint, the residue endpoint, and the
suffix are all empty:
$$
I\hsame\emp\land \rho\hsame\emp\land q\hsame\emp.
$$
\end{theorem}
\begin{proof}
Transport $\Cont(f,q,f_q)$ along $f_q\hsame\emp$ and apply empty-result
inversion for continuations. This gives $f\hsame\emp$ and $q\hsame\emp$.
The empty-function endpoint theorem for residue pole data then reflects
$f\hsame\emp$ back to the integral and residue endpoints.
\end{proof}
\leanchecked{BEDC.Derived.ResidueUp.ResiduePoleData\_empty\_suffix\_reflects\_endpoints}

\begin{theorem}[Residue pole data prefix empty reflects endpoints]
\label{thm:residue-pole-data-prefix-empty-reflects-endpoints}
If $\mathsf{ResiduePoleData}(f,z_0,r,p,g,I,\rho)$ holds, $\Cont(q,f,q_f)$,
and $q_f\hsame\emp$, then the prefix, integral endpoint, and residue endpoint
are all empty:
$$
q\hsame\emp\land I\hsame\emp\land \rho\hsame\emp.
$$
\end{theorem}
\begin{proof}
Transport $\Cont(q,f,q_f)$ along $q_f\hsame\emp$ and apply empty-result
inversion for continuations. This gives $q\hsame\emp$ and $f\hsame\emp$.
The empty-function endpoint theorem for residue pole data then reflects
$f\hsame\emp$ to the integral and residue endpoints.
\end{proof}
\leanchecked{BEDC.Derived.ResidueUp.ResiduePoleData\_prefix\_empty\_reflects\_endpoints}

\begin{theorem}[Residue pole data suffix empty function endpoints]
\label{thm:residue-pole-data-suffix-empty-function-endpoints}
If \(\mathsf{ResiduePoleData}(f,z_0,r,p,g,I,\rho)\) holds, \(q\) is unary,
\(\Cont(f,q,f_q)\), and \(f_q\hsame\emp\), then \(I\hsame\emp\) and there is
a shifted residue \(\rho_q\) such that
$$
\Cont(\rho,q,\rho_q)\land \rho_q\hsame\emp.
$$
\end{theorem}
\begin{proof}
Apply residue suffix closure to obtain the shifted pole data and the residue
suffix witness. The empty-function endpoint theorem applied to the shifted
data and \(f_q\hsame\emp\) identifies \(I\) and \(\rho_q\) with the empty
history, and the residue suffix witness supplies the displayed continuation.
\end{proof}
\leanchecked{BEDC.Derived.ResidueUp.ResiduePoleData\_suffix\_empty\_function\_endpoints}

\begin{theorem}[Residue pole data integral prefix closure]
\label{thm:residue-pole-data-integral-prefix-closure}
If \(\mathsf{ResiduePoleData}(f,z_0,r,p,g,I,\rho)\) holds, \(q\) is unary,
and \(\Cont(q,f,q_f)\), then there is an integral history \(I_q\) such that
\(\mathsf{ResiduePoleData}(q_f,z_0,r,p,g,I_q,\rho)\), with
\(\Cont(q,I,I_q)\) and \(\Cont(I_q,\rho,q_f)\).
\end{theorem}
\begin{proof}
Take \(I_q=\mathsf{append}(q,I)\). The complex integral carrier is closed under
unary prefixing. The two continuation witnesses \(\Cont(I,\rho,f)\) and
\(\Cont(q,f,q_f)\), together with append associativity, identify
\(\Cont(I_q,\rho,q_f)\). The product carrier for \(q_f\) is then rebuilt from
the prefixed integral carrier, the original residue carrier, and that
continuation.
\end{proof}
\leanchecked{BEDC.Derived.ResidueUp.ResiduePoleData\_integral\_prefix\_closure}

\begin{theorem}[Residue pole data integral prefix empty function endpoints]
\label{thm:residue-pole-data-integral-prefix-empty-function-endpoints}
If \(\mathsf{ResiduePoleData}(f,z_0,r,p,g,I,\rho)\) holds, \(q\) is unary,
\(\Cont(q,f,q_f)\), and \(q_f\hsame\emp\), then there is an integral history
\(I_q\) such that \(\mathsf{ResiduePoleData}(q_f,z_0,r,p,g,I_q,\rho)\),
\(\Cont(q,I,I_q)\), and
$$
I_q\hsame\emp\land \rho\hsame\emp.
$$
\end{theorem}
\begin{proof}
Apply integral prefix closure to obtain the prefixed integral witness \(I_q\)
and the residue pole data over \(q_f\). The empty-function endpoint theorem
applied to that prefixed pole data and \(q_f\hsame\emp\) identifies both
\(I_q\) and \(\rho\) with the empty history, while the prefix continuation
\(\Cont(q,I,I_q)\) is preserved from the closure witness.
\end{proof}
\leanchecked{BEDC.Derived.ResidueUp.ResiduePoleData\_integral\_prefix\_empty\_function\_endpoints}

\section{The certificate}
\label{sec:residue-certificate}

\begin{theorem}[Residue pole data integral suffix closure]
\label{thm:residue-pole-data-integral-suffix-closure}
If $\mathsf{ResiduePoleData}(f,z_0,r,p,g,I,\rho)$ holds, $q$ is unary,
and $\Cont(f,q,f_q)$, then there is an integral history $I_q$ such that
$\mathsf{ResiduePoleData}(f_q,z_0,r,p,g,I_q,\rho)$, with
$\Cont(I,q,I_q)$ and $\Cont(I_q,\rho,f_q)$.
\end{theorem}
\begin{proof}
Unpack the residue pole data of \autoref{def:residue-pole-data}. This gives the
open-disk witness, complex carrier witnesses for $I$ and $\rho$, the product
carrier for $f$, and the continuation $\Cont(I,\rho,f)$. Set
$I_q=\mathsf{append}(I,q)$. The right unary context closure for complex carriers
from \autoref{thm:complex-history-carrier-append-unary-closed} gives a complex
carrier for $I_q$, and $\Cont(I,q,I_q)$ is the defining continuation.

It remains to identify the final continuation. By
\autoref{thm:complex-history-carrier-unary}, the residue carrier makes $\rho$
unary. Since $q$ is unary, unary append commutativity from
\autoref{lem:unary-append-comm-left-induction} swaps $q$ and $\rho$. Let $J$ be
the history supplied by the defining continuation of $I_q$ by $\rho$, so
$\Cont(I_q,\rho,J)$. Combining this continuation with the swap, append
associativity, and the two continuations $\Cont(I,\rho,f)$ and $\Cont(f,q,f_q)$
gives
$$
J
\hsame
\mathsf{append}(\mathsf{append}(I,q),\rho)
\hsame
\mathsf{append}(\mathsf{append}(I,\rho),q)
\hsame f_q.
$$
The continuation result transport theorem
\autoref{thm:cont-result-hsame-transport} applied to $\Cont(I_q,\rho,J)$ and
$J\hsame f_q$ gives $\Cont(I_q,\rho,f_q)$. The same open-disk witness and the
original residue carrier are retained; the product carrier for $f_q$ is rebuilt
from the complex carriers for $I_q$ and $\rho$ together with
$\Cont(I_q,\rho,f_q)$. Hence
$\mathsf{ResiduePoleData}(f_q,z_0,r,p,g,I_q,\rho)$ holds, with the two displayed
continuations.
\end{proof}
\leanchecked{BEDC.Derived.ResidueUp.ResiduePoleData\_integral\_suffix\_closure}

\begin{theorem}[Residue pole data result determinism]
\label{thm:residue-pole-data-result-determinism}
If two residue pole data witnesses share the displayed pole histories $z_0,r,p,g,I,\rho$ and have function histories $f$ and $f'$, so that
$$
\mathsf{ResiduePoleData}(f,z_0,r,p,g,I,\rho)
\quad\text{and}\quad
\mathsf{ResiduePoleData}(f',z_0,r,p,g,I,\rho),
$$
then $f\hsame f'$.
\end{theorem}
\leanchecked{BEDC.Derived.ResidueUp.ResiduePoleData\_result\_determinism\_with\_continuations}
\begin{proof}
Unpack both residue pole data witnesses using \autoref{def:residue-pole-data}. The first witness exposes a continuation ledger $C:\Cont(I,\rho,f)$, and the second exposes a continuation ledger $C':\Cont(I,\rho,f')$. Apply \autoref{thm:cont-transport-result-classified} to $C$ and $C'$ with the reflexive endpoint comparisons $I\hsame I$, $\rho\hsame\rho$, and $f\hsame f$. The transported first ledger has the same source, right endpoint, and result as $C$, while the classification clause compares that result with the result of $C'$. Hence $f\hsame f'$.
\end{proof}
\leanchecked{BEDC.Derived.ResidueUp.ResiduePoleData\_result\_determinism}
