\begin{theorem}[Measure null event addition neutral]
\label{thm:measure-null-event-addition-neutral}
For a $\MeasureUp$ carrier as in \autoref{sec:measure-carrier}, let $E$ be a
measurable event, let $\varnothing$ be the empty measurable event, and let
$0_R$ be the additive unit in the $\RealUp$ additive $\AbGroupUp$ row. Then
the finite measure endpoint satisfies both neutral readbacks
$$
\mu(E)+\mu(\varnothing)\sim_R\mu(E)
$$
and
$$
\mu(\varnothing)+\mu(E)\sim_R\mu(E).
$$
\end{theorem}
\leanchecked{BEDC.Derived.MeasureUp.MeasureCountableZeroTail\_canonical}
\begin{proof}
\autoref{thm:measure-empty-set-zero-from-sigma-additivity} gives
$\mu(\varnothing)\sim_R0_R$ from the all-empty countable disjoint family. The
$\RealUp$ classifier transport row carries this comparison through addition
with the fixed endpoint $\mu(E)$.

For the right neutral readback, additive congruence gives
$\mu(E)+\mu(\varnothing)\sim_R\mu(E)+0_R$. The right-unit row of the additive
$\AbGroupUp$ structure then gives $\mu(E)+0_R\sim_R\mu(E)$, and transitivity
of $\sim_R$ composes the two comparisons.

For the left neutral readback, the same transport gives
$\mu(\varnothing)+\mu(E)\sim_R0_R+\mu(E)$. The left-unit row of the additive
$\AbGroupUp$ structure gives $0_R+\mu(E)\sim_R\mu(E)$, so transitivity gives
the displayed comparison.
\end{proof}

\begin{theorem}[Measure finite disjoint-union symmetric readback]
\label{thm:measure-finite-disjoint-union-symmetric-readback}
For a $\MeasureUp$ carrier as in \autoref{sec:measure-carrier}, let $A$ and
$B$ be disjoint measurable events, and let $A\cup B$ be the measurable binary
union row. If the ordered finite readback gives
$$
\mu(A\cup B)\sim_R\mu(A)+\mu(B),
$$
then the symmetric readback also holds:
$$
\mu(A\cup B)\sim_R\mu(B)+\mu(A).
$$
\end{theorem}
\begin{proof}
Apply \autoref{thm:measure-finite-disjoint-union-additivity} to the displayed
disjoint measurable pair. This supplies the ordered comparison
$\mu(A\cup B)\sim_R\mu(A)+\mu(B)$ from empty-zero, head-tail readback, and
finite-support countable additivity.

The additive commutativity row of the $\RealUp$ additive $\AbGroupUp$ surface
gives $\mu(A)+\mu(B)\sim_R\mu(B)+\mu(A)$. Transitivity of the $\RealUp$
classifier composes the ordered finite readback with this commutativity row
and gives the symmetric readback.
\end{proof}

\begin{theorem}[Measure finite three-support countable readback]
\label{thm:measure-finite-three-support-countable-readback}
For a $\MeasureUp$ carrier as in \autoref{sec:measure-carrier}, let $\mathcal{A}$ be its designated $\sigma$-algebra, let $\mu$ be its $\RealUp$-valued measure map, write $0_R$ for the additive zero in the target $\RealUp$ row, and write $x\sim_R y$ for the $\RealUp$ classifier of \autoref{def:real-classifier-specification}. Let $E_\bullet$ be a pairwise-disjoint measurable sequence such that $E_n=\varnothing$ for every $n\geq3$. Suppose the $\MeasureUp$ certificate supplies $\sigma$-additivity for pairwise-disjoint measurable sequences, classifier-stable head-tail readback and pointwise classifier stability for the resulting countable $\RealUp$ sums, and the target $\RealUp$ addition is carried by its additive $\AbGroupUp$ row. Then
$$
\Sigma_\mu(E_\bullet)\sim_R(\mu(E_0)+\mu(E_1))+\mu(E_2).
$$
\end{theorem}
\begin{proof}
Let
$$
S:=\Sigma_\mu(E_\bullet),\qquad
T:=\Sigma_\mu(E_{\bullet+1}),\qquad
U:=\Sigma_\mu(E_{\bullet+2}),\qquad
V:=\Sigma_\mu(E_{\bullet+3}).
$$

Apply head-tail readback to the original sequence, to the once-shifted sequence, and to the twice-shifted sequence. These comparisons classify $S$ with $\mu(E_0)+T$, classify $T$ with $\mu(E_1)+U$, and classify $U$ with $\mu(E_2)+V$.

The three-shifted tail is the all-empty sequence by the support hypothesis. Pointwise countable-sum stability transports $V$ to the canonical empty-sequence endpoint, and \autoref{thm:measure-countable-zero-tail-canonical} classifies that endpoint with $0_R$.

Use additive congruence from \autoref{def:abgroup-carrier} to transport the comparison $V\sim_R0_R$ through the third head-tail equation. The right-unit law reads $\mu(E_2)+0_R$ as $\mu(E_2)$, so $U\sim_R\mu(E_2)$.

Transport this comparison through the second and first head-tail equations. Associativity from the additive $\AbGroupUp$ row rewrites $\mu(E_0)+(\mu(E_1)+\mu(E_2))$ as $(\mu(E_0)+\mu(E_1))+\mu(E_2)$, giving the displayed classification.
\end{proof}

\begin{theorem}[Measure triple disjoint-union additivity]
\label{thm:measure-triple-disjoint-union-additivity}
For a $\MeasureUp$ carrier as in \autoref{sec:measure-carrier}, let $\mathcal{A}$ be its designated $\sigma$-algebra, let $\mu$ be its $\RealUp$-valued measure map, and write $x\sim_R y$ for the $\RealUp$ classifier of \autoref{def:real-classifier-specification}. Let $A,B,C\in\mathcal{A}$ be pairwise disjoint. Suppose the measurable binary-union rows display $A\cup B\in\mathcal{A}$ and $(A\cup B)\cup C\in\mathcal{A}$, and suppose the $\MeasureUp$ certificate supplies $\sigma$-additivity for pairwise-disjoint measurable sequences, classifier-stable head-tail readback and pointwise classifier stability for the resulting countable $\RealUp$ sums, and the target $\RealUp$ addition is carried by its additive $\AbGroupUp$ row. Then
$$
\mu((A\cup B)\cup C)\sim_R(\mu(A)+\mu(B))+\mu(C).
$$
\end{theorem}
\begin{proof}
Define $E_0:=A$, $E_1:=B$, $E_2:=C$, and $E_n:=\varnothing$ for every $n\geq3$. The displayed measurable rows make each term measurable, and pairwise disjointness of $A,B,C$ is the only nonempty disjointness check.

The countable union of this sequence is the displayed event $(A\cup B)\cup C$. Applying $\sigma$-additivity gives a classifier comparison between $\mu((A\cup B)\cup C)$ and $\Sigma_\mu(E_\bullet)$.

The finite three-support readback theorem \autoref{thm:measure-finite-three-support-countable-readback} classifies $\Sigma_\mu(E_\bullet)$ with $(\mu(E_0)+\mu(E_1))+\mu(E_2)$. Substituting the three displayed heads and composing the $\RealUp$ classifier comparisons gives the claimed triple disjoint-union additivity.
\end{proof}
