\section{Root premeasure event-value bridge}
\label{sec:measure-root-premeasure-event-value-bridge}

\begin{definition}[Measure root premeasure event-value bridge]
\label{def:measure-root-premeasure-event-value-bridge}
The $\MeasureUp$ root premeasure event-value bridge is the package whose
fields are the countable premeasure carrier and classifier rows of
\autoref{def:measure-root-countable-premeasure-front}, the countable
additivity row of
\autoref{thm:measure-root-countable-premeasure-additivity-obligation}, the
public obligation basis rows of
\autoref{def:measure-root-public-obligation-basis}, and the $\NameCert$
obligation package of \autoref{thm:measure-namecert-obligation-package}.
Its visible event-value surface consists only of covered measurable-event
$\mathsf{BHist}$ rows, their $\RealUp$ nonnegative value endpoints, the
classifier transports for those rows, and the ledger exactness rows that tie
hidden measurable witnesses to displayed events.
\end{definition}

\begin{theorem}[Measure root premeasure finite-additivity bridge]
\label{thm:measure-root-premeasure-finite-additivity-bridge}
The package of \autoref{def:measure-root-premeasure-event-value-bridge}
exports finite disjoint-union additivity by composing
\autoref{thm:measure-root-countable-premeasure-additivity-obligation},
\autoref{thm:measure-root-public-obligation-basis-additivity}, and
\autoref{thm:measure-namecert-obligation-package}.
\end{theorem}
\begin{proof}
Project the countable additivity field from the bridge package. By
\autoref{thm:measure-root-countable-premeasure-additivity-obligation}, the
countable-premeasure front reads finite disjoint unions as finite-support
instances of the same countable disjoint-additivity row.

Next project the public additive face. Apply
\autoref{thm:measure-root-public-obligation-basis-additivity}; it exposes
empty-zero, finite readback, relative-difference additivity, countable
disjoint-additivity, and countable-sum transport over the same displayed
event-value ledger.

Finally read the $\NameCert$ obligation package through
\autoref{thm:measure-namecert-obligation-package}. Its finite disjoint-union
theorem is attached to the carrier, classifier, nonnegative endpoint, and
ledger exactness rows already present in the bridge, so the finite additivity
claim is the exported projection of the package.
\end{proof}

\begin{theorem}[Measure root premeasure downstream threshold bridge]
\label{thm:measure-root-premeasure-downstream-threshold-bridge}
Downstream $\IntegralUp$, $\ProbSpaceUp$, and $\FunctionalAnalysisUp$
consumers may read only the event carrier and classifier, countable
additivity, finite additivity, nonnegative value, and ledger exactness rows
exposed by the bridge of
\autoref{def:measure-root-premeasure-event-value-bridge}.
\end{theorem}
\begin{proof}
Start with the bridge package and discard every field not named by its
event-value surface. The remaining carrier and classifier rows are exactly
the covered measurable-event $\mathsf{BHist}$ rows, value endpoint rows, and
transport rows inherited from
\autoref{def:measure-root-countable-premeasure-front}.

For additive information, use
\autoref{thm:measure-root-countable-premeasure-additivity-obligation} and
\autoref{thm:measure-root-premeasure-finite-additivity-bridge}. These give
the countable disjoint-additivity row and its finite disjoint-union
projection without adding a consumer-specific carrier.

For public visibility and exactness, apply
\autoref{thm:measure-root-public-obligation-basis-exactness} and
\autoref{thm:measure-namecert-obligation-package}. The first pins the rows
to the displayed public basis, while the second supplies the nonnegative
value and ledger exactness obligations consumed by the three downstream
chapters.

The package definition has no further outgoing field for analytic completion,
host set equality, quotient event equality, or a standard bridge. Hence
$\IntegralUp$, $\ProbSpaceUp$, and $\FunctionalAnalysisUp$ can read exactly
the listed rows and no larger root surface from this bridge.
\end{proof}
