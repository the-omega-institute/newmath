\begin{definition}[Tagged option payload descent image classifier]
\label{def:tagged-option-payload-descent-image-classifier}
Let $S,T:\mathsf{BHist}\to\mathsf{Prop}$ be BEDC history source carriers, let
$\sim_S$ and $\sim_T$ be source classifiers, let
$\delta:\mathsf{DescentCertificate}(\mathsf{BHist},\mathsf{BHist},\sim_S,\sim_T)$,
and assume payload preservation
$$
\rho_{\delta}:\forall a.\;S(a)\to T(\delta.\mathsf{map}(a)).
$$
For target endpoints $k,k':\mathsf{BHist}$, define the tagged option payload
descent image classifier by
$$
\mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
(S,T,\rho_{\delta};k,k')
:\!\!\Longleftrightarrow
\exists h,h':\mathsf{BHist}.\;
\mathsf{TaggedOptionHistoryClassifier}_{S,\sim_S}(h,h')
\land
\mathsf{TaggedOptionMapRel}_{\delta}(S,T,\rho_{\delta};h,k)
\land
\mathsf{TaggedOptionMapRel}_{\delta}(S,T,\rho_{\delta};h',k').
$$
\end{definition}
\leandef{BEDC.Derived.OptionUp.TaggedOptionPayloadDescentImageClassifier}

\begin{theorem}[Tagged option payload descent image classifier branch exactness]
\label{thm:tagged-option-payload-descent-image-classifier-branch-exactness}
This theorem fits the per-object template as follows: the traditional target is
classified image readback for option payload descent; the BEDC source is
$\mathsf{BHist}$ with $\hsame$ endpoint presentations; the carrier is
$\mathsf{TaggedOptionPayloadDescentImageCarrier}$; the classifier is
$\mathsf{TaggedOptionPayloadDescentImageClassifier}$; the stability obligation is
branch analysis of the source tagged classifier and the two
$\mathsf{TaggedOptionMapRel}$ witnesses; the ledger is the pair of source
endpoints and their transported visible payloads; the core theorem is exact
absent/present pair readback; exactness is the displayed equivalence; the
standard bridge is $\mathsf{none}$ and $\mathsf{some}$ under payload descent; and
the status is $\mathsf{ProofTarget}$.

Let $S,T:\mathsf{BHist}\to\mathsf{Prop}$ be BEDC history source carriers, let
$\sim_S$ and $\sim_T$ be source classifiers, let
$\delta:\mathsf{DescentCertificate}(\mathsf{BHist},\mathsf{BHist},\sim_S,\sim_T)$,
and assume payload preservation
$$
\rho_{\delta}:\forall a.\;S(a)\to T(\delta.\mathsf{map}(a)).
$$
Let $\mathsf{cert}:\NameCert(S,\sim_S)$ and assume
$\mathsf{TaggedOptionSourceHsameCompatible}(S,\sim_S)$. For target endpoints
$k,k':\mathsf{BHist}$, the image classifier is exactly the absent target pair or
the transported present-payload pair:
$$
\mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
(S,T,\rho_{\delta};k,k')
\Longleftrightarrow
\bigl(\hsame(k,\emp)\land\hsame(k',\emp)\bigr)
\lor
\bigl(\exists a,b:\mathsf{BHist}.\;
S(a)\land S(b)\land a\sim_S b\land
T(\delta.\mathsf{map}(a))\land T(\delta.\mathsf{map}(b))\land
\hsame(k,\Eone(\delta.\mathsf{map}(a)))\land
\hsame(k',\Eone(\delta.\mathsf{map}(b)))\bigr).
$$
\end{theorem}
\begin{proof}
For the forward direction, unfold
Definition~\ref{def:tagged-option-payload-descent-image-classifier}. This
exposes source endpoints $h,h'$, a source tagged classifier witness, and the two
map-relation witnesses from $h$ to $k$ and from $h'$ to $k'$.

In the absent source-classifier branch, both source endpoints are presented as
$\emp$. Splitting the two map-relation witnesses rules out any present
map-relation branch by empty-versus-$\Eone$ no-confusion, so both target
endpoints are presented as $\emp$. This gives the absent side of the displayed
disjunction.

In the present source-classifier branch, the classifier exposes source payloads
$p,q$ with $S(p)$, $S(q)$, $p\sim_S q$, and presentations of $h,h'$ as
$\Eone(p),\Eone(q)$. The two map relations cannot be absent, again by
empty-versus-$\Eone$ no-confusion, and their present branches expose payloads
$a,b$ with $S(a)$, $S(b)$, the target carrier witnesses
$T(\delta.\mathsf{map}(a))$, $T(\delta.\mathsf{map}(b))$, and the displayed
target presentations.

It remains to identify the source payload relation for $a$ and $b$. Invert the
two $\Eone$ presentations around $h$ and $h'$ to obtain $\hsame(p,a)$ and
$\hsame(q,b)$. Source hsame compatibility gives $p\sim_S a$ and $q\sim_S b$;
symmetry and transitivity from $\mathsf{cert}$ compose these with
$p\sim_S q$ to obtain $a\sim_S b$. The present side of the displayed
disjunction follows.

For the reverse direction, split the displayed disjunction. In the absent case,
choose source endpoints $\emp$ and $\emp$, use the absent-absent source
classifier constructor, and use the two absent map-relation constructors with
the displayed target endpoint presentations.

In the present case, unpack $a,b$, the two source carrier witnesses, the source
classifier witness $a\sim_S b$, the two transported target carrier witnesses,
and the two target endpoint presentations. Choose source endpoints $\Eone(a)$
and $\Eone(b)$; the present-present source classifier constructor and the two
present map-relation constructors rebuild exactly the existential data required
by the image classifier definition.
\end{proof}
\leanchecked{BEDC.Derived.OptionUp.TaggedOptionPayloadDescentImageClassifier\_branch\_exactness}

\begin{theorem}[Tagged option payload descent image classifier right visible-branch inversion]
\label{thm:tagged-option-payload-descent-image-classifier-right-visible-branch-inversion}
Let $S,T:\mathsf{BHist}\to\mathsf{Prop}$ be BEDC history source carriers, let
$\sim_S$ and $\sim_T$ be source classifiers, let
$\delta:\mathsf{DescentCertificate}(\mathsf{BHist},\mathsf{BHist},\sim_S,\sim_T)$,
and assume payload preservation
$$
\rho_{\delta}:\forall a.\;S(a)\to T(\delta.\mathsf{map}(a)).
$$
Let $\mathsf{cert}:\NameCert(S,\sim_S)$ and assume
$\mathsf{TaggedOptionSourceHsameCompatible}(S,\sim_S)$. For target endpoints
$k,k':\mathsf{BHist}$, the right visible branch determines the matching left
image branch. The absent clause is
$$
\mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
(S,T,\rho_{\delta};k,k')
\to
\hsame(k',\emp)
\to
\hsame(k,\emp).
$$
For every payload history $a:\mathsf{BHist}$, the present clause is
$$
\mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
(S,T,\rho_{\delta};k,k')
\to
T(a)
\to
\hsame(k',\Eone(a))
\to
\exists b,c:\mathsf{BHist}.\;
S(b)\land S(c)\land b\sim_S c\land
T(\delta.\mathsf{map}(b))\land T(\delta.\mathsf{map}(c))\land
\hsame(a,\delta.\mathsf{map}(c))\land
\hsame(k,\Eone(\delta.\mathsf{map}(b))).
$$
\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:tagged-option-payload-descent-image-classifier-branch-exactness}
to the image-classifier witness. This rewrites the witness into the disjunction
whose absent side presents both target endpoints as $\emp$ and whose present side
exposes source payloads $b,c$ together with the transported endpoint
presentations.

For the absent clause, split that disjunction. The absent side gives
$\hsame(k,\emp)$ directly. The present side also presents $k'$ as
$\Eone(\delta.\mathsf{map}(c))$ for some source payload $c$; composing this with
the displayed witness $\hsame(k',\emp)$ gives empty-versus-$\Eone$
no-confusion, so the mixed branch is impossible.

For the present clause, split the same disjunction. The absent side presents
$k'$ as $\emp$, contradicting the displayed witness $\hsame(k',\Eone(a))$ by
empty-versus-$\Eone$ no-confusion.

In the present side, keep the exposed source payloads $b,c$. The branch-exactness
data already supplies $S(b)$, $S(c)$, $b\sim_S c$,
$T(\delta.\mathsf{map}(b))$, $T(\delta.\mathsf{map}(c))$, and
$\hsame(k,\Eone(\delta.\mathsf{map}(b)))$. Compose the right-endpoint
presentations $\hsame(k',\Eone(a))$ and
$\hsame(k',\Eone(\delta.\mathsf{map}(c)))$, and invert the $\Eone$ constructor
to obtain
$\hsame(a,\delta.\mathsf{map}(c))$. These fields are exactly the displayed
existential witness. The hypothesis $T(a)$ records the visible right payload as
a target carrier payload and is preserved as part of the public premise.
\end{proof}

$\hsame(u,\delta.\mathsf{map}(a))$ and
$\hsame(v,\delta.\mathsf{map}(b))$, so the same $a,b$ provide the required

\begin{theorem}[Tagged option payload descent image classifier stability fields]

$\sim_S$ and $\sim_T$ be source classifiers, let
$\delta:\mathsf{DescentCertificate}(\mathsf{BHist},\mathsf{BHist},\sim_S,\sim_T)$,

\rho_{\delta}:\forall a.\;S(a)\to T(\delta.\mathsf{map}(a)).

\begin{enumerate}
  \item For every target endpoint $k$, if
  $$
  \mathsf{TaggedOptionPayloadDescentImageCarrier}_{\delta}
  (S,T,\rho_{\delta};k),
  $$
  then
  $$
  \mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
  (S,T,\rho_{\delta};k,k).
  $$
  \item For target endpoints $k,k'$, if
  $$
  \mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
  (S,T,\rho_{\delta};k,k'),
  $$
  then
  $$
  \mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
  (S,T,\rho_{\delta};k',k).
  $$
  \item For target endpoints $k,k'$, if
  $$
  \mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
  (S,T,\rho_{\delta};k,k'),

  $$
  \mathsf{TaggedOptionPayloadDescentImageCarrier}_{\delta}
  (S,T,\rho_{\delta};k)
  \land
  \mathsf{TaggedOptionPayloadDescentImageCarrier}_{\delta}

$\mathsf{TaggedOptionHistoryCarrier}_S(h)$, and a map witness from $h$ to $k$.

Theorem~\ref{thm:tagged-option-history-stability-fields} turns the source

$\mathsf{TaggedOptionHistoryClassifier}_{S,\sim_S}(h,h)$.

the same source endpoint on both sides. The source classifier witness just built

$\mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
(S,T,\rho_{\delta};k,k)$.

For symmetry, unpack the image-classifier witness. This exposes source endpoints
$h,h'$, a source classifier witness
$\mathsf{TaggedOptionHistoryClassifier}_{S,\sim_S}(h,h')$, and map witnesses
from $h$ to $k$ and from $h'$ to $k'$. The symmetry clause of
Theorem~\ref{thm:tagged-option-history-stability-fields} gives the reversed
source classifier witness
$\mathsf{TaggedOptionHistoryClassifier}_{S,\sim_S}(h',h)$.

Repack the image-classifier existential with the two source endpoints exchanged
and the two map witnesses exchanged. This gives the displayed relation from
$k'$ to $k$.

For endpoint exposure, apply
Theorem~\ref{thm:tagged-option-payload-descent-image-classifier-certificate} to
the image-classifier witness with the same $\mathsf{cert}$ and source hsame
compatibility hypothesis. Projecting the first two components of its conclusion
gives exactly the two displayed image-carrier witnesses.
\end{proof}

\begin{theorem}[Tagged option payload descent image classifier reflective transitivity]
\label{thm:tagged-option-payload-descent-image-classifier-reflective-transitivity}
Let $S,T:\mathsf{BHist}\to\mathsf{Prop}$ be BEDC history source carriers, let
$\sim_S$ and $\sim_T$ be source classifiers, let
$\delta:\mathsf{DescentCertificate}(\mathsf{BHist},\mathsf{BHist},\sim_S,\sim_T)$,
and assume payload preservation
$$
\rho_{\delta}:\forall a.\;S(a)\to T(\delta.\mathsf{map}(a)).
$$
Let $\mathsf{cert}:\NameCert(S,\sim_S)$, assume
$\mathsf{TaggedOptionSourceHsameCompatible}(S,\sim_S)$, and assume
$$
\mathsf{TaggedOptionPayloadDescentReflectsSource}_{\delta}(S,\sim_S).
$$
For target endpoints $k,\ell,m:\mathsf{BHist}$, if
$$
\mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
(S,T,\rho_{\delta};k,\ell)
$$
and
$$
\mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
(S,T,\rho_{\delta};\ell,m),
$$
then
$$
\mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
(S,T,\rho_{\delta};k,m).
$$
\end{theorem}
\begin{proof}
Apply
Theorem~\ref{thm:tagged-option-payload-descent-image-classifier-branch-exactness}
to the two image-classifier witnesses. This rewrites each witness into the
absent target pair or the present transported-payload pair, with the middle
endpoint $\ell$ appearing in both branch analyses.

\begin{enumerate}
  \item In the absent/absent case, the first branch gives $\hsame(k,\emp)$ and
  $\hsame(\ell,\emp)$, while the second gives $\hsame(\ell,\emp)$ and
  $\hsame(m,\emp)$. The absent side of the reverse branch-exactness equivalence
  rebuilds
  $\mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
  (S,T,\rho_{\delta};k,m)$.

  \item In the absent/present case, the first branch presents $\ell$ as $\emp$.
  The second branch presents the same endpoint as
  $\Eone(\delta.\mathsf{map}(c))$ for some source payload $c$. Empty-versus-
  $\Eone$ no-confusion makes this case impossible.

  \item In the present/absent case, the first branch presents $\ell$ as
  $\Eone(\delta.\mathsf{map}(b))$ for some source payload $b$, while the second
  branch presents $\ell$ as $\emp$. Empty-versus-$\Eone$ no-confusion again
  rules out the mixed case.

  \item In the present/present case, the first branch exposes source payloads
  $a,b$ with $S(a)$, $S(b)$, $a\sim_S b$, and endpoint presentations
  $\hsame(k,\Eone(\delta.\mathsf{map}(a)))$ and
  $\hsame(\ell,\Eone(\delta.\mathsf{map}(b)))$. The second branch exposes
  source payloads $c,d$ with $S(c)$, $S(d)$, $c\sim_S d$, and endpoint
  presentations $\hsame(\ell,\Eone(\delta.\mathsf{map}(c)))$ and
  $\hsame(m,\Eone(\delta.\mathsf{map}(d)))$.

  Compose the two presentations of $\ell$ and invert the $\Eone$ constructor to
  obtain
  $$
  \hsame(\delta.\mathsf{map}(b),\delta.\mathsf{map}(c)).
  $$
  The reflection hypothesis turns this visible target equality, together with
  $S(b)$ and $S(c)$, into $b\sim_S c$.

  Transitivity from $\mathsf{cert}$ composes $a\sim_S b$, $b\sim_S c$, and
  $c\sim_S d$ into $a\sim_S d$. The present side of the reverse
  branch-exactness equivalence, using the already exposed target carrier
  witnesses and the endpoint presentations for $k$ and $m$, rebuilds
  $\mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
  (S,T,\rho_{\delta};k,m)$.
\end{enumerate}
\end{proof}
