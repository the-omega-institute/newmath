\section{Composite image classifier normalized one-sided public completion}
\label{sec:option-composite-image-classifier-normalized-one-sided-public-completion}

\begin{theorem}[Tagged option composite image classifier normalized one-sided public completion]
\label{thm:tagged-option-composite-image-classifier-normalized-one-sided-public-completion}
This theorem fits the per-object template as follows: the traditional target is
endpoint-normalized one-sided public completion for functorial
$\mathsf{OptionUp}$ image classification; the BEDC source is $\mathsf{BHist}$
with two composable $\mathsf{DescentCertificate}$ maps; the carrier is
$\mathsf{TaggedOptionPayloadDescentImageCarrier}$; the classifier is
$\mathsf{TaggedOptionPayloadDescentImageClassifier}$; the stability obligations
are public readback completion, endpoint visibility, empty-versus-present
no-confusion, and the $\hsame$ symmetry and transitivity spine; the ledger
records total left and right visible-endpoint consumers with synchronized
opposite endpoints; the core theorem is one-sided public completion from any
visible normalized endpoint; exactness says the completed opposite endpoint
carries the same target classifier and the full descent factorization package;
the standard bridge is $\mathsf{some}$ normalized public readback; and the
status is $\mathsf{ProofTarget}$.

Let $S,T,U:\mathsf{BHist}\to\mathsf{Prop}$ be BEDC history source,
intermediate, and target carriers, let $\sim_S$, $\sim_T$, and $\sim_U$ be their
classifiers, and let
$$
\delta:\mathsf{DescentCertificate}(\mathsf{BHist},\mathsf{BHist},\sim_S,\sim_T)
$$
and
$$
\epsilon:\mathsf{DescentCertificate}(\mathsf{BHist},\mathsf{BHist},\sim_T,\sim_U).
$$
Assume payload preservation fields
$$
\rho_{\delta}:\forall a.\;S(a)\to T(\delta.\mathsf{map}(a))
$$
and
$$
\rho_{\epsilon}:\forall b.\;T(b)\to U(\epsilon.\mathsf{map}(b)).
$$
Assume endpoint stability for the second descent:
$$
\eta_{\epsilon}:\forall x\,y.\;
T(x)\to T(y)\to \hsame(x,y)\to
\hsame(\epsilon.\mathsf{map}(x),\epsilon.\mathsf{map}(y)).
$$
Write $\epsilon\circ\delta$ for the descent certificate whose map field is
$a\mapsto\epsilon.\mathsf{map}(\delta.\mathsf{map}(a))$ and whose respects field
sends $r:a\sim_S b$ to
$\epsilon.\mathsf{respects}(\delta.\mathsf{respects}(r))$. Define
$$
\rho_{\epsilon\circ\delta}(a,Sa)=
\rho_{\epsilon}(\delta.\mathsf{map}(a),\rho_{\delta}(a,Sa)).
$$
Let $\mathsf{cert}_S:\NameCert(S,\sim_S)$,
$\mathsf{cert}_T:\NameCert(T,\sim_T)$, and
$\mathsf{cert}_U:\NameCert(U,\sim_U)$. Assume
$\mathsf{TaggedOptionSourceHsameCompatible}(S,\sim_S)$,
$\mathsf{TaggedOptionSourceHsameCompatible}(T,\sim_T)$, and
$\mathsf{TaggedOptionSourceHsameCompatible}(U,\sim_U)$. For final endpoints
$m,m':\mathsf{BHist}$ and normalized endpoints $u,u':\mathsf{BHist}$, if
$$
\mathsf{TaggedOptionPayloadDescentImageClassifier}_{\epsilon\circ\delta}
(S,U,\rho_{\epsilon\circ\delta};m,m')
$$
and
$$
\hsame(m,u)\land \hsame(m',u'),
$$
then the following one-sided public completion clauses hold:
\begin{enumerate}
\item for every left visible payload $x_L:\mathsf{BHist}$, if $U(x_L)$ and
$\hsame(u,\Eone(x_L))$, then there exist $y_R,k,k',p,q,a,b:\mathsf{BHist}$ such
that
\begin{enumerate}
\item $U(y_R)$, $\hsame(u',\Eone(y_R))$, and $x_L\sim_U y_R$;
\item $\mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
(S,T,\rho_{\delta};k,k')$;
\item $\mathsf{TaggedOptionMapRel}_{\epsilon}(T,U,\rho_{\epsilon};k,u)$ and
$\mathsf{TaggedOptionMapRel}_{\epsilon}(T,U,\rho_{\epsilon};k',u')$;
\item $T(p)$, $T(q)$, $U(\epsilon.\mathsf{map}(p))$, and
$U(\epsilon.\mathsf{map}(q))$;
\item $\hsame(k,\Eone(p))$, $\hsame(k',\Eone(q))$,
$\hsame(x_L,\epsilon.\mathsf{map}(p))$, and
$\hsame(y_R,\epsilon.\mathsf{map}(q))$;
\item $S(a)$, $S(b)$, $a\sim_S b$, $T(\delta.\mathsf{map}(a))$, and
$T(\delta.\mathsf{map}(b))$;
\item $\hsame(p,\delta.\mathsf{map}(a))$ and
$\hsame(q,\delta.\mathsf{map}(b))$;
\item $p\sim_T q$;
\item $\neg\hsame(u,\emp)$ and $\neg\hsame(u',\emp)$;
\item for every $y':\mathsf{BHist}$, if $U(y')$ and
$\hsame(u',\Eone(y'))$, then $\hsame(y_R,y')$ and $x_L\sim_U y'$.
\end{enumerate}
\item for every right visible payload $y_R:\mathsf{BHist}$, if $U(y_R)$ and
$\hsame(u',\Eone(y_R))$, then there exist $x_L,k,k',p,q,a,b:\mathsf{BHist}$ such
that
\begin{enumerate}
\item $U(x_L)$, $\hsame(u,\Eone(x_L))$, and $x_L\sim_U y_R$;
\item $\mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
(S,T,\rho_{\delta};k,k')$;
\item $\mathsf{TaggedOptionMapRel}_{\epsilon}(T,U,\rho_{\epsilon};k,u)$ and
$\mathsf{TaggedOptionMapRel}_{\epsilon}(T,U,\rho_{\epsilon};k',u')$;
\item $T(p)$, $T(q)$, $U(\epsilon.\mathsf{map}(p))$, and
$U(\epsilon.\mathsf{map}(q))$;
\item $\hsame(k,\Eone(p))$, $\hsame(k',\Eone(q))$,
$\hsame(x_L,\epsilon.\mathsf{map}(p))$, and
$\hsame(y_R,\epsilon.\mathsf{map}(q))$;
\item $S(a)$, $S(b)$, $a\sim_S b$, $T(\delta.\mathsf{map}(a))$, and
$T(\delta.\mathsf{map}(b))$;
\item $\hsame(p,\delta.\mathsf{map}(a))$ and
$\hsame(q,\delta.\mathsf{map}(b))$;
\item $p\sim_T q$;
\item $\neg\hsame(u,\emp)$ and $\neg\hsame(u',\emp)$;
\item for every $x':\mathsf{BHist}$, if $U(x')$ and
$\hsame(u,\Eone(x'))$, then $\hsame(x_L,x')$ and $x'\sim_U y_R$.
\end{enumerate}
\end{enumerate}
\end{theorem}
\begin{proof}
Apply
Theorem~\ref{thm:tagged-option-composite-image-classifier-normalized-public-readback-completion-package}
to the composite image-classifier witness, the endpoint normalization witnesses,
and the displayed descent, preservation, stability, certificate, and
compatibility data. This gives either the normalized absent public branch or a
present visible branch carrying the two one-sided completion functions.

For the left clause, fix $x_L$ with $U(x_L)$ and $\hsame(u,\Eone(x_L))$. In the
absent branch, the supplied $\hsame(u,\emp)$ composes with the visible endpoint
witness, after symmetry if needed, to identify $\emp$ with $\Eone(x_L)$. This
contradicts BEDC history no-confusion via $not\_hsame\_emp\_e1$, so the absent
branch cannot occur.

In the present branch, instantiate the left one-sided completion function at
$x_L$. It supplies $y_R,k,k',p,q,a,b$ with the target endpoint, the synchronized
target classifier, the two-stage descent factorization, the public map-relation
links, the source and intermediate payload witnesses, the non-absence facts, and
the universal competing-right-payload clause.

For the right clause, fix $y_R$ with $U(y_R)$ and $\hsame(u',\Eone(y_R))$. In
the absent branch, the supplied $\hsame(u',\emp)$ composes with the visible
endpoint witness, after symmetry if needed, to identify $\emp$ with
$\Eone(y_R)$. This contradicts BEDC history no-confusion via
$not\_hsame\_emp\_e1$, so the absent branch cannot occur.

In the present branch, instantiate the right one-sided completion function at
$y_R$. It supplies $x_L,k,k',p,q,a,b$ with the symmetric target endpoint,
classifier, descent factorization, public map-relation links, non-absence
facts, and universal competing-left-payload clause. These are exactly the two
displayed one-sided public completion clauses.
\end{proof}
