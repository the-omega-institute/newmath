\begin{theorem}[Option classifier $\hsame$ symmetry]
\label{thm:option-classifier-hsame-symm}
For typed option carrier values $x,y:\mathsf{OptionCarrier}(\mathsf{BHist})$, if $\mathsf{OptionClassifierSpec}(\hsame,x,y)$ holds, then $\mathsf{OptionClassifierSpec}(\hsame,y,x)$ holds.
\end{theorem}
\begin{proof}
The proof case-analyzes typed $\mathsf{none}$ and $\mathsf{some}$ carrier values: the none--none branch returns the existing classifier witness, the mixed branches contradict the typed classifier clauses, and the some--some branch reverses the BEDC history witness by $\hsame$ symmetry.
\end{proof}
\leanchecked{BEDC.Derived.OptionUp.OptionClassifierSpec\_hsame\_symm}

\begin{theorem}[Option classifier $\hsame$ $e1$ map equivalence]
\label{thm:option-classifier-spec-hsame-e1-map-iff}
For typed option carrier values $x,y:\mathsf{OptionCarrier}(\mathsf{BHist})$,
mapping both visible payloads through the BEDC history constructor $\Eone$
preserves and reflects the specialized option classifier:
$$
\mathsf{OptionClassifierSpec}(\hsame,\mathsf{map}(\Eone,x),\mathsf{map}(\Eone,y))
\Longleftrightarrow
\mathsf{OptionClassifierSpec}(\hsame,x,y).
$$
\end{theorem}
\begin{proof}
Case-analyze the two option endpoints. The none--none branch is unchanged, and
the mixed branches contradict the typed classifier clauses. In the
some--some branch, the forward direction uses injectivity of $\Eone$ up to
$\hsame$, and the reverse direction applies constructor congruence for
$\Eone$.
\end{proof}
\leanchecked{BEDC.Derived.OptionUp.OptionClassifierSpec\_hsame\_e1\_map\_iff}

\begin{theorem}[Option classifier $\hsame$ transport]
\label{thm:option-classifier-spec-hsame-transport}
For typed option carrier values
$x,x',y,y':\mathsf{OptionCarrier}(\mathsf{BHist})$, if
$\mathsf{OptionClassifierSpec}(\hsame,x,y)$,
$\mathsf{OptionClassifierSpec}(\hsame,x,x')$, and
$\mathsf{OptionClassifierSpec}(\hsame,y,y')$ hold, then
$\mathsf{OptionClassifierSpec}(\hsame,x',y')$ holds.
\end{theorem}
\begin{proof}
Case-analyze the four option endpoints. Mixed $\mathsf{none}$--$\mathsf{some}$
branches contradict one of the displayed classifier witnesses. The
$\mathsf{none}$--$\mathsf{none}$ branch reuses the existing typed classifier
witness. In the $\mathsf{some}$--$\mathsf{some}$ branch, compose
$\hsame(x,x')$ symmetrically with $\hsame(x,y)$ and then with $\hsame(y,y')$.
\end{proof}
\leanchecked{BEDC.Derived.OptionUp.OptionClassifierSpec\_hsame\_transport}

\begin{theorem}[Option classifier $\hsame$ right determinacy]
\label{thm:option-classifier-spec-hsame-right-deterministic}

\end{proof}
\leanchecked{BEDC.Derived.OptionUp.OptionClassifierSpec\_hsame\_right\_deterministic}

\begin{theorem}[Option classifier $\hsame$ some-right confluence]
\label{thm:option-classifier-spec-hsame-some-right-confluence}
For a typed option carrier value $x:\mathsf{OptionCarrier}(\mathsf{BHist})$ and
histories $h,k:\mathsf{BHist}$, if
$\mathsf{OptionClassifierSpec}(\hsame,x,\mathsf{some}(h))$ and
$\mathsf{OptionClassifierSpec}(\hsame,x,\mathsf{some}(k))$ hold, then
$\hsame(h,k)$ holds.
\end{theorem}
\begin{proof}
Case-analyze the shared endpoint $x$. If $x$ is $\mathsf{none}$, the first
classifier witness is impossible because it relates $\mathsf{none}$ to a
present endpoint. If $x=\mathsf{some}(a)$, the two witnesses are
$\hsame(a,h)$ and $\hsame(a,k)$; reverse the first and compose it with the
second.
\end{proof}
\leanchecked{BEDC.Derived.OptionUp.OptionClassifierSpec\_hsame\_some\_right\_confluence}

\begin{theorem}[Option classifier $\hsame$ some-right inversion]
\label{thm:option-classifier-spec-hsame-some-right-inversion}
For a typed option carrier value $x:\mathsf{OptionCarrier}(\mathsf{BHist})$ and
a history $b:\mathsf{BHist}$, if
$\mathsf{OptionClassifierSpec}(\hsame,x,\mathsf{some}(b))$ holds, then there is
a history $a:\mathsf{BHist}$ such that $x=\mathsf{some}(a)$ and
$\hsame(a,b)$ holds.
\end{theorem}
\begin{proof}
Case-analyze $x$. The none branch contradicts the typed none--some classifier
clause. In the some branch, the exposed payload is the required witness and the
classifier clause is exactly the displayed $\hsame$ witness.
\end{proof}
\leanchecked{BEDC.Derived.OptionUp.OptionClassifierSpec\_hsame\_some\_right\_inversion}

\begin{theorem}[Option classifier $\hsame$ right payload determinacy]

\end{theorem}
\begin{proof}
Case-analyze the shared left option value $x$. The none branch contradicts the

\end{proof}
\leanchecked{BEDC.Derived.OptionUp.OptionClassifierSpec\_hsame\_right\_payload\_deterministic}

\begin{theorem}[Option classifier symmetry from source symmetry]
\label{thm:option-classifier-symm-from-source-symmetry}
Let $A$ be an entry type, let $R:A\to A\to\mathsf{Prop}$ be a source relation, and let $x,y:\mathsf{OptionCarrier}(A)$. If
$\forall a\,b.\; R(a,b)\to R(b,a)$ and
$\mathsf{OptionClassifierSpec}(R,x,y)$ hold, then
$\mathsf{OptionClassifierSpec}(R,y,x)$ holds.
\end{theorem}
\begin{proof}
Case-analyze $x$ and $y$. The none--none case is immediate from the none classifier clause. The mixed none--some and some--none cases are impossible by the typed option classifier clauses. In the some--some case, apply the displayed source-symmetry hypothesis to the payload witness.
\end{proof}
\leanchecked{BEDC.Derived.OptionUp.OptionClassifierSpec\_symm\_from\_source\_symmetry}

\begin{definition}[Option stability certificate]
\label{def:option-stability-certificate}
The stability certificate separates the typed optional classifier fields from
the nullable history-envelope fields.

For the typed $\mathsf{OptionClassifierSpec}$ classifier, the fields are:
\begin{enumerate}
  \item reflexivity for $\mathsf{none}$ and for $\mathsf{some}(a)$ from source reflexivity;
  \item symmetry from a source-symmetry hypothesis, as in Theorem~\ref{thm:option-classifier-symm-from-source-symmetry};
  \item transitivity from a source-transitivity hypothesis;
  \item no-confusion between $\mathsf{none}$ and $\mathsf{some}(a)$ in either order;
  \item present inversion, exposing the source classifier witness from a $\mathsf{some}$--$\mathsf{some}$ classification.
\end{enumerate}
The checked $\hsame$ specialization is the Lean-marked instance of this
symmetry field.

For the nullable $\mathsf{OptionHistoryCarrier}$ envelope and its
$\mathsf{OptionHistoryClassifier}$, the certificate fields are:
\begin{enumerate}
  \item carrier transport across $\hsame$;
  \item endpoint transport for the classifier across $\hsame$ on both endpoints;
  \item classifier symmetry;
  \item classifier transitivity;
  \item ledger transport across raw and visible endpoint $\hsame$ links.
\end{enumerate}
\end{definition}
\leanchecked{BEDC.Derived.OptionUp.option\_stability\_certificate\_fields}

\begin{theorem}[Option classifier transitivity]
\label{thm:option-classifier-transitivity}
Let $A$ be an entry type, let $R:A\to A\to\mathsf{Prop}$ be a source
classifier relation, and let $x,y,z:\mathsf{OptionCarrier}(A)$. If $R$ is
transitive and both $\mathsf{OptionClassifierSpec}(R,x,y)$ and
$\mathsf{OptionClassifierSpec}(R,y,z)$ hold, then
$\mathsf{OptionClassifierSpec}(R,x,z)$ holds.
\end{theorem}
\leanchecked{BEDC.Derived.OptionUp.OptionClassifierSpec\_trans}
\begin{proof}
Proceed by constructor case analysis on $x$, $y$, and $z$.  The none--none--none
case is immediate, mixed none--some cases contradict the classifier clauses, and
the some--some--some case is exactly the supplied transitivity of $R$.
\end{proof}

\begin{definition}[Option ledger policy]
\label{def:option-ledger-policy}
The ledger keeps a raw optional-history carrier witness and an $\hsame$ link from the raw history to the visible history.
\end{definition}
\leandef{BEDC.Derived.OptionUp.OptionHistoryLedgerPolicy}

\begin{theorem}[Option history ledger policy empty raw iff]
\label{thm:option-history-ledger-policy-empty-raw-iff}
Let $S:\mathsf{BHist}\to\mathsf{Prop}$ and let $v:\mathsf{BHist}$. Then
$$
\mathsf{OptionHistoryLedgerPolicy}(S,\mathsf{Empty},v)
\Longleftrightarrow
\hsame(v,\mathsf{Empty}).
$$
\end{theorem}
\begin{proof}
Unpack the ledger at raw history $\mathsf{Empty}$. A ledger witness stores an
$\hsame(\mathsf{Empty},v)$ link, whose symmetry gives
$\hsame(v,\mathsf{Empty})$. Conversely, an $\hsame(v,\mathsf{Empty})$ witness
provides the empty carrier branch for the raw endpoint and, by symmetry, the
ledger link from $\mathsf{Empty}$ to $v$.
\end{proof}
\leanchecked{BEDC.Derived.OptionUp.OptionHistoryLedgerPolicy\_empty\_raw\_iff}

\begin{theorem}[Option visible ledger carrier]
\label{thm:option-ledger-policy-visible-carrier}
Let $S:\mathsf{BHist}\to\mathsf{Prop}$ and let $\rho,v:\mathsf{BHist}$. If
$source\_transport : \forall h\,k.\;\hsame(h,k)\to S(h)\to S(k)$ holds, then
$\mathsf{OptionHistoryLedgerPolicy}(S,\rho,v)$ implies
$\mathsf{OptionHistoryCarrier}_S(v)$.
\end{theorem}
\begin{proof}
Unpack the raw carrier and raw-to-visible $\hsame$ link. The absent case composes the symmetric link with the raw empty-history witness; the present case transports the raw source witness along the stored link.
\end{proof}
\leanchecked{BEDC.Derived.OptionUp.OptionHistoryLedgerPolicy\_visible\_carrier}

\begin{theorem}[Option ledger policy source visible iff]
\label{thm:option-history-ledger-policy-source-visible-iff}
Let $S:\mathsf{BHist}\to\mathsf{Prop}$ and let
$\rho,v:\mathsf{BHist}$. Assume
$$
  \forall h\,k.\;\hsame(h,k)\to S(h)\to S(k).
$$
If $\mathsf{OptionHistoryLedgerPolicy}(S,\rho,v)$ holds, then
$$
S(\rho)\Longleftrightarrow S(v).
$$
\end{theorem}
\begin{proof}
Unpack the ledger to obtain the raw-visible link $\hsame(\rho,v)$. The forward
direction transports a raw source witness along this link. The reverse
direction transports a visible source witness along the symmetric link.
\end{proof}
\leanchecked{BEDC.Derived.OptionUp.OptionHistoryLedgerPolicy\_source\_visible\_iff}

\begin{theorem}[Option ledger visible source exactness]
\label{thm:option-history-ledger-policy-visible-source-exactness}
Let $S:\mathsf{BHist}\to\mathsf{Prop}$ exclude the empty history, and let
$\rho,v:\mathsf{BHist}$. If $S(v)$ holds, then the option ledger over
$\rho$ and $v$ is exactly raw source membership together with the stored
raw-visible equality:
$$
\mathsf{OptionHistoryLedgerPolicy}(S,\rho,v)
\Longleftrightarrow
S(\rho)\land\hsame(\rho,v).
$$
\end{theorem}
\begin{proof}
Unpack the ledger into the raw option carrier and the raw-visible $\hsame$
link. If the raw carrier is absent, the link transports the empty witness to
$v$, contradicting empty-source exclusion and $S(v)$. Hence the raw carrier is
the source branch, and the link is the displayed equality. Conversely, the raw
source witness rebuilds the source branch of the carrier and pairs it with the
given raw-visible link.
\end{proof}
\leanchecked{BEDC.Derived.OptionUp.OptionHistoryLedgerPolicy\_visible\_source\_exactness}

\begin{theorem}[Option ledger exclusive readback]
\label{thm:option-ledger-policy-exclusive-readback}
Let $S:\mathsf{BHist}\to\mathsf{Prop}$ exclude the empty history, and let
$\rho,v:\mathsf{BHist}$. If
$\mathsf{OptionHistoryLedgerPolicy}(S,\rho,v)$ holds, then exactly one branch is
read back from the ledger:
$$
(\hsame(\rho,\mathsf{Empty})\land \hsame(v,\mathsf{Empty})
\land \neg S(\rho)\land \neg S(v))
\lor
(S(\rho)\land S(v)\land \neg\hsame(\rho,\mathsf{Empty})
\land \neg\hsame(v,\mathsf{Empty})\land \hsame(\rho,v)).
$$
\end{theorem}
\begin{proof}
Unpack the ledger into the raw optional-history carrier and the raw-to-visible
$\hsame$ link. The raw carrier readback gives either the absent branch or the
source branch. In the absent branch, the stored link transfers the empty witness
to the visible endpoint, and source exclusion rules out visible source
membership. In the source branch, the stored $\hsame$ link identifies raw and
visible endpoints, so the source witness transports by case analysis on that
link; empty witnesses on either endpoint contradict source exclusion.
\end{proof}
\leanchecked{BEDC.Derived.OptionUp.OptionHistoryLedgerPolicy\_exclusive\_readback}

$$
  \forall h\,k.\;\hsame(h,k)\to S(h)\to S(k).

If $\mathsf{OptionHistoryLedgerPolicy}(S,\rho,v)$ holds, then
$\mathsf{OptionHistoryClassifier}_S(\rho,w) \Longleftrightarrow

\begin{corollary}[OptionUp certificate scope]
\label{cor:option-certificate-scope}
Citations to Theorem~\ref{thm:concrete-option-namecert} assert only the
nullable empty-or-source history envelope. The constructor-tagged history
certificate is Theorem~\ref{thm:tagged-option-history-semantic-certificate},
under a source $\NameCert$ and
$\mathsf{TaggedOptionSourceHsameCompatible}(S,\sim_S)$, or
Corollary~\ref{cor:tagged-option-history-semantic-certificate-hsame} in the
$\hsame$ specialization. The typed $\mathsf{OptionClassifierSpec}$ is a
separate host-option classifier; it is not a field that licenses the tagged
history semantic certificate, and neither certificate implies the other carrier
presentation without the bridge data of
Definition~\ref{def:tagged-option-to-nullable-bridge-hypotheses}.
\end{corollary}
