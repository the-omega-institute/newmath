\begin{definition}[Tagged option-to-nullable bridge hypotheses]
\label{def:tagged-option-to-nullable-bridge-hypotheses}
For a BEDC history source carrier $S:\mathsf{BHist}\to\mathsf{Prop}$ and source
classifier $\sim_S$, define
$\mathsf{TaggedOptionToNullableBridge}(S,\sim_S)$ to be the pair of conditions:
\begin{enumerate}
  \item present endpoint closure: for all histories $a,h$, if $S(a)$ and $\hsame(h,\Eone(a))$, then $S(h)$;
  \item classifier hsame soundness: for all histories $a,b$, if $S(a)$, $S(b)$, and $a\sim_S b$, then $\hsame(a,b)$.
\end{enumerate}
\end{definition}
\leandef{BEDC.Derived.OptionUp.TaggedOptionToNullableBridge}

\begin{theorem}[Tagged option-to-nullable classifier bridge]
\label{thm:tagged-option-to-nullable-classifier-bridge}
Let $S:\mathsf{BHist}\to\mathsf{Prop}$ be a source carrier and let $\sim_S$ be a
source classifier. If $\mathsf{TaggedOptionToNullableBridge}(S,\sim_S)$ holds,
then for every history $h$,
$\mathsf{TaggedOptionHistoryCarrier}_S(h)$ implies
$\mathsf{OptionHistoryCarrier}_S(h)$, and for every pair of histories $h,k$,
$\mathsf{TaggedOptionHistoryClassifier}_{S,\sim_S}(h,k)$ implies
$\mathsf{OptionHistoryClassifier}_S(h,k)$.
\end{theorem}
\begin{proof}
Unpack the bridge into present endpoint closure and classifier hsame soundness.
For the carrier implication, case-analyze
$\mathsf{TaggedOptionHistoryCarrier}_S(h)$. The absent branch is already the
empty side of $\mathsf{OptionHistoryCarrier}_S(h)$. In the present branch, a
payload $a$ with $S(a)$ and $\hsame(h,\Eone(a))$ descends to $S(h)$ by present
endpoint closure, giving the source side of the nullable envelope.

For the classifier implication, use
Theorem~\ref{thm:tagged-option-history-classifier-branch-partition}. In the
absent-absent branch, the two empty endpoint witnesses give the nullable
carriers and compose through $\emp$ to obtain $\hsame(h,k)$. In the
present-present branch, the carrier part descends for both endpoints by present
endpoint closure. Classifier hsame soundness converts $a\sim_S b$ into
$\hsame(a,b)$; applying $e1$ congruence gives
$\hsame(\Eone(a),\Eone(b))$, which composes with the endpoint witnesses
$\hsame(h,\Eone(a))$ and $\hsame(k,\Eone(b))$ to yield $\hsame(h,k)$. These are
exactly the carrier and endpoint-equality fields of
$\mathsf{OptionHistoryClassifier}_S(h,k)$.
\end{proof}

\begin{theorem}[Tagged option history classifier induces nullable classification]
\label{thm:tagged-option-history-classifier-option-history-classifier}
Let $S:\mathsf{BHist}\to\mathsf{Prop}$ be a source carrier. If two histories
are classified by the tagged option classifier over $\hsame$, then they are
classified by the nullable option classifier whose source side is the tagged
option history carrier:
$$
\mathsf{TaggedOptionHistoryClassifier}_{S,\hsame}(h,k)
\Longrightarrow
\mathsf{OptionHistoryClassifier}_{\mathsf{TaggedOptionHistoryCarrier}_S}(h,k).
$$
\end{theorem}
\begin{proof}
Analyze the tagged classifier. In the absent branch, both endpoints are
history-same to $\emp$, so the nullable carriers use their absent clauses and
the classifier equality composes through $\emp$. In the present branch, the
payload witnesses $a$ and $b$ rebuild the tagged carrier source clauses for
the nullable classifier. The endpoint equality is obtained by composing the
left endpoint link, congruence of $\Eone$ over $\hsame(a,b)$, and the reverse
of the right endpoint link.
\end{proof}
\leanchecked{BEDC.Derived.OptionUp.TaggedOptionHistoryClassifier\_option\_history\_classifier}

\begin{theorem}[Tagged option-to-nullable hsame bridge criterion]
\label{thm:tagged-option-to-nullable-hsame-bridge-criterion}
The ten-field fit is: traditional target nullable comparison for
constructor-tagged options; BEDC source $\mathsf{BHist}$ with $\hsame$; carrier
$\mathsf{TaggedOptionHistoryCarrier}$; classifier
$\mathsf{TaggedOptionHistoryClassifier}_{S,\hsame}$; stability obligation
present-endpoint absorption; ledger visible absent/present tag; core theorem the
bridge criterion; exactness one-way comparison to the nullable envelope; standard
bridge none/some readback; status $\mathsf{ProofTarget}$. Let
$S:\mathsf{BHist}\to\mathsf{Prop}$ be a source carrier. If
$$
  \forall a\,h.\;
    S(a)\to\hsame(h,\Eone(a))\to S(h),
$$
then $\mathsf{TaggedOptionToNullableBridge}(S,\hsame)$ holds. Consequently, for
every history $h$,
$$
  \mathsf{TaggedOptionHistoryCarrier}_S(h)\to
  \mathsf{OptionHistoryCarrier}_S(h),
$$
and for every pair of histories $h,k$,
$$
  \mathsf{TaggedOptionHistoryClassifier}_{S,\hsame}(h,k)\to
  \mathsf{OptionHistoryClassifier}_S(h,k).
$$
\end{theorem}
\begin{proof}
Pair the displayed present-endpoint absorption hypothesis with identity
classifier hsame soundness: from $S(a)$, $S(b)$, and $\hsame(a,b)$, the required
history-sameness conclusion is the same witness. This gives
$\mathsf{TaggedOptionToNullableBridge}(S,\hsame)$. The two descent conclusions
are exactly Theorem~\ref{thm:tagged-option-to-nullable-classifier-bridge}
specialized to the source classifier $\hsame$.
\end{proof}

\begin{theorem}[Tagged option public certificate compares with nullable envelope]
\label{thm:tagged-option-public-certificate-descends-to-nullable-envelope}
Let $S:\mathsf{BHist}\to\mathsf{Prop}$ be a source carrier, let $\sim_S$ be a
source classifier, let $\mathsf{cert}:\NameCert(S,\sim_S)$, and assume
$\mathsf{TaggedOptionToNullableBridge}(S,\sim_S)$. Then the constructor-tagged
public certificate
$$
\mathsf{SemanticNameCert}(\mathsf{TaggedOptionHistoryCarrier}_S,
\mathsf{TaggedOptionHistoryPattern}_S,
\mathsf{TaggedOptionHistoryLedger}_S,
\mathsf{TaggedOptionHistoryClassifier}_{S,\sim_S})
$$
holds, and the tagged carrier and classifier compare to the nullable envelope:
$$
\forall h.\;
\mathsf{TaggedOptionHistoryCarrier}_S(h)\to
\mathsf{OptionHistoryCarrier}_S(h),
$$
and
$$
\forall h\,k.\;
\mathsf{TaggedOptionHistoryClassifier}_{S,\sim_S}(h,k)\to
\mathsf{OptionHistoryClassifier}_S(h,k).
$$
This is one-way compatibility of the constructor-tagged public certificate with
the checked nullable envelope; it is not an equality of the two carrier
presentations.
\end{theorem}
\begin{proof}
First apply Theorem~\ref{thm:tagged-option-history-semantic-certificate} to the
source certificate $\mathsf{cert}$, obtaining the carrier-only tagged semantic
certificate. The core certificate is reused unchanged.

For the displayed public certificate, use
Theorem~\ref{thm:tagged-option-history-carrier-pattern-ledger-equivalence} to
convert each tagged carrier witness into the visible pattern witness and the
visible ledger witness, then repack the core certificate with these two
soundness maps.

The two displayed implications are exactly
Theorem~\ref{thm:tagged-option-to-nullable-classifier-bridge}, applied to the
bridge hypothesis $\mathsf{TaggedOptionToNullableBridge}(S,\sim_S)$. The bridge
records one-way carrier and classifier compatibility with the checked nullable
envelope; it does not construct a nullable semantic certificate from the tagged
one.
\end{proof}

\begin{theorem}[Tagged option hsame public certificate descends to nullable envelope]
\label{thm:tagged-option-hsame-public-certificate-descends-to-nullable-envelope}
The ten-field fit is: traditional target nullable comparison for a public
option object; BEDC source $\mathsf{BHist}$ with $\hsame$; carrier
$\mathsf{TaggedOptionHistoryCarrier}_S$; classifier
$\mathsf{TaggedOptionHistoryClassifier}_{S,\hsame}$; stability obligation
present-endpoint absorption; ledger visible absent/present tag; core theorem the
tagged $\hsame$ semantic certificate; exactness one-way comparison to the
nullable envelope; standard bridge none/some readback; status
$\mathsf{ProofTarget}$. Let $S:\mathsf{BHist}\to\mathsf{Prop}$ be a source
carrier, let $\mathsf{cert}:\NameCert(S,\hsame)$, and assume
$$
\forall a\,h.\;
S(a)\to\hsame(h,\Eone(a))\to S(h).
$$
Then the constructor-tagged public certificate
$$
\mathsf{SemanticNameCert}(\mathsf{TaggedOptionHistoryCarrier}_S,
\mathsf{TaggedOptionHistoryPattern}_S,
\mathsf{TaggedOptionHistoryLedger}_S,
\mathsf{TaggedOptionHistoryClassifier}_{S,\hsame})
$$
holds, and the tagged carrier and classifier descend to the nullable envelope:
$$
\forall h.\;
\mathsf{TaggedOptionHistoryCarrier}_S(h)\to
\mathsf{OptionHistoryCarrier}_S(h),
$$
and
$$
\forall h\,k.\;
\mathsf{TaggedOptionHistoryClassifier}_{S,\hsame}(h,k)\to
\mathsf{OptionHistoryClassifier}_S(h,k).
$$
\end{theorem}
\begin{proof}
The tagged certificate is the $\hsame$ certificate corollary
Corollary~\ref{cor:tagged-option-history-semantic-certificate-hsame}, applied to
$\mathsf{cert}$. The displayed absorption hypothesis gives
$\mathsf{TaggedOptionToNullableBridge}(S,\hsame)$ by
Theorem~\ref{thm:tagged-option-to-nullable-hsame-bridge-criterion}. Applying
Theorem~\ref{thm:tagged-option-to-nullable-classifier-bridge} to that bridge
datum gives the carrier and classifier implications into the nullable envelope.
\end{proof}
