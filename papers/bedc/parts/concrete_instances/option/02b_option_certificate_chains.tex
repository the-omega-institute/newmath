\begin{theorem}[Option history classifier endpoint transport]
\label{thm:option-history-classifier-hsame-transport}
If the optional history classifier relates \(h\) to \(k\), and
\(\hsame(h,h')\) and \(\hsame(k,k')\), then it relates \(h'\) to \(k'\).
\end{theorem}
\begin{proof}
Unpack the classifier into its endpoint carrier witnesses and the stored
\(\hsame(h,k)\) link. Transport both carriers along the endpoint sameness
links, then compose symmetry of \(\hsame(h,h')\), the stored link, and
\(\hsame(k,k')\).
\end{proof}
\leanchecked{BEDC.Derived.OptionUp.OptionHistoryClassifier\_hsame\_transport}

\begin{theorem}[Option history classifier symmetry]
\label{thm:option-history-classifier-symmetry}
For any source predicate on BEDC histories, the optional history classifier is
symmetric.
\end{theorem}
\begin{proof}
Unpack the classifier into endpoint carrier witnesses and the stored
\(\hsame(h,k)\) link. Swap the endpoint witnesses and reverse the stored link
by symmetry of \(\hsame\).
\end{proof}
\leanchecked{BEDC.Derived.OptionUp.OptionHistoryClassifier\_symm}

\begin{theorem}[Option history classifier transitivity]
\label{thm:option-history-classifier-transitivity}
For every $S:\mathsf{BHist}\to\mathsf{Prop}$ and $h,k,r:\mathsf{BHist}$, $\mathsf{OptionHistoryClassifier}_S(h,k)$ and $\mathsf{OptionHistoryClassifier}_S(k,r)$ imply
$\mathsf{OptionHistoryClassifier}_S(h,r)$.
\end{theorem}
\begin{proof}
Unpack the two classifier witnesses. The source and target carrier witnesses
are preserved, and the underlying \(\hsame\) links compose across the shared
middle history.
\end{proof}
\leanchecked{BEDC.Derived.OptionUp.OptionHistoryClassifier\_trans}

\begin{theorem}[Option history classifier self exactness]
\label{thm:option-history-classifier-self-exactness}
For every BEDC history source predicate $S:\mathsf{BHist}\to\mathsf{Prop}$
and every history $h$, the nullable option classifier at equal endpoints is
exactly the nullable option carrier:
$$
\mathsf{OptionHistoryClassifier}_S(h,h)
\Longleftrightarrow
\mathsf{OptionHistoryCarrier}_S(h).
$$
\end{theorem}
\begin{proof}
The forward direction projects the left endpoint carrier from the classifier
witness. For the reverse direction, use the displayed carrier witness for both
endpoints and close the classifier equality field by reflexivity of $\hsame$ at
$h$.
\end{proof}
\leanchecked{BEDC.Derived.OptionUp.OptionHistoryClassifier\_self\_exactness}

\begin{theorem}[Option history classifier empty left iff]
\label{thm:option-history-classifier-empty-left-iff}
For any source predicate on BEDC histories, the optional history classifier
relates the empty history on the left to \(k\) exactly when \(k\) is
history-same as the empty history.
\end{theorem}
\begin{proof}
In the forward direction, unpack the classifier and reverse its stored
\(\hsame(\emp,k)\) link. In the reverse direction, both endpoint carrier
witnesses are supplied by the absent-history branch, and the classifier link is
the symmetric form of the assumed \(\hsame(k,\emp)\).
\end{proof}
\leanchecked{BEDC.Derived.OptionUp.OptionHistoryClassifier\_empty\_left\_iff}

\begin{corollary}[Option history classifier empty right iff]

Theorem~\ref{thm:option-history-classifier-empty-left-iff} to obtain

\end{proof}

\begin{definition}[Option source excludes empty]
\label{def:option-source-excludes-empty}
For a BEDC history source predicate $S:\mathsf{BHist}\to\mathsf{Prop}$,

$$
\mathsf{OptionSourceExcludesEmpty}(S)
\quad:\Longleftrightarrow\quad
\forall h.\; S(h)\to\hsame(h,\emp)\to\bot.

\begin{theorem}[Nullable option carrier empty-or-source partition]
\label{thm:option-history-carrier-exclusive-readback}
For any BEDC history source predicate $S:\mathsf{BHist}\to\mathsf{Prop}$ and
history $h$, if $\mathsf{OptionSourceExcludesEmpty}(S)$ and
$\mathsf{OptionHistoryCarrier}_S(h)$ hold, then
$$
(\hsame(h,\emp)\land\neg S(h))\lor(S(h)\land\neg\hsame(h,\emp)).

\end{theorem}
\begin{proof}
Case-analyze the carrier witness. In the empty branch, keep
$\hsame(h,\emp)$ and use
Theorem~\ref{thm:option-history-carrier-absent-source-separation} to rule out
$S(h)$. In the source branch, keep $S(h)$ and use the same theorem to rule out
$\hsame(h,\emp)$.
\end{proof}

\begin{theorem}[Nullable option classifier exclusive readback]
\label{thm:option-history-classifier-exclusive-readback}
For any BEDC history source predicate $S:\mathsf{BHist}\to\mathsf{Prop}$ and
histories $h,k$, if $\mathsf{OptionSourceExcludesEmpty}(S)$ and
$\mathsf{OptionHistoryClassifier}_S(h,k)$ hold, then either both endpoints are
the empty option branch and neither endpoint lies in $S$,
$$
\hsame(h,\emp)\land\hsame(k,\emp)\land\neg S(h)\land\neg S(k),
$$
or both endpoints lie in the source branch, neither is empty, and the classifier
keeps the endpoint equality:
$$
S(h)\land S(k)\land\neg\hsame(h,\emp)\land\neg\hsame(k,\emp)\land\hsame(h,k).
$$
\end{theorem}
\begin{proof}
Unpack the classifier into the two endpoint carrier witnesses and the displayed
$\hsame(h,k)$ equality. Apply Theorem~\ref{thm:option-history-carrier-exclusive-readback}
to both endpoint carriers. The empty--empty and source--source cases give the
two displayed alternatives. The mixed cases contradict
$\mathsf{OptionSourceExcludesEmpty}(S)$ after transporting the empty readback
across $\hsame(h,k)$.
\end{proof}
\leanchecked{BEDC.Derived.OptionUp.OptionHistoryClassifier\_exclusive\_readback}

\begin{theorem}[Option history classifier source-left exactness]
\label{thm:option-history-classifier-source-left-exactness}
For any BEDC history source predicate $S:\mathsf{BHist}\to\mathsf{Prop}$,
if $\mathsf{OptionSourceExcludesEmpty}(S)$ and $S(h)$ hold, then the nullable
option classifier from $h$ to $k$ is exactly source membership of $k$ together
with endpoint sameness:
$$
\mathsf{OptionHistoryClassifier}_S(h,k)
\quad\Longleftrightarrow\quad
S(k)\land\hsame(h,k).
$$
\end{theorem}
\begin{proof}
Unpack the classifier into endpoint carrier witnesses and the stored
$\hsame(h,k)$ link. Since $S(h)$ excludes the empty branch, any empty carrier
witness for $k$ would transport across $\hsame(h,k)$ to contradict
$\mathsf{OptionSourceExcludesEmpty}(S)$. Hence the right carrier is the source
branch $S(k)$, and the stored classifier link supplies $\hsame(h,k)$. Conversely,
$S(h)$ and $S(k)$ give the two source carrier branches, and the displayed
$\hsame(h,k)$ link reassembles the classifier.
\end{proof}

\begin{theorem}[Option classifier \(\hsame\) symmetry]
\label{thm:option-classifier-hsame-symm}
For typed option carrier values $x,y:\mathsf{OptionCarrier}(\mathsf{BHist})$, if $\mathsf{OptionClassifierSpec}(\hsame,x,y)$ holds, then $\mathsf{OptionClassifierSpec}(\hsame,y,x)$ holds.
\end{theorem}
\begin{proof}
The proof case-analyzes typed $\mathsf{none}$ and $\mathsf{some}$ carrier values: the none--none branch returns the existing classifier witness, the mixed branches contradict the typed classifier clauses, and the some--some branch reverses the BEDC history witness by $\hsame$ symmetry.

\begin{theorem}[Option classifier symmetry from source symmetry]
\label{thm:option-classifier-symm-from-source-symmetry}
Let $A$ be an entry type, let $R:A\to A\to\mathsf{Prop}$ be a source relation, and let $x,y:\mathsf{OptionCarrier}(A)$. If

$\mathsf{OptionClassifierSpec}(R,y,x)$ holds.
\end{theorem}
\begin{proof}
Case-analyze $x$ and $y$. The none--none case is immediate from the none classifier clause. The mixed none--some and some--none cases are impossible by the typed option classifier clauses. In the some--some case, apply the displayed source-symmetry hypothesis to the payload witness.

\leanchecked{BEDC.Derived.OptionUp.OptionClassifierSpec\_symm\_from\_source\_symmetry}

\begin{definition}[Option stability certificate]
\label{def:option-stability-certificate}
The stability certificate separates the typed optional classifier fields from
the nullable history-envelope fields.

For the typed $\mathsf{OptionClassifierSpec}$ classifier, the fields are:
\begin{enumerate}
  \item reflexivity for $\mathsf{none}$ and for $\mathsf{some}(a)$ from source reflexivity;
  \item symmetry from a source-symmetry hypothesis, as in Theorem~\ref{thm:option-classifier-symm-from-source-symmetry};
  \item transitivity from a source-transitivity hypothesis;
  \item no-confusion between $\mathsf{none}$ and $\mathsf{some}(a)$ in either order;
  \item present inversion, exposing the source classifier witness from a $\mathsf{some}$--$\mathsf{some}$ classification.
\end{enumerate}
The checked $\hsame$ specialization is the Lean-marked instance of this
symmetry field.

For the nullable $\mathsf{OptionHistoryCarrier}$ envelope and its
$\mathsf{OptionHistoryClassifier}$, the certificate fields are:
\begin{enumerate}
  \item carrier transport across $\hsame$;
  \item endpoint transport for the classifier across $\hsame$ on both endpoints;
  \item classifier symmetry;
  \item classifier transitivity;
  \item ledger transport across raw and visible endpoint $\hsame$ links.
\end{enumerate}
\end{definition}
\leanchecked{BEDC.Derived.OptionUp.option\_stability\_certificate\_fields}

\begin{theorem}[Option classifier transitivity]
\label{thm:option-classifier-transitivity}
Let $A$ be an entry type, let $R:A\to A\to\mathsf{Prop}$ be a source
classifier relation, and let $x,y,z:\mathsf{OptionCarrier}(A)$. If $R$ is
transitive and both $\mathsf{OptionClassifierSpec}(R,x,y)$ and
$\mathsf{OptionClassifierSpec}(R,y,z)$ hold, then
$\mathsf{OptionClassifierSpec}(R,x,z)$ holds.
\end{theorem}
\leanchecked{BEDC.Derived.OptionUp.OptionClassifierSpec\_trans}
\begin{proof}
Proceed by constructor case analysis on $x$, $y$, and $z$.  The none--none--none
case is immediate, mixed none--some cases contradict the classifier clauses, and
the some--some--some case is exactly the supplied transitivity of $R$.
\end{proof}

\begin{theorem}[Option visible ledger carrier]
\label{thm:option-ledger-policy-visible-carrier}

$\mathsf{OptionHistoryCarrier}_S(v)$.
\end{theorem}
\begin{proof}

\begin{theorem}[Option ledger exclusive readback]
\label{thm:option-ledger-policy-exclusive-readback}
Let $S:\mathsf{BHist}\to\mathsf{Prop}$ exclude the empty history, and let
$\rho,v:\mathsf{BHist}$. If
$\mathsf{OptionHistoryLedgerPolicy}(S,\rho,v)$ holds, then exactly one branch is
read back from the ledger:
$$
\begin{aligned}
&\hsame(\rho,\mathsf{Empty})\land \hsame(v,\mathsf{Empty})
  \land \neg S(\rho)\land \neg S(v)\\
&\quad\lor\\
&S(\rho)\land S(v)\land \neg\hsame(\rho,\mathsf{Empty})
  \land \neg\hsame(v,\mathsf{Empty})\land \hsame(\rho,v).
\end{aligned}
$$
\end{theorem}
\begin{proof}
Unpack the ledger into the raw optional-history carrier and the raw-to-visible
$\hsame$ link. The raw carrier readback gives either the absent branch or the
source branch. In the absent branch, the stored link transfers the empty witness
to the visible endpoint, and source exclusion rules out visible source
membership. In the source branch, the stored $\hsame$ link identifies raw and
visible endpoints, so the source witness transports by case analysis on that
link; empty witnesses on either endpoint contradict source exclusion.
\end{proof}
\leanchecked{BEDC.Derived.OptionUp.OptionHistoryLedgerPolicy\_exclusive\_readback}

\begin{theorem}[Option ledger classifier endpoint equivalence]
\label{thm:option-ledger-classifier-endpoint-equivalence}

\begin{corollary}[OptionUp certificate scope]
\label{cor:option-certificate-scope}

nullable empty-or-source history envelope. The constructor-tagged history
certificate is Theorem~\ref{thm:tagged-option-history-semantic-certificate},

separate host-option classifier; it is not a field that licenses the tagged
history semantic certificate, and neither certificate implies the other carrier
presentation without the bridge data of
Definition~\ref{def:tagged-option-to-nullable-bridge-hypotheses}.
\end{corollary}
