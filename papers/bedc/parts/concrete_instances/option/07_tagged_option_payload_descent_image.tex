\begin{definition}[Tagged option payload descent image carrier]
\label{def:tagged-option-payload-descent-image-carrier}
Let $S,T:\mathsf{BHist}\to\mathsf{Prop}$ be BEDC history source carriers, let
$\sim_S$ and $\sim_T$ be source classifiers, let
$\delta:\mathsf{DescentCertificate}(\mathsf{BHist},\mathsf{BHist},\sim_S,\sim_T)$,
and assume payload preservation
$$
\rho_{\delta}:\forall a.\;S(a)\to T(\delta.\mathsf{map}(a)).
$$
For a target endpoint $k:\mathsf{BHist}$, define the tagged option payload
descent image carrier by
$$
\mathsf{TaggedOptionPayloadDescentImageCarrier}_{\delta}(S,T,\rho_{\delta};k)
:\!\!\Longleftrightarrow
\exists h:\mathsf{BHist}.\;
\mathsf{TaggedOptionHistoryCarrier}_S(h)\land
\mathsf{TaggedOptionMapRel}_{\delta}(S,T,\rho_{\delta};h,k).
$$
\end{definition}

\begin{theorem}[Tagged option payload descent image branch exactness]
\label{thm:tagged-option-payload-descent-image-branch-exactness}
Let $S,T:\mathsf{BHist}\to\mathsf{Prop}$ be BEDC history source carriers, let
$\sim_S$ and $\sim_T$ be source classifiers, let
$\delta:\mathsf{DescentCertificate}(\mathsf{BHist},\mathsf{BHist},\sim_S,\sim_T)$,
and assume payload preservation
$$
\rho_{\delta}:\forall a.\;S(a)\to T(\delta.\mathsf{map}(a)).
$$
For every target endpoint $k:\mathsf{BHist}$, the payload descent image carrier is
exactly the absent endpoint together with the transported present payload
endpoints:
$$
\mathsf{TaggedOptionPayloadDescentImageCarrier}_{\delta}(S,T,\rho_{\delta};k)
\Longleftrightarrow
\hsame(k,\emp)\lor
\exists a:\mathsf{BHist}.\;
S(a)\land T(\delta.\mathsf{map}(a))\land
\hsame(k,\Eone(\delta.\mathsf{map}(a))).
$$
This result is not a variant of a checked theorem: it combines the image-carrier
definition with the concrete map relation and supplies coverage and completeness
for the image predicate.
\end{theorem}
\begin{proof}
For the forward direction, split the stored image-carrier witness. This exposes a
source endpoint $h$, a source tagged-option carrier witness, and a
$\mathsf{TaggedOptionMapRel}_{\delta}(S,T,\rho_{\delta};h,k)$ witness.

Now split the map-relation witness. In the absent branch, the relation gives the
target presentation $\hsame(k,\emp)$, so the left side of the displayed
disjunction holds. In the present branch, the relation exposes a payload
$a:\mathsf{BHist}$, the source witness $S(a)$, the transported target witness
$T(\delta.\mathsf{map}(a))$, and the endpoint presentation
$\hsame(k,\Eone(\delta.\mathsf{map}(a)))$, giving the right side.

For the reverse direction, split the displayed disjunction. In the absent case,
choose the source endpoint $\emp$. The absent tagged carrier constructor and the
absent map-relation constructor use reflexivity of $\hsame$ at $\emp$ together
with the displayed target endpoint witness to build the image-carrier
existential.

In the present case, unpack the payload $a$, the source witness $S(a)$, the target
witness $T(\delta.\mathsf{map}(a))$, and the presentation
$\hsame(k,\Eone(\delta.\mathsf{map}(a)))$. Choose the source endpoint
$\Eone(a)$; the present tagged carrier constructor uses $S(a)$ and reflexivity of
$\hsame$ at $\Eone(a)$, while the present map-relation constructor uses the
displayed target endpoint witness and the payload witnesses. These data repack
exactly into the image-carrier existential.
\end{proof}

\begin{theorem}[Tagged option payload descent image certificate]
\label{thm:tagged-option-payload-descent-image-certificate}
Let $S,T:\mathsf{BHist}\to\mathsf{Prop}$ be BEDC history source carriers, let
$\sim_S$ and $\sim_T$ be source classifiers, let
$\delta:\mathsf{DescentCertificate}(\mathsf{BHist},\mathsf{BHist},\sim_S,\sim_T)$,
and assume payload preservation
$$
\rho_{\delta}:\forall a.\;S(a)\to T(\delta.\mathsf{map}(a)).
$$
Let $\mathsf{cert}:\NameCert(S,\sim_S)$ and assume
$\mathsf{TaggedOptionSourceHsameCompatible}(S,\sim_S)$. Then the image carrier
of Definition~\ref{def:tagged-option-payload-descent-image-carrier} satisfies
the following certificate clauses.
\begin{enumerate}
  \item For every source endpoint $h$, if
  $\mathsf{TaggedOptionHistoryCarrier}_S(h)$, then there exists
  $k:\mathsf{BHist}$ such that
  $$
  \mathsf{TaggedOptionMapRel}_{\delta}(S,T,\rho_{\delta};h,k)
  $$
  and
  $$
  \mathsf{TaggedOptionPayloadDescentImageCarrier}_{\delta}
  (S,T,\rho_{\delta};k).
  $$
  \item For every endpoint $k$, if
  $$
  \mathsf{TaggedOptionPayloadDescentImageCarrier}_{\delta}
  (S,T,\rho_{\delta};k),
  $$
  then
  $$
  \mathsf{TaggedOptionHistoryCarrier}_T(k)\land
  \mathsf{TaggedOptionHistoryPattern}_T(k)\land
  \mathsf{TaggedOptionHistoryLedger}_T(k).
  $$
  \item For histories $h,h',k,k'$, if
  $$
  \mathsf{TaggedOptionHistoryClassifier}_{S,\sim_S}(h,h'),
  $$
  $$
  \mathsf{TaggedOptionMapRel}_{\delta}(S,T,\rho_{\delta};h,k),
  $$
  and
  $$
  \mathsf{TaggedOptionMapRel}_{\delta}(S,T,\rho_{\delta};h',k'),
  $$
  then
  $$
  \mathsf{TaggedOptionHistoryClassifier}_{T,\sim_T}(k,k').
  $$
\end{enumerate}
\end{theorem}
\begin{proof}
For the endpoint existence clause, apply
Theorem~\ref{thm:tagged-option-payload-descent-total-endpoint-transport} to
the displayed source carrier witness. It gives an endpoint $k$ together with
the required $\mathsf{TaggedOptionMapRel}$ witness; the same source carrier
witness and map witness are exactly the existential data required by the image
carrier definition.

For the target endpoint-scope clause, split the stored image-carrier witness.
This exposes a source endpoint $h$, a source carrier witness, and a
$\mathsf{TaggedOptionMapRel}_{\delta}(S,T,\rho_{\delta};h,k)$ witness.

Split the map-relation witness. In the absent branch, the target presentation
is the absent endpoint $\emp$; in the present branch, it exposes a payload
$a$ with $S(a)$, the transported target payload witness
$T(\delta.\mathsf{map}(a))$, and the target presentation
$\hsame(k,\Eone(\delta.\mathsf{map}(a)))$. These two branches give
$\mathsf{TaggedOptionHistoryCarrier}_T(k)$.

Apply
Theorem~\ref{thm:tagged-option-history-carrier-pattern-ledger-equivalence} to
that target carrier witness. The equivalence supplies the visible tag pattern
and visible tag ledger fields, yielding the displayed conjunction.

For the classifier clause, use
Theorem~\ref{thm:tagged-option-payload-descent-preserves-classification} with
the source classifier witness, the two map-relation witnesses,
$\mathsf{cert}$, and the source hsame compatibility hypothesis. Its conclusion
is the required target tagged option classifier.
\end{proof}
