\begin{theorem}[Tagged option identity payload descent exactness]
\label{thm:tagged-option-payload-descent-identity-exactness}
This theorem fits the per-object template as follows: the traditional target is
constructor-tagged option identity map; the BEDC source is $\mathsf{BHist}$ with
$\hsame$; the carrier is $\mathsf{TaggedOptionHistoryCarrier}$; the classifier
is $\mathsf{TaggedOptionHistoryClassifier}$ specialized to $\hsame$; the
stability obligation is identity preservation of visible absent and present
endpoints; the ledger is the visible tag and unchanged payload; the core theorem
is identity exactness for $\mathsf{TaggedOptionMapRel}$; exactness is branch
analysis; the standard bridge is none/some identity readback; and the status is
$\mathsf{ProofTarget}$. Let $S:\mathsf{BHist}\to\mathsf{Prop}$. Write
$\delta_{\mathsf{id}}$ for the descent certificate
$$
\delta_{\mathsf{id}}:
\mathsf{DescentCertificate}(\mathsf{BHist},\mathsf{BHist},\hsame,\hsame)
$$
whose map field is $a\mapsto a$ and whose respects field sends $p$ to $p$. Let
$$
\rho_{\mathsf{id}}:\forall a.\;S(a)\to S(\delta_{\mathsf{id}}.\mathsf{map}(a))
$$
be the preservation field that reuses the source witness. Then for all histories
$h,k$,
$$
\mathsf{TaggedOptionMapRel}_{\delta_{\mathsf{id}}}
(S,S,\rho_{\mathsf{id}};h,k)
\Longleftrightarrow
\mathsf{TaggedOptionHistoryCarrier}_S(h)\land\hsame(h,k).
$$
\end{theorem}
\begin{proof}
For the forward direction, split the map relation by its visible branch.

In the absent branch, the witness gives $\hsame(h,\emp)$ and $\hsame(k,\emp)$.
The first field builds the absent carrier for $h$. Composing $\hsame(h,\emp)$
with symmetry of $\hsame(k,\emp)$ gives $\hsame(h,k)$.

In the present branch, the witness exposes a payload $a$ with $S(a)$,
$S(\delta_{\mathsf{id}}.\mathsf{map}(a))$, $\hsame(h,\Eone(a))$, and
$\hsame(k,\Eone(\delta_{\mathsf{id}}.\mathsf{map}(a)))$. Since
$\delta_{\mathsf{id}}.\mathsf{map}(a)$ is $a$, the first payload witness and the
source endpoint presentation build the present carrier for $h$, and composing
the two endpoint presentations gives $\hsame(h,k)$.

For the reverse direction, split the carrier of $h$ by its tagged branch and use
the assumed $\hsame(h,k)$.

In the absent branch, compose symmetry of $\hsame(h,k)$ with the absent
presentation of $h$ to obtain $\hsame(k,\emp)$. Together with the original absent
presentation, this is the absent branch of
$\mathsf{TaggedOptionMapRel}_{\delta_{\mathsf{id}}}$.

In the present branch, keep the exposed payload $a$ and the witness $S(a)$.
The field $\rho_{\mathsf{id}}$ supplies the target payload witness, and
composing symmetry of $\hsame(h,k)$ with $\hsame(h,\Eone(a))$ gives
$\hsame(k,\Eone(a))$. Because the identity descent leaves $a$ unchanged, these
fields build the present branch of
$\mathsf{TaggedOptionMapRel}_{\delta_{\mathsf{id}}}$.
\end{proof}

\label{thm:tagged-option-maprel-target-carrier}

$$
\rho_{\delta}:\forall a.\;S(a)\to T(\delta.\mathsf{map}(a)).

$$

$$
exposes the target endpoint as a tagged option carrier:
$$
\mathsf{TaggedOptionHistoryCarrier}_T(k).

\end{theorem}
\begin{proof}
Split the map-relation witness. In the absent branch, the target endpoint is
presented as $\emp$, so it satisfies the absent carrier branch. In the present
branch, the witness exposes a payload $a$ with
$T(\delta.\mathsf{map}(a))$ and
$\hsame(k,\Eone(\delta.\mathsf{map}(a)))$, which are exactly the payload and
endpoint fields of the present carrier branch for $T$.
\end{proof}

\begin{theorem}[Tagged option payload descent preserves classification]
\label{thm:tagged-option-payload-descent-preserves-classification}
Let $S,T:\mathsf{BHist}\to\mathsf{Prop}$ be BEDC history source carriers, let
$\sim_S$ and $\sim_T$ be source classifiers, let
$\delta:\mathsf{DescentCertificate}(\mathsf{BHist},\mathsf{BHist},\sim_S,\sim_T)$,
and assume
$$
\rho_{\delta}:\forall a.\;S(a)\to T(\delta.\mathsf{map}(a)).
$$
For histories $h,h',k,k'$, if
$$
\mathsf{TaggedOptionHistoryClassifier}_{S,\sim_S}(h,h'),
$$
$$
\mathsf{TaggedOptionMapRel}_{\delta}(S,T,\rho_{\delta};h,k),
$$
$$
\mathsf{TaggedOptionMapRel}_{\delta}(S,T,\rho_{\delta};h',k'),
$$
$$
\mathsf{cert}:\NameCert(S,\sim_S),
$$
and
$$
\mathsf{TaggedOptionSourceHsameCompatible}(S,\sim_S),
$$
then
$$
\mathsf{TaggedOptionHistoryClassifier}_{T,\sim_T}(k,k').
$$
\end{theorem}
\begin{proof}
Split the source classifier witness by its tagged branch and split the two
endpoint map-relation witnesses. In the absent classifier branch, both source
endpoints are presented as $\emp$. A present map-relation branch would also
present one of those endpoints as $\Eone(a)$, contradicting empty-versus-one
no-confusion; hence both map-relation witnesses are absent branches, and the
target classifier is rebuilt by the absent-absent constructor for $k$ and $k'$.

In the present classifier branch, the source witness exposes payloads $a,b$ with
$S(a)$, $S(b)$, endpoint presentations
$\hsame(h,\Eone(a))$ and $\hsame(h',\Eone(b))$, and classifier witness
$a\sim_S b$. The absent map-relation branch is impossible by the same
constructor no-confusion argument, so the two map-relation witnesses expose
payloads $c,d$ with target endpoints
$\hsame(k,\Eone(\delta.\mathsf{map}(c)))$ and
$\hsame(k',\Eone(\delta.\mathsf{map}(d)))$. Endpoint inversion between
$\Eone(a)$ and $\Eone(c)$, and between $\Eone(b)$ and $\Eone(d)$, gives source
history sameness between the corresponding payloads. Source hsame compatibility
turns these sameness witnesses into source classifier witnesses; symmetry and
transitivity from $\mathsf{cert}$ compose them with $a\sim_S b$ to obtain
$c\sim_S d$. Applying $\delta.\mathsf{respects}$ yields
$\delta.\mathsf{map}(c)\sim_T\delta.\mathsf{map}(d)$, while
$\rho_{\delta}$ supplies both target carrier witnesses. These fields rebuild the
present-present target classifier for $k$ and $k'$.
\end{proof}
\leanchecked{BEDC.Derived.OptionUp.TaggedOptionMapRel\_preserves\_classification}

\begin{theorem}[Tagged option payload descent target branch readback]
\label{thm:tagged-option-payload-descent-target-branch-readback}
Let $S,T:\mathsf{BHist}\to\mathsf{Prop}$ be BEDC history source carriers, let
$\sim_S$ and $\sim_T$ be source classifiers, let
$\delta:\mathsf{DescentCertificate}(\mathsf{BHist},\mathsf{BHist},\sim_S,\sim_T)$,
and assume payload preservation
$$
\rho_{\delta}:\forall a.\;S(a)\to T(\delta.\mathsf{map}(a)).
$$
For histories $h,h',k,k'$, assume a source classifier witness
$$
\mathsf{TaggedOptionHistoryClassifier}_{S,\sim_S}(h,h'),
$$
the two map-relation witnesses
$$
\mathsf{TaggedOptionMapRel}_{\delta}(S,T,\rho_{\delta};h,k)
$$
and
$$
\mathsf{TaggedOptionMapRel}_{\delta}(S,T,\rho_{\delta};h',k'),
$$
a source certificate
$$
\mathsf{cert}:\NameCert(S,\sim_S),
$$
and
$$
\mathsf{TaggedOptionSourceHsameCompatible}(S,\sim_S).
$$
Then the target endpoints carry both the transported classifier and the concrete
absent-or-present branch exposure:
$$
\mathsf{TaggedOptionHistoryClassifier}_{T,\sim_T}(k,k')
\land
\left(
\bigl(\hsame(k,\emp)\land\hsame(k',\emp)\bigr)
\lor
\exists c,d:\mathsf{BHist}.\;
T(c)\land T(d)\land
\hsame(k,\Eone(c))\land \hsame(k',\Eone(d))\land c\sim_T d
\right).
$$
\end{theorem}
\begin{proof}
First apply Theorem~\ref{thm:tagged-option-payload-descent-preserves-classification}
to the source classifier witness, the two map-relation witnesses, the source
certificate, and source hsame compatibility. This gives
$\mathsf{TaggedOptionHistoryClassifier}_{T,\sim_T}(k,k')$.

Next apply
Theorem~\ref{thm:tagged-option-history-classifier-branch-exactness} with carrier
$T$, classifier $\sim_T$, and endpoints $k,k'$ to the transported classifier.
The forward direction of that exactness theorem gives exactly the displayed
target branch disjunction. Conjoining the transported classifier with this
branch exposure gives the stated target readback.
\end{proof}

\begin{theorem}[Tagged option map relation common-source target classification]
\label{thm:tagged-option-maprel-common-source-target-classification}
Let $S,T:\mathsf{BHist}\to\mathsf{Prop}$ be BEDC history source carriers, let
$\sim_S$ and $\sim_T$ be source classifiers, let
$\delta:\mathsf{DescentCertificate}(\mathsf{BHist},\mathsf{BHist},\sim_S,\sim_T)$,
and assume payload-carrier preservation
$$
\rho_{\delta}:\forall a.\;S(a)\to T(\delta.\mathsf{map}(a)).
$$
Assume also $\mathsf{TaggedOptionSourceHsameCompatible}(S,\sim_S)$. For
histories $h,k,k'$, if
$$
\mathsf{TaggedOptionMapRel}_{\delta}(S,T,\rho_{\delta};h,k)
$$
and
$$
\mathsf{TaggedOptionMapRel}_{\delta}(S,T,\rho_{\delta};h,k')
$$
then
$$
\mathsf{TaggedOptionHistoryClassifier}_{T,\sim_T}(k,k').
$$
\end{theorem}
\begin{proof}
Split the two map-relation witnesses. If both witnesses are absent, their
target endpoint presentations are $\hsame(k,\emp)$ and $\hsame(k',\emp)$, so
the absent-absent tagged classifier branch gives the conclusion.

In a mixed branch, one witness presents the shared source endpoint $h$ as
$\emp$, while the other presents it as $\Eone(a)$ for some payload $a$.
Symmetry and transitivity of $\hsame$ compose these two presentations into
$\hsame(\emp,\Eone(a))$ or its reverse, contradicting generated-history
empty-versus-one no-confusion. Hence mixed branches cannot occur.

In the present/present branch, the two witnesses expose payloads $a,b$ with
$S(a)$, $S(b)$, target carrier witnesses
$T(\delta.\mathsf{map}(a))$ and $T(\delta.\mathsf{map}(b))$, source endpoint
presentations $\hsame(h,\Eone(a))$ and $\hsame(h,\Eone(b))$, and target
endpoint presentations
$\hsame(k,\Eone(\delta.\mathsf{map}(a)))$ and
$\hsame(k',\Eone(\delta.\mathsf{map}(b)))$. Composing the two source endpoint
presentations and inverting the common one-step constructor yields
$\hsame(a,b)$. Source hsame compatibility converts this to $a\sim_S b$, and
$\delta.\mathsf{respects}$ sends it to
$\delta.\mathsf{map}(a)\sim_T\delta.\mathsf{map}(b)$. Together with the stored
target carrier and endpoint witnesses, this rebuilds the present-present
target classifier for $k$ and $k'$.
\end{proof}
\leanchecked{BEDC.Derived.OptionUp.TaggedOptionMapRel\_common\_source\_target\_classification}

\begin{theorem}[Tagged option map relation common-source visible target completion]
\label{thm:tagged-option-maprel-common-source-visible-target-completion}
Let $S,T:\mathsf{BHist}\to\mathsf{Prop}$ be BEDC history source carriers, let
$\sim_S$ and $\sim_T$ be source classifiers, let
$\delta:\mathsf{DescentCertificate}(\mathsf{BHist},\mathsf{BHist},\sim_S,\sim_T)$,
and assume payload-carrier preservation
$$
\rho_{\delta}:\forall a.\;S(a)\to T(\delta.\mathsf{map}(a)).
$$
Let $\mathsf{cert}_T:\NameCert(T,\sim_T)$, and assume
$\mathsf{TaggedOptionSourceHsameCompatible}(S,\sim_S)$ and
$\mathsf{TaggedOptionSourceHsameCompatible}(T,\sim_T)$. For histories
$h,k,k'$, if
$$
\mathsf{TaggedOptionMapRel}_{\delta}(S,T,\rho_{\delta};h,k)
$$
and
$$
\mathsf{TaggedOptionMapRel}_{\delta}(S,T,\rho_{\delta};h,k'),
$$
then the two target endpoints complete visible payload readback in both
directions:
\begin{enumerate}
  \item for every visible payload $x:\mathsf{BHist}$, if $T(x)$ and
  $\hsame(k,\Eone(x))$, then there exists $y:\mathsf{BHist}$ such that
  $$
  T(y)\land\hsame(k',\Eone(y))\land x\sim_T y;
  $$
  \item for every visible payload $y:\mathsf{BHist}$, if $T(y)$ and
  $\hsame(k',\Eone(y))$, then there exists $x:\mathsf{BHist}$ such that
  $$
  T(x)\land\hsame(k,\Eone(x))\land x\sim_T y.
  $$
\end{enumerate}
\end{theorem}
\begin{proof}
First apply
Theorem~\ref{thm:tagged-option-maprel-common-source-target-classification} to
the two common-source map-relation witnesses and the source hsame compatibility
hypothesis. This gives
$\mathsf{TaggedOptionHistoryClassifier}_{T,\sim_T}(k,k')$.

For a visible payload $x$ at the left target endpoint, apply
Theorem~\ref{thm:tagged-option-history-classifier-visible-branch-inversion} to
that target classifier, the witness $T(x)$, and $\hsame(k,\Eone(x))$. The
visible-branch inversion exposes payloads $x_0,y$ with $T(x_0)$, $T(y)$,
$\hsame(x,x_0)$, $x_0\sim_T y$, and $\hsame(k',\Eone(y))$. Target hsame
compatibility converts $\hsame(x,x_0)$ to $x\sim_T x_0$, and the transitivity
field of $\mathsf{cert}_T$ composes it with $x_0\sim_T y$. The payload $y$ is
the required matching visible endpoint.

For a visible payload $y$ at the right target endpoint, apply
Theorem~\ref{thm:tagged-option-history-classifier-right-visible-branch-inversion}
to the same target classifier, the witness $T(y)$, and
$\hsame(k',\Eone(y))$. The right visible-branch inversion exposes payloads
$x,y_0$ with $T(x)$, $T(y_0)$, $\hsame(y,y_0)$, $x\sim_T y_0$, and
$\hsame(k,\Eone(x))$. Target hsame compatibility converts $\hsame(y,y_0)$ to
$y\sim_T y_0$; the symmetry and transitivity fields of $\mathsf{cert}_T$ then
compose $x\sim_T y_0$ with $y_0\sim_T y$. The payload $x$ gives the reverse
matching visible endpoint.
\end{proof}

\begin{theorem}[Tagged option payload descent transports semantic fields]
\label{thm:tagged-option-payload-descent-semantic-fields}
Let $S,T:\mathsf{BHist}\to\mathsf{Prop}$ be BEDC history source carriers, let
$\sim_S$ and $\sim_T$ be source classifiers, let
$\delta:\mathsf{DescentCertificate}(\mathsf{BHist},\mathsf{BHist},\sim_S,\sim_T)$,
and assume payload-carrier preservation
$$
\rho_{\delta}:\forall a.\;S(a)\to T(\delta.\mathsf{map}(a)).
$$
Let $\mathsf{cert}:\NameCert(S,\sim_S)$ and assume
$\mathsf{TaggedOptionSourceHsameCompatible}(S,\sim_S)$. For histories
$h,h',k,k'$, if
$$
\mathsf{TaggedOptionMapRel}_{\delta}(S,T,\rho_{\delta};h,k)
$$
and
$$
\mathsf{TaggedOptionMapRel}_{\delta}(S,T,\rho_{\delta};h',k'),
$$
then the two source endpoints satisfy their three unary tagged option semantic
fields:
$$
\mathsf{TaggedOptionHistoryCarrier}_S(h)\land
\mathsf{TaggedOptionHistoryPattern}_S(h)\land
\mathsf{TaggedOptionHistoryLedger}_S(h),
$$
$$
\mathsf{TaggedOptionHistoryCarrier}_S(h')\land
\mathsf{TaggedOptionHistoryPattern}_S(h')\land
\mathsf{TaggedOptionHistoryLedger}_S(h'),
$$
and the two target endpoints satisfy the corresponding target fields:
$$
\mathsf{TaggedOptionHistoryCarrier}_T(k)\land
\mathsf{TaggedOptionHistoryPattern}_T(k)\land
\mathsf{TaggedOptionHistoryLedger}_T(k),
$$
$$
\mathsf{TaggedOptionHistoryCarrier}_T(k')\land
\mathsf{TaggedOptionHistoryPattern}_T(k')\land
\mathsf{TaggedOptionHistoryLedger}_T(k').
$$
Moreover, the binary tagged option classifier field descends along the same
two map-relation witnesses:
$$
\mathsf{TaggedOptionHistoryClassifier}_{S,\sim_S}(h,h')\to
\mathsf{TaggedOptionHistoryClassifier}_{T,\sim_T}(k,k').
$$
\end{theorem}
\begin{proof}
Split each instance of
Definition~\ref{def:tagged-option-payload-descent-relation}. In an absent
branch, the witness stores the presentation of the source endpoint as $\emp$
and the target endpoint as $\emp$; these two presentations build the absent
carrier witnesses for the corresponding source and target tagged option
endpoints. In a present branch, the witness exposes a payload $a$ with
$S(a)$, the target payload witness
$T(\delta.\mathsf{map}(a))$ supplied by $\rho_{\delta}$, and the endpoint
presentations $\hsame(h,\Eone(a))$ and
$\hsame(k,\Eone(\delta.\mathsf{map}(a)))$; these fields build the present
carrier witnesses for the source and target endpoints. The same argument
applies to the primed map-relation witness.

For each of the four endpoint carriers just obtained, apply
Theorem~\ref{thm:tagged-option-history-carrier-pattern-ledger-equivalence} to
convert the carrier witness into the corresponding visible tag pattern and
visible tag ledger witnesses. The binary classifier clause is exactly
Theorem~\ref{thm:tagged-option-payload-descent-preserves-classification},
applied to the displayed source classifier witness, the two map-relation
witnesses, the source certificate $\mathsf{cert}$, and the source hsame
compatibility hypothesis.
\end{proof}

\begin{theorem}[Tagged option payload descent public map certificate]
\label{thm:tagged-option-payload-descent-public-map-certificate}
This theorem fits the per-object template as follows: the traditional target is
constructor-tagged option payload descent; the BEDC source is $\mathsf{BHist}$
with $\NameCert$ and
$\mathsf{DescentCertificate}(\mathsf{BHist},\mathsf{BHist},\sim_S,\sim_T)$;
the carrier is tagged option histories; the classifier is the tagged option
history classifier; the stability fields are totality, target single-valuedness,
and classifier preservation; the ledger is visible absent/present tags and
transported payloads; the core theorem is this public map certificate; exactness
is $\mathsf{TaggedOptionMapRel}$ branch analysis; the standard bridge is
none/some payload mapping; and the status is $\mathsf{ProofTarget}$. Let
$S,T:\mathsf{BHist}\to\mathsf{Prop}$ be BEDC history source carriers, let
$\sim_S$ and $\sim_T$ be source classifiers, let
$$
\delta:\mathsf{DescentCertificate}(\mathsf{BHist},\mathsf{BHist},\sim_S,\sim_T),
$$
and assume payload preservation
$$
\rho_{\delta}:\forall a.\;S(a)\to T(\delta.\mathsf{map}(a)).
$$
Let $\mathsf{cert}:\NameCert(S,\sim_S)$ and assume
$\mathsf{TaggedOptionSourceHsameCompatible}(S,\sim_S)$. Then the
constructor-tagged option map relation satisfies the following public map
certificate clauses.
\begin{enumerate}
  \item For every history $h$, if
  $\mathsf{TaggedOptionHistoryCarrier}_S(h)$, then there exists $k:\mathsf{BHist}$
  such that
  $$
  \mathsf{TaggedOptionMapRel}_{\delta}(S,T,\rho_{\delta};h,k)
  $$
  and
  $$
  \mathsf{TaggedOptionHistoryCarrier}_T(k)\land
  \mathsf{TaggedOptionHistoryPattern}_T(k)\land
  \mathsf{TaggedOptionHistoryLedger}_T(k).
  $$
  \item For histories $h,k,k'$, if the same source endpoint $h$ maps to $k$ and
  $k'$, so that
  $$
  \mathsf{TaggedOptionMapRel}_{\delta}(S,T,\rho_{\delta};h,k)
  $$
  and
  $$
  \mathsf{TaggedOptionMapRel}_{\delta}(S,T,\rho_{\delta};h,k'),
  $$
  then
  $$
  \mathsf{TaggedOptionHistoryClassifier}_{T,\sim_T}(k,k').
  $$
  \item For histories $h,h',k,k'$, if $h$ maps to $k$, $h'$ maps to $k'$, and
  the source endpoints are classified together, so that
  $$
  \mathsf{TaggedOptionMapRel}_{\delta}(S,T,\rho_{\delta};h,k),
  $$
  $$
  \mathsf{TaggedOptionMapRel}_{\delta}(S,T,\rho_{\delta};h',k'),
  $$
  and
  $$
  \mathsf{TaggedOptionHistoryClassifier}_{S,\sim_S}(h,h'),
  $$
  then
  $$
  \mathsf{TaggedOptionHistoryClassifier}_{T,\sim_T}(k,k').
  $$
\end{enumerate}

\begin{proof}
For the endpoint totality clause, apply
Theorem~\ref{thm:tagged-option-payload-descent-total-endpoint-transport} to the
source carrier witness. It supplies the target endpoint, the
$\mathsf{TaggedOptionMapRel}$ witness, and the target carrier, pattern, and
ledger fields.

For the same-source target classification clause, apply
Theorem~\ref{thm:tagged-option-maprel-common-source-target-classification} to
the two map-relation witnesses and the source hsame compatibility hypothesis.
This gives the target classifier directly.

For the classifier preservation clause, apply
Theorem~\ref{thm:tagged-option-payload-descent-preserves-classification} to the
source classifier witness, the two map-relation witnesses, the source
certificate $\mathsf{cert}$, and the source hsame compatibility hypothesis.
\end{proof}
