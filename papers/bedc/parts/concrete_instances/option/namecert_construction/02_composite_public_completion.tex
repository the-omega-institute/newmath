\begin{theorem}[Tagged option composite image classifier normalized factored public readback completion]
\label{thm:tagged-option-composite-image-classifier-normalized-factored-public-readback-completion}
Let $S,T,U:\mathsf{BHist}\to\mathsf{Prop}$ be BEDC history source,
intermediate, and target carriers, let $\sim_S$, $\sim_T$, and $\sim_U$ be their
classifiers, and let
$$
\delta:\mathsf{DescentCertificate}(\mathsf{BHist},\mathsf{BHist},\sim_S,\sim_T)
$$
and
$$
\epsilon:\mathsf{DescentCertificate}(\mathsf{BHist},\mathsf{BHist},\sim_T,\sim_U).
$$
Assume payload preservation fields
$$
\rho_{\delta}:\forall a.\;S(a)\to T(\delta.\mathsf{map}(a))
$$
and
$$
\rho_{\epsilon}:\forall b.\;T(b)\to U(\epsilon.\mathsf{map}(b)).
$$
Assume endpoint stability for the second descent:
$$
\eta_{\epsilon}:\forall x\,y.\;
T(x)\to T(y)\to \hsame(x,y)\to
\hsame(\epsilon.\mathsf{map}(x),\epsilon.\mathsf{map}(y)).
$$
Write $\epsilon\circ\delta$ for the descent certificate whose map field is
$a\mapsto\epsilon.\mathsf{map}(\delta.\mathsf{map}(a))$ and whose respects field
sends $r:a\sim_S b$ to
$\epsilon.\mathsf{respects}(\delta.\mathsf{respects}(r))$. Define
$$
\rho_{\epsilon\circ\delta}(a,Sa)=
\rho_{\epsilon}(\delta.\mathsf{map}(a),\rho_{\delta}(a,Sa)).
$$
Let $\mathsf{cert}_S:\NameCert(S,\sim_S)$,
$\mathsf{cert}_T:\NameCert(T,\sim_T)$, and
$\mathsf{cert}_U:\NameCert(U,\sim_U)$. Assume
$\mathsf{TaggedOptionSourceHsameCompatible}(S,\sim_S)$,
$\mathsf{TaggedOptionSourceHsameCompatible}(T,\sim_T)$, and
$\mathsf{TaggedOptionSourceHsameCompatible}(U,\sim_U)$. For final endpoints
$m,m':\mathsf{BHist}$ and normalized endpoints $u,u':\mathsf{BHist}$, if
$$
\mathsf{TaggedOptionPayloadDescentImageClassifier}_{\epsilon\circ\delta}
(S,U,\rho_{\epsilon\circ\delta};m,m')
$$
and
$$
\hsame(m,u)\land \hsame(m',u'),
$$
then either
\begin{enumerate}
\item $\hsame(u,\emp)\land \hsame(u',\emp)$;
\item $\neg\exists x,y:\mathsf{BHist}.\;
U(x)\land U(y)\land
\hsame(u,\Eone(x))\land
\hsame(u',\Eone(y))$;
\item there exist $k,k':\mathsf{BHist}$ such that
\begin{enumerate}
\item $\mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
(S,T,\rho_{\delta};k,k')$;
\item $\mathsf{TaggedOptionMapRel}_{\epsilon}(T,U,\rho_{\epsilon};k,u)$ and
$\mathsf{TaggedOptionMapRel}_{\epsilon}(T,U,\rho_{\epsilon};k',u')$;
\item $\hsame(k,\emp)\land \hsame(k',\emp)$;
\item $\neg\exists p,q:\mathsf{BHist}.\;
T(p)\land T(q)\land
\hsame(k,\Eone(p))\land
\hsame(k',\Eone(q))$.
\end{enumerate}
\end{enumerate}
or there exist $x,y,k,k',p,q,a,b:\mathsf{BHist}$ such that
\begin{enumerate}
\item $U(x)$, $U(y)$, $\hsame(u,\Eone(x))$, $\hsame(u',\Eone(y))$, and
$x\sim_U y$;
\item $\neg\hsame(u,\emp)$ and $\neg\hsame(u',\emp)$;
\item $\mathsf{TaggedOptionPayloadDescentImageClassifier}_{\delta}
(S,T,\rho_{\delta};k,k')$;
\item $\mathsf{TaggedOptionMapRel}_{\epsilon}(T,U,\rho_{\epsilon};k,u)$ and
$\mathsf{TaggedOptionMapRel}_{\epsilon}(T,U,\rho_{\epsilon};k',u')$;
\item $T(p)$, $T(q)$, $U(\epsilon.\mathsf{map}(p))$, and
$U(\epsilon.\mathsf{map}(q))$;
\item $\hsame(k,\Eone(p))$, $\hsame(k',\Eone(q))$,
$\hsame(x,\epsilon.\mathsf{map}(p))$, and
$\hsame(y,\epsilon.\mathsf{map}(q))$;
\item $S(a)$, $S(b)$, $a\sim_S b$, $T(\delta.\mathsf{map}(a))$, and
$T(\delta.\mathsf{map}(b))$;
\item $\hsame(p,\delta.\mathsf{map}(a))$ and
$\hsame(q,\delta.\mathsf{map}(b))$;
\item $p\sim_T q$;
\item for all $x',y':\mathsf{BHist}$, if $U(x')$, $U(y')$,
$\hsame(u,\Eone(x'))$, and $\hsame(u',\Eone(y'))$, then
$\hsame(x,x')\land \hsame(y,y')\land x'\sim_U y'$;
\item for every $x_L:\mathsf{BHist}$, if $U(x_L)$ and
$\hsame(u,\Eone(x_L))$, then $\hsame(x,x_L)$ and $x_L\sim_U y$;
\item for every $y_R:\mathsf{BHist}$, if $U(y_R)$ and
$\hsame(u',\Eone(y_R))$, then $\hsame(y,y_R)$ and $x\sim_U y_R$.
\end{enumerate}
\end{theorem}
\begin{proof}
Apply
Theorem~\ref{thm:tagged-option-composite-image-classifier-normalized-public-readback-completion-package}
to the composite image-classifier witness, the endpoint normalization witnesses,
and the displayed descent, preservation, stability, certificate, and
compatibility data. This gives an exhaustive absent-or-present public readback
completion at $(u,u')$.

In the absent branch, retain the two normalized empty presentations and the
exclusion of every visible $U$-payload pair at $(u,u')$. Apply
Theorem~\ref{thm:tagged-option-composite-image-classifier-normalized-absent-public-factorization}
to the same composite witness, endpoint normalizations, displayed data, and
absent endpoint presentations. This supplies intermediate endpoints $k,k'$ with
the $\delta$ classifier factor, the two $\epsilon$ public map-relation links,
the intermediate absent presentations, and the excluded intermediate visible
pair. Package these data as the absent factored branch.

In the present branch, retain the selected payloads $x,y$, their visible target
presentations, the target classification $x\sim_U y$, the two non-absence facts,
and the universal competing visible-pair uniqueness clause. Apply
Theorem~\ref{thm:tagged-option-composite-image-classifier-normalized-selected-visible-public-factorization}
to the same composite witness, endpoint normalizations, displayed data, and
selected visible presentation. This supplies $k,k',p,q,a,b$ with the complete
selected visible public factorization and the one-sided competing-payload
uniqueness clauses. Package those witnesses together with the retained
completion uniqueness clause.
\end{proof}

Theorem~\ref{thm:tagged-option-payload-descent-image-classifier-branch-exactness}
to the transported classifier witness. This leaves the two possible public
branches at the normalized endpoints.

$\hsame(u',\emp)$. Any presentation $\hsame(u,\Ezero(t))$ or
$\hsame(u',\Ezero(t))$ composes with the corresponding empty presentation and
contradicts $\Ezero$-versus-empty generated-history no-confusion.

In the present branch, exactness exposes payloads $x,y:\mathsf{BHist}$ with
$\hsame(u,\Eone(x))$ and $\hsame(u',\Eone(y))$. Any zero-tag presentation at
either endpoint composes with the corresponding one-tag presentation and
contradicts zero-versus-one generated-history no-confusion.

Since the branch analysis covers every transported normalized classifier
witness, both normalized endpoints reject every malformed zero-tag tail.
\end{proof}
