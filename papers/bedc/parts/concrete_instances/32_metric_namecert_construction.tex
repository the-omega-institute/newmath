\chapter{A Concrete Naming Certificate for \(\MetricSpaceUp\)}
\label{ch:concrete-instances-metric-namecert}

The metric space interface \(\MetricSpaceUp\) has certificate obligations over a carrier together with a distance function \(d : C \times C \to \RealUp\) satisfying non-negativity, identity-of-indiscernibles (with apartness in place of decidable equality), symmetry, and the triangle inequality, all expressed in Bishop-style with explicit real moduli. It depends on \autoref{ch:concrete-instances-real-namecert}.

\section{Carrier definition}

\begin{definition}[MetricSpace carrier]
\label{def:metric-carrier}
Use an abstract carrier \(C\), classifier, apartness, and distance \(d:C\times C\to\RealUp\).
\end{definition}

\section{The certificate}

\begin{definition}[MetricSpace source specification]
\label{def:metric-source-specification}
The source is \(C\) together with the Bishop real distance codomain.
\end{definition}

\begin{definition}[MetricSpace pattern specification]
\label{def:metric-pattern-specification}
The pattern is metric judgment formation from carrier pairs to real distance witnesses.
\end{definition}

\begin{definition}[MetricSpace classifier specification]
\label{def:metric-classifier-specification}
The classifier is the supplied carrier sameness, with distance stable under classifier replacement.
\end{definition}

\begin{definition}[MetricSpace stability certificate]
\label{def:metric-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item distance non-negativity
  \item identity of indiscernibles using classifier and apartness
  \item symmetry
  \item triangle inequality
  \item distance congruence in both arguments
  \item explicit real moduli for all inequality fields
\end{enumerate}
\end{definition}

\begin{theorem}[Metric empty distance witness exactness]
\label{thm:metric-distance-witness-empty-distance-iff}
For histories $x,y$, a concrete distance witness with empty distance ledger is
equivalent to both endpoint histories being empty up to the history classifier:
$$
\mathsf{MetricDistanceWitness}(x,y,\emp)
\Longleftrightarrow
\hsame(x,\emp)\wedge\hsame(y,\emp).
$$
\end{theorem}
\begin{proof}
In the forward direction, unpack the concrete distance witness. Its
continuation field is $\mathsf{Cont}(x,y,\emp)$, and empty-result inversion for
continuation forces both endpoint histories to be empty.

Conversely, case analysis on the two endpoint sameness witnesses reduces the
claim to the empty history. The empty endpoint, empty target, and empty distance
are unary, and the continuation ledger is the reflexive empty concatenation.
\end{proof}
\leanchecked{BEDC.Derived.MetricUp.MetricDistanceWitness\_empty\_distance\_iff}

\begin{theorem}[Metric distance witness depth additivity]
\label{thm:metric-distance-witness-depth-add}
Let $\mathsf{MetricDistanceDepth}$ be the structural depth of a BEDC history,
counting each visible constructor above $\emp$. For histories $x,y,d$, every
concrete metric distance witness carries exactly the append-depth ledger:
$$
\mathsf{MetricDistanceWitness}(x,y,d)
\Longrightarrow
\mathsf{MetricDistanceDepth}(d)
=
\mathsf{MetricDistanceDepth}(x)+\mathsf{MetricDistanceDepth}(y).
$$
\end{theorem}
\begin{proof}
Unpack the concrete witness. Its continuation field is precisely the BEDC
append ledger $d=x\cdot y$. The structural-depth equation follows by induction
on the right history $y$, with the empty case reflexive and both visible
constructor cases obtained by applying successor congruence to the induction
hypothesis.
\end{proof}

\begin{theorem}[Metric distance witness symmetric classifier]
\label{thm:metric-distance-witness-symmetric-classifier}
For unary endpoint histories $x,y$, if $d_{xy}$ is witnessed by the
continuation ledger from $x$ to $y$ and $d_{yx}$ is witnessed by the reversed
ledger from $y$ to $x$, then the two distance histories classify together by
$\hsame$:
$$
\mathsf{MetricDistanceWitness}(x,y,d_{xy})
\Longrightarrow
\mathsf{MetricDistanceWitness}(y,x,d_{yx})
\Longrightarrow
\hsame(d_{xy},d_{yx}).
$$
\end{theorem}
\begin{proof}
Unpack the two concrete distance witnesses. The endpoint histories are unary in
both directions, and the two displayed continuations are precisely the forward
and reversed unary continuation ledgers. Unary continuation commutativity gives
$\hsame(d_{xy},d_{yx})$.
\end{proof}

\begin{theorem}[Metric distance witness prefix closure]
\label{thm:metric-distance-witness-prefix-closed}
For unary histories $p,x,y,d$, a unary visible prefix on the first endpoint
and the distance ledger preserves the concrete metric distance witness:
$$
\mathsf{UnaryHistory}(p)
\Longrightarrow
\mathsf{MetricDistanceWitness}(x,y,d)
\Longrightarrow
\mathsf{MetricDistanceWitness}(p\cdot x,y,p\cdot d).
$$
\end{theorem}
\begin{proof}
Unpack the witness into unary endpoint and distance histories together with the
continuation ledger $d=x\cdot y$. Unary append closure gives unary histories
for $p\cdot x$ and $p\cdot d$. The prefixed continuation ledger is exactly the
associativity equality $(p\cdot x)\cdot y=p\cdot(x\cdot y)$.
\end{proof}
\leanchecked{BEDC.Derived.MetricUp.MetricDistanceWitness\_prefix\_closed}

\begin{theorem}[Metric distance witness prefix iff]
\label{thm:metric-distance-witness-prefix-iff}
For histories $p,x,y,d$, a prefixed source and distance witness is equivalent
to unary evidence for the prefix together with the unprefixed distance witness:
$$
\mathsf{MetricDistanceWitness}(p\cdot x,y,p\cdot d)
\Longleftrightarrow
\mathsf{UnaryHistory}(p)\ \wedge\ \mathsf{MetricDistanceWitness}(x,y,d).
$$
\end{theorem}
\begin{proof}
From the prefixed witness, unary append factorization recovers unary evidence
for $p$, $x$, and $d$, and continuation prefix cancellation recovers
$\mathsf{Cont}(x,y,d)$.

Conversely, unary append closure builds the prefixed source and distance
histories, and continuation associativity transports the original ledger
$d=x\cdot y$ to $p\cdot d=(p\cdot x)\cdot y$.
\end{proof}
\leanchecked{BEDC.Derived.MetricUp.MetricDistanceWitness\_prefix\_iff}

\begin{theorem}[Metric distance witness suffix iff]
\label{thm:metric-distance-witness-suffix-iff}
For histories $p,x,y,d$, a common right suffix on the target and distance is
equivalent to unary evidence for the suffix together with the unextended
distance witness:
$$
\mathsf{MetricDistanceWitness}(x,y\cdot p,d\cdot p)
\Longleftrightarrow
\mathsf{UnaryHistory}(p)\ \wedge\ \mathsf{MetricDistanceWitness}(x,y,d).
$$
\end{theorem}
\begin{proof}
From the suffixed witness, unary append factorization recovers unary evidence
for $y$, $p$, and $d$. Reassociation exposes a common right suffix $p$, and
right append cancellation recovers the base continuation
$\mathsf{Cont}(x,y,d)$.

Conversely, unary append closure builds the suffixed target and distance
histories, and associativity transports the original ledger
$d=x\cdot y$ to $d\cdot p=x\cdot(y\cdot p)$.
\end{proof}
\leanchecked{BEDC.Derived.MetricUp.MetricDistanceWitness\_suffix\_iff}

\begin{theorem}[Metric distance witness visible-context exactness]
\label{thm:metric-distance-witness-visible-context-iff}
For histories $p,q,x,y,d$, simultaneous visible ledgers on the source side and
target side are exactly equivalent to unary evidence for the two ledgers
together with the central distance witness:
$$
\mathsf{MetricDistanceWitness}(p\cdot x,y\cdot q,(p\cdot d)\cdot q)
\Longleftrightarrow
\mathsf{UnaryHistory}(p)\ \wedge\ \mathsf{UnaryHistory}(q)\ \wedge\
\mathsf{MetricDistanceWitness}(x,y,d).
$$
\end{theorem}
\begin{proof}
In the forward direction, apply
\autoref{thm:metric-distance-witness-suffix-iff} with source $p\cdot x$,
target $y$, distance $p\cdot d$, and suffix $q$. This recovers
$\mathsf{UnaryHistory}(q)$ together with
$\mathsf{MetricDistanceWitness}(p\cdot x,y,p\cdot d)$. Then apply
\autoref{thm:metric-distance-witness-prefix-iff} to the remaining witness to
recover $\mathsf{UnaryHistory}(p)$ and
$\mathsf{MetricDistanceWitness}(x,y,d)$.

Conversely, assume unary evidence for $p$ and $q$ together with the central
distance witness. Apply \autoref{thm:metric-distance-witness-prefix-iff} from
right to left to build
$\mathsf{MetricDistanceWitness}(p\cdot x,y,p\cdot d)$. Then apply
\autoref{thm:metric-distance-witness-suffix-iff} from right to left with suffix
$q$ to append the visible target ledger and obtain
$\mathsf{MetricDistanceWitness}(p\cdot x,y\cdot q,(p\cdot d)\cdot q)$.
\end{proof}
\leanchecked{BEDC.Derived.MetricUp.MetricDistanceWitness\_visible\_context\_iff}

\begin{theorem}[Metric distance witness visible-context result determinacy]
\label{thm:metric-distance-witness-visible-context-result-deterministic}
For histories $p,q,x,y,d,d'$, if the same visible endpoint histories carry two
visible distance ledgers with central parts $d$ and $d'$, then those central
parts classify together:
$$
\mathsf{MetricDistanceWitness}(p\cdot x,y\cdot q,(p\cdot d)\cdot q)
\Longrightarrow
\mathsf{MetricDistanceWitness}(p\cdot x,y\cdot q,(p\cdot d')\cdot q)
\Longrightarrow
\hsame(d,d').
$$
\end{theorem}
\begin{proof}
Apply \autoref{thm:metric-distance-witness-visible-context-iff} to the first
visible witness. This recovers $\mathsf{MetricDistanceWitness}(x,y,d)$ together
with unary evidence for the visible prefix and suffix.

Apply the same visible-context exactness theorem to the second visible witness.
This recovers $\mathsf{MetricDistanceWitness}(x,y,d')$ over the same central
endpoints.

Unpack the two central witnesses. Their continuation ledgers have the same
source endpoint $x$ and target endpoint $y$, namely $\ContR(x,y,d)$ and
$\ContR(x,y,d')$. By continuation determinacy
\autoref{thm:continuation-determinacy}, their results classify together, giving
$\hsame(d,d')$.

The per-object template is as follows. The traditional target is visible metric
distance readback. The BEDC source is $\mathsf{BHist}$ with $\ContR$ and
$\mathsf{UnaryHistory}$. The carrier is $\mathsf{MetricDistanceWitness}$ and the
classifier is $\hsame$. The ledger is the visible prefix and suffix context
around the central endpoints. The core theorem is central-result determinacy,
exactness is uniqueness of the central distance, and the status is
$\mathsf{ProofTarget}$.
\end{proof}
\leanchecked{BEDC.Derived.MetricUp.MetricDistanceWitness\_visible\_context\_result\_deterministic}

\begin{theorem}[Metric distance witness visible-context determinacy]
\label{thm:metric-distance-witness-visible-context-determinacy}

visible endpoints determine the same central distance:
$$
\begin{array}{l}
\mathsf{MetricDistanceWitness}(p\cdot x,y\cdot q,(p\cdot d)\cdot q)
\Longrightarrow
\\
\mathsf{MetricDistanceWitness}(p\cdot x,y\cdot q,(p\cdot e)\cdot q)
\Longrightarrow
\hsame(d,e).
\end{array}
$$
\end{theorem}
\begin{proof}
Apply \autoref{thm:metric-distance-witness-visible-context-iff} to the first
contextual witness to recover the central witness
$\mathsf{MetricDistanceWitness}(x,y,d)$. Apply the same theorem to the second
contextual witness to recover
$\mathsf{MetricDistanceWitness}(x,y,e)$.

Unpack the two central distance witnesses. They provide continuation ledgers
$\ContR(x,y,d)$ and $\ContR(x,y,e)$ over the same source and target histories;
continuation determinacy, Theorem~\ref{thm:continuation-determinacy}, yields
$\hsame(d,e)$.
\end{proof}
