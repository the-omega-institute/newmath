\chapter{A Concrete Naming Certificate for \(\AddUp\)}
\label{ch:concrete-instances-add-namecert}

The addition-like interface is licensed by continuation of unary histories. It is not a primitive operation.

\section{Unary continuation}

\begin{lemma}[Continuation preserves unary histories]
If \(\mathsf{Unary}(h)\), \(\mathsf{Unary}(k)\), and \(\ContR(h,k,r)\), then \(\mathsf{Unary}(r)\).
\end{lemma}
\leanchecked{BEDC.FKernel.Unary.unary\_cont\_preserves\_unary\_by\_induction}
\begin{proof}
By induction on the generated proof of \(\ContR(h,k,r)\). In the empty case, \(r=k\), hence \(\mathsf{Unary}(r)\) follows from the second premise. In the \(\Eone\)-continuation case, use the induction hypothesis on the subcontinuation and then apply unary closure under \(\Eone\). The \(\Ezero\)-continuation case makes the second input history have shape \(\Ezero(k_0)\), contradicting the premise \(\mathsf{Unary}(k)\) by unary no-zero-extension.
\end{proof}

\begin{lemma}[Unary continuation unit]
\label{lem:unary-continuation-unit}
For every unary history \(h\), continuation by the empty history and continuation from the empty history produce unary histories internally same to the expected unary source.
\end{lemma}
\leanchecked{BEDC.FKernel.Unary.unary\_cont\_unit}
\leanvariant{BEDC.FKernel.Unary.unary\_cont\_unit\_same\_pair}
\begin{proof}
The left unit is the empty continuation generator. The right unit follows from the continuation unit theorem in the proof-standing layer, restricted to unary histories. Unary closure ensures the resulting histories remain within the unary source discipline.
\end{proof}

\begin{lemma}[Unary continuation associativity]
\label{lem:unary-continuation-associativity}
Unary continuation is associative up to \(\HSame\) when all participating histories are unary.
\end{lemma}
\leanchecked{BEDC.FKernel.Unary.unary\_continuation\_associativity}
\begin{proof}
This is the continuation associativity theorem specialized to unary histories, together with continuation-preserves-unary-histories for the intermediate results.
\end{proof}

\section{The certificate}

\begin{definition}[Additive source specification]
\label{def:additive-source-specification}
The source specification \(\mathsf{AddSource}\) consists of ordered pairs of unary histories together with a continuation witness.
\end{definition}
\leandef{BEDC.FKernel.Unary.AddSourceSpec}
\leanvariant{BEDC.FKernel.Unary.AddSourceSpec\_from\_unary\_cont}
\leanchecked{BEDC.FKernel.Unary.AddSourceSpec\_result\_unary}

\begin{definition}[Additive pattern specification]
\label{def:additive-pattern-specification}
The pattern specification \(\mathsf{AddPattern}\) is unary continuation:
$$
  \ContR(h,k,r)
$$
with \(\mathsf{Unary}(h)\) and \(\mathsf{Unary}(k)\).
\end{definition}
\leanchecked{BEDC.FKernel.Unary.additive\_pattern\_result\_unary\_iff\_inputs}

\begin{definition}[Additive classifier specification]
\label{def:additive-classifier-specification}
The classifier specification \(\mathsf{AddClassifier}\) classifies additive outputs by \(\HSame\) of the generated result histories.
\end{definition}
\leandef{BEDC.FKernel.Unary.AddClassifierSpec}

\begin{definition}[Additive stability certificate]
\label{def:additive-stability-certificate}
The stability certificate \(\mathsf{AddStabilityCert}\) consists of:
\begin{enumerate}
  \item continuation preserves unary histories;
  \item unary continuation unit;
  \item unary continuation associativity;
  \item continuation determinacy up to \(\HSame\).
\end{enumerate}
\end{definition}
\leanchecked{BEDC.FKernel.Unary.additive\_stability\_certificate\_fields}
\leanvariant{BEDC.FKernel.Unary.additive\_stability\_certificate\_determinacy\_field}
\leanvariant{BEDC.FKernel.Unary.additive\_stability\_certificate\_unit\_assoc\_determinacy}

\begin{definition}[Additive ledger policy]
\label{def:additive-stability-and-ledger-policy}
The additive ledger policy \(\mathsf{AddLedger}\) records the two input unary histories and the continuation witness generating the output history. For composite additive uses, the ledger composes by the composite gap principle.
\end{definition}
\leandef{BEDC.FKernel.Unary.AddLedgerPolicy}
\leanvariant{BEDC.FKernel.Unary.AddLedgerPolicy\_iff\_unary\_cont\_closed}
\leanvariant{BEDC.FKernel.Unary.addLedgerPolicy\_from\_unary\_cont\_closed}
\leanvariant{BEDC.FKernel.Unary.unary\_addition\_seed\_from\_policy}
\leanchecked{BEDC.FKernel.Unary.addLedgerPolicy\_from\_unary\_cont\_closed}

\begin{theorem}[Concrete \(\AddUp\) naming certificate]
\label{thm:concrete-addup-namecert}
The tuple
$$
  \mathsf{NameCert}_{\AddUp}
  :=
  (\mathsf{AddSource},
   \mathsf{AddPattern},
   \mathsf{AddClassifier},
   \mathsf{AddStabilityCert},
   \mathsf{AddLedger})
$$
forms a valid naming certificate for the derived interface \(\AddUp\).
\end{theorem}
\leanchecked{BEDC.FKernel.Unary.add\_up\_name\_certificate\_exists}
\leanvariant{BEDC.FKernel.Unary.add\_up\_name\_certificate}
\begin{proof}
The source, pattern, classifier, stability, and ledger fields are supplied by the preceding definitions. Stability is justified by the three unary continuation lemmas and continuation determinacy. The ledger policy records the exact continuation witness rather than erasing it. Thus the naming protocol is satisfied.
\end{proof}

\begin{corollary}[\(\AddUp\) is licensed by continuation]
The additive interface \(\AddUp\) is licensed by unary continuation. It is not a primitive operation of the finite kernel.
\end{corollary}
\leanchecked{BEDC.FKernel.Unary.add\_up\_licensed\_not\_primitive}

\begin{theorem}[Unary continuation monoid activation]
\label{thm:unary-continuation-monoid-activation}
Let the typed naming-certificate template of \autoref{def:typed-naming-certificate} be instantiated on the unary carrier with classifier \(\hsame\). Assume the core carrier certificate \(\NameCert(\Unary,\hsame)\) from \autoref{def:lean-carrier-naming-certificate}; take the source, pattern, classifier, and ledger policies to be the continuation policies of \autoref{def:additive-source-specification}, \autoref{def:additive-pattern-specification}, \autoref{def:additive-classifier-specification}, and \autoref{def:additive-stability-and-ledger-policy}; assume pattern soundness and ledger soundness for those policies; and assume that \(\mathsf{AddStabilityCert}\) supplies unary closure, empty-history unit, associativity up to \(\hsame\), and same-source determinacy up to \(\hsame\). Then
\[
  \mathsf{License}(\UnaryContMonoidUp).
\]
\end{theorem}
\begin{proof}
By \autoref{def:typed-naming-certificate}, a derived interface is licensed by a certificate once its source, pattern, classifier, stability, and ledger fields are supplied. The assumed \(\NameCert(\Unary,\hsame)\) provides the carrier-and-classifier core described in \autoref{def:lean-carrier-naming-certificate}. The source is the ordered unary source with a continuation witness from \autoref{def:additive-source-specification}; the pattern is unary continuation from \autoref{def:additive-pattern-specification}; the classifier is \(\hsame\) on continuation results from \autoref{def:additive-classifier-specification}; and the ledger is the recorded pair of unary inputs together with the continuation witness from \autoref{def:additive-stability-and-ledger-policy}. Pattern soundness sends each source witness to the displayed unary continuation pattern, and ledger soundness records the same source witness in the ledger policy.

The remaining certificate component is stability. Unary closure is the closure field in \(\mathsf{AddStabilityCert}\). The empty-history unit is \autoref{lem:unary-continuation-unit}. Associativity up to \(\hsame\) is \autoref{lem:unary-continuation-associativity}. Same-source determinacy is the determinacy field of \(\mathsf{AddStabilityCert}\), so two continuation results generated from the same ordered unary source pair are classified together by \(\hsame\). These fields give a closed carrier, a unit, associative continuation, and deterministic result classification for fixed sources. That is exactly the certificate data for the unary continuation monoid-like interface. Since no order-swap commutativity field is used, the activated upward name is \(\UnaryContMonoidUp\).
\end{proof}

\begin{theorem}[Same-source additive determinacy bridge]
\label{thm:same-source-additive-determinacy-bridge}
For histories \(h\), \(k\), \(r\), and \(r'\), if \(\mathsf{AddSource}(h,k,r)\) and \(\mathsf{AddSource}(h,k,r')\), then \(\mathsf{AddClassifier}(r,r')\).
\end{theorem}
\begin{proof}
By \autoref{def:additive-source-specification}, the first source witness supplies unary inputs \(h\) and \(k\) together with \(\ContR(h,k,r)\). The second source witness has the same ordered input pair, so it supplies the same unary inputs together with \(\ContR(h,k,r')\). The determinacy field listed in \autoref{def:additive-stability-certificate} applies to these two continuation witnesses and gives \(\HSame(r,r')\). By \autoref{def:additive-classifier-specification}, \(\mathsf{AddClassifier}\) classifies additive outputs precisely by \(\HSame\) of their generated result histories, so this witness is \(\mathsf{AddClassifier}(r,r')\).
\end{proof}
