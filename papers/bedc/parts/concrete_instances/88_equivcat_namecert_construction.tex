\chapter{A Concrete Naming Certificate for $\EquivCatUp$}
\label{ch:concrete-instances-equivcat-namecert}

The categorical-equivalence interface $\EquivCatUp$ packages a pair of functors that are mutually quasi-inverse together with the natural-isomorphism witnesses. It depends on $\NameCert_{\AdjunctionUp}$ (\autoref{ch:concrete-instances-adjunction-namecert}) and $\NameCert_{\NatTransUp}$ (\autoref{ch:concrete-instances-nattrans-namecert}), and is the structural notion of sameness for $\NameCert_{\CategoryUp}$ instances.

\section{Carrier definition}
\label{sec:equivcat-carrier}

\begin{theorem}[Empty roundtrip identity components]
\label{thm:equivcat-adjunction-empty-roundtrip-identity-components}
For histories $p,q,a,unit,counit,left,right:\mathsf{BHist}$, if an
adjunction unit-counit carrier has both displayed roundtrip results
$left$ and $right$ classifier-equal to $\emp$, then the unit and counit
components are the empty identity components:
$$
\mathsf{NatTransPrefixComponentCarrier}(p,q,a,\emp)
\quad\text{and}\quad
\mathsf{NatTransPrefixComponentCarrier}(q,p,a,\emp).
$$
\end{theorem}
\begin{proof}
The empty left roundtrip continuation forces both $unit$ and $counit$ to be
$\emp$. The empty right roundtrip gives the same collapse in the opposite
order. After these endpoint inversions, the component fields of the
adjunction unit-counit carrier are exactly the two empty identity components.
\end{proof}
\leanchecked{BEDC.Derived.EquivCatUp.EquivCatAdjunction\_empty\_roundtrip\_identity\_components}

\begin{theorem}[Equivalence empty roundtrip prefix determinacy]
\label{thm:equivcat-empty-roundtrip-prefix-determinacy}
For histories $p,q,a,unit,counit,left,right:\mathsf{BHist}$, if
$\mathsf{AdjunctionUnitCounitCarrier}(p,q,a,unit,counit,left,right)$ holds
and $left\hsame\emp$ and $right\hsame\emp$, then
$$
p\hsame q,\qquad
unit\hsame\emp,\qquad
counit\hsame\emp,
$$
and
$$
\mathsf{NatTransPrefixComponentCarrier}(p,q,a,\emp)
\quad\text{and}\quad
\mathsf{NatTransPrefixComponentCarrier}(q,p,a,\emp).
$$
\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:equivcat-adjunction-empty-roundtrip-identity-components}
to the adjunction unit-counit carrier and the two empty displayed triangle
results. This gives the two empty natural-transformation prefix components
from $p$ to $q$ and from $q$ back to $p$ at the object history $a$.

The NatTrans empty-identity exactness theorem applied to the first empty
component extracts the prefix classifier equality $p\hsame q$.

The stored triangle continuations in the adjunction unit-counit carrier have
results $left$ and $right$. Empty-result inversion for the displayed
continuations, together with $left\hsame\emp$ and $right\hsame\emp$, collapses
the unit and counit histories to $unit\hsame\emp$ and $counit\hsame\emp$.
Collecting these conclusions gives the stated determinacy package.
\end{proof}
\leanchecked{BEDC.Derived.EquivCatUp.EquivCatAdjunction\_empty\_roundtrip\_prefix\_determinacy}

\begin{theorem}[EquivCat empty roundtrip target-prefix determinacy]
\label{thm:equivcat-adjunction-empty-roundtrip-target-prefix-deterministic}
For histories $p,q,r,a,unit,counit,left,right:\mathsf{BHist}$, if
$\mathsf{AdjunctionUnitCounitCarrier}(p,q,a,unit,counit,left,right)$ holds,
$left\hsame\emp$ and $right\hsame\emp$, and
$$
\mathsf{NatTransPrefixComponentCarrier}(p,r,a,\emp)
$$
is also displayed, then $q\hsame r$.
\end{theorem}
\begin{proof}
The empty roundtrip identity-components theorem extracts the reverse empty
component
$$
\mathsf{NatTransPrefixComponentCarrier}(q,p,a,\emp)
$$
from the adjunction carrier and the two empty displayed roundtrips. Composing
that empty component with the displayed empty component from $p$ to $r$ by
NatTrans empty-identity prefix transitivity identifies the target prefix
$q$ with $r$.
\end{proof}
\leanchecked{BEDC.Derived.EquivCatUp.EquivCatAdjunction\_empty\_roundtrip\_target\_prefix\_deterministic}

\section{The certificate}
\label{sec:equivcat-certificate}

\begin{theorem}[Equivalence cycle identity carrier semantic certificate]
\label{thm:equivcat-cycle-identity-carrier-semantic-namecert}
For BEDC categorical hom carriers
$left:\mathsf{CategoryHomCarrier}(a,b,f)$ and
$right:\mathsf{CategoryHomCarrier}(b,a,g)$, the carrier of displayed cycle
identities
$$
\mathsf{CategoryHomCarrier}(a,b,t)\wedge
\mathsf{CategoryHomCarrier}(b,a,g)\wedge
t\hsame\emp\wedge g\hsame\emp
$$
has a semantic naming certificate with classifier $\hsame$.
\end{theorem}
\begin{proof}
The cycle-tail theorem for categorical hom carriers identifies both cycle
tails with the empty history. Transporting $left$ along the displayed tail
comparison gives the witness at $\emp$, while $right$ supplies the fixed
reverse carrier. The classifier laws are the finite-kernel equivalence laws
for $\hsame$. Carrier transport keeps the endpoints fixed, transports the
first hom carrier along the classifier, and composes the inverse classifier
with the stored empty-tail proof.
\end{proof}
\leanchecked{BEDC.Derived.EquivCatUp.EquivCatCycleIdentityCarrier\_semanticNameCert}
