\chapter{A Concrete Naming Certificate for \(\GroupUp\)}
\label{ch:concrete-instances-group-namecert}

The group interface \(\GroupUp\) extends \autoref{ch:concrete-instances-monoid-namecert} with a unary inverse operation \({}^{-1} : C \to C\) satisfying left and right inverse laws under the carrier's classifier. It is not a primitive structure; it is licensed only by composing the monoid certificate with the inverse field.

\begin{remark}[Roadmap chapter]
\label{rem:concrete-instances-group-namecert-roadmap}
This chapter records the licensing target. Lean markers identify the
algebraic fields and derived laws that are formalized for the certificate.
\end{remark}

\section{Carrier definition}

\begin{definition}[Group carrier]
\label{def:group-carrier}
Use a monoid setup extended with inverse operation \(a^{-1}\).
\end{definition}

\section{The certificate}

\begin{definition}[Group source specification]
\label{def:group-source-specification}
The source is a certified monoid carrier plus inverse data.
\end{definition}

\begin{definition}[Group pattern specification]
\label{def:group-pattern-specification}
The pattern is monoid expression formation extended by inverse formation.
\end{definition}

\begin{definition}[Group classifier specification]
\label{def:group-classifier-specification}
The classifier is the monoid carrier classifier, with inverse congruence.
\end{definition}

\begin{definition}[Group stability certificate]
\label{def:group-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item all monoid stability fields
  \item inverse congruence
  \item left inverse law
  \item right inverse law
  \item closure under inverse
\end{enumerate}
\end{definition}

\begin{theorem}[Left inverse involutivity]
\label{thm:group-left-inverse-involutive}
For a history-level group operation with associativity, left and right
identity, multiplication congruence, and left inverses, the left inverse
operation is involutive: \((x^{-1})^{-1}\hsame x\).
\end{theorem}
\begin{proof}
Rewrite \((x^{-1})^{-1}\) through the right identity. Replace the identity
by \(x^{-1}x\), reassociate, collapse
\((x^{-1})^{-1}x^{-1}\) by the left inverse law, and finish with the left
identity.
\end{proof}
\leanchecked{BEDC.Derived.GroupUp.group\_left\_inverse\_involutive}

\begin{theorem}[Right inverse involutivity]
\label{thm:group-right-inverse-involutive}
For a history-level group operation with associativity, left and right
identity, multiplication congruence, and right inverses, the right inverse
operation is involutive: $(x^{-1})^{-1}\hsame x$.
\end{theorem}
\begin{proof}
Rewrite $(x^{-1})^{-1}$ through the left identity. Replace the identity
by $xx^{-1}$, reassociate, collapse $x^{-1}(x^{-1})^{-1}$ by the right
inverse law, and finish with the right identity.
\end{proof}
\leanchecked{BEDC.Derived.GroupUp.group\_right\_inverse\_involutive}

\begin{theorem}[Left and right inverse witnesses agree]
\label{thm:group-left-right-inverse-uniqueness}
For a history-level group operation with associativity, left and right identity,
and multiplication congruence, if histories $y$ and $z$ satisfy
$$
\hsame(yx,e)
$$
and
$$
\hsame(xz,e),
$$
then $\hsame(y,z)$.
\end{theorem}
\begin{proof}
Chain $y$ through the right identity, replace the identity $e$ by $xz$,
reassociate to $(yx)z$, replace $yx$ by $e$, and finish with the left identity
on $z$. Each replacement is licensed by multiplication congruence and the
displayed inverse hypotheses.
\end{proof}
\leanchecked{BEDC.Derived.GroupUp.group\_left\_right\_inverse\_uniqueness}
\leanchecked{BEDC.Derived.GroupUp.group\_left\_right\_inverse\_unique}

\begin{theorem}[Group inverse congruence from laws]
\label{thm:group-inverse-congruence-from-laws}
For a history-level group operation with associativity, left and right identity,
multiplication congruence, and left and right inverses, \(\hsame(x,y)\) implies
\(\hsame(x^{-1},y^{-1})\).
\end{theorem}
\begin{proof}
Transport the left inverse witness for \(x^{-1}\) from \(x\) to \(y\) using
multiplication congruence. Combine it with the right inverse witness for
\(y^{-1}\) at \(y\), and apply uniqueness of left and right inverse witnesses.
\end{proof}
\leanchecked{BEDC.Derived.GroupUp.group\_inverse\_congruence\_from\_laws}

\begin{theorem}[Group left cancellation]
\label{thm:group-left-cancellation}
For a history-level group operation with associativity, left identity,
multiplication congruence, and left inverses, equality of \(ab\) and \(ac\)
under \(\hsame\) implies equality of \(b\) and \(c\) under \(\hsame\).
\end{theorem}
\begin{proof}
Chain \(b\) through the left identity, replace that identity by
\(a^{-1}a\), reassociate to \(a^{-1}(ab)\), use multiplication congruence on
the assumed product equality, reassociate back to \((a^{-1}a)c\), collapse the
left inverse, and finish by the left identity on \(c\).
\end{proof}
\leanchecked{BEDC.Derived.GroupUp.group\_left\_cancel}

\begin{theorem}[Group right cancellation]
\label{thm:group-right-cancellation}
For a history-level group operation with associativity, right identity,
multiplication congruence, and right inverses, equality of \(ba\) and \(ca\)
under \(\hsame\) implies equality of \(b\) and \(c\) under \(\hsame\).
\end{theorem}
\begin{proof}
Chain \(b\) through the right identity, replace that identity by
\(aa^{-1}\), reassociate to \((ba)a^{-1}\), use multiplication congruence on
the assumed product equality, reassociate back to \(c(aa^{-1})\), collapse the
right inverse, and finish by the right identity on \(c\).
\end{proof}
\leanchecked{BEDC.Derived.GroupUp.group\_right\_cancel}
