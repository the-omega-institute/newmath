\chapter{A Concrete Naming Certificate for \(\GroupUp\)}
\label{ch:concrete-instances-group-namecert}

The group interface \(\GroupUp\) extends \autoref{ch:concrete-instances-monoid-namecert} with a unary inverse operation \({}^{-1} : C \to C\) satisfying left and right inverse laws under the carrier's classifier. It is not a primitive structure; it is licensed only by composing the monoid certificate with the inverse field.

\begin{remark}[Roadmap chapter]
\label{rem:concrete-instances-group-namecert-roadmap}
This chapter records the licensing target. Lean markers identify the
algebraic fields and derived laws that are formalized for the certificate.
\end{remark}

\section{Carrier definition}

\begin{definition}[Group carrier]
\label{def:group-carrier}
Use a monoid setup extended with inverse operation \(a^{-1}\).
\end{definition}

\section{The certificate}

\begin{definition}[Group source specification]
\label{def:group-source-specification}
The source is a certified monoid carrier plus inverse data.
\end{definition}

\begin{definition}[Group pattern specification]
\label{def:group-pattern-specification}
The pattern is monoid expression formation extended by inverse formation.
\end{definition}

\begin{definition}[Group classifier specification]
\label{def:group-classifier-specification}
The classifier is the monoid carrier classifier, with inverse congruence.
\end{definition}

\begin{definition}[Group stability certificate]
\label{def:group-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item all monoid stability fields
  \item inverse congruence
  \item left inverse law
  \item right inverse law
  \item closure under inverse
\end{enumerate}
\end{definition}

\begin{theorem}[Group inverse of identity]
\label{thm:group-inverse-identity}
For a history-level group operation with right identity and left inverses, the
inverse of the identity is internally the same as the identity:
\(\hsame(e^{-1},e)\).
\end{theorem}
\begin{proof}
Chain \(e^{-1}\) backward through the right identity
\(e^{-1}\cdot e\hsame e^{-1}\), then apply the left inverse law at \(e\).
\end{proof}
\leanchecked{BEDC.Derived.GroupUp.group\_inverse\_identity}

\begin{theorem}[Left inverse involutivity]
\label{thm:group-left-inverse-involutive}
For a history-level group operation with associativity, left and right
identity, multiplication congruence, and left inverses, the left inverse
operation is involutive: \((x^{-1})^{-1}\hsame x\).
\end{theorem}
\begin{proof}
Rewrite \((x^{-1})^{-1}\) through the right identity. Replace the identity
by \(x^{-1}x\), reassociate, collapse
\((x^{-1})^{-1}x^{-1}\) by the left inverse law, and finish with the left
identity.
\end{proof}
\leanchecked{BEDC.Derived.GroupUp.group\_left\_inverse\_involutive}

\begin{theorem}[Right inverse involutivity]
\label{thm:group-right-inverse-involutive}
For a history-level group operation with associativity, left and right
identity, multiplication congruence, and right inverses, the right inverse
operation is involutive: $(x^{-1})^{-1}\hsame x$.
\end{theorem}
\begin{proof}
Rewrite $(x^{-1})^{-1}$ through the left identity. Replace the identity
by $xx^{-1}$, reassociate, collapse $x^{-1}(x^{-1})^{-1}$ by the right
inverse law, and finish with the right identity.
\end{proof}
\leanchecked{BEDC.Derived.GroupUp.group\_right\_inverse\_involutive}

\begin{theorem}[Left and right inverse witnesses agree]
\label{thm:group-left-right-inverse-uniqueness}
For a history-level group operation with associativity, left and right identity,
and multiplication congruence, if histories $y$ and $z$ satisfy
$$
\hsame(yx,e)
$$
and
$$
\hsame(xz,e),
$$
then $\hsame(y,z)$.
\end{theorem}
\begin{proof}
Chain $y$ through the right identity, replace the identity $e$ by $xz$,
reassociate to $(yx)z$, replace $yx$ by $e$, and finish with the left identity
on $z$. Each replacement is licensed by multiplication congruence and the
displayed inverse hypotheses.
\end{proof}
\leanchecked{BEDC.Derived.GroupUp.group\_left\_right\_inverse\_uniqueness}

\begin{theorem}[Group inverse congruence from laws]
\label{thm:group-inverse-congruence-from-laws}
For a history-level group operation with associativity, left and right identity,
multiplication congruence, and left and right inverses, \(\hsame(x,y)\) implies
\(\hsame(x^{-1},y^{-1})\).
\end{theorem}
\begin{proof}
Transport the left inverse witness for \(x^{-1}\) from \(x\) to \(y\) using
multiplication congruence. Combine it with the right inverse witness for
\(y^{-1}\) at \(y\), and apply uniqueness of left and right inverse witnesses.
\end{proof}
\leanchecked{BEDC.Derived.GroupUp.group\_inverse\_congruence\_from\_laws}

\begin{theorem}[Group inverse cancellation from laws]
\label{thm:group-inverse-cancel-from-laws}
For a history-level group operation with associativity, left and right
identity, multiplication congruence, and left and right inverses, equality of
inverses under $\hsame$ implies equality of the original histories:
$$
\hsame(x^{-1},y^{-1})\to\hsame(x,y).
$$
\end{theorem}
\begin{proof}
Apply the derived inverse congruence theorem to the displayed equality of
inverses, yielding $\hsame((x^{-1})^{-1},(y^{-1})^{-1})$. Chain the left
inverse involutivity law backward at $x$, across this double-inverse equality,
and forward at $y$.
\end{proof}
\leanchecked{BEDC.Derived.GroupUp.group\_inverse\_cancel\_from\_laws}

\begin{theorem}[Group multiplication by a right inverse cancels]
\label{thm:group-mul-right-inverse-cancel}
For a history-level group operation with associativity, right identity,
multiplication congruence, and right inverses, every pair of histories satisfies
$$
(ab)b^{-1}\hsame a.
$$
\end{theorem}
\begin{proof}
Reassociate $(ab)b^{-1}$ to $a(bb^{-1})$. The right inverse law collapses
$bb^{-1}$ to the identity, and the right identity law leaves $a$.
\end{proof}
\leanchecked{BEDC.Derived.GroupUp.group\_mul\_right\_inverse\_cancel}

\begin{theorem}[Group multiplication by a left inverse cancels]
\label{thm:group-mul-left-inverse-cancel}
For a history-level group operation with associativity, left identity,
multiplication congruence, and left inverses, every pair of histories satisfies
$$
a^{-1}(ab)\hsame b.
$$
\end{theorem}
\begin{proof}
Reassociate $a^{-1}(ab)$ back to $(a^{-1}a)b$. The left inverse law collapses
$a^{-1}a$ to the identity, and the left identity law leaves $b$.
\end{proof}
\leanchecked{BEDC.Derived.GroupUp.group\_mul\_left\_inverse\_cancel}

\begin{theorem}[Group inverse reverses multiplication]
\label{thm:group-inverse-mul-reverse}
For a history-level group operation with associativity, left and right identity,
multiplication congruence, and left and right inverses, the inverse of a product
is internally the same as the reversed product of inverses:
$$
\hsame((xy)^{-1},y^{-1}x^{-1}).
$$
\end{theorem}
\begin{proof}
The product $y^{-1}x^{-1}$ is a right inverse of $xy$: reassociate
$(xy)(y^{-1}x^{-1})$ to $x(y(y^{-1}x^{-1}))$, reassociate the inner product to
$(yy^{-1})x^{-1}$, collapse $yy^{-1}$ to the identity, and finish with the
left identity followed by the right inverse law for $x$. The left inverse law
for $xy$ and uniqueness of left and right inverse witnesses identify
$(xy)^{-1}$ with $y^{-1}x^{-1}$.
\end{proof}
\leanchecked{BEDC.Derived.GroupUp.group\_inverse\_mul\_reverse}

\begin{theorem}[Group left cancellation]
\label{thm:group-left-cancellation}
For a history-level group operation with associativity, left identity,
multiplication congruence, and left inverses, equality of \(ab\) and \(ac\)
under \(\hsame\) implies equality of \(b\) and \(c\) under \(\hsame\).
\end{theorem}
\begin{proof}
Chain \(b\) through the left identity, replace that identity by
\(a^{-1}a\), reassociate to \(a^{-1}(ab)\), use multiplication congruence on
the assumed product equality, reassociate back to \((a^{-1}a)c\), collapse the
left inverse, and finish by the left identity on \(c\).
\end{proof}
\leanchecked{BEDC.Derived.GroupUp.group\_left\_cancel}

\begin{theorem}[Group left-multiplication equation solution]
\label{thm:group-left-mul-equation-solution}
For a history-level group operation with associativity, left identity,
multiplication congruence, and left inverses, $ax\hsame b$ implies
$x\hsame a^{-1}b$.
\end{theorem}
\begin{proof}
Chain $x$ backward through the left identity $ex\hsame x$, replace $e$ by
$a^{-1}a$ using the left inverse law, reassociate to $a^{-1}(ax)$, and
transport the supplied equality $ax\hsame b$ through multiplication congruence.
\end{proof}

\end{theorem}
\begin{proof}

\begin{theorem}[Group multiplication equation exactness]
\label{thm:group-mul-equation-exactness}
For a history-level group operation with associativity, left and right
identities, multiplication congruence, and two-sided inverses, the displayed
left and right multiplication solutions classify the original equations:
$$
ax\hsame b \Longleftrightarrow x\hsame a^{-1}b
$$
and
$$
xa\hsame b \Longleftrightarrow x\hsame ba^{-1}.
$$
\end{theorem}
\begin{proof}
The forward implications are Theorem~\ref{thm:group-left-mul-equation-solution}
and Theorem~\ref{thm:group-right-mul-equation-solution}.

For the reverse direction of the left equation, transport
$x\hsame a^{-1}b$ by left multiplication with $a$. Reassociate
$a(a^{-1}b)$ to $(aa^{-1})b$, collapse $aa^{-1}$ by the right inverse law,
and use the left identity on $b$.

For the reverse direction of the right equation, transport
$x\hsame ba^{-1}$ by right multiplication with $a$. Reassociate
$(ba^{-1})a$ to $b(a^{-1}a)$, collapse $a^{-1}a$ by the left inverse law,
and use the right identity on $b$.
\end{proof}

\begin{theorem}[Group left absorption forces the right factor to the unit]
\label{thm:group-left-absorb-right-factor-unit}
For a history-level group operation with associativity, left and right
identities, multiplication congruence, and left inverses, if $ab\hsame a$,
then $b\hsame e$.
\end{theorem}
\begin{proof}
Chain $ab$ through the assumption to $a$, then backward through the right
identity $ae\hsame a$. Left cancellation of the common factor $a$ yields
$b\hsame e$.
\end{proof}

\begin{theorem}[Group two-sided cancellation]
\label{thm:group-two-sided-cancellation}
For a history-level group operation with associativity, left and right
identities, multiplication congruence, and two-sided inverses, equality of
$(ab)c$ and $(ad)c$ under $\hsame$ implies equality of $b$ and $d$ under
$\hsame$.
\end{theorem}
\begin{proof}
First apply right cancellation to the common final factor $c$, reducing the
displayed equality to $\hsame(ab,ad)$. Then apply left cancellation to the
common initial factor $a$, yielding $\hsame(b,d)$.
\end{proof}
