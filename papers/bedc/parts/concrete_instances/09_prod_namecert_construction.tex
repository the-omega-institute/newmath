\chapter{A Concrete Naming Certificate for \(\ProdUp\)}
\label{ch:concrete-instances-prod-namecert}

The product interface \(\ProdUp\) is licensed over two source carriers each endowed with a \(\NameCert\). The carrier is the pair constructor of source witnesses; the classifier is the conjunction of the component classifiers under their respective internal sameness judgments. It is not a primitive cartesian product; it is licensed only as a derived constructor over the two sources.

\section{Carrier definition}

\begin{definition}[Prod carrier]
\label{def:prod-carrier}
For licensed source carriers \(A\) and \(B\), define a pair carrier with constructor \(\mathsf{pair}(a,b)\).
\end{definition}
\leandef{BEDC.Derived.ProdUp.ProdCarrier}

\section{The certificate}

\begin{definition}[Prod source specification]
\label{def:prod-source-specification}
The source is the ordered product of the two source certificates.
\end{definition}
\leandef{BEDC.Derived.ProdUp.ProdHistoryCarrier}

\begin{theorem}[Product history carrier append introduction]
\label{thm:prod-history-carrier-append-intro}
If \(l\) lies in the left source carrier and \(r\) lies in the right source
carrier, then \(\mathsf{append}(l,r)\) lies in the product history carrier.
\end{theorem}
\begin{proof}
Use \(l\) and \(r\) as the two component witnesses. The BEDC continuation
judgment is witnessed by reflexivity of the defining append result, so the
product history carrier is rebuilt from the two source proofs and that
continuation witness.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdHistoryCarrier\_append\_intro}

\begin{theorem}[Product history carrier continuation introduction]
\label{thm:prod-history-carrier-cont-intro}
If \(l\) lies in the left source carrier, \(r\) lies in the right source
carrier, and \(\Cont(l,r,h)\), then \(h\) lies in the product history carrier.
\end{theorem}
\begin{proof}
Use \(l\) and \(r\) as the component witnesses. The supplied continuation
proof is exactly the result witness required by the product history carrier.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdHistoryCarrier\_cont\_intro}

\begin{theorem}[Product history carrier transport]
\label{thm:prod-history-carrier-hsame-transport}
If two product histories are internally the same and the first is generated from a left witness, a right witness, and a continuation proof, then the second is generated from the same component witnesses and the transported continuation proof.
\end{theorem}
\begin{proof}
Invert the product-history carrier into its left history, right history, component witnesses, and continuation proof. Transport the continuation result across \(\hsame\), and rebuild the product-history carrier with the same component witnesses.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdHistoryCarrier\_hsame\_transport}

\begin{theorem}[Product history carrier unary from components]
\label{thm:prod-history-carrier-unary-of-components}
If the left and right source predicates each determine unary histories, then
every product-history carrier generated from those predicates is unary.
\end{theorem}
\begin{proof}
Unpack the product-history carrier into component histories \(l\) and \(r\),
their source witnesses, and the BEDC continuation witness for the result.
The component assumptions give unary witnesses for \(l\) and \(r\), and BEDC
unary continuation closure gives unary membership of the product result.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdHistoryCarrier\_unary\_of\_components}

\begin{definition}[Prod pattern specification]
\label{def:prod-pattern-specification}
The pattern is paired generation from one \(A\)-witness and one \(B\)-witness.
\end{definition}

\begin{definition}[Prod classifier specification]
\label{def:prod-classifier-specification}
The classifier relates pairs componentwise using the classifiers of \(A\) and \(B\).
\end{definition}
\leandef{BEDC.Derived.ProdUp.ProdClassifier}
\leandef{BEDC.Derived.ProdUp.ProdHistoryClassifier}
\leanchecked{BEDC.Derived.ProdUp.prodClassifier\_iff}

\begin{theorem}[Product history classifier right transport]
\label{thm:prod-history-classifier-right-hsame-transport}
If a product-history classifier relates $h$ to $k$, and $k$ is internally the same as $k'$, then the classifier relates $h$ to $k'$.
\end{theorem}
\begin{proof}
Invert the product-history classifier into the left carrier, the right carrier, and the stored $\hsame(h,k)$ witness. Transport the right carrier across $\hsame(k,k')$, then compose $\hsame(h,k)$ with $\hsame(k,k')$ for the classifier witness.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdHistoryClassifier\_right\_hsame\_transport}

\begin{theorem}[Product history classifier endpoint transport]
\label{thm:prod-history-classifier-hsame-transport}
If a product-history classifier relates $h$ to $k$, and $h$ is internally the same as $h'$ while $k$ is internally the same as $k'$, then the classifier relates $h'$ to $k'$.
\end{theorem}
\begin{proof}
Invert the product-history classifier into endpoint carriers and the stored $\hsame(h,k)$ witness. Transport both endpoint carriers across their respective $\hsame$ witnesses, then compose symmetry of $\hsame(h,h')$, the stored link, and $\hsame(k,k')$ to obtain $\hsame(h',k')$.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdHistoryClassifier\_hsame\_transport}

\begin{theorem}[Product history classifier symmetry]
\label{thm:prod-history-classifier-symmetry}
If a product-history classifier relates \(h\) to \(k\), then it relates
\(k\) to \(h\).
\end{theorem}
\begin{proof}
Unpack the classifier into the two endpoint product carriers and the stored
\(\hsame(h,k)\) link. Exchange the endpoint carriers and reverse the stored
history-sameness link by symmetry.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdHistoryClassifier\_symm}

\begin{theorem}[Product history classifier left transport]
\label{thm:prod-history-classifier-left-hsame-transport}
If a product-history classifier relates $h$ to $k$, and $h$ is internally the
same as $h'$, then the classifier relates $h'$ to $k$.
\end{theorem}
\begin{proof}
Invert the classifier witness. Transport the left carrier through the
$\hsame(h,h')$ witness, keep the right carrier, and compose the symmetric
$\hsame(h',h)$ link with the stored $\hsame(h,k)$ classifier witness.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdHistoryClassifier\_left\_hsame\_transport}

\begin{theorem}[Product history classifier transitivity]
\label{thm:prod-history-classifier-transitivity}
For source-history predicates $L$ and $R$ on $\mathsf{BHist}$, if $\mathsf{ProdHistoryClassifier}(L,R,h,k)$ and $\mathsf{ProdHistoryClassifier}(L,R,k,u)$, then $\mathsf{ProdHistoryClassifier}(L,R,h,u)$.
\end{theorem}
\begin{proof}
Unpack both classifier witnesses. Keep the left carrier from the first witness and the right carrier from the second witness, then compose the stored $\hsame(h,k)$ and $\hsame(k,u)$ witnesses by transitivity to obtain the classifier link from $h$ to $u$.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdHistoryClassifier\_trans}

\begin{definition}[Prod stability certificate]
\label{def:prod-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item componentwise reflexivity
  \item componentwise transitivity
  \item closure under pair construction
  \item projection stability for first and second components
\end{enumerate}
\end{definition}

\begin{theorem}[History product classifier is symmetric]
\label{thm:prod-classifier-hsame-symm}
For pairs of finite-kernel histories, if the product classifier relates \(x\) to \(y\) componentwise by \(\hsame\), then it relates \(y\) to \(x\).
\end{theorem}
\begin{proof}
Invert the componentwise product classifier. Each component witness reverses by symmetry of \(\hsame\), and the reversed witnesses rebuild the product classifier in the opposite direction.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdClassifierSpec\_hsame\_symm}

\begin{definition}[Prod ledger policy]
\label{def:prod-ledger-policy}
Record a raw product history carrying both component ledgers together with an
\(\hsame\) link to the visible history exposed by the ledger.
\end{definition}
\leandef{BEDC.Derived.ProdUp.ProdHistoryLedgerPolicy}

\begin{theorem}[Prod ledger visible carrier]
\label{thm:prod-ledger-visible-carrier}
If a visible history is linked by \(\hsame\) to a raw product-ledger history,
then the visible history is itself a product carrier.
\end{theorem}
\begin{proof}
Unpack the raw product carrier and transport it across the stored \(\hsame\)
link by product-carrier transport.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdHistoryLedgerPolicy\_visible\_carrier}

\begin{theorem}[Prod ledger raw-visible classifier]
\label{thm:prod-ledger-raw-visible-classifier}
If $\mathsf{ProdHistoryLedgerPolicy}(L,R,\rho,v)$ holds, then
$\mathsf{ProdHistoryClassifier}(L,R,\rho,v)$ holds.
\end{theorem}
\begin{proof}
Unpack the ledger into the raw product carrier for $\rho$ and the stored
$\hsame(\rho,v)$ link. The visible-carrier theorem supplies the product
carrier for $v$. These three witnesses are exactly the product classifier
fields for the raw ledger history and the visible history.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdHistoryLedgerPolicy\_raw\_visible\_classifier}

\begin{theorem}[Concrete \(\ProdUp\) naming certificate]
\label{thm:concrete-prod-namecert}
For source-history predicates $L$ and $R$ on $\mathsf{BHist}$, assume there
are carrier witnesses $l$ and $r$ with $L(l)$ and $R(r)$. Equivalently, the
witnesses may be supplied by source certificates whose carrier-inhabited
fields provide them. Then the componentwise tuple forms the product semantic
naming certificate for $\mathsf{ProdHistoryCarrier}(L,R)$ and
$\mathsf{ProdHistoryClassifier}(L,R)$, hence a valid NameCert for
$\ProdUp(A,B)$.
\end{theorem}
\begin{proof}
Take the left and right witnesses from the two source certificates, or from
the explicit carrier-witness assumptions. BEDC continuation joins them into a
product history. The product classifier stores the two carrier witnesses and
an $\hsame$ witness between product histories, so the certificate fields follow
by $\hsame$ reflexivity, symmetry, transitivity, and carrier transport.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.prod\_history\_semantic\_name\_certificate}

\begin{corollary}[\(\ProdUp\) is licensed, not primitive]
\label{cor:prod-licensed-not-primitive}
\(\ProdUp\) is licensed as a derived binary constructor over two sources.
\end{corollary}
