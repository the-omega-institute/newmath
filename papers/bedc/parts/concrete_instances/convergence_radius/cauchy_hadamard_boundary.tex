\section{Cauchy--Hadamard Boundary}
\label{sec:convrad-cauchy-hadamard-boundary}

\begin{definition}[Cauchy--Hadamard exactness row]
\label{def:convrad-cauchy-hadamard-exactness-row}
For a coefficient family $a$ and a unary radius history $R$, let
$\mathsf{CHExact}(a,R)$ denote the analytic ledger row asserting that the
$\mathsf{ConvRad}(a,R)$ endpoint is exact for the Cauchy--Hadamard limsup
expression,
$$
R = 1/\mathsf{LimSup}_n |a_n|^{1/n}.
$$
This is the Cauchy--Hadamard item in \autoref{def:conv-rad-ledger-policy}. It
is a ledger-policy row, not one of the checked carrier, classifier, transport,
or closure rows of \autoref{thm:conv-rad-namecert}.
\end{definition}
\leandef{BEDC.Derived.ConvergenceRadiusUp.ConvRadCauchyHadamardExactnessRow}

\begin{definition}[Convergence-radius checked-row reduct]
\label{def:convrad-checked-row-reduct}
The checked-row reduct of the $\ConvergenceRadiusUp$ package is the finite row
set $\mathcal Q_{\mathrm{chk}}$ consisting of the following data:
\begin{enumerate}
  \item the carrier row $\mathsf{ConvRad}(a,R)$ and the $\hsame$ classifier on
  radius histories;
  \item radius transport, as in \autoref{lem:conv-rad-radius-transport};
  \item coefficient-family classifier transport, as in
  \autoref{lem:conv-rad-radius-coefficient-classifier-transport};
  \item extraction of power-series carrier data from geometric bounds and from
  convergence-radius witnesses, as in
  \autoref{lem:geom-bound-power-series-carrier} and
  \autoref{lem:conv-rad-power-series-carrier-witness};
  \item append-unary coefficient closure, as in
  \autoref{thm:conv-rad-append-unary-coeff-closed}.
\end{enumerate}
A policy interpretation over this reduct is a row assignment $\Pi$ with
witnesses for every row in $\mathcal Q_{\mathrm{chk}}$. The assignment is
constrained only on $\mathcal Q_{\mathrm{chk}}$; analytic ledger rows such as
$\mathsf{CHExact}(a,R)$ are not constrained unless the ledger policy of
\autoref{def:conv-rad-ledger-policy} is also supplied.
\end{definition}
\leandef{BEDC.Derived.ConvergenceRadiusUp.ConvRadCheckedRowReduct}

\begin{theorem}[Checked convergence-radius rows do not force Cauchy--Hadamard]
\label{thm:convrad-checked-rows-do-not-force-cauchy-hadamard}
Fix a coefficient family $a$ and a unary radius history $R$ for which the
checked-row reduct of \autoref{def:convrad-checked-row-reduct} has the
displayed carrier row $\mathsf{ConvRad}(a,R)$. In the
$\ConvergenceRadiusUp$ setup reduced to that checked row set, there are two
policy interpretations $\Pi_{+}$ and $\Pi_{-}$ over the same history carrier
and the same radius classifier such that
$$
\forall q\in\mathcal Q_{\mathrm{chk}},\quad \Pi_{+}(q)=\Pi_{-}(q),
$$
but
$$
\Pi_{+}(\mathsf{CHExact}(a,R))=\mathsf{accept}
\qquad\text{and}\qquad
\Pi_{-}(\mathsf{CHExact}(a,R))=\mathsf{reject}.
$$
Consequently, no finite derivation whose nonlogical inputs are only the
checked rows of \autoref{def:convrad-checked-row-reduct} can prove
$\mathsf{CHExact}(a,R)$.
\end{theorem}
\begin{proof}
By \autoref{def:typed-naming-certificate}, the naming-certificate template
separates source, pattern, classifier, stability, and ledger fields. The
concrete convergence-radius package in \autoref{thm:conv-rad-namecert} lists
the checked carrier, classifier, transport, witness, and append-unary closure
rows, and states that the analytic laws of \autoref{def:conv-rad-ledger-policy}
remain ledger-policy obligations.

Let $\chi_{a,R}$ be the row name represented by
$\mathsf{CHExact}(a,R)$. Since $\chi_{a,R}$ is not an element of
$\mathcal Q_{\mathrm{chk}}$, define $\Pi_{+}$ and $\Pi_{-}$ to use the same
witness for every checked row $q\in\mathcal Q_{\mathrm{chk}}$. On the
Cauchy--Hadamard row, set $\Pi_{+}(\chi_{a,R})=\mathsf{accept}$ and
$\Pi_{-}(\chi_{a,R})=\mathsf{reject}$; all other analytic row names may be
assigned identically in the two interpretations.

Every premise and conclusion of a checked row is read from
$\mathcal Q_{\mathrm{chk}}$, so the two interpretations validate exactly the
same checked carrier, classifier, transport, witness, and closure data. They
therefore share the full checked-row reduct. They disagree only on the analytic
ledger row $\mathsf{CHExact}(a,R)$, which the reduct does not mention.

If the checked rows alone forced $\mathsf{CHExact}(a,R)$, any two
interpretations that agree on all rows used by the derivation would agree on
that conclusion. Applying this to $\Pi_{+}$ and $\Pi_{-}$ gives the displayed
contradiction.
\end{proof}
\leanchecked{BEDC.Derived.ConvergenceRadiusUp.ConvRad\_checked\_rows\_do\_not\_force\_cauchy\_hadamard}
