\chapter{A Concrete Naming Certificate for $\RepresentationRingUp$}
\label{ch:concrete-instances-representationring-namecert}

The representation-ring interface $\RepresentationRingUp$ packages the Grothendieck ring of finite-dimensional representations of a group under direct sum and tensor product. It depends on $\NameCert_{\GroupUp}$ (\autoref{ch:concrete-instances-group-namecert}) and $\NameCert_{\RingUp}$ (\autoref{ch:concrete-instances-ring-namecert}).

\section{Carrier definition}
\label{sec:representationring-carrier}

\begin{definition}[RepresentationRingUp BHist carrier]\label{def:representationring-bhist-carrier}
The $\RepresentationRingUp$ carrier is the finite $\mathsf{BHist}$ packet
$(G,R,V,S,T,\chi,\rho,\lambda)$. The row $G$ is the consumed
$\NameCert_{\GroupUp}$ source from \autoref{ch:concrete-instances-group-namecert},
and $R$ is the consumed $\NameCert_{\RingUp}$ source from
\autoref{ch:concrete-instances-ring-namecert}. The row $V$ is the finite
representation-endpoint row: for each displayed endpoint $v$ it carries a
finite action ledger
$$
\alpha_g(v):\mathsf{BHist}
$$
for every carried group-history row $g$ projected from $G$, together with the
unit and product compatibility rows for that action. The row $S$ records the
direct-sum endpoint, and $T$ records the tensor-product endpoint. The row
$\chi$ is the ring-normalization classifier row that reads $S$ as addition and
$T$ as multiplication in the consumed ring source $R$. The row $\rho$ is the
$\mathsf{Pkg}$ provenance naming the two consumed dependencies and the visible
representation endpoints. The finite ledger $\lambda$ contains exactly the
$\Cont$ rows for the group action, direct sum, tensor product, classifier
normalization, and dependency projections.

No host Grothendieck quotient, host equality of representations, or standard
representation-ring bridge is part of this carrier. Equality inside the carrier
means only transport through the displayed $\mathsf{BHist}$, $\Cont$, $\hsame$,
$\mathsf{Pkg}$, and consumed $\NameCert$ rows.
\end{definition}

\begin{definition}[RepresentationRingUp classifier]\label{def:representationring-classifier}
For two $\RepresentationRingUp$ packets
$(G,R,V,S,T,\chi,\rho,\lambda)$ and
$(G',R',V',S',T',\chi',\rho',\lambda')$, the classifier requires
componentwise $\hsame$ transport of $G,R,V,S,T,\chi,\rho,\lambda$ to the
primed rows. The transported package must preserve the group-action rows
projected from $G$, the direct-sum row $S$, the tensor-product row $T$, and the
ring-normalization row $\chi$ projected through $R$.

Thus a comparison is accepted exactly when the same finite action ledger,
sum ledger, tensor ledger, ring ledger, dependency provenance, and
$\Cont$ ledger reappear after $\hsame$ transport. The classifier does not
identify host representation objects; it identifies only the BEDC rows visible
in \autoref{def:representationring-bhist-carrier}.
\end{definition}

\section{The certificate}
\label{sec:representationring-certificate}

\begin{theorem}[RepresentationRingUp group-action boundary]\label{thm:representationring-group-action-boundary}
Every representation action row exposed by $\RepresentationRingUp$ is projected
from the $\NameCert_{\GroupUp}$ dependency of
\autoref{ch:concrete-instances-group-namecert} together with the finite
$\mathsf{BHist}$ action ledger in
\autoref{def:representationring-bhist-carrier}. In particular, the certificate
has no independent source for group multiplication, unit action, or action
composition beyond $G$ and the corresponding $\Cont$ entries of $\lambda$.
\end{theorem}

\begin{proof}
Unpack the source, carrier, dependency, and ledger fields of the ten-field
roadmap template. The only group source named by the carrier is $G$, the
consumed $\NameCert_{\GroupUp}$ row from
\autoref{ch:concrete-instances-group-namecert}.

Project a displayed action row through $G$. The endpoint $V$ supplies the
acted-on history, and the ledger $\lambda$ supplies the finite $\Cont$ rows
recording unit action and product compatibility for the action endpoints.

It remains to rule out extra action sources. The carrier of
\autoref{def:representationring-bhist-carrier} contains no additional group
field, and the classifier of \autoref{def:representationring-classifier}
transports only the listed rows. Hence every visible action row is bounded by
$G$ and the BHist action ledger.
\end{proof}

\begin{theorem}[RepresentationRingUp sum-tensor ledger exactness]\label{thm:representationring-sum-tensor-ledger-exactness}
The direct-sum row $S$ and tensor-product row $T$ of
$\RepresentationRingUp$ are exactly the additive and multiplicative ledger
entries consumed from the $\NameCert_{\RingUp}$ dependency of
\autoref{ch:concrete-instances-ring-namecert}. The certificate exposes no
additional additive, multiplicative, quotient, or host-equality row for the
representation ring.
\end{theorem}

\begin{proof}
Read the operation fields of the ten-field roadmap template after the carrier
has been fixed. The only ring source named by the carrier is $R$, the consumed
$\NameCert_{\RingUp}$ row from
\autoref{ch:concrete-instances-ring-namecert}.

The direct-sum endpoint $S$ is therefore read as the additive row of $R$ after
transport through the finite representation endpoint row $V$. The tensor
endpoint $T$ is read as the multiplicative row of $R$ after the same
dependency and endpoint transport.

The normalization row $\chi$ records that these two rows are the additive and
multiplicative rows consumed by the ring dependency. Since the carrier has no
host quotient field and no independent operation field, ledger coverage and
soundness coincide exactly with $S$, $T$, $R$, $\chi$, and the relevant
$\Cont$ entries of $\lambda$.
\end{proof}

\begin{theorem}[RepresentationRingUp classifier stability]\label{thm:representationring-classifier-stability}
Componentwise $\hsame$ transport, together with the dependency classifiers of
$\NameCert_{\GroupUp}$ and $\NameCert_{\RingUp}$, preserves the
$\RepresentationRingUp$ carrier rows $G,R,V,S,T,\chi,\rho,\lambda$. The
transported packet again satisfies the group-action, direct-sum,
tensor-product, ring-normalization, provenance, and finite ledger clauses of
\autoref{def:representationring-bhist-carrier}.
\end{theorem}

\begin{proof}
Start with the classifier field of the ten-field roadmap template. The
definition of \autoref{def:representationring-classifier} requires $\hsame$
transport on each of the eight carrier rows.

For the group-action rows, apply the $\NameCert_{\GroupUp}$ classifier from
\autoref{ch:concrete-instances-group-namecert} to transport the source group
row and then transport the action endpoints through the corresponding entries
of $\lambda$.

For direct sum, tensor product, and ring normalization, apply the
$\NameCert_{\RingUp}$ classifier from
\autoref{ch:concrete-instances-ring-namecert}. The rows $S$, $T$, and $\chi$
therefore remain the additive, multiplicative, and normalization rows of the
transported ring source.

Finally transport $\rho$ and $\lambda$ componentwise. Repacking the transported
dependencies, operation rows, classifier row, provenance row, and ledger row
gives the primed carrier packet with no appeal to host equality.
\end{proof}

\begin{theorem}[RepresentationRingUp dependency boundary]\label{thm:representationring-dependency-boundary}
The dependency surface of $\RepresentationRingUp$ consists exactly of the
$\GroupUp$ source of \autoref{ch:concrete-instances-group-namecert}, the
$\RingUp$ source of \autoref{ch:concrete-instances-ring-namecert}, the carrier
and classifier rows of \autoref{def:representationring-bhist-carrier} and
\autoref{def:representationring-classifier}, the classifier-stability rows,
and the finite ledger and exactness rows of
\autoref{thm:representationring-group-action-boundary},
\autoref{thm:representationring-sum-tensor-ledger-exactness}, and
\autoref{thm:representationring-classifier-stability}. It does not claim a
standard Grothendieck-ring bridge.
\end{theorem}

\begin{proof}
Unpack the dependency field of the ten-field roadmap template. The
introduction names exactly two consumed certificate sources:
$\NameCert_{\GroupUp}$ and $\NameCert_{\RingUp}$, cited through
\autoref{ch:concrete-instances-group-namecert} and
\autoref{ch:concrete-instances-ring-namecert}.

The carrier and classifier fields contribute the eight-row packet and its
componentwise $\hsame$ transport. The ledger and exactness fields contribute
only the group-action rows, the direct-sum and tensor-product rows, the
ring-normalization row, the provenance row, and the finite $\Cont$ ledger.

Coverage projects any accepted $\RepresentationRingUp$ packet into exactly
those entries. Soundness repacks exactly those entries into the carrier and
classifier. Completeness follows because no standard Grothendieck quotient,
host representation equality, or bridge row occurs in the displayed package.
\end{proof}
