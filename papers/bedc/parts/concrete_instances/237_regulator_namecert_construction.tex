\chapter{A Concrete Naming Certificate for $\RegulatorUp$}
\label{ch:concrete-instances-regulator-namecert}
\origin{human}

The regulator interface $\RegulatorUp$ packages the determinant of logarithms of fundamental units (for number fields) and its Beilinson generalisation to higher K-theory. It depends on $\NameCert_{\DirichletUnitUp}$ (\autoref{ch:concrete-instances-dirichletunit-namecert}) and $\NameCert_{\NumFieldUp}$ (\autoref{ch:concrete-instances-numfield-namecert}).

\section{Carrier definition}
\label{sec:regulator-carrier}

\begin{definition}[Regulator root input packet]
\label{def:regulator-root-input-packet}
A $\RegulatorUp$ root input packet is a displayed $\mathsf{BHist}$ tuple
$$
\mathcal{R}=(\mathcal{D},\mathcal{N},u,\iota,\nu,\beta,\delta,\rho)
$$
whose source rows are the dependency certificates
$\mathcal{D}:\NameCert_{\DirichletUnitUp}$ and
$\mathcal{N}:\NameCert_{\NumFieldUp}$. The unit-facing rows $u$ and $\iota$ are
the visible unit and inverse/unit-product rows read from
$\mathcal{D}$; the unary row $\nu$ is the Dirichlet-unit rank witness; the row
$\beta$ is the finite basis and logarithmic ledger supplied by
$\mathcal{D}$; the row $\delta$ is the displayed determinant-layout ledger
linking $\nu$, $\beta$, and the $\NumFieldUp$ source rows; and $\rho$ is the
$\mathsf{Pkg}$ provenance tying the two dependency certificates to the
regulator packet.

The packet also carries finite $\Cont$ readbacks from $\mathcal{D}$ to the unit,
inverse, rank, and logarithmic ledger rows, from $\mathcal{N}$ to the
number-field carrier, embedding, singleton-basis, and ledger rows, and from
$\rho$ to every exported dependency slot. It contains no analytic regulator
value, no host logarithm map, no determinant evaluation in a host field, and no
standard bridge row.
\end{definition}
\leanchecked{BEDC.Derived.RegulatorUp.RegulatorTasteGate\_single\_carrier\_alignment}

\begin{theorem}[Regulator DirichletUnit input boundary]
\label{thm:regulator-dirichletunit-input-boundary}
Every accepted packet of \autoref{def:regulator-root-input-packet} projects to
exactly the $\DirichletUnitUp$ rows needed by a root regulator source: visible
unit rows, inverse/unit-product rows, the unary rank witness, the finite
basis/logarithmic ledger row, finite $\Cont$ readbacks, and the $\mathsf{Pkg}$
provenance attached to $\NameCert_{\DirichletUnitUp}$. It does not project a
determinant, analytic regulator value, host logarithm equality, or standard
bridge from the $\DirichletUnitUp$ dependency.
\end{theorem}
\leanchecked{BEDC.Derived.RegulatorUp.RegulatorRootInputPacket\_dirichletunit\_input\_boundary}
\begin{proof}
Project the source row $\mathcal{D}$ from
\autoref{def:regulator-root-input-packet}. The available unit, inverse, rank,
and logarithmic rows are precisely the rows named by
\autoref{thm:dirichletunit-regulator-consumer-boundary} and
\autoref{thm:dirichletunit-root-regulator-input-exhaustion}.

The packet adds no second Dirichlet-unit source. Its $\Cont$ readbacks point
from $\mathcal{D}$ to $u$, $\iota$, $\nu$, and $\beta$, and its provenance row
$\rho$ records those projections as the same package boundary.

Inspecting the packet fields gives the exclusion clause. Determinant layout is
recorded only as the separate regulator-side row $\delta$, while analytic
regulator values, host logarithm equality, and bridge rows are not fields of
$\mathcal{R}$.
\end{proof}

\begin{theorem}[Regulator NumField carrier boundary]
\label{thm:regulator-numfield-carrier-boundary}
Every accepted packet of \autoref{def:regulator-root-input-packet} projects to
the $\NumFieldUp$ source rows through $\mathcal{N}$: the RatUp history carrier,
the RatUp endpoint classifier, the identity finite-extension row, the
singleton-basis row, the scalar-action row, exact endpoint classification, and
the inherited ledger provenance of the public $\NameCert_{\NumFieldUp}$
surface. No ambient host number field, quotient of embeddings, or analytic
place map is exported by the regulator root packet.
\end{theorem}
\leanchecked{BEDC.Derived.RegulatorUp.RegulatorRootInputPacket\_numfield\_finite\_basis\_boundary}
\begin{proof}
Project the source row $\mathcal{N}$ from the regulator packet. It is a
$\NameCert_{\NumFieldUp}$ dependency row, so its public carrier and classifier
surface is read through
\autoref{thm:numfield-rat-reflexive-public-namecert-interface}.

The exactness and ledger clauses are the rows isolated by
\autoref{thm:numfield-reflexive-rational-exactness-package}. They give the
identity finite-extension row, singleton basis, scalar action, exact endpoint
classification, and ledger provenance available to a downstream packet.

Repack these projections with the $\Cont$ readbacks and $\Pkg$ provenance of
\autoref{def:regulator-root-input-packet}. Since no host number-field carrier,
embedding quotient, or analytic place map is a component of $\mathcal{R}$, none
is exported at the root boundary.
\end{proof}

\begin{theorem}[Regulator ledger exactness]
\label{thm:regulator-ledger-exactness}
The determinant and logarithm data visible in a
\autoref{def:regulator-root-input-packet} are finite displayed ledger rows:
the Dirichlet-unit logarithmic row $\beta$, the unary rank row $\nu$, the
number-field source ledger read from $\mathcal{N}$, and the determinant-layout
row $\delta$. These rows specify the shape of the regulator input matrix; they
do not assert an analytic regulator value or equality of host logarithms.
\end{theorem}
\leanchecked{BEDC.Derived.RegulatorUp.RegulatorLedgerSpine\_rows\_unary}
\leanchecked{BEDC.Derived.RegulatorUp.RegulatorRootInputPacket\_ledger\_exactness}
\begin{proof}
First project the Dirichlet-unit side by
\autoref{thm:regulator-dirichletunit-input-boundary}. This gives $\nu$ and
$\beta$ as finite $\mathsf{BHist}$ ledger rows, read through $\Cont$ and
package provenance.

Next project the number-field side by
\autoref{thm:regulator-numfield-carrier-boundary}. The source rows available to
the regulator are the displayed carrier, classifier, finite-extension,
singleton-basis, exactness, and ledger rows of $\NameCert_{\NumFieldUp}$.

The remaining determinant-facing row is $\delta$. By the packet definition it
records only a finite layout over $\nu$, $\beta$, and the source ledger of
$\mathcal{N}$. Repacking these three sources exhausts the regulator ledger, and
there is no field from which an analytic regulator value or host logarithm
equality could be read.
\end{proof}

\section{The certificate}
\label{sec:regulator-certificate}

\begin{theorem}[Regulator root NameCert threshold]
\label{thm:regulator-root-namecert-threshold}
The root $\NameCert_{\RegulatorUp}$ threshold is assembled from the packet of
\autoref{def:regulator-root-input-packet}: the
$\NameCert_{\DirichletUnitUp}$ source rows, the $\NameCert_{\NumFieldUp}$ source
rows, the visible unit and inverse rows, the unary rank witness, the finite
logarithmic ledger, the determinant-layout ledger, the finite $\Cont$
readbacks, the $\hsame$ transport rows on displayed packet entries, and the
$\mathsf{Pkg}$ provenance tying the two dependency certificates to the packet.
The threshold exports no analytic regulator value, no host determinant
evaluation, no host logarithm equality, and no standard bridge.
\end{theorem}
\leanchecked{BEDC.Derived.RegulatorUp.RegulatorRootInputPacket\_root\_namecert\_threshold}
\begin{proof}
Build the carrier source from \autoref{def:regulator-root-input-packet}. Its
entries provide the two dependency certificates, the Dirichlet-unit visible
rows, the number-field source rows, the determinant-layout row, and the finite
readback and provenance rows.

Use \autoref{thm:regulator-dirichletunit-input-boundary} for the exact
Dirichlet-unit projections and
\autoref{thm:regulator-numfield-carrier-boundary} for the exact number-field
projections. These two boundaries bind every dependency row to
$\mathsf{BHist}$, $\Cont$, $\hsame$, $\NameCert$, and $\mathsf{Pkg}$ data.

Finally use \autoref{thm:regulator-ledger-exactness}. It says that the
logarithm and determinant-facing entries are finite displayed ledger rows
rather than analytic values. Repacking the carrier, dependency, classifier,
ledger, readback, and provenance rows gives the stated root threshold, with no
additional host determinant, logarithm, or bridge row to export.
\end{proof}

\begin{theorem}[Regulator dependency-ledger factorization]
\label{thm:regulator-dependency-ledger-factorization}
Every public read of the root $\NameCert_{\RegulatorUp}$ threshold factors
through exactly three rows: the $\DirichletUnitUp$ projection of
\autoref{thm:regulator-dirichletunit-input-boundary}, the $\NumFieldUp$
projection of \autoref{thm:regulator-numfield-carrier-boundary}, and the
finite determinant/logarithm layout ledger of
\autoref{thm:regulator-ledger-exactness}. The factorization exports the
finite $\mathsf{BHist}$, $\Cont$, $\hsame$, $\NameCert$, and $\mathsf{Pkg}$
rows of those three sources, and it excludes analytic regulator values, host
determinant evaluation, host logarithm equality, and standard bridge rows.
\end{theorem}
\leanchecked{BEDC.Derived.RegulatorUp.RegulatorRootInputPacket\_dependency\_ledger\_factorization}
\begin{proof}
Project the packet fields from \autoref{def:regulator-root-input-packet}. The
root threshold reads the two dependency certificates, the visible unit-facing
entries, the number-field source entries, the determinant-layout entry, the
finite readbacks, and the package provenance.

Apply \autoref{thm:regulator-dirichletunit-input-boundary} and
\autoref{thm:regulator-numfield-carrier-boundary}. These two boundary
theorems identify the dependency-facing rows that can be read from the packet:
the $\DirichletUnitUp$ unit, inverse, rank, logarithmic, readback, and
provenance rows, and the $\NumFieldUp$ carrier, classifier,
finite-extension, singleton-basis, scalar-action, exactness, ledger, and
provenance rows.

Now apply \autoref{thm:regulator-ledger-exactness}. It isolates the remaining
determinant/logarithm information as the finite displayed layout over the
Dirichlet-unit rank and logarithmic rows and the number-field source ledger.

Repack the three projected sources as the public threshold surface. Since
\autoref{def:regulator-root-input-packet} has no field for an analytic
regulator value, host determinant evaluation, host logarithm equality, or
standard bridge, no public read of $\NameCert_{\RegulatorUp}$ can factor
through such a row.
\end{proof}

\begin{definition}[Regulator rank-zero displayed layout]
\label{def:regulator-rank-zero-displayed-layout}
Let
$$
\mathcal{R}=(\mathcal{D},\mathcal{N},u,\iota,\nu,\beta,\delta,\rho)
$$
be a $\RegulatorUp$ root input packet in the sense of
\autoref{def:regulator-root-input-packet}. The packet is rank-zero displayed
when the unary rank witness is the empty unary row,
$$
\nu\hsame\emp,
$$
and the determinant-layout row $\delta$ is read only as the finite displayed
layout indexed by positions licensed by $\nu$, the Dirichlet-unit ledger
$\beta$, and the public $\NumFieldUp$ source ledger of $\mathcal{N}$.
\end{definition}

\begin{theorem}[Regulator rank-zero layout degeneracy]
\label{thm:regulator-rank-zero-layout-degeneracy}
Let
$$
\mathcal{R}=(\mathcal{D},\mathcal{N},u,\iota,\nu,\beta,\delta,\rho)
$$
be an accepted $\RegulatorUp$ root input packet. If $\mathcal{R}$ is rank-zero
displayed in the sense of
\autoref{def:regulator-rank-zero-displayed-layout}, then the determinant-layout
ledger $\delta$ is the empty finite layout over the
$\DirichletUnitUp$ and $\NumFieldUp$ projections. Consequently the rank-zero
packet exports no analytic regulator value, no host determinant evaluation,
and no host logarithm equality.
\end{theorem}
\leanchecked{BEDC.Derived.RegulatorUp.RegulatorRootInputPacket\_rank\_zero\_layout\_degeneracy}
\begin{proof}
Project the accepted packet through
\autoref{def:regulator-root-input-packet}. Its determinant-facing entries are
only the unary rank witness $\nu$, the finite Dirichlet-unit basis and
logarithmic ledger $\beta$, the $\NumFieldUp$ source ledger read from
$\mathcal{N}$, and the displayed determinant-layout ledger $\delta$ linking
those rows.

Apply \autoref{thm:regulator-dirichletunit-input-boundary}. The
$\DirichletUnitUp$ side available to the regulator consists of visible unit
rows, inverse and unit-product rows, the unary rank witness, the finite
basis/logarithmic ledger, finite $\Cont$ readbacks, and package provenance. In
particular, the Dirichlet-unit dependency contributes no determinant row, no
analytic regulator value, and no host logarithm equality.

Apply \autoref{thm:regulator-numfield-carrier-boundary} to the source
$\mathcal{N}$. The $\NumFieldUp$ side contributes the RatUp reflexive carrier,
classifier, identity finite-extension row, singleton-basis row, scalar-action
row, exactness row, and ledger provenance. It contributes no ambient host
number field, quotient of embeddings, analytic place map, determinant
evaluation, or logarithm equality.

It remains to identify the determinant-layout row under the rank-zero
hypothesis. By \autoref{def:regulator-rank-zero-displayed-layout}, the rank
row satisfies $\nu\hsame\emp$, and $\delta$ is indexed only by positions
licensed by $\nu$ together with the displayed $\DirichletUnitUp$ ledger
$\beta$ and the displayed $\NumFieldUp$ source ledger. The empty unary row has
no positive basis position to license. Hence there is no row at which a
logarithmic column, determinant minor, or host evaluation entry can be placed.
The only finite determinant-layout compatible with the displayed index set is
therefore the empty layout over the two dependency projections.

Finally use \autoref{thm:regulator-ledger-exactness}. It says that all
regulator determinant and logarithm data visible at the packet boundary are
finite displayed ledger rows shaping the input matrix, not analytic values.
Since the rank-zero layout is empty and the packet definition contains no
field for an analytic regulator value, host determinant evaluation, host
logarithm equality, or bridge row, none of those rows can be exported from
$\mathcal{R}$.
\end{proof}
