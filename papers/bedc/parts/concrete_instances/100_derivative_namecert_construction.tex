\chapter{A Concrete Naming Certificate for $\DerivativeUp$}
\label{ch:concrete-instances-derivative-namecert}

The derivative interface $\DerivativeUp$ packages the constructive pointwise derivative of a function between metric carriers, with explicit modulus relating $\varepsilon$ to the difference-quotient tolerance. It is the real and general counterpart of $\NameCert_{\ComplexDiffUp}$ (\autoref{ch:concrete-instances-complex-diff-namecert}).

\section{Carrier definition}
\label{sec:derivative-carrier}

\begin{definition}[Derivative metric quotient]
\label{def:derivative-metric-quotient}
For histories $f,z,h,q,dist:\mathsf{BHist}$, the derivative metric quotient
carrier holds when $f$, $z$, $q$, $h$, and $dist$ are unary histories, $h$ is
complex-nonzero, $q$ is the continuation result of $f$ by the step $h$, and
$dist$ is the continuation result of $h$ by the quotient $q$:
$$
\mathsf{UnaryHistory}(f)\wedge
\mathsf{UnaryHistory}(z)\wedge
\mathsf{CplxNonZero}(h)\wedge
\mathsf{UnaryHistory}(q)\wedge
\mathsf{Cont}(f,h,q)\wedge
\mathsf{UnaryHistory}(h)\wedge
\mathsf{UnaryHistory}(dist)\wedge
\mathsf{Cont}(h,q,dist).
$$
\end{definition}
\leandef{BEDC.Derived.DerivativeUp.DerivativeMetricQuotient}

\begin{theorem}[Derivative metric quotient history transport]
\label{thm:derivative-metric-quotient-hsame-transport}
For histories $f,f',z,z',h,h',q,q',dist,dist':\mathsf{BHist}$, if each
source component is classifier-equal to its primed component, then a
derivative metric quotient for $(f,z,h,q,dist)$ transports to a derivative
metric quotient for $(f',z',h',q',dist')$. The transported package also reads
back the complex difference quotient, the metric distance witness, and the two
displayed continuation ledgers:
$$
\mathsf{CplxDiffQuot}(f',z',h',q'),\qquad
\mathsf{MetricDistanceWitness}(h',q',dist'),\qquad
\mathsf{Cont}(f',h',q'),\qquad
\mathsf{Cont}(h',q',dist').
$$
\end{theorem}
\begin{proof}
Destructure the derivative metric quotient into its unary carriers, the
complex nonzero step proof, and the two continuation ledgers. Transport unary
carriers along the five $\hsame$ assumptions. The nonzero proof for $h'$
composes a hypothetical $h'\hsame\emp$ with $h\hsame h'$. The difference
ledger transports along the function, step, and quotient equalities. The
metric ledger transports as the continuation field of the concrete metric
distance witness built from $h$, $q$, $dist$, and $\mathsf{Cont}(h,q,dist)$.
Reassembling these transported fields gives all stated components.
\end{proof}
\leanchecked{BEDC.Derived.DerivativeUp.DerivativeMetricQuotient\_hsame\_transport}

\begin{theorem}[Derivative metric quotient function prefix closure]
\label{thm:derivative-metric-quotient-function-prefix-closed}
For histories $p,f,z,h,q,dist:\mathsf{BHist}$, if $p$ is unary and
$\mathsf{DerivativeMetricQuotient}(f,z,h,q,dist)$ holds, then the function
and quotient histories may be prefixed together while the step and point
remain fixed:
$$
\mathsf{DerivativeMetricQuotient}
(\mathsf{append}(p,f),z,h,\mathsf{append}(p,q),
\mathsf{append}(h,\mathsf{append}(p,q))).
$$
\end{theorem}
\begin{proof}
Unpack the derivative metric quotient into unary carriers, the complex-nonzero
step proof, and the two continuation ledgers. Unary closure under append gives
the prefixed function and prefixed quotient carriers. The function ledger
$\Cont(f,h,q)$ is transported through the shared prefix by reassociating
$\mathsf{append}(p,\mathsf{append}(f,h))$ with
$\mathsf{append}(\mathsf{append}(p,f),h)$. The metric distance component is
then rebuilt canonically from the original unary step, the prefixed quotient,
and the defining continuation
$\Cont(h,\mathsf{append}(p,q),\mathsf{append}(h,\mathsf{append}(p,q)))$.
\end{proof}

\begin{theorem}[Derivative empty-function quotient step readback]
\label{thm:derivative-metric-quotient-empty-function-step-readback}
If $\mathsf{DerivativeMetricQuotient}(\emp,z,h,q,dist)$ holds, then the
quotient result reads back to the step, the metric ledger is $\Cont(h,q,dist)$,
and the displayed distance reads back to $\mathsf{append}(h,h)$:
$$
q\hsame h\land \Cont(h,q,dist)\land dist\hsame\mathsf{append}(h,h).
$$
\end{theorem}
\begin{proof}
The function continuation ledger is $\Cont(\emp,h,q)$, so left-unit readback
gives $q\hsame h$. The metric field is the displayed ledger $\Cont(h,q,dist)$,
hence $dist\hsame\mathsf{append}(h,q)$. Appending the recovered equality
$q\hsame h$ under the fixed left context $h$ identifies this with
$\mathsf{append}(h,h)$.
\end{proof}

\begin{theorem}[Derivative metric quotient distance result is nonempty]
\label{thm:derivative-metric-quotient-distance-result-nonempty}
For histories $f,z,h,q,dist:\mathsf{BHist}$, a derivative metric quotient
cannot have a metric-distance result classified as $\emp$.
\end{theorem}
\begin{proof}
Unpack the derivative metric quotient. Its metric ledger is a continuation
$\Cont(h,q,dist)$, while the step field says that $h$ is complex-nonzero.
Transport the continuation result along $dist\hsame\emp$. Empty-result
inversion for $\Cont$ forces $h\hsame\emp$, contradicting the complex-nonzero
step witness.
\end{proof}

\begin{theorem}[Derivative metric quotient result deterministic]
\label{thm:derivative-metric-quotient-result-deterministic}
For histories $f,z,h,q,q',dist,dist':\mathsf{BHist}$, two derivative metric
quotient witnesses with the same function, point, and step determine the same
quotient and displayed metric-distance result up to $\hsame$. The four
continuation ledgers
$$
\Cont(f,h,q),\qquad \Cont(f,h,q'),\qquad \Cont(h,q,dist),\qquad
\Cont(h,q',dist')
$$
are exposed together with $q\hsame q'$ and $dist\hsame dist'$.
\end{theorem}
\begin{proof}
Destructure the two derivative metric quotient witnesses. Determinism of
$\Cont(f,h,\cdot)$ identifies the two quotient histories. Transport the second
metric ledger along that quotient equality, then apply determinism of
$\Cont(h,q,\cdot)$ to identify the displayed distance histories.
\end{proof}

\begin{theorem}[Derivative metric quotient chain-product composition]
\label{thm:derivative-metric-quotient-chain-product-composition}
For histories $f,g,z,h_f,h_g,q_f,q_g,d_f,d_g:\mathsf{BHist}$, assume
$$
\mathsf{DerivativeMetricQuotient}(f,z,h_f,q_f,d_f),\qquad
\mathsf{DerivativeMetricQuotient}(g,q_f,h_g,q_g,d_g).
$$
Let $h_{fg},q_{fg},d_{fg},F_{fg}:\mathsf{BHist}$ be continuation witnesses with
$$
\Cont(h_f,h_g,h_{fg}),\qquad
\Cont(q_f,q_g,q_{fg}),\qquad
\Cont(d_f,d_g,d_{fg}),\qquad
\Cont(f,g,F_{fg}),
$$
and assume the displayed chain ledger
$$
\Cont(f,h_{fg},q_g).
$$
Then the product representative of the chain is a derivative metric quotient,
$$
\mathsf{DerivativeMetricQuotient}(F_{fg},z,h_{fg},q_{fg},d_{fg}),
$$
and its distance ledger is the continuation product of the two component
distance histories:
$$
\Cont(d_f,d_g,d_{fg}).
$$
The displayed chain ledger also factors through the inner quotient endpoint:
$$
\Cont(q_f,h_g,q_g)\land g\hsame q_f.
$$
\end{theorem}
\begin{proof}
Unpack the two derivative metric quotient witnesses using
\autoref{def:derivative-metric-quotient}. The first witness gives unary
carriers for $f,z,h_f,q_f,d_f$, the non-zero step witness for $h_f$, and the
continuation ledgers $\Cont(f,h_f,q_f)$ and $\Cont(h_f,q_f,d_f)$. The second
witness gives unary carriers for $g,q_f,h_g,q_g,d_g$, the non-zero step witness
for $h_g$, and the continuation ledgers $\Cont(g,h_g,q_g)$ and
$\Cont(h_g,q_g,d_g)$.

The supplied continuation ledgers give unary carriers for $F_{fg}$, $h_{fg}$,
$q_{fg}$, and $d_{fg}$. If $h_{fg}\hsame\emp$, continuation-empty inversion
applied to $\Cont(h_f,h_g,h_{fg})$ forces $h_f\hsame\emp$, contradicting the
non-zero step field of the first quotient. Thus $h_{fg}$ carries the required
complex non-zero witness.

The chain ledger factors through $q_f$ as follows. The supplied continuation
$\Cont(h_f,h_g,h_{fg})$, together with $\Cont(f,h_f,q_f)$ and the displayed
$\Cont(f,h_{fg},q_g)$, gives $\Cont(q_f,h_g,q_g)$ by
\autoref{thm:cont-composite-right-factor}. Comparing this ledger with the
second quotient ledger $\Cont(g,h_g,q_g)$ and cancelling the common right tail
$h_g$ gives $g\hsame q_f$.

It remains to assemble the two ledgers required by
$\mathsf{DerivativeMetricQuotient}(F_{fg},z,h_{fg},q_{fg},d_{fg})$. For the
quotient ledger, replace $q_f$ and $q_g$ by the results of
$\Cont(f,h_f,q_f)$ and $\Cont(g,h_g,q_g)$. After reassociation, the canonical
result of continuing $F_{fg}$ by $h_{fg}$ and the displayed quotient
representative $q_{fg}$ differ only by moving the unary factor $g$ past the
unary factor $h_f$. Unary continuation commutativity from
\autoref{thm:unary-commutativity-refined}, together with associativity from
\autoref{thm:continuation-associativity-up-to-hsame-proof-standing}, therefore
identifies the two results, and continuation result transport gives
$\Cont(F_{fg},h_{fg},q_{fg})$.

For the metric ledger, replace $d_f$ and $d_g$ by the results of
$\Cont(h_f,q_f,d_f)$ and $\Cont(h_g,q_g,d_g)$. The ledger
$\Cont(d_f,d_g,d_{fg})$ therefore classifies $d_{fg}$ with the product of the
two component metric results. The canonical metric result of continuing
$h_{fg}$ by $q_{fg}$ is obtained from the step product followed by the quotient
product. Reassociation and the same unary commutativity swap of $q_f$ with
$h_g$ identify these two histories, so continuation result transport gives
$\Cont(h_{fg},q_{fg},d_{fg})$. Repacking the unary carriers, the non-zero
composite step, and the two continuation ledgers gives the derivative metric
quotient. Finally, $\Cont(d_f,d_g,d_{fg})$ is the displayed distance product
ledger.
\end{proof}
\leanchecked{BEDC.Derived.DerivativeUp.DerivativeMetricQuotient\_chain\_product\_composition}

\begin{theorem}[Derivative empty-function quotient zero-headed step exclusion]
\label{thm:derivative-metric-quotient-empty-function-zero-step-absurd}
For histories $z,h,q,dist,s:\mathsf{BHist}$, if
$$
\mathsf{DerivativeMetricQuotient}(\emp,z,h,q,dist)
\qquad\text{and}\qquad
h\hsame\Ezero(s),
$$
then $\bot$.
\end{theorem}
\begin{proof}
Apply \autoref{thm:derivative-metric-quotient-empty-function-step-readback} to the derivative metric quotient. This gives $q\hsame h$. Composing with the displayed step classification gives $q\hsame\Ezero(s)$. The quotient carrier field in \autoref{def:derivative-metric-quotient} gives $\UnaryHistory(q)$, the unary-generation carrier of \autoref{def:unary-history-judgment}. By \autoref{lem:unary-transport}, this transports to $\UnaryHistory(\Ezero(s))$. This contradicts \autoref{lem:unary-no-zero-extension}.
\end{proof}
\leanchecked{BEDC.Derived.DerivativeUp.DerivativeMetricQuotient\_empty\_function\_zero\_step\_absurd}
