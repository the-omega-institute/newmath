\chapter{A Concrete Naming Certificate for \(\IntervalUp\)}
\label{ch:concrete-instances-interval-namecert}

The unit interval interface \(\IntervalUp\) is licensed over Bishop reals from \autoref{ch:concrete-instances-real-namecert} as the predicate "\(0 \le x\) and \(x \le 1\)" interpreted constructively (with explicit moduli for the inequalities). The classifier is real sameness restricted to interval witnesses. It is not a primitive geometric object; it is licensed only as a derived sub-carrier of \(\RealUp\).

\begin{remark}[Interval certificate scope]
\label{rem:concrete-instances-interval-namecert-roadmap}
This chapter records the interval licensing target as a bounded subcarrier
with classifier transport inherited from internal history sameness.
\end{remark}

\section{Carrier definition}

\begin{definition}[Interval carrier]
\label{def:interval-carrier}
Use real witnesses \(x:\RealUp\) paired with constructive proofs \(0\le x\) and \(x\le1\).
\end{definition}

\begin{theorem}[Interval empty boundary]
\label{thm:interval-empty-boundary-of-hsame}
If a history \(h\) is internally the same as \(\emp\), then \(h\) inhabits the interval carrier whose lower boundary is \(\hsame(\emp,-)\) and whose upper boundary is \(\hsame(-,\emp)\).
\end{theorem}
\leanchecked{BEDC.Derived.IntervalUp.IntervalCarrier\_empty\_boundary\_of\_hsame}
\begin{proof}
Transport the unary empty history across the symmetry of the given \(\hsame(h,\emp)\) witness. The lower boundary uses the symmetric witness, and the upper boundary uses the original witness.
\end{proof}

\begin{theorem}[Interval empty boundary hsame transport]
\label{thm:interval-empty-boundary-hsame-transport}
For the interval carrier whose lower boundary is \(\hsame(\emp,-)\) and whose upper boundary is \(\hsame(-,\emp)\), internal sameness transports carrier membership.
\end{theorem}
\begin{proof}
Unpack the interval carrier into unary membership and the two boundary witnesses. Unary membership transports along the given \(\hsame\) witness. The lower boundary composes \(\hsame(\emp,h)\) with \(\hsame(h,k)\), and the upper boundary composes the symmetric witness \(\hsame(k,h)\) with \(\hsame(h,\emp)\).
\end{proof}
\leanchecked{BEDC.Derived.IntervalUp.IntervalCarrier\_empty\_boundary\_hsame\_transport}

\begin{theorem}[Interval empty-boundary classifier]
\label{thm:interval-empty-boundary-classifier}
If \(h\) and \(k\) are both internally the same as \(\emp\), then they satisfy the interval classifier whose lower boundary is \(\hsame(\emp,-)\) and whose upper boundary is \(\hsame(-,\emp)\).
\end{theorem}
\leanchecked{BEDC.Derived.IntervalUp.IntervalClassifierSpec\_empty\_boundary\_of\_hsame}
\begin{proof}
Apply the empty-boundary carrier theorem to \(h\) and to \(k\). The classifier relation is \(\HSame(h,k)\), obtained by composing \(\HSame(h,\emp)\) with the symmetry of \(\HSame(k,\emp)\).
\end{proof}

\section{The certificate}

\begin{definition}[Interval source specification]
\label{def:interval-source-specification}
The source is the Bishop real source restricted to the closed unit interval predicate.
\end{definition}

\begin{definition}[Interval pattern specification]
\label{def:interval-pattern-specification}
The pattern is subcarrier formation by attaching lower and upper bound witnesses to a real.
\end{definition}

\begin{definition}[Interval classifier specification]
\label{def:interval-classifier-specification}
The classifier is \(\RealUp\) sameness restricted to interval witnesses.
\end{definition}

\begin{definition}[Interval stability certificate]
\label{def:interval-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item real classifier reflexivity and transitivity
  \item lower-bound preservation under real sameness
  \item upper-bound preservation under real sameness
  \item closure of endpoints \(0\) and \(1\)
  \item closure of interval operations explicitly proved to preserve bounds
\end{enumerate}
\end{definition}

\begin{definition}[Interval ledger policy]
\label{def:interval-ledger-policy}
Record the real ledger and both inequality moduli.
\end{definition}

\begin{theorem}[Concrete \(\IntervalUp\) naming certificate]
\label{thm:concrete-interval-namecert}
The bounded-real subcarrier tuple forms a valid NameCert for \(\IntervalUp\).
\end{theorem}
\leanchecked{BEDC.Derived.IntervalUp.interval\_name\_certificate}

\begin{corollary}[\(\IntervalUp\) is licensed, not primitive]
\label{cor:interval-licensed-not-primitive}
\(\IntervalUp\) is licensed as a subcarrier of \(\RealUp\).
\end{corollary}

\begin{remark}[Zero-axiom encoding note]
\label{rem:interval-zero-axiom-note}
Uses constructive inequality witnesses with moduli rather than classical order trichotomy.
\end{remark}
