\chapter{A Concrete Naming Certificate for \(\IntervalUp\)}
\label{ch:concrete-instances-interval-namecert}

The unit interval interface \(\IntervalUp\) is licensed over Bishop reals from \autoref{ch:concrete-instances-real-namecert} as the predicate "\(0 \le x\) and \(x \le 1\)" interpreted constructively (with explicit moduli for the inequalities). The classifier is real sameness restricted to interval witnesses. It is not a primitive geometric object; it is licensed only as a derived sub-carrier of \(\RealUp\).

\begin{remark}[Interval certificate scope]
\label{rem:concrete-instances-interval-namecert-roadmap}
This chapter records the interval licensing target as a bounded subcarrier
with classifier transport inherited from internal history sameness.
\end{remark}

\section{Carrier definition}

\begin{definition}[Interval carrier]
\label{def:interval-carrier}
Use real witnesses \(x:\RealUp\) paired with constructive proofs \(0\le x\) and \(x\le1\).
\end{definition}

\begin{theorem}[Interval empty boundary]
\label{thm:interval-empty-boundary-of-hsame}
If a history \(h\) is internally the same as \(\emp\), then \(h\) inhabits the interval carrier whose lower boundary is \(\hsame(\emp,-)\) and whose upper boundary is \(\hsame(-,\emp)\).
\end{theorem}
\leanchecked{BEDC.Derived.IntervalUp.IntervalCarrier\_empty\_boundary\_of\_hsame}
\begin{proof}
Transport the unary empty history across the symmetry of the given \(\hsame(h,\emp)\) witness. The lower boundary uses the symmetric witness, and the upper boundary uses the original witness.
\end{proof}

\begin{theorem}[Interval empty-boundary carrier iff endpoint]
\label{thm:interval-empty-boundary-carrier-iff-endpoint}
For the interval carrier whose lower boundary is \(\hsame(\emp,-)\) and whose
upper boundary is \(\hsame(-,\emp)\), carrier membership of \(h\) is equivalent
to the endpoint witness \(\hsame(h,\emp)\).
\end{theorem}
\begin{proof}
In the forward direction, eliminate the concrete interval-carrier conjunction
and project the upper-boundary witness. In the reverse direction, apply the
empty-boundary carrier constructor to the endpoint witness.
\end{proof}
\leanchecked{BEDC.Derived.IntervalUp.IntervalCarrier\_empty\_boundary\_iff\_hsame\_empty}

\begin{theorem}[Interval empty boundary hsame transport]
\label{thm:interval-empty-boundary-hsame-transport}
For the interval carrier whose lower boundary is \(\hsame(\emp,-)\) and whose upper boundary is \(\hsame(-,\emp)\), internal sameness transports carrier membership.
\end{theorem}
\begin{proof}
Unpack the interval carrier into unary membership and the two boundary witnesses. Unary membership transports along the given \(\hsame\) witness. The lower boundary composes \(\hsame(\emp,h)\) with \(\hsame(h,k)\), and the upper boundary composes the symmetric witness \(\hsame(k,h)\) with \(\hsame(h,\emp)\).
\end{proof}
\leanchecked{BEDC.Derived.IntervalUp.IntervalCarrier\_empty\_boundary\_hsame\_transport}

\begin{definition}[Interval empty-boundary ledger policy]
\label{def:interval-empty-boundary-ledger-policy}
The empty-boundary interval ledger records a raw endpoint in the concrete
carrier whose lower boundary is \(\hsame(\emp,-)\) and whose upper boundary is
\(\hsame(-,\emp)\), together with an internal history-sameness link from the
raw endpoint to the visible endpoint.
\end{definition}
\leandef{BEDC.Derived.IntervalUp.IntervalEmptyBoundaryLedgerPolicy}

\begin{theorem}[Interval empty-boundary ledger visible carrier]
\label{thm:interval-empty-boundary-ledger-visible-carrier}
If an empty-boundary interval ledger relates a raw endpoint to a visible
endpoint, then the visible endpoint remains in the same concrete
empty-boundary interval carrier.
\end{theorem}
\leanchecked{BEDC.Derived.IntervalUp.IntervalEmptyBoundaryLedgerPolicy\_visible\_carrier}
\begin{proof}
Unpack the ledger into raw carrier evidence and the raw-visible
\(\hsame\) link. The empty-boundary carrier transport theorem carries the raw
carrier evidence across that link to the visible endpoint.
\end{proof}

\begin{theorem}[Interval empty boundary carrier uniqueness]
\label{thm:interval-empty-boundary-carrier-uniqueness}
For the interval carrier whose lower boundary is \(\hsame(\emp,-)\) and whose
upper boundary is \(\hsame(-,\emp)\), any two carrier inhabitants are
internally the same history.
\end{theorem}
\leanchecked{BEDC.Derived.IntervalUp.IntervalCarrier\_empty\_boundary\_unique}
\begin{proof}
Unpack the two carrier witnesses and read their upper-bound witnesses
\(\hsame(h,\emp)\) and \(\hsame(k,\emp)\). Compose the first with the symmetry
of the second to obtain \(\hsame(h,k)\).
\end{proof}

\begin{theorem}[Interval empty-boundary classifier]
\label{thm:interval-empty-boundary-classifier}
If \(h\) and \(k\) are both internally the same as \(\emp\), then they satisfy the interval classifier whose lower boundary is \(\hsame(\emp,-)\) and whose upper boundary is \(\hsame(-,\emp)\).
\end{theorem}
\leanchecked{BEDC.Derived.IntervalUp.IntervalClassifierSpec\_empty\_boundary\_of\_hsame}
\begin{proof}
Apply the empty-boundary carrier theorem to \(h\) and to \(k\). The classifier relation is \(\HSame(h,k)\), obtained by composing \(\HSame(h,\emp)\) with the symmetry of \(\HSame(k,\emp)\).
\end{proof}

\begin{theorem}[Interval empty-boundary classifier characterization]
\label{thm:interval-empty-boundary-classifier-characterization}
For the interval classifier whose lower boundary is \(\hsame(\emp,-)\) and
whose upper boundary is \(\hsame(-,\emp)\), a pair \(h,k\) is classified
exactly when both endpoint histories are internally the same as \(\emp\).
\end{theorem}
\leanchecked{BEDC.Derived.IntervalUp.IntervalClassifierSpec\_empty\_boundary\_iff}
\begin{proof}
In the forward direction, eliminate the restricted classifier witness and read
the two upper-bound witnesses from the carrier data. In the reverse direction,
apply the empty-boundary classifier theorem to the two endpoint witnesses.
\end{proof}

\begin{theorem}[Interval empty-boundary classifier iff endpoints]
\label{thm:interval-empty-boundary-classifier-iff-endpoints}
For the interval classifier whose lower boundary is $\hsame(\emp,-)$ and whose upper boundary is $\hsame(-,\emp)$, a classifier witness between $h$ and $k$ is equivalent to endpoint witnesses $\hsame(h,\emp)$ and $\hsame(k,\emp)$.
\end{theorem}
\begin{proof}
In one direction, eliminate the restricted classifier witness and project the upper-boundary fields from the two interval carriers. In the other direction, apply the empty-boundary classifier constructor to the two endpoint witnesses.
\end{proof}
\leanchecked{BEDC.Derived.IntervalUp.IntervalClassifierSpec\_empty\_boundary\_iff\_endpoints}

\section{The certificate}

\begin{definition}[Interval source specification]
\label{def:interval-source-specification}
The source is the Bishop real source restricted to the closed unit interval predicate.
\end{definition}

\begin{definition}[Interval pattern specification]
\label{def:interval-pattern-specification}
The pattern is subcarrier formation by attaching lower and upper bound witnesses to a real.
\end{definition}

\begin{definition}[Interval classifier specification]
\label{def:interval-classifier-specification}
The classifier is \(\RealUp\) sameness restricted to interval witnesses.
\end{definition}

\begin{theorem}[Interval classifier transitivity]
\label{thm:interval-classifier-transitivity}
If interval classifier witnesses relate \(h\) to \(k\) and \(k\) to \(r\), then
they determine an interval classifier witness from \(h\) to \(r\).
\end{theorem}
\begin{proof}
Eliminate both restricted classifier witnesses. The first left carrier and the
second right carrier become the endpoint data for the composite classifier, and
the underlying history link is the transitive composite of the two \(\HSame\)
witnesses.
\end{proof}
\leanchecked{BEDC.Derived.IntervalUp.IntervalClassifierSpec\_trans}

\begin{theorem}[Interval classifier symmetry]
\label{thm:interval-classifier-symmetry}
If interval classifier witnesses relate \(h\) to \(k\), then they determine an
interval classifier witness from \(k\) to \(h\).
\end{theorem}
\begin{proof}
Eliminate the restricted classifier witness. The endpoint interval-carrier
witnesses are exchanged, and the stored history-sameness link is reversed by
symmetry of \(\hsame\).
\end{proof}
\leanchecked{BEDC.Derived.IntervalUp.IntervalClassifierSpec\_symm}

\begin{definition}[Interval stability certificate]
\label{def:interval-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item real classifier reflexivity and transitivity
  \item lower-bound preservation under real sameness
  \item upper-bound preservation under real sameness
  \item closure of endpoints \(0\) and \(1\)
  \item closure of interval operations explicitly proved to preserve bounds
\end{enumerate}
\end{definition}

\begin{definition}[Interval ledger policy]
\label{def:interval-ledger-policy}
Record the real ledger and both inequality moduli.
\end{definition}

\begin{definition}[\(\IntervalUp\) certificate obligations]
\label{def:interval-certificate-obligations}
The carrier, source, pattern, classifier, stability, and ledger data listed above are the obligations required for the interface name \(\IntervalUp\).
\end{definition}

\begin{theorem}[Concrete \(\IntervalUp\) naming certificate]
\label{thm:concrete-interval-namecert}
The bounded-real subcarrier tuple forms a valid NameCert for \(\IntervalUp\).
\end{theorem}
\leanchecked{BEDC.Derived.IntervalUp.interval\_name\_certificate}

\paragraph{Primitive boundary.}
\(\IntervalUp\) is not a BEDC primitive. It is admitted only when those obligations are supplied.

\end{remark}
