\chapter{A Concrete Naming Certificate for $\IdealUp$}
\label{ch:concrete-instances-ideal-namecert}

The ideal interface $\IdealUp$ packages a sub-rng of a ring carrier closed under multiplication by every ambient ring element. It depends on $\NameCert_{\RingUp}$ (\autoref{ch:concrete-instances-ring-namecert}) and supplies the kernel object for the quotient-ring construction $\NameCert_{\QuotientRingUp}$ (\autoref{ch:concrete-instances-quotientring-namecert}).

\section{Carrier definition}
\label{sec:ideal-carrier}

\begin{definition}[Ideal carried predicate rows]
\label{def:ideal-carried-predicate-rows}
Let $R$ carry a $\RingUp$ certificate over BEDC histories, with $\NameCert_{\RingUp}$ carrier predicate $C:\mathsf{BHist}\to\mathsf{Prop}$, classifier $\sim_R$, zero $0_R$, addition $+_R$, additive inverse $-_R$, and multiplication $\cdot_R$. An $\IdealUp$ carried predicate over $R$ is a predicate
$$
I:\mathsf{BHist}\to\mathsf{Prop}
$$
together with the following certificate rows:
\begin{enumerate}
\item carrier support: $I(x)\Longrightarrow C(x)$;
\item zero membership: $I(0_R)$;
\item additive closure: $I(x)\land I(y)\Longrightarrow I(x+_R y)$;
\item additive inverse closure: $I(x)\Longrightarrow I(-_R x)$;
\item internal multiplication closure: $I(x)\land I(y)\Longrightarrow I(x\cdot_R y)$;
\item classifier transport: $I(x)\land x\sim_R y\Longrightarrow I(y)$;
\item left and right ambient absorption: for every carried ambient endpoint $r$ with $C(r)$,
$$
I(x)\Longrightarrow I(r\cdot_R x)\land I(x\cdot_R r).
$$
\end{enumerate}
\end{definition}

\section{The certificate}
\label{sec:ideal-certificate}

\begin{definition}[Ideal obligation inventory]
\label{def:ideal-obligation-inventory}
An $\IdealUp$ certificate over a fixed $\RingUp$ certificate is the ten-field
$\NameCert_{\IdealUp}$ inventory whose rows are fixed as follows.
\begin{enumerate}
\item The traditional target is a two-sided ideal of a ring.
\item The BEDC source is $\RingUp$ over $\mathsf{BHist}$, with the ambient
carrier, classifier, addition, additive inverse, zero, and multiplication
already named by the ring certificate.
\item The carrier is a predicate $I:\mathsf{BHist}\to\mathsf{Prop}$ supported
by the ambient ring carrier $C$.
\item The classifier is inherited from the ambient ring classifier $\sim_R$.
\item The stability rows are zero membership, additive closure, additive
inverse closure, internal multiplication closure, classifier transport, and
two-sided absorption by every carried ambient ring endpoint.
\item The ledger records hidden membership witnesses for $I(x)$ and the
ambient absorber witnesses $C(r)$ consumed by the two absorption rows.
\item The core theorem list consists of
\autoref{thm:ideal-carrier-support-row-obligation},
\autoref{thm:ideal-zero-add-neg-obligation},
\autoref{thm:ideal-classifier-absorption-obligation},
\autoref{thm:ideal-intersection-closure},
\autoref{thm:finite-ideal-intersection-closure}, and
\autoref{thm:ring-map-ideal-preimage-ideal-closure}.
\item Exactness is row exhaustion: the certificate consumes exactly the carrier
support, zero, additive, inverse, internal multiplication, classifier
transport, and two-sided absorption rows displayed here.
\item The standard bridge row is absent at this paper site.
\item The boundary package is the scoped $\IdealUp$ obligation surface used by
quotient-ring, ideal-class, intersection, finite-meet, and preimage
constructions.
\end{enumerate}
\end{definition}

\begin{theorem}[Ideal carrier support obligation]
\label{thm:ideal-carrier-support-row-obligation}
Let $R$ carry a $\RingUp$ certificate over $\mathsf{BHist}$ with ambient carrier
$C$ and classifier $\sim_R$. Let $I:\mathsf{BHist}\to\mathsf{Prop}$ satisfy the
inventory of \autoref{def:ideal-obligation-inventory}. Then the inventory
supplies the carrier-support row
$$
I(x)\Longrightarrow C(x).
$$
Consequently every endpoint admitted by the ideal predicate is a carried
$\RingUp$ endpoint.
\end{theorem}
\begin{proof}
Project the carrier row of \autoref{def:ideal-obligation-inventory}. That row
states that $I$ is supported by the ambient $\RingUp$ carrier $C$.

Applying this projection to a membership witness $I(x)$ gives $C(x)$. The
endpoint $x$ therefore lies in the same $\mathsf{BHist}$ carrier scope as the
ambient ring operations and classifier $\sim_R$.
\end{proof}

\begin{theorem}[Ideal zero and additive obligations]
\label{thm:ideal-zero-add-neg-obligation}
Let $R$ carry a $\RingUp$ certificate over $\mathsf{BHist}$ with addition
$+_R$, additive inverse $-_R$, zero $0_R$, multiplication $\cdot_R$, ambient
carrier $C$, and classifier $\sim_R$. Let
$I:\mathsf{BHist}\to\mathsf{Prop}$ satisfy the inventory of
\autoref{def:ideal-obligation-inventory}. Then the inventory supplies
$$
I(0_R),
$$
$$
I(x)\land I(y)\Longrightarrow I(x+_R y),
$$
$$
I(x)\Longrightarrow I(-_R x),
$$
and
$$
I(x)\land I(y)\Longrightarrow I(x\cdot_R y).
$$
\end{theorem}
\begin{proof}
Project the stability field of \autoref{def:ideal-obligation-inventory}. Its
zero row is exactly $I(0_R)$, using the zero endpoint named by the surrounding
$\RingUp$ certificate.

The same stability field contains the additive closure and additive inverse
rows. Projecting those rows gives closure under $+_R$ and $-_R$ for every pair
or endpoint already carrying an $I$-membership witness.

Finally project the internal multiplication row from the same inventory. Since
the multiplication symbol $\cdot_R$ is the one named by the ambient $\RingUp$
certificate, the resulting endpoint remains in the ring carrier scope supplied
by the carrier-support row.
\end{proof}

\begin{theorem}[Ideal classifier and absorption obligations]
\label{thm:ideal-classifier-absorption-obligation}
Let $R$ carry a $\RingUp$ certificate over $\mathsf{BHist}$ with carrier $C$,
classifier $\sim_R$, and multiplication $\cdot_R$. Let
$I:\mathsf{BHist}\to\mathsf{Prop}$ satisfy the inventory of
\autoref{def:ideal-obligation-inventory}. Then the inventory supplies
classifier transport
$$
I(x)\land x\sim_R y\Longrightarrow I(y)
$$
and two-sided absorption by every carried ambient ring endpoint:
$$
C(r)\land I(x)\Longrightarrow I(r\cdot_R x)\land I(x\cdot_R r).
$$
\end{theorem}
\begin{proof}
Project the classifier row and the classifier-transport stability row of
\autoref{def:ideal-obligation-inventory}. The classifier row identifies the
ideal comparison relation with the ambient $\RingUp$ classifier $\sim_R$, and
the transport row carries $I$-membership along that comparison.

Project the absorption rows from the same stability field. Given an ambient
absorber witness $C(r)$ and an ideal membership witness $I(x)$, the left row
gives $I(r\cdot_R x)$ and the right row gives $I(x\cdot_R r)$.

The ledger row records exactly the membership witness for $x$ and the carried
ambient witness for $r$ consumed in these projections, so no extra absorber or
classifier datum is introduced outside the surrounding $\RingUp$ scope.
\end{proof}

\begin{definition}[Ambient carrier ideal predicate]
\label{def:ideal-ambient-carrier-predicate}
Let $R$ carry a $\RingUp$ certificate over $\mathsf{BHist}$ with ambient carrier
$C$, classifier $\sim_R$, zero $0_R$, addition $+_R$, additive inverse $-_R$,
and multiplication $\cdot_R$. The ambient carrier ideal predicate is
$$
W_R(x)\Longleftrightarrow C(x).
$$
Its carried-predicate certificate rows are the following ten-field package:
traditional target the whole ring ideal, BEDC source the surrounding $\RingUp$
certificate over $\mathsf{BHist}$, carrier $C$, classifier $\sim_R$, stability
from $\RingUp$ carrier transport and operation closure, ledger empty except for
the ambient carrier witnesses consumed by the rows, exactness by exhaustion of
the displayed row inventory, standard bridge absent, and status proof target.
\end{definition}

\begin{theorem}[Ambient carrier IdealUp closure]
\label{thm:ideal-ambient-carrier-ideal-closure}
Let $R$ carry a $\RingUp$ certificate over $\mathsf{BHist}$ with notation as in
\autoref{def:ideal-ambient-carrier-predicate}. Then $W_R$ is an $\IdealUp$
predicate over $R$: it has carrier support, zero membership, additive closure,
additive inverse closure, internal multiplication closure, classifier
transport, and two-sided absorption by every carried ambient ring endpoint.
\end{theorem}
\begin{proof}
Unfold \autoref{def:ideal-ambient-carrier-predicate}. Carrier support is
identity: a witness $W_R(x)$ is exactly a witness $C(x)$.

Zero membership is the zero-carrier row of the surrounding $\RingUp$
certificate. Additive closure, additive inverse closure, and internal
multiplication closure are exactly the carrier-closure rows for $+_R$, $-_R$,
and $\cdot_R$ after replacing $W_R$ by $C$.

Classifier transport is the carrier-transport row of the same $\RingUp$
certificate along $\sim_R$. If $W_R(x)$ and $x\sim_R y$, transport gives
$C(y)$, hence $W_R(y)$.

For absorption, assume $W_R(x)$ and $C(r)$. The multiplication-closure row of
the ring certificate gives both $C(r\cdot_R x)$ and $C(x\cdot_R r)$. By the
definition of $W_R$, these are $W_R(r\cdot_R x)$ and $W_R(x\cdot_R r)$.

These rows are precisely the $\IdealUp$ carried-predicate inventory over the
fixed $\RingUp$ certificate.
\end{proof}
\leanchecked{BEDC.Derived.IdealUp.IdealAmbientCarrier\_closure\_rows}

\begin{theorem}[Ambient ideal quotient-kernel carrier exactness]
\label{thm:ideal-ambient-quotient-kernel-carrier-exactness}
Let $R$ carry a $\RingUp$ certificate over $\mathsf{BHist}$ with notation as in
\autoref{def:ideal-ambient-carrier-predicate}. Let $Q_{W_R}$ be the
quotient-kernel source relation of
\autoref{def:ideal-quotient-kernel-source-package} formed from the ambient
carrier ideal predicate $W_R$. Then
$$
Q_{W_R}(x,y)\Longleftrightarrow C(x)\land C(y).
$$
\end{theorem}
\leanchecked{BEDC.Derived.IdealUp.IdealAmbientCarrier\_quotient\_kernel\_exactness}
\begin{proof}
Unfold \autoref{def:ideal-quotient-kernel-source-package} and
\autoref{def:ideal-ambient-carrier-predicate}.

For the forward direction, a witness $Q_{W_R}(x,y)$ consists of endpoint
carrier witnesses $C(x)$ and $C(y)$ together with the membership witness
$W_R(x-_R y)$. Projecting the first two components gives $C(x)\land C(y)$.

For the reverse direction, assume $C(x)$ and $C(y)$. The additive
inverse-closure row of the $\RingUp$ certificate gives $C(-_R y)$, and the
additive closure row gives
$$
C(x-_R y).
$$
By the definition of $W_R$, this is $W_R(x-_R y)$. Pairing it with the original
endpoint carrier witnesses gives $Q_{W_R}(x,y)$.
\end{proof}
\leanchecked{BEDC.Derived.IdealUp.IdealAmbientCarrier\_quotient\_kernel\_carrier\_exactness}
\leanchecked{BEDC.Derived.IdealUp.IdealAmbientQuotientKernel\_carrier\_exactness}

\begin{theorem}[Ideal zero-membership obligation]
\label{thm:ideal-zero-membership-obligation}
Let $R$ carry a $\RingUp$ certificate over $\mathsf{BHist}$, with $\NameCert_{\RingUp}$ carrier $C$, classifier $\sim_R$, zero $0_R$, addition $+_R$, additive inverse $-_R$, and multiplication $\cdot_R$. If $I:\mathsf{BHist}\to\mathsf{Prop}$ is an $\IdealUp$ carried predicate over $R$ in the sense of \autoref{def:ideal-carried-predicate-rows}, then the certificate contains the zero-membership row $I(0_R)$.
\end{theorem}
\begin{proof}
Unpack the $\IdealUp$ carried-predicate data from \autoref{def:ideal-carried-predicate-rows}. The underlying predicate is explicitly a predicate on $\mathsf{BHist}$, while the zero endpoint $0_R$ is the zero row of the ambient $\RingUp$ certificate.

Among the rows carried by the ideal certificate is the zero-membership row. Projecting that row gives exactly $I(0_R)$.
\end{proof}

\begin{theorem}[Ideal classifier-transport obligation]
\label{thm:ideal-classifier-transport-obligation}
Let $R$ carry a $\RingUp$ certificate over $\mathsf{BHist}$, with $\NameCert_{\RingUp}$ carrier $C$, classifier $\sim_R$, zero $0_R$, addition $+_R$, additive inverse $-_R$, and multiplication $\cdot_R$. If $I:\mathsf{BHist}\to\mathsf{Prop}$ is an $\IdealUp$ carried predicate over $R$ in the sense of \autoref{def:ideal-carried-predicate-rows}, then its certificate contains the classifier-transport row
$$
I(x)\land x\sim_R y\Longrightarrow I(y).
$$
\end{theorem}
\begin{proof}
Unpack the carried predicate rows. The endpoints $x$ and $y$ are BEDC histories, the relation $x\sim_R y$ is the classifier supplied by the ambient $\NameCert_{\RingUp}$, and $I$ is the $\mathsf{BHist}$ predicate being certified.

The classifier-transport row listed in \autoref{def:ideal-carried-predicate-rows} is one of the fields of the $\IdealUp$ certificate. Projecting it and applying it to a membership witness $I(x)$ and the classifier comparison $x\sim_R y$ yields $I(y)$.
\end{proof}

\begin{theorem}[Ideal two-sided absorption obligation]
\label{thm:ideal-two-sided-absorption-obligation}
Let $R$ carry a $\RingUp$ certificate over $\mathsf{BHist}$, with $\NameCert_{\RingUp}$ carrier $C$, classifier $\sim_R$, zero $0_R$, addition $+_R$, additive inverse $-_R$, and multiplication $\cdot_R$. If $I:\mathsf{BHist}\to\mathsf{Prop}$ is an $\IdealUp$ carried predicate over $R$ in the sense of \autoref{def:ideal-carried-predicate-rows}, then its certificate contains the left and right ambient absorption rows: for every carried ambient endpoint $r$ with $C(r)$,
$$
I(x)\Longrightarrow I(r\cdot_R x)\land I(x\cdot_R r).
$$
\end{theorem}
\begin{proof}
Unpack the $\IdealUp$ certificate rows from \autoref{def:ideal-carried-predicate-rows}. The ambient multiplier $r$ is not required to belong to the ideal predicate; it is only required to be carried by the $\RingUp$ carrier $C$ from the underlying $\NameCert_{\RingUp}$.

The two-sided absorption row is part of the listed certificate surface. Projecting it at the carried endpoint $r$ and a membership witness $I(x)$ gives the pair of conclusions $I(r\cdot_R x)$ and $I(x\cdot_R r)$.
\end{proof}

\begin{definition}[Ideal intersection predicate]
\label{def:ideal-intersection-predicate}
Let $R$ carry a $\RingUp$ certificate with carrier $C$, and let $I$ and $J$ be $\IdealUp$ carried predicates over $R$ in the sense of \autoref{sec:ideal-carrier}. The intersection predicate $I\cap_RJ$ is defined by
$$
(I\cap_RJ)(x)\Longleftrightarrow I(x)\land J(x).
$$
\end{definition}

\begin{theorem}[Ideal intersection closure]
\label{thm:ideal-intersection-closure}
Let $R$ carry a $\RingUp$ certificate as in \autoref{ch:concrete-instances-ring-namecert}, with carrier $C$, classifier $\sim_R$, addition $+_R$, additive inverse $-_R$, zero $0_R$, and multiplication $\cdot_R$. Let $I$ and $J$ be $\IdealUp$ predicates over $R$. Assume the two ideal certificates provide the rows specified in \autoref{sec:ideal-certificate}: carrier support, zero membership, additive closure, additive inverse closure, internal multiplication closure, classifier transport, and left and right absorption by every carried ring endpoint. For $K:=I\cap_RJ$ as in \autoref{def:ideal-intersection-predicate}, $K$ is an $\IdealUp$ predicate over $R$. Explicitly, $K$ has carrier support and the rows
$$
K(0_R),\qquad
K(x)\land K(y)\Longrightarrow K(x+_R y),\qquad
K(x)\Longrightarrow K(-_R x),
$$
$$
K(x)\land K(y)\Longrightarrow K(x\cdot_R y),\qquad
K(x)\land x\sim_R y\Longrightarrow K(y),
$$
and, for every $r$ with $C(r)$,
$$
K(x)\Longrightarrow K(r\cdot_R x)\land K(x\cdot_R r).
$$
\end{theorem}
\leanchecked{BEDC.Derived.IdealUp.IdealIntersection\_closure\_rows}
\begin{proof}
The carrier support row is immediate. If $K(x)$, then $I(x)$ and $J(x)$; the carrier-support row of either ideal gives $C(x)$.

For zero, the zero rows for $I$ and $J$ give $I(0_R)$ and $J(0_R)$, hence $K(0_R)$.

For additive closure, assume $K(x)$ and $K(y)$. Then $I(x)$, $I(y)$, $J(x)$, and $J(y)$. The additive-closure row for $I$ gives $I(x+_R y)$, and the additive-closure row for $J$ gives $J(x+_R y)$. Together these are $K(x+_R y)$.

For additive inverse closure, assume $K(x)$. The inverse row for $I$ gives $I(-_R x)$, and the inverse row for $J$ gives $J(-_R x)$. Thus $K(-_R x)$.

For internal multiplication closure, assume $K(x)$ and $K(y)$. Carrier support gives $C(y)$. Right absorption in $I$, applied to $I(x)$ and $C(y)$, gives $I(x\cdot_R y)$. Right absorption in $J$, applied to $J(x)$ and $C(y)$, gives $J(x\cdot_R y)$. Hence $K(x\cdot_R y)$, which supplies the multiplicative closure row required by the sub-rng part of $\IdealUp$.

For classifier transport, assume $K(x)$ and $x\sim_R y$. The transport row for $I$ gives $I(y)$, and the transport row for $J$ gives $J(y)$. Therefore $K(y)$.

For right absorption, assume $K(x)$ and $C(r)$. Right absorption in the two input ideal certificates gives $I(x\cdot_R r)$ and $J(x\cdot_R r)$, so $K(x\cdot_R r)$. For left absorption, the same argument with the left absorption rows gives $I(r\cdot_R x)$ and $J(r\cdot_R x)$, hence $K(r\cdot_R x)$.

These rows are exactly the carrier, additive sub-rng, classifier-stability, and ambient two-sided absorption rows required of an $\IdealUp$ predicate over the given $\RingUp$ certificate.
\end{proof}

\begin{definition}[Ideal sum predicate]
\label{def:ideal-sum-predicate}
Let $R$ carry a $\RingUp$ certificate with carrier $C$, classifier $\sim_R$, zero $0_R$, addition $+_R$, additive inverse $-_R$, and multiplication $\cdot_R$. Let $I$ and $J$ be $\IdealUp$ carried predicates over $R$ in the sense of \autoref{sec:ideal-carrier}. The ideal-sum predicate $I+_RJ$ is defined by the displayed additive decomposition
$$
(I+_RJ)(z)\Longleftrightarrow
\exists x,y:\mathsf{BHist}.\; I(x)\land J(y)\land z\sim_R x+_R y.
$$
The ambient carrier witnesses are obtained through the carrier-support rows of the two $\IdealUp$ certificates, the ring addition carrier row, and carrier transport along $\sim_R$. The hidden ledger for a sum membership contains only the two membership witnesses $I(x)$ and $J(y)$ together with the additive decomposition row $z\sim_R x+_R y$.
\end{definition}

\begin{theorem}[Ideal sum closure]
\label{thm:ideal-sum-closure}
Let $R$ carry a $\RingUp$ certificate as in \autoref{ch:concrete-instances-ring-namecert}, with carrier $C$, classifier $\sim_R$, addition $+_R$, additive inverse $-_R$, zero $0_R$, and multiplication $\cdot_R$. Let $I$ and $J$ be $\IdealUp$ predicates over $R$. Assume the two ideal certificates provide the rows specified in \autoref{sec:ideal-certificate}: carrier support, zero membership, additive closure, additive inverse closure, internal multiplication closure, classifier transport, and left and right absorption by every carried ring endpoint. For $K:=I+_RJ$ as in \autoref{def:ideal-sum-predicate}, $K$ is an $\IdealUp$ predicate over $R$. Explicitly, $K$ has carrier support and the rows
$$
K(0_R),\qquad
K(z)\land K(w)\Longrightarrow K(z+_R w),\qquad
K(z)\Longrightarrow K(-_R z),
$$
$$
K(z)\land K(w)\Longrightarrow K(z\cdot_R w),\qquad
K(z)\land z\sim_R z'\Longrightarrow K(z'),
$$
and, for every $r$ with $C(r)$,
$$
K(z)\Longrightarrow K(r\cdot_R z)\land K(z\cdot_R r).
$$
\end{theorem}
\leanchecked{BEDC.Derived.IdealUp.IdealSum\_closure\_rows}
\begin{proof}
For carrier support, unpack a witness for $K(z)$. The support rows of $I$ and $J$ give $C(x)$ and $C(y)$, the ring addition carrier row gives $C(x+_R y)$, and carrier transport along $z\sim_R x+_R y$ gives $C(z)$.

For zero membership, use the witness pair $(0_R,0_R)$. The zero rows give $I(0_R)$ and $J(0_R)$, and the additive-zero row of the ring certificate supplies the decomposition $0_R\sim_R0_R+_R0_R$.

For additive closure, choose witnesses $z\sim_R x+_R y$ for $K(z)$ and $w\sim_R x'+_R y'$ for $K(w)$. Addition congruence carries these decompositions to

$$
z+_R w\sim_R (x+_R y)+_R(x'+_R y').
$$

Additive associativity and commutativity in the $\RingUp$ rows regroup the right side as $(x+_R x')+_R(y+_R y')$. The additive-closure rows of $I$ and $J$ give $I(x+_R x')$ and $J(y+_R y')$, which are the required witnesses for $K(z+_R w)$.

For additive inverse closure, start from $z\sim_R x+_R y$. The inverse-congruence row gives $-_R z\sim_R -_R(x+_R y)$, and the inverse-of-sum row identifies the right side with $(-_R x)+_R(-_R y)$. The additive inverse rows of $I$ and $J$ supply $I(-_R x)$ and $J(-_R y)$, so $K(-_R z)$.

For classifier transport, assume $K(z)$ and $z\sim_R z'$. Symmetry and transitivity of the classifier compose $z'\sim_R z$ with the displayed decomposition for $z$, giving the same witness pair for $K(z')$.

For absorption, assume $K(z)$ with witness $z\sim_R x+_R y$ and let $r$ be carried. Distributivity gives

$$
r\cdot_R z\sim_R (r\cdot_R x)+_R(r\cdot_R y)
$$

and

$$
z\cdot_R r\sim_R (x\cdot_R r)+_R(y\cdot_R r).
$$

The left and right absorption rows of $I$ and $J$ supply the four displayed ideal memberships, so the two decompositions witness $K(r\cdot_R z)$ and $K(z\cdot_R r)$.

For internal multiplication closure, carrier support for $K(w)$ gives $C(w)$. Right absorption applied to $K(z)$ and $C(w)$ yields $K(z\cdot_R w)$, which supplies the multiplicative row required by the sub-rng part of $\IdealUp$.

These rows are exactly the carrier, additive sub-rng, multiplicative sub-rng, classifier-stability, and two-sided ambient absorption rows required of an $\IdealUp$ predicate over the fixed $\RingUp$ certificate.
\end{proof}
\leanchecked{BEDC.Derived.IdealUp.IdealSum\_ideal\_closure\_rows}

\begin{theorem}[Ideal sum quotient-ring kernel scope]
\label{thm:ideal-sum-quotientring-kernel-scope}
Let $R$ carry a $\RingUp$ certificate and let $I$ and $J$ be $\IdealUp$ carried predicates over $R$. Let $K:=I+_RJ$ be the ideal-sum predicate of \autoref{def:ideal-sum-predicate}, and let $Q_K$ be the quotient-kernel source package of \autoref{def:ideal-quotient-kernel-source-package}. The package $Q_K$ has the same diagonal and endpoint-transport scope as the quotient-kernel packages built from any other $\IdealUp$ predicate, so $\QuotientRingUp$ may consume ideal sums without adding quotient primitives.
\end{theorem}
\leanchecked{BEDC.Derived.IdealUp.IdealSumQuotientKernel\_scope}
\begin{proof}
By \autoref{thm:ideal-sum-closure}, the predicate $K$ is an $\IdealUp$ predicate over the same $\RingUp$ data as $I$ and $J$.

Apply \autoref{thm:ideal-quotient-kernel-diagonal-exactness} to $K$ for the diagonal row, and apply \autoref{thm:ideal-quotient-kernel-transport-surface} to $K$ for endpoint transport. The kernel relation is still the same concrete relation $Q_K(x,y)\Longleftrightarrow C(x)\land C(y)\land K(x-_R y)$, so the quotient-ring consumer sees no additional quotient carrier or quotient primitive.

The ten-field fit is: traditional target sum of two ideals; BEDC source $\RingUp$ plus two $\IdealUp$ predicates over $\mathsf{BHist}$; carrier the existential additive decomposition predicate; classifier $\sim_R$; stability \autoref{thm:ideal-sum-closure}; ledger decomposition witnesses; core theorem \autoref{thm:ideal-sum-closure}; exactness row exhaustion of the displayed witnesses; standard bridge absent; status $\mathsf{ProofTarget}$.
\end{proof}
\leanchecked{BEDC.Derived.IdealUp.IdealSumQuotientKernel\_diagonal\_transport\_scope}

\begin{definition}[Principal two-sided generated ideal predicate]
\label{def:ideal-principal-generated-predicate}
Let $R$ carry a $\RingUp$ certificate with carrier $C$, classifier $\sim_R$, addition $+_R$, additive inverse $-_R$, zero $0_R$, and multiplication $\cdot_R$. Let $a$ be a carried generator, so $C(a)$. The principal two-sided generated ideal predicate $\langle a\rangle_R$ is the predicate on endpoints $z$ for which a finite $\mathsf{HistSpine}$ carries signed two-sided monomial rows
$$
\ell_i\cdot_R a\cdot_R r_i
$$
with $C(\ell_i)$ and $C(r_i)$ at every spine position $i$, together with sign rows, such that the $\RingUp$ finite additive fold of those signed rows is classifier-equal to $z$. The empty spine folds to $0_R$. The certificate ledger records the spine, the left and right coefficient rows, the sign rows, and the final classifier edge from the finite fold to $z$.
\end{definition}
\leandef{BEDC.Derived.IdealUp.IdealPrincipalGenerated}

\begin{theorem}[Principal generated subtractive closure]
\label{thm:ideal-principal-generated-subtractive-closure}
Let $R$ carry a $\RingUp$ certificate, let $a$ be a carried generator, and let $\langle a\rangle_R$ be the predicate of \autoref{def:ideal-principal-generated-predicate}. Then $\langle a\rangle_R$ has carrier support, zero membership, additive closure, additive inverse closure, and classifier transport over the ambient $\RingUp$ data:
$$
\langle a\rangle_R(0_R),\qquad
\langle a\rangle_R(x)\land \langle a\rangle_R(y)\Longrightarrow \langle a\rangle_R(x+_R y),\qquad
\langle a\rangle_R(x)\Longrightarrow \langle a\rangle_R(-_R x),
$$
and
$$
\langle a\rangle_R(x)\land x\sim_R y\Longrightarrow \langle a\rangle_R(y),
$$
with every displayed membership carrying a RingUp endpoint carrier witness.
\end{theorem}
\leanchecked{BEDC.Derived.IdealUp.IdealPrincipalGenerated\_subtractive\_closure\_rows}
\begin{proof}
Carrier support is part of the final fold certificate: every signed monomial row is carried by the multiplication-closure rows, the finite additive fold is carried by the additive-closure rows, and classifier transport carries that support to the named endpoint $z$.

For zero membership, take the empty spine. Its fold is $0_R$, and the final classifier edge is the reflexive edge at $0_R$.

For additive closure, concatenate the two finite spines witnessing $x$ and $y$. The fold of the concatenated spine is classified, by RingUp finite-fold associativity, with the sum of the two earlier folds. Addition congruence then transports along the two final classifier edges and yields a final classifier edge to $x+_R y$.

For additive inverse closure, reverse the signs in the spine witnessing $x$. The RingUp inverse and finite-fold rows classify the new fold with the additive inverse of the old fold, and inverse congruence transports the final edge to $-_R x$.

For classifier transport, keep the same spine and compose its final classifier edge with the given edge $x\sim_R y$. The ledger data is unchanged, and the composed edge witnesses $\langle a\rangle_R(y)$.
\end{proof}

\begin{theorem}[Principal generated absorption closure]
\label{thm:ideal-principal-generated-absorption-closure}
Let $R$ carry a $\RingUp$ certificate, let $a$ be a carried generator, and let $\langle a\rangle_R$ be the predicate of \autoref{def:ideal-principal-generated-predicate}. If $t$ is carried, then $\langle a\rangle_R(x)$ implies both $\langle a\rangle_R(t\cdot_R x)$ and $\langle a\rangle_R(x\cdot_R t)$. Consequently, together with \autoref{thm:ideal-principal-generated-subtractive-closure}, the predicate $\langle a\rangle_R$ supplies the internal multiplication row required of an $\IdealUp$ predicate.
\end{theorem}
\leanchecked{BEDC.Derived.IdealUp.IdealPrincipalGenerated\_absorption\_closure\_rows}
\begin{proof}
Start with a finite signed spine witnessing $\langle a\rangle_R(x)$. For left absorption by a carried endpoint $t$, multiply every monomial row in the spine on the left by $t$.

Associativity rewrites each row
$$
t\cdot_R(\ell_i\cdot_R a\cdot_R r_i)
$$
as
$$
(t\cdot_R\ell_i)\cdot_R a\cdot_R r_i.
$$
Carrier support for $t$ and $\ell_i$ gives $C(t\cdot_R\ell_i)$, while $C(r_i)$ is retained. Distributivity and the finite-fold multiplication rows classify the fold of the transformed spine with $t\cdot_R x$, after composing with the original final classifier edge.

For right absorption, multiply every monomial row on the right by $t$. Associativity rewrites
$$
(\ell_i\cdot_R a\cdot_R r_i)\cdot_R t
$$
as
$$
\ell_i\cdot_R a\cdot_R(r_i\cdot_R t),
$$
and the multiplication-closure rows give $C(r_i\cdot_R t)$. The same distributivity and finite-fold argument supplies the final classifier edge to $x\cdot_R t$.

Internal multiplication is then a consequence of right absorption: from $\langle a\rangle_R(x)$ and $\langle a\rangle_R(y)$, carrier support from \autoref{thm:ideal-principal-generated-subtractive-closure} gives $C(y)$, and right absorption by $y$ gives $\langle a\rangle_R(x\cdot_R y)$.
\end{proof}

The internal multiplication row is obtained there from right absorption together with carrier support for the second factor. These rows are precisely the row package required of an $\IdealUp$ predicate over the ambient $\RingUp$ certificate.

\begin{definition}[Ring map zero-fiber predicate]
\label{def:ring-map-zero-fiber-predicate}
Let $S$ and $T$ carry $\RingUp$ certificates, with carriers $C_S,C_T$, classifiers $\sim_S,\sim_T$, zeros $0_S,0_T$, additions $+_S,+_T$, additive inverses $-_S,-_T$, and multiplications $\cdot_S,\cdot_T$. For a function $f:S\to T$, the ring-map zero-fiber predicate on the source carrier is
$$
\mathsf{Ker}^{\mathrm{ring}}_f(x)\Longleftrightarrow C_S(x)\land f(x)\sim_T0_T.
$$
The first component records source-carrier evidence; the second records the target-zero classification.
\end{definition}

\begin{theorem}[Ring map zero fiber IdealUp closure]
\label{thm:ring-map-zero-fiber-ideal-closure}
Let $S$ and $T$ carry $\RingUp$ certificates as in \autoref{ch:concrete-instances-ring-namecert}, with notation as in \autoref{def:ring-map-zero-fiber-predicate}. Let $f:S\to T$ satisfy carrier preservation and classifier preservation on carried source endpoints:
$$
C_S(x)\Longrightarrow C_T(f(x))
$$
and
$$
C_S(x)\land C_S(y)\land x\sim_S y\Longrightarrow f(x)\sim_T f(y).
$$
Assume also that $f$ preserves zero, addition, additive inverse, and multiplication on carried source endpoints:
$$
f(0_S)\sim_T0_T,
$$
$$
C_S(x)\land C_S(y)\Longrightarrow f(x+_S y)\sim_T f(x)+_T f(y),
$$
$$
C_S(x)\Longrightarrow f(-_S x)\sim_T -_T f(x),
$$
and
$$
C_S(x)\land C_S(y)\Longrightarrow f(x\cdot_S y)\sim_T f(x)\cdot_T f(y).
$$
Then $K:=\mathsf{Ker}^{\mathrm{ring}}_f$ from \autoref{def:ring-map-zero-fiber-predicate} is an $\IdealUp$ predicate over the source ring $S$. Explicitly, $K$ has carrier support and the rows
$$
K(0_S),\qquad K(x)\land K(y)\Longrightarrow K(x+_S y),\qquad K(x)\Longrightarrow K(-_S x),
$$
$$
K(x)\land K(y)\Longrightarrow K(x\cdot_S y),\qquad K(x)\land x\sim_S y\Longrightarrow K(y),
$$
and, for every $r$ with $C_S(r)$,
$$
K(x)\Longrightarrow K(r\cdot_S x)\land K(x\cdot_S r).
$$
\end{theorem}
\leanchecked{BEDC.Derived.IdealUp.RingMapZeroFiber\_ideal\_closure\_rows}
\begin{proof}
Unfold \autoref{def:ring-map-zero-fiber-predicate}. Carrier support is the first component of a $K$ witness.

For zero, the source $\RingUp$ certificate carries $0_S$, and the zero-preservation row gives $f(0_S)\sim_T0_T$. Hence $K(0_S)$.

For classifier transport, assume $K(x)$ and $x\sim_S y$. The source carrier-transport row from \autoref{def:lean-carrier-naming-certificate} gives $C_S(y)$. Classifier preservation gives $f(x)\sim_T f(y)$. Target classifier symmetry and \autoref{thm:namecert-equivalence-transitivity} compose this with the zero-fiber witness $f(x)\sim_T0_T$ to obtain $f(y)\sim_T0_T$. Thus $K(y)$.

For additive closure, assume $K(x)$ and $K(y)$. Source additive closure carries $x+_S y$. Addition preservation gives
$$
f(x+_S y)\sim_T f(x)+_T f(y).
$$
Target addition congruence, applied to the two zero-fiber witnesses, gives
$$
f(x)+_T f(y)\sim_T0_T+_T0_T.
$$
The target additive zero row contained in \autoref{def:ring-stability-certificate} reduces $0_T+_T0_T$ to $0_T$. Transitivity gives $f(x+_S y)\sim_T0_T$, so $K(x+_S y)$.

For additive inverse closure, assume $K(x)$. Source inverse closure carries $-_S x$. Additive-inverse preservation gives
$$
f(-_S x)\sim_T -_T f(x).
$$
Target inverse congruence transports $f(x)\sim_T0_T$ to $-_T f(x)\sim_T -_T0_T$, and \autoref{thm:ring-neg-zero}, read in the target additive component, identifies $-_T0_T$ with $0_T$. Composing the classifier edges yields $f(-_S x)\sim_T0_T$, hence $K(-_S x)$.

For internal multiplication closure, assume $K(x)$ and $K(y)$. Source multiplication closure carries $x\cdot_S y$. Multiplication preservation gives
$$
f(x\cdot_S y)\sim_T f(x)\cdot_T f(y).
$$
Target multiplication congruence transports the two zero-fiber witnesses to
$$
f(x)\cdot_T f(y)\sim_T0_T\cdot_T0_T.
$$
The zero-product rows of \autoref{def:ring-stability-certificate} reduce $0_T\cdot_T0_T$ to $0_T$. Thus $K(x\cdot_S y)$.

For left absorption, assume $K(x)$ and $C_S(r)$. Source multiplication closure carries $r\cdot_S x$, and multiplication preservation gives
$$
f(r\cdot_S x)\sim_T f(r)\cdot_T f(x).
$$
Carrier preservation gives $C_T(f(r))$. Target multiplication congruence transports the right factor along $f(x)\sim_T0_T$, and the target right zero-product row gives
$$
f(r)\cdot_T0_T\sim_T0_T.
$$
Therefore $K(r\cdot_S x)$. For right absorption, the same argument starts from
$$
f(x\cdot_S r)\sim_T f(x)\cdot_T f(r),
$$
transports the left factor along $f(x)\sim_T0_T$, and uses the target left zero-product row $0_T\cdot_T f(r)\sim_T0_T$. This gives $K(x\cdot_S r)$.

The displayed carrier support, zero, additive, inverse, multiplicative, classifier-transport, and two-sided absorption rows are exactly the $\IdealUp$ certificate rows listed in \autoref{sec:ideal-certificate} for a predicate over the source $\RingUp$ certificate.
\end{proof}
\leanchecked{BEDC.Derived.IdealUp.RingMapIdealPreimage\_closure\_rows}

The ideal quotient-kernel source package is the concrete $\mathsf{BHist}$ relation $Q_I$ on ring endpoints defined by

Q_I(x,y)\Longleftrightarrow C(x)\land C(y)\land I(x-_R y).

It is the quotient-free $\mathsf{BHist}$ relation later read as the ideal-coset relation in \autoref{def:quotientring-ideal-coset-relation}.
\end{definition}

\begin{theorem}[Ideal quotient-kernel diagonal exactness]
\label{thm:ideal-quotient-kernel-diagonal-exactness}
Let $R$ carry a $\RingUp$ certificate with notation as in \autoref{def:ideal-quotient-kernel-source-package}, and let $I$ be an $\IdealUp$ carried predicate over $R$. Assume the ideal certificate supplies zero membership and classifier transport, and the ring certificate supplies the additive inverse and zero rows. If $C(x)$, then $Q_I(x,x)$.
\end{theorem}
\begin{proof}
Unfold \autoref{def:ideal-quotient-kernel-source-package}. The two endpoint carrier witnesses are both the given evidence $C(x)$.

It remains to show $I(x-_R x)$. By the additive inverse row in the underlying ring certificate, $x+_R(-_R x)$ is classified with $0_R$. The ideal zero row gives $I(0_R)$, and classifier symmetry orients the comparison from $0_R$ to $x-_R x$.

Classifier transport for $I$ carries zero membership across this edge. Repacking the two carrier witnesses with the transported ideal membership gives $Q_I(x,x)$.
\end{proof}
\leanchecked{BEDC.Derived.IdealUp.IdealQuotientKernel\_diagonal\_exactness}

\begin{theorem}[Ideal quotient-kernel transport surface]
\label{thm:ideal-quotient-kernel-transport-surface}
Let $R$ carry a $\RingUp$ certificate with notation as in \autoref{def:ideal-quotient-kernel-source-package}, and let $I$ be an $\IdealUp$ carried predicate over $R$. Assume the ring certificate supplies carrier transport, additive inverse congruence, addition congruence, and the classifier-equivalence rows, and assume the ideal certificate supplies classifier transport. If $Q_I(x,y)$, $x\sim_R x'$, and $y\sim_R y'$, then $Q_I(x',y')$.
\end{theorem}
\begin{proof}
Unfold \autoref{def:ideal-quotient-kernel-source-package}. From $Q_I(x,y)$ we have $C(x)$, $C(y)$, and $I(x-_R y)$.

Carrier transport along $x\sim_R x'$ and $y\sim_R y'$ gives $C(x')$ and $C(y')$. Additive inverse congruence sends $y\sim_R y'$ to $-_R y\sim_R -_R y'$, and addition congruence combines this with $x\sim_R x'$ to obtain
$$
x-_R y\sim_R x'-_R y'.
$$

The ideal classifier-transport row carries $I(x-_R y)$ across this subtraction comparison, giving $I(x'-_R y')$. Pairing this membership with the transported carriers gives $Q_I(x',y')$.
\end{proof}
\leanchecked{BEDC.Derived.IdealUp.IdealQuotientKernel\_endpoint\_transport}

\begin{theorem}[Ideal quotient-kernel multiplication descent]
\label{thm:ideal-quotient-kernel-multiplication-descent}
Let $R$ carry a $\RingUp$ certificate with notation as in \autoref{def:ideal-quotient-kernel-source-package}, and let $I$ be an $\IdealUp$ carried predicate over $R$. Assume the ring certificate supplies multiplication carrier rows and the product-difference classifier row, and assume the ideal certificate supplies classifier transport, additive closure, and left and right absorption by every carried endpoint. If $Q_I(a,a')$ and $Q_I(b,b')$, then the IdealUp rows give
$$
I((a\cdot_R b)-_R(a'\cdot_R b')).
$$
Together with the RingUp multiplication carrier rows, this repacks as
$$
Q_I(a\cdot_R b,a'\cdot_R b').
$$
Ten-field fit: traditional target quotient-kernel multiplication descent; BEDC source $\RingUp$ plus $\IdealUp$ over $\mathsf{BHist}$; carrier $Q_I$ endpoints; classifier $\sim_R$; stability absorption, additive closure, and classifier transport; ledger two representative differences and two absorbed products; core theorem this block; exactness the displayed repacking; standard bridge absent; status proof target.
\end{theorem}
\leanchecked{BEDC.Derived.IdealUp.IdealQuotientKernel\_multiplication\_descent}
\begin{proof}
Unfold \autoref{def:ideal-quotient-kernel-source-package}. From $Q_I(a,a')$ and $Q_I(b,b')$ we obtain carried endpoints $C(a)$, $C(a')$, $C(b)$, and $C(b')$, together with ideal memberships for the representative differences. Write
$$
d_a:=a-_R a',
\qquad
d_b:=b-_R b'.
$$

Right absorption applies $I(d_a)$ against the carried endpoint $b$, giving $I(d_a\cdot_R b)$. Left absorption applies $I(d_b)$ against the carried endpoint $a'$, giving $I(a'\cdot_R d_b)$. Additive closure combines these two absorbed products into membership of their ring sum.

The product-difference classifier row identifies that sum with $(a\cdot_R b)-_R(a'\cdot_R b')$. Classifier transport for the ideal predicate carries the combined membership across this comparison, yielding
$$
I((a\cdot_R b)-_R(a'\cdot_R b')).
$$

Finally, multiplication carrier rows applied to $C(a),C(b)$ and to $C(a'),C(b')$ give the two product carrier witnesses. Pairing those witnesses with the transported product-difference membership is exactly $Q_I(a\cdot_R b,a'\cdot_R b')$.
\end{proof}
\leanchecked{BEDC.Derived.IdealUp.IdealQuotientKernel\_product\_difference\_membership}

\begin{definition}[Finite ideal-intersection predicate]
\label{def:finite-ideal-intersection-predicate}

Let $xs:\mathsf{HistSpine}$ be a finite BEDC-history spine, and let $\mathcal{I}$ assign to every position $i:\mathsf{Pos}(xs)$ a predicate $\mathcal{I}_i:\mathsf{BHist}\to\mathsf{Prop}$ over the same ring data. \mathsf{FinIdealMeet}_{C,xs,\mathcal{I}}(x)
\Longleftrightarrow

The carrier conjunct makes the empty finite spine denote the ambient carried ring predicate, while a nonempty spine is the usual pointwise intersection of the displayed ideal predicates.
\end{definition}

\begin{theorem}[Finite ideal intersection closure]
\label{thm:finite-ideal-intersection-closure}
Let $R$ carry a $\RingUp$ certificate as in \autoref{ch:concrete-instances-ring-namecert}, with carrier $C$, classifier $\sim_R$, addition $+_R$, additive inverse $-_R$, zero $0_R$, and multiplication $\cdot_R$. Let $xs:\mathsf{HistSpine}$ be a finite BEDC-history spine, and let $\mathcal{I}$ assign to every position $i:\mathsf{Pos}(xs)$ a predicate $\mathcal{I}_i:\mathsf{BHist}\to\mathsf{Prop}$. Assume that, for every $i:\mathsf{Pos}(xs)$, the predicate $\mathcal{I}_i$ is an $\IdealUp$ predicate over the same $\RingUp$ data, with the rows specified in \autoref{sec:ideal-certificate}: carrier support, zero membership, additive closure, additive inverse closure, internal multiplication closure, classifier transport, and left and right absorption by every carried ring endpoint. For
$$
K(x)\Longleftrightarrow \mathsf{FinIdealMeet}_{C,xs,\mathcal{I}}(x),
$$
the predicate $K$ is an $\IdealUp$ predicate over $R$. Explicitly, $K$ has carrier support and the rows
$$
K(0_R),\qquad
K(x)\land K(y)\Longrightarrow K(x+_R y),\qquad
K(x)\Longrightarrow K(-_R x),
$$
$$
K(x)\land K(y)\Longrightarrow K(x\cdot_R y),\qquad
K(x)\land x\sim_R y\Longrightarrow K(y),
$$
and, for every $r$ with $C(r)$,
$$
K(x)\Longrightarrow K(r\cdot_R x)\land K(x\cdot_R r).
$$
\end{theorem}
\leanchecked{BEDC.Derived.IdealUp.FiniteIdealMeet\_closure\_rows}
\begin{proof}
Unfold \autoref{def:finite-ideal-intersection-predicate}. The carrier support row is the first conjunct of $K(x)$.

For zero membership, the ring certificate supplies $C(0_R)$. Fix a position $i:\mathsf{Pos}(xs)$. The zero row in the $i$th ideal certificate gives $\mathcal{I}_i(0_R)$. Since $i$ was arbitrary, $0_R$ satisfies every predicate in the finite family, and together with $C(0_R)$ this is $K(0_R)$.

For additive closure, assume $K(x)$ and $K(y)$. The carrier conjuncts give $C(x)$ and $C(y)$, so the additive closure row of the ring certificate gives $C(x+_R y)$. For an arbitrary position $i$, the universal conjuncts of $K(x)$ and $K(y)$ give $\mathcal{I}_i(x)$ and $\mathcal{I}_i(y)$. The additive closure row of the $i$th ideal certificate gives $\mathcal{I}_i(x+_R y)$. Thus every indexed predicate holds at $x+_R y$, and the carrier witness just obtained repacks as $K(x+_R y)$.

For additive inverse closure, assume $K(x)$. The ring inverse-closure row gives $C(-_R x)$ from $C(x)$. For each position $i$, the inverse row of the $i$th ideal certificate gives $\mathcal{I}_i(-_R x)$ from $\mathcal{I}_i(x)$. Hence all finite-family predicates hold at $-_R x$, and $K(-_R x)$ follows.

For internal multiplication closure, assume $K(x)$ and $K(y)$. The ring multiplication-closure row gives $C(x\cdot_R y)$ from $C(x)$ and $C(y)$. For every position $i$, the universal conjuncts give $\mathcal{I}_i(x)$ and $\mathcal{I}_i(y)$, and the internal multiplication row of the $i$th ideal certificate gives $\mathcal{I}_i(x\cdot_R y)$. Therefore $K(x\cdot_R y)$.

For classifier transport, assume $K(x)$ and $x\sim_R y$. Carrier transport from the naming certificate underlying the $\RingUp$ data carries $C(x)$ along $x\sim_R y$ to obtain $C(y)$. For each position $i$, the classifier-transport row of the $i$th ideal certificate gives $\mathcal{I}_i(y)$ from $\mathcal{I}_i(x)$ and the same classifier witness. Pairing these facts gives $K(y)$.

For absorption, assume $K(x)$ and $C(r)$. The ring multiplication-closure row gives both $C(r\cdot_R x)$ and $C(x\cdot_R r)$. For each position $i$, left absorption in the $i$th ideal certificate gives $\mathcal{I}_i(r\cdot_R x)$ from $C(r)$ and $\mathcal{I}_i(x)$, and right absorption gives $\mathcal{I}_i(x\cdot_R r)$ from the same data. Hence the finite family holds at both absorbed endpoints. Together with the carrier witnesses, this gives $K(r\cdot_R x)$ and $K(x\cdot_R r)$.

These are exactly the carrier, additive sub-rng, multiplicative sub-rng, classifier-stability, and two-sided ambient absorption rows required of an $\IdealUp$ predicate over the fixed $\RingUp$ certificate.
\end{proof}

\begin{definition}[Ring map ideal preimage predicate]
\label{def:ring-map-ideal-preimage-predicate}
Let $S$ and $T$ carry $\RingUp$ certificates, with carriers $C_S,C_T$, classifiers $\sim_S,\sim_T$, zeros $0_S,0_T$, additions $+_S,+_T$, additive inverses $-_S,-_T$, and multiplications $\cdot_S,\cdot_T$. The first component records source-carrier evidence, and the second component records membership of the image endpoint in the target ideal.
\end{definition}

\begin{theorem}[Ring map ideal preimage IdealUp closure]
\label{thm:ring-map-ideal-preimage-ideal-closure}
Let $S$ and $T$ carry $\RingUp$ certificates as in \autoref{ch:concrete-instances-ring-namecert}, with notation as in \autoref{def:ring-map-ideal-preimage-predicate}. Let $f:S\to T$ satisfy carrier preservation and classifier preservation on carried source endpoints:
$$
C_S(x)\Longrightarrow C_T(f(x))
$$
and
$$
C_S(x)\land C_S(y)\land x\sim_S y\Longrightarrow f(x)\sim_T f(y).
$$
Assume also that $f$ preserves zero, addition, additive inverse, and multiplication on carried source endpoints:
$$
f(0_S)\sim_T0_T,
$$
$$
C_S(x)\land C_S(y)\Longrightarrow f(x+_S y)\sim_T f(x)+_T f(y),
$$
$$
C_S(x)\Longrightarrow f(-_S x)\sim_T -_T f(x),
$$
and
$$
C_S(x)\land C_S(y)\Longrightarrow f(x\cdot_S y)\sim_T f(x)\cdot_T f(y).
$$
Let $J$ be an $\IdealUp$ predicate over $T$. Assume its ideal certificate supplies the rows specified in \autoref{sec:ideal-certificate}: carrier support, zero membership, additive closure, additive inverse closure, internal multiplication closure, classifier transport, and left and right absorption by every carried target ring endpoint. For $K:=\mathsf{Preim}^{\mathrm{ideal}}_{f,J}$ from \autoref{def:ring-map-ideal-preimage-predicate}, $K$ is an $\IdealUp$ predicate over the source ring $S$. Explicitly, $K$ has carrier support and the rows
$$
K(0_S),\qquad K(x)\land K(y)\Longrightarrow K(x+_S y),\qquad K(x)\Longrightarrow K(-_S x),
$$
$$
K(x)\land K(y)\Longrightarrow K(x\cdot_S y),\qquad K(x)\land x\sim_S y\Longrightarrow K(y),
$$
and, for every $r$ with $C_S(r)$,
$$
K(x)\Longrightarrow K(r\cdot_S x)\land K(x\cdot_S r).
$$
\end{theorem}
\begin{proof}
Unfold \autoref{def:ring-map-ideal-preimage-predicate}. Carrier support is the first component of a $K$ witness.

For zero membership, the source $\RingUp$ certificate carries $0_S$. The target ideal zero row gives $J(0_T)$. The zero-preservation row gives $f(0_S)\sim_T0_T$, and target classifier symmetry from \autoref{def:lean-carrier-naming-certificate} gives $0_T\sim_T f(0_S)$. Target ideal classifier transport therefore gives $J(f(0_S))$, so $K(0_S)$.

For additive closure, assume $K(x)$ and $K(y)$. The source carrier components give $C_S(x)$ and $C_S(y)$, hence the source ring certificate carries $x+_S y$. The target ideal additive row gives $J(f(x)+_T f(y))$ from $J(f(x))$ and $J(f(y))$. The addition-preservation row gives $f(x+_S y)\sim_T f(x)+_T f(y)$, and target classifier symmetry reverses this comparison. Applying target ideal classifier transport gives $J(f(x+_S y))$. Together with the source carrier witness this is $K(x+_S y)$.

For additive inverse closure, assume $K(x)$. Source inverse closure carries $-_S x$. The target ideal inverse row gives $J(-_T f(x))$. The inverse-preservation row gives $f(-_S x)\sim_T -_T f(x)$, whose symmetric form transports $J(-_T f(x))$ to $J(f(-_S x))$. Thus $K(-_S x)$.

For internal multiplication closure, assume $K(x)$ and $K(y)$. Source multiplication closure carries $x\cdot_S y$. The target ideal internal multiplication row gives $J(f(x)\cdot_T f(y))$. The multiplication-preservation row gives $f(x\cdot_S y)\sim_T f(x)\cdot_T f(y)$, and target classifier symmetry orients this comparison for ideal transport. Hence $J(f(x\cdot_S y))$, so $K(x\cdot_S y)$.

For source classifier transport, assume $K(x)$ and $x\sim_S y$. Source carrier transport from \autoref{def:lean-carrier-naming-certificate} carries $C_S(x)$ along $x\sim_S y$ to obtain $C_S(y)$. Classifier preservation gives $f(x)\sim_T f(y)$. Target ideal classifier transport sends $J(f(x))$ along this comparison to $J(f(y))$. Therefore $K(y)$.

For left absorption, assume $K(x)$ and $C_S(r)$. Source multiplication closure carries $r\cdot_S x$, and carrier preservation gives $C_T(f(r))$. The target ideal left-absorption row gives $J(f(r)\cdot_T f(x))$ from $C_T(f(r))$ and $J(f(x))$. Multiplication preservation gives $f(r\cdot_S x)\sim_T f(r)\cdot_T f(x)$; after target classifier symmetry, ideal transport yields $J(f(r\cdot_S x))$. Thus $K(r\cdot_S x)$.

For right absorption, the same data carry $x\cdot_S r$ in the source, and the target ideal right-absorption row gives $J(f(x)\cdot_T f(r))$. Multiplication preservation for $x\cdot_S r$, followed by target classifier symmetry and ideal transport, gives $J(f(x\cdot_S r))$. Hence $K(x\cdot_S r)$.

The displayed carrier support, zero, additive, inverse, multiplicative, classifier-transport, and two-sided absorption rows are exactly the $\IdealUp$ certificate rows listed in \autoref{sec:ideal-certificate} for a predicate over the source $\RingUp$ certificate.
\end{proof}
