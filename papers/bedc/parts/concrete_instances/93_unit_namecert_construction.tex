\chapter{A Concrete Naming Certificate for $\UnitUp$}
\label{ch:concrete-instances-unit-namecert}

The unit-type interface $\UnitUp$ packages the unique naming certificate over the inductive carrier with a single constructor. It is the terminal object in the category of naming certificates and supplies the canonical trivial witness for placeholder constructions.

\section{Carrier definition}
\label{sec:unit-carrier}

\begin{definition}[Unit history carrier]
\label{def:unit-history-carrier}
For a history $h:\mathsf{BHist}$, the unit carrier is the continuation
endpoint that composes the empty history with $h$ and returns the empty
history:
$$
\mathsf{UnitHistoryCarrier}(h):=\mathsf{Cont}(\emp,h,\emp).
$$
\end{definition}
\leandef{BEDC.Derived.UnitUp.UnitHistoryCarrier}

\begin{definition}[Unit history classifier]
\label{def:unit-history-classifier}
For histories $h,k:\mathsf{BHist}$, the unit classifier requires both
endpoints to lie in the unit carrier and relates them by $\hsame$:
$$
\mathsf{UnitHistoryClassifier}(h,k):=
\mathsf{UnitHistoryCarrier}(h)\land
\mathsf{UnitHistoryCarrier}(k)\land
h\hsame k.
$$
\end{definition}
\leandef{BEDC.Derived.UnitUp.UnitHistoryClassifier}

\begin{theorem}[Unit history classifier empty endpoints iff]
\label{thm:unit-history-classifier-empty-endpoints-iff}
For histories \(h,k:\mathsf{BHist}\), the unit classifier is equivalent to
both endpoints being empty:
$$
\mathsf{UnitHistoryClassifier}(h,k)
\Longleftrightarrow h\hsame\emp\land k\hsame\emp.
$$
\end{theorem}
\begin{proof}
In the forward direction, each carrier component is a continuation
\(\Cont(\emp,-,\emp)\); empty-result inversion identifies the endpoint with
\(\emp\). In the reverse direction, the endpoint equalities reduce both carrier
continuations to the empty left-unit continuation, and transitivity of
\(\hsame\) supplies the classifier equality.
\end{proof}
\leanchecked{BEDC.Derived.UnitUp.UnitHistoryClassifier\_empty\_endpoints\_iff}

\begin{theorem}[Unit history classifier visible endpoint absurd]
\label{thm:unit-history-classifier-visible-endpoint-absurd}
No visible endpoint can occur on either side of the unit history classifier.
For all histories $p,q,k$, each classifier instance
$\mathsf{UnitHistoryClassifier}(\mathsf{e0}(p),k)$,
$\mathsf{UnitHistoryClassifier}(\mathsf{e1}(p),k)$,
$\mathsf{UnitHistoryClassifier}(k,\mathsf{e0}(q))$, and
$\mathsf{UnitHistoryClassifier}(k,\mathsf{e1}(q))$ is impossible.
\end{theorem}
\begin{proof}
Each classifier witness contains the unit carrier witness for the visible
endpoint. The left-unit continuation result identifies that visible endpoint
with the empty history, contradicting constructor disjointness for $\mathsf{e0}$
and $\mathsf{e1}$.
\end{proof}
\leanchecked{BEDC.Derived.UnitUp.UnitHistoryClassifier\_visible\_endpoint\_absurd}

\begin{theorem}[Unit history carrier empty equivalence]
\label{thm:unit-history-carrier-empty-iff}
For a history $h:\mathsf{BHist}$, membership in the unit history carrier is
equivalent to history-sameness with the empty history:
$$
\mathsf{UnitHistoryCarrier}(h)\Longleftrightarrow \HSame(h,\emp).
$$
\end{theorem}
\begin{proof}
The forward direction applies the right-unit uniqueness law to
$\mathsf{Cont}(\emp,h,\emp)$. Conversely, a witness $\HSame(h,\emp)$ reduces
the endpoint to the empty history, where the left-unit continuation law gives
$\mathsf{Cont}(\emp,\emp,\emp)$.
\end{proof}
\leanchecked{BEDC.Derived.UnitUp.UnitHistoryCarrier\_empty\_iff}

\section{The certificate}
\label{sec:unit-certificate}

\begin{theorem}[Unit history classifier semantic name certificate]
\label{thm:unit-history-classifier-semantic-namecert}
The unit history carrier, used as source, pattern, and ledger policy, forms a
semantic name certificate with the unit history classifier.
\end{theorem}
\begin{proof}
The empty history inhabits the carrier by the left-unit continuation law.
Classifier reflexivity, symmetry, and transitivity are obtained by packaging
the carrier endpoints together with the corresponding $\hsame$ laws. Carrier
transport follows from the target carrier endpoint already present in the
classifier witness.
\end{proof}
\leanchecked{BEDC.Derived.UnitUp.UnitHistoryClassifier\_semanticNameCert}
