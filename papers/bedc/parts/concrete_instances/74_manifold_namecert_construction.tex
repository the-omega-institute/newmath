\chapter{A Concrete Naming Certificate for $\ManifoldUp$}
\label{ch:concrete-instances-manifold-namecert}

The manifold interface $\ManifoldUp$ packages a topological carrier with a finite atlas of charts into $(\RealUp)^n$ whose transition maps are constructively smooth. It depends on $\NameCert_{\TopologyUp}$ (\autoref{ch:concrete-instances-topology-namecert}) and the real-analytic certificate (\autoref{ch:concrete-instances-real-analytic-namecert}), and supplies the substrate for the bundle and de~Rham certificates.

\section{Carrier definition}
\label{sec:manifold-carrier}

\begin{definition}[Charted history atlas source]
\label{def:manifoldup-charted-history-atlas-source}
A charted history atlas source for $\ManifoldUp$ is a predicate
$\mathsf{ManAtlasPkg}(P)$ on $\mathsf{BHist}$ packages $P$ whose displayed rows
are the following data.
\begin{enumerate}
\item The traditional target is a finite-atlas topological manifold.
\item The BEDC carrier histories are points carried by the underlying
$\TopologyUp$ carrier row.
\item A chart row for an index $i$ consists of a $\TopologyUp$ indexed-open
witness $\mathsf{Dom}_i$ on the carrier, a coordinate history
$\varphi_i$ into a finite product window $(\RealUp)^{n_i}$, and the local
chart ledger that binds the coordinate history to the displayed domain.
\item A transition row for an ordered pair $i,j$ consists of the overlap
predicate $\mathsf{Dom}_i(x)\wedge\mathsf{Dom}_j(x)$, the coordinate change
history $\tau_{ij}$ between the two $\RealUp$ windows, and a
$\RealAnalyticUp$ local transition certificate on that overlap.
\item The classifier row is componentwise $\hsame$ on the carrier history, the
finite chart-index history, every indexed-open domain row, every coordinate
history, and every real-analytic transition ledger.
\item The stability row says that componentwise $\hsame$ transport of those
fields carries $\mathsf{ManAtlasPkg}$ packages to
$\mathsf{ManAtlasPkg}$ packages.
\item The visible ledger records the topology-open witnesses, coordinate
windows, chart-cover rows, overlap rows, and real-analytic transition rows.
\item The standard bridge row is absent: this source names the BEDC obligation
surface and does not assert an external manifold equivalence.
\end{enumerate}
Equivalently, $\mathsf{ManAtlasPkg}(P)$ is the ten-field
$\NameCert_{\ManifoldUp}$ source object whose public carrier is the finite
charted atlas over the $\TopologyUp$ carrier, whose classifier is the displayed
componentwise $\hsame$ classifier, and whose visible ledger is the
topology-coordinate-transition ledger just listed.
\end{definition}

\begin{theorem}[ManifoldUp chart cover coverage]
\label{thm:manifoldup-chart-cover-coverage}
For every package $P$ with $\mathsf{ManAtlasPkg}(P)$ and every carried point
history $x$ in the underlying $\TopologyUp$ carrier, the visible atlas ledger
exhibits a chart index $i$ together with the displayed chart-domain row
$\mathsf{Dom}_i(x)$.
\end{theorem}
\begin{proof}
Unpack $\mathsf{ManAtlasPkg}(P)$ by
\autoref{def:manifoldup-charted-history-atlas-source}. The carrier row says
that $x$ is a $\mathsf{BHist}$ point carried by the same $\TopologyUp$ carrier
that supports every chart domain.

The visible atlas ledger is part of the charted source object. Its chart-cover
row is indexed by the finite chart history and states that every carried point
is displayed in at least one domain row.

Project the cover witness for $x$. The projection returns a chart index $i$
and the corresponding membership evidence $\mathsf{Dom}_i(x)$, both living in
the same $\TopologyUp$ carrier row as $x$.
\end{proof}

\begin{theorem}[ManifoldUp chart domain carrier transport]
\label{thm:manifoldup-chart-domain-carrier-transport}
Let $P$ be a package with $\mathsf{ManAtlasPkg}(P)$. If carried point histories
$x$ and $y$ are related by the $\TopologyUp$ carrier classifier and
$\mathsf{Dom}_i(x)$ holds for a displayed chart-domain row, then
$\mathsf{Dom}_i(y)$ holds after transporting the row along the classifier.
\end{theorem}
\begin{proof}
The chart-domain row $\mathsf{Dom}_i$ is an indexed-open witness supplied by
the $\TopologyUp$ part of
\autoref{def:manifoldup-charted-history-atlas-source}. Its membership predicate
is therefore classified by the same carrier comparison as the underlying
topological points.

Apply the carrier classifier to the evidence that $x$ and $y$ are the same
carried point for the topology row. The indexed-open stability clause transports
membership in $\mathsf{Dom}_i$ from $x$ to $y$.

The transported domain row is still the displayed row for chart $i$, because
the chart classifier is componentwise $\hsame$ on chart-domain histories. Thus
the transported membership is exactly $\mathsf{Dom}_i(y)$.
\end{proof}

\begin{theorem}[ManifoldUp atlas transition composition]
\label{thm:manifoldup-atlas-transition-composition}
Let $P$ be a package with $\mathsf{ManAtlasPkg}(P)$. For chart indices
$i,j,k$ and a carried point history $x$ lying in the three displayed domains,
the overlap transition from $i$ to $k$ is witnessed by composing the supplied
$\RealAnalyticUp$ transition ledgers from $i$ to $j$ and from $j$ to $k$ on the
common overlap.
\end{theorem}
\begin{proof}
Unpack the transition rows of
\autoref{def:manifoldup-charted-history-atlas-source}. The hypotheses
$\mathsf{Dom}_i(x)$, $\mathsf{Dom}_j(x)$, and $\mathsf{Dom}_k(x)$ place $x$ in
both overlaps $(i,j)$ and $(j,k)$.

Project the two visible ledgers $\mathsf{Trans}_{ij}$ and
$\mathsf{Trans}_{jk}$. Each supplies a coordinate-change history between finite
$\RealUp$ windows together with a $\RealAnalyticUp$ local transition
certificate on its overlap.

Compose the coordinate histories at the shared $j$-coordinates. The
$\RealAnalyticUp$ local transition row is closed under this displayed
composition boundary, so the composed history carries a real-analytic
certificate on the triple-overlap row.

Record the composed certificate as the transition witness from $i$ to $k$ on
the common overlap. This witness is the atlas-transition composition required
by the $\ManifoldUp$ ledger.
\end{proof}

\section{The certificate}
\label{sec:manifold-certificate}

\begin{theorem}[ManifoldUp transition ledger stability]
\label{thm:manifoldup-transition-ledger-stability}
Let $P$ and $Q$ be $\mathsf{BHist}$ packages. Suppose
$\mathsf{ManAtlasPkg}(P)$ holds and the atlas rows of $P$ and $Q$ are related
componentwise by $\hsame$: the carrier histories, finite chart-index histories,
indexed-open domain rows, coordinate histories, and real-analytic transition
ledgers are all related. Then the transition ledger of $P$ transports to a
transition ledger for $Q$.
\end{theorem}
\begin{proof}
Start with a displayed transition row of $P$ for chart indices $i,j$. It
contains the overlap membership boundary, the coordinate-change history, and
the $\RealAnalyticUp$ local transition certificate.

Transport the two chart-domain rows by componentwise $\hsame$. The
$\TopologyUp$ indexed-open membership stability clause transports the overlap
predicate because it is the conjunction of the two transported domain
memberships.

Transport the source and target coordinate histories by their componentwise
$\hsame$ rows. The real-analytic transport boundary then carries the local
transition certificate across the transported coordinate histories.

Repack the transported overlap, coordinate-change history, and real-analytic
certificate. The result is the corresponding transition ledger row for $Q$.
\end{proof}

\begin{theorem}[Manifold name-certificate obligation surface]
\label{thm:manifoldup-namecert-obligation-threshold}
The carrier definition of
\autoref{def:manifoldup-charted-history-atlas-source}, together with
\autoref{thm:manifoldup-chart-cover-coverage},
\autoref{thm:manifoldup-chart-domain-carrier-transport},
\autoref{thm:manifoldup-atlas-transition-composition}, and
\autoref{thm:manifoldup-transition-ledger-stability}, assembles the paper-side
$\ManifoldUp$ $\NameCert$ obligation threshold. The threshold consists of the
carrier, classifier, chart cover, chart-domain transport, transition
composition, transition-ledger stability, and the visible atlas ledger.
\end{theorem}
\begin{proof}
The carrier definition supplies the named source object:
$\TopologyUp$ rows for the base carrier, $\RealUp$ coordinate rows for chart
targets, finite chart-index data, indexed-open chart domains, coordinate
histories, and transition ledgers.

\autoref{thm:manifoldup-chart-cover-coverage} supplies the cover obligation by
projecting, for every carried point, an index and domain membership from the
visible atlas ledger.

\autoref{thm:manifoldup-chart-domain-carrier-transport} supplies the
classifier boundary for chart-domain membership. The same componentwise
$\hsame$ classifier transports the carrier, domains, coordinate histories, and
ledger rows named in the definition.

\autoref{thm:manifoldup-atlas-transition-composition} supplies the atlas
composition obligation, and
\autoref{thm:manifoldup-transition-ledger-stability} supplies stability of the
visible transition ledger under transported chart rows. The absence of a
standard bridge is already part of the carrier definition, so the assembled
object is exactly the $\ManifoldUp$ $\NameCert$ obligation threshold listed in
the theorem statement.
\end{proof}
