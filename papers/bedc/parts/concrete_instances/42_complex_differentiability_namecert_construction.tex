\chapter{A Concrete Naming Certificate for \(\ComplexDiffUp\)}
\label{ch:concrete-instances-complex-diff-namecert}

This chapter gives a concrete naming-certificate construction for the complex-differentiability interface \(\ComplexDiffUp\) at the paper-proof level. The construction sits above the complex-limit certificate \(\ComplexLimitUp\) (\autoref{ch:concrete-instances-complex-limit-namecert}) and below the holomorphic certificate \(\HolomorphicUp\) (\autoref{ch:concrete-instances-holomorphic-namecert}), and licenses pointwise complex differentiability with an explicit modulus in Bishop's constructive style. Every claim of the form ``\(f\) is complex-differentiable at \(z\) with derivative \(f'\)'' is unfolded into an explicit unary modulus \(\delta : \mathsf{Unary} \to \mathsf{Unary}\) controlling the difference quotient. The Cauchy--Riemann equations are derived inside this chapter as a coupling of real-component partial derivatives, with both directions of the equivalence given as constructive proofs. No use of classical case analysis on complex equality is permitted; everywhere the data is a witness or an explicit refutation.

\section{The complex difference quotient}

\begin{definition}[Complex non-zero witness]
\label{def:cplx-non-zero}
For a complex history \(h\) (a pair of unary-coded real histories), the predicate \(\mathsf{CplxNonZero}(h)\) holds when there is an explicit unary precision \(k\) and a strict-prefix witness showing that the modulus of \(h\) exceeds \(2^{-k}\). Concretely, \(\mathsf{CplxNonZero}(h)\) is the closed inductive predicate generated by the witness pair \((k, w)\) where \(w\) is a \(\mathsf{NatUnaryStrictPrefix}\) certificate against the modulus history of \(h\).
\end{definition}
\leandef{BEDC.Derived.ComplexDiffUp.CplxNonZero}

\begin{definition}[Complex difference quotient]
\label{def:cplx-diff-quot}
For complex histories \(z, h\) with \(\mathsf{CplxNonZero}(h)\) and a function-history \(f\) on the complex carrier, the complex difference-quotient relation \(\mathsf{CplxDiffQuot}(f, z, h, q)\) holds when \(q\) is the complex history obtained by:
\begin{enumerate}
  \item evaluating \(f\) at the complex continuation \(z + h\) to obtain \(\mathsf{val}^+\);
  \item evaluating \(f\) at \(z\) to obtain \(\mathsf{val}\);
  \item forming the complex difference \(\mathsf{val}^+ - \mathsf{val}\); and
  \item dividing by \(h\) using the explicit reciprocal of \(h\) supplied by \(\mathsf{CplxNonZero}(h)\).
\end{enumerate}
The resulting \(q\) is the unique complex history (up to \(\hsame\)) representing \((f(z+h)-f(z))/h\).
\end{definition}
\leandef{BEDC.Derived.ComplexDiffUp.CplxDiffQuot}

\begin{lemma}[Complex difference quotient is well-defined]
\label{lem:cplx-diff-quot-well-defined}
For each \((f, z, h)\) with \(\mathsf{CplxNonZero}(h)\) there is exactly one complex history \(q\), up to \(\hsame\), with \(\mathsf{CplxDiffQuot}(f, z, h, q)\).
\end{lemma}
\leanstmt{BEDC.Derived.ComplexDiffUp.cplx\_diff\_quot\_well\_defined}
\begin{proof}
Functionality of complex addition, subtraction, and division on the rational-pair representation of complex histories (\autoref{ch:concrete-instances-complex-namecert}) gives a unique \(q\) up to \(\hsame\). The reciprocal step is governed by the explicit witness in \(\mathsf{CplxNonZero}(h)\), which fixes a precision below which the modulus is bounded away from zero.
\end{proof}

\begin{lemma}[Compatibility of the difference quotient with \(\hsame\)]
\label{lem:cplx-diff-quot-hsame}
If \(\mathsf{CplxDiffQuot}(f, z, h, q)\) and \(\hsame(z, z')\), \(\hsame(h, h')\), \(\mathsf{CplxNonZero}(h')\), then \(\mathsf{CplxDiffQuot}(f, z', h', q)\).
\end{lemma}
\leanstmt{BEDC.Derived.ComplexDiffUp.cplx\_diff\_quot\_hsame}
\begin{proof}
Each of the four sub-steps in \autoref{def:cplx-diff-quot} is \(\hsame\)-stable on its complex inputs, by transport along the complex-history closure laws.
\end{proof}

\section{Pointwise complex differentiability}

\begin{definition}[Pointwise complex differentiability]
\label{def:cplx-diff-at}
For a complex-valued function-history \(f\), a complex history \(z\), and a candidate derivative \(f'\), the pointwise complex differentiability predicate \(\mathsf{CplxDiffAt}(f, z, f')\) is the data of an explicit modulus \(\delta : \mathsf{Unary} \to \mathsf{Unary}\) such that for every unary precision \(k\) and every complex history \(h\) with \(\mathsf{CplxNonZero}(h)\) and modulus less than \(2^{-\delta(k)}\), the unique difference quotient \(q\) of \autoref{def:cplx-diff-quot} satisfies \(|q - f'| < 2^{-k}\) as a complex-modulus comparison.
\end{definition}
\leandef{BEDC.Derived.ComplexDiffUp.CplxDiffAt}

\begin{lemma}[Uniqueness of the derivative]
\label{lem:cplx-diff-at-unique}
If \(\mathsf{CplxDiffAt}(f, z, f')\) and \(\mathsf{CplxDiffAt}(f, z, g')\), then \(\hsame(f', g')\).
\end{lemma}
\leanstmt{BEDC.Derived.ComplexDiffUp.cplx\_diff\_at\_unique}
\begin{proof}
For any unary precision \(k\), pick \(h\) with modulus below \(2^{-(\max(\delta_{f'}(k+1), \delta_{g'}(k+1)))}\). Both moduli supply that the difference quotient is within \(2^{-(k+1)}\) of \(f'\) and of \(g'\); the triangle inequality on complex moduli then yields \(|f' - g'| < 2^{-k}\). Since \(k\) is arbitrary, \(\hsame(f', g')\) follows from the complex-history closure law that two complex histories agreeing to every unary precision are \(\hsame\).
\end{proof}

\begin{lemma}[Compatibility with \(\hsame\)]
\label{lem:cplx-diff-at-hsame}
If \(\mathsf{CplxDiffAt}(f, z, f')\), \(\hsame(z, z')\), and \(\hsame(f', g')\), then \(\mathsf{CplxDiffAt}(f, z', g')\) with the same modulus \(\delta\).
\end{lemma}
\leanstmt{BEDC.Derived.ComplexDiffUp.cplx\_diff\_at\_hsame}
\begin{proof}
Transport via \autoref{lem:cplx-diff-quot-hsame} and the complex-modulus comparison's stability under \(\hsame\).
\end{proof}

\section{Closure laws for complex differentiability}

\begin{lemma}[Sum rule]
\label{lem:cplx-diff-sum}
If \(\mathsf{CplxDiffAt}(f, z, f')\) with modulus \(\delta_f\) and \(\mathsf{CplxDiffAt}(g, z, g')\) with modulus \(\delta_g\), then \(\mathsf{CplxDiffAt}(f + g, z, f' + g')\) with modulus \(k \mapsto \max(\delta_f(k+1), \delta_g(k+1))\).
\end{lemma}
\leanstmt{BEDC.Derived.ComplexDiffUp.cplx\_diff\_sum}
\begin{proof}
The difference quotient of \(f + g\) decomposes into the sum of the two individual difference quotients. The pasted modulus halves each error budget at precision \(k+1\), giving total error below \(2^{-k}\).
\end{proof}

\begin{lemma}[Scalar rule]
\label{lem:cplx-diff-scalar}
If \(\mathsf{CplxDiffAt}(f, z, f')\) with modulus \(\delta_f\) and \(c\) is a complex history, then \(\mathsf{CplxDiffAt}(c \cdot f, z, c \cdot f')\) with modulus \(k \mapsto \delta_f(k + |c|)\), where \(|c|\) is an explicit unary upper bound on the modulus of \(c\).
\end{lemma}
\leanstmt{BEDC.Derived.ComplexDiffUp.cplx\_diff\_scalar}
\begin{proof}
Linear scaling of the difference quotient absorbs \(c\); the modulus pastes by adding the explicit bound on \(|c|\) at each precision.
\end{proof}

\begin{lemma}[Product rule with Leibniz modulus pasting]
\label{lem:cplx-diff-product}
If \(\mathsf{CplxDiffAt}(f, z, f')\) and \(\mathsf{CplxDiffAt}(g, z, g')\), then \(\mathsf{CplxDiffAt}(f \cdot g, z, f' \cdot g(z) + f(z) \cdot g')\) with modulus pasted by an explicit Leibniz-style combination of \(\delta_f\), \(\delta_g\), an upper bound \(|f|_z\) for the modulus of \(f\) near \(z\), and an upper bound \(|g|_z\) for the modulus of \(g\) near \(z\).
\end{lemma}
\leanstmt{BEDC.Derived.ComplexDiffUp.cplx\_diff\_product}
\begin{proof}
Write the difference quotient of \(f \cdot g\) as
\[
\frac{(f(z+h) - f(z)) \cdot g(z+h) + f(z) \cdot (g(z+h) - g(z))}{h}.
\]
Each summand contributes one of the two derivative terms up to an explicit error controlled by the local moduli of \(f\) and \(g\) near \(z\); the local-modulus bounds are obtained from \(\mathsf{CplxDiffAt}(f, z, f')\) and \(\mathsf{CplxDiffAt}(g, z, g')\) by integrating the difference-quotient bound on a small explicit disk around \(z\).
\end{proof}

\begin{lemma}[Chain rule with explicit composite modulus]
\label{lem:cplx-diff-chain}
If \(\mathsf{CplxDiffAt}(g, z, g')\) with modulus \(\delta_g\) and \(\mathsf{CplxDiffAt}(f, g(z), f')\) with modulus \(\delta_f\), then \(\mathsf{CplxDiffAt}(f \circ g, z, f' \cdot g')\) with modulus \(k \mapsto \delta_g(\delta_f(k+1) + |g'|_z + 1) + 1\), where \(|g'|_z\) is an explicit bound on the modulus of \(g'\).
\end{lemma}
\leanstmt{BEDC.Derived.ComplexDiffUp.cplx\_diff\_chain}
\begin{proof}
For \(h\) small enough (controlled by the inner \(\delta_g\)), \(g(z+h) - g(z)\) is close to \(g' \cdot h\); plugging this into the outer difference quotient via \(\delta_f\) yields the composite estimate. The \(+1\) corrections absorb the rounding from each linear-approximation step.
\end{proof}

\section{Real-component partial derivatives}

\begin{definition}[Real-coordinate projections of a complex function]
\label{def:cplx-real-proj}
For a complex-valued function-history \(f\), write \(f = u + i \cdot v\) where the real part \(u\) and the imaginary part \(v\) are explicit real-valued function-histories obtained by projecting the rational-pair representation of \(f\). Each of \(u\) and \(v\) takes a complex input \((x, y)\) and returns a real history.
\end{definition}
\leandef{BEDC.Derived.ComplexDiffUp.CplxRealProj}

\begin{definition}[One-variable real partial derivative]
\label{def:partial-real}
For a real-valued function-history \(u\) on the complex carrier and a complex history \(z = (x, y)\), the partial derivative with respect to the real coordinate, \(\mathsf{PartialReal}(u, z, p)\), is the data of an explicit modulus \(\delta : \mathsf{Unary} \to \mathsf{Unary}\) such that for every unary precision \(k\) and every real history \(t\) with \(0 < |t| < 2^{-\delta(k)}\), the difference \(\bigl(u(x + t, y) - u(x, y)\bigr)/t\) is within \(2^{-k}\) of \(p\) as a real-modulus comparison.
\end{definition}
\leandef{BEDC.Derived.ComplexDiffUp.PartialReal}

\begin{definition}[One-variable imaginary partial derivative]
\label{def:partial-imag}
\(\mathsf{PartialImag}(u, z, q)\) is defined analogously with respect to the imaginary coordinate: the difference \(\bigl(u(x, y + s) - u(x, y)\bigr)/s\) approaches \(q\) under the modulus \(\delta\).
\end{definition}
\leandef{BEDC.Derived.ComplexDiffUp.PartialImag}

\begin{lemma}[Uniqueness of real-coordinate partials]
\label{lem:partial-unique}
The real and imaginary partial derivatives of \autoref{def:partial-real} and \autoref{def:partial-imag} are unique up to \(\hsame\), and respect \(\hsame\) on the input position \(z\).
\end{lemma}
\leanstmt{BEDC.Derived.ComplexDiffUp.partial\_unique}
\begin{proof}
The argument is the one-variable specialization of \autoref{lem:cplx-diff-at-unique}: triangle-inequality combination of the two moduli halves the precision budget.
\end{proof}

\section{Cauchy--Riemann equations, constructive form}

\begin{theorem}[Cauchy--Riemann, forward direction]
\label{thm:cauchy-riemann-forward}
If \(\mathsf{CplxDiffAt}(f, z, f')\) holds with \(f = u + i \cdot v\) and \(f' = a + i \cdot b\), then there exist explicit moduli witnessing all four real-coordinate partial derivatives \(\partial_x u\), \(\partial_x v\), \(\partial_y u\), \(\partial_y v\) at \(z\), and the two coupling equations
\(\hsame(\partial_x u, \partial_y v)\) and \(\hsame(\partial_x v, -\partial_y u)\)
hold up to \(\hsame\) on real histories.
\end{theorem}
\leanstmt{BEDC.Derived.ComplexDiffUp.cauchy\_riemann\_forward}
\begin{proof}[Sketch]
Specializing \(\mathsf{CplxDiffAt}(f, z, f')\) to a real \(h\) (i.e., \(h = (t, 0)\)) yields \(\mathsf{PartialReal}\) for both \(u\) and \(v\) at \(z\), with witnesses \(\partial_x u = a\) and \(\partial_x v = b\). Specializing to a purely imaginary \(h = (0, s)\) and dividing by \(i\) yields \(\mathsf{PartialImag}\) for both \(u\) and \(v\), with witnesses \(\partial_y u = -b\) and \(\partial_y v = a\). The coupling \(\hsame(\partial_x u, \partial_y v)\) follows directly from \(a = a\), and \(\hsame(\partial_x v, -\partial_y u)\) from \(b = -(-b)\). Both are uniqueness-up-to-\(\hsame\) reductions via \autoref{lem:partial-unique}.
\end{proof}

\begin{theorem}[Cauchy--Riemann, backward direction]
\label{thm:cauchy-riemann-backward}
Suppose \(f = u + i \cdot v\) admits the four real-coordinate partial derivatives at \(z\) with explicit moduli, and the two coupling equations \(\hsame(\partial_x u, \partial_y v)\) and \(\hsame(\partial_x v, -\partial_y u)\) hold. If in addition the partials are continuous at \(z\) with an explicit local modulus, then \(\mathsf{CplxDiffAt}(f, z, \partial_x u + i \cdot \partial_x v)\) holds, with an explicit composite modulus.
\end{theorem}
\leanstmt{BEDC.Derived.ComplexDiffUp.cauchy\_riemann\_backward}
\begin{proof}[Sketch]
For complex \(h = (t, s)\), the difference \(f(z+h) - f(z)\) decomposes via the mean-value form (one-variable) along the path \((x, y) \to (x + t, y) \to (x + t, y + s)\). Each segment contributes a partial-derivative term up to an explicit error controlled by the continuity modulus. The two coupling equations are exactly what is needed to reassemble the result into the complex-linear form \((\partial_x u + i \cdot \partial_x v) \cdot h\), giving the complex difference quotient with an explicit error bound.
\end{proof}

\begin{lemma}[Cauchy--Riemann is an iff under continuity]
\label{lem:cauchy-riemann-iff}
Under the hypothesis that \(f\) has continuous real-coordinate partial derivatives at \(z\), \(\mathsf{CplxDiffAt}(f, z, f')\) is equivalent to the conjunction of the two coupling equations of \autoref{thm:cauchy-riemann-forward}.
\end{lemma}
\leanstmt{BEDC.Derived.ComplexDiffUp.cauchy\_riemann\_iff}
\begin{proof}
Combine \autoref{thm:cauchy-riemann-forward} and \autoref{thm:cauchy-riemann-backward}.
\end{proof}

\section{The certificate}

\begin{definition}[Complex-differentiability source specification]
\label{def:cplx-diff-source-specification}
The source specification carries a complex function-history \(f\), a complex history \(z\), and a candidate derivative \(f'\), constrained by \(\mathsf{CplxDiffAt}(f, z, f')\) of \autoref{def:cplx-diff-at}. The generator is the explicit modulus \(\delta\).
\end{definition}
\leandef{BEDC.Derived.ComplexDiffUp.CplxDiffSourceSpec}

\begin{definition}[Complex-differentiability pattern specification]
\label{def:cplx-diff-pattern-specification}
The pattern of \(\ComplexDiffUp\) is the explicit difference-quotient witness of \autoref{def:cplx-diff-quot}: for an input \((f, z, h)\) with \(\mathsf{CplxNonZero}(h)\), the pattern is the unique \(q\) with \(\mathsf{CplxDiffQuot}(f, z, h, q)\), parametrically in the precision \(k\).
\end{definition}
\leandef{BEDC.Derived.ComplexDiffUp.CplxDiffPatternSpec}

\begin{definition}[Complex-differentiability classifier specification]
\label{def:cplx-diff-classifier-specification}
The classifier on the differentiability carrier is \(\hsame\) lifted to the triple \((f, z, f')\): two triples classify together iff each component is \(\hsame\)-equal as a history (with the function-history part compared on its rational-pair representation).
\end{definition}
\leandef{BEDC.Derived.ComplexDiffUp.CplxDiffClassifierSpec}

\begin{definition}[Complex-differentiability stability certificate]
\label{def:cplx-diff-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item closure of \(\mathsf{CplxDiffAt}\) under \(\hsame\) on each input slot (\autoref{lem:cplx-diff-at-hsame});
  \item uniqueness of the derivative up to \(\hsame\) (\autoref{lem:cplx-diff-at-unique});
  \item compatibility of the difference-quotient pattern with \(\hsame\) (\autoref{lem:cplx-diff-quot-hsame}).
\end{enumerate}
\end{definition}
\leandef{BEDC.Derived.ComplexDiffUp.CplxDiffStabilityCert}

\begin{definition}[Complex-differentiability ledger policy]
\label{def:cplx-diff-ledger-policy}
The ledger policy records:
\begin{enumerate}
  \item sum, scalar, product, and chain closure (\autoref{lem:cplx-diff-sum}--\autoref{lem:cplx-diff-chain});
  \item the Cauchy--Riemann coupling (\autoref{thm:cauchy-riemann-forward}, \autoref{thm:cauchy-riemann-backward}, \autoref{lem:cauchy-riemann-iff});
  \item explicit modulus pasting laws so that any finite combination of differentiable pieces yields an explicit composite modulus.
\end{enumerate}
\end{definition}
\leandef{BEDC.Derived.ComplexDiffUp.CplxDiffLedgerPolicy}

\begin{theorem}[Concrete \(\ComplexDiffUp\) naming certificate]
\label{thm:complex-diff-namecert}
The data \((\mathsf{CplxDiffAt}, \mathsf{CplxDiffQuot}, \hsame, \mathsf{CplxDiffStabilityCert}, \mathsf{CplxDiffLedgerPolicy})\) of \autoref{def:cplx-diff-source-specification}--\autoref{def:cplx-diff-ledger-policy} forms a naming certificate over \(\Hist\) for the complex-differentiability interface \(\ComplexDiffUp\). The certificate is constructive (every modulus is an explicit unary function), unique-up-to-\(\hsame\) on the derivative slot, and closed under sum, scalar, product, and chain rule.
\end{theorem}
\leanstmt{BEDC.Derived.ComplexDiffUp.complex\_diff\_namecert}
\begin{proof}[Sketch]
The five fields satisfy the closure laws of the naming-certificate framework (\autoref{ch:core-typed-naming-certificates}). Constructivity is explicit in the modulus data of \autoref{def:cplx-diff-at}. Uniqueness is \autoref{lem:cplx-diff-at-unique}. The four closure laws are \autoref{lem:cplx-diff-sum}--\autoref{lem:cplx-diff-chain}. The Cauchy--Riemann coupling sits inside the ledger policy as a derived fact, not as an additional axiom. The certificate uses no \(\mathsf{Classical.choice}\), no \(\mathsf{Quot.sound}\), and no \(\mathsf{propext}\), so the certificate's Lean targets fall under the project's axiom-purity gate.
\end{proof}

\begin{theorem}[Semantic \(\ComplexDiffUp\) naming certificate]
\label{thm:complex-diff-semantic-namecert}
The concrete certificate of \autoref{thm:complex-diff-namecert} satisfies the semantic naming-certificate schema of \(\NameCert\) (\autoref{ch:core-typed-naming-certificates}): the carrier is a \(\Hist\)-predicate on triples \((f, z, f')\), the classifier is a restriction of \(\hsame\), and every closure law is given by an explicit constructive proof. The certificate uses no \(\mathsf{Classical.choice}\), no \(\mathsf{Quot.sound}\), and no \(\mathsf{propext}\), so the certificate's Lean targets fall under the project's axiom-purity gate.
\end{theorem}
\leanstmt{BEDC.Derived.ComplexDiffUp.complex\_diff\_semantic\_namecert}
\begin{proof}[Sketch]
Direct verification of the four \(\NameCert\) field types using the constructions of this chapter, together with transport along \(\hsame\) on each input slot.
\end{proof}
