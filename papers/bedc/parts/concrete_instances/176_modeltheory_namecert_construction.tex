\chapter{A Concrete Naming Certificate for $\ModelTheoryUp$}
\label{ch:concrete-instances-modeltheory-namecert}

The model-theory interface $\ModelTheoryUp$ packages a structure for a first-order signature, satisfaction of formulas, elementary equivalence, and elementary substructures. It depends on $\NameCert_{\FirstOrderUp}$ (\autoref{ch:concrete-instances-firstorder-namecert}).

\section{Carrier definition}
\label{sec:modeltheory-carrier}

\begin{definition}[ModelTheoryUp BHist structure packet]\label{def:modeltheory-bhist-structure-packet}
A $\ModelTheoryUp$ source packet is a finite displayed $\mathsf{BHist}$ tuple
$(S,A,\mathsf{Val},\mathsf{Sat},\mathsf{Elem},\rho,\lambda)$. The row $S$ is
the $\FirstOrderUp$ signature and formula row, $A$ is the displayed structure
row, $\mathsf{Val}$ records variable-assignment observations, $\mathsf{Sat}$
records the satisfaction ledgers for named formulas and assignments, and
$\mathsf{Elem}$ records elementary-equivalence and elementary-substructure
faces. The row $\rho$ is the $\Pkg$ provenance tying the packet to the visible
certificate, and $\lambda$ is the $\Cont$ and $\hsame$ transport ledger.

Acceptance of the packet exports only these displayed rows. A host
satisfaction relation, an ambient model equality, or an external comparison
of structures is not part of the carrier.
\end{definition}

\begin{definition}[ModelTheoryUp satisfaction ledger classifier]\label{def:modeltheory-satisfaction-ledger-classifier}
Two $\ModelTheoryUp$ source packets classify the same row exactly when
$\hsame$ transports their $\FirstOrderUp$ signature rows, displayed carrier
rows, valuation rows, satisfaction ledgers, elementary-equivalence rows, and
$\Pkg$ provenance through the same $\Cont$ ledger. The comparison is therefore
rowwise: formula and substitution reads move through the $\FirstOrderUp$
source, assignments move through $\mathsf{Val}$, truth records move through
$\mathsf{Sat}$, and elementary faces move through $\mathsf{Elem}$.
\end{definition}

\section{The certificate}
\label{sec:modeltheory-certificate}

\begin{theorem}[ModelTheoryUp FirstOrder dependency surface]\label{thm:modeltheory-firstorder-dependency-surface}
Every admitted $\ModelTheoryUp$ packet projects its signature row,
term/formula grammar row, and substitution-read rows through
$\NameCert_{\FirstOrderUp}$. The structure, valuation, satisfaction, and
elementary rows may consume those projected first-order endpoints, but they
do not export an independent first-order syntax source.
\end{theorem}
\begin{proof}
Invert the carried packet of
\autoref{def:modeltheory-bhist-structure-packet}. The first projection is the
row $S$, so the available symbols, terms, formulas, and substitution reads
are exactly the rows already visible from $\NameCert_{\FirstOrderUp}$.

Project the displayed structure row $A$ and the valuation row
$\mathsf{Val}$. Each assignment observation names variables and formula
endpoints by referring back to $S$, so no separate grammar or substitution
operation appears in the model-theory carrier.

Finally project the satisfaction and elementary ledgers. Their formula
occurrences are read through the same $S$ row, and their provenance is the
same $\Pkg$ row $\rho$. Repacking these projections gives the admitted
$\ModelTheoryUp$ packet with the first-order dependency surface unchanged.
\end{proof}

\begin{theorem}[ModelTheoryUp elementary transport]\label{thm:modeltheory-elementary-equivalence-transport}
If two admitted $\ModelTheoryUp$ packets are related by $\hsame$ transport of
their $\FirstOrderUp$ rows and satisfaction ledgers, then their
elementary-equivalence and elementary-substructure rows transport as
displayed $\mathsf{Elem}$ rows. This transport uses no host model equality.
\end{theorem}
\begin{proof}
Start with the classifier relation of
\autoref{def:modeltheory-satisfaction-ledger-classifier}. It supplies the
$\hsame$ rows for the first-order source, the displayed structure row, the
valuation ledger, the satisfaction ledger, the elementary ledger, and the
$\Pkg$ provenance.

Use the transported $\FirstOrderUp$ row to move formula endpoints and
substitution reads. Use the transported $\mathsf{Sat}$ row to move each
satisfaction record for the corresponding assignment observation.

The elementary-equivalence and elementary-substructure faces are finite
records in $\mathsf{Elem}$. Project those records from the source packet,
replace each formula and satisfaction reference by the transported row just
obtained, and repack them as the target $\mathsf{Elem}$ row. No step compares
ambient carriers by a host equality; every comparison is one of the displayed
$\hsame$ rows.
\end{proof}

\begin{theorem}[ModelTheoryUp NameCert obligation surface]\label{thm:modeltheory-namecert-obligation-surface}
The public $\NameCert_{\ModelTheoryUp}$ obligation surface consists exactly of
the carrier of \autoref{def:modeltheory-bhist-structure-packet}, the
classifier of \autoref{def:modeltheory-satisfaction-ledger-classifier}, the
first-order dependency surface of
\autoref{thm:modeltheory-firstorder-dependency-surface}, the elementary
transport of \autoref{thm:modeltheory-elementary-equivalence-transport},
stability transport, the satisfaction exactness ledger, the
elementary-equivalence ledger, and no additional exported field.
\end{theorem}
\begin{proof}
Collect the carrier rows first. The finite $\mathsf{BHist}$ packet provides
$S$, $A$, $\mathsf{Val}$, $\mathsf{Sat}$, $\mathsf{Elem}$, $\rho$, and
$\lambda$, while the first-order dependency theorem identifies all syntax and
substitution reads as projections through $\NameCert_{\FirstOrderUp}$.

Next collect the classifier rows. The satisfaction ledger classifier gives
rowwise $\hsame$ transport for the first-order source, the structure carrier,
valuation observations, satisfaction records, elementary faces, and
provenance. The elementary transport theorem repacks the transported
satisfaction data into elementary-equivalence and elementary-substructure
rows.

It remains to check public coverage. A consumer of
$\NameCert_{\ModelTheoryUp}$ can ask for a carrier row, a classifier row,
stability under $\hsame$, exact readback of satisfaction records, or the
elementary ledger. Each request is one of the collected rows, and none exposes
a host satisfaction predicate, an ambient model equality, or another public
field outside the displayed packet.
\end{proof}

\begin{definition}[ModelTheoryUp assignment observation row]\label{def:modeltheory-assignment-observation-row}
A $\ModelTheoryUp$ assignment observation row is the displayed
$\mathsf{BHist}$ entry

$$
(S,\varphi,x,\mathsf{Val}_{x,\varphi},\rho_{\mathsf{Val}},
\lambda_{\mathsf{Val}})
$$

whose variable endpoint $x$ and formula endpoint $\varphi$ are read only
through $\NameCert_{\FirstOrderUp}$. The valuation component
$\mathsf{Val}_{x,\varphi}$ is the visible assignment row for that endpoint
pair, $\lambda_{\mathsf{Val}}$ records the $\Cont$ and $\hsame$ transport
available for the observation, and $\rho_{\mathsf{Val}}$ is the $\Pkg$
provenance tying the row back to the admitted packet of
\autoref{def:modeltheory-bhist-structure-packet}.
\end{definition}

\begin{theorem}[ModelTheoryUp satisfaction exactness row]\label{thm:modeltheory-satisfaction-exactness-row}
Every accepted $\ModelTheoryUp$ satisfaction record is exactly the visible
$\mathsf{Sat}$ row for a formula endpoint read through
$\NameCert_{\FirstOrderUp}$ and an assignment observation of
\autoref{def:modeltheory-assignment-observation-row}. Conversely, every such
visible $\mathsf{Sat}$ row repacks as the satisfaction record for the same
admitted packet. No host satisfaction predicate is exported.
\end{theorem}
\begin{proof}
Unpack the finite packet of
\autoref{def:modeltheory-bhist-structure-packet}. The projections expose the
first-order row $S$, the valuation ledger $\mathsf{Val}$, the satisfaction
ledger $\mathsf{Sat}$, and the provenance row $\rho$.

The formula endpoint is read from $S$ through
$\NameCert_{\FirstOrderUp}$, while the assignment endpoint is read from the
displayed valuation row described in
\autoref{def:modeltheory-assignment-observation-row}. Thus the satisfaction
record has no data source other than the visible pair of endpoints and the
packet provenance.

Repack those projections as the same $\mathsf{Sat}$ row with its $\Pkg$,
$\Cont$, and $\hsame$ witnesses. This gives the exact readback direction, and
the converse is the same repacking step starting from the displayed row. At
no point is a semantic satisfaction relation outside the $\mathsf{BHist}$
packet introduced.
\end{proof}

\begin{theorem}[ModelTheoryUp elementary substructure boundary]\label{thm:modeltheory-elementary-substructure-boundary}
The elementary-equivalence and elementary-substructure faces of an admitted
$\ModelTheoryUp$ packet are finite displayed $\mathsf{Elem}$ rows transported
through the satisfaction exactness row and $\hsame$. Every formula occurrence
in those faces is a $\NameCert_{\FirstOrderUp}$ endpoint, and every assignment
occurrence is a transported valuation row.
\end{theorem}
\begin{proof}
Begin with the classifier of
\autoref{def:modeltheory-satisfaction-ledger-classifier}. It supplies the
rowwise $\hsame$ transports for the first-order source, valuation ledger,
satisfaction ledger, elementary ledger, and $\Pkg$ provenance.

Apply \autoref{thm:modeltheory-satisfaction-exactness-row} to each
satisfaction record referenced by an elementary face. This replaces the
record by the visible $\mathsf{Sat}$ row for a named formula endpoint and a
named assignment observation.

It remains to move the elementary face itself. Replace every formula
reference by the transported first-order row and every assignment reference
by the transported $\mathsf{Val}$ row, then repack the finite face as the
target $\mathsf{Elem}$ row. The boundary is therefore a displayed ledger
boundary, not an ambient elementary-substructure predicate.
\end{proof}

\begin{theorem}[ModelTheoryUp downstream consumer row]\label{thm:modeltheory-downstream-consumer-row}
A downstream consumer of $\NameCert_{\ModelTheoryUp}$ can ask only for the
carrier packet, the satisfaction-ledger classifier, the assignment
observation row, the satisfaction exactness row, the elementary substructure
boundary, stability transport, and $\Pkg$ provenance. Any request for a host
truth predicate, an ambient model equality, or an external elementary
comparison lies outside the public obligation surface.
\end{theorem}
\begin{proof}
Project the public surface of
\autoref{thm:modeltheory-namecert-obligation-surface}. The visible rows are
the carrier packet, the classifier, first-order dependency rows, elementary
transport rows, stability transport, satisfaction readback, and provenance.

Refine the satisfaction readback by
\autoref{thm:modeltheory-satisfaction-exactness-row}. Refine the assignment
input by \autoref{def:modeltheory-assignment-observation-row}, and refine the
elementary request by
\autoref{thm:modeltheory-elementary-substructure-boundary}.

These refinements exhaust the consumer-facing rows. Since each row is a
projection from the displayed packet or a repacking through $\Cont$,
$\hsame$, and $\Pkg$, no consumer receives an operation beyond the listed
certificate surface.
\end{proof}
