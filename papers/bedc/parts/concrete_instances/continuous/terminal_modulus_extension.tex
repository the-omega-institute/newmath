\begin{theorem}[Continuous function carrier terminal modulus extension]
\label{thm:continuous-function-carrier-terminal-modulus-extension}
For histories
$source,map,target,modulus,cert,extra,modulus',cert'$, a continuous-function
carrier remains a continuous-function carrier after the terminal certificate
is extended by one more modulus witness and the modulus ledger is composed:
$$
\mathsf{ContinuousFunctionCarrier}(source,map,target,modulus,cert)
\Longrightarrow
\mathsf{ContinuousModulusWitness}(cert,extra,cert')
\Longrightarrow
\ContR(modulus,extra,modulus')
\Longrightarrow
\mathsf{ContinuousFunctionCarrier}(source,map,target,modulus',cert').
$$
\end{theorem}
\begin{proof}
Unpack the given carrier. This exposes the graph ledger
$\ContR(source,map,target)$ and the terminal modulus-certificate ledger
$\ContR(target,modulus,cert)$, together with the unary carrier data for the
displayed histories.

Unpack the additional modulus witness
$\mathsf{ContinuousModulusWitness}(cert,extra,cert')$. It supplies
$\ContR(cert,extra,cert')$, so the two ledgers
$\ContR(target,modulus,cert)$ and $\ContR(cert,extra,cert')$ assemble the chain
$\mathsf{ContinuousModulusChain}(target,modulus,extra,cert')$ with middle
history $cert$.

Apply Theorem~\ref{thm:continuous-function-carrier-modulus-chain-replacement}
to the original carrier, this chain, and the displayed composition ledger
$\ContR(modulus,extra,modulus')$. The replacement theorem repacks the preserved
graph data with the composite terminal modulus certificate, giving the displayed
carrier conclusion.

The ten-field certificate fit is the following. The traditional target is
terminal modulus extension for continuous maps. The BEDC source is
$\mathsf{BHist}$ with $\ContR$. The carrier is
$\mathsf{ContinuousFunctionCarrier}$ and the classifier is $\hsame$. The
stability obligation is preservation under one terminal modulus extension. The
ledger is source, map, target, original modulus and certificate, extra modulus,
composite modulus, and terminal certificate. The core theorem is this closure,
exactness is the displayed carrier conclusion, the standard bridge is deferred,
and the status is $\mathsf{ProofTarget}$.
\end{proof}

\begin{theorem}[Continuous function carrier visible terminal modulus extension]
\label{thm:continuous-function-carrier-visible-terminal-modulus-extension}
For histories $p,q,source,map,target,modulus,cert,extra,modulus',cert'$,
visible continuous-function carrier data remain carrier data after the central
terminal certificate is extended and the terminal modulus ledger is composed:
$$
\mathsf{ContinuousFunctionCarrier}(p\cdot source,map,p\cdot target,
modulus\cdot q,(p\cdot cert)\cdot q)
\Longrightarrow
\mathsf{ContinuousModulusWitness}(cert,extra,cert')
\Longrightarrow
\ContR(modulus,extra,modulus')
\Longrightarrow
\mathsf{ContinuousFunctionCarrier}(p\cdot source,map,p\cdot target,
modulus'\cdot q,(p\cdot cert')\cdot q).
$$
\end{theorem}
\begin{proof}
Peel the visible presentation by
Theorem~\ref{thm:continuous-function-carrier-visible-modulus-context-iff}. This
recovers the unary context histories $\mathsf{UnaryHistory}(p)$ and
$\mathsf{UnaryHistory}(q)$ together with the central carrier
$\mathsf{ContinuousFunctionCarrier}(source,map,target,modulus,cert)$.

Apply
Theorem~\ref{thm:continuous-function-carrier-terminal-modulus-extension} to the
central carrier, the witness
$\mathsf{ContinuousModulusWitness}(cert,extra,cert')$, and the composition
ledger $\ContR(modulus,extra,modulus')$. The conclusion is the central carrier
$\mathsf{ContinuousFunctionCarrier}(source,map,target,modulus',cert')$.

Use the reverse direction of
Theorem~\ref{thm:continuous-function-carrier-visible-modulus-context-iff} with
the recovered unary histories. It repacks the same prefix $p$ and suffix $q$
around the central carrier, producing the displayed visible carrier conclusion.
\end{proof}
\leanchecked{BEDC.Derived.ContinuousUp.ContinuousFunctionCarrier\_visible\_terminal\_modulus\_extension}

\begin{theorem}[Continuous function carrier visible terminal modulus extension output determinacy]
\label{thm:continuous-function-carrier-visible-terminal-modulus-extension-output-determinacy}
For histories
$p,q,source,map,target,target',modulus,cert,extra,modulus',cert',cert''$,
suppose
$$
\mathsf{ContinuousFunctionCarrier}(p\cdot source,map,p\cdot target,
modulus\cdot q,(p\cdot cert)\cdot q)
$$
holds, together with
$\mathsf{ContinuousModulusWitness}(cert,extra,cert')$ and
$\ContR(modulus,extra,modulus')$. If the post-extension visible carrier
$$
\mathsf{ContinuousFunctionCarrier}(p\cdot source,map,p\cdot target',
modulus'\cdot q,(p\cdot cert'')\cdot q)
$$
also holds, then the visible target and terminal certificate are determined:
$$
\hsame(target,target')\land\hsame(cert',cert'').
$$
\end{theorem}
\begin{proof}
Apply
Theorem~\ref{thm:continuous-function-carrier-visible-terminal-modulus-extension}
to the original visible carrier, the central witness
$\mathsf{ContinuousModulusWitness}(cert,extra,cert')$, and the terminal modulus
ledger $\ContR(modulus,extra,modulus')$. This constructs the canonical
post-extension visible carrier with target $target$ and terminal certificate
$cert'$.

Compare that canonical carrier with the independently displayed post-extension
carrier by
Theorem~\ref{thm:continuous-function-carrier-visible-modulus-context-target-cert-deterministic}.
The two carrier presentations share $p$, $q$, $source$, $map$, and $modulus'$,
so the determinacy theorem returns exactly
$\hsame(target,target')\land\hsame(cert',cert'')$.
\end{proof}
\leanchecked{BEDC.Derived.ContinuousUp.ContinuousFunctionCarrier\_visible\_terminal\_modulus\_extension\_output\_determinacy}

\begin{theorem}[Continuous function carrier visible terminal modulus extension transitive]
\label{thm:continuous-function-carrier-visible-terminal-modulus-extension-transitive}
For histories
$p,q,source,map,target,modulus_0,cert_0,extra_1,modulus_1,cert_1,
extra_2,modulus_2,cert_2$, visible continuous-function carrier data remain
carrier data after two consecutive terminal certificate extensions and their
two terminal modulus ledgers:
$$
\mathsf{ContinuousFunctionCarrier}(p\cdot source,map,p\cdot target,
modulus_0\cdot q,(p\cdot cert_0)\cdot q)
\Longrightarrow
\mathsf{ContinuousModulusWitness}(cert_0,extra_1,cert_1)
\Longrightarrow
\ContR(modulus_0,extra_1,modulus_1)
\Longrightarrow
\mathsf{ContinuousModulusWitness}(cert_1,extra_2,cert_2)
\Longrightarrow
\ContR(modulus_1,extra_2,modulus_2)
\Longrightarrow
\mathsf{ContinuousFunctionCarrier}(p\cdot source,map,p\cdot target,
modulus_2\cdot q,(p\cdot cert_2)\cdot q).
$$
\end{theorem}
\leanchecked{BEDC.Derived.ContinuousUp.ContinuousFunctionCarrier\_visible\_terminal\_modulus\_extension\_transitive}
\begin{proof}
Apply
Theorem~\ref{thm:continuous-function-carrier-visible-terminal-modulus-extension}
to the displayed visible carrier, the witness
$\mathsf{ContinuousModulusWitness}(cert_0,extra_1,cert_1)$, and the first
composition ledger $\ContR(modulus_0,extra_1,modulus_1)$. This gives the
visible carrier with terminal modulus $modulus_1$ and terminal certificate
$cert_1$.

Apply the same theorem again to that intermediate carrier, the witness
$\mathsf{ContinuousModulusWitness}(cert_1,extra_2,cert_2)$, and the second
composition ledger $\ContR(modulus_1,extra_2,modulus_2)$. The result is the
visible carrier whose terminal modulus is $modulus_2$ and whose terminal
certificate is $cert_2$, which is the displayed conclusion.

The ten-field certificate fit is the following. The traditional target is
two-step terminal modulus extension for continuous maps. The BEDC source is
$\mathsf{BHist}$ with $\ContR$,
$\mathsf{ContinuousModulusWitness}$, and
$\mathsf{ContinuousFunctionCarrier}$. The carrier is
$\mathsf{ContinuousFunctionCarrier}$ and the classifier is $\hsame$. The
stability obligation is preservation under two consecutive terminal modulus
extensions. The ledger is the two extra modulus witnesses and the two composite
modulus rows. The core theorem is this closure, exactness is the displayed
carrier conclusion, the standard bridge is deferred, and the status is
$\mathsf{ProofTarget}$.
\end{proof}

\begin{theorem}[Continuous function carrier visible terminal modulus reassociation]
\label{thm:continuous-function-carrier-visible-terminal-modulus-reassociation}
For histories
$p,q,source,map,target,modulus_0,cert_0,extra_1,modulus_1,cert_1,
extra_2,extra_{12},modulus_2,modulus_{12},cert_2$, sequential two-step terminal
extension and the corresponding composite one-step terminal extension produce
the same visible terminal modulus data:
$$
\begin{array}{l}
\mathsf{ContinuousFunctionCarrier}(p\cdot source,map,p\cdot target,
modulus_0\cdot q,(p\cdot cert_0)\cdot q)
\Longrightarrow\\
\mathsf{ContinuousModulusWitness}(cert_0,extra_1,cert_1)
\Longrightarrow
\mathsf{ContinuousModulusWitness}(cert_1,extra_2,cert_2)
\Longrightarrow\\
\ContR(modulus_0,extra_1,modulus_1)
\Longrightarrow
\ContR(modulus_1,extra_2,modulus_2)
\Longrightarrow\\
\ContR(extra_1,extra_2,extra_{12})
\Longrightarrow
\ContR(modulus_0,extra_{12},modulus_{12})
\Longrightarrow\\
\mathsf{ContinuousFunctionCarrier}(p\cdot source,map,p\cdot target,
modulus_2\cdot q,(p\cdot cert_2)\cdot q)
\land\\
\mathsf{ContinuousFunctionCarrier}(p\cdot source,map,p\cdot target,
modulus_{12}\cdot q,(p\cdot cert_2)\cdot q)
\land\\
\hsame(modulus_2,modulus_{12}).
\end{array}
$$
\end{theorem}
\leanchecked{BEDC.Derived.ContinuousUp.ContinuousFunctionCarrier\_visible\_terminal\_modulus\_reassociation}
\begin{proof}
Apply
Theorem~\ref{thm:continuous-function-carrier-visible-terminal-modulus-extension-transitive}
to the displayed visible carrier, the two certificate-extension witnesses, and
the two sequential terminal modulus ledgers. This constructs the sequential
visible carrier with terminal data $(modulus_2,cert_2)$.

Next assemble the certificate witnesses
$\mathsf{ContinuousModulusWitness}(cert_0,extra_1,cert_1)$ and
$\mathsf{ContinuousModulusWitness}(cert_1,extra_2,cert_2)$ into
$\mathsf{ContinuousModulusChain}(cert_0,extra_1,extra_2,cert_2)$ with middle
history $cert_1$. By
Theorem~\ref{thm:continuous-modulus-chain-composite-closed} and the ledger
$\ContR(extra_1,extra_2,extra_{12})$, this gives the composite witness
$\mathsf{ContinuousModulusWitness}(cert_0,extra_{12},cert_2)$.

Apply
Theorem~\ref{thm:continuous-function-carrier-visible-terminal-modulus-extension}
to the original visible carrier, this composite witness, and
$\ContR(modulus_0,extra_{12},modulus_{12})$. The result is the composite
one-step visible carrier with terminal data $(modulus_{12},cert_2)$.

Finally use Theorem~\ref{thm:continuation-associativity} on
$\ContR(modulus_0,extra_1,modulus_1)$,
$\ContR(modulus_1,extra_2,modulus_2)$,
$\ContR(extra_1,extra_2,extra_{12})$, and
$\ContR(modulus_0,extra_{12},modulus_{12})$. It identifies the sequential and
composite terminal moduli, giving $\hsame(modulus_2,modulus_{12})$, and the
three displayed conclusions are then packed together.
\end{proof}

\label{thm:continuous-function-carrier-visible-terminal-modulus-reassociation-output-determinacy}
For histories
$p,q,source,map,target,target',modulus_0,cert_0,extra_1,modulus_1,cert_1,
extra_2,extra_{12},modulus_2,modulus_{12},cert_2,cert''$, the displayed
composite visible output is determined by the reassociated terminal modulus
construction:
$$
\begin{array}{l}
\mathsf{ContinuousFunctionCarrier}(p\cdot source,map,p\cdot target,
modulus_0\cdot q,(p\cdot cert_0)\cdot q)
\Longrightarrow\\
\mathsf{ContinuousModulusWitness}(cert_0,extra_1,cert_1)
\Longrightarrow
\mathsf{ContinuousModulusWitness}(cert_1,extra_2,cert_2)
\Longrightarrow\\
\ContR(modulus_0,extra_1,modulus_1)
\Longrightarrow
\ContR(modulus_1,extra_2,modulus_2)
\Longrightarrow\\

\mathsf{ContinuousFunctionCarrier}(p\cdot source,map,p\cdot target',

the four terminal modulus ledgers. This supplies the canonical composite

$\mathsf{ContinuousFunctionCarrier}(p\cdot source,map,p\cdot target,

$\hsame(modulus_2,modulus_{12})$.

Next compare that canonical composite carrier with the independently displayed
composite carrier
$\mathsf{ContinuousFunctionCarrier}(p\cdot source,map,p\cdot target',

Theorem~\ref{thm:continuous-function-carrier-visible-modulus-context-target-cert-deterministic},
the common visible source, map, and terminal modulus determine the public
target and certificate, giving $\hsame(target,target')$ and
$\hsame(cert_2,cert'')$.

Finally package those two target-certificate comparisons with the modulus
comparison obtained from reassociation. The result is the displayed triple
$\hsame(target,target')\land\hsame(cert_2,cert'')\land
\hsame(modulus_2,modulus_{12})$.
\end{proof}

\begin{theorem}[Continuous function carrier visible terminal modulus three-step reassociation]
\label{thm:continuous-function-carrier-visible-terminal-modulus-three-step-reassociation}
For histories
$p,q,source,map,target,modulus_0,cert_0,extra_1,modulus_1,cert_1,
extra_2,modulus_2,cert_2,extra_3,modulus_3,cert_3,extra_{12},
modulus_{12},extra_{123},modulus_{123}$, sequential three-step terminal
extension and the corresponding composite one-step terminal extension produce
the same visible terminal modulus data:
$$
\begin{array}{l}
\mathsf{ContinuousFunctionCarrier}(p\cdot source,map,p\cdot target,
modulus_0\cdot q,(p\cdot cert_0)\cdot q)
\Longrightarrow\\
\mathsf{ContinuousModulusWitness}(cert_0,extra_1,cert_1)
\Longrightarrow
\mathsf{ContinuousModulusWitness}(cert_1,extra_2,cert_2)
\Longrightarrow\\
\mathsf{ContinuousModulusWitness}(cert_2,extra_3,cert_3)
\Longrightarrow\\
\ContR(modulus_0,extra_1,modulus_1)
\Longrightarrow
\ContR(modulus_1,extra_2,modulus_2)
\Longrightarrow\\
\ContR(modulus_2,extra_3,modulus_3)
\Longrightarrow
\ContR(extra_1,extra_2,extra_{12})
\Longrightarrow\\
\ContR(modulus_0,extra_{12},modulus_{12})
\Longrightarrow
\ContR(extra_{12},extra_3,extra_{123})
\Longrightarrow\\
\ContR(modulus_0,extra_{123},modulus_{123})
\Longrightarrow\\
\mathsf{ContinuousFunctionCarrier}(p\cdot source,map,p\cdot target,
modulus_3\cdot q,(p\cdot cert_3)\cdot q)
\land\\
\mathsf{ContinuousFunctionCarrier}(p\cdot source,map,p\cdot target,
modulus_{123}\cdot q,(p\cdot cert_3)\cdot q)
\land\\
\hsame(modulus_3,modulus_{123}).
\end{array}
$$
\end{theorem}
\begin{proof}
Apply
Theorem~\ref{thm:continuous-function-carrier-visible-terminal-modulus-extension-transitive}
to the displayed visible carrier, the first two certificate-extension
witnesses, and the first two sequential terminal modulus ledgers. This gives
the visible carrier with terminal data $(modulus_2,cert_2)$. Apply
Theorem~\ref{thm:continuous-function-carrier-visible-terminal-modulus-extension}
to that carrier, the third certificate-extension witness, and
$\ContR(modulus_2,extra_3,modulus_3)$. The sequential visible carrier with
terminal data $(modulus_3,cert_3)$ follows.

Next compose the first two certificate witnesses. They form the chain
$\mathsf{ContinuousModulusChain}(cert_0,extra_1,extra_2,cert_2)$ with middle
history $cert_1$, so
Theorem~\ref{thm:continuous-modulus-chain-composite-closed} and
$\ContR(extra_1,extra_2,extra_{12})$ supply
$\mathsf{ContinuousModulusWitness}(cert_0,extra_{12},cert_2)$.

Compose that witness with the third one. The chain
$\mathsf{ContinuousModulusChain}(cert_0,extra_{12},extra_3,cert_3)$ with
middle history $cert_2$, together with
$\ContR(extra_{12},extra_3,extra_{123})$, gives
$\mathsf{ContinuousModulusWitness}(cert_0,extra_{123},cert_3)$ by the same
composite-closure theorem.

Apply
Theorem~\ref{thm:continuous-function-carrier-visible-terminal-modulus-extension}
to the original visible carrier, the composite witness just constructed, and
$\ContR(modulus_0,extra_{123},modulus_{123})$. This produces the composite
one-step visible carrier with terminal data $(modulus_{123},cert_3)$.

It remains to identify the two terminal moduli. First use
Theorem~\ref{thm:continuation-associativity} on
$\ContR(modulus_0,extra_1,modulus_1)$,
$\ContR(modulus_1,extra_2,modulus_2)$,
$\ContR(extra_1,extra_2,extra_{12})$, and
$\ContR(modulus_0,extra_{12},modulus_{12})$, obtaining
$\hsame(modulus_2,modulus_{12})$.

Transport the third terminal continuation along that comparison, carrying
$\ContR(modulus_2,extra_3,modulus_3)$ to the row
$\ContR(modulus_{12},extra_3,modulus_3)$. Then apply
Theorem~\ref{thm:continuation-associativity} to
$\ContR(modulus_0,extra_{12},modulus_{12})$, the transported row,
$\ContR(extra_{12},extra_3,extra_{123})$, and
$\ContR(modulus_0,extra_{123},modulus_{123})$. The resulting comparison is
$\hsame(modulus_3,modulus_{123})$, and the two visible carriers and this
comparison form the displayed triple.
\end{proof}
