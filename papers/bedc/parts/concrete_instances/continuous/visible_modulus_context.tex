\begin{theorem}[Continuous function carrier visible modulus context exactness]
\label{thm:continuous-function-carrier-visible-modulus-context-iff}
For histories $p,q,source,map,target,modulus,cert$, a visible prefix on the
source, target, and certificate ledgers together with a visible suffix on the
modulus and certificate ledgers is exactly unary evidence for the two context
histories together with the central continuous-function carrier:
$$
\mathsf{ContinuousFunctionCarrier}(p\cdot source,map,p\cdot target,
modulus\cdot q,(p\cdot cert)\cdot q)
\Longleftrightarrow
\mathsf{UnaryHistory}(p)\wedge\mathsf{UnaryHistory}(q)\wedge
\mathsf{ContinuousFunctionCarrier}(source,map,target,modulus,cert).
$$
\end{theorem}
\begin{proof}
Peel the right suffix from the modulus and certificate ledgers by modulus
suffix exactness. The remaining carrier has the common left prefix on source,
target, and certificate, so carrier prefix exactness recovers the unary prefix
and the central carrier. Conversely, rebuild the prefixed carrier first, then
reattach the modulus suffix.
\end{proof}
\leanchecked{BEDC.Derived.ContinuousUp.ContinuousFunctionCarrier\_visible\_modulus\_context\_iff}

\begin{theorem}[Continuous modulus chain visible context exactness]
\label{thm:continuous-modulus-chain-visible-context-iff}
For histories $p,q,source,first,second,target$, a visible prefix on the source
and target together with a visible suffix on the second modulus and target is
exactly unary evidence for the two contexts together with the central
two-link chain:
$$
\mathsf{ContinuousModulusChain}(p\cdot source,first,second\cdot q,
(p\cdot target)\cdot q)
\Longleftrightarrow
\mathsf{UnaryHistory}(p)\wedge\mathsf{UnaryHistory}(q)\wedge
\mathsf{ContinuousModulusChain}(source,first,second,target).
$$
\end{theorem}
\begin{proof}
Peel the right suffix from the second modulus and target by chain suffix
exactness. The remaining chain has the common left prefix on source and
target, so chain prefix exactness exposes unary $p$ and the central chain.
Conversely, rebuild the prefixed chain first, then reattach the suffix.
\end{proof}
\leanchecked{BEDC.Derived.ContinuousUp.ContinuousModulusChain\_visible\_context\_iff}

\begin{theorem}[Continuous modulus chain visible context second determinacy]
\label{thm:continuous-modulus-chain-visible-context-second-deterministic}
For histories $p,q,source,first,second,second',target$, if two visible-context
two-link modulus chains share the same visible source, first modulus, and
visible target,
$$
\mathsf{ContinuousModulusChain}(p\cdot source,first,second\cdot q,
(p\cdot target)\cdot q)
\quad\hbox{and}\quad
\mathsf{ContinuousModulusChain}(p\cdot source,first,second'\cdot q,
(p\cdot target)\cdot q),
$$
then $second\hsame second'$.
\end{theorem}
\begin{proof}
Apply the visible-context exactness statement to both chains. This removes the
common prefix and suffix contexts and exposes two central chains with the same
source, first modulus, and target. The ordinary second-modulus determinacy
theorem identifies the two central second moduli.
\end{proof}
\leanchecked{BEDC.Derived.ContinuousUp.ContinuousModulusChain\_visible\_context\_second\_deterministic}

\begin{theorem}[Continuous function carrier visible-context target-certificate determinacy]
\label{thm:continuous-function-carrier-visible-modulus-context-target-cert-deterministic}
For histories $p,q,source,map,target,target',modulus,cert,cert'$, if both
visible-modulus-context carrier presentations
$$
\mathsf{ContinuousFunctionCarrier}(p\cdot source,map,p\cdot target,
modulus\cdot q,(p\cdot cert)\cdot q)
$$
and
$$
\mathsf{ContinuousFunctionCarrier}(p\cdot source,map,p\cdot target',
modulus\cdot q,(p\cdot cert')\cdot q)
$$
hold, then the public target and certificate ledgers are determined by the
central presentation:
$$
\hsame(target,target')\land\hsame(cert,cert').
$$
\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:continuous-function-carrier-visible-modulus-context-iff}
to the first visible carrier. This recovers the unary context histories
$\mathsf{UnaryHistory}(p)$ and $\mathsf{UnaryHistory}(q)$ together with the
central witness
$\mathsf{ContinuousFunctionCarrier}(source,map,target,modulus,cert)$.

Apply the same exactness theorem to the second visible carrier. The common
visible prefix and suffix are peeled in the same way, producing the central
witness
$\mathsf{ContinuousFunctionCarrier}(source,map,target',modulus,cert')$.

Now apply
Theorem~\ref{thm:continuous-function-carrier-target-cert-deterministic} to the
two central carrier witnesses. It gives the displayed
$\hsame(target,target')$ and $\hsame(cert,cert')$ conclusions, which are already
the public target and certificate comparisons required here.

The ten-field certificate fit is the following. The traditional target is
public target-certificate determinacy for visible continuous-function carrier
presentations. The BEDC source is $\mathsf{BHist}$ equipped with
$\mathsf{Cont}$. The carrier is $\mathsf{ContinuousFunctionCarrier}$ and the classifier is
$\hsame$. The stability obligation is visible-context public readback. The
ledger is source, map, target, modulus, and certificate histories together
with the visible prefix and suffix contexts. The core theorem is this result.
The exactness predicate is the displayed pair of $\hsame$ conclusions. The
standard bridge is deferred, and the status is $\mathsf{ProofTarget}$.
\end{proof}

\begin{theorem}[Continuous function carrier visible-context graph-modulus determinacy]
\label{thm:continuous-function-carrier-visible-modulus-context-graph-modulus-deterministic}
For histories $p,q,source,map,map',target,modulus,modulus',cert$, if both
visible-modulus-context carrier presentations
$$
\mathsf{ContinuousFunctionCarrier}(p\cdot source,map,p\cdot target,
modulus\cdot q,(p\cdot cert)\cdot q)
$$
and
$$
\mathsf{ContinuousFunctionCarrier}(p\cdot source,map',p\cdot target,
modulus'\cdot q,(p\cdot cert)\cdot q)
$$
hold, then the public graph and modulus ledgers are simultaneously determined:
$$
\hsame(map,map')\land\hsame(modulus,modulus').
$$
\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:continuous-function-carrier-visible-modulus-context-iff}
to both visible carrier hypotheses. The common displayed prefix and suffix are
peeled away, leaving central carriers over
$(source,map,target,modulus,cert)$ and
$(source,map',target,modulus',cert)$.

Apply
Theorem~\ref{thm:continuous-function-carrier-graph-modulus-cont-readback} to
the two central carriers. This exposes graph ledgers
$\mathsf{Cont}(source,map,target)$ and
$\mathsf{Cont}(source,map',target)$, and certificate ledgers
$\mathsf{Cont}(target,modulus,cert)$ and
$\mathsf{Cont}(target,modulus',cert)$.

Continuation left cancellation on the two graph ledgers gives
$\hsame(map,map')$. The same cancellation principle on the two
target-certificate ledgers gives $\hsame(modulus,modulus')$, and the two
comparisons are paired as the displayed conclusion.

The ten-field certificate fit is: traditional target visible
continuous-function determinacy; BEDC source $\mathsf{BHist}$ with
$\mathsf{Cont}$; carrier $\mathsf{ContinuousFunctionCarrier}$; classifier
$\hsame$; stability obligation visible-context public readback; ledger source,
map, target, modulus, certificate, prefix, and suffix; core theorem this
statement; exactness the displayed pair of $\hsame$ conclusions; standard
bridge deferred; status $\mathsf{ProofTarget}$.
\end{proof}
