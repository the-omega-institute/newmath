\section{The certificate}

\begin{definition}[ContinuousMap source specification]
\label{def:continuous-source-specification}
The source is a pair of certified metric spaces and the totally bounded subset discipline on the domain.
\end{definition}
\leandef{BEDC.Derived.ContinuousUp.ContinuousSourceSpec}

\begin{definition}[ContinuousMap pattern specification]
\label{def:continuous-pattern-specification}
The pattern is graph-with-modulus formation over certified metric spaces and totally bounded source witnesses. A witness supplies the graph relation, the domain and codomain classifier ledgers, and the unary modulus relation that converts a target precision request into the source precision obligation needed for the graph.
\end{definition}
\leandef{BEDC.Derived.ContinuousUp.ContinuousPatternSpec}
\leanchecked{BEDC.Derived.ContinuousUp.ContinuousPatternSpec\_comp\_carrier}

\begin{definition}[ContinuousMap classifier specification]
\label{def:continuous-classifier-specification}
The classifier relates two continuous-map witnesses when their graph outputs agree under the codomain classifier on every pair of source histories related by the source classifier, and when their unary modulus ledgers mutually refine each other for every precision history admitted by the precision classifier.
\end{definition}

\begin{theorem}[Continuous modulus chain factorization]
\label{thm:continuous-modulus-chain-factorizes}
For histories $s,\delta_1,\delta_2,t$, assume the two-link chain
$$
\mathsf{ContinuousModulusChain}(s,\delta_1,\delta_2,t).
$$
Then there is a history $m$ such that both one-step links are certified:
$$
\mathsf{ContinuousModulusWitness}(s,\delta_1,m)
\quad\text{and}\quad
\mathsf{ContinuousModulusWitness}(m,\delta_2,t).
$$
\end{theorem}
\begin{proof}
Unpack the chain into unary data for $s$, $\delta_1$, $\delta_2$, and $t$,
together with a middle history $m$ and continuation ledgers
$s\cdot\delta_1=m$ and $m\cdot\delta_2=t$.

The first ledger and the unary source and modulus data assemble
$$
\mathsf{ContinuousModulusWitness}(s,\delta_1,m).
$$

Unary continuation closure then makes the middle history unary. The second
ledger and the unary $\delta_2$ and target data assemble
$$
\mathsf{ContinuousModulusWitness}(m,\delta_2,t).
$$

Taking this middle history as the existential witness gives the displayed
factorization of the chain into its two certified one-step modulus witnesses.
\end{proof}
\leanchecked{BEDC.Derived.ContinuousUp.ContinuousModulusChain\_factorizes}

\begin{theorem}[Continuous modulus chain middle deterministic factorization]
\label{thm:continuous-modulus-chain-middle-deterministic-factorization}
For histories $s,\delta_1,\delta_2,t$, a continuous modulus chain determines a
middle history $m$ such that the two one-step modulus witnesses factor the
chain, and every other first-leg middle endpoint is the same BEDC history:
$$
\mathsf{ContinuousModulusChain}(s,\delta_1,\delta_2,t)
\Longrightarrow
\exists m.\;
\mathsf{ContinuousModulusWitness}(s,\delta_1,m)\land
\mathsf{ContinuousModulusWitness}(m,\delta_2,t)
$$
$$
\land\;
\forall m'.\;
\ContR(s,\delta_1,m')\Longrightarrow
\ContR(m',\delta_2,t)\Longrightarrow
\hsame(m,m').
$$
\end{theorem}
\begin{proof}
Unpack the chain into unary data and a displayed middle history $m$ with
continuation ledgers $s\cdot\delta_1=m$ and $m\cdot\delta_2=t$. As in
Theorem~\ref{thm:continuous-modulus-chain-factorizes}, the first ledger gives
the first witness and unary continuation closure supplies unary data for $m$,
so the second ledger gives the second witness. If another $m'$ satisfies the
same first continuation ledger, $\mathsf{Cont}$ determinacy for
$s\cdot\delta_1$ yields $\hsame(m,m')$.
\end{proof}
\leanchecked{BEDC.Derived.ContinuousUp.ContinuousModulusChain\_middle\_deterministic\_factorization}

\begin{theorem}[Continuous modulus chain target determinacy]
\label{thm:continuous-modulus-chain-target-deterministic}
For histories $source,first,second,target,target'$, two modulus-chain witnesses
with the same source and the same two modulus histories determine the same
target up to $\hsame$:
$$
\mathsf{ContinuousModulusChain}(source,first,second,target)
\Longrightarrow
\mathsf{ContinuousModulusChain}(source,first,second,target')
\Longrightarrow
\hsame(target,target').
$$
\end{theorem}
\begin{proof}
Unpack both chains. The first continuation ledgers share $source$ and $first$,
so continuation determinacy identifies the two middle histories up to
$\hsame$. Transport the second continuation comparison along this middle
sameness, keeping the common $second$ history fixed; continuation respect for
$\hsame$ then gives $\hsame(target,target')$.
\end{proof}
\leanchecked{BEDC.Derived.ContinuousUp.ContinuousModulusChain\_target\_deterministic}

\begin{theorem}[Continuous modulus chain result determinacy]
\label{thm:continuous-modulus-chain-result-deterministic}
For histories $s,\delta_1,\delta_2,t,t'$, two continuous modulus chains with the
same source and modulus ledgers determine the same target history:
$$
\mathsf{ContinuousModulusChain}(s,\delta_1,\delta_2,t)
\Longrightarrow
\mathsf{ContinuousModulusChain}(s,\delta_1,\delta_2,t')
\Longrightarrow
\hsame(t,t').
$$
\end{theorem}
\begin{proof}
Unpack the two chains into middle histories $m$ and $m'$ with first ledgers
$s\cdot\delta_1=m$ and $s\cdot\delta_1=m'$, and second ledgers
$m\cdot\delta_2=t$ and $m'\cdot\delta_2=t'$. Determinacy of the first
continuation gives $\hsame(m,m')$. Continuation compatibility with this middle
history equality and reflexivity of $\delta_2$ transports the two second ledgers
to obtain $\hsame(t,t')$.
\end{proof}

\begin{theorem}[Continuous modulus chain source determinacy]
\label{thm:continuous-modulus-chain-source-deterministic}
For histories $source,source',first,second,target$, two modulus-chain witnesses
with the same two modulus histories and the same target determine the same
source up to $\hsame$:
$$
\mathsf{ContinuousModulusChain}(source,first,second,target)
\Longrightarrow
\mathsf{ContinuousModulusChain}(source',first,second,target)
\Longrightarrow
\hsame(source,source').
$$
\end{theorem}
\begin{proof}
Unpack both chains into middle histories and continuation ledgers. The second
continuation ledgers have common suffix $second$ and common target, so right
cancellation identifies the two middle histories. The first continuation
ledgers then have common suffix $first$ and endpoints identified by that middle
sameness; common-suffix cancellation gives $\hsame(source,source')$.
\end{proof}

\begin{theorem}[Continuous modulus chain composite closure]
\label{thm:continuous-modulus-chain-composite-closed}
For unary histories $s,\delta_1,\delta_2,t$, a two-link modulus chain from
$s$ through $\delta_1$ and $\delta_2$ to $t$, together with a continuation
ledger composing $\delta_1$ and $\delta_2$ into $\delta$, determines a single
continuous modulus witness from $s$ to $t$ using $\delta$:
$$
\mathsf{ContinuousModulusChain}(s,\delta_1,\delta_2,t)
\Longrightarrow
\mathsf{Cont}(\delta_1,\delta_2,\delta)
\Longrightarrow
\mathsf{ContinuousModulusWitness}(s,\delta,t).
$$
\end{theorem}
\begin{proof}
Unpack the chain into unary endpoint and modulus data plus a middle history
with ledgers $m=s\cdot\delta_1$ and $t=m\cdot\delta_2$. The composite modulus
is unary by closure of unary continuation. The source-to-target continuation is
the associativity equality $(s\cdot\delta_1)\cdot\delta_2
=s\cdot(\delta_1\cdot\delta_2)$.
\end{proof}

\begin{theorem}[Continuous modulus witness composite public readback]
\label{thm:continuous-modulus-witness-composite-public-readback}
For histories $source,first,second,middle,target,composite$, two certified
one-step modulus witnesses
$$
\mathsf{ContinuousModulusWitness}(source,first,middle)
$$
and
$$
\mathsf{ContinuousModulusWitness}(middle,second,target)
$$
together with $\mathsf{Cont}(first,second,composite)$ determine the composite
witness
$$
\mathsf{ContinuousModulusWitness}(source,composite,target).
$$
Any displayed witness from $source$ through the same $composite$ has target
history $\hsame$ to $target$.
\end{theorem}
\begin{proof}
The two one-step witnesses supply the source, modulus, middle, and target
carriers plus the two continuation ledgers, hence assemble a
$\mathsf{ContinuousModulusChain}(source,first,second,target)$. Composite
closure turns this chain and the displayed modulus composition into the
one-step composite witness. If another target is displayed for the same source
and composite modulus, target determinacy for continuous modulus witnesses
identifies it with $target$.
\end{proof}

\begin{theorem}[Continuous function carrier modulus-chain replacement]
\label{thm:continuous-function-carrier-modulus-chain-replacement}
For histories $source,map,target,oldMod,oldCert,\delta_1,\delta_2,\delta,cert$,
assume an existing continuous-function carrier, a two-step modulus chain from
the same target history to a certificate history, and a continuation ledger
composing the two modulus steps:

$$
\begin{array}{l}
\mathsf{ContinuousFunctionCarrier}(source,map,target,oldMod,oldCert),
\\
\mathsf{ContinuousModulusChain}(target,\delta_1,\delta_2,cert),
\\
\mathsf{Cont}(\delta_1,\delta_2,\delta).
\end{array}
$$

Then the carrier admits the composite modulus and the chain certificate:

$$
\mathsf{ContinuousFunctionCarrier}(source,map,target,\delta,cert).
$$
\end{theorem}
\begin{proof}
Unpack the given carrier into unary data for $source$, $map$, $target$,
$oldMod$, and $oldCert$. Keep the graph ledger
$\mathsf{Cont}(source,map,target)$; the old modulus and certificate are used
only to expose the source, map, and target carrier data.

Apply Theorem~\ref{thm:continuous-modulus-chain-composite-closed} to the chain
$\mathsf{ContinuousModulusChain}(target,\delta_1,\delta_2,cert)$ and the
composition ledger $\mathsf{Cont}(\delta_1,\delta_2,\delta)$. This yields the
replacement modulus witness

$$
\mathsf{ContinuousModulusWitness}(target,\delta,cert).
$$

Unpack this witness to obtain unary data for $\delta$ and $cert$, together with
the modulus-certificate ledger $\mathsf{Cont}(target,\delta,cert)$. Reassemble
$\mathsf{ContinuousFunctionCarrier}(source,map,target,\delta,cert)$ from the
preserved source, map, and target data, the preserved graph ledger, and this new
modulus witness.

The ten-field certificate fit is the following. The traditional target is
uniform-continuity data whose modulus certificate may be recomputed through a
two-link modulus chain. The BEDC source is $BHist$ equipped with $\mathsf{Cont}$.
The carrier is $\mathsf{ContinuousFunctionCarrier}$ and the classifier is
$\mathsf{hsame}$. The stability obligation is carrier preservation under
modulus-chain replacement. The ledger consists of source, map, target, old
modulus, old certificate, two modulus steps, the composite modulus, and the
terminal certificate. The core theorem is this carrier-level replacement,
exactness is the displayed carrier conclusion, the standard bridge is deferred,
and the status is $\mathsf{ProofTarget}$.
\end{proof}

\begin{theorem}[Prefixed continuous function carrier graph-chain closure]
\label{thm:continuous-function-carrier-prefixed-graph-chain-closed}
For histories $p,source,map,target,\delta_1,\delta_2,\delta,cert$, unary
source, map, target, and visible-prefix histories, a graph continuation ledger,
a two-step modulus chain from the target history to a certificate history, and a
continuation ledger composing the two modulus steps determine the prefixed graph
carrier with the composite modulus certificate:

$$
\begin{array}{l}
\mathsf{UnaryHistory}(p)
\Longrightarrow
\mathsf{UnaryHistory}(source)
\Longrightarrow
\\
\mathsf{UnaryHistory}(target)
\Longrightarrow
\mathsf{UnaryHistory}(map)
\Longrightarrow
\\
\mathsf{Cont}(source,map,target)
\Longrightarrow
\mathsf{ContinuousModulusChain}(target,\delta_1,\delta_2,cert)
\Longrightarrow
\mathsf{Cont}(\delta_1,\delta_2,\delta)
\Longrightarrow
\\
\mathsf{ContinuousFunctionCarrier}(p\cdot source,map,p\cdot target,\delta,p\cdot cert).
\end{array}
$$
\end{theorem}
\begin{proof}
First apply Theorem~\ref{thm:continuous-modulus-prefixed-composite-closed} to
the prefix $p$, the chain
$\mathsf{ContinuousModulusChain}(target,\delta_1,\delta_2,cert)$, and the
composition ledger $\mathsf{Cont}(\delta_1,\delta_2,\delta)$. This gives the
modulus witness
$$
\mathsf{ContinuousModulusWitness}(p\cdot target,\delta,p\cdot cert).
$$

Unpack this witness into unary data for $p\cdot target$, the composite modulus
$\delta$, and $p\cdot cert$, together with the certificate ledger
$\mathsf{Cont}(p\cdot target,\delta,p\cdot cert)$.

The graph ledger is prefixed directly from
$\mathsf{Cont}(source,map,target)$. Associativity of history append gives

$$
(p\cdot source)\cdot map
=
p\cdot(source\cdot map)
=
p\cdot target.
$$

Unary append closure applied to $p$ and $source$ supplies the unary source
history $p\cdot source$. The unary map hypothesis is reused, and the prefixed
target, composite modulus, and prefixed certificate data come from the modulus
witness. These fields assemble
$\mathsf{ContinuousFunctionCarrier}(p\cdot source,map,p\cdot target,\delta,
p\cdot cert)$.

The ten-field certificate fit is the following. The traditional target is
uniformly continuous map composition under visible finite-prefix evidence. The
BEDC source is $BHist$ equipped with $\mathsf{Cont}$. The carrier is
$\mathsf{ContinuousFunctionCarrier}$ and the classifier is $\mathsf{hsame}$. The
stability obligation is carrier preservation from graph data and a prefixed
two-link modulus chain. The ledger consists of source, map, target, two modulus
steps, the composite modulus, the terminal certificate, and the visible prefix.
The core theorem is this carrier-level closure, exactness is the displayed
carrier conclusion, the standard bridge is deferred, and the status is
$\mathsf{ProofTarget}$.
\end{proof}

\label{thm:continuous-function-carrier-prefixed-graph-chain-public-readback}

assume unary histories $p$, $source$, $target$, and $map$, a graph ledger
$\mathsf{Cont}(source,map,target)$, a two-link modulus chain
$\mathsf{ContinuousModulusChain}(target,\delta_1,\delta_2,cert)$, and a
composition ledger $\mathsf{Cont}(\delta_1,\delta_2,\delta)$. If a displayed
prefixed continuous-function carrier has target $p\cdot target'$ and
certificate $p\cdot cert'$,

$$
\mathsf{ContinuousFunctionCarrier}
(p\cdot source,map,p\cdot target',\delta,p\cdot cert'),
$$

then the public target and certificate are determined:

$$
\hsame(target,target')\quad\text{and}\quad\hsame(cert,cert').
$$
\end{theorem}
\begin{proof}
The graph ledger and the modulus chain first build the canonical prefixed
continuous-function carrier with target $p\cdot target$ and certificate
$p\cdot cert$. Carrier target-certificate determinacy compares this canonical
carrier with the displayed carrier, giving equality of the two prefixed targets
and equality of the two prefixed certificates. Left cancellation of the shared
prefix $p$ yields the displayed public readback equalities.
\end{proof}

\begin{theorem}[Prefixed continuous modulus composite closure]
\label{thm:continuous-modulus-prefixed-composite-closed}
For unary histories $p,s,\delta_1,\delta_2,t,\delta$, a unary prefix $p$, a
two-link modulus chain from $s$ through $\delta_1$ and $\delta_2$ to $t$, and a
continuation ledger composing $\delta_1$ and $\delta_2$ into $\delta$ determine
a prefixed continuous modulus witness:

$$
\begin{array}{l}
\mathsf{UnaryHistory}(p)
\Longrightarrow
\mathsf{ContinuousModulusChain}(s,\delta_1,\delta_2,t)
\Longrightarrow
\\
\mathsf{Cont}(\delta_1,\delta_2,\delta)
\Longrightarrow
\mathsf{ContinuousModulusWitness}(p\cdot s,\delta,p\cdot t).
\end{array}
$$

\end{theorem}
\begin{proof}
Unpack the chain into unary endpoint and modulus histories, together with the
continuation ledgers from $s$ to the middle history and from the middle history
to $t$.

The prefixed source and target histories are unary by unary append closure
applied to $p$ with $s$ and $t$. The composite modulus history is unary by unary
continuation closure applied to the two modulus histories and their composition
ledger.

The remaining continuation ledger is obtained by associating the concatenation
spine:

$$
(p\cdot s)\cdot\delta
=
p\cdot(s\cdot\delta)
=
p\cdot t.
$$

Thus the prefixed endpoints and the same composite modulus satisfy the public
continuous modulus witness predicate.
\end{proof}
\leanchecked{BEDC.Derived.ContinuousUp.ContinuousModulusWitness\_prefixed\_composite\_closed}

\begin{theorem}[Continuous modulus witness prefixed composite factorization]
\label{thm:continuous-modulus-witness-prefixed-composite-factorizes}
For histories $p,s,\delta_1,\delta_2,\delta,t$, if $\delta_1$ and $\delta_2$
are unary histories, $\mathsf{Cont}(\delta_1,\delta_2,\delta)$ holds, and
$$
\mathsf{ContinuousModulusWitness}(p\cdot s,\delta,p\cdot t)
$$
holds, then there is a middle history $m$ such that
$$
\mathsf{ContinuousModulusWitness}(s,\delta_1,m)
\quad\land\quad
\mathsf{ContinuousModulusWitness}(m,\delta_2,t).
$$
\end{theorem}
\begin{proof}
First remove the common prefix from the displayed continuous modulus witness
using Theorem~\ref{thm:continuous-modulus-witness-prefix-iff}. This leaves a
witness from $s$ through the composite modulus $\delta$ to $t$.

Choose $m=s\cdot\delta_1$. The first witness is immediate from
$\mathsf{Cont}(s,\delta_1,s\cdot\delta_1)$ and unary append closure. For the
second witness, case-analyze the two continuation ledgers
$\mathsf{Cont}(s,\delta,t)$ and $\mathsf{Cont}(\delta_1,\delta_2,\delta)$.
Associativity identifies $(s\cdot\delta_1)\cdot\delta_2$ with
$s\cdot(\delta_1\cdot\delta_2)$, hence with $t$. This supplies the required
continuation ledger from $m$ through $\delta_2$ to $t$.
\end{proof}
\leanchecked{BEDC.Derived.ContinuousUp.ContinuousModulusWitness\_prefixed\_composite\_factorizes}

\begin{theorem}[Continuous prefixed composite middle deterministic factorization]
\label{thm:continuous-modulus-witness-prefixed-composite-middle-deterministic-factorization}
For histories $p,s,\delta_1,\delta_2,\delta,t$, if $\delta_1$ and $\delta_2$
are unary histories, $\mathsf{Cont}(\delta_1,\delta_2,\delta)$ holds, and
$$
\mathsf{ContinuousModulusWitness}(p\cdot s,\delta,p\cdot t)
$$
holds, then there is a middle history $m$ such that
$$
\mathsf{ContinuousModulusWitness}(s,\delta_1,m)
\quad\land\quad
\mathsf{ContinuousModulusWitness}(m,\delta_2,t),
$$
and every other middle history carrying the same two witness legs is $\hsame$
to $m$.
\end{theorem}
\begin{proof}
First apply the prefixed composite factorization theorem to expose a middle
history and the two one-step continuous modulus witnesses. If another middle
history carries a witness from $s$ through $\delta_1$, both first legs have the
same source and modulus; one-step target determinacy therefore identifies the

\label{thm:continuous-modulus-witness-prefixed-composite-source-deterministic}

source determinacy gives $\hsame(s,s')$.
