\chapter{A Concrete Naming Certificate for $\QuadratureUp$}
\label{ch:concrete-instances-quadrature-namecert}

The numerical-quadrature interface $\QuadratureUp$ packages a finite weighted sum approximating an integral on a compact interval, classified by exactness degree (trapezoid, Simpson, Gauss-Legendre). It depends on $\NameCert_{\IntegralUp}$ (\autoref{ch:concrete-instances-integral-namecert}) and $\NameCert_{\FinSetUp}$ (\autoref{ch:concrete-instances-finset-namecert}).

\section{Carrier definition}
\label{sec:quadrature-carrier}

\begin{definition}[Quadrature carried rule source row]
\label{def:quadrature-carried-rule-source-row}
A carried $\QuadratureUp$ rule source row is the $\mathsf{BHist}$-visible tuple
$$
(xs,\alpha,\omega,\mathcal I_Q)
$$
where $xs$ is the $\FinSetUp$ node spine, each node endpoint $\alpha_i$ and weight endpoint $\omega_i$ is carried by the scalar endpoint rows, and $\mathcal I_Q$ is the endpoint map supplied by the $\IntegralUp$ polynomial row on the fixed compact interval. The polynomial inputs to the row are finite $\mathsf{ListCarrier}$ coefficient codes. Ordering choices for the node spine, numerical approximation choices, and convergence data are not carrier fields of the rule; when they are used, they are recorded only by ledger rows attached to the source tuple.
\end{definition}

\section{The certificate}
\label{sec:quadrature-certificate}

\begin{definition}[Quadrature finite weighted exactness degree]
\label{def:quadrature-finite-weighted-exactness-degree}
Fix a scalar carrier $R$ with classifier $\sim_R$, zero endpoint $0_R$, addition $+$, multiplication $\cdot$, canonical polynomial trim $\mathsf{trim}_R$, and Horner evaluation $\mathsf{Eval}_{\alpha,R}$ as in \autoref{def:polynomial-horner-evaluation}. A finite weighted $\QuadratureUp$ rule over $R$ is a tuple
$$
Q=(xs,\alpha,\omega,\mathcal I_Q)
$$
where $xs$ is the finite $\FinSetUp$ index spine for the nodes, $\alpha_i$ and $\omega_i$ are carried scalar endpoints for each $i:\mathsf{Pos}(xs)$, and $\mathcal I_Q(p)$ is the scalar endpoint supplied by the $\IntegralUp$ row for the polynomial code $p$ on the fixed compact interval. Its quadrature sum on a finite coefficient spine $p:\mathsf{ListCarrier}(R)$ is
$$
\mathsf{QSum}_Q(p):=
\mathsf{fold}_{+,\mathsf{Pos}(xs)}
\bigl(i\mapsto \omega_i\cdot\mathsf{Eval}_{\alpha_i,R}(p)\bigr).
$$
The pointwise exactness predicate is
$$
\mathsf{QExact}_Q(p)\Longleftrightarrow
\mathsf{QSum}_Q(p)\sim_R\mathcal I_Q(p).
$$
For a unary degree bound $n:\NatUp$, define
$$
\mathsf{PolyDegLe}_R(p,n)
\Longleftrightarrow
\forall k:\NatUp.\;
\mathsf{NatUnaryStrictPrefix}(n,k)
\Longrightarrow
\mathsf{trim}_R(p)[k]\sim_R0_R.
$$
The quadrature exactness-degree predicate is
$$
\mathsf{QuadratureExactUpTo}_R(Q,n)
\Longleftrightarrow
\forall p:\mathsf{ListCarrier}(R).\;
\mathsf{PolyDegLe}_R(p,n)
\Longrightarrow
\mathsf{QExact}_Q(p).
$$
Finally, the degree-bound comparison used by this $\QuadratureUp$ classifier is
$$
\mathsf{DegBoundLe}(e,d)
\Longleftrightarrow
\forall k:\NatUp.\;
\mathsf{NatUnaryStrictPrefix}(d,k)
\Longrightarrow
\mathsf{NatUnaryStrictPrefix}(e,k).
$$
This is the unary BEDC reading of the ordinary inequality $e\leq d$: every coefficient index above $d$ is also above $e$.
\end{definition}

\begin{theorem}[Quadrature carrier scope obligation]
\label{thm:quadrature-carrier-scope-obligation}
For the $\QuadratureUp$ certificate described by \autoref{def:quadrature-carried-rule-source-row} and \autoref{def:quadrature-finite-weighted-exactness-degree}, the carrier scope is exactly the $\FinSetUp$ node spine $xs$, the scalar endpoint rows carrying $\alpha_i$ and $\omega_i$, the $\IntegralUp$ endpoint map $\mathcal I_Q$, the $\mathsf{ListCarrier}$ polynomial code $p$, and the $\NatUp$ degree bound $n$. No host ordering of nodes, host analytic function, host real equality, convergence assumption, or approximation scheme is part of the carried source row.
\end{theorem}
\begin{proof}
The carrier row of \autoref{def:quadrature-carried-rule-source-row} names only the tuple $(xs,\alpha,\omega,\mathcal I_Q)$, with polynomial inputs supplied as finite $\mathsf{ListCarrier}$ codes. The exactness-degree definition then adds only the unary bound $n:\NatUp$ and the derived rows $\mathsf{QSum}_Q$, $\mathsf{QExact}_Q$, $\mathsf{PolyDegLe}_R$, and $\mathsf{QuadratureExactUpTo}_R$.

Each displayed row is built from the listed source objects: folding over $\mathsf{Pos}(xs)$, scalar multiplication and addition, Horner evaluation on a list code, the integral endpoint map, trimming, and strict-prefix comparison on unary naturals. Therefore any datum outside these rows can enter only as a ledger entry and not as carrier scope.
\end{proof}

\begin{theorem}[Quadrature classifier scope obligation]
\label{thm:quadrature-classifier-scope-obligation}
The exactness-degree classifier for the carried $\QuadratureUp$ rule is the preorder row $\mathsf{DegBoundLe}$ on $\NatUp$ bounds together with the weakening action on $\mathsf{QuadratureExactUpTo}_R$. Its source predicates are precisely $\mathsf{NatUnaryStrictPrefix}$ and $\mathsf{PolyDegLe}_R$ as displayed in \autoref{def:quadrature-finite-weighted-exactness-degree}.
\end{theorem}
\begin{proof}
The classifier comparison is the displayed predicate $\mathsf{DegBoundLe}(e,d)$. By \autoref{lem:quadrature-polynomial-degree-window-inclusion}, this comparison transports a polynomial degree window from the smaller bound $e$ to the larger bound $d$ using only $\mathsf{NatUnaryStrictPrefix}$ and the coefficient condition in $\mathsf{PolyDegLe}_R$.

Applying \autoref{thm:quadrature-exactness-degree-weakening} to that transported window gives the classifier action on $\mathsf{QuadratureExactUpTo}_R$. The reflexive and transitive rows recorded in \autoref{prop:quadrature-degree-bound-comparison-preorder-rows} identify the comparison surface as a preorder on the same unary bounds, so the classifier has no additional source objects.
\end{proof}

\begin{theorem}[Quadrature ledger exactness obligation]
\label{thm:quadrature-ledger-exactness-obligation}
For a carried $\QuadratureUp$ rule, the ledger exactness surface records the node spine, weights, integral endpoint provenance, polynomial trim and evaluation rows, and the pointwise predicate $\mathsf{QExact}_Q$. The ledger does not assert host real equality or host function equality; exactness is the carried scalar endpoint relation
$$
\mathsf{QSum}_Q(p)\sim_R\mathcal I_Q(p)
$$
on finite polynomial codes.
\end{theorem}
\begin{proof}
Unfold \autoref{def:quadrature-finite-weighted-exactness-degree}. The quadrature sum is a finite fold over $\mathsf{Pos}(xs)$ using the carried weights $\omega_i$ and Horner evaluation at the carried node endpoints $\alpha_i$. The integral side is the carried endpoint $\mathcal I_Q(p)$ supplied by the $\IntegralUp$ polynomial row.

The polynomial side is controlled by $\mathsf{trim}_R$ and $\mathsf{PolyDegLe}_R$, while exactness itself is exactly the predicate $\mathsf{QExact}_Q(p)$. The window inclusion lemma, the weakening theorem, and the preorder rows cited above transport this same predicate between unary degree bounds without replacing it by host equality of real numbers or functions.
\end{proof}

\begin{lemma}[Polynomial degree windows are nested]
\label{lem:quadrature-polynomial-degree-window-inclusion}
With the notation of \autoref{def:quadrature-finite-weighted-exactness-degree}, if $\mathsf{DegBoundLe}(e,d)$ holds, then every polynomial code with degree at most $e$ also has degree at most $d$:
$$
\mathsf{PolyDegLe}_R(p,e)
\Longrightarrow
\mathsf{PolyDegLe}_R(p,d).
$$
\end{lemma}
\leanchecked{BEDC.Derived.QuadratureUp.QuadraturePolynomialDegreeWindow\_inclusion}
\begin{proof}
Unfold $\mathsf{PolyDegLe}_R$ and $\mathsf{DegBoundLe}$. Assume $\mathsf{PolyDegLe}_R(p,e)$ and fix $k:\NatUp$ with $\mathsf{NatUnaryStrictPrefix}(d,k)$. The degree-bound comparison sends this strict-prefix witness to $\mathsf{NatUnaryStrictPrefix}(e,k)$. Applying the degree-at-most-$e$ hypothesis at this $k$ gives
$$
\mathsf{trim}_R(p)[k]\sim_R0_R.
$$
This is precisely the coefficient condition required for $\mathsf{PolyDegLe}_R(p,d)$.
\end{proof}

\begin{theorem}[Quadrature exactness degree weakening]
\label{thm:quadrature-exactness-degree-weakening}
Let $Q$ be a finite weighted $\QuadratureUp$ rule over the scalar carrier $R$ in the sense of \autoref{def:quadrature-finite-weighted-exactness-degree}. For unary degree bounds $e,d:\NatUp$, if
$$
\mathsf{QuadratureExactUpTo}_R(Q,d)
\qquad\text{and}\qquad
\mathsf{DegBoundLe}(e,d)
$$
hold, then the same rule is exact on every polynomial of degree at most $e$:
$$
\mathsf{QuadratureExactUpTo}_R(Q,e).
$$
\end{theorem}
\leanchecked{BEDC.Derived.QuadratureUp.QuadratureExactUpTo\_weakening}
\begin{proof}
Unfold the target exactness predicate. Fix a finite coefficient spine $p:\mathsf{ListCarrier}(R)$ and assume $\mathsf{PolyDegLe}_R(p,e)$. By \autoref{lem:quadrature-polynomial-degree-window-inclusion}, the degree-bound comparison gives $\mathsf{PolyDegLe}_R(p,d)$. The exactness hypothesis at degree $d$ applies to this same polynomial code and returns
$$
\mathsf{QExact}_Q(p).
$$
Unfolding $\mathsf{QExact}_Q$ identifies this row as
$$
\mathsf{QSum}_Q(p)\sim_R\mathcal I_Q(p),
$$
which is the required exactness row for $p$ under the degree-$e$ predicate. Since $p$ was arbitrary, $\mathsf{QuadratureExactUpTo}_R(Q,e)$ holds. The finite node spine, the weights, and the integral endpoint map are the original components of $Q$ throughout.
\end{proof}

\begin{theorem}[Quadrature exactness degree equivalence stability]
\label{thm:quadrature-exactness-degree-equivalence-stability}
Let $Q$ be a finite weighted $\QuadratureUp$ rule over the scalar carrier $R$ in the sense of \autoref{def:quadrature-finite-weighted-exactness-degree}. For unary degree bounds $e,d:\NatUp$, if
$$
\mathsf{DegBoundLe}(e,d)
\qquad\text{and}\qquad
\mathsf{DegBoundLe}(d,e)
$$
hold, then the exactness classifiers at $d$ and $e$ determine the same row:
$$
\mathsf{QuadratureExactUpTo}_R(Q,d)
\Longleftrightarrow
\mathsf{QuadratureExactUpTo}_R(Q,e).
$$
\end{theorem}
\leanchecked{BEDC.Derived.QuadratureUp.QuadratureExactUpTo\_degree\_equivalence\_stability}
\begin{proof}
We prove the two directions separately.

For the forward direction, assume $\mathsf{QuadratureExactUpTo}_R(Q,d)$. Apply \autoref{thm:quadrature-exactness-degree-weakening} to this degree-$d$ exactness row and to the comparison $\mathsf{DegBoundLe}(e,d)$. The result is $\mathsf{QuadratureExactUpTo}_R(Q,e)$.

For the reverse direction, assume $\mathsf{QuadratureExactUpTo}_R(Q,e)$. Apply \autoref{thm:quadrature-exactness-degree-weakening} to this degree-$e$ exactness row and to the comparison $\mathsf{DegBoundLe}(d,e)$. The result is $\mathsf{QuadratureExactUpTo}_R(Q,d)$.
\end{proof}

\begin{theorem}[Quadrature degree-bound comparison is transitive]
\label{thm:quadrature-degree-bound-comparison-transitive}
With the notation of \autoref{def:quadrature-finite-weighted-exactness-degree}, the unary degree-bound comparison used by the $\QuadratureUp$ exactness-degree classifier is transitive. For unary degree bounds $e,d,c:\NatUp$, if
$$
\mathsf{DegBoundLe}(e,d)
\qquad\text{and}\qquad
\mathsf{DegBoundLe}(d,c)
$$
hold, then
$$
\mathsf{DegBoundLe}(e,c)
$$
holds.
\end{theorem}
\leanchecked{BEDC.Derived.QuadratureUp.QuadratureDegBoundLe\_trans}
\begin{proof}
Unfold $\mathsf{DegBoundLe}$. To prove $\mathsf{DegBoundLe}(e,c)$, fix $k:\NatUp$ and assume
$$
\mathsf{NatUnaryStrictPrefix}(c,k).
$$
The comparison $\mathsf{DegBoundLe}(d,c)$ sends this witness to
$$
\mathsf{NatUnaryStrictPrefix}(d,k).
$$
Applying $\mathsf{DegBoundLe}(e,d)$ to that witness gives
$$
\mathsf{NatUnaryStrictPrefix}(e,k).
$$
This is exactly the strict-prefix condition required at the index $k$ for $\mathsf{DegBoundLe}(e,c)$. Since $k$ was arbitrary, the comparison from $e$ to $c$ holds.
\end{proof}

\begin{theorem}[Quadrature degree-bound comparison is reflexive]
\label{thm:quadrature-degree-bound-comparison-reflexive}
With the notation of \autoref{def:quadrature-finite-weighted-exactness-degree}, the exactness-degree comparison used by the $\QuadratureUp$ classifier is reflexive. For every unary degree bound $d:\NatUp$,
$$
\mathsf{DegBoundLe}(d,d)
$$
holds.
\end{theorem}
\begin{proof}
Unfold $\mathsf{DegBoundLe}$ at the pair $(d,d)$. The required statement is
$$
\forall k:\NatUp.\;
\mathsf{NatUnaryStrictPrefix}(d,k)
\Longrightarrow
\mathsf{NatUnaryStrictPrefix}(d,k).
$$
Fix $k:\NatUp$ and assume $\mathsf{NatUnaryStrictPrefix}(d,k)$. The conclusion is the same strict-prefix witness. Since $k$ was arbitrary, the displayed universal clause holds, and therefore $\mathsf{DegBoundLe}(d,d)$ holds.
\end{proof}

\begin{proposition}[Quadrature degree-bound comparison preorder rows]
\label{prop:quadrature-degree-bound-comparison-preorder-rows}
With the notation of \autoref{def:quadrature-finite-weighted-exactness-degree}, the unary degree-bound comparison attached to the $\QuadratureUp$ exactness-degree classifier supplies the reflexive and transitive rows on unary bounds:
$$
\forall d:\NatUp.\;\mathsf{DegBoundLe}(d,d)
$$
and
$$
\forall e,d,c:\NatUp.\;
\mathsf{DegBoundLe}(e,d)\Longrightarrow
\mathsf{DegBoundLe}(d,c)\Longrightarrow
\mathsf{DegBoundLe}(e,c).
$$
\end{proposition}
\begin{proof}
The reflexive row is \autoref{thm:quadrature-degree-bound-comparison-reflexive}. The transitive row is \autoref{thm:quadrature-degree-bound-comparison-transitive}. Both rows use the same concrete predicate $\mathsf{DegBoundLe}$ from \autoref{def:quadrature-finite-weighted-exactness-degree}, so together they give the stated preorder data for the $\QuadratureUp$ degree-bound classifier.
\end{proof}
\leanchecked{BEDC.Derived.QuadratureUp.QuadratureDegBoundLe\_preorder\_rows}
