\chapter{A Concrete Naming Certificate for \(\ComplexTopologyUp\)}
\label{ch:concrete-instances-complex-topology-namecert}

This chapter gives a concrete naming-certificate construction for the complex topology interface \(\ComplexTopologyUp\) at the paper-proof level. The interface promotes the inline open-disk predicate of \autoref{ch:concrete-instances-holomorphic-namecert} to a stand-alone licensed object and packages the open-set, closed-set, compact-subset, and density-witness predicates that the holomorphy, contour-integration, and analytic-continuation chapters consume. The construction sits above \(\ComplexUp\) (\autoref{ch:concrete-instances-complex-namecert}) for the rational-pair carrier and \(\RealAnalyticUp\) (\autoref{ch:concrete-instances-real-analytic-namecert}) for the radius and modulus arithmetic. Every set-membership predicate is decidable on rational representations through \(\mathsf{NatUnaryStrictPrefix}\) (\autoref{ch:concrete-instances-nat-namecert}); compactness is encoded by explicit dyadic \(\varepsilon\)-net data, in line with the constructive-compactness convention of the L10 sub-roadmap. The construction uses no \(\mathsf{Classical.choice}\), no \(\mathsf{Quot.sound}\), and no \(\mathsf{propext}\), so the certificate's Lean targets fall under the project's axiom-purity gate.

\section{Open disks}
\label{sec:cplx-topo-open-disks}

\begin{definition}[Open disk]
\label{def:cplx-topo-open-disk}
For complex history \(z_0\), unary radius \(r\), and complex history \(z\), the open-disk membership predicate \(\mathsf{OpenDisk}(z_0, r, z)\) holds when the rational representation of the complex modulus \(|z - z_0|\) is strictly less than \(r\) under \(\mathsf{NatUnaryStrictPrefix}\). The witness is an explicit unary gap \(\mathsf{Gap}(z_0, r, z) > \emp\).
\end{definition}
\leandef{BEDC.Derived.ComplexTopologyUp.ComplexTopologyOpenDiskGap}

\begin{remark}[Compatibility with the inline definition]
This definition coincides with the inline \texttt{BEDC.Derived.HolomorphicUp.OpenDisk} of \autoref{ch:concrete-instances-holomorphic-namecert} up to namespace; promoting it to \(\ComplexTopologyUp\) lets downstream chapters cite a single canonical home.
\end{remark}

\begin{theorem}[Open-disk point-radius suffix closure]
\label{thm:cplx-topo-open-disk-point-radius-suffix}
If \(\mathsf{OpenDisk}(z_0,r,z)\) holds and a unary suffix \(extra\) continues
both \(r\) to \(r'\) and \(z\) to \(z'\), then
\(\mathsf{OpenDisk}(z_0,r',z')\) holds. The same gap witness continues \(z'\)
to \(r'\), while its original ledger continues \(z\) to \(r\).
\end{theorem}
\begin{proof}
Unpack the witnessed open disk. Unary closure gives the shifted radius, complex
history closure gives the shifted point, and associativity of \(\Cont\) rewrites
the original gap ledger after the common unary suffix.
\end{proof}
\leanchecked{BEDC.Derived.ComplexTopologyUp.ComplexTopologyOpenDisk\_point\_radius\_suffix\_closed}

\begin{definition}[Closed disk]
\label{def:cplx-topo-closed-disk}
\(\mathsf{ClosedDisk}(z_0, r, z)\) holds when \(|z - z_0| \leq r\) on rational representations, decidable by non-strict prefix comparison.
\end{definition}

\begin{theorem}[Open-disk gap modulus]
\label{thm:cplx-topo-gap-modulus}
For every triple \((z_0, r, z)\) with \(\mathsf{OpenDisk}(z_0, r, z)\), the gap function \(\mathsf{Gap}(z_0, r, z)\) is a unary history giving an explicit positive distance from \(z\) to the boundary \(\partial \mathsf{ClosedDisk}(z_0, r)\).
\end{theorem}
\begin{proof}
Compute \(\mathsf{Gap}(z_0, r, z) := r \ominus |z - z_0|\) using \(\RealUp\) truncated subtraction. The strict inequality of \autoref{def:cplx-topo-open-disk} guarantees the result is non-empty unary.
\end{proof}
\leanchecked{BEDC.Derived.ComplexTopologyUp.ComplexTopologyOpenDisk\_gap\_modulus}

\begin{lemma}[Open-disk gap transport along \(\hsame\)]
\label{lem:cplx-topo-open-disk-gap-hsame-transport}
If a complex-topology open-disk gap witness is transported by the complex
history classifier at its center and point, and by \(\hsame\) at its radius
and gap histories, then the transported tuple is again an open-disk gap
witness and exposes the transported boundary ledger.
\end{lemma}
\begin{proof}
Unpack the open-disk gap witness. The complex classifiers provide the target
complex carriers for the center and point. Unary transport moves the radius
and gap carriers along \(\hsame\), and continuation transport moves
\(\Cont(z,g,r)\) to \(\Cont(z',g',r')\).
\end{proof}
\leanchecked{BEDC.Derived.ComplexTopologyUp.ComplexTopologyOpenDiskGap\_hsame\_transport}

\begin{theorem}[Open-disk gap radius extension]
\label{thm:cplx-topo-open-disk-gap-radius-extension}
If a complex-topology open-disk gap witness has radius \(r\), point \(z\),
gap \(g\), and \(z\cdot g=r\), then any common unary extension \(extra\) of
the radius and gap gives a new witness with radius \(r'\), gap \(g'\), and
\(\Cont(z,g',r')\).
\end{theorem}
\begin{proof}
Unpack the open-disk gap witness. Unary continuation closure gives the
extended radius and gap carriers. The boundary ledger follows by eliminating
the two continuation witnesses and reassociating the original point--gap
continuation.
\end{proof}
\leanchecked{BEDC.Derived.ComplexTopologyUp.ComplexTopologyOpenDiskGap\_radius\_extension\_closed}

\begin{theorem}[Open-disk gap radius is non-empty]
\label{thm:cplx-topo-open-disk-gap-radius-not-empty}
If a complex-topology open-disk gap witness has radius \(r\), then
\(r\) cannot be \(\hsame\) to \(\emp\).
\end{theorem}
\begin{proof}
Transport the point--gap continuation ledger along a hypothetical
\(\hsame(r,\emp)\). Empty-result inversion for continuation forces the point
history to be empty, contradicting the complex carrier carried by the open-disk
gap witness.
\end{proof}
\leanchecked{BEDC.Derived.ComplexTopologyUp.ComplexTopologyOpenDiskGap\_radius\_not\_empty}

\begin{theorem}[Open-disk gap unary suffix transport]
\label{thm:cplx-topo-open-disk-gap-unary-suffix-transport}
If a complex-topology open-disk gap witness has boundary ledger
\(\Cont(z,g,r)\), and a unary suffix \(q\) continues both \(z\) to \(zq\)
and \(r\) to \(rq\), then the shifted point \(zq\) and shifted radius \(rq\)
form another open-disk gap witness with the same center and gap, and expose
\(\Cont(zq,g,rq)\).
\end{theorem}
\begin{proof}
Unpack the witness. The point carrier is closed under unary suffix
continuation, the radius carrier is closed under unary continuation, and the
boundary ledger follows by reassociating \(\Cont\) after commuting the unary
gap and suffix histories.
\end{proof}
\leanchecked{BEDC.Derived.ComplexTopologyUp.ComplexTopologyOpenDiskGap\_unary\_suffix\_transport}

\section{Open sets and finite intersections}
\label{sec:cplx-topo-open-sets}

\begin{definition}[Open-set membership]
\label{def:cplx-topo-open-set}
A complex open set is encoded as a closed inductive predicate \(U : \ComplexUp\)-history \(\to \mathsf{Prop}\) generated by finite unions and finite intersections of open-disk predicates. \(\mathsf{IsOpen}(U)\) holds when every membership witness \(U(z)\) supplies an explicit interior open-disk \(\mathsf{OpenDisk}(z, r_z, z)\) of positive unary radius \(r_z\), with \(r_z\) given by an explicit total function on the membership-witness data.
\end{definition}

\begin{theorem}[Finite intersection of open sets is open]
\label{thm:cplx-topo-finite-intersection}
If \(\mathsf{IsOpen}(U_1)\) and \(\mathsf{IsOpen}(U_2)\), then \(\mathsf{IsOpen}(U_1 \cap U_2)\) with explicit interior radius \(r_{12}(z) := \min(r_1(z), r_2(z))\).
\end{theorem}
\begin{proof}
Combine the two interior-disk witnesses by taking the minimum unary radius pointwise; the minimum is decidable on unary representations.
\end{proof}

\begin{theorem}[Arbitrary union of open sets is open]
\label{thm:cplx-topo-union}
For any classifier-respecting family \(\{U_i\}_{i \in I}\) of open sets indexed by an inductive carrier \(I\), the union \(\bigcup_i U_i\) is open with interior radius inherited from the witnessing branch.
\end{theorem}
\begin{proof}
Membership in the union is by case analysis on \(I\); each case provides the interior-disk witness from the corresponding \(U_i\).
\end{proof}

\section{Compact subsets via dyadic \(\varepsilon\)-nets}
\label{sec:cplx-topo-compact}

\begin{definition}[Dyadic \(\varepsilon\)-net]
\label{def:cplx-topo-dyadic-net}
For a complex-set predicate \(K\) and unary precision \(k\), a dyadic \(2^{-k}\)-net is a finite list of complex histories \([z_1, \ldots, z_N]\) such that for every \(z\) with \(K(z)\), there exists an index \(i \leq N\) with \(\mathsf{OpenDisk}(z_i, 2^{-k}, z)\). The net function \(\mathsf{Net}_K : \mathsf{Unary} \to \mathsf{List}(\ComplexUp)\) is required to be a total explicit function on the precision domain.
\end{definition}

\begin{definition}[Compact subset]
\label{def:cplx-topo-compact}
\(\mathsf{IsCompact}(K)\) holds when \(K\) is bounded (contained in some \(\mathsf{ClosedDisk}(\emp, R)\) for explicit unary \(R\)) and possesses an explicit dyadic \(2^{-k}\)-net for every unary precision \(k\) (i.e.\ a witnessing total function \(\mathsf{Net}_K : \mathsf{Unary} \to \mathsf{List}(\ComplexUp)\)).
\end{definition}

\begin{theorem}[Closed disk is compact]
\label{thm:cplx-topo-closed-disk-compact}
For every \(z_0\) and unary \(r\), \(\mathsf{IsCompact}(\mathsf{ClosedDisk}(z_0, r))\) holds with explicit dyadic \(\varepsilon\)-net obtained by tiling the disk with rational-grid points at spacing \(2^{-k}\).
\end{theorem}
\begin{proof}
At precision \(k\), enumerate the rational pairs \((p, q)\) with \(|p - \mathsf{Re}(z_0)| \leq r\) and \(|q - \mathsf{Im}(z_0)| \leq r\) on the dyadic-rational grid of mesh \(2^{-k}/2\); the resulting finite list witnesses every disk point within distance \(2^{-k}/\sqrt{2} \leq 2^{-k}\). The list construction is by unary recursion over the grid bounds, which are given explicitly by \(R\) and \(k\).
\end{proof}

\begin{theorem}[Compact subsets of half-planes]
\label{thm:cplx-topo-half-plane-compact}
For unary parameters \(\sigma_a, \epsilon, T\), the compact sub-half-plane
$$
K(\sigma_a, \epsilon, T) \;:=\; \{ s : \mathsf{Re}(s) \geq \sigma_a + \epsilon, |\mathsf{Im}(s)| \leq T \}
$$
is compact in the sense of \autoref{def:cplx-topo-compact}, with explicit dyadic \(\varepsilon\)-nets given by tiling the rectangle \([\sigma_a + \epsilon, R(\sigma_a, T)] \times [-T, T]\) for explicit \(R(\sigma_a, T)\).
\end{theorem}
\begin{proof}
The half-plane predicate is a finite intersection of closed half-planes, each compact on bounded restriction. The bound \(R(\sigma_a, T)\) is taken to enclose the witnessing region needed by the consuming chapter (typically the half-plane \(\Re(s) > 1\) of the Dirichlet-series majorants in \autoref{ch:concrete-instances-dirichlet-series-namecert}).
\end{proof}

\section{Density witnesses and clustering points}
\label{sec:cplx-topo-density}

\begin{definition}[Quantitative density witness]
\label{def:cplx-topo-density}
For a center \(z_c \in \ComplexUp\) and an open set \(U\) with \(U(z_c)\), a quantitative density witness inside \(U\) is an explicit sequence \(\mathsf{Cluster} : \mathsf{Unary} \to \ComplexUp\) of distinct points \((z_n)_n\) with \(U(z_n)\), \(z_n \neq z_c\), and \(|z_n - z_c| \leq 2^{-n}\) for every unary \(n\). The witness is a total explicit function on the precision domain.
\end{definition}

\begin{theorem}[Open disks contain quantitative cluster sequences]
\label{thm:cplx-topo-open-disk-clustering}
For every \(z_0\) and unary \(r > 0\), the open disk \(\mathsf{OpenDisk}(z_0, r)\) admits a quantitative density witness centered at \(z_0\): the sequence \(z_n := z_0 + 2^{-n} r / 2\) is in the disk, distinct from \(z_0\), and within distance \(2^{-n} r / 2 \leq 2^{-n}\) for \(n \geq 1\).
\end{theorem}
\begin{proof}
Construct the sequence by explicit unary recursion. The membership condition holds because the point lies at distance \(2^{-n} r / 2 < r\) from \(z_0\); distinctness holds because the offset is non-empty unary for each \(n\).
\end{proof}

\begin{remark}[Clustering for the constructive identity theorem]
The quantitative density witness of \autoref{def:cplx-topo-density} is the precise data required by the constructive identity theorem of \autoref{ch:concrete-instances-holomorphic-namecert}: two holomorphic functions agreeing on an explicit cluster sequence agree on a smaller open disk. This chapter supplies the witness type that the holomorphy chapter consumes.
\end{remark}

\section{Domain compatibility for analytic continuation}
\label{sec:cplx-topo-domain-compat}

\begin{definition}[Domain compatibility]
\label{def:cplx-topo-domain-compat}
For two open sets \(D_1, D_2\) and an overlap region \(U\), \(\mathsf{DomCompat}(D_1, D_2, U)\) holds when \(\mathsf{IsOpen}(U)\), \(U \subseteq D_1 \cap D_2\) with explicit intersection-membership witness, and \(U\) contains an open disk \(\mathsf{OpenDisk}(z_c, r_c)\) with quantitative density witness \(\mathsf{Cluster}(z_c, U)\) of \autoref{def:cplx-topo-density}.
\end{definition}

\begin{remark}[Domain compatibility consumed by analytic continuation]
\autoref{ch:concrete-instances-analytic-continuation} requires exactly this data to invoke the constructive identity theorem when extending a holomorphic function across overlapping domains; the present definition fixes the schema so the analytic-continuation chapter cites a single canonical interface.
\end{remark}

\section{Certificate target}
\label{sec:cplx-topo-cert-target}

\begin{theorem}[Concrete \(\ComplexTopologyUp\) naming certificate]
\label{thm:concrete-cplx-topo-namecert}
The tuple \((\mathsf{OpenDisk}, \mathsf{ClosedDisk}, \mathsf{IsOpen}, \mathsf{IsCompact}, \mathsf{Cluster}, \mathsf{DomCompat})\), with sources, classifiers, stability obligations, ledger, and standard bridge as listed below, forms a paper-side \(\NameCert(\ComplexTopologyUp)\) over \(\ComplexUp\):
\begin{enumerate}
  \item \textbf{Source}: \(\ComplexUp\)-histories with rational-pair carrier representations.
  \item \textbf{Carrier}: each predicate of Sections~\ref{sec:cplx-topo-open-disks}--\ref{sec:cplx-topo-domain-compat} is a closed inductive predicate on \(\ComplexUp\)-histories.
  \item \textbf{Classifier}: \(\ComplexClassifier\)-equivariance for set membership inherited from \(\hsame\)-transport on the underlying complex carriers.
  \item \textbf{Stability obligations}: open-disk gap modulus (\autoref{thm:cplx-topo-gap-modulus}), finite-intersection closure (\autoref{thm:cplx-topo-finite-intersection}), arbitrary-union closure (\autoref{thm:cplx-topo-union}), closed-disk compactness (\autoref{thm:cplx-topo-closed-disk-compact}), half-plane compactness (\autoref{thm:cplx-topo-half-plane-compact}), open-disk clustering (\autoref{thm:cplx-topo-open-disk-clustering}).
  \item \textbf{Ledger}: explicit interior-radius function for every open-set witness; explicit dyadic \(\varepsilon\)-net function for every compact-set witness; explicit cluster sequence for every density witness.
  \item \textbf{Standard bridge}: when classical complex topology (Heine-Borel compactness, open covers) is supplied externally, the explicit-net witnesses interpret as classical compactness witnesses on bounded subsets only; the bridge slot records this restricted correspondence.
  \item \textbf{Status}: \(\mathsf{ProofTarget}\). Lean construction proceeds in \(\mathsf{BEDC.Derived.ComplexTopologyUp.*}\). The inline \(\mathsf{OpenDisk}\) of \autoref{ch:concrete-instances-holomorphic-namecert} is to be re-exported from \(\mathsf{BEDC.Derived.ComplexTopologyUp}\) as the canonical home; the holomorphic chapter retains a re-export alias for backward compatibility with existing markers.
\end{enumerate}
\end{theorem}
\begin{proof}
Each carrier predicate is closed inductive over \(\ComplexUp\)-histories with decidable membership on rational representations. The six stability obligations are the section-by-section theorems above, each established by explicit construction of the witness data (gap modulus, dyadic net, cluster sequence). The bridge slot is the standard interpretation under the bounded-subset restriction. No \(\mathsf{Classical.choice}\), no \(\mathsf{Quot.sound}\), and no \(\mathsf{propext}\) is required.
\end{proof}

\begin{corollary}[\(\ComplexTopologyUp\) is licensed, not primitive]
\label{cor:cplx-topo-licensed-not-primitive}
\(\ComplexTopologyUp\) is licensed by closed inductive predicates over rational-pair representations of complex histories, with constructive compactness encoded as explicit dyadic \(\varepsilon\)-nets. It is not a primitive of the BEDC kernel, and in particular does not assume Heine-Borel compactness as a general theorem.
\end{corollary}

\begin{remark}[Discharge of holomorphy/contour/analytic-continuation forward references]
\label{rem:cplx-topo-discharge-forward-refs}
The open-disk, compact-half-plane, cluster-sequence, and domain-compatibility witnesses inline-defined or implicitly used in \autoref{ch:concrete-instances-holomorphic-namecert}, \autoref{ch:concrete-instances-contour-integral}, \autoref{ch:concrete-instances-analytic-continuation}, and \autoref{ch:concrete-instances-zeta-basic-namecert} are now anchored at the present \(\ComplexTopologyUp\) certificate. Future restructuring should re-export the inline \(\mathsf{OpenDisk}\) from \(\mathsf{BEDC.Derived.HolomorphicUp}\) as an alias of the canonical \(\mathsf{BEDC.Derived.ComplexTopologyUp.OpenDisk}\).
\end{remark}
