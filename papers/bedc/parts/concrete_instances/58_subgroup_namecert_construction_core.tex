\chapter{A Concrete Naming Certificate for $\SubgroupUp$}
\label{ch:concrete-instances-subgroup-namecert}

The subgroup interface $\SubgroupUp$ packages a subset of a group carrier closed under the group operation, identity, and inverse. It depends on $\NameCert_{\GroupUp}$ (\autoref{ch:concrete-instances-group-namecert}) and is the first sub-object construction in BEDC, prerequisite to the quotient and ideal certificates that follow.

\section{Carrier definition}
\label{sec:subgroup-carrier}

\begin{definition}[Centralizer SubgroupUp instance]
\label{def:subgroup-centralizer-instance}
Fix a BEDC history-level $\GroupUp$ source over $\mathsf{BHist}$ with
operation $\mathsf{mul}$, unit $e$, inverse operation $\mathsf{inv}$, and
fixed element $a:\mathsf{BHist}$. The centralizer $\SubgroupUp$ instance uses
the carrier $Z_a$ from \autoref{def:group-centralizer-carrier}. Its classifier
is the restriction of $\hsame$ to that carrier: two carried histories $x$ and
$y$ represent the same subgroup element exactly when $x\hsame y$. The
operation, unit, and inverse are inherited from the ambient $\GroupUp$ source,
and the ledger records the fixed element $a$ together with the commuting
witnesses certifying membership in $Z_a$.
\end{definition}
\leandef{BEDC.Derived.SubgroupUp.SubgroupCentralizerCarrier}
\leandef{BEDC.Derived.SubgroupUp.SubgroupCentralizerClassifier}

\section{The certificate}
\label{sec:subgroup-certificate}

\begin{theorem}[Centralizer SubgroupUp certificate target]
\label{thm:subgroup-centralizer-certificate-target}
For a BEDC history-level $\GroupUp$ operation over $\mathsf{BHist}$ satisfying
associativity, left and right unit laws for $e$, multiplication congruence, and
two-sided inverse laws, the instance of
\autoref{def:subgroup-centralizer-instance} satisfies the $\SubgroupUp$ rows:
\begin{enumerate}
\item the carrier is inhabited by $e$, and $Z_a(e)$;
\item if $Z_a(x)$ and $Z_a(y)$, then $Z_a(\mathsf{mul}(x,y))$;
\item if $Z_a(x)$, then $Z_a(\mathsf{inv}(x))$;
\item if $Z_a(x)$ and $x\hsame y$, then $Z_a(y)$;
\item for every $x:\mathsf{BHist}$,
$$
Z_a(x) \Longleftrightarrow \mathsf{comm}(x,a)\hsame e.
$$
\end{enumerate}
Thus the centralizer carrier is a concrete certificate target for
$\NameCert_{\SubgroupUp}$ over the ambient history classifier.
\end{theorem}
\begin{proof}
The inhabited, product-closure, inverse-closure, and commutator-fiber rows are
the rows of \autoref{thm:group-centralizer-subgroup}. The inverse row uses the
ambient inverse congruence from
\autoref{thm:group-inverse-congruence-from-laws} and the reverse-product
description of inverses from \autoref{thm:group-inverse-mul-reverse}; these
are exactly the sources cited by the GroupUp centralizer proof.

For carrier transport, unfold $Z_a$. A witness $x\hsame y$ transports
$\mathsf{mul}(x,a)\hsame\mathsf{mul}(a,x)$ to
$\mathsf{mul}(y,a)\hsame\mathsf{mul}(a,y)$ by multiplication congruence on the
left and right arguments, followed by symmetry and transitivity of $\hsame$.
Therefore the restricted classifier preserves the subgroup carrier.

The exactness row is the commutator criterion from
\autoref{thm:group-commutator-trivial-iff-commutes} applied to the ordered
pair $(x,a)$, with the definition of $Z_a$ supplying the displayed
orientation. Repacking these rows with the inherited operation, unit, inverse,
and ledger gives the advertised $\SubgroupUp$ certificate target.
\end{proof}

\begin{definition}[Two-centralizer intersection carrier]
\label{def:subgroup-centralizer-intersection-carrier}
For two fixed histories $a,b:\mathsf{BHist}$ in the same BEDC
history-level $\GroupUp$ source, let $Z_a$ and $Z_b$ be the centralizer
carriers of \autoref{def:group-centralizer-carrier}. Define the
two-centralizer carrier by
$$
Z_{a,b}(x) :\Longleftrightarrow Z_a(x)\land Z_b(x).
$$
Its $\SubgroupUp$ classifier is the restriction of $\hsame$ to this
carrier: two carried histories represent the same element exactly when their
ambient histories satisfy $x\hsame y$.
\end{definition}
\leandef{BEDC.Derived.SubgroupUp.SubgroupCentralizerIntersectionCarrier}

\begin{theorem}[Centralizer intersection SubgroupUp certificate]
\label{thm:subgroup-centralizer-intersection-certificate}
For a BEDC history-level $\GroupUp$ operation over $\mathsf{BHist}$
satisfying the same associativity, left and right unit laws, multiplication
congruence, and two-sided inverse laws as in
\autoref{thm:subgroup-centralizer-certificate-target}, and for fixed histories
$a,b:\mathsf{BHist}$, the carrier $Z_{a,b}$ of
\autoref{def:subgroup-centralizer-intersection-carrier} has the following
$\SubgroupUp$ certificate rows:
\begin{enumerate}
\item $Z_{a,b}(e)$.
\item If $Z_{a,b}(x)$ and $Z_{a,b}(y)$, then
$Z_{a,b}(\mathsf{mul}(x,y))$.
\item If $Z_{a,b}(x)$, then $Z_{a,b}(\mathsf{inv}(x))$.
\item If $Z_{a,b}(x)$ and $x\hsame y$, then $Z_{a,b}(y)$.
\item For every $x:\mathsf{BHist}$,
$$
Z_{a,b}(x) \Longleftrightarrow
\mathsf{comm}(x,a)\hsame e\land \mathsf{comm}(x,b)\hsame e.
$$
\end{enumerate}
Thus the simultaneous centralizer of $a$ and $b$ is a reusable
$\SubgroupUp$ certificate target over the ambient history classifier.
\end{theorem}
\begin{proof}
Apply \autoref{thm:subgroup-centralizer-certificate-target} twice, once to
the fixed history $a$ and once to the fixed history $b$. The two identity
rows give $Z_a(e)$ and $Z_b(e)$, which package directly as $Z_{a,b}(e)$ by
\autoref{def:subgroup-centralizer-intersection-carrier}.

For product closure, unpack $Z_{a,b}(x)$ and $Z_{a,b}(y)$ into their two
centralizer components. The $a$-row gives
$Z_a(\mathsf{mul}(x,y))$, and the $b$-row gives
$Z_b(\mathsf{mul}(x,y))$; pairing these witnesses proves the displayed
product row for $Z_{a,b}$.
\leanchecked{BEDC.Derived.SubgroupUp.SubgroupCentralizerIntersectionCarrier\_mul\_closed\_from\_empty\_unit}

The inverse row is componentwise in the same sense. From $Z_a(x)$ obtain
$Z_a(\mathsf{inv}(x))$, and from $Z_b(x)$ obtain
$Z_b(\mathsf{inv}(x))$; the pair is exactly
$Z_{a,b}(\mathsf{inv}(x))$.
\leanchecked{BEDC.Derived.SubgroupUp.SubgroupCentralizerIntersectionCarrier\_inv\_closed\_from\_empty\_unit}

Carrier transport also separates into the two coordinates. Given
$Z_{a,b}(x)$ and $x\hsame y$, use the centralizer transport row for $a$ to
move $Z_a(x)$ to $Z_a(y)$, and use the centralizer transport row for $b$ to
move $Z_b(x)$ to $Z_b(y)$. The restricted $\hsame$ classifier on
$Z_{a,b}$ is therefore stable under ambient history equality.

Finally, unfold the carrier definition. The commutator exactness row for
$a$ rewrites $Z_a(x)$ as $\mathsf{comm}(x,a)\hsame e$, and the corresponding
row for $b$ rewrites $Z_b(x)$ as $\mathsf{comm}(x,b)\hsame e$. Combining the
two equivalences gives the advertised simultaneous commutator criterion.
\end{proof}

Per-object template: traditional target two-centralizer subgroup certificate;
BEDC source $\GroupUp$ over $\mathsf{BHist}$; carrier $Z_{a,b}$; classifier
restricted $\hsame$; stability identity, product, inverse, and carrier
transport; ledger fixed histories $a,b$, two commutator witnesses, and
ambient operation witnesses; core theorem list
\autoref{thm:subgroup-centralizer-certificate-target} and
\autoref{thm:subgroup-centralizer-intersection-certificate}; exactness
predicate simultaneous triviality of $\mathsf{comm}(x,a)$ and
$\mathsf{comm}(x,b)$; downstream consumer
\autoref{thm:ratup-fieldup-affine-center-centralizer-intersection}; standard
bridge center as an intersection of centralizers; status
$\mathsf{ProofTarget}$.

\begin{definition}[RatUp FieldUp affine center subgroup carrier]
\label{def:ratup-fieldup-affine-center-subgroup-carrier}
For the concrete $\FieldUp(\RatUp)$ affine exit package of
\autoref{thm:ratup-fieldup-canonical-exit-action-inverse-package}, take the
ambient group carrier to be $\mathsf{AffAut}_F$ with classifier
$\equiv_{\mathrm{aff}}$, multiplication $\circ_{\mathrm{aff}}$, identity
$\mathrm{id}_F$, and inverse row $\iota_F$. The associated $\SubgroupUp$
carrier is the affine center predicate on that same group:
$$
\mathsf{Carrier}_{\SubgroupUp}(Z) := \mathsf{Center}_F(Z)
\qquad (Z:\mathsf{AffAut}_F).
$$
Thus membership in the subgroup certificate is exactly center membership in
the concrete FieldUp affine automorphism group.
\end{definition}

\begin{theorem}[RatUp FieldUp affine center SubgroupUp instance]
\label{thm:ratup-fieldup-affine-center-subgroup-instance}
The carrier of
\autoref{def:ratup-fieldup-affine-center-subgroup-carrier} instantiates the
concrete $\SubgroupUp$ certificate fields over $\mathsf{AffAut}_F$. Namely:
\begin{enumerate}
\item $\mathsf{Carrier}_{\SubgroupUp}(\mathrm{id}_F)$.
\item If $\mathsf{Carrier}_{\SubgroupUp}(Z)$ and
$\mathsf{Carrier}_{\SubgroupUp}(W)$, then
$\mathsf{Carrier}_{\SubgroupUp}(Z\circ_{\mathrm{aff}}W)$.
\item If $\mathsf{Carrier}_{\SubgroupUp}(Z)$, then
$\mathsf{Carrier}_{\SubgroupUp}(\iota_F(Z))$.
\item If $Z\equiv_{\mathrm{aff}}W$ and
$\mathsf{Carrier}_{\SubgroupUp}(Z)$, then
$\mathsf{Carrier}_{\SubgroupUp}(W)$.
\item If $\mathsf{Carrier}_{\SubgroupUp}(Z)$, then for every
$T:\mathsf{AffAut}_F$,
$\mathsf{Carrier}_{\SubgroupUp}(\mathrm{Inn}_T(Z))$.
\end{enumerate}
The same data make this $\SubgroupUp$ object normal under inner affine
conjugation.
\end{theorem}
\begin{proof}
Fix the canonical exit action and inverse package of
\autoref{thm:ratup-fieldup-canonical-exit-action-inverse-package}. This fixes
the ambient group operations $\circ_{\mathrm{aff}}$, $\mathrm{id}_F$,
$\iota_F$, $\equiv_{\mathrm{aff}}$, and the inner affine conjugation row on
$\mathsf{AffAut}_F$.

By Definition~\ref{def:ratup-fieldup-affine-center-subgroup-carrier}, each
subgroup membership assertion is exactly the corresponding
$\mathsf{Center}_F$ assertion. Apply
\autoref{thm:ratup-fieldup-affine-center-normal-subgroup} to obtain identity
membership, product closure, inverse closure, classifier transport, and
normality under $\mathrm{Inn}_T$.

The classifier-transport field is compatible with the ambient ledger because
\autoref{thm:field-affine-pointwise-classifier-operation-congruence} supplies
operation and inverse congruence for $\equiv_{\mathrm{aff}}$. Therefore the
five displayed clauses are precisely the concrete $\SubgroupUp$ certificate
fields over the FieldUp affine exit package.
\end{proof}

Per-object template: traditional target normal subgroup of the affine

carrier $\mathsf{Center}_F$; classifier $\equiv_{\mathrm{aff}}$; stability
identity, product, inverse, classifier transport, and inner conjugation; ledger
affine operation and conjugation witnesses; core theorem list
\autoref{thm:ratup-fieldup-affine-center-subgroup-instance} and
\autoref{thm:ratup-fieldup-affine-center-normal-subgroup}; exactness predicate
center membership as the $\SubgroupUp$ carrier; standard bridge ordinary
affine center normal subgroup; status $\mathsf{ProofTarget}$.

\begin{definition}[RatUp FieldUp affine normalizer SubgroupUp carrier]
\label{def:ratup-fieldup-affine-normalizer-subgroupup-carrier}
For Definition~\ref{def:ratup-fieldup-concrete-exit-object}, fix a certified
affine automorphism $A:\mathsf{AffAut}_F$ in the concrete $\FieldUp(\RatUp)$
exit object. Let $C_F(A)$ be the affine centralizer carrier and let
$N_F(C_F(A))$ be the two-sided affine centralizer normalizer carrier. The
corresponding $\SubgroupUp$ carrier is
$$
\mathsf{Carrier}^{N}_{\SubgroupUp}(T) := N_F(C_F(A))(T)
\qquad (T:\mathsf{AffAut}_F).
$$
It uses the ambient affine classifier $\equiv_{\mathrm{aff}}$, operation
$\circ_{\mathrm{aff}}$, identity $\mathrm{id}_F$, and inverse $\iota_F$. Its
ledger records the fixed automorphism $A$, the centralizer membership tests
for $C_F(A)$, and the conjugation witnesses asserting preservation and
reflection of that centralizer under affine inner action.
\end{definition}

\begin{theorem}[RatUp FieldUp affine normalizer SubgroupUp instance and center action]
\label{thm:ratup-fieldup-affine-normalizer-subgroupup-inclusion-action}
The carrier of
\autoref{def:ratup-fieldup-affine-normalizer-subgroupup-carrier} is the
concrete $\SubgroupUp$ certificate target for the affine normalizer
$N_F(C_F(A))$ over $\mathsf{AffAut}_F$. Namely, for the fixed certified
$A:\mathsf{AffAut}_F$:
\begin{enumerate}
\item $\mathsf{Carrier}^{N}_{\SubgroupUp}(\mathrm{id}_F)$.
\item If $\mathsf{Carrier}^{N}_{\SubgroupUp}(T)$ and
$\mathsf{Carrier}^{N}_{\SubgroupUp}(U)$, then
$\mathsf{Carrier}^{N}_{\SubgroupUp}(T\circ_{\mathrm{aff}}U)$.
\item If $\mathsf{Carrier}^{N}_{\SubgroupUp}(T)$, then
$\mathsf{Carrier}^{N}_{\SubgroupUp}(\iota_F(T))$.
\item If $T\equiv_{\mathrm{aff}}U$ and
$\mathsf{Carrier}^{N}_{\SubgroupUp}(T)$, then
$\mathsf{Carrier}^{N}_{\SubgroupUp}(U)$.
\item If $\mathsf{Center}_F(Z)$, then
$\mathsf{Carrier}^{N}_{\SubgroupUp}(Z)$.
\item If $C_F(A)(X)$ and $\mathsf{Center}_F(Z)$, then the restricted
normalizer action satisfies
$\mathrm{Act}_Z(X)\equiv_{\mathrm{aff}}X$.
\end{enumerate}
Thus the affine center includes into this normalizer carrier and acts
pointwise as the classifier identity on the fixed centralizer.
\end{theorem}
\begin{proof}
By Definition~\ref{def:ratup-fieldup-affine-normalizer-subgroupup-carrier},
each $\mathsf{Carrier}^{N}_{\SubgroupUp}$ assertion is exactly the
$N_F(C_F(A))$ normalizer predicate for the fixed affine automorphism $A$.

Apply
\autoref{thm:ratup-fieldup-affine-centralizer-normalizer-subgroup}. Its four
displayed rows give identity membership, product closure, inverse closure, and
transport along $\equiv_{\mathrm{aff}}$ for the carrier
$\mathsf{Carrier}^{N}_{\SubgroupUp}$.

For the center inclusion and restricted action, apply
\autoref{thm:ratup-fieldup-affine-center-normalizes-centralizers}. The first
displayed implication gives
$\mathsf{Center}_F(Z)\Rightarrow\mathsf{Carrier}^{N}_{\SubgroupUp}(Z)$, and
the second displayed statement gives
$\mathrm{Act}_Z(X)\equiv_{\mathrm{aff}}X$ for every carried centralizer element
$X$.

The ambient classifier compatibility is the same compatibility used in
\autoref{thm:field-affine-pointwise-classifier-operation-congruence}: affine
operation and inverse rows respect $\equiv_{\mathrm{aff}}$, so the normalizer
witnesses repack as $\SubgroupUp$ certificate rows without changing the
underlying affine group data.
\end{proof}

\begin{definition}[Centralizer normalizer witness]
\label{def:subgroup-centralizer-normalizer-witness}
For the centralizer carrier $Z_a$ from
\autoref{def:group-centralizer-carrier}, define the history-level normalizer
witness predicate by
$$
\mathsf{Norm}_{Z_a}(t) :\Longleftrightarrow
\bigl(\forall x:\mathsf{BHist},\ Z_a(x)\Rightarrow
Z_a(\mathsf{mul}(\mathsf{mul}(t,x),\mathsf{inv}(t)))\bigr)
\land
\bigl(\forall x:\mathsf{BHist},\ Z_a(x)\Rightarrow
Z_a(\mathsf{mul}(\mathsf{mul}(\mathsf{inv}(t),x),
\mathsf{inv}(\mathsf{inv}(t))))\bigr).
$$
Thus $\mathsf{Norm}_{Z_a}(t)$ records that conjugation by $t$ and conjugation by
$\mathsf{inv}(t)$ both preserve the centralizer carrier. Equivalently, it is the
carrier $N_a$ of \autoref{def:group-centralizer-normalizer-carrier} with the
abbreviation $\mathsf{Conj}$ expanded.
\end{definition}

\begin{theorem}[Centralizer elements normalize the centralizer]
\label{thm:subgroup-centralizer-self-normalizer}
For a BEDC history-level $\GroupUp$ operation over $\mathsf{BHist}$ satisfying
the same associativity, unit, multiplication-congruence, and two-sided inverse
laws as in \autoref{thm:subgroup-centralizer-certificate-target}, every
centralizer element normalizes the centralizer. Explicitly, if $Z_a(t)$, then
$\mathsf{Norm}_{Z_a}(t)$.
\end{theorem}
\begin{proof}
Assume $Z_a(t)$. The inverse-closure row of
\autoref{thm:group-centralizer-subgroup} applied to $Z_a(t)$ gives
$Z_a(\mathsf{inv}(t))$.

For the forward conjugation clause, let $x:\mathsf{BHist}$ satisfy $Z_a(x)$.
The product-closure row of \autoref{thm:group-centralizer-subgroup} applied to
$Z_a(t)$ and $Z_a(x)$ gives $Z_a(\mathsf{mul}(t,x))$.

Apply product closure once more, now to $Z_a(\mathsf{mul}(t,x))$ and
$Z_a(\mathsf{inv}(t))$. The result is
$Z_a(\mathsf{mul}(\mathsf{mul}(t,x),\mathsf{inv}(t)))$, which is precisely
the first clause of $\mathsf{Norm}_{Z_a}(t)$.

For the inverse conjugation clause, let $x:\mathsf{BHist}$ satisfy $Z_a(x)$.
Product closure applied to $Z_a(\mathsf{inv}(t))$ and $Z_a(x)$ gives
$Z_a(\mathsf{mul}(\mathsf{inv}(t),x))$.

Apply product closure once more with the final factor $Z_a(t)$. The two-sided
inverse laws identify $t$ with $\mathsf{inv}(\mathsf{inv}(t))$ under $\hsame$,
and the carrier-transport row of \autoref{thm:group-centralizer-subgroup}
transports membership to
$Z_a(\mathsf{mul}(\mathsf{mul}(\mathsf{inv}(t),x),
\mathsf{inv}(\mathsf{inv}(t))))$. This is the second clause of
$\mathsf{Norm}_{Z_a}(t)$, so both preservation directions hold.
\end{proof}

\begin{lemma}[Centralizer normalizer kernel classifier reflexivity]
\label{lem:subgroup-centralizer-normalizer-kernel-classifier-refl}
For a BEDC history-level $\GroupUp$ operation over $\mathsf{BHist}$ satisfying
the same associativity, unit, multiplication-congruence, and two-sided inverse
laws as in \autoref{thm:subgroup-centralizer-certificate-target}, if
$x:\mathsf{BHist}$ is carried by the centralizer normalizer $N_a$, then the
normalizer endpoint classifies with itself and the inverse-left kernel word is
carried by the centralizer:
$$
N_a(x)\Rightarrow N_a(x)\land N_a(x)\land
Z_a(\mathsf{mul}(\mathsf{inv}(x),x)).
$$
\end{lemma}
\begin{proof}
The two normalizer endpoint rows are the given $N_a(x)$ witness. For the
kernel word, use the centralizer certificate target of
\autoref{thm:subgroup-centralizer-certificate-target}: its identity row gives
$Z_a(\emp)$ and its $\hsame$-transport row transports this membership along
the inverse-left law
$\mathsf{mul}(\mathsf{inv}(x),x)\hsame\emp$ read in the reverse direction.
Thus the inverse-left kernel endpoint lies in the lower centralizer carrier.
\end{proof}

\begin{definition}[Centralizer-normalizer SubgroupUp pair]
\label{def:subgroup-centralizer-normalizer-pair}
Fix the centralizer carrier $Z_a$ from
\autoref{def:group-centralizer-carrier} and the centralizer normalizer carrier
$N_a$ from \autoref{def:group-centralizer-normalizer-carrier}. The
centralizer-normalizer $\SubgroupUp$ pair has lower carrier $Z_a$ and upper
carrier $N_a$, both classified by the restriction of $\hsame$ to carried
histories. The operation, unit, and inverse are inherited from the ambient
history-level $\GroupUp$ operation $\mathsf{mul}$, $e$, and $\mathsf{inv}$.

The pair ledger records the fixed element $a$, the conjugation word
$\mathsf{Conj}_s(x)$, the witnesses that histories carried by $Z_a$ commute
with $a$, and the forward and inverse preservation witnesses appearing in the
definition of $N_a$. Thus a carried normalizer history acts on the carried
centralizer histories by the restricted conjugation word while keeping every
endpoint inside the same $\SubgroupUp$ lower carrier.
\end{definition}

\begin{theorem}[Centralizer-normalizer SubgroupUp pair certificate]
\label{thm:subgroup-centralizer-normalizer-pair-certificate}
For a BEDC history-level $\GroupUp$ operation over $\mathsf{BHist}$ satisfying
the same associativity, unit, multiplication-congruence, and two-sided inverse
laws as in
\autoref{thm:group-centralizer-normalizer-subgroup}, the pair of
\autoref{def:subgroup-centralizer-normalizer-pair} satisfies the following
concrete $\SubgroupUp$ certificate rows:
\begin{enumerate}
\item $Z_a$ is the centralizer $\SubgroupUp$ certificate target of
\autoref{thm:subgroup-centralizer-certificate-target}.
\item $N_a$ is a $\SubgroupUp$ certificate target over the same ambient
operation, with unit, product, inverse, and classifier-transport rows.
\item The lower carrier is included in the upper carrier: if $Z_a(s)$, then
$N_a(s)$.
\item If $N_a(s)$ and $Z_a(x)$, then
$Z_a(\mathsf{Conj}_s(x))$ and
$Z_a(\mathsf{Conj}_{\mathsf{inv}(s)}(x))$.
\item For every $s$ with $N_a(s)$, the restricted normalizer action on $Z_a$
is classifier-exact, its orbit classifier is stable under $\hsame$ replacement
of carried endpoints, and its fixed-action kernel satisfies
$$
(\forall x:\mathsf{BHist},\ Z_a(x)\Rightarrow
\hsame(\mathsf{Conj}_s(x),x))
\Longleftrightarrow
(\forall x:\mathsf{BHist},\ Z_a(x)\Rightarrow
\hsame(\mathsf{mul}(s,x),\mathsf{mul}(x,s))).
$$
\end{enumerate}
Hence the centralizer and its normalizer are available as one concrete
$\SubgroupUp$ pair over $\mathsf{BHist}$, with the restricted conjugation
action and orbit classifier attached to the same certificate object.
\end{theorem}
\begin{proof}
The lower certificate is
\autoref{thm:subgroup-centralizer-certificate-target}. It gives the unit,
product, inverse, classifier-transport, and commutator-fiber rows for $Z_a$
with the operation, unit, inverse, and restricted $\hsame$ classifier inherited
from the ambient $\GroupUp$.

The upper certificate and the carrier inclusion are supplied by
\autoref{thm:group-centralizer-normalizer-subgroup}. Its first four rows give
unit, product, inverse, and classifier transport for $N_a$, and its final row
places each centralizer element in the normalizer.

The action rows are the defining preservation clauses of $N_a$, recorded in
\autoref{thm:group-centralizer-normalizer-action-automorphism-kernel-exactness}.
That theorem also gives the inverse-action identities and the displayed
fixed-action kernel equivalence, so the restricted action is exact on the
centralizer carrier.

Finally,
\autoref{thm:group-centralizer-normalizer-orbit-classifier-certificate}
supplies the endpoint-transport row for the orbit classifier generated by the
same normalizer action. Repacking these cited rows with the ledger of
\autoref{def:subgroup-centralizer-normalizer-pair} gives one concrete
$\SubgroupUp$ pair over $\mathsf{BHist}$.
\end{proof}
