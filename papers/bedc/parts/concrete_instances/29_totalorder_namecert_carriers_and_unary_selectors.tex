\chapter{A Concrete Naming Certificate for \(\TotalOrderUp\)}
\label{ch:concrete-instances-totalorder-namecert}

The total order interface \(\TotalOrderUp\) extends \autoref{ch:concrete-instances-poset-namecert} with the trichotomy field, expressed constructively: for any \(a, b\) in the carrier, the carrier's apartness predicate decides \(a < b\), \(a = b\) (under the classifier), or \(a > b\). The classical formulation (LEM-style "either \(a \le b\) or \(b \le a\)") is replaced by decidable comparison so that the total-order certificate remains 0-axiom.

\section{Carrier definition}

\begin{definition}[TotalOrder carrier]
\label{def:totalorder-carrier}
Use a poset setup extended with constructive comparison data returning less, same, or greater.
\end{definition}

\section{Concrete singleton history total order instance}
\label{sec:totalorder-concrete-singleton-history-instance}

\begin{definition}[Concrete singleton history total order instance]
\label{def:concrete-singleton-history-totalorder-instance}
Let $A_{\emp}$ be the singleton history source of
Definition~\ref{def:list-singleton-history-source} at $a_0=\emp$. Define
$$
\rho_{\emp}(h,k):=
A_{\emp}(h)\land A_{\emp}(k)\land\hsame(h,k),
$$
and let the order relation on carried endpoints be
$$
h\le_{\emp}k:=\mathsf{PreorderPrefixLE}(h,k).
$$
The constructive comparison package has three branches:
$$
\mathsf{lt}_{\emp}(h,k):=\bot,\qquad
\mathsf{same}_{\emp}(h,k):=\rho_{\emp}(h,k),\qquad
\mathsf{gt}_{\emp}(h,k):=\bot.
$$
Thus the less and greater branches are empty, while the same branch stores the
singleton classifier witness.
\end{definition}

\begin{theorem}[Singleton history total order laws]
\label{thm:singleton-history-totalorder-laws}
For carried histories $h,k:\mathsf{BHist}$ with $A_{\emp}(h)$ and
$A_{\emp}(k)$, the comparison package of
\autoref{def:concrete-singleton-history-totalorder-instance} always returns
the same branch carrying $\rho_{\emp}(h,k)$. The resulting tuple satisfies the
$\TotalOrderUp$ carrier field, the $\POSetUp$ fields, classifier stability,
strict-to-apartness, trichotomy, and comparison stability.
\end{theorem}
\begin{proof}
The carrier data are inherited from the singleton history source. The
hypotheses $A_{\emp}(h)$ and $A_{\emp}(k)$ unfold to $\hsame(h,\emp)$ and
$\hsame(k,\emp)$; symmetry and transitivity of $\hsame$ compose these endpoint
witnesses into $\hsame(h,k)$. Together with the two carrier facts, this is
exactly $\rho_{\emp}(h,k)$.

The preorder comparisons in both directions are supplied by
$\mathsf{PreorderPrefixLE\_of\_hsame}$ from the displayed
$\hsame(h,k)$ witness and its symmetric form. These witnesses give the
$\POSetUp$ order relation on the restricted singleton carrier.

Antisymmetry is the concrete prefix collapse:
$\mathsf{PreorderPrefixLE\_antisymm\_hsame}$ sends opposing prefix comparisons
to $\hsame(h,k)$, which repackages with the carrier facts as the classifier
$\rho_{\emp}(h,k)$. Classifier stability and comparison stability use the same
endpoint transport through $\emp$.

The strict branches are false by definition, so strict-to-apartness is vacuous.
Trichotomy is the same branch just constructed; no classical comparison
principle is used.
\end{proof}

\begin{theorem}[Singleton total-order comparison exactness]
\label{thm:singleton-totalorder-comparison-exactness}
For carried histories $h,k:\mathsf{BHist}$ with $A_{\emp}(h)$ and
$A_{\emp}(k)$ in the concrete instance of
\autoref{def:concrete-singleton-history-totalorder-instance}, the comparison
branches are exact in the following sense:
\begin{enumerate}
  \item $\mathsf{lt}_{\emp}(h,k)\to\mathsf{False}$ and
  $\mathsf{gt}_{\emp}(h,k)\to\mathsf{False}$.
  \item $\mathsf{same}_{\emp}(h,k)\Longleftrightarrow\rho_{\emp}(h,k)$.
  \item $\mathsf{same}_{\emp}(h,k)\Longleftrightarrow
  (h\le_{\emp}k\land k\le_{\emp}h)$.
\end{enumerate}
\end{theorem}
\begin{proof}
Unfold the three comparison branches in
\autoref{def:concrete-singleton-history-totalorder-instance}. The strict
branches are both $\bot$, so each strict witness eliminates to
$\mathsf{False}$.

The same branch is defined to be $\rho_{\emp}(h,k)$, giving the second
equivalence by reflexive unfolding.

For the opposing-prefix characterization, apply the exactness part of
\autoref{thm:singleton-empty-poset-laws} to the carried histories $h$ and $k$.
Composing that equivalence with the displayed unfolding
$\mathsf{same}_{\emp}(h,k)\Longleftrightarrow\rho_{\emp}(h,k)$ gives the stated
equivalence with $h\le_{\emp}k$ and $k\le_{\emp}h$.
\end{proof}

\begin{theorem}[TotalOrder carrier selector classifier coherence]
\label{thm:totalorder-carrier-selector-classifier-coherence}
For carried singleton histories $h,k:\mathsf{BHist}$ with $A_{\emp}(h)$ and
$A_{\emp}(k)$, the branch selector of
\autoref{def:concrete-singleton-history-totalorder-instance} is coherent with
the carrier classifier:
\begin{enumerate}
  \item $\mathsf{same}_{\emp}(h,k)\Longleftrightarrow\rho_{\emp}(h,k)$.
  \item $\mathsf{lt}_{\emp}(h,k)\to\mathsf{False}$ and
  $\mathsf{gt}_{\emp}(h,k)\to\mathsf{False}$.
  \item For histories $h',k':\mathsf{BHist}$ with $\hsame(h,h')$ and
  $\hsame(k,k')$, endpoint replacement preserves the selected branch:
  $$
  \mathsf{same}_{\emp}(h',k')\Longleftrightarrow
  \mathsf{same}_{\emp}(h,k).
  $$
\end{enumerate}

The per-object fields are: traditional target singleton total order; BEDC
source the $\mathsf{BHist}$ singleton carrier; classifier $\rho_{\emp}$;
stability by $\hsame$ endpoint replacement; ledger branch selection; core
theorem this selector coherence; exactness same-branch equivalence with the
carrier classifier; standard bridge deferred; downstream use by
$\TotalOrderUp$ obligation closure; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Unfold the concrete singleton total-order instance. The selected same branch is
defined to be $\rho_{\emp}(h,k)$, while the strict-left and strict-right
branches are both $\bot$.

Apply \autoref{thm:singleton-history-totalorder-laws} to the carried histories
$h$ and $k$. It supplies the singleton comparison package and the carrier
classifier row built from the two carrier witnesses and $\hsame$ transitivity
through $\emp$.

Apply \autoref{thm:singleton-totalorder-comparison-exactness}. Its first clause
eliminates both strict branches to $\mathsf{False}$, and its second clause is
the displayed equivalence between the same branch and $\rho_{\emp}(h,k)$.

For endpoint replacement, transport the carrier facts through
$\hsame(h,h')$ and $\hsame(k,k')$. Symmetry and transitivity of $\hsame$
repackage the transported witnesses as $\rho_{\emp}(h',k')$ exactly when the
original witnesses repackage as $\rho_{\emp}(h,k)$.

Composing this transported classifier equivalence with the same-branch
equivalences for $(h,k)$ and $(h',k')$ proves preservation of the selected
branch.
\end{proof}

\begin{theorem}[TotalOrder singleton branch transport determinacy]
\label{thm:totalorder-singleton-branch-transport-determinacy}
For histories $h,k,h',k':\mathsf{BHist}$, assume $A_{\emp}(h)$,
$A_{\emp}(k)$, $\hsame(h,h')$, and $\hsame(k,k')$. Then endpoint
replacement along $\hsame$ determines the singleton comparison data:
\begin{enumerate}
  \item $A_{\emp}(h')\land A_{\emp}(k')$.
  \item $\mathsf{lt}_{\emp}(h',k')\to\mathsf{False}$ and
  $\mathsf{gt}_{\emp}(h',k')\to\mathsf{False}$.
  \item $\mathsf{same}_{\emp}(h',k')\Longleftrightarrow
  \rho_{\emp}(h',k')$.
  \item $\rho_{\emp}(h',k')\Longleftrightarrow\rho_{\emp}(h,k)$.
\end{enumerate}
\end{theorem}
\begin{proof}
Transport the two carrier facts through the endpoint replacements. Since
$A_{\emp}(h)$ means $\hsame(h,\emp)$, symmetry of $\hsame(h,h')$ and
transitivity with $\hsame(h,\emp)$ give $\hsame(h',\emp)$, hence
$A_{\emp}(h')$. The same argument using $\hsame(k,k')$ gives
$A_{\emp}(k')$.

Apply \autoref{thm:singleton-totalorder-comparison-exactness} to the carried
histories $h'$ and $k'$. Its strict-branch clause gives both eliminations to
$\mathsf{False}$, and its same-branch clause gives
$\mathsf{same}_{\emp}(h',k')\Longleftrightarrow\rho_{\emp}(h',k')$.

It remains to compare the singleton classifiers before and after transport.
The carrier coordinates have already been transported. For the history
coordinate, compose symmetry of $\hsame(h,h')$, the original singleton
collapse $\hsame(h,k)$ obtained from $A_{\emp}(h)$ and $A_{\emp}(k)$, and
$\hsame(k,k')$ to obtain $\hsame(h',k')$; conversely compose
$\hsame(h,h')$, $\hsame(h',k')$, and symmetry of $\hsame(k,k')$ to obtain
$\hsame(h,k)$. Repackaging these witnesses with the corresponding carrier
facts proves $\rho_{\emp}(h',k')\Longleftrightarrow\rho_{\emp}(h,k)$.
\end{proof}

\section{Concrete unary prefix total order instance}
\label{sec:totalorder-concrete-unary-prefix-instance}

\begin{definition}[Concrete unary prefix total order instance]
\label{def:concrete-unary-prefix-totalorder-instance}
Let the carrier be the concrete unary-history preorder carrier:
$$
\mathsf{Carrier}_{\mathsf{u}}:=\mathsf{PreorderCarrier}
=\mathsf{UnaryHistory}.
$$
Let the carrier classifier be
$$
\rho_{\mathsf{u}}(h,k):=
\mathsf{PreorderCarrierClassifierSpec}(\mathsf{PreorderCarrier})(h,k),
$$
and let the order relation be the concrete prefix relation
$$
h\le_{\mathsf{u}}k:=\mathsf{PreorderPrefixLE}(h,k).
$$
The constructive comparison package has the three branches
$$
\mathsf{lt}_{\mathsf{u}}(h,k):=\mathsf{NatUnaryStrictPrefix}(h,k),
\qquad
\mathsf{same}_{\mathsf{u}}(h,k):=\rho_{\mathsf{u}}(h,k),
\qquad
\mathsf{gt}_{\mathsf{u}}(h,k):=\mathsf{NatUnaryStrictPrefix}(k,h).
$$
Thus strict-left means that $k$ extends $h$ by a nonempty unary tail, same is
the carrier classifier, and strict-right is the opposite strict unary prefix.
\end{definition}

\begin{theorem}[Concrete unary prefix total order laws]
\label{thm:concrete-unary-prefix-totalorder-laws}
For carried unary histories $h,k$ in
$\mathsf{Carrier}_{\mathsf{u}}$, the branches of
\autoref{def:concrete-unary-prefix-totalorder-instance} supply constructive
total comparison. Each strict branch implies carrier-classifier apartness, the
same branch satisfies
$$
\mathsf{same}_{\mathsf{u}}(h,k)\Longleftrightarrow
(\mathsf{PreorderPrefixLE}(h,k)\land\mathsf{PreorderPrefixLE}(k,h)),
$$
and the inherited preorder and antisymmetry fields assemble these data into a
concrete $\TotalOrderUp$ law package over unary histories. The strict comparison
row is the isolated branch transitivity of
\autoref{thm:totalorder-strict-branch-transitivity}.
\end{theorem}
\begin{proof}
Start from the total unary prefix comparison of
\autoref{thm:nat-unary-prefix-total}. For carried histories $h$ and $k$, this
returns a unary continuation from $h$ to $k$ or one from $k$ to $h$.

Split the selected unary tail by the unary-generation cases. A nonempty tail
chooses the corresponding strict branch
$\mathsf{NatUnaryStrictPrefix}(h,k)$ or
$\mathsf{NatUnaryStrictPrefix}(k,h)$. An empty tail collapses the continuation
to $\hsame(h,k)$, which repackages with the two carrier witnesses as
$\rho_{\mathsf{u}}(h,k)$ and therefore chooses the same branch.

The strict branches are apart from the same branch. If a strict-left witness and
the carrier classifier both held, the classifier exactness of
\autoref{thm:poset-prefix-classifier-exactness} would give the reverse prefix
comparison as well; splitting that reverse continuation into empty or nonempty
form either collapses the strict tail to $\emp$ or gives opposing strict
prefixes, both ruled out by \autoref{thm:nat-unary-strict-prefix-asymm}. The
strict-right case is symmetric.

The displayed equivalence for the same branch is exactly the prefix classifier
exactness theorem \autoref{thm:poset-prefix-classifier-exactness} after
unfolding $\mathsf{same}_{\mathsf{u}}$. Relation stability, reflexivity, and
transitivity are inherited from
\autoref{thm:concrete-unary-prefix-preorder-laws}, while antisymmetry is the
same exactness field of \autoref{thm:poset-prefix-classifier-exactness}.

For strict-comparison transitivity, use
\autoref{thm:totalorder-strict-branch-transitivity}. It applies the concrete
strict unary-prefix transitivity field to the strict-left branch and the same
argument with reversed endpoints to the strict-right branch. These fields repack
into the concrete $\TotalOrderUp$ law package.
\end{proof}

\begin{theorem}[Concrete unary prefix total-order comparison partition]
\label{thm:concrete-unary-prefix-totalorder-comparison-partition}
For carried unary histories $h,k$ in
$\mathsf{Carrier}_{\mathsf{u}}$, exactly one of the three concrete branches
$$
\mathsf{NatUnaryStrictPrefix}(h,k),
\qquad
\rho_{\mathsf{u}}(h,k),
\qquad
\mathsf{NatUnaryStrictPrefix}(k,h)
$$
supplies the comparison branch. The middle branch is exact:
$$
\rho_{\mathsf{u}}(h,k)\Longleftrightarrow
(\mathsf{PreorderPrefixLE}(h,k)\land\mathsf{PreorderPrefixLE}(k,h)).
$$
The strict-left and strict-right branches cannot both hold, and neither strict
branch is compatible with $\rho_{\mathsf{u}}(h,k)$.

The per-object fields are: traditional target constructive total-order branch
partition; BEDC source $\mathsf{UnaryHistory}$, $\mathsf{PreorderPrefixLE}$, and
$\mathsf{NatUnaryStrictPrefix}$ over $\mathsf{BHist}$; carrier
$\mathsf{Carrier}_{\mathsf{u}}$; classifier $\rho_{\mathsf{u}}$; stability
obligations branch totality and incompatibility; ledger selected comparison
branch; core theorem this partition; exactness the same-branch equivalence;
standard bridge deferred; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Use the total unary prefix comparison of
\autoref{thm:nat-unary-prefix-total}. For carried histories $h$ and $k$, it gives
a unary continuation in one direction, and the unary tail is then split into
empty and nonempty cases.

If the selected tail is nonempty from $h$ to $k$, the selected branch is
$\mathsf{NatUnaryStrictPrefix}(h,k)$. If the selected tail is nonempty from $k$
to $h$, the selected branch is $\mathsf{NatUnaryStrictPrefix}(k,h)$.

If the selected tail is empty, the continuation collapses to the same-history
case. The carrier classifier is exactly the same-history branch, and
\autoref{thm:poset-prefix-classifier-exactness} identifies it with the pair of
prefix comparisons in both directions.

The two strict branches are incompatible by
\autoref{thm:nat-unary-strict-prefix-asymm}. A strict branch together with
$\rho_{\mathsf{u}}(h,k)$ would, by
\autoref{thm:poset-prefix-classifier-exactness}, also provide the opposite
prefix comparison; splitting that opposite comparison yields either an empty
tail contradicting strictness or the opposing strict branch, again ruled out by
\autoref{thm:nat-unary-strict-prefix-asymm}. The strict-right case is symmetric.

These cases give total coverage, the exact middle branch, and pairwise
incompatibility, so the three branches form the concrete constructive
partition.
\end{proof}

\begin{theorem}[Concrete unary prefix total-order least upper selector]
\label{thm:totalorder-prefix-least-upper-selector}
For carried unary histories $h,k$ in $\mathsf{Carrier}_{\mathsf{u}}$, there
exists a carried history $u$ in $\mathsf{Carrier}_{\mathsf{u}}$ such that
$$
h\le_{\mathsf{u}}u,
\qquad
k\le_{\mathsf{u}}u,
$$
and, for every carried history $r$ in $\mathsf{Carrier}_{\mathsf{u}}$,
$$
\bigl(h\le_{\mathsf{u}}r\land k\le_{\mathsf{u}}r\bigr)
\Rightarrow u\le_{\mathsf{u}}r.
$$

The per-object fields are: traditional target finite unary-word
maximum/least-upper selector; BEDC source $\mathsf{UnaryHistory}$ over
$\mathsf{BHist}$; carrier $\mathsf{Carrier}_{\mathsf{u}}$; classifier
$\rho_{\mathsf{u}}$; relation $\le_{\mathsf{u}}$; ledger the selected
comparison branch; core theorem this block; exactness least-upper property;
standard bridge finite unary prefix maximum; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Apply
\autoref{thm:concrete-unary-prefix-totalorder-comparison-partition} to the
carried pair $h,k$. The proof splits according to the selected branch.

In the strict-left branch $\mathsf{NatUnaryStrictPrefix}(h,k)$, choose $u=k$.
Forgetting the nonempty tail gives $h\le_{\mathsf{u}}k$, while reflexivity from
\autoref{thm:concrete-unary-prefix-preorder-laws} gives
$k\le_{\mathsf{u}}k$. If $r$ is any common upper bound of $h$ and $k$, the
second premise already supplies $k\le_{\mathsf{u}}r$, which is the required
leastness clause for the selected $u$.

In the same branch $\rho_{\mathsf{u}}(h,k)$, choose $u=k$. The exactness row in
\autoref{thm:concrete-unary-prefix-totalorder-comparison-partition} gives
$h\le_{\mathsf{u}}k$ and $k\le_{\mathsf{u}}h$; the displayed upper-bound clauses
again use $h\le_{\mathsf{u}}k$ and reflexivity of $k$. For every common upper
bound $r$, the premise $k\le_{\mathsf{u}}r$ is exactly the required comparison
from $u$ to $r$.

In the strict-right branch $\mathsf{NatUnaryStrictPrefix}(k,h)$, choose $u=h$.
The branch supplies $k\le_{\mathsf{u}}h$ after forgetting strictness, and
reflexivity gives $h\le_{\mathsf{u}}h$. For a common upper bound $r$, the first
premise supplies $h\le_{\mathsf{u}}r$, so the selected $u$ is below every common
upper bound.

These three cases construct the carried selector and verify both upper-bound
rows and the least-upper property.
\end{proof}
