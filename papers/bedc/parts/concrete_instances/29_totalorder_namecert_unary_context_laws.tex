\begin{theorem}[Concrete unary prefix total-order greatest lower selector]
\label{thm:totalorder-prefix-greatest-lower-selector}
For carried unary histories $h,k$ in $\mathsf{Carrier}_{\mathsf{u}}$, there
exists a carried history $l$ in $\mathsf{Carrier}_{\mathsf{u}}$ such that
$$
l\le_{\mathsf{u}}h,
\qquad
l\le_{\mathsf{u}}k,
$$
and, for every carried history $r$ in $\mathsf{Carrier}_{\mathsf{u}}$,
$$
\bigl(r\le_{\mathsf{u}}h\land r\le_{\mathsf{u}}k\bigr)
\Rightarrow r\le_{\mathsf{u}}l.
$$

The per-object fields are: traditional target finite unary-word
minimum/greatest-lower selector; BEDC source $\mathsf{UnaryHistory}$ over
$\mathsf{BHist}$; carrier $\mathsf{Carrier}_{\mathsf{u}}$; classifier
$\rho_{\mathsf{u}}$; relation $\le_{\mathsf{u}}$; ledger selected comparison
branch; core theorem this block; exactness greatest-lower property; downstream
consumer $\LatticeUp$ meet and unary interval certificates; status
$\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Apply
\autoref{thm:concrete-unary-prefix-totalorder-comparison-partition} to the
carried pair $h,k$. The proof splits according to the selected branch.

In the strict-left branch $\mathsf{NatUnaryStrictPrefix}(h,k)$, choose $l=h$.
Reflexivity from \autoref{thm:concrete-unary-prefix-preorder-laws} gives
$h\le_{\mathsf{u}}h$, and forgetting the nonempty tail gives
$h\le_{\mathsf{u}}k$. If $r$ is any common lower bound of $h$ and $k$, the
first premise already supplies $r\le_{\mathsf{u}}h$, which is the required
greatestness clause for the selected $l$.

In the same branch $\rho_{\mathsf{u}}(h,k)$, choose $l=h$. The exactness row in
\autoref{thm:concrete-unary-prefix-totalorder-comparison-partition} gives
$h\le_{\mathsf{u}}k$ and $k\le_{\mathsf{u}}h$; the displayed lower-bound clauses
use reflexivity of $h$ and the comparison $h\le_{\mathsf{u}}k$. For every
common lower bound $r$, the premise $r\le_{\mathsf{u}}h$ is exactly the
required comparison from $r$ to $l$.

In the strict-right branch $\mathsf{NatUnaryStrictPrefix}(k,h)$, choose $l=k$.
Forgetting strictness gives $k\le_{\mathsf{u}}h$, and reflexivity gives
$k\le_{\mathsf{u}}k$. For a common lower bound $r$, the second premise supplies
$r\le_{\mathsf{u}}k$, so every common lower bound lies below the selected $l$.

These three cases construct the carried selector and verify both lower-bound
rows and the greatest-lower property.
\end{proof}

\begin{theorem}[Concrete unary prefix total-order branch transport]
\label{thm:concrete-unary-prefix-totalorder-branch-transport}
For carried unary histories $h,h',k,k'$ in
$\mathsf{Carrier}_{\mathsf{u}}$, suppose
$\rho_{\mathsf{u}}(h,h')$ and $\rho_{\mathsf{u}}(k,k')$. Then classifier
replacement transports the three concrete comparison branches exactly:
$$
\mathsf{NatUnaryStrictPrefix}(h,k)\Longleftrightarrow
\mathsf{NatUnaryStrictPrefix}(h',k'),
$$
$$
\rho_{\mathsf{u}}(h,k)\Longleftrightarrow\rho_{\mathsf{u}}(h',k'),
$$
$$
\mathsf{NatUnaryStrictPrefix}(k,h)\Longleftrightarrow
\mathsf{NatUnaryStrictPrefix}(k',h').
$$
The per-object fields are: traditional target constructive total-order branch
transport; BEDC source $\mathsf{UnaryHistory}$ plus
$\mathsf{NatUnaryStrictPrefix}$ and $\rho_{\mathsf{u}}$; carrier
$\mathsf{Carrier}_{\mathsf{u}}$; classifier $\rho_{\mathsf{u}}$; stability
obligation comparison transport; ledger endpoint replacement witnesses; core
theorem this block; exactness the three displayed iff rows; standard bridge
finite unary word total order; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Unpack the two classifier witnesses. The witness
$\rho_{\mathsf{u}}(h,h')$ gives the endpoint comparison $\hsame(h,h')$, and
$\rho_{\mathsf{u}}(k,k')$ gives $\hsame(k,k')$.

For the strict-left row, the forward direction applies
\autoref{thm:nat-unary-strict-prefix-cont-hsame-transport} to the strict
witness from $h$ to $k$ with endpoint witnesses $\hsame(h,h')$ and
$\hsame(k,k')$.

The reverse direction uses the same transport theorem with the symmetric
endpoint witnesses $\hsame(h',h)$ and $\hsame(k',k)$.

For the classifier row, compose the endpoint comparison $\hsame(h,h')$, the
middle comparison extracted from $\rho_{\mathsf{u}}(h,k)$, and the symmetric
form of $\hsame(k,k')$. This gives $\hsame(h',k')$, which repackages with the
transported carrier witnesses as $\rho_{\mathsf{u}}(h',k')$.

The reverse classifier direction is the same composition with the two endpoint
replacement witnesses reversed. Symmetry and transitivity of the carrier
classifier recover $\rho_{\mathsf{u}}(h,k)$.

For the strict-right row, read the strict witness as a strict unary prefix from
$k$ to $h$. Apply the strict-prefix transport theorem with endpoints
$\hsame(k,k')$ and $\hsame(h,h')$.

The reverse strict-right direction again uses the symmetric endpoint witnesses.
Thus all three comparison branches are preserved and reflected under concrete
classifier replacement.
\end{proof}

\begin{definition}[Concrete unary successor comparison package]
\label{def:concrete-unary-prefix-totalorder-successor-package}
For a carried unary history $h$ in $\mathsf{Carrier}_{\mathsf{u}}$, the concrete
successor comparison package is the pair
$(h,\Eone(h))$ over the carrier $\mathsf{Carrier}_{\mathsf{u}}$, classifier
$\rho_{\mathsf{u}}$, and order $\le_{\mathsf{u}}$ from
\autoref{def:concrete-unary-prefix-totalorder-instance}. Its selected comparison
tail is the visible one-step ledger element $\Eone(\emp)$, so the branch data are
fixed as
$$
\mathsf{NatUnaryStrictPrefix}(h,\Eone(h)),
\qquad
\rho_{\mathsf{u}}(h,\Eone(h)),
\qquad
\mathsf{NatUnaryStrictPrefix}(\Eone(h),h).
$$
The package records the first branch as the selected concrete successor branch
and retains the other two branches as explicit exclusion targets.
\end{definition}

\begin{theorem}[Concrete unary successor total-order strict branch]
\label{thm:concrete-unary-prefix-totalorder-successor-strict}
For every $h$ with $\mathsf{Carrier}_{\mathsf{u}}(h)$, the successor target
$\Eone(h)$ is carried by $\mathsf{Carrier}_{\mathsf{u}}$ and the concrete
successor comparison package of
\autoref{def:concrete-unary-prefix-totalorder-successor-package} satisfies
$$
\mathsf{NatUnaryStrictPrefix}(h,\Eone(h)).
$$
Moreover,
$$
\rho_{\mathsf{u}}(h,\Eone(h))\to\mathsf{False}
\qquad
\text{and}
\qquad
\mathsf{NatUnaryStrictPrefix}(\Eone(h),h)\to\mathsf{False}.
$$

The per-object fields are: traditional target one-step successor separation; BEDC
source $\mathsf{BHist}$, $\mathsf{UnaryHistory}$,
$\mathsf{NatUnaryStrictPrefix}$, and $\mathsf{PreorderPrefixLE}$; carrier
$\mathsf{Carrier}_{\mathsf{u}}$; classifier $\rho_{\mathsf{u}}$; ledger tail
$\Eone(\emp)$; core theorem this successor strict-branch law; exactness the
exclusion of same and reverse-strict branches; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Unary closure of $\mathsf{Carrier}_{\mathsf{u}}=\mathsf{UnaryHistory}$ under the
$\Eone$ constructor gives the carried target $\Eone(h)$.

Apply \autoref{lem:nat-unary-strict-prefix-one-step} to the unary witness for
$h$. The selected tail is $\Eone(\emp)$, so this produces
$\mathsf{NatUnaryStrictPrefix}(h,\Eone(h))$ for the concrete successor pair.

Now instantiate
\autoref{thm:concrete-unary-prefix-totalorder-comparison-partition} at
$h$ and $\Eone(h)$. Its incompatibility clause says that the selected strict-left
branch cannot coexist with the same branch
$\rho_{\mathsf{u}}(h,\Eone(h))$.

The same partition also excludes coexistence of the selected strict-left branch
with the strict-right branch
$\mathsf{NatUnaryStrictPrefix}(\Eone(h),h)$. Therefore any witness of either
excluded branch yields $\mathsf{False}$.
\end{proof}

For carried unary histories $p,h$ in $\mathsf{Carrier}_{\mathsf{u}}$, both

\rho_{\mathsf{u}}(\mathsf{append}(p,h),\mathsf{append}(p,\Eone(h)))

The per-object fields are: traditional target context-stable one-step successor

$\mathsf{append}$, $\mathsf{NatUnaryStrictPrefix}$, and
$\mathsf{PreorderPrefixLE}$; carrier $\mathsf{Carrier}_{\mathsf{u}}$;

Apply \autoref{lem:nat-unary-strict-prefix-one-step} to the unary witness for
$h$, obtaining the strict successor tail $\Eone(\emp)$ from $h$ to $\Eone(h)$.
Continuation associativity transports this same nonempty tail through the left

$\mathsf{NatUnaryStrictPrefix}$ witness between the contextual endpoints.

Instantiate
\autoref{thm:concrete-unary-prefix-totalorder-comparison-partition} at
$\mathsf{append}(p,h)$ and $\mathsf{append}(p,\Eone(h))$. Its incompatibility

$\rho_{\mathsf{u}}(\mathsf{append}(p,h),\mathsf{append}(p,\Eone(h)))$ and with

$\mathsf{NatUnaryStrictPrefix}
(\mathsf{append}(p,\Eone(h)),\mathsf{append}(p,h))$. Therefore either excluded
branch yields $\mathsf{False}$.
\end{proof}

\begin{theorem}[Concrete unary strict branch transitivity]
\label{thm:totalorder-strict-branch-transitivity}
For carried unary histories $h,k,l$ in $\mathsf{Carrier}_{\mathsf{u}}$,
strict-left comparison is transitive:
$$
\mathsf{NatUnaryStrictPrefix}(h,k)\land
\mathsf{NatUnaryStrictPrefix}(k,l)
\to
\mathsf{NatUnaryStrictPrefix}(h,l).
$$
The strict-right branch is transitive in the same concrete carrier:
$$
\mathsf{NatUnaryStrictPrefix}(k,h)\land
\mathsf{NatUnaryStrictPrefix}(l,k)
\to
\mathsf{NatUnaryStrictPrefix}(l,h).
$$
Consequently $\mathsf{lt}_{\mathsf{u}}$ and $\mathsf{gt}_{\mathsf{u}}$ are
transitive on the concrete unary prefix $\TotalOrderUp$ instance.
\end{theorem}
\begin{proof}
Unfold the strict-left branch of
\autoref{def:concrete-unary-prefix-totalorder-instance}. The two hypotheses are
exactly a strict unary prefix from $h$ to $k$ and a strict unary prefix from $k$
to $l$. Apply
\autoref{thm:nat-unary-strict-prefix-trans-composite-tail}; its composite tail
witness gives $\mathsf{NatUnaryStrictPrefix}(h,l)$.

For strict-right comparison, unfold
$\mathsf{gt}_{\mathsf{u}}(x,y):=\mathsf{NatUnaryStrictPrefix}(y,x)$. The
hypotheses become strict unary prefixes from $l$ to $k$ and from $k$ to $h$.
Applying the same composite-tail transitivity theorem yields
$\mathsf{NatUnaryStrictPrefix}(l,h)$.

The carrier witnesses for $h$, $k$, and $l$ are not consumed by the strict-prefix
composition itself; they keep the conclusion inside the concrete
$\mathsf{Carrier}_{\mathsf{u}}$ total-order package. The two displayed
implications are therefore precisely the transitivity fields for
$\mathsf{lt}_{\mathsf{u}}$ and $\mathsf{gt}_{\mathsf{u}}$.
\end{proof}

\begin{theorem}[Concrete unary total-order branch context equivalence]
\label{thm:totalorder-branch-two-sided-context-equivalence}
Let $h,k,\ell,r$ be unary histories in
$\mathsf{Carrier}_{\mathsf{u}}=\mathsf{UnaryHistory}$, and set
$$
H:=\mathsf{append}(\ell,\mathsf{append}(h,r)),
\qquad
K:=\mathsf{append}(\ell,\mathsf{append}(k,r)).
$$
Then $H$ and $K$ are carried unary histories, and the three comparison branches
of \autoref{def:concrete-unary-prefix-totalorder-instance} are invariant under
this common two-sided unary context:
\begin{enumerate}
  \item $\mathsf{lt}_{\mathsf{u}}(H,K)\Longleftrightarrow
  \mathsf{lt}_{\mathsf{u}}(h,k)$.
  \item $\mathsf{same}_{\mathsf{u}}(H,K)\Longleftrightarrow
  \mathsf{same}_{\mathsf{u}}(h,k)$.
  \item $\mathsf{gt}_{\mathsf{u}}(H,K)\Longleftrightarrow
  \mathsf{gt}_{\mathsf{u}}(h,k)$.
\end{enumerate}
\end{theorem}
\begin{proof}
Unary closure under append first gives
$\mathsf{UnaryHistory}(\mathsf{append}(h,r))$ and
$\mathsf{UnaryHistory}(\mathsf{append}(k,r))$. Applying the same closure once
more with the unary left context $\ell$ gives the carrier facts for $H$ and $K$.

For the same branch, unfold
$\mathsf{same}_{\mathsf{u}}:=\rho_{\mathsf{u}}$. By
\autoref{thm:poset-prefix-classifier-exactness}, the classifier on carried
unary histories is equivalent to the pair of opposing prefix comparisons.
Apply \autoref{thm:preorder-prefix-append-both-context-iff} to the comparison
$H\le_{\mathsf{u}}K$ and again to $K\le_{\mathsf{u}}H$; the two applications
identify the contextual pair of prefix comparisons exactly with
$h\le_{\mathsf{u}}k$ and $k\le_{\mathsf{u}}h$. A second use of
\autoref{thm:poset-prefix-classifier-exactness} repacks these comparisons as
$\rho_{\mathsf{u}}(h,k)$.

For the strict-left branch, unfold
$\mathsf{lt}_{\mathsf{u}}$ as
$\mathsf{NatUnaryStrictPrefix}$. In the forward direction from
$\mathsf{NatUnaryStrictPrefix}(h,k)$, unpack the nonempty unary tail from $h$ to
$k$. Right-context preservation composes this tail past $r$ by continuation
associativity and unary-tail commutativity; left-context preservation then
places $\ell$ outside both endpoints. The tail remains the same nonempty unary
tail, so the result is $\mathsf{NatUnaryStrictPrefix}(H,K)$.

For the reverse strict-left direction, unpack the nonempty unary tail from $H$
to $K$. Its underlying prefix comparison cancels through
\autoref{thm:preorder-prefix-cancel-both-context}, leaving a prefix comparison
from $h$ to $k$. If the recovered central tail were empty, append-empty
exactness would collapse $h$ and $k$, hence also $H$ and $K$ after the common
unary context. Comparing that collapse with the original strict contextual tail
would force the original tail empty, contradicting its strictness. Thus the
central prefix tail is nonempty and gives
$\mathsf{NatUnaryStrictPrefix}(h,k)$.

The strict-right branch is the same argument with endpoints reversed, since
$\mathsf{gt}_{\mathsf{u}}(x,y)$ unfolds to
$\mathsf{NatUnaryStrictPrefix}(y,x)$. Applying the strict-left preservation and
reflection just proved to the ordered pair $(k,h)$ gives the displayed
equivalence for $\mathsf{gt}_{\mathsf{u}}$.
\end{proof}

\begin{theorem}[Concrete unary successor strict branch under two-sided context]
\label{thm:concrete-unary-prefix-totalorder-successor-two-sided-context-strict}
For carried unary histories $\ell,h,r$ in
$\mathsf{Carrier}_{\mathsf{u}}$, set
$$
H:=\mathsf{append}(\ell,\mathsf{append}(h,r)),
\qquad
K:=\mathsf{append}(\ell,\mathsf{append}(\Eone(h),r)).
$$
Then $H$ and $K$ are carried by $\mathsf{Carrier}_{\mathsf{u}}$, and the
contextual successor pair satisfies
$$
\mathsf{lt}_{\mathsf{u}}(H,K).
$$
Moreover,
$$
\mathsf{same}_{\mathsf{u}}(H,K)\to\mathsf{False}
\qquad
\text{and}
\qquad
\mathsf{gt}_{\mathsf{u}}(H,K)\to\mathsf{False}.
$$

The per-object fields are: traditional target context-stable one-step successor
separation; BEDC source $\mathsf{BHist}$, $\mathsf{UnaryHistory}$,
$\mathsf{append}$, $\Eone$, $\mathsf{lt}_{\mathsf{u}}$,
$\mathsf{same}_{\mathsf{u}}$, and $\mathsf{gt}_{\mathsf{u}}$; carrier
$\mathsf{Carrier}_{\mathsf{u}}$; classifier $\rho_{\mathsf{u}}$; ledger common
two-sided context $(\ell,r)$ plus tail $\Eone(\emp)$; core theorem this
two-sided contextual successor strict-branch law; exactness the exclusion of
same and reverse-strict branches; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Apply
\autoref{thm:concrete-unary-prefix-totalorder-successor-strict} to the carried
history $h$. This gives the carried successor $\Eone(h)$, the base branch
$\mathsf{lt}_{\mathsf{u}}(h,\Eone(h))$, and the two base exclusions
$\mathsf{same}_{\mathsf{u}}(h,\Eone(h))\to\mathsf{False}$ and
$\mathsf{gt}_{\mathsf{u}}(h,\Eone(h))\to\mathsf{False}$.

Instantiate
\autoref{thm:totalorder-branch-two-sided-context-equivalence} with the carried
quadruple $h,\Eone(h),\ell,r$. Its carrier conclusion gives that the displayed
$H$ and $K$ are carried unary histories.

Use the right-to-left direction of the strict-left branch equivalence from the
same theorem. Transporting the base successor branch
$\mathsf{lt}_{\mathsf{u}}(h,\Eone(h))$ through the common context
$(\ell,r)$ gives $\mathsf{lt}_{\mathsf{u}}(H,K)$.

For a same-branch witness at the contextual endpoints, use the left-to-right
direction of the same-branch equivalence to obtain
$\mathsf{same}_{\mathsf{u}}(h,\Eone(h))$. The base same-branch exclusion then
returns $\mathsf{False}$.

For a reverse strict-branch witness at the contextual endpoints, use the
left-to-right direction of the strict-right branch equivalence to obtain
$\mathsf{gt}_{\mathsf{u}}(h,\Eone(h))$. The base reverse-branch exclusion then
returns $\mathsf{False}$.
\end{proof}

comparison branch certificate; exactness same branch iff opposing prefix

The $\TotalOrderUp$ schema is represented by this concrete $\mathsf{BHist}$

\end{proof}

\begin{theorem}[Concrete unary prefix total-order finite minimum selector]
\label{thm:totalorder-prefix-finite-minimum-selector}
Let $L$ be a nonempty finite spine of histories in the finite-list helper of
\autoref{def:list-carrier}. Assume every entry of $L$ is carried by
$\mathsf{Carrier}_{\mathsf{u}}$ for the concrete $\TotalOrderUp$ certificate of
\autoref{thm:concrete-unary-prefix-totalorder-namecert}. Then there is an entry
$m\in L$ such that, for every entry $x\in L$,
$$
m\le_{\mathsf{u}}x.
$$

The per-object fields are: traditional target finite-list minimum extraction in
a total order; BEDC source finite spines of carried unary histories; carrier
$\mathsf{Carrier}_{\mathsf{u}}$; classifier $\rho_{\mathsf{u}}$; relation
$\le_{\mathsf{u}}$; stability obligations total comparison, reflexivity, and
transitivity; ledger the tail-minimum witness and the selected comparison branch
at each cons step; core theorem this finite minimum selector; exactness
membership plus the displayed lower-bound row; standard bridge finite unary-word
total order; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Induct on the nonempty finite spine $L$.

For the singleton spine $L=[a]$, choose $m=a$. The membership clause is
immediate, and the only entry to check is $a$ itself. Reflexivity from
\autoref{thm:concrete-unary-prefix-preorder-laws} gives $a\le_{\mathsf{u}}a$.

For the cons step, write $L=a::T$ with $T$ nonempty and all its entries carried
by $\mathsf{Carrier}_{\mathsf{u}}$. By the induction hypothesis applied to
$T$, choose $m_T\in T$ such that $m_T\le_{\mathsf{u}}x$ for every $x\in T$.
Apply \autoref{thm:concrete-unary-prefix-totalorder-comparison-partition} to
the carried pair $a,m_T$.

In the strict-left branch $\mathsf{NatUnaryStrictPrefix}(a,m_T)$, choose
$m=a$. By \autoref{def:nat-unary-strict-prefix} and
\autoref{def:concrete-unary-prefix-totalorder-instance}, forgetting the
nonempty-tail condition gives $a\le_{\mathsf{u}}m_T$. For $x=a$, reflexivity
gives $a\le_{\mathsf{u}}a$. For $x\in T$, compose
$a\le_{\mathsf{u}}m_T$ with the induction-hypothesis comparison
$m_T\le_{\mathsf{u}}x$ using \autoref{thm:preorder-prefix-transitivity}. Thus
the head $a$ is a lower bound for every entry of $L$.

In the same branch $\rho_{\mathsf{u}}(a,m_T)$, choose $m=a$. The exact
middle-branch row of
\autoref{thm:concrete-unary-prefix-totalorder-comparison-partition} gives
$a\le_{\mathsf{u}}m_T$ and $m_T\le_{\mathsf{u}}a$. The entry $a$ is below
itself by reflexivity, and it is below every tail entry by transitivity through
$m_T$, exactly as in the strict-left case.

In the strict-right branch $\mathsf{NatUnaryStrictPrefix}(m_T,a)$, choose
$m=m_T$. Its membership in $L$ follows from $m_T\in T$. Forgetting the
nonempty-tail condition gives $m_T\le_{\mathsf{u}}a$, which handles the head
entry. Every tail entry is handled by the induction hypothesis. Hence the
selected tail minimum is below every entry of $L$.

All comparison branches construct an entry of $L$ satisfying the displayed
lower-bound condition, so the finite minimum selector exists for every nonempty
finite spine of carried concrete $\TotalOrderUp$ elements.
\end{proof}

\begin{theorem}[Concrete unary prefix total-order finite maximum index selector]
\label{thm:totalorder-prefix-finite-maximum-index-selector}
Let $L$ be a nonempty finite spine of histories in the finite-list helper of
\autoref{def:list-carrier}. Assume every entry of $L$ is carried by
$\mathsf{Carrier}_{\mathsf{u}}$ for the concrete $\TotalOrderUp$ certificate of
\autoref{thm:concrete-unary-prefix-totalorder-namecert}. Then there are a
finite-spine membership witness $i$ and a carried entry $M=L_i$ such that, for
every entry $x\in L$,
$$
x\le_{\mathsf{u}}M.
$$

The per-object fields are: traditional target finite-list maximum extraction in
a total order; BEDC source finite spines of carried unary histories; carrier
$\mathsf{Carrier}_{\mathsf{u}}$; classifier $\rho_{\mathsf{u}}$; relation
$\le_{\mathsf{u}}$; stability obligations total comparison, reflexivity, and
transitivity; ledger the tail-maximum membership witness and the selected
comparison branch at each cons step; core theorem this finite maximum selector;
exactness the displayed membership witness plus upper-bound row; standard
bridge finite unary-word total order; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Induct on the nonempty finite spine $L$.

For the singleton spine $L=[a]$, choose the unique membership position and set
$M=a$. The only entry to check is $a$ itself, and reflexivity from
\autoref{thm:concrete-unary-prefix-preorder-laws} gives $a\le_{\mathsf{u}}a$.

For the cons step, write $L=a::T$ with $T$ nonempty and all its entries carried
by $\mathsf{Carrier}_{\mathsf{u}}$. By the induction hypothesis applied to
$T$, choose a tail membership witness $i_T$ and an entry $M_T\in T$ such that
$x\le_{\mathsf{u}}M_T$ for every $x\in T$. Apply
\autoref{thm:concrete-unary-prefix-totalorder-comparison-partition} to the
carried pair $a,M_T$.

In the strict-left branch $\mathsf{NatUnaryStrictPrefix}(a,M_T)$, choose the
same tail membership witness, viewed as a membership witness in $a::T$, and set
$M=M_T$. By \autoref{def:nat-unary-strict-prefix} and
\autoref{def:concrete-unary-prefix-totalorder-instance}, forgetting the
nonempty-tail condition gives $a\le_{\mathsf{u}}M_T$, which handles the head
entry. Every tail entry is below $M_T$ by the induction hypothesis.

In the same branch $\rho_{\mathsf{u}}(a,M_T)$, again choose the tail membership
witness and set $M=M_T$. The exact middle-branch row of
\autoref{thm:concrete-unary-prefix-totalorder-comparison-partition} gives
$a\le_{\mathsf{u}}M_T$ and $M_T\le_{\mathsf{u}}a$. The first comparison handles
the head entry, and every tail entry is handled by the induction hypothesis.

In the strict-right branch $\mathsf{NatUnaryStrictPrefix}(M_T,a)$, choose the
head membership witness and set $M=a$. Forgetting strictness gives
$M_T\le_{\mathsf{u}}a$. For the head entry, reflexivity gives
$a\le_{\mathsf{u}}a$. For a tail entry $x\in T$, the induction hypothesis gives
$x\le_{\mathsf{u}}M_T$; composing this with $M_T\le_{\mathsf{u}}a$ by
\autoref{thm:preorder-prefix-transitivity} gives $x\le_{\mathsf{u}}a$.

All comparison branches construct a located entry of $L$ above every entry of
$L$, so the finite maximum selector exists for every nonempty finite spine of
carried concrete $\TotalOrderUp$ elements.
\end{proof}

\begin{definition}[Concrete unary finite minimum witness]
\label{def:totalorder-prefix-finite-minimum-witness}
For a finite BEDC-history spine $L:\mathsf{ListCarrier}(\mathsf{BHist})$, a visible readback $m$ is a concrete unary finite minimum witness when
$$
\mathsf{FiniteMinimumWitness}_{\mathsf{u}}(L,m)
:=
m\in L
\land

The carrier side condition is supplied by the finite-carrier list hypothesis used with the witness.
\end{definition}

\begin{theorem}[Concrete unary prefix finite minimum classifier uniqueness]
\label{thm:totalorder-prefix-finite-minimum-classifier-unique}
Let $L:\mathsf{ListCarrier}(\mathsf{BHist})$ be a nonempty finite spine in the helper of \autoref{def:list-carrier}. Assume every entry of $L$ is carried by $\mathsf{Carrier}_{\mathsf{u}}$ for the concrete $\TotalOrderUp$ certificate of \autoref{thm:concrete-unary-prefix-totalorder-namecert}:
$$
\forall x.\; x\in L\to \mathsf{Carrier}_{\mathsf{u}}(x).
$$
If $m$ and $n$ satisfy $\mathsf{FiniteMinimumWitness}_{\mathsf{u}}(L,m)$ and $\mathsf{FiniteMinimumWitness}_{\mathsf{u}}(L,n)$, then
$$
\rho_{\mathsf{u}}(m,n).
$$

The per-object fields are: traditional target finite-list minimum witness uniqueness in a total order; BEDC source finite spines of carried unary histories; carrier $\mathsf{Carrier}_{\mathsf{u}}$; classifier $\rho_{\mathsf{u}}$; relation $\le_{\mathsf{u}}$; stability obligations finite-list lower-bound readout and inherited $\POSetUp$ antisymmetry; ledger the two minimum witness records and the two extracted comparisons; core theorem this block; exactness classifier equality of competing minimum readbacks; standard bridge finite unary-word minimum uniqueness; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Unpack the two finite minimum witness records of \autoref{def:totalorder-prefix-finite-minimum-witness}. The first supplies $m\in L$ and the lower-bound row $\forall x.\; x\in L\to m\le_{\mathsf{u}}x$; the second supplies $n\in L$ and the lower-bound row $\forall x.\; x\in L\to n\le_{\mathsf{u}}x$.

From the entry-carrier hypothesis, the membership rows give carrier witnesses for $m$ and $n$. Instantiate the first lower-bound row at the entry $n$; this gives $m\le_{\mathsf{u}}n$. Instantiate the second lower-bound row at the entry $m$; this gives $n\le_{\mathsf{u}}m$.

By \autoref{def:concrete-unary-prefix-totalorder-instance}, these comparisons are the concrete prefix relation of the unary total-order package. The exact middle-branch row of \autoref{thm:concrete-unary-prefix-totalorder-comparison-partition}, equivalently the $\POSetUp$ classifier exactness of \autoref{thm:poset-prefix-classifier-exactness} in this notation, sends the carried opposing comparisons to $\rho_{\mathsf{u}}(m,n)$.

The strict-left and strict-right branches of the total comparison ledger are therefore excluded by the incompatibility clauses of \autoref{thm:concrete-unary-prefix-totalorder-comparison-partition}. Together with the existence selector of \autoref{thm:totalorder-prefix-finite-minimum-selector}, this gives a classifier-stable finite minimum readback for every nonempty finite carried spine.
\end{proof}

\begin{theorem}[Concrete unary prefix finite extrema certificate]
\label{thm:totalorder-prefix-finite-extrema-certificate}
Let $L:\mathsf{ListCarrier}(\mathsf{BHist})$ be a nonempty finite BEDC-history
spine in the helper of \autoref{def:list-carrier}. Assume every entry of $L$ is
carried by $\mathsf{Carrier}_{\mathsf{u}}$ for the concrete $\TotalOrderUp$
certificate of \autoref{thm:concrete-unary-prefix-totalorder-namecert}:
$$
\forall x.\; x\in L\to \mathsf{Carrier}_{\mathsf{u}}(x).
$$
Then there are histories $m$ and $M$ such that
$$
\mathsf{FiniteMinimumWitness}_{\mathsf{u}}(L,m)
\land
\mathsf{FiniteMaximumWitness}_{\mathsf{u}}(L,M)
$$
and the selected readbacks are classifier-stable: every other minimum witness
$m'$ satisfying $\mathsf{FiniteMinimumWitness}_{\mathsf{u}}(L,m')$ obeys
$\rho_{\mathsf{u}}(m,m')$, and every other maximum witness $M'$ satisfying
$\mathsf{FiniteMaximumWitness}_{\mathsf{u}}(L,M')$ obeys
$\rho_{\mathsf{u}}(M,M')$.

The per-object fields are: traditional target finite extrema in a constructive
total order; BEDC source finite $\mathsf{ListCarrier}$ spines and unary
$\TotalOrderUp$; carrier $\mathsf{Carrier}_{\mathsf{u}}$; classifier
$\rho_{\mathsf{u}}$; stability selector existence plus classifier uniqueness;
ledger the selected minimum and maximum witnesses; core theorem this finite
extrema certificate; exactness classifier uniqueness of extrema; standard
bridge finite unary-word extrema; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Apply \autoref{thm:totalorder-prefix-finite-minimum-selector} to the nonempty
carried spine $L$. This gives an entry $m\in L$ whose lower-bound row is exactly
$\mathsf{FiniteMinimumWitness}_{\mathsf{u}}(L,m)$ after repacking through
\autoref{def:totalorder-prefix-finite-minimum-witness}.

Apply \autoref{thm:totalorder-prefix-finite-maximum-index-selector} to the same
carried spine. Its membership index and selected entry give a history $M\in L$
whose upper-bound row is exactly
$\mathsf{FiniteMaximumWitness}_{\mathsf{u}}(L,M)$ after repacking through
\autoref{def:totalorder-prefix-finite-maximum-witness}.

For any competing minimum witness $m'$, apply
\autoref{thm:totalorder-prefix-finite-minimum-classifier-unique} to the two
minimum witness records for $m$ and $m'$. The result is the displayed classifier
agreement $\rho_{\mathsf{u}}(m,m')$.

For any competing maximum witness $M'$, apply
\autoref{thm:totalorder-prefix-finite-maximum-classifier-unique} to the two
maximum witness records for $M$ and $M'$. This gives
$\rho_{\mathsf{u}}(M,M')$, completing the paired finite-extrema certificate.
\end{proof}
