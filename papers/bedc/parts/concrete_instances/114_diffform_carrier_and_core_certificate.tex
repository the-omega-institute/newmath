\chapter{A Concrete Naming Certificate for $\DiffFormUp$}
\label{ch:concrete-instances-diffform-namecert}

The differential-form interface $\DiffFormUp$ packages the carrier of antisymmetric multilinear maps on the tangent bundle of a manifold, together with the wedge product. It depends on $\NameCert_{\ManifoldUp}$ (\autoref{ch:concrete-instances-manifold-namecert}) and $\NameCert_{\TensorProductUp}$ (\autoref{ch:concrete-instances-tensorproduct-namecert}), and is the substrate for the de~Rham complex.

\section{Carrier definition}
\label{sec:diffform-carrier}

\begin{definition}[Differential-form BHist form carrier]
\label{def:diffform-bhist-form-carrier}
Fix a $\NameCert_{\ManifoldUp}$ source $M$ and a
$\NameCert_{\TensorProductUp}$ source $T$ whose tensor rows are read over the
tangent-bundle histories of $M$. A differential-form BHist row is a tuple of
histories
$$
(d,\mathcal{P},\tau,\sigma,\alpha,\lambda)
:
\mathsf{BHist}^6
$$
accepted by the predicate $\mathsf{DiffFormBHistCarrier}_{M,T}$ exactly when
the following BEDC rows are all present:
\begin{enumerate}
  \item $\mathsf{UnaryHistory}(d)$ records the form degree.
  \item $\mathcal{P}:\mathsf{ProbeBundle}$ is the visible finite spine of tangent
  probe names, and each bundled probe row is carried by the tangent-source
  surface exposed by $\NameCert_{\ManifoldUp}$.
  \item $\tau$ is a $\TensorProductUp$ multilinear row over the bundled
  tangent probes, with continuation ledgers $\Cont$ recording the ordered
  tensor concatenations used by the row.
  \item $\sigma$ is the scalar endpoint produced from that tensor row,
  classified by the scalar classifier retained by the manifold and tensor
  sources.
  \item $\alpha$ is the visible permutation ledger: for every displayed
  adjacent swap of bundled probe rows, the transported scalar endpoint is
  classified with the signed endpoint demanded by antisymmetry.
  \item $\lambda$ records the source names $M$ and $T$, the degree row, the
  probe bundle, the tensor row, the scalar endpoint row, and the antisymmetry
  rows as the only accepted provenance for the form row.
\end{enumerate}
\end{definition}

\begin{definition}[Differential-form BHist form classifier]
\label{def:diffform-bhist-form-classifier}
For carried form rows
$$
\omega=(d,\mathcal{P},\tau,\sigma,\alpha,\lambda)
$$
and
$$
\omega'=(d',\mathcal{P}',\tau',\sigma',\alpha',\lambda'),
$$
the classifier $\omega\sim_{\DiffFormUp}\omega'$ holds when the degree rows
are $\hsame$, the scalar endpoints agree pointwise under the retained scalar
classifier for every matched multilinear probe row, and the antisymmetry and
tensor-source ledgers match as visible BHist rows:
$$
\hsame(d,d')
$$
$$
\text{and for every matched bundled multilinear row, }
\sigma\sim_R\sigma'
$$
$$
\text{and }
\hsame(\alpha,\alpha')\wedge\hsame(\tau,\tau')
\wedge\hsame(\lambda,\lambda').
$$
Here $\sim_R$ is the scalar classifier already present in the upstream scalar
surface; the differential-form classifier introduces no classifier outside
$\hsame$ and the displayed source classifiers.
\end{definition}

\section{The certificate}
\label{sec:diffform-certificate}

\begin{definition}[Differential-form BHist NameCert core rows]
\label{def:diffform-bhist-namecert-core-rows}
Fix a $\NameCert_{\ManifoldUp}$ source $M$ and a
$\NameCert_{\TensorProductUp}$ source $T$ as in
\autoref{def:diffform-bhist-form-carrier}. The $\DiffFormUp$ BHist
NameCert core-row package consists of the following ten displayed template
fields:
\begin{enumerate}
  \item the carrier row
  $\mathsf{DiffFormBHistCarrier}_{M,T}$;
  \item the classifier row $\sim_{\DiffFormUp}$ of
  \autoref{def:diffform-bhist-form-classifier};
  \item the componentwise $\hsame$ transport row for carried forms;
  \item the degree history row $d$;
  \item the tangent-probe $\mathsf{ProbeBundle}$ row $\mathcal{P}$;
  \item the $\TensorProductUp$ multilinear row $\tau$;
  \item the tensor-continuation ledger $\Cont$ attached to $\tau$;
  \item the scalar endpoint classifier row retained from the upstream
  manifold and tensor sources;
  \item the antisymmetry permutation ledger $\alpha$;
  \item the standard-bridge slot, fixed to absent data.
\end{enumerate}
These fields are read as template data for the certificate only. The package
does not introduce an exterior-derivative row, a closedness predicate, a
nondegeneracy predicate, or a standard bridge.
\end{definition}

\begin{theorem}[Differential-form BHist carrier obligation]
\label{thm:diffform-bhist-carrier-obligation}
The carrier row in
\autoref{def:diffform-bhist-namecert-core-rows} is exhausted by the six
coordinates
$$
(d,\mathcal{P},\tau,\sigma,\alpha,\lambda)
$$
of \autoref{def:diffform-bhist-form-carrier}. In particular, accepting a
$\DiffFormUp$ carrier row adds no seventh coordinate and no hidden source
outside the displayed degree, probe-bundle, tensor, scalar-endpoint,
antisymmetry, and source-ledger histories.
\end{theorem}
\begin{proof}
Invert the acceptance predicate
$\mathsf{DiffFormBHistCarrier}_{M,T}$ displayed in
\autoref{def:diffform-bhist-form-carrier}. A carried row is presented as the
six-tuple $(d,\mathcal{P},\tau,\sigma,\alpha,\lambda)$, and each acceptance
clause projects one of those six histories or one relation among histories
already named by the tuple.

The degree clause projects $d$, the bundled tangent-probe clause projects
$\mathcal{P}$, the multilinear tensor clause projects $\tau$, the scalar
endpoint clause projects $\sigma$, the permutation ledger clause projects
$\alpha$, and the source-provenance clause projects $\lambda$. No clause has
a constructor that allocates another coordinate.

It remains to check that the core-row package of
\autoref{def:diffform-bhist-namecert-core-rows} does not enlarge this
carrier. Its transport, classifier, continuation, endpoint-classifier, and
bridge slots refer back to the six displayed carrier histories and to
upstream certificate sources. Thus carrier acceptance is exactly the six
coordinate surface of \autoref{def:diffform-bhist-form-carrier}.
\end{proof}

\begin{theorem}[Differential-form BHist classifier obligation]
\label{thm:diffform-bhist-classifier-obligation}
The classifier row in
\autoref{def:diffform-bhist-namecert-core-rows} uses only degree
$\hsame$, matched tensor and scalar comparisons, and visible ledger
$\hsame$. It introduces no classifier other than $\hsame$ on displayed BHist
rows and the scalar classifier already retained by the manifold and tensor
sources.
\end{theorem}
\begin{proof}
Project the classifier definition from
\autoref{def:diffform-bhist-form-classifier}. The first clause is exactly
$\hsame(d,d')$, so degree comparison is a visible BHist comparison and not a
new differential-form relation.

For the tensor and scalar part, the statement quantifies only over matched
bundled multilinear rows. The tensor match is the visible comparison of the
displayed tensor histories, while the endpoint comparison is the retained
scalar classifier $\sim_R$ supplied by the upstream scalar surface. The
classifier therefore reuses source comparisons already present in $M$ and
$T$.

The remaining clauses are
$\hsame(\alpha,\alpha')$, $\hsame(\tau,\tau')$, and
$\hsame(\lambda,\lambda')$. These are visible ledger comparisons on the
antisymmetry, tensor, and source-provenance histories. Hence every accepted
classifier step is one of the claimed degree, matched tensor/scalar, or
ledger comparisons.
\end{proof}

\begin{theorem}[Differential-form BHist ledger exactness obligation]
\label{thm:diffform-bhist-ledger-exactness-obligation}
Every ledger row accepted by the core-row package of
\autoref{def:diffform-bhist-namecert-core-rows} is visible BHist provenance
from $BHist$, $\hsame$, $\Cont$, $\mathsf{ProbeBundle}$,
$\NameCert_{\ManifoldUp}$, and $\NameCert_{\TensorProductUp}$. No accepted
ledger row may cite a hidden differential-form primitive or a standard bridge.
\end{theorem}
\begin{proof}
First inspect the carrier ledger. By
\autoref{def:diffform-bhist-form-carrier}, $\lambda$ records only the source
names $M$ and $T$, the degree row, the probe bundle, the tensor row, the
scalar endpoint row, and the antisymmetry rows. These are all BHist
provenance entries or rows exported by the two named upstream certificates.

Next inspect the tensor and probe ledgers. The probe data are a finite
$\mathsf{ProbeBundle}$ spine whose tangent rows are supplied by
$\NameCert_{\ManifoldUp}$. The tensor data are supplied by
$\NameCert_{\TensorProductUp}$, and their ordered concatenations are recorded
by $\Cont$. Thus the only tensor-continuation provenance admitted by the
core package is visible continuation data over the displayed probe rows.

Finally inspect the classifier and bridge slots. Classifier ledgers are
$\hsame$ comparisons on degree, tensor, antisymmetry, and source histories,
together with the scalar endpoint classifier retained from the upstream
sources. The standard-bridge slot in
\autoref{def:diffform-bhist-namecert-core-rows} is fixed to absent data, so
there is no additional bridge ledger to accept.
\end{proof}

\begin{theorem}[Differential-form classifier reflexivity obligation]
\label{thm:diffform-classifier-reflexivity-obligation}
Every row carried by $\mathsf{DiffFormBHistCarrier}_{M,T}$ satisfies its own
$\DiffFormUp$ classifier:
$$
\omega\sim_{\DiffFormUp}\omega.
$$
Equivalently, the BHist row accepted by the displayed carrier gives a
reflexive NameCert classifier row for the $\DiffFormUp$ certificate over the
fixed $\NameCert_{\ManifoldUp}$ and $\NameCert_{\TensorProductUp}$ sources.
\end{theorem}
\begin{proof}
Unpack the classifier in \autoref{def:diffform-bhist-form-classifier}. The
degree, tensor, antisymmetry, and source-ledger clauses are all reflexive
$\hsame$ comparisons between the corresponding visible $\mathsf{BHist}$ rows.

For each matched multilinear probe row in the displayed $\mathsf{ProbeBundle}$,
the scalar endpoint is compared with itself by reflexivity of the scalar-source
classifier retained by the upstream $\NameCert_{\ManifoldUp}$ and
$\NameCert_{\TensorProductUp}$ data. These componentwise reflexivity clauses
are exactly the clauses required by the $\DiffFormUp$ classifier.
\end{proof}
