\chapter{A Concrete Naming Certificate for $\SheafUp$}
\label{ch:concrete-instances-sheaf-namecert}

The sheaf interface $\SheafUp$ packages a presheaf satisfying the locality and gluing axioms over open covers. It depends on $\NameCert_{\PreSheafUp}$ (\autoref{ch:concrete-instances-presheaf-namecert}) and $\NameCert_{\TopologyUp}$ (\autoref{ch:concrete-instances-topology-namecert}), and is the substrate for ringed-space and scheme certificates.

\begin{definition}[Indexed section presheaf carrier]
\label{def:sheaf-indexed-section-presheaf-carrier}
Let $\mathcal{T}$ be an indexed open topology carrier for $\TopologyUp$ in the sense of \autoref{def:topology-indexed-open-carrier}. Write
$$
j\le_{\mathcal{T}} i
:\Longleftrightarrow
\forall x:X,\ \mathsf{OpenAt}(j,x)\Longrightarrow\mathsf{OpenAt}(i,x)
$$
for inclusion of indexed opens. An indexed section presheaf carrier for $\PreSheafUp$ over $\mathcal{T}$ is a tuple
$$
\mathcal{P}=(\mathsf{Sec},\sim,\rho,\mathsf{Eqv},\mathsf{Id},\mathsf{Comp})
$$
with the following data. For every open index $i:\mathsf{OpenIx}$ there is a section carrier $\mathsf{Sec}(i)$ and a classifier $\sim_i$ on $\mathsf{Sec}(i)$; the field $\mathsf{Eqv}$ says that each $\sim_i$ is reflexive, symmetric, and transitive. For every inclusion $j\le_{\mathcal{T}} i$ there is a restriction map
$$
\rho_{i,j}:\mathsf{Sec}(i)\to\mathsf{Sec}(j).
$$
The identity field says $\rho_{i,i}(s)\sim_i s$ for every $s:\mathsf{Sec}(i)$, and the composition field says that whenever $k\le_{\mathcal{T}}j\le_{\mathcal{T}}i$,
$$
\rho_{j,k}(\rho_{i,j}(s))\sim_k\rho_{i,k}(s)
$$
for every $s:\mathsf{Sec}(i)$. The restriction classifier-stability field says that, for every inclusion $j\le_{\mathcal{T}}i$ and every $s,t:\mathsf{Sec}(i)$,
$$
s\sim_i t
\Longrightarrow
\rho_{i,j}(s)\sim_j\rho_{i,j}(t).
$$
This is the section-level concrete form of the contravariant restriction data described in \autoref{ch:concrete-instances-presheaf-namecert}.
\end{definition}

\begin{definition}[Indexed sheaf cover, compatibility, and locality]
\label{def:sheaf-indexed-cover-compatibility-locality}
Let $\mathcal{P}$ be an indexed section presheaf carrier over $\mathcal{T}$ as in \autoref{def:sheaf-indexed-section-presheaf-carrier}. An indexed open cover of an open index $i:\mathsf{OpenIx}$ is a tuple $\mathcal{U}=(A,\iota,h,\gamma)$ consisting of an indexing type $A$, a family $\iota:A\to\mathsf{OpenIx}$, inclusion witnesses
$$
h_a:\iota(a)\le_{\mathcal{T}}i
$$
for all $a:A$, and the pointwise cover law
$$
\gamma:
\forall x:X,\ \mathsf{OpenAt}(i,x)\Longrightarrow
\exists a:A,\ \mathsf{OpenAt}(\iota(a),x).
$$
For a family $s_a:\mathsf{Sec}(\iota(a))$, compatibility over $\mathcal{U}$ means that for all $a,b:A$ the two restrictions to the indexed intersection agree:
$$
\rho_{\iota(a),\iota(a)\wedge_{\TopologyUp}\iota(b)}(s_a)
\sim_{\iota(a)\wedge_{\TopologyUp}\iota(b)}
\rho_{\iota(b),\iota(a)\wedge_{\TopologyUp}\iota(b)}(s_b),
$$
where the needed inclusions are supplied by the finite-intersection law of \autoref{def:topology-indexed-open-carrier}. A sheaf locality field for $\mathcal{P}$ says that, for every such cover $\mathcal{U}$ of $i$ and every pair $g,h:\mathsf{Sec}(i)$,
$$
\bigl(\forall a:A,\ \rho_{i,\iota(a)}(g)\sim_{\iota(a)}\rho_{i,\iota(a)}(h)\bigr)
\Longrightarrow
g\sim_i h.
$$
A sheaf gluing field says that every compatible local family over $\mathcal{U}$ has a section $g:\mathsf{Sec}(i)$ whose restriction to each $\iota(a)$ is classifier-equal to $s_a$. An indexed sheaf carrier for $\SheafUp$ is an indexed section presheaf carrier equipped with both of these fields.
\end{definition}

\begin{theorem}[Sheaf gluing uniqueness from locality]
\label{thm:sheaf-gluing-uniqueness-from-locality}
Let $\mathcal{S}$ be an indexed sheaf carrier for $\SheafUp$ over an indexed open topology carrier $\mathcal{T}$. Let $\mathcal{U}=(A,\iota,h,\gamma)$ be an indexed open cover of an open index $i:\mathsf{OpenIx}$, and let $s_a:\mathsf{Sec}(\iota(a))$ be a compatible local family over $\mathcal{U}$. If $g,h:\mathsf{Sec}(i)$ both restrict to that family on every cover member, in the sense that
$$
\forall a:A,\quad \rho_{i,\iota(a)}(g)\sim_{\iota(a)}s_a
$$
and
$$
\forall a:A,\quad \rho_{i,\iota(a)}(h)\sim_{\iota(a)}s_a,
$$
then the two cover-global sections are classifier-equal:
$$
g\sim_i h.
$$
Thus the gluing supplied by the $\SheafUp$ carrier is unique up to the section classifier on the covered open.
\end{theorem}
\begin{proof}
Fix a cover member $a:A$. The first restriction hypothesis gives
$$
\rho_{i,\iota(a)}(g)\sim_{\iota(a)}s_a.
$$
The second restriction hypothesis gives
$$
\rho_{i,\iota(a)}(h)\sim_{\iota(a)}s_a.
$$
By symmetry of the section classifier on $\mathsf{Sec}(\iota(a))$, supplied by the equivalence field of \autoref{def:sheaf-indexed-section-presheaf-carrier}, this yields
$$
s_a\sim_{\iota(a)}\rho_{i,\iota(a)}(h).
$$
Transitivity of the same classifier then gives
$$
\rho_{i,\iota(a)}(g)\sim_{\iota(a)}\rho_{i,\iota(a)}(h).
$$
Since $a$ was arbitrary, the restrictions of $g$ and $h$ agree on every member of $\mathcal{U}$. Applying the sheaf locality field from \autoref{def:sheaf-indexed-cover-compatibility-locality} gives $g\sim_i h$.
\end{proof}

\begin{theorem}[Sheaf global restrictions form a compatible local family]
\label{thm:sheaf-global-restrictions-compatible-local-family}
Let $\mathcal{S}$ be an indexed sheaf carrier for $\SheafUp$ over an indexed open topology carrier $\mathcal{T}$. Let $\mathcal{U}=(A,\iota,h,\gamma)$ be an indexed open cover of an open index $i:\mathsf{OpenIx}$, and let $g:\mathsf{Sec}(i)$. Define the local family $s_a:\mathsf{Sec}(\iota(a))$ by
$$
s_a=\rho_{i,\iota(a)}(g).
$$
Then $s_a$ is compatible over $\mathcal{U}$ in the sense of \autoref{def:sheaf-indexed-cover-compatibility-locality}. Explicitly, for every $a,b:A$, putting $k_{a,b}:=\iota(a)\wedge_{\TopologyUp}\iota(b)$, one has
$$
\rho_{\iota(a),k_{a,b}}(\rho_{i,\iota(a)}(g))
\sim_{k_{a,b}}
\rho_{\iota(b),k_{a,b}}(\rho_{i,\iota(b)}(g)),
$$
where the restriction maps use the inclusions supplied by the finite-intersection field and the cover inclusions $h_a$ and $h_b$.
\end{theorem}
\begin{proof}
Fix $a,b:A$ and write $k=\iota(a)\wedge_{\TopologyUp}\iota(b)$. The finite-intersection field of \autoref{def:topology-indexed-open-carrier} gives
$$
\mathsf{OpenAt}(k,x)
\Longrightarrow
\bigl(\mathsf{OpenAt}(\iota(a),x)\wedge\mathsf{OpenAt}(\iota(b),x)\bigr)
$$
for every $x:X$. Projecting the two components gives inclusions $k\le_{\mathcal{T}}\iota(a)$ and $k\le_{\mathcal{T}}\iota(b)$. Together with the cover inclusions $h_a:\iota(a)\le_{\mathcal{T}} i$ and $h_b:\iota(b)\le_{\mathcal{T}} i$, these are the two chains needed for presheaf restriction composition.

Apply the composition field of \autoref{def:sheaf-indexed-section-presheaf-carrier} to the chain $k\le_{\mathcal{T}}\iota(a)\le_{\mathcal{T}}i$ and the section $g$. This gives
$$
\rho_{\iota(a),k}(\rho_{i,\iota(a)}(g))
\sim_k
\rho_{i,k}(g).
$$
Applying the same composition field to $k\le_{\mathcal{T}}\iota(b)\le_{\mathcal{T}}i$ gives
$$
\rho_{\iota(b),k}(\rho_{i,\iota(b)}(g))
\sim_k
\rho_{i,k}(g).
$$
By symmetry of the section classifier on $\mathsf{Sec}(k)$, supplied by the equivalence field of \autoref{def:sheaf-indexed-section-presheaf-carrier}, the second displayed relation yields
$$
\rho_{i,k}(g)
\sim_k
\rho_{\iota(b),k}(\rho_{i,\iota(b)}(g)).
$$
Transitivity of the same classifier combines this with the first displayed relation and gives
$$
\rho_{\iota(a),k}(\rho_{i,\iota(a)}(g))
\sim_k
\rho_{\iota(b),k}(\rho_{i,\iota(b)}(g)).
$$
Since $a$ and $b$ were arbitrary, this is exactly compatibility of the family $a\mapsto\rho_{i,\iota(a)}(g)$ over $\mathcal{U}$ as defined in \autoref{def:sheaf-indexed-cover-compatibility-locality}.
\end{proof}

\begin{corollary}[Sheaf gluing of global restrictions is idempotent]
\label{cor:sheaf-gluing-global-restrictions-idempotent}
Let $\mathcal{S}$ be an indexed sheaf carrier for $\SheafUp$ over an indexed open topology carrier $\mathcal{T}$. Let $\mathcal{U}=(A,\iota,\eta,\gamma)$ be an indexed open cover of an open index $i:\mathsf{OpenIx}$, and let $g:\mathsf{Sec}(i)$. Define the local family $s_a:\mathsf{Sec}(\iota(a))$ by
$$
s_a=\rho_{i,\iota(a)}(g).
$$
If $h:\mathsf{Sec}(i)$ is a gluing of this family, in the sense that
$$
\forall a:A,\quad \rho_{i,\iota(a)}(h)\sim_{\iota(a)}s_a,
$$
then
$$
h\sim_i g.
$$
\end{corollary}
\begin{proof}
By \autoref{thm:sheaf-global-restrictions-compatible-local-family}, the family $s_a$ is compatible over $\mathcal{U}$. For each $a:A$, the displayed definition of $s_a$ and reflexivity of the classifier on $\mathsf{Sec}(\iota(a))$, supplied by \autoref{def:sheaf-indexed-section-presheaf-carrier}, give
$$
\rho_{i,\iota(a)}(g)\sim_{\iota(a)}s_a.
$$
The gluing hypothesis gives
$$
\rho_{i,\iota(a)}(h)\sim_{\iota(a)}s_a
$$
for the same $a$. Since $a$ was arbitrary, both cover-global sections $h$ and $g$ restrict to the compatible local family $s_a$ on every member of $\mathcal{U}$. Applying \autoref{thm:sheaf-gluing-uniqueness-from-locality} with $h$ as the first cover-global section and $g$ as the second gives $h\sim_i g$.
\end{proof}

\begin{theorem}[Sheaf restriction classifier transport]
\label{thm:sheaf-restriction-classifier-transport}
Let $\mathcal{S}$ be an indexed sheaf carrier for $\SheafUp$ over an indexed open topology carrier $\mathcal{T}$. For every inclusion $j\le_{\mathcal{T}}i$ and every pair $s,t:\mathsf{Sec}(i)$, classifier equality on the larger open transports along restriction:
$$
s\sim_i t
\Longrightarrow
\rho_{i,j}(s)\sim_j\rho_{i,j}(t).
$$
Thus restriction rows are morphisms of the local section classifiers.
\end{theorem}
\begin{proof}
The indexed sheaf carrier contains the indexed section presheaf carrier of \autoref{def:sheaf-indexed-section-presheaf-carrier}. Apply its restriction classifier-stability field to the inclusion $j\le_{\mathcal{T}}i$ and the given classifier row $s\sim_i t$.
\end{proof}

\begin{theorem}[Sheaf compatible-family stability]
\label{thm:sheaf-gluing-compatible-family-stability}
Let $\mathcal{S}$ be an indexed sheaf carrier for $\SheafUp$ over an indexed open topology carrier $\mathcal{T}$. Let $\mathcal{U}=(A,\iota,h,\gamma)$ be an indexed open cover of an open index $i:\mathsf{OpenIx}$. Suppose $s_a:\mathsf{Sec}(\iota(a))$ is compatible over $\mathcal{U}$ and $t_a:\mathsf{Sec}(\iota(a))$ is a second local family with
$$
\forall a:A,\quad t_a\sim_{\iota(a)}s_a.
$$
Then $t_a$ is compatible over $\mathcal{U}$. Explicitly, for all $a,b:A$, writing $k_{a,b}:=\iota(a)\wedge_{\TopologyUp}\iota(b)$, one has
$$
\rho_{\iota(a),k_{a,b}}(t_a)
\sim_{k_{a,b}}
\rho_{\iota(b),k_{a,b}}(t_b).
$$
\end{theorem}
\begin{proof}
Fix $a,b:A$ and write $k=\iota(a)\wedge_{\TopologyUp}\iota(b)$. The finite-intersection field of \autoref{def:topology-indexed-open-carrier} supplies the inclusions $k\le_{\mathcal{T}}\iota(a)$ and $k\le_{\mathcal{T}}\iota(b)$ needed to restrict both local sections to the overlap.

Apply \autoref{thm:sheaf-restriction-classifier-transport} to the classifier row $t_a\sim_{\iota(a)}s_a$ along $k\le_{\mathcal{T}}\iota(a)$. This gives
$$
\rho_{\iota(a),k}(t_a)
\sim_k
\rho_{\iota(a),k}(s_a).
$$
Apply the same transport row to $t_b\sim_{\iota(b)}s_b$ along $k\le_{\mathcal{T}}\iota(b)$, and then use symmetry of the classifier on $\mathsf{Sec}(k)$, to get
$$
\rho_{\iota(b),k}(s_b)
\sim_k
\rho_{\iota(b),k}(t_b).
$$

The compatibility of the original family gives
$$
\rho_{\iota(a),k}(s_a)
\sim_k
\rho_{\iota(b),k}(s_b).
$$
Transitivity of $\sim_k$ first composes the transported $a$-side row with the original compatibility row, and then composes the result with the reversed transported $b$-side row. The resulting row is exactly
$$
\rho_{\iota(a),k}(t_a)
\sim_k
\rho_{\iota(b),k}(t_b).
$$
Since $a$ and $b$ were arbitrary, the family $t_a$ is compatible over $\mathcal{U}$.
\end{proof}

\begin{theorem}[Sheaf NameCert obligation surface]
\label{thm:sheaf-namecert-obligation-surface}
The concrete $\SheafUp$ certificate surface is the following ten-field obligation surface.
\begin{enumerate}
\item The traditional target is a sheaf of sections over indexed opens of an indexed topology carrier.
\item The BEDC source data are $\TopologyUp$ together with $\PreSheafUp$.
\item The carrier is the indexed section presheaf carrier of \autoref{def:sheaf-indexed-section-presheaf-carrier}.
\item The classifier family is the open-indexed relation $\sim_i$ on $\mathsf{Sec}(i)$.
\item The stability obligations are classifier equivalence, identity restriction, composition of restriction, restriction classifier transport, locality, gluing, uniqueness of gluing, compatibility of global restrictions, and compatible-family stability.
\item The ledger consists of open-cover indices, cover-member inclusions, finite-intersection overlap indices, and restriction rows to those overlaps.
\item The core theorem list is \autoref{thm:sheaf-restriction-classifier-transport}, \autoref{thm:sheaf-gluing-compatible-family-stability}, \autoref{thm:sheaf-gluing-uniqueness-from-locality}, \autoref{thm:sheaf-global-restrictions-compatible-local-family}, and \autoref{cor:sheaf-gluing-global-restrictions-idempotent}.
\item Exactness is locality together with existence and uniqueness of gluing for compatible local families.
\item There is no standard bridge attached to this surface.
\item The status is a paper-side obligation surface for downstream certificates using $\NameCert_{\SheafUp}$.
\end{enumerate}
\end{theorem}
\begin{proof}
Rows one through four are the data fixed by \autoref{def:sheaf-indexed-section-presheaf-carrier} and \autoref{def:sheaf-indexed-cover-compatibility-locality}: indexed opens come from $\TopologyUp$, section carriers and contravariant restriction rows come from $\PreSheafUp$, and classifier equality is indexed by the same opens as the section carriers.

Row five is exactly the list of fields and consequences exposed in this section. Equivalence, identity, composition, and restriction transport are part of the carrier surface; locality and gluing are the sheaf fields; uniqueness, compatible global restrictions, idempotence of global gluing, and compatible-family stability are the theorem rows cited in row seven.

Rows six through ten record how downstream constructions consume the surface. A ringed-space or scheme certificate can name cover indices, overlaps, and restriction rows without choosing a standard-model bridge; the exactness content available to it is precisely local detection plus existence and uniqueness of gluing.
\end{proof}
