\chapter{A Concrete Naming Certificate for $\AdjunctionUp$}
\label{ch:concrete-instances-adjunction-namecert}

The adjunction interface $\AdjunctionUp$ packages a pair of functors $F \dashv G$ between two categories together with the natural-isomorphism witness on hom-sets. It depends on $\NameCert_{\FunctorUp}$ (\autoref{ch:concrete-instances-functor-namecert}) and $\NameCert_{\NatTransUp}$ (\autoref{ch:concrete-instances-nattrans-namecert}), and is the canonical structure underlying free/forgetful and Galois-style correspondences.

\section{Carrier definition}
\label{sec:adjunction-carrier}

\begin{definition}[Adjunction unit-counit carrier]
\label{def:adjunction-unit-counit-carrier}
For histories $p,q,a,unit,counit,left,right:\mathsf{BHist}$,
$\mathsf{AdjunctionUnitCounitCarrier}(p,q,a,unit,counit,left,right)$ holds
when $unit$ is a natural-transformation prefix component from $p$ to $q$ at
$a$, $counit$ is the corresponding component from $q$ to $p$, and the two
triangle witnesses are the continuations $\mathsf{Cont}(unit,counit,left)$ and
$\mathsf{Cont}(counit,unit,right)$.
\end{definition}
\leandef{BEDC.Derived.AdjunctionUp.AdjunctionUnitCounitCarrier}

\begin{theorem}[Adjunction unit-counit diagonal empty components exactness]
\label{thm:adjunction-unit-counit-empty-components-exact}
For histories $p,q,a,composite:\mathsf{BHist}$,
$$
\begin{array}{l}
\mathsf{AdjunctionUnitCounitCarrier}(p,q,a,\emp,\emp,composite,composite)\\
\quad\Longleftrightarrow\quad
\mathsf{UnaryHistory}(p)\land
\mathsf{UnaryHistory}(q)\land
\mathsf{UnaryHistory}(a)\land
p\sim q\land composite\sim\emp.
\end{array}
$$
\end{theorem}
\leanchecked{BEDC.Derived.AdjunctionUp.AdjunctionUnitCounitCarrier\_empty\_components\_exact}
\begin{proof}
Both displayed natural-transformation components are empty identity components.
Their exactness theorem gives unary histories for $p,q,a$ and $p\sim q$. The
diagonal continuation result of the two empty components is history-same as
$\emp$. Conversely, $p\sim q$ supplies both empty identity components, and
$composite\sim\emp$ supplies both empty continuation witnesses.
\end{proof}

\begin{theorem}[Adjunction empty unit-counit prefix exactness]
\label{thm:adjunction-empty-unit-counit-prefix}
For histories $p,q,a:\mathsf{BHist}$, empty unit and counit prefix components,
together with the empty composite continuation, are exact for unary prefixes,
a unary object, and prefix equality:
$$
\begin{array}{c}
\mathsf{NatTransPrefixComponentCarrier}(p,q,a,\emp)
\land
\mathsf{NatTransPrefixComponentCarrier}(q,p,a,\emp)
\land
\ContR(\emp,\emp,\emp)
\\[2pt]
\Longleftrightarrow
\mathsf{Unary}(p)\land\mathsf{Unary}(q)\land\mathsf{Unary}(a)\land\hsame(p,q).
\end{array}
$$
\end{theorem}
\begin{proof}
Each empty prefix component is read through the natural-transformation
empty-identity exactness theorem. The unit component supplies the unary data and
$p\hsame q$. Conversely, the unary data and $p\hsame q$ reconstruct the unit
component; symmetry reconstructs the counit component; the empty composite is
the reflexive continuation at $\emp$.
\end{proof}
\leanchecked{BEDC.Derived.AdjunctionUp.Adjunction\_empty\_unit\_counit\_prefix\_iff}

\begin{theorem}[Adjunction unit-counit empty component exactness]
\label{thm:adjunction-unit-counit-empty-components-exactness}
For histories $p,q,a,left,right:\mathsf{BHist}$,
$$
\mathsf{AdjunctionUnitCounitCarrier}(p,q,a,\emp,\emp,left,right)
$$
holds exactly when $p,q,a$ are unary histories, $p \hsame q$, and both triangle
results satisfy $left \hsame \emp$ and $right \hsame \emp$.
\end{theorem}
\begin{proof}
Each empty component is read by the natural-transformation empty-identity
theorem, yielding the unary prefix data and $p \hsame q$. The two triangle
continuations then compare with the empty right-unit continuation. Conversely,
the same unary data and prefix equality rebuild both empty component carriers,
and the displayed empty triangle results rebuild the two continuations.
\end{proof}
\leanchecked{BEDC.Derived.AdjunctionUp.AdjunctionUnitCounitCarrier\_empty\_components\_iff}

\begin{theorem}[Adjunction unit-counit triangle result determinism]
\label{thm:adjunction-unit-counit-triangle-results-deterministic}
If an adjunction unit-counit carrier displays triangle results $left$ and
$right$, then any other continuations with the same unit-counit inputs display
the same results up to $\hsame$.
\end{theorem}
\begin{proof}
Project the two continuation witnesses from the carrier and apply continuation
determinism separately to the left and right triangle rows.
\end{proof}
\leanchecked{BEDC.Derived.AdjunctionUp.AdjunctionUnitCounitCarrier\_triangle\_results\_deterministic}
