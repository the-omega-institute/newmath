\chapter{A Concrete Naming Certificate for \(\NatTransUp\)}
\label{ch:concrete-instances-nattrans-namecert}

The natural transformation interface \(\NatTransUp\) has certificate obligations between two functors from \autoref{ch:concrete-instances-functor-namecert} sharing source and target categories. The carrier is, for each object \(X\) of the source category, a morphism in the target category from the first functor's image of \(X\) to the second functor's image of \(X\), satisfying the naturality square under the codomain category's morphism classifier. It is not a primitive 2-cell; it is admitted only as an object-indexed family of morphisms with the naturality field.

\section{Carrier definition}

\begin{definition}[NatTrans carrier]
\label{def:nattrans-carrier}
Use an object-indexed family of codomain morphisms \(\eta_X:F(X)\to G(X)\) plus naturality witnesses for every source morphism.
\end{definition}
\leandef{BEDC.Derived.NatTransUp.NatTransPrefixComponentCarrier}

\begin{theorem}[NatTrans prefix component unary object suffix exactness]
\label{thm:nattrans-prefix-component-carrier-unary-object-suffix-iff}
For histories $p,q,a,\eta$, a prefix component over the shared object suffix
$a$ is equivalent to unary membership of $a$ together with the unsuffixed
category hom carrier:
$$
\mathsf{NatTransPrefixComponentCarrier}(p,q,a,\eta)
\Longleftrightarrow
\mathsf{UnaryHistory}(a)\land\mathsf{CategoryHomCarrier}(p,q,\eta).
$$
\end{theorem}
\begin{proof}
Unfold the prefix component carrier. The category hom-carrier suffix theorem
peels the common unary object suffix from
$\mathsf{append}(p,a)$ and $\mathsf{append}(q,a)$, and the reverse direction
repackages that suffixed hom-carrier with the component fields.
\end{proof}
\leanchecked{BEDC.Derived.NatTransUp.NatTransPrefixComponentCarrier\_unary\_object\_suffix\_iff}

\begin{theorem}[NatTrans prefix component empty identity exactness]
\label{thm:nattrans-prefix-component-carrier-empty-identity-iff}
For prefix components over unary histories, the empty component from
$\mathsf{append}(p,a)$ to $\mathsf{append}(q,a)$ exists exactly when $p$, $q$,
and $a$ are unary histories and $\HSame(p,q)$.
\end{theorem}
\begin{proof}
Unfold the prefix component carrier to its category hom-carrier over
$\mathsf{append}(p,a)$ and $\mathsf{append}(q,a)$. Category empty-identity
exactness identifies the two prefixed endpoints; cancellation of the shared
suffix $a$ recovers $\HSame(p,q)$. Conversely, unary append closure supplies
the two prefixed endpoint carriers, and the equality of $p$ and $q$ transports
the category empty identity carrier to the component carrier.
\end{proof}
\leanchecked{BEDC.Derived.NatTransUp.NatTransPrefixComponentCarrier\_empty\_identity\_iff}

\begin{theorem}[NatTrans empty identity prefix transitivity]
\label{thm:nattrans-prefix-component-empty-identity-prefix-trans}
For prefix components over unary histories, if the empty component identifies
\(\mathsf{append}(p,a)\) with \(\mathsf{append}(q,a)\) and the empty component
also identifies \(\mathsf{append}(q,a)\) with \(\mathsf{append}(r,a)\), then
\(\HSame(p,r)\).
\end{theorem}
\begin{proof}
Apply empty-identity exactness to the two component witnesses. The first
component gives \(\HSame(p,q)\), and the second gives \(\HSame(q,r)\).
Transitivity of history sameness identifies \(p\) and \(r\).
\end{proof}
\leanchecked{BEDC.Derived.NatTransUp.NatTransPrefixComponentCarrier\_empty\_identity\_prefix\_trans}

\begin{theorem}[NatTrans prefix component endomorphism component empty exactness]
\label{thm:nattrans-prefix-component-endomorphism-component-empty}
For prefix components whose source and target prefixes agree, a component
history \(\eta\) is carried exactly when the prefix, object, and component
histories are unary and the component is internally empty:
$$
\mathsf{NatTransPrefixComponentCarrier}(p,p,a,\eta)
\quad\Longleftrightarrow\quad
\mathsf{UnaryHistory}(p)\wedge\mathsf{UnaryHistory}(a)\wedge
\mathsf{UnaryHistory}(\eta)\wedge\eta\hsame\emp.
$$
\end{theorem}
\leanchecked{BEDC.Derived.NatTransUp.NatTransPrefixComponentCarrier\_endomorphism\_component\_empty\_iff}
\begin{proof}
Unfold the prefix component into its category hom-carrier
\(\mathsf{CategoryHomCarrier}(\mathsf{append}(p,a),\mathsf{append}(p,a),\eta)\).
The continuation field has matching source and target, hence right-unit
uniqueness forces \(\eta\hsame\emp\), while the component fields provide unary
membership of \(p\), \(a\), and \(\eta\). Conversely, an empty component is
rebuilt by the prefix empty-identity exactness theorem with reflexivity of
\(\hsame(p,p)\).
\end{proof}

\begin{theorem}[NatTrans one-headed source empty component exactness]
\label{thm:nattrans-prefix-component-e1-source-empty-component-exactness}
For histories $p,q,a$, an empty prefix component from $\Eone(p)$ to $q$ over
$a$ exists exactly when $p$ and $a$ are unary and the target prefix is literally
$\Eone(p)$:
$$
\mathsf{NatTransPrefixComponentCarrier}(\Eone(p),q,a,\emp)
\quad\Longleftrightarrow\quad
\mathsf{UnaryHistory}(p)\wedge\mathsf{UnaryHistory}(a)\wedge q=\Eone(p).
$$
\end{theorem}
\begin{proof}
Apply empty-identity component exactness. Unary inversion on the one-headed
source prefix extracts unary membership of $p$, and the prefix sameness
specializes to the equality $q=\Eone(p)$. Conversely, substituting this equality
rebuilds the one-headed target prefix, and empty-identity exactness packages
the empty component.
\end{proof}
\leanchecked{BEDC.Derived.NatTransUp.NatTransPrefixComponentCarrier\_e1\_source\_empty\_component\_iff}

\begin{theorem}[NatTrans one-headed object empty component exactness]
\label{thm:nattrans-prefix-component-e1-object-empty-component-iff}
For histories \(p,q,a\), an empty prefix component over the one-headed object
\(\Eone(a)\) exists exactly when both prefixes and the object tail are unary and
the two prefixes are history-same:
$$
\mathsf{NatTransPrefixComponentCarrier}(p,q,\Eone(a),\emp)
\quad\Longleftrightarrow\quad
\mathsf{UnaryHistory}(p)\wedge\mathsf{UnaryHistory}(q)\wedge
\mathsf{UnaryHistory}(a)\wedge p\hsame q.
$$
\end{theorem}
\begin{proof}
Apply empty-identity component exactness at the object \(\Eone(a)\). The forward
direction uses unary inversion on the object carrier to expose \(a\). The reverse
direction closes the one-headed object by unary closure and repacks the
empty-identity component with the displayed prefix classifier.
\end{proof}
\leanchecked{BEDC.Derived.NatTransUp.NatTransPrefixComponentCarrier\_e1\_object\_empty\_component\_iff}

\begin{theorem}[NatTrans one-headed target empty component exactness]
\label{thm:nattrans-prefix-component-e1-target-empty-component-iff}
For histories $p,q,a$, an empty prefix component from $p$ to $\Eone(q)$ over
$a$ exists exactly when $q$ and $a$ are unary and the source prefix is literally
$\Eone(q)$:
$$
\mathsf{NatTransPrefixComponentCarrier}(p,\Eone(q),a,\emp)
\quad\Longleftrightarrow\quad
\mathsf{UnaryHistory}(q)\wedge\mathsf{UnaryHistory}(a)\wedge p=\Eone(q).
$$
\end{theorem}
\begin{proof}
Apply empty-identity component exactness. Unary inversion on the one-headed
target prefix extracts unary membership of $q$, and the prefix sameness
specializes to the equality $p=\Eone(q)$. Conversely, substituting this equality
rebuilds the one-headed source prefix, and empty-identity exactness packages
the empty component.
\end{proof}
\leanchecked{BEDC.Derived.NatTransUp.NatTransPrefixComponentCarrier\_e1\_target\_empty\_component\_iff}

\begin{theorem}[NatTrans prefix component empty-source prefix exactness]
\label{thm:nattrans-prefix-component-carrier-empty-source-prefix}
For histories $q,a,\eta$, a prefix component from the empty prefix to $q$ over
$a$ is exactly a unary target prefix, a unary object, and a category hom-carrier
from $a$ to $\mathsf{append}(q,a)$:
$$
\mathsf{NatTransPrefixComponentCarrier}(\emp,q,a,\eta)
\quad\Longleftrightarrow\quad
\mathsf{UnaryHistory}(q)\wedge
\mathsf{UnaryHistory}(a)\wedge
\mathsf{CategoryHomCarrier}(a,\mathsf{append}(q,a),\eta).
$$
\end{theorem}
\begin{proof}
Unfold the prefix component carrier. The source endpoint is
$\mathsf{append}(\emp,a)$, which is history-same as $a$ by the left unit law for
append. Category hom-carrier transport across this endpoint equality gives the
unprefixed source form. The converse uses the same endpoint equality in the
opposite direction and packages the empty prefix carrier.
\end{proof}
\leanchecked{BEDC.Derived.NatTransUp.NatTransPrefixComponentCarrier\_empty\_source\_prefix\_iff}

\begin{theorem}[NatTrans prefix component empty-target prefix exactness]
\label{thm:nattrans-prefix-component-empty-target-prefix-iff}
For histories $p,a,\eta$, a prefix component from $p$ to the empty prefix over
$a$ is exactly a unary source prefix, a unary object, and a category hom-carrier
from $\mathsf{append}(p,a)$ to $a$:
$$
\mathsf{NatTransPrefixComponentCarrier}(p,\emp,a,\eta)
\quad\Longleftrightarrow\quad
\mathsf{UnaryHistory}(p)\wedge
\mathsf{UnaryHistory}(a)\wedge
\mathsf{CategoryHomCarrier}(\mathsf{append}(p,a),a,\eta).
$$
\end{theorem}
\begin{proof}
Unfold the prefix component carrier. The target endpoint is
$\mathsf{append}(\emp,a)$, which is history-same as $a$ by the left unit law
for append. Category hom-carrier transport across this target endpoint
equality gives the unprefixed target form. The converse uses the same endpoint
equality in the opposite direction and packages the empty target prefix
carrier.
\end{proof}
\leanchecked{BEDC.Derived.NatTransUp.NatTransPrefixComponentCarrier\_empty\_target\_prefix\_iff}

\begin{theorem}[NatTrans prefix component empty-prefix exactness]
\label{thm:nattrans-prefix-component-carrier-empty-prefixes-iff}
For histories $a,\eta$, a prefix component with both prefixes empty is exactly
a unary object together with a category hom-carrier from $a$ to itself:
$$
\mathsf{NatTransPrefixComponentCarrier}(\emp,\emp,a,\eta)
\quad\Longleftrightarrow\quad
\mathsf{UnaryHistory}(a)\wedge
\mathsf{CategoryHomCarrier}(a,a,\eta).
$$
\end{theorem}
\begin{proof}
Unfold the prefix component carrier. Both endpoints are
$\mathsf{append}(\emp,a)$, which is history-same as $a$ by the left unit law
for append. Category hom-carrier transport across those endpoint equalities
gives the displayed carrier from $a$ to itself. The converse transports this
carrier back to the two empty-prefixed endpoints and packages the two empty
prefix carriers.
\end{proof}
\leanchecked{BEDC.Derived.NatTransUp.NatTransPrefixComponentCarrier\_empty\_prefixes\_iff}

\begin{theorem}[NatTrans prefix component zero-headed component impossibility]
\label{thm:nattrans-prefix-component-carrier-zero-headed-component-absurd}
For histories $p,q,a,\eta$, a prefix component carrier has no zero-headed
source prefix, target prefix, object, or component morphism:
$$
\mathsf{NatTransPrefixComponentCarrier}(p,q,a,\eta)
\Longrightarrow
\left(
\exists z,\ p=\Ezero(z)
\right)
\lor
\left(
\exists z,\ q=\Ezero(z)
\right)
\lor
\left(
\exists z,\ a=\Ezero(z)
\right)
\lor
\left(
\exists z,\ \eta=\Ezero(z)
\right)
\Longrightarrow
\mathsf{False}.
$$
\end{theorem}
\begin{proof}
Unfold the prefix component carrier. The first three fields give unary-history
witnesses for $p$, $q$, and $a$, and the underlying category hom-carrier gives
the unary-history witness for the component morphism $\eta$. A zero-headed
component in any of those positions contradicts unary zero-extension
separation.
\end{proof}
\leanchecked{BEDC.Derived.NatTransUp.NatTransPrefixComponentCarrier\_zero\_headed\_component\_absurd}
