\chapter{A Concrete Naming Certificate for $\FisherInfoUp$}
\label{ch:concrete-instances-fisherinfo-namecert}

The Fisher-information interface $\FisherInfoUp$ packages, for a parametric family of distributions, the Riemannian metric whose components are expectations of squared score functions. It depends on $\NameCert_{\DistributionUp}$ (\autoref{ch:concrete-instances-distribution-namecert}) and $\NameCert_{\RiemannianMetricUp}$ (\autoref{ch:concrete-instances-riemannianmetric-namecert}).

\section{Carrier definition}
\label{sec:fisherinfo-carrier}

\begin{definition}[FisherInfoUp score packet carrier]
\label{def:fisherinfo-score-packet-carrier}
A carried $\FisherInfoUp$ score packet is a displayed $\mathsf{BHist}$
eight-tuple
$$
(\mathcal{D},\mathcal{G},\theta,s,\eta,g,\rho,\lambda).
$$
Here $\mathcal{D}$ is a displayed $\NameCert_{\DistributionUp}$ packet for the
parametric distribution family, $\mathcal{G}$ is a displayed
$\NameCert_{\RiemannianMetricUp}$ fibre packet, $\theta$ is the parameter
history, $s$ is the score-function history over the family, $\eta$ is the
squared-score expectation history, $g$ is the metric-component history, $\rho$
is the shared $\mathsf{Pkg}$ provenance row, and $\lambda$ is a finite
$\Cont$ ledger connecting these displayed rows. The packet is accepted exactly
when each coordinate is a $\mathsf{BHist}$ row, all comparisons used by the
ledger are $\hsame$ comparisons, $\lambda$ is a finite $\Cont$ composite,
$\rho$ is the common $\mathsf{Pkg}$ provenance, and the only named dependency
rows used by the carrier are the displayed $\NameCert_{\DistributionUp}$ and
$\NameCert_{\RiemannianMetricUp}$ rows. The carrier includes no host density,
integral, derivative, or coordinate metric primitive.
\end{definition}

\begin{theorem}[FisherInfoUp distribution score source row]
\label{thm:fisherinfo-distribution-score-source-row}
Every accepted packet of
\autoref{def:fisherinfo-score-packet-carrier} projects to the
$\DistributionUp$ pushforward and probability row carried by $\mathcal{D}$.
This row is the expectation source for the squared score $s^2$ recorded by
$\eta$, with provenance supplied by $\rho$ and readback supplied by the finite
$\Cont$ ledger $\lambda$.
\end{theorem}
\begin{proof}
Project the accepted eight-tuple onto its distribution-side coordinates. The
first projection is the carried $\NameCert_{\DistributionUp}$ packet
$\mathcal{D}$, so the family row is read through the pushforward carrier and
probability-package rows already exposed by $\DistributionUp$.

The score observable is the displayed $\mathsf{BHist}$ row $s$ over the same
parameter history $\theta$. Squaring the displayed score stays inside the
accepted packet because the squared-score endpoint is the expectation history
$\eta$, not a host analytic object.

The finite $\Cont$ ledger $\lambda$ records the readback from the
$\DistributionUp$ endpoint to the expectation row, and $\rho$ is the shared
$\mathsf{Pkg}$ provenance. Repacking these projections gives exactly the
distribution source row required for the squared-score expectation.
\end{proof}

\begin{theorem}[FisherInfoUp Riemannian metric component row]
\label{thm:fisherinfo-riemannian-metric-component-row}
In every accepted packet of
\autoref{def:fisherinfo-score-packet-carrier}, the endpoint of the expected
squared score $\eta$ is exactly the metric-component $\mathsf{BHist}$ row
$g$ consumed by the inherited $\RiemannianMetricUp$ fibre surface
$\mathcal{G}$. Thus the Fisher information component is not an external
coordinate metric; it is the displayed expectation endpoint repacked as the
metric fibre component.
\end{theorem}
\begin{proof}
Project the accepted packet onto the expectation row $\eta$ and the
metric-component row $g$. By the carrier definition, both are displayed
$\mathsf{BHist}$ rows in the same finite $\Cont$ ledger and share the
provenance row $\rho$.

The distribution score source row identifies $\eta$ as the expectation
endpoint of $s^2$ through the carried $\DistributionUp$ probability row. The
RiemannianMetricUp dependency $\mathcal{G}$ consumes a displayed metric fibre
component rather than a host coordinate tensor.

The accepted FisherInfoUp packet equates these two displayed endpoints inside
the finite ledger: the expected squared score endpoint is repacked as $g$.
Consequently the inherited fibre surface reads the same $\mathsf{BHist}$ row
that the expectation source row produces.
\end{proof}

\section{The certificate}
\label{sec:fisherinfo-certificate}

\begin{theorem}[FisherInfoUp classifier transport row]
\label{thm:fisherinfo-classifier-transport-row}
Let $(\mathcal{D},\mathcal{G},\theta,s,\eta,g,\rho,\lambda)$ and
$(\mathcal{D}',\mathcal{G}',\theta',s',\eta',g',\rho',\lambda')$ be accepted
$\FisherInfoUp$ score packets. If $\hsame$ transport carries the distribution
family row $\mathcal{D}$ to $\mathcal{D}'$, the parameter row $\theta$ to
$\theta'$, the score row $s$ to $s'$, and the metric fibre row $\mathcal{G}$
to $\mathcal{G}'$, then it transports the FisherInfoUp metric endpoint
componentwise from $g$ to $g'$.
\end{theorem}
\begin{proof}
Use the carried $\DistributionUp$ classifier first. It transports the
pushforward and probability row that supplies the expectation source from
$\mathcal{D}$ to $\mathcal{D}'$, while the parameter and score hypotheses
transport the displayed score-function row from $s$ to $s'$ over
$\theta'$.

The expected squared score endpoint therefore transports from $\eta$ to
$\eta'$ along the same finite $\Cont$ readback pattern. No analytic
integration or derivative primitive enters this step because both endpoints
are displayed $\mathsf{BHist}$ rows in accepted packets.

Apply the metric component row theorem on both packets. It identifies $\eta$
with $g$ and $\eta'$ with $g'$ as the metric fibre components consumed by
$\mathcal{G}$ and $\mathcal{G}'$. The $\RiemannianMetricUp$ classifier then
transports the component row, giving the stated componentwise FisherInfoUp
metric endpoint transport.
\end{proof}

\begin{theorem}[FisherInfoUp public certificate surface]
\label{thm:fisherinfo-public-certificate-surface}
The public $\NameCert_{\FisherInfoUp}$ obligation surface consists exactly of
the score packet carrier of
\autoref{def:fisherinfo-score-packet-carrier}, the distribution source row of
\autoref{thm:fisherinfo-distribution-score-source-row}, the metric component
row of \autoref{thm:fisherinfo-riemannian-metric-component-row}, the
classifier transport row of
\autoref{thm:fisherinfo-classifier-transport-row}, the finite $\Cont$ ledger
$\lambda$, the shared $\mathsf{Pkg}$ ledger $\rho$, the exactness boundary
excluding host density, integral, derivative, and coordinate metric
primitives, and an absent bridge row. These data form the public
$\NameCert_{\FisherInfoUp}$ surface over the dependency certificates
$\NameCert_{\DistributionUp}$ and $\NameCert_{\RiemannianMetricUp}$.
\end{theorem}
\begin{proof}
The carrier component is exactly the displayed eight-coordinate packet of
\autoref{def:fisherinfo-score-packet-carrier}. Its accepted coordinates
expose the $\DistributionUp$ dependency, the $\RiemannianMetricUp$ dependency,
the parameter history, the score-function history, the squared-score
expectation history, the metric-component history, the shared $\mathsf{Pkg}$
provenance, and the finite $\Cont$ ledger.

The distribution-side public rows are supplied by
\autoref{thm:fisherinfo-distribution-score-source-row}. The metric endpoint
identified from the expected squared score is supplied by
\autoref{thm:fisherinfo-riemannian-metric-component-row}.

The classifier transport row is supplied by
\autoref{thm:fisherinfo-classifier-transport-row}. Repacking these rows with
the finite $\Cont$ ledger, the $\mathsf{Pkg}$ provenance ledger, the explicit
exactness boundary, and the absent bridge row gives the stated ten-field
$\NameCert_{\FisherInfoUp}$ obligation surface.
\end{proof}
