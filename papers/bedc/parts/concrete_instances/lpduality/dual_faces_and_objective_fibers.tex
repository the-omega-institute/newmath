\begin{theorem}[LPDuality optimal primal face binary convex closure]
\label{thm:lpduality-optimal-primal-face-binary-convex-closure}
Let an $\LPDualityUp$ row over an ordered field be given as in \autoref{def:lpduality-finite-ordered-field-feasibility-row}. Let $x,x':J\to F$ be primal feasible witnesses, let $\tau:F$ be a scalar endpoint, and suppose both witnesses lie in the same primal objective fiber:
$$
\mathsf{PrFeas}_{A,b}(x),
\qquad
\mathsf{PrFeas}_{A,b}(x'),
$$
$$
\mathsf{PrObj}_{A,c}(x)\sim_F\tau,
\qquad
\mathsf{PrObj}_{A,c}(x')\sim_F\tau.
$$
Let $\alpha,\beta:F$ satisfy
$$
\mathsf{NonNeg}_F(\alpha),
\qquad
\mathsf{NonNeg}_F(\beta),
\qquad
\alpha+_F\beta\sim_F1_F,
$$
and define the componentwise binary affine mixture
$$
\lambda_j:=\alpha\cdot_F x_j+_F\beta\cdot_F x'_j
\qquad (j\in J).
$$
Then $\lambda$ is primal feasible and remains in the same primal objective fiber:
$$
\mathsf{PrFeas}_{A,b}(\lambda),
\qquad
\mathsf{PrObj}_{A,c}(\lambda)\sim_F\tau.
$$
Consequently, when $\tau$ is the scalar endpoint certified as the primal optimum of the row, the certified primal optimal face at $\tau$ is closed under binary affine combinations with nonnegative coefficients whose sum is classifier-equal to $1_F$.
\end{theorem}
\leanchecked{BEDC.Derived.LPDualityUp.LPDualityOptimalPrimalFace\_binary\_convex\_closure}
\begin{proof}
Unpack the two primal-feasibility hypotheses using \autoref{def:lpduality-finite-ordered-field-feasibility-row}. They give
$$
\forall j\in J.\;\mathsf{NonNeg}_F(x_j),
\qquad
\forall j\in J.\;\mathsf{NonNeg}_F(x'_j),
$$
and
$$
\forall i\in I.\;\alpha_i(x)\preceq_F b_i,
\qquad
\forall i\in I.\;\alpha_i(x')\preceq_F b_i.
$$
First fix $j\in J$. Since $\mathsf{NonNeg}_F(x_j)$ and $\mathsf{NonNeg}_F(x'_j)$ mean $0_F\preceq_Fx_j$ and $0_F\preceq_Fx'_j$, multiplication monotonicity by the nonnegative scalars $\alpha$ and $\beta$ gives
$$
\alpha\cdot_F0_F\preceq_F\alpha\cdot_Fx_j,
\qquad
\beta\cdot_F0_F\preceq_F\beta\cdot_Fx'_j.
$$
The two-point finite-sum monotonicity instance in \autoref{def:lpduality-finite-ordered-field-feasibility-row} yields
$$
\alpha\cdot_F0_F+_F\beta\cdot_F0_F
\preceq_F
\alpha\cdot_Fx_j+_F\beta\cdot_Fx'_j.
$$
The $\FieldUp$ zero laws identify the left endpoint with $0_F$, and the definition of $\lambda_j$ identifies the right endpoint with $\lambda_j$. Classifier stability of $\preceq_F$ therefore gives
$$
0_F\preceq_F\lambda_j.
$$
Thus $\mathsf{NonNeg}_F(\lambda_j)$ holds for every $j\in J$.

For the row inequalities, apply \autoref{thm:lpduality-primal-feasibility-binary-convex-closure} to the finite row family $I$, with $\mathsf{Row}_i(z)=\alpha_i(z)$. The displayed row-constraint clauses for $x$ and $x'$, together with the same nonnegative coefficients and the classifier equality $\alpha+_F\beta\sim_F1_F$, give
$$
\forall i\in I.\;\alpha_i(\lambda)\preceq_F b_i.
$$
Combining this row-constraint clause with the nonnegativity clause just proved is exactly
$$
\mathsf{PrFeas}_{A,b}(\lambda)
$$
by \autoref{def:lpduality-finite-ordered-field-feasibility-row}.

It remains to read back the objective endpoint. By \autoref{lem:lpduality-primal-objective-binary-affine-readback},
$$
\mathsf{PrObj}_{A,c}(\lambda)
\sim_F
\alpha\cdot_F\mathsf{PrObj}_{A,c}(x)+_F
\beta\cdot_F\mathsf{PrObj}_{A,c}(x').
$$
Scalar classifier congruence transports the two objective-fiber hypotheses through multiplication and addition, giving
$$
\alpha\cdot_F\mathsf{PrObj}_{A,c}(x)+_F
\beta\cdot_F\mathsf{PrObj}_{A,c}(x')
\sim_F
\alpha\cdot_F\tau+_F\beta\cdot_F\tau.
$$
Distributivity, the coefficient-sum classifier equality, and the multiplicative unit law give
$$
\alpha\cdot_F\tau+_F\beta\cdot_F\tau
\sim_F
(\alpha+_F\beta)\cdot_F\tau
\sim_F
1_F\cdot_F\tau
\sim_F
\tau.
$$
Transitivity of the scalar classifier composes the displayed classifier equalities into
$$
\mathsf{PrObj}_{A,c}(\lambda)\sim_F\tau.
$$
The final sentence is the specialization of the same statement to a scalar endpoint already certified as optimal for the row.
\end{proof}

\begin{lemma}[LPDuality dual objective binary affine readback]
\label{lem:lpduality-dual-objective-binary-affine-readback}
Let an $\LPDualityUp$ row over an ordered field be given as in \autoref{def:lpduality-finite-ordered-field-feasibility-row}. Let $y,y':I\to F$ be dual vectors, let $\alpha,\beta:F$ be scalar coefficients, and define $\mu:I\to F$ componentwise by
$$
\mu_i:=\alpha\cdot_F y_i+_F\beta\cdot_F y'_i
\qquad (i\in I).
$$
Then the dual objective is affine over this binary mixture:
$$
\mathsf{DuObj}_{b}(\mu)
\sim_F
\alpha\cdot_F\mathsf{DuObj}_{b}(y)+_F
\beta\cdot_F\mathsf{DuObj}_{b}(y').
$$
\end{lemma}
\leanchecked{BEDC.Derived.LPDualityUp.LPDualityDualObjective\_binary\_affine\_readback}
\begin{proof}
Expand the dual objective endpoint from \autoref{def:lpduality-finite-ordered-field-feasibility-row}. By the definition of $\mu$, distributivity in the carried $\FieldUp$ row, associativity and commutativity of multiplication, and finite-sum compatibility with the scalar classifier, one has
$$
\begin{aligned}
\mathsf{DuObj}_{b}(\mu)
&=\sum_{i\in I}(\alpha\cdot_F y_i+_F\beta\cdot_F y'_i)\cdot_F b_i\\
&\sim_F
\sum_{i\in I}\bigl(\alpha\cdot_F(y_i\cdot_F b_i)+_F
\beta\cdot_F(y'_i\cdot_F b_i)\bigr)\\
&\sim_F
\alpha\cdot_F\sum_{i\in I}y_i\cdot_F b_i
+_F
\beta\cdot_F\sum_{i\in I}y'_i\cdot_F b_i\\
&=
\alpha\cdot_F\mathsf{DuObj}_{b}(y)+_F
\beta\cdot_F\mathsf{DuObj}_{b}(y').
\end{aligned}
$$
This is the asserted classifier equality.
\end{proof}

\begin{theorem}[LPDuality objective fiber sandwich closure]
\label{thm:lpduality-objective-fiber-sandwich-closure}
Let an $\LPDualityUp$ row over an ordered field be given as in
\autoref{def:lpduality-finite-ordered-field-feasibility-row}. Let
$x,x':J\to F$ be primal feasible witnesses, let $y,y':I\to F$ be dual feasible
witnesses, and let $\tau:F$ be a scalar endpoint. Suppose weak duality gives
the objective comparison for every displayed primal-dual pair, and suppose all
four displayed objective endpoints lie in the same scalar fiber:
$$
\mathsf{PrObj}_{A,c}(x)\sim_F\tau,
\qquad
\mathsf{PrObj}_{A,c}(x')\sim_F\tau,
$$
$$
\mathsf{DuObj}_{b}(y)\sim_F\tau,
\qquad
\mathsf{DuObj}_{b}(y')\sim_F\tau.
$$
Then the primal optimal-face closure row at $\tau$ and the dual optimal-face
closure row at $\tau$ are consumed by the same $\tau$-fiber sandwich: every
binary affine primal mixture of $x,x'$ and every binary affine dual mixture of
$y,y'$ with nonnegative coefficients whose sum is classifier-equal to $1_F$
remains feasible, remains classifier-equal to $\tau$ on its own objective
side, and sits between weak-duality comparisons whose endpoints are
classifier-stable copies of $\tau$.
\end{theorem}
\leanchecked{BEDC.Derived.LPDualityUp.LPDualityObjectiveFiber\_sandwich\_closure}
\begin{proof}
First apply \autoref{thm:lpduality-feasible-weak-duality} to each displayed
primal-dual pair. Classifier stability of $\preceq_F$ transports the primal
and dual objective endpoints along the four fiber hypotheses, so each weak
duality comparison becomes a comparison from a scalar endpoint
classifier-equal to $\tau$ to a scalar endpoint classifier-equal to $\tau$.

Next apply
\autoref{thm:lpduality-optimal-primal-face-binary-convex-closure} to the
primal witnesses $x,x'$ at the fiber endpoint $\tau$. This gives primal
feasibility and primal objective readback at $\tau$ for every allowed binary
affine primal mixture.

Apply
\autoref{thm:lpduality-optimal-dual-face-binary-convex-closure} to the dual
witnesses $y,y'$ at the same fiber endpoint. This gives dual feasibility and
dual objective readback at $\tau$ for every allowed binary affine dual mixture.

Finally use weak duality once more on the transported primal and dual mixtures.
The two closure rows provide the objective endpoint equalities to $\tau$, and
classifier stability of the ordered-field order transports the comparison
endpoints into the common $\tau$ fiber. Thus the primal and dual closure rows
are consumed by one objective-fiber sandwich rather than by separate scalar
certificates.
\end{proof}

\begin{theorem}[LPDuality primal-dual affine fiber product closure]
\label{thm:lpduality-primal-dual-affine-fiber-product-closure}
Let an $\LPDualityUp$ row over an ordered field be given as in
\autoref{def:lpduality-finite-ordered-field-feasibility-row}. Let
$x,x':J\to F$ be primal feasible witnesses, let $y,y':I\to F$ be dual
feasible witnesses, and let $\tau:F$ be a scalar endpoint. Suppose all four
objective endpoints lie in the same scalar fiber:
$$
\mathsf{PrObj}_{A,c}(x)\sim_F\tau,
\qquad
\mathsf{PrObj}_{A,c}(x')\sim_F\tau,
$$
$$
\mathsf{DuObj}_{b}(y)\sim_F\tau,
\qquad
\mathsf{DuObj}_{b}(y')\sim_F\tau.
$$
Let $\alpha,\beta:F$ be nonnegative coefficients with
$\alpha+_F\beta\sim_F1_F$, and form the paired binary affine mixture
$$
\lambda_j:=\alpha\cdot_Fx_j+_F\beta\cdot_Fx'_j
\qquad (j\in J),
\qquad
\mu_i:=\alpha\cdot_Fy_i+_F\beta\cdot_Fy'_i
\qquad (i\in I).
$$
Then $(\lambda,\mu)$ is a feasible primal-dual pair, both objective endpoints
remain classifier-equal to $\tau$, and the weak-duality comparison between
the mixed pair is a comparison between classifier-stable copies of $\tau$.
\end{theorem}
\begin{proof}
Apply
\autoref{thm:lpduality-optimal-primal-face-binary-convex-closure} to the
primal witnesses $x,x'$ with endpoint $\tau$ and coefficients
$\alpha,\beta$. This gives the primal feasibility of $\lambda$ and the
readback $\mathsf{PrObj}_{A,c}(\lambda)\sim_F\tau$.

Apply
\autoref{thm:lpduality-optimal-dual-face-binary-convex-closure} to the dual
witnesses $y,y'$ with the same endpoint and coefficients. This gives the dual
feasibility of $\mu$ and the readback
$\mathsf{DuObj}_{b}(\mu)\sim_F\tau$.

Finally use
\autoref{thm:lpduality-objective-fiber-sandwich-closure} on the same four
witnesses and the same scalar fiber. The two closure rows give the paired
feasible mixture, and weak duality for the mixed pair is transported along
the two objective readbacks to a comparison whose endpoints are
classifier-stable copies of $\tau$.
\end{proof}

\begin{theorem}[LPDuality weak-duality slackness optimality packet]
\label{thm:lpduality-weak-duality-slackness-optimality-packet}
Let an $\LPDualityUp$ row over an ordered field be given as in
\autoref{def:lpduality-finite-ordered-field-feasibility-row}. Let $x:J\to F$
be a primal feasible witness and let $y:I\to F$ be a dual feasible witness.
Suppose the two complementary-slackness product families are
classifier-equal termwise:
$$
\forall j\in J.\; c_j\cdot_Fx_j\sim_F\beta_j(y)\cdot_Fx_j,
\qquad
\forall i\in I.\; y_i\cdot_F\alpha_i(x)\sim_Fy_i\cdot_Fb_i.
$$
Then the row carries a certified optimality packet: the objectives
$\mathsf{PrObj}_{A,c}(x)$ and $\mathsf{DuObj}_{b}(y)$ are classifier-equal,
$x$ is primal optimal, and $y$ is dual optimal. Moreover, writing
$\tau:=\mathsf{PrObj}_{A,c}(x)$, any further primal feasible optimal witness
$x':J\to F$ and any further dual feasible optimal witness $y':I\to F$ whose
objective endpoints are classifier-equal to $\tau$ enter the common
objective-fiber sandwich at $\tau$: every binary affine primal mixture of
$x,x'$ and every binary affine dual mixture of $y,y'$ with nonnegative
coefficients whose sum is classifier-equal to $1_F$ remains feasible, remains
in its own $\tau$-objective fiber, and is bounded by the weak-duality
comparisons between classifier-stable copies of $\tau$.
\end{theorem}
\begin{proof}
Apply \autoref{thm:lpduality-complementary-slackness-objective-equality} to
the feasible witnesses $x$ and $y$ and to the two displayed slackness
families. This gives the scalar objective equality together with the primal
optimality of $x$ and the dual optimality of $y$.

The objective equality identifies $\mathsf{DuObj}_{b}(y)$ with the same fiber
endpoint $\tau:=\mathsf{PrObj}_{A,c}(x)$. Together with the assumed fiber
readbacks for the feasible optimal witnesses $x'$ and $y'$, the four
endpoints $x,x',y,y'$ have weak-duality comparisons by
\autoref{thm:lpduality-feasible-weak-duality}. Hence they satisfy the
hypotheses of \autoref{thm:lpduality-objective-fiber-sandwich-closure}.
Applying that theorem gives the shared primal and dual binary-affine closure
statement and the weak-duality sandwich over the common $\tau$ fiber.
\end{proof}
