\begin{theorem}[Category empty-middle displayed morphism determinacy]
\label{thm:category-hom-carrier-comp-empty-middle-displayed-morphism-deterministic}
For histories $a,c,f,g,fg,\mathit{displayed}$, if $f$ is a hom-carrier from $a$ to
$\emp$, $g$ is a hom-carrier from $\emp$ to $c$, $\ContR(f,g,fg)$, and
$\mathit{displayed}$ is also a hom-carrier from $\emp$ to $c$, then
$$
\mathsf{CategoryHomCarrier}(\emp,c,\mathit{displayed})\Longrightarrow
\HSame(\mathit{displayed},fg).
$$
\end{theorem}
\begin{proof}
Empty-middle inversion supplies the canonical hom-carrier from $\emp$ to $c$
over the composite morphism $fg$. Hom-carrier morphism determinacy compares
that canonical carrier with the displayed carrier over the same endpoints.
\end{proof}
