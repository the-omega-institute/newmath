\begin{theorem}[Category empty-source target morphism swap]
\label{thm:category-empty-source-target-morphism-swap}
For histories $b,f:\mathsf{BHist}$, an empty-source hom-carrier is unchanged
when its target and morphism fields are exchanged:
$$
\mathsf{CategoryHomCarrier}(\emp,b,f)
\Longleftrightarrow
\mathsf{CategoryHomCarrier}(\emp,f,b).
$$
\end{theorem}
\begin{proof}
Empty-source exactness rewrites the left side as unary membership of $b$
together with $\HSame(f,b)$. Transport unary membership across this sameness
to obtain unary membership of $f$, reverse the endpoint classifier, and apply
empty-source exactness again. The converse is the same argument with $b$ and
$f$ exchanged.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.CategoryHomCarrier\_empty\_source\_target\_morphism\_swap\_iff}

\begin{theorem}[Category empty-target source morphism swap]
\label{thm:category-hom-carrier-empty-target-source-morphism-swap}
For histories $a,f:\mathsf{BHist}$, a hom-carrier into the empty target is
unchanged when its source and morphism fields are exchanged:
$$
\mathsf{CategoryHomCarrier}(a,\emp,f)
\Longleftrightarrow
\mathsf{CategoryHomCarrier}(f,\emp,a).
$$
\end{theorem}
\begin{proof}
Empty-target exactness rewrites the left side as the pair
$\hsame(a,\emp)\wedge\hsame(f,\emp)$. Swapping the two components gives the
empty-target classifier data for the exchanged hom-carrier. The converse is
the same component swap in the other direction.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.CategoryHomCarrier\_empty\_target\_source\_morphism\_swap\_iff}

\begin{theorem}[Category empty-target morphism source determinacy]
\label{thm:category-empty-target-morphism-source-deterministic}
For the unary-continuation category, if two hom-carriers share the empty target
and their morphism histories are the same up to $\HSame$, then their source
histories are the same up to $\HSame$:
$$
\mathsf{CategoryHomCarrier}(a,\emp,f)\to
\mathsf{CategoryHomCarrier}(b,\emp,g)\to
\HSame(f,g)\to
\HSame(a,b).
$$
\end{theorem}
\begin{proof}
Transport the second empty-target hom-carrier across the morphism sameness so
that both hom-carriers use the same morphism history. Category hom-carrier
source determinacy then identifies the two sources.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.CategoryHomCarrier\_empty\_target\_morphism\_source\_deterministic}

\begin{corollary}[Category empty-source swap consumer determinacy]
\label{cor:category-empty-source-target-morphism-swap-consumer-determinacy}
For histories $b,f,c,g:\BHist$, suppose an accepted empty-source hom-carrier
fixes the target and morphism rows by
$$
\mathsf{CategoryHomCarrier}(\emp,b,f).
$$
If a consumer-facing use of the same packet reads the target row $b$ and the
morphism row $f$ through empty-target hom-carriers,
$$
\mathsf{CategoryHomCarrier}(c,\emp,b)
\quad\text{and}\quad
\mathsf{CategoryHomCarrier}(g,\emp,f),
$$
then the two consumer source coordinates are forced to be the same:
$$
\HSame(c,g).
$$
Consequently the empty-source target--morphism swap of
\autoref{thm:category-empty-source-target-morphism-swap} has only the finite
route, normal-form read, and exactness decision already displayed by the
chapter: an accepted consumer cannot introduce an alternate hidden coordinate
or a duplicate classifier source for the swapped target and morphism fields.
\end{corollary}
\begin{proof}
Apply \autoref{thm:category-empty-source-target-morphism-swap} to the displayed
empty-source carrier. The same packet also admits the swapped read
$$
\mathsf{CategoryHomCarrier}(\emp,f,b).
$$
The empty-source exactness row used in that swap reads the original carrier as
unary membership of $b$ together with the endpoint classifier $\HSame(f,b)$;
by symmetry this gives $\HSame(b,f)$.
Now apply \autoref{thm:category-empty-target-morphism-source-deterministic} to
the two empty-target carriers
$\mathsf{CategoryHomCarrier}(c,\emp,b)$ and
$\mathsf{CategoryHomCarrier}(g,\emp,f)$, using $\HSame(b,f)$ as the morphism
comparison. It identifies their source histories, giving $\HSame(c,g)$.
Thus every accepted consumer read factors through the displayed empty-source
swap row and the two empty-target readback rows; no unlisted coordinate remains
from which a second source classifier could be read.
\end{proof}
