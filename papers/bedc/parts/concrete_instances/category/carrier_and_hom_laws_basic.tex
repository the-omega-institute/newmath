\chapter{A Concrete Naming Certificate for \(\CategoryUp\)}
\label{ch:concrete-instances-category-namecert}

The category interface \(\CategoryUp\) has certificate obligations over an object carrier together with, for each ordered pair of objects, a morphism carrier, identity morphisms at each object, a binary composition operation respecting source and target, and the associativity and identity laws expressed as morphism-classifier sameness. The size discipline is BEDC-internal: object and morphism carriers are each covered by their own NameCert obligations, avoiding any global "set of all objects" assumption. It is not a primitive structure; it is admitted only as a multi-sorted carrier with relational composition.

\section{Carrier definition}

\begin{definition}[Category carrier]
\label{def:category-carrier}
Use a multi-sorted setup with object carrier, morphism carrier for each ordered object pair, identities, composition, and law fields.
\end{definition}

\begin{theorem}[Category empty identity hom carrier]
\label{thm:category-hom-carrier-empty-identity}
For the unary-continuation category, if \(h\) is unary, then the empty history
is a hom-carrier from \(h\) to \(h\).
\end{theorem}
\begin{proof}
The source and target carrier witnesses are the given unary witness for \(h\).
The morphism carrier is the unary empty history, and the continuation field is
the right-unit continuation \(\ContR(h,\emp,h)\).
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.CategoryHomCarrier\_empty\_identity}

\begin{theorem}[Category empty identity hom carrier exactness]
\label{thm:category-hom-carrier-empty-identity-iff}
For histories \(a,b\), the empty history is a hom-carrier from \(a\) to \(b\)
exactly when \(a\) and \(b\) are unary and internally the same:
$$
\mathsf{CategoryHomCarrier}(a,b,\emp)
\quad\Longleftrightarrow\quad
\mathsf{UnaryHistory}(a)\wedge \mathsf{UnaryHistory}(b)\wedge \hsame(a,b).
$$
\end{theorem}
\begin{proof}
Unpack the hom-carrier witness. Its continuation field is
\(\ContR(a,\emp,b)\); determinism against the right-unit continuation
\(\ContR(a,\emp,a)\) identifies \(a\) and \(b\). Conversely, unary source and
target witnesses together with \(\hsame(a,b)\) transport the right-unit
continuation to a continuation ending at \(b\), while \(\emp\) supplies the
empty morphism carrier.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.CategoryHomCarrier\_empty\_identity\_iff}

\begin{theorem}[Category one-headed source empty morphism exactness]
\label{thm:category-hom-carrier-e1-source-empty-morphism-iff}
For histories \(a\) and \(target\), the empty morphism is a hom-carrier from
\(\Eone(a)\) to \(target\) exactly when the target is \(\Eone(a)\) and the
tail \(a\) is unary:
$$
\mathsf{CategoryHomCarrier}(\Eone(a),target,\emp)
\quad\Longleftrightarrow\quad
target=\Eone(a)\wedge\mathsf{UnaryHistory}(a).
$$
\end{theorem}
\begin{proof}
Apply empty identity hom-carrier exactness to the source \(\Eone(a)\). The
resulting history-sameness witness identifies \(target\) with \(\Eone(a)\),
and unary inversion on the one-headed source gives unary membership of \(a\).
Conversely, after the displayed target equation is eliminated, unary closure
under \(\Eone\) and the empty identity carrier rebuild the hom-carrier.
\end{proof}

\begin{theorem}[Category empty-source hom carrier exactness]
\label{thm:category-hom-carrier-empty-source-iff}
For histories \(b,f\), a hom-carrier out of the empty source is exactly unary
membership of the target together with sameness of the morphism and target:
$$
\mathsf{CategoryHomCarrier}(\emp,b,f)
\quad\Longleftrightarrow\quad
\mathsf{UnaryHistory}(b)\wedge\hsame(f,b).
$$
\end{theorem}
\begin{proof}
Unpack the hom-carrier witness. Its continuation field is
\(\ContR(\emp,f,b)\), and the left-unit continuation exactness identifies
\(b\) with \(f\); symmetry gives \(\hsame(f,b)\), while the target unary
witness is already present. Conversely, transport the target unary witness
across \(\hsame(f,b)\) to obtain the morphism carrier, and use the left-unit
continuation equivalence to rebuild \(\ContR(\emp,f,b)\).
\end{proof}

\begin{theorem}[Category empty-target hom carrier exactness]
\label{thm:category-hom-carrier-empty-target-iff}
For histories $a,f$, a hom-carrier into the empty target is exactly sameness of
the source and morphism with the empty history:
$$
\mathsf{CategoryHomCarrier}(a,\emp,f)
\quad\Longleftrightarrow\quad
\hsame(a,\emp)\wedge\hsame(f,\emp).
$$
\end{theorem}
\begin{proof}
Unpack the hom-carrier witness. Its continuation field is
$\ContR(a,f,\emp)$, and empty-result inversion for continuation splits the
source and morphism into empty histories. Conversely, after those two sameness
witnesses are eliminated, the empty source, empty target, and empty morphism
are unary, and the right-unit continuation at $\emp$ rebuilds the hom-carrier.
\end{proof}

\begin{theorem}[Category hom carrier composition closure]
\label{thm:category-hom-carrier-comp-closed}
For the unary-continuation category, if \(f\) is a hom-carrier from \(a\) to \(b\), \(g\) is a hom-carrier from \(b\) to \(c\), and \(\ContR(f,g,fg)\), then \(fg\) is a hom-carrier from \(a\) to \(c\).
\end{theorem}
\begin{proof}
Unpack the two hom-carrier witnesses. The source, middle, target, and morphism histories are unary. Unary continuation closure gives unary membership for the composite morphism \(fg\), and associativity of the recursive append representative underlying \(\ContR\) supplies the continuation from \(a\) through \(fg\) to \(c\).
\end{proof}

$b$, $g$ is a hom-carrier from $b$ to $c$, and $\ContR(f,g,fg)$. If the

into a hom-carrier from $a$ to $c$. Its concrete continuation field is

$\ContR(a,fg,a)$, and right-unit uniqueness forces $fg$ to be $\emp$.
\end{proof}

\begin{theorem}[Category empty-target composite boundary inversion]
\label{thm:category-hom-carrier-empty-target-composite-boundary-inversion}
If $f$ is a hom-carrier from $a$ to $b$, $g$ is a hom-carrier from $b$ to
$\emp$, and $\ContR(f,g,fg)$, then $a$, $b$, $f$, $g$, and $fg$ are all

\end{theorem}
\begin{proof}
Empty-target exactness applied to the right hom-carrier identifies $b$ and
$g$ with $\emp$. With $b$ reduced to $\emp$, empty-target exactness applied to
the left hom-carrier identifies $a$ and $f$ with $\emp$. The displayed
continuation ledger then has both inputs empty, so its result is $\emp$.
\end{proof}

\begin{theorem}[Category hom-carrier composite result congruence]
\label{thm:category-hom-carrier-composite-result-congruence}
For unary-continuation hom carriers $f:a\to b$, $g:b\to c$,

$$
\HSame(fg,fg').
$$
\end{theorem}
\begin{proof}
Composition closure places $fg$ and $fg'$ as hom-carriers from the same source

endpoints then identifies the two composite morphism histories.
\end{proof}

\begin{theorem}[Category hom carrier same-composite middle-factor determinacy]
\label{thm:category-hom-carrier-comp-same-middle-factors-deterministic}
For unary-continuation hom carriers $f:a\to b$, $g:b\to c$,
$f':a\to b$, and $g':b\to c$, if $\ContR(f,g,fg)$ and
$\ContR(f',g',fg)$ hold, then
$\HSame(f,f')\land\HSame(g,g')$.
\end{theorem}
\begin{proof}
First apply
\autoref{thm:category-hom-carrier-morphism-deterministic} to the two
$a\to b$ carriers. This gives $\HSame(f,f')$.

Transport the first continuation ledger across this equality. The transported
ledger and $\ContR(f',g',fg)$ now have the same left continuation history and
the same result history, so continuation left cancellation identifies the two
right factors, giving $\HSame(g,g')$.
\end{proof}

$$

$$
\HSame(f,\emp)\wedge\HSame(g,\emp)\wedge\HSame(a,b)\wedge\HSame(b,c).

$\mathsf{CategoryHomCarrier}(a,b,\emp)$ and
$\mathsf{CategoryHomCarrier}(b,c,\emp)$. The empty identity exactness theorem
then reads back $\HSame(a,b)$ from the first carrier and $\HSame(b,c)$ from
the second.
\end{proof}

\begin{theorem}[Category hom carrier unary suffix equivalence]
\label{thm:category-hom-carrier-unary-suffix-iff}
For a unary suffix $q$, a displayed hom-carrier
$\mathsf{CategoryHomCarrier}(\mathsf{append}(a,q),\mathsf{append}(b,q),f)$
is equivalent to the pair consisting of unary evidence for $q$ and the
unsuffixed carrier $\mathsf{CategoryHomCarrier}(a,b,f)$.
\end{theorem}
\begin{proof}
The forward direction reads unary evidence from the suffixed endpoints, swaps
the unary suffix past the unary morphism, and cancels the common suffix in the
continuation ledger. The reverse direction uses unary append closure and the
same commutation to rebuild the suffixed continuation.
\end{proof}

\begin{theorem}[Category hom carrier unary context exactness]

\label{thm:category-hom-carrier-comp-right-factor}

from \(b\) to \(c\).
\end{theorem}
\begin{proof}

gives \(\ContR(a,f,b)\), the composite ledger gives \(\ContR(f,g,fg)\), and
the displayed carrier gives \(\ContR(a,fg,c)\). Associativity of append turns
these three ledgers into \(\ContR(b,g,c)\), which is the right-factor
hom-carrier ledger.
\end{proof}

\begin{theorem}[Category hom carrier two-sided identity law]

\mathsf{CategoryHomCarrier}(a,b,right)\wedge
\HSame(left,f)\wedge\HSame(right,f)\wedge\HSame(left,right).

\begin{proof}
The left composite is classified against the original morphism by the
left-unit result theorem: $\ContR(\emp,f,left)$ and

\texttt{cont\_right\_unit}. Continuation determinism gives
$\HSame(right,f)$.

Transport the displayed $\mathsf{CategoryHomCarrier}(a,b,f)$ witness across
the two history-sameness witnesses. This gives
$\mathsf{CategoryHomCarrier}(a,b,left)$ and
$\mathsf{CategoryHomCarrier}(a,b,right)$.

Finally compose $\HSame(left,f)$ with the symmetry of $\HSame(right,f)$.
Transitivity yields $\HSame(left,right)$, completing the two-sided identity
