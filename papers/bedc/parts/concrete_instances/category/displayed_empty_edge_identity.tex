\begin{theorem}[Category displayed empty composite exposes edge and identity carriers]
\label{thm:category-comp-result-empty-edge-identity-carriers}
For unary-continuation hom carriers $f:a\to b$ and $g:b\to c$, if
$\ContR(f,g,fg)$ and $\HSame(fg,\emp)$, then the empty morphism is displayed
both on the two original edges and as the three endpoint identity carriers:
$$
\mathsf{CategoryHomCarrier}(a,b,\emp)\wedge
\mathsf{CategoryHomCarrier}(b,c,\emp)\wedge
\mathsf{CategoryHomCarrier}(a,a,\emp)\wedge
\mathsf{CategoryHomCarrier}(b,b,\emp)\wedge
\mathsf{CategoryHomCarrier}(c,c,\emp).
$$
\end{theorem}
\begin{proof}
Transport the displayed composite ledger along $\HSame(fg,\emp)$. The empty
composite identity-factor theorem gives the two edge carriers, and the empty
composite identity-carrier theorem gives the three endpoint identity carriers.
\end{proof}
