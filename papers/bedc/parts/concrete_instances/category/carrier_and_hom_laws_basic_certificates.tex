\begin{theorem}[Category hom carrier identity semantic name certificate]
\label{thm:category-hom-carrier-identity-semantic-namecert}
For a unary object history $a$, the identity hom-carrier predicate
$f\mapsto\mathsf{CategoryHomCarrier}(a,a,f)$ carries a semantic naming
certificate whose source, pattern, and ledger predicates are this hom-carrier
predicate and whose classifier is $\hsame$.
\end{theorem}
\begin{proof}
The carrier witness is the empty history, using the empty identity hom-carrier
at $a$. The classifier laws are the reflexive, symmetric, and transitive laws
for $\hsame$. Carrier transport follows by transporting the concrete
hom-carrier witness across history sameness.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.CategoryHomCarrier\_identity\_semanticNameCert}

\begin{theorem}[Category hom carrier semantic name certificate from a hom]
\label{thm:category-hom-carrier-semantic-namecert-of-hom}
For displayed unary-continuation hom data
$\mathsf{CategoryHomCarrier}(a,b,f)$, the hom-carrier predicate
$g\mapsto\mathsf{CategoryHomCarrier}(a,b,g)$ carries a semantic naming
certificate whose source, pattern, and ledger predicates are this hom-carrier
predicate and whose classifier is $\hsame$.
\end{theorem}
\begin{proof}
The displayed morphism $f$ supplies carrier inhabitation. The classifier laws
are reflexivity, symmetry, and transitivity of $\hsame$. If a classified
morphism is transported across $\hsame$, endpoint histories stay fixed and
\autoref{thm:category-hom-carrier-hsame-transport} rebuilds the
hom-carrier witness. Pattern and ledger soundness are identities because the
three semantic predicates are the same hom-carrier predicate.
\end{proof}

continuation-morphism witness is rebuilt at the transported source.

\label{thm:continuation-morphism-target-semantic-namecert}
For a displayed continuation morphism $m:\mathsf{ContinuationMorphism}(a,b)$,

$tgt\mapsto\mathsf{Nonempty}(\mathsf{ContinuationMorphism}(a,tgt))$

\end{theorem}
\begin{proof}
The displayed morphism $m$ gives carrier inhabitation at target $b$.

and displayed tail fixed, thereby rebuilding a concrete continuation morphism
at the transported target. Pattern and ledger soundness are identities for the
same predicate.
\end{proof}
