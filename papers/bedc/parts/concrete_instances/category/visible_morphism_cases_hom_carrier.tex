\begin{theorem}[Category visible composable morphism target cases]
\label{thm:category-visible-composable-morphisms-target-cases}
For histories $a,b,c,l,m,k$, if
$$
\mathsf{CategoryHomCarrier}(a,b,\Eone(l)),\qquad
\mathsf{CategoryHomCarrier}(b,c,\Eone(m)),\qquad
\ContR(\Eone(l),\Eone(m),\Eone(k)),
$$
then there exist histories $r,s$ such that
$$
b=\Eone(r)\land c=\Eone(s)\land
\mathsf{CategoryHomCarrier}(a,r,l)\land
\mathsf{CategoryHomCarrier}(\Eone(r),s,m)\land
\ContR(\Eone(l),\Eone(m),\Eone(k)).
$$
\end{theorem}
\begin{proof}
Apply the one-headed morphism target-case theorem to the left visible
hom-carrier and to the right visible hom-carrier. The two applications expose
tails $r$ and $s$ with $b=\Eone(r)$ and $c=\Eone(s)$. Substitute the first
target equality into the right descended hom-carrier, retain the left
descended hom-carrier, and carry the original visible continuation ledger into
the existential package.
\end{proof}

\begin{theorem}[Category hom-carrier nonempty morphism target visible]
\label{thm:category-hom-carrier-nonempty-morphism-target-visible}
For histories $source,target,morph:\mathsf{BHist}$, if
$$
\mathsf{CategoryHomCarrier}(source,target,morph)
$$
and $morph$ is not internally empty, then the morphism and target are both
one-headed and the hom-continuation ledger descends through their exposed
payloads:
$$
\exists k,r.\;
morph=\Eone(k)\land target=\Eone(r)\land
\mathsf{UnaryHistory}(k)\land \mathsf{UnaryHistory}(r)\land
\ContR(source,k,r).
$$
\end{theorem}
\begin{proof}
Unpack the hom-carrier witness. The morphism cannot be empty by the nonempty
hypothesis, and it cannot be zero-headed because unary histories have no
zero-headed extension. In the one-headed morphism case, the continuation
ledger forces the result target to be one-headed; constructor inversion then
exposes the target payload, both unary payload witnesses, and the descended
continuation ledger.
\end{proof}

\begin{theorem}[Category hom-carrier nonempty morphism visible endpoint cases]
\label{thm:category-hom-carrier-nonempty-morphism-visible-endpoint-cases}
For histories $source,target,morph:\mathsf{BHist}$, if
$$
\mathsf{CategoryHomCarrier}(source,target,morph)
$$
and $morph$ is not internally empty, then either the source is empty and the
visible target and morphism coincide, or all three visible endpoints are
one-headed:
$$
\begin{array}{l}
\exists r.\;source=\emp\land target=\Eone(r)\land morph=\Eone(r)\land
\mathsf{UnaryHistory}(r)
\\
\vee
\exists a,k,r.\;source=\Eone(a)\land morph=\Eone(k)\land target=\Eone(r)\land
\mathsf{UnaryHistory}(a)\land\mathsf{UnaryHistory}(k)\land
\mathsf{UnaryHistory}(r)\land\ContR(\Eone(a),k,r).
\end{array}
$$
\end{theorem}
\begin{proof}
Inspect the source constructor. In the empty-source branch, the empty-source
hom-carrier law identifies the morphism with the target; the nonempty morphism
hypothesis removes the empty target case, and unary generation removes the
zero-headed target case. In the one-headed source branch, the nonempty and
unary hypotheses force the morphism to be one-headed, and the continuation
ledger then forces the target to be one-headed. Constructor inversion exposes
the three unary payloads and the descended continuation.
\end{proof}

\begin{theorem}[Category hom-carrier nonempty composite target visible]
\label{thm:category-hom-carrier-comp-result-nonempty-target-visible}
For the unary-continuation category, suppose
$$
\mathsf{CategoryHomCarrier}(a,b,f),\qquad
\mathsf{CategoryHomCarrier}(b,c,g),\qquad
\ContR(f,g,fg).
$$
If the displayed composite morphism $fg$ is not internally empty, then $fg$
and the target object $c$ are both one-headed, and the composite continuation
ledger is visible at that boundary:
$$
\exists k,r.\;
fg=\Eone(k)\land c=\Eone(r)\land
\mathsf{UnaryHistory}(k)\land \mathsf{UnaryHistory}(r)\land
\ContR(a,\Eone(k),\Eone(r)).
$$
\end{theorem}
\begin{proof}
Composition closure turns the two hom-carriers and the displayed composite
ledger into a hom-carrier from $a$ to $c$ over $fg$. The composite morphism is
unary. The empty case contradicts the nonempty hypothesis, and the zero-headed
case contradicts unary generation. In the one-headed case, the target cannot
be empty or zero-headed because the continuation ledger has one-headed result
shape; the remaining target case exposes $c=\Eone(r)$ and carries the original
continuation witness at the visible boundary.
\end{proof}
