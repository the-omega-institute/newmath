\begin{theorem}[Continuation morphism empty-source one-headed target tail iff]
\label{thm:continuation-morphism-empty-source-e1-target-tail}
For histories $r,k:\mathsf{BHist}$ and a continuation morphism
$m:\mathsf{ContinuationMorphism}(\emp,\Eone(r))$, the displayed tail of $m$
has one-headed payload $k$ exactly when the target payload is $k$:
$$
\hsame(\mathsf{tail}(m),\Eone(k))
\Longleftrightarrow
\hsame(r,k).
$$
\end{theorem}
\begin{proof}
The empty-source tail target exactness identifies $\mathsf{tail}(m)$ with
$\Eone(r)$. In the forward direction, compose this equality with the displayed
tail equality and invert the one-headed constructor. Conversely, lift
$r\hsame k$ through the one-headed constructor and compose with the tail-target
exactness.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.ContinuationMorphism\_empty\_source\_e1\_target\_tail\_iff}
