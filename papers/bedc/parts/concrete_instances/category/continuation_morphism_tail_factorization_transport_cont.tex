\begin{theorem}[Continuation morphism tail factorization transport continuation]
\label{thm:continuation-morphism-tail-factorization-transport-cont}
For continuation morphisms $left:a\to b$ and $right:b\to c$, if displayed
tails $l$ and $r$ classify the two component tails, then the canonical
composite tail is the continuation product of $l$ and $r$:
$$
\hsame(\mathsf{tail}(left),l)
\Longrightarrow
\hsame(\mathsf{tail}(right),r)
\Longrightarrow
\ContR(l,r,\mathsf{tail}(left\circ right)).
$$
\end{theorem}
\begin{proof}
The composite-tail continuation theorem gives
$\ContR(\mathsf{tail}(left),\mathsf{tail}(right),
\mathsf{tail}(left\circ right))$. Transport this continuation along the two
tail classifier witnesses and keep the composite result endpoint fixed.
\end{proof}
