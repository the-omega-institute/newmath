\begin{theorem}[Category hom-carrier one-headed endpoint morphism payload determinacy]
\label{thm:category-e1-source-e1-target-morphism-payload-deterministic}
For histories $a,a',r,r',k,k'$, if
$\hsame(a,a')$ and $\hsame(r,r')$, then any two hom-carriers
$$
\mathsf{CategoryHomCarrier}(\Eone(a),\Eone(r),\Eone(k))
\qquad\text{and}\qquad
\mathsf{CategoryHomCarrier}(\Eone(a'),\Eone(r'),\Eone(k'))
$$
have internally same visible morphism payloads: $\hsame(k,k')$.
\end{theorem}
\begin{proof}
Lift the source and target endpoint equalities through the one-headed
constructor. Endpoint-sensitive morphism determinacy gives
$\hsame(\Eone(k),\Eone(k'))$ for the displayed morphisms. One-headed
constructor inversion removes the visible tag and yields $\hsame(k,k')$.
\end{proof}
