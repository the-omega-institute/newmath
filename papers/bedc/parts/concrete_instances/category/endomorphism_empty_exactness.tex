\begin{theorem}[Category endomorphism tail empty exactness]
\label{thm:category-hom-carrier-endomorphism-tail-empty}
For a unary-continuation endomorphism carrier, the morphism tail is externally
empty exactly when the source object and morphism tail are unary and the tail is
internally the empty history:
$$
\mathsf{CategoryHomCarrier}(a,a,f)
\quad\Longleftrightarrow\quad
\mathsf{UnaryHistory}(a)\wedge\mathsf{UnaryHistory}(f)\wedge f\hsame\emp.
$$
\end{theorem}
\leanchecked{BEDC.Derived.CategoryUp.CategoryHomCarrier\_endomorphism\_tail\_empty\_iff}
\begin{proof}
Unpack the endomorphism carrier. Its continuation field is
\(\ContR(a,f,a)\), so right-unit uniqueness for continuation forces
\(f\hsame\emp\), while the carrier fields give unary membership of \(a\) and
\(f\). Conversely, if \(f\hsame\emp\), substituting the empty tail rebuilds
the canonical empty identity hom-carrier at \(a\).
\end{proof}

\begin{theorem}[Category hom carrier endomorphism empty exactness]
\label{thm:category-hom-carrier-endomorphism-empty-iff}
For histories $a,f:\mathsf{BHist}$, an endomorphism hom carrier at $a$ is
exactly a unary object witness for $a$ together with an empty morphism tail:
$$
\mathsf{CategoryHomCarrier}(a,a,f)
\quad\Longleftrightarrow\quad
\mathsf{UnaryHistory}(a)\wedge\hsame(f,\emp).
$$
\end{theorem}
\leanchecked{BEDC.Derived.CategoryUp.CategoryHomCarrier\_endomorphism\_empty\_iff}
\begin{proof}
Unpack the hom-carrier witness. Its continuation field is
$\ContR(a,f,a)$, so right-unit uniqueness forces $f\hsame\emp$, while the
source unary witness is already present. Conversely, after eliminating the
empty morphism-tail sameness witness, the empty identity hom carrier at the
unary object $a$ gives the required endomorphism carrier.
\end{proof}
