\begin{theorem}[Continuation morphism one-headed source to one-headed target tail cases]
\label{thm:continuation-morphism-e1-source-e1-target-tail-cases}
For histories $a,r:\mathsf{BHist}$, every continuation morphism
$m:\mathsf{ContinuationMorphism}(\Eone(a),\Eone(r))$ has either the empty-tail
case with the exposed endpoints classified together,
$$
\mathsf{tail}(m)=\emp\land\HSame(a,r),
$$
or a one-headed tail whose payload still continues from $\Eone(a)$ to $r$:
$$
\exists k:\mathsf{BHist}.\;
\mathsf{tail}(m)=\Eone(k)\land \ContR(\Eone(a),k,r).
$$
\end{theorem}
\begin{proof}
Unpack the displayed continuation tail. The empty-tail branch is the right-unit
continuation, so constructor injectivity exposes $\HSame(a,r)$. A zero-headed
tail would force a one-headed result to be zero-headed, contradicting
constructor disjointness. In the one-headed branch, constructor injectivity on
the continuation result removes the outer one constructor and leaves
$\ContR(\Eone(a),k,r)$.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.ContinuationMorphism\_e1\_source\_e1\_target\_tail\_cases}

\begin{theorem}[Continuation morphism one-headed endpoints reject zero-headed tail]
\label{thm:continuation-morphism-e1-source-e1-target-tail-e0-absurd}
For histories $a,r,k:\mathsf{BHist}$ and
$m:\mathsf{ContinuationMorphism}(\Eone(a),\Eone(r))$,
$$
\HSame(\mathsf{tail}(m),\Ezero(k))\to\mathsf{False}.
$$
\end{theorem}
\begin{proof}
Apply the preceding one-headed endpoint tail cases. The empty-tail branch gives
$\emp\hsame\Ezero(k)$, and the visible branch gives
$\Eone(t)\hsame\Ezero(k)$ for some $t$; both contradict constructor separation.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.ContinuationMorphism\_e1\_source\_e1\_target\_tail\_e0\_absurd}

\begin{theorem}[Continuation morphism one-headed endpoints one-tail exactness]
\label{thm:continuation-morphism-e1-source-e1-target-tail-e1-iff}
For histories $a,r,k:\mathsf{BHist}$ and
$m:\mathsf{ContinuationMorphism}(\Eone(a),\Eone(r))$, the displayed tail is
one-headed with payload $k$ exactly when the payload continues from the
one-headed source to the target payload:
$$
\HSame(\mathsf{tail}(m),\Eone(k))
\Longleftrightarrow
\ContR(\Eone(a),k,r).
$$
\end{theorem}
\begin{proof}
For the forward direction, apply the one-headed endpoint tail cases. The
empty-tail branch contradicts the assumed one-headed tail comparison. In the
visible branch, the exposed one-headed tail payload and the assumed tail
comparison identify the two payloads, so continuation transport gives
$\ContR(\Eone(a),k,r)$. Conversely, step the displayed payload continuation
through the one constructor and compare it with the record continuation by
left cancellation.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.ContinuationMorphism\_e1\_source\_e1\_target\_tail\_e1\_iff}
