\section{The certificate}

\begin{definition}[Category source specification]
\label{def:category-source-specification}
The source is the licensed object carrier and each licensed hom-carrier.
\end{definition}

\begin{definition}[Category pattern specification]
\label{def:category-pattern-specification}
The pattern is morphism expression formation from identities and composable pairs.
\end{definition}

\begin{definition}[Category classifier specification]
\label{def:category-classifier-specification}
The classifier is object sameness for objects and hom-carrier sameness for morphisms, indexed by source and target.
\end{definition}

\begin{definition}[Category stability certificate]
\label{def:category-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item object classifier reflexivity and transitivity
  \item morphism classifier reflexivity and transitivity
  \item identity morphism stability
  \item composition closure
  \item associativity under morphism classifier
  \item left and right identity laws
  \item composition congruence
\end{enumerate}
\end{definition}
\leanchecked{BEDC.Derived.CategoryUp.ContinuationMorphism\_identity\_comp\_closure}

\begin{definition}[Category ledger policy]
\label{def:category-ledger-policy}
Record object ledgers, hom-carrier ledgers, and composition trees.
\end{definition}

\begin{definition}[\(\CategoryUp\) certificate obligations]
\label{def:category-certificate-obligations}
The carrier, source, pattern, classifier, stability, and ledger data listed above are the obligations required for the interface name \(\CategoryUp\).
\end{definition}

\begin{theorem}[Category left zero-headed continuation result cases]
\label{thm:category-cont-left-e0-result-cases}
If $\ContR(\Ezero(h),k,\Ezero(r))$, then either $k=\emp$ and
$\HSame(h,r)$, or $k=\Ezero(k_0)$ for some $k_0$ and
$\ContR(\Ezero(h),k_0,r)$.
\end{theorem}
\leanchecked{BEDC.Derived.CategoryUp.category\_cont\_left\_e0\_result\_cases}
\begin{proof}
Classify the continuation history $k$. The empty branch identifies the inner
histories. The zero branch exposes the predecessor of $k$ and removes the
matching outer constructor from the continuation result. The one branch would
identify a zero-headed history with a one-headed history, which is impossible
by generated-history no-confusion.
\end{proof}

\begin{theorem}[Category left one-headed continuation result cases]
\label{thm:category-cont-left-e1-result-cases}
If $\ContR(\Eone(h),k,\Eone(r))$, then either $k=\emp$ and
$\HSame(h,r)$, or $k=\Eone(k_1)$ for some $k_1$ and
$\ContR(\Eone(h),k_1,r)$.
\end{theorem}
\leanchecked{BEDC.Derived.CategoryUp.category\_cont\_left\_e1\_result\_cases}
\begin{proof}
Classify the continuation history $k$. The empty branch removes the matching
one-headed constructor from the displayed result. The zero branch is ruled out
by constructor disjointness. The one branch exposes the predecessor of $k$ and
removes the matching outer constructor from the continuation result.
\end{proof}

\begin{theorem}[Category hom-carrier zero-headed component impossibility]
\label{thm:category-hom-carrier-zero-headed-component-absurd}
For histories $source,target,morph$, a category hom-carrier has no
zero-headed source, target, or morphism component:
$$
\mathsf{CategoryHomCarrier}(source,target,morph)
\Longrightarrow
\left(
\exists z,\ source=\Ezero(z)
\right)
\lor
\left(
\exists z,\ target=\Ezero(z)
\right)
\lor
\left(
\exists z,\ morph=\Ezero(z)
\right)
\Longrightarrow
\mathsf{False}.
$$
\end{theorem}
\begin{proof}
Unpack the hom-carrier into its source, target, and morphism unary-history
fields. A zero-headed source, target, or morphism rewrites the corresponding
field to a unary witness for a zero-headed extension, contradicting unary
separation for zero-headed histories.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.CategoryHomCarrier\_zero\_headed\_component\_absurd}

\begin{theorem}[Category hom-carrier composite zero-headed result impossibility]
\label{thm:category-hom-carrier-comp-zero-result-absurd}
For histories $a,b,c,f,g,z:\mathsf{BHist}$, if $f$ is a hom-carrier from $a$
to $b$, $g$ is a hom-carrier from $b$ to $c$, and the displayed composite
continuation has zero-headed result
$$
\ContR(f,g,\Ezero(z)),
$$
then contradiction follows.
\end{theorem}
\begin{proof}
Composition closure makes $\Ezero(z)$ a hom-carrier morphism from $a$ to $c$.
Its morphism-carrier field is therefore a unary-history witness for a
zero-headed history, contradicting unary separation for zero-headed histories.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.CategoryHomCarrier\_comp\_zero\_result\_absurd}

\begin{theorem}[Category hom-carrier left one-headed result cases]
\label{thm:category-hom-carrier-left-e1-result-cases}
If
$$
\mathsf{CategoryHomCarrier}(\Eone(a),\Eone(r),k)
$$
then either $k=\emp$, $a$ is unary, and $\HSame(a,r)$, or there is a history
$k_0$ such that $k=\Eone(k_0)$, $k_0$ is unary, and
$$
\ContR(\Eone(a),k_0,r).
$$
\end{theorem}
\begin{proof}
Unpack the hom-carrier into unary source, unary morphism-tail, and continuation
fields. Classify the tail $k$. The empty branch cancels the visible one
constructor in the result and inverts the source unary witness. The zero-headed
tail branch is impossible by constructor no-confusion. The one-headed tail
branch removes the outer constructor from the continuation result and inverts
the morphism-tail unary witness to obtain the unary carrier for $k_0$.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.CategoryHomCarrier\_left\_e1\_result\_cases}

\begin{theorem}[Category hom-carrier one-headed source and target morphism exactness]
\label{thm:category-hom-carrier-e1-source-e1-target-morphism-iff}
For histories $a,r,morph$, a hom-carrier from $\Eone(a)$ to $\Eone(r)$ is
exactly either the empty morphism identifying the two exposed endpoints, or a
one-headed morphism whose tail continues the visible source to the exposed
target:
$$
\mathsf{CategoryHomCarrier}(\Eone(a),\Eone(r),morph)
\Longleftrightarrow
\left(morph=\emp\land\mathsf{UnaryHistory}(a)\land\hsame(a,r)\right)
\lor
\left(\exists k:\mathsf{BHist}.\;
morph=\Eone(k)\land\mathsf{UnaryHistory}(a)\land
\mathsf{UnaryHistory}(k)\land\ContR(\Eone(a),k,r)\right).
$$
\end{theorem}
\begin{proof}
The forward direction applies the visible-source and visible-target case split:
the empty branch gives the exposed endpoint classifier, while the one-headed
branch gives a tail $k$ and a continuation ledger $\ContR(\Eone(a),k,r)$.
Unary inversion on the source carrier supplies $\mathsf{UnaryHistory}(a)$.
Conversely, the empty branch is rebuilt by the right-unit continuation after
transport along $\hsame(a,r)$; the one-headed branch uses unary continuation
closure for the exposed target and the one-step continuation rule for the
visible morphism.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.CategoryHomCarrier\_e1\_source\_e1\_target\_morphism\_iff}

morphism into a one-headed shape, and constructor inversion gives $\HSame$
between the exposed morphism tails.
\end{proof}

\begin{theorem}[Continuation morphism zero-headed source to one-headed target tail cases]
\label{thm:category-continuation-morphism-e0-source-e1-target-tail-cases}
For histories $a,r:\mathsf{BHist}$, every continuation morphism
$m:\mathsf{ContinuationMorphism}(\Ezero(a),\Eone(r))$ has a one-headed tail
whose exposed tail still continues from $\Ezero(a)$ to $r$:
$$
\exists k:\mathsf{BHist}.\;
\HSame(\mathsf{tail}(m),\Eone(k))\land \ContR(\Ezero(a),k,r).
$$
\end{theorem}
\begin{proof}
Unpack the displayed tail of $m$. The empty-tail case would force
$\Ezero(a)\hsame\Eone(r)$, contradicting constructor disjointness. The
zero-headed tail case would force a one-headed result to be zero-headed, again
contradicting constructor disjointness. Therefore the tail is one-headed, and
continuation step-result inversion exposes the residual continuation
$\ContR(\Ezero(a),k,r)$.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.ContinuationMorphism\_e0\_source\_e1\_target\_tail\_cases}

\begin{theorem}[Continuation morphism zero-headed source to one-headed target tail core determinacy]
\label{thm:continuation-morphism-e0-source-e1-target-tail-core-deterministic}
For histories $a,r,k,l:\mathsf{BHist}$ and continuation morphisms
$m,n:\mathsf{ContinuationMorphism}(\Ezero(a),\Eone(r))$, if the two displayed
tails are classified with one-headed presentations
$\Eone(k)$ and $\Eone(l)$, then the exposed tail cores are classified:
$$
\HSame(\mathsf{tail}(m),\Eone(k))\to
\HSame(\mathsf{tail}(n),\Eone(l))\to
\HSame(k,l).
$$
\end{theorem}
\begin{proof}
Tail determinacy for continuation morphisms with the same endpoints gives
$\HSame(\mathsf{tail}(m),\mathsf{tail}(n))$. Compose this comparison with the
two exposed one-headed presentations. Constructor inversion for $\Eone$ then
descends the comparison to the exposed cores $k$ and $l$.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.ContinuationMorphism\_e0\_source\_e1\_target\_tail\_core\_deterministic}

\begin{theorem}[Continuation morphism zero-headed source to one-headed target rejects zero-headed tail]
\label{thm:category-continuation-e0-source-e1-target-tail-e0-absurd}
For histories $a,r,k:\mathsf{BHist}$ and
$m:\mathsf{ContinuationMorphism}(\Ezero(a),\Eone(r))$,
$$
\HSame(\mathsf{tail}(m),\Ezero(k))\to\mathsf{False}.

\end{theorem}
\begin{proof}

$t$ with $\mathsf{tail}(m)\hsame\Eone(t)$. If the displayed tail were also
$\HSame$ to $\Ezero(k)$, transitivity would give
$\Eone(t)\hsame\Ezero(k)$, contradicting constructor separation.
\end{proof}

\begin{theorem}[Continuation morphism zero-headed source to one-headed target one-tail exactness]

$m:\mathsf{ContinuationMorphism}(\Ezero(a),\Eone(r))$, the displayed tail is
one-headed with payload $k$ exactly when that payload continues from the
zero-headed source to the one-headed target payload:
$$
\HSame(\mathsf{tail}(m),\Eone(k))
\Longleftrightarrow
\ContR(\Ezero(a),k,r).
$$
\end{theorem}
\leanchecked{BEDC.Derived.CategoryUp.ContinuationMorphism\_e0\_source\_e1\_target\_tail\_e1\_iff}
\begin{proof}
For the forward direction, inspect the displayed tail of $m$. Empty and
zero-headed tails contradict the assumed one-headed tail comparison. In the
one-headed branch, constructor inversion on the comparison identifies the
payloads, and constructor inversion on the continuation result exposes
$\ContR(\Ezero(a),k,r)$.

Conversely, step the displayed payload continuation through the one constructor
to form a continuation morphism with tail $\Eone(k)$. Tail determinacy against
$m$ gives the displayed tail comparison.
\end{proof}
