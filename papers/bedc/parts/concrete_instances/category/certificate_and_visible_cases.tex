\section{The certificate}

\begin{definition}[Category source specification]
\label{def:category-source-specification}
The source is the licensed object carrier and each licensed hom-carrier.
\end{definition}

\begin{definition}[Category pattern specification]
\label{def:category-pattern-specification}
The pattern is morphism expression formation from identities and composable pairs.
\end{definition}

\begin{definition}[Category classifier specification]
\label{def:category-classifier-specification}
The classifier is object sameness for objects and hom-carrier sameness for morphisms, indexed by source and target.
\end{definition}

\begin{definition}[Category stability certificate]
\label{def:category-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item object classifier reflexivity and transitivity
  \item morphism classifier reflexivity and transitivity
  \item identity morphism stability
  \item composition closure
  \item associativity under morphism classifier
  \item left and right identity laws
  \item composition congruence
\end{enumerate}
\end{definition}
\leanchecked{BEDC.Derived.CategoryUp.ContinuationMorphism\_identity\_comp\_closure}

\begin{definition}[Category ledger policy]
\label{def:category-ledger-policy}
Record object ledgers, hom-carrier ledgers, and composition trees.
\end{definition}

\begin{definition}[\(\CategoryUp\) certificate obligations]
\label{def:category-certificate-obligations}
The carrier, source, pattern, classifier, stability, and ledger data listed above are the obligations required for the interface name \(\CategoryUp\).
\end{definition}

\begin{theorem}[Category left zero-headed continuation result cases]
\label{thm:category-cont-left-e0-result-cases}
If $\ContR(\Ezero(h),k,\Ezero(r))$, then either $k=\emp$ and
$\HSame(h,r)$, or $k=\Ezero(k_0)$ for some $k_0$ and
$\ContR(\Ezero(h),k_0,r)$.
\end{theorem}
\leanchecked{BEDC.Derived.CategoryUp.category\_cont\_left\_e0\_result\_cases}
\begin{proof}
Classify the continuation history $k$. The empty branch identifies the inner
histories. The zero branch exposes the predecessor of $k$ and removes the
matching outer constructor from the continuation result. The one branch would
identify a zero-headed history with a one-headed history, which is impossible
by generated-history no-confusion.
\end{proof}

\begin{theorem}[Category hom-carrier zero-headed source impossibility]
\label{thm:category-hom-carrier-e0-source-absurd}
For histories $a,target,f$, no hom-carrier can have source $\Ezero(a)$:
$$
\mathsf{CategoryHomCarrier}(\Ezero(a),target,f)\to\mathsf{False}.
$$
\end{theorem}
\begin{proof}
Unpack the hom-carrier witness. Its source-carrier field says that
$\Ezero(a)$ is unary, contradicting the BEDC unary-history separation theorem
for zero-headed extensions.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.CategoryHomCarrier\_e0\_source\_absurd}

\begin{theorem}[Category hom-carrier left one-headed result cases]
\label{thm:category-hom-carrier-left-e1-result-cases}
If
$$
\mathsf{CategoryHomCarrier}(\Eone(a),\Eone(r),k)
$$
then either $k=\emp$, $a$ is unary, and $\HSame(a,r)$, or there is a history
$k_0$ such that $k=\Eone(k_0)$, $k_0$ is unary, and
$$
\ContR(\Eone(a),k_0,r).
$$
\end{theorem}
\begin{proof}
Unpack the hom-carrier into unary source, unary morphism-tail, and continuation
fields. Classify the tail $k$. The empty branch cancels the visible one
constructor in the result and inverts the source unary witness. The zero-headed
tail branch is impossible by constructor no-confusion. The one-headed tail
branch removes the outer constructor from the continuation result and inverts
the morphism-tail unary witness to obtain the unary carrier for $k_0$.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.CategoryHomCarrier\_left\_e1\_result\_cases}

\begin{theorem}[Category hom-carrier one-headed source and target morphism exactness]
\label{thm:category-hom-carrier-e1-source-e1-target-morphism-iff}
For histories $a,r,morph$, a hom-carrier from $\Eone(a)$ to $\Eone(r)$ is
exactly either the empty morphism identifying the two exposed endpoints, or a
one-headed morphism whose tail continues the visible source to the exposed
target:
$$
\mathsf{CategoryHomCarrier}(\Eone(a),\Eone(r),morph)
\Longleftrightarrow
\left(morph=\emp\land\mathsf{UnaryHistory}(a)\land\hsame(a,r)\right)
\lor
\left(\exists k:\mathsf{BHist}.\;
morph=\Eone(k)\land\mathsf{UnaryHistory}(a)\land
\mathsf{UnaryHistory}(k)\land\ContR(\Eone(a),k,r)\right).
$$
\end{theorem}
\begin{proof}
The forward direction applies the visible-source and visible-target case split:
the empty branch gives the exposed endpoint classifier, while the one-headed
branch gives a tail $k$ and a continuation ledger $\ContR(\Eone(a),k,r)$.
Unary inversion on the source carrier supplies $\mathsf{UnaryHistory}(a)$.
Conversely, the empty branch is rebuilt by the right-unit continuation after
transport along $\hsame(a,r)$; the one-headed branch uses unary continuation
closure for the exposed target and the one-step continuation rule for the
visible morphism.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.CategoryHomCarrier\_e1\_source\_e1\_target\_morphism\_iff}

\begin{theorem}[Category empty morphism between one-headed endpoints]
\label{thm:category-hom-carrier-e1-source-e1-target-empty-morphism-iff}
For histories $a,r$, the empty morphism is a hom-carrier from $\Eone(a)$ to
$\Eone(r)$ exactly when the exposed source tail is unary and the two exposed
tails are classified together:
$$
\mathsf{CategoryHomCarrier}(\Eone(a),\Eone(r),\emp)
\Longleftrightarrow
\mathsf{UnaryHistory}(a)\land\HSame(a,r).
$$
\end{theorem}
\begin{proof}
The forward direction applies the one-headed result case split to the displayed
empty morphism. The visible morphism branch contradicts constructor
no-confusion, and the empty branch exposes unary membership of $a$ and
$\HSame(a,r)$. Conversely, unary one-step closure supplies the two endpoint
carriers, the tail classifier lifts through $\Eone$, and empty-identity
exactness rebuilds the hom-carrier.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.CategoryHomCarrier\_e1\_source\_e1\_target\_empty\_morphism\_iff}

\begin{theorem}[Category hom-carrier one-headed morphism target exactness]
\label{thm:category-hom-carrier-e1-morphism-target-iff}
For histories $a,k,r$,
$$
\mathsf{CategoryHomCarrier}(a,\Eone(r),\Eone(k))
\quad\Longleftrightarrow\quad
\mathsf{UnaryHistory}(a)\land \mathsf{UnaryHistory}(k)\land \ContR(a,k,r).
$$
\end{theorem}
\begin{proof}
Unpack the hom-carrier witness. The forward direction keeps the unary source
witness, inverts the one-headed morphism unary witness to obtain unary
membership for $k$, and removes the outer one constructor from the
continuation result. Conversely, unary continuation closure gives unary
membership for $r$, one-step closure gives the one-headed target and morphism
carriers, and the continuation one-step rule lifts $\ContR(a,k,r)$ to
$\ContR(a,\Eone(k),\Eone(r))$.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.CategoryHomCarrier\_e1\_morphism\_target\_iff}

\begin{theorem}[Category empty-source one-headed target exactness]
\label{thm:category-hom-carrier-empty-source-e1-target-iff}
For histories $r,morph$, an empty-source hom-carrier into the one-headed
target $\Eone(r)$ is exactly equality of the morphism with that target
together with unary payload membership:
$$
\mathsf{CategoryHomCarrier}(\emp,\Eone(r),morph)
\quad\Longleftrightarrow\quad
morph=\Eone(r)\land\mathsf{UnaryHistory}(r).
$$
\end{theorem}
\begin{proof}
Category empty-source exactness gives unary membership of the one-headed target
and equality of the morphism with that target. Unary inversion removes the
visible constructor from the target carrier. Conversely, unary one-step closure
rebuilds the target carrier and the displayed equality supplies the
empty-source continuation classifier.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.CategoryHomCarrier\_empty\_source\_e1\_target\_iff}

\begin{theorem}[Category empty-source one-headed morphism exactness]
\label{thm:category-hom-carrier-empty-source-e1-morphism-iff}
For histories $k,target$, an empty-source hom-carrier whose morphism is
$\Eone(k)$ is exactly equality of the target with that morphism together with
unary payload membership:
$$
\mathsf{CategoryHomCarrier}(\emp,target,\Eone(k))
\quad\Longleftrightarrow\quad
target=\Eone(k)\land\mathsf{UnaryHistory}(k).
$$
\end{theorem}
\begin{proof}
Category empty-source exactness identifies the one-headed morphism with the
target. Eliminating that equality turns the target carrier into a one-headed
unary witness, whose payload is unary by inversion. Conversely, one-step unary
closure rebuilds the target carrier, and symmetry of the displayed equality
provides the required empty-source classifier.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.CategoryHomCarrier\_empty\_source\_e1\_morphism\_iff}

\begin{theorem}[Category hom-carrier one-headed target morphism cases]
\label{thm:category-hom-carrier-e1-target-morphism-cases}
For histories $source,r,morph$, a hom-carrier whose target is one-headed has
only the empty morphism case or a one-headed morphism tail:
$$
\mathsf{CategoryHomCarrier}(source,\Eone(r),morph)
\Longrightarrow
\left(
morph=\emp\land source=\Eone(r)\land \mathsf{UnaryHistory}(r)
\right)
$$
or there exists a history $k$ such that
$$
morph=\Eone(k)\land\mathsf{UnaryHistory}(source)\land
\mathsf{UnaryHistory}(k)\land\ContR(source,k,r).
$$
\end{theorem}
\begin{proof}
Unpack the hom-carrier witness into unary source, target, and morphism fields
and the continuation ledger. Classify the morphism history. In the empty
morphism branch, the continuation right-unit shape identifies the source with
$\Eone(r)$, and unary inversion on the target gives the unary history for $r$.
The zero-headed morphism branch contradicts unary-history separation for zero
extensions. In the one-headed branch, unary inversion gives the tail carrier
for $k$, and constructor inversion on the continuation ledger yields
$\ContR(source,k,r)$.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.CategoryHomCarrier\_e1\_target\_morphism\_cases}

\begin{theorem}[Category hom-carrier one-headed target source and morphism cases]
\label{thm:category-hom-carrier-e1-target-source-morphism-cases}
For histories $source,r,morph$, a hom-carrier whose target is $\Eone(r)$
separates the empty source branch from the one-headed source branch:
$$
\mathsf{CategoryHomCarrier}(source,\Eone(r),morph)
\Longrightarrow
\left(
source=\emp\land morph=\Eone(r)\land \mathsf{UnaryHistory}(r)
\right)
$$
or there exists a history $a$ such that
$$
source=\Eone(a)\land\mathsf{UnaryHistory}(a)\land
\left(
(morph=\emp\land\HSame(a,r))\lor
\exists k,\ morph=\Eone(k)\land\mathsf{UnaryHistory}(k)\land
\ContR(\Eone(a),k,r)
\right).
$$
\end{theorem}
\begin{proof}
Classify the source endpoint. If it is empty, the continuation ledger and the
one-headed target force the morphism to be $\Eone(r)$, and target unary
inversion gives $\mathsf{UnaryHistory}(r)$. If it is zero-headed, the source
unary field contradicts zero-extension separation. If it is one-headed, apply
the left-one-headed result case split: the empty morphism branch gives
$\HSame(a,r)$, and the visible morphism branch exposes a tail $k$ with
$\ContR(\Eone(a),k,r)$.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.CategoryHomCarrier\_e1\_target\_source\_morphism\_cases}

\begin{theorem}[Category hom-carrier one-headed morphism target cases]
\label{thm:category-hom-carrier-one-headed-morphism-target-cases}
For histories $a,target,k$, a hom-carrier whose morphism endpoint is
one-headed forces the target endpoint to be one-headed and descends the
hom-carrier to the exposed target and morphism tails:
$$
\mathsf{CategoryHomCarrier}(a,target,\Eone(k))
\Longrightarrow
\exists r:\mathsf{BHist}.\;
target=\Eone(r)\land\mathsf{CategoryHomCarrier}(a,r,k).
$$
\end{theorem}
\begin{proof}
Unpack the hom-carrier into its unary source, target, and morphism fields and
its continuation ledger. The continuation step-result inversion for a
one-headed right input exposes a target tail $r$ with $target=\Eone(r)$ and a
tail continuation $\ContR(a,k,r)$. Unary inversion on the displayed target and
morphism fields gives the two tail carriers, which repack with the source
carrier and tail continuation as $\mathsf{CategoryHomCarrier}(a,r,k)$.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.CategoryHomCarrier\_e1\_morphism\_target\_cases}

\begin{theorem}[Category hom-carrier one-headed target and morphism descent]
\label{thm:category-hom-carrier-e1-target-morphism-exactness}
For histories $a,k,r$, a hom-carrier whose target and morphism histories are
both one-headed descends to the corresponding tail hom-carrier:
$$
\mathsf{CategoryHomCarrier}(a,\Eone(r),\Eone(k))
\Longrightarrow
\mathsf{CategoryHomCarrier}(a,r,k).
$$
\end{theorem}
\begin{proof}
Unpack the hom-carrier fields. The source unary witness is unchanged. Unary
inversion removes the visible constructor from both the target carrier and the
morphism carrier, and the right step-rule inversion for $\ContR$ removes the
same visible constructor from the continuation ledger. Repacking these fields
gives $\mathsf{CategoryHomCarrier}(a,r,k)$.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.CategoryHomCarrier\_e1\_target\_morphism\_exactness}

\begin{theorem}[Category hom-carrier one-headed source and morphism target exactness]
\label{thm:category-hom-e1-source-e1-morphism-target-iff}
For histories $a,k,t$,
$$
\begin{aligned}
&\mathsf{CategoryHomCarrier}(\Eone(a),t,\Eone(k)) \\
&\quad\Longleftrightarrow\quad
\exists r,\ t=\Eone(r)\land \mathsf{UnaryHistory}(a)\land
\mathsf{UnaryHistory}(k)\land \mathsf{UnaryHistory}(r)\land
\ContR(\Eone(a),k,r).
\end{aligned}
$$
\end{theorem}
\begin{proof}
Unpack the hom-carrier witness. The continuation field forces the target to
have a visible one constructor; removing that constructor gives
\(\ContR(\Eone(a),k,r)\). The source, morphism, and target unary witnesses
invert through the same visible constructors. Conversely, the displayed target
equation, the three unary witnesses, and the one-step continuation rule rebuild
the hom-carrier fields.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.CategoryHomCarrier\_e1\_source\_e1\_morphism\_target\_iff}

\begin{theorem}[Category hom-carrier one-headed source empty target impossibility]
\label{thm:category-hom-e1-source-empty-target-absurd}
For histories $a,f$, no hom-carrier can have source $\Eone(a)$ and target
$\emp$:
$$
\mathsf{CategoryHomCarrier}(\Eone(a),\emp,f)\to\mathsf{False}.
$$
\end{theorem}
\begin{proof}
Unpack the hom-carrier witness and classify the morphism history \(f\). In
each constructor branch, the continuation field would identify the empty
target with an append whose source already has a visible one constructor,
which is impossible by generated-history no-confusion.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.CategoryHomCarrier\_e1\_source\_empty\_target\_absurd}

\begin{theorem}[Category hom-carrier one-headed source morphism cases]
\label{thm:category-hom-carrier-e1-source-morphism-cases}
For histories $a,target,morph$, a hom-carrier with source $\Eone(a)$ has one
of the following forms:
$$
\mathsf{CategoryHomCarrier}(\Eone(a),target,morph)
\Longrightarrow
\left(
morph=\emp\land target=\Eone(a)\land \mathsf{UnaryHistory}(a)
\right)
$$
or there exist histories $k,r$ such that
$$
morph=\Eone(k)\land target=\Eone(r)\land
\mathsf{UnaryHistory}(a)\land\mathsf{UnaryHistory}(k)\land
\mathsf{UnaryHistory}(r)\land\ContR(\Eone(a),k,r).
$$
\end{theorem}
\begin{proof}
Unpack the hom-carrier witness into unary source, target, and morphism fields
and the continuation ledger. Classify the morphism history. The empty branch
forces the target to be exactly $\Eone(a)$ by the right-unit continuation
shape, and unary inversion gives unary history for $a$. The zero-headed
morphism branch contradicts unary-history separation for zero extensions. In
the one-headed branch, classify the target history. Empty and zero-headed
targets contradict the continuation ledger; a one-headed target exposes
histories $k$ and $r$, unary inversion gives the three unary witnesses, and
constructor inversion on the continuation ledger yields $\ContR(\Eone(a),k,r)$.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.CategoryHomCarrier\_e1\_source\_morphism\_cases}

\paragraph{Primitive boundary.}
\(\CategoryUp\) is not a BEDC primitive. It is admitted only when those obligations are supplied.

\begin{remark}[Zero-axiom encoding note]
\label{rem:category-zero-axiom-note}
Avoids a global universe of all objects by using explicit licensed carriers.
\end{remark}
