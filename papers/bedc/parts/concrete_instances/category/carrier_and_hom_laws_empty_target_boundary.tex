\begin{theorem}[Category empty-target composite boundary inversion]
\label{thm:category-hom-carrier-empty-target-composite-boundary-inversion}
If $f$ is a hom-carrier from $a$ to $b$, $g$ is a hom-carrier from $b$ to
$\emp$, and $\ContR(f,g,fg)$, then $a$, $b$, $f$, $g$, and $fg$ are all
$\hsame$ to $\emp$.
\end{theorem}
\begin{proof}
Empty-target exactness applied to the right hom-carrier identifies $b$ and
$g$ with $\emp$. With $b$ reduced to $\emp$, empty-target exactness applied to
the left hom-carrier identifies $a$ and $f$ with $\emp$. The displayed
continuation ledger then has both inputs empty, so its result is $\emp$.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.CategoryHomCarrier\_empty\_target\_comp\_boundary\_inversion}

\begin{theorem}[Category hom-carrier composite empty-target boundary]
\label{thm:category-hom-carrier-comp-empty-target-boundary}
If $f$ is a hom-carrier from $a$ to $b$, $g$ is a hom-carrier from $b$ to
$c$, $\ContR(f,g,fg)$, and $c\hsame\emp$, then $a$, $b$, $c$, $f$, $g$, and
$fg$ are all $\hsame$ to $\emp$.
\end{theorem}
\begin{proof}
Transport the right hom-carrier along $c\hsame\emp$ so that it has empty
target. Empty-target exactness applied to that right carrier identifies $b$
and $g$ with $\emp$. Transporting the left carrier along $b\hsame\emp$ lets
empty-target exactness identify $a$ and $f$ with $\emp$. The displayed
continuation then has both inputs empty, so its result is $\emp$.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.CategoryHomCarrier\_comp\_empty\_target\_boundary}
