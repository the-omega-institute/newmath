\begin{theorem}[Continuation morphism composite endpoint cycle tail empty iff]
\label{thm:continuation-morphism-comp-endpoint-cycle-tail-empty-iff}
For histories $a,b,c$, let $left:\mathsf{ContinuationMorphism}(a,b)$ and
$right:\mathsf{ContinuationMorphism}(b,c)$ be continuation morphisms. Then the
outer endpoints close exactly when the canonical composite displayed tail is
empty:
$$
\hsame(a,c)\Longleftrightarrow
\hsame(\mathsf{tail}(left\circ right),\emp).
$$
\end{theorem}
\begin{proof}
The forward implication is the endpoint-cycle empty-tail theorem. Conversely,
if the composite tail is empty, transport the stored continuation relation
$\Cont(a,\mathsf{tail}(left\circ right),c)$ across that empty-tail comparison
to obtain $\Cont(a,\emp,c)$. Determinacy against the right-unit continuation
$\Cont(a,\emp,a)$ identifies $c$ with $a$, hence $a\hsame c$ by symmetry.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.ContinuationMorphism\_comp\_endpoint\_cycle\_tail\_empty\_iff}

\begin{theorem}[Continuation morphism composite source empty tail target iff]
\label{thm:continuation-morphism-comp-source-empty-tail-target-iff}
For histories $a,b,c$, let $left:\mathsf{ContinuationMorphism}(a,b)$ and
$right:\mathsf{ContinuationMorphism}(b,c)$ be continuation morphisms. The
source is empty exactly when the canonical composite displayed tail is the
target:
$$
\hsame(a,\emp)\Longleftrightarrow
\hsame(\mathsf{tail}(left\circ right),c).
$$
\end{theorem}
\begin{proof}
If $a\hsame\emp$, the stored composite continuation ledger has empty source.
The left-unit result law therefore identifies the target with the composite
tail, and symmetry gives the displayed tail-target comparison.

Conversely, transport the stored composite continuation ledger along the
tail-target comparison to obtain $\ContR(a,c,c)$. Comparing this ledger with
the canonical left-unit continuation $\ContR(\emp,c,c)$ by right cancellation
identifies $a$ with $\emp$.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.ContinuationMorphism\_comp\_source\_empty\_tail\_target\_iff}
