\begin{theorem}[Category hom carrier unary-context middle-object determinacy]
\label{thm:category-hom-carrier-unary-context-comp-middle-object-deterministic}
For the unary-continuation category, suppose the two displayed hom-carrier
ledgers have a shared left prefix $p$ and a shared right suffix $q$:
$$
\mathsf{CategoryHomCarrier}((p\cdot a)\cdot q,(p\cdot b)\cdot q,f)
$$
and
$$
\mathsf{CategoryHomCarrier}((p\cdot b')\cdot q,(p\cdot c)\cdot q,g).
$$
If $\ContR(f,g,fg)$ and $fg$ is displayed as a hom-carrier from
$(p\cdot a)\cdot q$ to $(p\cdot c)\cdot q$, then $\HSame(b,b')$.
\end{theorem}
\begin{proof}
Apply the unary suffix equivalence of
\autoref{thm:category-hom-carrier-unary-suffix-iff} to remove the shared
right suffix from the two input ledgers and the displayed composite ledger.
Each remaining ledger has the same left prefix $p$ on its source and target
endpoints; unary append factorization gives the base endpoint carriers, and
prefix cancellation removes $p$ from the continuation evidence. The base
hom-carriers now satisfy the hypotheses of
\autoref{thm:category-hom-carrier-comp-middle-object-deterministic}, which
identifies the middle objects by $\HSame(b,b')$.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.CategoryHomCarrier\_unary\_context\_comp\_middle\_object\_deterministic}

\begin{theorem}[Category hom carrier parallel-factor composite middle determinacy]
\label{thm:category-hom-carrier-parallel-factor-comp-tail-middle-deterministic}
For the unary-continuation category, suppose the two left hom-carriers share
source endpoint $a$ and have history-same morphisms $f$ and $f'$, while the two
right hom-carriers share target endpoint $c$ and have history-same morphisms
$g$ and $g'$:
$$
\mathsf{CategoryHomCarrier}(a,b,f),\quad
\mathsf{CategoryHomCarrier}(b,c,g),\quad
\mathsf{CategoryHomCarrier}(a,b',f'),\quad
\mathsf{CategoryHomCarrier}(b',c,g').
$$
Then the four underlying continuation fields are present, the middle endpoints
are history-same, and the two append composite tails are history-same:
$$
\Cont(a,f,b)\wedge\Cont(b,g,c)\wedge\Cont(a,f',b')\wedge\Cont(b',g',c)
\wedge\HSame(b,b')\wedge\HSame(\mathsf{append}(f,g),\mathsf{append}(f',g')).
$$
\end{theorem}
\begin{proof}
Unpack the four hom-carrier records to read their continuation fields. The
comparison $\HSame(f,f')$ transports the left morphism history, so target
determinacy of the two left hom-carriers identifies $b$ and $b'$. The two
morphism-history comparisons identify the append products
$\mathsf{append}(f,g)$ and $\mathsf{append}(f',g')$.
\end{proof}

\begin{theorem}[Category hom carrier identity square closure]
\label{thm:category-hom-carrier-identity-square-closed}

these two history equalities gives the two carrier witnesses, and transitivity
of \(\HSame\) identifies \(left\) and \(right\).

\begin{theorem}[Category hom carrier two-sided identity law]
\label{thm:category-hom-carrier-two-sided-identity-law}
For the unary-continuation category, if
$\mathsf{CategoryHomCarrier}(a,b,f)$, $\ContR(\emp,f,left)$, and
$\ContR(f,\emp,right)$, then
$$
\mathsf{CategoryHomCarrier}(a,b,left)\wedge
\mathsf{CategoryHomCarrier}(a,b,right)\wedge
\HSame(left,f)\wedge\HSame(right,f)\wedge\HSame(left,right).
$$
\end{theorem}
\begin{proof}
The left composite is classified against the original morphism by the
left-unit result theorem: $\ContR(\emp,f,left)$ and
\texttt{cont\_left\_unit\_result} give $\HSame(left,f)$.

For the right composite, compare $\ContR(f,\emp,right)$ with the canonical
right-unit continuation $\ContR(f,\emp,f)$ supplied by
\texttt{cont\_right\_unit}. Continuation determinism gives
$\HSame(right,f)$.

Transport the displayed $\mathsf{CategoryHomCarrier}(a,b,f)$ witness across
the two history-sameness witnesses. This gives
$\mathsf{CategoryHomCarrier}(a,b,left)$ and
$\mathsf{CategoryHomCarrier}(a,b,right)$.

Finally compose $\HSame(left,f)$ with the symmetry of $\HSame(right,f)$.
Transitivity yields $\HSame(left,right)$, completing the two-sided identity
classification.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.CategoryHomCarrier\_two\_sided\_identity\_law}
