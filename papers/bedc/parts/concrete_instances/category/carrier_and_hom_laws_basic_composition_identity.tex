\begin{theorem}[Category hom carrier right-append fixed identity]
\label{thm:category-hom-carrier-right-append-fixed-identity}
For the unary-continuation category, if $\mathsf{CategoryHomCarrier}(a,b,f)$,
then for every history $g:\mathsf{BHist}$,
$$
\HSame(\mathsf{append}(f,g),g)
\Longleftrightarrow
\HSame(f,\emp)\wedge\HSame(a,b).
$$
\end{theorem}
\begin{proof}
In the forward direction, the displayed fixed append equality is a continuation
witness $\ContR(f,g,g)$, so left-unit uniqueness forces $\HSame(f,\emp)$.
Transporting the hom-carrier continuation relation along this empty-tail
equality gives $\ContR(a,\emp,b)$, and comparison with the canonical right-unit
continuation identifies $a$ and $b$. Conversely, if $f\hsame\emp$, the equality
reduces to the kernel empty-left append law.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.CategoryHomCarrier\_right\_append\_fixed\_identity\_iff}

\begin{theorem}[Category hom carrier two-sided identity law]
\label{thm:category-hom-carrier-two-sided-identity-law}
For the unary-continuation category, if
$\mathsf{CategoryHomCarrier}(a,b,f)$, $\ContR(\emp,f,left)$, and
$\ContR(f,\emp,right)$, then
$$
\mathsf{CategoryHomCarrier}(a,b,left)\wedge
\mathsf{CategoryHomCarrier}(a,b,right)\wedge
\HSame(left,f)\wedge\HSame(right,f)\wedge\HSame(left,right).
$$
\end{theorem}
\begin{proof}
The left composite is classified against the original morphism by the
left-unit result theorem: $\ContR(\emp,f,left)$ and
\texttt{cont\_left\_unit\_result} give $\HSame(left,f)$.

For the right composite, compare $\ContR(f,\emp,right)$ with the canonical
right-unit continuation $\ContR(f,\emp,f)$ supplied by
\texttt{cont\_right\_unit}. Continuation determinism gives
$\HSame(right,f)$.

Transport the displayed $\mathsf{CategoryHomCarrier}(a,b,f)$ witness across
the two history-sameness witnesses. This gives
$\mathsf{CategoryHomCarrier}(a,b,left)$ and
$\mathsf{CategoryHomCarrier}(a,b,right)$.

Finally compose $\HSame(left,f)$ with the symmetry of $\HSame(right,f)$.
Transitivity yields $\HSame(left,right)$, completing the two-sided identity
classification.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.CategoryHomCarrier\_two\_sided\_identity\_law}

\begin{theorem}[Category hom carrier left identity carrier result]
\label{thm:category-hom-carrier-left-identity-carrier-result}
For the unary-continuation category, if
$\mathsf{CategoryHomCarrier}(a,a,id)$,
$\mathsf{CategoryHomCarrier}(a,b,f)$, and $\ContR(id,f,left)$, then
$$
\mathsf{CategoryHomCarrier}(a,b,left)\wedge\HSame(left,f).

\end{theorem}
\begin{proof}
Compare the displayed identity carrier $id$ with the canonical empty identity
carrier at $a$. Morphism determinacy gives $\HSame(id,\emp)$. Rewriting the
displayed continuation by this equality turns $\ContR(id,f,left)$ into an
empty-left continuation, so the kernel left-unit result gives
$\HSame(left,f)$. Transport the original hom-carrier witness for $f$ across
this morphism equality to obtain $\mathsf{CategoryHomCarrier}(a,b,left)$.
\end{proof}

\begin{theorem}[Continuation morphism left-right identity tail result]
\label{thm:continuation-morphism-left-right-identity-tail-result}
For continuation morphisms $leftId:a\to a$, $m:a\to b$, and
$rightId:b\to b$, both identity tails are empty, and composing $m$ with them
on the left and right leaves its tail unchanged up to $\hsame$:
$$
\HSame(leftId.tail,\emp)\wedge
\HSame(rightId.tail,\emp)\wedge
\HSame((leftId\circ m)\circ rightId,\;m.tail).
$$
\end{theorem}
\begin{proof}
Compare each identity morphism with the canonical empty continuation at its
endpoint to obtain the two empty-tail equalities. After destructing the three
continuation-morphism records, the composite tail is the displayed append of
the three tails. The append left-unit and right-unit laws reduce it to
$m.tail$.
\end{proof}

\begin{theorem}[Continuation morphism fixed left tail has right identity]
\label{thm:continuation-morphism-comp-left-tail-fixed-right-identity}
For continuation morphisms $left:a\to b$ and $right:b\to c$, if the composite
append tail of $left$ followed by $right$ is history-same to $left.tail$, then
$right.tail$ is empty and the endpoints $b$ and $c$ are history-same:
$$
\HSame(\mathsf{append}(left.tail,right.tail),left.tail)
\Longrightarrow
\HSame(right.tail,\emp)\wedge\HSame(b,c).
$$
\end{theorem}
\begin{proof}
The displayed fixed-left comparison is a self-continuation of $left.tail$ by
$right.tail$. Right-unit uniqueness for continuation forces $right.tail$ to be
empty. Transporting the right morphism's continuation relation to this empty
tail and comparing it with the canonical right-unit continuation at $b$
identifies $b$ and $c$.
\end{proof}

\begin{theorem}[Category hom carrier composite middle identity result]
\label{thm:category-hom-carrier-comp-middle-identity-result}
For the unary-continuation category, if
$\mathsf{CategoryHomCarrier}(a,b,f)$,
$\mathsf{CategoryHomCarrier}(b,b,id)$,
$\mathsf{CategoryHomCarrier}(b,c,g)$,
$\ContR(f,id,fid)$, $\ContR(fid,g,result)$, and $\ContR(f,g,fg)$, then
$$
\mathsf{CategoryHomCarrier}(a,c,result)\wedge
\mathsf{CategoryHomCarrier}(a,c,fg)\wedge\HSame(result,fg).
$$
\end{theorem}
\begin{proof}
Apply the right identity carrier result to $\ContR(f,id,fid)$, obtaining
$\mathsf{CategoryHomCarrier}(a,b,fid)$ and $\HSame(fid,f)$. Composition
closure with the right hom-carrier gives the displayed carrier for $result$,
while direct composition closure gives the displayed carrier for $fg$.
The two continuation witnesses have history-same left tails and the same
right tail $g$, so continuation result respect for $\hsame$ gives
$\HSame(result,fg)$.
\end{proof}
