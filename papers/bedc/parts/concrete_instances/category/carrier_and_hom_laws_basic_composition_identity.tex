\begin{theorem}[Category hom carrier right-append fixed identity]
\label{thm:category-hom-carrier-right-append-fixed-identity}
For the unary-continuation category, if $\mathsf{CategoryHomCarrier}(a,b,f)$,
then for every history $g:\mathsf{BHist}$,
$$
\HSame(\mathsf{append}(f,g),g)
\Longleftrightarrow
\HSame(f,\emp)\wedge\HSame(a,b).
$$
\end{theorem}
\begin{proof}
In the forward direction, the displayed fixed append equality is a continuation
witness $\ContR(f,g,g)$, so left-unit uniqueness forces $\HSame(f,\emp)$.
Transporting the hom-carrier continuation relation along this empty-tail
equality gives $\ContR(a,\emp,b)$, and comparison with the canonical right-unit
continuation identifies $a$ and $b$. Conversely, if $f\hsame\emp$, the equality
reduces to the kernel empty-left append law.
\end{proof}

\begin{theorem}[Category hom carrier composite result fixed by both factors]
\label{thm:category-hom-carrier-composite-result-fixed-both-identity}
For the unary-continuation category, if
$\mathsf{CategoryHomCarrier}(a,b,f)$,
$\mathsf{CategoryHomCarrier}(b,c,g)$, and $\ContR(f,g,fg)$, then
$$
\HSame(fg,f)\wedge\HSame(fg,g)
\Longleftrightarrow
\HSame(f,\emp)\wedge\HSame(g,\emp)\wedge\HSame(a,b)\wedge\HSame(b,c).
$$
\end{theorem}
\begin{proof}
The forward direction applies the one-sided fixed-composite identity laws to
the two displayed comparisons. Fixing the result against $f$ forces the right
factor tail empty and identifies $b$ with $c$; fixing it against $g$ forces the
left factor tail empty and identifies $a$ with $b$. Conversely, when both
factor tails are empty, the composite continuation reduces to the empty
history, hence it is history-same to each factor tail.
\end{proof}

\begin{theorem}[Category hom carrier two-sided identity law]
\label{thm:category-hom-carrier-two-sided-identity-law}
For the unary-continuation category, if
$\mathsf{CategoryHomCarrier}(a,b,f)$, $\ContR(\emp,f,left)$, and
$\ContR(f,\emp,right)$, then
$$
\mathsf{CategoryHomCarrier}(a,b,left)\wedge
\mathsf{CategoryHomCarrier}(a,b,right)\wedge
\HSame(left,f)\wedge\HSame(right,f)\wedge\HSame(left,right).
$$
\end{theorem}
\begin{proof}
The left composite is classified against the original morphism by the
left-unit result theorem: $\ContR(\emp,f,left)$ and
\texttt{cont\_left\_unit\_result} give $\HSame(left,f)$.

For the right composite, compare $\ContR(f,\emp,right)$ with the canonical
right-unit continuation $\ContR(f,\emp,f)$ supplied by
\texttt{cont\_right\_unit}. Continuation determinism gives
$\HSame(right,f)$.

Transport the displayed $\mathsf{CategoryHomCarrier}(a,b,f)$ witness across
the two history-sameness witnesses. This gives
$\mathsf{CategoryHomCarrier}(a,b,left)$ and
$\mathsf{CategoryHomCarrier}(a,b,right)$.

Finally compose $\HSame(left,f)$ with the symmetry of $\HSame(right,f)$.
Transitivity yields $\HSame(left,right)$, completing the two-sided identity
classification.
\end{proof}
\leanchecked{BEDC.Derived.CategoryUp.CategoryHomCarrier\_two\_sided\_identity\_law}

\label{thm:continuation-morphism-composite-tail-fixed-both-identity}
For continuation morphisms $left:a\to b$ and $right:b\to c$, the closed
composite tail is fixed by both factor tails exactly when both factors are
identity tails and both endpoint steps are history-same:
$$
\HSame((left\circ right).tail,left.tail)\wedge
\HSame((left\circ right).tail,right.tail)
\Longleftrightarrow
\HSame(left.tail,\emp)\wedge\HSame(right.tail,\emp)\wedge
\HSame(a,b)\wedge\HSame(b,c).

\begin{proof}
After destructing the two continuation-morphism records, the closed composite

left tail empty. Transporting the two relation fields across those empty-tail
equalities and comparing with canonical right-unit continuations gives the two
endpoint comparisons. The converse reduces the append tail to the empty
history and hence to each empty factor tail.
\end{proof}
