\chapter{A Concrete Naming Certificate for $\ModuleUp$}
\label{ch:concrete-instances-module-namecert}

The module interface $\ModuleUp$ has certificate obligations over a fixed scalar ring satisfying \autoref{ch:concrete-instances-ring-namecert} together with an additive abelian group from \autoref{ch:concrete-instances-abgroup-namecert} as carrier, equipped with a scalar multiplication $\cdot : R \times M \to M$ satisfying associativity with the ring product, distributivity over additions on both sides, and unit law for the ring identity. It is not a primitive structure; it is admitted only by composing the ring and abelian group certificates with the scalar action fields.

\section{Carrier definition}

\begin{definition}[Module carrier]
\label{def:module-carrier}
Use a scalar ring $R$, an additive abelian group $M$, and scalar multiplication $R\times M\to M$.
\end{definition}

\section{Concrete singleton empty-history module instance}
\label{sec:module-concrete-singleton-instance}

\begin{definition}[Singleton empty-history module instance]
\label{def:singleton-empty-history-module-instance}
Let the scalar carrier and module carrier both be the singleton history source
$A_{\emp}$ of Definition~\ref{def:list-singleton-history-source} at
$a_0=\emp$. Define the shared classifier
$$
\rho_{\emp}(h,k):=
A_{\emp}(h)\land A_{\emp}(k)\land\hsame(h,k).
$$
Use the singleton empty-history ring operations of
Theorem~\ref{thm:singleton-empty-history-ring-laws} for the scalar side, and
the singleton empty-history abelian-group operations of
Theorem~\ref{thm:singleton-empty-history-abgroup-laws} for the additive module
side. Define the scalar action by
$$
\mathsf{smul}(r,m):=\emp.
$$
This instance has traditional target the one-element module over the
one-element ring; BEDC source $\mathsf{BHist}$; scalar and module carrier
$A_{\emp}$; classifier $\rho_{\emp}$; stability obligations given by the
singleton ring laws, singleton abelian-group laws, and scalar-action laws;
ledger constant-action provenance; core theorem
Theorem~\ref{thm:module-singleton-scalar-action-laws}; exactness the singleton
endpoint classification $A_{\emp}(h)$; standard bridge the one-object module;
and status $\mathsf{ProofTarget}$.
\end{definition}

\begin{theorem}[Module singleton classifier empty endpoints]
\label{thm:module-singleton-classifier-empty-endpoints}
For histories $h,k:\mathsf{BHist}$, the concrete module singleton classifier
satisfies
$$
\rho_{\emp}(h,k)
\quad\Longleftrightarrow\quad
h\hsame\emp\land k\hsame\emp.
$$
\end{theorem}
\begin{proof}
The forward direction unpacks the two carrier components of the classifier,
which are exactly the two empty-endpoint comparisons.

Conversely, the two endpoint comparisons give the carrier components, and
composing $h\hsame\emp$ with the symmetry of $k\hsame\emp$ supplies the
$\hsame(h,k)$ component of the classifier.
\end{proof}
\leanchecked{BEDC.Derived.ModuleUp.ModuleSingletonClassifier\_empty\_endpoints\_iff}

\begin{theorem}[Module singleton scalar readback]
\label{thm:module-singleton-smul-classifier-readback-iff}
For histories $r,m,n:\mathsf{BHist}$, if $m$ lies in the singleton carrier
$A_{\emp}$, then the scalar action endpoint is history-same to $\emp$, and
classification after scalar action reads back to the original module endpoint:
$$
A_{\emp}(m)\Longrightarrow
\hsame(\mathsf{smul}(r,m),\emp)\land
\bigl(\rho_{\emp}(\mathsf{smul}(r,m),n)
\Longleftrightarrow
\rho_{\emp}(m,n)\bigr).
$$
\end{theorem}
\begin{proof}
The scalar action is definitionally the empty history. The given carrier
witness identifies $m$ with $\emp$; composing this witness with the classifier
relation supplies the forward readback, and composing its symmetry supplies the
reverse direction.
\end{proof}
\leanchecked{BEDC.Derived.ModuleUp.ModuleSingletonSmul\_classifier\_readback\_iff}

\begin{theorem}[Module singleton scalar action image carrier exactness]
\label{thm:module-singleton-smul-image-carrier-exact}
For the concrete singleton $\ModuleUp$ instance, let
$$
P(h)\;:\Longleftrightarrow\;\exists r,m:\mathsf{BHist},\,
  A_{\emp}(m)\wedge h\hsame\mathsf{ModuleSingletonSmul}(r,m).
$$
Then the scalar-action image pattern is exactly the singleton carrier:
$$
P(h)\quad\Longleftrightarrow\quad A_{\emp}(h).
$$
The ten-field fit is: traditional target the one-element module action image;
BEDC source $\mathsf{BHist}$ with $\mathsf{ModuleSingletonSmul}$; carrier
$A_{\emp}$; classifier $\rho_{\emp}$; stability from
Theorems~\ref{thm:module-singleton-classifier-empty-endpoints} and
\ref{thm:module-singleton-smul-classifier-readback-iff}; ledger scalar-action
witnesses; core theorem this block; exactness the displayed iff; standard
bridge the terminal module; and status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
For the forward direction, unpack a witness
$r,m:\mathsf{BHist}$ with $A_{\emp}(m)$ and
$h\hsame\mathsf{ModuleSingletonSmul}(r,m)$. The scalar action endpoint is
definitionally $\emp$, so this comparison composes to $h\hsame\emp$. By the
singleton carrier definition, that endpoint comparison is precisely
$A_{\emp}(h)$.

For the reverse direction, assume $A_{\emp}(h)$. Choose $r=\emp$ and $m=h$.
The module witness is the given carrier evidence, and
$\mathsf{ModuleSingletonSmul}(r,m)$ is definitionally $\emp$. The same
singleton carrier witness compares $h$ with this empty scalar-action endpoint,
so the chosen pair lies in the image pattern.
\end{proof}

\begin{theorem}[Module singleton scalar action image semantic certificate]
\label{thm:module-singleton-smul-image-semantic-name-cert}
For the concrete singleton $\ModuleUp$ instance, the source carrier $A_{\emp}$,
the scalar-action image pattern
$$
P(h)\;:\Longleftrightarrow\;\exists r,m:\mathsf{BHist},\,
  A_{\emp}(m)\wedge h\hsame\mathsf{ModuleSingletonSmul}(r,m),
$$
the ledger carrier $A_{\emp}$, and the singleton classifier $\rho_{\emp}$ form a
$\mathsf{SemanticNameCert}$.
\end{theorem}
\begin{proof}
The certificate core is the singleton module classifier from
Theorem~\ref{thm:module-singleton-scalar-action-laws}. Its carrier is
$A_{\emp}$, its classifier is $\rho_{\emp}$, and its stability fields give the
reflexive, symmetric, transitive, and carrier-transport clauses required by the
semantic certificate.

For pattern soundness, assume $A_{\emp}(h)$. Choose the scalar and module
witnesses $r=\emp$ and $m=h$. The module witness is the given carrier evidence,
and $\mathsf{ModuleSingletonSmul}(r,m)$ is definitionally the empty history, so
the same singleton evidence supplies $h\hsame\mathsf{ModuleSingletonSmul}(r,m)$.

Ledger soundness is the identity map on the singleton carrier $A_{\emp}$, with
classification read by the same $\rho_{\emp}$ endpoint relation.
\end{proof}

\begin{theorem}[Module singleton scalar action graph exactness]
\label{thm:module-singleton-smul-graph-exact}
For histories $r,m,n:\mathsf{BHist}$, define the concrete scalar-action graph
predicate
$$
G(r,m,n)\;:\Longleftrightarrow\;
A_{\emp}(r)\land A_{\emp}(m)\land
\rho_{\emp}(\mathsf{ModuleSingletonSmul}(r,m),n).
$$
Then this graph is exactly the singleton carrier on scalar input, module input,
and classified output:
$$
G(r,m,n)\quad\Longleftrightarrow\quad
A_{\emp}(r)\land A_{\emp}(m)\land A_{\emp}(n).
$$
The ten-field fit is: traditional target the terminal module action graph;
BEDC source $\mathsf{BHist}$ with $\mathsf{ModuleSingletonSmul}$; carrier
$A_{\emp}$; classifier $\rho_{\emp}$; stability from
Theorems~\ref{thm:module-singleton-classifier-empty-endpoints},
\ref{thm:module-singleton-smul-classifier-readback-iff}, and
\ref{thm:module-singleton-scalar-action-laws}; ledger scalar endpoint, module
endpoint, and classified result endpoint; core theorem this block; exactness
the displayed iff; downstream consumer the $\ModuleUp$ action-row certificate;
and status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Unfold $\mathsf{ModuleSingletonSmul}$ as the constant empty-history action.
The only nontrivial classifier read in the graph is therefore
$\rho_{\emp}(\emp,n)$.

For the forward direction, unpack $G(r,m,n)$. The first two projections are
already the scalar endpoint witness $A_{\emp}(r)$ and the module endpoint
witness $A_{\emp}(m)$. The classifier projection, together with
Theorem~\ref{thm:module-singleton-classifier-empty-endpoints}, reads
$\rho_{\emp}(\emp,n)$ as the two empty endpoint witnesses; the second one is
precisely the singleton carrier witness for $n$.

For the reverse direction, assume
$A_{\emp}(r)\land A_{\emp}(m)\land A_{\emp}(n)$. Reuse the scalar and module
carrier witnesses unchanged. Since
$\mathsf{ModuleSingletonSmul}(r,m)$ is definitionally $\emp$,
Theorem~\ref{thm:module-singleton-classifier-empty-endpoints} and the
singleton carrier witness for $n$ give
$\rho_{\emp}(\mathsf{ModuleSingletonSmul}(r,m),n)$, which repacks the graph
predicate.
\end{proof}

\begin{theorem}[Singleton empty-history module laws]
\label{thm:module-singleton-scalar-action-laws}
The tuple
$$
(A_{\emp},\rho_{\emp},\mathsf{add},0,\mathsf{neg},
\mathsf{mul},1,\mathsf{smul})
$$
satisfies the $\ModuleUp$ schema over the concrete $\mathsf{BHist}$ carrier:
the scalar ring fields, the additive abelian-group fields, scalar-action
closure, scalar-action congruence, scalar associativity, distributivity over
module addition, distributivity over scalar addition, and scalar unit law.
\end{theorem}
\begin{proof}
The scalar ring fields are supplied by
Theorem~\ref{thm:singleton-empty-history-ring-laws}. The additive module
fields are supplied by
Theorem~\ref{thm:singleton-empty-history-abgroup-laws}. These two certificates
fix the shared singleton carrier $A_{\emp}$ and classifier $\rho_{\emp}$.

For scalar-action closure, every endpoint
$\mathsf{smul}(r,m)$ is definitionally $\emp$. The singleton source of
Definition~\ref{def:list-singleton-history-source} gives
$A_{\emp}(\emp)$, hence the result lies in the module carrier.

For scalar-action congruence, suppose
$\rho_{\emp}(r,r')$ and $\rho_{\emp}(m,m')$. Both action endpoints reduce to
$\emp$. The two carrier witnesses are the same singleton witness
$A_{\emp}(\emp)$, and the relation component is reflexivity
$\hsame(\emp,\emp)$. Together these form
$\rho_{\emp}(\mathsf{smul}(r,m),\mathsf{smul}(r',m'))$.

For scalar associativity, both
$\mathsf{smul}(\mathsf{mul}(r,s),m)$ and
$\mathsf{smul}(r,\mathsf{smul}(s,m))$ are definitionally $\emp$. The same
singleton carrier witness and $\hsame$ reflexivity package the required
$\rho_{\emp}$ endpoint relation.

For distributivity over module addition, both
$\mathsf{smul}(r,\mathsf{add}(m,n))$ and
$\mathsf{add}(\mathsf{smul}(r,m),\mathsf{smul}(r,n))$ reduce to $\emp$.
Closure is the singleton carrier witness, and the classifier relation is
$\hsame(\emp,\emp)$.

For distributivity over scalar addition, both
$\mathsf{smul}(\mathsf{add}(r,s),m)$ and
$\mathsf{add}(\mathsf{smul}(r,m),\mathsf{smul}(s,m))$ reduce to $\emp$.
Again the singleton carrier witnesses and reflexivity assemble the
$\rho_{\emp}$ witness.

For the scalar unit law, $\mathsf{smul}(1,m)$ is definitionally $\emp$. The
input witness $A_{\emp}(m)$ identifies $m$ with $\emp$, and
$A_{\emp}(\emp)$ supplies the other endpoint. Hence
$\rho_{\emp}(\mathsf{smul}(1,m),m)$ follows from the carrier witnesses and the
corresponding $\hsame$ endpoint.
\end{proof}

\section{The certificate}

\begin{definition}[Module source specification]
\label{def:module-source-specification}
The source is a certified $\RingUp(R)$ and certified $\AbGroupUp(M)$ sharing a scalar action.
\end{definition}

\begin{definition}[Module pattern specification]
\label{def:module-pattern-specification}
The pattern is module expression formation from vector addition, zero, negation, and scalar multiplication.
\end{definition}

\begin{definition}[Module classifier specification]
\label{def:module-classifier-specification}
The classifier is the module carrier classifier, with scalar action congruent in both scalar and vector inputs.
\end{definition}

\begin{definition}[Module stability certificate]
\label{def:module-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item additive abelian-group laws on $M$
  \item ring laws on $R$
  \item scalar action congruence
  \item scalar associativity with ring multiplication
  \item scalar distributivity over module addition
  \item scalar distributivity over ring addition
  \item scalar unit law
\end{enumerate}
\end{definition}

\begin{definition}[Module ledger policy]
\label{def:module-ledger-policy}
Record scalar and module ledgers for every scalar-action expression.
\end{definition}

\begin{definition}[$\ModuleUp$ certificate obligations]
\label{def:module-certificate-obligations}
The carrier, source, pattern, classifier, stability, and ledger data listed above are the obligations required for the interface name $\ModuleUp$.
\end{definition}

\paragraph{Primitive boundary.}
$\ModuleUp$ is not a BEDC primitive. It is admitted only when those obligations are supplied.
