\begin{theorem}[Bounded $\hsame$ product pair-representation coherence]
\label{thm:prod-pair-rep-bounded-hsame-coherence}
Let $L,R:\mathsf{BHist}\to\mathsf{Prop}$ be source-history carriers. Suppose
there are histories $l_0,r_0:\mathsf{BHist}$ such that
$$
\forall l.\;L(l)\to\hsame(l,l_0)
$$
and
$$
\forall r.\;R(r)\to\hsame(r,r_0).
$$
With both source classifiers specialized to $\hsame$, the product
pair-representation coherence condition holds:
$$
\mathsf{ProdPairRepCoherent}_{L,R,\hsame,\hsame}.
$$
\end{theorem}
\begin{proof}
Take two displayed pair representations of the same product history $h$, with
components $(l,r)$ and $(l',r')$. The pair-representation witnesses include
$L(l)$, $L(l')$, $R(r)$, and $R(r')$. The boundedness hypotheses give
$\hsame(l,l_0)$, $\hsame(l',l_0)$, $\hsame(r,r_0)$, and $\hsame(r',r_0)$.
By symmetry of $\hsame$, turn $\hsame(l',l_0)$ into $\hsame(l_0,l')$ and
$\hsame(r',r_0)$ into $\hsame(r_0,r')$. By transitivity of $\hsame$, compose
$l$ through $l_0$ to obtain $\hsame(l,l')$, and compose $r$ through $r_0$ to
obtain $\hsame(r,r')$. These are exactly the two component classifier
conclusions required by $\mathsf{ProdPairRepCoherent}_{L,R,\hsame,\hsame}$.
\end{proof}

\begin{theorem}[Product pair-representation coherence from classifier-bounded sources]
\label{thm:prod-pair-rep-classifier-bounded-coherence}
The ten-field fit is: traditional target ordered-product component readback
through bounded source classifiers; BEDC source $\mathsf{BHist}$ with $\Cont$;
carriers $L$ and $R$; classifiers $\sim_L$ and $\sim_R$; stability obligation
pair-representation coherence; ledger displayed pair representation; core
theorem classifier-bounded coherence; exactness component comparison through
the source classifiers; standard bridge source certificates plus center
histories; status $\mathsf{ProofTarget}$.

Let $L,R:\mathsf{BHist}\to\mathsf{Prop}$ be source-history carriers, let
$\sim_L,\sim_R:\mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$ be source
classifiers, and assume
$\mathsf{cert}_L:\NameCert(L,\sim_L)$ and
$\mathsf{cert}_R:\NameCert(R,\sim_R)$. Suppose there are histories
$l_0,r_0:\mathsf{BHist}$ such that
$$
\forall l.\;L(l)\to l\sim_L l_0
$$
and
$$
\forall r.\;R(r)\to r\sim_R r_0.
$$
Then the product pair-representation coherence condition holds:
$$
\mathsf{ProdPairRepCoherent}_{L,R,\sim_L,\sim_R}.
$$
\end{theorem}
\begin{proof}
Take two displayed pair representations of the same product history $h$, with
components $(l,r)$ and $(l',r')$. The displayed witnesses contain the source
facts $L(l)$, $L(l')$, $R(r)$, and $R(r')$. The boundedness assumptions give
$l\sim_L l_0$, $l'\sim_L l_0$, $r\sim_R r_0$, and $r'\sim_R r_0$. Apply the
symmetry fields of $\mathsf{cert}_L$ and $\mathsf{cert}_R$ to turn the second
left and right bounds into $l_0\sim_L l'$ and $r_0\sim_R r'$. Apply the
transitivity fields of the two certificates to compose through the center
histories, obtaining $l\sim_L l'$ and $r\sim_R r'$. These two component
classifier witnesses are exactly the conclusion required by
$\mathsf{ProdPairRepCoherent}_{L,R,\sim_L,\sim_R}$.
\end{proof}

\begin{definition}[Product displayed pair ledger]\label{def:prod-displayed-pair-ledger}
Let $L,R:\mathsf{BHist}\to\mathsf{Prop}$ be source-history carriers. The product

$$
\mathsf{ProdDisplayedPairLedger}_{L,R}(h)
\Longleftrightarrow
\exists l,r:\mathsf{BHist}.\;\mathsf{ProdPairRep}_{L,R}(h,l,r).

\end{definition}

\begin{theorem}[Product displayed pair ledger soundness]\label{thm:prod-displayed-pair-ledger-soundness}
Let $L,R:\mathsf{BHist}\to\mathsf{Prop}$ be source-history carriers. For every
history $h:\mathsf{BHist}$,
$$
\mathsf{ProdHistoryCarrier}(L,R,h)
\to
\mathsf{ProdDisplayedPairLedger}_{L,R}(h).
$$
\end{theorem}
\begin{proof}
Assume $\mathsf{ProdHistoryCarrier}(L,R,h)$. Apply the forward direction of
Theorem~\ref{thm:prod-pair-rep-coverage} to this carrier witness.

The theorem returns histories $l,r:\mathsf{BHist}$ with
$\mathsf{ProdPairRep}_{L,R}(h,l,r)$. By
Definition~\ref{def:prod-displayed-pair-ledger}, this existential witness is
exactly $\mathsf{ProdDisplayedPairLedger}_{L,R}(h)$.
\end{proof}

\begin{theorem}[Product displayed pair ledger exactness and transport]
\label{thm:prod-displayed-pair-ledger-exact-and-transport}
Let $L,R:\mathsf{BHist}\to\mathsf{Prop}$ be source-history carriers. Write
$D(h):\Longleftrightarrow\mathsf{ProdDisplayedPairLedger}_{L,R}(h)$. For all
histories $h,k:\mathsf{BHist}$,
$$
D(h)\Longleftrightarrow\mathsf{ProdHistoryCarrier}(L,R,h)
$$
and
$$
D(h)\land\hsame(h,k)\to D(k).
$$
\end{theorem}
\begin{proof}
Unfold $D$ and Definition~\ref{def:prod-displayed-pair-ledger}. For the
forward direction of the equivalence, a displayed pair ledger witness is a pair
of histories $l,r$ with $\mathsf{ProdPairRep}_{L,R}(h,l,r)$. Apply the reverse
direction of Theorem~\ref{thm:prod-pair-rep-coverage} to obtain
$\mathsf{ProdHistoryCarrier}(L,R,h)$.

For the reverse direction, start with
$\mathsf{ProdHistoryCarrier}(L,R,h)$. Apply the forward direction of
Theorem~\ref{thm:prod-pair-rep-coverage} to obtain histories $l,r$ with
$\mathsf{ProdPairRep}_{L,R}(h,l,r)$. Repack this displayed representation by
Definition~\ref{def:prod-displayed-pair-ledger}; this is $D(h)$.

For endpoint transport, assume $D(h)$ and $\hsame(h,k)$. Unfold $D$ to obtain a
displayed pair representation $\mathsf{ProdPairRep}_{L,R}(h,l,r)$. Apply
Theorem~\ref{thm:prod-pair-rep-hsame-transport} along $\hsame(h,k)$, keeping the
same displayed component histories $l,r$, to obtain
$\mathsf{ProdPairRep}_{L,R}(k,l,r)$. Repack this representation by
Definition~\ref{def:prod-displayed-pair-ledger}; the result is $D(k)$.
\end{proof}

$\mathsf{ProdPairRepCoherent}_{L,R,\sim_L,\sim_R}$, the chosen representation,

$(l\sim_L l')\wedge(r\sim_R r')$.

\begin{definition}[Product displayed pair ledger chain]
\label{def:prod-history-ledger-chain}

\frac{\mathsf{ProdHistoryLedgerPolicy}(L,R,\rho,z)}

A displayed ledger followed by a chain through an intermediate history gives a

$$
\frac{\mathsf{ProdHistoryLedgerPolicy}(L,R,\rho,v)\quad

\end{definition}

\begin{theorem}[Product ledger chain envelope closure]
\label{thm:prod-ledger-chain-envelope-closure}
Let $L,R:\mathsf{BHist}\to\mathsf{Prop}$ be source-history carriers and let
$\rho,z:\mathsf{BHist}$. If
$\mathsf{ProdHistoryLedgerChain}_{L,R}(\rho,z)$ holds, then
$\mathsf{ProdHistoryCarrier}(L,R,z)$ and
$\mathsf{ProdHistoryClassifier}(L,R,\rho,z)$ hold. Moreover, for every
$w:\mathsf{BHist}$ both history-envelope endpoint-replacement clauses hold:
$$
\mathsf{ProdHistoryClassifier}(L,R,\rho,w)
\Longleftrightarrow
\mathsf{ProdHistoryClassifier}(L,R,z,w)
$$
and
$$
\mathsf{ProdHistoryClassifier}(L,R,w,\rho)
\Longleftrightarrow
\mathsf{ProdHistoryClassifier}(L,R,w,z).
$$
\end{theorem}
\begin{proof}
Induct on the derivation of
$\mathsf{ProdHistoryLedgerChain}_{L,R}(\rho,z)$.

In the one-ledger constructor, Theorem~\ref{thm:prod-ledger-visible-carrier}
gives $\mathsf{ProdHistoryCarrier}(L,R,z)$ from the displayed ledger, and
Theorem~\ref{thm:prod-ledger-raw-visible-classifier} gives
$\mathsf{ProdHistoryClassifier}(L,R,\rho,z)$. The left endpoint-replacement
clause is Theorem~\ref{thm:prod-ledger-classifier-endpoint-equivalence}. The
right endpoint-replacement clause follows by composing through the same
raw-visible classifier with symmetry and transitivity of the product history
classifier.

In the cons constructor, write the first displayed ledger as
$\mathsf{ProdHistoryLedgerPolicy}(L,R,\rho,v)$ and the remaining chain as
$\mathsf{ProdHistoryLedgerChain}_{L,R}(v,z)$. The induction hypothesis supplies
$\mathsf{ProdHistoryCarrier}(L,R,z)$,
$\mathsf{ProdHistoryClassifier}(L,R,v,z)$, and the two endpoint equivalences
from $v$ to $z$.

The first ledger supplies
$\mathsf{ProdHistoryClassifier}(L,R,\rho,v)$ by
Theorem~\ref{thm:prod-ledger-raw-visible-classifier}. Compose it with the
induction classifier by Theorem~\ref{thm:prod-history-classifier-transitivity}
to obtain $\mathsf{ProdHistoryClassifier}(L,R,\rho,z)$. Compose the one-ledger
endpoint equivalence from $\rho$ to $v$ with the induction endpoint
equivalence from $v$ to $z$ for the left clause, and use the same endpoint
transport through symmetry for the right clause.
\end{proof}

\begin{theorem}[Product componentwise public certificate package]
\label{thm:prod-componentwise-public-certificate-package}

package; BEDC source $\mathsf{BHist}$ with $\Cont$ and $\hsame$; carrier
$\mathsf{ProdHistoryCarrier}$ through $\mathsf{ProdPairRep}$; classifier
$\mathsf{ProdComponentHistoryClassifier}$; stability inputs
$\mathsf{ProdPairRepCoherent}$ and source $\hsame$-soundness; ledger
$\mathsf{ProdDisplayedPairLedger}$; core theorem the semantic certificate,
displayed pair ledger endpoints, and envelope descent together; exactness
componentwise endpoint ledgers plus descent to
$\mathsf{ProdHistoryClassifier}$; standard bridge ordered-pair public readback;
status $\mathsf{ProofTarget}$. This theorem is not a variant of any one
preceding theorem, because it composes certificate existence, ledger endpoints,

$\sim_L,\sim_R:\mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$ be their source

classifiers are $\hsame$-sound:
$$
\forall x\,y.\;x\sim_L y\to\hsame(x,y)
\qquad\hbox{and}\qquad
\forall x\,y.\;x\sim_R y\to\hsame(x,y).
$$
Then the following three facts hold:
\begin{enumerate}
\item
$$
\mathsf{SemanticNameCert}(\mathsf{ProdHistoryCarrier}(L,R),
\mathsf{ProdHistoryCarrier}(L,R),
\mathsf{ProdDisplayedPairLedger}_{L,R},
\mathsf{ProdComponentHistoryClassifier}_{L,R,\sim_L,\sim_R}).
$$
\item For all histories $h,k:\mathsf{BHist}$,
$$
\mathsf{ProdComponentHistoryClassifier}_{L,R,\sim_L,\sim_R}(h,k)
\to
\mathsf{ProdDisplayedPairLedger}_{L,R}(h)
\land
\mathsf{ProdDisplayedPairLedger}_{L,R}(k).
$$
\item For all histories $h,k:\mathsf{BHist}$,
$$
\mathsf{ProdComponentHistoryClassifier}_{L,R,\sim_L,\sim_R}(h,k)
\to
\mathsf{ProdHistoryClassifier}(L,R,h,k).
$$
\end{enumerate}
\end{theorem}
\begin{proof}
For the semantic certificate clause, apply
Theorem~\ref{thm:prod-component-history-semantic-certificate-target} to
$\mathsf{cert}_L$, $\mathsf{cert}_R$, and the
