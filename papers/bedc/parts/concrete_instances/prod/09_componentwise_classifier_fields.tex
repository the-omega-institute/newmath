\begin{theorem}[Product pair-representation coherence obstruction from separated presentations]
\label{thm:prod-pair-rep-coherence-obstruction-separated-presentations}
The ten-field fit is: traditional target ordered-pair presentation comparison;
BEDC source $\mathsf{BHist}$ with $\Cont$; carrier $\mathsf{ProdPairRep}$;
classifier pair $\sim_L,\sim_R$; stability obligation coherence; ledger
displayed pair boundary; core theorem separated-presentation obstruction;
exactness as failure of component comparison; standard bridge ordered-pair
readback; status $\mathsf{ProofTarget}$.

Let $L,R:\mathsf{BHist}\to\mathsf{Prop}$ be source-history carriers, let
$\sim_L,\sim_R:\mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$ be source
classifiers, and let $h,l,r,l',r':\mathsf{BHist}$. Suppose
$\mathsf{ProdPairRep}_{L,R}(h,l,r)$ and
$\mathsf{ProdPairRep}_{L,R}(h,l',r')$ hold. If either
$\neg(l\sim_L l')$ or $\neg(r\sim_R r')$ holds, then
$$
\neg\mathsf{ProdPairRepCoherent}_{L,R,\sim_L,\sim_R}
$$
holds.
\end{theorem}
\begin{proof}
Assume a coherence witness. Apply it to the two displayed representations of
the same product history $h$. The coherence witness returns both component
comparisons, namely $l\sim_L l'$ and $r\sim_R r'$. The separated component
clause supplies either the negation of the left comparison or the negation of
the right comparison, contradicting the corresponding returned component
witness.
\end{proof}

\begin{theorem}[Product componentwise history classifier carrier reflexivity from source certificates]
\label{thm:prod-component-history-classifier-carrier-reflexivity-from-source-certificates}
The ten-field fit is: traditional target ordered-pair classifier reflexivity;
BEDC source $\mathsf{BHist}$ with $\Cont$ and displayed pair representations;
carrier $\mathsf{ProdHistoryCarrier}$; classifier
$\mathsf{ProdComponentHistoryClassifier}$; stability obligation
carrier-relative reflexivity from the two source certificates; ledger one
displayed product pair representation; exactness component reflexivity at the
same endpoints; standard bridge diagonal ordered-pair comparison; status
$\mathsf{ProofTarget}$.

Let $L,R:\mathsf{BHist}\to\mathsf{Prop}$ be source-history carriers, let
$\sim_L,\sim_R:\mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$ be source
classifiers, and assume $\mathsf{cert}_L:\NameCert(L,\sim_L)$ and
$\mathsf{cert}_R:\NameCert(R,\sim_R)$. Then for every $h:\mathsf{BHist}$,
$$
\mathsf{ProdHistoryCarrier}(L,R,h)\to
\mathsf{ProdComponentHistoryClassifier}_{L,R,\sim_L,\sim_R}(h,h).
$$
\end{theorem}
\begin{proof}
Unpack the product carrier witness into displayed source histories $l$ and
$r$, source carrier witnesses for $L(l)$ and $R(r)$, and the corresponding
pair representation of $h$. Use that same pair representation for both
endpoints of the componentwise classifier. The source certificates provide
$l\sim_L l$ and $r\sim_R r$ through their reflexivity fields, and these two
component witnesses rebuild
$\mathsf{ProdComponentHistoryClassifier}_{L,R,\sim_L,\sim_R}(h,h)$.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdComponentHistoryClassifier\_carrier\_reflexivity\_from\_source\_certificates}

\begin{theorem}[Product componentwise history classifier symmetry from source certificates]
\label{thm:prod-component-history-classifier-symmetry-from-source-certificates}
The ten-field fit is: traditional target ordered-pair classifier symmetry;
BEDC source $\mathsf{BHist}$ with $\Cont$ and displayed pair representations;
carrier $\mathsf{ProdHistoryCarrier}$; classifier
$\mathsf{ProdComponentHistoryClassifier}$; stability obligation
source-certified symmetry for the two component classifiers; ledger displayed
pair representations at both endpoints; exactness reversed component
comparison; standard bridge symmetric ordered-pair comparison; status
$\mathsf{ProofTarget}$.

Let $L,R:\mathsf{BHist}\to\mathsf{Prop}$ be source-history carriers, let
$\sim_L,\sim_R:\mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$ be source
classifiers, and assume $\mathsf{cert}_L:\NameCert(L,\sim_L)$ and
$\mathsf{cert}_R:\NameCert(R,\sim_R)$. Then for all $h,k:\mathsf{BHist}$,
$$
\mathsf{ProdComponentHistoryClassifier}_{L,R,\sim_L,\sim_R}(h,k)\to
\mathsf{ProdComponentHistoryClassifier}_{L,R,\sim_L,\sim_R}(k,h).
$$
\end{theorem}
\begin{proof}
Unpack the componentwise classifier witness into displayed pair
representations of $h$ and $k$ and component classifier witnesses
$l_h\sim_L l_k$ and $r_h\sim_R r_k$. Apply
$\mathsf{cert}_L.\mathsf{equiv\_symm}$ and
$\mathsf{cert}_R.\mathsf{equiv\_symm}$ to reverse the two component
comparisons. Swap the endpoint pair representations and rebuild the
componentwise classifier from the reversed left and right witnesses.
\end{proof}

\begin{theorem}[Product componentwise classifier self-readback exactness]
\label{thm:prod-component-history-classifier-self-readback-exactness}
The ten-field fit is: traditional target diagonal ordered-pair readback; BEDC
source $\mathsf{BHist}$ with $\Cont$ and displayed pair representations;
carrier $\mathsf{ProdHistoryCarrier}$; classifier
$\mathsf{ProdComponentHistoryClassifier}$; stability obligation endpoint-carrier
exposure together with carrier-relative reflexivity from the two source
certificates; ledger one visible history used at both endpoints; core theorem
self-readback exactness; exactness diagonal classifier truth if and only if
product carrier membership; standard bridge ordered-pair self comparison;
status $\mathsf{ProofTarget}$.

Let $L,R:\mathsf{BHist}\to\mathsf{Prop}$ be source-history carriers, let
$\sim_L,\sim_R:\mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$ be source
classifiers, and assume $\mathsf{cert}_L:\NameCert(L,\sim_L)$ and
$\mathsf{cert}_R:\NameCert(R,\sim_R)$. Then every $h:\mathsf{BHist}$
satisfies
$$
\mathsf{ProdComponentHistoryClassifier}_{L,R,\sim_L,\sim_R}(h,h)
\Longleftrightarrow
\mathsf{ProdHistoryCarrier}(L,R,h).
$$
\end{theorem}
\begin{proof}
For the forward direction, assume
$\mathsf{ProdComponentHistoryClassifier}_{L,R,\sim_L,\sim_R}(h,h)$. Apply
Theorem~\ref{thm:prod-component-history-classifier-endpoint-carriers} to this
diagonal classifier witness. It returns product-carrier witnesses for both
endpoints, and either projection gives $\mathsf{ProdHistoryCarrier}(L,R,h)$.

For the reverse direction, assume $\mathsf{ProdHistoryCarrier}(L,R,h)$. Apply
Theorem~\ref{thm:prod-component-history-classifier-carrier-reflexivity-from-source-certificates}
with the two source certificates to the carrier witness. The result is exactly
$\mathsf{ProdComponentHistoryClassifier}_{L,R,\sim_L,\sim_R}(h,h)$.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdComponentHistoryClassifier\_self\_readback\_iff}

\begin{theorem}[Product componentwise classifier is independent of displayed pair representatives]
\label{thm:prod-component-history-classifier-representative-independence}
The ten-field fit is: traditional target ordered-product componentwise readback
well-definedness; BEDC source $\mathsf{BHist}$ with $\Cont$ and displayed
$\mathsf{ProdPairRep}$ witnesses; carrier $\mathsf{ProdHistoryCarrier}$;
classifier $\mathsf{ProdComponentHistoryClassifier}$; stability obligation
representative independence under $\mathsf{ProdPairRepCoherent}$ and source
$\NameCert$ symmetry and transitivity; ledger displayed pair representations at
both endpoints; core theorem this law; exactness independence from chosen
coherent representatives; standard bridge ordered-pair readback; status
$\mathsf{ProofTarget}$.

Let $L,R:\mathsf{BHist}\to\mathsf{Prop}$ be source-history carriers, let
$\sim_L,\sim_R:\mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$ be source
classifiers, and assume $\mathsf{cert}_L:\NameCert(L,\sim_L)$,
$\mathsf{cert}_R:\NameCert(R,\sim_R)$, and
$\mathsf{ProdPairRepCoherent}_{L,R,\sim_L,\sim_R}$. For all
$h,k,l,r,l',r',u,v,u',v':\mathsf{BHist}$, if
$\mathsf{ProdPairRep}_{L,R}(h,l,r)$,
$\mathsf{ProdPairRep}_{L,R}(h,l',r')$,
$\mathsf{ProdPairRep}_{L,R}(k,u,v)$, and
$\mathsf{ProdPairRep}_{L,R}(k,u',v')$ hold, then
$$
(l\sim_L u)\wedge(r\sim_R v)\to
(l'\sim_L u')\wedge(r'\sim_R v').
$$
\end{theorem}
\begin{proof}
Apply $\mathsf{ProdPairRepCoherent}_{L,R,\sim_L,\sim_R}$ at the source endpoint
$h$ to compare the displayed pairs $(l',r')$ and $(l,r)$. This gives
$l'\sim_L l$ and $r'\sim_R r$. Apply the same coherence law at the target
endpoint $k$ to compare $(u,v)$ with $(u',v')$, giving $u\sim_L u'$ and
$v\sim_R v'$. The supplied component comparison gives $l\sim_L u$ and
$r\sim_R v$. Two uses of $\mathsf{cert}_L.\mathsf{equiv\_trans}$ compose
$l'\sim_L l\sim_L u\sim_L u'$, and two uses of
$\mathsf{cert}_R.\mathsf{equiv\_trans}$ compose
$r'\sim_R r\sim_R v\sim_R v'$. The resulting left and right witnesses form the
displayed pair of component comparisons.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdComponentHistoryClassifier\_representative\_independence}

\begin{theorem}[Product componentwise history classifier core fields]
\label{thm:prod-component-history-classifier-certificate-target}
Let $L,R:\mathsf{BHist}\to\mathsf{Prop}$ be source-history carriers, let
$\sim_L,\sim_R:\mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$ be their source
classifiers, and assume $\mathsf{cert}_L:\NameCert(L,\sim_L)$,
$\mathsf{cert}_R:\NameCert(R,\sim_R)$, and
$\mathsf{ProdPairRepCoherent}_{L,R,\sim_L,\sim_R}$. Write
$C(h,k)$ for
$\mathsf{ProdComponentHistoryClassifier}_{L,R,\sim_L,\sim_R}(h,k)$. Then its
componentwise endpoint exposure is:
$$
\forall h\,k.\;C(h,k)\to\mathsf{ProdHistoryCarrier}(L,R,h)\land
\mathsf{ProdHistoryCarrier}(L,R,k).
$$
Its five core $\NameCert$ fields for $\mathsf{ProdHistoryCarrier}(L,R)$
decompose as follows:
\begin{enumerate}
  \item carrier inhabitation:
  $$
  \exists h.\;\mathsf{ProdHistoryCarrier}(L,R,h);
  $$
  \item carrier-relative reflexivity:
  $$
  \forall h.\;\mathsf{ProdHistoryCarrier}(L,R,h)\to C(h,h);
  $$
  \item source-certified symmetry:
  $$
  \forall h\,k.\;C(h,k)\to C(k,h);
  $$
  \item source-certified transitivity:
  $$
  \forall h\,k\,u.\;C(h,k)\to C(k,u)\to C(h,u);
  $$
  \item projection-derived NameCert-shaped classifier-induced carrier
  transport:
  $$
  \forall h\,k.\;C(h,k)\to\mathsf{ProdHistoryCarrier}(L,R,h)\to
  \mathsf{ProdHistoryCarrier}(L,R,k).
  $$
\end{enumerate}
These five fields are the core $\NameCert$ data used by
Theorem~\ref{thm:prod-component-history-semantic-certificate-target}.
\end{theorem}
\begin{proof}
For carrier inhabitation, $\mathsf{cert}_L.\mathsf{carrier\_inhabited}$ and
$\mathsf{cert}_R.\mathsf{carrier\_inhabited}$ supply histories $l$ and $r$
with $L(l)$ and $R(r)$. Applying
Theorem~\ref{thm:prod-history-carrier-append-intro} to those two source
carrier witnesses gives
$\mathsf{ProdHistoryCarrier}(L,R,\mathsf{append}(l,r))$, so the product
carrier is inhabited. For carrier-relative reflexivity, unpack a
$\mathsf{ProdHistoryCarrier}(L,R,h)$ witness and apply
Theorem~\ref{thm:prod-component-history-classifier-carrier-reflexivity-from-source-certificates}.
For symmetry, apply
Theorem~\ref{thm:prod-component-history-classifier-symmetry-from-source-certificates}
to the componentwise classifier witness. For transitivity, unpack
$\mathsf{C}(h,k)$ and
$\mathsf{C}(k,u)$ into pair representations $(l_h,r_h)$, $(l_k,r_k)$,
$(l'_k,r'_k)$, and $(l_u,r_u)$. Pair-representation coherence at the middle
history $k$ gives both $l_k\sim_L l'_k$ and $r_k\sim_R r'_k$. The two source
$\NameCert.\mathsf{equiv\_trans}$ fields then compose the left chain
$l_h\sim_L l_k\sim_L l'_k\sim_L l_u$ and the right chain
$r_h\sim_R r_k\sim_R r'_k\sim_R r_u$ separately. For endpoint exposure, apply
Theorem~\ref{thm:prod-component-history-classifier-endpoint-carriers} directly
to the classifier witness $C(h,k)$. For carrier transport, apply that endpoint
exposure to $C(h,k)$, take the right projection
$\mathsf{ProdHistoryCarrier}(L,R,k)$, and ignore the already supplied left
carrier premise except as the required $\NameCert$ field input. Thus the
displayed endpoint clause and five-field tuple are exactly the componentwise
product core classifier data.
\end{proof}

\begin{theorem}[Product componentwise history classifier transitivity from pair coherence]
\label{thm:prod-component-history-classifier-trans-from-pair-coherence}
The ten-field fit is: traditional target ordered-pair classifier composition;
BEDC source $\mathsf{BHist}$ with $\Cont$ and displayed pair representations;
carrier $\mathsf{ProdHistoryCarrier}$; classifier
$\mathsf{ProdComponentHistoryClassifier}$; stability obligation
source-certified transitivity together with middle pair-representation
coherence; ledger displayed pair representations; exactness component
comparison through the shared middle endpoint; standard bridge ordered-pair
composition; status $\mathsf{ProofTarget}$.

Let $L,R:\mathsf{BHist}\to\mathsf{Prop}$ be source-history carriers, let
$\sim_L,\sim_R:\mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$ be source
classifiers, and assume $\mathsf{cert}_L:\NameCert(L,\sim_L)$,
$\mathsf{cert}_R:\NameCert(R,\sim_R)$, and
$\mathsf{ProdPairRepCoherent}_{L,R,\sim_L,\sim_R}$. Then for all
$h,k,u:\mathsf{BHist}$,
$$
\mathsf{ProdComponentHistoryClassifier}_{L,R,\sim_L,\sim_R}(h,k)\to
\mathsf{ProdComponentHistoryClassifier}_{L,R,\sim_L,\sim_R}(k,u)\to
\mathsf{ProdComponentHistoryClassifier}_{L,R,\sim_L,\sim_R}(h,u).
$$
\end{theorem}
\begin{proof}
Unpack the first componentwise classifier witness into displayed pair
representations of $h$ and $k$, and unpack the second into displayed pair
representations of $k$ and $u$. The two representations of the shared middle
history $k$ are compared by
$\mathsf{ProdPairRepCoherent}_{L,R,\sim_L,\sim_R}$, giving the middle left and
right component comparisons. On the left side,
$\mathsf{cert}_L.\mathsf{equiv\_trans}$ first composes the $h$--$k$
comparison with the middle left comparison, and then composes the result with
the $k$--$u$ comparison. On the right side,
$\mathsf{cert}_R.\mathsf{equiv\_trans}$ performs the same two-step composition
through the middle right comparison. Reuse the original displayed pair
representations of $h$ and $u$ together with these two composed component
comparisons to rebuild
$\mathsf{ProdComponentHistoryClassifier}_{L,R,\sim_L,\sim_R}(h,u)$.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdComponentHistoryClassifier\_trans\_from\_pair\_coherence}
