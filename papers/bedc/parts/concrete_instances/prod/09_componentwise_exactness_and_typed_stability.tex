\begin{theorem}[Product envelope and componentwise hsame classifiers coincide]
\label{thm:prod-envelope-componentwise-hsame-classifiers-coincide}
Let $L,R:\mathsf{BHist}\to\mathsf{Prop}$ be source-history carriers and assume
$\mathsf{ProdPairRepCoherent}_{L,R,\hsame,\hsame}$. For all
$h,k:\mathsf{BHist}$,
$$
\mathsf{ProdHistoryClassifier}(L,R,h,k)
\Longleftrightarrow
\mathsf{ProdComponentHistoryClassifier}_{L,R,\hsame,\hsame}(h,k).
$$
\end{theorem}
\begin{proof}
The forward implication is Theorem~\ref{thm:prod-history-envelope-to-componentwise-hsame}.
For the reverse implication, apply
Theorem~\ref{thm:prod-component-history-classifier-descends-to-envelope}
with both source classifiers specialized to $\hsame$. The two required
$\hsame$-soundness premises are discharged by the identity proof on the given
component links. Thus the envelope classifier and the componentwise $\hsame$
classifier have exactly the same histories under the displayed
pair-representation coherence.
\end{proof}

This exactness statement instantiates the roadmap template as traditional
ordered-pair equality agreement over the BEDC source $\mathsf{BHist}$ with
$\Cont$, carrier $\mathsf{ProdHistoryCarrier}$, classifier agreement between
the envelope and componentwise $\hsame$ classifiers, stability by
$\mathsf{ProdPairRepCoherent}_{L,R,\hsame,\hsame}$, ledger through displayed
pair representations, and status $\mathsf{ProofTarget}$.

\begin{theorem}[Product history classifier displayed component hsame exactness]
\label{thm:prod-history-classifier-displayed-component-hsame-iff}
Let $L,R:\mathsf{BHist}\to\mathsf{Prop}$ be source-history carriers and assume
$\mathsf{ProdPairRepCoherent}_{L,R,\hsame,\hsame}$. If
$L(l)$, $R(r)$, $\Cont(l,r,h)$, $L(l')$, $R(r')$, and
$\Cont(l',r',k)$ hold, then
$$
\mathsf{ProdHistoryClassifier}(L,R,h,k)
\Longleftrightarrow
\hsame(l,l')\wedge\hsame(r,r').
$$
\end{theorem}
\begin{proof}
For the forward direction, the displayed component data are
$\mathsf{ProdPairRep}_{L,R}(h,l,r)$ and
$\mathsf{ProdPairRep}_{L,R}(k,l',r')$. The product history classifier supplies
the endpoint link $\hsame(h,k)$, so pair-representation coherence transports
the endpoint link back to the two component links. For the reverse direction,
the two component links are pushed through the two continuation witnesses by
continuation respect for $\hsame$, producing $\hsame(h,k)$. The displayed
component data rebuild the two endpoint product carriers, and these three
fields are exactly $\mathsf{ProdHistoryClassifier}(L,R,h,k)$.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdHistoryClassifier\_displayed\_component\_hsame\_iff}

\begin{theorem}[Product componentwise history classifier displayed exactness]
\label{thm:prod-component-history-classifier-displayed-exactness}
Let $L,R:\mathsf{BHist}\to\mathsf{Prop}$ be source-history carriers, let
$\sim_L,\sim_R:\mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$ be their source
classifiers. Assume $\mathsf{ProdPairRepCoherent}_{L,R,\sim_L,\sim_R}$ and the
displayed source transitivity laws
$$
\forall a\,b\,c.\;a\sim_L b\to b\sim_L c\to a\sim_L c
\qquad\hbox{and}\qquad
\forall a\,b\,c.\;a\sim_R b\to b\sim_R c\to a\sim_R c.
$$
If
$\mathsf{ProdPairRep}_{L,R}(h,l,r)$ and
$\mathsf{ProdPairRep}_{L,R}(k,l',r')$, then
$$
\mathsf{ProdComponentHistoryClassifier}_{L,R,\sim_L,\sim_R}(h,k)
\Longleftrightarrow
(l\sim_L l')\wedge(r\sim_R r').
$$
\end{theorem}
\begin{proof}
For the forward direction, apply
Theorem~\ref{thm:prod-component-history-classifier-inversion} to the two
displayed pair representations and the assumed componentwise classifier. For
the reverse direction, use the displayed pair representations
$\mathsf{ProdPairRep}_{L,R}(h,l,r)$ and
$\mathsf{ProdPairRep}_{L,R}(k,l',r')$ as the existential witnesses in
Definition~\ref{def:prod-component-history-classifier}; the two displayed
component relations supply the remaining conjuncts.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdComponentHistoryClassifier\_displayed\_exactness}

\begin{definition}[Prod stability certificate]
\label{def:prod-stability-certificate}
The checked stability subcertificate consists of:
\begin{enumerate}
  \item componentwise reflexivity obtained from the two source reflexivity fields
  \item componentwise transitivity obtained from the two source transitivity fields
  \item coordinate exposure: any
  $\mathsf{ProdClassifierSpec}(R_A,R_B,x,y)$ witness yields
  $R_A(x.1,y.1)$ and $R_B(x.2,y.2)$
\end{enumerate}
Relation-level symmetry is supplied by
Theorem~\ref{thm:prod-classifier-symm-from-source-symmetry}; the
source-certificate theorem is obtained from the two $\mathsf{equiv\_symm}$
fields, while the $\hsame$ specialization is
Theorem~\ref{thm:prod-classifier-hsame-symm}.
\end{definition}
\leanchecked{BEDC.Derived.ProdUp.prod\_stability\_certificate\_fields}

\begin{theorem}[Product classifier transitivity at the typed level]
\label{thm:prod-classifier-transitivity}
Let $R_A : A \to A \to \mathsf{Prop}$ and
$R_B : B \to B \to \mathsf{Prop}$ be source classifier relations.  If both
$R_A$ and $R_B$ are transitive, then for product carriers $x$, $y$, and $z$,
$$
  \mathsf{ProdClassifierSpec}(R_A,R_B,x,y)
  \quad\text{and}\quad
  \mathsf{ProdClassifierSpec}(R_A,R_B,y,z)
$$
imply $\mathsf{ProdClassifierSpec}(R_A,R_B,x,z)$.
\end{theorem}
\leanchecked{BEDC.Derived.ProdUp.ProdClassifierSpec\_trans}
\begin{proof}
Invert the product classifier witnesses by their constructor. The first
components compose by transitivity of $R_A$, the second components compose by
transitivity of $R_B$, and the paired typed witness rebuilds the product
classifier.
\end{proof}

\begin{theorem}[Product classifier symmetry from source symmetry]
\label{thm:prod-classifier-symm-from-source-symmetry}
Let $R_A : A \to A \to \mathsf{Prop}$ and
$R_B : B \to B \to \mathsf{Prop}$ be source classifier relations. If
$$
\forall a\,a'.\; R_A(a,a')\to R_A(a',a)
$$
and
$$
\forall b\,b'.\; R_B(b,b')\to R_B(b',b)
$$
hold, then for product carriers $x$ and $y$,
$\mathsf{ProdClassifierSpec}(R_A,R_B,x,y)$ implies
$\mathsf{ProdClassifierSpec}(R_A,R_B,y,x)$.
\end{theorem}
\begin{proof}
Invert the product classifier witness into its two coordinate witnesses.
Apply source symmetry for $R_A$ to the first coordinate and source symmetry for
$R_B$ to the second coordinate, then rebuild the product classifier with the
same constructor in the opposite order.
\end{proof}

\begin{theorem}[Product classifier symmetry from source certificates]
\label{thm:prod-classifier-symm-from-namecert}
Let $L,R:\mathsf{BHist}\to\mathsf{Prop}$ be BEDC history carriers, let
$\sim_L,\sim_R:\mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$ be their
source classifiers, and let
$\mathsf{cert}_L:\NameCert(L,\sim_L)$ and
$\mathsf{cert}_R:\NameCert(R,\sim_R)$ be source name certificates. For product
carriers $x,y:\mathsf{ProdCarrier}(\mathsf{BHist},\mathsf{BHist})$,
$\mathsf{ProdClassifierSpec}(\sim_L,\sim_R,x,y)$ implies
$\mathsf{ProdClassifierSpec}(\sim_L,\sim_R,y,x)$.
\end{theorem}
\begin{proof}
Invert the product classifier witness into its left and right component
classifier proofs. Apply the $\mathsf{equiv\_symm}$ field of
$\mathsf{cert}_L$ to reverse the left component, and the
$\mathsf{equiv\_symm}$ field of $\mathsf{cert}_R$ to reverse the right
component. The product classifier constructor then rebuilds the componentwise
witness in the opposite order.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdClassifierSpec\_symm\_from\_nameCert}

\begin{theorem}[History product classifier is symmetric]
\label{thm:prod-classifier-hsame-symm}
For pairs of finite-kernel histories, if the product classifier relates \(x\) to \(y\) componentwise by \(\hsame\), then it relates \(y\) to \(x\).
\end{theorem}
\begin{proof}
Invert the componentwise product classifier. Each component witness reverses by symmetry of \(\hsame\), and the reversed witnesses rebuild the product classifier in the opposite direction.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdClassifierSpec\_hsame\_symm}
