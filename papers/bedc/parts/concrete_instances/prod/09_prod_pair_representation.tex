\begin{definition}[Product pair representation]
\label{def:prod-pair-rep}
For source-history predicates $L,R:\mathsf{BHist}\to\mathsf{Prop}$, define
$\mathsf{ProdPairRep}_{L,R}(h,l,r)$ to mean that $l$ is a left source
history, $r$ is a right source history, and their continuation presents $h$:
$$
  \mathsf{ProdPairRep}_{L,R}(h,l,r)
  \quad:\Longleftrightarrow\quad
  L(l)\wedge R(r)\wedge \Cont(l,r,h).
$$
\end{definition}
\leandef{BEDC.Derived.ProdUp.ProdPairRep}

\begin{theorem}[Product pair representation coverage]
\label{thm:prod-pair-rep-coverage}
For source-history predicates $L,R:\mathsf{BHist}\to\mathsf{Prop}$ and history $h$,
$$
  \mathsf{ProdHistoryCarrier}(L,R,h)
  \quad\Longleftrightarrow\quad
  \exists l,r:\mathsf{BHist}.\;\mathsf{ProdPairRep}_{L,R}(h,l,r).
$$
\end{theorem}
\begin{proof}
Unfold $\mathsf{ProdHistoryCarrier}$ and $\mathsf{ProdPairRep}$. In the forward direction, the carrier witness exposes component histories $l$ and $r$, source witnesses $L(l)$ and $R(r)$, and the continuation witness $\Cont(l,r,h)$; these are exactly the displayed pair representation witnesses. In the reverse direction, a displayed pair representation supplies the same two component histories, source witnesses, and continuation witness, so it rebuilds the product history carrier witness for $h$.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdPairRep\_coverage}

\begin{theorem}[Product pair representation endpoint transport]
\label{thm:prod-pair-rep-hsame-transport}
For source-history predicates $L,R:\mathsf{BHist}\to\mathsf{Prop}$ and histories
$h,k,l,r$, if $\mathsf{ProdPairRep}_{L,R}(h,l,r)$ and $\hsame(h,k)$, then
$\mathsf{ProdPairRep}_{L,R}(k,l,r)$ holds.
\end{theorem}
\begin{proof}
The per-object fit is: traditional target ordered-pair endpoint stability; BEDC
source $\mathsf{BHist}$ with $\Cont$; carrier $\mathsf{ProdPairRep}$;
classifier endpoint $\hsame$; stability obligation endpoint transport; ledger
displayed pair representation; core theorem this transport law; exactness pair
presentation preserved by endpoint sameness; standard bridge stable ordered-pair
endpoint; status $\mathsf{ProofTarget}$. Unfold $\mathsf{ProdPairRep}$ to keep
$L(l)$ and $R(r)$, and transport the $\Cont(l,r,h)$ result across $\hsame(h,k)$
using continuation result transport. The transported continuation witness
together with the unchanged source witnesses rebuilds
$\mathsf{ProdPairRep}_{L,R}(k,l,r)$.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdPairRep\_hsame\_transport}

\begin{theorem}[Product fixed-pair endpoint determinism]
\label{thm:prod-pair-rep-determinism}
For source-history predicates $L,R:\mathsf{BHist}\to\mathsf{Prop}$ and histories $h,k,l,r$,
if $\mathsf{ProdPairRep}_{L,R}(h,l,r)$ and
$\mathsf{ProdPairRep}_{L,R}(k,l,r)$, then
$\hsame(h,k)$ holds, and consequently
$\mathsf{ProdHistoryClassifier}(L,R,h,k)$ holds.
\end{theorem}
\begin{proof}
Unfold both displayed pair representations. The first supplies $L(l)$, $R(r)$,
and $\Cont(l,r,h)$; the second supplies the same component source witnesses and
$\Cont(l,r,k)$. Applying continuation determinism to the two continuation
witnesses gives $\hsame(h,k)$. The same source witnesses and continuation
witnesses rebuild $\mathsf{ProdHistoryCarrier}(L,R,h)$ and
$\mathsf{ProdHistoryCarrier}(L,R,k)$, and these endpoint carriers together with
the $\hsame$ witness rebuild
$\mathsf{ProdHistoryClassifier}(L,R,h,k)$.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdHistoryClassifier\_fixed\_pair\_determinism}
This statement fixes both component histories; comparison of two different pair presentations of one endpoint is exactly the separate coherence datum in Definition~\ref{def:prod-pair-rep-coherent}.

\begin{theorem}[Product fixed-pair endpoint exactness]
\label{thm:prod-pair-rep-fixed-endpoint-exactness}
For source-history predicates $L,R:\mathsf{BHist}\to\mathsf{Prop}$ and histories
$h,k,l,r$, if $\mathsf{ProdPairRep}_{L,R}(h,l,r)$ holds, then
$$
  \mathsf{ProdPairRep}_{L,R}(k,l,r)\Longleftrightarrow\hsame(h,k).
$$
\end{theorem}
\begin{proof}
The per-object fit is: traditional target ordered-pair endpoint readback; BEDC
source $\mathsf{BHist}$ with $\Cont$; carrier $\mathsf{ProdPairRep}$;
classifier endpoint $\hsame$; stability by endpoint transport and continuation
determinism; ledger fixed displayed pair; core theorem fixed-pair endpoint
exactness; exactness predicate the displayed iff; standard bridge fixed
ordered-pair endpoint. For the forward direction, apply
Theorem~\ref{thm:prod-pair-rep-determinism} to the displayed
base representation and the target representation, then project the
$\hsame(h,k)$ conjunct. For the reverse direction, apply
Theorem~\ref{thm:prod-pair-rep-hsame-transport} to the base representation and
the displayed endpoint sameness.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdPairRep\_fixed\_endpoint\_exactness}

\begin{definition}[Product pair-representation coherence]
\label{def:prod-pair-rep-coherent}
For source classifiers $\sim_L$ and $\sim_R$, define
$\mathsf{ProdPairRepCoherent}_{L,R,\sim_L,\sim_R}$ to mean that any two pair
representations of the same product history classify their corresponding
components:
$$
  \forall h\,l\,r\,l'\,r'.\;
  \mathsf{ProdPairRep}_{L,R}(h,l,r)\wedge
  \mathsf{ProdPairRep}_{L,R}(h,l',r')
  \to (l\sim_L l')\wedge(r\sim_R r').
$$
\end{definition}

\begin{theorem}[Product raw pair representation coherence is not automatic]
\label{thm:prod-pair-rep-coherence-not-automatic}
Let $\mathsf{TopCarrier}(h):=\mathsf{True}$ for every $h:\mathsf{BHist}$, and take both source classifiers to be $\hsame$. Then $\neg\mathsf{ProdPairRepCoherent}_{\mathsf{TopCarrier},\mathsf{TopCarrier},\hsame,\hsame}$ holds.

The per-object fit is: traditional target ordered-pair readback; BEDC source $\mathsf{BHist}$ with $\Cont$ and $\hsame$; carrier $\mathsf{ProdPairRep}$; classifier $\hsame$; stability obligation coherence; ledger displayed pair boundary; core theorem non-automaticity; exactness obstruction; standard bridge raw append ambiguity; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
The endpoint $\Ezero(\emp)$ has two raw pair presentations under $\mathsf{TopCarrier}$. The left-unit presentation uses components $(\emp,\Ezero(\emp))$, and the right-unit presentation uses components $(\Ezero(\emp),\emp)$. If product pair-representation coherence were automatic for the two $\hsame$ classifiers, applying it to these two presentations of the same endpoint would force $\hsame(\emp,\Ezero(\emp))$ on the left components, and also $\hsame(\Ezero(\emp),\emp)$ on the right components. BEDC history no-confusion separates the empty history from a zero-step endpoint, so the forced $\hsame(\emp,\Ezero(\emp))$ is impossible. Thus the componentwise ordered-pair certificate needs displayed coherence data and cannot be recovered from raw continuation alone.
\end{proof}

\begin{definition}[Product componentwise history classifier]
\label{def:prod-component-history-classifier}
For source-history carriers $L,R:\mathsf{BHist}\to\mathsf{Prop}$ and source
classifiers $\sim_L,\sim_R:\mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$,
define $\mathsf{ProdComponentHistoryClassifier}_{L,R,\sim_L,\sim_R}(h,k)$ by
the existence of component histories $l_h,r_h,l_k,r_k$ such that
$$
\mathsf{ProdPairRep}_{L,R}(h,l_h,r_h)
\wedge
\mathsf{ProdPairRep}_{L,R}(k,l_k,r_k)
\wedge
l_h\sim_L l_k
\wedge
r_h\sim_R r_k.
$$
\end{definition}
\leandef{BEDC.Derived.ProdUp.ProdComponentHistoryClassifier}

\begin{theorem}[Product componentwise history classifier source refinement]
\label{thm:prod-component-history-classifier-source-refinement}
The per-object fit is: traditional ordered-product classifier stability under
source refinement; BEDC source $\mathsf{BHist}$ with $\Cont$; carrier through
$\mathsf{ProdPairRep}$; classifier
$\mathsf{ProdComponentHistoryClassifier}$; ledger unchanged displayed pair
representations; one-way classifier preservation under carrier and classifier
inclusions; status $\mathsf{ProofTarget}$.

For carriers $L,R,L',R':\mathsf{BHist}\to\mathsf{Prop}$, classifiers
$\sim_L,\sim_R,\sim'_L,\sim'_R:\mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$,
and histories $h,k:\mathsf{BHist}$, suppose $\forall u.\;L(u)\to L'(u)$, $\forall u.\;R(u)\to R'(u)$,
$\forall x\,y.\;x\sim_L y\to x\sim'_L y$, and
$\forall x\,y.\;x\sim_R y\to x\sim'_R y$. Then
$\mathsf{ProdComponentHistoryClassifier}_{L,R,\sim_L,\sim_R}(h,k)$ implies
$\mathsf{ProdComponentHistoryClassifier}_{L',R',\sim'_L,\sim'_R}(h,k)$.
\end{theorem}
\begin{proof}
Unpack the componentwise classifier witness into displayed pair
representations of $h$ and $k$ with components $l_h,r_h,l_k,r_k$, together
with the two component classifier witnesses. Rebuild each displayed pair
representation for $L',R'$ by applying the carrier inclusions to the stored
component source witnesses and keeping the same continuation evidence.
Transport the two component classifier witnesses through the classifier
implications, then repack these two rebuilt displayed pair representations and
the transported component witnesses as a
$\mathsf{ProdComponentHistoryClassifier}_{L',R',\sim'_L,\sim'_R}(h,k)$
witness.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdComponentHistoryClassifier\_source\_refinement}
