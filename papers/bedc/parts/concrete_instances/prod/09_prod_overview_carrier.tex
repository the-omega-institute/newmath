\chapter{A Concrete Naming Certificate for \(\ProdUp\)}
\label{ch:concrete-instances-prod-namecert}

The product interface $\ProdUp$ has two classifier levels plus the checked envelope. $\mathsf{ProdClassifierSpec}$ is the typed finite-pair helper. $\mathsf{ProdComponentHistoryClassifier}_{L,R,\sim_L,\sim_R}$ is the public BEDC-history certificate classifier induced by $\mathsf{ProdPairRep}$ and the source classifiers, while $\mathsf{ProdHistoryClassifier}$ is the checked envelope classifier using endpoint $\hsame$. It is licensed only as a derived constructor over the two sources.

\section{Roadmap template status}

\begin{enumerate}
  \item \textbf{Traditional target}: ordered product data with one left component and one right component.
  \item \textbf{BEDC source}: the checked history-envelope certificate uses inhabited source-history predicates $L$ and $R$; the explicit typed interface uses two licensed source carriers $A$ and $B$, each with its own name certificate.
  \item \textbf{Carrier choice}: the paired history carrier records a left history, a right history, and the continuation that combines them.
  \item \textbf{Classifier choice}: $\mathsf{ProdClassifierSpec}$ is the typed finite-pair helper; $\mathsf{ProdComponentHistoryClassifier}_{L,R,\sim_L,\sim_R}$ is the public history-level classifier from $\mathsf{ProdPairRep}$ and source classifiers; $\mathsf{ProdHistoryClassifier}$ is the checked envelope classifier using endpoint $\hsame$.
  \item \textbf{Stability obligations}: component transport, continuation transport, unary closure from components, history-envelope classifier symmetry and transitivity, typed classifier symmetry and transitivity, typed coordinate inversion, $\mathsf{ProdPairRepCoherent}$, and componentwise-to-envelope descent.
  \item \textbf{Ledger}: no product quotient is used; pairing records both component witnesses, their generated continuation, and the displayed pair ledger soundness needed by the public history-level certificate.
  \item \textbf{Core theorem list}: append introduction, continuation introduction, carrier transport, unary closure, classifier equivalence, component-envelope inversion, $\mathsf{ProdPairRepCoherent}$ criteria, displayed pair ledger soundness, componentwise-to-envelope descent under $\hsame$-sound source classifiers, and the semantic name certificate.
  \item \textbf{Exactness predicate}: typed classification holds exactly when both finite-pair coordinates classify in their respective sources; public history-level classification holds through displayed $\mathsf{ProdPairRep}$ components and source classifiers; envelope classification uses endpoint $\hsame$.
  \item \textbf{Standard bridge}: the checked history-envelope bridge reads back only as a generated product endpoint with a displayed pair representation. Ordinary ordered-pair and componentwise history readback belong to the componentwise certificate target under $\mathsf{ProdPairRepCoherent}_{L,R,\sim_L,\sim_R}$ and $\mathsf{ProdComponentHistoryClassifier}_{L,R,\sim_L,\sim_R}$.

\end{enumerate}

\section{Carrier definition}

\begin{definition}[Prod carrier]
\label{def:prod-carrier}
For licensed source carriers \(A\) and \(B\), define a pair carrier with constructor \(\mathsf{pair}(a,b)\).
\end{definition}
\leandef{BEDC.Derived.ProdUp.ProdCarrier}

\section{The certificate}

\begin{definition}[Prod source specification]
\label{def:prod-source-specification}
For the checked history-envelope certificate, the source specification is a
pair of inhabited predicates $L,R:\mathsf{BHist}\to\mathsf{Prop}$. The carrier
is $\mathsf{ProdHistoryCarrier}(L,R)$, and the classifier is
$\mathsf{ProdHistoryClassifier}(L,R)$; no source $\NameCert$ is required at
this scope.

For the explicit ordered-product interface, the source data are two licensed
carriers with $\NameCert$ witnesses. Those certificates are used for typed
component transport, pair-representation coherence, and displayed component-exposure fields.
\end{definition}
\leandef{BEDC.Derived.ProdUp.ProdHistoryCarrier}

\begin{theorem}[Product history carrier append introduction]
\label{thm:prod-history-carrier-append-intro}
If \(l\) lies in the left source carrier and \(r\) lies in the right source
carrier, then \(\mathsf{append}(l,r)\) lies in the product history carrier.
\end{theorem}
\begin{proof}
Use \(l\) and \(r\) as the two component witnesses. The BEDC continuation
judgment is witnessed by reflexivity of the defining append result, so the
product history carrier is rebuilt from the two source proofs and that
continuation witness.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdHistoryCarrier\_append\_intro}

\begin{theorem}[Product history carrier continuation introduction]
\label{thm:prod-history-carrier-cont-intro}
If \(l\) lies in the left source carrier, \(r\) lies in the right source
carrier, and \(\Cont(l,r,h)\), then \(h\) lies in the product history carrier.
\end{theorem}
\begin{proof}
Use \(l\) and \(r\) as the component witnesses. The supplied continuation
proof is exactly the result witness required by the product history carrier.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdHistoryCarrier\_cont\_intro}

\begin{theorem}[Product empty-result component exposure]
\label{thm:prod-history-carrier-empty-result-components}
If $\mathsf{ProdHistoryCarrier}(L,R,\mathsf{Empty})$ holds, then there are
component histories $l$ and $r$ with $L(l)$ and $R(r)$ such that
$l\hsame\mathsf{Empty}$ and $r\hsame\mathsf{Empty}$.
\end{theorem}
\begin{proof}
Unpack the product carrier into component histories $l$ and $r$, their source
witnesses, and the continuation proof $\Cont(l,r,\mathsf{Empty})$. BEDC
empty-result inversion for continuations forces both component histories to be
the empty history, giving the two displayed $\hsame$ facts.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdHistoryCarrier\_empty\_result\_components}

\begin{theorem}[Product visible-result component cases]
\label{thm:prod-history-carrier-visible-result-cases}
If $\mathsf{ProdHistoryCarrier}(L,R,\Ezero(tail))$ holds, then its right
component is either $\emp$ and the left component is $\hsame$ to the visible
result, or the right component is itself $\Ezero$-headed and the stored
continuation exposes its tail. The same dichotomy holds for visible result
$\Eone(tail)$ with an $\Eone$-headed right component.
\end{theorem}
\begin{proof}
Unpack the product carrier into component histories $l$ and $r$, source
witnesses $L(l)$ and $R(r)$, and the continuation proof
$\Cont(l,r,\Ezero(tail))$ or $\Cont(l,r,\Eone(tail))$. Apply the corresponding
BEDC continuation visible-result inversion theorem to the stored continuation.
It yields exactly the empty-right case or the matching visible-headed right
component case.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdHistoryCarrier\_visible\_result\_cases}

\begin{theorem}[Product unary history closure]
\label{thm:prod-history-carrier-unary}
If a product history is generated from unary left and right histories, then the
resulting product history is unary.
\end{theorem}
\begin{proof}
Invert the product-history carrier into its left history, right history, unary
component witnesses, and continuation proof. Unary closure under continuation
turns the two unary component witnesses and the stored continuation proof into
unary membership of the result history.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdHistoryCarrier\_unary}

\begin{theorem}[Product history carrier transport]
\label{thm:prod-history-carrier-hsame-transport}
If two product histories are internally the same and the first is generated from a left witness, a right witness, and a continuation proof, then the second is generated from the same component witnesses and the transported continuation proof.
\end{theorem}
\begin{proof}
Invert the product-history carrier into its left history, right history, component witnesses, and continuation proof. Transport the continuation result across \(\hsame\), and rebuild the product-history carrier with the same component witnesses.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdHistoryCarrier\_hsame\_transport}

\begin{theorem}[Product history carrier unary from components]
\label{thm:prod-history-carrier-unary-of-components}
If the left and right source predicates each determine unary histories, then
every product-history carrier generated from those predicates is unary.
\end{theorem}
\begin{proof}
Unpack the product-history carrier into component histories \(l\) and \(r\),
their source witnesses, and the BEDC continuation witness for the result.
The component assumptions give unary witnesses for \(l\) and \(r\), and BEDC
unary continuation closure gives unary membership of the product result.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdHistoryCarrier\_unary\_of\_components}

\begin{definition}[Prod pattern specification]
\label{def:prod-pattern-specification}
The pattern is paired generation from one \(A\)-witness and one \(B\)-witness.
\end{definition}

\begin{definition}[Prod classifier specification]
\label{def:prod-classifier-specification}
The typed classifier $\mathsf{ProdClassifierSpec}$ relates pairs componentwise using the classifiers of $A$ and $B$. The typed wrapper $\mathsf{ProdClassifier}$ has the same componentwise content. Separately, the checked history-level scope is $\mathsf{ProdHistoryClassifier}(L,R)$: it relates histories $h$ and $k$ exactly when $h$ lies in $\mathsf{ProdHistoryCarrier}(L,R)$, $k$ lies in $\mathsf{ProdHistoryCarrier}(L,R)$, and $\hsame(h,k)$ holds.
\end{definition}
\leandef{BEDC.Derived.ProdUp.ProdClassifierSpec}
\leandef{BEDC.Derived.ProdUp.ProdHistoryClassifier}

\begin{theorem}[Product typed classifier coordinate exactness]
\label{thm:prod-classifier-coordinate-exactness}
For types $A$ and $B$, component classifiers $R_A:A\to A\to\mathsf{Prop}$
and $R_B:B\to B\to\mathsf{Prop}$, and elements $a,a':A$ and $b,b':B$,
the product typed classifier is exact on coordinates:
$$
\mathsf{ProdClassifierSpec}(R_A,R_B,\mathsf{pair}(a,b),\mathsf{pair}(a',b'))
\quad\Longleftrightarrow\quad
R_A(a,a')\land R_B(b,b').
$$
\end{theorem}
\begin{proof}
Unfold $\mathsf{ProdClassifierSpec}$. The forward direction is constructor
inversion on the product classifier witness, yielding the two coordinate
witnesses. The reverse direction pairs the two displayed coordinate witnesses
to rebuild the product classifier witness.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.prodClassifier\_iff}

\begin{definition}[Product typed source carrier]
\label{def:product-typed-source-carrier}
For source carriers $L,R:\mathsf{BHist}\to\mathsf{Prop}$ and
$x:\mathsf{ProdCarrier}(\mathsf{BHist},\mathsf{BHist})$, define
$\mathsf{ProdTypedCarrier}_{L,R}(x)$ by the displayed pair decomposition
$$
\mathsf{ProdTypedCarrier}_{L,R}(x)
\quad\Longleftrightarrow\quad
\exists l,r:\mathsf{BHist}.\ x=\mathsf{pair}(l,r)\land L(l)\land R(r).
$$
It is the typed source-carrier predicate for the explicit ordered-product
interface: the left coordinate is licensed by $L$ and the right coordinate is
licensed by $R$.
\end{definition}
\leandef{BEDC.Derived.ProdUp.ProdTypedCarrier}

\begin{theorem}[Product typed carrier transport from source certificates]
\label{thm:product-typed-carrier-transport-source-certificates}
Let $L,R:\mathsf{BHist}\to\mathsf{Prop}$ be source carriers and let
$\sim_L,\sim_R$ be source classifiers. If
$\mathsf{cert}_L:\NameCert(L,\sim_L)$,
$\mathsf{cert}_R:\NameCert(R,\sim_R)$,
$x,y:\mathsf{ProdCarrier}(\mathsf{BHist},\mathsf{BHist})$,
$\mathsf{ProdTypedCarrier}_{L,R}(x)$, and
$\mathsf{ProdClassifierSpec}(\sim_L,\sim_R,x,y)$, then
$\mathsf{ProdTypedCarrier}_{L,R}(y)$.
\end{theorem}
\begin{proof}
Invert the typed carrier witness for $x$ to obtain coordinates $l$ and $r$
with $x=\mathsf{pair}(l,r)$, $L(l)$, and $R(r)$. Invert the product carrier
$y$ as $\mathsf{pair}(l',r')$. Apply
Theorem~\ref{thm:prod-classifier-coordinate-exactness} to the classifier
witness for the two displayed pairs; this yields source classifier witnesses
$l\sim_L l'$ and $r\sim_R r'$. The
$\mathsf{carrier\_respects\_equiv}$ fields of $\mathsf{cert}_L$ and
$\mathsf{cert}_R$ transport $L(l)$ to $L(l')$ and $R(r)$ to $R(r')$.
Those transported witnesses rebuild $\mathsf{ProdTypedCarrier}_{L,R}(y)$.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdTypedCarrier\_transport\_from\_source\_certificates}
