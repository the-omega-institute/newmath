\chapter{The Signature Package Instance}
\label{ch:concrete-instances-signature-package}

The hardening layer introduced \(\PkgPol\) as an abstract interface. This chapter constructs the first concrete instance: package tokens generated directly from answer signatures.

\section{Signature tokens}

\begin{definition}[Signature token generator]
\label{def:signature-token-generator}
Fix a generated probe bundle \(\Pi\). For every generated answer history \(s\), introduce a formal package token
\[
  \mathsf{Tok}_{\Pi}(s).
\]
This notation is a naming convention for the package-token event associated with the answer history \(s\). It is not a host function in the object theory.
\end{definition}
\leandef{BEDC.FKernel.Package.SignatureTokenGenerator}

\begin{definition}[Signature package relation]
The concrete signature package relation \(\Pkg^{\mathrm{sig}}\) is generated by the rule
\[
  \Pkg^{\mathrm{sig}}(\Pi,s,\mathsf{Tok}_{\Pi}(s)).
\]
Package-sameness is generated by the rule
\[
  \hsame(s,t)\quad\Longrightarrow\quad
  \psame(\mathsf{Tok}_{\Pi}(s),\mathsf{Tok}_{\Pi}(t)).
\]
No additional package-sameness constructors are included.
\end{definition}

\begin{lemma}[Signature package existence]
\label{lem:signature-package-existence}
For every generated answer history \(s\), there exists a package token \(p\) such that
\[
  \Pkg^{\mathrm{sig}}(\Pi,s,p).
\]
\end{lemma}
\leanchecked{BEDC.FKernel.Package.signature\_package\_existence}
\begin{proof}
Take \(p=\mathsf{Tok}_{\Pi}(s)\) and apply the signature package rule.
\end{proof}

\begin{lemma}[Signature package extensionality]
If \(\hsame(s,t)\), \(\Pkg^{\mathrm{sig}}(\Pi,s,p)\), and \(\Pkg^{\mathrm{sig}}(\Pi,t,q)\), then \(\psame(p,q)\).
\end{lemma}
\begin{proof}
By the generation rule for \(\Pkg^{\mathrm{sig}}\), \(p\) is package-same to \(\mathsf{Tok}_{\Pi}(s)\) and \(q\) is package-same to \(\mathsf{Tok}_{\Pi}(t)\). The displayed package-sameness rule sends \(\hsame(s,t)\) to
\[
  \psame(\mathsf{Tok}_{\Pi}(s),\mathsf{Tok}_{\Pi}(t)).
\]
Using the internal equivalence behavior of \(\psame\), the conclusion follows.
\end{proof}

\begin{lemma}[Signature package grounding]
\label{lem:signature-package-grounding}
If \(\Pkg^{\mathrm{sig}}(\Pi,s,p)\), then \(p\) is generated from the signature history \(s\) by the displayed token rule.
\end{lemma}
\leanchecked{BEDC.FKernel.Package.signature\_package\_grounding}
\begin{proof}
By closed generation for \(\Pkg^{\mathrm{sig}}\), every package derivation is generated from a token \(\mathsf{Tok}_{\Pi}(s)\). No hidden token constructor is present.
\end{proof}

\begin{lemma}[Signature package sameness reflects history sameness]
\label{lem:signature-package-psame-implies-hsame}
For the concrete signature package instance, if \(\psame(p,q)\), then \(\hsame(p,q)\).
\end{lemma}
\leanchecked{BEDC.FKernel.Package.signature\_package\_psame\_implies\_hsame}
\begin{proof}
Invert the one package-sameness constructor. The displayed token-introduction witnesses identify \(p\) and \(q\) with the introducing signature histories, and transitivity of \(\hsame\) composes those witnesses with the constructor proof.
\end{proof}

\begin{theorem}[Concrete \(\PkgPol\) for signatures]
\label{thm:concrete-signature-package-policy}
The relation \(\Pkg^{\mathrm{sig}}\), together with the token-sameness rule above, satisfies the package existence, package extensionality, and package grounding requirements of \(\PkgPol\).
\end{theorem}
\leanchecked{BEDC.FKernel.Package.signature\_package\_policy}
\leanvariant{BEDC.FKernel.Package.signature\_package\_token\_exists}
\begin{proof}
Existence, extensionality, and grounding are exactly the three preceding lemmas.
\end{proof}

\begin{corollary}[The first concrete package interface]
\label{cor:first-concrete-package-interface}
Every generated probe bundle \(\Pi\) admits a concrete package interface whose visible tokens are generated by answer signatures.
\end{corollary}
\leanchecked{BEDC.FKernel.Package.first\_concrete\_package\_interface}
