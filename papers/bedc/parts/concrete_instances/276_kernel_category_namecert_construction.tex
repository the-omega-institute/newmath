\chapter{Kernel Category Name Certificate}
\label{ch:concrete-instances-kernel-as-category}

The object $\KernelCategoryUp$ names the categorical reading of the finite
kernel's unary-continuation surface.  Its rows do not introduce an external
category library.  They expose a source history, a target history, a displayed
morphism history, and the checked $\CategoryUp$ hom-carrier showing that the
morphism is a finite continuation from source to target.

\section{Source}
\label{sec:concrete-instances-kernel-as-category-source}

\begin{definition}[Source rows for $\KernelCategoryUp$]
\label{def:concrete-instances-kernel-as-category-source}
A source row is a tuple $(a,b,f,\gamma)$ where $a,b,f:\mathsf{BHist}$ and
$\gamma$ is a checked hom-carrier
$$
\begin{aligned}
  \gamma:\mathsf{CategoryHomCarrier}(a,b,f).
\end{aligned}
$$
Thus $a$ and $b$ are unary kernel objects, $f$ is the displayed morphism
history, and $\gamma$ records the finite continuation ledger from $a$ through
$f$ to $b$.
\end{definition}
\leandef{BEDC.Derived.KernelCategoryUp.SourceRows}

\section{Pattern}
\label{sec:concrete-instances-kernel-as-category-pattern}

\begin{definition}[Unary-continuation morphism pattern]
\label{def:concrete-instances-kernel-as-category-pattern}
The accepted pattern of a source row is exactly its displayed
$\CategoryUp$ hom-carrier:
$$
\begin{aligned}
  \mathsf{Pattern}(a,b,f,\gamma)
  \quad\Longleftrightarrow\quad
  \mathsf{CategoryHomCarrier}(a,b,f).
\end{aligned}
$$
Unfolded, this says that the source, target, and morphism histories are unary
and that $\Cont(a,f,b)$ is present in the row.  The categorical shape is
therefore the finite kernel composition surface itself.
\end{definition}
\leandef{BEDC.Derived.KernelCategoryUp.Pattern}

\section{Classifier}
\label{sec:concrete-instances-kernel-as-category-classifier}

\begin{definition}[Morphism classifier]
\label{def:concrete-instances-kernel-as-category-classifier}
Two $\KernelCategoryUp$ source rows classify together when their sources,
targets, and displayed morphism histories are componentwise $\hsame$:
$$
\begin{aligned}
  a\hsame a'
  \quad\text{and}\quad
  b\hsame b'
  \quad\text{and}\quad
  f\hsame f'.
\end{aligned}
$$
The classifier compares only the public histories carried by the finite
ledger.  It does not compare host records, categorical presentations, or
external morphism objects.
\end{definition}
\leandef{BEDC.Derived.KernelCategoryUp.Classifier}

\section{Stability}
\label{sec:concrete-instances-kernel-as-category-stability}

\begin{theorem}[Identity and composition stability]
\label{thm:concrete-instances-kernel-as-category-stability}
For any two source rows $a\xrightarrow{f} b$ and $b'\xrightarrow{g} c$ whose
middle objects classify by $b\hsame b'$, and for any displayed continuation
$\Cont(f,g,fg)$, the empty morphisms at $a$ and $c$ are accepted source-row
patterns and the composite row $a\xrightarrow{fg} c$ is accepted.  The
composite row is also classified with itself by the morphism classifier.
\end{theorem}
\leanchecked{BEDC.Derived.KernelCategoryUp.stability}

\begin{proof}
The identity rows are supplied by the checked
$\mathsf{CategoryHomCarrier\_empty\_identity}$ theorem.  The composite row is
supplied by $\mathsf{CategoryHomCarrier\_comp\_closed}$ after transporting the
middle object along the displayed $\hsame$ classifier.  Reflexivity of
$\hsame$ gives the final classifier judgment.  The proof uses only the finite
kernel history and continuation definitions.
\end{proof}

\section{Ledger}
\label{sec:concrete-instances-kernel-as-category-ledger}

\begin{definition}[Composition ledger]
\label{def:concrete-instances-kernel-as-category-ledger}
The ledger for $\KernelCategoryUp$ keeps a left source row, a right source row,
the middle-object classifier, the composite morphism history, and the displayed
continuation witness $\Cont(f,g,fg)$.  This records the full composition chain:
the two endpoint hom-carriers, the public reason that the middle endpoints
match, and the finite continuation witness generating the composite.
\end{definition}
\leandef{BEDC.Derived.KernelCategoryUp.Ledger}
