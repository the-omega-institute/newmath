\chapter{A Concrete Naming Certificate for \(\ListUp\)}
\label{ch:concrete-instances-list-namecert}

The list interface \(\ListUp\) is licensed over a source carrier endowed with a \(\NameCert\). The carrier is the inductive sequence built from a nil constructor and a cons constructor consuming one source witness and one shorter list. The classifier is structural sameness lifted from the source classifier: nil-with-nil and cons-with-cons under componentwise sameness. It is not a primitive sequence type; it is licensed only as a derived inductive over the source.

\section{Carrier definition}

\begin{definition}[List carrier]
\label{def:list-carrier}
For a licensed source carrier \(A\), define the inductive list carrier by \(\mathsf{nil}\) and \(\mathsf{cons}(a,\ell)\).
\end{definition}
\leandef{BEDC.Derived.ListUp.ListCarrier}

\section{The certificate}

\begin{definition}[List source specification]
\label{def:list-source-specification}
The source is finite generated sequencing over the source certificate for \(A\).
\end{definition}

\begin{definition}[List pattern specification]
\label{def:list-pattern-specification}
The pattern is nil-or-cons generation with structurally smaller tail.
\end{definition}

\begin{definition}[List classifier specification]
\label{def:list-classifier-specification}
The classifier relates nil with nil and cons with cons when heads are source-classified the same and tails are list-classified the same.
\end{definition}
\leandef{BEDC.Derived.ListUp.ListClassifierSpec}

\begin{theorem}[List classifier is transitive]
\label{thm:list-classifier-transitivity}
For any transitive source classifier on \(A\), the structural list classifier lifted from it is transitive on \(\ListUp(A)\).
\end{theorem}
\begin{proof}
Induct on the first list. Shape mismatches are impossible by the classifier clauses. In the cons-cons-cons case, the head witnesses compose by source transitivity and the tail witnesses compose by the induction hypothesis.
\end{proof}
\leanchecked{BEDC.Derived.ListUp.ListClassifierSpec\_trans}

\begin{theorem}[History list classifier is transitive]
\label{thm:list-classifier-hsame-trans}
For lists of finite-kernel histories, if the structural list classifier relates \(x\) to \(y\) and \(y\) to \(z\) using \(\hsame\) on heads, then it relates \(x\) to \(z\).
\end{theorem}
\begin{proof}
Induct on the first list. The nil and cons shape mismatches are impossible by the classifier clauses. In the cons-cons-cons case, head witnesses compose by transitivity of \(\hsame\), and the tail witnesses compose by the induction hypothesis.
\end{proof}
\leanchecked{BEDC.Derived.ListUp.ListClassifierSpec\_hsame\_trans}

\begin{definition}[List stability certificate]
\label{def:list-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item reflexivity by list induction
  \item transitivity by paired source/list induction
  \item nil/cons separation
  \item closure under cons
  \item tail inversion and head inversion for classified cons values
\end{enumerate}
\end{definition}
\leanchecked{BEDC.Derived.ListUp.listClassifier\_refl}

\begin{theorem}[List classifier stability under append]
\label{thm:list-classifier-append-hsame}
For BEDC histories, if two list pairs are structurally classified by \(\HSame\),
then appending the first components and appending the second components
preserves structural classification.
\end{theorem}
\leanchecked{BEDC.Derived.ListUp.ListClassifierSpec\_append\_hsame}

\begin{proof}
Induct on the first list. The empty case returns the classification of the
second pair. In the cons case, invert the second list and the structural
classifier; the head \(\HSame\) witness is reused and the tail follows from
the induction hypothesis.
\end{proof}

\begin{theorem}[List classifier transitivity from a source certificate]
\label{thm:list-classifier-trans-from-namecert}
If the source classifier is supplied by a \(\NameCert\), then the lifted list classifier is transitive on finite list spines.
\end{theorem}
\leanchecked{BEDC.Derived.ListUp.ListClassifierSpec\_trans\_from\_nameCert}

\begin{proof}
Proceed by induction on the first list. The nil case forces the middle list to be nil, then the final list is nil. In the cons case, nil/cons mismatches are impossible by classifier inversion. For matching cons constructors, transitivity of the source certificate handles the heads and the induction hypothesis handles the tails.
\end{proof}

\begin{definition}[List ledger policy]
\label{def:list-ledger-policy}
Record the finite constructor spine and the source ledger for each element.
\end{definition}

\begin{theorem}[Concrete \(\ListUp\) naming certificate]
\label{thm:concrete-list-namecert}
The structural tuple forms a valid NameCert for \(\ListUp(A)\).
\end{theorem}

\begin{corollary}[\(\ListUp\) is licensed, not primitive]
\label{cor:list-licensed-not-primitive}
\(\ListUp\) is licensed as a finite inductive sequence over \(A\).
\end{corollary}
