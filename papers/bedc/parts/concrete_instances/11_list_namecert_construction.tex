\chapter{A Concrete Naming Certificate for \(\ListUp\)}
\label{ch:concrete-instances-list-namecert}

The list interface $\ListUp$ is licensed over a source carrier endowed with a $\NameCert$ and a spine-representation coherence witness. The carrier is the inductive sequence built from a nil constructor and a cons constructor consuming one source witness and one shorter list. The classifier is structural sameness lifted from the source classifier through finite spines: nil-with-nil and cons-with-cons under componentwise sameness. It is not a primitive sequence type; it is licensed only as a derived inductive over the source under this coherence datum.

\section{Carrier definition}

\begin{definition}[List carrier]
\label{def:list-carrier}
For a licensed source carrier \(A\), define the inductive list carrier by \(\mathsf{nil}\) and \(\mathsf{cons}(a,\ell)\).
\end{definition}

The checked $\ListUp$ lemmas below operate on a finite-spine classifier model, while the $\NameCert$ carrier remains the generated nil/cons BEDC carrier described in this definition.

\section{The certificate}

\begin{definition}[List source specification]

\begin{definition}[List history carrier]
\label{def:list-history-carrier}
For a source carrier $S:\mathsf{BHist}\to\mathsf{Prop}$, define $\mathsf{ListHistoryCarrier}_S(h)$ as the least history predicate generated by two clauses:
\begin{enumerate}
  \item nil: if $\hsame(h,\emp)$, then $\mathsf{ListHistoryCarrier}_S(h)$;
  \item cons: if there are histories $a,t,p$ with $S(a)$, $\mathsf{ListHistoryCarrier}_S(t)$, $\ContR(a,t,p)$, and $\hsame(h,\Eone(p))$, then $\mathsf{ListHistoryCarrier}_S(h)$.
\end{enumerate}
\end{definition}

\begin{definition}[List recursive history classifier]
\label{def:list-history-classifier}
For a source carrier $S:\mathsf{BHist}\to\mathsf{Prop}$ and source classifier
$\sim_S:\mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$, define
$\mathsf{ListHistoryClassifier}^{\mathrm{rec}}_{S,\sim_S}(h,k)$ as the least relation
generated by two clauses:
\begin{enumerate}
  \item nil-nil: if $\hsame(h,\emp)$ and $\hsame(k,\emp)$, then
  $\mathsf{ListHistoryClassifier}^{\mathrm{rec}}_{S,\sim_S}(h,k)$;
  \item cons-cons: if there are histories $a,b,t,u,p,q$ such that $S(a)$,
  $S(b)$, $a\sim_S b$, $\mathsf{ListHistoryClassifier}^{\mathrm{rec}}_{S,\sim_S}(t,u)$,
  $\ContR(a,t,p)$, $\ContR(b,u,q)$, $\hsame(h,\Eone(p))$, and
  $\hsame(k,\Eone(q))$, then
  $\mathsf{ListHistoryClassifier}^{\mathrm{rec}}_{S,\sim_S}(h,k)$.
\end{enumerate}
\end{definition}

\begin{theorem}[List recursive history classifier carrier endpoints]
\label{thm:list-history-classifier-carrier-endpoints}
For any source carrier $S:\mathsf{BHist}\to\mathsf{Prop}$ and source classifier
$\sim_S:\mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$, if
$\mathsf{ListHistoryClassifier}^{\mathrm{rec}}_{S,\sim_S}(h,k)$ holds, then both
$\mathsf{ListHistoryCarrier}_S(h)$ and $\mathsf{ListHistoryCarrier}_S(k)$
hold.
\end{theorem}
\begin{proof}
Induct on the displayed classifier witness. In the nil-nil clause, the two
endpoint witnesses rebuild the nil clause of $\mathsf{ListHistoryCarrier}_S$
on both sides. In the cons-cons clause, the induction hypothesis gives carrier
witnesses for the two tails $t$ and $u$; combine those with the source carrier
witnesses $S(a)$ and $S(b)$, the continuation witnesses $\ContR(a,t,p)$ and
$\ContR(b,u,q)$, and the endpoint witnesses $\hsame(h,\Eone(p))$ and
$\hsame(k,\Eone(q))$ to rebuild the two cons carrier clauses.
\end{proof}

\begin{theorem}[List recursive history classifier constructor exactness]
\label{thm:list-history-classifier-constructor-exactness}
For any source carrier $S:\mathsf{BHist}\to\mathsf{Prop}$, source classifier
$\sim_S:\mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$, and histories $h,k$,
every witness of $\mathsf{ListHistoryClassifier}^{\mathrm{rec}}_{S,\sim_S}(h,k)$ is exactly one
of the two displayed constructor cases: either nil-nil endpoint witnesses
$\hsame(h,\emp)$ and $\hsame(k,\emp)$, or cons-cons witnesses $a,b,t,u,p,q$ with
$S(a)$, $S(b)$, $a\sim_S b$, $\mathsf{ListHistoryClassifier}^{\mathrm{rec}}_{S,\sim_S}(t,u)$,
$\ContR(a,t,p)$, $\ContR(b,u,q)$, $\hsame(h,\Eone(p))$, and
$\hsame(k,\Eone(q))$. A nil endpoint classified against a cons endpoint, or a
cons endpoint classified against a nil endpoint, has no classifier case.
\end{theorem}
\begin{proof}
Induct on the generated classifier witness. The nil-nil constructor gives
exactly the two empty endpoint witnesses. The cons-cons constructor gives
exactly the source head witnesses, the recursive tail-classifier witness, the
two continuation witnesses, and the two visible $e1$ endpoint witnesses. The
mixed endpoint claims reduce to BEDC history no-confusion:
$\mathsf{not\_hsame\_emp\_e1}$ eliminates nil against cons, and
$\mathsf{not\_hsame\_e1\_empty}$ eliminates cons against nil.
\end{proof}

\begin{theorem}[List history carrier generated cases]
\label{thm:list-history-carrier-generated-cases}
For any source carrier $S:\mathsf{BHist}\to\mathsf{Prop}$ and any history $h$, if $\mathsf{ListHistoryCarrier}_S(h)$, then either $\hsame(h,\emp)$, or there exist histories $a,t,p$ such that $S(a)$, $\mathsf{ListHistoryCarrier}_S(t)$, $\ContR(a,t,p)$, and $\hsame(h,\Eone(p))$.
\end{theorem}
\begin{proof}
Apply the induction principle for the least predicate displayed in the preceding definition. The nil constructor gives the left disjunct directly. The cons constructor exposes its witnesses $a,t,p$ together with $S(a)$, the recursive carrier witness for $t$, the continuation witness $\ContR(a,t,p)$, and the endpoint witness $\hsame(h,\Eone(p))$, which are exactly the data required for the right disjunct.
\end{proof}

\begin{theorem}[List history carrier transport]
\label{thm:list-history-carrier-hsame-transport}
For any source carrier $S:\mathsf{BHist}\to\mathsf{Prop}$, if $\hsame(h,k)$ and $\mathsf{ListHistoryCarrier}_S(h)$, then $\mathsf{ListHistoryCarrier}_S(k)$.
\end{theorem}
\begin{proof}
Proceed by cases on the displayed generation of $\mathsf{ListHistoryCarrier}_S(h)$. In the nil case, compose the symmetry of $\hsame(h,k)$ with the endpoint witness $\hsame(h,\emp)$ to obtain $\hsame(k,\emp)$, then use the nil clause. In the cons case, keep the witnesses $a,t,p$, the source witness $S(a)$, the recursive carrier witness for $t$, and the continuation witness $\ContR(a,t,p)$. Compose the symmetry of $\hsame(h,k)$ with the endpoint witness $\hsame(h,\Eone(p))$ to obtain $\hsame(k,\Eone(p))$, then rebuild the cons clause for $k$.
\end{proof}

\begin{theorem}[List recursive history classifier basic fields from a source certificate]\label{thm:list-recursive-history-classifier-equivalence-fields}
Let $A : \mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier, let $\sim_A : \mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$ be a source classifier, and let $\mathsf{cert} : \NameCert(A,\sim_A)$. The generated classifier $\mathsf{ListHistoryClassifier}^{\mathrm{rec}}_{A,\sim_A}$ has the following basic fields from the source certificate:
\begin{enumerate}
  \item if $\mathsf{ListHistoryCarrier}_A(h)$ holds, then $\mathsf{ListHistoryClassifier}^{\mathrm{rec}}_{A,\sim_A}(h,h)$ holds;
  \item if $\mathsf{ListHistoryClassifier}^{\mathrm{rec}}_{A,\sim_A}(h,k)$ holds, then $\mathsf{ListHistoryClassifier}^{\mathrm{rec}}_{A,\sim_A}(k,h)$ holds;
  \item if $\mathsf{ListHistoryClassifier}^{\mathrm{rec}}_{A,\sim_A}(h,k)$ holds, then both $\mathsf{ListHistoryCarrier}_A(h)$ and $\mathsf{ListHistoryCarrier}_A(k)$ hold.
\end{enumerate}
\end{theorem}
\begin{proof}
For reflexivity, induct on the displayed generation of $\mathsf{ListHistoryCarrier}_A(h)$. The nil clause is rebuilt from $\hsame$ reflexivity and the given endpoint link. In the cons clause, the source certificate supplies head reflexivity from the head carrier witness, the induction hypothesis supplies tail reflexivity, and the stored $\ContR$ witness and endpoint $\hsame$ link rebuild the cons-cons classifier clause.
For endpoint membership, induct on the displayed generation of $\mathsf{ListHistoryClassifier}^{\mathrm{rec}}_{A,\sim_A}(h,k)$. The nil-nil clause rebuilds two nil carrier witnesses. The cons-cons clause combines the two source carrier witnesses, the induction hypotheses for the two tails, the two $\ContR$ witnesses, and the two endpoint $\hsame$ links to rebuild both cons carrier witnesses.
For symmetry, induct on the classifier generation. The nil-nil case swaps the two endpoint $\hsame$ witnesses. The cons-cons case uses the symmetry field of $\mathsf{cert}$ on the head classifier witness, applies the induction hypothesis to the tail classifier witness, swaps the two $\ContR$ witnesses and endpoint links, and rebuilds the cons-cons clause.
\end{proof} Transitivity for the certificate classifier is the separate spine-coherent result Theorem~\ref{thm:list-history-classifier-trans-from-spine-coherence}.

\begin{theorem}[List classifier case exactness]
\label{thm:list-classifier-case-exactness}
For any source relation $\mathsf{sameA}$ on $A$ and any finite spines $xs,ys : \mathsf{ListCarrier}(A)$, $\mathsf{ListClassifierSpec}(\mathsf{sameA},xs,ys)$ holds exactly in the matching constructor cases: both spines are $\mathsf{nil}$, or both spines are $\mathsf{cons}$ spines with a source-head witness $\mathsf{sameA}(a,b)$ and a recursive tail witness $\mathsf{ListClassifierSpec}(\mathsf{sameA},xs',ys')$. It is false for the mixed shapes $\mathsf{nil}$ against $\mathsf{cons}(b,ys')$ and $\mathsf{cons}(a,xs')$ against $\mathsf{nil}$.
\end{theorem}
\begin{proof}
Case-analyze the two finite-spine constructors. In the nil-nil case the classifier is the nil clause. In the cons-cons case the classifier unfolds to its head source witness together with the recursive tail classifier witness, and conversely those two witnesses rebuild the cons classifier clause. The nil-cons and cons-nil cases have no classifier clause, so any assumed witness is eliminated by constructor case analysis.
\end{proof}

\begin{lemma}[List classifier nil-cons no-confusion]
\label{lem:list-classifier-nil-cons-no-confusion}
For any source classifier on $A$, $\mathsf{ListClassifierSpec}$ never relates
$\mathsf{nil}$ to $\mathsf{cons}(a,xs)$, and never relates
$\mathsf{cons}(a,xs)$ to $\mathsf{nil}$.
\end{lemma}
\begin{proof}
Unfold the structural classifier by list constructors. The only nil clause is
the nil-nil clause, and the only cons clause is the cons-cons clause. The two
mixed constructor shapes therefore have no classifier constructor, so both
directions are discharged directly by case analysis on the impossible witness.
\end{proof}

\begin{theorem}[History list classifier constructor-map embeddings]
\label{thm:list-classifier-constructor-map-embeddings}
For BEDC history lists $xs$ and $ys$, structural classification by $\hsame$ is
equivalent to structural classification after mapping both lists through the
$e0$ history constructor, and likewise equivalent to structural classification
after mapping both lists through the $e1$ history constructor.
\end{theorem}
\begin{proof}
For the $e0$ equivalence, the forward implication is exactly the checked
preservation theorem for $e0$-mapping, and the reverse implication is exactly
the checked $e0$-mapping inversion theorem. The $e1$ equivalence is obtained in
the same way from the checked preservation theorem for $e1$-mapping and the
checked $e1$-mapping inversion theorem. Thus each BEDC history constructor
preserves and reflects the list classifier, giving a classifier embedding in
both constructor-map cases.
\end{proof}

\begin{theorem}[History list classifier separates mixed mapped tags]
\label{thm:list-classifier-mixed-tag-empty}

\end{proof}
\leanchecked{BEDC.Derived.ListUp.ListClassifierSpec\_hsame\_map\_e0\_map\_e1\_empty}

\begin{theorem}[List classifier preserves length]
\label{thm:list-classifier-length-eq}
For any source relation $sameA$ on BEDC history entries and finite list spines
$xs,ys$, if $\mathsf{ListClassifierSpec}(sameA,xs,ys)$ holds, then $xs$ and
$ys$ have the same finite length.
\end{theorem}
\begin{proof}
Induct on the first list spine. The nil case can classify only against nil by
the mixed-shape clauses of $\mathsf{ListClassifierSpec}$. In the cons case,
the classifier can classify only against a cons spine; its tail witness is again
a $\mathsf{ListClassifierSpec}$ witness for the source relation $sameA$, so the
induction hypothesis gives equality of tail lengths, and the two cons lengths
are equal.
\end{proof}

\begin{theorem}[List classifier stability under append]
\label{thm:list-classifier-append-source-relation}
For any source entry type $A$, source classifier $sameA:A\to A\to\mathsf{Prop}$,

\begin{proof}

classifier witness. In the cons clause, keep the head classifier witness and

classifier witness, then rebuild the cons classifier clause.
\end{proof}

\begin{theorem}[Classified list prefix cancellation]
\label{thm:list-classifier-hsame-append-left-cancel-classified}
For BEDC history lists, if \(p\) and \(p'\) are structurally classified by
\(\hsame\), and \(p{++}x\) is structurally classified with \(p'{++}y\), then
\(x\) is structurally classified with \(y\).
\end{theorem}
\begin{proof}
Induct on \(p\). The empty case forces \(p'\) to be empty by the prefix
classifier, leaving the appended classifier. In the cons case, the prefix
classifier and appended classifier both expose matching classified tails, and
the induction hypothesis applies to those tails.
\end{proof}
\leanchecked{BEDC.Derived.ListUp.ListClassifierSpec\_hsame\_append\_left\_cancel\_classified}

\begin{theorem}[List classifier stability under reverse]
\label{thm:list-classifier-reverse-hsame}
For BEDC history lists, structural classification by $\HSame$ is preserved when both lists are reversed.
\end{theorem}
\begin{proof}
Induct on the first list. Shape mismatches are impossible by the classifier clauses. In the cons-cons case, reverse turns each list into the reversed tail followed by a singleton head; the induction hypothesis handles the tails and append stability combines it with the head witness.
\end{proof}
\leanchecked{BEDC.Derived.ListUp.ListClassifierSpec\_reverse\_hsame}

\begin{theorem}[List classifier reverse inversion]
\label{thm:list-classifier-reverse-hsame-inversion}
For BEDC history lists, if the reversed lists are structurally classified by
\(\HSame\), then the original lists are structurally classified by \(\HSame\).
\end{theorem}
\begin{proof}
Apply reverse stability to the classified reversed lists. The double reverse of
each list is definitionally identified with the original list by list
reverse-reverse induction, giving the desired structural classifier.
\end{proof}
\leanchecked{BEDC.Derived.ListUp.ListClassifierSpec\_reverse\_hsame\_inversion}

\begin{theorem}[List classifier reverse equivalence]
\label{thm:list-classifier-reverse-hsame-equiv}
For BEDC history lists $xs$ and $ys$, structural classification by $\HSame$ is equivalent before and after reversing both lists:
$$
  \mathsf{ListClassifierSpec}(\hsame,xs,ys)
  \Longleftrightarrow
  \mathsf{ListClassifierSpec}(\hsame,\mathsf{reverse}(xs),\mathsf{reverse}(ys)).
$$
\end{theorem}
\begin{proof}
The forward direction is the reverse-stability theorem. The reverse direction is the reverse-inversion theorem applied to the classified reversed lists.
\end{proof}

\begin{theorem}[List classifier append right cancellation]
\label{thm:list-classifier-append-right-cancel-hsame}
For BEDC history lists, if adding the same right suffix to two lists gives a
structural \(\HSame\)-classification, then the original lists are structurally
classified.
\end{theorem}
\begin{proof}
Reverse the classified append witnesses. The common right suffix becomes a
common left prefix, so left-append cancellation removes it. Reversing the
remaining classifier witness returns the original list order.
\end{proof}
\leanchecked{BEDC.Derived.ListUp.ListClassifierSpec\_append\_right\_cancel\_hsame}

\begin{theorem}[List classifier right cancellation with equivalent suffixes]
\label{thm:list-classifier-append-right-cancel-with-hsame-suffix}
For BEDC history lists, if two right suffixes are structurally classified by
\(\HSame\), and appending those suffixes to \(x\) and \(y\) gives a structural
\(\HSame\)-classification, then \(x\) and \(y\) are structurally classified.
\end{theorem}
\begin{proof}
Use symmetry on the suffix classifier and append stability with reflexivity on
the left list \(y\) to classify \(y{++}\mathit{suffix}'\) with
\(y{++}\mathit{suffix}\). Transitivity aligns the appended classification to a
common right suffix, and ordinary right-append cancellation removes it.
\end{proof}
\leanchecked{BEDC.Derived.ListUp.ListClassifierSpec\_append\_right\_cancel\_with\_hsame\_suffix}

\begin{theorem}[List classifier two-sided cancellation with classified boundaries]
\label{thm:list-classifier-append-two-sided-cancel-classified}
For BEDC history lists $p$, $p'$, $x$, $y$, $s$, and $s'$, if
$$
  \mathsf{ListClassifierSpec}(\hsame,p,p'),
  \quad
  \mathsf{ListClassifierSpec}(\hsame,s,s'),
  \quad
  \mathsf{ListClassifierSpec}(\hsame,p{++}x{++}s,p'{++}y{++}s')
$$
hold, then $\mathsf{ListClassifierSpec}(\hsame,x,y)$ holds.
\end{theorem}
\begin{proof}
Reassociate both appended histories as $(p{++}x){++}s$ and
$(p'{++}y){++}s'$. Apply right cancellation with the classified suffixes to
obtain $\mathsf{ListClassifierSpec}(\hsame,p{++}x,p'{++}y)$. Then apply
classified left cancellation with the classified prefixes $p$ and $p'$ to
obtain $\mathsf{ListClassifierSpec}(\hsame,x,y)$.
\end{proof}

\begin{theorem}[List classifier classified append context equivalence]
\label{thm:list-classifier-append-context-equiv}
For BEDC history lists $p$, $p'$, $x$, $y$, $s$, and $s'$, if the prefix
contexts $p$ and $p'$ are structurally classified by $\HSame$ and the suffix
contexts $s$ and $s'$ are structurally classified by $\HSame$, then
$$
  \mathsf{ListClassifierSpec}(\hsame,x,y)
  \quad\text{if and only if}\quad
  \mathsf{ListClassifierSpec}(\hsame,p{++}x{++}s,p'{++}y{++}s')
$$
holds.
\end{theorem}
\begin{proof}
For the forward direction, apply append stability first to the classified
prefixes and middle lists, obtaining a classification of $p{++}x$ with
$p'{++}y$, and then apply append stability again with the classified suffixes.
For the reverse direction, apply the preceding two-sided cancellation theorem
to the classified bordered histories, the classified prefixes, and the
classified suffixes.
\end{proof}

\begin{theorem}[List classifier transitivity from a source certificate]
\label{thm:list-classifier-trans-from-namecert}
If the source classifier is supplied by a \(\NameCert\), then the lifted list classifier is transitive on finite list spines.
\end{theorem}
\leanchecked{BEDC.Derived.ListUp.ListClassifierSpec\_trans\_from\_nameCert}

\begin{proof}
Proceed by induction on the first list. The nil case forces the middle list to be nil, then the final list is nil. In the cons case, nil/cons mismatches are impossible by classifier inversion. For matching cons constructors, transitivity of the source certificate handles the heads and the induction hypothesis handles the tails.
\end{proof}

\begin{theorem}[List classifier symmetry from a source certificate]
\label{thm:list-classifier-symm-from-namecert}
For a BEDC history carrier $A : \mathsf{BHist} \to \mathsf{Prop}$, source classifier $\sim_A$, and source certificate $\mathsf{cert} : \NameCert(A,\sim_A)$, if $\mathsf{ListClassifierSpec}(\sim_A,xs,ys)$ holds for finite list spines $xs$ and $ys$, then $\mathsf{ListClassifierSpec}(\sim_A,ys,xs)$ holds.
\end{theorem}
\begin{proof}
Induct on the structural list-classifier witness. The nil clause reverses to nil. In the cons clause, apply $\mathsf{cert.equiv\_symm}$ to the head witness and apply the induction hypothesis to the tail witness; the cons constructor of the lifted classifier rebuilds the reversed structural witness.
\end{proof}

\begin{theorem}[List carrier transport from a source certificate]
\label{thm:list-carrier-transport-from-namecert}
Let $C:\mathsf{BHist}\to\mathsf{Prop}$ be a source carrier predicate, let $\sim_C:\mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$ be a source classifier, and assume $\mathsf{cert}:\NameCert(C,\sim_C)$. For finite BEDC-history spines $xs,ys:\mathsf{HistSpine}$, if every entry of $xs$ satisfies $C$ and $\mathsf{ListClassifierSpec}(\sim_C,xs,ys)$ holds, then every entry of $ys$ satisfies $C$.
\end{theorem}
\begin{proof}
Induct on the structural classifier derivation. The nil clause has no entries to transport. In the cons clause, the source certificate field $\mathsf{NameCert.carrier\_respects\_equiv}$ transports the head carrier witness across the head classifier witness, and the induction hypothesis transports all tail entries across the tail classifier witness.
\end{proof}

\begin{definition}[List ledger policy]
\label{def:list-ledger-policy}
Record the finite constructor spine and the source ledger for each element.
\end{definition}

\begin{definition}[List history ledger]
\label{def:list-history-ledger}
For a source carrier $S:\mathsf{BHist}\to\mathsf{Prop}$,
$\mathsf{ListHistoryLedger}_S(h)$ records the visible history endpoint as a
finite nil/cons spine: the empty endpoint is recorded by $\hsame(h,\emp)$, and
a cons endpoint is recorded by histories $a,t,p$ with $S(a)$,
$\mathsf{ListHistoryLedger}_S(t)$, $\ContR(a,t,p)$, and
$\hsame(h,\Eone(p))$.
\end{definition}

\begin{theorem}[List history carrier, pattern, and ledger equivalence]
\label{thm:list-history-carrier-pattern-ledger-equivalence}
For every source carrier $S:\mathsf{BHist}\to\mathsf{Prop}$ and history $h$,
the three displayed list-history fields determine the same endpoints:
$$
  \mathsf{ListHistoryCarrier}_S(h)
  \Longleftrightarrow
  \mathsf{ListHistoryPattern}_S(h),
$$
and
$$
  \mathsf{ListHistoryCarrier}_S(h)
  \Longleftrightarrow
  \mathsf{ListHistoryLedger}_S(h).
$$
\end{theorem}
\begin{proof}
For the carrier-pattern equivalence, use the displayed generated cases for
$\mathsf{ListHistoryCarrier}_S$. The nil case gives $\hsame(h,\emp)$, which is
the nil pattern branch. The cons case exposes histories $a,t,p$ with $S(a)$,
$\mathsf{ListHistoryCarrier}_S(t)$, $\ContR(a,t,p)$, and
$\hsame(h,\Eone(p))$, which is exactly the cons pattern branch. Conversely,
each pattern branch is one of the two carrier constructors, using the same
$\hsame$, $S(a)$, $\ContR(a,t,p)$, and tail carrier witnesses.

For the carrier-ledger equivalence, induct over the displayed nil/cons
generation in each direction. From carrier to ledger, the nil constructor
becomes the nil ledger clause, and the cons constructor reuses $S(a)$,
$\ContR(a,t,p)$, the endpoint witness $\hsame(h,\Eone(p))$, and the induction
hypothesis on the tail. From ledger to carrier, the nil ledger clause rebuilds
the nil carrier clause, and the cons ledger clause reuses the same source,
continuation, endpoint, and tail witnesses, with the induction hypothesis
turning the tail ledger witness into the required tail carrier witness.
\end{proof}

\begin{definition}[List spine representation]\label{def:list-spine-representation}
Let $\mathsf{HistSpine}$ be the finite-spine carrier whose entries are BEDC histories. For a BEDC history source carrier $A:\mathsf{BHist}\to\mathsf{Prop}$, the relation $\mathsf{ListSpineRep}_A(h,xs)$ has $xs:\mathsf{HistSpine}$ and says that $h$ represents $xs$ by the same nil and cons clauses used by $\mathsf{ListHistoryCarrier}_A$; the source predicate is carried by the clauses:
\begin{enumerate}
  \item nil: $xs=\mathsf{nil}$ and $\hsame(h,\emp)$;
  \item cons: $xs=\mathsf{cons}(a,xs')$ and there are histories $t,p$ with $A(a)$, $\mathsf{ListSpineRep}_A(t,xs')$, $\ContR(a,t,p)$, and $\hsame(h,\Eone(p))$.
\end{enumerate}
The spine-bridge classifier used by the certificate is
$$
  \mathsf{ListSpineBridgeClassifier}_{A,\sim_A}(h,k)
  \quad\Longleftrightarrow\quad
  \exists xs,ys:\mathsf{HistSpine}.\;
  \mathsf{ListSpineRep}_A(h,xs)\land
  \mathsf{ListSpineRep}_A(k,ys)\land
  \mathsf{ListClassifierSpec}(\sim_A,xs,ys).
$$
The certificate classifier is this bridge classifier:
$$
  \mathsf{ListHistoryClassifier}_{A,\sim_A}(h,k)
  \quad\Longleftrightarrow\quad
  \mathsf{ListSpineBridgeClassifier}_{A,\sim_A}(h,k).
$$
Thus $\mathsf{ListClassifierSpec}$ remains the finite-spine classifier, $\mathsf{ListSpineBridgeClassifier}_{A,\sim_A}$ names the BEDC-history bridge through finite spines, and $\mathsf{ListHistoryClassifier}_{A,\sim_A}$ is the certificate classifier whose endpoints are BEDC histories. The recursive classifier $\mathsf{ListHistoryClassifier}^{\mathrm{rec}}_{S,\sim_S}$ from Definition~\ref{def:list-history-classifier} is a paper-side generator for carrier and endpoint reasoning.
\end{definition}

\begin{theorem}[List spine nil-cons no-confusion]\label{thm:list-spine-rep-nil-cons-no-confusion}
For every BEDC history source carrier $A:\mathsf{BHist}\to\mathsf{Prop}$, history $h$, source history $a$, and tail spine $xs:\mathsf{HistSpine}$, $\mathsf{ListSpineRep}_A(h,\mathsf{nil})$ and $\mathsf{ListSpineRep}_A(h,\mathsf{cons}(a,xs))$ are contradictory.
\end{theorem}
\begin{proof}
Unpack the nil representation to obtain $\hsame(h,\emp)$. Unpack the cons representation to obtain an endpoint witness $\hsame(h,\Eone(p))$ for some continuation package $p$. Compose the symmetry of the nil endpoint with the cons endpoint to get $\hsame(\emp,\Eone(p))$, then close by BEDC history constructor no-confusion between the empty history and a one-step endpoint.
\end{proof}

\begin{theorem}[List spine representation exactness]
\label{thm:list-spine-representation-exactness}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier and let $\sim_A$ be a source classifier. The spine representation of Definition~\ref{def:list-spine-representation} is exact for the generated list carrier and classifier:
\begin{enumerate}
  \item if $\mathsf{ListSpineRep}_A(h,xs)$ holds, then $\mathsf{ListHistoryCarrier}_A(h)$ holds;
  \item if $\mathsf{ListHistoryCarrier}_A(h)$ holds, then there is a finite spine $xs$ such that $\mathsf{ListSpineRep}_A(h,xs)$ holds;
  \item for all histories $h,k$, the spine-induced classifier $\mathsf{ListSpineBridgeClassifier}_{A,\sim_A}(h,k)$ holds exactly when the recursive generated classifier $\mathsf{ListHistoryClassifier}^{\mathrm{rec}}_{A,\sim_A}(h,k)$ holds.
\end{enumerate}
\end{theorem}
\begin{proof}
For soundness, induct on the displayed derivation of $\mathsf{ListSpineRep}_A(h,xs)$. The nil representation is exactly the nil generator of $\mathsf{ListHistoryCarrier}_A$, and the cons representation carries the head witness $A(a)$, the recursive representation of the tail, the continuation witness $\ContR(a,t,p)$, and the endpoint witness $\hsame(h,\Eone(p))$, so the cons generator rebuilds the carrier.

For completeness, induct on the displayed generation of $\mathsf{ListHistoryCarrier}_A(h)$. The nil carrier chooses $xs=\mathsf{nil}$ and the stored empty endpoint witness. The cons carrier first applies the induction hypothesis to the tail carrier, then chooses $xs=\mathsf{cons}(a,xs')$ and reuses the source, continuation, and endpoint witnesses.

For classifier equivalence, unfold both classifier presentations. A bridge witness exposes finite spines and a $\mathsf{ListClassifierSpec}$ witness; applying carrier soundness to both representation endpoints gives the generated list histories, and induction on the finite-spine classifier derivation rebuilds the nil-nil and cons-cons classifier clauses. Conversely, a generated classifier derivation is handled by induction on its nil/cons clauses: the nil case chooses two nil spines, and the cons case uses the induction hypotheses for the tail classifier together with the stored source head classification and continuation witnesses to construct the finite-spine classifier witness.
\end{proof}

\begin{definition}[List spine representation coherence]
\label{def:list-spine-rep-coherence}
For a BEDC history source carrier $A:\mathsf{BHist}\to\mathsf{Prop}$ and
source classifier $\sim_A$, define
$\mathsf{ListSpineRepCoherent}_{A,\sim_A}$ as the proposition that every two
finite-spine presentations of the same history are classified by the finite
spine classifier. Concretely, for every history $m$ and spines $ys,ys':\mathsf{HistSpine}$,
if $\mathsf{ListSpineRep}_A(m,ys)$ and $\mathsf{ListSpineRep}_A(m,ys')$, then
$\mathsf{ListClassifierSpec}(\sim_A,ys,ys')$.
\end{definition}

\begin{theorem}[List history classifier transitivity from spine coherence]
\label{thm:list-history-classifier-trans-from-spine-coherence}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier, let
$\sim_A$ be a source classifier, and let
$\mathsf{cert}:\NameCert(A,\sim_A)$. If
$\mathsf{ListSpineRepCoherent}_{A,\sim_A}$ holds, then the existential
spine-representation classifier of Definition~\ref{def:list-spine-representation}
is transitive: whenever
$\mathsf{ListSpineBridgeClassifier}_{A,\sim_A}(h,k)$ and
$\mathsf{ListSpineBridgeClassifier}_{A,\sim_A}(k,u)$ hold, then
$\mathsf{ListSpineBridgeClassifier}_{A,\sim_A}(h,u)$ holds.
\end{theorem}
\begin{proof}
Unpack the two history-classifier witnesses. The first gives spines $xs,ys$
with representations of $h$ and $k$ and a witness
$\mathsf{ListClassifierSpec}(\sim_A,xs,ys)$. The second gives spines $ys',zs$
with representations of $k$ and $u$ and a witness
$\mathsf{ListClassifierSpec}(\sim_A,ys',zs)$. Apply
Definition~\ref{def:list-spine-rep-coherence} to the two displayed
representations of the middle history $k$ to obtain
$\mathsf{ListClassifierSpec}(\sim_A,ys,ys')$. The checked finite-spine theorem
$\mathsf{ListClassifierSpec\_trans\_from\_nameCert}$, applied with
$\mathsf{cert}$, first composes $xs$ with $ys'$ through $ys$ and then composes
the result with $zs$ through $ys'$. Repack the original representation of $h$,
the original representation of $u$, and the composed finite-spine classifier
witness to obtain $\mathsf{ListSpineBridgeClassifier}_{A,\sim_A}(h,u)$.
\end{proof}

\begin{theorem}[List spine classifier preserves represented length]
\label{thm:list-spine-classifier-represented-length}
Let $A:\mathsf{BHist}\to\mathsf{Prop}$ be a BEDC history source carrier, let
$\sim_A$ be a source classifier, and let
$\mathsf{cert}:\NameCert(A,\sim_A)$. If
$\mathsf{ListSpineRepCoherent}_{A,\sim_A}$ holds, then represented finite
spines have equal length along the spine-induced history classifier: whenever
$\mathsf{ListSpineRep}_A(h,xs_0)$,
$\mathsf{ListSpineRep}_A(k,ys_0)$, and
$\mathsf{ListHistoryClassifier}_{A,\sim_A}(h,k)$ hold, the finite spines
$xs_0$ and $ys_0$ have the same length.
\end{theorem}
\begin{proof}
Unpack the history-classifier witness to finite spines $xs,ys$ with
$\mathsf{ListSpineRep}_A(h,xs)$, $\mathsf{ListSpineRep}_A(k,ys)$, and
$\mathsf{ListClassifierSpec}(\sim_A,xs,ys)$. Coherence applied at $h$
classifies $xs_0$ with $xs$, and coherence applied at $k$ classifies $ys$ with
$ys_0$. Compose these three finite-spine classifier witnesses by
$\mathsf{ListClassifierSpec\_trans\_from\_nameCert}$ and $\mathsf{cert}$ to get
$\mathsf{ListClassifierSpec}(\sim_A,xs_0,ys_0)$. The paper-side arbitrary-source finite-spine length lemma in Theorem~\ref{thm:list-classifier-length-eq} gives equality of the represented finite lengths.
\end{proof}

\begin{theorem}[List spine representation coverage]
\label{thm:list-spine-representation-coverage}
For every BEDC history source carrier $A:\mathsf{BHist}\to\mathsf{Prop}$ and
history $h$, the generated history carrier is covered exactly by finite spine
representations:
$$
\mathsf{ListHistoryCarrier}_A(h)
\quad\Longleftrightarrow\quad
\exists xs:\mathsf{HistSpine}.\;\mathsf{ListSpineRep}_A(h,xs).
$$
\end{theorem}
\begin{proof}
For the forward direction, induct on the generated
$\mathsf{ListHistoryCarrier}_A(h)$ witness. The nil constructor is represented by
the empty finite spine. The cons constructor uses the induction hypothesis for
the tail carrier witness to obtain a tail spine, then prefixes it by the source
history and reuses the continuation and endpoint witnesses. For the reverse
direction, induct on the $\mathsf{ListSpineRep}_A(h,xs)$ witness. The nil
representation rebuilds the nil carrier clause, and the cons representation
rebuilds the cons carrier clause from the source witness, the recursively
recovered tail carrier witness, the continuation witness, and the displayed
$\hsame(h,\Eone(p))$ endpoint.
\end{proof}

\begin{theorem}[Conditional $\ListUp$ semantic certificate from the spine bridge]
\label{thm:concrete-list-namecert}
Given a BEDC history source carrier $A : \mathsf{BHist} \to \mathsf{Prop}$,
source classifier $\sim_A$, source certificate
$\mathsf{cert} : \NameCert(A,\sim_A)$, and a witness of
$\mathsf{ListSpineRepCoherent}_{A,\sim_A}$, define
$\mathsf{ListSource}(A)$ as the history carrier
$\mathsf{ListHistoryCarrier}_A$, take the certificate classifier
$\mathsf{ListHistoryClassifier}_{A,\sim_A}$ to be the spine-induced bridge
classifier from Definition~\ref{def:list-spine-representation}, use
Theorem~\ref{thm:list-spine-representation-exactness} as its finite-spine
exactness theorem, take $\mathsf{ListHistoryPattern}_A$ from
Definition~\ref{def:list-history-pattern} as the pattern, and take
$\mathsf{ListHistoryLedger}_A$ from Definition~\ref{def:list-history-ledger}
as the ledger. The host-list spine model is a presentation witness, not the exported carrier. Under the source $\NameCert$ and the spine-coherence witness,
these fields form the paper-side target
$\mathsf{SemanticNameCert}(\mathsf{ListHistoryCarrier}_A,\mathsf{ListHistoryPattern}_A,\mathsf{ListHistoryLedger}_A,\mathsf{ListHistoryClassifier}_{A,\sim_A})$
for $\ListUp(A)$.
\end{theorem}
\begin{proof}
The source $\NameCert$ supplies the head-level reflexivity, symmetry,
transitivity, and carrier transport fields. The generated nil/cons carrier
supplies the BEDC histories, and
Theorem~\ref{thm:list-spine-representation-exactness} supplies the exact
finite-spine presentation through which the checked structural classifier is
applied. Classifier reflexivity chooses a witness $\mathsf{ListSpineRep}_A(h,xs)$ and inducts on that displayed representation: the nil branch is immediate, and the cons branch uses the carried source proof $A(a)$ with $\mathsf{cert.equiv\_refl}$ before recursing on the tail witness. Symmetry unpacks the bridge witness and applies
Theorem~\ref{thm:list-classifier-symm-from-namecert} to the displayed finite-spine classifier.
Transitivity is Theorem~\ref{thm:list-history-classifier-trans-from-spine-coherence}:
the two existential witnesses may expose different finite spines for the middle
history, and Definition~\ref{def:list-spine-rep-coherence} supplies the
finite-spine classifier witness that lets
Theorem~\ref{thm:list-classifier-trans-from-namecert} compose the two endpoint
classifiers. Carrier transport for the certificate core unpacks
$\mathsf{ListHistoryClassifier}_{A,\sim_A}(h,k)$ as a $\mathsf{ListSpineBridgeClassifier}_{A,\sim_A}$, uses the displayed target representation $\mathsf{ListSpineRep}_A(k,ys)$, and applies the soundness direction of
Theorem~\ref{thm:list-spine-representation-exactness}. Source-certificate transport is used only for entrywise source evidence and ledger data along the finite-spine classifier. Nil/cons case analysis uses
Theorem~\ref{thm:list-history-carrier-generated-cases} for the generated history
carrier and Lemma~\ref{lem:list-classifier-nil-cons-no-confusion} inside the
finite-spine bridge, and exactness is the matching-shape statement of
Theorem~\ref{thm:list-classifier-case-exactness} transported through the
representation relation.
\end{proof}

\end{corollary}
