\chapter{A Concrete Naming Certificate for \(\MonoidUp\)}
\label{ch:concrete-instances-monoid-namecert}

The monoid interface \(\MonoidUp\) is licensed by an abstract carrier endowed with a binary operation \(\cdot : C \times C \to C\), an identity element \(e : C\), associativity of \(\cdot\), and left and right identity laws of \(e\) under the carrier's source classifier. It is the simplest algebraic NameCert and is self-contained: no other licensed object is required as a dependency. It is not a primitive structure; it is licensed only when the carrier's classifier admits the algebraic equational fields.

\section{Carrier definition}

\begin{definition}[Monoid carrier]
\label{def:monoid-carrier}
Use an abstract setup structure with carrier \(C\), classifier \(\sim_C\), operation \(\cdot\), identity \(e\), and law fields.
\end{definition}

\section{Concrete history instance}
\label{sec:monoid-concrete-history-instance}

\begin{definition}[Concrete history monoid instance]
\label{def:concrete-history-monoid-instance}
The BEDC unary-history monoid instance takes $Carrier := \mathsf{UnaryHistory}$,
$\mathsf{mul}(h,k) := \mathsf{append}(h,k)$, equivalently witnessed by
$\Cont(h,k,\mathsf{append}(h,k))$, $e := \emp$, and
$classifier := \mathsf{MonoidHistoryClassifier}(\mathsf{UnaryHistory})$.
\end{definition}

\begin{theorem}[Concrete history monoid laws]
\label{thm:concrete-history-monoid-laws}
The concrete instance of \autoref{def:concrete-history-monoid-instance} has:
\begin{enumerate}
  \item left unit of $\emp$ for unary histories;
  \item right unit of $\emp$ for unary histories;
  \item associativity of $\mathsf{append}$ up to $\hsame$;
  \item multiplication congruence under
  $\mathsf{MonoidHistoryClassifier}(\mathsf{UnaryHistory})$.
\end{enumerate}
\end{theorem}
\begin{proof}
The left-unit and right-unit clauses are the BEDC continuation unit laws
\texttt{BEDC.FKernel.Cont.cont\_left\_unit} and
\texttt{BEDC.FKernel.Cont.cont\_right\_unit}, restricted to unary histories.

Associativity is the continuation associativity theorem
\texttt{BEDC.FKernel.Cont.cont\_assoc\_hsame}; equivalently, the two
parenthesizations compute to the same BEDC history append tree up to $\hsame$.

For congruence, the classifier supplies unary endpoint witnesses together with
the relevant $\hsame$ transport. Unary append closure keeps both products in the
carrier, and transport of $\hsame$ across the append arguments gives the
required multiplication congruence.
\end{proof}

The $\MonoidUp$ schema is instantiated by this concrete $BHist$ carrier and
operation, so subsequent certificates can refer to this instance rather than
only to abstract operator fields.

\begin{theorem}[Unary append monoid commutative classifier]
\label{thm:unary-append-monoid-commutative-classifier}
For unary histories $a$ and $b$, the concrete monoid classifier identifies the
two append products:
$$
\mathsf{MonoidHistoryClassifier}(\mathsf{UnaryHistory})
(\mathsf{append}(a,b),\mathsf{append}(b,a)).
$$
\end{theorem}
\begin{proof}
Unary append commutativity gives
$\mathsf{append}(a,b)\hsame\mathsf{append}(b,a)$, and unary append closure
supplies the two endpoint carrier witnesses required by
$\mathsf{MonoidHistoryClassifier}(\mathsf{UnaryHistory})$.
\end{proof}
\leanchecked{BEDC.Derived.MonoidUp.unary\_append\_monoid\_commutative\_classifier}

\section{The certificate}

\begin{definition}[Monoid source specification]
\label{def:monoid-source-specification}
The source is the carrier \(C\) with its classifier and operation data.
\end{definition}

\begin{definition}[Monoid pattern specification]
\label{def:monoid-pattern-specification}
The pattern is closed expression formation from \(e\), carrier witnesses, and binary multiplication.
\end{definition}

\begin{definition}[Monoid classifier specification]
\label{def:monoid-classifier-specification}
The classifier is the supplied carrier sameness relation, required to respect multiplication.
\end{definition}
\leandef{BEDC.Derived.MonoidUp.MonoidHistoryClassifier}

\begin{definition}[Monoid stability certificate]
\label{def:monoid-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item classifier reflexivity
  \item classifier transitivity
  \item associativity under classifier
  \item left identity law
  \item right identity law
  \item multiplication congruence in both arguments
\end{enumerate}
\end{definition}
\leanchecked{BEDC.Derived.MonoidUp.monoid\_stability\_certificate\_fields}

\begin{theorem}[Monoid identity uniqueness]
\label{thm:monoid-identity-unique}
For BEDC history multiplication, if \(e\) is a left identity and \(e'\) is a
right identity under \(\hsame\), then \(e\hsame e'\).
\end{theorem}
\begin{proof}
Use the right-identity law for \(e'\) at \(e\) to identify \(e\) with
\(e\cdot e'\), then use the left-identity law for \(e\) at \(e'\).
\end{proof}
\leanchecked{BEDC.Derived.MonoidUp.monoid\_identity\_unique}

\begin{theorem}[Monoid history classifier identity uniqueness]
\label{thm:monoid-history-classifier-identity-unique}
For a BEDC history-level monoid classifier over carrier $C$, if $e,e'\in C$,
$e$ acts as a left identity and $e'$ acts as a right identity under the monoid
classifier, then the classifier identifies $e$ and $e'$.
\end{theorem}
\begin{proof}
The right-identity field at $e$ gives $e\cdot e'\hsame e$, while the
left-identity field at $e'$ gives $e\cdot e'\hsame e'$. Symmetry and
transitivity of $\hsame$ identify $e$ and $e'$, and the carrier witnesses for
$e$ and $e'$ supply the classifier endpoints.
\end{proof}
\leanchecked{BEDC.Derived.MonoidUp.MonoidHistoryClassifier\_identity\_unique}

\begin{theorem}[Unary append monoid left identity is empty]
\label{thm:unary-append-monoid-left-identity-empty}
For the concrete unary-history append monoid, any unary history $e$ whose
append action is a left identity under
$\mathsf{MonoidHistoryClassifier}(\mathsf{UnaryHistory})$ is classified with
$\emp$:
$$
\mathsf{MonoidHistoryClassifier}(\mathsf{UnaryHistory})(e,\emp).
$$
\end{theorem}
\begin{proof}
Apply classifier identity uniqueness with $e$ as the left identity and $\emp$
as the right identity. The right-identity classifier for $\emp$ follows from
unary closure of append, $\mathsf{UnaryHistory}(\emp)$, and the BEDC append law
$\mathsf{append}(h,\emp)=h$.
\end{proof}
\leanchecked{BEDC.Derived.MonoidUp.unary\_append\_monoid\_left\_identity\_empty}

\begin{theorem}[Unary append monoid right identity is empty]
\label{thm:unary-append-monoid-right-identity-empty}
For the concrete unary-history append monoid, any unary history $e$ whose
append action is a right identity under
$\mathsf{MonoidHistoryClassifier}(\mathsf{UnaryHistory})$ is classified with
$\emp$:
$$
\mathsf{MonoidHistoryClassifier}(\mathsf{UnaryHistory})(e,\emp).
$$
\end{theorem}
\begin{proof}
Apply classifier identity uniqueness with $\emp$ as the left identity and $e$
as the right identity. The left-identity classifier for $\emp$ follows from
unary closure of append, $\mathsf{UnaryHistory}(\emp)$, and the BEDC append law
$\mathsf{append}(\emp,h)=h$. Symmetry of the resulting $\hsame$ component gives
the displayed classifier orientation.
\end{proof}
\leanchecked{BEDC.Derived.MonoidUp.unary\_append\_monoid\_right\_identity\_empty}

\begin{theorem}[Unary append monoid identity is empty iff]
\label{thm:unary-append-monoid-identity-empty-iff}
For the concrete unary-history append monoid, a unary history $e$ is both a
left and right identity under
$\mathsf{MonoidHistoryClassifier}(\mathsf{UnaryHistory})$ exactly when it is
classified with $\emp$.
\end{theorem}
\begin{proof}
The forward implication uses the left-identity uniqueness result for the
concrete empty right identity. Conversely, a classifier witness
$\mathsf{MonoidHistoryClassifier}(\mathsf{UnaryHistory})(e,\emp)$ gives
$e\hsame\emp$. Replacing $e$ by $\emp$ reduces the two identity classifier
families to the BEDC append laws
$\mathsf{append}(\emp,h)=h$ and $\mathsf{append}(h,\emp)=h$, with unary closure
supplying the endpoint carrier fields.
\end{proof}
\leanchecked{BEDC.Derived.MonoidUp.unary\_append\_monoid\_identity\_empty\_iff}

\begin{theorem}[Unary append monoid idempotent iff empty]
\label{thm:unary-append-monoid-idempotent-empty-iff}
For every unary history $e$, the concrete unary append monoid classifies the
duplicated append factor as $e$ exactly when $e$ is classified with $\emp$:
$$
\mathsf{MonoidHistoryClassifier}(\mathsf{UnaryHistory})
(\mathsf{append}(e,e),e)
\quad\text{if and only if}\quad
\mathsf{MonoidHistoryClassifier}(\mathsf{UnaryHistory})(e,\emp).
$$
Here the BEDC source is $\mathsf{BHist}$ with the carrier
$\mathsf{UnaryHistory}$ and operation $\mathsf{append}$; the classifier is
$\mathsf{MonoidHistoryClassifier}(\mathsf{UnaryHistory})$. The ledger records
the duplicated factor $e$ and the empty right unit, stability is supplied by
common-context congruence and cancellation, exactness is the displayed iff, and
the usual finite unary-word monoid is the standard bridge.
\end{theorem}
\begin{proof}
For the forward direction, rewrite the right endpoint $e$ as
$\mathsf{append}(e,\emp)$ by the empty right-unit law. The hypothesis then
classifies two histories with common left context $e$ and common right context
$\emp$, with middle endpoints $e$ and $\emp$. Apply
Theorem~\ref{thm:unary-append-monoid-classifier-cancel-context} to obtain
$\mathsf{MonoidHistoryClassifier}(\mathsf{UnaryHistory})(e,\emp)$.

For the reverse direction, start from the classifier witness
$\mathsf{MonoidHistoryClassifier}(\mathsf{UnaryHistory})(e,\emp)$. Apply
Theorem~\ref{thm:unary-append-monoid-classifier-append-context} with left
context $e$ and right context $\emp$ to classify the contextual pair
$\mathsf{append}(e,\mathsf{append}(e,\emp))$ and
$\mathsf{append}(e,\mathsf{append}(\emp,\emp))$. The empty right-unit and
left-unit laws identify these endpoints with $\mathsf{append}(e,e)$ and $e$,
giving the displayed classifier.
\end{proof}
\leanchecked{BEDC.Derived.MonoidUp.unary\_append\_monoid\_idempotent\_empty\_iff}

\begin{theorem}[History continuation monoid laws]
\label{thm:history-continuation-monoid-laws}
The BEDC history continuation relation has empty history as a left and right unit, and the two parenthesizations of a triple continuation have the same resulting history.
\end{theorem}
\leanchecked{BEDC.Derived.MonoidUp.history\_continuation\_monoid\_laws}
\begin{proof}
The unit fields are the empty-history continuation constructors. For associativity, each continuation witness unfolds to history append, and the result follows from associativity of BEDC history append.
\end{proof}

\begin{theorem}[Nonempty continuation suffix changes the source]
\label{thm:history-continuation-nonempty-suffix-source-absurd}
For any source history \(h\), no continuation by a suffix of the form \(\Ezero(k)\)
or \(\Eone(k)\) can have result \(h\) itself.
\end{theorem}
\leanchecked{BEDC.Derived.MonoidUp.history\_continuation\_nonempty\_suffix\_source\_absurd}
\begin{proof}
Right-unit uniqueness for continuation forces any suffix whose result is the
source to be \(\emp\). BEDC history no-confusion separates \(\Ezero(k)\) and
\(\Eone(k)\) from \(\emp\).
\end{proof}

\begin{theorem}[Nonempty continuation prefix changes the target]
\label{thm:history-continuation-nonempty-prefix-target-absurd}
For any target history $k$, no continuation with source of the form
$\Ezero(h)$ or $\Eone(h)$ can have result $k$ itself.
\end{theorem}
\begin{proof}
Left-unit uniqueness for continuation forces any source whose continuation
leaves the target unchanged to be $\emp$. BEDC history no-confusion separates
$\Ezero(h)$ and $\Eone(h)$ from $\emp$.
\end{proof}
\leanchecked{BEDC.Derived.MonoidUp.history\_continuation\_nonempty\_prefix\_target\_absurd}

\begin{theorem}[Unary append monoid semantic name certificate]
\label{thm:unary-append-monoid-semantic-name-certificate}
The unary-history carrier, empty history, BEDC history append, and the
history-sameness classifier instantiate the semantic naming certificate for
\(\MonoidUp\). The certificate exports the empty-left, empty-right,
associativity, and append-congruence fields under the classifier
\(\mathsf{MonoidHistoryClassifier}(\mathsf{UnaryHistory})\).
\end{theorem}
\begin{proof}
Unary history is inhabited by empty history and is closed under BEDC append.
The empty-history laws and associativity are the primitive append laws for
histories. Congruence follows by transporting both append arguments across
\(\hsame\), leaving the same appended history on both sides.
\end{proof}
\leanchecked{BEDC.Derived.MonoidUp.unary\_append\_monoid\_semantic\_name\_certificate}

\begin{theorem}[Unary append monoid classifier appends common context]
\label{thm:unary-append-monoid-classifier-append-context}
For unary histories $l,r$, if the monoid classifier identifies $a$ with $b$,
then it identifies the two contextual histories
$$
l(a r)
$$
and
$$
l(b r).
$$
\end{theorem}
\begin{proof}
The classifier supplies unary witnesses for $a$ and $b$ together with
$a\hsame b$. Unary append closure builds the two contextual endpoint carriers.
After the endpoint equality is exposed, reflexivity of $\hsame$ identifies the
two appended histories.
\end{proof}
\leanchecked{BEDC.Derived.MonoidUp.unary\_append\_monoid\_classifier\_append\_context}

\begin{theorem}[Unary append monoid classifier preserves context]
\label{thm:unary-append-monoid-classifier-context-congr}
For unary histories $l,r,a,b$, if the monoid classifier identifies $a$ with
$b$, then it identifies the common-context appends
$$
l(a r)
$$
and
$$
l(b r).
$$
\end{theorem}
\begin{proof}
The classifier gives unary endpoint witnesses for $a$ and $b$ and an $\hsame$
witness between them. Unary histories are closed under BEDC append, so appending
the common right and left contexts supplies the endpoint carriers. The $\hsame$
witness identifies the middle history, leaving equal contextual appends.
\end{proof}
\leanchecked{BEDC.Derived.MonoidUp.unary\_append\_monoid\_classifier\_context\_congr}

\begin{theorem}[Unary append monoid classifier cancels common context]
\label{thm:unary-append-monoid-classifier-cancel-context}
For unary histories $l,r,a,b$, if the monoid classifier identifies
$$
l(a r)
$$
with
$$
l(b r),
$$
then it identifies $a$ with $b$.
\end{theorem}
\begin{proof}
The classifier supplies the endpoint unary witnesses and a BEDC history equality
between the two contextual appends. Left append cancellation removes the common
left context $l$, and right append cancellation removes the common suffix $r$.
The given unary witnesses for $a$ and $b$ supply the classifier endpoints.
\end{proof}
\leanchecked{BEDC.Derived.MonoidUp.unary\_append\_monoid\_classifier\_cancel\_context}

\begin{theorem}[Unary append monoid classifier context equivalence]
\label{thm:unary-append-monoid-classifier-context-iff}
For unary histories $l,r,a,b$, the monoid classifier identifies
$$
l(a r)
$$
with
$$
l(b r)
$$
exactly when it identifies $a$ with $b$.
\end{theorem}
\begin{proof}
The forward direction is common-context cancellation: left append cancellation
removes $l$, and right append cancellation removes $r$. The reverse direction
is common-context congruence using unary append closure on both endpoints.
\end{proof}
\leanchecked{BEDC.Derived.MonoidUp.unary\_append\_monoid\_classifier\_context\_iff}

\begin{theorem}[Concrete \(\MonoidUp\) naming certificate]
\label{thm:concrete-monoid-namecert}
Any setup satisfying the monoid fields forms a valid NameCert for \(\MonoidUp(C)\).
\end{theorem}
\begin{proof}
The identity element inhabits the carrier. The monoid classifier records source and target carrier witnesses together with the BEDC history equality witness, giving the certificate core by \(\hsame\) reflexivity, symmetry, transitivity, and carrier projection. Closure under multiplication supplies carrier witnesses for the exported identity, associativity, and multiplication-congruence fields.
\end{proof}
