\chapter{A Concrete Naming Certificate for \(\MonoidUp\)}
\label{ch:concrete-instances-monoid-namecert}

The monoid interface \(\MonoidUp\) is licensed by an abstract carrier endowed with a binary operation \(\cdot : C \times C \to C\), an identity element \(e : C\), associativity of \(\cdot\), and left and right identity laws of \(e\) under the carrier's source classifier. It is the simplest algebraic NameCert and is self-contained: no other licensed object is required as a dependency. It is not a primitive structure; it is licensed only when the carrier's classifier admits the algebraic equational fields.

\section{Carrier definition}

\begin{definition}[Monoid carrier]
\label{def:monoid-carrier}
Use an abstract setup structure with carrier \(C\), classifier \(\sim_C\), operation \(\cdot\), identity \(e\), and law fields.
\end{definition}

\section{Concrete history instance}
\label{sec:monoid-concrete-history-instance}

\begin{definition}[Concrete history monoid instance]
\label{def:concrete-history-monoid-instance}
The BEDC unary-history monoid instance takes $Carrier := \mathsf{UnaryHistory}$,
$\mathsf{mul}(h,k) := \mathsf{append}(h,k)$, equivalently witnessed by
$\Cont(h,k,\mathsf{append}(h,k))$, $e := \emp$, and
$classifier := \mathsf{MonoidHistoryClassifier}(\mathsf{UnaryHistory})$.
\end{definition}

\begin{theorem}[Concrete history monoid laws]
\label{thm:concrete-history-monoid-laws}
The concrete instance of \autoref{def:concrete-history-monoid-instance} has:
\begin{enumerate}
  \item left unit of $\emp$ for unary histories;
  \item right unit of $\emp$ for unary histories;
  \item associativity of $\mathsf{append}$ up to $\hsame$;
  \item multiplication congruence under
  $\mathsf{MonoidHistoryClassifier}(\mathsf{UnaryHistory})$.
\end{enumerate}
\end{theorem}
\leanchecked{BEDC.Derived.MonoidUp.concrete\_history\_monoid\_laws}
\begin{proof}
The left-unit and right-unit clauses are the BEDC continuation unit laws
\texttt{BEDC.FKernel.Cont.cont\_left\_unit} and
\texttt{BEDC.FKernel.Cont.cont\_right\_unit}, restricted to unary histories.

Associativity is the continuation associativity theorem
\texttt{BEDC.FKernel.Cont.cont\_assoc\_hsame}; equivalently, the two
parenthesizations compute to the same BEDC history append tree up to $\hsame$.

For congruence, the classifier supplies unary endpoint witnesses together with
the relevant $\hsame$ transport. Unary append closure keeps both products in the
carrier, and transport of $\hsame$ across the append arguments gives the
required multiplication congruence.
\end{proof}

The $\MonoidUp$ schema is instantiated by this concrete $BHist$ carrier and
operation, so subsequent certificates can refer to this instance rather than
only to abstract operator fields.

\begin{theorem}[Unary append monoid commutative classifier]
\label{thm:unary-append-monoid-commutative-classifier}

$$
\mathsf{MonoidHistoryClassifier}(\mathsf{UnaryHistory})
(\mathsf{append}(a,b),\mathsf{append}(b,a)).
$$
\end{theorem}
\begin{proof}
Unary append commutativity gives
$\mathsf{append}(a,b)\hsame\mathsf{append}(b,a)$, and unary append closure
supplies the two endpoint carrier witnesses required by
$\mathsf{MonoidHistoryClassifier}(\mathsf{UnaryHistory})$.
\end{proof}
\leanchecked{BEDC.Derived.MonoidUp.unary\_append\_monoid\_commutative\_classifier}

\begin{theorem}[Monoid identity uniqueness]
\label{thm:monoid-identity-unique}
For BEDC history multiplication, if \(e\) is a left identity and \(e'\) is a
right identity under \(\hsame\), then \(e\hsame e'\).
\end{theorem}
\begin{proof}
Use the right-identity law for \(e'\) at \(e\) to identify \(e\) with
\(e\cdot e'\), then use the left-identity law for \(e\) at \(e'\).
\end{proof}
\leanchecked{BEDC.Derived.MonoidUp.monoid\_identity\_unique}

\begin{theorem}[Monoid history classifier identity uniqueness]
\label{thm:monoid-history-classifier-identity-unique}
For a BEDC history-level monoid classifier over carrier $C$, if $e,e'\in C$,
$e$ acts as a left identity and $e'$ acts as a right identity under the monoid
classifier, then the classifier identifies $e$ and $e'$.
\end{theorem}
\begin{proof}
The right-identity field at $e$ gives $e\cdot e'\hsame e$, while the
left-identity field at $e'$ gives $e\cdot e'\hsame e'$. Symmetry and
transitivity of $\hsame$ identify $e$ and $e'$, and the carrier witnesses for
$e$ and $e'$ supply the classifier endpoints.
\end{proof}

\begin{theorem}[Unary append monoid idempotent iff empty]
\label{thm:unary-append-monoid-idempotent-empty-iff}
For every unary history $e$, the concrete unary append monoid classifies the
duplicated append factor as $e$ exactly when $e$ is classified with $\emp$:
$$
\mathsf{MonoidHistoryClassifier}(\mathsf{UnaryHistory})
(\mathsf{append}(e,e),e)
\quad\text{if and only if}\quad
\mathsf{MonoidHistoryClassifier}(\mathsf{UnaryHistory})(e,\emp).
$$
Here the BEDC source is $\mathsf{BHist}$ with the carrier
$\mathsf{UnaryHistory}$ and operation $\mathsf{append}$; the classifier is
$\mathsf{MonoidHistoryClassifier}(\mathsf{UnaryHistory})$. The ledger records
the duplicated factor $e$ and the empty right unit, stability is supplied by
common-context congruence and cancellation, exactness is the displayed iff, and
the usual finite unary-word monoid is the standard bridge.
\end{theorem}
\begin{proof}
For the forward direction, rewrite the right endpoint $e$ as
$\mathsf{append}(e,\emp)$ by the empty right-unit law. The hypothesis then
classifies two histories with common left context $e$ and common right context
$\emp$, with middle endpoints $e$ and $\emp$. Apply
Theorem~\ref{thm:unary-append-monoid-classifier-cancel-context} to obtain
$\mathsf{MonoidHistoryClassifier}(\mathsf{UnaryHistory})(e,\emp)$.

For the reverse direction, start from the classifier witness
$\mathsf{MonoidHistoryClassifier}(\mathsf{UnaryHistory})(e,\emp)$. Apply
Theorem~\ref{thm:unary-append-monoid-classifier-append-context} with left
context $e$ and right context $\emp$ to classify the contextual pair
$\mathsf{append}(e,\mathsf{append}(e,\emp))$ and
$\mathsf{append}(e,\mathsf{append}(\emp,\emp))$. The empty right-unit and
left-unit laws identify these endpoints with $\mathsf{append}(e,e)$ and $e$,
giving the displayed classifier.
\end{proof}
\leanchecked{BEDC.Derived.MonoidUp.unary\_append\_monoid\_idempotent\_empty\_iff}

\begin{theorem}[Unary append monoid one-sided inverse obstruction]
\label{thm:unary-append-monoid-one-sided-inverse-obstruction}
For every operation $\mathsf{inv}:\mathsf{BHist}\to\mathsf{BHist}$, the
concrete unary-history append monoid supports neither a global right-inverse
family
$$
\forall h:\mathsf{BHist},\
\mathsf{UnaryHistory}(h)\Rightarrow
\mathsf{MonoidHistoryClassifier}(\mathsf{UnaryHistory})
(\mathsf{append}(h,\mathsf{inv}(h)),\emp),
$$
nor a global left-inverse family
$$
\forall h:\mathsf{BHist},\
\mathsf{UnaryHistory}(h)\Rightarrow
\mathsf{MonoidHistoryClassifier}(\mathsf{UnaryHistory})
(\mathsf{append}(\mathsf{inv}(h),h),\emp).
$$

The ten-field fit is: traditional target group-upgrade obstruction for the
unary append monoid; BEDC source $\mathsf{BHist}$, $\mathsf{UnaryHistory}$,
$\mathsf{append}$, and
$\mathsf{MonoidHistoryClassifier}(\mathsf{UnaryHistory})$; carrier
$\mathsf{UnaryHistory}$; classifier the concrete monoid classifier; stability
from unary append closure and empty-result inversion; ledger the attempted
inverse endpoint; core theorem this block; exactness the contradiction from
the visible unary generator; standard bridge finite unary-word monoid has no
global inverse; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
For the right-inverse family, take the visible unary generator
$u=\Eone(\emp)$. Unary closure gives $\mathsf{UnaryHistory}(u)$, so the
assumed family classifies
$\mathsf{append}(u,\mathsf{inv}(u))$ with $\emp$. Unfolding the concrete
classifier exposes
$\mathsf{append}(u,\mathsf{inv}(u))\hsame\emp$.

Empty-result inversion for append then forces the left factor to satisfy
$u\hsame\emp$. This contradicts the BEDC no-confusion theorem separating the
$\Eone$ constructor from $\emp$.

The left-inverse family is symmetric in the obstruction. Specializing it at
the same $u$ exposes
$\mathsf{append}(\mathsf{inv}(u),u)\hsame\emp$. Empty-result inversion now
forces the right factor $u$ to be $\hsame$ to $\emp$, and the same
no-confusion contradiction follows.
\end{proof}
\leanchecked{BEDC.Derived.MonoidUp.unary\_append\_monoid\_one\_sided\_inverse\_law\_absurd}

\begin{definition}[Opposite monoid certificate data]
\label{def:opposite-monoid-certificate-data}
Let a $\MonoidUp$ certificate be presented by a carrier predicate $\mathcal C$, classifier $\sim_C$, multiplication $\cdot_C$, and identity $e_C$. Define the reversed multiplication by
$$
a\cdot_C^{\mathrm{op}} b:=b\cdot_C a.
$$
The opposite monoid data $C^{\mathrm{op}}$ have the same carrier predicate $\mathcal C$, the same classifier $\sim_C$, the same identity $e_C$, and multiplication $\cdot_C^{\mathrm{op}}$.
\end{definition}

\begin{theorem}[Opposite monoid certificate]
\label{thm:opposite-monoid-certificate}
If $(\mathcal C,\sim_C,\cdot_C,e_C)$ satisfies the $\MonoidUp$ certificate obligations of \autoref{def:monoid-certificate-obligations}, then the data of \autoref{def:opposite-monoid-certificate-data} also satisfy the $\MonoidUp$ certificate obligations. Thus the same carrier, identity, and classifier carry a monoid certificate with multiplication $\cdot_C^{\mathrm{op}}$.
\end{theorem}
\leanchecked{BEDC.Derived.MonoidUp.unary\_append\_monoid\_opposite\_semantic\_name\_certificate}
\begin{proof}
Unpack the monoid obligations through \autoref{def:monoid-certificate-obligations}. The carrier row of \autoref{def:monoid-carrier} is unchanged. The carrier-and-classifier core is the same $\NameCert$ data, so classifier reflexivity, symmetry, transitivity, and carrier transport are copied from \autoref{def:lean-carrier-naming-certificate}. The source row of \autoref{def:monoid-source-specification} is the same carrier and classifier equipped with the reversed multiplication symbol.

Closure for the reversed multiplication follows from original closure with the two inputs exchanged: if $\mathcal C(a)$ and $\mathcal C(b)$, then
$$
a\cdot_C^{\mathrm{op}} b=b\cdot_C a
$$
is carried because $b$ and $a$ are carried.

The identity rows are exchanged. The opposite left identity law is
$$
e_C\cdot_C^{\mathrm{op}} a=a\cdot_C e_C\sim_C a,
$$
which is the original right identity row. The opposite right identity law is
$$
a\cdot_C^{\mathrm{op}} e_C=e_C\cdot_C a\sim_C a,
$$
which is the original left identity row.

For associativity, the two opposite bracketings unfold to
$$
(a\cdot_C^{\mathrm{op}} b)\cdot_C^{\mathrm{op}} c=c\cdot_C(b\cdot_C a)
$$
and
$$
a\cdot_C^{\mathrm{op}}(b\cdot_C^{\mathrm{op}} c)=(c\cdot_C b)\cdot_C a.
$$
The associativity row in \autoref{def:monoid-stability-certificate}, applied to the original multiplication at the triple $(c,b,a)$, supplies
$$
(c\cdot_C b)\cdot_C a\sim_C c\cdot_C(b\cdot_C a).
$$
Classifier symmetry gives the required opposite orientation
$$
c\cdot_C(b\cdot_C a)\sim_C(c\cdot_C b)\cdot_C a.
$$

Multiplication congruence is also inherited with inputs exchanged. Given $a\sim_C a'$ and $b\sim_C b'$, the required comparison for the opposite product is
$$
a\cdot_C^{\mathrm{op}} b=b\cdot_C a\sim_C b'\cdot_C a'=a'\cdot_C^{\mathrm{op}} b',
$$
which is the original multiplication congruence applied to $b\sim_C b'$ and $a\sim_C a'$.

The classifier row of \autoref{def:monoid-classifier-specification} is unchanged because endpoint comparison is still $\sim_C$; the congruence paragraph is exactly the respect-for-multiplication field for the reversed operation. The pattern row of \autoref{def:monoid-pattern-specification} remains closed expression formation from $e_C$, carried endpoints, and binary multiplication, with each reversed multiplication node read as an original multiplication node whose children are exchanged. The ledger row of \autoref{def:monoid-ledger-policy} records the same expression tree with that exchanged child order at every reversed multiplication node. Hence the source-to-pattern and source-to-ledger soundness fields are the original operation-tree soundness checks read through the definition of $\cdot_C^{\mathrm{op}}$.

These carrier, source, pattern, classifier, stability, and ledger rows are precisely the five-field naming-certificate data required by \autoref{def:typed-naming-certificate} and the monoid obligations of \autoref{def:monoid-certificate-obligations}. Therefore $C^{\mathrm{op}}$ carries a $\MonoidUp$ certificate.
\end{proof}

\begin{theorem}[Double opposite monoid certificate data identity]
\label{thm:double-opposite-monoid-certificate-data-identity}
Let $D$ be a $\MonoidUp$ certificate satisfying the obligations of
\autoref{def:monoid-certificate-obligations}. Present its monoid data by a
carrier predicate $\mathcal C$, classifier $\sim_C$, multiplication $\cdot_C$,
and identity element $e_C$ as in \autoref{def:opposite-monoid-certificate-data}.
Let
$$
D^{\mathrm{opop}}=(D^{\mathrm{op}})^{\mathrm{op}}
$$
be obtained by applying \autoref{def:opposite-monoid-certificate-data} twice.
Then $D^{\mathrm{opop}}$ is a $\MonoidUp$ certificate, and its carrier
predicate, classifier, multiplication, and identity element are componentwise
identical to those of $D$:
$$
\begin{aligned}
\mathcal C^{\mathrm{opop}}(a)&=\mathcal C(a),&
 a\sim_C^{\mathrm{opop}} b&=a\sim_C b,\\
a\cdot_C^{\mathrm{opop}} b&=a\cdot_C b,&
 e_C^{\mathrm{opop}}&=e_C.
\end{aligned}
$$
\end{theorem}
\begin{proof}
Because $D$ satisfies the $\MonoidUp$ obligations,
\autoref{thm:opposite-monoid-certificate} makes $D^{\mathrm{op}}$ a certified
$\MonoidUp$ instance. Applying the same theorem to that certified opposite data
makes $D^{\mathrm{opop}}$ a certified $\MonoidUp$ instance.

It remains to compare the displayed data. The carrier predicate, classifier,
and identity element are fixed by one use of
\autoref{def:opposite-monoid-certificate-data}. Hence two unfoldings give
$$
\mathcal C^{\mathrm{opop}}(a)=\mathcal C^{\mathrm{op}}(a)=\mathcal C(a),
$$
$$
a\sim_C^{\mathrm{opop}} b=a\sim_C^{\mathrm{op}} b=a\sim_C b,
$$
and
$$
e_C^{\mathrm{opop}}=e_C^{\mathrm{op}}=e_C.
$$

For multiplication, the first opposite exchanges the two inputs and the second
opposite exchanges them back:
$$
a\cdot_C^{\mathrm{opop}} b
=b\cdot_C^{\mathrm{op}} a
=a\cdot_C b.
$$
Thus the raw multiplication field of the double opposite certificate is exactly
the original multiplication field.

The source row of \autoref{def:monoid-source-specification} uses only the
carrier, classifier, and multiplication just identified. The pattern row of
\autoref{def:monoid-pattern-specification} forms expressions from the same
identity element and the same restored multiplication. The classifier row of
\autoref{def:monoid-classifier-specification} is the same endpoint relation,
and multiplication respect is typed over the restored multiplication. The
ledger policy of \autoref{def:monoid-ledger-policy} records operation trees;
one opposite reading swaps the two children at each multiplication node, and
the second reading swaps the same children back. Therefore the displayed
component identities identify the double opposite monoid certificate data with
the monoid data of $D$.
\end{proof}
