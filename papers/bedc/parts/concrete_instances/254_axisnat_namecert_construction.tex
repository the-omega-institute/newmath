\chapter{A Concrete Naming Certificate for \(\AxisNatUp\)}
\label{ch:concrete-instances-axisnat-namecert}

This chapter gives a concrete naming-certificate construction for \(\AxisNatUp\) at the paper-proof level. The interface \(\AxisNatUp\) is a conservative parallel to \(\NatUp\) (Chapter~\ref{ch:concrete-instances-nat-namecert}): it is a finite-repetition counting seed whose carrier is the \(\Ezero\)-spine, not the \(\Eone\)-spine that \(\NatUp\) uses. The two interfaces are kernel-distinct; their relation is mediated by the bridge of Chapter~\ref{ch:concrete-instances-unary-direction-bridge}.

\section{The \(\Ezero\)-spine carrier}

\begin{definition}[Zero-spine history judgment]
\label{def:zero-spine-history-judgment}
The zero-spine history judgment \(\mathsf{ZeroSpine}(h)\) is generated by:
\begin{enumerate}
  \item \(\mathsf{ZeroSpine}(\emp)\);
  \item if \(\mathsf{ZeroSpine}(h)\), then \(\mathsf{ZeroSpine}(\Ezero(h))\).
\end{enumerate}
No rule generates \(\mathsf{ZeroSpine}(\Eone(h))\).
\end{definition}
\leandef{BEDC.Derived.AxisZeckendorf.Spine.ZeroSpine}

\begin{lemma}[Zero-spine empty base]
\label{lem:zero-spine-empty-base}
\(\mathsf{ZeroSpine}(\emp)\) holds.
\end{lemma}
\leandef{BEDC.Derived.AxisZeckendorf.Spine.zeroSpine\_empty}
\begin{proof}
First generator of the zero-spine judgment.
\end{proof}

\begin{lemma}[Zero-spine closure under zero-extension]
\label{lem:zero-spine-closure-under-zero-extension}
If \(\mathsf{ZeroSpine}(h)\), then \(\mathsf{ZeroSpine}(\Ezero(h))\).
\end{lemma}
\leandef{BEDC.Derived.AxisZeckendorf.Spine.zeroSpine\_e0\_closed}
\begin{proof}
Second generator of the zero-spine judgment.
\end{proof}

\begin{lemma}[Zero-spine zero-extension inversion]
\label{lem:zero-spine-e0-inversion}
If $\mathsf{ZeroSpine}(\Ezero(h))$, then $\mathsf{ZeroSpine}(h)$.
\end{lemma}
\leanchecked{BEDC.Derived.AxisZeckendorf.Spine.zeroSpine\_e0\_inversion}
\begin{proof}
Proceed by cases on the closed generation of $\mathsf{ZeroSpine}(\Ezero(h))$.

The empty constructor cannot produce an $\Ezero$ endpoint. The zero-extension constructor is the only matching generated shape, and it exposes exactly the predecessor witness $\mathsf{ZeroSpine}(h)$.
\end{proof}

\begin{lemma}[Zero-spine no one-extension]
\label{lem:zero-spine-no-one-extension}
No judgment \(\mathsf{ZeroSpine}(\Eone(h))\) is generated.
\end{lemma}
\leandef{BEDC.Derived.AxisZeckendorf.Spine.zeroSpine\_no\_e1\_extension}
\begin{proof}
By closed generation for \(\mathsf{ZeroSpine}\), the only extension rule uses \(\Ezero\). No \(\Eone\)-extension constructor is displayed.
\end{proof}

\begin{lemma}[Zero-spine \(\HSame\) transport]
\label{lem:zero-spine-hsame-transport}
If \(\mathsf{ZeroSpine}(h)\) and \(\HSame(h, k)\), then \(\mathsf{ZeroSpine}(k)\).
\end{lemma}
\leandef{BEDC.Derived.AxisZeckendorf.Spine.zeroSpine\_hsame\_transport}
\begin{proof}
\(\HSame\) is propositional history identity, so the witness transports along the equality.
\end{proof}

\section{The certificate}

\begin{definition}[Zero-spine source specification]
\label{def:axisnat-source-specification}
The source specification \(\mathsf{AxisNatSource}\) consists of histories \(h\) with \(\mathsf{ZeroSpine}(h)\).
\end{definition}
\leandef{BEDC.Derived.AxisZeckendorf.AxisNat.AxisNatSourceSpec}

\begin{definition}[Zero-spine pattern specification]
\label{def:axisnat-pattern-specification}
The pattern specification \(\mathsf{AxisNatPattern}\) is repeated generation by the zero-extension constructor \(\Ezero\) from \(\emp\).
\end{definition}
\leandef{BEDC.Derived.AxisZeckendorf.AxisNat.AxisNatPatternSpec}

\begin{definition}[Zero-spine classifier specification]
\label{def:axisnat-classifier-specification}
The classifier specification \(\mathsf{AxisNatClassifier}\) classifies zero-spine histories pairwise by \(\HSame\) restricted to the zero-spine carrier.
\end{definition}
\leandef{BEDC.Derived.AxisZeckendorf.AxisNat.AxisNatClassifierSpec}

\begin{definition}[Zero-spine stability certificate]
\label{def:axisnat-stability-certificate}
The stability certificate \(\mathsf{AxisNatStability}\) consists of:
\begin{enumerate}
  \item the base derivation \(\mathsf{ZeroSpine}(\emp)\);
  \item closure under \(\Ezero\);
  \item no one-extension, recording that \(\Eone\) is not part of the zero-spine generation discipline.
\end{enumerate}
\end{definition}
\leandef{BEDC.Derived.AxisZeckendorf.AxisNat.AxisNatStabilityCert}
\leanvariant{BEDC.Derived.AxisZeckendorf.AxisNat.axisNatStabilityCert}

\begin{definition}[Zero-spine ledger policy]
\label{def:axisnat-ledger-policy}
The ledger policy \(\mathsf{AxisNatLedger}\) records the source zero-spine history behind any visible zero-spine class. It does not erase the emitted history; it licenses a derived interface while retaining provenance.
\end{definition}
\leandef{BEDC.Derived.AxisZeckendorf.AxisNat.AxisNatLedgerPolicy}

\begin{theorem}[Concrete \(\AxisNatUp\) naming certificate]
\label{thm:concrete-axisnat-namecert}
The tuple
$$
  \mathsf{NameCert}_{\AxisNatUp}
  :=
  (\mathsf{AxisNatSource},
   \mathsf{AxisNatPattern},
   \mathsf{AxisNatClassifier},
   \mathsf{AxisNatStability},
   \mathsf{AxisNatLedger})
$$
forms a valid naming certificate for the derived interface \(\AxisNatUp\).
\end{theorem}
\leandef{BEDC.Derived.AxisZeckendorf.AxisNat.axisNat\_namecert}
\begin{proof}
The source, pattern, classifier, stability, and ledger fields are explicitly supplied by the preceding definitions. The stability field is justified by zero-spine base generation, zero-spine closure under \(\Ezero\), and zero-spine no-one-extension. Therefore the naming protocol is satisfied.
\end{proof}

\begin{corollary}[\(\AxisNatUp\) is licensed, not primitive]
\label{cor:axisnatup-licensed-not-primitive}
The interface \(\AxisNatUp\) is licensed by \(\mathsf{NameCert}_{\AxisNatUp}\). It is not a primitive object of the finite kernel.
\end{corollary}

\section{Distinction from \(\NatUp\)}

\begin{theorem}[Zero-spine and unary spine are kernel-distinct]
\label{thm:axisnat-natup-kernel-distinct}
The \(\Ezero\)-spine carrier \(\mathsf{ZeroSpine}\) and the \(\Eone\)-spine carrier \(\mathsf{Unary}\) intersect, as history predicates, only at the empty history \(\emp\).
\end{theorem}
\leanchecked{BEDC.Derived.AxisZeckendorf.AxisNat.ZeroSpine\_unaryHistory\_intersection\_empty}

\begin{proof}
A zero-spine history is either \(\emp\) or has the form \(\Ezero(h)\); a unary history is either \(\emp\) or has the form \(\Eone(h)\). The two non-empty shapes are \(\Ezero\)-headed and \(\Eone\)-headed respectively, so a non-empty history cannot satisfy both judgments.
\end{proof}

\begin{remark}[Bridge instead of identification]
The kernel-distinct theorem above ensures that any identification of \(\AxisNatUp\) with \(\NatUp\) must come from a separately certified bridge, not from internal kernel reasoning. Such a bridge is the subject of Chapter~\ref{ch:concrete-instances-unary-direction-bridge}.
\end{remark}

\begin{theorem}[Zero-spine source shape exhaustion]
\label{thm:zero-spine-source-shape-exhaustion}
For every history $h:\mathsf{BHist}$, if $\mathsf{ZeroSpine}(h)$, then
$$
\HSame(h,\emp)\vee
\exists h_{\mathrm{pred}}:\mathsf{BHist}.\;
\HSame(h,\Ezero(h_{\mathrm{pred}}))\land
\mathsf{ZeroSpine}(h_{\mathrm{pred}}).
$$
\end{theorem}
\begin{proof}
Induct on the displayed generation of $\mathsf{ZeroSpine}(h)$ from \autoref{def:zero-spine-history-judgment}.

In the empty-generator case, $h$ is $\emp$. The left disjunct follows from the empty generator for $\HSame(\emp,\emp)$.

In the zero-extension generator case, the derivation has the displayed form $h=\Ezero(p)$ together with a predecessor derivation $\mathsf{ZeroSpine}(p)$. Choose $h_{\mathrm{pred}}:=p$ in the right disjunct. The history-sameness component is $\HSame(\Ezero(p),\Ezero(p))$, obtained by applying the $\Ezero$ history-sameness constructor to reflexivity on $p$, and the remaining component is exactly the predecessor derivation $\mathsf{ZeroSpine}(p)$.
\end{proof}
