
\begin{lemma}[Padded coefficient extensionality for the polynomial classifier]
\label{lem:padded-coefficient-extensionality-polynomial-classifier}
Let $R$ carry a $\RingUp(R)$ certificate with classifier $\sim_R$, zero endpoint $0_R$, scalar certificate $\mathsf{cert}_R:\NameCert(R,\sim_R)$, and canonical trim $\mathsf{trim}_R$ from \autoref{def:polynomial-pattern-specification} and \autoref{def:polynomial-stability-certificate}. If finite coefficient spines $x,y:\mathsf{ListCarrier}(R)$ satisfy $x[n]\sim_R y[n]$ for every $n:\NatUp$, then
$$
\mathsf{PolySame}_R(x,y).
$$
\end{lemma}
\begin{proof}
Put $x_0:=\mathsf{trim}_R(x)$ and $y_0:=\mathsf{trim}_R(y)$. The trailing-zero removal clause in \autoref{def:polynomial-stability-certificate} gives decompositions $x=x_0{++}t_x$ and $y=y_0{++}t_y$, with $\mathsf{ZeroRem}_R(t_x)$ and $\mathsf{ZeroRem}_R(t_y)$ in the sense of \autoref{def:classified-zero-remainder-spine}. Hence $x[n]\sim_R x_0[n]$ and $y[n]\sim_R y_0[n]$ for every $n$: inside the trimmed prefix the coefficient is unchanged, inside the removed tail the zero-remainder witness classifies the coefficient with $0_R$, and after the whole spine both sides are the padding coefficient $0_R$. Classifier symmetry and transitivity from $\mathsf{cert}_R$ give $x_0[n]\sim_R y_0[n]$ for every $n$.

The trimmed spines are normalized representatives. If one normalized representative had a last displayed coefficient beyond the length of the other, the padded coefficient comparison at that last index would classify that displayed coefficient with $0_R$, contradicting the normalized no-trailing-zero condition in \autoref{def:polynomial-stability-certificate}. Thus $x_0$ and $y_0$ have the same finite length. Induction over this common length builds a structural classifier witness: at each head position the hypothesis $x_0[n]\sim_Ry_0[n]$ supplies the scalar head comparison, and the induction hypothesis supplies the tail classifier. Therefore
$$
\mathsf{ListClassifierSpec}(\sim_R,x_0,y_0).
$$
Unfolding $\mathsf{PolySame}_R$ by \autoref{def:polynomial-raw-add-comparison-data} gives the displayed polynomial classifier comparison.
\end{proof}

\begin{lemma}[Raw Cauchy product coefficient formula]
\label{lem:raw-cauchy-product-coefficient-formula}
Let $R$ carry a $\CommRingUp(R)$ certificate with classifier $\sim_R$, zero endpoint $0_R$, addition $+$, multiplication $\cdot$, and the raw operations $\mathsf{radd}_R$, $\mathsf{rscale}_R$, $\mathsf{shift}_R$, and $\mathsf{rmul}_R$ from \autoref{def:polynomial-raw-add-comparison-data} and \autoref{def:polynomial-raw-cauchy-product}. For finite coefficient spines $x,y:\mathsf{ListCarrier}(R)$ and every $n:\NatUp$,
$$
\mathsf{rmul}_R(x,y)[n]\sim_R\mathsf{Cauchy}_R(x,y;n).
$$
\end{lemma}
\begin{proof}
Read the scalar source as a $\RingUp(R)$ source by \autoref{prop:commring-forgets-ring-certificate}. The recursion defining $\mathsf{radd}_R$ gives the padded coefficient formula $\mathsf{radd}_R(u,v)[n]\sim_Ru[n]+v[n]$ by induction on the two finite spines. The definitions of $\mathsf{rscale}_R$ and $\mathsf{shift}_R$ give
$$
\mathsf{rscale}_R(a,u)[n]\sim_Ra\cdot u[n],\qquad \mathsf{shift}_R(u)[0]\sim_R0_R,\qquad \mathsf{shift}_R(u)[n+1]\sim_Ru[n].
$$
These comparisons use only zero self-classification, additive congruence, and the additive zero laws from \autoref{def:ring-stability-certificate}.

Induct on the first spine $x$. If $x=\mathsf{nil}$, the raw product is $\mathsf{nil}$ by \autoref{def:polynomial-raw-cauchy-product}. Every summand in $\mathsf{Cauchy}_R(\mathsf{nil},y;n)$ has the form $0_R\cdot y[j]$ and is classified with $0_R$ by the zero-multiplication row in \autoref{def:ring-stability-certificate}. The zero-summand fold theorem \autoref{thm:zero-convolution-summands-fold-zero} classifies the Cauchy fold with $0_R$, which is the padded coefficient of $\mathsf{nil}$.

For the cons step, write $x=\mathsf{cons}(a,x^{\prime})$. At coefficient $0$, the raw product recursion gives the sum of $a\cdot y[0]$ and the zero head of the shifted recursive product; the additive right-zero law classifies this with $a\cdot y[0]$, which is $\mathsf{Cauchy}_R(\mathsf{cons}(a,x^{\prime}),y;0)$. At coefficient $n+1$, the same recursion and the auxiliary coefficient formulas give
$$
\mathsf{rmul}_R(\mathsf{cons}(a,x^{\prime}),y)[n+1]\sim_R a\cdot y[n+1]+\mathsf{rmul}_R(x^{\prime},y)[n].
$$
The induction hypothesis and additive congruence replace the second summand by $\mathsf{Cauchy}_R(x^{\prime},y;n)$. Splitting the finite index spine $i+j=n+1$ into the head case $i=0$ and the shifted cases $i=i^{\prime}+1$ identifies the resulting fold with $\mathsf{Cauchy}_R(\mathsf{cons}(a,x^{\prime}),y;n+1)$ by \autoref{thm:finite-additive-fold-congruence-coefficient-lists}.
\end{proof}

\begin{lemma}[Raw Cauchy product coefficient associativity]
\label{lem:raw-cauchy-product-coefficient-associativity}
Let $R$ carry a $\CommRingUp(R)$ certificate with classifier $\sim_R$, zero endpoint $0_R$, addition $+$, multiplication $\cdot$, and the raw Cauchy product data of \autoref{def:polynomial-raw-cauchy-product}. For finite coefficient spines $x,y,z:\mathsf{ListCarrier}(R)$ and every $n:\NatUp$,
$$
\mathsf{rmul}_R(\mathsf{rmul}_R(x,y),z)[n]\sim_R\mathsf{rmul}_R(x,\mathsf{rmul}_R(y,z))[n].
$$
\end{lemma}
\begin{proof}
Fix $n$. By \autoref{lem:raw-cauchy-product-coefficient-formula}, the left coefficient is classified with $\mathsf{Cauchy}_R(\mathsf{rmul}_R(x,y),z;n)$. For each outer index $s+k=n$, apply \autoref{lem:raw-cauchy-product-coefficient-formula} to replace $\mathsf{rmul}_R(x,y)[s]$ by $\mathsf{Cauchy}_R(x,y;s)$. Multiplication congruence and \autoref{thm:finite-additive-fold-congruence-coefficient-lists} lift these replacements through the outer finite fold. The right-multiplication half of \autoref{lem:scalar-product-distributes-across-finite-additive-folds} then classifies $\mathsf{Cauchy}_R(x,y;s)\cdot z[k]$ with the inner fold over $(i,j)$ satisfying $i+j=s$ of $(x[i]\cdot y[j])\cdot z[k]$. The nested finite index spine $i+j=s$, $s+k=n$ is the same finite triple spine $i+j+k=n$ in lexicographic order after deleting the bookkeeping index $s$. The finite fold swap certificate of \autoref{def:two-index-finite-fold-swap-certificate}, together with fold congruence, classifies the resulting nested fold with $\mathsf{Triple}_R(x,y,z;n)$. Thus the left raw product coefficient is classified with $\mathsf{Triple}_R(x,y,z;n)$.

The right coefficient is handled symmetrically. The coefficient formula first gives $\mathsf{Cauchy}_R(x,\mathsf{rmul}_R(y,z);n)$. Replacing each coefficient of $\mathsf{rmul}_R(y,z)$ by its Cauchy fold and applying the left-multiplication half of \autoref{lem:scalar-product-distributes-across-finite-additive-folds} gives the finite fold over triples $i+j+k=n$ of $x[i]\cdot(y[j]\cdot z[k])$. Scalar multiplication associativity from \autoref{def:ring-stability-certificate} classifies each displayed summand with $(x[i]\cdot y[j])\cdot z[k]$. Fold congruence therefore classifies the right coefficient with the same $\mathsf{Triple}_R(x,y,z;n)$. Symmetry and transitivity of the scalar classifier compose the two comparisons.
\end{proof}

\begin{theorem}[Polynomial multiplication associativity over commutative-ring scalars]
\label{thm:polynomial-multiplication-associativity-commring-scalars}
Let $R$ carry a $\CommRingUp(R)$ certificate, read as a $\RingUp(R)$ certificate by \autoref{prop:commring-forgets-ring-certificate}. Let the classifier be $\sim_R$, the zero endpoint be $0_R$, the scalar operations be $+$ and $\cdot$, the canonical trim be $\mathsf{trim}_R$ from \autoref{def:polynomial-stability-certificate}, and let $\mathsf{rmul}_R$ and $\mathsf{PolyMul}_R$ be the raw and normalized products of \autoref{def:polynomial-raw-cauchy-product}. For finite coefficient spines $p,q,r:\mathsf{ListCarrier}(R)$,
$$
\mathsf{PolySame}_R\bigl(\mathsf{PolyMul}_R(\mathsf{PolyMul}_R(p,q),r),\mathsf{PolyMul}_R(p,\mathsf{PolyMul}_R(q,r))\bigr).
$$
\end{theorem}
\begin{proof}
Put $p_0:=\mathsf{trim}_R(p)$, $q_0:=\mathsf{trim}_R(q)$, and $r_0:=\mathsf{trim}_R(r)$. Set $A:=\mathsf{PolyMul}_R(\mathsf{PolyMul}_R(p,q),r)$ and $B:=\mathsf{PolyMul}_R(p,\mathsf{PolyMul}_R(q,r))$. By \autoref{lem:padded-coefficient-extensionality-polynomial-classifier}, it is enough to prove $A[n]\sim_RB[n]$ for every $n:\NatUp$.

Trim preserves padded coefficients by the trailing-zero decomposition in \autoref{def:polynomial-stability-certificate}: for every finite spine $u$ and every $n$, $u[n]\sim_R\mathsf{trim}_R(u)[n]$. The coefficient formula \autoref{lem:raw-cauchy-product-coefficient-formula} is functorial in both inputs with respect to such pointwise padded-coefficient comparisons, since multiplication congruence compares each Cauchy summand and \autoref{thm:finite-additive-fold-congruence-coefficient-lists} lifts the summand comparisons through the finite fold.

Unfolding \autoref{def:polynomial-raw-cauchy-product} and using trim idempotence from \autoref{def:polynomial-stability-certificate}, the left normalized product has the same padded coefficients as $\mathsf{rmul}_R(\mathsf{trim}_R(\mathsf{rmul}_R(p_0,q_0)),r_0)$. By trim preservation and the functoriality just recorded, its $n$th coefficient is classified with $\mathsf{rmul}_R(\mathsf{rmul}_R(p_0,q_0),r_0)[n]$. Similarly, the right normalized product has the same padded coefficients as $\mathsf{rmul}_R(p_0,\mathsf{trim}_R(\mathsf{rmul}_R(q_0,r_0)))$, and its $n$th coefficient is classified with $\mathsf{rmul}_R(p_0,\mathsf{rmul}_R(q_0,r_0))[n]$.

Applying \autoref{lem:raw-cauchy-product-coefficient-associativity} to $p_0,q_0,r_0$ gives
$$
\mathsf{rmul}_R(\mathsf{rmul}_R(p_0,q_0),r_0)[n]\sim_R\mathsf{rmul}_R(p_0,\mathsf{rmul}_R(q_0,r_0))[n].
$$
Classifier symmetry and transitivity compose the two trim comparisons with this raw coefficient associativity comparison, yielding $A[n]\sim_RB[n]$ for the fixed $n$. Since $n$ was arbitrary, \autoref{lem:padded-coefficient-extensionality-polynomial-classifier} gives $\mathsf{PolySame}_R(A,B)$, which is the displayed associativity statement.
\end{proof}

\begin{theorem}[Polynomial singleton evaluation]
\label{thm:polynomial-singleton-evaluation}
Let $R$ carry a $\RingUp(R)$ certificate with classifier $\sim_R$, zero endpoint $0_R$, addition $+$, multiplication $\cdot$, canonical trim $\mathsf{trim}_R$, and Horner evaluation as in \autoref{def:polynomial-horner-evaluation}. For every carried scalar coefficient $c$ and every carried scalar point $\alpha$,
$$
\mathsf{Eval}_{\alpha,R}(\mathsf{cons}(c,\mathsf{nil}))\sim_R c.
$$
\end{theorem}
\begin{proof}
Put $s=\mathsf{cons}(c,\mathsf{nil})$. First compute the unnormalized Horner value. The cons and nil recursion clauses in \autoref{def:polynomial-horner-evaluation} identify
$$
\mathsf{eval}_{\alpha,R}(s)=c+\alpha\cdot0_R.
$$
The right zero-absorption half of \autoref{thm:ring-zero-absorption-from-distributivity} gives
$$
\alpha\cdot0_R\sim_R0_R.
$$
Classifier reflexivity for $c$ and addition congruence from \autoref{def:ring-stability-certificate} give
$$
c+\alpha\cdot0_R\sim_Rc+0_R.
$$
The additive right-zero theorem \autoref{thm:ring-add-right-zero} gives $c+0_R\sim_Rc$, and transitivity yields
$$
\mathsf{eval}_{\alpha,R}(s)\sim_Rc.
$$
By \autoref{def:polynomial-horner-evaluation}, $\mathsf{Eval}_{\alpha,R}(s)=\mathsf{eval}_{\alpha,R}(\mathsf{trim}_R(s))$. The one-coefficient trimming boundary in \autoref{def:polynomial-normalized-singleton-representative}, read with \autoref{def:polynomial-stability-certificate}, has exactly the empty zero branch and the retained singleton branch. In the empty zero branch, $c\sim_R0_R$ and $\mathsf{trim}_R(s)=\mathsf{nil}$; the nil Horner equation gives $\mathsf{Eval}_{\alpha,R}(s)=0_R$, and classifier symmetry gives $0_R\sim_Rc$. In the retained singleton branch, $\mathsf{trim}_R(s)=s$, so the raw computation above is the required normalized comparison. These are the two canonical trim branches for a singleton coefficient spine.
\end{proof}
