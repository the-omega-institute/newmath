\begin{theorem}[Tagged option history classifier right visible-branch inversion]
\label{thm:tagged-option-history-classifier-right-visible-branch-inversion}
This theorem fits the per-object template as follows: the traditional target is
tagged option branch inversion from the right endpoint; the BEDC source is
$\mathsf{BHist}$ with $\hsame$; the carrier is
$\mathsf{TaggedOptionHistoryCarrier}$; the classifier is
$\mathsf{TaggedOptionHistoryClassifier}$; the ledger is the visible
absent/present tag plus payload; exactness is right-endpoint branch exposure;
and the status is $\mathsf{ProofTarget}$. For every BEDC history source carrier
$S:\mathsf{BHist}\to\mathsf{Prop}$, source classifier $\sim_S$, and histories
$h,k$, the classifier determines the visible branch presented by the right
endpoint:
$$
\mathsf{TaggedOptionHistoryClassifier}_{S,\sim_S}(h,k)\to
\hsame(k,\emp)\to
\hsame(h,\emp)
$$
and, for every payload history $a$,
$$
\mathsf{TaggedOptionHistoryClassifier}_{S,\sim_S}(h,k)\to
S(a)\to
\hsame(k,\Eone(a))\to
\exists b,a':\mathsf{BHist}.\;
S(b)\land S(a')\land \hsame(a,a')\land b\sim_S a'\land
\hsame(h,\Eone(b)).
$$
\end{theorem}
\begin{proof}
For the absent clause, case-analyze the tagged classifier witness. The
absent-absent branch already stores $\hsame(h,\emp)$. The present-present branch
stores $\hsame(k,\Eone(a'))$ for some right payload; composing it with the
displayed witness $\hsame(k,\emp)$ gives empty-versus-one no-confusion.

For the present clause, case-analyze the classifier witness again. The
absent-absent branch stores $\hsame(k,\emp)$ and contradicts the displayed
present endpoint $\hsame(k,\Eone(a))$. In the present-present branch, obtain
payloads $b$ and $a'$ with $S(b)$, $S(a')$, $b\sim_S a'$,
$\hsame(h,\Eone(b))$, and $\hsame(k,\Eone(a'))$. Compose the two presentations
of $k$ and invert the one-constructor equality to get $\hsame(a,a')$; the
remaining fields are exactly the displayed witnesses.
\end{proof}

\begin{theorem}[Tagged option history classifier absent-branch equivalence]
\label{thm:tagged-option-history-classifier-absent-branch-equivalence}
This theorem fits the per-object template as follows: the traditional target is
option absent-branch agreement; the BEDC source is $\mathsf{BHist}$ with
$\hsame$; the carrier is $\mathsf{TaggedOptionHistoryCarrier}$; the classifier
is $\mathsf{TaggedOptionHistoryClassifier}$; the stability obligation is branch
determinism; the ledger is the visible absent/present tag; the core theorem is
absentness equivalence; exactness is absent endpoints agree under classifier;
the standard bridge is $\mathsf{none}$/$\mathsf{some}$; and the status is
$\mathsf{ProofTarget}$. For every BEDC history source carrier
$S:\mathsf{BHist}\to\mathsf{Prop}$, source classifier $\sim_S$, and histories
$h,k$, classified endpoints agree on absentness:
$$
\mathsf{TaggedOptionHistoryClassifier}_{S,\sim_S}(h,k)\to
\bigl(\hsame(h,\emp)\Longleftrightarrow\hsame(k,\emp)\bigr).
$$
\end{theorem}
\begin{proof}
Assume a classifier witness between $h$ and $k$. The forward implication is
exactly the absent clause of
Theorem~\ref{thm:tagged-option-history-classifier-visible-branch-inversion}:
from $\hsame(h,\emp)$, left visible-branch inversion gives
$\hsame(k,\emp)$. The reverse implication is the absent clause of
Theorem~\ref{thm:tagged-option-history-classifier-right-visible-branch-inversion}:
from $\hsame(k,\emp)$, right visible-branch inversion gives
$\hsame(h,\emp)$. These two implications assemble the displayed
$\Longleftrightarrow$.
\end{proof}

\begin{theorem}[Tagged option history classifier present-branch equivalence]
\label{thm:tagged-option-history-classifier-present-branch-equivalence}
For every BEDC history source carrier $S:\mathsf{BHist}\to\mathsf{Prop}$, source classifier $\sim_S$, histories $h,k$, and classifier witness $\gamma:\mathsf{TaggedOptionHistoryClassifier}_{S,\sim_S}(h,k)$, classified endpoints agree on presentness:
$$
\bigl(\exists a:\mathsf{BHist}.\; S(a)\land\hsame(h,\Eone(a))\bigr)
\Longleftrightarrow
\bigl(\exists b:\mathsf{BHist}.\; S(b)\land\hsame(k,\Eone(b))\bigr).
$$
\end{theorem}
\begin{proof}
For the forward implication, unpack a source payload $a$ with $S(a)$ and
$\hsame(h,\Eone(a))$. Apply the present clause of
Theorem~\ref{thm:tagged-option-history-classifier-visible-branch-inversion} to
the witness $\gamma$; the exposed right payload supplies $S(b)$ and
$\hsame(k,\Eone(b))$, which is the required present witness for $k$. For the
reverse implication, unpack a right payload $b$ with $S(b)$ and
$\hsame(k,\Eone(b))$. Apply the present clause of
Theorem~\ref{thm:tagged-option-history-classifier-right-visible-branch-inversion}
to $\gamma$; the exposed left payload supplies $S(a)$ and
$\hsame(h,\Eone(a))$, giving the required present witness for $h$.
\end{proof}
