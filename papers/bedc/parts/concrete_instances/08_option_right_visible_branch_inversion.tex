\begin{theorem}[Tagged option history classifier right visible-branch inversion]
\label{thm:tagged-option-history-classifier-right-visible-branch-inversion}
This theorem fits the per-object template as follows: the traditional target is
tagged option branch inversion from the right endpoint; the BEDC source is
$\mathsf{BHist}$ with $\hsame$; the carrier is
$\mathsf{TaggedOptionHistoryCarrier}$; the classifier is
$\mathsf{TaggedOptionHistoryClassifier}$; the ledger is the visible
absent/present tag plus payload; exactness is right-endpoint branch exposure;
and the status is $\mathsf{ProofTarget}$. For every BEDC history source carrier
$S:\mathsf{BHist}\to\mathsf{Prop}$, source classifier $\sim_S$, and histories
$h,k$, the classifier determines the visible branch presented by the right
endpoint:
$$
\mathsf{TaggedOptionHistoryClassifier}_{S,\sim_S}(h,k)\to
\hsame(k,\emp)\to
\hsame(h,\emp)
$$
and, for every payload history $a$,
$$
\mathsf{TaggedOptionHistoryClassifier}_{S,\sim_S}(h,k)\to
S(a)\to
\hsame(k,\Eone(a))\to
\exists b,a':\mathsf{BHist}.\;
S(b)\land S(a')\land \hsame(a,a')\land b\sim_S a'\land
\hsame(h,\Eone(b)).
$$
\end{theorem}
\begin{proof}
For the absent clause, case-analyze the tagged classifier witness. The
absent-absent branch already stores $\hsame(h,\emp)$. The present-present branch
stores $\hsame(k,\Eone(a'))$ for some right payload; composing it with the
displayed witness $\hsame(k,\emp)$ gives empty-versus-one no-confusion.

For the present clause, case-analyze the classifier witness again. The
absent-absent branch stores $\hsame(k,\emp)$ and contradicts the displayed
present endpoint $\hsame(k,\Eone(a))$. In the present-present branch, obtain
payloads $b$ and $a'$ with $S(b)$, $S(a')$, $b\sim_S a'$,
$\hsame(h,\Eone(b))$, and $\hsame(k,\Eone(a'))$. Compose the two presentations
of $k$ and invert the one-constructor equality to get $\hsame(a,a')$; the
remaining fields are exactly the displayed witnesses.
\end{proof}

\begin{definition}[Tagged option-to-nullable bridge hypotheses]
\label{def:tagged-option-to-nullable-bridge-hypotheses}
For a BEDC history source carrier $S:\mathsf{BHist}\to\mathsf{Prop}$ and source
classifier $\sim_S$, define
$\mathsf{TaggedOptionToNullableBridge}(S,\sim_S)$ to be the pair of conditions:
\begin{enumerate}
  \item present endpoint closure: for all histories $a,h$, if $S(a)$ and $\hsame(h,\Eone(a))$, then $S(h)$;
  \item classifier hsame soundness: for all histories $a,b$, if $S(a)$, $S(b)$, and $a\sim_S b$, then $\hsame(a,b)$.
\end{enumerate}
\end{definition}

\begin{theorem}[Tagged option-to-nullable classifier bridge]
\label{thm:tagged-option-to-nullable-classifier-bridge}
Let $S:\mathsf{BHist}\to\mathsf{Prop}$ be a source carrier and let $\sim_S$ be a
source classifier. If $\mathsf{TaggedOptionToNullableBridge}(S,\sim_S)$ holds,
then for every history $h$,
$\mathsf{TaggedOptionHistoryCarrier}_S(h)$ implies
$\mathsf{OptionHistoryCarrier}_S(h)$, and for every pair of histories $h,k$,
$\mathsf{TaggedOptionHistoryClassifier}_{S,\sim_S}(h,k)$ implies
$\mathsf{OptionHistoryClassifier}_S(h,k)$.
\end{theorem}
\begin{proof}
Unpack the bridge into present endpoint closure and classifier hsame soundness.
For the carrier implication, case-analyze
$\mathsf{TaggedOptionHistoryCarrier}_S(h)$. The absent branch is already the
empty side of $\mathsf{OptionHistoryCarrier}_S(h)$. In the present branch, a
payload $a$ with $S(a)$ and $\hsame(h,\Eone(a))$ descends to $S(h)$ by present
endpoint closure, giving the source side of the nullable envelope.

For the classifier implication, use
Theorem~\ref{thm:tagged-option-history-classifier-branch-partition}. In the
absent-absent branch, the two empty endpoint witnesses give the nullable
carriers and compose through $\emp$ to obtain $\hsame(h,k)$. In the
present-present branch, the carrier part descends for both endpoints by present
endpoint closure. Classifier hsame soundness converts $a\sim_S b$ into
$\hsame(a,b)$; applying $e1$ congruence gives
$\hsame(\Eone(a),\Eone(b))$, which composes with the endpoint witnesses
$\hsame(h,\Eone(a))$ and $\hsame(k,\Eone(b))$ to yield $\hsame(h,k)$. These are
exactly the carrier and endpoint-equality fields of
$\mathsf{OptionHistoryClassifier}_S(h,k)$.
\end{proof}

\begin{theorem}[Tagged option-to-nullable hsame bridge criterion]
\label{thm:tagged-option-to-nullable-hsame-bridge-criterion}
The ten-field fit is: traditional target nullable comparison for
constructor-tagged options; BEDC source $\mathsf{BHist}$ with $\hsame$; carrier
$\mathsf{TaggedOptionHistoryCarrier}$; classifier

$S:\mathsf{BHist}\to\mathsf{Prop}$ be a source carrier. If
$$
  \forall a\,h.\;
    S(a)\to\hsame(h,\Eone(a))\to S(h),
$$
then $\mathsf{TaggedOptionToNullableBridge}(S,\hsame)$ holds. Consequently, for
every history $h$,
$$
  \mathsf{TaggedOptionHistoryCarrier}_S(h)\to
  \mathsf{OptionHistoryCarrier}_S(h),
$$
and for every pair of histories $h,k$,
$$
  \mathsf{TaggedOptionHistoryClassifier}_{S,\hsame}(h,k)\to
  \mathsf{OptionHistoryClassifier}_S(h,k).
$$
\end{theorem}
\begin{proof}
Pair the displayed present-endpoint absorption hypothesis with identity
classifier hsame soundness: from $S(a)$, $S(b)$, and $\hsame(a,b)$, the required
history-sameness conclusion is the same witness. This gives
$\mathsf{TaggedOptionToNullableBridge}(S,\hsame)$. The two descent conclusions
are exactly Theorem~\ref{thm:tagged-option-to-nullable-classifier-bridge}
specialized to the source classifier $\hsame$.
\end{proof}

\forall h\,k.\;
\mathsf{TaggedOptionHistoryClassifier}_{S,\sim_S}(h,k)\to

\end{theorem}
\begin{proof}
The tagged semantic certificate is exactly

envelope; it does not construct a nullable semantic certificate from the tagged

\end{proof}

\begin{theorem}[Tagged option history classifier absent-branch equivalence]

\end{theorem}
\begin{proof}

\begin{theorem}[Tagged option history classifier present-branch equivalence]
\label{thm:tagged-option-history-classifier-present-branch-equivalence}
For every BEDC history source carrier $S:\mathsf{BHist}\to\mathsf{Prop}$, source classifier $\sim_S$, histories $h,k$, and classifier witness $\gamma:\mathsf{TaggedOptionHistoryClassifier}_{S,\sim_S}(h,k)$, classified endpoints agree on presentness:
$$
\bigl(\exists a:\mathsf{BHist}.\; S(a)\land\hsame(h,\Eone(a))\bigr)
\Longleftrightarrow
\bigl(\exists b:\mathsf{BHist}.\; S(b)\land\hsame(k,\Eone(b))\bigr).
$$
\end{theorem}
\begin{proof}
For the forward implication, unpack a source payload $a$ with $S(a)$ and
$\hsame(h,\Eone(a))$. Apply the present clause of
Theorem~\ref{thm:tagged-option-history-classifier-visible-branch-inversion} to
the witness $\gamma$; the exposed right payload supplies $S(b)$ and
$\hsame(k,\Eone(b))$, which is the required present witness for $k$. For the
reverse implication, unpack a right payload $b$ with $S(b)$ and
$\hsame(k,\Eone(b))$. Apply the present clause of
Theorem~\ref{thm:tagged-option-history-classifier-right-visible-branch-inversion}
to $\gamma$; the exposed left payload supplies $S(a)$ and
$\hsame(h,\Eone(a))$, giving the required present witness for $h$.
\end{proof}

\begin{theorem}[Tagged option history carrier exclusive branch partition]
\label{thm:tagged-option-history-carrier-exclusive-branch-partition}
This theorem fits the per-object template as follows: the traditional target is
tagged optional carrier readback; the BEDC source is $\mathsf{BHist}$ with
$\hsame$; the carrier is $\mathsf{TaggedOptionHistoryCarrier}$; the classifier
is unused; the stability obligation is constructor no-confusion; the ledger is
the visible absent/present tag with payload; the core theorem is exclusive
carrier partition; exactness is absent-or-present coverage with disjointness;
the standard bridge is $\mathsf{none}$/$\mathsf{some}$; and the status is
$\mathsf{ProofTarget}$. For every source carrier
$S:\mathsf{BHist}\to\mathsf{Prop}$ and history $h$, if
$\mathsf{TaggedOptionHistoryCarrier}_S(h)$ holds, then
$$
\bigl(\hsame(h,\emp)\land
\neg\exists a:\mathsf{BHist}.\; S(a)\land\hsame(h,\Eone(a))\bigr)
\lor
\bigl(\exists a:\mathsf{BHist}.\;
S(a)\land\hsame(h,\Eone(a))\land\neg\hsame(h,\emp)\bigr).
$$
\end{theorem}
\begin{proof}
Case-analyze the tagged carrier witness. In the absent branch, the carrier
witness gives $\hsame(h,\emp)$; any present payload witness
$S(a)\land\hsame(h,\Eone(a))$ contradicts
Lemma~\ref{lem:tagged-option-history-carrier-constructor-separation}. In the
present branch, the carrier witness gives a payload $a$ with $S(a)$ and
$\hsame(h,\Eone(a))$; the same constructor-separation lemma rules out
$\hsame(h,\emp)$. These two carrier constructors give the displayed coverage
and disjointness.
\end{proof}

\begin{theorem}[Tagged option history classifier branch partition]
\label{thm:tagged-option-history-classifier-branch-partition}
This theorem fits the per-object template as follows: the traditional target is
tagged option classifier branch partition; the BEDC source is $\mathsf{BHist}$
with $\hsame$; the carrier is $\mathsf{TaggedOptionHistoryCarrier}$; the
classifier is $\mathsf{TaggedOptionHistoryClassifier}$; the ledger is the

classifier $\sim_S$, histories $h,k$, and classifier witness

$$
\bigl(\hsame(h,\emp)\land\hsame(k,\emp)\bigr)
\lor
\bigl(\exists a,b:\mathsf{BHist}.\;

\hsame(h,\Eone(a))\land \hsame(k,\Eone(b))\land a\sim_S b\bigr).
$$
\end{theorem}
\begin{proof}
Case-analyze the classifier witness $\gamma$. The absent-absent constructor

witnesses $S(a)$ and $S(b)$, endpoint witnesses $\hsame(h,\Eone(a))$ and
$\hsame(k,\Eone(b))$, and the source classifier witness $a\sim_S b$, giving the
right branch. These are the only constructors of the tagged classifier witness,
so the displayed disjunction exhausts the endpoints.
\end{proof}

\begin{theorem}[Tagged option source weakening]
\label{thm:tagged-option-source-weakening}

$$
\forall h.\; S(h)\to T(h)
$$
and
$$
\forall a\,b.\; S(a)\to S(b)\to a\sim_S b\to a\sim_T b.
$$
Then for every history $h$,
$$
\mathsf{TaggedOptionHistoryCarrier}_S(h)\to
\mathsf{TaggedOptionHistoryCarrier}_T(h),

\mathsf{TaggedOptionHistoryClassifier}_{S,\sim_S}(h,k)\to

\begin{proof}

implication to the two source witnesses to obtain $T(a)$ and $T(b)$, and apply
