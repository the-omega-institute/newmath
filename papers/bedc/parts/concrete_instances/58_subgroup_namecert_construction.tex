\begin{theorem}[Centralizer normalizer quotient classifiers coincide]
\label{thm:subgroup-centralizer-normalizer-quotient-classifiers-coincide}
For a BEDC history-level $\GroupUp$ operation over $\mathsf{BHist}$
satisfying the same associativity, unit, multiplication-congruence,
carrier-transport, and two-sided inverse laws as in
\autoref{thm:subgroup-centralizer-normalizer-tower-certificate}, every
$x,y:\mathsf{BHist}$ satisfies
$$
Q_{Z_a\leq N_a}(x,y)\Longleftrightarrow
Q_{Z_a\triangleleft N_a}(x,y).
$$
\end{theorem}
\begin{proof}
For the forward direction, unpack
$Q_{Z_a\leq N_a}(x,y)$. The endpoint clauses give $N_a(x)$ and $N_a(y)$, and
the right-witness clause gives a history $z$ with $Z_a(z)$ and
$\hsame(y,\mathsf{mul}(x,z))$.

It remains to produce the kernel word for
$Q_{Z_a\triangleleft N_a}(x,y)$. Multiply the endpoint comparison on the left
by $\mathsf{inv}(x)$. Associativity, the left inverse law, the unit law, and
multiplication congruence identify
$\mathsf{mul}(\mathsf{inv}(x),y)$ with $z$ up to $\hsame$. The carrier
transport row for $Z_a$ from
\autoref{thm:subgroup-centralizer-normalizer-tower-certificate} moves the
witness $Z_a(z)$ to the displayed relative kernel word.

For the reverse direction, unpack
$Q_{Z_a\triangleleft N_a}(x,y)$. The endpoint clauses already give the
$N_a(x)$ and $N_a(y)$ clauses needed by $Q_{Z_a\leq N_a}(x,y)$, and the kernel
clause gives $Z_a(\mathsf{mul}(\mathsf{inv}(x),y))$.

Choose the right witness
$z:=\mathsf{mul}(\mathsf{inv}(x),y)$. Associativity, the left inverse law, the
unit law, and multiplication congruence identify
$\mathsf{mul}(x,z)$ with $y$ up to $\hsame$, so the kernel clause supplies the
required $Z_a(z)$ witness and the endpoint comparison repacks the
coset-witness classifier.

The ten-field fit is the ordinary quotient coset/kernel classifier
equivalence for the centralizer-normalizer tower: the BEDC source is
$\GroupUp$ plus $\SubgroupUp$ over $\mathsf{BHist}$; the carrier is $N_a$; the
classifier is either displayed predicate; stability comes from the tower
certificate, inverse laws, unit laws, multiplication congruence, and carrier
transport; the ledger records the right witness and the relative-kernel word;
the core theorem is this statement; exactness is the displayed equivalence;
the downstream consumer is $\QuotientGroupUp$; and the status is
$\mathsf{ProofTarget}$.
\end{proof}

\begin{theorem}[Centralizer normalizer right quotient classifier transitivity]
\label{thm:subgroup-centralizer-right-quotient-classifier-trans}
For a BEDC history-level $\GroupUp$ operation over $\mathsf{BHist}$
satisfying the same associativity, unit, multiplication-congruence, and
two-sided inverse laws as in
\autoref{thm:subgroup-centralizer-normalizer-tower-certificate}, the
right-witness quotient classifier $Q_{Z_a\leq N_a}$ is transitive:
$$
Q_{Z_a\leq N_a}(x,y)\land Q_{Z_a\leq N_a}(y,z)
\Longrightarrow Q_{Z_a\leq N_a}(x,z).
$$
\end{theorem}
\begin{proof}
Convert both right-witness classifier hypotheses to the relative-kernel
classifier using the centralizer-normalizer quotient classifier equivalence.
The kernel classifier is transitive by the empty-unit centralizer quotient
kernel theorem: the two relative kernel words multiply, and the middle
endpoint cancels by associativity, unit, congruence, and inverse laws. Convert
the resulting kernel classifier back to the right-witness classifier.
\end{proof}

\begin{theorem}[Centralizer right quotient classifier hsame transport]
\label{thm:subgroup-right-quotient-classifier-hsame-transport}
For a BEDC history-level $\GroupUp$ operation over $\mathsf{BHist}$
satisfying the same associativity, unit, multiplication-congruence, and
two-sided inverse laws as in
\autoref{thm:subgroup-centralizer-normalizer-tower-certificate}, if
$Q_{Z_a\leq N_a}(x,y)$ is presented by the right-witness quotient classifier
and $x\hsame x'$ and $y\hsame y'$, then the same right-witness quotient
classifier holds at $(x',y')$.
\end{theorem}
\begin{proof}
Translate the right-witness quotient classifier through the quotient-kernel
equivalence to obtain $Q_{Z_a\triangleleft N_a}(x,y)$. The quotient-kernel
transport theorem moves both endpoints and the centralizer kernel row across
the two $\hsame$ comparisons. Translating back through the same equivalence
repackages the transported kernel as the right-witness quotient classifier at
$(x',y')$.
\end{proof}

\begin{theorem}[Centralizer right coset classifier transitivity]
\label{thm:subgroup-centralizer-right-coset-classifier-trans}
For a BEDC history-level $\GroupUp$ operation over $\mathsf{BHist}$ satisfying
associativity, right-unit, and multiplication-congruence laws, the centralizer
right-coset classifier is transitive:
$$
Q^{\mathrm{right}}_{Z_a\leq N_a}(x,y)\land
Q^{\mathrm{right}}_{Z_a\leq N_a}(y,z)
\Longrightarrow Q^{\mathrm{right}}_{Z_a\leq N_a}(x,z).
$$
\end{theorem}
\begin{proof}
Unpack the two right-coset witnesses. The first supplies a centralizer element
$u$ with $y\hsame \mathsf{mul}(x,u)$, and the second supplies a centralizer
element $v$ with $z\hsame \mathsf{mul}(y,v)$. Product closure for the
centralizer carrier gives a carried witness $\mathsf{mul}(u,v)$. Multiplication
congruence transports the second endpoint along the first comparison, and
associativity identifies
$\mathsf{mul}(\mathsf{mul}(x,u),v)$ with
$\mathsf{mul}(x,\mathsf{mul}(u,v))$. The endpoint clauses are inherited from
the two classifier hypotheses.
\end{proof}

\begin{theorem}[Centralizer quotient kernel empty fiber]
\label{thm:subgroup-centralizer-quotient-kernel-empty-fiber}
For a BEDC history-level $\GroupUp$ operation over $\mathsf{BHist}$
satisfying the same associativity, unit, multiplication-congruence,
carrier-transport, and two-sided inverse laws as in
\autoref{thm:subgroup-centralizer-normalizer-tower-certificate}, every
$x:\mathsf{BHist}$ satisfies
$$
Q_{Z_a\triangleleft N_a}(\mathsf{e},x)\Longleftrightarrow Z_a(x).
$$
\end{theorem}
\leanchecked{BEDC.Derived.SubgroupUp.SubgroupCentralizerQuotientKernel\_empty\_fiber\_iff}
\begin{proof}
For the forward direction, unpack the relative-kernel classifier
$Q_{Z_a\triangleleft N_a}(\mathsf{e},x)$. Its kernel clause gives
$Z_a(\mathsf{mul}(\mathsf{inv}(\mathsf{e}),x))$.

The inverse identity row identifies $\mathsf{inv}(\mathsf{e})$ with
$\mathsf{e}$, and the left-unit law identifies
$\mathsf{mul}(\mathsf{e},x)$ with $x$. The $Z_a$ carrier-transport row from
\autoref{thm:subgroup-centralizer-normalizer-tower-certificate} carries the
kernel witness across this endpoint comparison and gives $Z_a(x)$.

For the reverse direction, assume $Z_a(x)$. The empty centralizer row gives
$Z_a(\mathsf{e})$, and
\autoref{thm:subgroup-centralizer-self-normalizer} sends both $Z_a(\mathsf{e})$
and $Z_a(x)$ into the normalizer carrier $N_a$.

It remains to supply the kernel word. The same inverse identity and left-unit
comparison identify $\mathsf{mul}(\mathsf{inv}(\mathsf{e}),x)$ with $x$; by
transporting $Z_a(x)$ along the reverse comparison, we obtain the carried
kernel clause and package
$Q_{Z_a\triangleleft N_a}(\mathsf{e},x)$.
\end{proof}

\begin{theorem}[Centralizer normalizer coset classifier identity fiber]
\label{thm:subgroup-centralizer-normalizer-coset-classifier-identity-fiber}

satisfying the same associativity, unit, multiplication-congruence,
carrier-transport, and two-sided inverse laws as in
\autoref{thm:subgroup-centralizer-normalizer-tower-certificate}, every
$x:\mathsf{BHist}$ satisfies
$$
Q_{Z_a\leq N_a}(\mathsf{e},x)\Longleftrightarrow Z_a(x).
$$
\end{theorem}
\begin{proof}
Apply
\autoref{thm:subgroup-centralizer-normalizer-quotient-classifiers-coincide} at
the endpoint pair $(\mathsf{e},x)$. It identifies the right-coset classifier
$Q_{Z_a\leq N_a}(\mathsf{e},x)$ with the relative-kernel classifier
$Q_{Z_a\triangleleft N_a}(\mathsf{e},x)$.

Then apply
\autoref{thm:subgroup-centralizer-quotient-kernel-empty-fiber}. Its empty
fiber calculation identifies
$Q_{Z_a\triangleleft N_a}(\mathsf{e},x)$ with $Z_a(x)$. Transitivity of these
two equivalences gives the displayed identity fiber for
$Q_{Z_a\leq N_a}$.
\end{proof}
