\begin{theorem}[Product pair-representation coherence under classifier weakening]
\label{thm:prod-pair-rep-coherence-under-classifier-weakening}
The ten-field roadmap fit is: target displayed ordered-product coherence
transport; BEDC source $\mathsf{BHist}$ with $\Cont$; carriers $L$ and $R$;
source classifiers $\sim_L,\sim_R$; target classifiers $\approx_L,\approx_R$;
ledger displayed pair representation; stability datum
$\mathsf{ProdPairRepCoherent}$; exactness unchanged at the product endpoint;
standard bridge component classifier weakening; status $\mathsf{ProofTarget}$.

Let $L,R:\mathsf{BHist}\to\mathsf{Prop}$ be source-history carriers, and let
$\sim_L,\sim_R,\approx_L,\approx_R:\mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$
be component classifiers. Suppose
$\mathsf{ProdPairRepCoherent}_{L,R,\sim_L,\sim_R}$ holds, and suppose
$$
\forall x\,y.\;x\sim_L y\to x\approx_L y
$$
and
$$
\forall x\,y.\;x\sim_R y\to x\approx_R y.
$$
Then $\mathsf{ProdPairRepCoherent}_{L,R,\approx_L,\approx_R}$ holds.
\end{theorem}
\begin{proof}
Take two displayed pair representations
$\mathsf{ProdPairRep}_{L,R}(h,l,r)$ and
$\mathsf{ProdPairRep}_{L,R}(h,l',r')$ of the same product history $h$. Applying
the given $\mathsf{ProdPairRepCoherent}_{L,R,\sim_L,\sim_R}$ datum to these
two representations yields the component witnesses $l\sim_L l'$ and
$r\sim_R r'$. Transport the first witness through the left classifier
implication to obtain $l\approx_L l'$, and transport the second witness through
the right classifier implication to obtain $r\approx_R r'$. These two
transported witnesses are exactly the two component conclusions required by
$\mathsf{ProdPairRepCoherent}_{L,R,\approx_L,\approx_R}$.
\end{proof}

\begin{theorem}[Product componentwise history classifier core fields]
\label{thm:prod-component-history-classifier-certificate-target}
Let $L,R:\mathsf{BHist}\to\mathsf{Prop}$ be source-history carriers, let
$\sim_L,\sim_R:\mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$ be their source
classifiers, and assume $\mathsf{cert}_L:\NameCert(L,\sim_L)$,
$\mathsf{cert}_R:\NameCert(R,\sim_R)$, and
$\mathsf{ProdPairRepCoherent}_{L,R,\sim_L,\sim_R}$. Write
$C(h,k)$ for
$\mathsf{ProdComponentHistoryClassifier}_{L,R,\sim_L,\sim_R}(h,k)$. Then its
five core $\NameCert$ fields for $\mathsf{ProdHistoryCarrier}(L,R)$ decompose as
follows:
\begin{enumerate}
  \item carrier inhabitation:
  $$
  \exists h.\;\mathsf{ProdHistoryCarrier}(L,R,h);
  $$
  \item carrier-relative reflexivity:
  $$
  \forall h.\;\mathsf{ProdHistoryCarrier}(L,R,h)\to C(h,h);
  $$
  \item source-certified symmetry:
  $$
  \forall h\,k.\;C(h,k)\to C(k,h);
  $$
  \item source-certified transitivity:
  $$
  \forall h\,k\,u.\;C(h,k)\to C(k,u)\to C(h,u);
  $$
  \item NameCert-shaped classifier-induced carrier transport:
  $$
  \forall h\,k.\;C(h,k)\to\mathsf{ProdHistoryCarrier}(L,R,h)\to
  \mathsf{ProdHistoryCarrier}(L,R,k).
  $$
The stronger endpoint exposure
$C(h,k)\to\mathsf{ProdHistoryCarrier}(L,R,h)\land
\mathsf{ProdHistoryCarrier}(L,R,k)$ is
Theorem~\ref{thm:prod-component-history-classifier-endpoint-carriers}; the
displayed theorem implies this field after adding the required source-endpoint
carrier premise.
\end{enumerate}
These five fields are the core $\NameCert$ data used by
Theorem~\ref{thm:prod-component-history-semantic-certificate-target}.
\end{theorem}
\begin{proof}
For carrier inhabitation, $\mathsf{cert}_L.\mathsf{carrier\_inhabited}$ and
$\mathsf{cert}_R.\mathsf{carrier\_inhabited}$ supply histories $l$ and $r$
with $L(l)$ and $R(r)$. Applying
Theorem~\ref{thm:prod-history-carrier-append-intro} to those two source
carrier witnesses gives
$\mathsf{ProdHistoryCarrier}(L,R,\mathsf{append}(l,r))$, so the product
carrier is inhabited. For carrier-relative reflexivity, unpack a
$\mathsf{ProdHistoryCarrier}(L,R,h)$ witness into left and right histories and
reuse that pair representation on both sides; the two source $\NameCert$
witnesses give component reflexivity. For symmetry, invert the
componentwise classifier witness and apply source symmetry to the two component
classifier witnesses, keeping the two pair representations with endpoints
exchanged. For transitivity, unpack $\mathsf{C}(h,k)$ and
$\mathsf{C}(k,u)$ into pair representations $(l_h,r_h)$, $(l_k,r_k)$,
$(l'_k,r'_k)$, and $(l_u,r_u)$. Pair-representation coherence at the middle
history $k$ gives both $l_k\sim_L l'_k$ and $r_k\sim_R r'_k$. The two source
$\NameCert.\mathsf{equiv\_trans}$ fields then compose the left chain
$l_h\sim_L l_k\sim_L l'_k\sim_L l_u$ and the right chain
$r_h\sim_R r_k\sim_R r'_k\sim_R r_u$ separately. For carrier transport, the
classifier witness first exposes a displayed pair representation of the right
endpoint $k$, and pair-representation coverage rebuilds
$\mathsf{ProdHistoryCarrier}(L,R,k)$. The source-endpoint carrier premise is
the required $\NameCert$ field input and does not supply the right endpoint.
Thus the displayed five-field tuple is exactly the componentwise product core
classifier data.
\end{proof}

\begin{theorem}[Bounded $\hsame$ product pair-representation coherence]
\label{thm:prod-pair-rep-bounded-hsame-coherence}
Let $L,R:\mathsf{BHist}\to\mathsf{Prop}$ be source-history carriers. Suppose
there are histories $l_0,r_0:\mathsf{BHist}$ such that
$$
\forall l.\;L(l)\to\hsame(l,l_0)
$$
and
$$
\forall r.\;R(r)\to\hsame(r,r_0).
$$
With both source classifiers specialized to $\hsame$, the product
pair-representation coherence condition holds:
$$
\mathsf{ProdPairRepCoherent}_{L,R,\hsame,\hsame}.
$$
\end{theorem}
\begin{proof}
Take two displayed pair representations of the same product history $h$, with
components $(l,r)$ and $(l',r')$. The pair-representation witnesses include
$L(l)$, $L(l')$, $R(r)$, and $R(r')$. The boundedness hypotheses give
$\hsame(l,l_0)$, $\hsame(l',l_0)$, $\hsame(r,r_0)$, and $\hsame(r',r_0)$.
By symmetry of $\hsame$, turn $\hsame(l',l_0)$ into $\hsame(l_0,l')$ and
$\hsame(r',r_0)$ into $\hsame(r_0,r')$. By transitivity of $\hsame$, compose
$l$ through $l_0$ to obtain $\hsame(l,l')$, and compose $r$ through $r_0$ to
obtain $\hsame(r,r')$. These are exactly the two component classifier
conclusions required by $\mathsf{ProdPairRepCoherent}_{L,R,\hsame,\hsame}$.
\end{proof}

\begin{theorem}[Singleton product pair-representation coherence]
\label{thm:prod-pair-rep-singleton-coherence}
Fix histories $l_0,r_0:\mathsf{BHist}$. Let
$L(x):\Longleftrightarrow\hsame(x,l_0)$ and
$R(y):\Longleftrightarrow\hsame(y,r_0)$. With both source classifiers
specialized to $\hsame$, the singleton source carriers satisfy product
pair-representation coherence:
$$
\mathsf{ProdPairRepCoherent}_{L,R,\hsame,\hsame}.
$$
\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:prod-pair-rep-bounded-hsame-coherence}. The required
left bound sends a witness $L(l)$ to the displayed clause $\hsame(l,l_0)$, and
the required right bound sends a witness $R(r)$ to the displayed clause
$\hsame(r,r_0)$.
\end{proof}

\begin{theorem}[Product componentwise history semantic certificate target]
\label{thm:prod-component-history-semantic-certificate-target}
Let $L,R:\mathsf{BHist}\to\mathsf{Prop}$ be source-history carriers, let
$\sim_L,\sim_R:\mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$ be their source
classifiers, and assume $\mathsf{cert}_L:\NameCert(L,\sim_L)$,
$\mathsf{cert}_R:\NameCert(R,\sim_R)$, and
$\mathsf{ProdPairRepCoherent}_{L,R,\sim_L,\sim_R}$. Then the componentwise
classifier forms the semantic naming certificate with displayed pair ledger
$D(h):\Longleftrightarrow
\exists l,r:\mathsf{BHist}.\;\mathsf{ProdPairRep}_{L,R}(h,l,r)$:
$$
\mathsf{SemanticNameCert}(\mathsf{ProdHistoryCarrier}(L,R),
\mathsf{ProdHistoryCarrier}(L,R),
D,
\mathsf{ProdComponentHistoryClassifier}_{L,R,\sim_L,\sim_R}).
$$
\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:prod-component-history-classifier-certificate-target} to
obtain carrier inhabitation, carrier-relative reflexivity, source-certified
symmetry, source-certified transitivity, and classifier-induced carrier
transport for
$\mathsf{ProdComponentHistoryClassifier}_{L,R,\sim_L,\sim_R}$ over
$\mathsf{ProdHistoryCarrier}(L,R)$. Package these five fields as the core
$\NameCert$ certificate. Carrier transport is still the right-endpoint
transport supplied by the classifier witness and pair-representation coverage.
The semantic pattern carrier is $\mathsf{ProdHistoryCarrier}(L,R)$, so
$\mathsf{pattern\_sound}$ is identity on the carrier. Ledger soundness is the
forward direction of Theorem~\ref{thm:prod-pair-rep-coverage}: every product
history carrier witness exposes component histories $l,r$ and hence a displayed
pair ledger witness $D(h)$.
\end{proof}

\begin{theorem}[Singleton history source certificate]
\label{thm:singleton-history-source-namecert}
Fix a BEDC history $a_0:\mathsf{BHist}$, and define
$A_{a_0}:\mathsf{BHist}\to\mathsf{Prop}$ by
$A_{a_0}(x)\Longleftrightarrow\hsame(x,a_0)$. Then
$\NameCert(A_{a_0},\hsame)$ holds.
\end{theorem}
\begin{proof}
The carrier is inhabited by $a_0$ using reflexivity of $\hsame$. The
reflexivity, symmetry, and transitivity fields of the certificate are exactly
the corresponding fields of $\hsame$. For carrier transport, start with
$A_{a_0}(x)$, namely $\hsame(x,a_0)$, and a classifier witness
$\hsame(x,y)$. Symmetry turns the classifier witness into $\hsame(y,x)$, and
transitivity with $\hsame(x,a_0)$ yields $\hsame(y,a_0)$, which is
$A_{a_0}(y)$.
\end{proof}

\begin{theorem}[Singleton componentwise product semantic certificate]
\label{thm:prod-singleton-componentwise-semantic-certificate}
Fix histories $l_0,r_0:\mathsf{BHist}$. Define singleton source carriers
$L(x):\Longleftrightarrow\hsame(x,l_0)$ and
$R(y):\Longleftrightarrow\hsame(y,r_0)$. With displayed pair ledger
$D(h):\Longleftrightarrow
\exists l,r:\mathsf{BHist}.\;\mathsf{ProdPairRep}_{L,R}(h,l,r)$, the
componentwise product classifier specialized to $\hsame$ forms the semantic
certificate
$$
\mathsf{SemanticNameCert}(\mathsf{ProdHistoryCarrier}(L,R),
\mathsf{ProdHistoryCarrier}(L,R),
D,
\mathsf{ProdComponentHistoryClassifier}_{L,R,\hsame,\hsame}).
$$
\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:singleton-history-source-namecert} at $l_0$ and at
$r_0$ to obtain the two singleton $\NameCert$ records for $L$ and $R$.
Theorem~\ref{thm:prod-pair-rep-singleton-coherence} supplies the
pair-representation coherence hypothesis for these singleton carriers. Apply
Theorem~\ref{thm:prod-component-history-semantic-certificate-target} with the
two singleton certificates and this coherence witness.
\end{proof}

\begin{theorem}[Product componentwise certificate refines the checked envelope]
\label{thm:prod-componentwise-certificate-refines-checked-envelope}
Let $L,R:\mathsf{BHist}\to\mathsf{Prop}$ be source-history carriers, let
$\sim_L,\sim_R:\mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$ be their source
classifiers, and assume $\mathsf{cert}_L:\NameCert(L,\sim_L)$,
$\mathsf{cert}_R:\NameCert(R,\sim_R)$, and
$\mathsf{ProdPairRepCoherent}_{L,R,\sim_L,\sim_R}$. Assume also that the source
classifiers are $\hsame$-sound:
$$
\forall x\,y.\;x\sim_L y\to\hsame(x,y)
\qquad\hbox{and}\qquad
\forall x\,y.\;x\sim_R y\to\hsame(x,y).
$$
With $D(h):\Longleftrightarrow
\exists l,r:\mathsf{BHist}.\;\mathsf{ProdPairRep}_{L,R}(h,l,r)$, both of the
following hold:
$$
\mathsf{SemanticNameCert}(\mathsf{ProdHistoryCarrier}(L,R),
\mathsf{ProdHistoryCarrier}(L,R),
D,
\mathsf{ProdComponentHistoryClassifier}_{L,R,\sim_L,\sim_R})
$$
and, for all histories $h,k$,
$$
\mathsf{ProdComponentHistoryClassifier}_{L,R,\sim_L,\sim_R}(h,k)
\to
\mathsf{ProdHistoryClassifier}(L,R,h,k).
$$
\end{theorem}
\begin{proof}
The first conclusion is Theorem~\ref{thm:prod-component-history-semantic-certificate-target}
applied to the two source certificates and the pair-representation coherence
witness. The second conclusion is
Theorem~\ref{thm:prod-component-history-classifier-descends-to-envelope}
applied to the two $\hsame$-soundness assumptions.
\end{proof}
