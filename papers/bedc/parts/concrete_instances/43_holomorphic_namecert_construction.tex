\chapter{A Naming Seed for \texorpdfstring{$\HolomorphicUp$}{Holomorphic-up}}
\label{ch:concrete-instances-holomorphic-namecert}

This chapter records a naming seed for holomorphic functions on open disks in $\ComplexUp$. The ten fields of the seed are: traditional target, the ordinary holomorphic-on-open-disk interface; BEDC source, $\ComplexDiffUp$ together with $\ConvergenceRadiusUp$ and the open-disk predicate; carrier choice, triples $(f,z_0,r)$ equipped with a disk-local differentiability predicate; classifier choice, $\hsame$-equivalence of the restricted pointwise functions and disk data; stability obligations, transport of the carrier, classifier, moduli, and derivative data along $\hsame$; ledger, the retained function history, disk witness, modulus witness, and analytic-bounds witnesses; core theorem list, the obligations in \autoref{def:holo-core-theorem-obligations}; exactness predicate, soundness and coverage against the intended holomorphic-on-disk behavior; standard bridge, comparison with ordinary complex analysis through constructive modulus data; status, $\text{Seed}$.

\section{Open-disk carrier}

\begin{definition}[Open disk predicate]
\label{def:open-disk}
For complex history \(z_0\), unary radius \(r\), complex history \(z\), and unary gap \(g\), the open-disk predicate \(\mathsf{OpenDisk}(z_0, r, z, g)\) holds when the rational representation of the modulus \(|z - z_0|\) is strictly less than \(r\) under \(\mathsf{NatUnaryStrictPrefix}\). We write \(\mathsf{OpenDisk}(z_0,r,z)\) when the gap witness is existentially understood. The BEDC kernel anchor records the four histories as unary and records the boundary ledger by \(\Cont(z,g,r)\).
\end{definition}
\leandef{BEDC.Derived.HolomorphicUp.HolomorphicOpenDisk}

\begin{lemma}[Open disk carriers are unary]
\label{lem:open-disk-unary-witness}
For every $(z_0, r, z)$ with $\mathsf{OpenDisk}(z_0, r, z)$, the gap modulus $\mathsf{Gap}(z_0, r, z)$ is a unary history giving an explicit positive distance from $z$ to the boundary of the disk.
\end{lemma}
\begin{proof}
The strict-prefix witness immediately gives the gap as the difference history.
\end{proof}

\begin{lemma}[Open-disk transport along \(\hsame\)]
\label{lem:holomorphic-open-disk-hsame-transport}
If \(\mathsf{OpenDisk}(z_0,r,z,g)\) and each of \(z_0,r,z,g\) is replaced by an \(\hsame\)-equivalent history, then the transported tuple is again an open-disk witness. The transported witness exposes the unary carriers of the four target histories and the transported ledger \(\Cont(z',g',r')\).
\end{lemma}
\begin{proof}
Transport the four unary fields by unary-history stability under \(\hsame\), and transport the boundary ledger by \(\Cont\)-stability under \(\hsame\) on source, gap, and result.
\end{proof}
\leanchecked{BEDC.Derived.HolomorphicUp.HolomorphicOpenDisk\_hsame\_transport}

\begin{lemma}[Radius ledger enlargement]
\label{lem:holomorphic-open-disk-radius-extension-closed}
If \(\mathsf{OpenDisk}(z_0,r,z,g)\), \(e\) is unary, and \(\Cont(r,e,r')\), then \(\mathsf{OpenDisk}(z_0,r',z,g \cdot e)\), where \(g \cdot e\) denotes BEDC history append.
\end{lemma}
\begin{proof}
The radius carrier follows from unary closure of \(\Cont(r,e,r')\). The enlarged gap is unary by append closure, and the ledger \(\Cont(z,g \cdot e,r')\) is obtained by composing \(\Cont(z,g,r)\) with \(\Cont(r,e,r')\) using associativity of append.
\end{proof}
\leanchecked{BEDC.Derived.HolomorphicUp.HolomorphicOpenDisk\_radius\_extension\_closed}

\begin{theorem}[Open disk $e1$ gap radius inversion]
\label{thm:holo-open-disk-e1-gap-radius-inversion}
For complex histories $z_0,z$, unary gap history $gap$, and radius history
$radius$, if the displayed center-to-point continuation is
$\ContR(z_0,\mathsf{e1}(gap),z)$ and
$\mathsf{NatUnaryStrictPrefix}(\mathsf{e1}(gap),\mathsf{e1}(radius))$, then
$\mathsf{NatUnaryStrictPrefix}(z_0,z)$ and
$\mathsf{NatUnaryStrictPrefix}(gap,radius)$.
\end{theorem}
\begin{proof}
Use the displayed $e1$ gap as the strict positive tail from $z_0$ to $z$.
The radius comparison descends by $e1$ strict-prefix inversion.
\end{proof}
\leanchecked{BEDC.Derived.HolomorphicUp.HolomorphicOpenDisk\_e1\_gap\_radius\_inversion}

\begin{theorem}[Open disk radius continuation gap witness]
\label{thm:holo-open-disk-radius-extension-gap-witness}
If \(z\) lies in the unary open disk with center \(z_0\) and radius \(r\), and
\(r\) continues by a unary history \(e\) to \(r'\), then the original boundary
gap factors through a unary continued gap. Concretely, there are histories
\(g,g'\) with
$$
\mathsf{UnaryHistory}(g)\land\mathsf{UnaryHistory}(g')\land
\Cont(z,g,r)\land\Cont(g,e,g')\land\Cont(z,g',r').
$$
\end{theorem}
\begin{proof}
Unpack the open-disk witness as a unary gap \(g\) with \(\Cont(z,g,r)\).
Associativity of continuation supplies the middle history \(g'\) with
\(\Cont(g,e,g')\) and \(\Cont(z,g',r')\); unary continuation closure makes
\(g'\) unary.
\end{proof}
\leanchecked{BEDC.Derived.HolomorphicUp.HolomorphicOpenDisk\_radius\_extension\_gap\_witness}

\begin{theorem}[Open disk radius continuation closure]
\label{thm:holo-open-disk-radius-continuation-extend}
Under the same hypotheses, membership in the unary open disk is preserved when
the radius is continued from \(r\) to \(r'\).
\end{theorem}
\begin{proof}
Use the continued gap from
\autoref{thm:holo-open-disk-radius-extension-gap-witness}. The center and point
carriers are unchanged, and the continued radius is unary by continuation
closure.
\end{proof}
\leanchecked{BEDC.Derived.HolomorphicUp.HolomorphicOpenDisk\_radius\_continuation\_extend}

\section{Holomorphic predicate}

\begin{definition}[Holomorphic on disk seed predicate]
\label{def:holo-on-disk}
For function $f : \ComplexUp \to \ComplexUp$, center $z_0$, and unary radius $r$, the seed predicate $\mathsf{HoloOnDisk}(f, z_0, r)$ is the intended carrier condition saying that a uniform modulus function $\delta : \Unary \times \Unary \to \Unary$ controls complex differentiability at every point $z$ with $\mathsf{OpenDisk}(z_0, r, z)$ and every unary precision $k$, using the boundary gap $\mathsf{Gap}(z_0, r, z)$ as the second modulus input.
\end{definition}

\section{Core theorem obligations}

\begin{definition}[Holomorphic core theorem obligations]
\label{def:holo-core-theorem-obligations}
The $\HolomorphicUp$ seed carries the following proof obligations before it can be presented as a certificate:
\begin{enumerate}
  \item uniform-modulus obligation: any accepted carrier witness must expose explicit modulus data, and equivalent witnesses must admit pointwise refinements compatible with $\hsame$;
  \item iterated-derivative obligation: the derivative sequence $\mathsf{IteratedCplxDiff}(f, z_0, n, f^{(n)})$ must be generated by unary recursion from the local complex-differentiability data;
  \item closure obligations: sum, product, and composition must preserve the carrier, with explicit modulus transport and an explicit image-containment witness for composition;
  \item power-series obligation: Taylor coefficients computed from the iterated-derivative data must determine a power-series witness with a convergence-radius comparison to $r$;
  \item identity obligation: agreement on a quantitatively dense sequence inside the disk must transport to agreement on an explicitly smaller disk;
  \item Cauchy-bound obligation: local upper bounds for $|f|$ on smaller disks must yield unary bounds for the iterated derivatives at the center.
\end{enumerate}
\end{definition}

\section{The seed fields}

\begin{definition}[Holomorphic source seed field]
\label{def:holo-source-spec}
The source seed field on $(f, z_0, r)$ records the intended constraint $\mathsf{HoloOnDisk}(f, z_0, r)$ together with its open-disk gap data.
\end{definition}

\begin{definition}[Holomorphic pattern seed field]
\label{def:holo-pattern-spec}
The pattern seed field is the uniform-modulus complex-differentiability predicate over the open disk.
\end{definition}

\begin{definition}[Holomorphic classifier seed field]
\label{def:holo-classifier-spec}
The classifier seed field groups two holomorphic-on-disk witnesses when they restrict to $\hsame$-equivalent pointwise functions on the disk's interior and carry compatible disk data.
\end{definition}

\begin{definition}[Holomorphic stability seed field]
\label{def:holo-stability-cert}
The stability seed field lists the required $\hsame$-respect of $\mathsf{HoloOnDisk}$ on the input function and on the disk parameters, transport of moduli under $\hsame$, and compatibility with the iterated-derivative recursion.
\end{definition}

\begin{definition}[Holomorphic ledger seed field]
\label{def:holo-ledger-policy}
The ledger seed field records the originating function history, center, radius, disk-gap witnesses, modulus witnesses, closure witnesses, Taylor-data witnesses, Cauchy-bound witnesses, and identity-theorem density witnesses.
\end{definition}

\begin{definition}[\texorpdfstring{$\HolomorphicUp$}{Holomorphic-up} certificate seed]
\label{def:holo-certificate-seed}
The public status of this chapter is $\text{Seed}$. The fields in this chapter specify the intended carrier, classifier, stability, and ledger data for $\HolomorphicUp$, while \autoref{def:holo-core-theorem-obligations} records the theorem obligations that must be supplied before the object can be treated as a paper certificate or a machine-checked certificate.
\end{definition}
