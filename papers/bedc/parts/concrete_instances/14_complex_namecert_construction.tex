\chapter{A Concrete Naming Certificate for \(\ComplexUp\)}
\label{ch:concrete-instances-complex-namecert}

The complex number interface \(\ComplexUp\) is licensed here by the product of two certified rational-history components from \autoref{ch:concrete-instances-rat-namecert}. The carrier, pattern, and ledger predicates are the same rational-pair history carrier, and the classifier is product history sameness restricted to those two rational-history components. It is not a primitive number system; it is licensed only via the rational source and the pair constructor of \autoref{ch:concrete-instances-prod-namecert}.

\begin{remark}[Complex certificate scope]
\label{rem:concrete-instances-complex-namecert-roadmap}
This chapter records the concrete history-level license: a complex history is
the continuation product of two rational-history witnesses, and the classifier
is the corresponding product classifier over internal history sameness.
\end{remark}

\section{Carrier definition}

\begin{definition}[Complex carrier]
\label{def:complex-carrier}
Use the BEDC product history carrier over two certified rational-history
components.
\end{definition}
\leandef{BEDC.Derived.ComplexUp.ComplexHistoryCarrier}

\begin{theorem}[Complex history carrier is unary]
\label{thm:complex-history-carrier-unary}
Every complex history carrier is a unary BEDC history.
\end{theorem}
\begin{proof}
Unpack the complex carrier as a product of two rational-history carriers and a
continuation witness. Each rational component gives a positive unary
denominator, hence a unary history. Unary histories are closed under the stored
continuation, so the complex carrier endpoint is unary.
\end{proof}
\leanchecked{BEDC.Derived.ComplexUp.ComplexHistoryCarrier\_unary}

\begin{theorem}[Complex carrier positive components]
\label{thm:complex-history-carrier-positive-components}
Every complex history carrier exposes real and imaginary rational-history
components whose denominators are positive unary histories, together with the
continuation witness assembling those components into the complex history.
\end{theorem}
\leanchecked{BEDC.Derived.ComplexUp.ComplexHistoryCarrier\_positive\_components}
\begin{proof}
Unpack the product-history carrier into its real component, imaginary
component, component carrier witnesses, and continuation witness. Apply the
rational-history carrier characterization to each component to read back the
positive unary denominator evidence, and keep the same continuation witness.
\end{proof}

\begin{theorem}[Complex e1-tail carrier component readback]
\label{thm:complex-history-carrier-e1-tail-components}
If $\mathsf{ComplexHistoryCarrier}(\Eone(tail))$ holds, then there are
histories $real$ and $imagTail$ such that $real$ is a rational-history
component, $\Eone(imagTail)$ is a rational-history component, and
$\Cont(real,imagTail,tail)$ holds.
\end{theorem}
\begin{proof}
Unpack the complex carrier into rational real and imaginary components and the
continuation witness assembling them into $\Eone(tail)$. Continuation
$\Eone$-result inversion gives either an empty imaginary component or an
$\Eone$-headed imaginary component with tail continuation. The empty branch
contradicts rational-history carrier non-emptiness; the visible branch exposes
the required rational component and tail continuation.
\end{proof}
\leanchecked{BEDC.Derived.ComplexUp.ComplexHistoryCarrier\_e1\_tail\_components}

\begin{theorem}[Complex carrier component reflexive classifiers]
\label{thm:complex-history-carrier-component-reflexive-classifiers}
Every complex history carrier exposes real and imaginary rational-history
components whose own rational-history classifiers hold reflexively, together
with the continuation witness assembling them into the complex history.
\end{theorem}
\begin{proof}
Unpack the complex carrier as a product-history carrier over two
rational-history carriers. The exposed real and imaginary components already
carry rational-history carrier evidence. Pair each component carrier with
itself and the reflexive $\hsame$ witness for that component to form the two
rational-history classifiers, and keep the stored continuation witness.
\end{proof}
\leanchecked{BEDC.Derived.ComplexUp.ComplexHistoryCarrier\_component\_reflexive\_classifiers}

\begin{theorem}[Complex carrier left unary context closure]
\label{thm:complex-history-carrier-prepend-unary-closed}
If \(p\) is a unary history and \(h\) is a complex history carrier, then
\(\mathsf{append}(p,h)\) is again a complex history carrier.
\end{theorem}
\begin{proof}
Unpack \(h\) as a rational real component, a rational imaginary component,
and a continuation witness. Close the real component under the left unary
denominator context \(p\), keep the imaginary component fixed, and reassociate
the continuation by associativity of \(\mathsf{append}\).
\end{proof}
\leanchecked{BEDC.Derived.ComplexUp.ComplexHistoryCarrier\_prepend\_unary\_closed}

\begin{theorem}[Complex carrier right unary context closure]
\label{thm:complex-history-carrier-append-unary-closed}
If \(h\) is a complex history carrier and \(q\) is a unary history, then
\(\mathsf{append}(h,q)\) is again a complex history carrier.
\end{theorem}
\begin{proof}
Unpack \(h\) as a rational real component, a rational imaginary component,
and a continuation witness. Close the imaginary component under the right
unary denominator context \(q\), keep the real component fixed, and reassociate
the continuation by associativity of \(\mathsf{append}\).
\end{proof}
\leanchecked{BEDC.Derived.ComplexUp.ComplexHistoryCarrier\_append\_unary\_closed}

\begin{theorem}[Complex history carrier is not empty]
\label{thm:complex-history-carrier-not-empty}
No complex history carrier can be identified with the empty history.
\end{theorem}
\begin{proof}
Unpack the complex carrier into its real rational component, imaginary rational
component, and continuation witness assembling the two components. Transport
the continuation result to the empty endpoint, invert the empty-result
continuation, and obtain that the real rational component is empty. This
contradicts the rational-history carrier non-emptiness theorem.
\end{proof}
\leanchecked{BEDC.Derived.ComplexUp.ComplexHistoryCarrier\_not\_empty}

\section{The certificate}

\begin{definition}[Complex source specification]
\label{def:complex-source-specification}
The source is \(\ProdUp(\RatUp,\RatUp)\) at the history-carrier level.
\end{definition}

\begin{definition}[Complex pattern specification]
\label{def:complex-pattern-specification}
The pattern predicate is exactly the rational-pair history carrier: a history
matches the pattern when it is a product history assembled from two certified
rational-history components.
\end{definition}

\begin{definition}[Complex classifier specification]
\label{def:complex-classifier-specification}
The classifier is product history classification over two rational-history
carriers.
\end{definition}
\leandef{BEDC.Derived.ComplexUp.ComplexHistoryClassifier}

\begin{theorem}[Complex history classifier nonempty endpoints]
\label{thm:complex-history-classifier-nonempty-endpoints}
If $\mathsf{ComplexHistoryClassifier}(h,k)$ holds, then neither endpoint can
be identified with the empty history.
\end{theorem}
\begin{proof}
Unpack the classifier into the two complex-history carrier witnesses and the
stored endpoint $\hsame$ link. Apply complex carrier non-emptiness separately
to the source endpoint carrier and to the target endpoint carrier.
\end{proof}
\leanchecked{BEDC.Derived.ComplexUp.ComplexHistoryClassifier\_nonempty\_endpoints}

\begin{theorem}[Complex classifier positive components]
\label{thm:complex-history-classifier-positive-components}
Every complex history classifier between two histories exposes real and
imaginary rational-history components for both endpoints, the continuation
ledgers assembling each endpoint, the endpoint $\hsame$ witness, and positive
unary denominator evidence for all four rational components.
\end{theorem}
\begin{proof}
Unpack the classifier into its two complex carrier witnesses and the stored
endpoint $\hsame$ proof. Apply the positive-component readback theorem to
each carrier. The resulting rational component carriers, continuation
ledgers, endpoint sameness, and positive denominator witnesses assemble
directly into the displayed endpoint-pair statement.
\end{proof}
\leanchecked{BEDC.Derived.ComplexUp.ComplexHistoryClassifier\_positive\_components}

\begin{theorem}[Complex e1-endpoint classifier component readback]
\label{thm:complex-history-classifier-e1-endpoint-components}
If $\mathsf{ComplexHistoryClassifier}(\Eone(tail),h)$ holds, or if
$\mathsf{ComplexHistoryClassifier}(h,\Eone(tail))$ holds, then there are
histories $real$ and $imagTail$ such that $real$ is a rational-history
component, $\Eone(imagTail)$ is a rational-history component, and
$\Cont(real,imagTail,tail)$ holds.
\end{theorem}
\begin{proof}
Unpack the complex classifier into its two endpoint carrier witnesses and the
stored endpoint $\hsame$ proof. In either endpoint position, the
$\Eone$-headed endpoint supplies a complex carrier. Apply the
$\Eone$-tail carrier component readback theorem to that carrier.
\end{proof}
\leanchecked{BEDC.Derived.ComplexUp.ComplexHistoryClassifier\_e1\_endpoint\_components}

\begin{theorem}[Complex classifier excludes an \(e0\) endpoint]
\label{thm:complex-history-classifier-e0-endpoint-absurd}
No complex history classifier can have an \(e0\)-headed history as either
endpoint.
\end{theorem}
\begin{proof}
Read the \(e0\)-headed endpoint as a complex carrier and expose its rational
real and imaginary components. The continuation result ending in \(e0\) either
makes the imaginary component empty or makes it itself \(e0\)-headed. Both
cases contradict the positive unary denominator witness attached to the
imaginary rational component.
\end{proof}
\leanchecked{BEDC.Derived.ComplexUp.ComplexHistoryClassifier\_e0\_endpoint\_absurd}

\begin{theorem}[Complex classifier excludes empty endpoints]
\label{thm:complex-history-classifier-empty-endpoint-absurd}
No complex history classifier can identify either endpoint with the empty
history.
\end{theorem}
\begin{proof}
Unpack the classifier into its two complex carrier witnesses. Apply complex
carrier non-emptiness separately to the source endpoint carrier and to the
target endpoint carrier.
\end{proof}
\leanchecked{BEDC.Derived.ComplexUp.ComplexHistoryClassifier\_empty\_endpoint\_absurd}

\begin{theorem}[Complex history classifier transitivity]
\label{thm:complex-history-classifier-transitivity}
If a complex history classifier relates \(h\) to \(k\), and also relates
\(k\) to \(r\), then it relates \(h\) to \(r\). The left endpoint carrier is
taken from the first classification, the right endpoint carrier is taken from
the second classification, and the history-sameness witnesses compose.
\end{theorem}
\begin{proof}
Unfolding the classifier exposes the two endpoint carrier witnesses together
with two \(hsame\) links. The endpoint carriers are preserved at the outside
of the chain, and the internal links compose by transitivity of \(hsame\).
\end{proof}
\leanchecked{BEDC.Derived.ComplexUp.ComplexHistoryClassifier\_trans}

\begin{theorem}[Complex history classifier symmetry]
\label{thm:complex-history-classifier-symmetry}
If $\mathsf{ComplexHistoryClassifier}(h,k)$, then $\mathsf{ComplexHistoryClassifier}(k,h)$.
\end{theorem}
\begin{proof}
Unfold the classifier as the product history classifier over two $\mathsf{RatHistoryCarrier}$ endpoints. The endpoint carrier witnesses are swapped, and the stored $\hsame(h,k)$ witness is reversed by symmetry to obtain $\hsame(k,h)$.
\end{proof}
\leanchecked{BEDC.Derived.ComplexUp.ComplexHistoryClassifier\_symm}

\begin{theorem}[Complex history classifier unary context closure]
\label{thm:complex-history-classifier-unary-context-closed}
If \(p\) and \(q\) are unary context histories, \(p\hsame p'\),
\(q\hsame q'\), and \(\mathsf{ComplexHistoryClassifier}(h,k)\), then the
context-wrapped endpoints
\(\mathsf{append}(p,\mathsf{append}(h,q))\) and
\(\mathsf{append}(p',\mathsf{append}(k,q'))\) are again related by the complex
history classifier.
\end{theorem}
\begin{proof}
Unpack the complex classifier into carrier witnesses for \(h\) and \(k\)
together with the endpoint \(\hsame\) link. Right unary context closure carries
both endpoint carriers through the tail contexts, after transporting unary
evidence across \(q\hsame q'\). Left unary context closure then carries those
results through the prefix contexts, after transporting unary evidence across
\(p\hsame p'\). The three \(\hsame\) links identify the two fully wrapped
histories, giving the classifier endpoint relation.
\end{proof}
\leanchecked{BEDC.Derived.ComplexUp.ComplexHistoryClassifier\_unary\_context\_closed}

\begin{theorem}[Complex history classifier unary context positive components]
\label{thm:complex-history-classifier-unary-context-positive-components}
If \(p\) and \(q\) are unary context histories, \(p\hsame p'\),
\(q\hsame q'\), and \(\mathsf{ComplexHistoryClassifier}(h,k)\), then the
context-wrapped endpoints
\(\mathsf{append}(p,\mathsf{append}(h,q))\) and
\(\mathsf{append}(p',\mathsf{append}(k,q'))\) expose real and imaginary
rational-history components for both endpoints, the two continuation ledgers,
the wrapped endpoint \(\hsame\) witness, and positive unary denominator
evidence for all four rational components.
\end{theorem}
\begin{proof}
Apply unary context closure to the source complex classifier and the matched
prefix and suffix contexts. This gives a complex classifier between the two
wrapped endpoints. Apply complex classifier positive-component readback to that
wrapped classifier. The readback exposes the four rational component carriers,
the two continuation ledgers assembling the wrapped endpoints, the endpoint
\(\hsame\) witness, and the four positive-denominator witnesses.
\end{proof}
\leanchecked{BEDC.Derived.ComplexUp.ComplexHistoryClassifier\_unary\_context\_positive\_components}

\begin{theorem}[Complex component classifier introduction]
\label{thm:complex-history-classifier-component-classifier-intro}
If the real components are related by $\mathsf{RatHistoryClassifier}$, the
imaginary components are related by $\mathsf{RatHistoryClassifier}$, and the
two endpoint histories are assembled by the corresponding continuation
witnesses, then the endpoint histories satisfy
$\mathsf{ComplexHistoryClassifier}$.
\end{theorem}
\begin{proof}
The source carrier is built from the source real and imaginary rational
component carriers and the first continuation witness. The target carrier is
built from the target rational component carriers and the second continuation
witness. The two component $\hsame$ witnesses are transported through the two
continuation ledgers by $\mathsf{Cont}$ respecting $\hsame$, yielding the
endpoint $\hsame$ field required by the complex classifier.
\end{proof}
\leanchecked{BEDC.Derived.ComplexUp.ComplexHistoryClassifier\_component\_classifier\_intro}

\begin{definition}[Complex stability certificate]
\label{def:complex-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item an inhabited rational-pair history carrier
  \item reflexivity of the product classifier on carrier histories
  \item symmetry of the product classifier
  \item transitivity of the product classifier
  \item carrier transport along product-classifier sameness
  \item pattern soundness for the rational-pair history carrier
  \item ledger soundness for the rational-pair history carrier
\end{enumerate}
\end{definition}

\begin{definition}[Complex ledger policy]
\label{def:complex-ledger-policy}
Record a raw rational-pair complex history carrier together with an \(\hsame\)
link to the visible complex history.
\end{definition}
\leandef{BEDC.Derived.ComplexUp.ComplexHistoryLedgerPolicy}

\begin{theorem}[Complex ledger visible carrier]
\label{thm:complex-ledger-visible-carrier}
If a complex ledger relates a raw complex history to a visible history, then
the visible endpoint is a complex history carrier.
\end{theorem}
\leanchecked{BEDC.Derived.ComplexUp.ComplexHistoryLedgerPolicy\_visible\_carrier}
\begin{proof}
Unpack the raw complex carrier as a product of two rational-history carriers
and a continuation witness. Transport the continuation result across the
stored raw-visible \(\hsame\) link and rebuild the rational-pair carrier at
the visible endpoint.
\end{proof}

\begin{theorem}[Complex ledger excludes an \(e0\)-visible endpoint]
\label{thm:complex-ledger-e0-visible-absurd}
No complex ledger policy can expose an \(e0\)-headed visible endpoint.
\end{theorem}
\begin{proof}
The visible-carrier theorem transports the raw rational-pair carrier to the
visible endpoint. Exposing the visible endpoint's rational components and
inverting the continuation result gives the same two impossible imaginary
denominator cases: empty or \(e0\)-headed.
\end{proof}
\leanchecked{BEDC.Derived.ComplexUp.ComplexHistoryLedgerPolicy\_e0\_visible\_absurd}

\begin{theorem}[Complex ledger raw-visible classifier]
\label{thm:complex-ledger-raw-visible-classifier}
If a complex ledger relates a raw complex history to a visible history, then
the raw and visible endpoints satisfy the complex history classifier.
\end{theorem}
\leanchecked{BEDC.Derived.ComplexUp.ComplexHistoryLedgerPolicy\_raw\_visible\_classifier}
\begin{proof}
Unpack the ledger into the raw complex carrier and the raw-visible
\(\hsame\) link. The visible carrier theorem supplies the visible endpoint
carrier, and these two carrier witnesses with the stored history link rebuild
the restricted product classifier.
\end{proof}

\begin{theorem}[Complex ledger classifier extension]
\label{thm:complex-ledger-classifier-extension}
If a complex ledger relates a raw history to a visible history, and the visible
history is classified with a target history, then the raw history is classified
with the same target.
\end{theorem}
\begin{proof}
Unpack the complex ledger into the raw carrier and the raw-visible
\(\hsame\) link. Unpack the visible-target classifier to keep the target
carrier and the visible-target \(\hsame\) link. The raw carrier and target
carrier form the endpoint data, and transitivity of \(\hsame\) composes the
two links.
\end{proof}
\leanchecked{BEDC.Derived.ComplexUp.ComplexHistoryLedgerPolicy\_classifier\_extension}

\begin{theorem}[Concrete \(\ComplexUp\) naming certificate]

The rational-pair history tuple forms a valid semantic NameCert for \(\ComplexUp\).
\end{theorem}
