\chapter{A Concrete Naming Certificate for \(\ComplexUp\)}
\label{ch:concrete-instances-complex-namecert}

The complex number interface \(\ComplexUp\) is licensed here by the product of two certified rational-history components from \autoref{ch:concrete-instances-rat-namecert}. The carrier, pattern, and ledger predicates are the same rational-pair history carrier, and the classifier is product history sameness restricted to those two rational-history components. It is not a primitive number system; it is licensed only via the rational source and the pair constructor of \autoref{ch:concrete-instances-prod-namecert}.

\begin{remark}[Complex certificate scope]
\label{rem:concrete-instances-complex-namecert-roadmap}
This chapter records the concrete history-level license: a complex history is
the continuation product of two rational-history witnesses, and the classifier
is the corresponding product classifier over internal history sameness.
\end{remark}

\section{Carrier definition}

\begin{definition}[Complex carrier]
\label{def:complex-carrier}
Use the BEDC product history carrier over two certified rational-history
components.
\end{definition}
\leandef{BEDC.Derived.ComplexUp.ComplexHistoryCarrier}

\begin{theorem}[Complex history carrier is unary]
\label{thm:complex-history-carrier-unary}
Every complex history carrier is a unary BEDC history.
\end{theorem}
\begin{proof}
Unpack the complex carrier as a product of two rational-history carriers and a
continuation witness. Each rational component gives a positive unary
denominator, hence a unary history. Unary histories are closed under the stored
continuation, so the complex carrier endpoint is unary.
\end{proof}
\leanchecked{BEDC.Derived.ComplexUp.ComplexHistoryCarrier\_unary}

\begin{theorem}[Complex carrier positive components]
\label{thm:complex-history-carrier-positive-components}
Every complex history carrier exposes real and imaginary rational-history
components whose denominators are positive unary histories, together with the
continuation witness assembling those components into the complex history.
\end{theorem}
\leanchecked{BEDC.Derived.ComplexUp.ComplexHistoryCarrier\_positive\_components}
\begin{proof}
Unpack the product-history carrier into its real component, imaginary
component, component carrier witnesses, and continuation witness. Apply the
rational-history carrier characterization to each component to read back the
positive unary denominator evidence, and keep the same continuation witness.
\end{proof}

\begin{theorem}[Complex carrier components are nonempty]
\label{thm:complex-history-carrier-components-nonempty}
Every complex history carrier exposes rational real and imaginary components
whose component histories are not identified with $\emp$.
\end{theorem}
\begin{proof}
Unpack the complex carrier into rational real and imaginary components and the
continuation assembling them. Rational-history carrier non-emptiness rejects
$real\hsame\emp$ and $imag\hsame\emp$ separately.
\end{proof}
\leanchecked{BEDC.Derived.ComplexUp.ComplexHistoryCarrier\_components\_nonempty}

\begin{theorem}[Complex e1-tail carrier component readback]
\label{thm:complex-history-carrier-e1-tail-components}
If $\mathsf{ComplexHistoryCarrier}(\Eone(tail))$ holds, then there are
histories $real$ and $imagTail$ such that $real$ is a rational-history
component, $\Eone(imagTail)$ is a rational-history component, and
$\Cont(real,imagTail,tail)$ holds.
\end{theorem}
\begin{proof}
Unpack the complex carrier into rational real and imaginary components and the
continuation witness assembling them into $\Eone(tail)$. Continuation
$\Eone$-result inversion gives either an empty imaginary component or an
$\Eone$-headed imaginary component with tail continuation. The empty branch
contradicts rational-history carrier non-emptiness; the visible branch exposes
the required rational component and tail continuation.
\end{proof}
\leanchecked{BEDC.Derived.ComplexUp.ComplexHistoryCarrier\_e1\_tail\_components}

\begin{theorem}[Complex e1-tail carrier nonempty component readback]
\label{thm:complex-history-carrier-e1-tail-components-nonempty}
If $\mathsf{ComplexHistoryCarrier}(\Eone(tail))$ holds, then the exposed real
component and the exposed visible imaginary component are rational-history
carriers and neither is history-same to $\emp$.
\end{theorem}
\begin{proof}
Apply the e1-tail component readback to obtain the rational real component,
the rational visible imaginary component, and the tail continuation. Rational
history-carrier non-emptiness applies separately to the two exposed component
carriers while the same continuation witness is retained.
\end{proof}
\leanchecked{BEDC.Derived.ComplexUp.ComplexHistoryCarrier\_e1\_tail\_components\_nonempty}

\begin{theorem}[Complex e1-tail carrier positive component readback]
\label{thm:complex-history-carrier-e1-tail-positive-components}
If $\mathsf{ComplexHistoryCarrier}(\Eone(tail))$ holds, then there are histories
$real$ and $imagTail$ such that $real$ is a rational-history component,
$\Eone(imagTail)$ is a rational-history component, $\Cont(real,imagTail,tail)$
holds, and both exposed rational components carry positive unary denominators.
\end{theorem}
\begin{proof}
First apply Theorem~\ref{thm:complex-history-carrier-e1-tail-components} to
read back the real component, the visible imaginary component, and the tail
continuation. Then apply the rational-history carrier characterization to both
rational component witnesses and keep the same continuation.
\end{proof}
\leanchecked{BEDC.Derived.ComplexUp.ComplexHistoryCarrier\_e1\_tail\_positive\_components}

\begin{theorem}[Complex carrier excludes an \(e0\) endpoint]
\label{thm:complex-history-carrier-e0-endpoint-absurd}
No complex history carrier can be an \(e0\)-headed history.
\end{theorem}
\begin{proof}
Every complex carrier is unary because it is assembled from unary
rational-history components by continuation. A zero-headed BEDC history cannot
be unary, so \(\mathsf{ComplexHistoryCarrier}(\Ezero(tail))\) is impossible.
\end{proof}
\leanchecked{BEDC.Derived.ComplexUp.ComplexHistoryCarrier\_e0\_endpoint\_absurd}

\begin{theorem}[Complex carrier component reflexive classifiers]
\label{thm:complex-history-carrier-component-reflexive-classifiers}
Every complex history carrier exposes real and imaginary rational-history
components whose own rational-history classifiers hold reflexively, together
with the continuation witness assembling them into the complex history.
\end{theorem}
\begin{proof}
Unpack the complex carrier as a product-history carrier over two
rational-history carriers. The exposed real and imaginary components already
carry rational-history carrier evidence. Pair each component carrier with
itself and the reflexive $\hsame$ witness for that component to form the two
rational-history classifiers, and keep the stored continuation witness.
\end{proof}
\leanchecked{BEDC.Derived.ComplexUp.ComplexHistoryCarrier\_component\_reflexive\_classifiers}

\begin{theorem}[Complex carrier left unary context closure]
\label{thm:complex-history-carrier-prepend-unary-closed}
If \(p\) is a unary history and \(h\) is a complex history carrier, then
\(\mathsf{append}(p,h)\) is again a complex history carrier.
\end{theorem}
\begin{proof}
Unpack \(h\) as a rational real component, a rational imaginary component,
and a continuation witness. Close the real component under the left unary
denominator context \(p\), keep the imaginary component fixed, and reassociate
the continuation by associativity of \(\mathsf{append}\).
\end{proof}
\leanchecked{BEDC.Derived.ComplexUp.ComplexHistoryCarrier\_prepend\_unary\_closed}

\begin{theorem}[Complex carrier right unary context closure]
\label{thm:complex-history-carrier-append-unary-closed}
If \(h\) is a complex history carrier and \(q\) is a unary history, then
\(\mathsf{append}(h,q)\) is again a complex history carrier.
\end{theorem}
\begin{proof}
Unpack \(h\) as a rational real component, a rational imaginary component,
and a continuation witness. Close the imaginary component under the right
unary denominator context \(q\), keep the real component fixed, and reassociate
the continuation by associativity of \(\mathsf{append}\).
\end{proof}
\leanchecked{BEDC.Derived.ComplexUp.ComplexHistoryCarrier\_append\_unary\_closed}

\begin{theorem}[Complex carrier unary continuation closure]
\label{thm:complex-history-carrier-unary-continuation-closed}
If \(h\) is a complex history carrier, \(q\) is unary, and
\(\Cont(h,q,zq)\), then \(zq\) is again a complex history carrier.
\end{theorem}
\begin{proof}
Unpack \(h\) into rational real and imaginary components with the continuation
assembling them. Associativity of continuation splits the suffix \(q\) through
the imaginary component. The imaginary rational carrier is closed under the
unary suffix and transported along the split continuation result, while the
real rational carrier is kept fixed. The resulting component continuation
reassembles \(zq\) as a complex history carrier.
\end{proof}
\leanchecked{BEDC.Derived.ComplexUp.ComplexHistoryCarrier\_unary\_continuation\_closed}

\begin{theorem}[Complex history carrier is not empty]
\label{thm:complex-history-carrier-not-empty}
No complex history carrier can be identified with the empty history.
\end{theorem}
\begin{proof}
Unpack the complex carrier into its real rational component, imaginary rational
component, and continuation witness assembling the two components. Transport
the continuation result to the empty endpoint, invert the empty-result
continuation, and obtain that the real rational component is empty. This
contradicts the rational-history carrier non-emptiness theorem.
\end{proof}
\leanchecked{BEDC.Derived.ComplexUp.ComplexHistoryCarrier\_not\_empty}

\section{The certificate}

\begin{theorem}[Complex history classifier nonempty endpoints]
\label{thm:complex-history-classifier-nonempty-endpoints}
If $\mathsf{ComplexHistoryClassifier}(h,k)$ holds, then neither endpoint can
be identified with the empty history.
\end{theorem}
\begin{proof}
Unpack the classifier into the two complex-history carrier witnesses and the
stored endpoint $\hsame$ link. Apply complex carrier non-emptiness separately
to the source endpoint carrier and to the target endpoint carrier.
\end{proof}

\begin{theorem}[Complex history classifier endpoint e-one shapes]
\label{thm:complex-history-classifier-endpoint-e1-shapes}
If $\mathsf{ComplexHistoryClassifier}(h,k)$ holds, then both classifier
endpoints are visibly one-headed histories with unary tails:
$$
\bigl(\exists h_t,\ h=\Eone(h_t)\land\mathsf{UnaryHistory}(h_t)\bigr)
\land
\bigl(\exists k_t,\ k=\Eone(k_t)\land\mathsf{UnaryHistory}(k_t)\bigr).
$$
\end{theorem}
\begin{proof}
The classifier gives carrier witnesses for both endpoints. Complex carrier
unarity supplies unary-history evidence for each endpoint, while classifier
non-emptiness rules out the empty history on both sides. Applying unary
nonempty one-tail readback to each endpoint gives the displayed shapes.
\end{proof}

\begin{theorem}[Complex history classifier transitivity]
\label{thm:complex-history-classifier-transitivity}
If a complex history classifier relates \(h\) to \(k\), and also relates
\(k\) to \(r\), then it relates \(h\) to \(r\). The left endpoint carrier is
taken from the first classification, the right endpoint carrier is taken from
the second classification, and the history-sameness witnesses compose.
\end{theorem}
\begin{proof}
Unfolding the classifier exposes the two endpoint carrier witnesses together
with two \(hsame\) links. The endpoint carriers are preserved at the outside
of the chain, and the internal links compose by transitivity of \(hsame\).
\end{proof}
\leanchecked{BEDC.Derived.ComplexUp.ComplexHistoryClassifier\_trans}

\begin{theorem}[Complex history classifier symmetry]
\label{thm:complex-history-classifier-symmetry}
If $\mathsf{ComplexHistoryClassifier}(h,k)$, then $\mathsf{ComplexHistoryClassifier}(k,h)$.
\end{theorem}
\begin{proof}
Unfold the classifier as the product history classifier over two $\mathsf{RatHistoryCarrier}$ endpoints. The endpoint carrier witnesses are swapped, and the stored $\hsame(h,k)$ witness is reversed by symmetry to obtain $\hsame(k,h)$.

\begin{theorem}[Complex history classifier unary context closure]
\label{thm:complex-history-classifier-unary-context-closed}
If \(p\) and \(q\) are unary context histories, \(p\hsame p'\),
\(q\hsame q'\), and \(\mathsf{ComplexHistoryClassifier}(h,k)\), then the
context-wrapped endpoints
\(\mathsf{append}(p,\mathsf{append}(h,q))\) and
\(\mathsf{append}(p',\mathsf{append}(k,q'))\) are again related by the complex
history classifier.
\end{theorem}
\begin{proof}
Unpack the complex classifier into carrier witnesses for \(h\) and \(k\)
together with the endpoint \(\hsame\) link. Right unary context closure carries
both endpoint carriers through the tail contexts, after transporting unary
evidence across \(q\hsame q'\). Left unary context closure then carries those
results through the prefix contexts, after transporting unary evidence across
\(p\hsame p'\). The three \(\hsame\) links identify the two fully wrapped
histories, giving the classifier endpoint relation.
\end{proof}
\leanchecked{BEDC.Derived.ComplexUp.ComplexHistoryClassifier\_unary\_context\_closed}
