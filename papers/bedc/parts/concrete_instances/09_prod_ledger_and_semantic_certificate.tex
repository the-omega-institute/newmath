\begin{definition}[Prod ledger policy]
\label{def:prod-ledger-policy}
Record a raw product history carrying both component ledgers together with an
$\hsame$ link to the visible history exposed by the ledger.
\end{definition}
\leandef{BEDC.Derived.ProdUp.ProdHistoryLedgerPolicy}

\begin{theorem}[Prod ledger visible carrier]
\label{thm:prod-ledger-visible-carrier}
If a visible history is linked by $\hsame$ to a raw product-ledger history,
then the visible history is itself a product carrier.
\end{theorem}
\begin{proof}
Unpack the raw product carrier and transport it across the stored $\hsame$
link by product-carrier transport.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdHistoryLedgerPolicy\_visible\_carrier}

\begin{theorem}[Prod ledger raw-visible classifier]
\label{thm:prod-ledger-raw-visible-classifier}
If $\mathsf{ProdHistoryLedgerPolicy}(L,R,\rho,v)$ holds, then
$\mathsf{ProdHistoryClassifier}(L,R,\rho,v)$ holds.
\end{theorem}
\begin{proof}
Unpack the ledger into the raw product carrier for $\rho$ and the stored
$\hsame(\rho,v)$ link. The visible-carrier theorem supplies the product
carrier for $v$. These three witnesses are exactly the product classifier
fields for the raw ledger history and the visible history.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdHistoryLedgerPolicy\_raw\_visible\_classifier}

\begin{theorem}[Prod ledger history-same transport]
\label{thm:prod-ledger-hsame-transport}
Let $L,R:\mathsf{BHist}\to\mathsf{Prop}$ be source carriers and let

$\mathsf{BHist}$ with $\Cont$ and $\hsame$; carrier
$\mathsf{ProdHistoryCarrier}$; classifier $\mathsf{ProdHistoryClassifier}$;
ledger $\mathsf{ProdHistoryLedgerPolicy}$; exactness raw-visible transport;
standard bridge displayed ordered-pair ledger; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Unpack the ledger into a raw product carrier for $\rho$ and the stored
raw-visible link $\hsame(\rho,v)$. Transport the raw carrier across
$\hsame(\rho,\rho')$ by Theorem~\ref{thm:prod-history-carrier-hsame-transport}.
For the visible link, compose symmetry of $\hsame(\rho,\rho')$, the stored
raw-visible link, and $\hsame(v,v')$ to obtain $\hsame(\rho',v')$. The
transported raw carrier and this composed link rebuild
$\mathsf{ProdHistoryLedgerPolicy}(L,R,\rho',v')$.
\end{proof}

Induct on the first ledger-chain derivation. A one-ledger derivation is a

\begin{theorem}[Product ledger-chain boundary classifier decomposition]
\label{thm:prod-ledger-chain-boundary-classifier-decomposition}
Let $L,R:\mathsf{BHist}\to\mathsf{Prop}$ be source carriers, and let
$\rho,z:\mathsf{BHist}$. If
$\mathsf{ProdHistoryLedgerChain}_{L,R}(\rho,z)$ holds, then both boundary
clauses hold:
\begin{enumerate}
  \item either $\mathsf{ProdHistoryLedgerPolicy}(L,R,\rho,z)$ holds, or there
  exists $v:\mathsf{BHist}$ such that
  $\mathsf{ProdHistoryLedgerPolicy}(L,R,\rho,v)$ and
  $\mathsf{ProdHistoryClassifier}(L,R,v,z)$ hold;
  \item either $\mathsf{ProdHistoryLedgerPolicy}(L,R,\rho,z)$ holds, or there
  exists $v:\mathsf{BHist}$ such that
  $\mathsf{ProdHistoryClassifier}(L,R,\rho,v)$ and
  $\mathsf{ProdHistoryLedgerPolicy}(L,R,v,z)$ hold.
\end{enumerate}
The object fit is: traditional target finite ordered-product provenance; BEDC
source $\mathsf{BHist}$,
$\mathsf{ProdHistoryLedgerPolicy}$, and
$\mathsf{ProdHistoryLedgerChain}$; carrier
$\mathsf{ProdHistoryCarrier}$; classifier
$\mathsf{ProdHistoryClassifier}$; ledger finite product ledger chain; core
theorem boundary decomposition; exactness endpoint classifier preservation.
\end{theorem}
\begin{proof}
For the head boundary clause, perform cases on the finite ledger-chain
constructor. The one-step constructor is exactly
$\mathsf{ProdHistoryLedgerPolicy}(L,R,\rho,z)$.

In the cons constructor, the displayed first ledger has the form
$\mathsf{ProdHistoryLedgerPolicy}(L,R,\rho,v)$ and the tail has the form
$\mathsf{ProdHistoryLedgerChain}_{L,R}(v,z)$. Apply
Theorem~\ref{thm:prod-ledger-chain-envelope-closure} to the tail; its envelope
part gives $\mathsf{ProdHistoryClassifier}(L,R,v,z)$, so the witness $v$
exposes the first concrete ledger step and the classifier tail consumed by
readback.

For the terminal boundary clause, induct on the chain. The base case again is
the displayed one-step ledger
$\mathsf{ProdHistoryLedgerPolicy}(L,R,\rho,z)$.

In the cons step, first apply
Theorem~\ref{thm:prod-ledger-raw-visible-classifier} to the leading policy
$\mathsf{ProdHistoryLedgerPolicy}(L,R,\rho,v)$, obtaining
$\mathsf{ProdHistoryClassifier}(L,R,\rho,v)$. If the tail is a one-step ledger,
the same $v$ is the terminal-boundary witness. If the inductive tail has already
exposed a terminal ledger
$\mathsf{ProdHistoryLedgerPolicy}(L,R,u,z)$ with classifier prefix
$\mathsf{ProdHistoryClassifier}(L,R,v,u)$, compose the leading classifier with
that prefix by classifier transitivity to obtain
$\mathsf{ProdHistoryClassifier}(L,R,\rho,u)$. The witness $u$ exposes the last
concrete ledger step and the accumulated classifier prefix.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdHistoryLedgerChain\_boundary\_classifier\_decomposition}
