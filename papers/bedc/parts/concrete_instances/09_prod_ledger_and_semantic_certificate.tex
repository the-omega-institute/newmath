\begin{definition}[Prod ledger policy]
\label{def:prod-ledger-policy}
Record a raw product history carrying both component ledgers together with an
$\hsame$ link to the visible history exposed by the ledger.
\end{definition}
\leandef{BEDC.Derived.ProdUp.ProdHistoryLedgerPolicy}

\begin{theorem}[Prod ledger visible carrier]
\label{thm:prod-ledger-visible-carrier}
If a visible history is linked by $\hsame$ to a raw product-ledger history,
then the visible history is itself a product carrier.
\end{theorem}
\begin{proof}
Unpack the raw product carrier and transport it across the stored $\hsame$
link by product-carrier transport.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdHistoryLedgerPolicy\_visible\_carrier}

\begin{theorem}[Prod ledger raw-visible classifier]
\label{thm:prod-ledger-raw-visible-classifier}
If $\mathsf{ProdHistoryLedgerPolicy}(L,R,\rho,v)$ holds, then
$\mathsf{ProdHistoryClassifier}(L,R,\rho,v)$ holds.
\end{theorem}
\begin{proof}
Unpack the ledger into the raw product carrier for $\rho$ and the stored
$\hsame(\rho,v)$ link. The visible-carrier theorem supplies the product
carrier for $v$. These three witnesses are exactly the product classifier
fields for the raw ledger history and the visible history.
\end{proof}
\leanchecked{BEDC.Derived.ProdUp.ProdHistoryLedgerPolicy\_raw\_visible\_classifier}

\mathsf{ProdComponentHistoryClassifier}_{L,R,\sim_L,\sim_R}(\rho,w)
\Longleftrightarrow
\mathsf{ProdComponentHistoryClassifier}_{L,R,\sim_L,\sim_R}(v,w)
$$
and
$$
\mathsf{ProdComponentHistoryClassifier}_{L,R,\sim_L,\sim_R}(w,\rho)
\Longleftrightarrow
\mathsf{ProdComponentHistoryClassifier}_{L,R,\sim_L,\sim_R}(w,v).
$$

replacement at the public componentwise classifier; BEDC source
$\mathsf{BHist}$ with $\Cont$ and $\hsame$; carrier

representation back to $\rho$. The coherence witness
$\mathsf{ProdPairRepCoherent}_{L,R,\sim_L,\sim_R}$ compares the raw

and visible pair representations, these links build
$\mathsf{ProdComponentHistoryClassifier}_{L,R,\sim_L,\sim_R}(\rho,v)$.

Apply Theorem~\ref{thm:prod-component-history-classifier-symmetry-from-source-certificates}

Theorem~\ref{thm:prod-component-history-classifier-trans-from-pair-coherence}.
For the second displayed clause, use the same two bridge directions and the
same transitivity theorem, now composing on the right endpoint. The two source
certificates supply the required componentwise symmetry and transitivity inputs
through those product classifier theorems.
\end{proof}

\begin{definition}[Prod ledger chain]
\label{def:prod-ledger-chain}
For source carriers $L,R:\mathsf{BHist}\to\mathsf{Prop}$, the relation
$\mathsf{ProdHistoryLedgerChain}_{L,R}(\rho,z)$ is the inductively generated
finite ledger-provenance relation on product histories with these constructors.
A single displayed ledger gives a chain:
$$
\frac{\mathsf{ProdHistoryLedgerPolicy}(L,R,\rho,z)}
{\mathsf{ProdHistoryLedgerChain}_{L,R}(\rho,z)}.
$$
A displayed ledger followed by a chain gives a chain:
$$
\frac{\mathsf{ProdHistoryLedgerPolicy}(L,R,\rho,v)\quad
\mathsf{ProdHistoryLedgerChain}_{L,R}(v,z)}
{\mathsf{ProdHistoryLedgerChain}_{L,R}(\rho,z)}.
$$
\end{definition}
\leandef{BEDC.Derived.ProdUp.ProdHistoryLedgerChain}

\begin{theorem}[Product ledger chain envelope closure]
\label{thm:prod-ledger-chain-envelope-closure}
Let $L,R:\mathsf{BHist}\to\mathsf{Prop}$ be source carriers and let
$\rho,z:\mathsf{BHist}$. If
$\mathsf{ProdHistoryLedgerChain}_{L,R}(\rho,z)$ holds, then
$\mathsf{ProdHistoryCarrier}(L,R,z)$ and
$\mathsf{ProdHistoryClassifier}(L,R,\rho,z)$ hold. Moreover, for every
$w:\mathsf{BHist}$ both history-envelope endpoint-replacement clauses hold:
$$
\mathsf{ProdHistoryClassifier}(L,R,\rho,w)
\Longleftrightarrow
\mathsf{ProdHistoryClassifier}(L,R,z,w)
$$
and
$$
\mathsf{ProdHistoryClassifier}(L,R,w,\rho)
\Longleftrightarrow
\mathsf{ProdHistoryClassifier}(L,R,w,z).
$$
The object fit is: traditional target finite ordered-product ledger closure;
BEDC source $\mathsf{BHist}$ with $\Cont$ and $\hsame$; carrier
$\mathsf{ProdHistoryCarrier}$; classifier $\mathsf{ProdHistoryClassifier}$;
ledger $\mathsf{ProdHistoryLedgerChain}$; core theorem this envelope closure;
exactness classifier-invariant raw and final endpoints; standard bridge
displayed ordered-pair provenance; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Induct on the derivation of
$\mathsf{ProdHistoryLedgerChain}_{L,R}(\rho,z)$. In the one-step constructor,
Theorem~\ref{thm:prod-ledger-visible-carrier} gives
$\mathsf{ProdHistoryCarrier}(L,R,z)$ from the displayed ledger, and
Theorem~\ref{thm:prod-ledger-raw-visible-classifier} gives
$\mathsf{ProdHistoryClassifier}(L,R,\rho,z)$. The left endpoint-replacement
clause is exactly
Theorem~\ref{thm:prod-ledger-classifier-endpoint-equivalence}. The right
endpoint-replacement clause follows by applying product history classifier
symmetry to turn each right-endpoint statement into the corresponding
left-endpoint statement, using the same endpoint theorem, and turning back by
symmetry.

In the cons constructor, write the first displayed ledger as
$\mathsf{ProdHistoryLedgerPolicy}(L,R,\rho,v)$ and the remaining chain as
$\mathsf{ProdHistoryLedgerChain}_{L,R}(v,z)$. The induction hypothesis supplies
$\mathsf{ProdHistoryCarrier}(L,R,z)$,
$\mathsf{ProdHistoryClassifier}(L,R,v,z)$, and the two endpoint equivalences
from $v$ to $z$. The first ledger supplies
$\mathsf{ProdHistoryClassifier}(L,R,\rho,v)$ by
Theorem~\ref{thm:prod-ledger-raw-visible-classifier}; compose it with the
induction classifier by Theorem~\ref{thm:prod-history-classifier-transitivity}
to obtain $\mathsf{ProdHistoryClassifier}(L,R,\rho,z)$. Compose the one-step
endpoint equivalence from $\rho$ to $v$ with the induction endpoint equivalence
from $v$ to $z$ for the left clause. Use the same composition after the
symmetry conversion for the right clause. Thus the finite product ledger chain
closes inside the history envelope while preserving both raw and final endpoint
classifiers.
\end{proof}

\exists l,r:\mathsf{BHist}.\;\mathsf{ProdPairRep}_{L,R}(h,l,r)$, the following
two clauses hold:
\begin{enumerate}
  \item the componentwise public certificate
  $$
  \mathsf{SemanticNameCert}(\mathsf{ProdHistoryCarrier}(L,R),
  \mathsf{ProdHistoryCarrier}(L,R),

  \mathsf{ProdComponentHistoryClassifier}_{L,R,\sim_L,\sim_R});

  \item for all histories $\rho,z,l,r:\mathsf{BHist}$, if
  $\mathsf{ProdHistoryLedgerChain}_{L,R}(\rho,z)$ and
  $\mathsf{ProdPairRep}_{L,R}(\rho,l,r)$ hold, then
  $\mathsf{ProdHistoryClassifier}(L,R,\rho,z)$ holds and there exist
  $l',r':\mathsf{BHist}$ such that
  $$
  \mathsf{ProdPairRep}_{L,R}(z,l',r')
  \quad\hbox{and}\quad
  l\sim_L l'
  \quad\hbox{and}\quad
  r\sim_R r'.
  $$
\end{enumerate}
The object fit is: traditional target ordered-product public certificate with
finite ledger-chain component readback; BEDC source $\mathsf{BHist}$ with
$\Cont$ and $\hsame$; carrier $\mathsf{ProdHistoryCarrier}$ plus displayed
$\mathsf{ProdPairRep}$; classifier componentwise source classifiers; stability
by envelope closure; ledger $\mathsf{ProdHistoryLedgerChain}$; core theorem this
combined public-certificate and readback package; exactness envelope classifier
and final displayed component witnesses; standard bridge ordered-pair
provenance; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
The public certificate clause is
Theorem~\ref{thm:prod-component-history-semantic-certificate-target} applied to
$\mathsf{cert}_L$, $\mathsf{cert}_R$, and
$\mathsf{ProdPairRepCoherent}_{L,R,\sim_L,\sim_R}$, with the displayed pair ledger
$D$ as stated.

For the ledger-chain readback clause, fix $\rho,z,l,r:\mathsf{BHist}$ and assume
$\mathsf{ProdHistoryLedgerChain}_{L,R}(\rho,z)$ together with
$\mathsf{ProdPairRep}_{L,R}(\rho,l,r)$. Theorem~\ref{thm:prod-ledger-chain-envelope-closure}
applied to the finite chain supplies
$\mathsf{ProdHistoryClassifier}(L,R,\rho,z)$, which is the envelope-classifier
part of the conclusion.

It remains to expose the final displayed components. Apply
Theorem~\ref{thm:prod-ledger-chain-displayed-component-readback} to the same chain
and initial displayed pair representation, using the coherence witness. This
gives $l',r':\mathsf{BHist}$ with
$\mathsf{ProdPairRep}_{L,R}(z,l',r')$, $l\sim_L l'$, and $r\sim_R r'$; conjoining
these witnesses with the envelope classifier gives the required public-object
ledger guarantee.
\end{proof}

\begin{theorem}[Concrete product history-envelope semantic certificate]
\label{thm:concrete-prod-namecert}
For source-history predicates $L,R:\mathsf{BHist}\to\mathsf{Prop}$, assume
$\exists l:\mathsf{BHist}.\;L(l)$ and $\exists r:\mathsf{BHist}.\;R(r)$. Then
$$
\mathsf{SemanticNameCert}(\mathsf{ProdHistoryCarrier}(L,R),\mathsf{ProdHistoryCarrier}(L,R),\mathsf{ProdHistoryCarrier}(L,R),\mathsf{ProdHistoryClassifier}(L,R))
$$
holds. This is the checked ordered-product history-envelope certificate; source
certificates supply the two inhabited-carrier hypotheses through their fields.
\end{theorem}
\begin{proof}
Choose the left and right carrier witnesses first. BEDC continuation joins them
into a product history. The history-envelope classifier stores the two endpoint
carrier witnesses and an $\hsame$ witness between product histories, so the
certificate fields follow by $\hsame$ reflexivity, symmetry, transitivity, and
carrier transport. When source certificates are present, their carrier-inhabited
fields provide the same witnesses.
\end{proof}

\begin{corollary}[Product history envelope is licensed, not primitive]
\label{cor:prod-history-envelope-licensed-not-primitive}
The checked theorem licenses only $\mathsf{ProdHistoryCarrier}$ with
$\mathsf{ProdHistoryClassifier}$ under explicit left and right carrier
witnesses. The public ordered-product certificate uses the history-level
$\mathsf{ProdComponentHistoryClassifier}_{L,R,\sim_L,\sim_R}$ induced by
$\mathsf{ProdPairRep}$ and source classifiers, together with displayed
component exposure, $\hsame$-sound descent, and ledger fields from this chapter.
\end{corollary}
