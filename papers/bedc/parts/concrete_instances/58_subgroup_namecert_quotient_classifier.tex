\begin{theorem}[Centralizer normalizer SubgroupUp certificate]
\label{thm:subgroup-centralizer-normalizer-subgroup-certificate}
For a BEDC history-level $\GroupUp$ operation over $\mathsf{BHist}$ satisfying
the same associativity, unit, multiplication-congruence, and two-sided inverse
laws as in \autoref{thm:subgroup-centralizer-certificate-target}, the carrier
$\mathsf{Norm}_{Z_a}$ of
\autoref{def:subgroup-centralizer-normalizer-witness} supplies the concrete
$\SubgroupUp$ certificate rows for the normalizer of $Z_a$:
\begin{enumerate}
\item $\mathsf{Norm}_{Z_a}(e)$.
\item If $\mathsf{Norm}_{Z_a}(s)$ and $\mathsf{Norm}_{Z_a}(t)$, then
$\mathsf{Norm}_{Z_a}(\mathsf{mul}(s,t))$.
\item If $\mathsf{Norm}_{Z_a}(t)$, then
$\mathsf{Norm}_{Z_a}(\mathsf{inv}(t))$.
\item If $\mathsf{Norm}_{Z_a}(s)$ and $s\hsame t$, then
$\mathsf{Norm}_{Z_a}(t)$.
\item If $Z_a(t)$, then $\mathsf{Norm}_{Z_a}(t)$.
\end{enumerate}
Thus the two-sided normalizer carrier has identity membership, product closure,
inverse closure, $\hsame$-transport, and the inclusion row from the centralizer.
\end{theorem}
\begin{proof}
For identity membership, unfold
\autoref{def:subgroup-centralizer-normalizer-witness}. The unit laws make
conjugation by $e$ act as the identity on every carried $x$, and the two-sided
inverse laws identify $\mathsf{inv}(e)$ with $e$. Carrier transport from
\autoref{thm:group-centralizer-subgroup} returns both clauses.

For product closure, assume $\mathsf{Norm}_{Z_a}(s)$ and
$\mathsf{Norm}_{Z_a}(t)$, reading these as expanded $N_a$ witnesses. Apply
\autoref{thm:group-normalizer-conjugation-action-composition}: its forward
composition row sends $Z_a(x)$ first through the $t$ clause and then through the
$s$ clause, while its inverse-action row sends $Z_a(x)$ first through the
inverse clause for $s$ and then through the inverse clause for $t$. These are
the two clauses of $\mathsf{Norm}_{Z_a}(\mathsf{mul}(s,t))$.

For inverse closure, the first clause of
$\mathsf{Norm}_{Z_a}(\mathsf{inv}(t))$ is exactly the inverse-conjugation half
of $\mathsf{Norm}_{Z_a}(t)$. Its second clause conjugates by
$\mathsf{inv}(\mathsf{inv}(t))$; the two-sided inverse laws identify this word
with the original conjugation by $t$, and multiplication congruence plus
carrier transport move the first clause across the comparison.

For $\hsame$-transport, assume $\mathsf{Norm}_{Z_a}(s)$ and $s\hsame t$.
Multiplication congruence transports the forward conjugation word with $s$ to
the corresponding word with $t$, and inverse congruence transports the inverse
word. Carrier transport from \autoref{thm:group-centralizer-subgroup} therefore
gives both clauses of $\mathsf{Norm}_{Z_a}(t)$.

The inclusion row is exactly
\autoref{thm:subgroup-centralizer-self-normalizer}: every $Z_a(t)$ supplies both
normalizer clauses. Repacking the five rows with the restricted $\hsame$
classifier gives the advertised concrete $\SubgroupUp$ certificate for the
normalizer carrier.
\end{proof}

\autoref{thm:subgroup-centralizer-certificate-target}. It supplies the

\autoref{thm:subgroup-centralizer-self-normalizer} gives the same witness by

\begin{definition}[Centralizer normalizer quotient classifier]
\label{def:centralizer-normalizer-quotient-classifier}
For the centralizer carrier $Z_a$ and its normalizer carrier $N_a$, define
the concrete quotient classifier predicate $Q_{Z_a\leq N_a}$ on
$\mathsf{BHist}$ by
$$
Q_{Z_a\leq N_a}(x,y)
:\Longleftrightarrow
N_a(x)\land N_a(y)\land
\exists z:\mathsf{BHist},\ Z_a(z)\land
\hsame(y,\mathsf{mul}(x,z)).
$$
The endpoints are carried by $N_a$, while the right witness is carried by
$Z_a$. Thus $Q_{Z_a\leq N_a}$ is a classifier predicate over concrete BEDC
histories, not a host quotient object.
\end{definition}

\begin{theorem}[Centralizer normalizer quotient classifier laws]
\label{thm:centralizer-normalizer-quotient-classifier-laws}
For a BEDC history-level $\GroupUp$ operation over $\mathsf{BHist}$ satisfying
the same associativity, unit, multiplication-congruence, and two-sided inverse
laws as in \autoref{thm:subgroup-centralizer-certificate-target}, the
classifier $Q_{Z_a\leq N_a}$ of
\autoref{def:centralizer-normalizer-quotient-classifier} has the quotient
classifier laws on $N_a$:
\begin{enumerate}
\item if $N_a(x)$, then $Q_{Z_a\leq N_a}(x,x)$;
\item if $Q_{Z_a\leq N_a}(x,y)$, then $Q_{Z_a\leq N_a}(y,x)$;
\item if $Q_{Z_a\leq N_a}(x,y)$ and $Q_{Z_a\leq N_a}(y,u)$, then
$Q_{Z_a\leq N_a}(x,u)$;
\item if $Q_{Z_a\leq N_a}(x,y)$ and $Q_{Z_a\leq N_a}(u,v)$, then
$Q_{Z_a\leq N_a}(\mathsf{mul}(x,u),\mathsf{mul}(y,v))$.
\end{enumerate}
\end{theorem}
\begin{proof}
The endpoint rows are supplied by
\autoref{thm:subgroup-centralizer-normalizer-tower-certificate}. The identity
row of \autoref{thm:subgroup-centralizer-certificate-target} gives
$Z_a(\mathsf{e})$, and the right-unit law rewrites
$\mathsf{mul}(x,\mathsf{e})$ to $x$ up to $\hsame$. This is the required
witness for reflexivity on $N_a$.

For symmetry, suppose $Q_{Z_a\leq N_a}(x,y)$ is witnessed by $z$ with
$Z_a(z)$ and $\hsame(y,\mathsf{mul}(x,z))$. The inverse-closure row of
\autoref{thm:subgroup-centralizer-certificate-target} gives
$Z_a(\mathsf{inv}(z))$. The two-sided inverse laws and multiplication
congruence transport the displayed witness equation to
$\hsame(x,\mathsf{mul}(y,\mathsf{inv}(z)))$, so the inverse witness proves
$Q_{Z_a\leq N_a}(y,x)$.

For transitivity, take witnesses $z$ and $w$ for
$Q_{Z_a\leq N_a}(x,y)$ and $Q_{Z_a\leq N_a}(y,u)$. Product closure for $Z_a$
gives $Z_a(\mathsf{mul}(z,w))$. Associativity and multiplication congruence
compose the two classifier equations into
$\hsame(u,\mathsf{mul}(x,\mathsf{mul}(z,w)))$, giving
$Q_{Z_a\leq N_a}(x,u)$.

It remains to check multiplication compatibility. Let $z$ witness
$Q_{Z_a\leq N_a}(x,y)$ and let $w$ witness $Q_{Z_a\leq N_a}(u,v)$. Since
$u$ is carried by $N_a$, the normality row in
\autoref{thm:subgroup-centralizer-normalizer-tower-certificate}, equivalently
\autoref{thm:group-centralizer-normal-in-normalizer}, moves $z$ across the
right endpoint and gives
$Z_a(\mathsf{Conj}_{\mathsf{inv}(u)}(z))$.

Product closure for $Z_a$ combines this conjugated witness with $w$. The
associativity, inverse, and multiplication-congruence laws rewrite
$\mathsf{mul}(y,v)$ to
$\mathsf{mul}(\mathsf{mul}(x,u),
\mathsf{mul}(\mathsf{Conj}_{\mathsf{inv}(u)}(z),w))$ up to $\hsame$.
Together with product closure of $N_a$ from
\autoref{thm:group-centralizer-normalizer-subgroup}, this gives
$Q_{Z_a\leq N_a}(\mathsf{mul}(x,u),\mathsf{mul}(y,v))$.
\end{proof}

$$

middle $y,\mathsf{inv}(y)$ pair. The carrier transport row for $Z_a$ moves the
product witness to
$Z_a(\mathsf{mul}(\mathsf{inv}(x),z))$, which is exactly the kernel clause for
$Q_{Z_a\triangleleft N_a}(x,z)$.

\begin{theorem}[Centralizer normalizer quotient-kernel classifier congruence]
\label{thm:subgroup-centralizer-normalizer-quotient-kernel-congruence}
For a BEDC history-level $\GroupUp$ operation over $\mathsf{BHist}$ satisfying
the same associativity, unit, multiplication-congruence, and two-sided inverse
laws as in
\autoref{thm:subgroup-centralizer-normalizer-tower-certificate}, the relation
$Q_{Z_a\triangleleft N_a}$ of
\autoref{def:subgroup-centralizer-normalizer-quotient-kernel-relation} is a
quotient-ready classifier on the upper carrier:
\begin{enumerate}
\item if $N_a(x)$, then $Q_{Z_a\triangleleft N_a}(x,x)$;
\item if $Q_{Z_a\triangleleft N_a}(x,y)$, then
$Q_{Z_a\triangleleft N_a}(y,x)$;
\item if $Q_{Z_a\triangleleft N_a}(x,y)$ and
$Q_{Z_a\triangleleft N_a}(y,z)$, then
$Q_{Z_a\triangleleft N_a}(x,z)$;
\item if $Q_{Z_a\triangleleft N_a}(x,y)$ and
$Q_{Z_a\triangleleft N_a}(u,v)$, then
$Q_{Z_a\triangleleft N_a}(\mathsf{mul}(x,u),\mathsf{mul}(y,v))$.
\end{enumerate}
\end{theorem}
\begin{proof}
Reflexivity uses the identity row of
\autoref{thm:subgroup-centralizer-normalizer-tower-certificate}. From
$N_a(x)$ the endpoint clauses are immediate, and the two-sided inverse law
identifies $\mathsf{mul}(\mathsf{inv}(x),x)$ with $e$. Transport along
$\hsame$ carries the centralizer identity witness $Z_a(e)$ to the required
kernel witness.

For symmetry, expand $Q_{Z_a\triangleleft N_a}(x,y)$. The endpoint clauses
swap because both $x$ and $y$ already lie in $N_a$. The kernel witness
$Z_a(\mathsf{mul}(\mathsf{inv}(x),y))$ is inverted by the inverse-closure row
for $Z_a$, and inverse-product reversal identifies the inverse word with
$\mathsf{mul}(\mathsf{inv}(y),x)$ after the ambient inverse laws. The
$\hsame$-transport row for $Z_a$ gives the displayed symmetric witness.

For transitivity, take kernel witnesses
$Z_a(\mathsf{mul}(\mathsf{inv}(x),y))$ and
$Z_a(\mathsf{mul}(\mathsf{inv}(y),z))$. Product closure in $Z_a$ gives a
witness for their product. Associativity and the inverse laws cancel the
middle factors $y$ and $\mathsf{inv}(y)$, and transport turns the product
witness into $Z_a(\mathsf{mul}(\mathsf{inv}(x),z))$. The endpoint clauses are
the retained $N_a(x)$ and $N_a(z)$ clauses.

For multiplication compatibility, product closure in $N_a$ supplies
$N_a(\mathsf{mul}(x,u))$ and $N_a(\mathsf{mul}(y,v))$. The kernel word
$\mathsf{mul}(\mathsf{inv}(\mathsf{mul}(x,u)),\mathsf{mul}(y,v))$ is rewritten
by inverse-product reversal and associativity as the product of
$\mathsf{mul}(\mathsf{mul}(\mathsf{inv}(u),
\mathsf{mul}(\mathsf{inv}(x),y)),u)$ with
$\mathsf{mul}(\mathsf{inv}(u),v)$. Since $u$ lies in $N_a$, the normality row
