\begin{definition}[Finite-fold zero matrix data]
\label{def:finite-fold-zero-matrix-data}
Let $R$ be a scalar $\RingUp$ carrier with classifier $\sim_R$ and zero endpoint $0_R$. For $n,m:\NatUp$, the carried zero matrix datum over $\mathsf{Fin}(n)\times\mathsf{Fin}(m)$ is the table
$$
Z_{n,m}:\mathsf{Fin}(n)\times\mathsf{Fin}(m)\to R,
\qquad
(Z_{n,m})_{i,j}:=0_R.
$$
It is carried by the $\MatrixUp$ carrier of \autoref{def:matrix-carrier} exactly when every displayed entry is carried by the scalar source, and its classifier endpoint is the pointwise scalar-zero table specified by \autoref{def:matrix-classifier-specification}.
\end{definition}

\begin{theorem}[Finite-fold matrix zero multiplication law]
\label{thm:finite-fold-matrix-zero-multiplication-law}
Let $R$ be a scalar $\RingUp$ carrier with classifier $\sim_R$, zero endpoint $0_R$, addition $+_R$, multiplication $\cdot_R$, and finite additive folds over finite index spines as in \autoref{def:finite-additive-scalar-fold}. Let $n,m,p:\NatUp$, and consider the $\MatrixUp$ carriers of \autoref{def:matrix-carrier} with pointwise classifier from \autoref{def:matrix-classifier-specification}. Let
$$
A:\mathsf{Fin}(n)\times\mathsf{Fin}(m)\to R,
\qquad
B:\mathsf{Fin}(m)\times\mathsf{Fin}(p)\to R
$$
be carried matrices. Assume matrix multiplication is the finite scalar-fold operation
$$
(X\cdot_{\mathrm{Mat}}Y)_{u,w}:=
\Sigma_{\mathsf{Fin}(s)}(v\mapsto X_{u,v}\cdot_RY_{v,w})
$$
for every matching inner dimension $s$, using the multiplication closure row of \autoref{def:matrix-stability-certificate}. Assume the scalar classifier equivalence fields, multiplication closure and congruence, additive closure and congruence, and the left and right zero multiplication laws retained in \autoref{def:ring-stability-certificate}. Then $Z_{n,m}\cdot_{\mathrm{Mat}}B$ and $A\cdot_{\mathrm{Mat}}Z_{m,p}$ are carried $\MatrixUp$ matrices over $\mathsf{Fin}(n)\times\mathsf{Fin}(p)$, and
$$
Z_{n,m}\cdot_{\mathrm{Mat}}B\sim_{\mathrm{Mat}}Z_{n,p}
\qquad\text{and}\qquad
A\cdot_{\mathrm{Mat}}Z_{m,p}\sim_{\mathrm{Mat}}Z_{n,p}.
$$
\end{theorem}
\leanchecked{BEDC.Derived.MatrixUp.MatrixFiniteFoldZeroMatrix\_multiplication\_law}
\begin{proof}
The zero matrix data of \autoref{def:finite-fold-zero-matrix-data} carries $Z_{n,m}$ and $Z_{m,p}$ pointwise from the scalar zero endpoint. The multiplication closure row of \autoref{def:matrix-stability-certificate} therefore carries the two products $Z_{n,m}\cdot_{\mathrm{Mat}}B$ and $A\cdot_{\mathrm{Mat}}Z_{m,p}$.

By \autoref{def:matrix-classifier-specification}, it remains to compare entries with the pointwise zero table $Z_{n,p}$. Fix $i:\mathsf{Fin}(n)$ and $k:\mathsf{Fin}(p)$. The left product entry expands to
$$
\Sigma_{\mathsf{Fin}(m)}(j\mapsto (Z_{n,m})_{i,j}\cdot_RB_{j,k}).
$$
Each summand is scalar-classified with $0_R$ by the displayed definition of $Z_{n,m}$ and the left zero multiplication law in \autoref{def:ring-stability-certificate}.

Induct on the finite spine enumerating $\mathsf{Fin}(m)$. In the nil case, \autoref{def:finite-additive-scalar-fold} identifies the fold with $0_R$. In the cons case, the head summand is classified with $0_R$, and the induction hypothesis classifies the tail fold with $0_R$. Additive congruence transports these comparisons to the cons fold, and the additive zero law retained in \autoref{def:ring-stability-certificate} classifies the resulting endpoint with $0_R$.

The right product entry expands to
$$
\Sigma_{\mathsf{Fin}(m)}(j\mapsto A_{i,j}\cdot_R(Z_{m,p})_{j,k}).
$$
Now each summand is scalar-classified with $0_R$ by the displayed definition of $Z_{m,p}$ and the right zero multiplication law. The same finite-spine induction reduces this fold to $0_R$.

Since $i$ and $k$ were arbitrary, \autoref{def:matrix-classifier-specification} packages the two entrywise scalar-zero comparisons as matrix classifier witnesses against $Z_{n,p}$. The row, contraction, column, and scalar-source witnesses are unchanged, and \autoref{def:matrix-ledger-policy} records that the only non-structural provenance used by the endpoint comparison is scalar zero multiplication.
\end{proof}

The ledger keeps the row, inner, column, and scalar-source witnesses fixed and records scalar-zero provenance. The core theorem is \autoref{thm:finite-fold-matrix-zero-multiplication-law}. \begin{theorem}[Finite-fold zero matrix additive identity]
\label{thm:finite-fold-zero-matrix-additive-identity}
Let $R$ carry a scalar $\RingUp(R)$ certificate with classifier $\sim_R$, zero endpoint $0_R$, and addition $+_R$. Assume the scalar source carries $0_R$, and assume the scalar classifier equivalence fields, addition closure, and scalar additive left-zero and right-zero laws retained in \autoref{def:ring-stability-certificate}. Let $n,m:\NatUp$, and consider the $\MatrixUp$ carrier of \autoref{def:matrix-carrier} with pointwise classifier from \autoref{def:matrix-classifier-specification}. Let
$$
A:\mathsf{Fin}(n)\times\mathsf{Fin}(m)\to R
$$
be a carried matrix, and let $Z_{n,m}$ be the zero matrix datum of \autoref{def:finite-fold-zero-matrix-data}. Assume matrix addition is the pointwise operation
$$
(X+Y)_{i,j}:=X_{i,j}+_R Y_{i,j}
$$
using the addition closure row of \autoref{def:matrix-stability-certificate}. Then $A+Z_{n,m}$ and $Z_{n,m}+A$ are carried $\MatrixUp$ matrices over $\mathsf{Fin}(n)\times\mathsf{Fin}(m)$, and
$$
A+Z_{n,m}\sim_{\mathrm{Mat}}A
\qquad\text{and}\qquad
Z_{n,m}+A\sim_{\mathrm{Mat}}A.
$$
Consequently,
$$
A+Z_{n,m}\sim_{\mathrm{Mat}}Z_{n,m}+A.
$$
\end{theorem}
\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonZero\_additive\_identity}
\begin{proof}
The zero matrix datum of \autoref{def:finite-fold-zero-matrix-data} carries $Z_{n,m}$ pointwise from the carried scalar zero endpoint. Since $A$ is carried, the addition closure row of \autoref{def:matrix-stability-certificate} carries both pointwise sums $A+Z_{n,m}$ and $Z_{n,m}+A$.

By \autoref{def:matrix-classifier-specification}, it remains to compare entries. Fix $i:\mathsf{Fin}(n)$ and $j:\mathsf{Fin}(m)$. The pointwise addition rule and the definition of $Z_{n,m}$ give
$$
(A+Z_{n,m})_{i,j}=A_{i,j}+_R 0_R.
$$
The scalar additive right-zero law retained in \autoref{def:ring-stability-certificate} classifies this endpoint with $A_{i,j}$, so
$$
(A+Z_{n,m})_{i,j}\sim_R A_{i,j}.
$$
For the other order, the same pointwise rule gives
$$
(Z_{n,m}+A)_{i,j}=0_R+_R A_{i,j}.
$$
The scalar additive left-zero law classifies this endpoint with $A_{i,j}$, hence
$$
(Z_{n,m}+A)_{i,j}\sim_R A_{i,j}.
$$
Since $i$ and $j$ were arbitrary, \autoref{def:matrix-classifier-specification} packages the two entrywise scalar comparisons as the displayed matrix classifier witnesses against $A$. Applying scalar symmetry to the second entrywise comparison and scalar transitivity with the first gives
$$
(A+Z_{n,m})_{i,j}\sim_R (Z_{n,m}+A)_{i,j}
$$
for every index pair, so the same pointwise classifier packages the final displayed consequence. The dimensions, row witness, column witness, and scalar source are unchanged, as recorded by \autoref{def:matrix-ledger-policy}.
\end{proof}
\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonAddFold\_zero\_prefix\_absorption}
