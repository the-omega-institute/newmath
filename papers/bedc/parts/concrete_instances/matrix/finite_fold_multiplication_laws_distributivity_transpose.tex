\begin{lemma}[Finite additive fold splits pointwise sums]
\label{lem:finite-additive-fold-splits-pointwise-sums}
Let $R$ be a scalar $\RingUp$ carrier with classifier $\sim_R$, zero endpoint $0_R$, addition $+_R$, and finite additive fold $\mathsf{fold}_{+,R}$ from \autoref{def:finite-additive-scalar-fold}. Assume the scalar classifier equivalence fields and the additive abelian-group, additive congruence, and zero rows of \autoref{def:ring-stability-certificate}. For same-indexed finite carried scalar spines $xs$ and $ys$, write $xs+_R ys$ for the pointwise sum spine. Then
$$
\mathsf{fold}_{+,R}(xs+_R ys)
\sim_R
\mathsf{fold}_{+,R}(xs)+_R\mathsf{fold}_{+,R}(ys).
$$
\end{lemma}
\begin{proof}
Induct on the common finite spine carrying $xs$ and $ys$. In the nil case, all three folds reduce by \autoref{def:finite-additive-scalar-fold} to $0_R$. The additive zero rows classify $0_R$ with $0_R+_R0_R$, giving the displayed comparison.

In the cons case, write $xs=x::xs'$ and $ys=y::ys'$. The fold equation reduces the left endpoint to
$$
(x+_Ry)+_R\mathsf{fold}_{+,R}(xs'+_Rys').
$$
The induction hypothesis and additive congruence replace the tail fold by $\mathsf{fold}_{+,R}(xs')+_R\mathsf{fold}_{+,R}(ys')$. The additive associativity and commutativity fields in \autoref{def:ring-stability-certificate} rebracket and exchange the middle summands, classifying
$$
(x+_Ry)+_R(\mathsf{fold}_{+,R}(xs')+_R\mathsf{fold}_{+,R}(ys'))
$$
with
$$
(x+_R\mathsf{fold}_{+,R}(xs'))+_R(y+_R\mathsf{fold}_{+,R}(ys')).
$$
By \autoref{def:finite-additive-scalar-fold}, the latter endpoint is $\mathsf{fold}_{+,R}(xs)+_R\mathsf{fold}_{+,R}(ys)$. Classifier transitivity composes the displayed comparisons.
\end{proof}

\begin{theorem}[Matrix multiplication left-distributes over matrix addition]
\label{thm:matrix-multiplication-left-distributes-over-addition}
Let $R$ be a scalar $\RingUp$ carrier with classifier $\sim_R$, zero endpoint
$0_R$, addition $+_R$, multiplication $\cdot_R$, and finite additive folds over
finite index spines as in \autoref{thm:finite-additive-fold-congruence-coefficient-lists}. Let
$n,m,p:\NatUp$, and consider the $\MatrixUp$ carriers of
\autoref{def:matrix-carrier} with pointwise classifier from
\autoref{def:matrix-classifier-specification}. Let
$$
A:\mathsf{Fin}(n)\times\mathsf{Fin}(m)\to R,
\qquad
B,C:\mathsf{Fin}(m)\times\mathsf{Fin}(p)\to R
$$
be carried matrices. Assume matrix addition is pointwise and matrix
multiplication is the finite-fold operation, using the addition and
multiplication closure rows of \autoref{def:matrix-stability-certificate}.
Assume the scalar ring stability fields of \autoref{def:ring-stability-certificate}:
additive abelian-group laws, addition and multiplication congruence, and left
distributivity. Then the displayed matrices are carried and
$$
A\cdot_{\mathrm{Mat}}(B+C)\sim_{\mathrm{Mat}}(A\cdot_{\mathrm{Mat}}B)+(A\cdot_{\mathrm{Mat}}C).
$$
\end{theorem}
\begin{proof}
Closure follows from the matrix addition and multiplication rows of
\autoref{def:matrix-stability-certificate}. By
\autoref{def:matrix-classifier-specification}, compare entries.

Fix $i:\mathsf{Fin}(n)$ and $k:\mathsf{Fin}(p)$. Over the finite spine
enumerating $\mathsf{Fin}(m)$, define
$$
xs(j):=A_{i,j}\cdot_RB_{j,k},
\qquad
ys(j):=A_{i,j}\cdot_RC_{j,k}.
$$
The pointwise addition definition and scalar left distributivity classify each
summand $A_{i,j}\cdot_R(B+C)_{j,k}$ with $xs(j)+_Rys(j)$.

Fold congruence lifts these pointwise comparisons through the finite fold. Then
\autoref{lem:finite-additive-fold-separates-pointwise-scalar-sums} separates the
fold of pointwise sums into the sum of the two finite folds. Unfolding matrix
multiplication and pointwise matrix addition identifies the endpoint with
$((A\cdot_{\mathrm{Mat}}B)+(A\cdot_{\mathrm{Mat}}C))_{i,k}$. Since $i$ and $k$
were arbitrary, the pointwise classifier packages the displayed law.
\end{proof}

\begin{theorem}[Finite-fold matrix transpose reverses multiplication]
\label{thm:finite-fold-matrix-transpose-reverses-multiplication}
Let $R$ be a scalar $\RingUp$ carrier with opposite multiplication data as in
\autoref{def:opposite-ring-certificate-data}. Let $A$ be a carried $n\times m$
matrix and $B$ a carried $m\times p$ matrix. Assume matrix multiplication is the
finite scalar-fold operation and matrix transpose swaps the two finite index
coordinates pointwise. Then
$$
(A\cdot_{\mathrm{Mat}}B)^{T}
\sim_{\mathrm{Mat}}
B^{T}\cdot_{\mathrm{Mat}}A^{T}
$$
over the opposite scalar multiplication.
\end{theorem}
\begin{proof}
The transpose carrier row swaps the row and column witnesses, and multiplication
closure carries both displayed matrices. Entrywise, the left side at
$(k,i)$ is the fold of $A_{i,j}\cdot_RB_{j,k}$ over the shared inner finite
spine.

The right side in the opposite scalar multiplication is the fold of
$B_{j,k}\cdot_{R^{\mathrm{op}}}A_{i,j}$ over the same finite spine. By
\autoref{def:opposite-ring-certificate-data}, each opposite product is the same
scalar endpoint as $A_{i,j}\cdot_RB_{j,k}$. Fold congruence from
\autoref{thm:finite-additive-fold-congruence-coefficient-lists} compares the two
folds, and the pointwise matrix classifier packages the transpose law.
\end{proof}

\begin{theorem}[Matrix addition commutativity from scalar additive commutativity]
\label{thm:matrix-addition-commutativity-from-scalar-additive-commutativity}
Let $R$ carry a scalar $\RingUp(R)$ certificate with classifier $\sim_R$ and addition $+_R$. Assume the additive abelian-group rows retained in \autoref{def:ring-stability-certificate}, in particular the scalar additive commutativity row
$$
a+_R b\sim_R b+_R a
$$
for all carried scalar endpoints $a,b$. Let $n,m:\NatUp$, and consider the $\MatrixUp$ carriers of \autoref{def:matrix-carrier} with pointwise classifier from \autoref{def:matrix-classifier-specification}. For carried matrices
$$
A,B:\mathsf{Fin}(n)\times\mathsf{Fin}(m)\to R,
$$
assume matrix addition is the pointwise operation
$$
(X+Y)_{i,j}:=X_{i,j}+_R Y_{i,j}
$$
using the addition closure row of \autoref{def:matrix-stability-certificate}. Then both $A+B$ and $B+A$ are carried matrices and
$$
A+B\sim_{\mathrm{Mat}}B+A.
$$
\end{theorem}
\begin{proof}
The addition closure row of \autoref{def:matrix-stability-certificate} carries $A+B$ from the carried inputs $A$ and $B$, and carries $B+A$ from the same inputs in the opposite order.

By \autoref{def:matrix-classifier-specification}, it remains to compare corresponding scalar entries. Fix $i:\mathsf{Fin}(n)$ and $j:\mathsf{Fin}(m)$. Since $A$ and $B$ are carried matrices, the entries $A_{i,j}$ and $B_{i,j}$ are carried scalar endpoints. The pointwise addition rule gives
$$
(A+B)_{i,j}=A_{i,j}+_R B_{i,j}
$$
and
$$
(B+A)_{i,j}=B_{i,j}+_R A_{i,j}.
$$
The scalar additive commutativity row retained in \autoref{def:ring-stability-certificate} gives
$$
A_{i,j}+_R B_{i,j}\sim_R B_{i,j}+_R A_{i,j}.
$$
Substituting the two pointwise addition clauses gives
$$
(A+B)_{i,j}\sim_R(B+A)_{i,j}.
$$
Since $i$ and $j$ were arbitrary, \autoref{def:matrix-classifier-specification} packages these entrywise scalar comparisons as
$$
A+B\sim_{\mathrm{Mat}}B+A.
$$
The dimensions and scalar source are unchanged throughout, and \autoref{def:matrix-ledger-policy} records the same row and column witnesses for the two pointwise sums.
\end{proof}

\begin{theorem}[Matrix addition associativity from scalar additive associativity]
\label{thm:matrix-addition-associativity-from-scalar-additive-associativity}
Let $R$ carry a scalar $\RingUp(R)$ certificate with classifier $\sim_R$ and addition $+_R$. Assume the additive abelian-group rows retained in \autoref{def:ring-stability-certificate}, in particular the scalar additive associativity row
$$
(a+_R b)+_R c\sim_R a+_R(b+_R c)
$$
for all carried scalar endpoints $a,b,c$. Let $n,m:\NatUp$, and consider the $\MatrixUp$ carriers of \autoref{def:matrix-carrier} with pointwise classifier from \autoref{def:matrix-classifier-specification}. For carried matrices
$$
A,B,C:\mathsf{Fin}(n)\times\mathsf{Fin}(m)\to R,
$$
assume matrix addition is the pointwise operation
$$
(X+Y)_{i,j}:=X_{i,j}+_R Y_{i,j}
$$
using the addition closure row of \autoref{def:matrix-stability-certificate}. Then $A+B$, $B+C$, $(A+B)+C$, and $A+(B+C)$ are carried matrices, and
$$
(A+B)+C\sim_{\mathrm{Mat}}A+(B+C).
$$
\end{theorem}
\begin{proof}
The addition closure row of \autoref{def:matrix-stability-certificate} carries $A+B$ and $B+C$ from the carried inputs. Applying the same row once more carries $(A+B)+C$ and $A+(B+C)$.

By \autoref{def:matrix-classifier-specification}, it remains to compare corresponding scalar entries. Fix $i:\mathsf{Fin}(n)$ and $j:\mathsf{Fin}(m)$. Since $A$, $B$, and $C$ are carried matrices, the entries $A_{i,j}$, $B_{i,j}$, and $C_{i,j}$ are carried scalar endpoints. The pointwise addition rule gives
$$
((A+B)+C)_{i,j}=(A_{i,j}+_R B_{i,j})+_R C_{i,j}
$$
and
$$
(A+(B+C))_{i,j}=A_{i,j}+_R(B_{i,j}+_R C_{i,j}).
$$
The scalar additive associativity row retained in \autoref{def:ring-stability-certificate} therefore gives
$$
((A+B)+C)_{i,j}\sim_R(A+(B+C))_{i,j}.
$$
Since $i$ and $j$ were arbitrary, \autoref{def:matrix-classifier-specification} packages these entrywise scalar comparisons as
$$
(A+B)+C\sim_{\mathrm{Mat}}A+(B+C).
$$
The row and column witnesses are unchanged throughout, so \autoref{def:matrix-ledger-policy} records the same dimension ledger for the two parenthesized pointwise sums.
\end{proof}
