\begin{lemma}[Finite additive fold singleton identity]
\label{lem:finite-additive-fold-singleton-identity}
Let $R$ be a scalar $\RingUp$ carrier with classifier $\sim_R$, zero endpoint
$0_R$, addition $+_R$, and finite additive fold $\mathsf{fold}_{+,R}$ from
\autoref{thm:finite-additive-fold-congruence-coefficient-lists}. Assume the
additive zero and classifier-equivalence rows retained in
\autoref{def:ring-stability-certificate}.
For every carried scalar $a$, the singleton fold is classified with $a$:
$$
\mathsf{fold}_{+,R}(\mathsf{singleton}(a))\sim_R a.
$$
\end{lemma}
\begin{proof}
Unfold the singleton spine as $\mathsf{cons}(a,\mathsf{nil})$. The fold equation
identifies its endpoint with $a+_R0_R$.

The additive right-zero law classifies $a+_R0_R$ with $a$. Classifier symmetry
and transitivity then transport the displayed fold endpoint to the scalar $a$.
\end{proof}

\begin{lemma}[Finite additive fold empty identity]
\label{lem:finite-additive-fold-empty-identity}
Under the same scalar data, the empty finite fold is classified with the zero
endpoint:
$$
\mathsf{fold}_{+,R}(\mathsf{nil})\sim_R0_R.
$$
\end{lemma}
\begin{proof}
This is the nil clause recorded by
\autoref{thm:finite-additive-fold-congruence-coefficient-lists}, followed by
classifier reflexivity on the zero endpoint.
\end{proof}

\begin{definition}[Finite additive scalar fold]
\label{def:finite-additive-scalar-fold}
Let $R$ be a scalar $\RingUp$ carrier with classifier $\sim_R$, zero endpoint
$0_R$, addition $+_R$, and the additive stability rows of
\autoref{def:ring-stability-certificate}. For every finite carried scalar spine
$xs$, the fold $\mathsf{fold}_{+,R}(xs)$ is generated by the two clauses
$$
\mathsf{fold}_{+,R}(\mathsf{nil}) := 0_R,
\qquad
\mathsf{fold}_{+,R}(x::xs) := x+_R\mathsf{fold}_{+,R}(xs).
$$
The fold ledger records the finite spine order, the carriedness witness for
each entry, and the additive ledger used at every cons step.
\end{definition}

\begin{theorem}[Finite additive fold respects scalar classifier]
\label{thm:finite-additive-fold-respects-scalar-classifier}
Let $R$ be a scalar $\RingUp$ carrier with classifier $\sim_R$, zero endpoint
$0_R$, addition $+_R$, and the additive stability rows of
\autoref{def:ring-stability-certificate}. If finite carried scalar spines $xs$
and $ys$ are related by
$\mathsf{ListClassifierSpec}(\sim_R,xs,ys)$, then
$$
\mathsf{fold}_{+,R}(xs)\sim_R\mathsf{fold}_{+,R}(ys).
$$
\end{theorem}
\begin{proof}
Induct on the displayed finite-spine classifier witness. In the nil case, both
folds reduce by \autoref{def:finite-additive-scalar-fold} to $0_R$, and the
zero self-classification row of \autoref{def:ring-stability-certificate} gives
$0_R\sim_R0_R$.

In the cons case, the head comparison gives $x\sim_R y$, and the induction
hypothesis gives
$\mathsf{fold}_{+,R}(xs)\sim_R\mathsf{fold}_{+,R}(ys)$. Additive congruence from
\autoref{def:ring-stability-certificate} transports these two comparisons to
$x+_R\mathsf{fold}_{+,R}(xs)\sim_R y+_R\mathsf{fold}_{+,R}(ys)$, which is exactly
the folded comparison after unfolding
\autoref{def:finite-additive-scalar-fold}.
\end{proof}

\begin{theorem}[Singleton additive fold carrier readback]
\label{thm:matrix-singleton-add-fold-carrier-iff}
For the singleton matrix carrier, let
$\mathsf{MatrixSingletonAddFold}$ fold a finite spine by the concrete
singleton addition $\mathsf{MatrixSingletonAdd}$, and let
$\mathsf{MatrixSingletonAddFoldSpineCarrier}$ record that every spine entry
is carried. Then
$$
\mathsf{MatrixSingletonCarrier}(\mathsf{MatrixSingletonAddFold}(xs))
\Longleftrightarrow
\mathsf{MatrixSingletonAddFoldSpineCarrier}(xs).
$$
\end{theorem}
\begin{proof}
Induct on the finite spine. The nil fold is $\emp$, so both sides are the
empty endpoint readback. In the cons case, singleton addition is kernel
append; empty-append inversion splits the folded endpoint into the head
carrier and the tail-fold carrier, and the induction hypothesis identifies the
tail-fold carrier with the tail spine carrier.
\end{proof}

\leandef{BEDC.Derived.MatrixUp.MatrixSingletonAddFoldSpineCarrier}
\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonAddFold\_carrier\_iff}

\begin{theorem}[Singleton additive fold continuation result readback]
\label{thm:matrix-singleton-add-fold-continuation-result-iff}
For every finite spine $xs$ and histories $h,r:\mathsf{BHist}$, if
$$
\mathsf{Cont}\bigl(h,\mathsf{MatrixSingletonAddFold}(xs),r\bigr),
$$
then the result is carried by the singleton matrix carrier exactly when the
left input is carried and every entry of the fold spine is carried:
$$
\mathsf{MatrixSingletonCarrier}(r)
\Longleftrightarrow
\mathsf{MatrixSingletonCarrier}(h)\land
\mathsf{MatrixSingletonAddFoldSpineCarrier}(xs).
$$
\end{theorem}
\begin{proof}
For the forward direction, transport the displayed continuation along
$\mathsf{MatrixSingletonCarrier}(r)$ to a continuation ending at $\emp$.
Kernel empty-result inversion for $\mathsf{Cont}$ forces both $h$ and
$\mathsf{MatrixSingletonAddFold}(xs)$ to be empty. The first equality is the
left carrier, and the second is read through
\autoref{thm:matrix-singleton-add-fold-carrier-iff} to obtain the spine
carrier.

Conversely, the left carrier identifies $h$ with $\emp$, while
\autoref{thm:matrix-singleton-add-fold-carrier-iff} turns the spine carrier
into $\hsame(\mathsf{MatrixSingletonAddFold}(xs),\emp)$. Continuation
respect for $\hsame$, compared against the empty right-unit continuation,
forces $r$ to be carried by the singleton matrix carrier.
\end{proof}
\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonAddFold\_continuation\_result\_iff}

\begin{theorem}[Singleton additive fold visible head exclusion]
\label{thm:matrix-singleton-add-fold-visible-head-absurd}
For every history $m$ and finite spine $xs$, folding a singleton additive spine
whose head is $\Ezero(m)$ or $\Eone(m)$ cannot read back to the empty carrier:
$$
\neg\hsame(\mathsf{MatrixSingletonAddFold}(\Ezero(m)::xs),\emp)
\quad\wedge\quad
\neg\hsame(\mathsf{MatrixSingletonAddFold}(\Eone(m)::xs),\emp).
$$
\end{theorem}
\begin{proof}
Unfold the cons case of the finite singleton additive fold. The head is the
left factor of a kernel append, and empty-append inversion would identify
$\Ezero(m)$ or $\Eone(m)$ with $\emp$. Both alternatives contradict constructor
separation in $\mathsf{BHist}$.
\end{proof}
\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonAddFold\_visible\_head\_absurd}

\begin{theorem}[Singleton additive fold append carrier readback]
\label{thm:matrix-singleton-add-fold-append-carrier-iff}
For finite singleton-matrix spines $xs$ and $ys$,
$$
\mathsf{MatrixSingletonCarrier}(\mathsf{MatrixSingletonAddFold}(xs\,{++}\,ys))
\Longleftrightarrow
\mathsf{MatrixSingletonAddFoldSpineCarrier}(xs)\land
\mathsf{MatrixSingletonAddFoldSpineCarrier}(ys).
$$
\end{theorem}
\begin{proof}
Induct on $xs$. For the empty spine, the appended fold is the fold of $ys$,
and the base fold readback identifies its carrier with the spine carrier of
$ys$. For a cons spine, concrete singleton addition is kernel append, so
empty-append inversion splits the folded endpoint into the head carrier and
the appended tail-fold carrier. The induction hypothesis splits the tail-fold
carrier into the tail spine carrier and the carrier of $ys$, and the reverse
direction repacks those two splits through the same empty-append constructor.
\end{proof}
\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonAddFold\_append\_carrier\_iff}

\begin{theorem}[Singleton additive fold append readback]
\label{thm:matrix-singleton-add-fold-append-hsame}
For finite singleton-matrix spines $xs$ and $ys$, folding their appended
spine by concrete singleton addition is history-same as appending the two

$$
\mathsf{MatrixSingletonAddFold}(xs\mathbin{+\!\!+}ys)\hsame
\mathsf{append}(\mathsf{MatrixSingletonAddFold}(xs),
\mathsf{MatrixSingletonAddFold}(ys)).
$$
\end{theorem}
\begin{proof}
Induct on $xs$. The nil case is the left-unit law for kernel append. In the
cons case, unfold singleton addition as kernel append, apply the induction
hypothesis under the common head, and reassociate by the kernel append
associativity law.
\end{proof}

\begin{theorem}[Singleton additive fold append display classifier]
\label{thm:matrix-singleton-add-fold-append-display-classifier-iff}
For finite singleton-matrix spines $xs$ and $ys$, the classifier between
the fold of the appended spine and the append of the two individual folds is
equivalent to carriedness of both input spines:
$$
\mathsf{MatrixSingletonClassifier}
  (\mathsf{MatrixSingletonAddFold}(xs\mathbin{+\!\!+}ys),
   \mathsf{append}(\mathsf{MatrixSingletonAddFold}(xs),
                  \mathsf{MatrixSingletonAddFold}(ys)))
\Longleftrightarrow
\mathsf{MatrixSingletonAddFoldSpineCarrier}(xs)\land
\mathsf{MatrixSingletonAddFoldSpineCarrier}(ys).
$$
\end{theorem}
\begin{proof}
In the forward direction, project the classifier to its left carrier row and
use append-spine carrier readback. Conversely, read both spine carriers back
to carriers for the individual folds; empty-append inversion gives the carrier
of the displayed append, while fold append readback supplies the classifier
comparison between the appended fold and the displayed append endpoint.
\end{proof}

\begin{lemma}[Finite Kronecker selector sum]
\label{lem:matrix-finite-kronecker-selector-sum}
Let $J=\mathsf{Fin}(p)$ be a finite index carrier licensed by the $\NatUp$
certificate of \autoref{ch:concrete-instances-nat-namecert}. Let $R$ be a
scalar carrier from the $\CommRingUp$ certificate of
\autoref{ch:concrete-instances-commring-namecert}, with classifier $\sim_R$,
operations $+_R,\cdot_R$, and scalars $0_R,1_R$. Assume a finite-index sum
certificate $\Sigma_J$ satisfying classifier congruence and the selector row:
if $j_0:J$, $t:J\to R$ is carried pointwise, $t(j_0)\sim_R u$, and
$t(j)\sim_R 0_R$ for every $j\ne j_0$, then $\Sigma_J(t)\sim_R u$.
For every pointwise carried scalar family $x:J\to R$ and every $j_0:J$,
$$
\Sigma_J(j\mapsto \delta_{j_0,j}\cdot_R x(j))\sim_R x(j_0)
\qquad\text{and}\qquad
\Sigma_J(j\mapsto x(j)\cdot_R \delta_{j,j_0})\sim_R x(j_0),
$$
where $\delta_{a,b}$ is $1_R$ when the finite-index equality certificate
identifies $a$ with $b$, and $0_R$ otherwise.
\end{lemma}
\begin{proof}
For the first displayed sum, split the finite index $j$. At the selected index
$j=j_0$, the Kronecker entry is $1_R$, so the scalar left-one law gives
$\delta_{j_0,j_0}\cdot_R x(j_0)\sim_R x(j_0)$. At every other index, the
Kronecker entry is $0_R$, so scalar left-zero absorption gives
$\delta_{j_0,j}\cdot_R x(j)\sim_R 0_R$. These are exactly the selected and
off-selected hypotheses of the selector row for $\Sigma_J$, applied to the
summand family $j\mapsto \delta_{j_0,j}\cdot_R x(j)$ with $u=x(j_0)$.

The second displayed sum uses the same selector-row hypothesis for
$j\mapsto x(j)\cdot_R \delta_{j,j_0}$. The scalar right-one law gives the
selected comparison at $j=j_0$, and scalar right-zero absorption gives the
comparison with $0_R$ away from $j_0$.
\end{proof}

\begin{theorem}[Matrix identity multiplication unit]
\label{thm:matrix-identity-multiplication-unit}
Let $n,m:\NatUp$ and let $R$ be a scalar $\CommRingUp$ certificate. Consider
the $\MatrixUp$ carrier of \autoref{def:matrix-carrier} with the pointwise
classifier of \autoref{def:matrix-classifier-specification}. Assume the
stability data of \autoref{def:matrix-stability-certificate} supplies finite
product sums over $\mathsf{Fin}(n)$ and $\mathsf{Fin}(m)$, the scalar zero and
one multiplication laws over $R$, and carried Kronecker identity matrices
$I_n$ and $I_m$ with entries
$$
(I_n)_{i,a}\sim_R\delta_{i,a}
\qquad\text{and}\qquad
(I_m)_{b,j}\sim_R\delta_{b,j}.
$$
If $M:\mathsf{Fin}(n)\times\mathsf{Fin}(m)\to R$ is carried, then the products
$I_n\cdot M$ and $M\cdot I_m$ are carried and
$$
I_n\cdot M\sim_{\mathrm{Mat}} M
\qquad\text{and}\qquad
M\cdot I_m\sim_{\mathrm{Mat}} M.
$$
\end{theorem}
\begin{proof}
The carriedness of $I_n$ and $I_m$ is part of the Kronecker identity closure
field, and $M$ is carried by hypothesis. Multiplication closure in
\autoref{def:matrix-stability-certificate} therefore carries both products.
It remains to prove the pointwise scalar comparisons required by
\autoref{def:matrix-classifier-specification}.

Fix $i:\mathsf{Fin}(n)$ and $b:\mathsf{Fin}(m)$. The left product entry is the
finite sum
$$
(I_n\cdot M)_{i,b}
\sim_R
\Sigma_{\mathsf{Fin}(n)}(a\mapsto (I_n)_{i,a}\cdot_R M_{a,b}).
$$
For each inner index $a$, identity-entry closure gives
$(I_n)_{i,a}\sim_R\delta_{i,a}$, and scalar classifier congruence for
$\cdot_R$ gives
$$
(I_n)_{i,a}\cdot_R M_{a,b}
\sim_R
\delta_{i,a}\cdot_R M_{a,b}.
$$
The finite-sum classifier congruence for $\Sigma_{\mathsf{Fin}(n)}$ yields
$$
\Sigma_{\mathsf{Fin}(n)}(a\mapsto (I_n)_{i,a}\cdot_R M_{a,b})
\sim_R
\Sigma_{\mathsf{Fin}(n)}(a\mapsto \delta_{i,a}\cdot_R M_{a,b}).
$$
Applying \autoref{lem:matrix-finite-kronecker-selector-sum} to the carried
family $a\mapsto M_{a,b}$ selects the unique inner index $a=i$, hence
$(I_n\cdot M)_{i,b}\sim_R M_{i,b}$.

For the right product, the entry at the same outer indices is
$$
(M\cdot I_m)_{i,b}
\sim_R
\Sigma_{\mathsf{Fin}(m)}(a\mapsto M_{i,a}\cdot_R (I_m)_{a,b}).
$$
For each inner index $a$, identity-entry closure gives
$(I_m)_{a,b}\sim_R\delta_{a,b}$, and scalar classifier congruence for
$\cdot_R$ gives
$$
M_{i,a}\cdot_R (I_m)_{a,b}
\sim_R
M_{i,a}\cdot_R \delta_{a,b}.
$$
The finite-sum classifier congruence for $\Sigma_{\mathsf{Fin}(m)}$ yields
$$
\Sigma_{\mathsf{Fin}(m)}(a\mapsto M_{i,a}\cdot_R (I_m)_{a,b})
\sim_R
\Sigma_{\mathsf{Fin}(m)}(a\mapsto M_{i,a}\cdot_R \delta_{a,b}).
$$
The right-hand selector part of
\autoref{lem:matrix-finite-kronecker-selector-sum} applied to the carried
family $a\mapsto M_{i,a}$ selects $a=b$, giving
$(M\cdot I_m)_{i,b}\sim_R M_{i,b}$.

Since these scalar comparisons hold for every row and column index, the
pointwise matrix classifier gives both displayed matrix identity laws. The
ledger in \autoref{def:matrix-ledger-policy} records the same row and column
witnesses throughout the two products, so no dimension data is changed.
\end{proof}

\begin{definition}[Finite-fold matrix identity data]
\label{def:finite-fold-matrix-identity-data}
Let $R$ be a scalar $\RingUp$ carrier with classifier $\sim_R$, zero endpoint
$0_R$, unit endpoint $1_R$, addition $+_R$, multiplication $\cdot_R$, and finite
additive folds as in \autoref{thm:finite-additive-fold-congruence-coefficient-lists}. For each finite
dimension $n:\NatUp$, the finite-fold identity matrix $I_n$ is the carried
$n\times n$ matrix whose $(i,j)$ entry is $1_R$ when the finite indices agree
and $0_R$ otherwise.
\end{definition}

\begin{theorem}[Finite-fold matrix identity law]
\label{thm:finite-fold-matrix-identity-law}
Let $R$ be a scalar $\RingUp$ carrier with classifier $\sim_R$, zero endpoint
$0_R$, unit endpoint $1_R$, addition $+_R$, multiplication $\cdot_R$, and finite
additive folds over finite index spines as in
\autoref{thm:finite-additive-fold-congruence-coefficient-lists}. Let $A$ be a carried $n\times m$
$\MatrixUp$ matrix. Assume the scalar classifier equivalence fields, additive
zero laws, multiplication congruence, zero multiplication laws, and left and
right unit laws retained in \autoref{def:ring-stability-certificate}. Then the
finite-fold products with the identity matrices of
\autoref{def:finite-fold-matrix-identity-data} are carried and
$$
I_n\cdot_{\mathrm{Mat}} A\sim_{\mathrm{Mat}}A
\qquad\text{and}\qquad
A\cdot_{\mathrm{Mat}} I_m\sim_{\mathrm{Mat}}A.
$$
\end{theorem}
\begin{proof}
The multiplication closure row of \autoref{def:matrix-stability-certificate}
carries both products. By \autoref{def:matrix-classifier-specification}, it is
enough to compare scalar entries.

For the left identity, fix $i:\mathsf{Fin}(n)$ and $j:\mathsf{Fin}(m)$. The
finite fold for $(I_n\cdot_{\mathrm{Mat}}A)_{i,j}$ has one diagonal summand
$1_R\cdot_R A_{i,j}$ and all other summands $0_R\cdot_R A_{k,j}$. The unit law
classifies the diagonal summand with $A_{i,j}$, and the zero multiplication law
classifies every off-diagonal summand with $0_R$.

The finite additive fold over a spine with one carried scalar and all remaining
entries classified with $0_R$ reduces, by the additive zero laws and fold
congruence from \autoref{thm:finite-additive-fold-congruence-coefficient-lists},
to that single scalar. Hence $(I_n\cdot_{\mathrm{Mat}}A)_{i,j}\sim_R A_{i,j}$.

The right identity is symmetric. The fold for
$(A\cdot_{\mathrm{Mat}}I_m)_{i,j}$ has diagonal summand
$A_{i,j}\cdot_R1_R$ and off-diagonal summands classified with $0_R$. The right
unit law, zero multiplication laws, additive zero laws, and fold congruence
again collapse the fold to $A_{i,j}$. Packaging the two entrywise comparisons
gives the displayed matrix classifier laws.
\end{proof}
