\begin{lemma}[Finite additive fold singleton identity]
\label{lem:finite-additive-fold-singleton-identity}
Let $R$ be a scalar $\RingUp$ carrier with classifier $\sim_R$, zero endpoint
$0_R$, addition $+_R$, and finite additive fold $\mathsf{fold}_{+,R}$ from
\autoref{thm:finite-additive-fold-congruence-coefficient-lists}. Assume the
additive zero and classifier-equivalence rows retained in
\autoref{def:ring-stability-certificate}.
For every carried scalar $a$, the singleton fold is classified with $a$:
$$
\mathsf{fold}_{+,R}(\mathsf{singleton}(a))\sim_R a.
$$
\end{lemma}
\begin{proof}
Unfold the singleton spine as $\mathsf{cons}(a,\mathsf{nil})$. The fold equation
identifies its endpoint with $a+_R0_R$.

The additive right-zero law classifies $a+_R0_R$ with $a$. Classifier symmetry
and transitivity then transport the displayed fold endpoint to the scalar $a$.
\end{proof}

\begin{lemma}[Finite additive fold empty identity]
\label{lem:finite-additive-fold-empty-identity}
Under the same scalar data, the empty finite fold is classified with the zero
endpoint:
$$
\mathsf{fold}_{+,R}(\mathsf{nil})\sim_R0_R.
$$
\end{lemma}
\begin{proof}
This is the nil clause recorded by
\autoref{thm:finite-additive-fold-congruence-coefficient-lists}, followed by
classifier reflexivity on the zero endpoint.
\end{proof}
