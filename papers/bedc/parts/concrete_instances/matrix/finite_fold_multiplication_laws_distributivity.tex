\begin{theorem}[Matrix multiplication classifier congruence]
\label{thm:matrix-multiplication-classifier-congruence}
Let $R$ be a scalar $\RingUp$ carrier with classifier $\sim_R$, multiplication $\cdot_R$, addition $+_R$, zero endpoint $0_R$, and finite additive fold from \autoref{thm:finite-additive-fold-congruence-coefficient-lists}. Let $n,m,p:\NatUp$, and consider the $\MatrixUp$ carriers of \autoref{def:matrix-carrier} with pointwise classifier from \autoref{def:matrix-classifier-specification}. Let
$$
A,A':\mathsf{Fin}(n)\times\mathsf{Fin}(m)\to R,
\qquad
B,B':\mathsf{Fin}(m)\times\mathsf{Fin}(p)\to R
$$
be carried matrices over the same scalar source and the same matching inner dimension $m$. Assume matrix multiplication is the finite-fold operation
$$
(X\cdot Y)_{i,k}:=\Sigma_{\mathsf{Fin}(m)}(j\mapsto X_{i,j}\cdot_RY_{j,k})
$$
for the shared inner finite index carrier, using the multiplication closure field of \autoref{def:matrix-stability-certificate}. Assume the scalar multiplication congruence field of \autoref{def:ring-stability-certificate} and the fold congruence theorem \autoref{thm:finite-additive-fold-congruence-coefficient-lists}. If
$$
A\sim_{\mathrm{Mat}}A'
\qquad\text{and}\qquad
B\sim_{\mathrm{Mat}}B',
$$
then $A\cdot B$ and $A'\cdot B'$ are carried $\MatrixUp$ matrices, and
$$
A\cdot B\sim_{\mathrm{Mat}}A'\cdot B'.
$$
\end{theorem}
\begin{proof}
The multiplication closure item in \autoref{def:matrix-stability-certificate} applies to the carried compatible pairs $(A,B)$ and $(A',B')$, so both products $A\cdot B$ and $A'\cdot B'$ are carried by the $\mathsf{Fin}(n)\times\mathsf{Fin}(p)$ matrix carrier.

By \autoref{def:matrix-classifier-specification}, it remains to prove the scalar classifier comparison entrywise. Fix $i:\mathsf{Fin}(n)$ and $k:\mathsf{Fin}(p)$. The matrix classifier hypotheses give, for every $j:\mathsf{Fin}(m)$,
$$
A_{i,j}\sim_R A'_{i,j}
\qquad\text{and}\qquad
B_{j,k}\sim_R B'_{j,k}.
$$
The multiplication congruence field in \autoref{def:ring-stability-certificate} therefore gives the summand comparison
$$
A_{i,j}\cdot_R B_{j,k}
\sim_R
A'_{i,j}\cdot_R B'_{j,k}
$$
for every inner index $j$.

The shared inner dimension supplies the same finite enumeration of $\mathsf{Fin}(m)$ on both sides. Hence these summand comparisons form a same-length finite-spine classifier witness between
$$
j\mapsto A_{i,j}\cdot_R B_{j,k}
\qquad\text{and}\qquad
j\mapsto A'_{i,j}\cdot_R B'_{j,k}.
$$
Applying \autoref{thm:finite-additive-fold-congruence-coefficient-lists} to that finite-spine witness yields
$$
\Sigma_{\mathsf{Fin}(m)}(j\mapsto A_{i,j}\cdot_R B_{j,k})
\sim_R
\Sigma_{\mathsf{Fin}(m)}(j\mapsto A'_{i,j}\cdot_R B'_{j,k}).
$$
By the displayed definition of matrix multiplication, this is exactly
$$
(A\cdot B)_{i,k}\sim_R(A'\cdot B')_{i,k}.
$$
Since $i$ and $k$ were arbitrary, \autoref{def:matrix-classifier-specification} packages the entrywise scalar comparisons as $A\cdot B\sim_{\mathrm{Mat}}A'\cdot B'$. The dimension and scalar-source ledgers are unchanged and are recorded by \autoref{def:matrix-ledger-policy}.
\end{proof}

\begin{lemma}[Finite additive fold splits pointwise sums]
\label{lem:finite-additive-fold-splits-pointwise-sums}
Let $R$ be a scalar $\RingUp$ carrier with classifier $\sim_R$, zero endpoint $0_R$, addition $+_R$, and finite additive fold $\mathsf{fold}_{+,R}$ from \autoref{def:finite-additive-scalar-fold}. Assume the scalar classifier equivalence fields and the additive abelian-group, additive congruence, and zero rows of \autoref{def:ring-stability-certificate}. For same-indexed finite carried scalar spines $xs$ and $ys$, write $xs+_R ys$ for the pointwise sum spine. Then
$$
\mathsf{fold}_{+,R}(xs+_R ys)
\sim_R
\mathsf{fold}_{+,R}(xs)+_R\mathsf{fold}_{+,R}(ys).
$$
\end{lemma}
\begin{proof}
Induct on the common finite spine carrying $xs$ and $ys$. In the nil case, all three folds reduce by \autoref{def:finite-additive-scalar-fold} to $0_R$. The additive zero rows classify $0_R$ with $0_R+_R0_R$, giving the displayed comparison.

In the cons case, write $xs=x::xs'$ and $ys=y::ys'$. The fold equation reduces the left endpoint to
$$
(x+_Ry)+_R\mathsf{fold}_{+,R}(xs'+_Rys').
$$
The induction hypothesis and additive congruence replace the tail fold by $\mathsf{fold}_{+,R}(xs')+_R\mathsf{fold}_{+,R}(ys')$. The additive associativity and commutativity fields in \autoref{def:ring-stability-certificate} rebracket and exchange the middle summands, classifying
$$
(x+_Ry)+_R(\mathsf{fold}_{+,R}(xs')+_R\mathsf{fold}_{+,R}(ys'))
$$
with
$$
(x+_R\mathsf{fold}_{+,R}(xs'))+_R(y+_R\mathsf{fold}_{+,R}(ys')).
$$
By \autoref{def:finite-additive-scalar-fold}, the latter endpoint is $\mathsf{fold}_{+,R}(xs)+_R\mathsf{fold}_{+,R}(ys)$. Classifier transitivity composes the displayed comparisons.
\end{proof}

\begin{theorem}[Matrix multiplication left-distributes over matrix addition]
\label{thm:matrix-multiplication-left-distributes-over-addition}
Let $R$ be a scalar $\RingUp$ carrier with classifier $\sim_R$, zero endpoint $0_R$, addition $+_R$, multiplication $\cdot_R$, and finite additive fold $\mathsf{fold}_{+,R}$. Write $\Sigma_{\mathsf{Fin}(m)}$ for that fold applied to the finite spine enumerating $\mathsf{Fin}(m)$. Let $n,m,p:\NatUp$, and consider the $\MatrixUp$ carriers of \autoref{def:matrix-carrier} with pointwise classifier from \autoref{def:matrix-classifier-specification}. Let
$$
A:\mathsf{Fin}(n)\times\mathsf{Fin}(m)\to R,
\qquad
B,C:\mathsf{Fin}(m)\times\mathsf{Fin}(p)\to R
$$
be carried matrices. Assume matrix addition is the pointwise operation
$$
(X+Y)_{u,w}:=X_{u,w}+_RY_{u,w}
$$
using the addition closure field of \autoref{def:matrix-stability-certificate}. Assume matrix multiplication is the finite-fold operation
$$
(X\cdot Y)_{u,w}:=\Sigma_{\mathsf{Fin}(m)}(v\mapsto X_{u,v}\cdot_RY_{v,w})
$$
for the matching inner finite index carrier, using the multiplication closure field of \autoref{def:matrix-stability-certificate}. Assume the scalar classifier equivalence fields, the scalar ring stability fields of \autoref{def:ring-stability-certificate}, and the fold congruence theorem \autoref{thm:finite-additive-fold-congruence-coefficient-lists}. Then $A\cdot(B+C)$ and $(A\cdot B)+(A\cdot C)$ are carried and
$$
A\cdot(B+C)\sim_{\mathrm{Mat}}(A\cdot B)+(A\cdot C).
$$
\end{theorem}
\begin{proof}
The addition closure item of \autoref{def:matrix-stability-certificate} carries $B+C$. Multiplication closure carries $A\cdot(B+C)$, $A\cdot B$, and $A\cdot C$, and addition closure carries $(A\cdot B)+(A\cdot C)$. By \autoref{def:matrix-classifier-specification}, it remains to prove the scalar classifier comparison at each entry.

Fix $i:\mathsf{Fin}(n)$ and $k:\mathsf{Fin}(p)$. Over the shared finite enumeration of $\mathsf{Fin}(m)$, define carried scalar spines
$$
xs_{i,k}(v):=A_{i,v}\cdot_RB_{v,k},
\qquad
ys_{i,k}(v):=A_{i,v}\cdot_RC_{v,k},
$$
and
$$
zs_{i,k}(v):=A_{i,v}\cdot_R(B_{v,k}+_RC_{v,k}).
$$
The pointwise definition of matrix addition gives
$$
(B+C)_{v,k}\sim_RB_{v,k}+_RC_{v,k}
$$
for every $v:\mathsf{Fin}(m)$. Scalar multiplication congruence and scalar left distributivity from \autoref{def:ring-stability-certificate} therefore give
$$
A_{i,v}\cdot_R(B+C)_{v,k}
\sim_R
(A_{i,v}\cdot_RB_{v,k})+_R(A_{i,v}\cdot_RC_{v,k})
$$
for every $v$. These comparisons form a same-length finite-spine classifier witness. Applying \autoref{thm:finite-additive-fold-congruence-coefficient-lists} to the shared inner enumeration yields
$$
\Sigma_{\mathsf{Fin}(m)}(v\mapsto A_{i,v}\cdot_R(B+C)_{v,k})
\sim_R
\Sigma_{\mathsf{Fin}(m)}(v\mapsto xs_{i,k}(v)+_Rys_{i,k}(v)).
$$

By \autoref{lem:finite-additive-fold-splits-pointwise-sums},
$$
\Sigma_{\mathsf{Fin}(m)}(v\mapsto xs_{i,k}(v)+_Rys_{i,k}(v))
\sim_R
\Sigma_{\mathsf{Fin}(m)}(v\mapsto xs_{i,k}(v))
+_R
\Sigma_{\mathsf{Fin}(m)}(v\mapsto ys_{i,k}(v)).
$$
The displayed definitions of matrix multiplication and matrix addition identify the leftmost endpoint with $(A\cdot(B+C))_{i,k}$ and the rightmost endpoint with $((A\cdot B)+(A\cdot C))_{i,k}$. Classifier transitivity composes the two fold comparisons with these operation readbacks, so
$$
(A\cdot(B+C))_{i,k}
\sim_R
((A\cdot B)+(A\cdot C))_{i,k}.
$$
Since $i$ and $k$ were arbitrary, \autoref{def:matrix-classifier-specification} packages the pointwise scalar comparisons as the displayed matrix classifier. The dimension and scalar-source witnesses are unchanged and are recorded by \autoref{def:matrix-ledger-policy}.
\end{proof}

\begin{lemma}[Finite additive fold separates pointwise scalar sums]
\label{lem:finite-additive-fold-separates-pointwise-scalar-sums}
Let $R$ be a scalar $\RingUp$ carrier with classifier $\sim_R$, zero endpoint $0_R$, addition $+_R$, and finite additive fold $\mathsf{fold}_{+,R}$ from \autoref{def:finite-additive-scalar-fold}. Assume the additive abelian-group laws and additive congruence fields retained in \autoref{def:ring-stability-certificate}. For any two finite carried scalar spines $xs$ and $ys$ over the same finite enumeration, write $xs+_Rys$ for the spine obtained by adding matching entries. Then
$$
\mathsf{fold}_{+,R}(xs+_Rys)\sim_R
\mathsf{fold}_{+,R}(xs)+_R\mathsf{fold}_{+,R}(ys).
$$
\end{lemma}
\begin{proof}
Induct on the shared finite enumeration. In the nil case, all three spines are $\mathsf{nil}$. The left endpoint reduces by \autoref{def:finite-additive-scalar-fold} to $0_R$, while the right endpoint reduces to $0_R+_R0_R$. The additive left-zero law in \autoref{def:ring-stability-certificate}, with classifier symmetry from the additive abelian-group laws, gives $0_R\sim_R0_R+_R0_R$.

In the cons case, write $xs=\mathsf{cons}(a,xs')$ and $ys=\mathsf{cons}(b,ys')$. The induction hypothesis gives
$$
\mathsf{fold}_{+,R}(xs'+_Rys')\sim_R
\mathsf{fold}_{+,R}(xs')+_R\mathsf{fold}_{+,R}(ys').
$$
Source reflexivity on $a+_Rb$ and additive congruence transport this comparison to
$$
(a+_Rb)+_R\mathsf{fold}_{+,R}(xs'+_Rys')
\sim_R
(a+_Rb)+_R(\mathsf{fold}_{+,R}(xs')+_R\mathsf{fold}_{+,R}(ys')).
$$
Apply \autoref{thm:abgroup-middle-four-interchange}, read for the additive abelian-group operation of the scalar ring, to classify the right endpoint with
$$
(a+_R\mathsf{fold}_{+,R}(xs'))+_R(b+_R\mathsf{fold}_{+,R}(ys')).
$$
The cons clauses of \autoref{def:finite-additive-scalar-fold} identify this last endpoint with
$$
\mathsf{fold}_{+,R}(xs)+_R\mathsf{fold}_{+,R}(ys),
$$
and identify the first endpoint with $\mathsf{fold}_{+,R}(xs+_Rys)$. Classifier transitivity gives the cons case.
\end{proof}

\begin{theorem}[Matrix multiplication right-distributes over matrix addition]
\label{thm:matrix-multiplication-right-distributes-over-addition}
Let $R$ be a scalar $\RingUp$ carrier with classifier $\sim_R$, zero endpoint $0_R$, addition $+_R$, multiplication $\cdot_R$, and finite additive folds over finite index spines as in \autoref{def:finite-additive-scalar-fold}. Let $n,m,p:\NatUp$, and consider the $\MatrixUp$ carriers of \autoref{def:matrix-carrier} with pointwise classifier from \autoref{def:matrix-classifier-specification}. Let
$$
A,B:\mathsf{Fin}(n)\times\mathsf{Fin}(m)\to R,
\qquad
C:\mathsf{Fin}(m)\times\mathsf{Fin}(p)\to R
$$
be carried matrices. Assume matrix addition is the pointwise operation
$$
(X+Y)_{u,v}:=X_{u,v}+_RY_{u,v}
$$
using the addition closure field of \autoref{def:matrix-stability-certificate}, and assume matrix multiplication is the finite-fold operation
$$
(X\cdot Y)_{u,w}:=\Sigma_{\mathsf{Fin}(m)}(j\mapsto X_{u,j}\cdot_RY_{j,w})
$$
for the shared inner finite index carrier, using the multiplication closure field of \autoref{def:matrix-stability-certificate}. Assume the scalar ring stability fields of \autoref{def:ring-stability-certificate}: additive abelian-group laws, addition and multiplication congruence, and right distributivity. Then the displayed matrices are carried and
$$
(A+B)\cdot C\sim_{\mathrm{Mat}}(A\cdot C)+(B\cdot C).
$$
\end{theorem}
\begin{proof}
The addition closure item of \autoref{def:matrix-stability-certificate} carries $A+B$. Multiplication closure carries $(A+B)\cdot C$, $A\cdot C$, and $B\cdot C$, and addition closure then carries $(A\cdot C)+(B\cdot C)$.

By \autoref{def:matrix-classifier-specification}, it remains to compare corresponding scalar entries. Fix $i:\mathsf{Fin}(n)$ and $k:\mathsf{Fin}(p)$. Over the finite spine enumerating $\mathsf{Fin}(m)$, define
$$
xs(j):=A_{i,j}\cdot_RC_{j,k},
\qquad
ys(j):=B_{i,j}\cdot_RC_{j,k}.
$$
The pointwise matrix-addition definition and scalar right distributivity give, for each $j:\mathsf{Fin}(m)$,
$$
(A+B)_{i,j}\cdot_RC_{j,k}
\sim_R
(A_{i,j}\cdot_RC_{j,k})+_R(B_{i,j}\cdot_RC_{j,k}).
$$
These pointwise comparisons form a finite-spine classifier witness. Applying \autoref{thm:finite-additive-fold-respects-scalar-classifier} gives
$$
\Sigma_{\mathsf{Fin}(m)}(j\mapsto (A+B)_{i,j}\cdot_RC_{j,k})
\sim_R
\Sigma_{\mathsf{Fin}(m)}(j\mapsto xs(j)+_Rys(j)).
$$
By \autoref{lem:finite-additive-fold-separates-pointwise-scalar-sums}, the right endpoint is classified with
$$
\Sigma_{\mathsf{Fin}(m)}(j\mapsto xs(j))+_R
\Sigma_{\mathsf{Fin}(m)}(j\mapsto ys(j)).
$$
Unfolding the displayed matrix-multiplication operation identifies these two folded summands with $(A\cdot C)_{i,k}$ and $(B\cdot C)_{i,k}$, and the pointwise matrix-addition operation identifies their scalar sum with $((A\cdot C)+(B\cdot C))_{i,k}$. Classifier transitivity therefore gives
$$
((A+B)\cdot C)_{i,k}\sim_R((A\cdot C)+(B\cdot C))_{i,k}.
$$
Since $i$ and $k$ were arbitrary, \autoref{def:matrix-classifier-specification} packages these entrywise scalar comparisons as
$$
(A+B)\cdot C\sim_{\mathrm{Mat}}(A\cdot C)+(B\cdot C).
$$
The dimensions and shared scalar source are unchanged throughout; \autoref{def:matrix-ledger-policy} records the common row, contraction, and column witnesses.
\end{proof}
