\section{The certificate}

\begin{definition}[Matrix source specification]
\label{def:matrix-source-specification}
The source is the finite index carrier from \(\NatUp\) and the scalar source \(\RingUp(R)\).
\end{definition}

\begin{definition}[Matrix pattern specification]
\label{def:matrix-pattern-specification}
The pattern is finite table formation assigning a scalar witness to each index pair.
\end{definition}

\begin{definition}[Matrix classifier specification]
\label{def:matrix-classifier-specification}
The classifier relates matrices pointwise by scalar classifier at every index pair.
\end{definition}

\begin{lemma}[Matrix singleton classifier append visible-factor exclusion]
\label{lem:matrix-singleton-classifier-append-visible-factor-exclusion}
For histories $h,p,q:\mathsf{BHist}$, the singleton matrix classifier admits
no witness whose source or target endpoint is an append with a visible factor:
\begin{enumerate}
  \item $\rho_{\emp}(\mathsf{append}(\Ezero(p),q),h)\to\bot$ and
        $\rho_{\emp}(h,\mathsf{append}(\Ezero(p),q))\to\bot$;
  \item $\rho_{\emp}(\mathsf{append}(\Eone(p),q),h)\to\bot$ and
        $\rho_{\emp}(h,\mathsf{append}(\Eone(p),q))\to\bot$;
  \item $\rho_{\emp}(\mathsf{append}(p,\Ezero(q)),h)\to\bot$ and
        $\rho_{\emp}(h,\mathsf{append}(p,\Ezero(q)))\to\bot$;
  \item $\rho_{\emp}(\mathsf{append}(p,\Eone(q)),h)\to\bot$ and
        $\rho_{\emp}(h,\mathsf{append}(p,\Eone(q)))\to\bot$.
\end{enumerate}
\end{lemma}
\begin{proof}
Unfold the singleton matrix classifier. Any displayed classifier witness
projects to the singleton carrier endpoint on its source and to the singleton
carrier endpoint on its target.

For an endpoint $\mathsf{append}(\Ezero(p),q)$ or
$\mathsf{append}(\Eone(p),q)$, append empty-result inversion forces the left
visible factor to be empty. Constructor separation contradicts the resulting
$\Ezero(p)\hsame\emp$ or $\Eone(p)\hsame\emp$ endpoint.

For an endpoint $\mathsf{append}(p,\Ezero(q))$ or
$\mathsf{append}(p,\Eone(q))$, the same inversion forces the right visible
factor to be empty. Constructor separation gives the source-side contradiction,
and the target-side pair is exactly
\autoref{thm:matrix-singleton-classifier-append-visible-right-absurd}.
\end{proof}
\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonClassifier\_append\_visible\_factor\_exclusion}

\begin{definition}[Matrix stability certificate]
\label{def:matrix-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item pointwise reflexivity
  \item pointwise transitivity
  \item finite index closure
  \item addition closure by scalar addition
  \item multiplication closure by finite scalar sums and products
  \item identity matrix closure when dimensions match
\end{enumerate}
\end{definition}

\begin{definition}[Matrix ledger policy]
\label{def:matrix-ledger-policy}
Record the dimension witnesses, index discipline, and scalar ledger for each entry.
\end{definition}

\begin{definition}[\(\MatrixUp\) certificate obligations]
\label{def:matrix-certificate-obligations}
The carrier, source, pattern, classifier, stability, and ledger data listed above are the obligations required for the interface name \(\MatrixUp\).
\end{definition}
\begin{theorem}[Matrix singleton carrier semantic name certificate]
\label{thm:matrix-singleton-carrier-semantic-namecert}
The singleton matrix carrier \(A_{\emp}(h)\equiv h\hsame\emp\), used as
source, pattern, and ledger policy, together with the ambient classifier
\(\hsame\), forms a semantic name certificate:
$$
\mathsf{SemanticNameCert}(A_{\emp},A_{\emp},A_{\emp},\hsame).
$$
\end{theorem}
\begin{proof}
The carrier is inhabited by \(\emp\). The classifier rows are the reflexive,
symmetric, and transitive laws for \(\hsame\). Carrier transport sends
\(h\hsame k\) and \(h\hsame\emp\) to \(k\hsame\emp\) by symmetry followed by
transitivity. Since source, pattern, and ledger policy are the same concrete
singleton carrier, the two semantic soundness maps are identity projections.
\end{proof}

\begin{theorem}[Singleton empty-history MatrixUp certificate obligations]
\label{thm:matrix-singleton-empty-history-certificate-obligations}
For the singleton empty-history matrix data of
\autoref{def:singleton-empty-history-matrix-instance}, the $\MatrixUp$
obligations of \autoref{def:matrix-certificate-obligations} are instantiated
as follows:
\begin{enumerate}
  \item the source, pattern, and ledger carriers are the singleton carrier
  $A_{\emp}$ on $\mathsf{BHist}$;
  \item the classifier is $\rho_{\emp}$;
  \item the semantic $\NameCert$ rows are supplied by
  \autoref{thm:matrix-singleton-carrier-semantic-namecert};
  \item the zero, identity, operation-closure, unit, associativity, and
  congruence rows are supplied by
  \autoref{thm:singleton-empty-history-matrix-laws}.
\end{enumerate}
\end{theorem}
\begin{proof}
Unpack \autoref{def:matrix-certificate-obligations}. The source, pattern, and
ledger slots are all the displayed singleton history carrier $A_{\emp}$ from
\autoref{def:singleton-empty-history-matrix-instance}, and the classifier slot
is the displayed relation $\rho_{\emp}$. The ledger is the same singleton
dimension/source ledger, with row and column witnesses represented by $\emp$.

The carrier transport and semantic soundness rows are exactly
\autoref{thm:matrix-singleton-carrier-semantic-namecert}. That theorem uses
the same carrier in the source, pattern, and ledger positions, so its semantic
$\NameCert$ data supplies the certificate-level carrier and classifier core.

It remains to fill the stability rows. The law package of
\autoref{thm:singleton-empty-history-matrix-laws} carries $0_M$ and $I_M$,
closes $A_{\emp}$ under $\oplus$ and $\odot$, supplies the unit and identity
laws, proves associativity under $\rho_{\emp}$, and gives classifier
congruence for both operations. These rows exhaust the stability portion of
the displayed $\MatrixUp$ obligations.
\end{proof}

\paragraph{Primitive boundary.}
\(\MatrixUp\) is not a BEDC primitive. It is admitted only when those obligations are supplied.

\end{remark}
