\section{The certificate}

\begin{definition}[Matrix source specification]
\label{def:matrix-source-specification}
The source is the finite index carrier from \(\NatUp\) and the scalar source \(\RingUp(R)\).
\end{definition}

\begin{definition}[Matrix pattern specification]
\label{def:matrix-pattern-specification}
The pattern is finite table formation assigning a scalar witness to each index pair.
\end{definition}

\begin{definition}[Matrix classifier specification]
\label{def:matrix-classifier-specification}
The classifier relates matrices pointwise by scalar classifier at every index pair.
\end{definition}

\begin{definition}[Matrix stability certificate]
\label{def:matrix-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item pointwise reflexivity
  \item pointwise transitivity
  \item finite index closure
  \item addition closure by scalar addition
  \item multiplication closure by finite scalar sums and products
  \item identity matrix closure when dimensions match
\end{enumerate}
\end{definition}

\begin{definition}[Matrix ledger policy]
\label{def:matrix-ledger-policy}
Record the dimension witnesses, index discipline, and scalar ledger for each entry.
\end{definition}

\begin{definition}[\(\MatrixUp\) certificate obligations]
\label{def:matrix-certificate-obligations}
The carrier, source, pattern, classifier, stability, and ledger data listed above are the obligations required for the interface name \(\MatrixUp\).
\end{definition}
\begin{theorem}[Matrix singleton carrier semantic name certificate]
\label{thm:matrix-singleton-carrier-semantic-namecert}
The singleton matrix carrier \(A_{\emp}(h)\equiv h\hsame\emp\), used as
source, pattern, and ledger policy, together with the ambient classifier
\(\hsame\), forms a semantic name certificate:
$$
\mathsf{SemanticNameCert}(A_{\emp},A_{\emp},A_{\emp},\hsame).
$$
\end{theorem}
\begin{proof}
The carrier is inhabited by \(\emp\). The classifier rows are the reflexive,
symmetric, and transitive laws for \(\hsame\). Carrier transport sends
\(h\hsame k\) and \(h\hsame\emp\) to \(k\hsame\emp\) by symmetry followed by
transitivity. Since source, pattern, and ledger policy are the same concrete
singleton carrier, the two semantic soundness maps are identity projections.
\end{proof}
\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonCarrier\_semanticNameCert}

\paragraph{Primitive boundary.}
\(\MatrixUp\) is not a BEDC primitive. It is admitted only when those obligations are supplied.

\begin{remark}[Zero-axiom encoding note]
\label{rem:matrix-zero-axiom-note}
Avoids funext by using pointwise classifier sameness; finite indices are inductive, not quotient indices.
\end{remark}
