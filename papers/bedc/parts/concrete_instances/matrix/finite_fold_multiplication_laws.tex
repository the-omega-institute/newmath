\begin{definition}[Finite additive scalar fold]
\label{def:finite-additive-scalar-fold}
Let $R$ be a scalar $\RingUp$ carrier with classifier $\sim_R$, zero endpoint
$0_R$, addition $+_R$, and the additive stability rows of
\autoref{def:ring-stability-certificate}. For every finite carried scalar spine
$xs$, the fold $\mathsf{fold}_{+,R}(xs)$ is generated by the two clauses
$$
\mathsf{fold}_{+,R}(\mathsf{nil}) := 0_R,
\qquad
\mathsf{fold}_{+,R}(x::xs) := x+_R\mathsf{fold}_{+,R}(xs).
$$
The fold ledger records the finite spine order, the carriedness witness for
each entry, and the additive ledger used at every cons step.
\end{definition}

\begin{theorem}[Finite additive fold respects scalar classifier]
\label{thm:finite-additive-fold-respects-scalar-classifier}
Let $R$ be a scalar $\RingUp$ carrier with classifier $\sim_R$, zero endpoint
$0_R$, addition $+_R$, and the additive stability rows of
\autoref{def:ring-stability-certificate}. If finite carried scalar spines $xs$
and $ys$ are related by
$\mathsf{ListClassifierSpec}(\sim_R,xs,ys)$, then
$$
\mathsf{fold}_{+,R}(xs)\sim_R\mathsf{fold}_{+,R}(ys).
$$
\end{theorem}
\begin{proof}
Induct on the displayed finite-spine classifier witness. In the nil case, both
folds reduce by \autoref{def:finite-additive-scalar-fold} to $0_R$, and the
zero self-classification row of \autoref{def:ring-stability-certificate} gives
$0_R\sim_R0_R$.

In the cons case, the head comparison gives $x\sim_R y$, and the induction
hypothesis gives
$\mathsf{fold}_{+,R}(xs)\sim_R\mathsf{fold}_{+,R}(ys)$. Additive congruence from
\autoref{def:ring-stability-certificate} transports these two comparisons to
$x+_R\mathsf{fold}_{+,R}(xs)\sim_R y+_R\mathsf{fold}_{+,R}(ys)$, which is exactly
the folded comparison after unfolding
\autoref{def:finite-additive-scalar-fold}.
\end{proof}

\begin{theorem}[Singleton additive fold carrier readback]
\label{thm:matrix-singleton-add-fold-carrier-iff}
For the singleton matrix carrier, let
\(\mathsf{MatrixSingletonAddFold}\) fold a finite spine by the concrete
singleton addition \(\mathsf{MatrixSingletonAdd}\), and let
\(\mathsf{MatrixSingletonAddFoldSpineCarrier}\) record that every spine entry
is carried. Then
$$
\mathsf{MatrixSingletonCarrier}(\mathsf{MatrixSingletonAddFold}(xs))
\Longleftrightarrow
\mathsf{MatrixSingletonAddFoldSpineCarrier}(xs).
$$
\end{theorem}
\begin{proof}
Induct on the finite spine. The nil fold is \(\emp\), so both sides are the
empty endpoint readback. In the cons case, singleton addition is kernel
append; empty-append inversion splits the folded endpoint into the head
carrier and the tail-fold carrier, and the induction hypothesis identifies the
tail-fold carrier with the tail spine carrier.
\end{proof}

\begin{theorem}[Singleton additive fold append carrier readback]
\label{thm:matrix-singleton-add-fold-append-carrier-iff}
For finite singleton-matrix spines $xs$ and $ys$,
$$
\mathsf{MatrixSingletonCarrier}(\mathsf{MatrixSingletonAddFold}(xs\,{++}\,ys))
\Longleftrightarrow
\mathsf{MatrixSingletonAddFoldSpineCarrier}(xs)\land
\mathsf{MatrixSingletonAddFoldSpineCarrier}(ys).
$$
\end{theorem}
\begin{proof}
Induct on $xs$. For the empty spine, the appended fold is the fold of $ys$,
and the base fold readback identifies its carrier with the spine carrier of
$ys$. For a cons spine, concrete singleton addition is kernel append, so
empty-append inversion splits the folded endpoint into the head carrier and
the appended tail-fold carrier. The induction hypothesis splits the tail-fold
carrier into the tail spine carrier and the carrier of $ys$, and the reverse
direction repacks those two splits through the same empty-append constructor.
\end{proof}

\begin{lemma}[Finite Kronecker selector sum]
\label{lem:matrix-finite-kronecker-selector-sum}
Let $J=\mathsf{Fin}(p)$ be a finite index carrier licensed by the $\NatUp$
certificate of \autoref{ch:concrete-instances-nat-namecert}. Let $R$ be a
scalar carrier from the $\CommRingUp$ certificate of
\autoref{ch:concrete-instances-commring-namecert}, with classifier $\sim_R$,
operations $+_R,\cdot_R$, and scalars $0_R,1_R$. Assume a finite-index sum
certificate $\Sigma_J$ satisfying classifier congruence and the selector row:
if $j_0:J$, $t:J\to R$ is carried pointwise, $t(j_0)\sim_R u$, and
$t(j)\sim_R 0_R$ for every $j\ne j_0$, then $\Sigma_J(t)\sim_R u$.
For every pointwise carried scalar family $x:J\to R$ and every $j_0:J$,
$$
\Sigma_J(j\mapsto \delta_{j_0,j}\cdot_R x(j))\sim_R x(j_0)
\qquad\text{and}\qquad
\Sigma_J(j\mapsto x(j)\cdot_R \delta_{j,j_0})\sim_R x(j_0),
$$
where $\delta_{a,b}$ is $1_R$ when the finite-index equality certificate
identifies $a$ with $b$, and $0_R$ otherwise.
\end{lemma}
\begin{proof}
For the first displayed sum, split the finite index $j$. At the selected index
$j=j_0$, the Kronecker entry is $1_R$, so the scalar left-one law gives
$\delta_{j_0,j_0}\cdot_R x(j_0)\sim_R x(j_0)$. At every other index, the
Kronecker entry is $0_R$, so scalar left-zero absorption gives
$\delta_{j_0,j}\cdot_R x(j)\sim_R 0_R$. These are exactly the selected and
off-selected hypotheses of the selector row for $\Sigma_J$, applied to the
summand family $j\mapsto \delta_{j_0,j}\cdot_R x(j)$ with $u=x(j_0)$.

The second displayed sum uses the same selector-row hypothesis for
$j\mapsto x(j)\cdot_R \delta_{j,j_0}$. The scalar right-one law gives the
selected comparison at $j=j_0$, and scalar right-zero absorption gives the
comparison with $0_R$ away from $j_0$.
\end{proof}

\begin{theorem}[Matrix identity multiplication unit]
\label{thm:matrix-identity-multiplication-unit}
Let $n,m:\NatUp$ and let $R$ be a scalar $\CommRingUp$ certificate. Consider
the $\MatrixUp$ carrier of \autoref{def:matrix-carrier} with the pointwise
classifier of \autoref{def:matrix-classifier-specification}. Assume the
stability data of \autoref{def:matrix-stability-certificate} supplies finite
product sums over $\mathsf{Fin}(n)$ and $\mathsf{Fin}(m)$, the scalar zero and
one multiplication laws over $R$, and carried Kronecker identity matrices
$I_n$ and $I_m$ with entries
$$
(I_n)_{i,a}\sim_R\delta_{i,a}
\qquad\text{and}\qquad
(I_m)_{b,j}\sim_R\delta_{b,j}.
$$
If $M:\mathsf{Fin}(n)\times\mathsf{Fin}(m)\to R$ is carried, then the products
$I_n\cdot M$ and $M\cdot I_m$ are carried and
$$
I_n\cdot M\sim_{\mathrm{Mat}} M
\qquad\text{and}\qquad
M\cdot I_m\sim_{\mathrm{Mat}} M.
$$
\end{theorem}
\begin{proof}
The carriedness of $I_n$ and $I_m$ is part of the Kronecker identity closure
field, and $M$ is carried by hypothesis. Multiplication closure in
\autoref{def:matrix-stability-certificate} therefore carries both products.
It remains to prove the pointwise scalar comparisons required by
\autoref{def:matrix-classifier-specification}.

Fix $i:\mathsf{Fin}(n)$ and $b:\mathsf{Fin}(m)$. The left product entry is the
finite sum
$$
(I_n\cdot M)_{i,b}
\sim_R
\Sigma_{\mathsf{Fin}(n)}(a\mapsto (I_n)_{i,a}\cdot_R M_{a,b}).
$$
For each inner index $a$, identity-entry closure gives
$(I_n)_{i,a}\sim_R\delta_{i,a}$, and scalar classifier congruence for
$\cdot_R$ gives
$$
(I_n)_{i,a}\cdot_R M_{a,b}
\sim_R
\delta_{i,a}\cdot_R M_{a,b}.
$$
The finite-sum classifier congruence for $\Sigma_{\mathsf{Fin}(n)}$ yields
$$
\Sigma_{\mathsf{Fin}(n)}(a\mapsto (I_n)_{i,a}\cdot_R M_{a,b})
\sim_R
\Sigma_{\mathsf{Fin}(n)}(a\mapsto \delta_{i,a}\cdot_R M_{a,b}).
$$
Applying \autoref{lem:matrix-finite-kronecker-selector-sum} to the carried
family $a\mapsto M_{a,b}$ selects the unique inner index $a=i$, hence
$(I_n\cdot M)_{i,b}\sim_R M_{i,b}$.

For the right product, the entry at the same outer indices is
$$
(M\cdot I_m)_{i,b}
\sim_R
\Sigma_{\mathsf{Fin}(m)}(a\mapsto M_{i,a}\cdot_R (I_m)_{a,b}).
$$
For each inner index $a$, identity-entry closure gives
$(I_m)_{a,b}\sim_R\delta_{a,b}$, and scalar classifier congruence for
$\cdot_R$ gives
$$
M_{i,a}\cdot_R (I_m)_{a,b}
\sim_R
M_{i,a}\cdot_R \delta_{a,b}.
$$
The finite-sum classifier congruence for $\Sigma_{\mathsf{Fin}(m)}$ yields
$$
\Sigma_{\mathsf{Fin}(m)}(a\mapsto M_{i,a}\cdot_R (I_m)_{a,b})
\sim_R
\Sigma_{\mathsf{Fin}(m)}(a\mapsto M_{i,a}\cdot_R \delta_{a,b}).
$$
The right-hand selector part of
\autoref{lem:matrix-finite-kronecker-selector-sum} applied to the carried
family $a\mapsto M_{i,a}$ selects $a=b$, giving
$(M\cdot I_m)_{i,b}\sim_R M_{i,b}$.

Since these scalar comparisons hold for every row and column index, the
pointwise matrix classifier gives both displayed matrix identity laws. The
ledger in \autoref{def:matrix-ledger-policy} records the same row and column
witnesses throughout the two products, so no dimension data is changed.
\end{proof}

\begin{definition}[Two-index finite fold swap certificate]
\label{def:two-index-finite-fold-swap-certificate}
Let $R$ be a scalar $\RingUp$ carrier with classifier $\sim_R$, zero endpoint $0_R$, addition $+_R$, and finite additive fold from \autoref{thm:finite-additive-fold-congruence-coefficient-lists}. Let $J$ and $K$ be finite index carriers, and write $\Sigma_J$ and $\Sigma_K$ for that fold applied to the finite spines enumerating $J$ and $K$. A two-index finite fold swap certificate for $(J,K)$ is the following stability field: for every pointwise carried scalar family $s:J\times K\to R$ and every pointwise classifier-compatible transpose $s^\tau:K\times J\to R$ satisfying
$$
s^\tau(k,j)\sim_R s(j,k)
$$
for all $j:J$ and $k:K$, the two iterated folds are scalar-classified:
$$
\Sigma_J(j\mapsto \Sigma_K(k\mapsto s(j,k)))
\sim_R
\Sigma_K(k\mapsto \Sigma_J(j\mapsto s^\tau(k,j))).
$$
\end{definition}

\begin{lemma}[Scalar product distributes across finite additive folds]
\label{lem:scalar-product-distributes-across-finite-additive-folds}
Let $R$ be a scalar $\RingUp$ carrier with classifier $\sim_R$, zero endpoint $0_R$, addition $+_R$, multiplication $\cdot_R$, and finite additive fold from \autoref{thm:finite-additive-fold-congruence-coefficient-lists}. Assume the ring stability fields of \autoref{def:ring-stability-certificate}: additive congruence, multiplication congruence, left and right distributivity, and zero multiplication laws. For every carried scalar $c$ and every finite carried scalar spine $xs$, let $xs\cdot_R c$ denote the spine obtained by multiplying each entry of $xs$ on the right by $c$, and let $c\cdot_R xs$ denote the spine obtained by multiplying each entry on the left by $c$. Then
$$
\mathsf{fold}_{+,R}(xs)\cdot_R c
\sim_R
\mathsf{fold}_{+,R}(xs\cdot_R c)
$$
and
$$
c\cdot_R\mathsf{fold}_{+,R}(xs)
\sim_R
\mathsf{fold}_{+,R}(c\cdot_R xs).
$$
\end{lemma}
\begin{proof}
Induct on the finite spine $xs$. In the nil case, \autoref{thm:finite-additive-fold-congruence-coefficient-lists} identifies the fold with $0_R$. The right and left zero multiplication laws from \autoref{def:ring-stability-certificate} identify $0_R\cdot_R c$ and $c\cdot_R0_R$ with $0_R$, which is the fold of the corresponding empty product spine.

In the cons case, write $xs=\mathsf{cons}(a,xs')$. For right multiplication, the fold equation gives $(a+_R\mathsf{fold}_{+,R}(xs'))\cdot_R c$. Right distributivity identifies this scalar with
$$
(a\cdot_R c)+_R(\mathsf{fold}_{+,R}(xs')\cdot_R c).
$$
The induction hypothesis identifies the second summand with $\mathsf{fold}_{+,R}(xs'\cdot_R c)$. Additive congruence then gives
$$
(a\cdot_R c)+_R\mathsf{fold}_{+,R}(xs'\cdot_R c),
$$
which is the cons fold of $xs\cdot_R c$.

The left multiplication proof is the same induction with left distributivity applied to $c\cdot_R(a+_R\mathsf{fold}_{+,R}(xs'))$. It yields
$$
(c\cdot_R a)+_R(c\cdot_R\mathsf{fold}_{+,R}(xs')),
$$
then uses the induction hypothesis and additive congruence to identify the result with the cons fold of $c\cdot_R xs$.
\end{proof}

\begin{theorem}[Matrix multiplication associativity from finite folds]
\label{thm:matrix-multiplication-associativity-finite-folds}
Let $R$ be a scalar $\RingUp$ carrier with classifier $\sim_R$, addition $+_R$, multiplication $\cdot_R$, zero endpoint $0_R$, and finite additive fold from \autoref{thm:finite-additive-fold-congruence-coefficient-lists}. Let $n,m,p,q:\NatUp$, and consider the $\MatrixUp$ carriers of \autoref{def:matrix-carrier} with pointwise classifier from \autoref{def:matrix-classifier-specification}. Let
$$
A:\mathsf{Fin}(n)\times\mathsf{Fin}(m)\to R,
\qquad
B:\mathsf{Fin}(m)\times\mathsf{Fin}(p)\to R,
\qquad
C:\mathsf{Fin}(p)\times\mathsf{Fin}(q)\to R
$$
be carried matrices. Assume matrix multiplication is the finite-fold operation
$$
(X\cdot Y)_{u,w}:=\Sigma_V(v\mapsto X_{u,v}\cdot_RY_{v,w})
$$
for the matching inner finite index carrier $V$, using the multiplication closure field of \autoref{def:matrix-stability-certificate}. Assume the scalar ring stability fields of \autoref{def:ring-stability-certificate}, the fold congruence theorem \autoref{thm:finite-additive-fold-congruence-coefficient-lists}, and the two-index finite fold swap certificate of \autoref{def:two-index-finite-fold-swap-certificate} for $\mathsf{Fin}(p)$ and $\mathsf{Fin}(m)$. Then the two matched products are carried and
$$
(A\cdot B)\cdot C\sim_{\mathrm{Mat}}A\cdot(B\cdot C).
$$
\end{theorem}
\begin{proof}
The multiplication closure item of \autoref{def:matrix-stability-certificate} carries $A\cdot B$ and $B\cdot C$, and then carries the two outer products $(A\cdot B)\cdot C$ and $A\cdot(B\cdot C)$. By \autoref{def:matrix-classifier-specification}, it remains to prove the scalar classifier comparison at each entry.

Fix $i:\mathsf{Fin}(n)$ and $l:\mathsf{Fin}(q)$, and set
$$
t(b,a):=A_{i,a}\cdot_R(B_{a,b}\cdot_R C_{b,l})
$$
for $a:\mathsf{Fin}(m)$ and $b:\mathsf{Fin}(p)$. Expanding the left-associated matrix product gives
$$
((A\cdot B)\cdot C)_{i,l}
\sim_R
\Sigma_{b:\mathsf{Fin}(p)}((A\cdot B)_{i,b}\cdot_R C_{b,l}).
$$
For each fixed $b$, expanding $(A\cdot B)_{i,b}$ and applying the right-multiplication half of \autoref{lem:scalar-product-distributes-across-finite-additive-folds} gives
$$
(A\cdot B)_{i,b}\cdot_R C_{b,l}
\sim_R
\Sigma_{a:\mathsf{Fin}(m)}((A_{i,a}\cdot_R B_{a,b})\cdot_R C_{b,l}).
$$
Scalar associativity from the multiplicative monoid part of \autoref{def:ring-stability-certificate} gives a pointwise comparison
$$
(A_{i,a}\cdot_R B_{a,b})\cdot_R C_{b,l}
\sim_R
A_{i,a}\cdot_R(B_{a,b}\cdot_R C_{b,l}).
$$
The finite-fold congruence theorem \autoref{thm:finite-additive-fold-congruence-coefficient-lists} lifts these pointwise comparisons through the inner and outer folds. Hence
$$
((A\cdot B)\cdot C)_{i,l}
\sim_R
\Sigma_{b:\mathsf{Fin}(p)}\Sigma_{a:\mathsf{Fin}(m)}t(b,a).
$$

Expanding the right-associated matrix product gives
$$
(A\cdot(B\cdot C))_{i,l}
\sim_R
\Sigma_{a:\mathsf{Fin}(m)}(A_{i,a}\cdot_R(B\cdot C)_{a,l}).
$$
For each fixed $a$, expanding $(B\cdot C)_{a,l}$ and applying the left-multiplication half of \autoref{lem:scalar-product-distributes-across-finite-additive-folds} gives
$$
A_{i,a}\cdot_R(B\cdot C)_{a,l}
\sim_R
\Sigma_{b:\mathsf{Fin}(p)}A_{i,a}\cdot_R(B_{a,b}\cdot_R C_{b,l}).
$$
Again \autoref{thm:finite-additive-fold-congruence-coefficient-lists} lifts the pointwise comparisons through the finite folds, so
$$
(A\cdot(B\cdot C))_{i,l}
\sim_R
\Sigma_{a:\mathsf{Fin}(m)}\Sigma_{b:\mathsf{Fin}(p)}t(b,a).
$$

Apply the fold swap certificate of \autoref{def:two-index-finite-fold-swap-certificate} to the family $t$ and its transpose $t^\tau(a,b):=t(b,a)$. This gives
$$
\Sigma_{b:\mathsf{Fin}(p)}\Sigma_{a:\mathsf{Fin}(m)}t(b,a)
\sim_R
\Sigma_{a:\mathsf{Fin}(m)}\Sigma_{b:\mathsf{Fin}(p)}t(b,a).
$$
Classifier transitivity composes the left expansion, this fold swap comparison, and the symmetry of the right expansion. Thus every entry of $(A\cdot B)\cdot C$ is scalar-classified with the corresponding entry of $A\cdot(B\cdot C)$. The pointwise matrix classifier of \autoref{def:matrix-classifier-specification} gives the displayed matrix classification, and the unchanged dimension witnesses are recorded by the ledger policy of \autoref{def:matrix-ledger-policy}.
\end{proof}

\begin{theorem}[Matrix multiplication classifier congruence]
\label{thm:matrix-multiplication-classifier-congruence}
Let $R$ be a scalar $\RingUp$ carrier with classifier $\sim_R$, multiplication $\cdot_R$, addition $+_R$, zero endpoint $0_R$, and finite additive fold from \autoref{thm:finite-additive-fold-congruence-coefficient-lists}. Let $n,m,p:\NatUp$, and consider the $\MatrixUp$ carriers of \autoref{def:matrix-carrier} with pointwise classifier from \autoref{def:matrix-classifier-specification}. Let
$$
A,A':\mathsf{Fin}(n)\times\mathsf{Fin}(m)\to R,
\qquad
B,B':\mathsf{Fin}(m)\times\mathsf{Fin}(p)\to R
$$
be carried matrices over the same scalar source and the same matching inner dimension $m$. Assume matrix multiplication is the finite-fold operation
$$
(X\cdot Y)_{i,k}:=\Sigma_{\mathsf{Fin}(m)}(j\mapsto X_{i,j}\cdot_RY_{j,k})
$$
for the shared inner finite index carrier, using the multiplication closure field of \autoref{def:matrix-stability-certificate}. Assume the scalar multiplication congruence field of \autoref{def:ring-stability-certificate} and the fold congruence theorem \autoref{thm:finite-additive-fold-congruence-coefficient-lists}. If
$$
A\sim_{\mathrm{Mat}}A'
\qquad\text{and}\qquad
B\sim_{\mathrm{Mat}}B',
$$
then $A\cdot B$ and $A'\cdot B'$ are carried $\MatrixUp$ matrices, and
$$
A\cdot B\sim_{\mathrm{Mat}}A'\cdot B'.
$$
\end{theorem}
\begin{proof}
The multiplication closure item in \autoref{def:matrix-stability-certificate} applies to the carried compatible pairs $(A,B)$ and $(A',B')$, so both products $A\cdot B$ and $A'\cdot B'$ are carried by the $\mathsf{Fin}(n)\times\mathsf{Fin}(p)$ matrix carrier.

By \autoref{def:matrix-classifier-specification}, it remains to prove the scalar classifier comparison entrywise. Fix $i:\mathsf{Fin}(n)$ and $k:\mathsf{Fin}(p)$. The matrix classifier hypotheses give, for every $j:\mathsf{Fin}(m)$,
$$
A_{i,j}\sim_R A'_{i,j}
\qquad\text{and}\qquad
B_{j,k}\sim_R B'_{j,k}.
$$
The multiplication congruence field in \autoref{def:ring-stability-certificate} therefore gives the summand comparison
$$
A_{i,j}\cdot_R B_{j,k}
\sim_R
A'_{i,j}\cdot_R B'_{j,k}
$$
for every inner index $j$.

The shared inner dimension supplies the same finite enumeration of $\mathsf{Fin}(m)$ on both sides. Hence these summand comparisons form a same-length finite-spine classifier witness between
$$
j\mapsto A_{i,j}\cdot_R B_{j,k}
\qquad\text{and}\qquad
j\mapsto A'_{i,j}\cdot_R B'_{j,k}.
$$
Applying \autoref{thm:finite-additive-fold-congruence-coefficient-lists} to that finite-spine witness yields
$$
\Sigma_{\mathsf{Fin}(m)}(j\mapsto A_{i,j}\cdot_R B_{j,k})
\sim_R
\Sigma_{\mathsf{Fin}(m)}(j\mapsto A'_{i,j}\cdot_R B'_{j,k}).
$$
By the displayed definition of matrix multiplication, this is exactly
$$
(A\cdot B)_{i,k}\sim_R(A'\cdot B')_{i,k}.
$$
Since $i$ and $k$ were arbitrary, \autoref{def:matrix-classifier-specification} packages the entrywise scalar comparisons as $A\cdot B\sim_{\mathrm{Mat}}A'\cdot B'$. The dimension and scalar-source ledgers are unchanged and are recorded by \autoref{def:matrix-ledger-policy}.
\end{proof}

\begin{lemma}[Finite additive fold splits pointwise sums]
\label{lem:finite-additive-fold-splits-pointwise-sums}
Let $R$ be a scalar $\RingUp$ carrier with classifier $\sim_R$, zero endpoint $0_R$, addition $+_R$, and finite additive fold $\mathsf{fold}_{+,R}$ from \autoref{def:finite-additive-scalar-fold}. Assume the scalar classifier equivalence fields and the additive abelian-group, additive congruence, and zero rows of \autoref{def:ring-stability-certificate}. For same-indexed finite carried scalar spines $xs$ and $ys$, write $xs+_R ys$ for the pointwise sum spine. Then
$$
\mathsf{fold}_{+,R}(xs+_R ys)
\sim_R
\mathsf{fold}_{+,R}(xs)+_R\mathsf{fold}_{+,R}(ys).
$$
\end{lemma}
\begin{proof}
Induct on the common finite spine carrying $xs$ and $ys$. In the nil case, all three folds reduce by \autoref{def:finite-additive-scalar-fold} to $0_R$. The additive zero rows classify $0_R$ with $0_R+_R0_R$, giving the displayed comparison.

In the cons case, write $xs=x::xs'$ and $ys=y::ys'$. The fold equation reduces the left endpoint to
$$
(x+_Ry)+_R\mathsf{fold}_{+,R}(xs'+_Rys').
$$
The induction hypothesis and additive congruence replace the tail fold by $\mathsf{fold}_{+,R}(xs')+_R\mathsf{fold}_{+,R}(ys')$. The additive associativity and commutativity fields in \autoref{def:ring-stability-certificate} rebracket and exchange the middle summands, classifying
$$
(x+_Ry)+_R(\mathsf{fold}_{+,R}(xs')+_R\mathsf{fold}_{+,R}(ys'))
$$
with
$$
(x+_R\mathsf{fold}_{+,R}(xs'))+_R(y+_R\mathsf{fold}_{+,R}(ys')).
$$
By \autoref{def:finite-additive-scalar-fold}, the latter endpoint is $\mathsf{fold}_{+,R}(xs)+_R\mathsf{fold}_{+,R}(ys)$. Classifier transitivity composes the displayed comparisons.
\end{proof}

\begin{theorem}[Matrix multiplication left-distributes over matrix addition]
\label{thm:matrix-multiplication-left-distributes-over-addition}
Let $R$ be a scalar $\RingUp$ carrier with classifier $\sim_R$, zero endpoint $0_R$, addition $+_R$, multiplication $\cdot_R$, and finite additive fold $\mathsf{fold}_{+,R}$. Write $\Sigma_{\mathsf{Fin}(m)}$ for that fold applied to the finite spine enumerating $\mathsf{Fin}(m)$. Let $n,m,p:\NatUp$, and consider the $\MatrixUp$ carriers of \autoref{def:matrix-carrier} with pointwise classifier from \autoref{def:matrix-classifier-specification}. Let
$$
A:\mathsf{Fin}(n)\times\mathsf{Fin}(m)\to R,
\qquad
B,C:\mathsf{Fin}(m)\times\mathsf{Fin}(p)\to R
$$
be carried matrices. Assume matrix addition is the pointwise operation
$$
(X+Y)_{u,w}:=X_{u,w}+_RY_{u,w}
$$
using the addition closure field of \autoref{def:matrix-stability-certificate}. Assume matrix multiplication is the finite-fold operation
$$
(X\cdot Y)_{u,w}:=\Sigma_{\mathsf{Fin}(m)}(v\mapsto X_{u,v}\cdot_RY_{v,w})
$$
for the matching inner finite index carrier, using the multiplication closure field of \autoref{def:matrix-stability-certificate}. Assume the scalar classifier equivalence fields, the scalar ring stability fields of \autoref{def:ring-stability-certificate}, and the fold congruence theorem \autoref{thm:finite-additive-fold-congruence-coefficient-lists}. Then $A\cdot(B+C)$ and $(A\cdot B)+(A\cdot C)$ are carried and
$$
A\cdot(B+C)\sim_{\mathrm{Mat}}(A\cdot B)+(A\cdot C).
$$
\end{theorem}
\begin{proof}
The addition closure item of \autoref{def:matrix-stability-certificate} carries $B+C$. Multiplication closure carries $A\cdot(B+C)$, $A\cdot B$, and $A\cdot C$, and addition closure carries $(A\cdot B)+(A\cdot C)$. By \autoref{def:matrix-classifier-specification}, it remains to prove the scalar classifier comparison at each entry.

Fix $i:\mathsf{Fin}(n)$ and $k:\mathsf{Fin}(p)$. Over the shared finite enumeration of $\mathsf{Fin}(m)$, define carried scalar spines
$$
xs_{i,k}(v):=A_{i,v}\cdot_RB_{v,k},
\qquad
ys_{i,k}(v):=A_{i,v}\cdot_RC_{v,k},
$$
and
$$
zs_{i,k}(v):=A_{i,v}\cdot_R(B_{v,k}+_RC_{v,k}).
$$
The pointwise definition of matrix addition gives
$$
(B+C)_{v,k}\sim_RB_{v,k}+_RC_{v,k}
$$
for every $v:\mathsf{Fin}(m)$. Scalar multiplication congruence and scalar left distributivity from \autoref{def:ring-stability-certificate} therefore give
$$
A_{i,v}\cdot_R(B+C)_{v,k}
\sim_R
(A_{i,v}\cdot_RB_{v,k})+_R(A_{i,v}\cdot_RC_{v,k})
$$
for every $v$. These comparisons form a same-length finite-spine classifier witness. Applying \autoref{thm:finite-additive-fold-congruence-coefficient-lists} to the shared inner enumeration yields
$$
\Sigma_{\mathsf{Fin}(m)}(v\mapsto A_{i,v}\cdot_R(B+C)_{v,k})
\sim_R
\Sigma_{\mathsf{Fin}(m)}(v\mapsto xs_{i,k}(v)+_Rys_{i,k}(v)).
$$

By \autoref{lem:finite-additive-fold-splits-pointwise-sums},
$$
\Sigma_{\mathsf{Fin}(m)}(v\mapsto xs_{i,k}(v)+_Rys_{i,k}(v))
\sim_R
\Sigma_{\mathsf{Fin}(m)}(v\mapsto xs_{i,k}(v))
+_R
\Sigma_{\mathsf{Fin}(m)}(v\mapsto ys_{i,k}(v)).
$$
The displayed definitions of matrix multiplication and matrix addition identify the leftmost endpoint with $(A\cdot(B+C))_{i,k}$ and the rightmost endpoint with $((A\cdot B)+(A\cdot C))_{i,k}$. Classifier transitivity composes the two fold comparisons with these operation readbacks, so
$$
(A\cdot(B+C))_{i,k}
\sim_R
((A\cdot B)+(A\cdot C))_{i,k}.
$$
Since $i$ and $k$ were arbitrary, \autoref{def:matrix-classifier-specification} packages the pointwise scalar comparisons as the displayed matrix classifier. The dimension and scalar-source witnesses are unchanged and are recorded by \autoref{def:matrix-ledger-policy}.
\end{proof}

\begin{lemma}[Finite additive fold separates pointwise scalar sums]
\label{lem:finite-additive-fold-separates-pointwise-scalar-sums}
Let $R$ be a scalar $\RingUp$ carrier with classifier $\sim_R$, zero endpoint $0_R$, addition $+_R$, and finite additive fold $\mathsf{fold}_{+,R}$ from \autoref{def:finite-additive-scalar-fold}. Assume the additive abelian-group laws and additive congruence fields retained in \autoref{def:ring-stability-certificate}. For any two finite carried scalar spines $xs$ and $ys$ over the same finite enumeration, write $xs+_Rys$ for the spine obtained by adding matching entries. Then
$$
\mathsf{fold}_{+,R}(xs+_Rys)\sim_R
\mathsf{fold}_{+,R}(xs)+_R\mathsf{fold}_{+,R}(ys).
$$
\end{lemma}
\begin{proof}
Induct on the shared finite enumeration. In the nil case, all three spines are $\mathsf{nil}$. The left endpoint reduces by \autoref{def:finite-additive-scalar-fold} to $0_R$, while the right endpoint reduces to $0_R+_R0_R$. The additive left-zero law in \autoref{def:ring-stability-certificate}, with classifier symmetry from the additive abelian-group laws, gives $0_R\sim_R0_R+_R0_R$.

In the cons case, write $xs=\mathsf{cons}(a,xs')$ and $ys=\mathsf{cons}(b,ys')$. The induction hypothesis gives
$$
\mathsf{fold}_{+,R}(xs'+_Rys')\sim_R
\mathsf{fold}_{+,R}(xs')+_R\mathsf{fold}_{+,R}(ys').
$$
Source reflexivity on $a+_Rb$ and additive congruence transport this comparison to
$$
(a+_Rb)+_R\mathsf{fold}_{+,R}(xs'+_Rys')
\sim_R
(a+_Rb)+_R(\mathsf{fold}_{+,R}(xs')+_R\mathsf{fold}_{+,R}(ys')).
$$
Apply \autoref{thm:abgroup-middle-four-interchange}, read for the additive abelian-group operation of the scalar ring, to classify the right endpoint with
$$
(a+_R\mathsf{fold}_{+,R}(xs'))+_R(b+_R\mathsf{fold}_{+,R}(ys')).
$$
The cons clauses of \autoref{def:finite-additive-scalar-fold} identify this last endpoint with
$$
\mathsf{fold}_{+,R}(xs)+_R\mathsf{fold}_{+,R}(ys),
$$
and identify the first endpoint with $\mathsf{fold}_{+,R}(xs+_Rys)$. Classifier transitivity gives the cons case.
\end{proof}

\begin{theorem}[Matrix multiplication right-distributes over matrix addition]
\label{thm:matrix-multiplication-right-distributes-over-addition}
Let $R$ be a scalar $\RingUp$ carrier with classifier $\sim_R$, zero endpoint $0_R$, addition $+_R$, multiplication $\cdot_R$, and finite additive folds over finite index spines as in \autoref{def:finite-additive-scalar-fold}. Let $n,m,p:\NatUp$, and consider the $\MatrixUp$ carriers of \autoref{def:matrix-carrier} with pointwise classifier from \autoref{def:matrix-classifier-specification}. Let
$$
A,B:\mathsf{Fin}(n)\times\mathsf{Fin}(m)\to R,
\qquad
C:\mathsf{Fin}(m)\times\mathsf{Fin}(p)\to R
$$
be carried matrices. Assume matrix addition is the pointwise operation
$$
(X+Y)_{u,v}:=X_{u,v}+_RY_{u,v}
$$
using the addition closure field of \autoref{def:matrix-stability-certificate}, and assume matrix multiplication is the finite-fold operation
$$
(X\cdot Y)_{u,w}:=\Sigma_{\mathsf{Fin}(m)}(j\mapsto X_{u,j}\cdot_RY_{j,w})
$$
for the shared inner finite index carrier, using the multiplication closure field of \autoref{def:matrix-stability-certificate}. Assume the scalar ring stability fields of \autoref{def:ring-stability-certificate}: additive abelian-group laws, addition and multiplication congruence, and right distributivity. Then the displayed matrices are carried and
$$
(A+B)\cdot C\sim_{\mathrm{Mat}}(A\cdot C)+(B\cdot C).
$$
\end{theorem}
\begin{proof}
The addition closure item of \autoref{def:matrix-stability-certificate} carries $A+B$. Multiplication closure carries $(A+B)\cdot C$, $A\cdot C$, and $B\cdot C$, and addition closure then carries $(A\cdot C)+(B\cdot C)$.

By \autoref{def:matrix-classifier-specification}, it remains to compare corresponding scalar entries. Fix $i:\mathsf{Fin}(n)$ and $k:\mathsf{Fin}(p)$. Over the finite spine enumerating $\mathsf{Fin}(m)$, define
$$
xs(j):=A_{i,j}\cdot_RC_{j,k},
\qquad
ys(j):=B_{i,j}\cdot_RC_{j,k}.
$$
The pointwise matrix-addition definition and scalar right distributivity give, for each $j:\mathsf{Fin}(m)$,
$$
(A+B)_{i,j}\cdot_RC_{j,k}
\sim_R
(A_{i,j}\cdot_RC_{j,k})+_R(B_{i,j}\cdot_RC_{j,k}).
$$
These pointwise comparisons form a finite-spine classifier witness. Applying \autoref{thm:finite-additive-fold-respects-scalar-classifier} gives
$$
\Sigma_{\mathsf{Fin}(m)}(j\mapsto (A+B)_{i,j}\cdot_RC_{j,k})
\sim_R
\Sigma_{\mathsf{Fin}(m)}(j\mapsto xs(j)+_Rys(j)).
$$
By \autoref{lem:finite-additive-fold-separates-pointwise-scalar-sums}, the right endpoint is classified with
$$
\Sigma_{\mathsf{Fin}(m)}(j\mapsto xs(j))+_R
\Sigma_{\mathsf{Fin}(m)}(j\mapsto ys(j)).
$$
Unfolding the displayed matrix-multiplication operation identifies these two folded summands with $(A\cdot C)_{i,k}$ and $(B\cdot C)_{i,k}$, and the pointwise matrix-addition operation identifies their scalar sum with $((A\cdot C)+(B\cdot C))_{i,k}$. Classifier transitivity therefore gives
$$
((A+B)\cdot C)_{i,k}\sim_R((A\cdot C)+(B\cdot C))_{i,k}.
$$
Since $i$ and $k$ were arbitrary, \autoref{def:matrix-classifier-specification} packages these entrywise scalar comparisons as
$$
(A+B)\cdot C\sim_{\mathrm{Mat}}(A\cdot C)+(B\cdot C).
$$
The dimensions and shared scalar source are unchanged throughout; \autoref{def:matrix-ledger-policy} records the common row, contraction, and column witnesses.
\end{proof}

\begin{lemma}[Rectangular scalar fold reindexing]
\label{lem:rectangular-scalar-fold-reindexing}
Let $R$ be a scalar $\RingUp$ carrier with classifier $\sim_R$, zero endpoint $0_R$, addition $+_R$, and finite additive fold $\mathsf{fold}_{+,R}$ from \autoref{def:finite-additive-scalar-fold}. Let $J$ and $K$ be finite index carriers with finite fold sums $\Sigma_J$ and $\Sigma_K$ obtained from their carried index spines. Assume the scalar classifier equivalence fields, the additive abelian-group laws, additive congruence, and zero rows listed in \autoref{def:ring-stability-certificate}. For every pointwise carried scalar family $s:J\times K\to R$, the two iterated scalar folds satisfy
$$
\Sigma_J(j\mapsto\Sigma_K(k\mapsto s(j,k)))\sim_R
\Sigma_K(k\mapsto\Sigma_J(j\mapsto s(j,k))).
$$
\end{lemma}
\begin{proof}
Induct on the finite spine enumerating $J$. In the nil case, the left endpoint is the fold of the empty spine and hence reduces to $0_R$. The right endpoint is the $K$-fold of the constant family whose inner $J$-fold is $0_R$; an induction on the finite spine enumerating $K$, using the additive zero rows and additive congruence from \autoref{def:ring-stability-certificate}, classifies this fold with $0_R$.

In the cons case, write the outer index spine as $j_0::J'$. The left iterated fold unfolds to
$$
\Sigma_K(k\mapsto s(j_0,k))+_R
\Sigma_{J'}(j\mapsto\Sigma_K(k\mapsto s(j,k))).
$$
The induction hypothesis classifies the tail term with
$$
\Sigma_K(k\mapsto\Sigma_{J'}(j\mapsto s(j,k))).
$$
Additive congruence therefore transports the left endpoint to
$$
\Sigma_K(k\mapsto s(j_0,k))+_R
\Sigma_K(k\mapsto\Sigma_{J'}(j\mapsto s(j,k))).
$$
On the right endpoint, the inner $J$-fold at each $k$ unfolds as
$$
s(j_0,k)+_R\Sigma_{J'}(j\mapsto s(j,k)).
$$
Applying \autoref{lem:finite-additive-fold-splits-pointwise-sums} to the shared $K$-indexed spine classifies the outer fold with the same sum of the selected row fold and the transposed tail fold. Classifier symmetry and transitivity compose these comparisons, giving the cons case.
\end{proof}

\begin{theorem}[Matrix product associativity from scalar folds]
\label{thm:matrix-product-associativity-from-scalar-folds}
Let $R$ be a scalar $\RingUp$ carrier with classifier $\sim_R$, zero endpoint $0_R$, addition $+_R$, multiplication $\cdot_R$, and finite additive folds over finite index spines as in \autoref{def:finite-additive-scalar-fold}. Let $n,m,p,q:\NatUp$, and consider the $\MatrixUp$ carriers of \autoref{def:matrix-carrier} with pointwise classifier from \autoref{def:matrix-classifier-specification}. Let
$$
A:\mathsf{Fin}(n)\times\mathsf{Fin}(m)\to R,
\qquad
B:\mathsf{Fin}(m)\times\mathsf{Fin}(p)\to R,
\qquad
C:\mathsf{Fin}(p)\times\mathsf{Fin}(q)\to R
$$
be carried matrices. Assume matrix multiplication is the finite scalar-fold operation
$$
(X\cdot_{\mathrm{Mat}}Y)_{u,w}:=
\Sigma_{\mathsf{Fin}(s)}(v\mapsto X_{u,v}\cdot_RY_{v,w})
$$
for every matching inner dimension $s$, using the multiplication closure row of \autoref{def:matrix-stability-certificate}. Assume the scalar classifier equivalence fields, multiplication congruence, multiplicative associativity, left and right distributivity, and additive laws retained in \autoref{def:ring-stability-certificate}. Then the two matched products are carried and
$$
(A\cdot_{\mathrm{Mat}}B)\cdot_{\mathrm{Mat}}C
\sim_{\mathrm{Mat}}
A\cdot_{\mathrm{Mat}}(B\cdot_{\mathrm{Mat}}C).
$$
\end{theorem}
\begin{proof}
The multiplication closure row of \autoref{def:matrix-stability-certificate} carries $A\cdot_{\mathrm{Mat}}B$ and $B\cdot_{\mathrm{Mat}}C$, and then carries the two outer products. By \autoref{def:matrix-classifier-specification}, it remains to compare corresponding scalar entries.

Fix $i:\mathsf{Fin}(n)$ and $l:\mathsf{Fin}(q)$. For $a:\mathsf{Fin}(m)$ and $b:\mathsf{Fin}(p)$, set
$$
t(a,b):=A_{i,a}\cdot_R(B_{a,b}\cdot_RC_{b,l}).
$$
Expanding the left-associated product gives
$$
((A\cdot_{\mathrm{Mat}}B)\cdot_{\mathrm{Mat}}C)_{i,l}
\sim_R
\Sigma_{b:\mathsf{Fin}(p)}((A\cdot_{\mathrm{Mat}}B)_{i,b}\cdot_RC_{b,l}).
$$
For each fixed $b$, the inner product definition and the right-multiplication half of \autoref{lem:scalar-product-distributes-across-finite-additive-folds} classify the summand with
$$
\Sigma_{a:\mathsf{Fin}(m)}((A_{i,a}\cdot_RB_{a,b})\cdot_RC_{b,l}).
$$
Scalar multiplicative associativity from \autoref{def:ring-stability-certificate} gives the pointwise comparison
$$
(A_{i,a}\cdot_RB_{a,b})\cdot_RC_{b,l}
\sim_R
A_{i,a}\cdot_R(B_{a,b}\cdot_RC_{b,l}).
$$
Using \autoref{thm:finite-additive-fold-respects-scalar-classifier} for the inner and outer finite spines yields
$$
((A\cdot_{\mathrm{Mat}}B)\cdot_{\mathrm{Mat}}C)_{i,l}
\sim_R
\Sigma_{b:\mathsf{Fin}(p)}\Sigma_{a:\mathsf{Fin}(m)}t(a,b).
$$

Expanding the right-associated product gives
$$
(A\cdot_{\mathrm{Mat}}(B\cdot_{\mathrm{Mat}}C))_{i,l}
\sim_R
\Sigma_{a:\mathsf{Fin}(m)}(A_{i,a}\cdot_R(B\cdot_{\mathrm{Mat}}C)_{a,l}).
$$
For each fixed $a$, the inner product definition and the left-multiplication half of \autoref{lem:scalar-product-distributes-across-finite-additive-folds} classify the summand with
$$
\Sigma_{b:\mathsf{Fin}(p)}A_{i,a}\cdot_R(B_{a,b}\cdot_RC_{b,l}).
$$
Again \autoref{thm:finite-additive-fold-respects-scalar-classifier} lifts the pointwise comparisons through the finite folds, so
$$
(A\cdot_{\mathrm{Mat}}(B\cdot_{\mathrm{Mat}}C))_{i,l}
\sim_R
\Sigma_{a:\mathsf{Fin}(m)}\Sigma_{b:\mathsf{Fin}(p)}t(a,b).
$$

Apply \autoref{lem:rectangular-scalar-fold-reindexing} to the family $s(b,a):=t(a,b)$ over $\mathsf{Fin}(p)\times\mathsf{Fin}(m)$. It gives
$$
\Sigma_{b:\mathsf{Fin}(p)}\Sigma_{a:\mathsf{Fin}(m)}t(a,b)
\sim_R
\Sigma_{a:\mathsf{Fin}(m)}\Sigma_{b:\mathsf{Fin}(p)}t(a,b).
$$
Classifier transitivity composes the left expansion, this rectangular fold reindexing, and the symmetry of the right expansion. Thus every entry of $(A\cdot_{\mathrm{Mat}}B)\cdot_{\mathrm{Mat}}C$ is scalar-classified with the corresponding entry of $A\cdot_{\mathrm{Mat}}(B\cdot_{\mathrm{Mat}}C)$. The pointwise classifier of \autoref{def:matrix-classifier-specification} packages these entrywise comparisons as the displayed matrix classifier, and \autoref{def:matrix-ledger-policy} records that the dimension witnesses are unchanged.
\end{proof}
