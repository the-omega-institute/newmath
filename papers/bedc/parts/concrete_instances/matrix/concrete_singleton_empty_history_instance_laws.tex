\begin{theorem}[Matrix singleton classifier append split empty]
\label{thm:matrix-singleton-classifier-append-split-empty}
For $M,N,h:\mathsf{BHist}$,
$$
\rho_{\emp}(\mathsf{append}(M,N),h)
\quad\Longleftrightarrow\quad
M\hsame\emp\land N\hsame\emp\land A_{\emp}(h).
$$
\end{theorem}
\begin{proof}
The forward direction reads the left carrier component of
$\rho_{\emp}(\mathsf{append}(M,N),h)$ as
$\mathsf{append}(M,N)\hsame\emp$. Empty-result inversion for append splits this
endpoint witness into $M\hsame\emp$ and $N\hsame\emp$, while the right carrier
component supplies $A_{\emp}(h)$.

Conversely, $M\hsame\emp$ and $N\hsame\emp$ reconstruct the singleton carrier
witness for $\mathsf{append}(M,N)$. Together with $A_{\emp}(h)$, transitivity of
$\hsame$ through $\emp$ gives the classifier comparison.
\end{proof}
\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonClassifier\_append\_split\_empty\_iff}

\begin{theorem}[Matrix singleton add-mul classifier iff]
\label{thm:matrix-singleton-add-mul-classifier-iff}
For $M,N:\mathsf{BHist}$, the singleton matrix classifier relates the displayed
addition and multiplication results exactly when both inputs are singleton
empty endpoints:
$$
\rho_{\emp}(M\oplus N,M\odot N)
\quad\Longleftrightarrow\quad
M\hsame\emp\land N\hsame\emp.
$$
\end{theorem}
\begin{proof}
Both displayed operations are the same BEDC append spine
$\mathsf{append}(M,N)$. In the forward direction, the left carrier component
of $\rho_{\emp}(M\oplus N,M\odot N)$ says that this append spine is
$\hsame$ to $\emp$. Empty-result inversion for append splits it into the two
endpoint witnesses $M\hsame\emp$ and $N\hsame\emp$.

Conversely, the two endpoint witnesses reconstruct the singleton carrier for
$\mathsf{append}(M,N)$. Since $M\oplus N$ and $M\odot N$ unfold to that same
append spine, reflexivity of $\hsame$ supplies the classifier relation between
the displayed operation results.
\end{proof}
\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonAddMul\_classifier\_iff}

\begin{theorem}[Matrix singleton add-mul continuation result iff]
\label{thm:matrix-singleton-addmul-continuation-result-iff}
For singleton-carrier inputs $M$ and $N$, a continuation from the displayed
addition result to the displayed multiplication result has result $R$ exactly
when $R$ is itself a singleton carrier:
$$
\ContR(M\oplus N,M\odot N,R)
\quad\Longleftrightarrow\quad
A_{\emp}(R).
$$
\end{theorem}
\begin{proof}
The two displayed operations both unfold to the same append spine
$\mathsf{append}(M,N)$. The singleton witnesses for $M$ and $N$ collapse that
spine to $\emp$. The forward direction compares the displayed continuation
with the right-unit continuation at $\emp$; the reverse direction transports
the result carrier back to that same empty continuation.
\end{proof}
\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonAddMul\_continuation\_result\_iff}

\begin{theorem}[Matrix singleton add-mul continuation empty-result factors]
\label{thm:matrix-singleton-addmul-continuation-empty-result-factors}
For histories $M,N,R:\mathsf{BHist}$, if
$\ContR(M\oplus N,M\odot N,R)$, then the displayed continuation result is
empty exactly when both matrix inputs are empty:
$$
R\hsame\emp
\quad\Longleftrightarrow\quad
M\hsame\emp\land N\hsame\emp.
$$
\end{theorem}
\begin{proof}
For the forward direction, transport the displayed continuation along
$R\hsame\emp$ and invert the empty continuation result. The source factor is
$M\oplus N$, which unfolds to $\mathsf{append}(M,N)$, so append empty-result
inversion gives $M\hsame\emp$ and $N\hsame\emp$.

Conversely, the two input endpoint witnesses collapse both displayed
operations to the empty history. Continuation determinacy compares the given
continuation with the BEDC right-unit continuation at $\emp$, identifying the
result with $\emp$.
\end{proof}
\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonAddMul\_continuation\_empty\_result\_factors\_iff}

\begin{theorem}[Matrix singleton multiplication continuation result carrier]
\label{thm:matrix-singleton-mul-continuation-result-carrier}
For histories $M,N,h,r:\mathsf{BHist}$, any continuation from $h$ through the
singleton multiplication endpoint has singleton result exactly when the source
and both multiplication inputs are singleton carriers:
$$
\ContR(h,M\odot N,r)
\Longrightarrow
\bigl(A_{\emp}(r)\Longleftrightarrow
A_{\emp}(h)\land A_{\emp}(M)\land A_{\emp}(N)\bigr).
$$
\end{theorem}
\begin{proof}
In the forward direction, transport the displayed continuation result along
$r\hsame\emp$ and invert the empty continuation result. The target endpoint is
$M\odot N$, which unfolds to $\mathsf{append}(M,N)$, so append empty-result
inversion supplies the two multiplication input carriers. The same continuation
inversion supplies the source carrier.

Conversely, the source carrier and the two multiplication input carriers
collapse the target endpoint to $\emp$. Continuation respect for $\hsame$
compares the displayed continuation with the right-unit continuation at
$\emp$, giving the singleton result carrier.
\end{proof}
\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonMulContinuation\_result\_carrier\_iff}

\begin{definition}[Matrix singleton entry projection package]
\label{def:matrix-singleton-entry-projection-package}
The matrix singleton entry projection package fixes the matrix carrier to
$A_{\emp}$, fixes the matrix classifier to $\rho_{\emp}$, and reads the
displayed row and column dimensions from the $\NatUp$ rows already carried by
the singleton matrix instance. Its entry projection operation sends every
carried unary row to the $\mathsf{BHist}$ endpoint $\emp$.
\end{definition}

\begin{theorem}[Matrix singleton entry projection laws]
\label{thm:matrix-singleton-entry-projection-laws}
In the matrix singleton entry projection package, every projected entry of a
carried singleton matrix is carried by $A_{\emp}$. Projected entries are stable
under $\rho_{\emp}$ and under the displayed $\hsame$ ledger. Moreover, the
projected entries of the displayed singleton addition result and of the
displayed singleton multiplication result are classified with $\emp$.
\end{theorem}
\begin{proof}
Unfold the package. The carrier is $A_{\emp}$, the classifier is
$\rho_{\emp}$, and every entry projection is the constant endpoint $\emp$ on
the carried unary rows. Hence each projected entry is immediately an
$A_{\emp}$ entry and is related to the singleton endpoint by $\rho_{\emp}$.

For stability, use the definition of $\rho_{\emp}$ together with the displayed
$\hsame$ ledger. Transporting a carried singleton matrix along $\rho_{\emp}$
keeps the projected endpoint equal to $\emp$, so the projected entry remains in
the same singleton classifier row.

For the displayed addition and multiplication results, apply
\autoref{thm:matrix-singleton-addmul-continuation-result-iff},
\autoref{thm:matrix-singleton-addmul-continuation-empty-result-factors}, and
\autoref{thm:matrix-singleton-mul-continuation-result-carrier}. The local
continuation laws identify the result endpoints with $A_{\emp}$, and the
constant entry projection classifies their projected entries with $\emp$. No
function-space carrier or $\mathsf{Fin}$-indexed matrix carrier is introduced.
\end{proof}

\begin{theorem}[Matrix singleton add-mul visible-left continuation result nonempty]
\label{thm:matrix-singleton-addmul-visible-left-continuation-result-nonempty}
For any histories $m,N,R$, if the shared left input of the displayed singleton
addition and multiplication has a visible head, then a continuation from the
addition endpoint to the multiplication endpoint cannot have empty result:
$$
\ContR(\Ezero(m)\oplus N,\Ezero(m)\odot N,R)\land R\hsame\emp
\Longrightarrow\bot,
$$
and the same statement holds with $\Eone(m)$ in place of $\Ezero(m)$.
\end{theorem}
\begin{proof}
Apply the continuation empty-result factor theorem for displayed singleton
addition and multiplication. An empty result forces the left input factor to
be empty. This contradicts constructor separation for $\Ezero(m)$ in the first
case and for $\Eone(m)$ in the second case.
\end{proof}

\begin{theorem}[Matrix singleton classifier visible right append exclusion]
\label{thm:matrix-singleton-classifier-append-visible-right-absurd}

cannot target an append whose right factor has a visible head:
$$
\rho_{\emp}(h,\mathsf{append}(p,\mathsf{e0}(q)))\to\bot
\qquad\text{and}\qquad
\rho_{\emp}(h,\mathsf{append}(p,\mathsf{e1}(q)))\to\bot.

\begin{proof}
Unfold the singleton matrix classifier. A classifier witness supplies the
target carrier endpoint
$\mathsf{append}(p,\mathsf{e0}(q))\hsame\emp$ in the first case, or
$\mathsf{append}(p,\mathsf{e1}(q))\hsame\emp$ in the second case. Empty-result
inversion for append forces the visible right factor itself to be $\emp$,
contradicting the disjointness of each visible constructor from $\emp$.
\end{proof}

The forward implication splits the source carrier of the classifier twice by

\end{theorem}
\begin{proof}
In the forward direction, the two carrier fields say that
$\mathsf{append}(R,P)$ and $\mathsf{append}(R,Q)$ are empty. Empty-result

$\rho_{\emp}(P,Q)$.

Conversely, unpack $A_{\emp}(R)$ and $\rho_{\emp}(P,Q)$. The shared left
endpoint and each singleton classified endpoint rebuild both empty append
endpoints. Transitivity of $\hsame$ through $\emp$ then supplies the classifier
comparison between the two appended histories.

\begin{theorem}[Singleton empty-history matrix laws]
\label{thm:singleton-empty-history-matrix-laws}
For the singleton instance of
Definition~\ref{def:singleton-empty-history-matrix-instance}, the displayed
matrix carrier and operations satisfy the concrete $\MatrixUp$ law package:
\begin{enumerate}
  \item $0_M$ and $I_M$ are carried by $A_{\emp}$.
  \item If $A_{\emp}(M)$ and $A_{\emp}(N)$, then
  $A_{\emp}(M\oplus N)$ and $A_{\emp}(M\odot N)$.
  \item $0_M$ is a left and right unit for $\oplus$ under $\rho_{\emp}$.
  \item $I_M$ is a left and right identity for $\odot$ under $\rho_{\emp}$.
  \item $\oplus$ and $\odot$ are associative under $\rho_{\emp}$.
  \item $\rho_{\emp}$ is reflexive, symmetric, and transitive on carried
  matrix codes, and it is respected by $\oplus$ and $\odot$.
\end{enumerate}
\end{theorem}
\begin{proof}
The carrier endpoint is supplied by the singleton source witness from
Definition~\ref{def:list-singleton-history-source} at $a_0=\emp$. The row and
column witnesses use the same empty unary history, so the dimension ledger is
closed at the singleton endpoint.

For operation closure, carried inputs are classified as singleton endpoints.
The displayed operations are either $\emp$ or $\mathsf{append}$ applied to
singleton endpoints. The BEDC continuation unit laws cited in
Theorem~\ref{thm:concrete-history-monoid-laws} identify the resulting append
endpoint with $\emp$, so the singleton source again carries the result.

The zero laws for $\oplus$ and the identity laws for $\odot$ are the same
unit calculation. On the left and on the right, the operation compares
$\mathsf{append}(\emp,M)$ or $\mathsf{append}(M,\emp)$ with $M$ under
$\rho_{\emp}$, and the continuation unit laws provide the $\hsame$ component.
The endpoint components come from $A_{\emp}(M)$ and $A_{\emp}(\emp)$.

Associativity of $\oplus$ and $\odot$ uses the continuation associativity law
already used by the singleton monoid, ring, and vector-space chapters. Both
operation symbols are represented by $\mathsf{append}$, so the two
parenthesized products share a single BEDC append spine up to $\hsame$.

For classifier congruence, unpack the two $\rho_{\emp}$ hypotheses. Closure
keeps both composite endpoints in $A_{\emp}$, and the relation components are
combined by reflexivity, symmetry, and transitivity of $\hsame$ after the
append unit and associativity reductions. This gives the required
$\rho_{\emp}$ comparison for both $\oplus$ and $\odot$.
\end{proof}
\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonEmptyHistory\_laws}

\begin{theorem}[Singleton empty-history matrix endpoint collapse]
\label{thm:singleton-empty-history-matrix-endpoint-collapse}
With the notation of
Definition~\ref{def:singleton-empty-history-matrix-instance}, if
$A_{\emp}(M)$ and $A_{\emp}(N)$, then
\begin{enumerate}
  \item $\rho_{\emp}(M\oplus N,\emp)$.
  \item $\rho_{\emp}(M\odot N,\emp)$.
  \item $\rho_{\emp}(M\oplus N,M\odot N)$.
\end{enumerate}
\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:singleton-empty-history-matrix-laws} to the carried
inputs $M$ and $N$. Its closure clause puts both $M\oplus N$ and $M\odot N$
back in $A_{\emp}$, while the singleton carrier witnesses identify $M$ and
$N$ with the endpoint $\emp$.

By the displayed definitions, both operations are
$\mathsf{append}(M,N)$. Replace the carried inputs by their singleton endpoint
witnesses, then use the left and right continuation unit laws cited in
Theorem~\ref{thm:concrete-history-monoid-laws} to reduce the resulting append
history to $\emp$.

For $\rho_{\emp}(M\oplus N,\emp)$ and
$\rho_{\emp}(M\odot N,\emp)$, combine the closure witnesses with the
corresponding $\hsame$ endpoint calculation. Symmetry is used when the unit
witness is oriented from $\emp$ to the displayed operation rather than from
the displayed operation to $\emp$.

For $\rho_{\emp}(M\oplus N,M\odot N)$, compose the first endpoint comparison
with the symmetry of the second by transitivity of $\hsame$. The two carrier
witnesses are the closure witnesses already supplied by the law package, so
the classifier definition gives the required comparison.
\end{proof}

\begin{theorem}[Singleton empty-history matrix endpoint exactness]
\label{thm:singleton-empty-history-matrix-endpoint-exactness}
For the singleton instance of
Definition~\ref{def:singleton-empty-history-matrix-instance}, if
$A_{\emp}(M)$ and $A_{\emp}(N)$, then all displayed matrix endpoints are
classified by the singleton empty history:
\begin{enumerate}
  \item $\rho_{\emp}(M,\emp)$.
  \item $\rho_{\emp}(0_M,\emp)$.
  \item $\rho_{\emp}(I_M,\emp)$.
  \item $\rho_{\emp}(M\oplus N,\emp)$.
  \item $\rho_{\emp}(M\odot N,\emp)$.
\end{enumerate}
\end{theorem}
\begin{proof}
Unfold the singleton carrier $A_{\emp}$ and the classifier
$\rho_{\emp}$. The hypotheses say that $M$ and $N$ are carried by the
singleton endpoint, so the first classification is obtained by pairing the
two carrier witnesses with the corresponding singleton endpoint witness in
$\hsame$.

of $0_M$, $I_M$, $M\oplus N$, and $M\odot N$ in $A_{\emp}$. These closure
witnesses supply the carrier components required by the classifier for the
four remaining endpoints.

For the $\hsame$ components, compose the singleton endpoint witnesses with
symmetry and transitivity of $\hsame$. Thus each displayed endpoint is
identified with $\emp$, and repacking the carrier witnesses with the
$\hsame$ witness gives the five asserted instances of $\rho_{\emp}$.
\end{proof}
