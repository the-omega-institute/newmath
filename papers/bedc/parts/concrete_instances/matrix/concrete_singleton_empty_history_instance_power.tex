\section{Concrete singleton empty-history matrix instance}
\label{sec:matrix-concrete-singleton-instance}

\begin{definition}[Concrete singleton empty-history matrix instance]
\label{def:singleton-empty-history-matrix-instance}
Let $A_{\emp}$ be the singleton history source of
Definition~\ref{def:list-singleton-history-source} at $a_0=\emp$. Use the
classifier
$$
\rho_{\emp}(h,k)\Longleftrightarrow
A_{\emp}(h)\land A_{\emp}(k)\land\hsame(h,k).
$$
The row-dimension witness and column-dimension witness are both represented by
the empty unary history $\emp$. The scalar side is the singleton empty-history
ring of Definition~\ref{def:singleton-empty-history-ring-instance}, and the
row and column vector layers are read through the singleton empty-history
vector-space instance of
Definition~\ref{def:singleton-empty-history-vecspace-instance}.

The matrix carrier is the same singleton source $A_{\emp}$ on
$\mathsf{BHist}$ codes. Define the displayed matrix operations by
$$
0_M:=\emp,\qquad I_M:=\emp,\qquad
M\oplus N:=\mathsf{append}(M,N),\qquad
M\odot N:=\mathsf{append}(M,N).
$$
The certificate fit has ten fields:
\begin{enumerate}
  \item Traditional target: the terminal singleton matrix algebra.
  \item BEDC source: $\mathsf{BHist}$ together with the singleton ring and
  $\VecSpaceUp$ layers.
  \item Carrier choice: $A_{\emp}$.
  \item Classifier choice: $\rho_{\emp}$.
  \item Stability obligations: closure, zero laws, identity laws,
  associativity, and congruence for the displayed operations.
  \item Ledger: the singleton endpoint together with row and column witnesses
  represented by $\emp$.
  \item Core theorem list:
  Theorem~\ref{thm:singleton-empty-history-matrix-laws}.
  \item Exactness predicate: singleton endpoint classification.
  \item Standard bridge: the terminal matrix object.
  \item Status: $\mathsf{ProofTarget}$.
\end{enumerate}
\end{definition}

\begin{definition}[Matrix singleton power]
\label{def:matrix-singleton-pow}
For histories $M$ and $e$, $\mathsf{MatrixSingletonPow}(M,e)$ is the concrete
power ledger obtained by recursion over the exponent history: the empty
exponent returns $I_M$, a one-headed exponent multiplies the previous ledger
by $M$, and the zero-headed branch returns the singleton zero endpoint.
\end{definition}
\leandef{BEDC.Derived.MatrixUp.MatrixSingletonPow}

\begin{theorem}[Matrix singleton power carrier closure]
\label{thm:matrix-singleton-pow-carrier-closed}
If $M$ is carried by the singleton matrix carrier and $e$ is unary, then the
power ledger is again carried by the singleton matrix carrier:
$$
A_{\emp}(M)\land \Unary(e)
\quad\Longrightarrow\quad
A_{\emp}(\mathsf{MatrixSingletonPow}(M,e)).
$$
\end{theorem}
\begin{proof}
Proceed by induction on the unary exponent. The empty exponent returns
$I_M=\emp$, hence is carried by $A_{\emp}$. In the one-headed case, the
induction hypothesis carries the previous power ledger, while the hypothesis
carries $M$; the displayed multiplication is append, so append empty-result
exactness carries the product ledger.
\end{proof}
\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonPow\_carrier\_closed}

\begin{theorem}[Matrix singleton power nonempty unary input carrier exactness]
\label{thm:matrix-singleton-pow-carrier-nonempty-unary-input}
If the exponent $e$ is unary and not history-same as $\emp$, then the singleton
carrier condition on the power ledger is exactly the singleton carrier
condition on the input matrix:
$$
\Unary(e)\land(\hsame(e,\emp)\to\bot)
\quad\Longrightarrow\quad
\bigl(
A_{\emp}(\mathsf{MatrixSingletonPow}(M,e))
\Longleftrightarrow
A_{\emp}(M)
\bigr).
$$
\end{theorem}
\begin{proof}
The unary, nonempty exponent has a one-headed form. In that branch the power
ledger is the displayed append product of the previous power ledger and $M$.
Append empty-result exactness therefore recovers $A_{\emp}(M)$ from the carried
power ledger. Conversely, carrier closure for singleton powers carries the
power ledger from $A_{\emp}(M)$ and $\Unary(e)$.
\end{proof}
\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonPow\_carrier\_nonempty\_unary\_input\_iff}

\begin{theorem}[Matrix singleton power nonempty unary suffix base carrier]
\label{thm:matrix-singleton-pow-nonempty-unary-suffix-base-carrier}
If $tail$ is unary and not history-same as $\emp$, then a singleton carrier
witness for the power ledger over $\mathsf{append}(e,tail)$ recovers the base
matrix carrier:
$$
\Unary(tail)\land(\hsame(tail,\emp)\to\bot)\land
A_{\emp}(\mathsf{MatrixSingletonPow}(M,\mathsf{append}(e,tail)))
\quad\Longrightarrow\quad
A_{\emp}(M).
$$
\end{theorem}
\begin{proof}
The unary, nonempty suffix cannot be $\emp$ and cannot start with $\Ezero$.
Thus it has the form $\Eone(t)$, so the appended exponent also exposes a
one-headed power branch. In that branch the power ledger is the append product
of the previous power ledger and $M$, and append empty-result inversion
projects the right factor carrier $A_{\emp}(M)$.
\end{proof}
\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonPow\_nonempty\_unary\_suffix\_base\_carrier}

\begin{theorem}[Matrix singleton positive-exponent continuation readback]
\label{thm:matrix-singleton-pow-positive-exponent-cont-readback}
If the exponent $e$ is unary and not history-same as $\emp$, then its
one-headed tail exposes the concrete successor continuation for the power
ledger:
$$
\Unary(e)\land(\hsame(e,\emp)\to\bot)
\quad\Longrightarrow\quad
\exists t.\ \Unary(t)\land
\ContR(\mathsf{MatrixSingletonPow}(M,t),M,\mathsf{MatrixSingletonPow}(M,e)).
$$
\end{theorem}
\begin{proof}
Unary nonempty histories have a one-headed tail. In that branch the recursive
definition of $\mathsf{MatrixSingletonPow}$ is the append product of the
predecessor power ledger with $M$, so the continuation is the displayed append
step.
\end{proof}
\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonPow\_positive\_exponent\_cont\_readback}

\begin{theorem}[Matrix singleton positive-exponent empty continuation base carrier]
\label{thm:matrix-singleton-pow-positive-exponent-empty-continuation-base-carrier}
If \(e\) is unary and not history-same as \(\emp\), and a continuation from
\(\mathsf{MatrixSingletonPow}(M,e)\) has result history-same to \(\emp\), then
the base matrix \(M\) lies in the singleton matrix carrier:
$$
A_{\emp}(M).
$$
\end{theorem}
\begin{proof}
Transport the continuation result to the concrete empty endpoint and invert
the empty-result continuation. This carries the power ledger itself by
\(A_{\emp}\). The nonempty unary power carrier exactness theorem then reads
the carriedness of the base matrix \(M\) back from the carried power ledger.
\end{proof}
\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonPow\_positive\_exponent\_empty\_continuation\_base\_carrier}

\begin{theorem}[Matrix singleton power append positive suffix continuation readback]
\label{thm:matrix-singleton-pow-append-positive-suffix-cont-readback}
If $suffix$ is unary and not history-same as $\emp$, then the power ledger
over $\mathsf{append}(pref,suffix)$ exposes the successor continuation at the
same prefix:
$$
\exists tail.\ \Unary(tail)\land
\ContR(\mathsf{MatrixSingletonPow}(M,\mathsf{append}(pref,tail)),M,
\mathsf{MatrixSingletonPow}(M,\mathsf{append}(pref,suffix))).
$$
\end{theorem}
\begin{proof}
Unary nonempty suffixes have a one-headed tail. After substituting
$suffix=\Eone(tail)$, the appended exponent is itself one-headed with payload
$\mathsf{append}(pref,tail)$. The recursive power ledger therefore unfolds to
the displayed continuation by $M$.
\end{proof}
\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonPow\_append\_positive\_suffix\_cont\_readback}

\begin{theorem}[Matrix singleton power append positive suffix classifier readback]
\label{thm:matrix-singleton-pow-append-positive-suffix-classifier-readback}
If $M$ is carried by the singleton matrix carrier, $pref$ and $suffix$ are
unary, and $suffix$ is not history-same as $\emp$, then the append-positive
suffix readback supplies a unary tail whose successor multiplication classifies
the original appended power ledger:
$$
\exists tail.\ \Unary(tail)\land
\rho_{\emp}\bigl(\mathsf{MatrixSingletonPow}(M,\mathsf{append}(pref,suffix)),
\mathsf{MatrixSingletonMul}(\mathsf{MatrixSingletonPow}(M,\mathsf{append}(pref,tail)),M)\bigr).
$$
\end{theorem}
\begin{proof}
The positive-suffix readback exposes a unary tail and the continuation from
$\mathsf{MatrixSingletonPow}(M,\mathsf{append}(pref,tail))$ by $M$ to the
original appended power ledger. Carrier closure gives the source power carrier,
and append empty-result exactness carries the displayed multiplication endpoint.
The continuation equation gives the classifier comparison.
\end{proof}
\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonPow\_append\_positive\_suffix\_classifier\_readback}

\begin{theorem}[Matrix singleton power successor classifier]
\label{thm:matrix-singleton-pow-succ-classifier}
For histories $M,e:\mathsf{BHist}$, if $M$ is carried by the singleton matrix
carrier and $e$ is unary, then the one-headed power ledger is classified
against the displayed successor multiplication:
$$
A_{\emp}(M)\land\Unary(e)
\quad\Longrightarrow\quad
\rho_{\emp}\bigl(\mathsf{MatrixSingletonPow}(M,\Eone(e)),
\mathsf{MatrixSingletonMul}(\mathsf{MatrixSingletonPow}(M,e),M)\bigr).
$$
\end{theorem}
\begin{proof}
The one-headed power branch unfolds to the displayed multiplication endpoint.
The carrier closure theorem carries the predecessor power ledger, and the
input hypothesis carries $M$. Since multiplication is append, append
empty-result exactness carries the displayed multiplication endpoint, and
reflexivity of $\hsame$ supplies the classifier comparison.
\end{proof}
\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonPow\_succ\_classifier}

\begin{theorem}[Matrix singleton power successor continuation classifier]
\label{thm:matrix-singleton-pow-succ-continuation-classifier}
For histories $M,e,r:\mathsf{BHist}$, if $M$ is carried by the singleton matrix
carrier, $e$ is unary, and
$$
\ContR(\mathsf{MatrixSingletonPow}(M,e),M,r),
$$
then the one-headed power ledger is classified against $r$:
$$
\rho_{\emp}\bigl(\mathsf{MatrixSingletonPow}(M,\Eone(e)),r\bigr).
$$
\end{theorem}
\begin{proof}
Carrier closure carries the predecessor power ledger. Together with the carrier
of $M$, append empty-result exactness carries the one-headed power endpoint.
The displayed continuation identifies $r$ with the same append endpoint, so
$r$ is carried by the singleton matrix carrier and is $\hsame$ to the
one-headed power ledger.
\end{proof}
\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonPow\_succ\_continuation\_classifier}

\begin{theorem}[Matrix singleton power visible-base successor continuation empty-result absurd]
\label{thm:matrix-singleton-pow-visible-base-succ-continuation-empty-result-absurd}
For any singleton-carried continuation right input $y$, a successor power whose
base has a visible head cannot continue to the empty result:
$$
\ContR(\mathsf{MatrixSingletonPow}(\Ezero(m),\Eone(e)),y,r)
\land r\hsame\emp\Longrightarrow\bot,
$$
and the same statement holds with $\Eone(m)$ in place of $\Ezero(m)$.
\end{theorem}
\begin{proof}
Transport the displayed continuation along $r\hsame\emp$ and invert the empty
continuation result. The source side is the successor power endpoint, which
unfolds to an append whose right factor is the visible base. Append
empty-result inversion would identify $\Ezero(m)$, respectively $\Eone(m)$,
with $\emp$, contradicting BEDC constructor separation.
\end{proof}
\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonPow\_visible\_base\_succ\_continuation\_empty\_result\_absurd}

\begin{theorem}[Matrix singleton power positive-exponent visible-base empty-result exclusion]
\label{thm:matrix-singleton-pow-positive-exponent-visible-base-continuation-empty-result-absurd}
For unary nonempty exponent $e$, a singleton matrix power whose base has a
visible head cannot continue to an empty result:
$$
\ContR(\mathsf{MatrixSingletonPow}(\Ezero(m),e),y,r)
\land r\hsame\emp\Longrightarrow\bot,
$$
and the same statement holds with $\Eone(m)$ in place of $\Ezero(m)$.
\end{theorem}
\begin{proof}
Transport the continuation along $r\hsame\emp$ and invert the empty result.
The nonempty-unary power carrier exactness theorem then turns the empty power
endpoint into a singleton carrier witness for the visible base, contradicting
constructor separation.
\end{proof}
\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonPow\_positive\_exponent\_visible\_base\_continuation\_empty\_result\_absurd}

\begin{theorem}[Matrix singleton power nonempty-base empty-result exclusion]
\label{thm:matrix-singleton-pow-nonempty-base-continuation-result-absurd}

$$
\ContR(\mathsf{MatrixSingletonPow}(M,e),y,r)\land r\hsame\emp\Longrightarrow\bot.
$$
\end{theorem}
\begin{proof}
Transport the continuation result to $\emp$ and invert the empty continuation. The source carrier for $\mathsf{MatrixSingletonPow}(M,e)$ transfers through positive-unary power exactness to a singleton carrier witness for $M$, contradicting the base non-carrier hypothesis.
\end{proof}
\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonPow\_nonempty\_base\_continuation\_result\_absurd}

\begin{theorem}[Matrix singleton power append exponent classifier]
\label{thm:matrix-singleton-pow-append-exponent-classifier}
If $M$ is carried by the singleton matrix carrier and $w,q$ are unary
histories, then the power ledger over the appended exponent is classified with
the product of the two factor power ledgers:
$$
A_{\emp}(M)\land\Unary(w)\land\Unary(q)
\quad\Longrightarrow\quad
\rho_{\emp}\bigl(\mathsf{MatrixSingletonPow}(M,\mathsf{append}(w,q)),
\mathsf{MatrixSingletonMul}(\mathsf{MatrixSingletonPow}(M,w),
\mathsf{MatrixSingletonPow}(M,q))\bigr).
$$
\end{theorem}
\begin{proof}
Append closure makes $\mathsf{append}(w,q)$ unary, so power carrier closure
carries the left ledger. The same closure carries each factor ledger; append
empty-result exactness then carries their displayed product. Since both
endpoints are carried by $A_{\emp}$, transitivity through $\emp$ supplies the
classifier comparison.
\end{proof}
\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonPow\_append\_exponent\_classifier}

\begin{theorem}[Matrix singleton power append successor right exponent classifier]
\label{thm:matrix-singleton-pow-append-succ-right-exponent-classifier}
If $M$ is carried by the singleton matrix carrier and $w,q$ are unary histories,
then the append exponent with right successor is classified by the left power
and the nested right multiplication endpoint:
$$
\rho_{\emp}\bigl(\mathsf{MatrixSingletonPow}(M,\mathsf{append}(w,\Eone(q))),
\mathsf{MatrixSingletonMul}(\mathsf{MatrixSingletonPow}(M,w),
\mathsf{MatrixSingletonMul}(\mathsf{MatrixSingletonPow}(M,q),M))\bigr).
$$
\end{theorem}
\begin{proof}
Unary closure carries $\mathsf{append}(w,\Eone(q))$. Singleton power closure
carries the composite power, the left power, and the right predecessor power.
Append empty-result exactness carries the inner multiplication by $M$ and then
the outer multiplication by the left power. Since both endpoints are carried by
$A_{\emp}$, transitivity through $\emp$ supplies the classifier comparison.
\end{proof}
\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonPow\_append\_succ\_right\_exponent\_classifier}

\begin{theorem}[Matrix singleton power append exponent classifier iff]
\label{thm:matrix-singleton-pow-append-exponent-classifier-iff}
If $M$ is carried by the singleton matrix carrier and $a,b$ are unary
histories, then the classifier of the append-exponent power ledger against
$h$ is exactly the singleton carrier condition on $h$:
$$
\mathsf{MatrixSingletonClassifier}
  (\mathsf{MatrixSingletonPow}(M,\mathsf{append}(a,b)),h)
\Longleftrightarrow
\mathsf{MatrixSingletonCarrier}(h).
$$
\end{theorem}
\begin{proof}
Unary closure under append makes $\mathsf{append}(a,b)$ a unary exponent. The
power carrier theorem therefore carries
$\mathsf{MatrixSingletonPow}(M,\mathsf{append}(a,b))$. The forward direction
reads the right carrier field from the classifier. In the reverse direction,
that carried power endpoint and the carrier witness for $h$ assemble the
restricted singleton classifier.
\end{proof}
\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonPow\_append\_exponent\_classifier\_iff}

\begin{theorem}[Matrix singleton power successor endpoint exactness]
\label{thm:matrix-singleton-pow-successor-endpoint-exactness}
For histories $M,e:\mathsf{BHist}$, if $A_{\emp}(M)$ and $\Unary(e)$, then
the successor power is carried:
$$
A_{\emp}(\mathsf{MatrixSingletonPow}(M,\Eone(e))).
$$
Moreover, the successor power is exactly classified with both displayed
singleton multiplication endpoints:
$$
\rho_{\emp}(\mathsf{MatrixSingletonPow}(M,\Eone(e)),
\mathsf{MatrixSingletonPow}(M,e)\odot M)
\quad\text{and}\quad
\rho_{\emp}(\mathsf{MatrixSingletonPow}(M,\Eone(e)),
M\odot\mathsf{MatrixSingletonPow}(M,e)).
$$
\end{theorem}
\begin{proof}
Unfold the one-headed exponent clause of $\mathsf{MatrixSingletonPow}$. The
successor ledger is the displayed singleton product
$\mathsf{MatrixSingletonPow}(M,e)\odot M$.

The carrier closure theorem for powers gives
$A_{\emp}(\mathsf{MatrixSingletonPow}(M,e))$ from $A_{\emp}(M)$ and
$\Unary(e)$. Combining this carrier with $A_{\emp}(M)$, the singleton
multiplication closure in
Theorem~\ref{thm:singleton-empty-history-matrix-laws} carries both append
endpoints
$\mathsf{MatrixSingletonPow}(M,e)\odot M$ and
$M\odot\mathsf{MatrixSingletonPow}(M,e)$.

For the first classifier, the unfolded successor ledger is the same append
endpoint, so the singleton classifier closes by reflexivity and the carried
endpoint witness. For the second classifier, both append endpoints classify as
empty singleton endpoints; transitivity through $\emp$ compares them and
supplies $\rho_{\emp}$.
\end{proof}
\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonPow\_succ\_endpoint\_exactness}
\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonPow\_successor\_endpoint\_exactness}
