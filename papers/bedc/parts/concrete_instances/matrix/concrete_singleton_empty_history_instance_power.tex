\section{Concrete singleton empty-history matrix instance}
\label{sec:matrix-concrete-singleton-instance}

\begin{definition}[Concrete singleton empty-history matrix instance]
\label{def:singleton-empty-history-matrix-instance}
Let $A_{\emp}$ be the singleton history source of
Definition~\ref{def:list-singleton-history-source} at $a_0=\emp$. Use the
classifier
$$
\rho_{\emp}(h,k)\Longleftrightarrow
A_{\emp}(h)\land A_{\emp}(k)\land\hsame(h,k).
$$
The row-dimension witness and column-dimension witness are both represented by
the empty unary history $\emp$. The scalar side is the singleton empty-history
ring of Definition~\ref{def:singleton-empty-history-ring-instance}, and the
row and column vector layers are read through the singleton empty-history
vector-space instance of
Definition~\ref{def:singleton-empty-history-vecspace-instance}.

The matrix carrier is the same singleton source $A_{\emp}$ on
$\mathsf{BHist}$ codes. Define the displayed matrix operations by
$$
0_M:=\emp,\qquad I_M:=\emp,\qquad
M\oplus N:=\mathsf{append}(M,N),\qquad
M\odot N:=\mathsf{append}(M,N).
$$
The certificate fit has ten fields:
\begin{enumerate}
  \item Traditional target: the terminal singleton matrix algebra.
  \item BEDC source: $\mathsf{BHist}$ together with the singleton ring and
  $\VecSpaceUp$ layers.
  \item Carrier choice: $A_{\emp}$.
  \item Classifier choice: $\rho_{\emp}$.
  \item Stability obligations: closure, zero laws, identity laws,
  associativity, and congruence for the displayed operations.
  \item Ledger: the singleton endpoint together with row and column witnesses
  represented by $\emp$.
  \item Core theorem list:
  Theorem~\ref{thm:singleton-empty-history-matrix-laws}.
  \item Exactness predicate: singleton endpoint classification.
  \item Standard bridge: the terminal matrix object.
  \item Status: $\mathsf{ProofTarget}$.
\end{enumerate}
\end{definition}

\begin{definition}[Matrix singleton power]
\label{def:matrix-singleton-pow}
For histories $M$ and $e$, $\mathsf{MatrixSingletonPow}(M,e)$ is the concrete
power ledger obtained by recursion over the exponent history: the empty
exponent returns $I_M$, a one-headed exponent multiplies the previous ledger
by $M$, and the zero-headed branch returns the singleton zero endpoint.
\end{definition}
\leandef{BEDC.Derived.MatrixUp.MatrixSingletonPow}

\begin{theorem}[Matrix singleton power empty-exponent classifier iff]
\label{thm:matrix-singleton-pow-empty-exponent-classifier-iff}
For histories $M,h:\mathsf{BHist}$, the singleton classifier of the empty
exponent power ledger against $h$ is exactly the singleton carrier condition
on $h$:
$$
\mathsf{MatrixSingletonClassifier}(\mathsf{MatrixSingletonPow}(M,\emp),h)
\Longleftrightarrow
\mathsf{MatrixSingletonCarrier}(h).
$$
\end{theorem}
\begin{proof}
The empty exponent branch of $\mathsf{MatrixSingletonPow}$ is
$\mathsf{MatrixSingletonOne}=\emp$. The forward implication reads the right
carrier field from the classifier. Conversely, a singleton carrier witness
for $h$ combines with the empty carrier of $\emp$ and symmetry of $\hsame$ to
assemble the singleton classifier.
\end{proof}
\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonPow\_empty\_exponent\_classifier\_iff}

\begin{theorem}[Matrix singleton power carrier closure]
\label{thm:matrix-singleton-pow-carrier-closed}
If $M$ is carried by the singleton matrix carrier and $e$ is unary, then the
power ledger is again carried by the singleton matrix carrier:
$$
A_{\emp}(M)\land \Unary(e)
\quad\Longrightarrow\quad
A_{\emp}(\mathsf{MatrixSingletonPow}(M,e)).
$$
\end{theorem}
\begin{proof}
Proceed by induction on the unary exponent. The empty exponent returns
$I_M=\emp$, hence is carried by $A_{\emp}$. In the one-headed case, the
induction hypothesis carries the previous power ledger, while the hypothesis
carries $M$; the displayed multiplication is append, so append empty-result
exactness carries the product ledger.
\end{proof}
\leanchecked{BEDC.Derived.MatrixUp.MatrixSingletonPow\_carrier\_closed}

\end{proof}

\rho_{\emp}(\mathsf{MatrixSingletonPow}(M,\Eone(e)),
\mathsf{MatrixSingletonPow}(M,e)\odot M)
\quad\text{and}\quad
\rho_{\emp}(\mathsf{MatrixSingletonPow}(M,\Eone(e)),
M\odot\mathsf{MatrixSingletonPow}(M,e)).
$$
\end{theorem}
\begin{proof}
Unfold the one-headed exponent clause of $\mathsf{MatrixSingletonPow}$. The
successor ledger is the displayed singleton product
$\mathsf{MatrixSingletonPow}(M,e)\odot M$.

The carrier closure theorem for powers gives
$A_{\emp}(\mathsf{MatrixSingletonPow}(M,e))$ from $A_{\emp}(M)$ and
$\Unary(e)$. Combining this carrier with $A_{\emp}(M)$, the singleton
multiplication closure in
Theorem~\ref{thm:singleton-empty-history-matrix-laws} carries both append
endpoints
$\mathsf{MatrixSingletonPow}(M,e)\odot M$ and
$M\odot\mathsf{MatrixSingletonPow}(M,e)$.

For the first classifier, the unfolded successor ledger is the same append
endpoint, so the singleton classifier closes by reflexivity and the carried
endpoint witness. For the second classifier, both append endpoints classify as
empty singleton endpoints; transitivity through $\emp$ compares them and
supplies $\rho_{\emp}$.
\end{proof}

\begin{definition}[Matrix singleton identity-power concrete instance]
\label{def:matrix-singleton-identity-power-concrete-instance}
The concrete identity-power schema for $\MatrixUp$ uses
$\mathsf{MatrixSingletonCarrier}$ as carrier,
$\mathsf{MatrixSingletonClassifier}$ as classifier, $\mathsf{UnaryHistory}$ as
the exponent carrier, and $\mathsf{MatrixSingletonPow}$ as the action. Its
source rows are the singleton matrix carrier row for the base, the unary
history row for the exponent, the empty-exponent identity row, and the
successor-step multiplication row over the singleton power ledger.
\end{definition}

\begin{theorem}[Matrix singleton identity-power law]
\label{thm:matrix-singleton-identity-power-law}
For the concrete instance of
\autoref{def:matrix-singleton-identity-power-concrete-instance}, the empty
exponent acts as the singleton identity endpoint, unary exponents preserve the
singleton matrix carrier, and the successor exponent is classified by the
displayed multiplication step:
$$
\begin{gathered}
\mathsf{MatrixSingletonClassifier}
  (\mathsf{MatrixSingletonPow}(M,\emp),h)
\Longleftrightarrow
\mathsf{MatrixSingletonCarrier}(h),\\
\mathsf{MatrixSingletonCarrier}(M)\land\mathsf{UnaryHistory}(e)
\Longrightarrow
\mathsf{MatrixSingletonCarrier}(\mathsf{MatrixSingletonPow}(M,e)),\\
\mathsf{MatrixSingletonCarrier}(M)\land\mathsf{UnaryHistory}(e)
\Longrightarrow
\mathsf{MatrixSingletonClassifier}
  (\mathsf{MatrixSingletonPow}(M,\Eone(e)),
   \mathsf{MatrixSingletonMul}(\mathsf{MatrixSingletonPow}(M,e),M)).
\end{gathered}
$$
Thus the singleton power rows expose a concrete $\mathsf{BHist}$ carrier and
action surface for $\MatrixUp$ rather than only an abstract operator schema.
\end{theorem}
\begin{proof}
The empty-exponent identity row is exactly
\autoref{thm:matrix-singleton-pow-empty-exponent-classifier-iff}. It unfolds
the empty branch of $\mathsf{MatrixSingletonPow}$ to the singleton identity
endpoint and identifies its classifier with the carrier condition on the
comparison endpoint.

Carrier closure under unary exponents is
\autoref{thm:matrix-singleton-pow-carrier-closed}. It gives the carried power
ledger from a singleton-carried base and a unary exponent, so the action stays
inside $\mathsf{MatrixSingletonCarrier}$.

The successor step is
\autoref{thm:matrix-singleton-pow-succ-classifier}, strengthened by
\autoref{thm:matrix-singleton-pow-successor-endpoint-exactness}. These rows
classify the one-headed exponent ledger by the displayed singleton
multiplication endpoint, completing the concrete carrier/action schema.
\end{proof}
