\begin{definition}[Matrix transpose over a certified scalar carrier]
\label{def:matrix-transpose-certified-scalar-carrier}
Let $R$ be a scalar $\RingUp$ carrier and let $n,m:\NatUp$. For a carried matrix $X:\mathsf{Fin}(n)\times\mathsf{Fin}(m)\to R$ from \autoref{def:matrix-carrier}, define its transpose by
$$
(X^{\mathsf{T}})_{j,i}:=X_{i,j}
$$
for $j:\mathsf{Fin}(m)$ and $i:\mathsf{Fin}(n)$. The carriedness witness at the transposed index $(j,i)$ is the scalar witness already assigned to the original entry $X_{i,j}$ by the matrix pattern of \autoref{def:matrix-pattern-specification}.
\end{definition}

\begin{theorem}[Matrix transpose reverses product over the opposite ring]
\label{thm:matrix-transpose-reverses-product-over-opposite-ring}
Let $R$ be a scalar $\RingUp$ carrier with classifier $\sim_R$, zero endpoint $0_R$, addition $+_R$, multiplication $\cdot_R$, and finite additive fold $\Sigma_{\mathsf{Fin}(m)}$ as in \autoref{thm:finite-additive-fold-congruence-coefficient-lists}. Let $R^{\mathrm{op}}$ be the opposite ring data of \autoref{def:opposite-ring-certificate-data}, certified by \autoref{thm:opposite-ring-certificate}; thus $R^{\mathrm{op}}$ has the same carrier, classifier, zero endpoint, and addition as $R$, and multiplication
$$
x\cdot_{R^{\mathrm{op}}}y:=y\cdot_Rx.
$$
Let $n,m,p:\NatUp$ and let
$$
A:\mathsf{Fin}(n)\times\mathsf{Fin}(m)\to R,
\qquad
B:\mathsf{Fin}(m)\times\mathsf{Fin}(p)\to R
$$
be carried matrices. Consider the $\MatrixUp$ carriers of \autoref{def:matrix-carrier} with pointwise classifier from \autoref{def:matrix-classifier-specification}, and write $\sim_{\mathrm{Mat},R^{\mathrm{op}}}$ for that classifier over the opposite scalar source. Assume matrix multiplication over any certified scalar ring is the finite-fold operation
$$
(X\cdot_S Y)_{u,w}:=\Sigma_V(v\mapsto X_{u,v}\cdot_SY_{v,w})
$$
for the matching inner finite index carrier $V$, using the multiplication closure field of \autoref{def:matrix-stability-certificate}. Then $(A\cdot_RB)^{\mathsf{T}}$ and $B^{\mathsf{T}}\cdot_{R^{\mathrm{op}}}A^{\mathsf{T}}$ are carried $\MatrixUp$ matrices over $R^{\mathrm{op}}$, and
$$
(A\cdot_RB)^{\mathsf{T}}
\sim_{\mathrm{Mat},R^{\mathrm{op}}}
B^{\mathsf{T}}\cdot_{R^{\mathrm{op}}}A^{\mathsf{T}}.
$$
\end{theorem}
\begin{proof}
The product $A\cdot_RB$ is carried by the multiplication closure row of \autoref{def:matrix-stability-certificate}. Since the opposite ring has the same scalar carrier and classifier by \autoref{def:opposite-ring-certificate-data}, \autoref{def:matrix-transpose-certified-scalar-carrier} carries $(A\cdot_RB)^{\mathsf{T}}$ over $R^{\mathrm{op}}$. The same transpose definition carries $B^{\mathsf{T}}$ and $A^{\mathsf{T}}$ over $R^{\mathrm{op}}$, and the multiplication closure row over the certified opposite ring carries $B^{\mathsf{T}}\cdot_{R^{\mathrm{op}}}A^{\mathsf{T}}$.

It remains to compare entries. Fix $k:\mathsf{Fin}(p)$ and $i:\mathsf{Fin}(n)$. Expanding the left endpoint gives
$$
((A\cdot_RB)^{\mathsf{T}})_{k,i}=(A\cdot_RB)_{i,k}\sim_R\Sigma_{\mathsf{Fin}(m)}(j\mapsto A_{i,j}\cdot_RB_{j,k}).
$$
Expanding the product over the opposite ring gives
$$
(B^{\mathsf{T}}\cdot_{R^{\mathrm{op}}}A^{\mathsf{T}})_{k,i}\sim_R\Sigma_{\mathsf{Fin}(m)}(j\mapsto (B^{\mathsf{T}})_{k,j}\cdot_{R^{\mathrm{op}}}(A^{\mathsf{T}})_{j,i}).
$$
By transpose readback and opposite multiplication, this last finite spine is the same as
$$
\Sigma_{\mathsf{Fin}(m)}(j\mapsto B_{j,k}\cdot_{R^{\mathrm{op}}}A_{i,j})=\Sigma_{\mathsf{Fin}(m)}(j\mapsto A_{i,j}\cdot_RB_{j,k}).
$$
The two displayed expansions end at the same finite scalar spine. The finite additive fold over $R^{\mathrm{op}}$ uses the same zero endpoint and the same addition as the fold over $R$, so the pointwise reflexive scalar classifier witness, lifted through \autoref{thm:finite-additive-fold-congruence-coefficient-lists}, compares the folded endpoints. Classifier transitivity composes the left expansion with this fold comparison and the symmetry of the right expansion.

Thus for every $k$ and $i$,
$$
((A\cdot_RB)^{\mathsf{T}})_{k,i}
\sim_R
(B^{\mathsf{T}}\cdot_{R^{\mathrm{op}}}A^{\mathsf{T}})_{k,i}.
$$
The pointwise matrix classifier of \autoref{def:matrix-classifier-specification} packages these scalar comparisons as the displayed $\MatrixUp$ classification over $R^{\mathrm{op}}$. The row-column swap and the unchanged scalar source are recorded by the dimension and scalar ledger of \autoref{def:matrix-ledger-policy}.
\end{proof}

\begin{theorem}[Matrix transpose double-readback involution]
\label{thm:matrix-transpose-double-readback-involution}
Let $R$ be a scalar $\RingUp$ carrier with classifier $\sim_R$, and let $n,m:\NatUp$. Consider the $\MatrixUp$ carriers of \autoref{def:matrix-carrier} with pointwise classifier from \autoref{def:matrix-classifier-specification}. If
$$
X:\mathsf{Fin}(n)\times\mathsf{Fin}(m)\to R
$$
is a carried matrix, then $(X^{\mathsf{T}})^{\mathsf{T}}$ is carried as an $n\times m$ $\MatrixUp$ matrix, and
$$
(X^{\mathsf{T}})^{\mathsf{T}}\sim_{\mathrm{Mat}}X.
$$
\end{theorem}
\begin{proof}
By \autoref{def:matrix-transpose-certified-scalar-carrier}, the first transpose $X^{\mathsf{T}}$ is carried over $\mathsf{Fin}(m)\times\mathsf{Fin}(n)$: at an index $(j,i)$ its scalar witness is the witness for $X_{i,j}$ supplied by the matrix pattern of \autoref{def:matrix-pattern-specification}. Applying the same transpose carrier rule once more carries $(X^{\mathsf{T}})^{\mathsf{T}}$ over $\mathsf{Fin}(n)\times\mathsf{Fin}(m)$. At an index $(i,j)$ the second carriedness witness is the first-transpose witness for $(X^{\mathsf{T}})_{j,i}$, which is exactly the original scalar witness for $X_{i,j}$.

It remains to prove the pointwise classifier comparison. Fix $i:\mathsf{Fin}(n)$ and $j:\mathsf{Fin}(m)$. Expanding \autoref{def:matrix-transpose-certified-scalar-carrier} twice gives
$$
((X^{\mathsf{T}})^{\mathsf{T}})_{i,j}=(X^{\mathsf{T}})_{j,i}=X_{i,j}.
$$
The original carried matrix supplies a scalar carrier witness for $X_{i,j}$, and the scalar classifier reflexivity row retained by the certified $\RingUp$ carrier gives
$$
((X^{\mathsf{T}})^{\mathsf{T}})_{i,j}\sim_R X_{i,j}.
$$
Since $i$ and $j$ were arbitrary, the pointwise classifier of \autoref{def:matrix-classifier-specification} packages these scalar comparisons as $(X^{\mathsf{T}})^{\mathsf{T}}\sim_{\mathrm{Mat}}X$. The dimension and scalar-source ledger is unchanged after the two coordinate swaps, as recorded by \autoref{def:matrix-ledger-policy}.
\end{proof}

\begin{theorem}[Matrix transpose preserves matrix addition]
\label{thm:matrix-transpose-preserves-addition}
Let $R$ be a scalar $\RingUp$ carrier with classifier $\sim_R$ and addition $+_R$. Let $n,m:\NatUp$, and consider the $\MatrixUp$ carriers of \autoref{def:matrix-carrier} with pointwise classifier from \autoref{def:matrix-classifier-specification}. For carried matrices
$$
A,B:\mathsf{Fin}(n)\times\mathsf{Fin}(m)\to R,
$$
assume matrix addition on each matching $\MatrixUp$ carrier is the pointwise operation
$$
(X+Y)_{u,v}:=X_{u,v}+_R Y_{u,v}
$$
for matching indices $u,v$, using the addition closure row of \autoref{def:matrix-stability-certificate}. Then $(A+B)^{\mathsf{T}}$ and $A^{\mathsf{T}}+B^{\mathsf{T}}$ are carried $\MatrixUp$ matrices over $\mathsf{Fin}(m)\times\mathsf{Fin}(n)$, and
$$
(A+B)^{\mathsf{T}}\sim_{\mathrm{Mat}}A^{\mathsf{T}}+B^{\mathsf{T}}.
$$
\end{theorem}
\begin{proof}
The addition closure row of \autoref{def:matrix-stability-certificate} carries $A+B$ over $\mathsf{Fin}(n)\times\mathsf{Fin}(m)$. Applying \autoref{def:matrix-transpose-certified-scalar-carrier} carries $(A+B)^{\mathsf{T}}$ over $\mathsf{Fin}(m)\times\mathsf{Fin}(n)$: at the transposed index $(j,i)$ its scalar witness is the carriedness witness for $(A+B)_{i,j}$ supplied by the matrix pattern of \autoref{def:matrix-pattern-specification}.

The same transpose definition carries $A^{\mathsf{T}}$ and $B^{\mathsf{T}}$ over $\mathsf{Fin}(m)\times\mathsf{Fin}(n)$. Applying the pointwise addition closure row in that swapped matrix carrier carries $A^{\mathsf{T}}+B^{\mathsf{T}}$. The ledger records the same scalar source and the exchanged finite index witnesses, as in \autoref{def:matrix-ledger-policy}.

It remains to compare entries. Fix $j:\mathsf{Fin}(m)$ and $i:\mathsf{Fin}(n)$. Expanding the transpose and pointwise addition clauses gives
$$
((A+B)^{\mathsf{T}})_{j,i}=(A+B)_{i,j}=A_{i,j}+_R B_{i,j}.
$$
Expanding the transposed pointwise sum gives
$$
(A^{\mathsf{T}}+B^{\mathsf{T}})_{j,i}=(A^{\mathsf{T}})_{j,i}+_R(B^{\mathsf{T}})_{j,i}=A_{i,j}+_R B_{i,j}.
$$
The two displayed endpoints are the same carried scalar sum, so scalar classifier reflexivity gives
$$
((A+B)^{\mathsf{T}})_{j,i}
\sim_R
(A^{\mathsf{T}}+B^{\mathsf{T}})_{j,i}.
$$
Since $j$ and $i$ were arbitrary, the pointwise classifier of \autoref{def:matrix-classifier-specification} packages these scalar comparisons as
$$
(A+B)^{\mathsf{T}}\sim_{\mathrm{Mat}}A^{\mathsf{T}}+B^{\mathsf{T}}.
$$
\end{proof}
