\section{Holomorphic predicate}

\begin{definition}[Holomorphic on disk seed predicate]
\label{def:holo-on-disk}
For function $f : \ComplexUp \to \ComplexUp$, center $z_0$, and unary radius $r$, the seed predicate $\mathsf{HoloOnDisk}(f, z_0, r)$ is the intended carrier condition saying that a uniform modulus function $\delta : \Unary \times \Unary \to \Unary$ controls complex differentiability at every point $z$ with $\mathsf{OpenDisk}(z_0, r, z)$ and every unary precision $k$, using the boundary gap $\mathsf{Gap}(z_0, r, z)$ as the second modulus input.
\end{definition}

\section{Core theorem obligations}

\begin{definition}[Holomorphic core theorem obligations]
\label{def:holo-core-theorem-obligations}
The $\HolomorphicUp$ seed carries the following proof obligations before it can be presented as a certificate:
\begin{enumerate}
  \item uniform-modulus obligation: any accepted carrier witness must expose explicit modulus data, and equivalent witnesses must admit pointwise refinements compatible with $\hsame$;
  \item iterated-derivative obligation: the derivative sequence $\mathsf{IteratedCplxDiff}(f, z_0, n, f^{(n)})$ must be generated by unary recursion from the local complex-differentiability data;
  \item closure obligations: sum, product, and composition must preserve the carrier, with explicit modulus transport and an explicit image-containment witness for composition;
  \item power-series obligation: Taylor coefficients computed from the iterated-derivative data must determine a power-series witness with a convergence-radius comparison to $r$;
  \item identity obligation: agreement on a quantitatively dense sequence inside the disk must transport to agreement on an explicitly smaller disk;
  \item Cauchy-bound obligation: local upper bounds for $|f|$ on smaller disks must yield unary bounds for the iterated derivatives at the center.
\end{enumerate}
\end{definition}

\begin{lemma}[Iterated complex differentiability preserves unary seed histories]
\label{lem:iterated-cplx-diff-unary-of-seed}
If an iterated complex-differentiability chain starts from a unary seed history and each successor stage is recorded by a unary continuation step, then every selected iterate is unary.
\end{lemma}
\begin{proof}
Induct on the iteration index. At zero, transport the unary seed along the displayed $\hsame$ endpoint. At a successor index, unpack the previous iterate, the unary step, and the continuation witness; the induction hypothesis gives the previous iterate as unary, and unary continuation closure gives the selected successor history.
\end{proof}
\leanchecked{BEDC.Derived.HolomorphicUp.IteratedCplxDiff\_unary\_of\_seed}

\begin{theorem}[Iterated complex differentiability transports unary readback]
\label{thm:iterated-cplx-diff-hsame-transport-unary-readback}
If the seed endpoint and selected iterate of an iterated complex-differentiability chain are each replaced by $\hsame$-equivalent histories, then the whole chain has the corresponding transported endpoint, and unary readback from the transported seed still proves the transported selected iterate unary.
\end{theorem}
\begin{proof}
Induct on the iteration index. At zero, compose the seed comparison, the displayed endpoint comparison, and the chain's zero endpoint. At a successor stage, transport the predecessor chain by the induction hypothesis and move the successor continuation witness across the endpoint comparison by continuation $\hsame$-stability. Unary readback then applies to the transported chain.
\end{proof}
\leanchecked{BEDC.Derived.HolomorphicUp.IteratedCplxDiff\_hsame\_transport\_unary\_readback}

\begin{theorem}[Iterated complex differentiability is closed under additive concatenation]
\label{thm:iterated-cplx-diff-add-closed-unary-readback}
If an iterated complex-differentiability chain carries a seed to a middle
history in $n$ stages, and a second chain carries that middle history to a
target history in $m$ stages, then their concatenation carries the seed to
the target in $n+m$ stages. Unary readback from the seed still proves the
target history unary.
\end{theorem}
\begin{proof}
Induct on $m$. At $m=0$, the second chain is exactly an $\hsame$ endpoint,
so transport the first chain to the target and reuse unary readback. At a
successor stage, apply the induction hypothesis to the predecessor endpoint,
append the displayed unary continuation step, and close the target by unary
continuation closure.
\end{proof}
\leanchecked{BEDC.Derived.HolomorphicUp.IteratedCplxDiff\_add\_closed\_unary\_readback}

\begin{theorem}[Strict iterated complex differentiability preserves unary seed histories]

If a strict iterated complex-differentiability chain starts from a unary seed

\end{theorem}
\begin{proof}
Induct on the iteration index. The zero case transports unary membership along

\end{proof}
\leanchecked{BEDC.Derived.HolomorphicUp.IteratedStrictCplxDiff\_unary\_of\_seed}

\begin{theorem}[Strict iterated complex differentiability forgets strict steps]
\label{thm:iterated-strict-cplx-diff-forget-strict-unary-readback}
Every strict iterated complex-differentiability chain has an underlying
ordinary iterated complex-differentiability chain with the same seed, index,
and selected endpoint. Consequently, unary membership of the seed still reads
back to unary membership of the selected endpoint through the ordinary chain.
\end{theorem}
\begin{proof}
Induct on the iteration index. At zero, the strict and ordinary chains have
the same seed endpoint classifier. At a successor stage, unpack the predecessor
and strict nonempty unary step, apply the induction hypothesis to the
predecessor chain, and keep the same unary continuation witness while erasing
only the nonemptiness field. Unary readback then applies to the resulting
ordinary iterated chain.
\end{proof}
\leanchecked{BEDC.Derived.HolomorphicUp.IteratedStrictCplxDiff\_forget\_strict\_unary\_readback}

\begin{theorem}[Strict iterated complex differentiability has ordinary readback]
\label{thm:iterated-strict-cplxdiff-relaxation}
Every strict iterated complex-differentiability chain determines the ordinary
iterated complex-differentiability chain with the same seed, index, and
selected history. Consequently, a unary seed still reads back unary membership
of the selected history:
$$
\mathsf{IteratedStrictCplxDiff}(seed,n,h)
\Longrightarrow
\mathsf{IteratedCplxDiff}(seed,n,h)\wedge
\bigl(\mathsf{UnaryHistory}(seed)\Rightarrow\mathsf{UnaryHistory}(h)\bigr).
$$
\end{theorem}
\begin{proof}
Induct on the iteration index. At zero, both chains are the same endpoint
classifier from the seed to the selected history, and unary readback is
transport along that classifier. At a successor index, unpack the predecessor,
the unary step, the nonempty-step witness, and the continuation. The induction
hypothesis supplies the ordinary predecessor chain; discarding only the
nonempty-step witness reconstructs the ordinary successor chain, and unary
continuation closure gives the displayed readback.
\end{proof}
\leanchecked{BEDC.Derived.HolomorphicUp.IteratedStrictCplxDiff\_iteratedCplxDiff\_unary\_readback}

\begin{lemma}[Strict iterated complex differentiability exposes a strict prefix]
\label{lem:iterated-strict-cplx-diff-strict-prefix}
If a strict iterated complex-differentiability chain starts from a unary seed
history and advances by at least one nonempty unary continuation step, then the
seed is a strict unary prefix of the selected iterate.
\end{lemma}
\begin{proof}
Induct on the number of remaining successor stages. The first successor
unpacks a nonempty unary step and transports its source along the seed
classifier. At a later successor, the induction hypothesis gives a strict
prefix from the seed to the previous iterate, and strict-prefix transitivity
composes it with the displayed nonempty continuation tail.
\end{proof}
\leanchecked{BEDC.Derived.HolomorphicUp.IteratedStrictCplxDiff\_strict\_prefix}

\begin{theorem}[Strict iterated complex differentiability zero-or-prefix dichotomy]
\label{thm:iterated-strict-cplx-diff-zero-or-strict-prefix}
For any strict iterated complex-differentiability chain from a unary seed, the
zero-stage case identifies the selected endpoint with the seed, while every
positive stage exposes a strict unary prefix from the seed into the selected
endpoint:
$$
\mathsf{IteratedStrictCplxDiff}(seed,n,h)
\Longrightarrow
\hsame(seed,h)\vee\mathsf{NatUnaryStrictPrefix}(seed,h).
$$
\end{theorem}
\begin{proof}
Split on the natural iteration index. The zero branch is exactly the endpoint
classifier stored by the definition. The successor branch applies the
strict-prefix readback theorem to the displayed nonempty chain.
\end{proof}
\leanchecked{BEDC.Derived.HolomorphicUp.IteratedStrictCplxDiff\_zero\_or\_strict\_prefix}

\begin{theorem}[Strict iterated complex differentiability endpoint contradiction]
\label{thm:iterated-strict-cplxdiff-endpoint-hsame-absurd}
If a strict iterated complex-differentiability chain starts from a unary seed
and advances by at least one nonempty continuation step, then its final endpoint
cannot be $\hsame$ to the seed:
$$
\mathsf{UnaryHistory}(seed)
\Longrightarrow
\mathsf{IteratedStrictCplxDiff}(seed,n+1,h)
\Longrightarrow
\hsame(seed,h)
\Longrightarrow
\bot.
$$
\end{theorem}
\begin{proof}
The strict-prefix readback theorem extracts a strict unary prefix from the seed
to the final endpoint. Unpacking this prefix gives a nonempty unary tail and a
continuation from the seed to the endpoint. Endpoint $\hsame$ cancellation
forces that tail to be empty, contradicting strictness.
\end{proof}
\leanchecked{BEDC.Derived.HolomorphicUp.IteratedStrictCplxDiff\_endpoint\_hsame\_absurd}

\begin{theorem}[Strict iterated complex differentiability result nonemptiness]
\label{thm:iterated-strict-cplxdiff-result-not-empty}
If a strict iterated complex-differentiability chain advances by at least one
nonempty continuation step, then its final endpoint cannot be $\hsame$ to
$\emp$.
\end{theorem}
\begin{proof}
Invert the successor strict-iteration witness to expose the previous endpoint,
the final step, and its nonempty-step field. Transport the final continuation
result along the displayed empty endpoint comparison. Empty-result inversion
forces the final step to be empty, contradicting the stored strictness field.
\end{proof}
\leanchecked{BEDC.Derived.HolomorphicUp.IteratedStrictCplxDiff\_result\_not\_empty}

\begin{theorem}[Strict iterated complex differentiability transports unary readback]
\label{thm:iterated-strict-cplx-diff-hsame-transport-unary-readback}
If a strict iterated complex-differentiability chain has $\hsame$-equivalent
seed and selected endpoints, then the strict chain transports to those
endpoints. Consequently, a unary transported seed still reads back unary
membership of the transported selected endpoint.
\end{theorem}
\begin{proof}
Induct over the iteration index. At zero, compose the seed comparison with
the displayed zero endpoint and the target comparison. At a successor index,
transport the predecessor chain by the induction hypothesis and transport the
final continuation step across the target endpoint comparison. Unary readback
then applies to the transported strict chain.
\end{proof}
\leanchecked{BEDC.Derived.HolomorphicUp.IteratedStrictCplxDiff\_hsame\_transport\_unary\_readback}

\begin{theorem}[Strict iterated complex differentiability transports strict prefix readback]
\label{thm:iterated-strict-cplx-diff-hsame-transport-strict-prefix}
If a nonempty strict iterated complex-differentiability chain has $\hsame$-equivalent seed and target endpoints, then the strict chain transports to those endpoints, and a unary transported seed exposes a strict unary prefix into the transported target.
\end{theorem}
\begin{proof}
Induct over the successor count of the strict chain. The first successor composes the seed endpoint comparison with the zero-stage endpoint and transports the displayed nonempty unary continuation. The successor case applies the induction hypothesis to the previous iterate and transports the final continuation step. The strict-prefix readback theorem then applies to the transported strict chain.
\end{proof}
\leanchecked{BEDC.Derived.HolomorphicUp.IteratedStrictCplxDiff\_hsame\_transport\_strict\_prefix}

\begin{theorem}[Strict iterated complex differentiability is closed under additive concatenation]
\label{thm:iterated-strict-cplx-diff-add-closed-strict-prefix-readback}
If a strict iterated complex-differentiability chain carries a seed to a
middle history in $n$ stages, and a second strict chain carries that middle
history to a target history in at least one stage, then their concatenation is
a strict chain from the seed to the target in $n+m+1$ stages. Unary readback
from the seed exposes a strict unary prefix into the target.
\end{theorem}
\begin{proof}
Induct over the positive right-hand chain. In the first stage, transport the
left chain to the predecessor endpoint of the displayed strict continuation
and append that continuation. In the successor case, apply the induction
hypothesis to the previous endpoint and append the final strict continuation.
The strict-prefix readback theorem applies to the concatenated nonempty chain.
\end{proof}
\leanchecked{BEDC.Derived.HolomorphicUp.IteratedStrictCplxDiff\_add\_closed\_strict\_prefix\_readback}

\begin{theorem}[Strict iterated complex differentiability with ordinary suffix readback]
\label{thm:iterated-strict-cplxdiff-ordinary-suffix-strict-prefix-readback}
If a strict iterated complex-differentiability chain carries a seed to a
middle history in at least one stage, and an ordinary iterated chain carries
that middle history to a target history, then the combined ordinary chain
starts at the seed and has the summed length. Unary readback from the seed
still exposes a strict unary prefix into the target.
\end{theorem}
\begin{proof}
Forget the strict left chain to an ordinary iterated chain and concatenate it
with the ordinary suffix. For the prefix readback, induct over the ordinary
suffix. The zero case transports the strict prefix through the endpoint
comparison. In the successor case, the induction hypothesis gives a strict
prefix into the previous endpoint, and the final ordinary continuation gives
a unary weak prefix from that endpoint to the target; strict prefix is closed
under such a right extension.
\end{proof}
\leanchecked{BEDC.Derived.HolomorphicUp.IteratedStrictCplxDiff\_ordinary\_suffix\_strict\_prefix\_readback}

\begin{theorem}[Strict iterated concatenation rejects a closed endpoint]
\label{thm:iterated-strict-cplxdiff-add-closed-endpoint-hsame-absurd}
If a strict iterated chain carries a unary seed to a middle history and a
second strict chain carries that middle history through at least one nonempty
step to a target, then the concatenated target cannot be history-same to the
seed.
\end{theorem}
\begin{proof}
The additive strict-prefix readback theorem concatenates the two chains and
extracts a strict unary prefix from the seed to the target. Unpacking that
prefix gives a nonempty unary tail and a continuation from the seed to the
target. Endpoint $\hsame$ cancellation forces that tail to be empty,
contradicting strictness.
\end{proof}
\leanchecked{BEDC.Derived.HolomorphicUp.IteratedStrictCplxDiff\_add\_closed\_endpoint\_hsame\_absurd}

\section{The seed fields}

\begin{definition}[Holomorphic source seed field]
\label{def:holo-source-spec}
The source seed field on $(f, z_0, r)$ records the intended constraint $\mathsf{HoloOnDisk}(f, z_0, r)$ together with its open-disk gap data.
\end{definition}
\leanchecked{BEDC.Derived.HolomorphicUp.HolomorphicSourceSpec\_semantic\_name\_certificate}

\begin{definition}[Holomorphic pattern seed field]
\label{def:holo-pattern-spec}
The pattern seed field is the uniform-modulus complex-differentiability predicate over the open disk.
\end{definition}
\leanchecked{BEDC.Derived.HolomorphicUp.HolomorphicPatternSpec\_semantic\_name\_certificate}

\begin{definition}[Holomorphic classifier seed field]
\label{def:holo-classifier-spec}
The classifier seed field groups two holomorphic-on-disk witnesses when they restrict to $\hsame$-equivalent pointwise functions on the disk's interior and carry compatible disk data.
\end{definition}
\leanchecked{BEDC.Derived.HolomorphicUp.HolomorphicClassifierSpec\_semantic\_name\_certificate}

\begin{definition}[Holomorphic stability seed field]
\label{def:holo-stability-cert}
The stability seed field lists the required $\hsame$-respect of $\mathsf{HoloOnDisk}$ on the input function and on the disk parameters, transport of moduli under $\hsame$, and compatibility with the iterated-derivative recursion.
\end{definition}

\begin{definition}[Holomorphic ledger seed field]
\label{def:holo-ledger-policy}
The ledger seed field records the originating function history, center, radius, disk-gap witnesses, modulus witnesses, closure witnesses, Taylor-data witnesses, Cauchy-bound witnesses, and identity-theorem density witnesses.
\end{definition}

\begin{definition}[\texorpdfstring{$\HolomorphicUp$}{Holomorphic-up} certificate seed]
\label{def:holo-certificate-seed}
The public status of this chapter is $\text{Seed}$. The fields in this chapter specify the intended carrier, classifier, stability, and ledger data for $\HolomorphicUp$, while \autoref{def:holo-core-theorem-obligations} records the theorem obligations that must be supplied before the object can be treated as a paper certificate or a machine-checked certificate.
\end{definition}
