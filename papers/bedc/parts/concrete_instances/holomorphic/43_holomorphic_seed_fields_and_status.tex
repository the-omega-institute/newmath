\section{The seed fields}

\begin{definition}[Holomorphic source seed field]
\label{def:holo-source-spec}
The source seed field on $(f, z_0, r)$ records the intended constraint $\mathsf{HoloOnDisk}(f, z_0, r)$ together with its open-disk gap data.
\end{definition}
\leanchecked{BEDC.Derived.HolomorphicUp.HolomorphicSourceSpec\_semantic\_name\_certificate}

\begin{definition}[Holomorphic pattern seed field]
\label{def:holo-pattern-spec}
The pattern seed field is the uniform-modulus complex-differentiability predicate over the open disk.
\end{definition}
\leanchecked{BEDC.Derived.HolomorphicUp.HolomorphicPatternSpec\_semantic\_name\_certificate}

\begin{definition}[Holomorphic classifier seed field]
\label{def:holo-classifier-spec}
The classifier seed field groups two holomorphic-on-disk witnesses when they restrict to $\hsame$-equivalent pointwise functions on the disk's interior and carry compatible disk data.
\end{definition}
\leanchecked{BEDC.Derived.HolomorphicUp.HolomorphicClassifierSpec\_semantic\_name\_certificate}

\begin{definition}[Holomorphic stability seed field]
\label{def:holo-stability-cert}
The stability seed field lists the required $\hsame$-respect of $\mathsf{HoloOnDisk}$ on the input function and on the disk parameters, transport of moduli under $\hsame$, and compatibility with the iterated-derivative recursion.
\end{definition}
\leanchecked{BEDC.Derived.HolomorphicUp.HolomorphicStabilityCert\_openDisk\_transport\_fields}
\leanchecked{BEDC.Derived.HolomorphicUp.HolomorphicPatternSpec\_hsame\_iterate\_transport}
\leanchecked{BEDC.Derived.HolomorphicUp.HolomorphicOpenDisk\_stability\_certificate\_fields}
\leanchecked{BEDC.Derived.HolomorphicUp.HolomorphicOpenDiskCarrier\_stability\_certificate}
\leanchecked{BEDC.Derived.HolomorphicUp.HolomorphicStabilityCert\_suffix\_gap\_comm}
\leanchecked{BEDC.Derived.HolomorphicUp.HolomorphicStabilityCert\_radius\_extension\_hsame\_transport}

\begin{definition}[Holomorphic ledger seed field]
\label{def:holo-ledger-policy}
The ledger seed field records the originating function history, center, radius, disk-gap witnesses, modulus witnesses, closure witnesses, Taylor-data witnesses, Cauchy-bound witnesses, and identity-theorem density witnesses.
\end{definition}
\leanchecked{BEDC.Derived.HolomorphicUp.HolomorphicLedgerPolicy\_extension\_gap\_no\_zero\_head}

\begin{definition}[\texorpdfstring{$\HolomorphicUp$}{Holomorphic-up} obligation surface]
\label{def:holo-certificate-seed}
The public status of this chapter is $\scopedClosure$. The fields in this chapter bind the carrier, classifier, stability, ledger, open-disk, and iterated-differentiability data for $\HolomorphicUp$ to the paper-side scope witnessed by the sibling certificate files.
\end{definition}
