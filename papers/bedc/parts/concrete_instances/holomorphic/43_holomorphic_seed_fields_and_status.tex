\section{The seed fields}

\begin{definition}[Holomorphic source seed field]
\label{def:holo-source-spec}
The source seed field on $(f, z_0, r)$ records the intended constraint $\mathsf{HoloOnDisk}(f, z_0, r)$ together with its open-disk gap data.
\end{definition}
\leanchecked{BEDC.Derived.HolomorphicUp.HolomorphicSourceSpec\_semantic\_name\_certificate}

\begin{definition}[Holomorphic pattern seed field]
\label{def:holo-pattern-spec}
The pattern seed field is the uniform-modulus complex-differentiability predicate over the open disk.
\end{definition}
\leanchecked{BEDC.Derived.HolomorphicUp.HolomorphicPatternSpec\_semantic\_name\_certificate}

\begin{definition}[Holomorphic classifier seed field]
\label{def:holo-classifier-spec}
The classifier seed field groups two holomorphic-on-disk witnesses when they restrict to $\hsame$-equivalent pointwise functions on the disk's interior and carry compatible disk data.
\end{definition}
\leanchecked{BEDC.Derived.HolomorphicUp.HolomorphicClassifierSpec\_semantic\_name\_certificate}

\begin{definition}[Holomorphic stability seed field]
\label{def:holo-stability-cert}
The stability seed field lists the required $\hsame$-respect of $\mathsf{HoloOnDisk}$ on the input function and on the disk parameters, transport of moduli under $\hsame$, and compatibility with the iterated-derivative recursion.
\end{definition}

\begin{definition}[Holomorphic ledger seed field]
\label{def:holo-ledger-policy}
The ledger seed field records the originating function history, center, radius, disk-gap witnesses, modulus witnesses, closure witnesses, Taylor-data witnesses, Cauchy-bound witnesses, and identity-theorem density witnesses.
\end{definition}

\begin{definition}[\texorpdfstring{$\HolomorphicUp$}{Holomorphic-up} obligation surface]
\label{def:holo-certificate-seed}
The public status of this chapter is $\publicClosure$. The fields in this chapter bind the carrier, classifier, stability, ledger, open-disk, and iterated-differentiability data for $\HolomorphicUp$ to the paper-side scope witnessed by the sibling certificate files.
\end{definition}

\begin{theorem}[HolomorphicUp public NameCert export]
\label{thm:holomorphic-public-namecert-export}
The fields of \autoref{def:holo-source-spec},
\autoref{def:holo-pattern-spec}, \autoref{def:holo-classifier-spec},
\autoref{def:holo-stability-cert}, \autoref{def:holo-ledger-policy}, and
\autoref{def:holo-core-theorem-obligations} assemble the public
$\NameCert_{\HolomorphicUp}$ export over $\mathsf{BHist}$,
$\UnaryHistory$, $\Cont$, $\hsame$, $\Pkg$, and the open-disk gap rows. The
exported surface consists exactly of the holomorphic-on-disk source row,
uniform complex-differentiability pattern row, disk-restricted classifier row,
transport and iterated-derivative stability rows, ledger provenance rows, and
the core theorem obligations named by those fields.
\end{theorem}
\begin{proof}
Project the source, pattern, classifier, stability, and ledger fields from the
five seed definitions. Each field is already stated over the displayed
$\mathsf{BHist}$, $\UnaryHistory$, $\Cont$, $\hsame$, $\Pkg$, and open-disk gap
surface.

Add the core theorem obligations from
\autoref{def:holo-core-theorem-obligations}. These obligations name the closure
and readback rows that the public certificate may expose, without adding
Cauchy integral theory, analytic continuation, or a standard bridge.

Repack the projected rows as the public $\NameCert_{\HolomorphicUp}$ export.
The packet is public because every consumer-visible row is named by one of the
displayed fields and every transport row remains inside the same BEDC
open-disk surface.
\end{proof}
