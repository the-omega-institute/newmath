\section{Open-disk carrier}

\begin{definition}[Open disk predicate]
\label{def:open-disk}
For complex history \(z_0\), unary radius \(r\), complex history \(z\), and unary gap \(g\), the open-disk predicate \(\mathsf{OpenDisk}(z_0, r, z, g)\) holds when the rational representation of the modulus \(|z - z_0|\) is strictly less than \(r\) under \(\mathsf{NatUnaryStrictPrefix}\). We write \(\mathsf{OpenDisk}(z_0,r,z)\) when the gap witness is existentially understood. The BEDC kernel anchor records the four histories as unary and records the boundary ledger by \(\Cont(z,g,r)\).
\end{definition}

\begin{lemma}[Open disk carriers are unary]
\label{lem:open-disk-unary-witness}
For every $(z_0, r, z)$ with $\mathsf{OpenDisk}(z_0, r, z)$, the gap modulus $\mathsf{Gap}(z_0, r, z)$ is a unary history giving an explicit positive distance from $z$ to the boundary of the disk.
\end{lemma}
\begin{proof}
The strict-prefix witness immediately gives the gap as the difference history.
\end{proof}

\begin{lemma}[Open disk gap transport certificate]
\label{lem:open-disk-gap-transport-certificate}
If $\mathsf{HolomorphicOpenDiskGap}(z_0,r,z,g)$ holds and
$z_0\hsame z_0'$, $r\hsame r'$, $z\hsame z'$, and $g\hsame g'$,
then $\mathsf{HolomorphicOpenDiskGap}(z_0',r',z',g')$ holds, $g'$ is
unary, and $\Cont(z',g',r')$ holds.
\end{lemma}
\begin{proof}
Transport the four unary carriers along the four $\hsame$ witnesses.

The boundary continuation ledger transports along the point, gap, and radius
$\hsame$ witnesses, giving $\Cont(z',g',r')$. Repack these transported fields
as the target open-disk gap certificate, retaining the unary witness for $g'$.
\end{proof}

\begin{theorem}[Open-disk gap zero-headed component impossibility]
\label{thm:holomorphic-open-disk-gap-zero-headed-component-absurd}
For histories \(z_0,r,z,g\), an open-disk gap witness has no zero-headed
center, radius, point, or gap component:
$$
\mathsf{HolomorphicOpenDiskGap}(z_0,r,z,g)
\Longrightarrow
\left(\exists w,\ z_0=\Ezero(w)\right)\lor
\left(\exists w,\ r=\Ezero(w)\right)\lor
\left(\exists w,\ z=\Ezero(w)\right)\lor
\left(\exists w,\ g=\Ezero(w)\right)
\Longrightarrow
\mathsf{False}.
$$
\end{theorem}
\begin{proof}
Unpack the open-disk gap witness into its four unary-history fields. Rewriting
any component as a zero-headed history turns the corresponding unary witness
into a unary zero-extension, contradicting unary zero-extension separation.
\end{proof}

\begin{theorem}[Open disk zero-headed component impossibility]

\end{theorem}
\begin{proof}

contradicting complex component endpoint separation.
\end{proof}
