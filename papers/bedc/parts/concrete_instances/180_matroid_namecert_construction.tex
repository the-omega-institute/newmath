\begin{definition}[FinSetUp subset and intersection over one ground]
\label{def:matroid-finset-subset-and-intersection}
Fix a ground predicate $E:\mathsf{BHist}\to\mathsf{Prop}$ with classifier $\sim_E$. For predicates $U,V,J:\mathsf{BHist}\to\mathsf{Prop}$, define finite-set subset over this ground by
$$
\mathsf{FinSetSubset}_{E,\sim_E}(U,V)
\Longleftrightarrow
\exists xs:\mathsf{HistSpine}.\;
\mathsf{FinSetSpineEnumerates}_{E,\sim_E}(xs,U)
\land
\forall z:\mathsf{BHist}.\;U(z)\to V(z).
$$
Define the $\FinSetUp$ intersection predicate by
$$
\mathsf{FinSetIntersection}_{E,\sim_E}(J;U,V)
\Longleftrightarrow
\exists zs:\mathsf{HistSpine}.\;
\mathsf{FinSetSpineEnumerates}_{E,\sim_E}(zs,J)
\land
\forall z:\mathsf{BHist}.\;J(z)\Longleftrightarrow U(z)\land V(z).
$$
Thus the intersection row is an exact finite-spine presentation of pointwise membership conjunction in the common ground.
\end{definition}

\begin{definition}[MatroidUp independent-set certificate surface]
\label{def:matroid-independence-certificate-surface}
A $\MatroidUp$ independent-set certificate over $(E,\sim_E)$ consists of a finite-set predicate family
$$
\mathsf{Ind}_M:(\mathsf{BHist}\to\mathsf{Prop})\to\mathsf{Prop}
$$
and the rows
$$
\begin{gathered}
\mathsf{finite}_M:
\forall I.\;\mathsf{Ind}_M(I)\to
\exists xs:\mathsf{HistSpine}.\;\mathsf{FinSetSpineEnumerates}_{E,\sim_E}(xs,I),\\
\mathsf{ground}_M:
\forall I\,z.\;\mathsf{Ind}_M(I)\to I(z)\to E(z),\\
\mathsf{empty}_M:\mathsf{Ind}_M(\lambda z.\;\bot),\\
\mathsf{hereditary}_M:
\forall I\,J.\;\mathsf{Ind}_M(I)\to
\mathsf{FinSetSubset}_{E,\sim_E}(J,I)\to
\mathsf{Ind}_M(J).
\end{gathered}
$$
The exchange row is the certified finite-spine exchange clause on these same finite predicates: independent finite sets with certified smaller cardinal admit a carried element of the larger set whose $\FinSetUp$ insertion into the smaller set remains independent. The hereditary row is the subset-closure field named in the $\MatroidUp$ certificate paragraph of \autoref{sec:matroid-certificate}.
\end{definition}

\begin{lemma}[FinSetUp intersection projects to the left factor]
\label{lem:matroid-finset-intersection-left-subset}
Fix a ground predicate $E:\mathsf{BHist}\to\mathsf{Prop}$ with classifier $\sim_E$. For predicates $I,K,J:\mathsf{BHist}\to\mathsf{Prop}$, if
$$
\mathsf{FinSetIntersection}_{E,\sim_E}(J;I,K)
$$
holds, then
$$
\mathsf{FinSetSubset}_{E,\sim_E}(J,I)
$$
holds. This is the finite-set form of the left lower-bound role for meet in \autoref{def:lattice-stability-certificate}, read here from the exact $\FinSetUp$ membership row.
\end{lemma}
\begin{proof}
Unpack \autoref{def:matroid-finset-subset-and-intersection}. The intersection hypothesis supplies a finite spine $zs$ with
$$
\mathsf{FinSetSpineEnumerates}_{E,\sim_E}(zs,J)
$$
and a pointwise equivalence
$$
\forall z:\mathsf{BHist}.\;J(z)\Longleftrightarrow I(z)\land K(z).
$$
To prove $\mathsf{FinSetSubset}_{E,\sim_E}(J,I)$, keep the same finite spine $zs$ as the finite-set witness for $J$. It remains only to prove the inclusion row. Fix $z:\mathsf{BHist}$ and assume $J(z)$. The forward half of the displayed pointwise equivalence gives $I(z)\land K(z)$, and projecting its first component gives $I(z)$. Thus every member of $J$ is a member of $I$, with the finite-spine witness already supplied.
\end{proof}

\begin{theorem}[Matroid intersection preserves independence]
\label{thm:matroid-intersection-preserves-independence}
Fix a $\MatroidUp$ independent-set certificate over a ground predicate $E:\mathsf{BHist}\to\mathsf{Prop}$ with classifier $\sim_E$, with independent-family row $\mathsf{Ind}_M$ as in \autoref{def:matroid-independence-certificate-surface}. Let $I,K,J:\mathsf{BHist}\to\mathsf{Prop}$ be finite-set predicates over the same ground. Suppose
$$
\mathsf{Ind}_M(I)
$$
and suppose there are finite spines $xs$ and $ys$ such that
$$
\mathsf{FinSetSpineEnumerates}_{E,\sim_E}(xs,I)
\qquad\text{and}\qquad
\mathsf{FinSetSpineEnumerates}_{E,\sim_E}(ys,K).
$$
If $J$ is the $\FinSetUp$ intersection of $I$ with $K$,
$$
\mathsf{FinSetIntersection}_{E,\sim_E}(J;I,K),
$$
then
$$
\mathsf{Ind}_M(J).
$$
\end{theorem}
\begin{proof}
The finite-spine witnesses $xs$ and $ys$ record that $I$ and $K$ are finite predicates over the same ground $(E,\sim_E)$ in the sense of \autoref{def:finset-spine-enumerates-predicate}. The independence hypothesis gives the $\MatroidUp$ independent set $I$.

Apply \autoref{lem:matroid-finset-intersection-left-subset} to the displayed intersection row. This gives
$$
\mathsf{FinSetSubset}_{E,\sim_E}(J,I).
$$
The hereditary field $\mathsf{hereditary}_M$ of the $\MatroidUp$ certificate in \autoref{def:matroid-independence-certificate-surface} applies to $\mathsf{Ind}_M(I)$ and this finite-set subset row, and yields
$$
\mathsf{Ind}_M(J).
$$
The exchange row and all cardinal comparison data are irrelevant for this closure step; independence is preserved solely by the certified subset-closure field.
\end{proof}
