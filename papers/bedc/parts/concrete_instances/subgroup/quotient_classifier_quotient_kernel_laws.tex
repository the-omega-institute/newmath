\begin{theorem}[Centralizer quotient-kernel centralizer endpoint invariance]
\label{thm:subgroup-centralizer-quotient-kernel-centralizer-endpoint-invariance}
Under the same BEDC history-level $\GroupUp$ laws, if $Z_a(c)$ and $Z_a(d)$,
then
$$
Q_{Z_a\triangleleft N_a}(\mathsf{mul}(x,c),\mathsf{mul}(y,d))
\Longleftrightarrow Q_{Z_a\triangleleft N_a}(x,y).
$$
\end{theorem}
\begin{proof}
For the direction from the original endpoints to the translated endpoints,
apply
\autoref{thm:subgroup-centralizer-quotient-kernel-two-sided-centralizer-endpoint-mul-closed}
with the endpoint factors $c$ and $d$.

For the direction from the translated endpoints back to the original
endpoints, use inverse closure for $Z_a$ from
\autoref{thm:subgroup-centralizer-normalizer-tower-certificate} to obtain
$Z_a(\mathsf{inv}(c))$ and $Z_a(\mathsf{inv}(d))$. Apply
\autoref{thm:subgroup-centralizer-quotient-kernel-two-sided-centralizer-endpoint-mul-closed}
to the translated quotient-kernel relation with these inverse factors.

The resulting relation has endpoints
$\mathsf{mul}(\mathsf{mul}(x,c),\mathsf{inv}(c))$ and
$\mathsf{mul}(\mathsf{mul}(y,d),\mathsf{inv}(d))$. Associativity, the
two-sided inverse laws, the unit laws, and multiplication congruence compare
these histories with $x$ and $y$ up to $\hsame$. The transport theorem
\autoref{thm:subgroup-centralizer-quotient-kernel-hsame-transport} then moves
the quotient-kernel witness back to $Q_{Z_a\triangleleft N_a}(x,y)$.
\end{proof}

\begin{theorem}[Centralizer normalizer quotient-kernel transitivity]
\label{thm:subgroup-centralizer-quotient-kernel-transitivity}
For a BEDC history-level $\GroupUp$ operation over $\mathsf{BHist}$ satisfying
the same associativity, unit, multiplication-congruence, and two-sided inverse
laws as in
\autoref{thm:subgroup-centralizer-normalizer-tower-certificate}, the relation
$Q_{Z_a\triangleleft N_a}$ of
\autoref{def:subgroup-centralizer-normalizer-quotient-kernel-relation} is
transitive:
$$
Q_{Z_a\triangleleft N_a}(x,y)\land Q_{Z_a\triangleleft N_a}(y,z)
\Longrightarrow Q_{Z_a\triangleleft N_a}(x,z).
$$
\end{theorem}
\begin{proof}
Expand the two hypotheses. Their endpoint rows provide the required
$N_a(x)$ and $N_a(z)$ clauses. Their kernel rows give
$$
Z_a(\mathsf{mul}(\mathsf{inv}(x),y))
\qquad\text{and}\qquad
Z_a(\mathsf{mul}(\mathsf{inv}(y),z)).
$$
Product closure for $Z_a$ gives a centralizer witness for the product of these
two relative kernel words. Associativity rewrites that product to
$$
\mathsf{mul}(\mathsf{inv}(x),
  \mathsf{mul}(y,\mathsf{mul}(\mathsf{inv}(y),z))),
$$
and the right inverse law for $y$ followed by the left unit law cancels the
middle $y,\mathsf{inv}(y)$ pair. The carrier transport row for $Z_a$ moves the
product witness to
$Z_a(\mathsf{mul}(\mathsf{inv}(x),z))$, which is exactly the kernel clause for
$Q_{Z_a\triangleleft N_a}(x,z)$.
\end{proof}

inverse closure gives $Z_a(\mathsf{inv}(x))$, and the right unit law transports
that witness to the displayed kernel word.

\begin{theorem}[Append centralizer carrier unary word]
\label{thm:subgroup-centralizer-carrier-append-e1-unit-unary}
For the concrete BEDC append operation, centralizing the visible singleton
$\Eone(\emp)$ forces the carried word to be unary:
$$
Z_{\Eone(\emp)}^{\mathsf{append}}(w)
\Longrightarrow
\mathsf{UnaryHistory}(w).

\end{theorem}
\begin{proof}
Unfold the centralizer carrier as the displayed comparison
$\mathsf{append}(w,\Eone(\emp))\hsame
\mathsf{append}(\Eone(\emp),w)$. Induct on the concrete history $w$. The
empty word is unary. A zero-headed word displays opposite visible constructors
on the two sides, contradicting constructor separation. A one-headed word has
matching $\Eone$ constructors; injectivity descends the same centralizer
comparison to the tail, and the induction hypothesis gives unary evidence.

\begin{theorem}[Centralizer normalizer quotient-kernel equivalence laws]
\label{thm:subgroup-centralizer-normalizer-quotient-kernel-congruence}
For a BEDC history-level $\GroupUp$ operation over $\mathsf{BHist}$ satisfying
the same associativity, unit, multiplication-congruence, and two-sided inverse
laws as in
\autoref{thm:subgroup-centralizer-normalizer-tower-certificate}, the relation
$Q_{Z_a\triangleleft N_a}$ of
\autoref{def:subgroup-centralizer-normalizer-quotient-kernel-relation} is a
quotient-kernel equivalence relation on the upper carrier:
\begin{enumerate}
\item if $N_a(x)$, then $Q_{Z_a\triangleleft N_a}(x,x)$;
\item if $Q_{Z_a\triangleleft N_a}(x,y)$, then
$Q_{Z_a\triangleleft N_a}(y,x)$;
\item if $Q_{Z_a\triangleleft N_a}(x,y)$ and
$Q_{Z_a\triangleleft N_a}(y,z)$, then
$Q_{Z_a\triangleleft N_a}(x,z)$;
\end{enumerate}
\end{theorem}
\begin{proof}
Reflexivity uses the identity row of
\autoref{thm:subgroup-centralizer-normalizer-tower-certificate}. From
$N_a(x)$ the endpoint clauses are immediate, and the two-sided inverse law
identifies $\mathsf{mul}(\mathsf{inv}(x),x)$ with $e$. Transport along
$\hsame$ carries the centralizer identity witness $Z_a(e)$ to the required
kernel witness.

For symmetry, expand $Q_{Z_a\triangleleft N_a}(x,y)$. The endpoint clauses
swap because both $x$ and $y$ already lie in $N_a$. The kernel witness
$Z_a(\mathsf{mul}(\mathsf{inv}(x),y))$ is inverted by the inverse-closure row
for $Z_a$, and inverse-product reversal identifies the inverse word with
$\mathsf{mul}(\mathsf{inv}(y),x)$ after the ambient inverse laws. The
$\hsame$-transport row for $Z_a$ gives the displayed symmetric witness.

For transitivity, take kernel witnesses
$Z_a(\mathsf{mul}(\mathsf{inv}(x),y))$ and
$Z_a(\mathsf{mul}(\mathsf{inv}(y),z))$. Product closure in $Z_a$ gives a
witness for their product. Associativity and the inverse laws cancel the
middle factors $y$ and $\mathsf{inv}(y)$, and transport turns the product
witness into $Z_a(\mathsf{mul}(\mathsf{inv}(x),z))$. The endpoint clauses are
the retained $N_a(x)$ and $N_a(z)$ clauses.
\end{proof}
