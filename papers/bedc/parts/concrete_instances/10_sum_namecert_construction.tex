\chapter{A Concrete Naming Certificate for \(\SumUp\)}
\label{ch:concrete-instances-sum-namecert}

The disjoint sum interface \(\SumUp\) is licensed over two source carriers each endowed with a \(\NameCert\). The carrier is the inductive sum with left and right constructors wrapping a witness of either source; the classifier is the constructor-discriminating relation that within each constructor falls back to the corresponding source classifier. It is not a primitive coproduct; it is licensed only as a derived constructor over the two sources.

\section{Roadmap template status}

\begin{enumerate}
  \item \textbf{Traditional target}: disjoint sum data with left and right tagged injections.
  \item \textbf{BEDC source}: two licensed source carriers $A$ and $B$, each with its own name certificate.
  \item \textbf{Carrier choice}: tagged left carriers use $e0$ histories and tagged right carriers use $e1$ histories.
  \item \textbf{Classifier choice}: the sum classifier relates left tags through the $A$ classifier, right tags through the $B$ classifier, and excludes mixed tags.
  \item \textbf{Stability obligations}: tagged carrier transport, left/right inversion, mixed-tag exclusion, classifier symmetry, classifier transitivity, and endpoint transport.
  \item \textbf{Ledger}: the tag is visible data; payload evidence is retained from the corresponding source certificate.
  \item \textbf{Core theorem list}: tagged injections, carrier transport, left and right inversions, classifier inversion, mixed-tag exclusion, and the semantic name certificate.
  \item \textbf{Exactness predicate}: classification holds exactly for same-tag payloads classified by the matching source classifier.
  \item \textbf{Standard bridge}: tagged histories read back as ordinary $\mathsf{inl}(a)$ and $\mathsf{inr}(b)$ values.
  \item \textbf{Status}: $\mathsf{CheckedCert}$, with the sum history semantic certificate checked in Lean.
\end{enumerate}

\section{Carrier definition}

\begin{definition}[Sum carrier]
\label{def:sum-carrier}
For licensed source carriers \(A\) and \(B\), define an inductive carrier with constructors \(\mathsf{inl}(a)\) and \(\mathsf{inr}(b)\).
\end{definition}
\leandef{BEDC.Derived.SumUp.SumCarrier}
\leandef{BEDC.Derived.SumUp.SumHistoryCarrier}

\begin{theorem}[Sum history carrier transport]
\label{thm:sum-history-carrier-hsame-transport}
If $h$ and $k$ are internally the same BEDC histories, then membership of $h$ in the tagged sum-history carrier transfers to membership of $k$ in the same carrier.
\end{theorem}
\begin{proof}
Invert the sum carrier witness. In the left-tag case, compose symmetry of the given $\hsame(h,k)$ witness with the stored $\hsame(h,e0(l))$ tag witness, and keep the same left source-carrier proof. The right-tag case is identical with $e1(r)$.
\end{proof}
\leanchecked{BEDC.Derived.SumUp.SumHistoryCarrier\_hsame\_transport}

\begin{theorem}[Sum history left-tag inversion]
\label{thm:sum-history-carrier-left-tag-inversion}
If an \(e0\)-tagged history belongs to the tagged sum-history carrier, then
its payload is history-same to a left carrier witness.
\end{theorem}
\begin{proof}
Invert the tagged carrier witness. In the left branch, BEDC \(e0\)-constructor
inversion recovers the payload sameness and keeps the stored left carrier
proof. In the right branch, the \(e0/e1\) no-confusion theorem contradicts the
stored tag witness.
\end{proof}
\leanchecked{BEDC.Derived.SumUp.SumHistoryCarrier\_e0\_inversion}

\begin{theorem}[Sum history right-tag inversion]
\label{thm:sum-history-carrier-e1-inversion}
If an $e1$-tagged history belongs to the tagged sum-history carrier, then
its payload is history-same to a right carrier witness.
\end{theorem}
\begin{proof}
Invert the tagged carrier witness. In the left branch, the $e1/e0$
no-confusion theorem contradicts the stored tag witness. In the right branch,
BEDC $e1$-constructor inversion recovers the payload sameness and keeps the
stored right carrier proof.
\end{proof}
\leanchecked{BEDC.Derived.SumUp.SumHistoryCarrier\_e1\_inversion}

\section{The certificate}

\begin{definition}[Sum source specification]
\label{def:sum-source-specification}
The source is the disjoint union of the two source certificates.
\end{definition}

\begin{definition}[Sum pattern specification]
\label{def:sum-pattern-specification}
The pattern is left-or-right injection from the corresponding source witness.
\end{definition}
\leanchecked{BEDC.Derived.SumUp.SumHistoryCarrier\_tagged\_injections}

\begin{definition}[Sum classifier specification]
\label{def:sum-classifier-specification}
The classifier relates left values through the \(A\)-classifier, right values through the \(B\)-classifier, and never relates opposite constructors.
\end{definition}
\leandef{BEDC.Derived.SumUp.SumHistoryClassifier}
\leandef{BEDC.Derived.SumUp.SumClassifierSpec}
\leanchecked{BEDC.Derived.SumUp.sum\_classifier\_inversion}

\begin{theorem}[Sum history classifier left-tag inversion]
\label{thm:sum-history-classifier-left-hsame-inversion}
When the history-level sum classifier uses \(\hsame\) on both components and
relates two \(e0\)-tagged histories, the two left payload histories are
internally the same.
\end{theorem}
\begin{proof}
Invert the tagged classifier witness. The left branch gives two \(e0\)-tag
witnesses and a stored component \(\hsame\) witness; constructor inversion for
\(e0\) composes them into payload sameness. The right branch contradicts
constructor separation between \(e0\) and \(e1\).
\end{proof}
\leanchecked{BEDC.Derived.SumUp.SumHistoryClassifier\_left\_hsame\_inversion}

\begin{theorem}[Sum history classifier right-tag inversion]
\label{thm:sum-history-classifier-right-hsame-inversion}
When the history-level sum classifier uses \(\hsame\) on both components and
relates two \(e1\)-tagged histories, the two right payload histories are
internally the same.
\end{theorem}
\begin{proof}
Invert the tagged classifier witness. The left branch contradicts constructor
separation between \(e1\) and \(e0\). The right branch gives two \(e1\)-tag
witnesses and a stored component \(\hsame\) witness; constructor inversion for
\(e1\) composes them into payload sameness.
\end{proof}
\leanchecked{BEDC.Derived.SumUp.SumHistoryClassifier\_right\_hsame\_inversion}

\begin{theorem}[Sum history classifier symmetry under history sameness]
\label{thm:sum-history-classifier-hsame-symmetry}
When both component classifiers are \(\hsame\), the history-level sum classifier is symmetric.
\end{theorem}
\leanchecked{BEDC.Derived.SumUp.SumHistoryClassifier\_hsame\_symm}
\begin{proof}
Invert the classifier witness. In the left-tagged case, swap the two tagged witnesses and use symmetry of \(\hsame\) on the left component. The right-tagged case is identical with the right component.
\end{proof}

\begin{theorem}[Sum history classifier history transport]
\label{thm:sum-history-classifier-hsame-transport}
If both endpoints of a history-level \(\SumUp\) classifier are transported by
\(\hsame\), then the same classifier witness transports to the transported
endpoints.
\end{theorem}
\begin{proof}
Invert the tagged classifier witness. In the left branch, keep the same left
component witnesses and compose the reversed endpoint transports with the two
stored \(e0\)-tag witnesses. The right branch is identical using the \(e1\)-tag
witnesses.
\end{proof}
\leanchecked{BEDC.Derived.SumUp.SumHistoryClassifier\_hsame\_transport}

\begin{theorem}[Sum history classifier tag separation]
\label{thm:sum-history-classifier-tag-separation}
If \(h\) is \(e0\)-tagged and \(k\) is \(e1\)-tagged, then the
history-level \(\SumUp\) classifier cannot relate \(h\) to \(k\).
\end{theorem}
\begin{proof}
Invert the classifier witness. In the left branch \(k\) is also
\(e0\)-tagged, contradicting its \(e1\)-tag witness. In the right branch
\(h\) is \(e1\)-tagged, contradicting its \(e0\)-tag witness.
\end{proof}
\leanchecked{BEDC.Derived.SumUp.SumHistoryClassifier\_mixed\_tags\_absurd}

\begin{theorem}[Sum history classifier transitivity]
\label{thm:sum-history-classifier-transitivity}
If the left and right source classifiers are transitive, then the
history-level \(\SumUp\) classifier is transitive.
\end{theorem}
\begin{proof}
Invert both classifier witnesses. Matching left--left and right--right branches
identify the shared tagged middle history by BEDC constructor inversion and
compose the corresponding source-classifier witnesses. Mixed branches
contradict the \(e0/e1\) no-confusion rule.
\end{proof}
\leanchecked{BEDC.Derived.SumUp.SumHistoryClassifier\_trans}

\begin{definition}[Sum stability certificate]
\label{def:sum-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item lifted reflexivity on left and right constructors
  \item lifted transitivity within each constructor
  \item constructor separation between left and right
  \item closure under both injections
\end{enumerate}
\end{definition}
\leanchecked{BEDC.Derived.SumUp.sum\_stability\_certificate\_fields}

\begin{theorem}[Sum classifier transitivity at the typed level]
\label{thm:sum-classifier-transitivity}
Let $R_A : A \to A \to \mathsf{Prop}$ and
$R_B : B \to B \to \mathsf{Prop}$ be source classifier relations.  If both
$R_A$ and $R_B$ are transitive, then for sum carriers $x$, $y$, and $z$,
$$
  \mathsf{SumClassifierSpec}(R_A,R_B,x,y)
  \quad\text{and}\quad
  \mathsf{SumClassifierSpec}(R_A,R_B,y,z)
$$
imply $\mathsf{SumClassifierSpec}(R_A,R_B,x,z)$.
\end{theorem}
\leanchecked{BEDC.Derived.SumUp.SumClassifierSpec\_trans}
\begin{proof}
Invert the sum classifier witnesses by their constructors. Matching left and
right branches compose through the corresponding source transitivity assumption,
while mixed tags are impossible by the classifier clauses.
\end{proof}

\begin{definition}[Sum ledger policy]
\label{def:sum-ledger-policy}
Record a raw sum history carrying the selected injection tag and source ledger
together with an \(\hsame\) link to the visible history exposed by the ledger.
\end{definition}
\leandef{BEDC.Derived.SumUp.SumHistoryLedgerPolicy}

\begin{theorem}[Sum ledger visible carrier]
\label{thm:sum-ledger-visible-carrier}
If a visible history is linked by \(\hsame\) to a raw sum-ledger history, then
the visible history is itself a sum carrier.
\end{theorem}
\begin{proof}
Unpack the raw sum carrier and transport it across the stored \(\hsame\) link
by sum-carrier transport.
\end{proof}
\leanchecked{BEDC.Derived.SumUp.SumHistoryLedgerPolicy\_visible\_carrier}

\begin{theorem}[Sum ledger raw-visible classifier]
\label{thm:sum-ledger-raw-visible-classifier}
If $\mathsf{SumHistoryLedgerPolicy}(L,R,\rho,v)$ holds, then
$\mathsf{SumHistoryClassifier}(L,R,\hsame,\hsame,\rho,v)$ holds.
\end{theorem}
\begin{proof}
Unpack the ledger into its raw sum carrier and the stored $\hsame(\rho,v)$
link. The raw carrier determines whether $\rho$ is left-tagged or right-tagged;
transport the tag witness across the stored link and use reflexivity of
$\hsame$ on the selected component to build the matching classifier branch.
\end{proof}
\leanchecked{BEDC.Derived.SumUp.SumHistoryLedgerPolicy\_raw\_visible\_classifier}

\begin{theorem}[Concrete $\SumUp$ semantic name certificate]
\label{thm:concrete-sum-namecert}
Let $Left,Right:\Hist\to\mathsf{Prop}$ be source carriers with classifiers $LeftEq$ and $RightEq$. If $\NameCert(Left,LeftEq)$ and $\NameCert(Right,RightEq)$ are supplied, then the tagged sum carrier, the tagged sum classifier, and the carrier-as-pattern and carrier-as-ledger tuple form a $\mathsf{SemanticNameCert}$ for $\mathsf{SumHistoryCarrier}(Left,Right)$ and $\mathsf{SumHistoryClassifier}(Left,Right,LeftEq,RightEq)$.
\end{theorem}
\begin{proof}
The carrier is represented by $e0$-tagged left histories and $e1$-tagged right histories. The classifier delegates within matching tags to the corresponding source certificate. Reflexivity is built by tagging the source certificate reflexivity field on the selected component. Symmetry and transitivity invert the tagged classifier witnesses, use the matching source certificate fields inside a left--left or right--right branch, and discharge mixed tags by constructor no-confusion. Carrier transport in each branch applies the corresponding source field $\mathsf{NameCert.carrier\_respects\_equiv}$ to the payload classifier witness, then rebuilds the tagged carrier witness. The carrier-as-pattern and carrier-as-ledger components use the same tagged carrier evidence, so the displayed tuple satisfies the semantic certificate fields.
\end{proof}
\leanchecked{BEDC.Derived.SumUp.sum\_history\_semantic\_name\_certificate}

\begin{corollary}[\(\SumUp\) is licensed, not primitive]
\label{cor:sum-licensed-not-primitive}
\(\SumUp\) is licensed as a derived disjoint sum, not as a primitive coproduct.
\end{corollary}
