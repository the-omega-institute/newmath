\chapter{A Concrete Naming Certificate for \(\SumUp\)}
\label{ch:concrete-instances-sum-namecert}

The disjoint sum interface \(\SumUp\) is licensed over two source carriers each endowed with a \(\NameCert\). The carrier is the inductive sum with left and right constructors wrapping source-carrier evidence. The classifier is the constructor-discriminating payload relation: it stores visible tag endpoints together with the matching source classifier witness, while actual source membership remains in $\mathsf{SumHistoryCarrier}$ and is the evidence used by the semantic certificate. It is not a primitive coproduct; it is licensed only as a derived constructor over the two sources.

\section{Roadmap template status}

\begin{enumerate}
  \item \textbf{Traditional target}: disjoint sum data with left and right tagged injections.
  \item \textbf{BEDC source}: two licensed source carriers $A$ and $B$, each with its own name certificate.
  \item \textbf{Carrier choice}: tagged left carriers use $e0$ histories and tagged right carriers use $e1$ histories.
  \item \textbf{Classifier choice}: the sum classifier stores tag endpoints plus the matching $A$ or $B$ classifier witness, and excludes mixed tags; source-carrier evidence is supplied by $\mathsf{SumHistoryCarrier}$.
  \item \textbf{Stability obligations}: tagged carrier transport, source-certified classifier-induced carrier transport, left/right inversion, mixed-tag exclusion, classifier symmetry, classifier transitivity, and endpoint transport.
  \item \textbf{Ledger}: the tag is visible data; payload evidence is retained from the corresponding source certificate.
  \item \textbf{Core theorem list}: tagged injections, $\hsame$ carrier transport, source-certified classifier-induced carrier transport, left and right inversions, classifier inversion, source-parameter irrelevance of the classifier, generic same-tag exactness, carrier-aware same-tag exactness, mixed-tag exclusion, and the semantic name certificate.
  \item \textbf{Exactness predicate}: the left-tag and right-tag iff clauses identify classifier-only payload witnesses and the matching source classifier, while carrier-aware exactness combines $\mathsf{SumHistoryCarrier}$ with explicit carrier transport along $\hsame$.
  \item \textbf{Standard bridge}: tagged histories read back as ordinary $\mathsf{inl}(a)$ and $\mathsf{inr}(b)$ values.
  \item \textbf{Status}: the $\mathsf{SemanticNameCert}$ at Theorem~\ref{thm:concrete-sum-namecert} is $\mathsf{CheckedCert}$; Theorem~\ref{thm:sum-history-classifier-source-same-tag-exactness} is the paper-side matching-source formulation, and Theorem~\ref{thm:sum-history-classifier-generic-same-tag-exactness} records the witness-level same-tag formulation.
\end{enumerate}

\section{Carrier definition}

\begin{definition}[Sum carrier]
\label{def:sum-carrier}
For licensed source carriers \(A\) and \(B\), define an inductive carrier with constructors \(\mathsf{inl}(a)\) and \(\mathsf{inr}(b)\).
\end{definition}

\begin{theorem}[Sum history carrier transport]
\label{thm:sum-history-carrier-hsame-transport}
If $h$ and $k$ are internally the same BEDC histories, then membership of $h$ in the tagged sum-history carrier transfers to membership of $k$ in the same carrier.
\end{theorem}
\begin{proof}
Invert the sum carrier witness. In the left-tag case, compose symmetry of the given $\hsame(h,k)$ witness with the stored $\hsame(h,e0(l))$ tag witness, and keep the same left source-carrier proof. The right-tag case is identical with $e1(r)$.
\end{proof}
\leanchecked{BEDC.Derived.SumUp.SumHistoryCarrier\_hsame\_transport}

\begin{theorem}[Sum history carrier branch exclusivity]
\label{thm:sum-history-carrier-branch-exclusivity}
For source carriers $Left,Right:\mathsf{BHist}\to\mathsf{Prop}$, no BEDC history $h$ has both a left carrier presentation and a right carrier presentation:
$$
\neg\Bigl(
(\exists l:\mathsf{BHist}.\; \hsame(h,\Ezero(l))\land Left(l))
\land
(\exists r:\mathsf{BHist}.\; \hsame(h,\Eone(r))\land Right(r))
\Bigr).
$$
\end{theorem}
\begin{proof}
Assume both presentations. The left witness gives $\hsame(h,\Ezero(l))$, and the right witness gives $\hsame(h,\Eone(r))$. Compose the symmetry of the left endpoint witness with the right endpoint witness to obtain $\hsame(\Ezero(l),\Eone(r))$, then close by BEDC $e0/e1$ no-confusion.
\end{proof}

\begin{theorem}[Sum history left-tag inversion]
\label{thm:sum-history-carrier-left-tag-inversion}
If an \(e0\)-tagged history belongs to the tagged sum-history carrier, then
its payload is history-same to a left carrier witness.
\end{theorem}
\begin{proof}
Invert the tagged carrier witness. In the left branch, BEDC \(e0\)-constructor
inversion recovers the payload sameness and keeps the stored left carrier
proof. In the right branch, the \(e0/e1\) no-confusion theorem contradicts the
stored tag witness.
\end{proof}
\leanchecked{BEDC.Derived.SumUp.SumHistoryCarrier\_e0\_inversion}

\begin{theorem}[Sum history right-tag inversion]
\label{thm:sum-history-carrier-e1-inversion}
If an $e1$-tagged history belongs to the tagged sum-history carrier, then
its payload is history-same to a right carrier witness.
\end{theorem}
\begin{proof}
Invert the tagged carrier witness. In the left branch, the $e1/e0$
no-confusion theorem contradicts the stored tag witness. In the right branch,
BEDC $e1$-constructor inversion recovers the payload sameness and keeps the
stored right carrier proof.
\end{proof}
\leanchecked{BEDC.Derived.SumUp.SumHistoryCarrier\_e1\_inversion}

\begin{theorem}[Sum history carrier visible-tag exactness]
\label{thm:sum-history-carrier-visible-tag-exactness}
For source carriers $Left,Right:\mathsf{BHist}\to\mathsf{Prop}$, the visible
tags of the tagged sum-history carrier are exactly the corresponding source
carrier payloads:
$$
\mathsf{SumHistoryCarrier}(Left,Right,\Ezero(l))
\Longleftrightarrow
\exists a:\mathsf{BHist}.\; Left(a)\land \hsame(l,a),
$$
and
$$
\mathsf{SumHistoryCarrier}(Left,Right,\Eone(r))
\Longleftrightarrow
\exists b:\mathsf{BHist}.\; Right(b)\land \hsame(r,b).
$$
\end{theorem}
\begin{proof}
For the forward implications, apply the checked left-tag and right-tag inversion
theorems to the carrier witness. For the reverse implications, start from the
payload witness, use the checked tagged-injection constructor for the matching
branch, convert the payload endpoint by constructor congruence for $\hsame$,
and transport carrier membership along the resulting history sameness by
$\mathsf{SumHistoryCarrier\_hsame\_transport}$.
\end{proof}

\begin{definition}[Sum classifier specification]
\label{def:sum-classifier-specification}
The classifier records left-tag endpoint payloads through the \(A\)-classifier and right-tag endpoint payloads through the \(B\)-classifier, and never relates opposite constructors. Source membership is carried by $\mathsf{SumHistoryCarrier}$ rather than by this classifier relation.
\end{definition}

\begin{theorem}[Sum classifier does not carry source membership]
\label{thm:sum-classifier-no-source-membership}
Let $\mathsf{FalseCarrier}(h)$ mean $\mathsf{False}$ for every BEDC history $h$. Then
$$
\mathsf{SumHistoryClassifier}(\mathsf{FalseCarrier},\mathsf{FalseCarrier},\hsame,\hsame,\Ezero(\emp),\Ezero(\emp))
$$
holds, while
$$
\neg \mathsf{SumHistoryCarrier}(\mathsf{FalseCarrier},\mathsf{FalseCarrier},\Ezero(\emp)).
$$
\end{theorem}
\begin{proof}
Build the classifier by the left-tag branch, using $\emp$ as both stored left payloads, reflexivity of $\hsame$ for the two visible $\Ezero(\emp)$ endpoints, and reflexivity of $\hsame$ for the payload relation. A carrier witness for $\Ezero(\emp)$ would be either a left branch with some payload satisfying $\mathsf{FalseCarrier}$ or a right branch with some payload satisfying $\mathsf{FalseCarrier}$. Both cases unfold to $\mathsf{False}$, so no carrier witness exists.
\end{proof}

\begin{theorem}[Sum history classifier source-parameter irrelevance]
\label{thm:sum-history-classifier-source-parameter-irrelevance}
For source carriers $Left,Right,LeftAlt,RightAlt:\mathsf{BHist}\to\mathsf{Prop}$,
component classifiers $LeftEq,RightEq:\mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$,
and histories $h,k:\mathsf{BHist}$,
$$
\mathsf{SumHistoryClassifier}(Left,Right,LeftEq,RightEq,h,k)
\Longleftrightarrow
\mathsf{SumHistoryClassifier}(LeftAlt,RightAlt,LeftEq,RightEq,h,k).
$$
\end{theorem}
\begin{proof}
For the forward direction, case-analyze the classifier witness. In the
left-tag branch, keep the same stored left payloads, endpoint tag witnesses,
and $LeftEq$ witness, and build the left branch for the alternate source
carriers. In the right-tag branch, keep the same stored right payloads,
endpoint tag witnesses, and $RightEq$ witness, and build the right branch for
the alternate source carriers. For the reverse direction, exchange the two
source-carrier pairs and repeat the same branch analysis. No source-carrier
proof occurs inside the classifier witness; those proofs belong to
$\mathsf{SumHistoryCarrier}$ and to certificate transport.
\end{proof}

In the roadmap template this records traditional disjoint-sum branch
determinism for BEDC source $\mathsf{BHist}$ with $\hsame$, carrier
$\mathsf{SumHistoryCarrier}$, classifier $\mathsf{SumHistoryClassifier}$,
visible ledger tag, and $\mathsf{ProofTarget}$ status.

\begin{theorem}[Sum history classifier branch determinism]
\label{thm:sum-history-classifier-branch-determinism}
For source classifiers $LeftEq,RightEq:\mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$,
histories $h,k,l,l',r,r':\mathsf{BHist}$, a single endpoint pair cannot carry
both a left-branch presentation and a right-branch presentation:
$$
\neg\Bigl(
\hsame(h,\Ezero(l))\land
\hsame(k,\Ezero(l'))\land
LeftEq(l,l')\land
\hsame(h,\Eone(r))\land
\hsame(k,\Eone(r'))\land
RightEq(r,r')
\Bigr).
$$
\end{theorem}
\begin{proof}
Assume both presentations. Compose symmetry of $\hsame(h,\Ezero(l))$ with
$\hsame(h,\Eone(r))$ to obtain $\hsame(\Ezero(l),\Eone(r))$, then close by BEDC
$e0/e1$ no-confusion. The endpoint $k$ and the stored component classifier
witnesses are retained in the presentation but are not needed for the
constructor contradiction.
\end{proof}

\begin{theorem}[Sum history classifier same-tag exactness]
\label{thm:sum-history-classifier-same-tag-exactness}
For source carriers $Left,Right:\mathsf{BHist}\to\mathsf{Prop}$, the history-level sum
classifier with $\hsame$ on both components is exact on each visible tag:
$$
\mathsf{SumHistoryClassifier}(Left,Right,\hsame,\hsame,\Ezero(l),\Ezero(l'))
\quad\Longleftrightarrow\quad
\hsame(l,l')
$$
and
$$
\mathsf{SumHistoryClassifier}(Left,Right,\hsame,\hsame,\Eone(r),\Eone(r'))
\quad\Longleftrightarrow\quad
\hsame(r,r').
$$
\end{theorem}
\begin{proof}
For the forward directions, apply the left-tag and right-tag inversion
theorems above. For the reverse directions, rebuild the matching classifier
branch using the given payload $\hsame$ witness, the reflexive visible-tag
witnesses for $\Ezero(l)$ and $\Ezero(l')$ or for $\Eone(r)$ and
$\Eone(r')$, and the corresponding left or right classifier constructor.
\end{proof}

\begin{theorem}[Sum history classifier same-tag source exactness]
\label{thm:sum-history-classifier-source-same-tag-exactness}
For source carriers $Left,Right:\mathsf{BHist}\to\mathsf{Prop}$ and source classifiers
$LeftEq,RightEq:\mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$, suppose the source classifiers
transport along endpoint history sameness:
$$
\forall l,l_0,l',l_0'.\;
\hsame(l,l_0)\to\hsame(l',l_0')\to LeftEq(l_0,l_0')\to LeftEq(l,l')
$$
and
$$
\forall r,r_0,r',r_0'.\;
\hsame(r,r_0)\to\hsame(r',r_0')\to RightEq(r_0,r_0')\to RightEq(r,r').
$$
Then the history-level sum classifier is exact for same-tag endpoints through
the matching source classifier. This theorem is a classifier-only same-tag
exactness statement; it returns the matching source classifier witness and does
not assert $Left(l)$, $Left(l')$, $Right(r)$, or $Right(r')$:
$$
\mathsf{SumHistoryClassifier}(Left,Right,LeftEq,RightEq,\Ezero(l),\Ezero(l'))
\quad\Longleftrightarrow\quad
LeftEq(l,l')
$$
and
$$
\mathsf{SumHistoryClassifier}(Left,Right,LeftEq,RightEq,\Eone(r),\Eone(r'))
\quad\Longleftrightarrow\quad
RightEq(r,r').
$$
\end{theorem}
\begin{proof}
Invert the same-tag classifier witness. In the left-tag forward direction, the
right branch is impossible by $e0/e1$ no-confusion. The left branch stores
endpoint payloads $l_0,l_0'$ together with $LeftEq(l_0,l_0')$; constructor
inversion for $e0$ yields $\hsame(l,l_0)$ and $\hsame(l',l_0')$, and the left
transport hypothesis gives $LeftEq(l,l')$. The right-tag forward direction is
identical using $e1$-constructor inversion and the right transport hypothesis.
For the reverse directions, use the visible endpoints as the stored payloads,
the reflexive tag witnesses for $\Ezero$ or $\Eone$, and the supplied matching
source-classifier witness to rebuild the corresponding classifier branch.
\end{proof}

\begin{theorem}[Sum history classifier hsame same-tag carrier exactness]
\label{thm:sum-history-classifier-hsame-same-tag-carrier-exactness}
For source carriers $Left,Right:\mathsf{BHist}\to\mathsf{Prop}$, the $\hsame$-specialized
history-level sum classifier is exact for same-tag endpoints together with the
matching source-carrier evidence:
$$
\mathsf{SumHistoryCarrier}(Left,Right,\Ezero(l))
\land
\mathsf{SumHistoryCarrier}(Left,Right,\Ezero(l'))
\land
\mathsf{SumHistoryClassifier}(Left,Right,\hsame,\hsame,\Ezero(l),\Ezero(l'))
\quad\Longleftrightarrow\quad
\exists a,a'.\; Left(a)\land Left(a')\land \hsame(l,a)\land \hsame(l',a')\land \hsame(a,a')
$$
and
$$
\mathsf{SumHistoryCarrier}(Left,Right,\Eone(r))
\land
\mathsf{SumHistoryCarrier}(Left,Right,\Eone(r'))
\land
\mathsf{SumHistoryClassifier}(Left,Right,\hsame,\hsame,\Eone(r),\Eone(r'))
\quad\Longleftrightarrow\quad
\exists b,b'.\; Right(b)\land Right(b')\land \hsame(r,b)\land \hsame(r',b')\land \hsame(b,b').
$$
Its roadmap-template role is: traditional target disjoint-sum same-tag payload
exactness; BEDC source $\mathsf{BHist}$ with $\hsame$; carrier
$\mathsf{SumHistoryCarrier}$; classifier the $\hsame$ specialization of
$\mathsf{SumHistoryClassifier}$; ledger visible tag plus payload carrier
witnesses; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
For the forward left-tag direction, apply the two $\Ezero$ carrier inversions to
obtain payload witnesses $a,a'$ with $Left(a)$, $Left(a')$, $\hsame(l,a)$, and
$\hsame(l',a')$. Apply the left $\hsame$ classifier inversion to the same-tag
classifier witness to obtain $\hsame(l,l')$. Symmetry and transitivity of
$\hsame$ compose the displayed endpoint links with $\hsame(l,l')$ to derive
$\hsame(a,a')$. The right-tag direction is identical using the two $\Eone$
carrier inversions and the right $\hsame$ classifier inversion. For the reverse
left-tag direction, the displayed witnesses rebuild the two left carrier
branches and the left classifier branch. The reverse right-tag direction
rebuilds the two right carrier branches and the right classifier branch.
\end{proof}

\begin{theorem}[Sum history classifier generic same-tag exactness]
\label{thm:sum-history-classifier-generic-same-tag-exactness}
For source carriers $Left,Right:\mathsf{BHist}\to\mathsf{Prop}$ and source classifiers $LeftEq,RightEq:\mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$, the history-level sum classifier is exact on each visible tag:
$$
\mathsf{SumHistoryClassifier}(Left,Right,LeftEq,RightEq,\Ezero(l),\Ezero(l'))
\quad\Longleftrightarrow\quad
\exists a,a'.\;\hsame(l,a)\wedge\hsame(l',a')\wedge LeftEq(a,a')
$$
and
$$
\mathsf{SumHistoryClassifier}(Left,Right,LeftEq,RightEq,\Eone(r),\Eone(r'))
\quad\Longleftrightarrow\quad
\exists b,b'.\;\hsame(r,b)\wedge\hsame(r',b')\wedge RightEq(b,b').
$$
\end{theorem}
\begin{proof}
Unfold $\mathsf{SumHistoryClassifier}$. In the left-tag forward direction, the left branch already stores payload witnesses for the two endpoints together with a $LeftEq$ witness; the opposite right branch is impossible by $e0/e1$ no-confusion, and $e0$-constructor inversion identifies the visible endpoints with the stored payload tags. The right-tag forward direction is identical with $e1$-constructor inversion and $RightEq$. For the reverse directions, the displayed payload witnesses rebuild the matching left or right classifier branch by constructor congruence for $e0$ or $e1$, together with the stored source-classifier witness. The surviving branch is constructor-stable, and the impossible branch is discharged by the same tag no-confusion principle.
\end{proof}

\begin{theorem}[Sum history classifier carrier-aware same-tag exactness]
\label{thm:sum-history-classifier-carrier-aware-same-tag-exactness}
Let $Left,Right:\mathsf{BHist}\to\mathsf{Prop}$ be source carriers, let
$LeftEq,RightEq:\mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$ be source
classifiers, and assume the two carriers transport along $\hsame$:
$$
\forall x\,y.\; Left(x)\to \hsame(x,y)\to Left(y)
$$
and
$$
\forall x\,y.\; Right(x)\to \hsame(x,y)\to Right(y).
$$
Then the visible same-tag carrier and classifier evidence is exact for the
matching source payloads:
$$
\begin{array}{l}
\mathsf{SumHistoryCarrier}(Left,Right,\Ezero(l))
\land
\mathsf{SumHistoryCarrier}(Left,Right,\Ezero(l'))
\land
\mathsf{SumHistoryClassifier}(Left,Right,LeftEq,RightEq,\Ezero(l),\Ezero(l'))
\end{array}
\quad\Longleftrightarrow\quad
\exists a,a'.\; Left(a)\land Left(a')\land \hsame(l,a)\land \hsame(l',a')\land LeftEq(a,a')
$$
and
$$
\begin{array}{l}
\mathsf{SumHistoryCarrier}(Left,Right,\Eone(r))
\land
\mathsf{SumHistoryCarrier}(Left,Right,\Eone(r'))
\land
\mathsf{SumHistoryClassifier}(Left,Right,LeftEq,RightEq,\Eone(r),\Eone(r'))
\end{array}
\quad\Longleftrightarrow\quad
\exists b,b'.\; Right(b)\land Right(b')\land \hsame(r,b)\land \hsame(r',b')\land RightEq(b,b').
$$
Its roadmap-template role is: traditional target disjoint-sum same-tag payload
exactness; BEDC source $\mathsf{BHist}$ with $\mathsf{SumHistoryCarrier}$ and
carrier transport along $\hsame$; carrier the tagged sum carrier; classifier
$\mathsf{SumHistoryClassifier}$; ledger visible tag plus payload evidence;
exactness the displayed iff clauses; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
For the forward left-tag direction, invert the two displayed carrier witnesses
to obtain payloads $c,c'$ with $Left(c)$ and $Left(c')$ plus $\hsame(l,c)$ and
$\hsame(l',c')$. Apply
Theorem~\ref{thm:sum-history-classifier-generic-same-tag-exactness} to the
classifier premise to obtain payloads $a,a'$ with $\hsame(l,a)$,
$\hsame(l',a')$, and $LeftEq(a,a')$. Symmetry and transitivity of $\hsame$
align $c$ with $a$ and $c'$ with $a'$. Apply the explicit left
carrier-transport field to $Left(c)$ and $\hsame(c,a)$, and to $Left(c')$ and
$\hsame(c',a')$. This gives the displayed left payload witnesses. The forward
right-tag direction is the same argument with $\Eone$, $Right$, $RightEq$, and
the explicit right carrier-transport field applied in the same order. For the
reverse directions, the displayed payload witnesses rebuild the two tagged
carrier premises and the matching classifier branch by the generic same-tag
exactness theorem, using the visible tag constructors to supply the endpoint
evidence.
\end{proof}

\begin{theorem}[Sum history classifier source-certified carrier transport]
\label{thm:sum-history-classifier-carrier-transport-from-namecert}
Let $Left,Right:\mathsf{BHist}\to\mathsf{Prop}$ be source carriers, let
$LeftEq,RightEq:\mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$ be source
classifiers, and assume source certificates $\NameCert(Left,LeftEq)$ and
$\NameCert(Right,RightEq)$. If
$\mathsf{SumHistoryCarrier}(Left,Right,h)$ and
$\mathsf{SumHistoryClassifier}(Left,Right,LeftEq,RightEq,h,k)$ hold, then
$\mathsf{SumHistoryCarrier}(Left,Right,k)$ holds.
\end{theorem}
\begin{proof}
Case-analyze the displayed carrier witness and the tagged classifier witness.
Mixed source tags are impossible by BEDC $e0/e1$ no-confusion. In the left-left
case, let $c$ be the payload supplied by the carrier witness, and let $a,a'$ be
the source payloads supplied by the classifier witness. The carrier branch stores
$\hsame(h,\Ezero(c))$ and $Left(c)$, while the classifier branch stores
$\hsame(h,\Ezero(a))$, $\hsame(k,\Ezero(a'))$, and $LeftEq(a,a')$.
Composing the symmetric carrier endpoint witness with the classifier endpoint
witness gives $\hsame(\Ezero(c),\Ezero(a))$; $e0$ constructor inversion aligns
the carrier payload with the classifier source payload, so the left carrier
witness may be read as $Left(a)$. Apply
$\mathsf{NameCert.carrier\_respects\_equiv}$ from the left source certificate
along $LeftEq(a,a')$ to obtain $Left(a')$, then rebuild the target left tagged
carrier using the stored endpoint witness $\hsame(k,\Ezero(a'))$. The right-right
case is identical with payloads aligned through $e1$ constructor inversion and
the right source certificate applied along $RightEq$.
\end{proof}

\begin{theorem}[Sum history classifier source-certified symmetry]
\label{thm:sum-history-classifier-symm-from-namecert}
Let $Left,Right:\mathsf{BHist}\to\mathsf{Prop}$ be source carriers, let
$LeftEq,RightEq:\mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$ be source
classifiers, and assume source certificates $\NameCert(Left,LeftEq)$ and
$\NameCert(Right,RightEq)$. If
$\mathsf{SumHistoryClassifier}(Left,Right,LeftEq,RightEq,h,k)$ holds, then
$\mathsf{SumHistoryClassifier}(Left,Right,LeftEq,RightEq,k,h)$ holds.
\end{theorem}
\begin{proof}
Invert the tagged classifier witness. In the left branch, swap the two visible
$e0$ endpoint witnesses and apply the left source certificate field
$\mathsf{equiv\_symm}$ to the stored $LeftEq$ witness. In the right branch,
swap the two visible $e1$ endpoint witnesses and apply the right source
certificate field $\mathsf{equiv\_symm}$ to the stored $RightEq$ witness.
\end{proof}

\begin{theorem}[Sum history classifier tag separation]
\label{thm:sum-history-classifier-tag-separation}
If \(h\) is \(e0\)-tagged and \(k\) is \(e1\)-tagged, then the
history-level \(\SumUp\) classifier cannot relate \(h\) to \(k\).
\end{theorem}
\begin{proof}
Invert the classifier witness. In the left branch \(k\) is also
\(e0\)-tagged, contradicting its \(e1\)-tag witness. In the right branch
\(h\) is \(e1\)-tagged, contradicting its \(e0\)-tag witness.
\end{proof}
\leanchecked{BEDC.Derived.SumUp.SumHistoryClassifier\_mixed\_tags\_absurd}

\begin{theorem}[Sum history classifier reverse tag separation]
\label{thm:sum-history-classifier-reverse-tag-separation}
Let $Left,Right:\mathsf{BHist}\to\mathsf{Prop}$ be source carriers with classifiers $LeftEq$ and $RightEq$. If $h$ is $e1$-tagged and $k$ is $e0$-tagged, then
$$
\mathsf{SumHistoryClassifier}(Left,Right,LeftEq,RightEq,h,k)
$$
is impossible.
\end{theorem}
\begin{proof}
Invert the classifier witness. In the left branch, the classifier forces $h$ to be $e0$-tagged, contradicting the displayed $e1$-tag witness by $e1/e0$ no-confusion. In the right branch, the classifier forces $k$ to be $e1$-tagged, contradicting the displayed $e0$-tag witness by $e0/e1$ no-confusion.
\end{proof}

\begin{theorem}[Sum history classifier transitivity]

history-level \(\SumUp\) classifier is transitive.
\end{theorem}
\begin{proof}
Invert both classifier witnesses. Matching left--left and right--right branches

contradict the \(e0/e1\) no-confusion rule.
\end{proof}

\begin{theorem}[Sum history classifier source-certified stability fields]
\label{thm:sum-history-classifier-source-certified-stability-fields}
Let $Left,Right:\mathsf{BHist}\to\mathsf{Prop}$ be source carriers, let
$LeftEq,RightEq:\mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}$ be source
classifiers, and assume source certificates $\NameCert(Left,LeftEq)$ and
$\NameCert(Right,RightEq)$. For the carrier
$\mathsf{SumHistoryCarrier}(Left,Right)$ and classifier
$\mathsf{SumHistoryClassifier}(Left,Right,LeftEq,RightEq)$, the following
stability fields hold:
\begin{enumerate}
  \item carrier-relative reflexivity:
  $$
  \begin{array}{l}
  \forall h.\;\mathsf{SumHistoryCarrier}(Left,Right,h)\to\\
  \mathsf{SumHistoryClassifier}(Left,Right,LeftEq,RightEq,h,h);
  \end{array}
  $$
  \item source-certified symmetry:
  $$
  \begin{array}{l}
  \forall h\,k.\;\mathsf{SumHistoryClassifier}(Left,Right,LeftEq,RightEq,h,k)\to\\
  \mathsf{SumHistoryClassifier}(Left,Right,LeftEq,RightEq,k,h);
  \end{array}
  $$
  \item source-certified transitivity:
  $$
  \begin{array}{l}
  \forall h\,k\,u.\;\mathsf{SumHistoryClassifier}(Left,Right,LeftEq,RightEq,h,k)\to\\
  \mathsf{SumHistoryClassifier}(Left,Right,LeftEq,RightEq,k,u)\to\\
  \mathsf{SumHistoryClassifier}(Left,Right,LeftEq,RightEq,h,u);
  \end{array}
  $$
  \item classifier-induced carrier transport:
  $$
  \begin{array}{l}
  \forall h\,k.\;\mathsf{SumHistoryCarrier}(Left,Right,h)\to\\
  \mathsf{SumHistoryClassifier}(Left,Right,LeftEq,RightEq,h,k)\to\\
  \mathsf{SumHistoryCarrier}(Left,Right,k).
  \end{array}
  $$
\end{enumerate}
\end{theorem}
\begin{proof}
For reflexivity, invert the displayed sum carrier witness. In the left branch,
use the left source certificate reflexivity field on the source payload and
rebuild the $e0$-tagged classifier witness with the same endpoint evidence; the
right branch is the same with the right source certificate and $e1$ tag. The
symmetry field is Theorem~\ref{thm:sum-history-classifier-symm-from-namecert}.
The transitivity field is Theorem~\ref{thm:sum-history-classifier-transitivity}
applied with the transitivity fields of the two source certificates. The
carrier transport field is
Theorem~\ref{thm:sum-history-classifier-carrier-transport-from-namecert}.
\end{proof}

  \item typed classifier transitivity
  \item mixed $\mathsf{inl}/\mathsf{inr}$ separation in both directions

Carrier transport, endpoint transport, and history-level transitivity are

\end{definition}

\begin{theorem}[Sum ledger history-same transport]

  \mathsf{SumHistoryLedgerPolicy}(Left,Right,raw',visible')
$$

stored $\hsame(raw,visible)$ link, and $\hsame(visible,visible')$ to obtain
$\hsame(raw',visible')$.
\end{proof}

\begin{theorem}[Sum ledger classifier composition]
\label{thm:sum-ledger-classifier-extension}
Let $Left,Right:\mathsf{BHist}\to\mathsf{Prop}$ be source carriers and let
$\rho,v,w:\mathsf{BHist}$ be histories. If
$\mathsf{SumHistoryLedgerPolicy}(Left,Right,\rho,v)$ holds and

\mathsf{SumHistoryClassifier}(Left,Right,\hsame,\hsame,\rho,w)

First apply the raw-visible classifier theorem to the ledger policy, obtaining

\mathsf{SumHistoryClassifier}(Left,Right,\hsame,\hsame,\rho,v).

Then compose this classifier witness with the given visible-target classifier
witness using sum-history classifier transitivity. The branch classifiers are
both $\hsame$, so their required transitivity inputs are the left and right
instances of BEDC history equality transitivity.
\end{proof}

\begin{theorem}[Sum ledger preserves visible tag separation]
\label{thm:sum-ledger-visible-tag-separation}
Let $Left,Right:\mathsf{BHist}\to\mathsf{Prop}$ be source carriers and let
$\rho,v,l,r:\mathsf{BHist}$ be histories. If
$\mathsf{SumHistoryLedgerPolicy}(Left,Right,\rho,v)$ holds, then both visible
mixed-tag endpoint pairs are impossible:
$$
(\hsame(\rho,\Ezero(l))\land \hsame(v,\Eone(r)))\to\bot
$$
and
$$
(\hsame(\rho,\Eone(r))\land \hsame(v,\Ezero(l)))\to\bot.
$$
\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:sum-ledger-raw-visible-classifier} to the ledger policy
to obtain
$\mathsf{SumHistoryClassifier}(Left,Right,\hsame,\hsame,\rho,v)$. The
left-to-right mixed endpoint pair is then discharged by
$\mathsf{SumHistoryClassifier\_mixed\_tags\_absurd}$ with the displayed
$\hsame(\rho,\Ezero(l))$ and $\hsame(v,\Eone(r))$ witnesses. For the symmetric
right-to-left pair, use classifier symmetry from
Theorem~\ref{thm:sum-history-classifier-hsame-symmetry} and apply the same mixed-tag
separation theorem to the exchanged endpoints.
\end{proof}

\begin{theorem}[Sum ledger same-tag payload exactness]
\label{thm:sum-ledger-same-tag-payload-exactness}
Let $Left,Right:\mathsf{BHist}\to\mathsf{Prop}$ be source carriers and let
$\rho,v,l,l',r,r':\mathsf{BHist}$ be histories. If
$\mathsf{SumHistoryLedgerPolicy}(Left,Right,\rho,v)$ holds, then the visible
same-tag endpoint pairs determine their payloads:
$$
(\hsame(\rho,\Ezero(l))\land\hsame(v,\Ezero(l')))\to\hsame(l,l')
$$
and
$$
(\hsame(\rho,\Eone(r))\land\hsame(v,\Eone(r')))\to\hsame(r,r').
$$
\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:sum-ledger-raw-visible-classifier} to obtain the
raw-visible $\mathsf{SumHistoryClassifier}(Left,Right,\hsame,\hsame,\rho,v)$
witness. For the left-tagged endpoint pair, transport this witness along the
displayed $\hsame(\rho,\Ezero(l))$ and $\hsame(v,\Ezero(l'))$ links by
Theorem~\ref{thm:sum-history-classifier-hsame-transport}, then apply the
left inversion Theorem~\ref{thm:sum-history-classifier-left-hsame-inversion}
to extract $\hsame(l,l')$. The right-tagged case is identical with
Theorem~\ref{thm:sum-history-classifier-right-hsame-inversion}.
\end{proof}

\begin{theorem}[Sum ledger visible payload readback]
\label{thm:sum-ledger-visible-payload-readback}
Let $Left,Right:\mathsf{BHist}\to\mathsf{Prop}$ be source carriers and let
$\rho,v,l,r:\mathsf{BHist}$ be histories. If
$\mathsf{SumHistoryLedgerPolicy}(Left,Right,\rho,v)$ holds, then visible tagged
endpoints read back source payload evidence:
$$
(\hsame(v,\Ezero(l))\to\exists a:\mathsf{BHist}.\;Left(a)\land\hsame(l,a))
$$
and
$$
(\hsame(v,\Eone(r))\to\exists b:\mathsf{BHist}.\;Right(b)\land\hsame(r,b)).
$$
\end{theorem}
\begin{proof}
The traditional target is disjoint-sum ledger readback; the BEDC source is
$\mathsf{BHist}$; the carrier is $\mathsf{SumHistoryCarrier}$; no classifier is
needed for this carrier-readback obligation; the ledger is
$\mathsf{SumHistoryLedgerPolicy}$; the exactness principle is visible-tag
carrier readback; the status is $\mathsf{ProofTarget}$. Apply
Theorem~\ref{thm:sum-ledger-visible-carrier} to the ledger, transport the visible
carrier witness along the displayed $\hsame(v,\Ezero(l))$ or
$\hsame(v,\Eone(r))$ endpoint equality, and then apply
Theorem~\ref{thm:sum-history-carrier-visible-tag-exactness} to extract the left
or right source payload witness.
\end{proof}

\begin{theorem}[Concrete $\SumUp$ semantic name certificate]
\label{thm:concrete-sum-namecert}
Let $Left,Right:\mathsf{BHist}\to\mathsf{Prop}$ be source carriers with classifiers
$LeftEq$ and $RightEq$. If $\NameCert(Left,LeftEq)$ and
$\NameCert(Right,RightEq)$ are supplied, then the tagged sum carrier, the tagged
sum classifier, and the carrier-as-pattern and carrier-as-ledger tuple form
$$
\begin{array}{l}
\mathsf{SemanticNameCert}( \\
\quad \mathsf{SumHistoryCarrier}(Left,Right), \\
\quad \mathsf{SumHistoryCarrier}(Left,Right), \\
\quad \mathsf{SumHistoryCarrier}(Left,Right), \\
\quad \mathsf{SumHistoryClassifier}(Left,Right,LeftEq,RightEq)).
\end{array}
$$
\end{theorem}
\begin{proof}
The carrier is represented by $e0$-tagged left histories and $e1$-tagged right histories. Theorem~\ref{thm:sum-classifier-no-source-membership} marks the split: the classifier delegates within matching tags to the corresponding source certificate, while source-carrier evidence is taken from $\mathsf{SumHistoryCarrier}$ when certificate fields require it. Reflexivity is built by tagging the source certificate reflexivity field on the selected component. Symmetry and transitivity invert the tagged classifier witnesses, use the matching source certificate fields inside a left--left or right--right branch, and discharge mixed tags by constructor no-confusion.

For carrier transport in the left branch, the carrier witness stores $\hsame(h,\Ezero(l))$ and $Left(l)$, while the classifier endpoint stores $\hsame(h,\Ezero(a))$, $\hsame(k,\Ezero(b))$, and $LeftEq(a,b)$. Symmetry of the carrier endpoint and transitivity with the classifier endpoint give $\hsame(\Ezero(l),\Ezero(a))$; $e0$ constructor inversion identifies the carrier payload with the classifier payload. The left source certificate field $\mathsf{NameCert.carrier\_respects\_equiv}$ then transports $Left(l)$ along this payload alignment and the stored $LeftEq(a,b)$ witness to obtain $Left(b)$, and the target left carrier is rebuilt with $\hsame(k,\Ezero(b))$.

For carrier transport in the right branch, the carrier witness stores $\hsame(h,\Eone(r))$ and $Right(r)$, while the classifier endpoint stores $\hsame(h,\Eone(a))$, $\hsame(k,\Eone(b))$, and $RightEq(a,b)$. Symmetry and transitivity produce $\hsame(\Eone(r),\Eone(a))$; $e1$ constructor inversion aligns the carrier payload with the classifier payload. The right source certificate field $\mathsf{NameCert.carrier\_respects\_equiv}$ transports $Right(r)$ along that alignment and the stored $RightEq(a,b)$ witness to obtain $Right(b)$, and the target right carrier is rebuilt with $\hsame(k,\Eone(b))$. The carrier-as-pattern and carrier-as-ledger components use the same tagged carrier evidence, so the displayed tuple satisfies the semantic certificate fields.
\end{proof}
\leanchecked{BEDC.Derived.SumUp.sum\_history\_semantic\_name\_certificate}
