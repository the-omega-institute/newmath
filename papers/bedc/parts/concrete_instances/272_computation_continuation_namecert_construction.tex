\chapter{Naming Certificate for $\ComputationContinuationUp$}
\label{ch:concrete-instances-computation-as-continuation}

The object $\ComputationContinuationUp$ names the operational reading of the
kernel continuation primitive $\Cont$.  It packages a small-step computation as
finite BEDC history data, with composition supplied only by the displayed
continuation ledger.

\section{Source}
\label{sec:concrete-instances-computation-as-continuation-source}

\begin{definition}[Source rows for $\ComputationContinuationUp$]
\label{def:concrete-instances-computation-as-continuation-source}
A source row is a tuple $(h,k,r,\ell)$ where $h$ is the input history, $k$ is
the displayed step or continuation tail, $r$ is the output history, and
$\ell:\Cont(h,k,r)$ is the finite transition ledger.  This is the whole source
surface of the NameCert: program syntax, host evaluation contexts, and external
determinism are not part of the row.
\end{definition}
\leandef{BEDC.Derived.ComputationContinuationUp.SourceRows}

\section{Pattern}
\label{sec:concrete-instances-computation-as-continuation-pattern}

\begin{definition}[Continuation transition pattern]
\label{def:concrete-instances-computation-as-continuation-pattern}
The pattern accepted by $\ComputationContinuationUp$ is exactly the
small-step-shaped continuation judgment
$$
\begin{aligned}
  \Cont(h,k,r).
\end{aligned}
$$
Reading $h$ as the current configuration, $k$ as the displayed transition tail,
and $r$ as the next configuration gives the Plotkin-style operational surface,
but the certificate records only the BEDC continuation row.
\end{definition}
\leandef{BEDC.Derived.ComputationContinuationUp.Pattern}

\section{Classifier}
\label{sec:concrete-instances-computation-as-continuation-classifier}

\begin{definition}[Computation-continuation classifier]
\label{def:concrete-instances-computation-as-continuation-classifier}
Two source rows classify together when their input histories, transition tails,
and output histories are componentwise history-same:
$$
\begin{aligned}
  h\hsame h',
  \qquad
  k\hsame k',
  \qquad
  r\hsame r'.
\end{aligned}
$$
Thus the public name is insensitive to the host identity of a row record and
depends only on the three BEDC histories exposed by the transition ledger.
\end{definition}
\leandef{BEDC.Derived.ComputationContinuationUp.Classifier}

\section{Stability}
\label{sec:concrete-instances-computation-as-continuation-stability}

\begin{theorem}[Continuation computation is stable under sequential composition]
\label{thm:concrete-instances-computation-as-continuation-stability}
If one source row records $\Cont(h,k,r)$ and a second source row records
$\Cont(r,k',u)$, then their sequential composite is again a
$\ComputationContinuationUp$ row:
$$
\begin{aligned}
  \Cont(h,k\mathbin{\mathrm{append}}k',u).
\end{aligned}
$$
The composite row is classified with itself by componentwise $\hsame$.
\end{theorem}
\leanchecked{BEDC.Derived.ComputationContinuationUp.stability}

\begin{proof}
The proof rewrites the second transition through the first transition and uses
associativity of finite continuation append.  The classifier part is
componentwise reflexivity of $\hsame$ on the composite source, composite tail,
and composite target.
\end{proof}

\section{Ledger}
\label{sec:concrete-instances-computation-as-continuation-ledger}

\begin{definition}[Ledger for $\ComputationContinuationUp$]
\label{def:concrete-instances-computation-as-continuation-ledger}
The ledger retains the original source row: input history, transition tail,
output history, and the displayed $\Cont$ witness.  A public operational
transition can therefore always be audited back to the finite continuation row
that generated it.
\end{definition}
\leandef{BEDC.Derived.ComputationContinuationUp.Ledger}
