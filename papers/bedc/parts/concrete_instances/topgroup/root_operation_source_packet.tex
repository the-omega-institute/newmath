\section{Root operation source packet}
\label{sec:topgroup-root-operation-source-packet}

\begin{definition}[TopGroupUp root operation-source packet]
\label{def:topgroup-root-operation-source-packet}
The $\TopGroupUp$ root operation-source packet is the shared
$\mathsf{BHist}$ packet
$$
\mathcal R=(G,T,m,i,n,\lambda,\rho)
$$
of \autoref{def:topgroup-bhist-carrier}, read as a root-facing source for the
operation rows. The coordinate $G$ is the displayed
$\NameCert_{\GroupUp}$ row, $T$ is the displayed
$\NameCert_{\TopologyUp}$ row, $m$ is the product row, $i$ is the inverse row,
$n$ is the neighbourhood row used by both operation-continuity families,
$\lambda$ is the finite $\Cont$ ledger, and $\rho$ is the $\Pkg$ provenance
row tying the group, topology, operation, neighbourhood, classifier, and
continuity entries to one packet.

Two such packets are compared only by the componentwise $\hsame$ rows already
present in the carrier and by the displayed $\Cont$ ledger comparison. The
packet contains no host topological-group object and no ambient continuity
predicate outside the shown BEDC rows.
\end{definition}

\begin{theorem}[TopGroupUp root source-pair exactness]
\label{thm:topgroup-root-source-pair-exactness}
Every accepted $\TopGroupUp$ root operation-source packet has exactly the
displayed $\NameCert_{\GroupUp}$ and $\NameCert_{\TopologyUp}$ source pair.
The product row, inverse row, neighbourhood row, finite $\Cont$ ledger, and
$\Pkg$ provenance row are projections from the same $\mathsf{BHist}$ packet,
so the root source cannot be read as two independent carrier certificates.
\end{theorem}
\leanchecked{BEDC.Derived.TopGroupUp.TopGroupRootOperationSourcePacket\_source\_pair\_exactness}
\begin{proof}
Project the packet of \autoref{def:topgroup-root-operation-source-packet}.
Its coordinates are the source pair $G,T$ together with the operation rows
$m,i$, the neighbourhood row $n$, the ledger $\lambda$, and the provenance row
$\rho$.

Apply \autoref{thm:topgroup-source-fiber-product-carrier}. It identifies the
group source, topology source, product graph, inverse graph, neighbourhood
row, finite $\Cont$ ledger, and $\Pkg$ provenance as projections from one
shared $\mathsf{BHist}$ carrier.

Finally use \autoref{thm:topgroup-root-shared-carrier-obligation}. A root
consumer therefore receives the two source rows only through that shared
packet, with no second carrier from which an independent source certificate
could be assembled.
\end{proof}
\leanchecked{BEDC.Derived.TopGroupUp.TopGroupRootThresholdPackage\_root\_source\_pair\_exactness}

\begin{theorem}[TopGroupUp root operation-continuity exhaustion]
\label{thm:topgroup-root-operation-continuity-exhaustion}
For an accepted $\TopGroupUp$ root operation-source packet, every
operation-continuity row exported at the root is either a product-continuity
row for $m$ or an inverse-continuity row for $i$. Both families are carried by
the displayed neighbourhood row $n$ and finite $\Cont$ ledger $\lambda$, and
there is no further operation-continuity predicate outside the packet.
\end{theorem}
\leanchecked{BEDC.Derived.TopGroupUp.TopGroupRootOperationSourcePacket\_operation\_continuity\_exhaustion}
\begin{proof}
Start with the exact source-pair packet supplied by
\autoref{thm:topgroup-root-source-pair-exactness}. It places the product row,
inverse row, neighbourhood row, and ledger over the same displayed
$\NameCert_{\GroupUp}$ and $\NameCert_{\TopologyUp}$ sources.

Apply \autoref{thm:topgroup-operation-continuity-obligation}. It says that
the exported operation-continuity obligations are exactly multiplication
continuity and inversion continuity.

Apply \autoref{thm:topgroup-ledger-exactness-obligation}. The finite $\Cont$
ledger covers all accepted product-neighbourhood and inverse-neighbourhood
continuity reads, and each listed entry is sound through the shared carrier
packet. Thus the root operation-continuity surface is exhausted by those two
families.
\end{proof}

\begin{theorem}[TopGroupUp root classifier ledger stability]
\label{thm:topgroup-root-classifier-ledger-stability}
If two accepted $\TopGroupUp$ root operation-source packets are related by the
root classifier, then the product row, inverse row, neighbourhood row, finite
$\Cont$ ledger, and $\Pkg$ provenance row transport componentwise by the
displayed $\hsame$ witnesses and the same ledger comparison. The transported
packet still exposes only the displayed source pair and its two
operation-continuity families.
\end{theorem}
\leanchecked{BEDC.Derived.TopGroupUp.TopGroupRootThreshold\_classifier\_ledger\_transport\_packet}
\begin{proof}
Unpack the two root operation-source packets. The classifier compares the
group source, topology source, product row, inverse row, neighbourhood row,
ledger row, and provenance row componentwise through the carried $\hsame$ and
$\Cont$ data.

Use \autoref{thm:topgroup-root-continuity-classifier-stability}. It transports
product continuity and inverse continuity through the same classifier packet,
with the product and inverse arities read from the shared carrier rows.

Then apply \autoref{thm:topgroup-root-operation-continuity-exhaustion}. Since
the transported root surface has only the product and inverse continuity
families listed in the finite ledger, classifier transport cannot introduce a
host topological-group object or an ambient continuity predicate.
\end{proof}

\begin{theorem}[TopGroupUp root operation-source continuity coupling]
\label{thm:topgroup-root-operation-source-continuity-coupling}
For an accepted root operation-source packet
$\mathcal R=(G,T,m,i,n,\lambda,\rho)$, the displayed
$\NameCert_{\GroupUp}$ row $G$ and displayed $\NameCert_{\TopologyUp}$ row $T$
determine the product-continuity and inverse-continuity families through the
same neighbourhood row $n$, finite $\Cont$ ledger $\lambda$, and $\Pkg$
provenance row $\rho$. Classifier transport of $\mathcal R$ transports both
continuity families through those same rows and cannot create an ambient
continuity predicate outside the packet.
\end{theorem}
\leanchecked{BEDC.Derived.TopGroupUp.TopGroupRootOperationSourcePacket\_continuity\_coupling}
\begin{proof}
Apply \autoref{thm:topgroup-root-source-pair-exactness}. It identifies $G$ and
$T$ as the displayed source pair of a single $\mathsf{BHist}$ packet, with
$m$, $i$, $n$, $\lambda$, and $\rho$ read as projections from that same packet.

Next use \autoref{thm:topgroup-root-operation-continuity-exhaustion}. The only
operation-continuity rows over the displayed source pair are the
product-continuity family for $m$ and the inverse-continuity family for $i$,
both indexed by the same neighbourhood row and finite $\Cont$ ledger.

Finally apply \autoref{thm:topgroup-root-classifier-ledger-stability}. The
classifier transports the product row, inverse row, neighbourhood row, ledger,
and provenance componentwise, so it transports both continuity families
together and has no row from which to synthesize an ambient continuity
predicate.
\end{proof}
\leanchecked{BEDC.Derived.TopGroupUp.TopGroupRootThresholdPackage\_root\_operation\_source\_continuity\_coupling}
