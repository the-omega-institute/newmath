\section{Effective primality testing}

\begin{definition}[Trial-division witness up to \(b\)]
\label{def:trial-division-witness}
For unary histories \(b, n\), the trial-division witness \(\mathsf{TrialDiv}(b, n)\) is the closed inductive predicate generated by:
\begin{enumerate}
  \item \(\mathsf{TrialDiv}(\Eone(\emp), n)\) for every unary \(n\) (the unit is examined first);
  \item if \(\mathsf{TrialDiv}(b, n)\), and either \(\neg \mathsf{NatDivides}(b, n)\) or \(\hsame(b, \Eone(\emp))\) or \(\hsame(b, n)\), and \(\Cont(b, \Eone(\emp), b')\), then \(\mathsf{TrialDiv}(b', n)\) (advance the test to \(b'\)).
\end{enumerate}
\end{definition}
\leandef{BEDC.Derived.PrimeUp.TrialDiv}

\begin{theorem}[Trial-division step preserves unary closure]
\label{thm:trialdiv-step-unary-closed}
If \(\mathsf{TrialDiv}(b,n)\) has been reached and the admissible step
condition for \(b\) holds, then every continuation
\(\Cont(b,\Eone(\emp),b')\) produces a unary \(b'\) and the next witness
\(\mathsf{TrialDiv}(b',n)\).
\end{theorem}
\begin{proof}
Induct on the trial-division witness. The base witness starts at
\(\Eone(\emp)\), which is unary; the step case uses unary closure of
\(\Cont\) with the unit continuation and rebuilds the trial-division step.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.TrialDiv\_step\_unary\_closed}

\begin{theorem}[Trial-division bounds are unary]
\label{thm:trial-div-bound-unary}
If \(\mathsf{TrialDiv}(b,n)\), then \(b\) is a unary history.
\end{theorem}
\begin{proof}
Induct on the trial-division witness. The unit constructor gives \(\mathsf{Unary}(\Eone(\emp))\). The step constructor extends the previous unary bound by the unary unit through \(\Cont\), so unary closure under continuation gives the next bound.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.TrialDiv\_bound\_unary}

\begin{theorem}[Trial-division bounds have positive shape]
\label{thm:trial-div-bound-positive-shape}
If \(\mathsf{TrialDiv}(b,n)\), then there is a unary tail \(t\) such that
\(b\hsame\Eone(t)\).
\end{theorem}
\begin{proof}
Inspect the trial-division witness. The unit constructor has
\(b=\Eone(\emp)\). In the step constructor, the continuation
\(\Cont(b,\Eone(\emp),b')\) exposes \(b'\) as \(\Eone(b)\), and the previous
trial-division witness supplies unary closure of the tail by
\autoref{thm:trial-div-bound-unary}.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.TrialDiv\_bound\_positive\_shape}

\begin{theorem}[Trial-division bound unit or successor shape]
\label{thm:trialdiv-bound-unit-or-successor-shape}
If \(\mathsf{TrialDiv}(b,n)\), then the bound is either the unit history
\(\Eone(\emp)\) or has the shape \(\Eone(\Eone(t))\) for a unary tail \(t\).
\end{theorem}
\begin{proof}
Apply the positive-shape theorem to write \(b\hsame\Eone(tail)\) with unary
\(tail\). Case-analyze this tail. The empty tail gives the unit branch; a
zero-headed tail contradicts unary generation; a one-headed tail exposes the
displayed \(\Eone(\Eone(t))\) branch, and unary inversion supplies
\(\mathsf{UnaryHistory}(t)\).
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.TrialDiv\_bound\_unit\_or\_successor\_shape}

\begin{theorem}[Trial-division bounds have unit prefix or are unit]
\label{thm:trial-div-bound-unit-prefix-or-self}
If $\mathsf{TrialDiv}(b,n)$, then either $\Eone(\emp)$ is a strict unary prefix
of $b$, or $b\hsame\Eone(\emp)$.
\end{theorem}
\begin{proof}
Induct on the trial-division witness. The unit constructor gives the self
branch. In the step constructor, the continuation by $\Eone(\emp)$ supplies a
strict one-step prefix. If the induction hypothesis already has a strict prefix,
compose the two strict prefixes; if it has the self branch, transport the
one-step prefix across that $\HSame$ endpoint.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.TrialDiv\_bound\_unit\_prefix\_or\_self}

\begin{theorem}[Trial-division unit step yields strict unit prefix]
\label{thm:trialdiv-bound-unit-strict-prefix-after-cont}
If $\mathsf{TrialDiv}(b,n)$ and $\Cont(b,\Eone(\emp),b')$, then
$\Eone(\emp)$ is a strict unary prefix of $b'$.
\end{theorem}
\begin{proof}
The displayed continuation supplies a one-step strict prefix from $b$ to $b'$.
The bound classification theorem gives either a strict unit prefix of $b$ or
$b\hsame\Eone(\emp)$. In the strict-prefix branch, compose the previous strict
prefix with the one-step prefix. In the unit branch, transport the one-step
strict prefix across the displayed endpoint classifier.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.TrialDiv\_bound\_unit\_strict\_prefix\_after\_cont}

\begin{theorem}[Trial-division target and bound trichotomy]
\label{thm:trialdiv-target-unary-bound-trichotomy}
If \(\mathsf{TrialDiv}(b,n)\), then the target \(n\) is unary and the current
bound \(b\) is comparable with \(n\) by endpoint equality or by one of the two
strict unary-prefix directions:
$$
\mathsf{UnaryHistory}(n)\land
\bigl(b\hsame n\lor \mathsf{NatUnaryStrictPrefix}(b,n)\lor
\mathsf{NatUnaryStrictPrefix}(n,b)\bigr).
$$
\end{theorem}
\begin{proof}
Induct on the trial-division witness to recover unaryness of the fixed target.
The bound is unary by \autoref{thm:trial-div-bound-unary}. Applying unary-prefix
trichotomy to these two unary histories gives the displayed comparison.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.TrialDiv\_target\_unary\_and\_bound\_trichotomy}

\begin{theorem}[Trial-division bounds are not empty]
\label{thm:trial-div-bound-not-empty}
If \(\mathsf{TrialDiv}(b,n)\), then \(b\) is not \(\emp\).
\end{theorem}
\begin{proof}
Use \autoref{thm:trial-div-bound-positive-shape} to write
\(b\hsame\Eone(t)\) for a unary tail \(t\). An additional endpoint
\(b\hsame\emp\) would identify \(\Eone(t)\) with \(\emp\), contradicting the
history no-confusion principle.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.TrialDiv\_bound\_not\_empty}

\begin{theorem}[Trial-division unit step result is not unit]
\label{thm:trialdiv-unit-step-result-not-unit}
If $\mathsf{TrialDiv}(b,n)$ and $\Cont(b,\Eone(\emp),b')$, then
$b'$ cannot be equal to the unit history $\Eone(\emp)$.
\end{theorem}
\begin{proof}
Transport the displayed continuation result across
$b'\hsame\Eone(\emp)$, obtaining a continuation
$\Cont(b,\Eone(\emp),\Eone(\emp))$. Compare it with the left-unit
continuation $\Cont(\emp,\Eone(\emp),\Eone(\emp))$ and cancel the common
right unit history. This gives $b\hsame\emp$, contradicting
\autoref{thm:trial-div-bound-not-empty}.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.TrialDiv\_unit\_step\_result\_not\_unit}

\begin{theorem}[Terminal trial division excludes non-trivial divisors]
\label{thm:trialdiv-terminal-excludes-nontrivial-divisors}
Let $n$ be unary with $\mathsf{NatGT}(n,\Eone(\emp))$. If
$\mathsf{TrialDiv}(n,n)$, then every unary $d$ with
$\mathsf{NatDivides}(d,n)$ satisfies $d\hsame\Eone(\emp)$ or $d\hsame n$.
Consequently $\mathsf{NatPrime}(n)$.
\end{theorem}
\begin{proof}
Fix a unary divisor $d$ of $n$. Since $\mathsf{NatGT}(n,\Eone(\emp))$ supplies
a nonempty endpoint, \autoref{lem:natdivides-result-boundary} applies to the
divisor witness.

If the boundary lemma gives $d\hsame n$, the desired divisor alternative is the
right branch. It remains to handle the strict-prefix branch
$\mathsf{NatUnaryStrictPrefix}(d,n)$.

Induct on the terminal witness $\mathsf{TrialDiv}(n,n)$ while transporting the
strict-prefix endpoint along each unit-continuation step. The induction
recovers the trial-division step at the bound represented by $d$. Its
admissible condition is
$$
\neg\mathsf{NatDivides}(d,n)
\quad\text{or}\quad
d\hsame\Eone(\emp)
\quad\text{or}\quad
d\hsame n.
$$

The first branch contradicts the divisor hypothesis. The last branch
contradicts strict-prefix asymmetry after transporting across the equality
$d\hsame n$. Therefore the unit branch remains, so
$d\hsame\Eone(\emp)$.

The prime predicate now follows by unfolding \autoref{def:nat-prime}: unaryness
and strict greater-than are the hypotheses, and the preceding argument supplies
the required alternative for every unary divisor.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.TrialDiv\_terminal\_excludes\_nontrivial\_divisors}

\begin{theorem}[Decidability of primality]
\label{thm:prime-decidable}
For every unary history \(n\), the proposition \(\mathsf{NatPrime}(n)\) is decidable: there is an explicit unary algorithm that returns either a proof of \(\mathsf{NatPrime}(n)\) or an explicit non-trivial divisor of \(n\).
\end{theorem}
\begin{proof}[Sketch]
The algorithm constructs $\mathsf{TrialDiv}(b,n)$ by unary induction on $b$
up to $n$ itself. At each step, it checks divisibility of $n$ by $b$ using
\autoref{lem:nat-mul-total} and \autoref{lem:nat-mul-functional}: this is
decidable because the search for a quotient $q$ of $\mathsf{NatMul}(b,q,n)$ is
bounded above by $n$.

If a non-trivial divisor is found, meaning $b$ divides $n$ and is neither the
unit nor $n$, the algorithm returns it and refutes primality. Otherwise,
$\mathsf{TrialDiv}$ reaches $b=n$, and
\autoref{thm:trialdiv-terminal-excludes-nontrivial-divisors} gives
$\mathsf{NatPrime}(n)$.
\end{proof}
