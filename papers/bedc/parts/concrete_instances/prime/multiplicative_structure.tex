\section{Multiplicative structure on unary histories}

\begin{definition}[Unary multiplication relation]
\label{def:nat-multiplication}
For unary histories \(d, q, n\) with \(\mathsf{Unary}(d)\), \(\mathsf{Unary}(q)\), and \(\mathsf{Unary}(n)\), the multiplicative relation \(\mathsf{NatMul}(d, q, n)\) holds when \(n\) is the unary continuation of \(d\) repeated \(q\) times. Concretely, \(\mathsf{NatMul}\) is the closed inductive predicate generated by:
\begin{enumerate}
  \item \(\mathsf{NatMul}(d, \emp, \emp)\) for every unary \(d\) (zero multiplied by anything is zero);
  \item if \(\mathsf{NatMul}(d, q, n)\) and \(\Cont(n, d, n')\), then \(\mathsf{NatMul}(d, \Eone(q), n')\) (one more copy of \(d\) appended).
\end{enumerate}
\end{definition}
\leandef{BEDC.Derived.PrimeUp.NatMul}

\begin{lemma}[Unary multiplication successor inversion]
\label{lem:nat-mul-succ-inversion}
If $\mathsf{NatMul}(d,\Eone(q),n')$, then there is a predecessor product
$n$ such that $\mathsf{NatMul}(d,q,n)$ and $\Cont(n,d,n')$.
\end{lemma}
\begin{proof}
Invert the closed successor clause in the inductive definition of
$\mathsf{NatMul}$. The displayed predecessor product and continuation
witness are exactly the constructor payloads.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatMul\_succ\_inversion}

\begin{lemma}[Unary multiplication exponent-result cases]
\label{lem:nat-mul-exponent-result-cases}
Every witness of \(\mathsf{NatMul}(d,q,n)\) is either the empty exponent
case, with \(q\hsame\emp\) and \(n\hsame\emp\), or a successor exponent case:
there are histories \(q_0\) and \(n_0\) such that
\(q=\Eone(q_0)\), \(\mathsf{NatMul}(d,q_0,n_0)\), and
\(\Cont(n_0,d,n)\).
\end{lemma}
\begin{proof}
Invert the closed multiplication witness. The zero constructor gives the
empty exponent and empty result. The successor constructor exposes exactly the
predecessor exponent, predecessor product, and final continuation payload.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatMul\_exponent\_result\_cases}

\begin{lemma}[Empty left unary multiplication has empty result]
\label{lem:natmul-empty-left-result-empty}
If \(\mathsf{NatMul}(\emp,q,n)\), then \(n\hsame\emp\).
\end{lemma}
\begin{proof}
Induct on the concrete multiplication witness. The zero constructor is
immediate. In the successor constructor, the induction hypothesis identifies
the previous product with \(\emp\), and the final continuation is therefore an
empty-left continuation whose result is \(\emp\).
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatMul\_empty\_left\_result\_empty}

\begin{theorem}[Successor multiplication with empty result has empty left factor]
\label{thm:natmul-succ-result-empty-left-empty}
If \(\mathsf{NatMul}(d,\Eone(q),n)\) and \(n\hsame\emp\), then \(d\hsame\emp\).
\end{theorem}
\begin{proof}
Invert the successor multiplication witness. Its final step is a continuation
\(\Cont(n_0,d,n)\). Since the result is empty, empty-result inversion for
\(\Cont\) forces both inputs of that continuation to be empty, in particular
the left multiplicand \(d\).
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatMul\_succ\_result\_empty\_left\_empty}

\begin{theorem}[Successor multiplication has positive result shape]
\label{thm:natmul-succ-result-positive-shape}
If $\mathsf{NatMul}(d,\Eone(q),n)$ and $d$ is not history-same to
$\emp$, then $n$ has one-headed shape:
$$
\exists r,\ \hsame(n,\Eone(r))\wedge\mathsf{UnaryHistory}(r).
$$
\end{theorem}
\begin{proof}
Invert the successor multiplication witness. The final step is a continuation
$\Cont(n_0,d,n)$, and the multiplication witness supplies unary closure of
$d$ and $n_0$. The empty and zero-headed cases for $d$ are impossible by
non-emptiness and unary exactness. Thus $d=\Eone(t)$, and the continuation
has result $\Eone(\mathsf{append}(n_0,t))$, whose tail is unary by append
closure.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatMul\_succ\_result\_positive\_shape}

\begin{theorem}[Nonempty multiplier multiplication has positive result shape]
\label{thm:natmul-nonempty-multiplier-result-positive-shape}
If $\mathsf{NatMul}(d,q,n)$, $q$ is not history-same to $\emp$, and $d$ is not
history-same to $\emp$, then the result has one-headed shape:
$$
\exists t,\ \hsame(n,\Eone(t))\wedge\mathsf{UnaryHistory}(t).
$$
\end{theorem}
\begin{proof}
The multiplication witness carries a unary classifier for $q$. The empty case
contradicts the displayed non-emptiness, and the zero-headed case contradicts
unary exactness. Hence $q=\Eone(q_0)$, and
\autoref{thm:natmul-succ-result-positive-shape} applies to the same
multiplication witness and the nonempty multiplicand.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatMul\_nonempty\_multiplier\_result\_positive\_shape}

\begin{theorem}[Successor multiplication empty result exactness]
\label{thm:nat-mul-succ-result-empty-left-empty-iff}
If $\mathsf{NatMul}(d,\Eone(q),n)$, then the result is empty exactly when the
left multiplicand is empty:
$$
\hsame(n,\emp)\Longleftrightarrow \hsame(d,\emp).
$$
\end{theorem}
\begin{proof}
The forward implication is the successor empty-result inversion above. For the
reverse implication, identify $d$ with $\emp$ and apply the empty-left
multiplication result theorem to the same multiplication witness.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatMul\_succ\_result\_empty\_left\_empty\_iff}

\begin{lemma}[Unary multiplication right operand is unary]
\label{lem:nat-mul-right-unary}
Every closed witness of $\mathsf{NatMul}(d,q,n)$ carries a unary multiplier
$q$.
\end{lemma}
\begin{proof}
Induct on the $\mathsf{NatMul}$ witness. The zero constructor has multiplier
$\emp$, and the successor constructor applies $\Eone$ to the previous unary
multiplier.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatMul\_right\_unary}

\begin{lemma}[Unary multiplication is functional]
\label{lem:nat-mul-functional}
For each pair of unary histories \((d, q)\), there is at most one unary \(n\) with \(\mathsf{NatMul}(d, q, n)\), up to \(\hsame\).
\end{lemma}
\begin{proof}
By induction on the second argument \(q\), using the determinacy of \(\Cont\) at each appending step.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatMul\_functional}

\begin{theorem}[Unary multiplication commutes up to history sameness]
\label{thm:natmul-comm-hsame}
For unary histories \(d\) and \(q\), if \(\mathsf{NatMul}(d,q,n)\) and
\(\mathsf{NatMul}(q,d,m)\), then \(n\hsame m\).
\end{theorem}
\begin{proof}
The concrete multiplication witness fixes the BEDC word length of its result
as the product of the word lengths of its two unary inputs. The two displayed
products therefore have equal lengths by commutativity of natural-number
multiplication. Since unary histories are determined by their word length, the
two multiplication ledgers have history-same results.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatMul\_comm\_hsame}

\begin{lemma}[Unary multiplication is total on unary inputs]
\label{lem:nat-mul-total}
For every pair of unary histories \((d, q)\) there exists a unary \(n\) with \(\mathsf{NatMul}(d, q, n)\).
\end{lemma}
\begin{proof}
By unary induction on \(q\). The base case yields \(n = \emp\); the step appends one further copy of \(d\) using \(\Cont\). Closure of unary histories under \(\Cont\) (\autoref{ch:concrete-instances-nat-namecert}) keeps \(n\) unary.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatMul\_total}

\begin{lemma}[Unary multiplication results are unary]
\label{lem:nat-mul-result-unary}
For every unary multiplicand $d$, every derivation
$\mathsf{NatMul}(d,q,n)$ has a unary result $n$.
\end{lemma}
\begin{proof}
Induct on the $\mathsf{NatMul}$ derivation. The zero constructor returns
$\emp$, and the successor constructor appends the unary multiplicand to the
unary inductive result, so unary closure under $\Cont$ applies.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatMul\_result\_unary}

\begin{lemma}[Unit left multiplication reads back the multiplier]
\label{lem:nat-mul-unit-left-hsame}
For every unary history $q$, if $\mathsf{NatMul}(\Eone(\emp), q, n)$, then $\hsame(n,q)$.
\end{lemma}
\begin{proof}
Induct on the concrete multiplication derivation for $q$. The zero case is immediate. In the successor case, unary inversion exposes the tail multiplier, the induction hypothesis identifies the previous product with that tail, and determinacy of $\Cont$ identifies the final appended unit with the canonical successor.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatMul\_unit\_left\_hsame}

\begin{lemma}[Unit right multiplication reads back the multiplicand]
\label{lem:natmul-unit-right-hsame}
For every history $d$, if $\mathsf{NatMul}(d,\Eone(\emp),n)$, then
$\hsame(n,d)$.
\end{lemma}
\begin{proof}
Invert the successor multiplication witness at multiplier $\Eone(\emp)$. Its
predecessor multiplier is $\emp$, so the predecessor product is the zero
constructor result $\emp$. The final continuation is therefore the empty-left
append of $d$, which identifies the result with $d$.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatMul\_unit\_right\_hsame}

\begin{lemma}[Unit right multiplication iff]
\label{lem:nat-mul-unit-right-iff}
For every unary history $d$, multiplication on the right by the unit history
returns exactly the multiplicand up to history-sameness:
$$
\mathsf{UnaryHistory}(d)
\Longrightarrow
\bigl(\mathsf{NatMul}(d,\Eone(\emp),n)\Longleftrightarrow \hsame(n,d)\bigr).
$$
\end{lemma}
\begin{proof}
The forward direction is
\autoref{lem:natmul-unit-right-hsame}. For the reverse direction, reduce along
the displayed history-sameness. The zero multiplication constructor gives
$\mathsf{NatMul}(d,\emp,\emp)$ from the unary carrier, and the successor
constructor appends $d$ to the empty product using the left-unit continuation
law.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatMul\_unit\_right\_iff}

\begin{theorem}[Unit multiplication result forces unit factors]
\label{thm:natmul-unit-result-factors-unit}
If a unary multiplication witness returns the unit history, then both the
multiplicand and the multiplier are the unit history:
$$
\mathsf{NatMul}(d,q,\Eone(\emp))
\Longrightarrow
\HSame(d,\Eone(\emp))\wedge \HSame(q,\Eone(\emp)).
$$
\end{theorem}
\begin{proof}
Invert the multiplication witness at the unit result. The zero constructor is
impossible. In the successor case, continuation inversion at an \(\Eone\)
result either makes the multiplicand empty, contradicting empty-left
multiplication, or exposes a one-headed multiplicand whose tail continuation
has empty result. Empty-result inversion then forces that tail and the
predecessor product to be empty, so the multiplier tail is empty by the
nonempty-multiplicand empty-result exactness theorem.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatMul\_unit\_result\_factors\_unit}

\begin{lemma}[\(\Ezero\)-multiplier exclusion]
\label{thm:natmul-e0-multiplier-absurd}
No unary multiplication witness has an \(\Ezero\)-headed multiplier:
for all histories \(d,q,n\), \(\mathsf{NatMul}(d,\Ezero(q),n)\) is impossible.
\end{lemma}
\begin{proof}
By constructor inversion on the closed relation \(\mathsf{NatMul}\): its
multiplier argument is generated only by \(\emp\) and \(\Eone\)-successor cases.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatMul\_e0\_multiplier\_absurd}

\begin{theorem}[Unary multiplication zero-headed component impossibility]
\label{thm:natmul-zero-headed-component-absurd}
If \(\mathsf{NatMul}(d,q,n)\) holds, then none of \(d\), \(q\), or \(n\) can
be a zero-headed history.
\end{theorem}
\begin{proof}
The multiplication witness exposes unary evidence for the multiplicand,
multiplier, and result. Each possible zero-headed equality contradicts the
corresponding unary-history witness by the zero-extension exclusion theorem.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatMul\_zero\_headed\_component\_absurd}

\begin{theorem}[Empty multiplier exactness for unary multiplication]
\label{thm:natmul-empty-right-iff}
For all histories $d,n$,
$$
\mathsf{NatMul}(d,\emp,n)
\quad\Longleftrightarrow\quad
\mathsf{UnaryHistory}(d)\land \hsame(n,\emp).
$$
\end{theorem}
\begin{proof}
The forward direction inverts the closed $\mathsf{NatMul}$ witness. The only
constructor with empty multiplier is the empty-multiplier row, which records
the unary field for $d$ and returns the empty result. Conversely, if $d$ is
unary and $n\hsame\emp$, identify the endpoint with $\emp$ and apply the
empty-multiplier constructor.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatMul\_empty\_right\_iff}

\begin{theorem}[Nonempty multiplicand and empty multiplication result]
\label{thm:natmul-nonempty-multiplicand-empty-result-iff}
For every unary multiplicand $d$ with $d\not\hsame\emp$, the empty-result
case of $\mathsf{NatMul}(d,q,\emp)$ is equivalent to the empty multiplier
$q\hsame\emp$.
\end{theorem}
\begin{proof}
The forward direction inverts the $\mathsf{NatMul}$ derivation. The zero
constructor gives the empty multiplier. A successor constructor would include
a continuation $\Cont(n,d,\emp)$; empty-result inversion for $\Cont$ forces
$d\hsame\emp$, contradicting the nonempty multiplicand hypothesis. The reverse
direction is the zero constructor after identifying $q$ with $\emp$.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatMul\_nonempty\_multiplicand\_empty\_result\_iff}

\begin{lemma}[Empty unary multiplication result exposes an empty factor]
\label{lem:natmul-empty-result-factor-empty-or-multiplier-empty}
If \(\mathsf{NatMul}(d,q,\emp)\), then either the multiplicand is empty
\(d\hsame\emp\), or the multiplier is empty \(q\hsame\emp\).
\end{lemma}
\begin{proof}
Invert the closed multiplication witness. The zero constructor has multiplier
\(\emp\). A successor constructor ends with a continuation \(\Cont(n,d,\emp)\),
and empty-result inversion for \(\Cont\) forces \(d\hsame\emp\).
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatMul\_empty\_result\_factor\_empty\_or\_multiplier\_empty}

\begin{theorem}[Unary multiplication empty result iff an input is empty]
\label{thm:natmul-empty-result-factor-empty-or-multiplier-empty-iff}
If \(\mathsf{NatMul}(d,q,n)\), then
\[
n\hsame\emp
\Longleftrightarrow
d\hsame\emp \vee q\hsame\emp .
\]
\end{theorem}
\begin{proof}
For the forward implication, transport the displayed result equality to the
empty-result case and apply
\autoref{lem:natmul-empty-result-factor-empty-or-multiplier-empty}.

For the converse implication, split on the empty input. If the multiplicand is
empty, \autoref{lem:natmul-empty-left-result-empty} identifies the product
result with \(\emp\). If the multiplier is empty, the empty-right
multiplication iff gives the canonical empty product for the same multiplicand;
functionality of \(\mathsf{NatMul}\) identifies the displayed result with that
empty product.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatMul\_empty\_result\_iff\_factor\_empty\_or\_multiplier\_empty}

\begin{theorem}[Nonempty unary multiplication factors have nonempty result]
\label{thm:natmul-nonempty-factors-result-not-empty}
If neither \(d\) nor \(q\) is history-same to \(\emp\), then no witness
\(\mathsf{NatMul}(d,q,n)\) can have \(n\hsame\emp\).
\end{theorem}
\begin{proof}
Transport the result equality to the empty-result case and apply the preceding
empty-result factor theorem. Its two alternatives are exactly the two excluded
empty-factor cases.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatMul\_nonempty\_factors\_result\_not\_empty}

\begin{theorem}[Append-continuation multiplication empty result split]
\label{thm:natmul-append-cont-empty-result-empty-factors-iff}
Let \(d\) be unary and not history-same to \(\emp\). If
\(\mathsf{NatMul}(d,w,n)\), \(\mathsf{NatMul}(d,q,e)\), and
\(\mathsf{Cont}(n,e,r)\), then \(r\hsame\emp\) exactly when both
\(w\hsame\emp\) and \(q\hsame\emp\).
\end{theorem}
\begin{proof}
Append-continuation closure first gives a multiplication witness for
\(\mathsf{append}(w,q)\) with result \(r\). The nonempty-multiplicand
empty-result theorem reduces the empty result to an empty appended multiplier,
and append empty-result exactness splits that condition into the two factors.
The converse builds the empty appended multiplier and uses multiplication
functionality to identify the continuation result with \(\emp\).
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatMul\_append\_cont\_empty\_result\_empty\_factors\_iff}

\begin{lemma}[Append multiplier right factor readback]
\label{lem:nat-mul-append-multiplier-right-factor}
Let $d$ and $q$ be unary histories. If $\mathsf{NatMul}(d,w,n)$ and
$\mathsf{NatMul}(d,\mathsf{append}(w,q),r)$ hold, and if $\ContR(n,e,r)$
displays $e$ as the right factor of the result continuation, then
$\mathsf{NatMul}(d,q,e)$.
\end{lemma}
\begin{proof}
Use totality of unary multiplication to build a product $e'$ for $d$ and $q$.
Append-continuation closure then produces a product for
$\mathsf{append}(w,q)$ with result $\mathsf{append}(n,e')$. Functionality of
$\mathsf{NatMul}$ identifies this result with the displayed $r$. Transport the
continuation $\ContR(n,e',\mathsf{append}(n,e'))$ across that result equality,
left-cancel it against $\ContR(n,e,r)$, and transport the product for $q$ from
$e'$ to $e$.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatMul\_append\_multiplier\_right\_factor}

\begin{definition}[Divisibility relation]
\label{def:nat-divides}
For unary histories \(d, n\), the divisibility relation \(\mathsf{NatDivides}(d, n)\) holds when there exists a unary \(q\) with \(\mathsf{NatMul}(d, q, n)\). The witness \(q\) is the explicit quotient.
\end{definition}
\leandef{BEDC.Derived.PrimeUp.NatDivides}

\begin{theorem}[Divisibility result unary carrier]
\label{thm:nat-divides-result-unary}
If $\mathsf{NatDivides}(d,n)$ holds, then $n$ is a unary history.
\end{theorem}
\begin{proof}
Unpack the quotient witness. The multiplication witness exposes unary $d$,
and multiplication result closure then proves the result history $n$ unary.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatDivides\_result\_unary}

\begin{lemma}[Divisibility result cases]
\label{lem:nat-divides-result-cases}
Every witness of \(\mathsf{NatDivides}(d,n)\) is either an empty-result
witness, with \(n\hsame\emp\) and \(d\) unary, or a successor-product witness:
there is a predecessor \(n_0\) with \(\mathsf{NatDivides}(d,n_0)\) and
\(\Cont(n_0,d,n)\).
\end{lemma}
\begin{proof}
Unpack the explicit quotient in the divisibility witness. If its
\(\mathsf{NatMul}\) witness is zero, the result is empty and the multiplicand
field gives unary \(d\). If it is a successor witness, unary inversion on the
quotient supplies a unary predecessor quotient, and the predecessor
multiplication witness repackages as \(\mathsf{NatDivides}(d,n_0)\).
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatDivides\_result\_cases}

\begin{lemma}[Empty left divisibility has empty result]
\label{lem:natdivides-empty-left-result-empty}
If \(\mathsf{NatDivides}(\emp,n)\), then \(n\hsame\emp\).
\end{lemma}
\begin{proof}
Unpack the quotient witness for divisibility. Its multiplication field is a
witness of \(\mathsf{NatMul}(\emp,q,n)\), so empty-left multiplication
exactness identifies the result \(n\) with \(\emp\).
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatDivides\_empty\_left\_result\_empty}

\begin{theorem}[Successor-result divisors have positive shape]
\label{thm:natdivides-successor-result-divisor-positive-shape}
If \(\mathsf{NatDivides}(d,\Eone(n))\), then \(d\) has positive unary shape:
there is a unary history \(t\) with \(d\hsame\Eone(t)\).
\end{theorem}
\begin{proof}
Unpack the divisibility witness and read the unaryness of \(d\) from its
multiplication field. If \(d\) were empty, empty-left divisibility would force
\(\Eone(n)\hsame\emp\), contradicting successor non-emptiness. The \(e0\)
case is excluded by unary-history inversion. Hence \(d\) is \(e1\)-headed, and
the unary tail is obtained by unary inversion.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatDivides\_successor\_result\_divisor\_positive\_shape}

\begin{theorem}[Empty-left divisibility exactness]
\label{thm:nat-divides-empty-left-iff}
For every history $n$, divisibility by the empty unary history holds exactly
when the result is empty:
$$
\mathsf{NatDivides}(\emp,n)\Longleftrightarrow \hsame(n,\emp).
$$
\end{theorem}
\begin{proof}
The forward implication is empty-left divisibility result exactness. For the
reverse implication, identify $n$ with $\emp$; then the empty quotient and the
zero multiplication constructor witness $\mathsf{NatDivides}(\emp,\emp)$.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatDivides\_empty\_left\_iff}

\begin{lemma}[Empty right divisibility is unary carrier]
\label{thm:natdivides-empty-right-iff}
For any history $d$, $\mathsf{NatDivides}(d,\emp)$ holds exactly when $d$ is a
unary history.
\end{lemma}
\begin{proof}
In the forward direction, unpack the quotient witness; its multiplication
field carries the unary multiplicand field for $d$. In the reverse direction,
use the empty quotient and the zero constructor of $\mathsf{NatMul}$.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatDivides\_empty\_right\_iff}

\begin{theorem}[Unit-left divisibility is unary carrier]
\label{thm:natdivides-unit-left-iff}
For any history $n$, $\mathsf{NatDivides}(\Eone(\emp),n)$ holds exactly when
$n$ is a unary history.
\end{theorem}
\begin{proof}
For the forward direction, unpack the quotient witness $q$. Multiplication by
the unit identifies the result with $q$, and unaryness transports across that
endpoint identification. Conversely, the unit-divisibility half of reflexivity
gives $\mathsf{NatDivides}(\Eone(\emp),n)$ for every unary $n$.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatDivides\_unit\_left\_iff}

\begin{theorem}[Unit-right divisibility exactness]
\label{thm:nat-divides-unit-right-iff}
For any history $d$, $\mathsf{NatDivides}(d,\Eone(\emp))$ holds exactly when
$d\hsame\Eone(\emp)$.
\end{theorem}
\begin{proof}
In the forward direction, unpack the quotient witness and apply unit-result
multiplication factor exactness: both factors of a product equal to
\(\Eone(\emp)\) are the unary unit. Conversely, identify \(d\) with the unary
unit and use reflexive divisibility for the unit history.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatDivides\_unit\_right\_iff}

\begin{lemma}[Divisibility is reflexive on the unit and on every unary history]
\label{lem:divides-reflexive}
For every unary \(n\), \(\mathsf{NatDivides}(\Eone(\emp), n)\) and \(\mathsf{NatDivides}(n, n)\).
\end{lemma}
\begin{proof}
The unit case takes \(q = n\) in \(\mathsf{NatMul}(\Eone(\emp), n, n)\) by induction. The self-divisibility case takes \(q = \Eone(\emp)\) and uses the continuation unit at the empty quotient.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatDivides\_reflexive}
\leanchecked{BEDC.Derived.PrimeUp.NatDivides\_unit\_self\_reflexive}
\leanchecked{BEDC.Derived.PrimeUp.NatDivides\_reflexive\_pair}

\begin{lemma}[Divisibility is transitive]
\label{lem:divides-transitive}
If \(\mathsf{NatDivides}(d, e)\) and \(\mathsf{NatDivides}(e, n)\), then \(\mathsf{NatDivides}(d, n)\).
\end{lemma}
\begin{proof}
From the witnessing quotients \(q_1\) with \(\mathsf{NatMul}(d, q_1, e)\) and \(q_2\) with \(\mathsf{NatMul}(e, q_2, n)\), the composite quotient is \(q_1 \cdot q_2\), itself constructed as a unary history by \autoref{lem:nat-mul-total}.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatDivides\_transitive}

\begin{lemma}[Divisibility is closed under displayed continuation]
\label{thm:nat-divides-cont-closed}
If $\mathsf{NatDivides}(d,x)$ and $\mathsf{NatDivides}(d,y)$ hold, and
$\Cont(x,y,z)$ displays $z$ as the continuation of the two divisible histories,
then $\mathsf{NatDivides}(d,z)$ holds.
\end{lemma}
\begin{proof}
Unpack the two quotient witnesses, say $qx$ and $qy$. Their multiplication
witnesses have results $x$ and $y$. The quotient for $z$ is
$\mathsf{append}(qx,qy)$; unary append closure makes it an admissible quotient,
and append-continuation multiplication closure combines the two multiplication
witnesses over the displayed $\Cont(x,y,z)$.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatDivides\_cont\_closed}

\begin{lemma}[Divisors of a nonempty unary product are bounded by the result]
\label{lem:natdivides-result-boundary}
If $\mathsf{NatDivides}(d,n)$, $n$ is unary, and $n\not\hsame\emp$, then
either $d\hsame n$ or $\mathsf{NatUnaryStrictPrefix}(d,n)$.
\end{lemma}
\begin{proof}
Unpack the quotient $q$ in the divisor witness and inspect the corresponding
$\mathsf{NatMul}(d,q,n)$ witness. The empty quotient would make the product
result empty, contradicting $n\not\hsame\emp$.

For the unit quotient, multiplication by the unit identifies the product result
with $d$, so the divisor is the whole result up to $\hsame$.

For a successor quotient, the multiplication witness writes $n$ as the previous
product followed by one more copy of $d$. The final $\Cont$ payload gives a
strict unary prefix from that copy into the result; unary append commutativity
and strict-prefix transport move the prefix endpoint to $d$, yielding
$\mathsf{NatUnaryStrictPrefix}(d,n)$.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatDivides\_nonempty\_result\_boundary}

\begin{theorem}[Prime divisor of a unary product exposes a factor]
\label{thm:prime-divides-product-factor}
For histories \(p,a,b,n\), if \(\mathsf{NatPrime}(p)\), \(\mathsf{NatMul}(a,b,n)\), and \(\mathsf{NatDivides}(p,n)\), then \(\mathsf{NatDivides}(p,a)\) or \(\mathsf{NatDivides}(p,b)\).
\end{theorem}
\begin{proof}[Sketch]
Induct on the concrete unary multiplication witness \(\mathsf{NatMul}(a,b,n)\), viewing the right factor \(b\) as the induction spine.

In the zero-product case, the result is \(\emp\). A divisor witness for \(p\) into \(\emp\) would make \(\emp\) a multiple of the prime \(p\). The non-empty condition in \(\mathsf{NatPrime}(p)\), together with the empty-result inversion for \(\mathsf{NatMul}\), rules this case out, so the conclusion follows vacuously.

In the successor case, the product is obtained by appending one more copy of \(a\) to the previous product. If \(p\) already divides the previous product, the induction hypothesis gives \(\mathsf{NatDivides}(p,a)\) or divisibility into the predecessor of \(b\); in the latter subcase, the successor multiplication constructor and \autoref{lem:divides-transitive} transport the divisor to \(\mathsf{NatDivides}(p,b)\).

It remains when the final appended copy is the first place where the divisor can occur. The \(\mathsf{NatMul}\) result and functionality facts identify the last continuation with the factor \(a\) up to \(\hsame\). Applying the defining divisor alternative in \(\mathsf{NatPrime}(p)\) to this non-trivial divisor yields either the unit case, which is incompatible with the prime's strict-greater field, or \(\hsame\)-equality with \(a\). This gives \(\mathsf{NatDivides}(p,a)\), completing the induction.
\end{proof}
