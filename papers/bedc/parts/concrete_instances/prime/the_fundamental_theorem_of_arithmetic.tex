\section{The fundamental theorem of arithmetic}

\begin{definition}[Prime factorization list]
\label{def:prime-factorization-list}
A prime factorization of a unary history \(n\) is a finite list of unary primes \([p_1, \dots, p_k]\) (in non-decreasing unary order) such that the iterated multiplicative continuation \(p_1 \cdot p_2 \cdots p_k\) yields a unary history \(\hsame\)-equal to \(n\).
\end{definition}
\leandef{BEDC.Derived.PrimeUp.NatPrimeFactorization}
\leandef{BEDC.Derived.PrimeUp.PrimeFactorizationProduct}
\leandef{BEDC.Derived.PrimeUp.PrimeFactorization}

\begin{theorem}[Prime factorization cons head divides]
\label{thm:prime-factorization-cons-head-divides}
If a prime factorization list of \(n\) has head \(p\), then \(p\) is prime and
divides \(n\):
$$
\mathsf{PrimeFactorization}(n,p::ps)
\Longrightarrow
\mathsf{NatPrime}(p)\land\mathsf{NatDivides}(p,n).
$$
\end{theorem}
\begin{proof}
Unfold the cons clause of the concrete factorization-product predicate. It
exposes the head primality proof, a tail product, and the multiplication
witness \(\mathsf{NatMul}(p,\mathsf{tail},n)\). The multiplier unary field of
that witness supplies the quotient carrier required by \(\mathsf{NatDivides}\).
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.PrimeFactorization\_cons\_head\_divides}

\begin{theorem}[Prime factorization product result transport]
\label{thm:prime-factorization-product-result-hsame-transport}
If a prime factorization product yields \(n\), and \(n\hsame n'\), then the
same factorization product yields \(n'\).
\end{theorem}
\begin{proof}
Inspect the concrete list-product predicate. The empty product clause is
transported by transitivity of \(\hsame\). In the cons clause, the tail product
is unchanged, and the displayed \(\mathsf{NatMul}\) result witness is
transported along \(n\hsame n'\).
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.PrimeFactorizationProduct\_result\_hsame\_transport}

\begin{theorem}[Prime factorization products have unary result]
\label{thm:prime-factorization-product-result-unary}
If \(\mathsf{PrimeFactorizationProduct}(ps,n)\), then \(n\) is a unary
history.
\end{theorem}
\begin{proof}
Induct on the factor list. The empty list identifies the product with the
unit endpoint \(\Eone(\emp)\), which is unary, and unary carrier transport
moves this along \(\hsame\). In the cons case, the head prime supplies a unary
multiplicand and the displayed \(\mathsf{NatMul}\) witness supplies unary
closure of the result.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.PrimeFactorizationProduct\_result\_unary}

\begin{theorem}[Prime factorization products have nonempty result]
\label{thm:prime-factorization-product-result-not-empty}
If \(\mathsf{PrimeFactorizationProduct}(ps,n)\), then \(n\) is not
history-same to \(\emp\).
\end{theorem}
\begin{proof}
Induct on the list \(ps\). The empty product is \(\Eone(\emp)\), which is
separated from \(\emp\) by constructor no-confusion. In the cons case, the
head prime is nonempty, the induction hypothesis makes the tail product
nonempty, and the nonempty-factor multiplication theorem applies to the
displayed \(\mathsf{NatMul}\) witness.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.PrimeFactorizationProduct\_result\_not\_empty}

\begin{definition}[Natural prime product relation]
\label{def:nat-prime-product-relation}
The relation \(\mathsf{NatPrimeProduct}(ps,n)\) is the closed list product
relation generated by the empty product
\(\mathsf{NatPrimeProduct}([], \Eone(\emp))\), and by the cons rule: if
\(\mathsf{NatPrime}(p)\), \(\mathsf{NatPrimeProduct}(ps,m)\), and
\(\mathsf{NatMul}(p,m,n)\), then
\(\mathsf{NatPrimeProduct}(p::ps,n)\).
\end{definition}
\leandef{BEDC.Derived.PrimeUp.NatPrimeProduct}

\begin{theorem}[Natural prime products have nonempty result]
\label{thm:nat-prime-product-result-not-empty}
If \(\mathsf{NatPrimeProduct}(ps,n)\), then \(n\) is not history-same to
\(\emp\).
\end{theorem}
\begin{proof}
Induct on the list-product witness. The empty product is \(\Eone(\emp)\), so
history no-confusion separates it from \(\emp\). In the cons case, primality
exposes the head factor as a nonempty one-headed unary history, the induction
hypothesis gives nonemptiness of the tail product, and the nonempty-factor
multiplication result theorem applies to the displayed \(\mathsf{NatMul}\)
witness.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatPrimeProduct\_result\_not\_empty}

\begin{theorem}[Natural prime factorizations have nonempty result]
\label{thm:nat-prime-factorization-result-not-empty}
If \(\mathsf{NatPrimeFactorization}(ps,n)\), then \(n\) is not history-same
to \(\emp\).
\end{theorem}
\begin{proof}
Induct on the concrete factorization witness. The unit factorization has
result \(\Eone(\emp)\), separated from \(\emp\) by constructor no-confusion.
The singleton case is prime nonemptiness. In the cons case, primality makes
the head factor nonempty, the induction hypothesis makes the tail product
nonempty, and nonempty-factor multiplication excludes an empty result.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatPrimeFactorization\_result\_not\_empty}

\begin{theorem}[Natural prime products have unary result]
\label{thm:nat-prime-product-result-unary}
If \(\mathsf{NatPrimeProduct}(ps,n)\), then \(n\) is a unary history.
\end{theorem}
\begin{proof}
Induct on the list-product witness. The empty product is \(\Eone(\emp)\),
hence unary by unary successor closure over the empty history. In the cons
case, primality supplies the unary carrier of the head factor, and the
multiplication-result carrier theorem applies to the displayed
\(\mathsf{NatMul}\) witness.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatPrimeProduct\_result\_unary}

\begin{theorem}[Natural prime product cons tail divides]
\label{thm:nat-prime-product-cons-tail-divides}
If \(\mathsf{NatPrimeProduct}(p::ps,n)\), then \(p\) is unary and there is a
tail product \(m\) such that \(\mathsf{NatPrimeProduct}(ps,m)\) and
\(\mathsf{NatDivides}(m,n)\).
\end{theorem}
\begin{proof}
Invert the cons constructor of \(\mathsf{NatPrimeProduct}\). The head prime
gives the unary carrier of \(p\), while the tail witness gives the product
\(m\). The preceding unary-result theorem supplies the unary carrier of \(m\),
and the multiplication closure theorem turns the displayed
\(\mathsf{NatMul}(p,m,n)\) witness into \(\mathsf{NatDivides}(m,n)\).
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatPrimeProduct\_cons\_tail\_divides}

\begin{theorem}[Existence of prime factorization]
\label{thm:ftoa-existence}
For every unary history \(n\) with \(\mathsf{NatGT}(n, \emp)\), there exists a prime factorization list of \(n\). The list is constructed explicitly by repeated trial division using \autoref{thm:prime-decidable}.
\end{theorem}
\begin{proof}[Sketch]
Strong induction on the unary history \(n\) along \(\mathsf{NatUnaryStrictPrefix}\). If \(n = \Eone(\emp)\), the empty list factors it. Otherwise apply \autoref{thm:prime-decidable}: either \(n\) is prime (return \([n]\)) or a non-trivial divisor \(d\) is exhibited; recurse on \(d\) and on the quotient \(q = n / d\), both strictly smaller than \(n\), and concatenate the resulting lists.
\end{proof}

\begin{theorem}[Uniqueness of prime factorization]
\label{thm:ftoa-uniqueness}
If \([p_1, \dots, p_j]\) and \([q_1, \dots, q_k]\) are two prime factorization lists of the same unary history \(n\), then \(j = k\) and the lists agree pointwise up to \(\hsame\).
\end{theorem}
\begin{proof}[Sketch]
Apply \autoref{thm:prime-divides-product-factor} to the leading prime of one factorization and the iterated product represented by the other factorization. Repeating this product-factor exposure across the second list locates a matching prime factor up to \(\hsame\).

Cancel the matched factor using the \(\mathsf{NatMul}\) result and functionality facts, then continue by induction on the remaining lists. The induction gives pointwise agreement up to \(\hsame\) and length agreement \(j = k\).
\end{proof}
