\section{The certificate}

\begin{definition}[Prime source specification]
\label{def:prime-source-specification}
The source specification on \(\mathsf{Unary}\) carries the additional constraint \(\mathsf{NatPrime}(p)\). Its generator is the prime predicate of \autoref{def:nat-prime}.
\end{definition}

\begin{definition}[Prime pattern specification]
\label{def:prime-pattern-specification}
The pattern of \(\PrimeUp\) is trial-division-bounded primality: for an input unary \(n\), the pattern is the trial-division predicate \(\mathsf{TrialDiv}(n, n)\) restricted to inputs satisfying \(\mathsf{NatPrime}\) at the upper bound.
\end{definition}

\begin{definition}[Prime classifier specification]
\label{def:prime-classifier-specification}
The classifier on prime carriers is \(\hsame\) restricted to unary primes. Two unary primes classify together iff they are \(\hsame\)-equal as histories.
\end{definition}

\begin{definition}[Prime stability certificate]
\label{def:prime-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item closure of the prime predicate under \(\hsame\): if \(\mathsf{NatPrime}(p)\) and \(\hsame(p, p')\), then \(\mathsf{NatPrime}(p')\);
  \item transport of divisibility under \(\hsame\): if \(\mathsf{NatDivides}(d, n)\), \(\hsame(d, d')\), and \(\hsame(n, n')\), then \(\mathsf{NatDivides}(d', n')\);
  \item compatibility of the trial-division witness with \(\hsame\) on bounds.
\end{enumerate}
\end{definition}
\leanchecked{BEDC.Derived.PrimeUp.PrimeStabilityCert\_fields}

\begin{definition}[Prime ledger policy]
\label{def:prime-ledger-policy}
The ledger policy records:
\begin{enumerate}
  \item composition of factorizations: concatenation of prime factorization lists for products;
  \item product-factor exposure by \autoref{thm:prime-divides-product-factor};
  \item infinitude closure: \(\mathsf{NatNextPrime}\) of \autoref{thm:euclid-infinite} produces a new prime exceeding any given bound.
\end{enumerate}
\end{definition}

\begin{theorem}[Concrete \(\PrimeUp\) naming certificate]
\label{thm:prime-namecert}
The data \((\mathsf{NatPrime}, \mathsf{TrialDiv}, \hsame, \mathsf{PrimeStabilityCert}, \mathsf{PrimeLedgerPolicy})\) of \autoref{def:prime-source-specification}--\autoref{def:prime-ledger-policy} forms a naming certificate over \(\Hist\) for the prime number interface \(\PrimeUp\). The certificate is decidable (\autoref{thm:prime-decidable}), unbounded (\autoref{thm:euclid-infinite}), and exhaustive (\autoref{thm:ftoa-existence}, \autoref{thm:ftoa-uniqueness}).
\end{theorem}
\begin{proof}[Sketch]
The five fields satisfy the closure laws of the naming-certificate framework (\autoref{ch:core-typed-naming-certificates}). Decidability is \autoref{thm:prime-decidable}. Unboundedness is \autoref{thm:euclid-infinite}. Exhaustiveness follows from existence (\autoref{thm:ftoa-existence}) together with uniqueness (\autoref{thm:ftoa-uniqueness}). The certificate uses no \(\mathsf{Classical.choice}\), no \(\mathsf{Quot.sound}\), and no \(\mathsf{propext}\), so its Lean targets fall under the project's axiom-purity gate.
\end{proof}

\begin{theorem}[Semantic \(\PrimeUp\) naming certificate]
\label{thm:prime-semantic-namecert}
The concrete certificate of \autoref{thm:prime-namecert} satisfies the semantic naming-certificate schema of \(\NameCert\) (\autoref{ch:core-typed-naming-certificates}): the carrier is a \(\Hist\)-predicate, the classifier is a restriction of \(\hsame\), and every closure law is given by an explicit constructive proof.
\end{theorem}
\begin{proof}[Sketch]
Direct verification of the four \(\NameCert\) field types using the constructions of this chapter.
\end{proof}

\begin{theorem}[Unary multiplication result semantic name certificate]
\label{thm:natmul-result-semantic-namecert}
For unary histories $d$ and $q$, the result predicate
$\mathsf{NatMul}(d,q,-)$, used as source, pattern, and ledger policy, together
with $\hsame$ as classifier, forms a semantic naming certificate.
\end{theorem}
\begin{proof}
Totality of unary multiplication supplies a carried result witness. The
classifier laws are exactly reflexivity, symmetry, and transitivity of
$\hsame$. Carrier transport is result transport for $\mathsf{NatMul}$ along
history-sameness, while the pattern and ledger maps are the shared concrete
result predicate.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatMul\_result\_semanticNameCert}

\begin{theorem}[Divisibility semantic name certificate]
\label{thm:nat-divides-semantic-namecert}
For a unary divisor $d$, the predicate $\mathsf{NatDivides}(d,-)$, used as
source, pattern, and ledger policy, together with $\hsame$ as classifier,
forms a semantic naming certificate.
\end{theorem}
\begin{proof}
The empty dividend is inhabited exactly by the empty-right divisibility
characterization. The classifier laws are inherited from $\hsame$, and carrier
transport is dividend transport for $\mathsf{NatDivides}$ along
history-sameness. Since the source, pattern, and ledger predicates coincide,
the semantic soundness fields are direct projections of the same witness.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatDivides\_semanticNameCert}

\begin{lemma}[Empty left factor has empty unary product]
\label{lem:natmul-empty-left-empty-result}
If $\mathsf{NatMul}(\emp,q,n)$, then $n\hsame\emp$.
\end{lemma}
\begin{proof}
Induct on the multiplication witness from \autoref{def:nat-multiplication}. The zero clause has result $\emp$. In the successor clause, the inductive result identifies the predecessor product with $\emp$, and the final continuation has empty right input. By the continuation unit law of \autoref{lem:unary-continuation-unit}, the successor result is history-same to that predecessor, hence to $\emp$.
\end{proof}

\begin{lemma}[Nonempty quotient exposes a multiplicand extension]
\label{lem:natmul-nonempty-quotient-product-extension}
Let $d$ and $q$ be unary histories. If $\mathsf{NatMul}(d,q,n)$ and $q\not\hsame\emp$, then there is a unary history $t$ with $\Cont(d,t,n)$.
\end{lemma}
\begin{proof}
Induct on the unary generation of $q$ from \autoref{def:unary-history-judgment}. The empty case contradicts $q\not\hsame\emp$, and the zero-headed case is impossible by \autoref{lem:unary-no-zero-extension}. In the one-headed case write $q=\Eone(q_0)$. By \autoref{lem:nat-mul-succ-inversion}, the multiplication witness has a predecessor product $p$ with $\mathsf{NatMul}(d,q_0,p)$ and $\Cont(p,d,n)$.

If $q_0\hsame\emp$, the zero clause of \autoref{def:nat-multiplication} identifies $p$ with $\emp$. The displayed continuation is therefore the left-unit continuation from $\emp$ by $d$, so $n\hsame d$; transporting the right-unit continuation of $d$ across this identification gives $\Cont(d,\emp,n)$, with unary tail $\emp$.

If $q_0\not\hsame\emp$, the induction hypothesis applied to $\mathsf{NatMul}(d,q_0,p)$ gives a unary tail $t$ with $\Cont(d,t,p)$. Composing this continuation with $\Cont(p,d,n)$ and using continuation associativity gives $\Cont(d,\mathsf{append}(t,d),n)$. The tail $\mathsf{append}(t,d)$ is unary because continuation preserves unary histories.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatMul\_nonempty\_quotient\_product\_extension}

\begin{lemma}[Divisibility exposes unary divisor]
\label{lem:nat-divides-divisor-unary}
If $\mathsf{NatDivides}(d,n)$ holds, then $d$ is a unary history.
\end{lemma}
\begin{proof}
Unpack the quotient witness in \(\mathsf{NatDivides}(d,n)\). The
corresponding multiplication witness has \(d\) as its multiplicand, and
the multiplicand readback for closed multiplication derivations reads the
unary carrier directly from it.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatDivides\_divisor\_unary}

\begin{theorem}[Divisibility antisymmetry up to history-sameness]
\label{thm:nat-divides-antisymmetry-hsame}
Let $d$ and $e$ be unary histories. If $\mathsf{NatDivides}(d,e)$ and $\mathsf{NatDivides}(e,d)$, then $d\hsame e$.
\end{theorem}
\begin{proof}
Unpack \autoref{def:nat-divides}. Choose unary quotients $q$ and $r$ with $\mathsf{NatMul}(d,q,e)$ and $\mathsf{NatMul}(e,r,d)$.

If $q\hsame\emp$, the first multiplication witness has empty multiplier, so $e\hsame\emp$. After replacing $e$ by $\emp$ in the second witness, \autoref{lem:natmul-empty-left-empty-result} gives $d\hsame\emp$, hence $d\hsame e$. The case $r\hsame\emp$ is symmetric.

It remains to consider $q\not\hsame\emp$ and $r\not\hsame\emp$. By \autoref{lem:natmul-nonempty-quotient-product-extension}, there are unary tails $t$ and $s$ with $\Cont(d,t,e)$ and $\Cont(e,s,d)$. By closed unary generation, each tail is either $\emp$ or a one-extension. If $t\hsame\emp$, the first continuation is a right-unit continuation and gives $e\hsame d$ by \autoref{lem:unary-continuation-unit}; symmetry gives $d\hsame e$. If $s\hsame\emp$, the second continuation gives $d\hsame e$ directly. If both tails are one-extensions, the two continuations are strict unary prefix witnesses in the sense of \autoref{def:nat-unary-strict-prefix}, contradicting \autoref{thm:nat-unary-strict-prefix-asymm}. From that contradiction the desired history-sameness follows.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatDivides\_antisymmetry\_hsame}

\begin{lemma}[Unary multiplication transports along the multiplicand]
\label{lem:nat-mul-multiplicand-hsame-transport}
Let $d,q,n$ be unary histories, and let $d'$ be a history with $d\hsame d'$. If $\mathsf{NatMul}(d,q,n)$, then $\mathsf{NatMul}(d',q,n)$.
\end{lemma}
\begin{proof}
By \autoref{lem:unary-transport}, $d'$ is unary. Induct on the closed $\mathsf{NatMul}(d,q,n)$ witness from \autoref{def:nat-multiplication}. In the zero-multiplier case, the result is $\emp$, and the zero constructor with the unary witness for $d'$ gives $\mathsf{NatMul}(d',\emp,\emp)$. In the successor case, the induction hypothesis gives $\mathsf{NatMul}(d',q,n_0)$ for the predecessor product. The original last step is a continuation $\Cont(n_0,d,n_1)$. Transport that continuation along reflexivity on $n_0$, the hypothesis $d\hsame d'$, and reflexivity on $n_1$ using \autoref{thm:cont-hsame-transport}; this gives $\Cont(n_0,d',n_1)$. Applying the successor constructor of \autoref{def:nat-multiplication} yields $\mathsf{NatMul}(d',\Eone(q),n_1)$.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatMul\_multiplicand\_hsame\_transport}

\begin{lemma}[Unary multiplication transports along the multiplier]
\label{lem:natmul-multiplier-hsame-transport}
Let $d,q,n$ be unary multiplication histories, and let $q'$ be a history with
$q\hsame q'$. If $\mathsf{NatMul}(d,q,n)$, then $q'$ is unary and
$\mathsf{NatMul}(d,q',n)$.
\end{lemma}
\begin{proof}
Case-analyze the displayed history-sameness $q\hsame q'$. In the reflexive
case the original multiplication witness already has multiplier $q'$, and the
right-unary theorem for $\mathsf{NatMul}$ supplies the unary witness for that
multiplier.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatMul\_multiplier\_hsame\_transport}

\begin{lemma}[Unary multiplication successor result-continuation transport]
\label{lem:nat-mul-succ-result-cont-transport}
Let $d,q,n$ be unary histories, and let $n'$ be a history with $n\hsame n'$. If $\mathsf{NatMul}(d,\Eone(q),n)$, then there is a predecessor product $p$ with $\mathsf{NatMul}(d,q,p)$ and $\Cont(p,d,n')$.
\end{lemma}
\begin{proof}
Invert the successor multiplication witness to obtain a predecessor product $p$ and the final continuation $\Cont(p,d,n)$. Transport this continuation along the displayed result history-sameness, leaving the predecessor product unchanged.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatMul\_succ\_result\_cont\_transport}

\begin{theorem}[Divisibility transports along divisor history-sameness]
\label{thm:nat-divides-divisor-hsame-transport}
Let $d$ and $e$ be unary histories, and let $d'$ be any history with $d\hsame d'$. If $\mathsf{NatDivides}(d,e)$, then $\mathsf{NatDivides}(d',e)$.
\end{theorem}
\begin{proof}
Unpack \autoref{def:nat-divides}. Choose a unary quotient $q$ with $\mathsf{NatMul}(d,q,e)$. By \autoref{lem:nat-mul-multiplicand-hsame-transport}, the same quotient satisfies $\mathsf{NatMul}(d',q,e)$. Since the quotient witness itself remains unary, the existential clause in \autoref{def:nat-divides} gives $\mathsf{NatDivides}(d',e)$.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatDivides\_divisor\_hsame\_transport}

\begin{lemma}[Unary multiplication transports along the result]
\label{lem:nat-mul-result-hsame-transport}
Let $d,q,n$ be unary histories, and let $n'$ be a history with $n\hsame n'$. If $\mathsf{NatMul}(d,q,n)$, then $\mathsf{NatMul}(d,q,n')$.
\end{lemma}
\begin{proof}
Case-analyze the closed witness for $\mathsf{NatMul}(d,q,n)$ from \autoref{def:nat-multiplication}. In the zero-multiplier case the result is $\emp$; by closed generation for $\hsame$ in \autoref{def:history-sameness}, the target $n'$ is also $\emp$, so the zero constructor gives $\mathsf{NatMul}(d,\emp,n')$.

In the successor case, the witness consists of a predecessor product $\mathsf{NatMul}(d,q_0,n_0)$ and a continuation $\Cont(n_0,d,n_1)$. The result hypothesis is $n_1\hsame n'$. Apply \autoref{thm:cont-hsame-transport} to this continuation, using reflexivity from \autoref{thm:history-sameness-equivalence} on $n_0$ and $d$ and the displayed result sameness. This yields $\Cont(n_0,d,n')$, and the successor constructor of \autoref{def:nat-multiplication} gives $\mathsf{NatMul}(d,\Eone(q_0),n')$.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatMul\_result\_hsame\_transport}

\begin{lemma}[Unary multiplication append multiplier totality]
\label{lem:nat-mul-append-multiplier-total}
Let $d,w,n,q$ be histories. If $\mathsf{NatMul}(d,w,n)$ and $q$ is unary, then
there is a unary result $r$ with
$$
\mathsf{NatMul}(d,\mathsf{append}(w,q),r).
$$
\end{lemma}
\begin{proof}
Read the unary multiplicand from the existing multiplication witness. Totality
of unary multiplication supplies a unary product $e$ for the appended
multiplier tail $q$. The continuation from $n$ to $e$ gives the displayed result
$r=\mathsf{append}(n,e)$, and append-continuation closure for
$\mathsf{NatMul}$ combines the two multiplication witnesses into the product
over $\mathsf{append}(w,q)$.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatMul\_append\_multiplier\_total}

\begin{theorem}[Divisibility transports along dividend history-sameness]
\label{thm:nat-divides-dividend-hsame-transport}
Let $d$ and $e$ be unary histories, and let $e'$ be a history with $e\hsame e'$. If $\mathsf{NatDivides}(d,e)$, then $\mathsf{NatDivides}(d,e')$.
\end{theorem}
\begin{proof}
By \autoref{lem:unary-transport}, the history $e'$ is unary, so it is an admissible dividend endpoint. Unpack \autoref{def:nat-divides}. Choose a unary quotient $q$ with $\mathsf{NatMul}(d,q,e)$. By \autoref{lem:nat-mul-result-hsame-transport}, the same quotient satisfies $\mathsf{NatMul}(d,q,e')$. The quotient witness remains unary, so the existential clause in \autoref{def:nat-divides} gives $\mathsf{NatDivides}(d,e')$.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatDivides\_dividend\_hsame\_transport}

\begin{corollary}[Divisibility transports along both endpoints]
\label{cor:nat-divides-endpoint-hsame-transport}
Let $d$ and $e$ be unary histories, and let $d'$ and $e'$ be histories with $d\hsame d'$ and $e\hsame e'$. If $\mathsf{NatDivides}(d,e)$, then $\mathsf{NatDivides}(d',e')$.
\end{corollary}
\begin{proof}
Unary transport puts $d'$ and $e'$ in the unary carrier. First apply \autoref{thm:nat-divides-divisor-hsame-transport} to transport the divisor from $d$ to $d'$, obtaining $\mathsf{NatDivides}(d',e)$. Then apply \autoref{thm:nat-divides-dividend-hsame-transport} to transport the dividend from $e$ to $e'$. The result is $\mathsf{NatDivides}(d',e')$.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatDivides\_endpoint\_hsame\_transport}
