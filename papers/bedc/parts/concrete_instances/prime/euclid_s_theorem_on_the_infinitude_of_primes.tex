\section{Euclid's theorem on the infinitude of primes}

\begin{definition}[Unary factorial]
\label{def:unary-factorial}
The unary factorial \(\mathsf{NatFact}(n, n!)\) is the closed inductive predicate generated by:
\begin{enumerate}
  \item \(\mathsf{NatFact}(\emp, \Eone(\emp))\) (zero factorial is one);
  \item if \(\mathsf{NatFact}(n, m)\) and \(\mathsf{NatMul}(\Eone(n), m, m')\), then \(\mathsf{NatFact}(\Eone(n), m')\).
\end{enumerate}
\end{definition}
\leandef{BEDC.Derived.PrimeUp.NatFact}

\begin{lemma}[Unary factorial inputs are unary]
\label{lem:natfact-input-unary}
Every closed witness of $\mathsf{NatFact}(n,m)$ has a unary input $n$.
\end{lemma}
\begin{proof}
Induct on the $\mathsf{NatFact}$ witness. The zero constructor has input
$\emp$, which is unary. The successor constructor changes the input from $n$
to $\Eone(n)$, so unary closure under $\Eone$ applied to the induction
hypothesis gives the result.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatFact\_input\_unary}

\begin{lemma}[Unary factorial successor inversion]
\label{lem:natfact-successor-inversion}
If $\mathsf{NatFact}(\Eone(n),m')$, then there is an $m$ such that
$\mathsf{NatFact}(n,m)$ and $\mathsf{NatMul}(\Eone(n),m,m')$.
\end{lemma}
\begin{proof}
Invert the closed factorial witness. The zero constructor has empty input and
therefore cannot produce an $\Eone$ input. The successor constructor exposes
exactly the predecessor factorial witness and the multiplication witness.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatFact\_successor\_inversion}

\begin{theorem}[Factorial successor input divides its result]
\label{thm:natfact-successor-input-divides-result}
If \(\mathsf{NatFact}(\Eone(n),m)\), then the successor input
\(\Eone(n)\) divides the factorial result \(m\):
$$
\mathsf{NatFact}(\Eone(n),m)\Longrightarrow
\mathsf{NatDivides}(\Eone(n),m).
$$
\end{theorem}
\begin{proof}
Successor inversion exposes a predecessor factorial value \(m_0\) together
with a multiplication witness
\(\mathsf{NatMul}(\Eone(n),m_0,m)\). The predecessor factorial result is unary
by factorial result closure. Thus \(m_0\) is a valid quotient witness in the
existential definition of \(\mathsf{NatDivides}(\Eone(n),m)\).
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatFact\_successor\_input\_divides\_result}

\begin{theorem}[Factorial successor step predecessor divides product]
\label{thm:natfact-successor-step-predecessor-divides}
If \(\mathsf{NatFact}(n,m)\) and
\(\mathsf{NatMul}(\Eone(n),m,m')\), then the predecessor factorial value
divides the displayed successor-step product:
$$
\mathsf{NatDivides}(m,m').
$$
\end{theorem}
\begin{proof}
The factorial witness supplies unary evidence for \(n\) and for the
predecessor factorial result \(m\). Thus the displayed multiplication step has
two unary inputs, and right-factor divisibility for unary multiplication
applies directly to the factor \(m\).
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatFact\_successor\_step\_predecessor\_divides}

\begin{theorem}[Factorial absorbs every positive bounded unary predecessor]
\label{thm:natfact-bounded-positive-predecessor-divides}
Let $n,m,d$ be histories. If $\mathsf{NatFact}(n,m)$,
$\mathsf{UnaryHistory}(d)$, $\mathsf{NatGT}(d,\emp)$, and either $d\hsame n$
or $\mathsf{NatUnaryStrictPrefix}(d,n)$, then
$\mathsf{NatDivides}(d,m)$.
\end{theorem}
\begin{proof}[Sketch]
Induct on the closed $\mathsf{NatFact}$ witness. In the empty-input case,
positive boundedness would require either $d\hsame\emp$ or a strict unary
prefix of $\emp$, contradicting $\mathsf{NatGT}(d,\emp)$ and strict-prefix
inversion.

For the successor step, write the input as $\Eone(n_0)$ and the result as the
product of $\Eone(n_0)$ with the predecessor factorial result. If
$d\hsame\Eone(n_0)$, endpoint transport turns
\autoref{thm:natfact-successor-input-divides-result} into the required
divisibility statement for $d$.

It remains to treat the strict-predecessor branch. Invert the
$\mathsf{NatUnaryStrictPrefix}(d,\Eone(n_0))$ witness. The endpoint case is the
previous paragraph, while the proper predecessor case gives either
$d\hsame n_0$ or $\mathsf{NatUnaryStrictPrefix}(d,n_0)$. The induction
hypothesis therefore gives divisibility of the predecessor factorial result by
$d$.

Finally, push that divisibility through the final multiplication witness. Use
the left-closure of $\mathsf{NatDivides}$ under multiplication, and transport
along $\hsame$ endpoints where the constructor inversion changed the displayed
representatives. Transitivity of $\mathsf{NatDivides}$ yields
$\mathsf{NatDivides}(d,m)$ for the successor factorial result.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatFact\_bounded\_positive\_predecessor\_divides}

\begin{lemma}[Unary factorial results are nonempty]
\label{lem:nat-fact-result-not-empty}
Every closed witness of $\mathsf{NatFact}(n,m)$ has a result $m$ distinct
from $\emp$ up to $\hsame$.
\end{lemma}
\begin{proof}
Induct on the $\mathsf{NatFact}$ witness. The zero case is the singleton
$\Eone(\emp)$, which cannot classify as empty. In the successor case, an empty
product result would force the previous factorial result to be empty by the
nonempty-multiplicand empty-result inversion for $\mathsf{NatMul}$, contradicting
the induction hypothesis.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatFact\_result\_not\_empty}

\begin{lemma}[Factorial successor results have positive shape]
\label{thm:natfact-successor-result-positive-shape}
If $\mathsf{NatFact}(\Eone(n),m)$, then the factorial result has one-headed
shape:
$$
\exists t,\ \hsame(m,\Eone(t))\wedge\mathsf{UnaryHistory}(t).
$$
\end{lemma}
\begin{proof}
Successor inversion exposes a predecessor factorial value $m_0$ and a
multiplication witness $\mathsf{NatMul}(\Eone(n),m_0,m)$. Factorial
non-emptiness rules out $m_0\hsame\emp$, and $\Eone(n)$ is not history-same to
$\emp$. The positive-shape theorem for multiplication with a nonempty
multiplier therefore gives the displayed one-headed result.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatFact\_successor\_result\_positive\_shape}

\begin{lemma}[Factorial results are unary]
\label{lem:natfact-result-unary}
If \(\mathsf{NatFact}(n,m)\), then \(m\) is a unary history.
\end{lemma}
\begin{proof}
Induct on the factorial witness. The base constructor yields \(\Eone(\emp)\), which is unary. The successor constructor carries a multiplication witness, whose left factor is unary by multiplicand readback and whose result is unary by multiplication result closure.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatFact\_result\_unary}

\begin{theorem}[Unary factorial zero-headed component impossibility]
\label{thm:natfact-zero-headed-component-absurd}
For histories \(n,m\), a closed witness of \(\mathsf{NatFact}(n,m)\) cannot
have either its input or result displayed as a zero-headed history.
\end{theorem}
\begin{proof}
Factorial input readback gives \(\mathsf{UnaryHistory}(n)\), and factorial
result readback gives \(\mathsf{UnaryHistory}(m)\). In either displayed
zero-headed branch, substituting the equality turns a zero-headed history into
a unary history, contradicting unary zero-extension separation.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatFact\_zero\_headed\_component\_absurd}

\begin{theorem}[Factorial results have positive shape]
\label{thm:natfact-result-positive-shape}
If \(\mathsf{NatFact}(n,m)\), then there is a unary tail \(t\) such that
\(\hsame(m,\Eone(t))\).
\end{theorem}
\begin{proof}
Combine factorial result nonemptiness with factorial result unary closure.
Case analysis on the generated history \(m\) eliminates the empty case by
nonemptiness and the \(\Ezero\) case by unary inversion. The remaining
\(\Eone(t)\) case exposes the tail \(t\), and unary inversion supplies the
tail carrier.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatFact\_result\_positive\_shape}

\begin{lemma}[Factorial is total and functional on unary histories]
\label{lem:fact-total}
For every unary \(n\), there is exactly one unary \(m\) with \(\mathsf{NatFact}(n, m)\).
\end{lemma}
\begin{proof}
By unary induction on \(n\), using the totality and functionality of \(\mathsf{NatMul}\) (\autoref{lem:nat-mul-total}, \autoref{lem:nat-mul-functional}) at each step.
\end{proof}
\leanchecked{BEDC.Derived.PrimeUp.NatFact\_total\_functional}

\begin{lemma}[Bounded unary divisor divides factorial result]
\label{lem:natfact-bounded-divisor-divides-result}
Let $n$, $m$, and $d$ be histories. If $\mathsf{NatFact}(n,m)$,
$\mathsf{UnaryHistory}(d)$, $\neg\hsame(d,\Eone(\emp))$, and
$\mathsf{NatGT}(n,d)\vee\hsame(d,n)$, then $\mathsf{NatDivides}(d,m)$.
\end{lemma}
\begin{proof}[Sketch]
Induct on the factorial witness. In the base case, the bound by $\emp$ gives
no nonunit unary divisor, so the hypotheses cannot all hold.

For a successor factorial witness, use \autoref{lem:natfact-successor-inversion}
to expose the predecessor factorial and the displayed multiplication step. If
$d\hsame n$ at the successor endpoint, endpoint transport and
\autoref{thm:natfact-successor-input-divides-result} give
$\mathsf{NatDivides}(d,m)$.

If $d$ is strictly below the successor endpoint, successor inversion of the
bound either identifies the endpoint case above or produces a predecessor
bound. The induction hypothesis gives divisibility into the predecessor
factorial result. Then \autoref{lem:nat-mul-total},
\autoref{lem:nat-mul-functional}, and the displayed product divisibility
closures \autoref{thm:nat-divides-mul-right-closed} and
\autoref{thm:nat-divides-mul-left-closed} carry the inherited divisor through
the successor multiplication display.
\end{proof}

\begin{lemma}[Factorial successor has no bounded nonunit divisor]
\label{lem:natfact-successor-no-bounded-nonunit-divisor}
Let $n$, $m$, and $d$ be histories. If $\mathsf{NatFact}(n,m)$,
$\mathsf{UnaryHistory}(d)$, $\neg\hsame(d,\Eone(\emp))$, and
$\mathsf{NatGT}(n,d)\vee\hsame(d,n)$, then
$\neg\mathsf{NatDivides}(d,\Eone(m))$.
\end{lemma}
\begin{proof}[Sketch]
Assume toward contradiction that $d$ divides $\Eone(m)$. By
\autoref{lem:natfact-bounded-divisor-divides-result}, the same bounded
nonunit divisor also divides the factorial result $m$.

The continuation $\Cont(m,\Eone(\emp),\Eone(m))$ displays the successor as the
factorial result followed by the unit. Divisible prefix cancellation
\autoref{lem:nat-divides-prefix-cancellation} removes the common divisible
prefix and forces $\mathsf{NatDivides}(d,\Eone(\emp))$.

Finally, unit-right divisibility exactness
\autoref{thm:nat-divides-unit-right-iff} identifies $d$ with
$\Eone(\emp)$, contradicting the nonunit hypothesis.
\end{proof}

\begin{lemma}[Successor of factorial is coprime to all bounded predecessors]
\label{lem:fact-plus-one-coprime}
Let \(m\) be the unary factorial of \(n\), and let \(m'\) be the successor \(\Eone(m)\) (i.e., \(n! + 1\)). Then for every unary \(d\) with \(\Eone(\emp) \neq d\) and \(\mathsf{NatGT}(\Eone(n), d) \vee \hsame(d, \Eone(n)) \vee \mathsf{NatGT}(d, \Eone(n)) \wedge \mathsf{NatDivides}(d, n)\), we have \(\neg \mathsf{NatDivides}(d, m')\).
\end{lemma}
\begin{proof}[Sketch]
For every positive unary $d$ bounded by $n$, the factorial absorption theorem
\autoref{thm:natfact-bounded-positive-predecessor-divides} supplies
$\mathsf{NatDivides}(d,m)$. If the same $d$ also divides $m'=\Eone(m)$, then
the unit-successor subtraction argument shows that $d$ divides the unit.

Unit divisibility forces $d\hsame\Eone(\emp)$, contradicting the displayed
non-unit hypothesis. Thus no such bounded non-unit divisor can divide
$m'$.
\end{proof}

\begin{theorem}[Euclid's theorem, constructive form]
\label{thm:euclid-infinite}
There exists an explicit total unary function \(\mathsf{NatNextPrime} : \mathsf{Unary} \to \mathsf{Unary}\) such that for every unary \(n\), \(\mathsf{NatPrime}(\mathsf{NatNextPrime}(n))\) and \(\mathsf{NatGT}(\mathsf{NatNextPrime}(n), n)\).
\end{theorem}
\begin{proof}[Sketch]
Given \(n\), compute the unary factorial \(m = n!\) by \autoref{lem:fact-total}, then form \(m' = m + 1\). By \autoref{lem:fact-plus-one-coprime}, every prime divisor of \(m'\) exceeds \(n\). The primality test of \autoref{thm:prime-decidable} either finds \(m'\) itself prime (return it) or returns an explicit non-trivial divisor; in the latter case, recursing on the divisor's prime factor (also exceeding \(n\)) terminates in a prime by well-foundedness of \(\mathsf{NatUnaryStrictPrefix}\). The function is total because the search is bounded.
\end{proof}
