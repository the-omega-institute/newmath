\begin{definition}[UniformCauchyCriterion Real-completion finite-envelope carrier]
\label{def:uniform-cauchy-criterion-real-completion-finite-envelope-carrier}
Let
$$
U=(I,W,M,Q,T,E,H,C,P,N)
$$
be an accepted $\UniformCauchyCriterionUp$ packet. The Real-completion finite-envelope carrier of $U$ is the packet
$$
\mathcal{E}_{\RealUp}(U)=
(U,W_{\StreamNameUp},R_{\RegSeqRatUp},Q_{\DyadicRatCoreUp},
S_{\CauchyLimitSealUp},E_{\RealCauchyCompletionUp},H,C,P,N).
$$
Its rows are exactly the accepted $\UniformCauchyCriterionUp$ packet, the $\StreamNameUp$ window family $W$, the $\RegSeqRatUp$ source rows $R$ reached by the uniform tail ledger, the $\DyadicRatCoreUp$ tolerance ledger $Q$, the $\CauchyLimitSealUp$ tail-meet route $S$, the $\RealCauchyCompletionUp$ public $\RealUp$ export $E$, and the visible $\BHist$, $\hsame$, $\Cont$, $\Pkg$, and $\NameCert_{\UniformCauchyCriterionUp}$ rows $H,C,P,N$. A row is admitted to the carrier only when it is displayed by this packet or by one of the named sibling routes attached to the same $U$.
\end{definition}

\begin{theorem}[UniformCauchyCriterion Real-completion finite-envelope NameCert obligations]
\label{thm:uniform-cauchy-criterion-real-completion-finite-envelope-namecert-obligations}

\item window-family stability: $\hsame$ transport of a $\StreamNameUp$ window request remains inside the same displayed $W$ family and is replayed only through the visible $\Cont$ row;
\item regular-rational handoff: every $\RegSeqRatUp$ readback row used by a consumer is reached from the uniform tail ledger of $U$ and not from a second stream source;
\item dyadic tolerance ledger exactness: every tolerance comparison consumed at the completion boundary is read from $Q_{\DyadicRatCoreUp}$ before the tail-meet route is sealed;

\end{theorem}
\leanchecked{BEDC.Derived.UniformCauchyCriterionUp.UniformCauchyCriterionPacket\_real\_completion\_finite\_envelope\_namecert\_obligations}

\begin{proof}
