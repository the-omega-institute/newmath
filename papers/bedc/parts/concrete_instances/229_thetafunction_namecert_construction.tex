\chapter{A Concrete Naming Certificate for $\ThetaFunctionUp$}
\label{ch:concrete-instances-thetafunction-namecert}

The theta-function interface $\ThetaFunctionUp$ packages the holomorphic functions $\theta(z, \tau)$ on $\mathbb{C} \times \mathbb{H}$ satisfying the heat equation and the lattice-quasiperiodicity laws. It depends on $\NameCert_{\HolomorphicUp}$ (\autoref{ch:concrete-instances-holomorphic-namecert}) and $\NameCert_{\EllipticCurveUp}$ (\autoref{ch:concrete-instances-ellipticcurve-namecert}).

\section{Carrier definition}
\label{sec:thetafunction-carrier}

\begin{definition}[ThetaFunction carrier source]
\label{def:thetafunction-carrier-source}
A $\ThetaFunctionUp$ carrier source is a finite $\mathsf{BHist}$ packet
$\Theta=(E,H,Q,\Pi,C)$ whose rows consist of an $\EllipticCurveUp$ period
ledger $E$, a $\HolomorphicUp$ local-function row $H$ on the displayed
$(z,\tau)$ chart, a finite $q$-expansion coefficient window $Q$ compatible with
the $\ComplexSeriesUp$ coefficient ledger, $\mathsf{Pkg}$ provenance $\Pi$ for
the elliptic-curve, holomorphic, and complex-series dependencies, and finite
$\Cont$ readback rows $C$ tying the period, chart, and coefficient observations
to the same packet.
\end{definition}

\begin{definition}[ThetaFunction classifier]
\label{def:thetafunction-classifier}
Two carrier sources $\Theta=(E,H,Q,\Pi,C)$ and
$\Theta'=(E',H',Q',\Pi',C')$ are classified by the $\ThetaFunctionUp$ source
classifier when the $\EllipticCurveUp$ period rows are transported by the same
$\hsame$-stable period comparison, the finite coefficient windows $Q$ and
$Q'$ agree under the displayed $\ComplexSeriesUp$ coefficient-window
classifier, the $\HolomorphicUp$ chart rows $H$ and $H'$ agree on the same
local-function source, and the $\mathsf{Pkg}$ and $\Cont$ provenance rows are
componentwise the displayed rows for those transports.
\end{definition}

\section{The certificate}
\label{sec:thetafunction-certificate}

\begin{theorem}[ThetaFunction hsame stability]
\label{thm:thetafunction-hsame-stability}
The classifier of \autoref{def:thetafunction-classifier} is stable under
rowwise $\hsame$ transport of the period ledger, coefficient window,
holomorphic chart row, $\mathsf{Pkg}$ provenance, and $\Cont$ readback rows.
\end{theorem}
\begin{proof}
Unpack the classifier into its displayed coordinates. The period coordinate is
transported by the $\EllipticCurveUp$ period row, the chart coordinate by the
$\HolomorphicUp$ local-function row, and the coefficient coordinate by the
$\ComplexSeriesUp$ window row.

Transport each coordinate along the given $\hsame$ witness and keep the same
$\Cont$ readback route and $\mathsf{Pkg}$ provenance. Repacking these transported
rows gives a classifier witness for the target packet, with no appeal to host
function equality or analytic continuation.
\end{proof}

\begin{theorem}[ThetaFunction ledger exactness]
\label{thm:thetafunction-ledger-exactness}
Every accepted $\ThetaFunctionUp$ carrier source exposes exactly the
$\EllipticCurveUp$ period rows, finite $q$-expansion coefficient rows,
$\HolomorphicUp$ chart rows, $\mathsf{Pkg}$ provenance, and $\Cont$ readback
rows of \autoref{def:thetafunction-carrier-source}. It exposes no analytic
continuation datum, completed modular form, host holomorphic-function equality,
or infinite coefficient object.
\end{theorem}
\begin{proof}
Project an accepted carrier source to the five components in
\autoref{def:thetafunction-carrier-source}. These projections are the period
ledger, holomorphic chart, coefficient window, provenance, and finite readback
rows.

Conversely, any ledger read admitted by the carrier must enter through one of
those five coordinates. The $\EllipticCurveUp$, $\HolomorphicUp$, and
$\ComplexSeriesUp$ dependency anchors account for the mathematical rows, while
$\mathsf{Pkg}$ and $\Cont$ account for provenance and finite readback.

There is no constructor field for analytic continuation, completed modular
forms, host equality of functions, or an infinite coefficient object. Therefore
none of those objects is exported by the ledger.
\end{proof}

\begin{theorem}[ThetaFunction NameCert boundary]
\label{thm:thetafunction-namecert-boundary}
The carrier source of \autoref{def:thetafunction-carrier-source}, the
classifier of \autoref{def:thetafunction-classifier}, the stability theorem
\autoref{thm:thetafunction-hsame-stability}, and the ledger exactness theorem
\autoref{thm:thetafunction-ledger-exactness} assemble the $\ThetaFunctionUp$
$\NameCert$ boundary over $\mathsf{BHist}$, $\hsame$, $\Cont$, and
$\mathsf{Pkg}$. The boundary consumes only the displayed $\EllipticCurveUp$,
$\HolomorphicUp$, and $\ComplexSeriesUp$ source rows and supplies exactly the
finite period, chart, and coefficient observation surface.
\end{theorem}
\begin{proof}
The carrier definition supplies the source object for the certificate. The
classifier definition supplies the comparison relation, and
\autoref{thm:thetafunction-hsame-stability} gives transport of accepted rows
along the componentwise $\hsame$ comparisons.

The exact ledger is supplied by
\autoref{thm:thetafunction-ledger-exactness}. Its projections are precisely the
period, chart, coefficient, provenance, and readback rows required by the
displayed $\NameCert$ surface.

Combining these rows gives the advertised boundary: a finite $\mathsf{BHist}$
packet with dependency provenance through $\mathsf{Pkg}$, readback through
$\Cont$, and no additional analytic-continuation or completed-modular-form
field.
\end{proof}
