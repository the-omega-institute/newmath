\chapter{A Concrete Naming Certificate for \(\ZetaBasicUp\) on \(\Re(s) > 1\)}
\label{ch:concrete-instances-zeta-basic-namecert}

This chapter gives a concrete naming-certificate construction for the Riemann zeta function \(\zeta(s) = \sum_{n=1}^{\infty} n^{-s}\) on the open half-plane \(\Re(s) > 1\), where the defining Dirichlet series converges absolutely. It is a direct specialization of \(\DirichletSeriesUp\) (\autoref{ch:concrete-instances-dirichlet-series-namecert}) at the unit coefficient sequence \(a_n = \Eone(\emp)\). The chapter does NOT extend \(\zeta\) to \(\ComplexUp \setminus \{1\}\) --- that is a separate operation chapter (\(\ZetaContUp\), \autoref{ch:concrete-instances-zeta-continuation}). Two faces of the same object are presented: the Dirichlet sum and the Euler product over primes; non-vanishing on the half-plane is supplied with explicit lower bound; holomorphicity is inherited from \(\DirichletSeriesUp\). The construction is fully constructive and uses no \(\mathsf{Classical.choice}\), no \(\mathsf{Quot.sound}\), and no \(\mathsf{propext}\), so the certificate's Lean targets fall under the project's axiom-purity gate.

\section{The constant-coefficient Dirichlet sum}

\begin{definition}[Zeta basic predicate]
\label{def:zeta-basic}
For complex history \(s\) and complex history \(z\), the predicate \(\mathsf{ZetaBasic}(s, z)\) holds when \(z\) is the explicit complex-history limit of the Dirichlet partial sums of the constant-unit coefficient sequence \(a_n = \Eone(\emp)\) at \(s\); equivalently, \(\mathsf{ZetaBasic}(s, z)\) is the specialization of the \(\DirichletSeriesUp\) carrier (\autoref{ch:concrete-instances-dirichlet-series-namecert}) at unit coefficients.
\end{definition}

\begin{definition}[Unit Dirichlet part sum]
\label{def:zeta-basic-part-sum}
For a complex history \(s\), the unit term at unary index \(n\) is the visible unit extension \(\Eone(s)\), and \(\mathsf{ZetaBasicPartSum}(s,n,z)\) is the specialized \(\mathsf{DirichletPartSum}\) witness for this unit term.
\end{definition}
\leandef{BEDC.Derived.ZetaBasicUp.ZetaBasicUnitTerm}
\leandef{BEDC.Derived.ZetaBasicUp.ZetaBasicPartSum}

\begin{theorem}[Unit Dirichlet part sums are unary]
\label{thm:zeta-basic-part-sum-unary-result}
If \(s\) and \(n\) are unary histories and \(\mathsf{ZetaBasicPartSum}(s,n,z)\) holds, then \(z\) is a unary history.
\end{theorem}
\begin{proof}
Induct on the concrete \(\mathsf{DirichletPartSum}\) witness. The zero result is \(\emp\), and the successor case appends the prior unary partial sum to the visible unit term \(\Eone(s)\), which is closed by unary continuation.
\end{proof}
\leanchecked{BEDC.Derived.ZetaBasicUp.ZetaBasicPartSum\_unary\_result}

\begin{theorem}[Unit Dirichlet successor part sums are nonempty]
\label{thm:zeta-basic-part-sum-successor-result-nonempty}
If \(\mathsf{ZetaBasicPartSum}(s,\Eone(n),z)\) holds, then \(z\) is not
\(\hsame\) to \(\emp\).
\end{theorem}
\begin{proof}
Invert the successor partial-sum witness. Its final step appends the previous
sum to \(\Eone(s)\). If the result were empty, empty-result inversion for
\(\Cont\) would force \(\Eone(s)\hsame\emp\), contradicting visible-successor
separation.
\end{proof}
\leanchecked{BEDC.Derived.ZetaBasicUp.ZetaBasicPartSum\_successor\_result\_nonempty}

\begin{lemma}[Zeta basic well-defined for \(\Re(s) > 1\)]
\label{lem:zeta-basic-well-defined}
For every \(s\) with rational representation of \(\Re(s)\) strictly greater than \(\Eone(\emp)\) (decidable on the rational representation), the predicate \(\mathsf{ZetaBasic}(s, z)\) holds for some complex history \(z\), and \(z\) is unique up to \(\hsame\) on both real components.
\end{lemma}
\begin{proof}[Sketch]
Specialize \autoref{thm:half-plane-convergence} of \(\DirichletSeriesUp\) at unit coefficients with abscissa of absolute convergence exactly \(\Eone(\emp)\); existence and uniqueness of the limit follow from the \(\ComplexLimitUp\) certificate (\autoref{ch:concrete-instances-complex-limit-namecert}).
\end{proof}

\begin{theorem}[Zeta basic unique]
\label{thm:zeta-basic-unique}
If \(\mathsf{ZetaBasic}(s, z)\) and \(\mathsf{ZetaBasic}(s, z')\) both hold for the same \(s\), then \(\hsame\) holds on both real components of \(z\) and \(z'\).
\end{theorem}
\begin{proof}
Direct uniqueness from \(\DirichletSeriesUp\) limit uniqueness.
\end{proof}

\section{Convergence and abscissa}

\begin{theorem}[Abscissa is exactly 1]
\label{thm:zeta-abscissa-one}
The abscissa of absolute convergence for the unit Dirichlet series is the rational unit \(\Eone(\emp)\): \(\mathsf{AbsConvAbscissa}(\mathbf{1}, \Eone(\emp))\). The harmonic series at \(s = 1\) diverges with explicit lower-bound logarithmic-rate witness, so the abscissa cannot be smaller; the geometric majorant for \(s > 1\) shows that the abscissa is exactly 1.
\end{theorem}
\begin{proof}[Sketch]
The harmonic divergence is constructively witnessed by the explicit comparison \(\sum_{n=1}^{2^k} 1/n \geq k/2\), proving abscissa is at least 1; the half-plane convergence theorem at any \(s\) with \(\Re(s) > 1\) shows the abscissa is at most 1.
\end{proof}

\section{The Euler product over primes}

\begin{theorem}[Euler product for \(\zeta\)]
\label{thm:zeta-euler-product}
For every \(s\) with \(\Re(s) > 1\), the explicit factorization
$$\zeta(s) = \prod_p (1 - p^{-s})^{-1}$$
holds, where the product runs over primes \(p\) in the sense of \(\PrimeUp\) (\autoref{ch:concrete-instances-prime-namecert}). The product converges absolutely with explicit modulus over compact sub-half-planes \(\{s : \Re(s) \geq 1 + \epsilon, |\Im(s)| \leq T\}\) for unary \(\epsilon, T\).
\end{theorem}
\begin{proof}[Sketch]
Direct specialization of the Euler product theorem of \(\DirichletSeriesUp\) (\autoref{thm:euler-product}) at completely multiplicative coefficients \(a_n = \Eone(\emp)\) with \(a_p = \Eone(\emp)\) for each prime \(p\).
\end{proof}

\section{Non-vanishing on the half-plane}

\begin{theorem}[\(\zeta\) non-vanishing for \(\Re(s) > 1\)]
\label{thm:non-vanishing-half-plane}
For every \(s\) with \(\Re(s) > 1\), \(\zeta(s) \neq 0\) in the strict sense: the modulus \(|\zeta(s)|\) admits an explicit positive lower bound depending on \(\Re(s)\), given by the geometric tail of the Euler product.
\end{theorem}
\begin{proof}[Sketch]
The Euler product factors are each bounded below in modulus by \(1 - p^{-\Re(s)} > 0\); the infinite product converges with explicit modulus, so the limiting modulus is bounded below by an explicit positive constant.
\end{proof}

\section{Holomorphicity on the half-plane}

\begin{theorem}[\(\zeta\) holomorphic on \(\Re(s) > 1\)]
\label{thm:zeta-half-plane-holomorphic}
The function \(s \mapsto \zeta(s)\) is holomorphic in the open half-plane \(\Re(s) > 1\) in the sense of \(\HolomorphicUp\) (\autoref{ch:concrete-instances-holomorphic-namecert}), with uniform modulus over compact sub-half-planes.
\end{theorem}
\begin{proof}
Direct specialization of \autoref{thm:half-plane-holomorphic} of \(\DirichletSeriesUp\) at unit coefficients.
\end{proof}

\section{The certificate}

\begin{definition}[Zeta basic source specification]
\label{def:zeta-basic-source-spec}
The source specification on \(s\) carries the constraint \(\Re(s) > 1\) (decidable on the rational representation).
\end{definition}

\begin{definition}[Zeta basic pattern specification]
\label{def:zeta-basic-pattern-spec}
The pattern is the unit-coefficient Dirichlet partial-sum convergence predicate, equivalently the Euler-product convergence predicate.
\end{definition}

\begin{definition}[Zeta basic classifier specification]
\label{def:zeta-basic-classifier-spec}
Two \(\zeta\)-values at the same \(s\) are classified together when they agree under \(\hsame\) on both real components.
\end{definition}

\begin{definition}[Zeta basic stability certificate]
\label{def:zeta-basic-stability-cert}
The stability certificate consists of \(\hsame\)-respect on \(s\), transport of partial-sum modulus under \(\hsame\), and equivalence of the Dirichlet-sum and Euler-product values.
\end{definition}

\begin{definition}[Zeta basic ledger policy]
\label{def:zeta-basic-ledger-policy}
The ledger policy records: (i) abscissa exactly 1, (ii) Euler product factorization, (iii) non-vanishing on the half-plane with explicit lower bound, (iv) holomorphicity over compact sub-half-planes.
\end{definition}

The five fields above form a naming certificate over \(\Hist\) for the basic \(\zeta\) interface \(\ZetaBasicUp\) on \(\Re(s) > 1\).
