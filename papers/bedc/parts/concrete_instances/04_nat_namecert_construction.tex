\begin{definition}[Unary history judgment]

\begin{lemma}[Unary closure under one-extension]

\end{lemma}

\begin{lemma}[Unary no-zero-extension]

No judgment \(\mathsf{Unary}(\Ezero(h))\) is generated.

\end{proof}

\begin{theorem}[Unary prefix totality]
\label{thm:nat-unary-prefix-total}
If \(h\) and \(k\) are unary histories, then either \(h\) continues by a unary
tail to \(k\), or \(k\) continues by a unary tail to \(h\).
\end{theorem}
\leanchecked{BEDC.Derived.NatUp.NatUnaryPrefix\_total}
\begin{proof}
Induct on the unary generation of \(h\). The empty case uses \(k\) itself as
the tail. In the \(\Eone\) case, split \(k\); the empty and \(\Ezero\) cases
are immediate by the continuation unit and unary no-zero rule. When
\(k=\Eone(k')\), apply the induction hypothesis to \(h'\) and \(k'\), then
lift the resulting unary tail across the common \(\Eone\) boundary by the
unary one-extension append law.
\end{proof}

\begin{theorem}[Unary prefix directed common upper]
\label{thm:nat-unary-prefix-directed-common-upper}
If $h$ and $k$ are unary histories, then there is a unary history $u$ such
that both $h$ and $k$ continue to $u$ by unary tails.
\end{theorem}
\begin{proof}
Apply unary prefix totality. If $h$ continues to $k$, choose $k$ as the
common upper and use the empty tail on the $k$-side. If $k$ continues to
$h$, choose $h$ as the common upper and use the empty tail on the $h$-side.
\end{proof}
\leanchecked{BEDC.Derived.NatUp.NatUnaryPrefix\_directed\_common\_upper}

\begin{definition}[Strict unary prefix]
\label{def:nat-unary-strict-prefix}
A strict unary prefix from $h$ to $k$ is a unary tail $t$ such that $t$ is not
$\emp$ and $\mathsf{Cont}(h,t,k)$.
\end{definition}
\leandef{BEDC.Derived.NatUp.NatUnaryStrictPrefix}

\begin{lemma}[Unary continuation tail cases]
\label{lem:nat-unary-prefix-cont-tail-cases}
If a unary tail \(t\) displays \(\Cont(h,t,k)\), then either \(h\hsame k\) or \(h\) is a strict unary prefix of \(k\).
\end{lemma}
\begin{proof}
Split the unary tail. The empty tail gives the right-unit continuation and therefore endpoint history-sameness. A zero-headed tail contradicts unary generation. A one-headed tail is nonempty, so the displayed continuation is exactly the witness for \(\mathsf{NatUnaryStrictPrefix}(h,k)\).
\end{proof}
\leanchecked{BEDC.Derived.NatUp.NatUnaryPrefix\_cont\_tail\_cases}

\begin{theorem}[Unary prefix trichotomy into sameness or strict prefix]
\label{thm:nat-unary-prefix-trichotomy-hsame-strict}
If $h$ and $k$ are unary histories, then either $h\hsame k$, or $h$ is a
strict unary prefix of $k$, or $k$ is a strict unary prefix of $h$.
\end{theorem}
\begin{proof}
Apply unary prefix totality to obtain a unary continuation tail in one
direction. If the tail is $\emp$, the continuation is the right-unit
continuation and the endpoints are history-same. A zero-headed tail contradicts
unary generation. A one-headed tail is nonempty, so the same continuation
witness gives the corresponding strict unary prefix branch.
\end{proof}
\leanchecked{BEDC.Derived.NatUp.NatUnaryPrefix\_trichotomy\_hsame\_strict}

\begin{lemma}[Strict unary prefix one-step]
\label{lem:nat-unary-strict-prefix-one-step}
If \(h\) is unary, then \(h\) is a strict unary prefix of \(\Eone(h)\).
\end{lemma}
\begin{proof}
Use the tail \(\Eone(\emp)\). It is unary, it is not \(\emp\), and
\(\Cont(h,\Eone(\emp),\Eone(h))\) holds by the continuation constructor.
\end{proof}
\leanchecked{BEDC.Derived.NatUp.NatUnaryStrictPrefix\_one\_step}

\begin{theorem}[Strict unary prefix asymmetry]
\label{thm:nat-unary-strict-prefix-asymm}
No pair of histories can strictly prefix each other through nonempty unary
tails.
\end{theorem}
\begin{proof}
Unpack the two strict witnesses. Mutual continuation makes the endpoints
history-same. The right-unit continuation at the source then transports to the
forward target, so left cancellation identifies the forward tail with $\emp$,
contradicting its strictness.
\end{proof}
\leanchecked{BEDC.Derived.NatUp.NatUnaryStrictPrefix\_asymm}

\begin{theorem}[Strict unary prefix has nonempty target]
\label{thm:nat-unary-strict-prefix-empty-right-absurd}
No history can be a strict unary prefix of \(\emp\).
\end{theorem}
\begin{proof}
Unpack the strict-prefix witness. The continuation into \(\emp\) forces the
tail to be \(\emp\), contradicting the nonempty-tail condition.
\end{proof}
\leanchecked{BEDC.Derived.NatUp.NatUnaryStrictPrefix\_empty\_right\_absurd}

\begin{lemma}[Strict unary prefix tail endpoint collapse is impossible]
\label{lem:nat-unary-strict-prefix-tail-endpoint-hsame-absurd}
If a unary tail \(t\) is nonempty, \(\Cont(h,t,k)\), and \(h\hsame k\), then
the displayed continuation is impossible.
\end{lemma}
\begin{proof}
Transport the right-unit continuation \(\Cont(h,\emp,h)\) across the endpoint
identification \(h\hsame k\), giving \(\Cont(h,\emp,k)\). Left cancellation
against the supplied continuation \(\Cont(h,t,k)\) identifies \(t\) with
\(\emp\), contradicting the strict tail witness.
\end{proof}
\leanchecked{BEDC.Derived.NatUp.NatUnaryStrictPrefix\_tail\_endpoint\_hsame\_absurd}

\begin{theorem}[Strict unary prefix one-boundary inversion]
\label{thm:nat-unary-strict-prefix-e1-inversion}
If \(\Eone(h)\) is a strict unary prefix of \(\Eone(k)\), then \(h\) is a strict unary prefix of \(k\).
\end{theorem}
\begin{proof}
Unpack the strict witness tail. The empty tail contradicts strictness, and an \(\Ezero\)-headed tail contradicts unary generation. For an \(\Eone(t)\) tail, invert the result boundary of the continuation and commute the visible \(\Eone\) across the unary tail \(t\), yielding the strict witness \(\Eone(t)\) from \(h\) to \(k\).
\end{proof}
\leanchecked{BEDC.Derived.NatUp.NatUnaryStrictPrefix\_e1\_inversion}

\begin{lemma}[Strict unary prefix one-boundary endpoint collapse is impossible]
\label{lem:nat-unary-strict-prefix-e1-endpoint-hsame-absurd}
If \(\Eone(h)\) is a strict unary prefix of \(\Eone(k)\) and \(h\hsame k\),
then the displayed strict prefix is impossible.
\end{lemma}
\begin{proof}
Apply one-boundary inversion to obtain a strict unary prefix from \(h\) to
\(k\). The strict witness tail together with the endpoint identification
\(h\hsame k\) contradicts the tail endpoint collapse lemma.
\end{proof}
\leanchecked{BEDC.Derived.NatUp.NatUnaryStrictPrefix\_e1\_endpoint\_hsame\_absurd}

\begin{theorem}[Strict unary prefix one-boundary congruence]
\label{thm:nat-unary-strict-prefix-e1-congr}
If $h$ is a strict unary prefix of $k$, then $\Eone(h)$ is a strict unary
prefix of $\Eone(k)$.
\end{theorem}
\leanchecked{BEDC.Derived.NatUp.NatUnaryStrictPrefix\_e1\_congr}
\begin{proof}
Unpack the nonempty unary tail witnessing the strict prefix from $h$ to
$k$. The same tail remains unary and nonempty. Commuting append across the
visible $\Eone$ boundary re-expresses the continuation as one from
$\Eone(h)$ to $\Eone(k)$.
\end{proof}

\begin{theorem}[Strict unary prefix zero-boundary inversion]
\label{thm:nat-unary-strict-prefix-e0-inversion}
If \(\Ezero(h)\) is a strict unary prefix of \(\Ezero(k)\), then \(h\) is a strict unary prefix of \(k\).
\end{theorem}
\begin{proof}
Unpack the strict witness tail. The empty tail contradicts strictness, an
\(\Ezero\)-headed tail contradicts unary generation, and an \(\Eone\)-headed
tail would force a zero-headed continuation result to be one-headed. Hence the
premise has no inhabitant, so the displayed implication follows.
\end{proof}
\leanchecked{BEDC.Derived.NatUp.NatUnaryStrictPrefix\_e0\_inversion}

\begin{theorem}[Strict unary prefix append-tail transitivity]
\label{thm:nat-unary-strict-prefix-append-tail-trans}
If \(h\) continues by a nonempty unary tail to \(k\), and \(k\) continues by a
unary tail to \(l\), then \(h\) is a strict unary prefix of \(l\). The witness
tail is \(\mathsf{append}(t_1,t_2)\), and it also gives
\(\mathsf{Cont}(h,\mathsf{append}(t_1,t_2),l)\).
\end{theorem}
\begin{proof}
Invert the two continuation witnesses and reassociate append. Unary closure
under append gives a unary composite tail, and append empty-result exactness
shows that the composite tail cannot be empty because its left component is
nonempty.
\end{proof}
\leanchecked{BEDC.Derived.NatUp.NatUnaryStrictPrefix\_append\_tail\_trans}

\begin{lemma}[Strict unary prefix target is unary]
\label{lem:nat-unary-strict-prefix-target-unary}
If \(h\) is unary and \(h\) is a strict unary prefix of \(k\), then \(k\) is unary.
\end{lemma}
\begin{proof}
Unpack the strict-prefix witness as a unary tail \(t\) with \(\Cont(h,t,k)\).
Unary continuation closure transports unary generation from \(h\) across \(t\)
to the target \(k\).
\end{proof}
\leanchecked{BEDC.Derived.NatUp.NatUnaryStrictPrefix\_target\_unary}

\begin{theorem}[Strict unary prefix composite-tail transitivity]
\label{thm:nat-unary-strict-prefix-trans-composite-tail}
If $h$ is a strict unary prefix of $k$ and $k$ is a strict unary prefix of $l$,
then the two nonempty unary tails compose into a concrete tail from $h$ to
$l$. This composite tail witnesses both $\mathsf{Cont}(h,t,l)$ and the strict
unary prefix relation from $h$ to $l$.
\end{theorem}
\begin{proof}
Unpack the two strict-prefix witnesses. Append their unary tails and apply
append-tail transitivity. Unary append closure gives the composite tail
carrier, and append empty-result exactness shows that the composite tail is
nonempty because its left component is nonempty.
\end{proof}
\leanchecked{BEDC.Derived.NatUp.NatUnaryStrictPrefix\_trans\_composite\_tail}

\begin{theorem}[Strict unary prefix continuation transport]
\label{thm:nat-unary-strict-prefix-cont-hsame-transport}
If a nonempty unary tail carries \(h\) to \(k\), and \(h\hsame h'\) and
\(k\hsame k'\), then \(h'\) is a strict unary prefix of \(k'\).
\end{theorem}
\begin{proof}
Reuse the same tail. The continuation component transports across the two
endpoint \(\hsame\) witnesses, while unary generation and nonemptiness of the
tail are unchanged.
\end{proof}
\leanchecked{BEDC.Derived.NatUp.NatUnaryStrictPrefix\_cont\_hsame\_transport}

\begin{theorem}[Concrete \(\NatUp\) naming certificate]
\label{thm:concrete-natup-namecert}

\begin{theorem}[NatUp unary standard bridge]
\label{thm:natup-unary-standard-bridge}
The unary-history certificate for $\NatUp$ carries a conservative standard
bridge to finite unary words and their natural-number length readback. The
bridge sends $\emp$ to zero, sends $\Eone(h)$ to the successor of the readback
of $h$, admits no $\Ezero$ branch, identifies $h\hsame k$ on unary histories
exactly when the external readback lengths are equal, and sends a unary
continuation $\Cont(h,t,k)$ to addition of the readback length of $t$ to the
readback length of $h$.
\end{theorem}
\begin{proof}
Define the decode field by unary-generation induction. The base generator reads
$\emp$ as zero. The one-extension generator reads $\Eone(h)$ as the successor
of the already decoded tail. The no-zero-extension lemma excludes an $\Ezero$
case, so every decoded value is obtained from a finite unary word and no other
BEDC history is admitted by the bridge carrier.

The encode field sends an external finite unary word back to its BEDC unary
history by the same two constructors. Induction on the word gives the
decode-encode round trip, while induction on the unary-history generator gives
the encode-decode round trip up to $h\hsame k$. The semantic NameCert theorem
supplies the source, pattern, and ledger preservation needed to keep these
witnesses inside the licensed $\NatUp$ surface.

For exactness, use strict-prefix and continuation transport. If two unary
histories are $h\hsame k$, the readback induction transports the same length
across the endpoint identification. Conversely, if the readback lengths agree,
unary prefix trichotomy leaves only the sameness branch: either strict-prefix
branch would add a nonempty unary tail and hence a positive successor length on
one side. Thus $\hsame$ matches equality of external readback lengths.

For continuation, induct on the unary tail. The empty tail gives addition by
zero. A one-headed tail applies the induction hypothesis and then one-extension
supplies the successor step on the result. The zero-headed case is ruled out by
unary no-zero-extension. Therefore unary $\Cont$ tails read as addition of
finite unary lengths, and the theorem-preservation list is limited to these
constructor, equality, and continuation clauses.

The no-host-leak field follows because the bridge uses the external
natural-number model only in these encode, decode, round-trip, and preservation
fields, while the BEDC public surface remains the unary-history NameCert
grounded by $\mathsf{nat\_up\_semantic\_name\_certificate}$.
\end{proof}
\leanchecked{BEDC.FKernel.Unary.nat\_up\_unary\_standard\_bridge}
\leanchecked{BEDC.Derived.NatUp.NatUp\_unary\_standard\_bridge}

\begin{theorem}[NatUp mature unary recursion package]
\label{thm:natup-mature-unary-recursion-package}
The unary-history certificate supplies the mature paper-level $\NatUp$ package
over BEDC $\mathsf{BHist}$ and $\hsame$. Its zero object is $\emp$, its
successor operation is $\Eone$, no carried element is a zero-extension, unary
generation gives induction, primitive recursion is defined by the empty and
one-extension generation cases, and continuation length is addition of the
unary tail length. The operation bridges consumed by this package are the
already cited $\AddUp$, $\MulUp$, and $\OrderUp$ interfaces, not host
arithmetic.
\end{theorem}
\leanchecked{BEDC.Derived.NatUp.NatUp\_mature\_unary\_recursion\_package}
\begin{proof}
The carrier is the closed unary-generation carrier of the semantic
$\NatUp$ certificate. Its base constructor gives $\emp$, its step constructor
gives $\Eone$, and \autoref{lem:unary-no-zero-extension} removes every
$\Ezero$ branch from the carried surface.

Induction is the eliminator for that unary-generation predicate. Primitive
recursion is obtained by the same cases: assign the empty value at $\emp$, and
in the one-extension case apply the specified successor clause to the value
already computed for the predecessor history.

The endpoint equality used by the recursion clauses is $\hsame$, not equality
of host natural numbers. Apply
\autoref{thm:nat-unary-prefix-trichotomy-hsame-strict}: the sameness branch is
the transport case, and either strict-prefix branch contributes a nonempty
unary continuation tail.

Finally, \autoref{thm:natup-unary-standard-bridge} reads continuation tails as
addition of unary lengths. Induction on the tail gives the empty-tail zero law
and the one-headed successor step. Thus the package exposes the NatUp
induction, recursion, and continuation-as-addition principles internally over
$\mathsf{BHist}$, while arithmetic consumers enter only through the cited
$\AddUp$, $\MulUp$, and $\OrderUp$ interfaces.
\end{proof}
