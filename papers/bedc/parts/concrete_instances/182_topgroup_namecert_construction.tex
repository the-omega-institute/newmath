\chapter{A Concrete Naming Certificate for $\TopGroupUp$}
\label{ch:concrete-instances-topgroup-namecert}

The topological-group interface $\TopGroupUp$ packages a group carrier with a topology making multiplication and inversion continuous. It depends on $\NameCert_{\GroupUp}$ (\autoref{ch:concrete-instances-group-namecert}) and $\NameCert_{\TopologyUp}$ (\autoref{ch:concrete-instances-topology-namecert}).

\section{Carrier definition}
\label{sec:topgroup-carrier}

\begin{definition}[TopGroupUp BHist carrier]
\label{def:topgroup-bhist-carrier}
A $\TopGroupUp$ $\mathsf{BHist}$ carrier is a finite packet
$$
\mathcal G=(G,T,m,i,n,\lambda,\rho)
$$
over the displayed $\NameCert_{\GroupUp}$ and $\NameCert_{\TopologyUp}$
source rows. The row $G$ is the group carrier and classifier surface. The row
$T$ is the topology carrier containing the public open-neighbourhood rows. The
row $m$ is the multiplication graph row, with endpoints read in $G$ and
neighbourhoods read in $T$. The row $i$ is the inversion graph row with the
same carrier and neighbourhood interface. The row $n$ records the open
neighbourhood transport rows used by multiplication and inversion. The row
$\lambda$ is the finite $\Cont$ ledger for continuity transport, and $\rho$ is
the $\Pkg$ provenance row for the group, topology, operation, neighbourhood,
continuity, and ledger entries.

The classifier between two carriers compares the $G$, $T$, $m$, $i$, $n$, and
$\rho$ rows by $\hsame$ and compares the continuation ledgers by $\Cont$. No
external topological group object is part of the carrier.
\end{definition}

\section{The certificate}
\label{sec:topgroup-certificate}

\begin{theorem}[TopGroupUp multiplication continuity carrier]
\label{thm:topgroup-multiplication-continuity-carrier}
Let $\mathcal G=(G,T,m,i,n,\lambda,\rho)$ be a $\TopGroupUp$
$\mathsf{BHist}$ carrier. If $x$ and $y$ are carried by the group row $G$, and
if an open-neighbourhood row of $T$ is displayed around the product endpoint
$m(x,y)$, then the multiplication graph row keeps $m(x,y)$ in the topology
carrier. Transporting $x$, $y$, the product row, and the target neighbourhood
by the classifier uses only the displayed $\hsame$ rows and the finite
$\Cont$ continuity ledger.
\end{theorem}
\begin{proof}
Project the group row, topology row, multiplication graph row,
neighbourhood row, continuation ledger, and provenance row from
\autoref{def:topgroup-bhist-carrier}. These projections name both operation
endpoints and the target open-neighbourhood row.

The classifier transports the two group endpoints and the multiplication graph
componentwise by $\hsame$. It also transports the relevant open-neighbourhood
row in $T$ and records the continuity step in the finite $\Cont$ ledger.

Repack the transported multiplication graph with the same topology carrier and
provenance row. The resulting product endpoint is still read through $T$, so
the multiplication continuity row remains inside the $\TopGroupUp$ carrier.
\end{proof}

\begin{theorem}[TopGroupUp inversion continuity carrier]
\label{thm:topgroup-inversion-continuity-carrier}
Let $\mathcal G=(G,T,m,i,n,\lambda,\rho)$ be a $\TopGroupUp$
$\mathsf{BHist}$ carrier. If $x$ is carried by the group row $G$, and if an
open-neighbourhood row of $T$ is displayed around the inverse endpoint
$i(x)$, then the inversion graph row keeps $i(x)$ in the topology carrier.
Transporting $x$, its inverse row, and the target neighbourhood by the
classifier uses only the displayed $\hsame$ rows and the same finite $\Cont$
continuity ledger.
\end{theorem}
\begin{proof}
Project the inversion graph row from
\autoref{def:topgroup-bhist-carrier}, together with the group carrier,
topology carrier, neighbourhood row, continuation ledger, and provenance row.

The endpoint transport is componentwise: $x$ is transported inside $G$, the
inverse graph row is transported by $\hsame$, and the target neighbourhood is
transported inside $T$. The continuity entry is recorded in the finite
$\Cont$ ledger already carried by $\mathcal G$.

Repack these rows with the unchanged source certificates. Since the carrier
contains no additional neighbourhood object or external topology, the inverse
endpoint remains in the displayed $\TopGroupUp$ carrier.
\end{proof}

\begin{theorem}[TopGroupUp NameCert obligation surface]
\label{thm:topgroup-namecert-obligation-surface}
The public $\NameCert_{\TopGroupUp}$ obligation surface consists exactly of
the $\mathsf{BHist}$ carrier of \autoref{def:topgroup-bhist-carrier}, its
componentwise classifier, the inherited $\GroupUp$ operation rows, the
$\TopologyUp$ open-neighbourhood rows, the multiplication and inversion
continuity rows, and the finite $\Cont$/$\Pkg$ ledger scope. It exports no
group-topology data beyond those displayed BEDC rows.
\end{theorem}
\begin{proof}
Unpack the carrier packet. The projections expose the group source, topology
source, multiplication graph, inversion graph, open-neighbourhood transport,
continuation ledger, and provenance row.

Apply \autoref{thm:topgroup-multiplication-continuity-carrier} for the
product continuity row and
\autoref{thm:topgroup-inversion-continuity-carrier} for the inverse
continuity row. Both results use only the componentwise classifier and the
finite $\Cont$ ledger.

Repack the projected rows as the public certificate surface. The source rows
are the displayed $\GroupUp$ and $\TopologyUp$ certificates, while every
operation and continuity row is carried over those sources, so the obligation
surface is exhausted by the listed finite packet.
\end{proof}

\begin{theorem}[TopGroupUp Cont ledger continuity exhaustion]
\label{thm:topgroup-cont-ledger-continuity-exhaustion}
Let $\mathcal G=(G,T,m,i,n,\lambda,\rho)$ be a $\TopGroupUp$ packet with
displayed $\NameCert_{\GroupUp}$ and $\NameCert_{\TopologyUp}$ source rows.
The finite $\Cont$ ledger $\lambda$ exhausts exactly the multiplication and
inversion continuity rows over those displayed sources: every public
continuity obligation of $\mathcal G$ is one of the product rows of
\autoref{thm:topgroup-multiplication-continuity-carrier} or one of the
inverse rows of \autoref{thm:topgroup-inversion-continuity-carrier}, and
those two families are carried by $\lambda$ with no extra topology object.
\end{theorem}
\begin{proof}
Project the ledger coordinate $\lambda$ from the carrier packet of
\autoref{def:topgroup-bhist-carrier}. The same projection exposes the
displayed $\GroupUp$ source, $\TopologyUp$ source, multiplication row,
inversion row, neighbourhood row, and provenance row.

Apply \autoref{thm:topgroup-multiplication-continuity-carrier} to the product
row and \autoref{thm:topgroup-inversion-continuity-carrier} to the inverse
row. These are the only operation-continuity rows exported by the public
surface of \autoref{thm:topgroup-namecert-obligation-surface}.

Repack the two projected families with the unchanged source certificates.
Because the carrier has no further topological component, the ledger is
exhausted by those two finite $\Cont$ families.
\end{proof}

\begin{theorem}[TopGroupUp classifier continuity transport]
\label{thm:topgroup-classifier-continuity-transport}
For classifier-related $\TopGroupUp$ packets
$\mathcal G=(G,T,m,i,n,\lambda,\rho)$ and
$\mathcal G'=(G',T',m',i',n',\lambda',\rho')$, the public classifier
transports $m$, $i$, $n$, $\lambda$, and $\rho$ componentwise by the carried
$\hsame$ rows and the finite $\Cont$ ledger. In particular, continuity
transport for the product and inverse rows is exactly the transport supplied
by \autoref{thm:topgroup-cont-ledger-continuity-exhaustion}.
\end{theorem}
\begin{proof}
Unpack the two packets with \autoref{def:topgroup-bhist-carrier}. The
component projections give the group carrier, topology carrier, multiplication
graph, inversion graph, neighbourhood row, ledger row, and provenance row on
both sides.

Transport the operation rows and neighbourhood rows componentwise by the
displayed $\hsame$ witnesses. The product and inverse continuity entries then
transport through the same finite ledger by
\autoref{thm:topgroup-cont-ledger-continuity-exhaustion}.

The provenance row $\rho$ is read only as the displayed source package, so it
transports with the same source certificates. Repacking these transported
coordinates gives the classifier-related $\TopGroupUp$ packet without adding
an external continuity predicate.
\end{proof}

\begin{theorem}[TopGroupUp public certificate boundary]
\label{thm:topgroup-public-certificate-boundary}
The complete public $\NameCert_{\TopGroupUp}$ boundary is formed by
\autoref{def:topgroup-bhist-carrier},
\autoref{thm:topgroup-multiplication-continuity-carrier},
\autoref{thm:topgroup-inversion-continuity-carrier},
\autoref{thm:topgroup-cont-ledger-continuity-exhaustion}, and
\autoref{thm:topgroup-classifier-continuity-transport}. No host
topological group and no external continuity predicate is read outside those
displayed BEDC rows.
\end{theorem}
\begin{proof}
Combine the carrier definition with the two continuity carrier rows. The
carrier supplies the finite packet, while the multiplication and inversion
theorems identify the only operation-continuity rows over the displayed
$\GroupUp$ and $\TopologyUp$ sources.

The ledger exhaustion theorem records that the finite $\Cont$ coordinate
contains exactly those rows. The classifier transport theorem records that
the same boundary is stable under componentwise $\hsame$ transport.

Thus the public certificate boundary is precisely the listed carrier,
continuity, ledger, transport, and provenance data. Since each component is
projected from the displayed packet, no host topological-group structure is
available to the certificate.
\end{proof}

\begin{theorem}[TopGroupUp source-coupled carrier scope]
\label{thm:topgroup-source-coupled-carrier-scope}
Every accepted $\TopGroupUp$ packet projects one shared $\mathsf{BHist}$ group
source row, topology source row, multiplication row, inversion row, $\Cont$
ledger, and $\Pkg$ provenance row. These rows are the source of the public
carrier and classifier scope; no independent host topological-group object is
available outside the displayed packet.
\end{theorem}
\begin{proof}
Unpack the accepted packet by \autoref{def:topgroup-bhist-carrier}. The
projection exposes the single group source $G$, topology source $T$,
multiplication row $m$, inversion row $i$, neighbourhood row $n$, ledger
$\lambda$, and provenance row $\rho$.

Apply \autoref{thm:topgroup-multiplication-continuity-carrier} and
\autoref{thm:topgroup-inversion-continuity-carrier} to read the operation
rows over those same sources. Apply
\autoref{thm:topgroup-cont-ledger-continuity-exhaustion} to read the finite
$\Cont$ scope and
\autoref{thm:topgroup-classifier-continuity-transport} for the carried
classifier transport.

Finally use \autoref{thm:topgroup-public-certificate-boundary}. Since that
boundary is exhausted by the projected rows and their provenance, there is no
second carrier from which an external topological group could be read.
\end{proof}

\begin{theorem}[TopGroupUp operation-continuity ledger scope]
\label{thm:topgroup-operation-continuity-ledger-scope}
For every accepted $\TopGroupUp$ packet, the product and inverse continuity
rows are exactly the two operation-neighbourhood transport families carried
by the same finite $\Cont$ ledger. Under endpoint $\hsame$ transport, both
families remain in that ledger and no additional continuity family is
introduced.
\end{theorem}
\begin{proof}
Project the packet using \autoref{def:topgroup-bhist-carrier}. The
continuity coordinate is the single finite ledger $\lambda$, and its
operation entries are attached to the displayed multiplication and inversion
rows.

Apply \autoref{thm:topgroup-cont-ledger-continuity-exhaustion}. It identifies
the public continuity obligations with the product rows of
\autoref{thm:topgroup-multiplication-continuity-carrier} and the inverse rows
of \autoref{thm:topgroup-inversion-continuity-carrier}.

For transport, apply
\autoref{thm:topgroup-classifier-continuity-transport}. The endpoint
$\hsame$ witnesses move the operation and neighbourhood endpoints
componentwise, while the product and inverse continuity entries remain the
same two ledger families.
\end{proof}

\begin{theorem}[TopGroupUp public NameCert readback exactness]
\label{thm:topgroup-public-namecert-readback-exactness}
The public $\NameCert_{\TopGroupUp}$ readback is exhausted by the carrier,
classifier, product continuity rows, inverse continuity rows, finite $\Cont$
ledger, and $\Pkg$ provenance. Reading a public certificate therefore returns
exactly those displayed BEDC rows and no further topological-group datum.
\end{theorem}
\begin{proof}
Start from the public boundary theorem
\autoref{thm:topgroup-public-certificate-boundary}. Its listed sources are
the carrier definition, the product and inverse continuity carrier rows, the
ledger exhaustion row, and the classifier transport row.

Use \autoref{def:topgroup-bhist-carrier} to identify the carrier,
classifier, finite $\Cont$ ledger, and $\Pkg$ provenance coordinates. Use
\autoref{thm:topgroup-multiplication-continuity-carrier} and
\autoref{thm:topgroup-inversion-continuity-carrier} for the two operation
continuity families, and
\autoref{thm:topgroup-cont-ledger-continuity-exhaustion} to see that the
ledger contains no further public continuity obligations.

Thus every public readback component is one of the named coordinates or one
of the two operation-continuity families. The classifier transport theorem
\autoref{thm:topgroup-classifier-continuity-transport} gives stability of
exactly those rows under $\hsame$, so the readback has no additional datum.
\end{proof}

\begin{theorem}[TopGroupUp product neighbourhood transport]
\label{thm:topgroup-product-neighbourhood-transport}
Let $\mathcal G=(G,T,m,i,n,\lambda,\rho)$ be a $\TopGroupUp$ packet with
displayed $\NameCert_{\GroupUp}$ and $\NameCert_{\TopologyUp}$ source rows.
For carried $\mathsf{BHist}$ endpoints $x,x',y,y',z,z'$, suppose
$m(x,y,z)$ is the displayed product row, $z$ lies in a neighbourhood row of
$T$, and $\hsame(x,x')$, $\hsame(y,y')$, and $\hsame(z,z')$ are the carried
endpoint transports. Then the product neighbourhood row transports to the
endpoint $z'$ using only the componentwise $\hsame$ rows, the
$\NameCert_{\GroupUp}$ operation source, the $\NameCert_{\TopologyUp}$
neighbourhood source, and the finite $\Cont$ ledger carried by
$\mathcal G$.
\end{theorem}
\begin{proof}
Project the packet $\mathcal G$. The projections expose the carried
$\mathsf{BHist}$ endpoints, the $\NameCert_{\GroupUp}$ multiplication row,
the $\NameCert_{\TopologyUp}$ neighbourhood row around the product endpoint,
and the relevant entry of the finite $\Cont$ ledger.

Transport the two input endpoints and the product endpoint componentwise by
the displayed $\hsame$ rows. The group source transports the multiplication
graph row, while the topology source transports the target neighbourhood row
inside $T$.

Repack the transported multiplication graph with the transported
neighbourhood row and the same continuity ledger entry. No outside
topological-group datum is introduced, so the transported row remains a
$\TopGroupUp$ product neighbourhood row over $\mathcal G$.
\end{proof}

\begin{theorem}[TopGroupUp inverse neighbourhood transport]
\label{thm:topgroup-inverse-neighbourhood-transport}
Let $\mathcal G=(G,T,m,i,n,\lambda,\rho)$ be a $\TopGroupUp$ packet with
displayed $\NameCert_{\GroupUp}$ and $\NameCert_{\TopologyUp}$ source rows.
For carried $\mathsf{BHist}$ endpoints $x,x',y,y'$, suppose $i(x,y)$ is the
displayed inverse row, $y$ lies in a neighbourhood row of $T$, and
$\hsame(x,x')$ and $\hsame(y,y')$ are the carried endpoint transports. Then
the inverse neighbourhood row transports to the endpoint $y'$ using only the
componentwise $\hsame$ rows, the $\NameCert_{\GroupUp}$ inverse source, the
$\NameCert_{\TopologyUp}$ neighbourhood source, and the finite $\Cont$ ledger
carried by $\mathcal G$.
\end{theorem}
\begin{proof}
Unpack $\mathcal G$ to obtain the group source, topology source, inverse graph
row, neighbourhood row around the inverse endpoint, and the matching
continuity entry in the $\Cont$ ledger.

The endpoint transport is componentwise. The $\hsame(x,x')$ row transports the
input, the $\hsame(y,y')$ row transports the inverse endpoint, the
$\NameCert_{\GroupUp}$ source carries the inverse graph across those rows, and
the $\NameCert_{\TopologyUp}$ source carries the neighbourhood row inside
$T$.

Repack the transported inverse graph and transported neighbourhood with the
same finite continuity entry. The result is still a row of the displayed
$\TopGroupUp$ packet, because every field used came from $\mathcal G$.
\end{proof}

\begin{theorem}[TopGroupUp continuity ledger coverage]
\label{thm:topgroup-continuity-ledger-coverage}
For a $\TopGroupUp$ packet
$\mathcal G=(G,T,m,i,n,\lambda,\rho)$ with displayed
$\NameCert_{\GroupUp}$ and $\NameCert_{\TopologyUp}$ sources, the finite
$\Cont$ ledger covers exactly the product and inverse neighbourhood transport
rows over carried $\mathsf{BHist}$ endpoints. Its rows are the transported
operation neighbourhood entries of
\autoref{thm:topgroup-product-neighbourhood-transport}, the transported
inverse neighbourhood entries of
\autoref{thm:topgroup-inverse-neighbourhood-transport}, and the source
provenance rows already displayed by
\autoref{thm:topgroup-namecert-obligation-surface}. It exports no additional
topological-group structure.
\end{theorem}
\begin{proof}
Start from the public obligation surface. Apply
\autoref{thm:topgroup-namecert-obligation-surface} to project the
$\mathsf{BHist}$ carrier, componentwise classifier, group source, topology
source, multiplication and inversion rows, and finite $\Cont$/$\Pkg$ ledger.

The product rows are covered by
\autoref{thm:topgroup-product-neighbourhood-transport}. Each entry uses the
displayed carried endpoints, the componentwise $\hsame$ transport rows, and
the same $\NameCert_{\GroupUp}$ and $\NameCert_{\TopologyUp}$ sources.

The inverse rows are covered by
\autoref{thm:topgroup-inverse-neighbourhood-transport}. These entries use the
same packet projections and therefore sit in the same finite $\Cont$ ledger.

Repack the two transport families with the provenance rows of $\mathcal G$.
Since the ledger contains only those product and inverse continuity rows, its
coverage is exactly the displayed $\TopGroupUp$ continuity surface.
\end{proof}

\begin{definition}[TopGroupUp source-coupled continuity packet]
\label{def:topgroup-source-coupled-continuity-packet}
The public source-coupled continuity packet for $\TopGroupUp$ is the finite
$\mathsf{BHist}$ packet consisting of the displayed
$\NameCert_{\GroupUp}$ row, the displayed $\NameCert_{\TopologyUp}$ row, the
product graph row, the inverse graph row, the source and target neighbourhood
rows, the finite $\Cont$ continuity ledger, and the $\mathsf{Pkg}$ provenance
row connecting those displayed sources. A packet is accepted only when these
rows are the rows projected by the $\TopGroupUp$ carrier
$\mathcal G=(G,T,m,i,n,\lambda,\rho)$.
\end{definition}

\begin{theorem}[TopGroupUp source-coupled continuity boundary]
\label{thm:topgroup-source-coupled-continuity-boundary}
Every accepted source-coupled continuity packet projects exactly its
$\NameCert_{\GroupUp}$ source row, its $\NameCert_{\TopologyUp}$ source row,
the product and inverse graph rows, the componentwise $\hsame$ endpoint
transport rows, the finite $\Cont$ continuity rows, and the $\mathsf{Pkg}$
provenance row. It exports no ambient topological-group object, no external
topology carrier equality, and no group operation beyond the carried graph
rows.
\end{theorem}
\begin{proof}
Project the accepted packet of
\autoref{def:topgroup-source-coupled-continuity-packet}. Its displayed
coordinates are precisely the two source certificates, the product and inverse
graph rows, the neighbourhood rows, the finite continuity ledger, and the
provenance row.

The componentwise $\hsame$ rows are the endpoint transports used by those
coordinates. They are already accounted for by the product transport theorem
\autoref{thm:topgroup-product-neighbourhood-transport} and the inverse
transport theorem \autoref{thm:topgroup-inverse-neighbourhood-transport}.

Completeness is the reverse packaging. Given exactly the projected rows, repack
them as the finite $\mathsf{BHist}$ packet of the definition and attach the
same $\Cont$ and $\mathsf{Pkg}$ ledgers. No extra ambient object is available,
because every public projection is one of the listed rows.
\end{proof}

\begin{theorem}[TopGroupUp continuity consumer exhaustion]
\label{thm:topgroup-continuity-consumer-exhaustion}
A downstream consumer of a $\TopGroupUp$ source-coupled continuity packet
receives only the product-neighbourhood transport family of
\autoref{thm:topgroup-product-neighbourhood-transport}, the inverse-neighbourhood
transport family of \autoref{thm:topgroup-inverse-neighbourhood-transport}, and
their finite ledger coverage in \autoref{thm:topgroup-continuity-ledger-coverage}.
The consumer cannot read a topological-group object, an external continuity
predicate, or any row outside the displayed $\mathsf{BHist}$, $\hsame$,
$\Cont$, $\mathsf{Pkg}$, $\NameCert_{\GroupUp}$, and
$\NameCert_{\TopologyUp}$ packet.
\end{theorem}
\begin{proof}
Start with the boundary theorem
\autoref{thm:topgroup-source-coupled-continuity-boundary}. It restricts the
consumer input to the displayed source certificates, endpoint transports,
finite continuity rows, and provenance row.

Apply \autoref{thm:topgroup-continuity-ledger-coverage}. The finite $\Cont$
ledger is exhausted by the product-neighbourhood and inverse-neighbourhood
transport rows, with provenance supplied by the same source certificates.

Finally use the two transport theorems named in the statement to repack the
consumer surface. Since every available row is either one of those transport
families or its $\mathsf{BHist}$ provenance, the consumer has no access to an
unlisted ambient continuity datum.
\end{proof}

\begin{theorem}[TopGroupUp root source compatibility]
\label{thm:topgroup-root-source-compatibility}
For every accepted $\TopGroupUp$ source-coupled continuity packet, the displayed
$\NameCert_{\GroupUp}$ row and displayed $\NameCert_{\TopologyUp}$ row share the
same $\mathsf{BHist}$ carrier endpoints of
\autoref{def:topgroup-bhist-carrier}. Classifier transport between the two
source rows is supplied by the componentwise $\hsame$ endpoint rows together
with the finite $\Cont$ continuity ledger.
\end{theorem}
\begin{proof}
Unpack the carrier by \autoref{def:topgroup-bhist-carrier}. The group source
certificate and topology source certificate are projections of the same
$\mathsf{BHist}$ packet, so their carrier endpoints are not independent
objects; they are the shared endpoints carried by
$\mathcal G=(G,T,m,i,n,\lambda,\rho)$.

The componentwise $\hsame$ rows are the classifier transports attached to those
shared endpoints. The product and inverse transport theorems,
\autoref{thm:topgroup-product-neighbourhood-transport} and
\autoref{thm:topgroup-inverse-neighbourhood-transport}, use those rows as the
source and target classifiers for their continuity statements.

Apply \autoref{thm:topgroup-continuity-ledger-coverage}. The finite $\Cont$
ledger records exactly the product and inverse continuity rows over the shared
carrier endpoints. Repack these rows to get the stated source compatibility.
\end{proof}

\begin{theorem}[TopGroupUp product-inverse obligation triad]
\label{thm:topgroup-product-inverse-obligation-triad}
The public $\TopGroupUp$ operation surface consists exactly of three
obligations: carrier compatibility between $\NameCert_{\GroupUp}$ and
$\NameCert_{\TopologyUp}$, product-neighbourhood transport, and
inverse-neighbourhood transport. These are the rows supplied by
\autoref{thm:topgroup-root-source-compatibility},
\autoref{thm:topgroup-product-neighbourhood-transport},
\autoref{thm:topgroup-inverse-neighbourhood-transport}, and
\autoref{thm:topgroup-continuity-ledger-coverage}.
\end{theorem}
\begin{proof}
Project the accepted packet through \autoref{def:topgroup-bhist-carrier}. The
carrier row gives the common source surface for the group and topology
certificates, while the operation graph rows expose only multiplication and
inversion.

\autoref{thm:topgroup-root-source-compatibility} identifies the carrier
compatibility obligation. The product-neighbourhood transport obligation is
exactly \autoref{thm:topgroup-product-neighbourhood-transport}, and the
inverse-neighbourhood transport obligation is exactly
\autoref{thm:topgroup-inverse-neighbourhood-transport}.

It remains to show that the list is exhaustive. Apply
\autoref{thm:topgroup-continuity-ledger-coverage}; the $\Cont$ ledger contains
only the product and inverse continuity rows. Together with the shared source
carrier row, these three obligations are the whole public operation surface.
\end{proof}

\begin{theorem}[TopGroupUp public consumer minimality]
\label{thm:topgroup-public-consumer-minimality}
Every downstream consumer of a $\TopGroupUp$ packet receives only the displayed
source certificate rows, the product and inverse operation rows, the finite
$\Cont$ rows, and the $\mathsf{Pkg}$ provenance. It receives no ambient
topological-group object and no order, topology, or operation row outside the
rows listed by \autoref{thm:topgroup-product-inverse-obligation-triad}.
\end{theorem}
\begin{proof}
Start from the source-coupled packet of
\autoref{def:topgroup-source-coupled-continuity-packet} and unpack its carrier
by \autoref{def:topgroup-bhist-carrier}. Its projections are the two source
certificates, the product and inverse graph rows, the finite $\Cont$ ledger,
and the $\mathsf{Pkg}$ provenance row.

\autoref{thm:topgroup-product-inverse-obligation-triad} repacks the operation
rows as carrier compatibility, product-neighbourhood transport, and
inverse-neighbourhood transport. No fourth operation row is present in the
finite ledger.

Finally apply \autoref{thm:topgroup-continuity-consumer-exhaustion}. It states
that downstream readback is exhausted by the same transport families, their
$\mathsf{BHist}$ provenance, $\hsame$ endpoint rows, $\Cont$ rows, and source
$\NameCert$ rows. Thus the public consumer receives exactly the listed data and
no ambient topological-group object.
\end{proof}
