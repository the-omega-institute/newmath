\chapter{A Concrete Naming Certificate for $\DiffFormUp$}
\label{ch:concrete-instances-diffform-namecert}

The differential-form interface $\DiffFormUp$ packages the carrier of antisymmetric multilinear maps on the tangent bundle of a manifold, together with the wedge product. It depends on $\NameCert_{\ManifoldUp}$ (\autoref{ch:concrete-instances-manifold-namecert}) and $\NameCert_{\TensorProductUp}$ (\autoref{ch:concrete-instances-tensorproduct-namecert}), and is the substrate for the de~Rham complex.

\section{Carrier definition}
\label{sec:diffform-carrier}

\begin{definition}[Differential-form BHist form carrier]
\label{def:diffform-bhist-form-carrier}
Fix a $\NameCert_{\ManifoldUp}$ source $M$ and a
$\NameCert_{\TensorProductUp}$ source $T$ whose tensor rows are read over the
tangent-bundle histories of $M$. A differential-form BHist row is a tuple of
histories
$$
(d,\mathcal{P},\tau,\sigma,\alpha,\lambda)
:
\mathsf{BHist}^6
$$
accepted by the predicate $\mathsf{DiffFormBHistCarrier}_{M,T}$ exactly when
the following BEDC rows are all present:
\begin{enumerate}
  \item $\mathsf{UnaryHistory}(d)$ records the form degree.
  \item $\mathcal{P}:\mathsf{ProbeBundle}$ is the visible finite spine of tangent
  probe names, and each bundled probe row is carried by the tangent-source
  surface exposed by $\NameCert_{\ManifoldUp}$.
  \item $\tau$ is a $\TensorProductUp$ multilinear row over the bundled
  tangent probes, with continuation ledgers $\Cont$ recording the ordered
  tensor concatenations used by the row.
  \item $\sigma$ is the scalar endpoint produced from that tensor row,
  classified by the scalar classifier retained by the manifold and tensor
  sources.
  \item $\alpha$ is the visible permutation ledger: for every displayed
  adjacent swap of bundled probe rows, the transported scalar endpoint is
  classified with the signed endpoint demanded by antisymmetry.
  \item $\lambda$ records the source names $M$ and $T$, the degree row, the
  probe bundle, the tensor row, the scalar endpoint row, and the antisymmetry
  rows as the only accepted provenance for the form row.
\end{enumerate}
\end{definition}

\begin{definition}[Differential-form BHist form classifier]
\label{def:diffform-bhist-form-classifier}
For carried form rows
$$
\omega=(d,\mathcal{P},\tau,\sigma,\alpha,\lambda)
$$
and
$$
\omega'=(d',\mathcal{P}',\tau',\sigma',\alpha',\lambda'),
$$
the classifier $\omega\sim_{\DiffFormUp}\omega'$ holds when the degree rows
are $\hsame$, the scalar endpoints agree pointwise under the retained scalar
classifier for every matched multilinear probe row, and the antisymmetry and
tensor-source ledgers match as visible BHist rows:
$$
\hsame(d,d')
$$
$$
\text{and for every matched bundled multilinear row, }
\sigma\sim_R\sigma'
$$
$$
\text{and }
\hsame(\alpha,\alpha')\wedge\hsame(\tau,\tau')
\wedge\hsame(\lambda,\lambda').
$$
Here $\sim_R$ is the scalar classifier already present in the upstream scalar
surface; the differential-form classifier introduces no classifier outside
$\hsame$ and the displayed source classifiers.
\end{definition}

\section{The certificate}
\label{sec:diffform-certificate}

\begin{theorem}[Differential-form carrier hsame transport]
\label{thm:diffform-carrier-hsame-transport}
Let $\omega$ be a row accepted by
$\mathsf{DiffFormBHistCarrier}_{M,T}$, and let $\omega'$ be obtained by
replacing its six displayed histories with $\hsame$-equal histories while
keeping the same $\NameCert_{\ManifoldUp}$ and $\NameCert_{\TensorProductUp}$
sources. Then $\omega'$ is also accepted by
$\mathsf{DiffFormBHistCarrier}_{M,T}$.
\end{theorem}
\begin{proof}
Unpack the carrier row of \autoref{def:diffform-bhist-form-carrier}. The
degree condition is transported by $\hsame$ for $\mathsf{UnaryHistory}$.

The probe bundle is a closed $\mathsf{ProbeBundle}$ spine, so its bundled
tangent rows are transported pointwise through the displayed manifold source
classifier. The tensor row is then transported through the
$\TensorProductUp$ classifier, retaining the same $\Cont$ ledgers after
source, probe, and endpoint histories are replaced by $\hsame$-equal rows.

It remains to transport the scalar and antisymmetry rows. The scalar endpoint
uses the retained scalar classifier, and its pointwise transport follows from
classifier stability in the source surface. Each antisymmetry row is a visible
permutation ledger over the same probe bundle; transporting the two scalar
endpoints of the swap row gives the required signed comparison. Repacking
these transported fields gives the carried row for $\omega'$.
\end{proof}

\begin{theorem}[Differential-form classifier stability]
\label{thm:diffform-form-classifier-stability}
For rows carried by $\mathsf{DiffFormBHistCarrier}_{M,T}$, the classifier of
\autoref{def:diffform-bhist-form-classifier} is stable under carrier
transport: if $\omega\sim_{\DiffFormUp}\eta$ and both rows are transported to
$\omega'$ and $\eta'$ by componentwise $\hsame$ comparisons, then
$\omega'\sim_{\DiffFormUp}\eta'$.
\end{theorem}
\begin{proof}
The degree comparison is the transitive composite of the transport comparison
for $\omega$, the original degree comparison, and the symmetric transport
comparison for $\eta$.

For each bundled multilinear probe row, use the tensor-source classifier to
identify the matched tensor rows after transport. The scalar endpoints are
then compared by the scalar classifier stability already retained in the
upstream scalar surface.

The antisymmetry and source-ledger clauses are pure $\hsame$ clauses on
visible BHist rows. Composing their original comparisons with the two
transport comparisons gives the required ledger matches for the transported
rows.
\end{proof}

\begin{theorem}[Differential-form probe-slot arity transport]
\label{thm:diffform-probe-slot-arity-transport}
Let
$$
\omega=(d,\mathcal{P},\tau,\sigma,\alpha,\lambda)
$$
and
$$
\omega'=(d',\mathcal{P}',\tau',\sigma',\alpha',\lambda')
$$
be rows carried by $\mathsf{DiffFormBHistCarrier}_{M,T}$. Suppose
$\hsame(d,d')$ and the displayed $\mathsf{ProbeBundle}$ spines of
$\mathcal{P}$ and $\mathcal{P}'$ are matched componentwise through the
tangent-source classifier supplied by $\NameCert_{\ManifoldUp}$. Then the
visible arity of the probe slots in $\mathcal{P}$ is the same as the visible
arity of the probe slots in $\mathcal{P}'$. Moreover, every transported
multilinear tensor row remains classified by the $\TensorProductUp$ source
over those transported tangent probes.
\end{theorem}
\begin{proof}
Unpack the carrier rows in
\autoref{def:diffform-bhist-form-carrier}. The degree entries are unary
history rows, so $\hsame(d,d')$ transports the displayed degree without
creating or deleting a probe slot.

The probe component is a closed $\mathsf{ProbeBundle}$ spine. Induct on this
spine. The empty spine has arity zero on both sides. In the cons case, the
head tangent probe is transported through the classifier retained by
$\NameCert_{\ManifoldUp}$, and the induction hypothesis preserves the arity of
the tail. Therefore the visible slot count is preserved at each spine
constructor.

It remains to place the transported multilinear row back in the tensor source.
The tensor component of a carried differential-form row is accepted only as a
$\TensorProductUp$ row over the bundled tangent probes. Since the probe
classifier transport is componentwise and the tensor row retains its displayed
$\Cont$ provenance, tensor-source classifier transport reclassifies $\tau'$ as
a multilinear row over the transported probe bundle.
\end{proof}

\begin{theorem}[Differential-form adjacent-swap involution ledger]
\label{thm:diffform-adjacent-swap-involution-ledger}
Let
$$
\omega=(d,\mathcal{P},\tau,\sigma,\alpha,\lambda)
$$
be carried by $\mathsf{DiffFormBHistCarrier}_{M,T}$. If a displayed row of the
antisymmetry ledger $\alpha$ records an adjacent swap of two neighboring probe
slots and classifies the resulting scalar endpoint with the signed endpoint
required by antisymmetry, then $\alpha$ also determines the reverse adjacent
swap row. Applying the displayed adjacent swap and then its reverse returns
endpoints classified with the original scalar endpoint under the retained
scalar classifier, and the provenance for both swap rows remains recorded in
$\lambda$.
\end{theorem}
\begin{proof}
Read the chosen adjacent-swap row from the visible antisymmetry ledger of
\autoref{def:diffform-bhist-form-carrier}. Its endpoints are scalar rows
classified by the scalar classifier retained from the manifold and tensor
sources, and its provenance is one of the rows listed in $\lambda$.

The reverse row uses the same two neighboring probe slots in the opposite
order. The scalar comparison for the reverse row is obtained by symmetry of
the retained scalar classifier, while the comparison between the double-swap
endpoint and the original endpoint is the transitive composite of the forward
and reverse scalar comparisons.

No hidden row is introduced in this construction. Both the displayed swap row
and the reverse swap row are retained as BHist provenance in $\lambda$, so the
two applications return to endpoints classified with the original scalar
endpoint inside the same carried differential-form row.
\end{proof}

\begin{theorem}[Differential-form wedge degree additivity ledger]
\label{thm:diffform-wedge-degree-additivity-ledger}
Let
$$
\omega=(d,\mathcal{P},\tau,\sigma,\alpha,\lambda)
$$
and
$$
\eta=(e,\mathcal{Q},\upsilon,\rho,\beta,\mu)
$$
be carried by $\mathsf{DiffFormBHistCarrier}_{M,T}$. If an admitted wedge row
$\omega\wedge_{\DiffFormUp}\eta$ is carried with output degree row $d\star e$,
then its visible degree ledger contains the unary continuation row
$\Cont(d,e,d\star e)$. If the two input form rows are transported by
componentwise classifier comparisons with degree rows $d'$ and $e'$ satisfying
$\hsame(d,d')$ and $\hsame(e,e')$, the wedge degree row is transported to a row
$d'\star e'$ classified with the same unary continuation/append degree
construction.
\end{theorem}
\begin{proof}
Start with the admitted wedge row and apply
\autoref{thm:diffform-wedge-ledger-coverage}. The wedge carrier contains the
visible tensor concatenation row joining the tensor component of $\omega$ to
the tensor component of $\eta$, and the carrier provenance records the degree
row of the wedge output.

The input degree rows are unary histories. Closure of unary histories under
continuation supplies the displayed degree ledger
$\Cont(d,e,d\star e)$ for the output degree row. This ledger is part of the
same BHist provenance that carries the wedge row, so it is visible to
downstream exterior-derivative consumers.

Now transport the two input form rows. The degree comparisons
$\hsame(d,d')$ and $\hsame(e,e')$ transport the two unary inputs, and the
componentwise classifier comparisons transport the probe, tensor, scalar, and
antisymmetry rows. Applying unary-history closure under the transported
continuation yields $d'\star e'$ with its corresponding $\Cont$ row.

Repackage the transported probe bundle, tensor concatenation, scalar endpoint,
antisymmetry ledger, and degree ledger as the wedge carrier row. The resulting
wedge row has degree additivity classified by the same differential-form
carrier and classifier surface.
\end{proof}

\begin{theorem}[Differential-form wedge ledger coverage]
\label{thm:diffform-wedge-ledger-coverage}
A wedge row $\omega\wedge_{\DiffFormUp}\eta$ is admitted by
$\mathsf{DiffFormBHistCarrier}_{M,T}$ only when its ledger visibly contains a
$\TensorProductUp$ concatenation row joining the tensor row of $\omega$ to the
tensor row of $\eta$. When that concatenation ledger is present, carrier and
classifier stability of the wedge row follow by transporting the two input
form rows through the tensor classifier and then through the retained scalar
classifier.
\end{theorem}
\begin{proof}
Read the wedge candidate through the carrier definition. The only provenance
accepted for a form row is its displayed ledger, and the tensor component of a
wedge must therefore be the visible tensor concatenation recorded there. If no
such $\TensorProductUp$ continuation row appears, the candidate has no accepted
carrier witness.

Assume the concatenation row is present. Transport the left and right form
rows by \autoref{thm:diffform-carrier-hsame-transport}. The
$\TensorProductUp$ source classifies the concatenated tensor endpoint after
the two input tensor rows have been transported, and the scalar endpoint is
classified by applying the retained scalar classifier to the transported
multilinear endpoints.

The antisymmetry ledger of the wedge is the concatenation of the two input
antisymmetry ledgers together with the displayed cross-permutation rows. Each
row is a visible BHist row, so $\hsame$ transport preserves it. These fields
reassemble into the carried wedge row and into the classifier comparison for
the transported wedge endpoints.
\end{proof}

\begin{definition}[Differential-form wedge associator ledger]
\label{def:diffform-wedge-associator-ledger}
For carried form rows $\omega$, $\eta$, and $\theta$ in
$\mathsf{DiffFormBHistCarrier}_{M,T}$, the visible wedge associator ledger is
the BHist package recording four rows: the $\TensorProductUp$ concatenation
spine for $(\omega\wedge_{\DiffFormUp}\eta)\wedge_{\DiffFormUp}\theta$, the
$\TensorProductUp$ concatenation spine for
$\omega\wedge_{\DiffFormUp}(\eta\wedge_{\DiffFormUp}\theta)$, the $\Cont$
associativity comparison between those tensor spines, and the common scalar
endpoint classifier obtained by transporting through the retained scalar
classifier. It also records the permutation-ledger comparison required to
identify the two antisymmetry ledgers after the tensor associativity
comparison has been applied.
\end{definition}

\begin{theorem}[Differential-form wedge associator closure]
\label{thm:diffform-wedge-associator-closure}
Let $\omega$, $\eta$, and $\theta$ be carried
$\mathsf{DiffFormBHistCarrier}_{M,T}$ rows. If the two binary wedge ledgers
needed for $(\omega\wedge_{\DiffFormUp}\eta)\wedge_{\DiffFormUp}\theta$ and
the two binary wedge ledgers needed for
$\omega\wedge_{\DiffFormUp}(\eta\wedge_{\DiffFormUp}\theta)$ are visible, then
the associator ledger of
\autoref{def:diffform-wedge-associator-ledger} carries both parenthesized
wedge rows and classifies them as equal under $\sim_{\DiffFormUp}$.
\end{theorem}
\begin{proof}
Apply \autoref{thm:diffform-wedge-ledger-coverage} to
$\omega\wedge_{\DiffFormUp}\eta$ and then to the wedge with $\theta$. This
constructs the left-parenthesized carrier row and its scalar endpoint
classifier from the two visible binary ledgers.

Apply the same theorem to $\eta\wedge_{\DiffFormUp}\theta$ and then to the
wedge with $\omega$. This constructs the right-parenthesized carrier row and
its scalar endpoint classifier from the other two visible binary ledgers.

The two tensor endpoints are compared by the $\Cont$ associativity row in the
visible $\TensorProductUp$ concatenation ledger. Transport the scalar endpoint
classifier across that comparison and the retained scalar classifier from
\autoref{def:diffform-bhist-form-classifier}.

It remains to identify the antisymmetry ledgers. The associator ledger carries
the required permutation comparison, and each permutation row is a visible
BHist row. Composing those rows by $\hsame$ transitivity gives the
$\DiffFormUp$ classifier equality between the two parenthesized wedge rows.
\end{proof}

\begin{theorem}[Differential-form exterior-algebra obligation surface]
\label{thm:diffform-exterior-algebra-obligation-surface}
The $\DiffFormUp$ exterior-algebra surface available to $\DeRhamUp$,
$\SymplecticUp$, and $\ContactUp$ consists of carrier transport, classifier
stability, binary wedge ledger coverage, and the wedge associator ledger. This
surface does not assert $d^2=0$, closedness, nondegeneracy, or contact
nonintegrability.
\end{theorem}
\begin{proof}
Carrier transport and classifier stability are supplied by
\autoref{thm:diffform-carrier-hsame-transport} and
\autoref{thm:diffform-form-classifier-stability}. Binary wedge admission and
transport are exactly \autoref{thm:diffform-wedge-ledger-coverage}.

Associativity of the exposed exterior product is not an unrecorded algebraic
law. It is the visible associator ledger of
\autoref{def:diffform-wedge-associator-ledger}, closed under the carrier and
classifier by \autoref{thm:diffform-wedge-associator-closure}.

Combining these rows with the NameCert surface of
\autoref{thm:diffform-namecert-obligation-surface} and the downstream scope
of \autoref{thm:diffform-exterior-derivative-downstream-scope} gives exactly
the surface available to de Rham, symplectic, and contact consumers. The
operator identity $d^2=0$ and the geometric conditions named in the statement
are not rows of this $\DiffFormUp$ surface, so they remain obligations of the
downstream certificates that use it.
\end{proof}

\begin{theorem}[Differential-form wedge degree classifier stability]
\label{thm:diffform-wedge-degree-classifier-stability}
Let $(\omega,\eta)$ and $(\omega',\eta')$ be two pairs of carried
$\DiffFormUp$ rows with $\omega\sim_{\DiffFormUp}\omega'$ and
$\eta\sim_{\DiffFormUp}\eta'$. If $\delta_{\omega,\eta}$ and
$\delta_{\omega',\eta'}$ are the wedge degree ledgers of
\autoref{def:diffform-wedge-degree-ledger}, and if the corresponding wedge
rows are admitted by the wedge ledger coverage of
\autoref{thm:diffform-wedge-ledger-coverage}, then
$$
\hsame(\delta_{\omega,\eta},\delta_{\omega',\eta'})
$$
and the wedge rows remain classified under $\sim_{\DiffFormUp}$.
\end{theorem}
\begin{proof}
The classifier comparison on the two input form pairs includes $\hsame$
comparisons on their degree rows by
\autoref{def:diffform-bhist-form-classifier}. Transporting
$\mathsf{UnaryHistory}$ along these comparisons gives the same unary degree
witnesses for the two transported inputs.

The displayed continuation row of
\autoref{def:diffform-wedge-degree-ledger} is a BHist row. Continuation
associativity and unit stability identify the output degree row obtained from
the transported input degrees with the output degree row in the second wedge
degree ledger, so the two degree ledgers are related by $\hsame$.

For the remaining wedge fields, apply
\autoref{thm:diffform-wedge-ledger-coverage}. The tensor concatenation row
transports through the $\TensorProductUp$ classifier, and the scalar endpoint
transports through the retained scalar classifier. Together with the degree
ledger comparison above and the visible antisymmetry ledger transport, these
rows repack as the required $\DiffFormUp$ classifier comparison for the wedge
rows.
\end{proof}

\begin{theorem}[Differential-form de Rham input boundary]
\label{thm:diffform-derham-input-boundary}
Any $\DeRhamUp$ exterior-derivative row sourced from $\DiffFormUp$ may depend
on the $\DiffFormUp$ source only through the carried $\mathsf{ProbeBundle}$,
the unary degree row, the multilinear $\TensorProductUp$ tensor row, the scalar
endpoint classifier, the antisymmetry ledger, wedge ledger coverage, and the
wedge degree ledger of \autoref{def:diffform-wedge-degree-ledger}. It cannot
import $d^2=0$, closedness, nondegeneracy, or contact nonintegrability from the
$\DiffFormUp$ certificate.
\end{theorem}
\begin{proof}
Inspect the carrier of \autoref{def:diffform-bhist-form-carrier}. The rows
visible to an exterior-derivative construction are the probe bundle, unary
degree row, tensor row, scalar endpoint, antisymmetry ledger, and source
ledger, all carried as BHist data.

The two wedge interfaces exported by $\DiffFormUp$ are exactly the tensor
coverage of \autoref{thm:diffform-wedge-ledger-coverage} and the degree ledger
of \autoref{def:diffform-wedge-degree-ledger}. Their stability is supplied by
\autoref{thm:diffform-wedge-degree-classifier-stability} together with the
carrier and classifier stability theorems above.

The identities $d^2=0$, closedness, nondegeneracy, and contact
nonintegrability require operator or geometric rows not listed in the
$\DiffFormUp$ carrier. Therefore a downstream $\DeRhamUp$, $\SymplecticUp$, or
$\ContactUp$ certificate must state those rows in its own surface.
\end{proof}

\begin{theorem}[Differential-form NameCert obligation surface]
\label{thm:diffform-namecert-obligation-surface}
The paper-side $\NameCert_{\DiffFormUp}$ obligation surface consists exactly
of the carrier of \autoref{def:diffform-bhist-form-carrier}, the classifier
of \autoref{def:diffform-bhist-form-classifier}, carrier transport,
classifier stability, wedge ledger coverage, the wedge associator ledger, the
wedge degree ledger, wedge degree classifier stability, the antisymmetry
permutation ledger, the de Rham input boundary of
\autoref{thm:diffform-derham-input-boundary}, and the downstream scope row of
\autoref{thm:diffform-exterior-derivative-downstream-scope}. It asserts no
standard bridge beyond these BEDC rows.
\end{theorem}
\begin{proof}
The carrier and classifier fields are the two displayed definitions. Carrier
transport is \autoref{thm:diffform-carrier-hsame-transport}, and classifier
stability is \autoref{thm:diffform-form-classifier-stability}.

The multiplicative operation exposed by the surface is the wedge row, whose
admission and stability are exactly the visible tensor-concatenation statement
of \autoref{thm:diffform-wedge-ledger-coverage}. Its associativity obligation
is the visible ledger of \autoref{def:diffform-wedge-associator-ledger} and
the closure statement of \autoref{thm:diffform-wedge-associator-closure}. Its
degree bookkeeping is the visible unary-history ledger of
\autoref{def:diffform-wedge-degree-ledger}, and classifier stability for that
ledger is \autoref{thm:diffform-wedge-degree-classifier-stability}.
Antisymmetry is not hidden in the wedge operation: it is the explicit
permutation ledger retained in every carrier row.

The remaining exported row is the downstream scope for exterior derivative
consumers, bounded by \autoref{thm:diffform-derham-input-boundary}. Since all
fields are stated with $BHist$, $\hsame$, $\Cont$, $\mathsf{ProbeBundle}$,
$\NameCert_{\ManifoldUp}$, and $\NameCert_{\TensorProductUp}$, this is a BEDC
obligation surface and not a claim of equivalence with a standard
differential-form object.
\end{proof}

\begin{theorem}[Differential-form exterior-derivative downstream scope]
\label{thm:diffform-exterior-derivative-downstream-scope}
Downstream exterior-derivative chapters may use a $\DiffFormUp$ row only
through its degree row, its carried $\mathsf{ProbeBundle}$ of tangent probes,
its multilinear $\TensorProductUp$ tensor row, its scalar endpoint classifier,
its antisymmetry ledger, wedge ledger coverage, the wedge associator ledger,
and the wedge degree ledger. In particular, the DiffForm surface supplies the
carrier, classifier, and exterior-algebra substrate for $\DeRhamUp$,
$\SymplecticUp$, and $\ContactUp$, but it does not by itself assert $d^2=0$,
closedness, nondegeneracy, or contact nonintegrability.
\end{theorem}
\begin{proof}
Inspect the exported fields of
\autoref{thm:diffform-namecert-obligation-surface}. Each accessible component
is one of the displayed carrier, classifier, stability, wedge, or ledger rows.

The exterior derivative needs the degree row and the multilinear scalar
endpoint surface in order to state degree raising and scalar endpoint
transport. Those rows are available from the carrier and classifier; the
wedge degree ledger supplies the visible input-degree bookkeeping allowed at
the de Rham boundary. Stability is supplied by the two transport theorems and
\autoref{thm:diffform-wedge-degree-classifier-stability}.

The identities $d^2=0$, closedness, nondegeneracy, and contact
nonintegrability require additional operator or geometric rows. They are not
fields of the $\DiffFormUp$ surface, so downstream chapters must state and
certify them in their own $\NameCert$ surfaces.
\end{proof}

\begin{theorem}[Differential-form source scope]
\label{thm:diffform-source-scope}
Every row accepted by $\mathsf{DiffFormBHistCarrier}_{M,T}$ depends only on
$BHist$, $\hsame$, $\Cont$, $\mathsf{ProbeBundle}$,
$\NameCert_{\ManifoldUp}$, and $\NameCert_{\TensorProductUp}$. No accepted
row may import a carrier component, classifier clause, or endpoint comparison
from outside those six displayed sources.
\end{theorem}
\begin{proof}
Project the accepted row through
\autoref{def:diffform-bhist-form-carrier}. Its six components are exactly the
degree history, probe bundle, tensor row, scalar endpoint, antisymmetry
ledger, and source ledger. The only structural rows mentioned there are
$BHist$, $\mathsf{ProbeBundle}$, the $\TensorProductUp$ continuation ledgers
$\Cont$, and the two upstream $\NameCert$ sources.

Now read the classifier clauses in
\autoref{def:diffform-bhist-form-classifier}. Degree, tensor, antisymmetry,
and source-ledger comparisons are $\hsame$ comparisons on visible histories,
while scalar endpoint comparison is the retained scalar classifier supplied by
the manifold and tensor sources. Thus every accepted classifier step remains
inside the stated source scope.
\end{proof}

\begin{theorem}[Differential-form wedge degree ledger]
\label{thm:diffform-wedge-degree-ledger}
For a carried wedge row $\omega\wedge_{\DiffFormUp}\eta$, the visible BHist
provenance exposes the degree row of $\omega$, the degree row of $\eta$, the
$\TensorProductUp$ concatenation ledger supplied by
\autoref{thm:diffform-wedge-ledger-coverage}, and the resulting unary degree
row of the wedge. This ledger states provenance for the wedge degree and does
not assert any exterior-derivative identity.
\end{theorem}
\begin{proof}
Unpack the wedge row by
\autoref{thm:diffform-wedge-ledger-coverage}. Admission of the wedge requires
a visible $\TensorProductUp$ concatenation row joining the tensor component of
the left carried form to the tensor component of the right carried form.

Read the two input degree rows from
\autoref{def:diffform-bhist-form-carrier}. They are the first carrier
components of $\omega$ and $\eta$, and the wedge row records them in its
source ledger together with the tensor concatenation row.

Repack those displayed histories as the wedge degree ledger. The output
degree row is a unary history visible in the carried wedge row, so downstream
chapters may cite the two input degree rows, the concatenation provenance, and
the output degree row without adding any law about $d$.
\end{proof}

\begin{theorem}[Differential-form scalar endpoint transport]
\label{thm:diffform-scalar-endpoint-transport}
Let $\omega$ and $\omega'$ be $\DiffFormUp$ rows with componentwise $\hsame$
comparisons on their probe bundles, tensor rows, antisymmetry ledgers, and
source ledgers. If the scalar endpoint of $\omega$ is accepted by the retained
scalar classifier from the upstream manifold and tensor sources, then the
scalar endpoint comparison transports to $\omega'$ through the
$\DiffFormUp$ classifier.
\end{theorem}
\begin{proof}
First transport the carrier row. The componentwise comparisons on the probe
bundle, tensor row, antisymmetry ledger, and source ledger are exactly the
visible histories required by
\autoref{thm:diffform-carrier-hsame-transport}, so the transported row is
again accepted by $\mathsf{DiffFormBHistCarrier}_{M,T}$.

Next apply classifier stability. By
\autoref{thm:diffform-form-classifier-stability}, the scalar endpoint clause
of \autoref{def:diffform-bhist-form-classifier} is preserved under the same
componentwise transport.

Finally, the scalar comparison used in that clause is not a new
$\DiffFormUp$ primitive. It is the retained scalar classifier from the
manifold and tensor sources, so endpoint comparison moves along the
transported probe, tensor, antisymmetry, and source ledgers already present in
the carried row.
\end{proof}

\begin{definition}[Differential-form antisymmetry chain ledger]
\label{def:diffform-antisymmetry-chain-ledger}
For carried rows
$$
\omega=(d,\mathcal{P},\tau,\sigma,\alpha,\lambda)
$$
and
$$
\omega'=(d,\mathcal{P}',\tau',\sigma',\alpha',\lambda')
$$
in $\mathsf{DiffFormBHistCarrier}_{M,T}$, a differential-form antisymmetry
chain ledger is a visible BHist row
$\mathsf{AntiChain}_{\omega,\omega'}$ accepted by
$\mathsf{DiffFormAntisymmetryChainLedger}_{M,T}$ exactly when it displays a
finite spine of adjacent-swap rows
$$
\mathcal{P}=\mathcal{P}_0,\mathcal{P}_1,\ldots,\mathcal{P}_n=\mathcal{P}'
$$
over the carried $\mathsf{ProbeBundle}$, with one scalar endpoint row and one
signed scalar comparison row recorded for each adjacent step
$\mathcal{P}_i\leadsto\mathcal{P}_{i+1}$. Its accepted source rows are only
$BHist$, $\hsame$, $\Cont$, the displayed $\mathsf{ProbeBundle}$ rows,
$\NameCert_{\ManifoldUp}$, and $\NameCert_{\TensorProductUp}$; the ledger does
not contain any permutation oracle outside those rows.
\end{definition}

\begin{theorem}[Differential-form antisymmetry chain coverage]
\label{thm:diffform-antisymmetry-chain-coverage}
\label{thm:diffform-antisymmetry-chain-ledger-well-formedness}
Every admitted finite permutation comparison between two carried
$\DiffFormUp$ rows over the same degree row is accepted only through a
differential-form antisymmetry chain ledger of
\autoref{def:diffform-antisymmetry-chain-ledger}. In particular, each
comparison row in the ledger is a visible adjacent-swap row, each adjacent
swap has the reverse row supplied by
\autoref{thm:diffform-adjacent-swap-involution-ledger}, and the composite
comparison uses no source beyond $BHist$, $\hsame$, $\Cont$,
$\mathsf{ProbeBundle}$, $\NameCert_{\ManifoldUp}$, and
$\NameCert_{\TensorProductUp}$.
\end{theorem}
\begin{proof}
Read the admitted permutation comparison inside the carrier row of
\autoref{def:diffform-bhist-form-carrier}. The only place where
antisymmetry data may enter a carried $\DiffFormUp$ row is the displayed
ledger $\alpha$ together with the provenance ledger $\lambda$, so a
permutation comparison without visible rows has no carrier witness.

Induct on the displayed finite adjacent-swap spine. The empty spine compares
the probe bundle with itself, and the endpoint comparison is the retained
scalar classifier clause already present in
\autoref{def:diffform-bhist-form-classifier}.

For the cons step, the head row is one adjacent swap of neighboring carried
probe slots. Apply
\autoref{thm:diffform-adjacent-swap-involution-ledger} to obtain the reverse
row and the scalar comparison returning the double swap to the original
endpoint. The induction hypothesis supplies coverage for the tail of the
chain, and $\hsame$ transitivity composes the visible BHist rows.

All rows used in the construction are the rows listed in
\autoref{def:diffform-antisymmetry-chain-ledger}. Therefore the finite
permutation comparison is exhausted by the adjacent-swap ledger, with no
hidden permutation rule or external classifier.
\end{proof}

\begin{theorem}[Differential-form antisymmetry chain classifier stability]
\label{thm:diffform-antisymmetry-chain-classifier-stability}
Let $\omega$ and $\eta$ be carried $\DiffFormUp$ rows whose permutation
comparison is witnessed by a differential-form antisymmetry chain ledger. If
$\omega$ and $\eta$ are transported componentwise by $\hsame$ comparisons on
their degree rows, probe bundles, tensor rows, scalar endpoint rows,
antisymmetry ledgers, and source ledgers, then the transported rows carry the
transported antisymmetry chain ledger and remain compared by
$\sim_{\DiffFormUp}$.
\end{theorem}
\begin{proof}
First transport the two carrier rows by
\autoref{thm:diffform-carrier-hsame-transport}. Componentwise $\hsame$
comparisons carry the probe bundles, tensor rows, scalar endpoints,
antisymmetry ledgers, and source ledgers to the transported rows while
retaining the same manifold and tensor source certificates.

Next transport the finite chain ledger. Each row of
\autoref{def:diffform-antisymmetry-chain-ledger} is a visible BHist row, and
each adjacent swap in the chain is a row over the transported
$\mathsf{ProbeBundle}$. Transport the scalar endpoint comparison for each
step through \autoref{thm:diffform-scalar-endpoint-transport}.

Apply \autoref{thm:diffform-antisymmetry-chain-coverage} to the transported
chain. The theorem identifies the transported finite permutation comparison
as exactly the composite of adjacent swaps in the visible ledger, and it
retains the reverse rows supplied by
\autoref{thm:diffform-adjacent-swap-involution-ledger}.

Finally apply \autoref{thm:diffform-form-classifier-stability}. The degree,
tensor, antisymmetry, and source-ledger clauses are $\hsame$ clauses, while
the scalar endpoint clause has already been transported through the retained
scalar classifier. These clauses reassemble into the required
$\sim_{\DiffFormUp}$ comparison for the transported rows.
\end{proof}
