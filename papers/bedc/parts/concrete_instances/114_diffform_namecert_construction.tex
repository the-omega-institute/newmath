\chapter{A Concrete Naming Certificate for $\DiffFormUp$}
\label{ch:concrete-instances-diffform-namecert}

The differential-form interface $\DiffFormUp$ packages the carrier of antisymmetric multilinear maps on the tangent bundle of a manifold, together with the wedge product. It depends on $\NameCert_{\ManifoldUp}$ (\autoref{ch:concrete-instances-manifold-namecert}) and $\NameCert_{\TensorProductUp}$ (\autoref{ch:concrete-instances-tensorproduct-namecert}), and is the substrate for the de~Rham complex.

\section{Carrier definition}
\label{sec:diffform-carrier}

\begin{definition}[Differential-form BHist form carrier]
\label{def:diffform-bhist-form-carrier}
Fix a $\NameCert_{\ManifoldUp}$ source $M$ and a
$\NameCert_{\TensorProductUp}$ source $T$ whose tensor rows are read over the
tangent-bundle histories of $M$. A differential-form BHist row is a tuple of
histories
$$
(d,\mathcal{P},\tau,\sigma,\alpha,\lambda)
:
\mathsf{BHist}^6
$$
accepted by the predicate $\mathsf{DiffFormBHistCarrier}_{M,T}$ exactly when
the following BEDC rows are all present:
\begin{enumerate}
  \item $\mathsf{UnaryHistory}(d)$ records the form degree.
  \item $\mathcal{P}:\mathsf{ProbeBundle}$ is the visible finite spine of tangent
  probe names, and each bundled probe row is carried by the tangent-source
  surface exposed by $\NameCert_{\ManifoldUp}$.
  \item $\tau$ is a $\TensorProductUp$ multilinear row over the bundled
  tangent probes, with continuation ledgers $\Cont$ recording the ordered
  tensor concatenations used by the row.
  \item $\sigma$ is the scalar endpoint produced from that tensor row,
  classified by the scalar classifier retained by the manifold and tensor
  sources.
  \item $\alpha$ is the visible permutation ledger: for every displayed
  adjacent swap of bundled probe rows, the transported scalar endpoint is
  classified with the signed endpoint demanded by antisymmetry.
  \item $\lambda$ records the source names $M$ and $T$, the degree row, the
  probe bundle, the tensor row, the scalar endpoint row, and the antisymmetry
  rows as the only accepted provenance for the form row.
\end{enumerate}
\end{definition}

\begin{definition}[Differential-form BHist form classifier]
\label{def:diffform-bhist-form-classifier}
For carried form rows
$$
\omega=(d,\mathcal{P},\tau,\sigma,\alpha,\lambda)
$$
and
$$
\omega'=(d',\mathcal{P}',\tau',\sigma',\alpha',\lambda'),
$$
the classifier $\omega\sim_{\DiffFormUp}\omega'$ holds when the degree rows
are $\hsame$, the scalar endpoints agree pointwise under the retained scalar
classifier for every matched multilinear probe row, and the antisymmetry and
tensor-source ledgers match as visible BHist rows:
$$
\hsame(d,d')
$$
$$
\text{and for every matched bundled multilinear row, }
\sigma\sim_R\sigma'
$$
$$
\text{and }
\hsame(\alpha,\alpha')\wedge\hsame(\tau,\tau')
\wedge\hsame(\lambda,\lambda').
$$
Here $\sim_R$ is the scalar classifier already present in the upstream scalar
surface; the differential-form classifier introduces no classifier outside
$\hsame$ and the displayed source classifiers.
\end{definition}

\section{The certificate}
\label{sec:diffform-certificate}

\begin{theorem}[Differential-form carrier hsame transport]
\label{thm:diffform-carrier-hsame-transport}
Let $\omega$ be a row accepted by
$\mathsf{DiffFormBHistCarrier}_{M,T}$, and let $\omega'$ be obtained by
replacing its six displayed histories with $\hsame$-equal histories while
keeping the same $\NameCert_{\ManifoldUp}$ and $\NameCert_{\TensorProductUp}$
sources. Then $\omega'$ is also accepted by
$\mathsf{DiffFormBHistCarrier}_{M,T}$.
\end{theorem}
\begin{proof}
Unpack the carrier row of \autoref{def:diffform-bhist-form-carrier}. The
degree condition is transported by $\hsame$ for $\mathsf{UnaryHistory}$.

The probe bundle is a closed $\mathsf{ProbeBundle}$ spine, so its bundled
tangent rows are transported pointwise through the displayed manifold source
classifier. The tensor row is then transported through the
$\TensorProductUp$ classifier, retaining the same $\Cont$ ledgers after
source, probe, and endpoint histories are replaced by $\hsame$-equal rows.

It remains to transport the scalar and antisymmetry rows. The scalar endpoint
uses the retained scalar classifier, and its pointwise transport follows from
classifier stability in the source surface. Each antisymmetry row is a visible
permutation ledger over the same probe bundle; transporting the two scalar
endpoints of the swap row gives the required signed comparison. Repacking
these transported fields gives the carried row for $\omega'$.
\end{proof}

\begin{theorem}[Differential-form classifier stability]
\label{thm:diffform-form-classifier-stability}
For rows carried by $\mathsf{DiffFormBHistCarrier}_{M,T}$, the classifier of
\autoref{def:diffform-bhist-form-classifier} is stable under carrier
transport: if $\omega\sim_{\DiffFormUp}\eta$ and both rows are transported to
$\omega'$ and $\eta'$ by componentwise $\hsame$ comparisons, then
$\omega'\sim_{\DiffFormUp}\eta'$.
\end{theorem}
\begin{proof}
The degree comparison is the transitive composite of the transport comparison
for $\omega$, the original degree comparison, and the symmetric transport
comparison for $\eta$.

For each bundled multilinear probe row, use the tensor-source classifier to
identify the matched tensor rows after transport. The scalar endpoints are
then compared by the scalar classifier stability already retained in the
upstream scalar surface.

The antisymmetry and source-ledger clauses are pure $\hsame$ clauses on
visible BHist rows. Composing their original comparisons with the two
transport comparisons gives the required ledger matches for the transported
rows.
\end{proof}

\begin{theorem}[Differential-form wedge ledger coverage]
\label{thm:diffform-wedge-ledger-coverage}
A wedge row $\omega\wedge_{\DiffFormUp}\eta$ is admitted by
$\mathsf{DiffFormBHistCarrier}_{M,T}$ only when its ledger visibly contains a
$\TensorProductUp$ concatenation row joining the tensor row of $\omega$ to the
tensor row of $\eta$. When that concatenation ledger is present, carrier and
classifier stability of the wedge row follow by transporting the two input
form rows through the tensor classifier and then through the retained scalar
classifier.
\end{theorem}
\begin{proof}
Read the wedge candidate through the carrier definition. The only provenance
accepted for a form row is its displayed ledger, and the tensor component of a
wedge must therefore be the visible tensor concatenation recorded there. If no
such $\TensorProductUp$ continuation row appears, the candidate has no accepted
carrier witness.

Assume the concatenation row is present. Transport the left and right form
rows by \autoref{thm:diffform-carrier-hsame-transport}. The
$\TensorProductUp$ source classifies the concatenated tensor endpoint after
the two input tensor rows have been transported, and the scalar endpoint is
classified by applying the retained scalar classifier to the transported
multilinear endpoints.

The antisymmetry ledger of the wedge is the concatenation of the two input
antisymmetry ledgers together with the displayed cross-permutation rows. Each
row is a visible BHist row, so $\hsame$ transport preserves it. These fields
reassemble into the carried wedge row and into the classifier comparison for
the transported wedge endpoints.
\end{proof}

\begin{definition}[Differential-form wedge associator ledger]
\label{def:diffform-wedge-associator-ledger}
For carried form rows $\omega$, $\eta$, and $\theta$ in
$\mathsf{DiffFormBHistCarrier}_{M,T}$, the visible wedge associator ledger is
the BHist package recording four rows: the $\TensorProductUp$ concatenation
spine for $(\omega\wedge_{\DiffFormUp}\eta)\wedge_{\DiffFormUp}\theta$, the
$\TensorProductUp$ concatenation spine for
$\omega\wedge_{\DiffFormUp}(\eta\wedge_{\DiffFormUp}\theta)$, the $\Cont$
associativity comparison between those tensor spines, and the common scalar
endpoint classifier obtained by transporting through the retained scalar
classifier. It also records the permutation-ledger comparison required to
identify the two antisymmetry ledgers after the tensor associativity
comparison has been applied.
\end{definition}

\begin{theorem}[Differential-form wedge associator closure]
\label{thm:diffform-wedge-associator-closure}
Let $\omega$, $\eta$, and $\theta$ be carried
$\mathsf{DiffFormBHistCarrier}_{M,T}$ rows. If the two binary wedge ledgers
needed for $(\omega\wedge_{\DiffFormUp}\eta)\wedge_{\DiffFormUp}\theta$ and
the two binary wedge ledgers needed for
$\omega\wedge_{\DiffFormUp}(\eta\wedge_{\DiffFormUp}\theta)$ are visible, then
the associator ledger of
\autoref{def:diffform-wedge-associator-ledger} carries both parenthesized
wedge rows and classifies them as equal under $\sim_{\DiffFormUp}$.
\end{theorem}
\begin{proof}
Apply \autoref{thm:diffform-wedge-ledger-coverage} to
$\omega\wedge_{\DiffFormUp}\eta$ and then to the wedge with $\theta$. This
constructs the left-parenthesized carrier row and its scalar endpoint
classifier from the two visible binary ledgers.

Apply the same theorem to $\eta\wedge_{\DiffFormUp}\theta$ and then to the
wedge with $\omega$. This constructs the right-parenthesized carrier row and
its scalar endpoint classifier from the other two visible binary ledgers.

The two tensor endpoints are compared by the $\Cont$ associativity row in the
visible $\TensorProductUp$ concatenation ledger. Transport the scalar endpoint
classifier across that comparison and the retained scalar classifier from
\autoref{def:diffform-bhist-form-classifier}.

It remains to identify the antisymmetry ledgers. The associator ledger carries
the required permutation comparison, and each permutation row is a visible
BHist row. Composing those rows by $\hsame$ transitivity gives the
$\DiffFormUp$ classifier equality between the two parenthesized wedge rows.
\end{proof}

\begin{theorem}[Differential-form exterior-algebra obligation surface]
\label{thm:diffform-exterior-algebra-obligation-surface}
The $\DiffFormUp$ exterior-algebra surface available to $\DeRhamUp$,
$\SymplecticUp$, and $\ContactUp$ consists of carrier transport, classifier
stability, binary wedge ledger coverage, and the wedge associator ledger. This
surface does not assert $d^2=0$, closedness, nondegeneracy, or contact
nonintegrability.
\end{theorem}
\begin{proof}
Carrier transport and classifier stability are supplied by
\autoref{thm:diffform-carrier-hsame-transport} and
\autoref{thm:diffform-form-classifier-stability}. Binary wedge admission and
transport are exactly \autoref{thm:diffform-wedge-ledger-coverage}.

Associativity of the exposed exterior product is not an unrecorded algebraic
law. It is the visible associator ledger of
\autoref{def:diffform-wedge-associator-ledger}, closed under the carrier and
classifier by \autoref{thm:diffform-wedge-associator-closure}.

Combining these rows with the NameCert surface of
\autoref{thm:diffform-namecert-obligation-surface} and the downstream scope
of \autoref{thm:diffform-exterior-derivative-downstream-scope} gives exactly
the surface available to de Rham, symplectic, and contact consumers. The
operator identity $d^2=0$ and the geometric conditions named in the statement
are not rows of this $\DiffFormUp$ surface, so they remain obligations of the
downstream certificates that use it.
\end{proof}

\begin{theorem}[Differential-form NameCert obligation surface]
\label{thm:diffform-namecert-obligation-surface}
The paper-side $\NameCert_{\DiffFormUp}$ obligation surface consists exactly
of the carrier of \autoref{def:diffform-bhist-form-carrier}, the classifier
of \autoref{def:diffform-bhist-form-classifier}, carrier transport,
classifier stability, wedge ledger coverage, the wedge associator ledger, the
antisymmetry permutation ledger, and the downstream scope row of
\autoref{thm:diffform-exterior-derivative-downstream-scope}. It asserts no
standard bridge beyond these BEDC rows.
\end{theorem}
\begin{proof}
The carrier and classifier fields are the two displayed definitions. Carrier
transport is \autoref{thm:diffform-carrier-hsame-transport}, and classifier
stability is \autoref{thm:diffform-form-classifier-stability}.

The multiplicative operation exposed by the surface is the wedge row, whose
admission and stability are exactly the visible tensor-concatenation statement
of \autoref{thm:diffform-wedge-ledger-coverage}. Its associativity obligation
is the visible ledger of \autoref{def:diffform-wedge-associator-ledger} and
the closure statement of \autoref{thm:diffform-wedge-associator-closure}.
Antisymmetry is not hidden in the wedge operation: it is the explicit
permutation ledger retained in every carrier row.

The remaining exported row is the downstream scope for exterior derivative
consumers. Since all fields are stated with $BHist$, $\hsame$, $\Cont$,
$\mathsf{ProbeBundle}$, $\NameCert_{\ManifoldUp}$, and
$\NameCert_{\TensorProductUp}$, this is a BEDC obligation surface and not a
claim of equivalence with a standard differential-form object.
\end{proof}

\begin{theorem}[Differential-form exterior-derivative downstream scope]
\label{thm:diffform-exterior-derivative-downstream-scope}
Downstream exterior-derivative chapters may use a $\DiffFormUp$ row only
through its degree row, its carried $\mathsf{ProbeBundle}$ of tangent probes,
its multilinear $\TensorProductUp$ tensor row, its scalar endpoint classifier,
its antisymmetry ledger, binary wedge ledger coverage, and the wedge
associator ledger. In particular, the DiffForm surface supplies the carrier,
classifier, and exterior-algebra substrate for
$\DeRhamUp$, $\SymplecticUp$, and $\ContactUp$, but it does not by itself
assert $d^2=0$, closedness, nondegeneracy, or contact nonintegrability.
\end{theorem}
\begin{proof}
Inspect the exported fields of
\autoref{thm:diffform-namecert-obligation-surface}. Each accessible component
is one of the displayed carrier, classifier, stability, wedge, or ledger rows.

The exterior derivative needs the degree row and the multilinear scalar
endpoint surface in order to state degree raising and scalar endpoint
transport. Those rows are available from the carrier and classifier, while
their stability is supplied by the two transport theorems.

The identities $d^2=0$, closedness, nondegeneracy, and contact
nonintegrability require additional operator or geometric rows. They are not
fields of the $\DiffFormUp$ surface, so downstream chapters must state and
certify them in their own $\NameCert$ surfaces.
\end{proof}
