\chapter{A Concrete Naming Certificate for $\AdeleUp$}
\label{ch:concrete-instances-adele-namecert}

The adele interface $\AdeleUp$ packages the restricted product of $\NameCert_{\RealUp}$ and $\NameCert_{\PadicUp}$ over all primes $p$. It depends on $\NameCert_{\PadicUp}$ (\autoref{ch:concrete-instances-padic-namecert}) and supplies the global object on which automorphic and L-function certificates are built.

\section{Carrier definition}
\label{sec:adele-carrier}

\begin{definition}[Adele history carrier]
\label{def:adele-history-carrier}
For a history $h$, the predicate $\mathsf{AdeleHistoryCarrier}(h)$ holds when
there are histories $r,p,e,s$ such that $r$ is a constant real history,
$\mathsf{PadicPrimeScale}(p,e,s)$ holds, and
$$
\hsame(h,\mathsf{append}(r,s)).
$$
\end{definition}
\leandef{BEDC.Derived.AdeleUp.AdeleHistoryCarrier}

\begin{theorem}[Adele history carrier nonempty]
\label{thm:adele-history-carrier-not-empty}
If $\mathsf{AdeleHistoryCarrier}(h)$ holds, then $h$ is not internally the same
as $\emp$.
\end{theorem}
\begin{proof}
Unpack the carrier into its constant real endpoint, $p$-adic scale endpoint,
and append witness. If $h\hsame\emp$, the append witness identifies the
appended real and scale endpoints with $\emp$, so append-empty exactness makes
the real endpoint empty. The constant real endpoint is one-sealed, and empty
cannot be internally the same as a one-sealed history.
\end{proof}
\leanchecked{BEDC.Derived.AdeleUp.AdeleHistoryCarrier\_not\_empty}

\begin{theorem}[Adele real-padic e1 real nonempty scale result package]
\label{thm:adele-real-padic-e1-real-nonempty-scale-result-package}
For histories $r,p,e,s:\mathsf{BHist}$, if $r$ is a rational-history carrier
and $\mathsf{PadicPrimeScale}(p,\mathsf{e1}(e),s)$ holds, then the real
component $\mathsf{e1}(r)$ is a real constant-history carrier and the padic
scale result is not history-same as $\emp$:
$$
\mathsf{RatHistoryCarrier}(r)\land
\mathsf{PadicPrimeScale}(p,\mathsf{e1}(e),s)
\Longrightarrow
\mathsf{RealConstantHistoryCarrier}(\mathsf{e1}(r))\land
(\hsame(s,\emp)\to\bot).
$$
The real component is certified by the $\RealUp$ $e1$ readback theorem from
the rational carrier. The padic component uses the empty-result iff for
$\PadicUp$ scale: an empty result would force
$\hsame(\mathsf{e1}(e),\emp)$, which is excluded by the BEDC history
constructor separation.
\end{theorem}
\begin{proof}
Apply the real constant-history $e1$ iff to the rational carrier witness for
$r$. For the padic result, assume $\hsame(s,\emp)$. The padic empty-result iff
turns this into $\hsame(\mathsf{e1}(e),\emp)$, contradicting the BEDC
$e1$-versus-empty separation theorem.
\end{proof}
\leanchecked{BEDC.Derived.AdeleUp.AdeleRealPadic\_e1\_real\_nonempty\_scale\_result\_package}

\begin{theorem}[Adele carrier from the first-prime unit scale]
\label{thm:adele-history-carrier-first-prime-unit-scale}
For any rational-history carrier tail \(r\), the history formed by appending
the one-sealed real component to the concrete first-prime unit scale result is
an adele-history carrier:
$$
\mathsf{RatHistoryCarrier}(r)\Longrightarrow
\mathsf{AdeleHistoryCarrier}(
  \mathsf{append}(\Eone(r),\Eone(\Eone(\emp)))).
$$
\end{theorem}
\begin{proof}
The real component is a constant real history by the \(e1\) readback theorem
for rational carriers. The padic component is the concrete first-prime unit
scale from \autoref{thm:padic-prime-scale-first-prime-unit-exponent-result}.
The final carrier equality is reflexive history sameness for the displayed
append history.
\end{proof}
\leanchecked{BEDC.Derived.AdeleUp.AdeleHistoryCarrier\_first\_prime\_unit\_scale}
