\chapter{A Concrete Naming Certificate for $\FilterUp$}
\label{ch:concrete-instances-filter-namecert}

The filter interface $\FilterUp$ packages a non-empty upward-closed family of subsets of a carrier closed under finite intersections. It supplies the structural substrate for convergence and limit notions independent of metric data, and is the constructive analogue of mathlib-style \texttt{Filter} for the BEDC topology and analysis layers.

\section{Carrier definition}
\label{sec:filter-carrier}

\begin{theorem}[Principal suffix unary intersection closure]
\label{thm:filter-principal-suffix-unary-intersection-closed}
Let $base,left,right,meet,leftPoint,rightPoint,meetPoint:\mathsf{BHist}$.
If $base$, $left$, and $right$ are unary histories,
$$
\ContR(left,right,meet),\qquad
\ContR(base,left,leftPoint),\qquad
\ContR(base,right,rightPoint),\qquad
\ContR(base,meet,meetPoint),
$$
then $meetPoint$ is unary, and it is simultaneously displayed as the
continuation of $leftPoint$ by $right$ and of $rightPoint$ by $left$.
\end{theorem}
\begin{proof}
Unary closure first gives unary $meet$ from the continuation
$\ContR(left,right,meet)$, and then gives unary $meetPoint$ from
$\ContR(base,meet,meetPoint)$. The left projection follows by associativity of
history append. The right projection uses unary commutativity of $left$ and
$right$, followed by the same associativity comparison.
\end{proof}
\leanchecked{BEDC.Derived.FilterUp.FilterPrincipalSuffix\_unary\_intersection\_closed}

\begin{theorem}[Principal suffix unary commuting square]
\label{thm:filter-principal-suffix-unary-commuting-square}
Let $base,left,right,leftPoint,rightPoint,lrPoint,rlPoint:\mathsf{BHist}$.
If $left$ and $right$ are unary histories and
$$
\ContR(base,left,leftPoint),\qquad
\ContR(base,right,rightPoint),\qquad
\ContR(leftPoint,right,lrPoint),\qquad
\ContR(rightPoint,left,rlPoint),
$$
then $lrPoint\hsame rlPoint$.
\end{theorem}
\begin{proof}
Unfold the four continuations as history appends. The left path displays
$\mathsf{append}(\mathsf{append}(base,left),right)$, and the right path
displays $\mathsf{append}(\mathsf{append}(base,right),left)$. Associativity
moves both paths to the common $base$ prefix, while unary commutativity of
$left$ and $right$ identifies the two suffix orders.
\end{proof}
\leanchecked{BEDC.Derived.FilterUp.FilterPrincipalSuffix\_unary\_commuting\_square}

\begin{theorem}[Principal suffix unary intersection meet commutes]
\label{thm:filter-principal-suffix-unary-intersection-meet-commutes}
Let $left,right,meet,commute:\mathsf{BHist}$. If $left$ and $right$ are unary
histories and
$$
\ContR(left,right,meet),\qquad
\ContR(right,left,commute),
$$
then $meet\hsame commute$.
\end{theorem}
\begin{proof}
Apply the unary commuting square with empty base. The left and right base legs
are the canonical left-unit continuations, so the two displayed intersection
results are precisely the two suffix orders, and the square identifies them by
$\hsame$.
\end{proof}
\leanchecked{BEDC.Derived.FilterUp.FilterPrincipalSuffix\_unary\_intersection\_meet\_commutes}

\begin{theorem}[Principal suffix unary commuting square base determinism]
\label{thm:filter-principal-suffix-unary-commuting-square-base-deterministic}
Let $base,base',left,right,leftPoint,rightPoint,meetPoint:\mathsf{BHist}$.
If $left$ and $right$ are unary histories and
$$
\ContR(base,left,leftPoint),\qquad
\ContR(base',right,rightPoint),\qquad
\ContR(leftPoint,right,meetPoint),\qquad
\ContR(rightPoint,left,meetPoint),
$$
then $base\hsame base'$.
\end{theorem}
\begin{proof}
Unfold the two paths to the common meet. The first path gives
$\mathsf{append}(\mathsf{append}(base,left),right)$ and the second gives
$\mathsf{append}(\mathsf{append}(base',right),left)$. Associativity moves both
to suffix form, unary commutativity aligns the common suffix
$\mathsf{append}(left,right)$, and right cancellation of that suffix identifies
the two bases.
\end{proof}
\leanchecked{BEDC.Derived.FilterUp.FilterPrincipalSuffix\_unary\_commuting\_square\_base\_deterministic}

\begin{theorem}[Principal suffix commuted meet closure]
\label{thm:filter-principal-suffix-commuted-meet-closed}
Let $base,left,right,meet,meetPoint:\mathsf{BHist}$. If $left$ and $right$
are unary histories and
$$
\ContR(left,right,meet),\qquad
\ContR(base,meet,meetPoint)
$$
hold, then the same $meetPoint$ is displayed by the commuted suffix:
$$
\ContR(base,\mathsf{append}(right,left),meetPoint).
$$
\end{theorem}
\begin{proof}
The first continuation identifies $meet$ with $\mathsf{append}(left,right)$.
Unary commutativity of $left$ and $right$ transports this suffix under the
fixed prefix $base$, giving the displayed continuation by
$\mathsf{append}(right,left)$.
\end{proof}
\leanchecked{BEDC.Derived.FilterUp.FilterPrincipalSuffix\_unary\_intersection\_commuted\_meet\_closed}

\begin{theorem}[Principal suffix commuted result determinism]
\label{thm:filter-principal-suffix-unary-intersection-commuted-result-deterministic}
Let $base,left,right,meet,meetPoint,displayed:\mathsf{BHist}$. If $left$ and
$right$ are unary histories and
$$
\ContR(left,right,meet),\qquad
\ContR(base,meet,meetPoint),\qquad
\ContR(base,\mathsf{append}(right,left),displayed),
$$
then $meetPoint\hsame displayed$.
\end{theorem}
\begin{proof}
The commuted meet closure theorem displays $meetPoint$ as a continuation of
$base$ by $\mathsf{append}(right,left)$. Continuation determinism compares this
canonical commuted display with the given $displayed$ history.
\end{proof}
\leanchecked{BEDC.Derived.FilterUp.FilterPrincipalSuffix\_unary\_intersection\_commuted\_result\_deterministic}

\begin{theorem}[Principal suffix unary intersection result determinism]
\label{thm:filter-principal-suffix-unary-intersection-result-deterministic}
In the situation of
\autoref{thm:filter-principal-suffix-unary-intersection-closed}, if another
history $displayed$ satisfies $\ContR(rightPoint,left,displayed)$, then
$meetPoint \hsame displayed$.
\end{theorem}
\begin{proof}
The closure theorem supplies the canonical continuation
$\ContR(rightPoint,left,meetPoint)$. Continuation determinism compares this
canonical display with the given displayed result.
\end{proof}
\leanchecked{BEDC.Derived.FilterUp.FilterPrincipalSuffix\_unary\_intersection\_result\_deterministic}

\begin{theorem}[Principal suffix unary intersection zero-result exclusion]
\label{thm:filter-principal-suffix-unary-intersection-zero-result-absurd}
Let $base,left,right,meet,leftPoint,rightPoint,z:\mathsf{BHist}$. If
$base$, $left$, and $right$ are unary histories and
$$
\ContR(left,right,meet),\qquad
\ContR(base,left,leftPoint),\qquad
\ContR(base,right,rightPoint),\qquad
\ContR(base,meet,\Ezero(z)),
$$
then $\bot$.
\end{theorem}
\begin{proof}
Unary closure transports through the concrete continuation
$\ContR(left,right,meet)$, so $meet$ is unary. Applying unary closure again to
$\ContR(base,meet,\Ezero(z))$ would make the zero-headed result unary, which
contradicts the unary no-zero-extension theorem.
\end{proof}
\leanchecked{BEDC.Derived.FilterUp.FilterPrincipalSuffix\_unary\_intersection\_zero\_result\_absurd}

\begin{theorem}[Principal suffix unary intersection e1-result cases]
\label{thm:filter-principal-suffix-unary-intersection-e1-result-cases}
Let $base,left,right,meet,leftPoint,rightPoint,z:\mathsf{BHist}$. If
$base$, $left$, and $right$ are unary histories and
$$
\ContR(left,right,meet),\qquad
\ContR(base,left,leftPoint),\qquad
\ContR(base,right,rightPoint),\qquad
\ContR(base,meet,\Eone(z)),
$$
then either $meet=\emp$ and $base\hsame\Eone(z)$, or there is a unary
$meetTail$ such that $meet=\Eone(meetTail)$ and
$\ContR(base,meetTail,z)$.
\end{theorem}
\begin{proof}
Unary closure along $\ContR(left,right,meet)$ first gives unary $meet$. The
kernel unary e1-result case theorem applied to
$\ContR(base,meet,\Eone(z))$ then splits the result into the empty-meet case
and the non-empty unary-tail case.
\end{proof}
\leanchecked{BEDC.Derived.FilterUp.FilterPrincipalSuffix\_unary\_intersection\_e1\_result\_cases}

\begin{theorem}[Principal suffix base points empty exactness]
\label{thm:filter-principal-suffix-base-points-empty-exact}
Let $base,left,right,leftPoint,rightPoint:\mathsf{BHist}$. If
$$
\ContR(base,left,leftPoint),\qquad
\ContR(base,right,rightPoint),
$$
and both displayed points satisfy
$$
leftPoint\hsame\emp,\qquad rightPoint\hsame\emp,
$$
then $base\hsame\emp$, $left\hsame\emp$, and $right\hsame\emp$.
\end{theorem}
\begin{proof}
Transport the two displayed endpoints to $\emp$. Empty-result inversion for the
first continuation identifies both $base$ and $left$ with $\emp$; the same
inversion for the second continuation identifies $right$ with $\emp$.
\end{proof}
\leanchecked{BEDC.Derived.FilterUp.FilterPrincipalSuffix\_base\_points\_empty\_exact}

\begin{theorem}[Principal suffix commuting square empty exactness]
\label{thm:filter-principal-suffix-unary-commuting-square-empty-exact}
Let $base,left,right,leftPoint,rightPoint,lrPoint,rlPoint:\mathsf{BHist}$.
If the two displayed continuations
$$
\ContR(base,left,leftPoint),\qquad
\ContR(base,right,rightPoint)
$$
are extended around the square by
$$
\ContR(leftPoint,right,lrPoint),\qquad
\ContR(rightPoint,left,rlPoint),
$$
and both square endpoints satisfy
$$
lrPoint\hsame\emp,\qquad rlPoint\hsame\emp,
$$
then $base\hsame\emp$, $left\hsame\emp$, and $right\hsame\emp$.
\end{theorem}
\begin{proof}
Transport the two square endpoints to $\emp$. Empty-result inversion of
$\ContR(leftPoint,right,lrPoint)$ identifies $leftPoint$ with $\emp$, and
empty-result inversion of $\ContR(rightPoint,left,rlPoint)$ identifies
$rightPoint$ with $\emp$. Apply the base-points empty exactness theorem to the
two base continuations and these two transported endpoint equalities.
\end{proof}
\leanchecked{BEDC.Derived.FilterUp.FilterPrincipalSuffix\_unary\_commuting\_square\_empty\_exact}

\begin{theorem}[Principal suffix intersection empty point exactness]
\label{thm:filter-principal-suffix-intersection-empty-point-exact}
Let $base,left,right,meet,meetPoint:\mathsf{BHist}$. If
$$
\ContR(left,right,meet),\qquad
\ContR(base,meet,meetPoint),
$$
and the displayed intersection point satisfies $meetPoint\hsame\emp$, then
all four histories collapse to the empty history:
$$
base\hsame\emp,\qquad meet\hsame\emp,\qquad
left\hsame\emp,\qquad right\hsame\emp.
$$
\end{theorem}
\begin{proof}
Transport $meetPoint$ to $\emp$ and apply empty-result inversion to
$\ContR(base,meet,meetPoint)$. This identifies both $base$ and $meet$ with
$\emp$. Substituting the empty meet into $\ContR(left,right,meet)$ and
applying the same empty-result inversion identifies $left$ and $right$ with
$\emp$.
\end{proof}
\leanchecked{BEDC.Derived.FilterUp.FilterPrincipalSuffix\_intersection\_empty\_point\_exact}

\begin{theorem}[Principal suffix intersection empty point forces base points empty]
\label{thm:filter-principal-suffix-intersection-empty-point-base-points-empty}
Let $base,left,right,meet,leftPoint,rightPoint,meetPoint:\mathsf{BHist}$.
If $meet$ is the continuation of $left$ by $right$, the base continuations
produce $leftPoint$, $rightPoint$, and $meetPoint$, and
$meetPoint\hsame\emp$, then both base points are empty:
$$
leftPoint\hsame\emp,\qquad rightPoint\hsame\emp.
$$
\end{theorem}
\begin{proof}
Intersection empty-point exactness identifies the shared base, the left
suffix, and the right suffix with $\emp$. Empty-append introduction applied
to $\ContR(base,left,leftPoint)$ gives the empty left base point, and the
same argument applied to $\ContR(base,right,rightPoint)$ gives the empty
right base point.
\end{proof}
\leanchecked{BEDC.Derived.FilterUp.FilterPrincipalSuffix\_intersection\_empty\_point\_base\_points\_empty}

\section{The certificate}

\begin{theorem}[Principal empty filter carrier semantic certificate]
\label{thm:filter-principal-empty-carrier-semantic-namecert}
The BEDC unary carrier restricted to the empty history,
$$
\mathsf{UnaryHistory}(h)\wedge h\hsame \emp,
$$
has a semantic naming certificate with source, pattern, and ledger predicates
all equal to this principal empty carrier, and with classifier $\hsame$.
\end{theorem}
\begin{proof}
The witness is the empty history, which is unary and classifies to itself.
Reflexivity, symmetry, and transitivity are the finite-kernel laws for
$\hsame$. If $h$ lies in the principal empty carrier and $h\hsame k$, unary
transport moves the unary carrier from $h$ to $k$, while symmetry followed by
transitivity sends the displayed comparison $h\hsame\emp$ to
$k\hsame\emp$.
\end{proof}
\leanchecked{BEDC.Derived.FilterUp.FilterPrincipalEmptyCarrier\_semanticNameCert}

\begin{theorem}[Principal visible carrier semantic certificate absurdity]
\label{thm:filter-principal-visible-carrier-semantic-namecert-absurd}
For every $tail:\mathsf{BHist}$, there is no semantic naming certificate whose
source, pattern, and ledger predicates all require a history to be unary, to
classify to $\emp$, and to classify to $\Eone(tail)$:
$$
\mathsf{SemanticNameCert}
(\{h\mid \mathsf{UnaryHistory}(h)\land h\hsame\emp\land h\hsame\Eone(tail)\},
 \{h\mid \mathsf{UnaryHistory}(h)\land h\hsame\emp\land h\hsame\Eone(tail)\},
 \{h\mid \mathsf{UnaryHistory}(h)\land h\hsame\emp\land h\hsame\Eone(tail)\},
 \hsame)
\Longrightarrow \bot .
$$
\end{theorem}
\begin{proof}
The semantic certificate supplies a ledger witness for the displayed carrier.
Its two classification comparisons compose by symmetry and transitivity to
give $\emp\hsame\Eone(tail)$, which is excluded by BEDC history constructor
separation.
\end{proof}
\leanchecked{BEDC.Derived.FilterUp.FilterPrincipalVisibleCarrier\_semanticNameCert\_absurd}

\begin{theorem}[ContinuousMap image of a principal suffix filter]
\label{thm:continuousmap-image-principal-suffix-filter}
Let $base,map,target,modulus,cert,distance,left,right,meet,leftPoint,rightPoint,meetPoint:\mathsf{BHist}$.
Assume
$$
\begin{array}{l}
\mathsf{ContinuousMapCarrier}(base,map,target,modulus,cert,distance),\\
\UnaryHistory(left),\quad \UnaryHistory(right),\\
\ContR(left,right,meet),\quad
\ContR(base,left,leftPoint),\quad
\ContR(base,right,rightPoint),\quad
\ContR(base,meet,meetPoint).
\end{array}
$$
Then there exist histories
$targetLeft,targetRight,targetMeet,imageLeft,imageRight,imageMeet:\mathsf{BHist}$
for which
$$
\begin{array}{l}
\ContR(target,left,targetLeft)\land
\ContR(target,right,targetRight)\land
\ContR(target,meet,targetMeet),\\
\UnaryHistory(targetMeet)\land
\ContR(targetLeft,right,targetMeet)\land
\ContR(targetRight,left,targetMeet),\\
\ContR(leftPoint,map,imageLeft)\land
\ContR(rightPoint,map,imageRight)\land
\ContR(meetPoint,map,imageMeet),\\
\hsame(targetLeft,imageLeft)\land
\hsame(targetRight,imageRight)\land
\hsame(targetMeet,imageMeet).
\end{array}
$$
Moreover, these target-side witnesses determine the intersection projections,
the target commuting square, and the graph-image displays:
$$
\begin{array}{l}
\forall z:\mathsf{BHist}.\;
\ContR(targetLeft,right,z)\Longrightarrow \hsame(targetMeet,z),\\
\forall z:\mathsf{BHist}.\;
\ContR(targetRight,left,z)\Longrightarrow \hsame(targetMeet,z),\\
\forall lr,rl:\mathsf{BHist}.\;
\ContR(targetLeft,right,lr)\Longrightarrow
\ContR(targetRight,left,rl)\Longrightarrow \hsame(lr,rl),\\
\forall z:\mathsf{BHist}.\;
\ContR(leftPoint,map,z)\Longrightarrow \hsame(targetLeft,z),\\
\forall z:\mathsf{BHist}.\;
\ContR(rightPoint,map,z)\Longrightarrow \hsame(targetRight,z),\\
\forall z:\mathsf{BHist}.\;
\ContR(meetPoint,map,z)\Longrightarrow \hsame(targetMeet,z).
\end{array}
$$
\end{theorem}
\begin{proof}
By \autoref{def:continuousmap-carrier}, the continuous-map carrier contains a
continuous-function carrier. Applying
\autoref{thm:continuous-function-carrier-graph-modulus-cont-readback} to that
component gives unary histories for $base$, $map$, and $target$, together with
the graph continuation $\ContR(base,map,target)$. The source hypotheses and
unary continuation closure give unary evidence for $meet$.

Choose $targetLeft$, $targetRight$, and $targetMeet$ from the generated
continuations of $target$ by $left$, $right$, and $meet$. Choose $imageLeft$,
$imageRight$, and $imageMeet$ from the generated continuations of $leftPoint$,
$rightPoint$, and $meetPoint$ by $map$.

Apply \autoref{thm:filter-principal-suffix-unary-intersection-closed} with
base $target$. Its inputs are the unary target history, the unary suffixes
$left$ and $right$, the source meet ledger $\ContR(left,right,meet)$, and the
three chosen target continuations. This gives $\UnaryHistory(targetMeet)$ and
the two projection continuations
$\ContR(targetLeft,right,targetMeet)$ and
$\ContR(targetRight,left,targetMeet)$.

For the image readback of $left$, apply
\autoref{thm:filter-principal-suffix-unary-commuting-square} to the square with
base $base$, suffixes $left$ and $map$, base legs
$\ContR(base,left,leftPoint)$ and $\ContR(base,map,target)$, and upper legs
$\ContR(leftPoint,map,imageLeft)$ and $\ContR(target,left,targetLeft)$. The
result identifies $imageLeft$ with $targetLeft$, and symmetry gives the stated
orientation. The same argument with $right$ gives
$\hsame(targetRight,imageRight)$. The meet case uses the unary meet evidence,
the base leg $\ContR(base,meet,meetPoint)$, and the same graph leg to obtain
$\hsame(targetMeet,imageMeet)$.

The two projection-determinacy clauses follow by comparing the canonical target
projection continuations with any displayed projection through
\autoref{thm:continuation-determinacy-up-to-history-sameness}. The universal
target commuting square is exactly
\autoref{thm:filter-principal-suffix-unary-commuting-square} applied at base
$target$ to arbitrary displayed upper legs. Finally, each graph-image
determinacy clause compares the chosen image continuation with an arbitrary
image continuation by continuation determinacy, and then composes that
comparison with the corresponding target-image equality using the equivalence
laws for $\hsame$ from \autoref{thm:history-sameness-equivalence}.
\end{proof}
\leanchecked{BEDC.Derived.FilterUp.FilterPrincipalSuffix\_continuousMap\_image\_point\_determinacy}
\leanchecked{BEDC.Derived.FilterUp.FilterPrincipalSuffix\_continuousMap\_image\_package}

\begin{theorem}[ContinuousMap image determinacy for principal suffix points]
\label{thm:continuousmap-image-principal-suffix-point-determinacy}
Let $base,map,target,modulus,cert,distance,suffix,sourcePoint,targetPoint,
target_0,target_1,modulus_0,modulus_1,cert_0,cert_1,distance_0,distance_1:
\mathsf{BHist}$. Assume the base continuous-map row
$$
\mathsf{ContinuousMapCarrier}(base,map,target,modulus,cert,distance),
$$
the principal suffix source and target displays
$$
\UnaryHistory(suffix),\qquad
\ContR(base,suffix,sourcePoint),\qquad
\ContR(target,suffix,targetPoint),
$$
and two displayed image rows over the same source point and the same graph
history:
$$
\begin{array}{l}
\mathsf{ContinuousMapCarrier}(sourcePoint,map,target_0,modulus_0,cert_0,distance_0),\\
\mathsf{ContinuousMapCarrier}(sourcePoint,map,target_1,modulus_1,cert_1,distance_1).
\end{array}
$$
Then the two displayed target histories are determined, and both read back to
the target-side principal suffix point:
$$
\hsame(target_0,target_1)\land
\hsame(targetPoint,target_0)\land
\hsame(targetPoint,target_1).
$$
\end{theorem}
\leanchecked{BEDC.Derived.FilterUp.ContinuousMap\_image\_principal\_suffix\_point\_determinacy}
\begin{proof}
Project the continuous-function component from the base carrier using
\autoref{def:continuousmap-carrier}. Applying
\autoref{thm:continuous-function-carrier-graph-modulus-cont-readback} to that
component gives the graph continuation $\ContR(base,map,target)$ and unary
membership of the graph history $map$.

Project the continuous-function components from the two displayed image rows in
the same way. Their graph fields give
$$
\ContR(sourcePoint,map,target_0)
\qquad\hbox{and}\qquad
\ContR(sourcePoint,map,target_1).
$$

Apply \autoref{thm:filter-principal-suffix-unary-commuting-square} with base
$base$, left suffix $suffix$, right suffix $map$, left point $sourcePoint$, and
right point $target$. The first displayed image row supplies the upper-left path
and $\ContR(target,suffix,targetPoint)$ supplies the upper-right path, so the
square gives $\hsame(target_0,targetPoint)$. The same square applied to the
second displayed image row gives $\hsame(target_1,targetPoint)$.

Symmetry and transitivity of $\hsame$ from
\autoref{thm:history-sameness-equivalence} orient these two comparisons as
$\hsame(targetPoint,target_0)$ and $\hsame(targetPoint,target_1)$ and compose
them to obtain $\hsame(target_0,target_1)$. Pairing the three comparisons gives
the displayed conclusion.
\end{proof}
\leanchecked{BEDC.Derived.FilterUp.ContinuousMapCarrier\_principal\_suffix\_point\_determinacy}

\begin{theorem}[Principal suffix upward closure]
\label{thm:filter-principal-suffix-upward-closed}
Let $base,suffix,point,extra,extended:\mathsf{BHist}$. If $base$, $suffix$,
and $extra$ are unary histories and
$$
\ContR(base,suffix,point),\qquad
\ContR(point,extra,extended),
$$
then $point$ and $extended$ are unary histories, and there is a unary history
$combined:\mathsf{BHist}$ such that
$$
\ContR(suffix,extra,combined)\qquad\text{and}\qquad
\ContR(base,combined,extended).
$$
Consequently, the principal suffix family generated by $base$ is closed under
unary continuation extensions of its points.
\end{theorem}
\begin{proof}
Unary continuation closure gives unary $point$ from the base display and unary
$extended$ from the point extension. Put
$combined=\mathsf{append}(suffix,extra)$. The continuation rules of
\autoref{def:continuation-relation} give $\ContR(suffix,extra,combined)$, and
unary continuation closure gives unary $combined$.

Let $direct=\mathsf{append}(base,combined)$, with the canonical continuation
$\ContR(base,combined,direct)$. Applying
\autoref{thm:continuation-associativity-up-to-hsame} to the four ledgers
$$
\ContR(base,suffix,point),\qquad
\ContR(point,extra,extended),\qquad
\ContR(suffix,extra,combined),\qquad
\ContR(base,combined,direct)
$$
gives $extended\hsame direct$. The reverse direction of
\autoref{thm:cont-result-hsame-iff} for $\ContR(base,combined,direct)$
transports the direct composite result to the displayed endpoint $extended$,
yielding $\ContR(base,combined,extended)$.
\end{proof}
