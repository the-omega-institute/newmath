\chapter{A Concrete Naming Certificate for $\FilterUp$}
\label{ch:concrete-instances-filter-namecert}

The filter interface $\FilterUp$ packages a non-empty upward-closed family of subsets of a carrier closed under finite intersections. It supplies the structural substrate for convergence and limit notions independent of metric data, and is the constructive analogue of mathlib-style \texttt{Filter} for the BEDC topology and analysis layers.

\section{Carrier definition}
\label{sec:filter-carrier}

\begin{theorem}[Principal suffix unary intersection closure]
\label{thm:filter-principal-suffix-unary-intersection-closed}
Let $base,left,right,meet,leftPoint,rightPoint,meetPoint:\mathsf{BHist}$.
If $base$, $left$, and $right$ are unary histories,
$$
\ContR(left,right,meet),\qquad
\ContR(base,left,leftPoint),\qquad
\ContR(base,right,rightPoint),\qquad
\ContR(base,meet,meetPoint),
$$
then $meetPoint$ is unary, and it is simultaneously displayed as the
continuation of $leftPoint$ by $right$ and of $rightPoint$ by $left$.
\end{theorem}
\begin{proof}
Unary closure first gives unary $meet$ from the continuation
$\ContR(left,right,meet)$, and then gives unary $meetPoint$ from
$\ContR(base,meet,meetPoint)$. The left projection follows by associativity of
history append. The right projection uses unary commutativity of $left$ and
$right$, followed by the same associativity comparison.
\end{proof}
\leanchecked{BEDC.Derived.FilterUp.FilterPrincipalSuffix\_unary\_intersection\_closed}

\begin{theorem}[Principal suffix unary commuting square]
\label{thm:filter-principal-suffix-unary-commuting-square}
Let $base,left,right,leftPoint,rightPoint,lrPoint,rlPoint:\mathsf{BHist}$.
If $left$ and $right$ are unary histories and
$$
\ContR(base,left,leftPoint),\qquad
\ContR(base,right,rightPoint),\qquad
\ContR(leftPoint,right,lrPoint),\qquad
\ContR(rightPoint,left,rlPoint),
$$
then $lrPoint\hsame rlPoint$.
\end{theorem}
\begin{proof}
Unfold the four continuations as history appends. The left path displays
$\mathsf{append}(\mathsf{append}(base,left),right)$, and the right path
displays $\mathsf{append}(\mathsf{append}(base,right),left)$. Associativity
moves both paths to the common $base$ prefix, while unary commutativity of
$left$ and $right$ identifies the two suffix orders.
\end{proof}
\leanchecked{BEDC.Derived.FilterUp.FilterPrincipalSuffix\_unary\_commuting\_square}

\begin{theorem}[Principal suffix unary commuting square base determinism]
\label{thm:filter-principal-suffix-unary-commuting-square-base-deterministic}
Let $base,base',left,right,leftPoint,rightPoint,meetPoint:\mathsf{BHist}$.
If $left$ and $right$ are unary histories and
$$
\ContR(base,left,leftPoint),\qquad
\ContR(base',right,rightPoint),\qquad
\ContR(leftPoint,right,meetPoint),\qquad
\ContR(rightPoint,left,meetPoint),
$$
then $base\hsame base'$.
\end{theorem}
\begin{proof}
Unfold the two paths to the common meet. The first path gives
$\mathsf{append}(\mathsf{append}(base,left),right)$ and the second gives
$\mathsf{append}(\mathsf{append}(base',right),left)$. Associativity moves both
to suffix form, unary commutativity aligns the common suffix
$\mathsf{append}(left,right)$, and right cancellation of that suffix identifies
the two bases.
\end{proof}
\leanchecked{BEDC.Derived.FilterUp.FilterPrincipalSuffix\_unary\_commuting\_square\_base\_deterministic}

\begin{theorem}[Principal suffix commuted meet closure]
\label{thm:filter-principal-suffix-commuted-meet-closed}
Let $base,left,right,meet,meetPoint:\mathsf{BHist}$. If $left$ and $right$
are unary histories and
$$
\ContR(left,right,meet),\qquad
\ContR(base,meet,meetPoint)
$$
hold, then the same $meetPoint$ is displayed by the commuted suffix:
$$
\ContR(base,\mathsf{append}(right,left),meetPoint).
$$
\end{theorem}
\begin{proof}
The first continuation identifies $meet$ with $\mathsf{append}(left,right)$.
Unary commutativity of $left$ and $right$ transports this suffix under the
fixed prefix $base$, giving the displayed continuation by
$\mathsf{append}(right,left)$.
\end{proof}
\leanchecked{BEDC.Derived.FilterUp.FilterPrincipalSuffix\_unary\_intersection\_commuted\_meet\_closed}

\begin{theorem}[Principal suffix unary intersection result determinism]
\label{thm:filter-principal-suffix-unary-intersection-result-deterministic}
In the situation of
\autoref{thm:filter-principal-suffix-unary-intersection-closed}, if another
history $displayed$ satisfies $\ContR(rightPoint,left,displayed)$, then
$meetPoint \hsame displayed$.
\end{theorem}
\begin{proof}
The closure theorem supplies the canonical continuation
$\ContR(rightPoint,left,meetPoint)$. Continuation determinism compares this
canonical display with the given displayed result.
\end{proof}
\leanchecked{BEDC.Derived.FilterUp.FilterPrincipalSuffix\_unary\_intersection\_result\_deterministic}

\begin{theorem}[Principal suffix unary intersection zero-result exclusion]
\label{thm:filter-principal-suffix-unary-intersection-zero-result-absurd}
Let $base,left,right,meet,leftPoint,rightPoint,z:\mathsf{BHist}$. If
$base$, $left$, and $right$ are unary histories and
$$
\ContR(left,right,meet),\qquad
\ContR(base,left,leftPoint),\qquad
\ContR(base,right,rightPoint),\qquad
\ContR(base,meet,\Ezero(z)),
$$
then $\bot$.
\end{theorem}
\begin{proof}
Unary closure transports through the concrete continuation
$\ContR(left,right,meet)$, so $meet$ is unary. Applying unary closure again to
$\ContR(base,meet,\Ezero(z))$ would make the zero-headed result unary, which
contradicts the unary no-zero-extension theorem.
\end{proof}
\leanchecked{BEDC.Derived.FilterUp.FilterPrincipalSuffix\_unary\_intersection\_zero\_result\_absurd}

\begin{theorem}[Principal suffix base points empty exactness]
\label{thm:filter-principal-suffix-base-points-empty-exact}
Let $base,left,right,leftPoint,rightPoint:\mathsf{BHist}$. If
$$
\ContR(base,left,leftPoint),\qquad
\ContR(base,right,rightPoint),
$$
and both displayed points satisfy
$$
leftPoint\hsame\emp,\qquad rightPoint\hsame\emp,
$$
then $base\hsame\emp$, $left\hsame\emp$, and $right\hsame\emp$.
\end{theorem}
\begin{proof}
Transport the two displayed endpoints to $\emp$. Empty-result inversion for the
first continuation identifies both $base$ and $left$ with $\emp$; the same
inversion for the second continuation identifies $right$ with $\emp$.
\end{proof}
\leanchecked{BEDC.Derived.FilterUp.FilterPrincipalSuffix\_base\_points\_empty\_exact}

\begin{theorem}[Principal suffix intersection empty point exactness]
\label{thm:filter-principal-suffix-intersection-empty-point-exact}
Let $base,left,right,meet,meetPoint:\mathsf{BHist}$. If
$$
\ContR(left,right,meet),\qquad
\ContR(base,meet,meetPoint),
$$
and the displayed intersection point satisfies $meetPoint\hsame\emp$, then
all four histories collapse to the empty history:
$$
base\hsame\emp,\qquad meet\hsame\emp,\qquad
left\hsame\emp,\qquad right\hsame\emp.
$$
\end{theorem}
\begin{proof}
Transport $meetPoint$ to $\emp$ and apply empty-result inversion to
$\ContR(base,meet,meetPoint)$. This identifies both $base$ and $meet$ with
$\emp$. Substituting the empty meet into $\ContR(left,right,meet)$ and
applying the same empty-result inversion identifies $left$ and $right$ with
$\emp$.
\end{proof}
\leanchecked{BEDC.Derived.FilterUp.FilterPrincipalSuffix\_intersection\_empty\_point\_exact}

\section{The certificate}
\label{sec:filter-certificate}

\begin{theorem}[Principal empty filter carrier semantic certificate]
\label{thm:filter-principal-empty-carrier-semantic-namecert}
The BEDC unary carrier restricted to the empty history,
$$
\mathsf{UnaryHistory}(h)\wedge h\hsame \emp,
$$
has a semantic naming certificate with source, pattern, and ledger predicates
all equal to this principal empty carrier, and with classifier $\hsame$.
\end{theorem}
\begin{proof}
The witness is the empty history, which is unary and classifies to itself.
Reflexivity, symmetry, and transitivity are the finite-kernel laws for
$\hsame$. If $h$ lies in the principal empty carrier and $h\hsame k$, unary
transport moves the unary carrier from $h$ to $k$, while symmetry followed by
transitivity sends the displayed comparison $h\hsame\emp$ to
$k\hsame\emp$.
\end{proof}
\leanchecked{BEDC.Derived.FilterUp.FilterPrincipalEmptyCarrier\_semanticNameCert}

\begin{theorem}[Principal visible carrier semantic certificate absurdity]
\label{thm:filter-principal-visible-carrier-semantic-namecert-absurd}
For every $tail:\mathsf{BHist}$, there is no semantic naming certificate whose
source, pattern, and ledger predicates all require a history to be unary, to
classify to $\emp$, and to classify to $\Eone(tail)$:
$$
\mathsf{SemanticNameCert}
(\{h\mid \mathsf{UnaryHistory}(h)\land h\hsame\emp\land h\hsame\Eone(tail)\},
 \{h\mid \mathsf{UnaryHistory}(h)\land h\hsame\emp\land h\hsame\Eone(tail)\},
 \{h\mid \mathsf{UnaryHistory}(h)\land h\hsame\emp\land h\hsame\Eone(tail)\},
 \hsame)
\Longrightarrow \bot .
$$
\end{theorem}
\begin{proof}
The semantic certificate supplies a ledger witness for the displayed carrier.
Its two classification comparisons compose by symmetry and transitivity to
give $\emp\hsame\Eone(tail)$, which is excluded by BEDC history constructor
separation.
\end{proof}
\leanchecked{BEDC.Derived.FilterUp.FilterPrincipalVisibleCarrier\_semanticNameCert\_absurd}
