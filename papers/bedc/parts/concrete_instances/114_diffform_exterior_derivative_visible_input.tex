\begin{definition}[Differential-form exterior-derivative visible input package]
\label{def:diffform-exterior-derivative-visible-input-package}
Let
$$
\omega=(d,\mathcal{P},\tau,\sigma,\alpha,\lambda)
$$
be a source row carried by $\mathsf{DiffFormBHistCarrier}_{M,T}$, and let
$$
D\omega=(d^+,\mathcal{P}^+,\tau^+,\sigma^+,\alpha^+,\lambda^+)
$$
be a carried output candidate. The package
$\mathsf{DiffFormExteriorInput}_{M,T}(\omega,D\omega)$ is accepted exactly
when its visible provenance consists of the two displayed $\DiffFormUp$
carrier rows, the unary degree rows $\mathsf{UnaryHistory}(d)$ and
$\mathsf{UnaryHistory}(d^+)$, the degree-raising continuation row
$$
\Cont(d,\Eone(\emp),d^+),
$$
the transported $\mathsf{ProbeBundle}$ rows connecting $\mathcal{P}$ with
$\mathcal{P}^+$, the retained $\TensorProductUp$ tensor row for the
multilinear output candidate, the retained scalar classifier comparison
between the source and output scalar endpoints, and the operator-owned source
ledger recording that these rows are the only inputs supplied to the
exterior-derivative construction. The package asserts no row for $d^2=0$,
closedness, nondegeneracy, or contact nonintegrability.
\end{definition}

\begin{theorem}[Differential-form exterior-derivative degree successor ledger]
\label{thm:diffform-exterior-derivative-degree-successor-ledger}
Every package accepted by
$\mathsf{DiffFormExteriorInput}_{M,T}(\omega,D\omega)$ exposes the source
degree row $\mathsf{UnaryHistory}(d)$, the output degree row
$\mathsf{UnaryHistory}(d^+)$, and the visible successor continuation row
$$
\Cont(d,\Eone(\emp),d^+).
$$
These are the only degree rows made available to a downstream
exterior-derivative consumer by the package.
\end{theorem}
\begin{proof}
Unpack the accepted package of
\autoref{def:diffform-exterior-derivative-visible-input-package}. Its degree
fields are listed as the two unary-history rows and the displayed
$\Cont(d,\Eone(\emp),d^+)$ row.

The source and output are carried rows of
\autoref{def:diffform-bhist-form-carrier}, so their degree components are
visible BHist carrier entries rather than hidden operator data. Applying the
source-scope boundary of \autoref{thm:diffform-source-scope} shows that no
other degree provenance may be imported through this package.
\end{proof}

\begin{theorem}[Differential-form exterior-derivative input classifier stability]
\label{thm:diffform-exterior-derivative-input-classifier-stability}
Let $\mathsf{DiffFormExteriorInput}_{M,T}(\omega,D\omega)$ be accepted. If
the source and output carrier rows are transported componentwise by $\hsame$
to $\omega'$ and $D\omega'$, then
$\mathsf{DiffFormExteriorInput}_{M,T}(\omega',D\omega')$ is accepted. Under
the same transport, the retained scalar endpoint comparison between the
source and output scalar endpoints remains the scalar comparison carried by
the transported package.
\end{theorem}
\begin{proof}
First transport the two carrier rows. By
\autoref{thm:diffform-carrier-hsame-transport}, the transported source
$\omega'$ and transported output $D\omega'$ remain accepted
$\DiffFormUp$ carrier rows.

Next transport the classifier data. Apply
\autoref{thm:diffform-form-classifier-stability} to the componentwise
$\hsame$ comparisons of the two carrier rows. The degree rows, probe bundles,
tensor rows, antisymmetry ledgers, and operator-owned source ledger remain the
same visible rows up to $\hsame$, and the successor continuation row
$\Cont(d,\Eone(\emp),d^+)$ transports to the corresponding continuation row
for the transported degree endpoints.

Finally, apply \autoref{thm:diffform-scalar-endpoint-transport}. The retained
scalar endpoint comparison in the accepted package is preserved by the same
componentwise transport, so the transported rows reassemble as
$\mathsf{DiffFormExteriorInput}_{M,T}(\omega',D\omega')$ with its scalar
comparison still visible.
\end{proof}
