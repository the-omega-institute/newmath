\chapter{A Concrete Naming Certificate for \(\ComplexSeriesUp\)}
\label{ch:concrete-instances-complex-series-namecert}

This chapter gives a concrete naming-certificate construction for convergent infinite complex sums \(\sum_{n=0}^{\infty} c_n\), where \(c\) is a unary-indexed family of complex histories. The certificate is distinct from \(\ConvergenceRadiusUp\) (\autoref{ch:concrete-instances-convergence-radius-namecert}): there is no formal variable, only a complex sequence and its tail behaviour. The construction sits above \(\ComplexLimitUp\) (\autoref{ch:concrete-instances-complex-limit-namecert}) and reuses the regular Cauchy machinery underlying \(\RealUp\) (\autoref{ch:concrete-instances-real-namecert}). Absolute convergence and conditional convergence are licensed by separate predicates with explicit unary moduli; the comparison, ratio, and root tests are recorded as constructive lemmas; and the Cauchy product of two absolutely convergent series is constructed term-wise with explicit modulus pasting. The certificate is fully constructive and uses no \(\mathsf{Classical.choice}\), no \(\mathsf{Quot.sound}\), and no \(\mathsf{propext}\), so its Lean targets fall under the project's axiom-purity gate.

\section{Term sequences and partial sums}

\begin{definition}[Complex term sequence]
\label{def:complex-term-seq}
A complex term sequence is a unary-indexed family \(c : \Unary \to \ComplexUp\)-history, i.e.\ a function from unary histories to complex carrier histories.
\end{definition}
\leandef{BEDC.Derived.ComplexSeriesUp.ComplexTermSeqCarrier}

\begin{lemma}[Complex term sequence carrier transport]
\label{lem:complex-term-seq-carrier-hsame-transport}
If two complex term families $c,d:\mathsf{BHist}\to\mathsf{BHist}$ are
pointwise related by $\hsame$ on every unary index, then the
$\mathsf{ComplexTermSeqCarrier}$ predicate transports from $c$ to $d$.
\end{lemma}
\begin{proof}
Fix a unary index $n$. The carrier for $c(n)$ expands to a complex-pair witness:
rational real and imaginary component carriers together with a continuation
ledger whose result is $c(n)$. Transport that continuation result along the
pointwise comparison $c(n)\hsame d(n)$ while keeping the same rational
component witnesses. The transported pair witness is exactly the complex carrier
for $d(n)$.
\end{proof}
\leanchecked{BEDC.Derived.ComplexSeriesUp.ComplexTermSeqCarrier\_hsame\_transport}

\begin{definition}[Complex partial sum]
\label{def:complex-part-sum}
For a term sequence \(c\), the complex partial-sum predicate \(\mathsf{ComplexPartSum}(c, n, S_n)\) is the closed inductive predicate generated by:
\begin{enumerate}
  \item \(\mathsf{ComplexPartSum}(c, \emp, \emp_{\ComplexUp})\) where \(\emp_{\ComplexUp}\) is the zero complex history (both real components empty);
  \item if \(\mathsf{ComplexPartSum}(c, n, S_n)\) and the complex addition relation gives \(S_n + c_n = S_{n+1}\), then \(\mathsf{ComplexPartSum}(c, \Eone(n), S_{n+1})\).
\end{enumerate}
\end{definition}
\leandef{BEDC.Derived.ComplexSeriesUp.ComplexPartSum}

\begin{lemma}[Complex partial-sum index cases]
\label{lem:complex-part-sum-index-cases}
Every closed witness of $\mathsf{ComplexPartSum}(c,n,S)$ either has the zero
index $n=\emp$ and identifies $S$ with the chosen zero history, or has a
successor index $n=\Eone(m)$ together with a predecessor partial sum and the
continuation step from that predecessor to $S$.
\end{lemma}
\begin{proof}
Invert the closed partial-sum witness. The zero constructor gives the empty
index and the base history. The step constructor exposes the predecessor index,
predecessor sum, and continuation payload directly.
\end{proof}
\leanchecked{BEDC.Derived.ComplexSeriesUp.ComplexPartSum\_index\_cases}

\begin{lemma}[Partial sums are complex histories]
\label{lem:complex-part-sum-well-defined}
For every term sequence \(c\) and unary \(n\), any two histories obtained from
the closed partial-sum recursion for \(c\) at \(n\) agree up to \(\hsame\).
\end{lemma}
\begin{proof}
Induct on the first partial-sum derivation. The zero case is immediate. In the
successor case, the second derivation has the same successor shape; the
induction hypothesis identifies the previous sums, and determinacy of the BEDC
continuation relation identifies the two successor sums.
\end{proof}
\leanchecked{BEDC.Derived.ComplexSeriesUp.ComplexPartSum\_deterministic}

\begin{lemma}[Complex partial-sum visible predecessor decomposition is deterministic]
\label{thm:complex-partsum-visible-previous-decomposition-deterministic}
If two displayed successor decompositions of the same complex partial-sum
history have the forms
$$
n=\mathsf{BHist.e1}(m)\land
\mathsf{ComplexPartSum}(zero,c,m,P)\land \mathsf{Cont}(P,c(m),S)
$$
and
$$
n=\mathsf{BHist.e1}(m')\land
\mathsf{ComplexPartSum}(zero,c,m',P')\land \mathsf{Cont}(P',c(m'),S),
$$
then $m=m'$ and $P\hsame P'$.
\end{lemma}
\begin{proof}
Invert both visible successor equalities. Constructor injectivity identifies
the exposed predecessor indices, and closed complex partial-sum determinacy
identifies the predecessor results under $\hsame$.
\end{proof}
\leanchecked{BEDC.Derived.ComplexSeriesUp.ComplexPartSum\_visible\_previous\_decomposition\_deterministic}

\begin{lemma}[Complex partial sums exist uniquely]
\label{lem:complex-part-sum-exists-unique}
For every chosen zero history, term family \(c\), and unary index \(n\), there
is a history \(S\) satisfying \(\mathsf{ComplexPartSum}(c,n,S)\), and every
other closed partial sum at \(n\) is \(\hsame\)-equal to \(S\).
\end{lemma}
\begin{proof}
Unary induction on the index constructs the zero partial sum at \(\emp\). At a
successor index, the induction hypothesis gives the predecessor sum and the
BEDC continuation relation appends the current term. The closed partial-sum
determinacy theorem identifies every competing derivation at the same index.
\end{proof}
\leanchecked{BEDC.Derived.ComplexSeriesUp.ComplexPartSum\_exists\_unique}

\begin{lemma}[Partial sums respect pointwise term equivalence]
\label{lem:complex-part-sum-term-hsame-deterministic}
If two term families \(c,d:\Hist\to\Hist\) are pointwise related by \(\hsame\)
on every unary index, and \(S,T\) are closed partial sums for \(c,d\) at the
same index \(n\), then \(S\hsame T\).
\end{lemma}
\begin{proof}
Induct on the first closed partial-sum witness. The base case forces the second
witness to be the zero constructor. In the successor case, the second witness
has the same successor index; the induction hypothesis identifies the
predecessor sums, closedness of the first witness supplies unaryness of the
predecessor index, and the pointwise term comparison identifies the terms. The
BEDC continuation relation then identifies the successor sums.
\end{proof}
\leanchecked{BEDC.Derived.ComplexSeriesUp.ComplexPartSum\_term\_hsame\_deterministic}

\begin{lemma}[Complex partial-sum successor term hsame determinacy]
\label{lem:complex-part-sum-successor-term-hsame-deterministic}
If two chosen zero histories agree under $\hsame$, and two term families agree
pointwise under $\hsame$, then a successor partial-sum step computed from the
first family has the same result as any closed successor partial-sum derivation
for the second family at the corresponding $\Eone(n)$ index.
\end{lemma}
\begin{proof}
First prove cross-family determinacy for partial sums by induction on the first
closed derivation. The zero constructor uses the zero-history comparison. The
successor constructor inverts the matching successor derivation on the other
side, applies the induction hypothesis to the predecessor sums, and then uses
determinacy of $\ContR$ with the pointwise term comparison. Applying this
cross-family result to the predecessor sums of the displayed successor witness
and the closed successor derivation identifies the final results.
\end{proof}
\leanchecked{BEDC.Derived.ComplexSeriesUp.ComplexPartSum\_successor\_term\_hsame\_deterministic}

\begin{lemma}[Partial sums respect pointwise history sameness]
\label{lem:complex-part-sum-pointwise-hsame-deterministic}
For two term families $c,d:\mathsf{BHist}\to\mathsf{BHist}$ with
$\hsame(c(m),d(m))$ at every unary index $m$, and for base histories
$z,z'$ with $z\hsame z'$, closed partial-sum derivations for $c$ and $d$ at
the same unary index have $\hsame$ results.
\end{lemma}
\begin{proof}
Induct on the first closed partial-sum derivation. The zero constructor uses
the base comparison $z\hsame z'$. In the successor case, invert the
second derivation, recover the predecessor unary witness, apply the induction
hypothesis to the predecessor sums, and use continuation determinacy with the
pointwise comparison of the current terms.
\end{proof}
\leanchecked{BEDC.Derived.ComplexSeriesUp.ComplexPartSum\_pointwise\_hsame\_deterministic}

\begin{lemma}[Complex partial-sum result nonempty from nonempty terms]
\label{lem:complex-part-sum-result-nonempty-of-nonempty-terms}
If the chosen zero history is apart from $\emp$, and every unary-indexed term
$c(m)$ is apart from $\emp$, then no closed
$\mathsf{ComplexPartSum}(zero,c,n,S)$ witness can have $S\hsame\emp$.
\end{lemma}
\begin{proof}
Induct on the closed partial-sum witness. The zero case is the supplied
apartness of the chosen zero history. In the successor case, transport the
final empty equality across the displayed continuation
$\Cont(S,c(m),T)$, invert the empty result, and use the unary index extracted
from the predecessor partial-sum chain to contradict the apartness of $c(m)$.
\end{proof}
\leanchecked{BEDC.Derived.ComplexSeriesUp.ComplexPartSum\_result\_nonempty\_of\_nonempty\_terms}

\begin{theorem}[Complex partial-sum results are unary]
\label{thm:complex-part-sum-result-unary}
If the chosen zero history is unary and each term history is unary on unary
indices, then every closed complex partial-sum witness has unary result.
\end{theorem}
\begin{proof}
Induct on the closed complex partial-sum witness. The base is the supplied
unary zero history. In the successor case, an inner induction on the
predecessor partial-sum witness supplies unaryness of the predecessor index,
the term hypothesis supplies unaryness of the next term, and unary continuation
closure gives the successor result.
\end{proof}
\leanchecked{BEDC.Derived.ComplexSeriesUp.ComplexPartSum\_result\_unary}

\section{Absolute convergence}

\begin{definition}[Modulus partial sum]
\label{def:abs-part-sum}
For a complex term sequence \(c\), the modulus partial sum \(\mathsf{AbsPartSum}(c, n, M_n)\) is the real-history value \(M_n = \sum_{k < n} |c_k|\) computed by an explicit unary recursion taking complex modulus pointwise and summing in \(\RealUp\).
\end{definition}
\leandef{BEDC.Derived.ComplexSeriesUp.ComplexAbsPartSum}

\begin{lemma}[Absolute partial-sum index cases]
\label{lem:complex-abs-part-sum-index-cases}
Every closed witness of $\mathsf{ComplexAbsPartSum}(\mathsf{modulus},n,M)$
either has the zero index $n=\emp$ and identifies $M$ with the chosen zero
history, or has a successor index $n=\Eone(m)$ together with a predecessor
absolute partial sum and the continuation step from that predecessor to $M$.
\end{lemma}
\begin{proof}
Invert the closed absolute partial-sum witness. The zero constructor gives the
empty index and base history; the step constructor exposes the predecessor
sum, predecessor index, and continuation payload directly.
\end{proof}
\leanchecked{BEDC.Derived.ComplexSeriesUp.ComplexAbsPartSum\_index\_cases}

\begin{lemma}[Absolute partial-sum successor result is deterministic]
\label{lem:complex-abs-part-sum-successor-result-deterministic}
If an absolute partial-sum derivation reaches $M$ at unary index $n$, and the
next modulus continuation sends $M$ and $\mathsf{modulus}(n)$ to $T$, then every
closed derivation of the successor absolute partial sum at $\Eone(n)$ has result
$\hsame$ to $T$.
\end{lemma}
\begin{proof}
Induct on the previous absolute partial-sum derivation to obtain determinacy at
each index. In the successor case, the other derivation has the same successor
shape; the induction hypothesis identifies the previous modulus sums, and
determinacy of $\mathsf{Cont}$ identifies the successor result.
\end{proof}
\leanchecked{BEDC.Derived.ComplexSeriesUp.ComplexAbsPartSum\_successor\_result\_deterministic}

\begin{lemma}[Absolute partial-sum unary index is deterministic]
\label{lem:complex-abs-part-sum-unary-index-deterministic}
If $n$ is a unary history, then any two closed derivations of
$\mathsf{ComplexAbsPartSum}(\mathsf{zero},\mathsf{modulus},n,-)$ have
$\hsame$ results.
\end{lemma}
\begin{proof}
Induct on the first absolute partial-sum derivation. The empty index case
forces the second derivation to be the same base constructor. In the successor
case, unary inversion exposes the predecessor index, the induction hypothesis
identifies the previous modulus sums, and determinacy of $\mathsf{Cont}$
identifies the successor results.
\end{proof}
\leanchecked{BEDC.Derived.ComplexSeriesUp.ComplexAbsPartSum\_unary\_index\_deterministic}

\begin{lemma}[Absolute partial sums exist uniquely]
\label{lem:complex-abs-part-sum-exists-unique}
For every chosen zero history, modulus family, and unary index \(n\), there is
a history \(M\) satisfying \(\mathsf{ComplexAbsPartSum}(zero,\mathsf{modulus},n,M)\),
and every other closed absolute partial sum at \(n\) is \(\hsame\)-equal to
\(M\).
\end{lemma}
\begin{proof}
Unary induction on \(n\) constructs the empty-index absolute partial sum from
the zero constructor. At a successor index, the induction hypothesis supplies a
predecessor sum, and the canonical BEDC continuation appends the current
modulus history. Unary-index determinacy identifies every competing closed
absolute partial sum at the same index.
\end{proof}
\leanchecked{BEDC.Derived.ComplexSeriesUp.ComplexAbsPartSum\_exists\_unique}

\begin{theorem}[Absolute partial-sum results are unary]
\label{thm:complex-abs-part-sum-result-unary}
If the chosen zero history is unary and every modulus history
$\mathsf{modulus}(m)$ is unary whenever $m$ is unary, then every closed
$\mathsf{ComplexAbsPartSum}(zero,\mathsf{modulus},n,M)$ witness has unary
result $M$.
\end{theorem}
\begin{proof}
Induct on the absolute partial-sum witness. The base result is the supplied
unary zero history. For a successor step, an inner induction on the predecessor
absolute partial-sum witness gives unaryness of the predecessor index; the
modulus hypothesis makes the appended modulus unary, and unary continuation
closure gives the successor result.
\end{proof}
\leanchecked{BEDC.Derived.ComplexSeriesUp.ComplexAbsPartSum\_result\_unary}

\begin{lemma}[Absolute partial-sum result is nonempty from nonempty terms]
\label{lem:complex-abs-part-sum-result-nonempty-of-nonempty-terms}
If the chosen zero history is not history-same to \(\emp\), and every modulus
term \(\mathsf{modulus}(m)\) at a unary index is not history-same to \(\emp\),
then no closed \(\mathsf{ComplexAbsPartSum}(zero,\mathsf{modulus},n,M)\)
witness can have \(M\hsame\emp\).
\end{lemma}
\begin{proof}
Induct on the absolute partial-sum witness. The base constructor reduces to
the nonempty zero assumption. In the successor case, empty-result inversion for
the continuation into \(M\) forces the current modulus term to be empty. A
local induction on the predecessor partial-sum witness supplies unaryness of
that predecessor index, contradicting the pointwise nonempty modulus
hypothesis.
\end{proof}
\leanchecked{BEDC.Derived.ComplexSeriesUp.ComplexAbsPartSum\_result\_nonempty\_of\_nonempty\_terms}

\begin{theorem}[Absolute partial-sum empty result boundary]
\label{thm:complex-abs-part-sum-empty-result-boundary}
If a closed $\mathsf{ComplexAbsPartSum}(zero,\mathsf{modulus},n,M)$ witness
has result $M\hsame\emp$, then either the chosen zero history is
history-same to $\emp$, or some unary predecessor index $m$ has
$\mathsf{modulus}(m)\hsame\emp$.
\end{theorem}
\begin{proof}
Induct on the absolute partial-sum witness. The base constructor reads the
empty result directly as $zero\hsame\emp$. In the successor constructor,

modulus term, while a local index induction supplies unaryness of the
predecessor index.
\end{proof}
\leanchecked{BEDC.Derived.ComplexSeriesUp.ComplexAbsPartSum\_empty\_result\_boundary}

\begin{lemma}[Absolute partial-sum successor modulus hsame determinacy]
\label{lem:complex-abs-part-sum-successor-modulus-hsame-deterministic}
If two chosen zero histories agree under $\hsame$, and two modulus families
agree pointwise under $\hsame$, then a successor absolute partial-sum step

successor absolute partial-sum derivation for the second family at the
corresponding $\Eone(n)$ index.
\end{lemma}
\begin{proof}

determinacy of predecessor modulus sums. The zero constructor is handled by the
zero-history comparison. In the successor constructor, invert the matching
successor derivation, compare predecessor sums by induction, and compare the
two continuation results using the pointwise modulus comparison. Applying this
determinacy statement to the predecessor of the displayed successor step and
the closed successor derivation gives the required result comparison.
\end{proof}
\leanchecked{BEDC.Derived.ComplexSeriesUp.ComplexAbsPartSum\_successor\_modulus\_hsame\_deterministic}

For two modulus families $m,m':\mathsf{BHist}\to\mathsf{BHist}$ with pointwise

\begin{lemma}[Absolute partial sums transport across modulus equivalence]

If two zero histories are related by $\hsame$ and two modulus functions are

\begin{definition}[Absolute convergence]
\label{def:abs-conv}
A complex series \(c\) converges absolutely with bound \(M\) and modulus \(N : \Unary \to \Unary\), written \(\mathsf{AbsConv}(c, M, N)\), when:
\begin{enumerate}
  \item for every unary \(n\), \(\mathsf{AbsPartSum}(c, n, M_n)\) gives a real-history bound \(M_n \leq M\);
  \item for every unary precision \(k\) and unary \(n, m \geq N(k)\), \(|M_n - M_m| < 2^{-k}\).
\end{enumerate}
\end{definition}
\leandef{BEDC.Derived.ComplexSeriesUp.ComplexAbsConv}

\begin{lemma}[Absolute convergence implies convergence]
\label{lem:abs-conv-implies-conv}
If \(\mathsf{AbsConv}(c, M, N)\) holds, the complex partial sums \((S_n)_n\) form a regular complex Cauchy sequence with the same modulus \(N\), and hence have an explicit complex-history limit \(S\) by Cauchy completeness (\autoref{ch:concrete-instances-complex-limit-namecert}).
\end{lemma}
\leanchecked{BEDC.Derived.ComplexSeriesUp.ComplexAbsConv\_complexSeriesConv}
\leanchecked{BEDC.Derived.ComplexSeriesUp.ComplexAbsConv\_complexSeriesConv\_of\_pointwise\_hsame}
\begin{proof}[Sketch]
The triangle inequality on each segment of consecutive partial sums bounds \(|S_n - S_m|\) by \(|M_n - M_m|\). Apply the modulus from absolute convergence directly.
\end{proof}

\section{Convergence and uniqueness}

\begin{definition}[Complex series convergence]
\label{def:complex-series-conv}
\(\mathsf{ComplexSeriesConv}(c, S)\) holds when the complex partial-sum sequence \((S_n)_n\) is a regular complex Cauchy sequence (in the sense of \autoref{ch:concrete-instances-complex-limit-namecert}) with explicit limit \(S\).
\end{definition}
\leandef{BEDC.Derived.ComplexSeriesUp.ComplexSeriesConv}

\begin{theorem}[Sum is unique]
\label{thm:complex-series-sum-unique}
If \(\mathsf{ComplexSeriesConv}(c, S)\) and \(\mathsf{ComplexSeriesConv}(c, S')\), then \(\hsame\) holds on both real components of \(S\) and \(S'\).
\end{theorem}
\begin{proof}[Sketch]
Direct application of complex-limit uniqueness.
\end{proof}

\section{Comparison, ratio, and root tests}

\begin{theorem}[Comparison test]
\label{thm:comparison-test}
Let \(c\) be a complex term sequence and \(d\) a real-history term sequence with \(d_n \geq 0\) for all \(n\). If for every unary \(n\), the modulus comparison \(|c_n| \leq d_n\) holds (decidable on each unary index), and \(\sum_n d_n\) converges in \(\RealUp\) with explicit modulus \(N_d\), then \(\mathsf{AbsConv}(c, M, N_d)\) holds with the same modulus \(N_d\) and bound \(M = \sum d_n\).
\end{theorem}
\begin{proof}[Sketch]
The comparison hypothesis transports to absolute partial sums: \(M_n \leq \sum_{k < n} d_k\). The modulus from \(d\)-convergence then bounds the differences of \(M_n\).
\end{proof}

\begin{theorem}[Ratio test]
\label{thm:ratio-test}
If there exists a rational \(r\) with \(0 \leq r < 1\) and a unary \(N_0\) such that for all unary \(n \geq N_0\), \(|c_{n+1}| \leq r \cdot |c_n|\) (decidable comparison), then \(\mathsf{AbsConv}\) holds with bound \(M = |c_{N_0}| / (1 - r)\) and modulus \(N(k) = N_0 + \lceil k / \log_2(1/r) \rceil\) computed as an explicit unary function of the precision.
\end{theorem}
\begin{proof}[Sketch]
Geometric majorant: starting from \(N_0\), the tail is bounded by a geometric series with ratio \(r\). Compute the modulus from the geometric tail sum.
\end{proof}

\begin{lemma}[Root test]
\label{lem:root-test}
If there exists a rational \(r < 1\) and unary \(N_0\) such that \(|c_n|^{1/n} \leq r\) (where the root is computed by explicit unary approximation) for all unary \(n \geq N_0\), then \(\mathsf{AbsConv}\) holds with explicit geometric modulus.
\end{lemma}
\begin{proof}[Sketch]
The hypothesis gives \(|c_n| \leq r^n\); apply comparison with the geometric series.
\end{proof}

\section{Cauchy product}

\begin{definition}[Cauchy product]
\label{def:cauchy-product}
For complex term sequences \(a, b\), the Cauchy product \(c = a \star b\) is the term sequence \(c_n = \sum_{k = 0}^{n} a_k \cdot b_{n-k}\) defined by an explicit unary recursion using complex multiplication.
\end{definition}
\leandef{BEDC.Derived.ComplexSeriesUp.CauchyProduct}

\begin{theorem}[Cauchy product preserves absolute convergence]
\label{thm:cauchy-product-conv}
If \(\mathsf{AbsConv}(a, M_a, N_a)\) and \(\mathsf{AbsConv}(b, M_b, N_b)\) hold, then \(\mathsf{AbsConv}(a \star b, M_a \cdot M_b, N_{ab})\) holds with explicit modulus \(N_{ab}\) computed from \(N_a, N_b, M_a, M_b\). Moreover the sum of the Cauchy product equals the product of sums: \(\sum (a \star b) = (\sum a) \cdot (\sum b)\).
\end{theorem}
\leanchecked{BEDC.Derived.ComplexSeriesUp.CauchyProduct\_result\_unary}
\begin{proof}[Sketch]
Reorder partial sums into a triangular block structure. Absolute convergence justifies the reordering with explicit modulus pasting.
\end{proof}

\section{Closure laws}

\begin{lemma}[Linearity of convergence]
\label{lem:linearity-of-conv}
If \(\mathsf{ComplexSeriesConv}(a, S_a)\) and \(\mathsf{ComplexSeriesConv}(b, S_b)\), and \(\alpha, \beta\) are complex histories, then \(\mathsf{ComplexSeriesConv}(\alpha a + \beta b, \alpha S_a + \beta S_b)\) with explicit modulus pasting.
\end{lemma}
\leanchecked{BEDC.Derived.ComplexSeriesUp.ComplexPartSum\_pointwise\_append\_partials}
\begin{proof}[Sketch]
Pointwise scalar multiple and pointwise sum of regular sequences remain regular with combined modulus.
\end{proof}

\begin{lemma}[Reordering for absolute convergence]
\label{lem:reordering-abs-conv}
If \(\mathsf{AbsConv}(c, M, N)\) holds and \(\sigma : \Unary \to \Unary\) is an explicit unary bijection (a permutation of the indexing), then the reordered series \(c \circ \sigma\) is also absolutely convergent with the same bound and an explicit modulus computable from \(N\) and \(\sigma\).
\end{lemma}
\begin{proof}[Sketch]
Absolute convergence ensures that any reordering converges to the same limit with adjusted modulus.
\end{proof}

\begin{lemma}[Pointwise sum is a complex term sequence]
\label{lem:complex-term-seq-pointwise-addition}
Let $c,d:\Hist\to\Hist$ be complex term sequences in the sense of \autoref{def:complex-term-seq}, and define their pointwise sum by
$$
(c\oplus d)(m):=\mathsf{append}(c(m),d(m)),
$$
as in \autoref{def:cplx-pointwise-sum}. Then $c\oplus d$ is again a complex term sequence.
\end{lemma}
\leanchecked{BEDC.Derived.ComplexSeriesUp.ComplexTermSeqCarrier\_pointwise\_append}
\begin{proof}
Fix a unary index $m$. Name the two term histories by fresh history variables $C_m$ and $D_m$, with $C_m=c(m)$ and $D_m=d(m)$. The term-sequence hypotheses give complex-history carriers for $C_m$ and $D_m$. By \autoref{thm:complex-history-carrier-unary}, $D_m$ is unary. The canonical append relation supplies a history $A_m$ such that
$$
\Cont(C_m,D_m,A_m)
\quad\text{and}\quad
A_m\hsame\mathsf{append}(C_m,D_m).
$$
This continuation witness falls under complex carrier unary continuation closure from \autoref{thm:complex-history-carrier-unary-continuation-closed}, so the named pointwise result $(c\oplus d)(m)$ is a complex-history carrier. Since $m$ was arbitrary among unary indices, the pointwise-sum family is a complex term sequence.
\end{proof}
\leanchecked{BEDC.Derived.ComplexSeriesUp.ComplexTermSeqCarrier\_pointwise\_append\_components}

\begin{theorem}[Complex partial sums commute with pointwise addition]
\label{thm:complex-part-sum-pointwise-addition}
Let $c,d:\Hist\to\Hist$ be complex term sequences, let $(c\oplus d)(m):=\mathsf{append}(c(m),d(m))$, and let $z_c,z_d,z_\oplus:\Hist$ be unary base histories with
$$
z_\oplus\hsame\mathsf{append}(z_c,z_d).
$$
If $n$ is unary and
$$
\mathsf{ComplexPartSum}(z_c,c,n,S_c)
\quad\text{and}\quad
\mathsf{ComplexPartSum}(z_d,d,n,S_d)
$$
hold, then there is a history $S_\oplus$ such that
$$
\mathsf{ComplexPartSum}(z_\oplus,c\oplus d,n,S_\oplus)
\land
S_\oplus\hsame\mathsf{append}(S_c,S_d).
$$
Moreover every closed partial sum $T$ of $c\oplus d$ at the same index satisfies
$$
\mathsf{ComplexPartSum}(z_\oplus,c\oplus d,n,T)
\Longrightarrow
T\hsame\mathsf{append}(S_c,S_d).
$$
\end{theorem}
\leanchecked{BEDC.Derived.ComplexSeriesUp.ComplexPartSum\_pointwise\_append\_partials\_unique}
\begin{proof}
By \autoref{lem:complex-term-seq-pointwise-addition}, $c\oplus d$ is a complex term sequence. Prove the displayed existence statement by unary induction on $n$, uniformly in the two input partial-sum witnesses.

In the base case $n=\emp$, index-case inversion from \autoref{lem:complex-part-sum-index-cases} identifies $S_c$ with $z_c$ and $S_d$ with $z_d$. The zero constructor gives
$$
\mathsf{ComplexPartSum}(z_\oplus,c\oplus d,\emp,z_\oplus).
$$
The base comparison $z_\oplus\hsame\mathsf{append}(z_c,z_d)$, append congruence from \autoref{thm:external-append-hsame-congruence}, and the two endpoint identifications give $z_\oplus\hsame\mathsf{append}(S_c,S_d)$.

For the successor case $n=\Eone(m)$, invert the two input partial-sum witnesses with \autoref{lem:complex-part-sum-index-cases}. This exposes predecessor sums $P_c,P_d$, bound term histories $C_m,D_m$ with $C_m\hsame c(m)$ and $D_m\hsame d(m)$, and continuations
$$
\Cont(P_c,C_m,S_c)
\quad\text{and}\quad
\Cont(P_d,D_m,S_d).
$$
The induction hypothesis at $m$ gives $P_\oplus$ with
$$
\mathsf{ComplexPartSum}(z_\oplus,c\oplus d,m,P_\oplus)
\land
P_\oplus\hsame\mathsf{append}(P_c,P_d).
$$
Let $A_m$ be the bound pointwise-sum term history supplied by the canonical append evidence
$$
\Cont(C_m,D_m,A_m)
\quad\text{and}\quad
A_m\hsame\mathsf{append}(C_m,D_m).
$$
Let $S_\oplus$ be the fresh successor history supplied by
$$
\Cont(P_\oplus,A_m,S_\oplus).
$$
The step constructor for \autoref{def:complex-part-sum} gives the closed partial sum for $c\oplus d$ at $\Eone(m)$.

It remains to compare this result with the sum of the two displayed successor results. The term-sequence hypotheses and \autoref{thm:complex-history-carrier-unary} make $C_m$ and $D_m$ unary, while \autoref{thm:complex-part-sum-result-unary} makes $P_c$ and $P_d$ unary. Associativity of unary continuation from \autoref{lem:unary-continuation-associativity} and commutativity from \autoref{thm:unary-cont-comm} give the middle-four comparison
$$
\mathsf{append}(\mathsf{append}(P_c,P_d),\mathsf{append}(C_m,D_m))
\hsame
\mathsf{append}(\mathsf{append}(P_c,C_m),\mathsf{append}(P_d,D_m)).
$$
The comparison $P_\oplus\hsame\mathsf{append}(P_c,P_d)$ and the term comparison $A_m\hsame\mathsf{append}(C_m,D_m)$ put the continuation witness for $S_\oplus$ in the same source and right-history class as the canonical append continuation to the left endpoint; \autoref{thm:continuation-determinacy-up-to-history-sameness} gives the corresponding $\hsame$ comparison.

The two displayed continuation witnesses for $S_c$ and $S_d$ are compared with the canonical append continuations by \autoref{thm:continuation-determinacy-up-to-history-sameness}, yielding
$$
\mathsf{append}(P_c,C_m)\hsame S_c
\quad\text{and}\quad
\mathsf{append}(P_d,D_m)\hsame S_d.
$$
Applying \autoref{thm:external-append-hsame-congruence} to these two comparisons gives
$$
\mathsf{append}(\mathsf{append}(P_c,C_m),\mathsf{append}(P_d,D_m))
\hsame
\mathsf{append}(S_c,S_d).
$$
Transitivity with the comparison for $S_\oplus$ and the middle-four comparison gives the required comparison for $S_\oplus$.

Finally, let $T$ be any closed partial sum of $c\oplus d$ at $n$. The determinacy theorem \autoref{lem:complex-part-sum-well-defined} compares $T$ with the constructed $S_\oplus$; transitivity with the comparison already proved gives $T\hsame\mathsf{append}(S_c,S_d)$.
\end{proof}
