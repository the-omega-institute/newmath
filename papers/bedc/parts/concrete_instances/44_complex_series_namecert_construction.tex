\chapter{A Concrete Naming Certificate for \(\ComplexSeriesUp\)}
\label{ch:concrete-instances-complex-series-namecert}

This chapter gives a concrete naming-certificate construction for convergent infinite complex sums \(\sum_{n=0}^{\infty} c_n\), where \(c\) is a unary-indexed family of complex histories. The certificate is distinct from \(\ConvergenceRadiusUp\) (\autoref{ch:concrete-instances-convergence-radius-namecert}): there is no formal variable, only a complex sequence and its tail behaviour. The construction sits above \(\ComplexLimitUp\) (\autoref{ch:concrete-instances-complex-limit-namecert}) and reuses the regular Cauchy machinery underlying \(\RealUp\) (\autoref{ch:concrete-instances-real-namecert}). Absolute convergence and conditional convergence are licensed by separate predicates with explicit unary moduli; the comparison, ratio, and root tests are recorded as constructive lemmas; and the Cauchy product of two absolutely convergent series is constructed term-wise with explicit modulus pasting. The certificate is fully constructive and uses no \(\mathsf{Classical.choice}\), no \(\mathsf{Quot.sound}\), and no \(\mathsf{propext}\), so its Lean targets fall under the project's axiom-purity gate.

\section{Term sequences and partial sums}

\begin{definition}[Complex term sequence]
\label{def:complex-term-seq}
A complex term sequence is a unary-indexed family \(c : \Unary \to \ComplexUp\)-history, i.e.\ a function from unary histories to complex carrier histories.
\end{definition}
\leandef{BEDC.Derived.ComplexSeriesUp.ComplexTermSeqCarrier}

\begin{lemma}[Complex term sequence carrier transport]
\label{lem:complex-term-seq-carrier-hsame-transport}
If two complex term families $c,d:\mathsf{BHist}\to\mathsf{BHist}$ are
pointwise related by $\hsame$ on every unary index, then the
$\mathsf{ComplexTermSeqCarrier}$ predicate transports from $c$ to $d$.
\end{lemma}
\begin{proof}
Fix a unary index $n$. The carrier for $c(n)$ expands to a complex-pair witness:
rational real and imaginary component carriers together with a continuation
ledger whose result is $c(n)$. Transport that continuation result along the
pointwise comparison $c(n)\hsame d(n)$ while keeping the same rational
component witnesses. The transported pair witness is exactly the complex carrier
for $d(n)$.
\end{proof}
\leanchecked{BEDC.Derived.ComplexSeriesUp.ComplexTermSeqCarrier\_hsame\_transport}

\begin{definition}[Complex partial sum]
\label{def:complex-part-sum}
For a term sequence \(c\), the complex partial-sum predicate \(\mathsf{ComplexPartSum}(c, n, S_n)\) is the closed inductive predicate generated by:
\begin{enumerate}
  \item \(\mathsf{ComplexPartSum}(c, \emp, \emp_{\ComplexUp})\) where \(\emp_{\ComplexUp}\) is the zero complex history (both real components empty);
  \item if \(\mathsf{ComplexPartSum}(c, n, S_n)\) and the complex addition relation gives \(S_n + c_n = S_{n+1}\), then \(\mathsf{ComplexPartSum}(c, \Eone(n), S_{n+1})\).
\end{enumerate}
\end{definition}
\leandef{BEDC.Derived.ComplexSeriesUp.ComplexPartSum}

\begin{lemma}[Complex partial-sum index cases]
\label{lem:complex-part-sum-index-cases}
Every closed witness of $\mathsf{ComplexPartSum}(c,n,S)$ either has the zero
index $n=\emp$ and identifies $S$ with the chosen zero history, or has a
successor index $n=\Eone(m)$ together with a predecessor partial sum and the
continuation step from that predecessor to $S$.
\end{lemma}
\begin{proof}
Invert the closed partial-sum witness. The zero constructor gives the empty
index and the base history. The step constructor exposes the predecessor index,
predecessor sum, and continuation payload directly.
\end{proof}
\leanchecked{BEDC.Derived.ComplexSeriesUp.ComplexPartSum\_index\_cases}

\begin{lemma}[Partial sums are complex histories]
\label{lem:complex-part-sum-well-defined}
For every term sequence \(c\) and unary \(n\), any two histories obtained from
the closed partial-sum recursion for \(c\) at \(n\) agree up to \(\hsame\).
\end{lemma}
\begin{proof}
Induct on the first partial-sum derivation. The zero case is immediate. In the
successor case, the second derivation has the same successor shape; the
induction hypothesis identifies the previous sums, and determinacy of the BEDC
continuation relation identifies the two successor sums.
\end{proof}
\leanchecked{BEDC.Derived.ComplexSeriesUp.ComplexPartSum\_deterministic}

\begin{lemma}[Partial sums respect pointwise term equivalence]
\label{lem:complex-part-sum-term-hsame-deterministic}
If two term families \(c,d:\Hist\to\Hist\) are pointwise related by \(\hsame\)
on every unary index, and \(S,T\) are closed partial sums for \(c,d\) at the
same index \(n\), then \(S\hsame T\).
\end{lemma}
\begin{proof}
Induct on the first closed partial-sum witness. The base case forces the second
witness to be the zero constructor. In the successor case, the second witness
has the same successor index; the induction hypothesis identifies the
predecessor sums, closedness of the first witness supplies unaryness of the
predecessor index, and the pointwise term comparison identifies the terms. The
BEDC continuation relation then identifies the successor sums.
\end{proof}
\leanchecked{BEDC.Derived.ComplexSeriesUp.ComplexPartSum\_term\_hsame\_deterministic}

\begin{lemma}[Complex partial-sum successor term hsame determinacy]
\label{lem:complex-part-sum-successor-term-hsame-deterministic}
If two chosen zero histories agree under $\hsame$, and two term families agree
pointwise under $\hsame$, then a successor partial-sum step computed from the
first family has the same result as any closed successor partial-sum derivation
for the second family at the corresponding $\Eone(n)$ index.
\end{lemma}
\begin{proof}
First prove cross-family determinacy for partial sums by induction on the first
closed derivation. The zero constructor uses the zero-history comparison. The
successor constructor inverts the matching successor derivation on the other
side, applies the induction hypothesis to the predecessor sums, and then uses
determinacy of $\ContR$ with the pointwise term comparison. Applying this
cross-family result to the predecessor sums of the displayed successor witness
and the closed successor derivation identifies the final results.
\end{proof}
\leanchecked{BEDC.Derived.ComplexSeriesUp.ComplexPartSum\_successor\_term\_hsame\_deterministic}

\section{Absolute convergence}

\begin{definition}[Modulus partial sum]
\label{def:abs-part-sum}
For a complex term sequence \(c\), the modulus partial sum \(\mathsf{AbsPartSum}(c, n, M_n)\) is the real-history value \(M_n = \sum_{k < n} |c_k|\) computed by an explicit unary recursion taking complex modulus pointwise and summing in \(\RealUp\).
\end{definition}
\leandef{BEDC.Derived.ComplexSeriesUp.ComplexAbsPartSum}

\begin{lemma}[Absolute partial-sum successor result is deterministic]
\label{lem:complex-abs-part-sum-successor-result-deterministic}
If an absolute partial-sum derivation reaches $M$ at unary index $n$, and the
next modulus continuation sends $M$ and $\mathsf{modulus}(n)$ to $T$, then every
closed derivation of the successor absolute partial sum at $\Eone(n)$ has result
$\hsame$ to $T$.
\end{lemma}
\begin{proof}
Induct on the previous absolute partial-sum derivation to obtain determinacy at
each index. In the successor case, the other derivation has the same successor
shape; the induction hypothesis identifies the previous modulus sums, and
determinacy of $\mathsf{Cont}$ identifies the successor result.
\end{proof}
\leanchecked{BEDC.Derived.ComplexSeriesUp.ComplexAbsPartSum\_successor\_result\_deterministic}

\begin{lemma}[Absolute partial-sum successor modulus hsame determinacy]
\label{lem:complex-abs-part-sum-successor-modulus-hsame-deterministic}
If two chosen zero histories agree under $\hsame$, and two modulus families
agree pointwise under $\hsame$, then a successor absolute partial-sum step
computed from the first modulus family has the same result as any closed
successor absolute partial-sum derivation for the second family at the
corresponding $\Eone(n)$ index.
\end{lemma}
\begin{proof}
Induct on the first absolute partial-sum derivation to obtain cross-family
determinacy of predecessor modulus sums. The zero constructor is handled by the
zero-history comparison. In the successor constructor, invert the matching
successor derivation, compare predecessor sums by induction, and compare the
two continuation results using the pointwise modulus comparison. Applying this
determinacy statement to the predecessor of the displayed successor step and
the closed successor derivation gives the required result comparison.
\end{proof}
\leanchecked{BEDC.Derived.ComplexSeriesUp.ComplexAbsPartSum\_successor\_modulus\_hsame\_deterministic}

\begin{definition}[Absolute convergence]
\label{def:abs-conv}
A complex series \(c\) converges absolutely with bound \(M\) and modulus \(N : \Unary \to \Unary\), written \(\mathsf{AbsConv}(c, M, N)\), when:
\begin{enumerate}
  \item for every unary \(n\), \(\mathsf{AbsPartSum}(c, n, M_n)\) gives a real-history bound \(M_n \leq M\);
  \item for every unary precision \(k\) and unary \(n, m \geq N(k)\), \(|M_n - M_m| < 2^{-k}\).
\end{enumerate}
\end{definition}

\begin{lemma}[Absolute convergence implies convergence]
\label{lem:abs-conv-implies-conv}
If \(\mathsf{AbsConv}(c, M, N)\) holds, the complex partial sums \((S_n)_n\) form a regular complex Cauchy sequence with the same modulus \(N\), and hence have an explicit complex-history limit \(S\) by Cauchy completeness (\autoref{ch:concrete-instances-complex-limit-namecert}).
\end{lemma}
\begin{proof}[Sketch]
The triangle inequality on each segment of consecutive partial sums bounds \(|S_n - S_m|\) by \(|M_n - M_m|\). Apply the modulus from absolute convergence directly.
\end{proof}

\section{Convergence and uniqueness}

\begin{definition}[Complex series convergence]
\label{def:complex-series-conv}
\(\mathsf{ComplexSeriesConv}(c, S)\) holds when the complex partial-sum sequence \((S_n)_n\) is a regular complex Cauchy sequence (in the sense of \autoref{ch:concrete-instances-complex-limit-namecert}) with explicit limit \(S\).
\end{definition}

\begin{theorem}[Sum is unique]
\label{thm:complex-series-sum-unique}
If \(\mathsf{ComplexSeriesConv}(c, S)\) and \(\mathsf{ComplexSeriesConv}(c, S')\), then \(\hsame\) holds on both real components of \(S\) and \(S'\).
\end{theorem}
\begin{proof}[Sketch]
Direct application of complex-limit uniqueness.
\end{proof}

\section{Comparison, ratio, and root tests}

\begin{theorem}[Comparison test]
\label{thm:comparison-test}
Let \(c\) be a complex term sequence and \(d\) a real-history term sequence with \(d_n \geq 0\) for all \(n\). If for every unary \(n\), the modulus comparison \(|c_n| \leq d_n\) holds (decidable on each unary index), and \(\sum_n d_n\) converges in \(\RealUp\) with explicit modulus \(N_d\), then \(\mathsf{AbsConv}(c, M, N_d)\) holds with the same modulus \(N_d\) and bound \(M = \sum d_n\).
\end{theorem}
\begin{proof}[Sketch]
The comparison hypothesis transports to absolute partial sums: \(M_n \leq \sum_{k < n} d_k\). The modulus from \(d\)-convergence then bounds the differences of \(M_n\).
\end{proof}

\begin{theorem}[Ratio test]
\label{thm:ratio-test}
If there exists a rational \(r\) with \(0 \leq r < 1\) and a unary \(N_0\) such that for all unary \(n \geq N_0\), \(|c_{n+1}| \leq r \cdot |c_n|\) (decidable comparison), then \(\mathsf{AbsConv}\) holds with bound \(M = |c_{N_0}| / (1 - r)\) and modulus \(N(k) = N_0 + \lceil k / \log_2(1/r) \rceil\) computed as an explicit unary function of the precision.
\end{theorem}
\begin{proof}[Sketch]
Geometric majorant: starting from \(N_0\), the tail is bounded by a geometric series with ratio \(r\). Compute the modulus from the geometric tail sum.
\end{proof}

\begin{lemma}[Root test]
\label{lem:root-test}
If there exists a rational \(r < 1\) and unary \(N_0\) such that \(|c_n|^{1/n} \leq r\) (where the root is computed by explicit unary approximation) for all unary \(n \geq N_0\), then \(\mathsf{AbsConv}\) holds with explicit geometric modulus.
\end{lemma}
\begin{proof}[Sketch]
The hypothesis gives \(|c_n| \leq r^n\); apply comparison with the geometric series.
\end{proof}

\section{Cauchy product}

\begin{definition}[Cauchy product]
\label{def:cauchy-product}
For complex term sequences \(a, b\), the Cauchy product \(c = a \star b\) is the term sequence \(c_n = \sum_{k = 0}^{n} a_k \cdot b_{n-k}\) defined by an explicit unary recursion using complex multiplication.
\end{definition}

\begin{theorem}[Cauchy product preserves absolute convergence]
\label{thm:cauchy-product-conv}
If \(\mathsf{AbsConv}(a, M_a, N_a)\) and \(\mathsf{AbsConv}(b, M_b, N_b)\) hold, then \(\mathsf{AbsConv}(a \star b, M_a \cdot M_b, N_{ab})\) holds with explicit modulus \(N_{ab}\) computed from \(N_a, N_b, M_a, M_b\). Moreover the sum of the Cauchy product equals the product of sums: \(\sum (a \star b) = (\sum a) \cdot (\sum b)\).
\end{theorem}
\begin{proof}[Sketch]
Reorder partial sums into a triangular block structure. Absolute convergence justifies the reordering with explicit modulus pasting.
\end{proof}

\section{Closure laws}

\begin{lemma}[Linearity of convergence]
\label{lem:linearity-of-conv}
If \(\mathsf{ComplexSeriesConv}(a, S_a)\) and \(\mathsf{ComplexSeriesConv}(b, S_b)\), and \(\alpha, \beta\) are complex histories, then \(\mathsf{ComplexSeriesConv}(\alpha a + \beta b, \alpha S_a + \beta S_b)\) with explicit modulus pasting.
\end{lemma}
\begin{proof}[Sketch]
Pointwise scalar multiple and pointwise sum of regular sequences remain regular with combined modulus.
\end{proof}

\begin{lemma}[Reordering for absolute convergence]
\label{lem:reordering-abs-conv}
If \(\mathsf{AbsConv}(c, M, N)\) holds and \(\sigma : \Unary \to \Unary\) is an explicit unary bijection (a permutation of the indexing), then the reordered series \(c \circ \sigma\) is also absolutely convergent with the same bound and an explicit modulus computable from \(N\) and \(\sigma\).
\end{lemma}
\begin{proof}[Sketch]
Absolute convergence ensures that any reordering converges to the same limit with adjusted modulus.
\end{proof}

\section{The certificate}

\begin{definition}[Complex series source specification]
\label{def:complex-series-source-spec}
The source specification on \(\ComplexUp\)-term-sequences carries the constraint \(\mathsf{ComplexSeriesConv}(c, S)\) for some \(S\). Its generator is the complex partial-sum predicate.
\end{definition}

\begin{definition}[Complex series pattern specification]
\label{def:complex-series-pattern-spec}
The pattern of \(\ComplexSeriesUp\) is regular Cauchy partial-sum convergence: a sequence \(c\) is admissible when its partial-sum sequence is regular Cauchy with explicit modulus.
\end{definition}

\begin{definition}[Complex series classifier specification]
\label{def:complex-series-classifier-spec}
Two convergent complex series are classified together when their limits agree under \(\hsame\) on both real components.
\end{definition}

\begin{definition}[Complex series stability certificate]
\label{def:complex-series-stability-cert}
The stability certificate consists of:
\begin{enumerate}
  \item closure of \(\mathsf{ComplexSeriesConv}\) under pointwise \(\hsame\)-equivalence on the term sequence;
  \item transport of moduli under \(\hsame\) on terms;
  \item compatibility of partial sums with \(\hsame\) on indexing.
\end{enumerate}
\end{definition}

\begin{definition}[Complex series ledger policy]
\label{def:complex-series-ledger-policy}
The ledger policy records:
\begin{enumerate}
  \item linearity of convergence under sum and complex scalar multiplication;
  \item Cauchy product of absolutely convergent series;
  \item reordering invariance for absolute convergence.
\end{enumerate}
\end{definition}

\begin{theorem}[Concrete \(\ComplexSeriesUp\) naming certificate]
\label{thm:complex-series-namecert}
The five fields \((\mathsf{ComplexSeriesSourceSpec}, \mathsf{ComplexSeriesPatternSpec}, \mathsf{ComplexSeriesClassifierSpec}, \mathsf{ComplexSeriesStabilityCert}, \mathsf{ComplexSeriesLedgerPolicy})\) form a naming certificate over \(\Hist\) for the complex series interface \(\ComplexSeriesUp\).
\end{theorem}
\begin{proof}[Sketch]
The five fields satisfy the closure laws of the naming-certificate framework (\autoref{ch:core-typed-naming-certificates}). All conversions between term sequence, partial-sum sequence, modulus, and limit are explicit unary functions, so the certificate uses no \(\mathsf{Classical.choice}\), no \(\mathsf{Quot.sound}\), and no \(\mathsf{propext}\) --- its Lean targets fall under the project's axiom-purity gate.
\end{proof}

\begin{theorem}[Semantic \(\ComplexSeriesUp\) naming certificate]
\label{thm:complex-series-semantic-namecert}
The concrete certificate of \autoref{thm:complex-series-namecert} satisfies the semantic naming-certificate schema of \(\NameCert\) (\autoref{ch:core-typed-naming-certificates}): the carrier is the complex partial-sum predicate, the classifier is \(\hsame\) on the limit, and every closure law is given by an explicit constructive proof.
\end{theorem}
\begin{proof}[Sketch]
Direct verification of the four \(\NameCert\) field types using the constructions of this chapter.
\end{proof}
