\begin{theorem}[Compact located refinement chain continuation witness]
\label{thm:compact-located-refinement-chain-continuation-witness}
If
$$
\mathsf{CompactLocatedRefinementChain}(finite,located,intermediate,compact,
finalLocated,finalIntermediate,finalCompact)
$$
holds and $\Cont(intermediate,finite,compact)$ is the starting compactness
ledger, then there is a unary refinement history $extra$ such that
$$
\mathsf{UnaryHistory}(extra)\land
\Cont(located,extra,finalLocated)\land
\Cont(intermediate,extra,finalIntermediate)\land
\Cont(finalIntermediate,finite,finalCompact).
$$
\end{theorem}
\begin{proof}
Induct on the located refinement chain. The base case uses the empty history as
the common locatedness and intermediate tail, together with the supplied
starting compactness ledger. In the step case, the induction hypothesis supplies
an accumulated tail for locatedness and the intermediate ledger. Associate that
tail with the step refinement on both ledgers. Determinism identifies the two
associated tails, unary continuation closure proves the accumulated tail unary,
and the step compactness continuation gives the final compact endpoint ledger.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactLocatedRefinementChain\_continuation\_witness}
\begin{theorem}[Compact located refinement chain final compact determinacy]\label{thm:compact-located-refinement-chain-final-compact-deterministic}
If two compact located refinement chains share the same starting compactness ledger and the same final located and intermediate endpoints, then their final compact endpoints are $\hsame$-equal.
\end{theorem}
\begin{proof}
Apply the continuation witness theorem to both chains using the same starting compactness ledger. Both witnesses end in $\Cont(finalIntermediate,finite,-)$, so continuation determinism identifies the final compact endpoints.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactLocatedRefinementChain\_final\_compact\_deterministic}
\begin{theorem}[Compact witness carrier located refinement chain closure]
\label{thm:compact-witness-carrier-located-refinement-chain-closed}
If
$$
\mathsf{CompactWitnessCarrier}(subset,located,finite,intermediate,compact)
$$
and
$$
\mathsf{CompactLocatedRefinementChain}(finite,located,intermediate,compact,finalLocated,finalIntermediate,finalCompact),
$$
then
$$
\mathsf{CompactWitnessCarrier}(subset,finalLocated,finite,finalIntermediate,finalCompact).
$$
\end{theorem}
\begin{proof}
Induct on the compact located refinement chain.

In the base case, the final locatedness, intermediate, and compactness ledgers
are the original ledgers $located$, $intermediate$, and $compact$. The desired
carrier is exactly the given compact witness carrier.

In the step case, the induction hypothesis gives the carrier at the prior triple
$(l,i,c)$. Apply
Theorem~\ref{thm:compact-witness-carrier-located-extension-closed} to that
carrier, the unary refinement history $e$, and the continuations
$\Cont(l,e,l')$, $\Cont(i,e,i')$, and $\Cont(i',finite,c')$.

The result is the required compact witness carrier at $(l',i',c')$.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactWitnessCarrier\_located\_refinement\_chain\_closed}

\begin{theorem}[Compact witness carrier located finite refinement chain closure]
\label{thm:compact-witness-carrier-located-finite-refinement-chain-closed}
If

$$
\mathsf{CompactWitnessCarrier}(subset,located,finite,intermediate,compact),
$$

and

$$
\mathsf{CompactLocatedRefinementChain}(finite,located,intermediate,compact,
finalLocated,finalIntermediate,finalCompact),
$$

and

$$
\mathsf{CompactFiniteRefinementChain}(finite,finalCompact,finalFinite,finalCompact'),
$$

then

$$
\mathsf{CompactWitnessCarrier}(subset,finalLocated,finalFinite,finalIntermediate,
finalCompact').
$$
\end{theorem}
\begin{proof}
First apply
Theorem~\ref{thm:compact-witness-carrier-located-refinement-chain-closed} to the
initial carrier and the located refinement chain. This gives

$$
\mathsf{CompactWitnessCarrier}(subset,finalLocated,finite,finalIntermediate,
finalCompact).
$$

Then apply
Theorem~\ref{thm:compact-witness-carrier-finite-refinement-chain-closed} to that
carrier and the finite refinement chain from $(finite,finalCompact)$ to
$(finalFinite,finalCompact')$. The result is the displayed compact witness
carrier.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactWitnessCarrier\_located\_finite\_refinement\_chain\_closed}

\begin{theorem}[Compact witness carrier located finite refinement chain final endpoints unary]
\label{thm:compact-witness-carrier-located-finite-refinement-chain-final-endpoints-unary}
Assume
$$
\mathsf{CompactWitnessCarrier}(subset,located,finite,intermediate,compact),
$$
a located refinement chain from $(located,intermediate,compact)$ to
$(finalLocated,finalIntermediate,finalCompact)$ over the same finite witness,
and a finite refinement chain from $(finite,finalCompact)$ to
$(finalFinite,finalCompact')$. Then
$$
\mathsf{UnaryHistory}(finalLocated)\land
\mathsf{UnaryHistory}(finalIntermediate)\land
\mathsf{UnaryHistory}(finalFinite)\land
\mathsf{UnaryHistory}(finalCompact').
$$
\end{theorem}
\begin{proof}
Unpack the compact witness carrier. The carrier provides unary witnesses for
$subset$, $located$, and $finite$ together with the continuation ledgers
$\Cont(subset,located,intermediate)$ and
$\Cont(intermediate,finite,compact)$. Closure of unary histories under
$\Cont$ gives unary witnesses for $intermediate$ and $compact$.

Apply the located-chain endpoint theorem to the located chain and these four
initial endpoint witnesses. This gives unary witnesses for $finalLocated$,
$finalIntermediate$, and $finalCompact$. Apply the finite-chain endpoint theorem
to the finite chain, the original unary witness for $finite$, and the obtained
unary witness for $finalCompact$. The result supplies unary witnesses for
$finalFinite$ and $finalCompact'$, which together form the displayed
conjunction.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactWitnessCarrier\_located\_finite\_refinement\_chain\_final\_endpoints\_unary}
