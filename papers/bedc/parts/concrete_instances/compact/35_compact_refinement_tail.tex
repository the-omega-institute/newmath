\begin{theorem}[Compact located refinement chain continuation witness]
\label{thm:compact-located-refinement-chain-continuation-witness}
If
$$
\mathsf{CompactLocatedRefinementChain}(finite,located,intermediate,compact,
finalLocated,finalIntermediate,finalCompact)
$$
holds and $\Cont(intermediate,finite,compact)$ is the starting compactness
ledger, then there is a unary refinement history $extra$ such that
$$
\mathsf{UnaryHistory}(extra)\land
\Cont(located,extra,finalLocated)\land
\Cont(intermediate,extra,finalIntermediate)\land
\Cont(finalIntermediate,finite,finalCompact).
$$
\end{theorem}
\begin{proof}
Induct on the located refinement chain. The base case uses the empty history as
the common locatedness and intermediate tail, together with the supplied
starting compactness ledger. In the step case, the induction hypothesis supplies
an accumulated tail for locatedness and the intermediate ledger. Associate that
tail with the step refinement on both ledgers. Determinism identifies the two
associated tails, unary continuation closure proves the accumulated tail unary,
and the step compactness continuation gives the final compact endpoint ledger.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactLocatedRefinementChain\_continuation\_witness}
\begin{theorem}[Compact located refinement chain final compact determinacy]\label{thm:compact-located-refinement-chain-final-compact-deterministic}
If two compact located refinement chains share the same starting compactness ledger and the same final located and intermediate endpoints, then their final compact endpoints are $\hsame$-equal.
\end{theorem}
\begin{proof}
Apply the continuation witness theorem to both chains using the same starting compactness ledger. Both witnesses end in $\Cont(finalIntermediate,finite,-)$, so continuation determinism identifies the final compact endpoints.
\end{proof}

\begin{theorem}[Compact witness carrier located finite refinement chain final endpoints unary]
\label{thm:compact-witness-carrier-located-finite-refinement-chain-final-endpoints-unary}
Assume
$$
\mathsf{CompactWitnessCarrier}(subset,located,finite,intermediate,compact),
$$
a located refinement chain from $(located,intermediate,compact)$ to
$(finalLocated,finalIntermediate,finalCompact)$ over the same finite witness,
and a finite refinement chain from $(finite,finalCompact)$ to
$(finalFinite,finalCompact')$. Then
$$
\mathsf{UnaryHistory}(finalLocated)\land
\mathsf{UnaryHistory}(finalIntermediate)\land
\mathsf{UnaryHistory}(finalFinite)\land
\mathsf{UnaryHistory}(finalCompact').
$$
\end{theorem}
\begin{proof}
Unpack the compact witness carrier. The carrier provides unary witnesses for
$subset$, $located$, and $finite$ together with the continuation ledgers
$\Cont(subset,located,intermediate)$ and
$\Cont(intermediate,finite,compact)$. Closure of unary histories under
$\Cont$ gives unary witnesses for $intermediate$ and $compact$.

Apply the located-chain endpoint theorem to the located chain and these four
initial endpoint witnesses. This gives unary witnesses for $finalLocated$,
$finalIntermediate$, and $finalCompact$. Apply the finite-chain endpoint theorem
to the finite chain, the original unary witness for $finite$, and the obtained
unary witness for $finalCompact$. The result supplies unary witnesses for
$finalFinite$ and $finalCompact'$, which together form the displayed
conjunction.
\end{proof}

\begin{theorem}[Compact witness carrier located finite refinement certificate package]
\label{thm:compact-witness-carrier-located-finite-refinement-complete-package}
Assume
$$
\mathsf{CompactWitnessCarrier}(subset,located,finite,intermediate,compact),
$$
a compact located refinement chain from
$(finite,located,intermediate,compact)$ to
$(finalLocated,finalIntermediate,finalCompact)$, and a compact finite
refinement chain from $(finite,finalCompact)$ to
$(finalFinite,finalCompact')$. Then these data determine one refinement
certificate with the following components:
\begin{enumerate}
\item the refined carrier
$$
\mathsf{CompactWitnessCarrier}(subset,finalLocated,finalFinite,
finalIntermediate,finalCompact');
$$
\item unary histories for $finalLocated$, $finalIntermediate$, $finalFinite$,
and $finalCompact'$;
\item endpoint classifier equivalences
$$
finalLocated\hsame finalIntermediate\leftrightarrow located\hsame intermediate
$$
and
$$
finalFinite\hsame finalCompact'\leftrightarrow finite\hsame finalCompact;
$$
\item terminal empty reflection: if $finalLocated$, $finalIntermediate$,
$finalFinite$, and $finalCompact'$ are each history-same to $\emp$, then
$subset$, $located$, $finite$, $intermediate$, and $compact$ are each
history-same to $\emp$;
\item nonempty propagation from $located$ to $finalLocated$, from
$intermediate$ to $finalIntermediate$, from $finite$ to $finalFinite$, and
from $finalCompact$ to $finalCompact'$.
\end{enumerate}
\end{theorem}
\begin{proof}
First apply
Theorem~\ref{thm:compact-witness-carrier-located-finite-refinement-chain-closed}
to the initial carrier and the two refinement chains. This supplies the
refined compact witness carrier.

Next apply
Theorem~\ref{thm:compact-witness-carrier-located-finite-refinement-chain-final-endpoints-unary}
to the same inputs. The four endpoint unary witnesses are exactly the unary
part of the certificate. The located endpoint classifier equivalence comes
from the compact located refinement chain endpoint theorem, and the finite
endpoint classifier equivalence comes from the compact finite refinement chain
endpoint theorem.

The terminal empty-reflection component is
Theorem~\ref{thm:compact-witness-carrier-located-finite-empty-final-reflects-empty-initial}.
It takes the four empty terminal endpoints and reflects them back to the five
initial carrier ledgers.

Finally, the nonempty propagation fields are supplied independently by
Theorem~\ref{thm:compact-located-refinement-chain-located-nonempty-forward},
Theorem~\ref{thm:compact-located-refinement-chain-intermediate-nonempty-forward},
the checked compact finite refinement chain nonempty-forward theorem, and the
checked compact finite refinement chain compact-nonempty-forward theorem.
Together these components form the advertised refinement certificate package.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactWitnessCarrier\_located\_finite\_refinement\_complete\_package}

\begin{corollary}[Compact located refinement chain continuation determinacy]
\label{cor:compact-located-refinement-chain-continuation-determinacy}
Fix the compact located refinement-chain rows
$$
\mathsf{CompactLocatedRefinementChain}(finite,located,intermediate,compact,
finalLocated,finalIntermediate,finalCompact)
$$
and the starting compactness ledger $\Cont(intermediate,finite,compact)$. Any
consumer-facing continuation use accepted by this chapter factors through the
following displayed rows and through no further coordinate: the located
refinement tail from $located$ to $finalLocated$, the intermediate refinement
tail from $intermediate$ to $finalIntermediate$, and the finite compactness
suffix $\Cont(finalIntermediate,finite,finalCompact)$. Consequently, if a second
accepted use has the same starting compactness ledger and the same named final
located and intermediate ledgers, then it names the same finite route, the same
normal-form endpoint, and the same exactness decision up to $\hsame$; in
particular, it cannot choose an alternate hidden coordinate or a duplicate
classifier source.
\end{corollary}
\begin{proof}
Apply \autoref{thm:compact-located-refinement-chain-continuation-witness} to the
accepted chain and the fixed starting compactness ledger. The witness exposes a
unary refinement history $extra$ together with exactly three continuation rows:
$$
\Cont(located,extra,finalLocated),\quad
\Cont(intermediate,extra,finalIntermediate),\quad
\Cont(finalIntermediate,finite,finalCompact).
$$
These are the only route, transport, and readback rows made available by the
continuation witness, so a consumer-facing use that is admitted by the compact
located refinement-chain surface must read its finite route from $extra$, its
normal form from the two named endpoints, and its compact exactness decision
from the displayed finite suffix.

Now compare with any second accepted use over the same starting compactness
ledger and the same final located and intermediate ledgers. By
\autoref{thm:compact-located-refinement-chain-common-source-final-located-deterministic},
the common source and final intermediate row identify the located endpoints and
supply a common located refinement witness. With the located and intermediate
rows fixed, \autoref{thm:compact-located-refinement-chain-final-compact-deterministic}
identifies the final compact endpoints. Thus the two accepted uses have the
same exposed refinement tail and the same compact suffix up to $\hsame$. Any
purported extra coordinate would have to change either the located endpoint,
the intermediate endpoint, or the final compact endpoint; the cited determinacy
rows identify all three, so no unlisted coordinate or second classifier source
survives the comparison.
\end{proof}
