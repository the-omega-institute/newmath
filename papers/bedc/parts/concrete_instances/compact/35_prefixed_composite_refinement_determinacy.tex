\begin{theorem}[Prefixed composite compact net refinement determinacy]
\label{thm:compact-net-witness-prefixed-composite-refinement-deterministic}
For histories $p,c,\epsilon,\eta,\eta',\theta,\theta',m$, assume
$\epsilon$, $\eta$, and $\eta'$ are unary histories, with continuation ledgers
$$
\mathsf{Cont}(\epsilon,\eta,\theta),
\qquad
\mathsf{Cont}(\epsilon,\eta',\theta').
$$
If both composed precisions give compact net witnesses with the same prefixed
center and the same prefixed refined net,
$$
\mathsf{CompactNetWitness}(p\cdot c,\theta,p\cdot m)
\quad\hbox{and}\quad
\mathsf{CompactNetWitness}(p\cdot c,\theta',p\cdot m),
$$
then the refinement precisions are history-same:
$$
\hsame(\eta,\eta').
$$
\end{theorem}
\begin{proof}
Unpack the two prefixed compact net witnesses. Their continuation ledgers have
the same source $p\cdot c$ and the same result $p\cdot m$, so continuation
left cancellation identifies the composed precisions $\theta$ and $\theta'$.
With the composed precision made common, the two ledgers out of $\epsilon$
have the same source and result, and another left-cancellation step yields
$\hsame(\eta,\eta')$.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactNetWitness\_prefixed\_composite\_refinement\_deterministic}

\begin{theorem}[Compact finite refinement flattening]
\label{thm:compact-finite-refinement-flattening}
Let $subset$, $located$, $finite$, $intermediate$, $compact$, $midFinite$,
$midCompact$, $finalFinite$, and $finalCompact$ be histories. Assume a compact
witness carrier
$$
\mathsf{CompactWitnessCarrier}(subset,located,finite,intermediate,compact),
$$
an outer finite refinement chain
$$
\mathsf{CompactFiniteRefinementChain}(finite,compact,midFinite,midCompact),
$$
and a second finite refinement chain over the terminal finite-refinement pair
$$
\mathsf{CompactFiniteRefinementChain}(midFinite,midCompact,finalFinite,finalCompact).
$$
Then the two finite-refinement layers flatten to a single finite refinement
chain with the same source:
$$
\mathsf{CompactFiniteRefinementChain}(finite,compact,finalFinite,finalCompact).
$$
Moreover the flattened endpoint keeps the original subset, locatedness, and
intermediate classifier data:
$$
\mathsf{CompactWitnessCarrier}(subset,located,finalFinite,intermediate,finalCompact).
$$
\end{theorem}
\begin{proof}
Apply Theorem~\autoref{thm:compact-finite-refinement-chain-transitivity} to
the two displayed finite refinement chains. This gives the single chain from
$(finite,compact)$ to $(finalFinite,finalCompact)$, so the source pair is the
original source pair from the carrier.

Apply Theorem~\autoref{thm:compact-witness-carrier-finite-refinement-chain-closed}
to the original compact witness carrier and this flattened chain. The closure
theorem changes only the finite-net and compactness ledgers. Its conclusion has
exactly the same $subset$, $located$, and $intermediate$ arguments, so every
locatedness classification witnessed by the original carrier is the same
classification at the flattened endpoint.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactFiniteRefinementChain\_flattening\_located\_cont}

\begin{corollary}[Prefixed composite compact net refinement readback determinacy]
\label{cor:compact-net-witness-prefixed-composite-refinement-readback-determinacy}
Fix histories $p,c,\epsilon,\eta,\eta',\theta,\theta',m$ and compact-refinement
histories $subset$, $located$, $finite$, $intermediate$, $compact$,
$midFinite$, $midCompact$, $finalFinite$, and $finalCompact$. Assume
$\epsilon$, $\eta$, and $\eta'$ are unary histories and that the two prefixed
composite compact-net uses expose exactly the continuation and witness rows
$$
\Cont(\epsilon,\eta,\theta),
\qquad
\Cont(\epsilon,\eta',\theta'),
$$
$$
\mathsf{CompactNetWitness}(p\cdot c,\theta,p\cdot m),
\qquad
\mathsf{CompactNetWitness}(p\cdot c,\theta',p\cdot m).
$$
Assume also that the finite compact-refinement packet exposes the carrier and
two finite-refinement rows
$$
\mathsf{CompactWitnessCarrier}(subset,located,finite,intermediate,compact),
$$
$$
\mathsf{CompactFiniteRefinementChain}(finite,compact,midFinite,midCompact),
\qquad
\mathsf{CompactFiniteRefinementChain}(midFinite,midCompact,finalFinite,finalCompact).
$$
Then every consumer-facing use admitted by these rows has the same displayed
readback data: the prefixed composite refinement route is $\eta$ up to
$\hsame$, the finite route is the flattened chain
$$
\mathsf{CompactFiniteRefinementChain}(finite,compact,finalFinite,finalCompact),
$$
the normal-form endpoint is the pair $(finalFinite,finalCompact)$, and the
exactness decision is the carrier
$$
\mathsf{CompactWitnessCarrier}(subset,located,finalFinite,intermediate,finalCompact).
$$
Thus no accepted use can replace the refinement route by an alternate hidden
coordinate or cite a duplicate classifier source for the finite endpoint.
\end{corollary}
\begin{proof}
The two prefixed composite compact-net uses have the same visible prefix
$p$, center $c$, and final net $m$. Their only difference is the refinement
continuation row out of the fixed first precision $\epsilon$. Applying
\autoref{thm:compact-net-witness-prefixed-composite-refinement-deterministic}
to the four displayed prefixed composite rows gives
$$
\hsame(\eta,\eta').
$$
Therefore any admitted composite readback that uses the second refinement row
names the same refinement route as the first one, up to the chapter classifier
$\hsame$.

For the finite compact-refinement part, apply
\autoref{thm:compact-finite-refinement-flattening} to the displayed carrier and
the two displayed finite-refinement chains. The result is exactly the single
finite route from $(finite,compact)$ to $(finalFinite,finalCompact)$, together
with the carrier at the same final endpoint:
$$
\mathsf{CompactFiniteRefinementChain}(finite,compact,finalFinite,finalCompact)
$$
and
$$
\mathsf{CompactWitnessCarrier}(subset,located,finalFinite,intermediate,finalCompact).
$$
The flattening theorem changes only the finite-net and compactness ledgers and
keeps the same $subset$, $located$, and $intermediate$ classifier data. Hence
the accepted use has no further coordinate from which to choose another route,
normal form, or exactness source: the route is the flattened finite chain, the
normal form is its displayed endpoint pair, and the exactness decision is the
flattened carrier over that endpoint. Combining this with the prefixed
composite determinacy proves the stated readback determinacy.
\end{proof}
