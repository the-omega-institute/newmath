\begin{theorem}[Compact finite refinement chain final cases]
\label{thm:compact-finite-refinement-chain-final-cases}
Every compact finite refinement chain reaches its final endpoints in exactly one
of the constructor-visible ways: either the chain is at the starting pair, with
$finalFinite\hsame finite$ and $finalCompact\hsame compact$, or it has an
immediate predecessor pair and a unary refinement history $extra$ with
$\Cont(currentFinite,extra,finalFinite)$ and
$\Cont(currentCompact,extra,finalCompact)$.
\end{theorem}
\begin{proof}
Case-split on the chain constructor. The base constructor gives the two
endpoint sameness proofs. The step constructor exposes the predecessor chain,
the unary refinement history, and the two displayed continuation ledgers.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactFiniteRefinementChain\_final\_cases}

\begin{theorem}[Compact located refinement chain final cases]
\label{thm:compact-located-refinement-chain-final-cases}
Every compact located refinement chain reaches its final endpoints either at
the starting triple, with all three final endpoints history-same to the
starting ledgers, or by a final refinement step. In the step case there are
current located, intermediate, and compact endpoints and a unary history
$extra$ such that the predecessor chain reaches the current endpoints,
$\Cont(currentLocated,extra,finalLocated)$,
$\Cont(currentIntermediate,extra,finalIntermediate)$, and
$\Cont(finalIntermediate,finite,finalCompact)$ hold.
\end{theorem}
\begin{proof}
Case-split on the located refinement-chain constructor. The base constructor
returns the three endpoint sameness proofs. The step constructor supplies the
predecessor chain, the unary refinement history, the located and intermediate
continuation ledgers, and the final compactness continuation ledger.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactLocatedRefinementChain\_final\_cases}
