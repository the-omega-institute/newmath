\begin{theorem}[Prefixed composite compact net witness closure]
\label{thm:compact-net-witness-prefixed-composite-closed}
For unary histories $p,c,\epsilon,\eta,\theta,n,m$, assume
$\mathsf{UnaryHistory}(p)$,
$\mathsf{CompactNetWitness}(c,\epsilon,n)$,
$\mathsf{CompactNetWitness}(n,\eta,m)$, and
$\mathsf{Cont}(\epsilon,\eta,\theta)$. Then the composed refinement is a
compact net witness under the same visible prefix:
$$
\mathsf{CompactNetWitness}(p \cdot c,\theta,p \cdot m).
$$

The certificate fit has ten fields: the traditional target is finite-net
precision composition; the BEDC source is $BHist$ equipped with $\mathsf{Cont}$;
the carrier is $\mathsf{CompactNetWitness}$; the classifier is $hsame$; the
stability law is closure under prefix and precision composition; the ledger
data are center, precision, and net continuations; the core theorem is this
prefixed composite closure; exactness is the direct composed witness; the
standard bridge is deferred; the status is $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Unpack the two compact net witnesses into unary center, precision, and net
histories, together with continuation ledgers $n=c\cdot\epsilon$ and
$m=n\cdot\eta$.

The continuation hypothesis $\mathsf{Cont}(\epsilon,\eta,\theta)$ gives the
composed precision history. Unary continuation closure supplies the unary
history for $\theta$, and associativity identifies
$(c\cdot\epsilon)\cdot\eta$ with $c\cdot(\epsilon\cdot\eta)$.

The identified composite ledger presents $m=c\cdot\theta$. Applying prefix
closure to this composed witness and using associativity once more yields the
ledger from $p\cdot c$ to $p\cdot m$ with precision $\theta$.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactNetWitness\_prefixed\_composite\_closed}

\begin{theorem}[Prefixed composite compact net witness factorization]
\label{thm:compact-net-witness-prefixed-composite-factorizes}
For histories $p,c,\epsilon,\eta,\theta,m$, assume
$\mathsf{UnaryHistory}(\epsilon)$,
$\mathsf{UnaryHistory}(\eta)$,
$\mathsf{Cont}(\epsilon,\eta,\theta)$, and
$\mathsf{CompactNetWitness}(p \cdot c,\theta,p \cdot m)$. Then there exists a
history $n$ such that
$$
\mathsf{CompactNetWitness}(c,\epsilon,n)
$$
and
$$
\mathsf{CompactNetWitness}(n,\eta,m).
$$
\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:compact-net-witness-prefix-iff} to the visible-prefix
witness. This removes the common prefix and recovers
$\mathsf{CompactNetWitness}(c,\theta,m)$, hence the unary histories for
$c,\theta,m$ and the continuation ledger $\mathsf{Cont}(c,\theta,m)$.

Choose the intermediate history $n=c\cdot\epsilon$. The unary history of
$\epsilon$ and the recovered unary history of $c$ give unary $n$, and the
continuation constructor gives $\mathsf{Cont}(c,\epsilon,n)$. These data form
$\mathsf{CompactNetWitness}(c,\epsilon,n)$.

It remains to build the second witness. The hypothesis
$\mathsf{Cont}(\epsilon,\eta,\theta)$ and the recovered continuation
$\mathsf{Cont}(c,\theta,m)$ identify the ledger $m$ with
$c\cdot(\epsilon\cdot\eta)$; associativity rewrites this as
$(c\cdot\epsilon)\cdot\eta=n\cdot\eta$. Together with the unary histories of
$n$, $\eta$, and $m$, this is $\mathsf{CompactNetWitness}(n,\eta,m)$.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactNetWitness\_prefixed\_composite\_factorizes}

\begin{theorem}[Prefixed composite compact net middle deterministic factorization]
\label{thm:compact-net-witness-prefixed-composite-middle-deterministic-factorization}
For histories $p,c,\epsilon,\eta,\theta,m$, assume
$\mathsf{UnaryHistory}(\epsilon)$,
$\mathsf{UnaryHistory}(\eta)$, and
$\mathsf{Cont}(\epsilon,\eta,\theta)$. If
$$
\mathsf{CompactNetWitness}(p \cdot c,\theta,p \cdot m),
$$
then there exists a history $n$ such that
$$
\mathsf{CompactNetWitness}(c,\epsilon,n)
$$
and
$$
\mathsf{CompactNetWitness}(n,\eta,m).
$$
Moreover any other history $n'$ supporting these two compact-net witnesses is
history-same to $n$.
\end{theorem}
\begin{proof}
Apply the prefixed composite factorization theorem to obtain an intermediate
history $n$ and the two compact-net witnesses through $\epsilon$ and $\eta$.
For any other displayed intermediate $n'$, the first compact-net witnesses for
$n$ and $n'$ share the same center $c$ and precision $\epsilon$; continuation
determinacy therefore gives $\hsame(n,n')$.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactNetWitness\_prefixed\_composite\_middle\_deterministic\_factorization}

\begin{definition}[Two-step compact net factor]
\label{def:compact-net-two-step-factor}
For histories $c,\epsilon,\eta,m$, the predicate
$\mathsf{CompactNetTwoStepFactor}(c,\epsilon,\eta,m)$ records an intermediate
net realizing the two visible precision steps:
$$
\mathsf{CompactNetTwoStepFactor}(c,\epsilon,\eta,m)
\Longleftrightarrow
\exists n.\;
\mathsf{CompactNetWitness}(c,\epsilon,n)\wedge
\mathsf{CompactNetWitness}(n,\eta,m).
$$
\end{definition}

\begin{theorem}[Prefixed composite compact net witness exactness]
\label{thm:compact-net-witness-prefixed-composite-iff}
For histories $p,c,\epsilon,\eta,\theta,m$, assume
$\mathsf{UnaryHistory}(\epsilon)$, $\mathsf{UnaryHistory}(\eta)$, and
$\mathsf{Cont}(\epsilon,\eta,\theta)$. Then the prefixed composed certificate
is exact at the compact-net witness level:
$$
\mathsf{CompactNetWitness}(p \cdot c,\theta,p \cdot m)
\Longleftrightarrow
\mathsf{UnaryHistory}(p)\wedge
\exists n,\mathsf{CompactNetWitness}(c,\epsilon,n)\wedge
\mathsf{CompactNetWitness}(n,\eta,m).
$$

The certificate fit has ten fields: the traditional target is finite-net
precision composition; the BEDC source is $BHist$ equipped with
$\mathsf{Cont}$; the carrier is $\mathsf{CompactNetWitness}$; the classifier is
$hsame$; the stability obligation is prefixed composite exactness; the ledger
data are center, precision, intermediate net, and final net continuations; the
core theorem is this equivalence; the exactness predicate is the displayed
equivalence; the standard bridge is deferred; the status is
$\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
For the forward direction, apply
Theorem~\ref{thm:compact-net-witness-prefix-iff} to the prefixed composed
witness. This recovers $\mathsf{UnaryHistory}(p)$ together with the unprefixed
witness $\mathsf{CompactNetWitness}(c,\theta,m)$. The hypotheses on
$\epsilon$, $\eta$, and $\theta$ then allow
Theorem~\ref{thm:compact-net-witness-prefixed-composite-factorizes} to produce
an intermediate history $n$ with the two compact-net witnesses required on the
right-hand side.

For the reverse direction, project the unary prefix evidence and the
intermediate history $n$ from the right-hand side. The two compact-net witnesses
for $c$ through $\epsilon$ and for $n$ through $\eta$, together with
$\mathsf{Cont}(\epsilon,\eta,\theta)$, satisfy the hypotheses of
Theorem~\ref{thm:compact-net-witness-prefixed-composite-closed}. Applying that
closure theorem yields
$\mathsf{CompactNetWitness}(p \cdot c,\theta,p \cdot m)$.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactNetWitness\_prefixed\_composite\_iff}

\begin{theorem}[Prefixed composite compact net result determinacy]
\label{thm:compact-net-witness-prefixed-composite-result-deterministic}
For histories $p,c,\epsilon,\eta,\theta,m,m'$, assume
$\mathsf{UnaryHistory}(\epsilon)$, $\mathsf{UnaryHistory}(\eta)$, and
$\mathsf{Cont}(\epsilon,\eta,\theta)$. If both prefixed composed net witnesses
start from the same prefixed center and composed precision,
$$
\mathsf{CompactNetWitness}(p \cdot c,\theta,p \cdot m)
\quad\hbox{and}\quad
\mathsf{CompactNetWitness}(p \cdot c,\theta,p \cdot m'),
$$
then the visible refined nets are history-same:
$$
\hsame(m,m').
$$
\end{theorem}
\begin{proof}
Apply the prefixed composite factorization theorem to each prefixed composed
witness. This produces two unprefixed two-step compact-net chains from the same
center $c$, through the same precision $\epsilon$, and then through the same
refinement precision $\eta$. Composite-result determinacy for compact net
witnesses compares those two unprefixed chains and yields $\hsame(m,m')$.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactNetWitness\_prefixed\_composite\_result\_deterministic}

\begin{theorem}[Prefixed composite compact net prefix-independent result determinacy]
\label{thm:compact-net-witness-prefixed-composite-prefix-independent-result-deterministic}
For histories $p,p',c,\epsilon,\eta,\theta,m,m'$, assume
$\mathsf{UnaryHistory}(\epsilon)$, $\mathsf{UnaryHistory}(\eta)$, and
$\mathsf{Cont}(\epsilon,\eta,\theta)$. If two composed compact-net witnesses
share the same unprefixed center and composed precision but use different
visible prefixes,
$$
\mathsf{CompactNetWitness}(p \cdot c,\theta,p \cdot m)
\quad\hbox{and}\quad
\mathsf{CompactNetWitness}(p' \cdot c,\theta,p' \cdot m'),
$$
then the refined nets are history-same:
$$
\hsame(m,m').
$$
\end{theorem}
\begin{proof}
Apply prefixed composite factorization to both witnesses, using the same
precision histories $\epsilon,\eta$ and composed precision $\theta$. This
removes the visible prefixes and recovers two unprefixed two-step compact-net
chains from the same center $c$ through the same precision sequence. The
unprefixed composite-result determinacy theorem compares those recovered chains
and gives $\hsame(m,m')$.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactNetWitness\_prefixed\_composite\_prefix\_independent\_result\_deterministic}

\begin{theorem}[Prefixed composite compact net center determinacy]
\label{thm:compact-net-witness-prefixed-composite-center-deterministic}
For histories $p,c,c',\epsilon,\eta,\theta,m$, assume
$\mathsf{UnaryHistory}(\epsilon)$, $\mathsf{UnaryHistory}(\eta)$, and
$\mathsf{Cont}(\epsilon,\eta,\theta)$. If two prefixed composed net witnesses

\mathsf{CompactNetWitness}(p \cdot c,\theta,p \cdot m)
\quad\hbox{and}\quad

\hsame(c,c').
$$
\end{theorem}
\begin{proof}
Apply prefixed composite factorization to both prefixed witnesses. This
recovers two unprefixed two-step compact-net chains ending at the same refined
net and using the same precision histories $\epsilon$ and $\eta$. The
unprefixed composite center determinacy theorem compares the recovered chains
and yields $\hsame(c,c')$.
\end{proof}

\begin{theorem}[Prefixed composite compact net precision determinacy]
\label{thm:compact-net-witness-prefixed-composite-precision-deterministic}
For histories $p,c,\epsilon,\epsilon',\eta,\theta,\theta',m$, assume
$\mathsf{UnaryHistory}(\epsilon)$, $\mathsf{UnaryHistory}(\epsilon')$,
$\mathsf{UnaryHistory}(\eta)$,
$\mathsf{Cont}(\epsilon,\eta,\theta)$, and
$\mathsf{Cont}(\epsilon',\eta,\theta')$. If two prefixed composed net witnesses
have the same prefixed center and the same prefixed refined net,
$$
\mathsf{CompactNetWitness}(p \cdot c,\theta,p \cdot m)
\quad\hbox{and}\quad
\mathsf{CompactNetWitness}(p \cdot c,\theta',p \cdot m),
$$
then their first precision histories are history-same:
$$
\hsame(\epsilon,\epsilon').
$$
\end{theorem}
\begin{proof}
Unpack the two prefixed compact-net witnesses. Since they share the same
prefixed center and prefixed refined net, left cancellation on their
continuation ledgers identifies the composite precision histories
$\theta$ and $\theta'$. The two composite ledgers then have the same right
suffix $\eta$, so continuation suffix cancellation identifies the first
precision histories.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactNetWitness\_prefixed\_composite\_precision\_deterministic}
