
\begin{theorem}[Prefixed composite compact net result determinacy]
\label{thm:compact-net-witness-prefixed-composite-result-deterministic}
For histories $p,c,\epsilon,\eta,\theta,m,m'$, assume
$\mathsf{UnaryHistory}(\epsilon)$, $\mathsf{UnaryHistory}(\eta)$, and
$\mathsf{Cont}(\epsilon,\eta,\theta)$. If both prefixed composed net witnesses
start from the same prefixed center and composed precision,
$$
\mathsf{CompactNetWitness}(p \cdot c,\theta,p \cdot m)
\quad\hbox{and}\quad
\mathsf{CompactNetWitness}(p \cdot c,\theta,p \cdot m'),
$$
then the visible refined nets are history-same:
$$
\hsame(m,m').
$$
\end{theorem}
\begin{proof}
Apply the prefixed composite factorization theorem to each prefixed composed
witness. This produces two unprefixed two-step compact-net chains from the same
center $c$, through the same precision $\epsilon$, and then through the same
refinement precision $\eta$. Composite-result determinacy for compact net
witnesses compares those two unprefixed chains and yields $\hsame(m,m')$.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactNetWitness\_prefixed\_composite\_result\_deterministic}

\begin{theorem}[Prefixed composite compact net prefix-independent result determinacy]
\label{thm:compact-net-witness-prefixed-composite-prefix-independent-result-deterministic}
For histories $p,p',c,\epsilon,\eta,\theta,m,m'$, assume
$\mathsf{UnaryHistory}(\epsilon)$, $\mathsf{UnaryHistory}(\eta)$, and
$\mathsf{Cont}(\epsilon,\eta,\theta)$. If two composed compact-net witnesses
share the same unprefixed center and composed precision but use different
visible prefixes,
$$
\mathsf{CompactNetWitness}(p \cdot c,\theta,p \cdot m)
\quad\hbox{and}\quad
\mathsf{CompactNetWitness}(p' \cdot c,\theta,p' \cdot m'),
$$
then the refined nets are history-same:
$$
\hsame(m,m').
$$
\end{theorem}
\begin{proof}
Apply prefixed composite factorization to both witnesses, using the same
precision histories $\epsilon,\eta$ and composed precision $\theta$. This
removes the visible prefixes and recovers two unprefixed two-step compact-net
chains from the same center $c$ through the same precision sequence. The
unprefixed composite-result determinacy theorem compares those recovered chains
and gives $\hsame(m,m')$.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactNetWitness\_prefixed\_composite\_prefix\_independent\_result\_deterministic}

\begin{theorem}[Prefixed composite compact net center determinacy]
\label{thm:compact-net-witness-prefixed-composite-center-deterministic}
For histories $p,c,c',\epsilon,\eta,\theta,m$, assume
$\mathsf{UnaryHistory}(\epsilon)$, $\mathsf{UnaryHistory}(\eta)$, and
$\mathsf{Cont}(\epsilon,\eta,\theta)$. If two prefixed composed net witnesses
have the same composed precision and the same prefixed refined net,
$$
\mathsf{CompactNetWitness}(p \cdot c,\theta,p \cdot m)
\quad\hbox{and}\quad
\mathsf{CompactNetWitness}(p \cdot c',\theta,p \cdot m),
$$
then their unprefixed centers are history-same:
$$
\hsame(c,c').
$$
\end{theorem}
\begin{proof}
Apply prefixed composite factorization to both prefixed witnesses. This
recovers two unprefixed two-step compact-net chains ending at the same refined
net and using the same precision histories $\epsilon$ and $\eta$. The
unprefixed composite center determinacy theorem compares the recovered chains
and yields $\hsame(c,c')$.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactNetWitness\_prefixed\_composite\_center\_deterministic}

\begin{theorem}[Prefixed composite compact net precision determinacy]
\label{thm:compact-net-witness-prefixed-composite-precision-deterministic}
For histories $p,c,\epsilon,\epsilon',\eta,\theta,\theta',m$, assume
$\mathsf{UnaryHistory}(\epsilon)$, $\mathsf{UnaryHistory}(\epsilon')$,
$\mathsf{UnaryHistory}(\eta)$,
$\mathsf{Cont}(\epsilon,\eta,\theta)$, and
$\mathsf{Cont}(\epsilon',\eta,\theta')$. If two prefixed composed net witnesses
have the same prefixed center and the same prefixed refined net,
$$
\mathsf{CompactNetWitness}(p \cdot c,\theta,p \cdot m)
\quad\hbox{and}\quad
\mathsf{CompactNetWitness}(p \cdot c,\theta',p \cdot m),
$$
then their first precision histories are history-same:
$$
\hsame(\epsilon,\epsilon').
$$
\end{theorem}
\begin{proof}
Unpack the two prefixed compact-net witnesses. Since they share the same
prefixed center and prefixed refined net, left cancellation on their
continuation ledgers identifies the composite precision histories
$\theta$ and $\theta'$. The two composite ledgers then have the same right
suffix $\eta$, so continuation suffix cancellation identifies the first
precision histories.
\end{proof}
\leanchecked{BEDC.Derived.CompactUp.CompactNetWitness\_prefixed\_composite\_precision\_deterministic}
