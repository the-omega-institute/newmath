\section{Carrier definition}
\label{sec:policy-carrier}

\begin{definition}[Policy action-ledger carrier]
\label{def:policy-action-ledger-carrier}
A $\PolicyUp$ carrier is a finite $\mathsf{BHist}$ action-ledger packet over
$\NameCert_{\BeliefUp}$, $\NameCert_{\MarkovChainUp}$,
$\NameCert_{\RandomVarUp}$, and $\NameCert_{\EstimatorUp}$. Its rows are
belief-state rows, Markov transition rows, estimator-score rows, selected
action rows, the $\hsame$ transport rows comparing corresponding ledger
entries, the $\Cont$ provenance rows explaining how each selected action is
read from the displayed belief, transition, and estimator rows, and the
$\mathsf{Pkg}$ source rows naming the four dependency certificates.

The carrier accepts only packets whose selected-action row is sourced from the
displayed dependency rows through the visible $\Cont$ ledger. It contains no
host Markov decision process, utility function, optimizer, or external policy
object.
\end{definition}
\begin{theorem}[Policy TasteGate source scope]
\label{thm:policy-tastegate-source-scope}
Every accepted $\PolicyUp$ action-ledger packet exposes exactly the four
dependency certificates $\NameCert_{\BeliefUp}$,
$\NameCert_{\MarkovChainUp}$, $\NameCert_{\RandomVarUp}$, and
$\NameCert_{\EstimatorUp}$ together with the finite action ledger of
\autoref{def:policy-action-ledger-carrier}. No host Markov decision process,
utility function, optimizer, or external policy object is exported by the
packet.
\end{theorem}
\begin{proof}
Project the carrier rows of
\autoref{def:policy-action-ledger-carrier}. The source rows are precisely the
four named dependency certificates and the finite $\mathsf{BHist}$ action
ledger whose entries display belief states, transition rows, estimator scores,
selected actions, $\hsame$ comparisons, $\Cont$ provenance, and
$\mathsf{Pkg}$ source rows.

The acceptance condition reads a selected action only through those displayed
rows. Since the carrier definition has no constructor for a host Markov
decision process, utility function, optimizer, or external policy object, no
accepted packet can expose such a row.
\end{proof}

\begin{theorem}[Policy local action transport]
\label{thm:policy-local-action-transport}
Let two finite $\PolicyUp$ action-ledger packets have componentwise $\hsame$
comparisons between their belief-state rows, Markov transition rows,
estimator-score rows, and selected-action rows. If the $\Cont$ provenance and
$\mathsf{Pkg}$ source rows are transported along those comparisons, then the
transported rows again form an accepted local $\PolicyUp$ carrier packet over
the same four dependency certificates.
\end{theorem}
\leanchecked{BEDC.Derived.PolicyUp.PolicyActionLedgerCarrier\_local\_action\_transport}
\begin{proof}
Transport the four dependency rows componentwise through the displayed
$\hsame$ comparisons. The belief, transition, estimator, and selected-action
entries remain inside the same finite $\mathsf{BHist}$ ledger shape.

Next transport the $\Cont$ provenance rows and the $\mathsf{Pkg}$ source rows.
The carrier acceptance condition only asks that the selected-action row be read
from the transported belief, transition, and estimator rows through that
visible provenance ledger. Therefore the transported packet satisfies the same
local carrier predicate.
\end{proof}

\begin{theorem}[Policy action classifier stability]
\label{thm:policy-action-classifier-stability}
Let $P$ and $P'$ be accepted $\PolicyUp$ action-ledger packets over the same
four displayed dependency certificates. Suppose the belief-state rows, Markov
transition rows, random-variable rows, estimator-score rows, selected-action
rows, $\Cont$ provenance rows, and $\mathsf{Pkg}$ source rows of $P$ are
transported componentwise to the corresponding rows of $P'$ by displayed
$\hsame$ comparisons. Then the transported packet is again accepted by the
PolicyUp action-ledger carrier of
\autoref{def:policy-action-ledger-carrier}, and the selected-action
classifier reads the same visible BEDC ledger surface.
\end{theorem}
\leanchecked{BEDC.Derived.PolicyUp.PolicyActionLedgerCarrier\_action\_classifier\_stability}
\begin{proof}
Inspect the finite action-ledger carrier. Its acceptance predicate names only
the displayed dependency certificates, the belief, transition,
random-variable, estimator, and selected-action rows, and the visible
$\hsame/\Cont/\mathsf{Pkg}$ provenance attached to those rows.

Transport each listed coordinate along the supplied $\hsame$ rows. The
transport preserves the finite $\mathsf{BHist}$ ledger shape and leaves the
source certificates inside the same four dependency packets.

The selected-action classifier has no access to any row outside that carrier.
After the $\Cont$ and $\mathsf{Pkg}$ rows are transported, the classifier
therefore reads the selected action through the transported visible ledger and
accepts the packet again.
\end{proof}

\begin{theorem}[Policy visible continuation provenance coverage]
\label{thm:policy-visible-cont-provenance-coverage}
Every selected-action row in an accepted $\PolicyUp$ packet is read through the
visible $\Cont$ provenance from the displayed $\BeliefUp$,
$\MarkovChainUp$, $\RandomVarUp$, and $\EstimatorUp$ source rows. The packet
exposes no host Markov decision process, optimizer, utility function, or
external policy object.
\end{theorem}
\begin{proof}
Project an accepted packet to the rows named by
\autoref{def:policy-action-ledger-carrier}. The selected-action coordinate is
accepted only when a visible $\Cont$ row relates it to the displayed belief,
transition, random-variable, and estimator rows, with $\mathsf{Pkg}$
provenance naming the four source certificates.

No other source is present in the carrier. A host Markov decision process,
optimizer, utility function, or external policy object would require an
additional readable coordinate, but the carrier has no such coordinate.
\end{proof}

\begin{theorem}[Policy NameCert obligation surface]
\label{thm:policy-namecert-obligation-surface}
The local $\NameCert_{\PolicyUp}$ obligation surface consists of the
action-ledger carrier of \autoref{def:policy-action-ledger-carrier}, the
TasteGate source scope of \autoref{thm:policy-tastegate-source-scope}, action
classifier stability from
\autoref{thm:policy-action-classifier-stability}, visible continuation
provenance coverage from
\autoref{thm:policy-visible-cont-provenance-coverage}, and local action
transport from \autoref{thm:policy-local-action-transport}.
\end{theorem}
\leanchecked{BEDC.Derived.PolicyUp.PolicyActionLedgerCarrier\_semantic\_name\_certificate}
\begin{proof}
Use the action-ledger carrier as the source predicate. It fixes the finite
$\mathsf{BHist}$ rows and the four displayed dependency certificates.

Use the TasteGate source-scope theorem to bound the exported sources. Use
classifier stability to transport the accepted action classifier along
componentwise $\hsame$ rows, and use visible provenance coverage to identify
the exact $\Cont$ rows through which selected actions are read.

Finally apply local action transport to repack transported rows as another
accepted carrier. These clauses provide exactly the local BEDC surface named
by $\NameCert_{\PolicyUp}$, without adding host MDP, optimizer, utility, or
external policy data.
\end{proof}

\begin{theorem}[Policy kernel scope boundary]
\label{thm:policy-kernel-scope-boundary}
The five local obligations of $\NameCert_{\PolicyUp}$ are scoped exactly to
the finite $\mathsf{BHist}$ action-ledger carrier of
\autoref{def:policy-action-ledger-carrier}, the four dependency certificates
$\NameCert_{\BeliefUp}$, $\NameCert_{\MarkovChainUp}$,
$\NameCert_{\RandomVarUp}$, and $\NameCert_{\EstimatorUp}$, and the displayed
$\hsame$, $\Cont$, and $\mathsf{Pkg}$ rows. No host Markov decision process,
optimizer, utility function, external policy object, hidden state space, or
unlogged action source is readable by those obligations.
\end{theorem}
\begin{proof}
Start from the action-ledger carrier of
\autoref{def:policy-action-ledger-carrier}. Its rows are the finite
$\mathsf{BHist}$ ledger, the four dependency certificate rows, the visible
$\hsame$ transport rows, the $\Cont$ provenance rows, and the $\mathsf{Pkg}$
source rows.

The source side is bounded by
\autoref{thm:policy-tastegate-source-scope}. The classifier and transported
carrier reads are bounded by
\autoref{thm:policy-local-action-transport} and
\autoref{thm:policy-action-classifier-stability}. The selected-action
provenance is bounded by
\autoref{thm:policy-visible-cont-provenance-coverage}.

Finally, \autoref{thm:policy-namecert-obligation-surface} says that these are
exactly the five local obligations named by $\NameCert_{\PolicyUp}$. Since
each source, classifier, transport, and ledger read has already been accounted
for by the listed rows, no host MDP, optimizer, utility, external policy
object, hidden state space, or unlogged action source is exposed.
\end{proof}
