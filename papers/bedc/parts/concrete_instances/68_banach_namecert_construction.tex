\chapter{A Concrete Naming Certificate for $\BanachUp$}
\label{ch:concrete-instances-banach-namecert}

The Banach-space interface $\BanachUp$ packages a normed vector space carrier together with a regular Cauchy completeness witness in the metric induced by the norm. It depends on $\NameCert_{\NormUp}$ (\autoref{ch:concrete-instances-norm-namecert}) and the real-completeness capstone (\autoref{ch:capstones-real-completeness}), and supplies the substrate for $\NameCert_{\HilbertUp}$ and the operator certificates downstream.

\section{Carrier definition}
\label{sec:banach-carrier}

\begin{definition}[Concrete Banach carrier]\label{def:banach-concrete-carrier}
A concrete Banach carrier is a history predicate $X:\Hist\to\mathsf{Prop}$
together with the $\NormUp$ carrier and classifier data supplied by
$\NameCert_{\NormUp}$, a displayed norm value row for each carried history,
and the metric predicate induced by that norm:
$$
\mathsf{BanachMetricWitness}_X(x,y,d)
$$
means that $x$ and $y$ are carried by $X$, their difference history is a
$\NormUp$ vector-history row, and $d$ is the $\RealUp$ norm value assigned to
that difference. The point classifier of the carrier is not a quotient:
carried representatives are identified by $\hsame$ transport of the displayed
vector, norm-value, and metric rows.

The completeness field is a family which assigns, to every regular Cauchy
$\StreamNameUp$ stream $s$ over $X$ for the norm-induced metric, histories
$\mu_s,\ell_s,L_s$ such that
$$
\mathsf{CompleteMetricLimitWitness}(X,s,\mu_s,\ell_s,L_s)
$$
holds. Thus the Banach datum is the tuple consisting of the carried
$\NormUp$ structure, the induced metric ledger, the $\hsame$ classifier on
representatives, and the displayed limit-witness family for all regular
Cauchy stream names over $X$.
\end{definition}

\begin{definition}[Complete normed-history Banach carrier]
\label{def:banach-complete-normed-carrier}
A $\BanachUp$ candidate is a $\VecSpaceUp$ carrier $C_V$ with vector
classifier $\sim_V$, zero endpoint $0_V$, additive difference operation
$x-_V y$, and scalar source read over $\NameCert_{\RealUp}$, together with a
$\NormUp$ norm
$$
\nu:C_V\to \RealUp
$$
whose classifier is the RealUp classifier. Its induced distance predicate is
the RealUp value
$$
d_\nu(x,y):=\nu(x-_V y)
$$
on carried vector endpoints $x,y:C_V$. A regular Cauchy completion witness for
the candidate is a row assigning to every regular Cauchy sequence
$(X,\mu):\mathbb{N}\to C_V$ for $d_\nu$ a carried limit endpoint
$\lim_\nu(X,\mu):C_V$, a convergence modulus, and a uniqueness certificate up
to $\sim_V$.

The BEDC source is $\NameCert_{\NormUp}$ together with the real-completeness
capstone of \autoref{ch:capstones-real-completeness}. The carrier choice is
$C_V$, the classifier is $\sim_V$, the stability obligations are norm transport
and Cauchy-sequence transport across $\sim_V$, the ledger records all moduli and
proof-index data used by the norm and completion rows, and the status is
$\mathsf{ProofTarget}$.
\end{definition}

\begin{definition}[Singleton empty-history Banach carrier]
\label{def:banach-singleton-empty-history-instance}
The concrete $\BanachUp$ carrier is the singleton empty-history norm instance
of \autoref{def:singleton-empty-history-norm-instance}. Its vector carrier is
$A_{\emp}$, its vector classifier is $\rho_{\emp}$, its norm is
$\nu_{\emp}$, and its metric distance is the constant RealUp zero endpoint
$$
d_{\emp}(m,n):=0_{\mathsf{Real}}^{c}
$$
for all displayed vector histories $m,n:\mathsf{BHist}$. The metric
classifier on all displayed distances is
$\mathsf{RealConstantHistoryClassifier}$.

A certified regular Cauchy stream $s$ in this metric is assigned the limit
$\emp$. Its convergence witness sends every requested real precision to the
same singleton carrier endpoint: each term of $s$ is $\rho_{\emp}$-classified
with $\emp$, and each displayed distance to the limit is
$\mathsf{RealConstantHistoryClassifier}$-classified with
$0_{\mathsf{Real}}^{c}$.
\end{definition}

\begin{theorem}[Banach norm-induced metric compatibility]
\label{thm:banach-norm-induced-metric-compatibility}
For the singleton empty-history $\NormUp$ instance of
\autoref{def:singleton-empty-history-norm-instance}, read through the
singleton metric carrier and the relation
$\mathsf{MetricDistanceWitness}$, the induced Banach metric row is the
constant RealUp zero endpoint row. Concretely, for carried vector histories
$m,n:\mathsf{BHist}$, the Banach distance between $m$ and $n$ is classified by
$0_{\mathsf{Real}}^{c}$ exactly when
$$
\rho_{\emp}(m,n)
$$
holds.
\end{theorem}
\begin{proof}
First identify the singleton vector endpoints. By
\autoref{thm:singleton-empty-history-norm-carrier-classifier}, any two carried
vectors in the singleton norm instance are related by $\rho_{\emp}$, and their
constant norm endpoints are related by the RealUp constant-history
classifier.

The zero side is exact by
\autoref{thm:singleton-empty-history-norm-zero-exactness}. Thus the displayed
norm value of the singleton difference is classified by
$0_{\mathsf{Real}}^{c}$ exactly when the underlying vector endpoint is
classified with the singleton zero vector.

Finally read the same endpoint row through the metric carrier. The metric row
with empty distance is equivalent, by
\autoref{thm:metric-distance-witness-empty-distance-iff}, to both endpoints
being $\hsame$ to $\emp$. The singleton vector classifier is precisely the
collapse relation on those carried endpoints, so the norm-zero row and the
empty-distance metric row classify the same pairs $m,n$.
\end{proof}

The vector endpoint collapse needed by downstream certificates is exactly

\end{proof}

\begin{theorem}[Singleton empty-history Banach Cauchy completeness]
\label{thm:banach-singleton-empty-history-cauchy-completeness}
Every regular Cauchy sequence in the singleton empty-history Banach carrier of
\autoref{def:banach-singleton-empty-history-instance} has the constant limit
$\emp$. For every carried term $s_i$ in the sequence, the convergence rows are
$$
\rho_{\emp}(s_i,\emp)
\quad\text{and}\quad
\mathsf{RealConstantHistoryClassifier}(d_{\emp}(s_i,\emp),0_{\mathsf{Real}}^{c}).
$$
\end{theorem}
\begin{proof}
The regular Cauchy hypothesis supplies carried terms in the norm carrier. By
\autoref{thm:singleton-empty-history-norm-laws}, that norm carrier is closed
over the singleton vector-space endpoint data and has the constant-zero norm
law package.

Apply \autoref{thm:singleton-empty-history-vecspace-endpoint-collapse} to each
carried term $s_i$ and to the singleton endpoint $\emp$. This gives
$\rho_{\emp}(s_i,\emp)$, so the chosen limit is accepted by the vector
classifier without any search over candidate limits.

It remains to verify the metric convergence rows. Unfold $d_{\emp}$: every
distance to the limit is definitionally $0_{\mathsf{Real}}^{c}$. The zero
exactness theorem
\autoref{thm:singleton-empty-history-norm-zero-exactness} supplies the same
constant-real zero boundary used by the norm, and the RealUp constant
classifier gives
$\mathsf{RealConstantHistoryClassifier}(d_{\emp}(s_i,\emp),0_{\mathsf{Real}}^{c})$.
\end{proof}

\begin{theorem}[Banach regular Cauchy limit witness]
\label{thm:banach-regular-cauchy-limit-witness}
Every regular Cauchy stream carried by the singleton norm-induced Banach
metric has a row
$$
\mathsf{CompleteMetricLimitWitness}(C_{\emp},s,\emp,\emp,L_{\emp})
$$
whose selected limit is $\emp$, whose modulus history is $\emp$, and whose
late-observation ledgers are the empty-distance rows transported through
\autoref{thm:banach-norm-induced-metric-compatibility}.
\end{theorem}
\begin{proof}
Unfold the carried observations of the singleton norm-induced metric. Each
observation of the regular Cauchy stream is a carried singleton vector
history, hence collapses to $\emp$ by the singleton vector classifier.

Use the selected singleton limit. The complete-metric singleton construction
of \autoref{thm:singleton-complete-metric-laws} supplies the carried limit
$\emp$, the empty modulus, and the convergence ledger with rows
$\mathsf{MetricDistanceWitness}(s(n),\emp,\emp)$ for every late observation.

Transport those metric rows back to the norm-induced presentation. By
\autoref{thm:banach-norm-induced-metric-compatibility}, the empty-distance
metric rows are exactly the rows whose norm endpoint is classified by the
constant RealUp zero endpoint. These transported ledgers are the required
$L_{\emp}$ family, so the displayed package is a
$\mathsf{CompleteMetricLimitWitness}$ for the singleton Banach stream.
\end{proof}

\begin{theorem}[Banach limit classifier uniqueness]
\label{thm:banach-limit-classifier-uniqueness}
For any singleton normed stream $s$, any two Banach limit witnesses
$$
\mathsf{CompleteMetricLimitWitness}(C_{\emp},s,\mu_0,\ell_0,L_0)
\quad\text{and}\quad
\mathsf{CompleteMetricLimitWitness}(C_{\emp},s,\mu_1,\ell_1,L_1)
$$
have limits related by $\hsame(\ell_0,\ell_1)$, and therefore by the
singleton vector classifier $\rho_{\emp}(\ell_0,\ell_1)$.
\end{theorem}
\begin{proof}
Apply \autoref{thm:completemetric-limit-witness-uniqueness} to the two
displayed limit witnesses for the same stream. In the singleton carrier the
separated point classifier is empty-history sameness, so the uniqueness
conclusion gives $\hsame(\ell_0,\ell_1)$.

It remains to read the same relation as a vector classifier row. Both limits
are carried singleton histories in their respective witness rows. Applying
\autoref{thm:singleton-empty-history-vecspace-endpoint-collapse} to the two
carried limits collapses them to the singleton endpoint and repacks their
$\hsame$ relation as $\rho_{\emp}(\ell_0,\ell_1)$.
\end{proof}

\begin{theorem}[Norm-induced metric completeness]
\label{thm:banach-norm-metric-completeness}
For every concrete Banach carrier $X$ of
\autoref{def:banach-concrete-carrier}, the supplied family of
$\mathsf{CompleteMetricLimitWitness}$ rows makes the norm-induced metric on
$X$ complete: every regular Cauchy $\StreamNameUp$ stream over $X$ for
$\mathsf{BanachMetricWitness}_X$ has a carried limit history, an explicit
modulus history, and late-observation distance ledgers bounded by each
positive $\RatUp$ tolerance.
\end{theorem}
\begin{proof}
Unfold the carrier tuple in \autoref{def:banach-concrete-carrier}. A regular
Cauchy stream over $X$ is already part of the input shape accepted by the
completeness field: its observations are carried by the $\NormUp$ carrier and
its pairwise distance rows are exactly the displayed
$\mathsf{BanachMetricWitness}_X$ rows obtained from norm values.

Apply the displayed completeness family to the stream. It returns the modulus
history, the carried limit history, and the late-observation ledger family
appearing in
$\mathsf{CompleteMetricLimitWitness}(X,s,\mu_s,\ell_s,L_s)$. By
\autoref{def:completemetric-limit-witness}, those rows are precisely the
complete-metric convergence data for the metric carrier.

The norm side supplies no hidden quotient step. The classifier used at the
limit is $\hsame$ transport on representatives, so the returned rows remain
inside the same BEDC history carrier and are accepted as the completeness
witness for the norm-induced metric.
\end{proof}

\begin{theorem}[Banach norm-induced metric obligations]
\label{thm:banach-metric-induced-by-norm}
For any $\BanachUp$ candidate of
\autoref{def:banach-complete-normed-carrier}, the distance predicate
$d_\nu(x,y)=\nu(x-_V y)$ satisfies the $\MetricSpaceUp$ witness obligations on
the vector carrier $C_V$: it is RealUp-valued and non-negative, it is zero
exactly on $\sim_V$-classified endpoints, it is symmetric, and it satisfies the
triangle inequality
$$
d_\nu(x,z)\le d_\nu(x,y)+d_\nu(y,z)
$$
for carried $x,y,z:C_V$.
\end{theorem}
\begin{proof}
The RealUp carrier and non-negativity rows are exactly the codomain and
non-negativity clauses of the supplied $\NormUp$ norm. For carried endpoints
$x$ and $y$, the vector-space difference $x-_V y$ is carried by the
$\VecSpaceUp$ additive stability rows, so applying $\nu$ gives a carried
RealUp distance.

Zero exactness is the positive-definiteness clause of $\NormUp$ applied to the
difference endpoint $x-_V y$. The vector-space additive inverse and zero laws
identify $x-_V y$ with $0_V$ exactly when $x\sim_V y$, and classifier transport
converts this vector statement into the displayed distance-zero statement.

For symmetry, the vector-space inverse law identifies $x-_V y$ with the
additive inverse of $y-_V x$. The $\NormUp$ absolute-homogeneity row for the
scalar $-1$ and the RealUp absolute-value row $|-1|=1$ show that the two norm
values are classified by the RealUp classifier.

For the triangle inequality, use the vector identity
$x-_V z\sim_V (x-_V y)+_V(y-_V z)$. Norm transport moves $\nu$ across this
classifier comparison, and the triangle-inequality row of $\NormUp$ bounds the
right-hand side by $\nu(x-_V y)+\nu(y-_V z)$. This is the required
$\MetricSpaceUp$ triangle row for $d_\nu$.
\end{proof}

\begin{theorem}[Banach classifier transport obligation]
\label{thm:banach-classifier-transport-obligation}
For any $\BanachUp$ candidate of
\autoref{def:banach-complete-normed-carrier}, if $x\sim_V x'$ and
$y\sim_V y'$ with all four endpoints carried by $C_V$, then the norm values and
induced distances transport:
$$
\mathsf{RealClassifier}(\nu(x),\nu(x'))
$$
and
$$
\mathsf{RealClassifier}(d_\nu(x,y),d_\nu(x',y')).
$$
\end{theorem}
\begin{proof}
The first displayed row is the norm-transport stability obligation contained
in the $\NormUp$ certificate. It reads the vector classifier comparison
$x\sim_V x'$ and returns the RealUp classifier comparison between the two norm
values.

For the distance row, apply the $\VecSpaceUp$ congruence laws to the two input
classifier comparisons. They transport additive inverse and addition, giving
$x-_V y\sim_V x'-_V y'$ in the vector classifier.

Applying the same norm-transport row to this difference comparison yields
$\mathsf{RealClassifier}(\nu(x-_V y),\nu(x'-_V y'))$. Unfolding the definition
of $d_\nu$ on both sides gives the stated distance transport.
\end{proof}

\begin{theorem}[Banach regular Cauchy completion obligation]
\label{thm:banach-regular-cauchy-completion-obligation}
For any $\BanachUp$ candidate of
\autoref{def:banach-complete-normed-carrier}, every regular Cauchy sequence
$(X,\mu):\mathbb{N}\to C_V$ for the induced metric $d_\nu$ has the supplied
limit endpoint $\lim_\nu(X,\mu):C_V$. The endpoint comes with an explicit
convergence modulus for $d_\nu$, and any two endpoints satisfying the same
regular Cauchy limit specification are classified by $\sim_V$.
\end{theorem}
\begin{proof}
The regular Cauchy input contains a carried vector endpoint $X_n$ at each
index and a modulus $\mu$ witnessing that the distances
$d_\nu(X_m,X_n)$ enter every requested positive RealUp tolerance after
$\mu(\varepsilon)$. By
\autoref{thm:banach-metric-induced-by-norm}, these distances are legitimate
$\MetricSpaceUp$ distances.

The completion row in the carrier definition is then applied directly to
$(X,\mu)$. It returns the carried endpoint $\lim_\nu(X,\mu)$, the convergence
modulus, and the ledger entries recording the modulus calls and the norm rows
used to read each distance.

If $p$ and $q$ are two carried endpoints satisfying the same limit
specification, the real-completeness capstone supplies uniqueness of the
RealUp distance-to-zero limit for the two difference sequences
$X_n-_V p$ and $X_n-_V q$. The vector-space subtraction identity rewrites
$p-_V q$ through those two tails, and the positive-definiteness row of
$\NormUp$ reflects distance zero back to $p\sim_V q$.
\end{proof}

\begin{theorem}[Banach zero-sequence limit obligation]
\label{thm:banach-zero-sequence-limit-obligation}
For any $\BanachUp$ candidate of
\autoref{def:banach-complete-normed-carrier}, the constant sequence
$Z_n=0_V$ is regular Cauchy for the induced metric $d_\nu$, and its supplied
completion endpoint is classified with the vector zero:
$$
\lim_\nu(Z,\mu_0)\sim_V 0_V
$$
for the constant zero modulus $\mu_0$.
\end{theorem}
\begin{proof}
The vector zero endpoint is carried by the $\VecSpaceUp$ zero row, so the
constant sequence $Z_n=0_V$ is a sequence in $C_V$. For any pair of indices,
$Z_m-_V Z_n$ is classified with $0_V$ by the additive identity and inverse
laws.

Positive definiteness and norm transport convert that vector zero comparison
into RealUp distance zero. Thus every requested tolerance is met by the
constant modulus $\mu_0$, and the sequence is regular Cauchy for $d_\nu$.

Apply
\autoref{thm:banach-regular-cauchy-completion-obligation} to obtain the
completion endpoint and its convergence modulus. The endpoint $0_V$ itself
satisfies the same limit specification by the preceding distance-zero row, so
the uniqueness part of the completion obligation classifies
$\lim_\nu(Z,\mu_0)$ with $0_V$.
\end{proof}

The data of a $\BanachUp$ candidate in

\begin{proof}
The carrier and classifier fields are copied from the underlying

\autoref{thm:banach-metric-induced-by-norm}, the expression

$\MetricSpaceUp$.

The stability fields are the transport rows of
\autoref{thm:banach-classifier-transport-obligation}. They show that both
norm values and induced distances are invariant under the vector classifier,
which is the exact endpoint-respect condition required by the naming
certificate.

The completion fields are supplied by
\autoref{thm:banach-regular-cauchy-completion-obligation}. The special zero
row of \autoref{thm:banach-zero-sequence-limit-obligation} anchors the

Combining these rows gives the ten-field fit: traditional target Banach

carrier $C_V$; classifier $\sim_V$; stability norm and distance transport;

\begin{theorem}[$\hsame$ transport of Banach carriers]
\label{thm:banach-hsame-transport}
Let $X$ be a concrete Banach carrier. If carried representatives, norm rows,
metric rows, regular Cauchy stream observations, modulus histories, limit
histories, and late-observation ledgers are transported componentwise along
$\hsame$, then the transported package is again a concrete Banach carrier.
The transported carrier preserves the $\NormUp$ classifier fields and the
complete-metric limit-witness fields.
\end{theorem}
\begin{proof}
Unpack the Banach carrier as the tuple of \autoref{def:banach-concrete-carrier}.
The first component is the $\NameCert_{\NormUp}$ carrier package, so its
carrier, classifier, norm value, and algebraic law rows transport along the
given $\hsame$ histories by the stability fields of that certificate.

The metric row is induced from the transported norm row. Transport the two
endpoints, the displayed difference history, and the $\RealUp$ norm value;
then repack these transported ledgers as
$\mathsf{BanachMetricWitness}_X$ for the transported representatives.

For the completeness component, apply
\autoref{thm:completemetric-limit-witness-stability}. The hypotheses named in
the theorem are exactly the transported stream observations, modulus history,
limit history, thresholds, distance histories, and $\RatUp$ comparison rows.
The result is a transported
$\mathsf{CompleteMetricLimitWitness}(X,s',\mu',\ell',L')$.

Combining the transported norm package, the induced metric row, and the
transported complete-metric witness family gives the transported concrete
Banach carrier. Since every field was moved by $\hsame$ and then repacked
componentwise, the norm and limit-witness fields are preserved exactly in the
paper-side classifier.
\end{proof}

\begin{theorem}[Complete-metric witness exactness for Banach carriers]
\label{thm:banach-complete-metric-witness-exactness}
For a concrete Banach carrier $X$, the displayed predicate
$$
\mathsf{CompleteMetricLimitWitness}(X,s,\mu,\ell,L)
$$
is exact for regular Cauchy streams over the norm-induced metric. Coverage
holds because every accepted regular Cauchy stream receives such a row from
the Banach completeness family. Soundness holds because any displayed row
contains carried observations, a carried limit, metric-distance rows, and
$\RatUp$ convergence bounds. Completeness holds because the real-completeness
capstone supplies the diagonal regular-Cauchy limit rows consumed by
$\NameCert_{\CompleteMetricUp}$.
\end{theorem}
\begin{proof}
For coverage, start with an accepted regular Cauchy stream over $X$. The last
field of \autoref{def:banach-concrete-carrier} assigns the histories
$\mu_s,\ell_s,L_s$ and the row
$\mathsf{CompleteMetricLimitWitness}(X,s,\mu_s,\ell_s,L_s)$, so the displayed
predicate covers the stream.

For soundness, unfold \autoref{def:completemetric-limit-witness}. A displayed
row contains the carried-observation ledger, the explicit modulus, the carried
limit, the threshold history for each positive tolerance, and the metric
distance and $\RatUp$ comparison rows for every late observation. Since the
metric row is $\mathsf{BanachMetricWitness}_X$, each distance is a norm value
of a difference history supplied by $\NameCert_{\NormUp}$.

For completeness of the displayed predicate, use the real-completeness
capstone through the $\CompleteMetricUp$ interface. The capstone supplies the
diagonal regular-Cauchy limit construction, while
\autoref{thm:completemetric-limit-witness-namecert-target} packages the
stream, modulus, limit, metric-distance rows, and rational bounds as a
complete-metric witness. Replacing the metric-distance rows by their
norm-induced representatives gives the Banach witness without changing the
BEDC histories being transported.
\end{proof}

\begin{theorem}[Banach NameCert obligation package]
\label{thm:banach-namecert-obligation-package}
The data of \autoref{def:banach-concrete-carrier}, together with
\autoref{thm:banach-norm-metric-completeness},
\autoref{thm:banach-hsame-transport}, and
\autoref{thm:banach-complete-metric-witness-exactness}, assemble the
paper-side $\NameCert_{\BanachUp}$ obligation package. Its fields are the
$\NormUp$ carrier, the $\hsame$ classifier on representatives, the
norm-induced metric ledger, the complete-metric limit-witness family, the
transport stability clauses, the exactness clauses for displayed regular
Cauchy limits, and the visible ledger policy inherited from
$\NameCert_{\NormUp}$ and $\NameCert_{\CompleteMetricUp}$.
\end{theorem}
\begin{proof}
The carrier field is the predicate $X:\Hist\to\mathsf{Prop}$ equipped with
the displayed $\NormUp$ data and the induced
$\mathsf{BanachMetricWitness}_X$ relation of
\autoref{def:banach-concrete-carrier}. The classifier field is $\hsame$
transport on representatives, so the certificate compares histories by
visible representative transport rather than by a host quotient.

The stability field is supplied by
\autoref{thm:banach-hsame-transport}. It moves the norm rows, metric rows,
stream rows, modulus histories, limit histories, and late-observation ledgers
componentwise and then repacks them as the same Banach carrier structure.

The completeness and exactness fields are supplied by
\autoref{thm:banach-norm-metric-completeness} and
\autoref{thm:banach-complete-metric-witness-exactness}. These theorems show
that every regular Cauchy stream over the norm-induced metric has a displayed
$\mathsf{CompleteMetricLimitWitness}$ row, and that every such row has exactly
the carried observations, carried limit, norm-distance rows, and rational
comparison bounds demanded by $\NameCert_{\CompleteMetricUp}$.

The ledger field is the union of the visible rows already named by
$\NameCert_{\NormUp}$ with the stream, modulus, threshold, distance, and
comparison rows of $\NameCert_{\CompleteMetricUp}$. The real-completeness
capstone supplies the limit construction consumed by the complete-metric
side, so all fields of the $\NameCert_{\BanachUp}$ obligation package are
accounted for constructively.
\end{proof}

The ten-field fit is as follows. The traditional target is Banach space. The
BEDC source is the singleton $\NormUp$ carrier, the $\CompleteMetricUp$
singleton limit witness, and the RealUp constant zero endpoint. The carrier is
the singleton normed history carrier. The classifier is $\rho_{\emp}$ on

\begin{theorem}[Banach norm-distance zero exactness]
\label{thm:banach-norm-distance-zero-exactness}
Let $X$ be a concrete Banach carrier and let
$\mathsf{BanachMetricWitness}_X(x,y,d)$ be a displayed norm-induced distance
row. If the $\RealUp$ classifier identifies $d$ with the zero real endpoint,
then the Banach point classifier identifies the carried representatives $x$
and $y$.
\end{theorem}
\begin{proof}
Unfold the carrier in \autoref{def:banach-concrete-carrier}. The metric row
is not independent data: it records carried representatives $x$ and $y$, a
displayed difference history, and the $\NormUp$ value of that difference as
the real endpoint $d$.

Use the input RealUp zero classification to read the norm value of the
difference as zero. The positive-definiteness row of $\NormUp$ then reflects
this zero norm back to the vector-history classifier, so the displayed
difference history is classified with the vector zero.

The vector-space zero-difference law turns that classifier row into equality
of the two carried representatives in the Banach point classifier. Repacking
the carried representatives, difference row, and norm ledger along $\hsame$
gives the required classifier comparison between $x$ and $y$.
\end{proof}

\begin{theorem}[Banach complete-metric limit uniqueness]
\label{thm:banach-limit-uniqueness}
Let $X$ be a concrete Banach carrier, and let $s$ be a regular Cauchy
$\StreamNameUp$ stream over $X$ for the norm-induced metric. If
$$
\mathsf{CompleteMetricLimitWitness}(X,s,\mu,\ell,L)
$$
and
$$
\mathsf{CompleteMetricLimitWitness}(X,s,\mu',\ell',L')
$$
are two displayed limit rows for the same stream, then the Banach point
classifier identifies the carried limit histories $L$ and $L'$.
\end{theorem}
\begin{proof}
First read both displayed rows through
\autoref{def:completemetric-limit-witness}. They give carried limit histories
$L$ and $L'$, convergence moduli, and norm-induced late-distance rows from
the same regular Cauchy stream to each candidate limit.

The previous theorem supplies the separated-metric zero-distance field for
the Banach metric: whenever a displayed
$\mathsf{BanachMetricWitness}_X(u,v,d)$ has zero RealUp distance, the Banach
point classifier identifies $u$ and $v$.

Apply
\autoref{thm:completemetric-limit-witness-uniqueness} to the two
complete-metric limit rows using that separatedness field. The complete-metric
uniqueness theorem returns classifier equality of the two limit histories, and
the classifier is exactly the $\hsame$-based Banach point classifier fixed in
\autoref{def:banach-concrete-carrier}.
\end{proof}

\begin{theorem}[Banach regular Cauchy stream transport]
\label{thm:banach-cauchy-stream-transport}
Let $X$ be a concrete Banach carrier. Componentwise $\hsame$ transport of a
regular Cauchy $\StreamNameUp$ stream over $X$, its observation rows, modulus
histories, candidate limit histories, norm-distance rows, and $\RatUp$
comparison ledgers preserves the displayed Banach complete-metric witness
package.
\end{theorem}
\begin{proof}
Transport the Banach carrier fields first. By
\autoref{thm:banach-hsame-transport}, the carried representatives, norm rows,
metric rows, stream observations, modulus histories, limit histories, and
late-observation ledgers remain a concrete Banach carrier package after
componentwise $\hsame$ transport.

Next transport the complete-metric row. Apply
\autoref{thm:completemetric-limit-witness-stability} to the transported
$\StreamNameUp$ observations, modulus history, candidate limit, distance
histories, and $\RatUp$ comparison rows. Its conclusion is the transported
$\mathsf{CompleteMetricLimitWitness}$ row for the same norm-induced metric
shape.

Finally combine the two transported packages. The Banach transport theorem
keeps the metric rows induced by the transported norm ledgers, and the
complete-metric stability theorem keeps the regular Cauchy and convergence
fields aligned with those rows. Repacking the fields gives the preserved
Banach complete-metric witness package.
\end{proof}
