\chapter{A Concrete Naming Certificate for $\HashUp$}
\label{ch:concrete-instances-hash-namecert}

The hash-function interface $\HashUp$ packages a deterministic finite-output map with collision-resistance, preimage-resistance, and second-preimage-resistance certificates against a bounded adversary. It depends on $\NameCert_{\FuncUp}$ (\autoref{ch:concrete-instances-funcobj-namecert}) and $\NameCert_{\FinSetUp}$ (\autoref{ch:concrete-instances-finset-namecert}).

\begin{definition}[Hash evaluation and success transcripts]
\label{def:hash-evaluation-success-transcripts}
Let $\mathcal H$ be a $\HashUp$ certificate as in \autoref{sec:hash-carrier} and \autoref{sec:hash-certificate}. Its carrier data consist of a message carrier $\mathsf{Msg}_{\mathcal H}$ with message classifier $\sigma_{\mathcal H}$, a finite digest carrier $\mathsf{Dig}_{\mathcal H}$ with digest classifier $\rho_{\mathcal H}$ from the finite-output $\FinSetUp$ presentation of \autoref{ch:concrete-instances-finset-namecert}, and a deterministic function row supplied by the $\FuncUp$ substrate of \autoref{ch:concrete-instances-funcobj-namecert}. Write
$$
\mathsf{HashEval}_{\mathcal H}(m,d)
$$
for the row witness that $m$ is a carried message, $d$ is a carried digest, and the hash row of $\mathcal H$ evaluates $m$ to the digest representative $d$.

For message histories $x$ and $x'$, define successful second-preimage transcript by
$$
\mathsf{HashSecondPreimageSuccess}_{\mathcal H}(x,x')
\Longleftrightarrow
\exists d,d':\mathsf{BHist}.\;
\mathsf{HashEval}_{\mathcal H}(x,d)\land
\mathsf{HashEval}_{\mathcal H}(x',d')\land
\neg\sigma_{\mathcal H}(x,x')\land
\rho_{\mathcal H}(d',d).
$$
Define successful collision transcript for the ordered pair $(x,x')$ by
$$
\mathsf{HashCollisionSuccess}_{\mathcal H}(x,x')
\Longleftrightarrow
\exists d,d':\mathsf{BHist}.\;
\mathsf{HashEval}_{\mathcal H}(x,d)\land
\mathsf{HashEval}_{\mathcal H}(x',d')\land
\neg\sigma_{\mathcal H}(x,x')\land
\rho_{\mathcal H}(d,d').
$$
The two predicates use the same evaluation row; the digest-classifier orientation is the only difference.
\end{definition}
\leandef{BEDC.Derived.HashUp.HashEvalTranscript}

\begin{theorem}[Hash evaluation transcript determinacy]
\label{thm:hash-evaluation-transcript-determinacy}
Let $\mathcal H$ be a $\HashUp$ certificate whose hash row is the deterministic $\FuncUp$ row carried by the certificate and whose digest endpoint classifier is the finite-output classifier $\rho_{\mathcal H}$ supplied by the $\FinSetUp$ dependency. For any message history $m$ and digest histories $d$ and $d'$, two carried evaluations over the same hash row determine the same digest class:
$$
\mathsf{HashEval}_{\mathcal H}(m,d)\land
\mathsf{HashEval}_{\mathcal H}(m,d')
\Longrightarrow
\rho_{\mathcal H}(d,d').
$$
\end{theorem}
\begin{proof}
Unfold the evaluation row of \autoref{def:hash-evaluation-success-transcripts}. The two hypotheses expose the same carried message row $m$, the same deterministic $\FuncUp$ function row, and two digest endpoint rows $d$ and $d'$ produced by that row.

Project determinacy from the $\FuncUp$ certificate. Since both evaluations read the same input history through the same finite $\Cont$ ledger and the same $\Pkg$ provenance, the function-row determinacy comparison identifies the two produced endpoint histories at the digest boundary.

Now transport that endpoint comparison through the finite-output $\FinSetUp$ classifier. The digest carrier row of $\mathcal H$ is exactly the finite-output row named in \autoref{def:hash-evaluation-success-transcripts}, so classifier transport yields $\rho_{\mathcal H}(d,d')$.

Repack the evidence without changing the transcript: the message row, digest endpoint rows, $\mathsf{BHist}$ witnesses, $\Cont$ readback, and $\Pkg$ provenance are the rows already carried by the two $\mathsf{HashEval}_{\mathcal H}$ hypotheses.
\end{proof}

\begin{theorem}[Hash second-preimage success induces collision]
\label{thm:hash-second-preimage-success-induces-collision}
Let $\mathcal H$ be a $\HashUp$ certificate whose finite-output digest classifier is supplied by its $\NameCert_{\FinSetUp}$ component. For message histories $x$ and $x'$, every successful second-preimage transcript for base message $x$ and returned message $x'$ induces a successful collision transcript for the ordered pair $(x,x')$ over the same hash row:
$$
\mathsf{HashSecondPreimageSuccess}_{\mathcal H}(x,x')
\Longrightarrow
\mathsf{HashCollisionSuccess}_{\mathcal H}(x,x').
$$
\end{theorem}
\leanchecked{BEDC.Derived.HashUp.HashSecondPreimageSuccess\_induces\_collision}
\begin{proof}
Unfold \autoref{def:hash-evaluation-success-transcripts}. A second-preimage success transcript gives digest histories $d$ and $d'$ with the two evaluation row witnesses
$$
\mathsf{HashEval}_{\mathcal H}(x,d)
\qquad\text{and}\qquad
\mathsf{HashEval}_{\mathcal H}(x',d'),
$$
the message distinction witness $\neg\sigma_{\mathcal H}(x,x')$, and the digest-classifier comparison $\rho_{\mathcal H}(d',d)$. The digest classifier is the classifier of the finite-output naming certificate from \autoref{ch:concrete-instances-finset-namecert}, so the classifier-symmetry field of \autoref{def:lean-carrier-naming-certificate} reverses the displayed comparison and gives
$$
\rho_{\mathcal H}(d,d').
$$
Using the same digest witnesses $d,d'$, the same two evaluation row witnesses, and the same message distinction witness, this is exactly the collision predicate over the same $\mathsf{HashEval}_{\mathcal H}$ row in \autoref{def:hash-evaluation-success-transcripts}.
\end{proof}

\begin{theorem}[Hash collision success is symmetric]
\label{thm:hash-collision-success-symmetric}
Let $\mathcal H$ be a $\HashUp$ certificate whose message classifier $\sigma_{\mathcal H}$ and finite-output digest classifier $\rho_{\mathcal H}$ have the classifier-symmetry fields supplied by their naming-certificate components. For message histories $x$ and $x'$, every successful collision transcript for the ordered pair $(x,x')$ induces a successful collision transcript for the ordered pair $(x',x)$ over the same hash-evaluation row:
$$
\mathsf{HashCollisionSuccess}_{\mathcal H}(x,x')
\Longrightarrow
\mathsf{HashCollisionSuccess}_{\mathcal H}(x',x).
$$
\end{theorem}
\leanchecked{BEDC.Derived.HashUp.HashCollisionSuccess\_symmetric}
\begin{proof}
Unfold \autoref{def:hash-evaluation-success-transcripts}. A collision transcript for $(x,x')$ gives digest histories $d$ and $d'$ with evaluation row witnesses
$$
\mathsf{HashEval}_{\mathcal H}(x,d)
\qquad\text{and}\qquad
\mathsf{HashEval}_{\mathcal H}(x',d'),
$$
the message distinction witness $\neg\sigma_{\mathcal H}(x,x')$, and the digest comparison $\rho_{\mathcal H}(d,d')$.

The digest-classifier symmetry field from \autoref{def:lean-carrier-naming-certificate} reverses the digest comparison to
$$
\rho_{\mathcal H}(d',d).
$$
For the reversed message distinction, suppose $\sigma_{\mathcal H}(x',x)$ held. The message-classifier symmetry field would give $\sigma_{\mathcal H}(x,x')$, contradicting the displayed witness $\neg\sigma_{\mathcal H}(x,x')$. Hence
$$
\neg\sigma_{\mathcal H}(x',x).
$$
Using the swapped digest witnesses $d'$ and $d$, the two original evaluation row witnesses read in the opposite order, the reversed message distinction, and the reversed digest comparison, the definition repacks exactly as $\mathsf{HashCollisionSuccess}_{\mathcal H}(x',x)$ over the same $\mathsf{HashEval}_{\mathcal H}$ row.
\end{proof}

\begin{theorem}[Hash compression round carrier]
\label{thm:hash-compression-round-carrier}
Each hash evaluation transcript in \autoref{def:hash-evaluation-success-transcripts} is a $\mathsf{BHist}$ packet over the $\FuncUp$ deterministic row, the message carrier, the finite digest carrier, the $\Cont$ readback row, and the $\mathsf{Pkg}$ provenance row. The transcript exports no separate host bitstring computation.
\end{theorem}
\begin{proof}
Unfold \autoref{def:hash-evaluation-success-transcripts}. A row $\mathsf{HashEval}_{\mathcal H}(m,d)$ contains a carried message endpoint, a carried digest endpoint, and the hash-function row supplied by the $\FuncUp$ substrate.

Project determinacy from \autoref{thm:hash-evaluation-transcript-determinacy}. The two digest rows produced by the same message and function row are compared only through the finite digest classifier inherited from $\FinSetUp$.

Finally read the evaluation route through the displayed $\Cont$ readback and $\mathsf{Pkg}$ provenance. Repacking these rows gives the compression-round carrier as a BEDC transcript rather than as an external bitstring computation.
\end{proof}

\begin{theorem}[Hash digest classifier stability]
\label{thm:hash-digest-classifier-stability}
Transporting digest endpoints by the finite-output $\FinSetUp$ digest classifier preserves $\mathsf{HashEval}_{\mathcal H}$, $\mathsf{HashSecondPreimageSuccess}_{\mathcal H}$, and $\mathsf{HashCollisionSuccess}_{\mathcal H}$. The preservation is a classifier transport of digest rows, not an equality of host strings.
\end{theorem}
\begin{proof}
Start with an evaluation transcript and a digest-classifier edge supplied by the finite-output $\FinSetUp$ certificate. The digest endpoint remains in the digest carrier, and the same $\FuncUp$ row and message carrier row still witness the evaluation.

For second-preimage success, unfold \autoref{def:hash-evaluation-success-transcripts}. The transported digest row replaces the original digest witness while the two evaluation rows and the message-distinction witness remain the same.

For collision success, use the same unpacking together with the digest-classifier symmetry used in \autoref{thm:hash-collision-success-symmetric}. Repacking the transported digest witnesses preserves the two success predicates through the classifier boundary.
\end{proof}

\begin{theorem}[Hash collision ledger boundary]
\label{thm:hash-collision-ledger-boundary}
The collision, second-preimage, and preimage-resistance ledgers of a $\HashUp$ certificate may read only message-carrier rows, digest-carrier rows, function determinacy rows, classifier comparisons, and bounded adversary transcripts. They may not read external probability spaces, host bitstring equality, or unrecorded randomness.
\end{theorem}
\begin{proof}
Unfold the transcript definition of \autoref{def:hash-evaluation-success-transcripts}. Collision and second-preimage success are built from two evaluation rows, one message-classifier distinction, and one digest-classifier comparison.

Apply \autoref{thm:hash-second-preimage-success-induces-collision} and \autoref{thm:hash-collision-success-symmetric}. These theorem blocks use only the evaluation rows, classifier symmetry, function determinacy, and the existing transcript witnesses.

The bounded adversary surface therefore has no additional ledger row to consult. Repacking the available witnesses gives the stated boundary for collision, second-preimage, and preimage-resistance ledgers.
\end{proof}

\label{thm:hash-collision-transcript-symmetry}
Let $\mathcal H$ be a $\HashUp$ certificate whose message classifier $\sigma_{\mathcal H}$ and finite-output digest classifier $\rho_{\mathcal H}$ have the classifier-symmetry fields supplied by their naming-certificate components. For message histories $x$ and $x'$, every successful collision transcript for the ordered pair $(x,x')$ induces a successful collision transcript for the ordered pair $(x',x)$ over the same two hash-evaluation rows:
$$
\mathsf{HashCollisionSuccess}_{\mathcal H}(x,x')
\Longrightarrow
\mathsf{HashCollisionSuccess}_{\mathcal H}(x',x).
$$
\end{theorem}
\leanchecked{BEDC.Derived.HashUp.HashCollisionTranscript\_symmetric}

\mathsf{HashEval}_{\mathcal H}(x,d)
\qquad\text{and}\qquad
\mathsf{HashEval}_{\mathcal H}(x',d'),
$$

The digest-classifier symmetry field from \autoref{def:lean-carrier-naming-certificate}, applied through the finite-output $\FinSetUp$ component cited in \autoref{ch:concrete-instances-finset-namecert}, reverses the digest comparison to

\rho_{\mathcal H}(d',d).
$$
For the reversed message distinction, suppose $\sigma_{\mathcal H}(x',x)$ held. The message-classifier symmetry field from the same naming-certificate discipline would give $\sigma_{\mathcal H}(x,x')$, contradicting $\neg\sigma_{\mathcal H}(x,x')$. Hence

\end{proof}

\begin{theorem}[Hash collision success induces reversed second-preimage]
\label{thm:hash-collision-success-induces-reversed-second-preimage}
Let $\mathcal H$ be a $\HashUp$ certificate whose message classifier $\sigma_{\mathcal H}$ has the classifier-symmetry field supplied by its naming-certificate component. For message histories $x$ and $x'$, every successful collision transcript for the ordered pair $(x,x')$ induces a successful second-preimage transcript for base message $x'$ and returned message $x$ over the same two hash-evaluation rows:
$$
\mathsf{HashCollisionSuccess}_{\mathcal H}(x,x')
\Longrightarrow
\mathsf{HashSecondPreimageSuccess}_{\mathcal H}(x',x).
$$
\end{theorem}
\leanchecked{BEDC.Derived.HashUp.HashCollisionSuccess\_induces\_reversed\_second\_preimage}
\begin{proof}
Unfold \autoref{def:hash-evaluation-success-transcripts}. A collision transcript for $(x,x')$ gives digest histories $d$ and $d'$ with the two evaluation row witnesses
$$
\mathsf{HashEval}_{\mathcal H}(x,d)
\qquad\text{and}\qquad
\mathsf{HashEval}_{\mathcal H}(x',d'),
$$
the message distinction witness $\neg\sigma_{\mathcal H}(x,x')$, and the digest comparison
$$
\rho_{\mathcal H}(d,d').
$$
To build the second-preimage transcript for $(x',x)$, take the digest witness for the base message $x'$ to be $d'$ and the digest witness for the returned message $x$ to be $d$. The two displayed evaluation row witnesses then occur in the required reversed order, and the required digest condition is still
$$
\rho_{\mathcal H}(d,d'),
$$
which is the collision digest comparison itself.

It remains only to reverse the message distinction. Suppose $\sigma_{\mathcal H}(x',x)$ held. The message-classifier symmetry field from \autoref{def:lean-carrier-naming-certificate} would give $\sigma_{\mathcal H}(x,x')$, contradicting $\neg\sigma_{\mathcal H}(x,x')$. Hence
$$
\neg\sigma_{\mathcal H}(x',x).
$$
Using the swapped digest witnesses $d'$ and $d$, the same two hash-evaluation rows, the reversed message distinction, and the unchanged digest comparison, the definition repacks exactly as $\mathsf{HashSecondPreimageSuccess}_{\mathcal H}(x',x)$.
\end{proof}

\begin{definition}[Hash collision-freeness]
\label{def:hash-collision-freeness}
Let $\mathcal H$ be a $\HashUp$ certificate as in \autoref{sec:hash-carrier} and \autoref{sec:hash-certificate}. For histories $x$ and $x'$, define
$$
\mathsf{CollFreeH}_{\mathcal H}(x,x')
\Longleftrightarrow
\neg\mathsf{HashCollisionSuccess}_{\mathcal H}(x,x').
$$
\end{definition}
\leandef{BEDC.Derived.HashUp.HashCollisionFree}

\begin{theorem}[Hash collision-freeness excludes second-preimage success]
\label{thm:hash-collision-freeness-excludes-second-preimage}
Let $\mathcal H$ be a $\HashUp$ certificate whose finite-output digest classifier is supplied by its $\NameCert_{\FinSetUp}$ component. For message histories $x$ and $x'$, if
$$
\mathsf{CollFreeH}_{\mathcal H}(x,x') \land \mathsf{HashSecondPreimageSuccess}_{\mathcal H}(x,x')
$$
holds, then
$$
\bot.
$$
\end{theorem}
\begin{proof}
Unfold $\mathsf{CollFreeH}_{\mathcal H}(x,x')$, so it is enough to show $\neg\mathsf{HashCollisionSuccess}_{\mathcal H}(x,x')$.
By \autoref{thm:hash-second-preimage-success-induces-collision}, the assumed second-preimage success gives
$$
\mathsf{HashCollisionSuccess}_{\mathcal H}(x,x').
$$
This directly contradicts the displayed collision-freeness hypothesis, so the conjunction is inconsistent.
\end{proof}
\leanchecked{BEDC.Derived.HashUp.HashCollisionFree\_excludes\_second\_preimage\_success}

\begin{theorem}[Hash collision-freeness is symmetric]
\label{thm:hash-collision-freeness-symmetric}
Let $\mathcal H$ be a $\HashUp$ certificate whose message classifier $\sigma_{\mathcal H}$ and finite-output digest classifier $\rho_{\mathcal H}$ have the classifier-symmetry fields supplied by their naming-certificate components. For message histories $x$ and $x'$, if
$$
\mathsf{CollFreeH}_{\mathcal H}(x,x')
$$
holds, then
$$
\mathsf{CollFreeH}_{\mathcal H}(x',x).
$$
\end{theorem}
\begin{proof}
Assume
$$
\mathsf{CollFreeH}_{\mathcal H}(x,x').
$$
By \autoref{def:hash-collision-freeness}, this means
$$
\neg\mathsf{HashCollisionSuccess}_{\mathcal H}(x,x').
$$
To prove $\mathsf{CollFreeH}_{\mathcal H}(x',x)$, it is enough by \autoref{def:hash-collision-freeness} to show
$$
\neg\mathsf{HashCollisionSuccess}_{\mathcal H}(x',x).
$$
Suppose instead that
$$
\mathsf{HashCollisionSuccess}_{\mathcal H}(x',x)
$$
held. Then \autoref{thm:hash-collision-success-symmetric}, applied with the ordered pair $(x',x)$ in place of $(x,x')$, gives
$$
\mathsf{HashCollisionSuccess}_{\mathcal H}(x,x').
$$
This contradicts the displayed collision-freeness hypothesis. Therefore
$$
\neg\mathsf{HashCollisionSuccess}_{\mathcal H}(x',x),
$$
and another use of \autoref{def:hash-collision-freeness} yields
$$
\mathsf{CollFreeH}_{\mathcal H}(x',x).
$$
\end{proof}
\leanchecked{BEDC.Derived.HashUp.HashCollisionFree\_symmetric}

\begin{theorem}[Hash collision success is irreflexive on the message]
\label{thm:hash-collision-success-irreflexive-on-message}
Let $\mathcal H$ be a $\HashUp$ certificate whose message classifier $\sigma_{\mathcal H}$ carries the reflexivity field supplied by its naming-certificate component. For every message history $x$, if
$$
\mathsf{HashCollisionSuccess}_{\mathcal H}(x,x)
$$
holds, then
$$
\bot.
$$
\end{theorem}
\begin{proof}
Unfold \autoref{def:hash-evaluation-success-transcripts}. A collision transcript for the diagonal pair $(x,x)$ gives digest histories $d$ and $d'$ with the two evaluation row witnesses
$$
\mathsf{HashEval}_{\mathcal H}(x,d)
\qquad\text{and}\qquad
\mathsf{HashEval}_{\mathcal H}(x,d'),
$$
the message distinction witness
$$
\neg\sigma_{\mathcal H}(x,x),
$$
and the digest comparison
$$
\rho_{\mathcal H}(d,d').
$$
The first evaluation row already exhibits $x$ as a carried message of $\mathcal H$. Since the message classifier is supplied by a naming certificate, the reflexivity field from \autoref{def:lean-carrier-naming-certificate} therefore gives
$$
\sigma_{\mathcal H}(x,x).
$$
This contradicts the displayed witness $\neg\sigma_{\mathcal H}(x,x)$. Hence no diagonal collision-success transcript can exist, and the assumption implies $\bot$.
\end{proof}
\leanchecked{BEDC.Derived.HashUp.HashCollisionSuccess\_irreflexive\_on\_message}

\end{proof}

\begin{theorem}[Hash resistance success boundary]
\label{thm:hash-resistance-success-boundary}
Let $\mathcal H$ be a $\HashUp$ certificate whose finite-output digest classifier $\rho_{\mathcal H}$ and message classifier $\sigma_{\mathcal H}$ are symmetric classifier fields supplied by the naming-certificate components of the certificate. Assume the digest classifier has the symmetry field used by \autoref{thm:hash-collision-success-induces-same-direction-second-preimage}. For message histories $x$ and $x'$, the two transcript predicates over the same two evaluation rows are equivalent:
$$
\mathsf{HashSecondPreimageSuccess}_{\mathcal H}(x,x')
\Longrightarrow
\mathsf{HashCollisionSuccess}_{\mathcal H}(x,x')
$$
and
$$
\mathsf{HashCollisionSuccess}_{\mathcal H}(x,x')
\Longrightarrow
\mathsf{HashSecondPreimageSuccess}_{\mathcal H}(x,x').
$$
Thus collision-freeness and second-preimage resistance share the same transcript boundary under the BEDC classifiers.
\end{theorem}
\begin{proof}
For the forward direction, assume $\mathsf{HashSecondPreimageSuccess}_{\mathcal H}(x,x')$. Apply \autoref{thm:hash-second-preimage-success-induces-collision}. It uses the same two $\mathsf{HashEval}_{\mathcal H}$ rows, the same message-classifier distinction, and the digest comparison already carried by the second-preimage transcript, so it returns $\mathsf{HashCollisionSuccess}_{\mathcal H}(x,x')$ without changing the transcript boundary.

For the reverse direction, assume $\mathsf{HashCollisionSuccess}_{\mathcal H}(x,x')$. Apply \autoref{thm:hash-collision-success-induces-same-direction-second-preimage}. The only transport it needs is the finite-output digest-classifier symmetry field from the $\NameCert_{\FinSetUp}$ component, while the message classifier and the two evaluation rows remain the same ordered pair. This returns $\mathsf{HashSecondPreimageSuccess}_{\mathcal H}(x,x')$.

The two implications identify the successful transcript boundary. Negating the collision side or the second-preimage side therefore excludes the same successful pair of BEDC evaluation rows under the certificate classifiers.
\end{proof}

\begin{theorem}[Hash resistance negated boundary equivalence]
\label{thm:hash-resistance-equivalence-negation-boundary}
Let $\mathcal H$ be a $\HashUp$ certificate whose finite-output digest classifier $\rho_{\mathcal H}$ and message classifier $\sigma_{\mathcal H}$ are symmetric classifier fields supplied by the naming-certificate components of the certificate. Assume the digest classifier has the symmetry field used by \autoref{thm:hash-collision-success-induces-same-direction-second-preimage}. For message histories $x$ and $x'$, collision-freeness for the ordered pair is equivalent to the negation of second-preimage success:
$$
\mathsf{CollFreeH}_{\mathcal H}(x,x')
\Longleftrightarrow
\neg\mathsf{HashSecondPreimageSuccess}_{\mathcal H}(x,x').
$$
\end{theorem}
\begin{proof}
Unfold \autoref{def:hash-collision-freeness}. The displayed claim becomes the equivalence between $\neg\mathsf{HashCollisionSuccess}_{\mathcal H}(x,x')$ and $\neg\mathsf{HashSecondPreimageSuccess}_{\mathcal H}(x,x')$ under the same certificate hypotheses.

For the forward direction, assume $\mathsf{CollFreeH}_{\mathcal H}(x,x')$ and suppose $\mathsf{HashSecondPreimageSuccess}_{\mathcal H}(x,x')$ held. By \autoref{thm:hash-second-preimage-success-induces-collision}, the second-preimage transcript induces $\mathsf{HashCollisionSuccess}_{\mathcal H}(x,x')$, contradicting the unfolded collision-freeness hypothesis.

For the reverse direction, assume $\neg\mathsf{HashSecondPreimageSuccess}_{\mathcal H}(x,x')$. To show $\mathsf{CollFreeH}_{\mathcal H}(x,x')$, unfold \autoref{def:hash-collision-freeness} and suppose $\mathsf{HashCollisionSuccess}_{\mathcal H}(x,x')$ held. By \autoref{thm:hash-collision-success-induces-same-direction-second-preimage}, that collision transcript induces $\mathsf{HashSecondPreimageSuccess}_{\mathcal H}(x,x')$, contradicting the assumed negation.
\end{proof}
