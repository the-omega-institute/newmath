\chapter{A Concrete Naming Certificate for $\DeRhamUp$}
\label{ch:concrete-instances-deRham-namecert}

The de~Rham interface $\DeRhamUp$ packages the exterior-derivative chain complex on differential forms, certifying $d^2 = 0$ and $\hsame$-stability. It depends on $\NameCert_{\DiffFormUp}$ (\autoref{ch:concrete-instances-diffform-namecert}) and $\NameCert_{\CohomologyUp}$ (\autoref{ch:concrete-instances-cohomology-namecert}), and supplies the de~Rham cohomology of a manifold.

\begin{definition}[De Rham chain-complex boundary row]
\label{def:derham-chain-complex-boundary-row}
Fix a $\NameCert_{\CohomologyUp}$ cochain differential
$d_{\mathrm{dR}}:\mathsf{BHist}\to\mathsf{BHist}$. A row
$b:\mathsf{BHist}$ is accepted by the $\DeRhamUp$ chain-complex boundary
predicate when it is in the image of the same cochain differential:
$$
\mathsf{DeRhamBoundary}_{d_{\mathrm{dR}}}(b)
:\Longleftrightarrow
\exists a:\mathsf{BHist},\ \hsame(b,d_{\mathrm{dR}}(a)).
$$
The witness $a$ is the cochain preimage row. The predicate adds no quotient
row and no host-level differential-form operation beyond the displayed
cochain differential.
\end{definition}
\leandef{BEDC.Derived.DeRhamUp.DeRhamBoundary}

\begin{definition}[De Rham double exterior-derivative packet]
\label{def:derham-double-exterior-derivative-packet}
Fix a $\NameCert_{\ManifoldUp}$ source $M$ and a
$\NameCert_{\TensorProductUp}$ source $T$ as in
\autoref{def:diffform-bhist-form-carrier}. Fix a
$\NameCert_{\CohomologyUp}$ cochain differential
$d_{\mathrm{dR}}:\mathsf{BHist}\to\mathsf{BHist}$ and a zero cochain endpoint
$z_{\mathrm{dR}}:\mathsf{BHist}$. For carried $\DiffFormUp$ rows
$\omega$, $\eta$, and $\theta$, the packet
$$
\mathsf{DeRhamDoubleExterior}_{M,T,d_{\mathrm{dR}}}
(\omega,\eta,\theta,z_{\mathrm{dR}})
$$
is accepted exactly when its visible fields are the following rows:
\begin{enumerate}
  \item $\omega$, $\eta$, and $\theta$ are carried by
  $\mathsf{DiffFormBHistCarrier}_{M,T}$.
  \item The two successive exterior-input rows
  $\mathsf{DiffFormExteriorInput}_{M,T}(\omega,\eta)$ and
  $\mathsf{DiffFormExteriorInput}_{M,T}(\eta,\theta)$ are accepted.
  \item The cochain graph rows identify the carried exterior outputs with the
  same cohomology differential:
  $$
  \eta\hsame d_{\mathrm{dR}}(\omega)
  \quad\text{and}\quad
  \theta\hsame d_{\mathrm{dR}}(\eta).
  $$
  \item The cohomology differential respects $\hsame$: for all
  $a,b:\mathsf{BHist}$,
  $$
  \hsame(a,b)\Longrightarrow
  \hsame(d_{\mathrm{dR}}(a),d_{\mathrm{dR}}(b)).
  $$
  \item The de Rham square-zero row and zero endpoint row are present:
  $$
  \forall a:\mathsf{BHist},\
  \hsame(d_{\mathrm{dR}}(d_{\mathrm{dR}}(a)),z_{\mathrm{dR}})
  \quad\text{and}\quad
  \hsame(z_{\mathrm{dR}},\emp).
  $$
\end{enumerate}
The two $\DiffFormUp$ exterior-input rows provide only the carried input-output
surface of \autoref{def:diffform-exterior-derivative-visible-input-package};
the square-zero row is a $\DeRhamUp$ field read through the
$\CohomologyUp$ cochain differential.
\end{definition}
\leandef{BEDC.Derived.DeRhamUp.DeRhamDoubleExteriorPacket}

\begin{theorem}[De Rham double exterior derivative boundary]
\label{thm:derham-double-exterior-derivative-boundary}
If
$\mathsf{DeRhamDoubleExterior}_{M,T,d_{\mathrm{dR}}}
(\omega,\eta,\theta,z_{\mathrm{dR}})$ is accepted, then the carried second
exterior derivative row $\theta$ is classified as the zero cochain endpoint
and is an accepted $\DeRhamUp$ chain-complex boundary row:
$$
\hsame(\theta,z_{\mathrm{dR}})
\quad\wedge\quad
\mathsf{DeRhamBoundary}_{d_{\mathrm{dR}}}(\theta).
$$
Moreover the intermediate carried row is a $\CohomologyUp$ cocycle:
$$
C_{d_{\mathrm{dR}}}(\eta).
$$
\end{theorem}
\leanchecked{BEDC.Derived.DeRhamUp.DeRhamDoubleExteriorPacket\_boundary}
\leanvariant{BEDC.Derived.DeRhamUp.DeRhamDoubleExteriorDerivative\_boundary}
\begin{proof}
Unpack the accepted packet of
\autoref{def:derham-double-exterior-derivative-packet}. The two accepted
$\DiffFormUp$ exterior-input rows are successive by
\autoref{def:diffform-exterior-derivative-visible-input-package}: the first
has source $\omega$ and output $\eta$, while the second has source $\eta$ and
output $\theta$. By \autoref{thm:diffform-exterior-derivative-degree-successor-ledger},
these rows expose only the one-step exterior-input ledgers. The nilpotence row
therefore comes from the de Rham packet, not from the $\DiffFormUp$ boundary
exhaustion of \autoref{thm:diffform-downstream-boundary-exhaustion}.

Let the graph comparisons be
$\eta\hsame d_{\mathrm{dR}}(\omega)$ and
$\theta\hsame d_{\mathrm{dR}}(\eta)$. Applying the $\hsame$-respect row of
$d_{\mathrm{dR}}$ to the first comparison gives
$$
\hsame(d_{\mathrm{dR}}(\eta),d_{\mathrm{dR}}(d_{\mathrm{dR}}(\omega))).
$$
The square-zero row at $\omega$ gives
$$
\hsame(d_{\mathrm{dR}}(d_{\mathrm{dR}}(\omega)),z_{\mathrm{dR}}).
$$
Transitivity with the second graph comparison yields
$\hsame(\theta,z_{\mathrm{dR}})$.

For the boundary predicate, use the cochain preimage $\eta$. The second graph
comparison is exactly the witness
$\hsame(\theta,d_{\mathrm{dR}}(\eta))$ required by
\autoref{def:derham-chain-complex-boundary-row}. Hence
$\mathsf{DeRhamBoundary}_{d_{\mathrm{dR}}}(\theta)$ holds.

Finally, the same differential comparison, the square-zero row at $\omega$,
and the zero endpoint comparison $\hsame(z_{\mathrm{dR}},\emp)$ give
$d_{\mathrm{dR}}(\eta)\hsame\emp$. By
\autoref{def:cohomology-cocycle-predicate}, this is exactly
$C_{d_{\mathrm{dR}}}(\eta)$.
\end{proof}

\begin{theorem}[De Rham zero-endpoint boundary transport]
\label{thm:derham-boundary-zero-endpoint-transport}
Fix a $\NameCert_{\CohomologyUp}$ cochain differential
$d_{\mathrm{dR}}:\mathsf{BHist}\to\mathsf{BHist}$. If a displayed row
$b:\mathsf{BHist}$ is accepted by
$\mathsf{DeRhamBoundary}_{d_{\mathrm{dR}}}$ through a cochain preimage
$a:\mathsf{BHist}$ and a transported endpoint $b':\mathsf{BHist}$ satisfies
$\hsame(b',b)$, then the transported row is again an accepted de Rham
boundary row:
$$
\mathsf{DeRhamBoundary}_{d_{\mathrm{dR}}}(b')
$$
with the same preimage witness $a$. In particular, if $b$ is compared by
$\hsame$ to the zero endpoint $z_{\mathrm{dR}}$, any displayed endpoint
transported to that zero endpoint remains in the same boundary predicate.
\end{theorem}
\leanchecked{BEDC.Derived.DeRhamUp.DeRhamBoundary\_zero\_endpoint\_hsame\_transport}
\leanvariant{BEDC.Derived.DeRhamUp.DeRhamBoundary\_zero\_endpoint\_transport}
\begin{proof}
Unpack \autoref{def:derham-chain-complex-boundary-row}. The accepted boundary
row supplies a witness $a$ and a comparison
$\hsame(b,d_{\mathrm{dR}}(a))$.

The transported endpoint supplies $\hsame(b',b)$. By transitivity of
$\hsame$, the row $b'$ compares with $d_{\mathrm{dR}}(a)$:
$$
\hsame(b',d_{\mathrm{dR}}(a)).
$$

Repack the same cochain preimage $a$ in
\autoref{def:derham-chain-complex-boundary-row}. No new differential-form row,
quotient row, or cohomology carrier is introduced; only the displayed endpoint
has been transported along $\hsame$.
\end{proof}

\begin{theorem}[De Rham cocycle readback from square-zero packets]
\label{thm:derham-cocycle-readback}
Every accepted packet
$$
\mathsf{DeRhamDoubleExterior}_{M,T,d_{\mathrm{dR}}}
(\omega,\eta,\theta,z_{\mathrm{dR}})
$$
reads its intermediate carried row $\eta$ as a $\CohomologyUp$ cocycle:
$$
C_{d_{\mathrm{dR}}}(\eta).
$$
The readback uses only the packet's cochain graph row
$\eta\hsame d_{\mathrm{dR}}(\omega)$, the square-zero row for
$d_{\mathrm{dR}}$, and the zero endpoint comparison
$\hsame(z_{\mathrm{dR}},\emp)$.
\end{theorem}
\begin{proof}
Unpack \autoref{def:derham-double-exterior-derivative-packet}. The graph row
identifies the intermediate carried row with the first cochain differential:
$\eta\hsame d_{\mathrm{dR}}(\omega)$.

Apply the $\hsame$-respect field of the cochain differential to this graph
row. It gives a comparison from $d_{\mathrm{dR}}(\eta)$ to
$d_{\mathrm{dR}}(d_{\mathrm{dR}}(\omega))$.

The square-zero row at $\omega$ compares
$d_{\mathrm{dR}}(d_{\mathrm{dR}}(\omega))$ with $z_{\mathrm{dR}}$, and the
zero endpoint comparison identifies $z_{\mathrm{dR}}$ with $\emp$. Chaining
these $\hsame$ comparisons yields $d_{\mathrm{dR}}(\eta)\hsame\emp$.

By \autoref{def:cohomology-cocycle-predicate}, this is exactly the
$\CohomologyUp$ cocycle predicate $C_{d_{\mathrm{dR}}}(\eta)$.
\end{proof}

\begin{theorem}[De Rham NameCert obligation surface]
\label{thm:derham-namecert-obligation-surface}
The de Rham rows in this chapter expose the
$\NameCert_{\DeRhamUp}$ obligation surface over $\DiffFormUp$ and
$\CohomologyUp$. The carrier is the differential-form carried row supplied by
\autoref{def:diffform-bhist-form-carrier}; the classifier is the boundary
predicate of \autoref{def:derham-chain-complex-boundary-row}; the stability
field is the endpoint transport of
\autoref{thm:derham-boundary-zero-endpoint-transport}; the exactness row is
the square-zero boundary theorem
\autoref{thm:derham-double-exterior-derivative-boundary}; and the ledger row
is the cocycle readback of \autoref{thm:derham-cocycle-readback}. Together
these rows supply the carrier, classifier, stability, exactness, and ledger
entries required by the named de Rham certificate surface, without asserting a
de Rham isomorphism or any quotient-level cohomology presentation.
\end{theorem}
\leanchecked{BEDC.Derived.DeRhamUp.DeRhamBoundary\_semanticNameCert}
\begin{proof}
The carrier component is inherited from the visible $\DiffFormUp$ form
carrier cited in the statement. The De Rham boundary row then classifies the
second exterior derivative by a concrete cochain preimage in
\autoref{def:derham-chain-complex-boundary-row}.

Stability is supplied by
\autoref{thm:derham-boundary-zero-endpoint-transport}: once a boundary row is
witnessed by a preimage, endpoint transport along $\hsame$ preserves the same
predicate using that witness.

Exactness is supplied by
\autoref{thm:derham-double-exterior-derivative-boundary}. It turns an accepted
double exterior packet into both the zero endpoint comparison for the second
row and the boundary predicate for that row.

The ledger is the readback
\autoref{thm:derham-cocycle-readback}. It records that the intermediate row
seen by the square-zero packet is a $\CohomologyUp$ cocycle. These named rows
are precisely the advertised certificate obligations and no additional bridge
or quotient surface is used.
\end{proof}

\begin{definition}[De Rham boundary source packet]
\label{def:derham-boundary-source-packet}
Fix a $\NameCert_{\CohomologyUp}$ cochain differential
$d_{\mathrm{dR}}:\mathsf{BHist}\to\mathsf{BHist}$ and a zero endpoint
$z_{\mathrm{dR}}:\mathsf{BHist}$. A De Rham boundary source packet consists of
the following $\mathsf{BHist}$ rows:
\begin{enumerate}
  \item the cochain differential row $d_{\mathrm{dR}}$ and its zero endpoint
  comparison $\hsame(z_{\mathrm{dR}},\emp)$;
  \item carried $\DiffFormUp$ input rows $\omega$, $\eta$, and $\theta$ with
  the two exterior-input rows of
  \autoref{def:derham-double-exterior-derivative-packet};
  \item the $\CohomologyUp$ cocycle row $C_{d_{\mathrm{dR}}}(\eta)$ read from
  the intermediate carried row;
  \item the boundary predicate
  $\mathsf{DeRhamBoundary}_{d_{\mathrm{dR}}}(\theta)$ of
  \autoref{def:derham-chain-complex-boundary-row};
  \item the finite ledger containing the cochain graph rows
  $\eta\hsame d_{\mathrm{dR}}(\omega)$ and
  $\theta\hsame d_{\mathrm{dR}}(\eta)$, the square-zero row for
  $d_{\mathrm{dR}}$, and the displayed $\DiffFormUp$ and $\CohomologyUp$
  source rows.
\end{enumerate}
The packet exports only these boundary-source rows. It does not add a quotient
cohomology carrier, a de Rham isomorphism row, or an ambient manifold-level
equality witness.
\end{definition}
\leandef{BEDC.Derived.DeRhamUp.DeRhamBoundarySourceLedgerPacket}

\begin{theorem}[De Rham boundary source stability]
\label{thm:derham-boundary-source-stability}
Let $P$ be a De Rham boundary source packet as in
\autoref{def:derham-boundary-source-packet}. If the visible boundary endpoint
$\theta$ is transported to $\theta'$ by $\hsame(\theta',\theta)$, then the same
packet data transport to a boundary source packet with boundary endpoint
$\theta'$. The cochain differential, zero endpoint row, $\DiffFormUp$ input
rows, $\CohomologyUp$ cocycle row, and finite ledger rows are unchanged except
for replacing the displayed boundary endpoint by $\theta'$.
\end{theorem}
\leanchecked{BEDC.Derived.DeRhamUp.DeRhamBoundarySourcePacket\_stability}
\begin{proof}
Unpack the packet. Its boundary component is
$\mathsf{DeRhamBoundary}_{d_{\mathrm{dR}}}(\theta)$, witnessed by the cochain
preimage row named in \autoref{def:derham-chain-complex-boundary-row}.

Apply \autoref{thm:derham-boundary-zero-endpoint-transport} to the displayed
comparison $\hsame(\theta',\theta)$. The theorem keeps the same preimage
witness and gives
$\mathsf{DeRhamBoundary}_{d_{\mathrm{dR}}}(\theta')$.

All other packet rows are source or ledger rows that do not mention the new
endpoint except through the boundary predicate just transported. Repacking the
unchanged cochain differential, zero endpoint row, $\DiffFormUp$ inputs,
cocycle row, and finite ledger with the transported boundary predicate gives
the claimed packet.
\end{proof}

\begin{theorem}[De Rham boundary consumer exactness]
\label{thm:derham-boundary-consumer-exactness}
Every downstream consumer of a De Rham boundary source packet reads exactly the
boundary row of \autoref{def:derham-chain-complex-boundary-row}, the zero
endpoint row of \autoref{def:derham-double-exterior-derivative-packet}, the
cocycle row supplied by \autoref{thm:derham-cocycle-readback}, and the finite
ledger rows displayed in \autoref{def:derham-boundary-source-packet}. The packet
therefore supplies no additional cohomology quotient, no de Rham isomorphism
row, and no hidden differential-form operation beyond the accepted
$\mathsf{BHist}$ boundary-source data.
\end{theorem}
\begin{proof}
Start with an accepted packet. Its boundary row is part of the data in
\autoref{def:derham-boundary-source-packet}, and
\autoref{thm:derham-double-exterior-derivative-boundary} identifies the
accepted double exterior packet as the source of that boundary predicate.

The zero endpoint row is also explicit: it is the comparison
$\hsame(z_{\mathrm{dR}},\emp)$ carried by
\autoref{def:derham-double-exterior-derivative-packet}. A consumer can read it
only through that packet field and through endpoint transport by
\autoref{thm:derham-boundary-zero-endpoint-transport}.

The cocycle component is fixed by \autoref{thm:derham-cocycle-readback}. It
reads the intermediate row $\eta$ as $C_{d_{\mathrm{dR}}}(\eta)$ from the
cochain graph row and the square-zero row, so the consumer receives exactly the
displayed $\CohomologyUp$ cocycle row.

Finally, inspect the finite ledger in
\autoref{def:derham-boundary-source-packet}. It lists the two cochain graph
rows, the square-zero row, and the visible $\DiffFormUp$ and $\CohomologyUp$
source rows. Since the definition lists no quotient, isomorphism, or additional
operation row, every downstream readback is exhausted by those rows.
\end{proof}

\begin{theorem}[De Rham root carrier obligation]
\label{thm:derham-root-carrier-obligation}
The root carrier obligation exposed by $\DeRhamUp$ is exactly the displayed
$\DiffFormUp$ row together with the $\mathsf{BHist}$ cochain differential
packet. Concretely, the carrier reads a differential-form row
$\omega:\mathsf{BHist}$ through the $\DiffFormUp$ source, a cochain
differential $d_{\mathrm{dR}}:\mathsf{BHist}\to\mathsf{BHist}$, and the
boundary preimage row of \autoref{def:derham-chain-complex-boundary-row};
the square-zero data are the packet of
\autoref{def:derham-double-exterior-derivative-packet}. No host manifold
carrier, quotient carrier, or external cochain complex carrier is part of the
root surface.
\end{theorem}
\begin{proof}
Read the carrier side of
\autoref{thm:derham-namecert-obligation-surface}. Its visible source row is
the $\DiffFormUp$ form row, while the cochain side is represented by the
displayed $\mathsf{BHist}$ differential $d_{\mathrm{dR}}$.

The boundary preimage component is exactly
\autoref{def:derham-chain-complex-boundary-row}. The square-zero endpoint
data are exactly the packet of
\autoref{def:derham-double-exterior-derivative-packet}.

Apply \autoref{thm:derham-double-exterior-derivative-boundary} to identify
the packet's second differential row as a boundary row, and apply
\autoref{thm:derham-cocycle-readback} to read the intermediate cochain row
as a cocycle. These two readings account for the carried De Rham data without
adding any carrier beyond the displayed $\DiffFormUp$ and $\mathsf{BHist}$
rows.
\end{proof}

\begin{theorem}[De Rham root classifier obligation]
\label{thm:derham-root-classifier-obligation}
The root classifier obligation for $\DeRhamUp$ is exactly
$\mathsf{DeRhamBoundary}_{d_{\mathrm{dR}}}$. If a boundary row is accepted by
this classifier and its endpoint is transported by $\hsame$, the transported
endpoint is again classified by
$\mathsf{DeRhamBoundary}_{d_{\mathrm{dR}}}$ through the same cochain preimage
witness. Thus classifier stability is precisely the endpoint-transport row
of \autoref{thm:derham-boundary-zero-endpoint-transport}.
\end{theorem}
\leanchecked{BEDC.Derived.DeRhamUp.DeRhamBoundary\_hsame\_transport\_with\_empty\_endpoint}
\begin{proof}
Unpack \autoref{def:derham-chain-complex-boundary-row}. Classification by
$\mathsf{DeRhamBoundary}_{d_{\mathrm{dR}}}$ is the assertion that the public
row has a displayed cochain preimage and compares by $\hsame$ with its
$d_{\mathrm{dR}}$ image.

The preservation clause is
\autoref{thm:derham-boundary-zero-endpoint-transport}. It keeps the same
preimage witness and changes only the visible endpoint along the supplied
$\hsame$ comparison.

The square-zero packet of
\autoref{def:derham-double-exterior-derivative-packet}, together with
\autoref{thm:derham-double-exterior-derivative-boundary}, supplies the
canonical boundary instance consumed by this classifier. The assembly theorem
\autoref{thm:derham-namecert-obligation-surface} records that no second

\end{proof}

\begin{theorem}[De Rham root ledger exactness]
\label{thm:derham-root-ledger-exactness}
The public ledger for the $\DeRhamUp$ root surface consists only of the
boundary preimage row of \autoref{def:derham-chain-complex-boundary-row}, the
square-zero endpoint row of
\autoref{def:derham-double-exterior-derivative-packet}, the cocycle readback
row of \autoref{thm:derham-cocycle-readback}, and the displayed source rows
imported from $\DiffFormUp$ and $\CohomologyUp$. Its exactness entry is the
boundary conclusion of
\autoref{thm:derham-double-exterior-derivative-boundary}, transported at
endpoints by \autoref{thm:derham-boundary-zero-endpoint-transport}.
\end{theorem}
\begin{proof}
List the public rows named by
\autoref{thm:derham-namecert-obligation-surface}. They are the
$\DiffFormUp$ carrier row, the $\CohomologyUp$ cochain and cocycle rows, the
boundary preimage row, the endpoint-transport row, the square-zero boundary
row, and the cocycle readback row.

The exactness component is provided by
\autoref{thm:derham-double-exterior-derivative-boundary}: from the displayed
double exterior packet it produces the accepted boundary predicate for the
second differential row.

The endpoint component is stable by
\autoref{thm:derham-boundary-zero-endpoint-transport}, and the readback
component is exactly \autoref{thm:derham-cocycle-readback}. Since each row is
already named in the obligation surface, the ledger has no additional bridge,
quotient, or cohomology-isomorphism entry.
\end{proof}

\begin{definition}[De Rham root consumer readback packet]
\label{def:derham-root-consumer-readback-packet}
The root consumer readback packet for $\DeRhamUp$ is the finite list of rows
that a downstream root consumer may inspect before any standard boundary
bridge. It consists of the $\DiffFormUp$ carried-form row, the
$\CohomologyUp$ cochain and cocycle rows, the $\ManifoldUp$ source row only as
the domain provenance already visible to $\DiffFormUp$, the
$\TensorProductUp$ row only as the algebraic source already visible to
$\DiffFormUp$, the $\mathsf{BHist}$ cochain endpoints, the $\hsame$ endpoint
transport rows, the $\Cont$ ledgers for the cochain differential and boundary
preimage, and the $\Pkg$ provenance row tying these entries to one package.
\end{definition}

\begin{theorem}[De Rham root unblock threshold surface]
\label{thm:derham-root-unblock-threshold-surface}
The root carrier obligation, root classifier obligation, root ledger exactness,
root consumer readback packet, and root consumer readback stability theorem are
exactly the $\DeRhamUp$ root-unblock surface. A downstream consumer may read
only the $\DiffFormUp$, $\CohomologyUp$, $\ManifoldUp$, $\TensorProductUp$,
$\mathsf{BHist}$, $\hsame$, $\Cont$, and $\Pkg$ rows displayed in those blocks.
\end{theorem}
\begin{proof}
Project the root carrier block
\autoref{thm:derham-root-carrier-obligation}. It supplies the carried
$\DiffFormUp$ row, the cochain differential, the boundary preimage row, and
the square-zero packet.

Project the classifier and ledger blocks
\autoref{thm:derham-root-classifier-obligation} and
\autoref{thm:derham-root-ledger-exactness}. These blocks account for
$\mathsf{DeRhamBoundary}_{d_{\mathrm{dR}}}$, endpoint transport by $\hsame$,
cocycle readback, and the finite ledger rows.

Finally add the readback packet and its stability theorem. They name the
allowed downstream rows and show that their $\hsame$ movement is closed within
the same finite packet. Therefore the listed blocks exhaust the root-unblock
surface and exclude quotient cohomology, de Rham isomorphism, and external
cochain-complex data.
\end{proof}

\begin{theorem}[De Rham boundary public classifier exactness]

This public surface exports the carried differential-form row, the boundary
classifier $\mathsf{DeRhamBoundary}_{d_{\mathrm{dR}}}$, endpoint stability

\end{theorem}
