\chapter{A Concrete Naming Certificate for $\DeRhamUp$}
\label{ch:concrete-instances-deRham-namecert}

The de~Rham interface $\DeRhamUp$ packages the exterior-derivative chain complex on differential forms, certifying $d^2 = 0$ and $\hsame$-stability. It depends on $\NameCert_{\DiffFormUp}$ (\autoref{ch:concrete-instances-diffform-namecert}) and $\NameCert_{\CohomologyUp}$ (\autoref{ch:concrete-instances-cohomology-namecert}), and supplies the de~Rham cohomology of a manifold.

\begin{definition}[De Rham chain-complex boundary row]
\label{def:derham-chain-complex-boundary-row}
Fix a $\NameCert_{\CohomologyUp}$ cochain differential
$d_{\mathrm{dR}}:\mathsf{BHist}\to\mathsf{BHist}$. A row
$b:\mathsf{BHist}$ is accepted by the $\DeRhamUp$ chain-complex boundary
predicate when it is in the image of the same cochain differential:
$$
\mathsf{DeRhamBoundary}_{d_{\mathrm{dR}}}(b)
:\Longleftrightarrow
\exists a:\mathsf{BHist},\ \hsame(b,d_{\mathrm{dR}}(a)).
$$
The witness $a$ is the cochain preimage row. The predicate adds no quotient
row and no host-level differential-form operation beyond the displayed
cochain differential.
\end{definition}
\leandef{BEDC.Derived.DeRhamUp.DeRhamBoundary}

\begin{definition}[De Rham double exterior-derivative packet]
\label{def:derham-double-exterior-derivative-packet}
Fix a $\NameCert_{\ManifoldUp}$ source $M$ and a
$\NameCert_{\TensorProductUp}$ source $T$ as in
\autoref{def:diffform-bhist-form-carrier}. Fix a
$\NameCert_{\CohomologyUp}$ cochain differential
$d_{\mathrm{dR}}:\mathsf{BHist}\to\mathsf{BHist}$ and a zero cochain endpoint
$z_{\mathrm{dR}}:\mathsf{BHist}$. For carried $\DiffFormUp$ rows
$\omega$, $\eta$, and $\theta$, the packet
$$
\mathsf{DeRhamDoubleExterior}_{M,T,d_{\mathrm{dR}}}
(\omega,\eta,\theta,z_{\mathrm{dR}})
$$
is accepted exactly when its visible fields are the following rows:
\begin{enumerate}
  \item $\omega$, $\eta$, and $\theta$ are carried by
  $\mathsf{DiffFormBHistCarrier}_{M,T}$.
  \item The two successive exterior-input rows
  $\mathsf{DiffFormExteriorInput}_{M,T}(\omega,\eta)$ and
  $\mathsf{DiffFormExteriorInput}_{M,T}(\eta,\theta)$ are accepted.
  \item The cochain graph rows identify the carried exterior outputs with the
  same cohomology differential:
  $$
  \eta\hsame d_{\mathrm{dR}}(\omega)
  \quad\text{and}\quad
  \theta\hsame d_{\mathrm{dR}}(\eta).
  $$
  \item The cohomology differential respects $\hsame$: for all
  $a,b:\mathsf{BHist}$,
  $$
  \hsame(a,b)\Longrightarrow
  \hsame(d_{\mathrm{dR}}(a),d_{\mathrm{dR}}(b)).
  $$
  \item The de Rham square-zero row and zero endpoint row are present:
  $$
  \forall a:\mathsf{BHist},\
  \hsame(d_{\mathrm{dR}}(d_{\mathrm{dR}}(a)),z_{\mathrm{dR}})
  \quad\text{and}\quad
  \hsame(z_{\mathrm{dR}},\emp).
  $$
\end{enumerate}
The two $\DiffFormUp$ exterior-input rows provide only the carried input-output
surface of \autoref{def:diffform-exterior-derivative-visible-input-package};
the square-zero row is a $\DeRhamUp$ field read through the
$\CohomologyUp$ cochain differential.
\end{definition}
\leandef{BEDC.Derived.DeRhamUp.DeRhamDoubleExteriorPacket}

\begin{theorem}[De Rham double exterior derivative boundary]
\label{thm:derham-double-exterior-derivative-boundary}
If
$\mathsf{DeRhamDoubleExterior}_{M,T,d_{\mathrm{dR}}}
(\omega,\eta,\theta,z_{\mathrm{dR}})$ is accepted, then the carried second
exterior derivative row $\theta$ is classified as the zero cochain endpoint
and is an accepted $\DeRhamUp$ chain-complex boundary row:
$$
\hsame(\theta,z_{\mathrm{dR}})
\quad\wedge\quad
\mathsf{DeRhamBoundary}_{d_{\mathrm{dR}}}(\theta).
$$
Moreover the intermediate carried row is a $\CohomologyUp$ cocycle:
$$
C_{d_{\mathrm{dR}}}(\eta).
$$
\end{theorem}
\leanchecked{BEDC.Derived.DeRhamUp.DeRhamDoubleExteriorPacket\_boundary}
\leanvariant{BEDC.Derived.DeRhamUp.DeRhamDoubleExteriorDerivative\_boundary}
\begin{proof}
Unpack the accepted packet of
\autoref{def:derham-double-exterior-derivative-packet}. The two accepted
$\DiffFormUp$ exterior-input rows are successive by
\autoref{def:diffform-exterior-derivative-visible-input-package}: the first
has source $\omega$ and output $\eta$, while the second has source $\eta$ and
output $\theta$. By \autoref{thm:diffform-exterior-derivative-degree-successor-ledger},
these rows expose only the one-step exterior-input ledgers. The nilpotence row
therefore comes from the de Rham packet, not from the $\DiffFormUp$ boundary
exhaustion of \autoref{thm:diffform-downstream-boundary-exhaustion}.

Let the graph comparisons be
$\eta\hsame d_{\mathrm{dR}}(\omega)$ and
$\theta\hsame d_{\mathrm{dR}}(\eta)$. Applying the $\hsame$-respect row of
$d_{\mathrm{dR}}$ to the first comparison gives
$$
\hsame(d_{\mathrm{dR}}(\eta),d_{\mathrm{dR}}(d_{\mathrm{dR}}(\omega))).
$$
The square-zero row at $\omega$ gives
$$
\hsame(d_{\mathrm{dR}}(d_{\mathrm{dR}}(\omega)),z_{\mathrm{dR}}).
$$
Transitivity with the second graph comparison yields
$\hsame(\theta,z_{\mathrm{dR}})$.

For the boundary predicate, use the cochain preimage $\eta$. The second graph
comparison is exactly the witness
$\hsame(\theta,d_{\mathrm{dR}}(\eta))$ required by
\autoref{def:derham-chain-complex-boundary-row}. Hence
$\mathsf{DeRhamBoundary}_{d_{\mathrm{dR}}}(\theta)$ holds.

Finally, the same differential comparison, the square-zero row at $\omega$,
and the zero endpoint comparison $\hsame(z_{\mathrm{dR}},\emp)$ give
$d_{\mathrm{dR}}(\eta)\hsame\emp$. By
\autoref{def:cohomology-cocycle-predicate}, this is exactly
$C_{d_{\mathrm{dR}}}(\eta)$.
\end{proof}

\begin{theorem}[De Rham zero-endpoint boundary transport]
\label{thm:derham-boundary-zero-endpoint-transport}
Fix a $\NameCert_{\CohomologyUp}$ cochain differential
$d_{\mathrm{dR}}:\mathsf{BHist}\to\mathsf{BHist}$. If a displayed row
$b:\mathsf{BHist}$ is accepted by
$\mathsf{DeRhamBoundary}_{d_{\mathrm{dR}}}$ through a cochain preimage
$a:\mathsf{BHist}$ and a transported endpoint $b':\mathsf{BHist}$ satisfies
$\hsame(b',b)$, then the transported row is again an accepted de Rham
boundary row:
$$
\mathsf{DeRhamBoundary}_{d_{\mathrm{dR}}}(b')
$$
with the same preimage witness $a$. In particular, if $b$ is compared by
$\hsame$ to the zero endpoint $z_{\mathrm{dR}}$, any displayed endpoint
transported to that zero endpoint remains in the same boundary predicate.
\end{theorem}
\begin{proof}
Unpack \autoref{def:derham-chain-complex-boundary-row}. The accepted boundary
row supplies a witness $a$ and a comparison
$\hsame(b,d_{\mathrm{dR}}(a))$.

The transported endpoint supplies $\hsame(b',b)$. By transitivity of
$\hsame$, the row $b'$ compares with $d_{\mathrm{dR}}(a)$:
$$
\hsame(b',d_{\mathrm{dR}}(a)).
$$

Repack the same cochain preimage $a$ in
\autoref{def:derham-chain-complex-boundary-row}. No new differential-form row,
quotient row, or cohomology carrier is introduced; only the displayed endpoint
has been transported along $\hsame$.
\end{proof}

\begin{theorem}[De Rham cocycle readback from square-zero packets]
\label{thm:derham-cocycle-readback}
Every accepted packet
$$
\mathsf{DeRhamDoubleExterior}_{M,T,d_{\mathrm{dR}}}
(\omega,\eta,\theta,z_{\mathrm{dR}})
$$
reads its intermediate carried row $\eta$ as a $\CohomologyUp$ cocycle:
$$
C_{d_{\mathrm{dR}}}(\eta).
$$
The readback uses only the packet's cochain graph row
$\eta\hsame d_{\mathrm{dR}}(\omega)$, the square-zero row for
$d_{\mathrm{dR}}$, and the zero endpoint comparison
$\hsame(z_{\mathrm{dR}},\emp)$.
\end{theorem}
\begin{proof}
Unpack \autoref{def:derham-double-exterior-derivative-packet}. The graph row
identifies the intermediate carried row with the first cochain differential:
$\eta\hsame d_{\mathrm{dR}}(\omega)$.

Apply the $\hsame$-respect field of the cochain differential to this graph
row. It gives a comparison from $d_{\mathrm{dR}}(\eta)$ to
$d_{\mathrm{dR}}(d_{\mathrm{dR}}(\omega))$.

The square-zero row at $\omega$ compares
$d_{\mathrm{dR}}(d_{\mathrm{dR}}(\omega))$ with $z_{\mathrm{dR}}$, and the
zero endpoint comparison identifies $z_{\mathrm{dR}}$ with $\emp$. Chaining
these $\hsame$ comparisons yields $d_{\mathrm{dR}}(\eta)\hsame\emp$.

By \autoref{def:cohomology-cocycle-predicate}, this is exactly the
$\CohomologyUp$ cocycle predicate $C_{d_{\mathrm{dR}}}(\eta)$.
\end{proof}

\begin{theorem}[De Rham NameCert obligation surface]
\label{thm:derham-namecert-obligation-surface}
The de Rham rows in this chapter expose the
$\NameCert_{\DeRhamUp}$ obligation surface over $\DiffFormUp$ and
$\CohomologyUp$. The carrier is the differential-form carried row supplied by
\autoref{def:diffform-bhist-form-carrier}; the classifier is the boundary
predicate of \autoref{def:derham-chain-complex-boundary-row}; the stability
field is the endpoint transport of
\autoref{thm:derham-boundary-zero-endpoint-transport}; the exactness row is
the square-zero boundary theorem
\autoref{thm:derham-double-exterior-derivative-boundary}; and the ledger row
is the cocycle readback of \autoref{thm:derham-cocycle-readback}. Together
these rows supply the carrier, classifier, stability, exactness, and ledger
entries required by the named de Rham certificate surface, without asserting a
de Rham isomorphism or any quotient-level cohomology presentation.
\end{theorem}
\begin{proof}
The carrier component is inherited from the visible $\DiffFormUp$ form
carrier cited in the statement. The De Rham boundary row then classifies the
second exterior derivative by a concrete cochain preimage in
\autoref{def:derham-chain-complex-boundary-row}.

Stability is supplied by
\autoref{thm:derham-boundary-zero-endpoint-transport}: once a boundary row is
witnessed by a preimage, endpoint transport along $\hsame$ preserves the same
predicate using that witness.

Exactness is supplied by
\autoref{thm:derham-double-exterior-derivative-boundary}. It turns an accepted
double exterior packet into both the zero endpoint comparison for the second
row and the boundary predicate for that row.

The ledger is the readback
\autoref{thm:derham-cocycle-readback}. It records that the intermediate row
seen by the square-zero packet is a $\CohomologyUp$ cocycle. These named rows
are precisely the advertised certificate obligations and no additional bridge
or quotient surface is used.
\end{proof}
