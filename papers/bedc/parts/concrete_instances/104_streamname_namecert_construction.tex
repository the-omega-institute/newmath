\chapter{A Concrete Naming Certificate for $\StreamNameUp$}
\label{ch:concrete-instances-streamname-namecert}

The stream-name interface $\StreamNameUp$ packages an abstract sequential carrier underlying the Bishop-style L10 real-number construction. It is the lowest-level data on which $\NameCert_{\TotallyBoundedUp}$, $\NameCert_{\CompleteMetricUp}$, $\NameCert_{\CompactMetricUp}$, and $\NameCert_{\ContinuousMapUp}$ are built.

\section{Carrier definition}
\label{sec:streamname-carrier}

\begin{definition}[Rational stream-name carrier]
\label{def:rat-stream-name-carrier}
For a certified observation map $s:\Hist\to\Hist$, the rational stream-name
carrier is the pointwise rational-history carrier over unary BEDC observation
histories:
$$
\mathsf{RatStreamNameCarrier}(s)
\Longleftrightarrow
\forall n,\ \mathsf{UnaryHistory}(n)\to\mathsf{RatHistoryCarrier}(s(n)).
$$
\end{definition}
\leandef{BEDC.Derived.StreamNameUp.RatStreamNameCarrier}

\begin{definition}[Rational stream-name classifier]
\label{def:rat-stream-name-classifier}
For certified observation maps $s,t:\Hist\to\Hist$, the rational stream-name
classifier bundles the two endpoint stream carriers with the pointwise
rational-history classifier over the same unary BEDC observation histories:
$$
\mathsf{RatStreamNameClassifier}(s,t)
\Longleftrightarrow
\mathsf{RatStreamNameCarrier}(s)\land
\mathsf{RatStreamNameCarrier}(t)\land
\forall n,\ \mathsf{UnaryHistory}(n)\to
\mathsf{RatHistoryClassifier}(s(n),t(n)).
$$
\end{definition}
\leandef{BEDC.Derived.StreamNameUp.RatStreamNameClassifier}

\begin{theorem}[Rational stream-name classifier unary context one-tail pair readback]
\label{thm:rat-streamname-classifier-unary-context-e1-pair-readback}
If $\mathsf{RatStreamNameClassifier}(s,t)$ holds at a unary observation $n$,
matching unary prefix and tail contexts are presented by continuation displays,
and the two outputs are one-extension histories, then the displayed tails are
unary and internally the same.
\end{theorem}
\begin{proof}
Read the stream-name classifier at the certified observation $n$. This gives
the RatUp point classifier for $s(n)$ and $t(n)$. Apply the RatUp continuation
context one-tail pair readback theorem to that point classifier and the four
continuation displays.
\end{proof}
\leanchecked{BEDC.Derived.StreamNameUp.RatStreamNameClassifier\_unary\_context\_e1\_pair\_readback}

\section{The certificate}
\label{sec:streamname-certificate}

\begin{theorem}[Rational stream-name certificate fields]
\label{thm:rat-stream-name-certificate-fields}
For certified observation maps $s,t,u:\Hist\to\Hist$, the carrier and
classifier above supply the rational stream-name certificate fields:
\begin{enumerate}
  \item $\mathsf{RatStreamNameCarrier}(s)$ implies
  $\mathsf{RatStreamNameClassifier}(s,s)$.
  \item $\mathsf{RatStreamNameClassifier}(s,t)$ implies
  $\mathsf{RatStreamNameClassifier}(t,s)$.
  \item $\mathsf{RatStreamNameClassifier}(s,t)$ and
  $\mathsf{RatStreamNameClassifier}(t,u)$ imply
  $\mathsf{RatStreamNameClassifier}(s,u)$.
  \item $\mathsf{RatStreamNameClassifier}(s,t)$ implies, for every $n$ with
  $\mathsf{UnaryHistory}(n)$, both $\mathsf{RatHistoryCarrier}(s(n))$ and
  $\mathsf{RatHistoryCarrier}(t(n))$.
  \item If $\mathsf{RatStreamNameClassifier}(s,t)$ and streams $s',t'$ satisfy
  $\forall n,\ \mathsf{UnaryHistory}(n)\to\hsame(s(n),s'(n))$ and
  $\forall n,\ \mathsf{UnaryHistory}(n)\to\hsame(t(n),t'(n))$, then
  $\mathsf{RatStreamNameClassifier}(s',t')$.
\end{enumerate}
\end{theorem}
\begin{proof}
For reflexivity, the two endpoint carrier rows of
$\mathsf{RatStreamNameClassifier}(s,s)$ are both the supplied carrier of $s$.
For its pointwise classifier row, fix a certified observation history $n$ with
witness $\nu:\mathsf{UnaryHistory}(n)$. The carrier gives
$\mathsf{RatHistoryCarrier}(s(n))$ from $\nu$, and RatUp classifier
reflexivity turns that pointwise rational witness into
$\mathsf{RatHistoryClassifier}(s(n),s(n))$.

For symmetry, unpack the supplied stream classifier into its carrier rows for
$s$ and $t$ and its pointwise classifier row. The reversed classifier uses the
$t$ carrier as its first carrier row and the $s$ carrier as its second carrier
row. For the pointwise row, fix $(n,\nu)$ and apply RatUp rational-history
classifier symmetry to the projected point classifier at $(n,\nu)$.

For transitivity, unpack $\mathsf{RatStreamNameClassifier}(s,t)$ and
$\mathsf{RatStreamNameClassifier}(t,u)$. The resulting classifier from $s$ to
$u$ uses the first carrier row of the first bundle and the second carrier row
of the second bundle. For the pointwise row, fix $(n,\nu)$ and compose the two
projected point classifiers at $(n,\nu)$ by RatUp rational-history classifier
transitivity.

For endpoint carriers, fix a certified observation history $n$ with witness
$\nu:\mathsf{UnaryHistory}(n)$. Project the two bundled carrier rows of
$\mathsf{RatStreamNameClassifier}(s,t)$ and read them at $(n,\nu)$; these are
exactly the rational-history carrier witnesses required at the two endpoint
observations.

For transport, first transport the two bundled carrier rows along the supplied
history-sameness witnesses, producing $\mathsf{RatStreamNameCarrier}(s')$ and
$\mathsf{RatStreamNameCarrier}(t')$. For the pointwise row, fix $(n,\nu)$ and
apply
$\mathsf{BEDC.Derived.RatUp.RatHistoryClassifier\_hsame\_transport}$ to the
projected point classifier at $(n,\nu)$ and to the two supplied endpoint
history-sameness witnesses at $(n,\nu)$. The transported carrier rows and the
transported pointwise classifier row repack as
$\mathsf{RatStreamNameClassifier}(s',t')$.
\end{proof}
\leanchecked{BEDC.Derived.StreamNameUp.RatStreamName\_certificate\_fields}

\begin{theorem}[Rational stream-name observation one-tail pair readback]
\label{thm:rat-stream-name-classifier-observation-e1-pair-readback}
For certified observation maps $s,t:\Hist\to\Hist$, if
$\mathsf{RatStreamNameClassifier}(s,t)$ and $n$ is a unary observation, then
displayed one-extension endpoints
$$
\hsame(s(n),\Eone(a))\quad\text{and}\quad \hsame(t(n),\Eone(b))
$$
determine unary tails with the internal comparison
$$
\mathsf{UnaryHistory}(a)\land\mathsf{UnaryHistory}(b)\land\hsame(a,b).
$$
\end{theorem}
\begin{proof}
Read the stream-name classifier at the certified observation $n$. Transport the
resulting RatUp point classifier to the displayed one-extension endpoints, then
apply rational one-tail classifier exactness.
\end{proof}

\begin{theorem}[Rational stream-name observation shape exclusions]
\label{thm:rat-stream-name-classifier-observation-shape-exclusions}
For certified observation maps $s,t:\Hist\to\Hist$ and an observation history
$n$, suppose
$$
\mathsf{RatStreamNameClassifier}(s,t)\quad\text{and}\quad
\mathsf{UnaryHistory}(n).
$$
Then the selected observations satisfy the positive-denominator and unary
endpoint package
$$
\mathsf{PositiveUnaryDenominator}(s(n))\land
\mathsf{PositiveUnaryDenominator}(t(n))\land
\mathsf{UnaryHistory}(s(n))\land
\mathsf{UnaryHistory}(t(n)).
$$
They also satisfy the two empty-history exclusions
$$
(\hsame(s(n),\emp)\to\mathsf{False})\land
(\hsame(t(n),\emp)\to\mathsf{False}),
$$
and for all histories $z_s,z_t:\Hist$ the two zero-headed exclusions
$$
(\hsame(s(n),\Ezero(z_s))\to\mathsf{False})\land
(\hsame(t(n),\Ezero(z_t))\to\mathsf{False}).
$$
\end{theorem}
\begin{proof}
Read the stream-name classifier at the certified observation $(n,\nu)$, where
$\nu:\mathsf{UnaryHistory}(n)$. Apply
\autoref{thm:rat-history-classifier-positive-denominators} to this point
classifier. This gives positive unary denominator witnesses for $s(n)$ and
$t(n)$.

Apply \autoref{thm:positive-unary-denominator-unary-and-nonempty} separately
to the two positive denominator witnesses. The unary components give
$\mathsf{UnaryHistory}(s(n))$ and $\mathsf{UnaryHistory}(t(n))$, and the
nonempty components give the two exclusions against $\emp$.

For a displayed zero-headed shape $\hsame(s(n),\Ezero(z_s))$, transport the
positive denominator witness for $s(n)$ across that sameness by
\autoref{thm:rat-positive-unary-denominator-hsame-transport}. The transported
witness contradicts \autoref{thm:rat-positive-denominator-e0-absurd}. The
argument for $\hsame(t(n),\Ezero(z_t))$ is identical, using the positive
denominator witness for $t(n)$.
\end{proof}

\begin{theorem}[Rational stream-name observation full endpoint package]
\label{thm:rat-stream-name-observation-full-endpoint-package}
For certified observation maps $s,t:\Hist\to\Hist$, an observation history $n$,
and histories $a,b:\Hist$, suppose
$$
\mathsf{RatStreamNameClassifier}(s,t)\quad\text{and}\quad
\mathsf{UnaryHistory}(n),
$$
and suppose the selected observations are displayed as one-extension endpoints
$$
\hsame(s(n),\Eone(a))\quad\text{and}\quad
\hsame(t(n),\Eone(b)).
$$
Then the following selected-observation endpoint package holds:
\begin{enumerate}
  \item $\mathsf{RatHistoryClassifier}(s(n),t(n))$.
  \item $\mathsf{PositiveUnaryDenominator}(s(n))\land
  \mathsf{PositiveUnaryDenominator}(t(n))$.
  \item $\mathsf{UnaryHistory}(s(n))\land\mathsf{UnaryHistory}(t(n))$.
  \item $(\hsame(s(n),\emp)\to\mathsf{False})\land
  (\hsame(t(n),\emp)\to\mathsf{False})$.
  \item For all histories $z_s,z_t:\Hist$,
  $(\hsame(s(n),\Ezero(z_s))\to\mathsf{False})\land
  (\hsame(t(n),\Ezero(z_t))\to\mathsf{False})$.
  \item $\mathsf{UnaryHistory}(a)\land\mathsf{UnaryHistory}(b)\land
  \hsame(a,b)$.
\end{enumerate}
\end{theorem}
\begin{proof}
Unpack \autoref{def:rat-stream-name-classifier} at the certified observation
$(n,\nu)$, where $\nu:\mathsf{UnaryHistory}(n)$. The pointwise row gives
$\mathsf{RatHistoryClassifier}(s(n),t(n))$.

Apply \autoref{thm:rat-stream-name-classifier-observation-shape-exclusions} to
the same stream classifier and unary observation. This supplies the two positive
unary denominator rows, the unary-history rows for the selected observations,
the empty-history exclusions, and the zero-headed exclusions for arbitrary
$z_s,z_t$.

Apply \autoref{thm:rat-stream-name-classifier-observation-e1-pair-readback} to
the stream classifier, the unary observation, and the two displayed
one-extension endpoint rows. This gives
$\mathsf{UnaryHistory}(a)\land\mathsf{UnaryHistory}(b)\land\hsame(a,b)$.

The point classifier, the shape rows, and the one-tail readback rows are exactly
the listed endpoint package.
\end{proof}

Fit. Traditional target: selected rational-stream observation endpoint package.
BEDC source: $\RatUp+\StreamNameUp$ over $\mathsf{BHist}$. Carrier:
$\mathsf{RatStreamNameCarrier}$. Classifier:
$\mathsf{RatStreamNameClassifier}$. Stability: pointwise $\hsame$ endpoint
display at the selected unary observation. Ledger: the selected observation, the
two one-extension displays, and arbitrary zero-headed probes. Core theorem:
\autoref{thm:rat-stream-name-observation-full-endpoint-package}. Exactness:
rational point classifier, endpoint shape exclusions, and one-tail equality
readback. Standard bridge: $\RealUp$ contextual endpoint certificates consume
this selected-observation package. Status: $\mathsf{ProofTarget}$.

(\hsame(t'(n),\emp)\to\mathsf{False}),

\begin{theorem}[Rational stream-name reindexing and map stability]
\label{thm:rat-stream-name-reindexing-and-map-stability}
For certified observation maps $s,t,r:\Hist\to\Hist$, suppose $r$ preserves
certified unary observations:
$$
\forall n,\ \mathsf{UnaryHistory}(n)\to\mathsf{UnaryHistory}(r(n)).
$$
Then $\mathsf{RatStreamNameCarrier}(s)$ implies
$\mathsf{RatStreamNameCarrier}(s\circ r)$, and
$\mathsf{RatStreamNameClassifier}(s,t)$ implies
$\mathsf{RatStreamNameClassifier}(s\circ r,t\circ r)$.

For a history map $F:\Hist\to\Hist$, suppose $F$ preserves RatUp carriers and
classifiers:
$$
\forall h,\ \mathsf{RatHistoryCarrier}(h)\to
\mathsf{RatHistoryCarrier}(F(h))
$$
and
$$
\forall h,k,\ \mathsf{RatHistoryClassifier}(h,k)\to
\mathsf{RatHistoryClassifier}(F(h),F(k)).
$$
Then $\mathsf{RatStreamNameCarrier}(s)$ implies
$\mathsf{RatStreamNameCarrier}(F\circ s)$, and
$\mathsf{RatStreamNameClassifier}(s,t)$ implies
$\mathsf{RatStreamNameClassifier}(F\circ s,F\circ t)$.
\end{theorem}
\begin{proof}
For reindexing, unfold $\mathsf{RatStreamNameCarrier}$. Fix an observation
history $n$ with witness $\nu:\mathsf{UnaryHistory}(n)$. The reindexer
hypothesis turns $\nu$ into a certified unary observation witness for $r(n)$,
and the carrier of $s$ gives the required rational-history carrier at
$s(r(n))$.

For the reindexed classifier, unfold
$\mathsf{RatStreamNameClassifier}$ and fix the same certified observation
data $(n,\nu)$. The reindexer hypothesis again supplies
$\mathsf{UnaryHistory}(r(n))$, so the classifier of $s$ and $t$ can be read at
$r(n)$, giving the point classifier between $s(r(n))$ and $t(r(n))$.

For mapped streams, unfold $\mathsf{RatStreamNameCarrier}$ and fix
$n$ with witness $\nu:\mathsf{UnaryHistory}(n)$. The carrier of $s$ gives
$\mathsf{RatHistoryCarrier}(s(n))$, and the carrier-preservation hypothesis
for $F$ transports it to $\mathsf{RatHistoryCarrier}(F(s(n)))$.

For mapped classifiers, unfold $\mathsf{RatStreamNameClassifier}$ and fix
$n$ with witness $\nu:\mathsf{UnaryHistory}(n)$. The classifier of $s$ and
$t$ gives $\mathsf{RatHistoryClassifier}(s(n),t(n))$, and the
classifier-preservation hypothesis for $F$ transports it to
$\mathsf{RatHistoryClassifier}(F(s(n)),F(t(n)))$.
\end{proof}

\begin{theorem}[Rational stream-name reindexing composition law]
\label{thm:rat-stream-name-reindexing-composition-law}
For certified observation maps $s,t,r,q:\Hist\to\Hist$, suppose $r$ and $q$
preserve certified unary observations:
$$
\forall n,\ \mathsf{UnaryHistory}(n)\to\mathsf{UnaryHistory}(r(n))
$$
and
$$
\forall n,\ \mathsf{UnaryHistory}(n)\to\mathsf{UnaryHistory}(q(n)).
$$
Then the composite observation map $n\mapsto q(r(n))$ preserves certified
unary observations. Consequently, $\mathsf{RatStreamNameCarrier}(s)$ implies
$\mathsf{RatStreamNameCarrier}(s\circ q\circ r)$, and
$\mathsf{RatStreamNameClassifier}(s,t)$ implies
$\mathsf{RatStreamNameClassifier}(s\circ q\circ r,t\circ q\circ r)$.
\end{theorem}
\begin{proof}
Fix a certified observation history $n$ with witness
$\nu:\mathsf{UnaryHistory}(n)$. The preservation witness for $r$ sends $\nu$
to a certified unary observation witness for $r(n)$, and the preservation
witness for $q$ sends that result to a certified unary observation witness for
$q(r(n))$. This proves unary preservation for the composite observation map.

For the carrier consequence, unfold $\mathsf{RatStreamNameCarrier}$ and fix
the same certified observation data $(n,\nu)$. The composite unary witness just
constructed lets the carrier of $s$ be read at $q(r(n))$, giving the required
rational-history carrier at $(s\circ q\circ r)(n)$.

For the classifier consequence, unfold $\mathsf{RatStreamNameClassifier}$ and
fix $(n,\nu)$. The composite unary witness again lets the classifier of $s$
and $t$ be read at $q(r(n))$, yielding the point classifier between
$(s\circ q\circ r)(n)$ and $(t\circ q\circ r)(n)$.
\end{proof}
\leanchecked{BEDC.Derived.StreamNameUp.RatStreamName\_reindexing\_composition\_law}

\begin{theorem}[Constant rational stream-name witness]
\label{thm:rat-stream-name-constant-witness}
For histories $d,e:\Hist$, if $\mathsf{RatHistoryCarrier}(d)$, then
$\mathsf{RatStreamNameCarrier}(\mathsf{RatConstStream}_d)$ and
$\mathsf{RatStreamNameClassifier}(\mathsf{RatConstStream}_d,
\mathsf{RatConstStream}_d)$. If $\mathsf{RatHistoryClassifier}(d,e)$, then
$\mathsf{RatStreamNameClassifier}(\mathsf{RatConstStream}_d,
\mathsf{RatConstStream}_e)$.
\end{theorem}
\begin{proof}
Unfold $\mathsf{RatConstStream}$ and
$\mathsf{RatStreamNameCarrier}$. For any certified observation history $n$ with
witness $\nu:\mathsf{UnaryHistory}(n)$, the stream value is $d$, so the supplied
$\mathsf{RatHistoryCarrier}(d)$ is the required pointwise carrier witness.

For the self-classifier, unfold $\mathsf{RatStreamNameClassifier}$ and fix
$n$ with witness $\nu:\mathsf{UnaryHistory}(n)$. The two endpoint observations
are both $d$, and RatUp carrier-reflexivity turns the certified rational
observation into $\mathsf{RatHistoryClassifier}(d,d)$.

For a classifier between two constant streams, unfold
$\mathsf{RatStreamNameClassifier}$ and fix $n$ with witness
$\nu:\mathsf{UnaryHistory}(n)$. The endpoint observations are $d$ and $e$, so
the supplied $\mathsf{RatHistoryClassifier}(d,e)$ is already the required
pointwise classifier witness.
\end{proof}

\begin{theorem}[Rational stream-name constant reindexing classifier]
\label{thm:rat-stream-name-constant-reindexing-classifier}
For histories $d,e:\Hist$ and an observation reindexer $r:\Hist\to\Hist$,

observation witness for $r(n)$, but the reindexed constant stream still has
value $d$, so the supplied $\mathsf{RatHistoryCarrier}(d)$ is the required
pointwise carrier witness.

For the classifier against the original constant stream, unfold
$\mathsf{RatStreamNameClassifier}$ and fix the same certified observation data
$(n,\nu)$. Both stream values are $d$ after expanding the constant stream, so
RatUp carrier-reflexivity turns the carrier witness for $d$ into the required

For two reindexed constant streams, unfold
$\mathsf{RatStreamNameClassifier}$ and fix $n$ with witness
$\nu:\mathsf{UnaryHistory}(n)$. The unary-preservation hypothesis certifies

$\mathsf{RatHistoryClassifier}(d,e)$ is therefore reused pointwise.

\begin{theorem}[Rational stream-name independently reindexed constant classifier]
\label{thm:rat-stream-name-independent-reindexed-constant-classifier}
For histories $d,e:\Hist$ and observation schedules
$r,q:\Hist\to\Hist$, if $\mathsf{RatHistoryCarrier}(d)$, then
$\mathsf{RatStreamNameCarrier}(\mathsf{RatConstStream}_d\circ r)$. If
$\mathsf{RatHistoryClassifier}(d,e)$, then
$\mathsf{RatStreamNameClassifier}(\mathsf{RatConstStream}_d\circ r,
\mathsf{RatConstStream}_e\circ q)$.

The ten-field fit is:
\begin{enumerate}
  \item the traditional target is independently scheduled constant rational
  stream names;
  \item the BEDC source is $\RatUp$ over $\mathsf{BHist}$;
  \item the carrier row is $\mathsf{RatStreamNameCarrier}$;
  \item the classifier row is $\mathsf{RatStreamNameClassifier}$;
  \item the schedule row is the pair of observation maps $r$ and $q$;
  \item the ledger is the two observation schedules together with the hidden
  rational seeds $d$ and $e$;
  \item the core theorem is this independently reindexed constant classifier;
  \item exactness is pointwise reduction to the $\RatUp$ classifier;
  \item the downstream consumer is the $\RealUp$ independent reindexed constant
  bridge;
  \item the status is $\mathsf{ProofTarget}$.
\end{enumerate}
\end{theorem}
\begin{proof}
For the carrier, unfold $\mathsf{RatConstStream}$ and
$\mathsf{RatStreamNameCarrier}$. Fix an arbitrary certified unary observation
$n$ with witness $\nu:\mathsf{UnaryHistory}(n)$. The composite
$\mathsf{RatConstStream}_d\circ r$ still evaluates to $d$, so the supplied
$\mathsf{RatHistoryCarrier}(d)$ is reused as the pointwise carrier witness.

For the classifier, unfold $\mathsf{RatConstStream}$ and
$\mathsf{RatStreamNameClassifier}$. At the same kind of certified unary
observation $n$, the left endpoint ignores $r(n)$ and evaluates to $d$, while
the right endpoint ignores $q(n)$ and evaluates to $e$. Thus the supplied
$\mathsf{RatHistoryClassifier}(d,e)$ is already the required pointwise
classifier.
\end{proof}

\begin{theorem}[Rational stream-name independently reindexed constant point exactness]
\label{thm:rat-stream-name-independent-reindexed-constant-point-exactness}
For histories $d,e:\Hist$ and observation maps $r,q:\Hist\to\Hist$, the
pointwise stream-name carrier and classifier of independently reindexed
constant streams recover exactly the RatUp history data:
$$
\mathsf{RatStreamNameCarrier}(\mathsf{RatConstStream}_d\circ r)
\Longleftrightarrow
\mathsf{RatHistoryCarrier}(d)
$$
and
$$
\mathsf{RatStreamNameClassifier}(\mathsf{RatConstStream}_d\circ r,
\mathsf{RatConstStream}_e\circ q)
\Longleftrightarrow
\mathsf{RatHistoryClassifier}(d,e).
$$
\end{theorem}
\begin{proof}
Unfold $\mathsf{RatConstStream}$ and $\mathsf{RatStreamNameCarrier}$. In the
forward carrier direction, read the stream carrier at the empty observation
$\emp$; the reindexed constant stream evaluates to the seed history $d$ there.

In the reverse carrier direction, fix an arbitrary certified unary observation.
The composite $\mathsf{RatConstStream}_d\circ r$ still evaluates to $d$, so the
seed carrier $\mathsf{RatHistoryCarrier}(d)$ supplies the pointwise stream
carrier witness.

For the classifier equivalence, unfold $\mathsf{RatConstStream}$ and
$\mathsf{RatStreamNameClassifier}$. In the forward direction, read the stream
classifier at $\emp$; both independently reindexed constant streams ignore their
observation maps and return the endpoint histories $d$ and $e$.

In the reverse direction, the supplied
$\mathsf{RatHistoryClassifier}(d,e)$ already contains the endpoint carrier data
and the point classifier between $d$ and $e$. At every certified unary
observation, the two stream endpoints again reduce to $d$ and $e$, so the same
point classifier repacks into the stream-name classifier.
\end{proof}

\begin{theorem}[Rational constant stream empty-observation exactness]
\label{thm:rat-const-stream-empty-point-exactness}
For histories $h,k:\Hist$, the pointwise stream-name carrier and classifier of
constant streams are equivalent to their RatUp point data at the empty unary
observation:
$$
\mathsf{RatStreamNameCarrier}(\mathsf{RatConstStream}_h)
\Longleftrightarrow
\mathsf{RatHistoryCarrier}(\mathsf{RatConstStream}_h(\emp))
$$
and
$$
\mathsf{RatStreamNameClassifier}(\mathsf{RatConstStream}_h,
\mathsf{RatConstStream}_k)
\Longleftrightarrow
\mathsf{RatHistoryClassifier}(\mathsf{RatConstStream}_h(\emp),
\mathsf{RatConstStream}_k(\emp)).
$$
\end{theorem}
\begin{proof}
Unfold $\mathsf{RatStreamNameCarrier}$ and $\mathsf{RatConstStream}$. In the
forward carrier direction, read the pointwise carrier witness at the empty
unary observation $\emp$, obtaining the RatUp carrier at
$\mathsf{RatConstStream}_h(\emp)$.

In the reverse carrier direction, fix an arbitrary certified unary
observation. A constant stream has the same value there as at $\emp$, so the
empty-observation carrier rebuilds the required pointwise carrier witness.

For the classifier equivalence, unfold $\mathsf{RatStreamNameClassifier}$ and
the two constant streams. The forward direction reads the pointwise classifier
witness at $\emp$.

In the reverse direction, the empty-observation classifier contains the endpoint
carrier data and the point classifier between
$\mathsf{RatConstStream}_h(\emp)$ and $\mathsf{RatConstStream}_k(\emp)$. At
every certified unary observation, the two constant endpoints again evaluate as
they do at $\emp$, so the same RatUp classifier repacks into the stream-name
classifier.
\end{proof}
\leanchecked{BEDC.Derived.StreamNameUp.RatConstStream\_empty\_point\_exactness}

\begin{theorem}[Rational stream-name constant point exactness]
\label{thm:rat-stream-name-constant-point-exactness}
For histories $h,k:\Hist$, the pointwise stream-name carrier and classifier of
constant streams recover exactly the RatUp history data:
$$
\mathsf{RatStreamNameCarrier}(\mathsf{RatConstStream}_h)
\Longleftrightarrow
\mathsf{RatHistoryCarrier}(h)
$$
and
$$
\mathsf{RatStreamNameClassifier}(\mathsf{RatConstStream}_h,
\mathsf{RatConstStream}_k)
\Longleftrightarrow
\mathsf{RatHistoryClassifier}(h,k).
$$
\end{theorem}
\begin{proof}
Unfold $\mathsf{RatStreamNameCarrier}$ and $\mathsf{RatConstStream}$. In the
forward direction, read the pointwise carrier witness at observation index $0$.
In the reverse direction, fix an arbitrary observation index $n$; since the
constant stream value is $h$, the supplied $\mathsf{RatHistoryCarrier}(h)$
witness is exactly the required pointwise witness.

For the classifier statement, unfold $\mathsf{RatStreamNameClassifier}$ and
the two constant streams. In the forward direction, read the pointwise
classifier witness at observation index $0$. In the reverse direction, fix an
arbitrary observation index $n$; since the endpoint stream values are $h$ and
$k$, the supplied $\mathsf{RatHistoryClassifier}(h,k)$ witness is exactly the
required pointwise classifier at that observation.
\end{proof}
\leanchecked{BEDC.Derived.StreamNameUp.RatStreamName\_constant\_point\_exactness}

\begin{theorem}[Rational stream-name public certificate]
\label{thm:rat-stream-name-public-certificate}
The carrier $\mathsf{RatStreamNameCarrier}$ and classifier
$\mathsf{RatStreamNameClassifier}$, read over unary $\mathsf{BHist}$
observations and RatUp point carriers and classifiers, supply the public
$\StreamNameUp$ certificate with the following ten fields:
\begin{enumerate}
  \item the traditional target is the certified countable rational-observation
  name interface;
  \item the BEDC source is $\RatUp$ over $\mathsf{BHist}$;
  \item the carrier row is $\mathsf{RatStreamNameCarrier}$;
  \item the classifier row is $\mathsf{RatStreamNameClassifier}$;
  \item reflexivity, symmetry, transitivity, endpoint-carrier recovery, and
  history-sameness transport are the rows of
  \autoref{thm:rat-stream-name-certificate-fields};
  \item reindexing stability and mapped-stream stability are the two clauses of
  \autoref{thm:rat-stream-name-reindexing-and-map-stability};
  \item composed reindexing stability is
  \autoref{thm:rat-stream-name-reindexing-composition-law};
  \item inhabited constant streams are supplied by
  \autoref{thm:rat-stream-name-constant-witness};
  \item constant reindexing classification is
  \autoref{thm:rat-stream-name-constant-reindexing-classifier};
  \item constant-point and independently reindexed constant-point exactness,
  with finite unary observation data in the ledger and exactness pointwise over
  the $\RatUp$ classifier, are
  \autoref{thm:rat-stream-name-constant-point-exactness} and
  \autoref{thm:rat-stream-name-independent-reindexed-constant-point-exactness}.
\end{enumerate}
\end{theorem}
\begin{proof}
Unfold the two public rows. The carrier row is the predicate assigning to each
stream $s:\Hist\to\Hist$ the pointwise requirement
$\mathsf{RatHistoryCarrier}(s(n))$ at every certified unary observation
$n:\mathsf{BHist}$, and the classifier row assigns to each pair of streams the
pointwise RatUp classifier between their two observed histories.

The equivalence-relation and transport rows are exactly the five clauses of
\autoref{thm:rat-stream-name-certificate-fields}. They read a stream
certificate at an arbitrary unary observation and apply the corresponding
RatUp point theorem, so no global stream data other than the two unfolded rows
is introduced.

The structural action rows come from
\autoref{thm:rat-stream-name-reindexing-and-map-stability} and
\autoref{thm:rat-stream-name-reindexing-composition-law}. The first theorem
handles a single certified observation reindexer and a RatUp-preserving history
map; the second repacks two unary-preserving reindexers into their composite and
then applies the same pointwise classifier and carrier reading.

The inhabited and constant-name rows are supplied by
\autoref{thm:rat-stream-name-constant-witness} and
\autoref{thm:rat-stream-name-constant-reindexing-classifier}. The witness row
uses a RatUp history carrier as the seed for a constant stream, while the
reindexing row observes that every reindexed constant stream still has the same
endpoint seed at each certified unary observation.

Finally,
\autoref{thm:rat-stream-name-constant-point-exactness} identifies the constant
stream carrier and classifier with the underlying RatUp carrier and classifier,
and
\autoref{thm:rat-stream-name-independent-reindexed-constant-point-exactness}
gives the same exactness after independent observation reindexing on the two
constant endpoints. Thus the ledger consists of finite unary observation data
together with the pointwise RatUp witnesses, and the exactness field is
pointwise over the RatUp classifier. These ten rows are precisely the public
$\StreamNameUp$ certificate assembled from the named component theorems.
\end{proof}
