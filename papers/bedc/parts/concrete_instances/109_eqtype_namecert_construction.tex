\chapter{A Concrete Naming Certificate for $\EqUp$}
\label{ch:concrete-instances-eqtype-namecert}

The equality-type interface $\EqUp$ packages the constructive identity type over a carrier, with reflexivity as the sole constructor and explicit transport for kernel-licit substitution. It supplies the foundational equality layer underlying every $\psame$ classifier.

\section{Carrier definition}
\label{sec:eqtype-carrier}

\begin{theorem}[EqType e1 anchor tail readback]
\label{thm:eqtype-class-carrier-e1-anchor-tail-readback}
For BEDC histories $anchor$, $tail$, and $h$, if $h$ belongs to the equality-type
class anchored at $\Eone(anchor)$ and $h\hsame\Eone(tail)$, then
$tail\hsame anchor$.
\end{theorem}
\begin{proof}
The class predicate is the transparent history-sameness class. Compose the class
comparison with the reverse of the displayed comparison to obtain
$\Eone(tail)\hsame\Eone(anchor)$, then invert the e1 constructor under $\hsame$.
\end{proof}
\leanchecked{BEDC.Derived.EqtypeUp.EqtypeClassCarrier\_e1\_anchor\_tail\_readback}

\begin{theorem}[EqType e0 anchor pair determinism]
\label{thm:eqtype-class-carrier-e0-anchor-pair-deterministic}
For BEDC histories $anchor$, $left$, and $right$, if both $left$ and $right$
belong to the equality-type class anchored at $\Ezero(anchor)$, then
$left\hsame right$.
\end{theorem}
\begin{proof}
Both carrier witnesses are concrete comparisons with the same anchor
$\Ezero(anchor)$. Compose the first comparison with the reverse of the second
comparison to obtain $left\hsame right$.
\end{proof}
\leanchecked{BEDC.Derived.EqtypeUp.EqtypeClassCarrier\_e0\_anchor\_pair\_deterministic}

\begin{theorem}[EqType e0 anchor e1 tail absurdity]
\label{thm:eqtype-class-carrier-e0-anchor-e1-tail-absurd}
For BEDC histories $anchor$, $tail$, and $h$, if $h$ belongs to the equality-type
class anchored at $\Ezero(anchor)$ and $h\hsame\Eone(tail)$, then contradiction
follows.
\end{theorem}
\begin{proof}
The carrier comparison identifies $h$ with $\Ezero(anchor)$. Composing it with
the reverse of the displayed comparison gives
$\Eone(tail)\hsame\Ezero(anchor)$, which is impossible by constructor
separation for history sameness.
\end{proof}
\leanchecked{BEDC.Derived.EqtypeUp.EqtypeClassCarrier\_e0\_anchor\_e1\_tail\_absurd}

\section{The certificate}
\label{sec:eqtype-certificate}

\begin{theorem}[EqType identity semantic name certificate]
\label{thm:eqtype-identity-semantic-namecert}
The identity classifier on histories forms a semantic name certificate when
source, pattern, and ledger are all the self-identity predicate:
$$
\mathsf{SemanticNameCert}
(\lambda h.\ h\hsame h,\lambda h.\ h\hsame h,\lambda h.\ h\hsame h,\hsame).
$$
\end{theorem}
\begin{proof}
The empty history supplies inhabitation. Reflexivity, symmetry, transitivity,
and carrier transport are exactly the corresponding closure laws of $\hsame$.
The pattern and ledger maps are the identity maps on the self-identity
predicate.
\end{proof}
\leanchecked{BEDC.Derived.EqtypeUp.EqtypeIdentity\_semanticNameCert}
