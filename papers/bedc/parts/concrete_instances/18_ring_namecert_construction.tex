\chapter{A Concrete Naming Certificate for \(\RingUp\)}
\label{ch:concrete-instances-ring-namecert}

The ring interface \(\RingUp\) is licensed by a carrier with two binary operations \(+\) and \(\cdot\) such that \((C, +)\) satisfies \autoref{ch:concrete-instances-abgroup-namecert} and \((C, \cdot)\) satisfies \autoref{ch:concrete-instances-monoid-namecert}, together with left and right distributivity of \(\cdot\) over \(+\). It is not a primitive structure; it is licensed only by composing the additive abelian group and multiplicative monoid certificates with the distributive law.

\begin{remark}[Concrete stability fields]
\label{rem:concrete-instances-ring-namecert-roadmap}
This chapter records the shared additive, multiplicative, congruence, and distributivity fields for the \(\RingUp\) naming certificate over BEDC histories.
\end{remark}

\section{Carrier definition}

\begin{definition}[Ring carrier]
\label{def:ring-carrier}
Use a setup with carrier \(C\), classifier, addition, zero, additive inverse, multiplication, one, and ring law fields.
\end{definition}

\section{Concrete singleton history ring instance}
\label{sec:ring-concrete-singleton-history-instance}

\begin{definition}[Concrete singleton history ring instance]
\label{def:concrete-singleton-history-ring-instance}
Let $A_{\emp}$ and $R_{\emp}$ be the singleton carrier and classifier from
Definition~\ref{def:concrete-singleton-history-group-instance}. Define the
operations by
$$
\mathsf{add}(h,k):=\mathsf{append}(h,k),\qquad
\mathsf{neg}(h):=\emp,\qquad
0:=\emp,
$$
and
$$
\mathsf{mul}(h,k):=\mathsf{append}(h,k),\qquad
1:=\emp.
$$
The resulting tuple
$(A_{\emp},R_{\emp},\mathsf{add},0,\mathsf{neg},\mathsf{mul},1)$ is the
concrete singleton history ring instance.
\end{definition}

\begin{theorem}[Concrete singleton history ring laws]
\label{thm:concrete-singleton-history-ring-laws}
The tuple
$(A_{\emp},R_{\emp},\mathsf{add},0,\mathsf{neg},\mathsf{mul},1)$ satisfies
the additive abelian-group laws, multiplicative monoid laws, addition and
multiplication congruence, left distributivity, and right distributivity under
$R_{\emp}$.
\end{theorem}
\begin{proof}
The additive component is the singleton history abelian group of
Theorem~\ref{thm:concrete-singleton-history-abgroup-laws}, read with its
binary operation as $\mathsf{add}$. Thus additive closure, zero, inverse,
associativity, commutativity, and additive congruence are already supplied over
$A_{\emp}$ and $R_{\emp}$.

The multiplicative component is the monoid part contained in
Theorem~\ref{thm:concrete-singleton-history-group-laws}, read with
$\mathsf{mul}(h,k):=\mathsf{append}(h,k)$ and $1:=\emp$. This gives
multiplication closure, left and right unit laws, associativity, and
multiplication congruence under the same classifier.

For left distributivity, assume $A_{\emp}(h)$, $A_{\emp}(k)$, and
$A_{\emp}(l)$. Closure for $\mathsf{add}$ and $\mathsf{mul}$ puts both
$\mathsf{mul}(h,\mathsf{add}(k,l))$ and
$\mathsf{add}(\mathsf{mul}(h,k),\mathsf{mul}(h,l))$ back in $A_{\emp}$.
The singleton witnesses identify each side with $\emp$, and transitivity of
$\hsame$ composes the two endpoint witnesses into the required
$R_{\emp}$ relation.

For right distributivity, the same argument applies to
$\mathsf{mul}(\mathsf{add}(h,k),l)$ and
$\mathsf{add}(\mathsf{mul}(h,l),\mathsf{mul}(k,l))$. Closure gives the two
carrier witnesses, and the singleton endpoint $\emp$ supplies the composed
$\hsame$ witness required by $R_{\emp}$.
\end{proof}
\leanchecked{BEDC.Derived.RingUp.concrete\_singleton\_history\_ring\_laws}

The RingUp schema is instantiated by this concrete $\mathsf{BHist}$ carrier and
operations, so later formalization can target BHist-anchored ring laws rather
than abstract operator fields.

\section{Concrete singleton empty-history ring instance}

\begin{definition}[Singleton empty-history ring instance]
\label{def:singleton-empty-history-ring-instance}
Let $A_{\emp}$ be the singleton history source of
Definition~\ref{def:list-singleton-history-source} at $a_0=\emp$. Define the
classifier
$$
\rho_{\emp}(h,k)\Longleftrightarrow
A_{\emp}(h)\land A_{\emp}(k)\land\hsame(h,k).
$$
Define the concrete ring operations by
$$
\mathsf{add}(h,k):=\emp,\qquad
\mathsf{neg}(h):=\emp,\qquad
0:=\emp,\qquad
\mathsf{mul}(h,k):=\emp,\qquad
1:=\emp.
$$
This is the one-element BEDC history ring over the concrete carrier
$A_{\emp}:\mathsf{BHist}\to\mathsf{Prop}$ and classifier $\rho_{\emp}$.
\end{definition}

\begin{theorem}[Singleton empty-history ring laws]
\label{thm:singleton-empty-history-ring-laws}
The tuple
$(A_{\emp},\rho_{\emp},\mathsf{add},0,\mathsf{neg},\mathsf{mul},1)$ satisfies
carrier inhabitance, additive abelian-group laws, multiplicative monoid laws,
addition and multiplication congruence, left and right distributivity, and zero
multiplication laws under $\rho_{\emp}$.
\end{theorem}
\begin{proof}
Carrier inhabitance is the reflexive endpoint witness $\hsame(\emp,\emp)$.
By Definition~\ref{def:list-singleton-history-source} at $a_0=\emp$, this is
exactly the witness that $\emp$ lies in $A_{\emp}$.

Every operation result in the displayed tuple is definitionally $\emp$. Hence
addition closure, additive inverse closure, multiplication closure, and the
closedness of $0$ and $1$ all use the same singleton carrier witness
$\hsame(\emp,\emp)$.

The additive unit, inverse, associativity, and commutativity laws reduce both
sides to $\emp$. The required endpoint facts for $\rho_{\emp}$ are the
singleton carrier witnesses; the relation component is reflexivity of
$\hsame$, with symmetry and transitivity used only to package classifier
endpoints already identified with $\emp$.

The multiplicative left and right unit laws, associativity, and operation
congruence are handled in the same way. Both source and target products are
definitionally $\emp$, so the classifier relation is $\hsame(\emp,\emp)$.

For left and right distributivity, both the product over a sum and the sum of
products reduce to $\emp$. The zero multiplication laws also reduce both sides
to $\emp$, so each law is supplied by reflexivity of $\hsame$ together with the
two singleton endpoint witnesses.
\end{proof}

\section{The certificate}

\begin{definition}[Ring source specification]
\label{def:ring-source-specification}
The source is an additive \(\AbGroupUp\) and multiplicative \(\MonoidUp\) on the same carrier and classifier.
\end{definition}

\begin{definition}[Ring pattern specification]
\label{def:ring-pattern-specification}
The pattern is expression formation from \(0,1,+,-,\cdot\).
\end{definition}

\begin{definition}[Ring classifier specification]
\label{def:ring-classifier-specification}
The classifier is the shared carrier sameness relation, congruent for both operations.
\end{definition}

\begin{definition}[Ring stability certificate]
\label{def:ring-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item additive abelian-group laws
  \item multiplicative monoid laws
  \item addition and multiplication congruence
  \item left distributivity
  \item right distributivity
  \item zero multiplication laws derived or supplied constructively
\end{enumerate}
\end{definition}
\leanchecked{BEDC.Derived.RingUp.ring\_stability\_certificate\_fields}

\begin{theorem}[Ring additive right inverse]
\label{thm:ring-add-right-inverse}
For BEDC history addition, additive commutativity and the supplied left inverse
law imply the matching right inverse law \(x+(-x)\hsame0\).
\end{theorem}
\begin{proof}
Commute \(x+(-x)\) to \((-x)+x\), then apply the left inverse law.
\end{proof}
\leanchecked{BEDC.Derived.RingUp.ring\_add\_right\_inverse}

\begin{theorem}[Ring additive inverse of zero]
\label{thm:ring-neg-zero}
For BEDC history addition, additive commutativity, the left-zero law, and the
left inverse law imply \(\hsame(-0,0)\).
\end{theorem}
\begin{proof}
Chain \(-0\) backward through the left-zero equation \(0+(-0)\hsame -0\),
commute \(0+(-0)\) to \((-0)+0\), and apply the left inverse law at \(0\).
\end{proof}
\leanchecked{BEDC.Derived.RingUp.ring\_neg\_zero}

\begin{theorem}[Ring duplicate addition collapses to zero]
\label{thm:ring-add-duplicate-eq-zero}
For BEDC history addition, if an element is the same as its duplicate sum
\(a+a\), then the element is the same as zero.
\end{theorem}
\begin{proof}
Add the left inverse of \(a\) on the left, reassociate to bring
\((-a)+a\) together, collapse that pair to zero, and use the left-zero law.
\end{proof}
\leanchecked{BEDC.Derived.RingUp.ring\_add\_duplicate\_eq\_zero}

\begin{theorem}[Ring additive duplicate zero exactness]
\label{thm:ring-add-duplicate-zero-iff}
For BEDC history addition satisfying additive associativity, left zero, left
inverse, and addition congruence, every history $a$ satisfies
$$
a\hsame a+a \Longleftrightarrow a\hsame0.
$$
\end{theorem}
\begin{proof}
The forward implication is exactly the duplicate-addition collapse: apply
\autoref{thm:ring-add-duplicate-eq-zero} to the hypothesis $a\hsame a+a$.

For the reverse implication, use symmetry of $a\hsame0$ to transport the left
summand from $a$ to $0$, and use the hypothesis itself to transport the right
summand from $a$ to $0$ by addition congruence. The resulting comparison
$a+a\hsame0+0$ reduces by the left-zero law to $0$, and transitivity with the
original hypothesis gives $a\hsame a+a$.
\end{proof}
\leanchecked{BEDC.Derived.RingUp.ring\_add\_duplicate\_empty\_iff}

\begin{theorem}[Ring additive right zero]
\label{thm:ring-add-right-zero}
For BEDC history addition, additive commutativity and the supplied left-zero
law imply the matching right-zero law $x+0\hsame x$.
\end{theorem}
\begin{proof}
Commute $x+0$ to $0+x$, then apply the left-zero law.
\end{proof}
\leanchecked{BEDC.Derived.RingUp.ring\_add\_right\_zero}

\begin{theorem}[Ring additive two-sided cancellation]
\label{thm:ring-additive-two-sided-cancellation}
For BEDC history addition satisfying additive associativity, commutativity,
left-zero, left-inverse, and addition congruence, every
$a,b,c:\mathsf{BHist}$ satisfies both cancellation implications:
$$
a+c\hsame b+c \Longrightarrow a\hsame b
$$
and
$$
c+a\hsame c+b \Longrightarrow a\hsame b.
$$
The ten-field fit is: traditional target additive cancellation in a ring; BEDC
source $\RingUp$ over $\mathsf{BHist}$ plus $\GroupUp$ cancellation; carrier
$\mathsf{BHist}$; classifier $\hsame$; stability from additive group laws and
addition congruence; ledger the common addend; core theorem this theorem;
exactness predicate the two displayed implications; standard bridge ordinary
ring additive cancellation; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
First derive the additive right-zero law from
Theorem~\ref{thm:ring-add-right-zero} and the additive right-inverse law from
Theorem~\ref{thm:ring-add-right-inverse}. Together with associativity,
commutativity, left zero, left inverse, and addition congruence, these are the
group-cancellation hypotheses for the additive operation.

For the first implication, read addition as the group operation and apply
Theorem~\ref{thm:group-right-cancellation} to the common right summand $c$.
The hypothesis $a+c\hsame b+c$ then yields $a\hsame b$.

For the second implication, apply
Theorem~\ref{thm:group-left-cancellation} to the common left summand $c$.
The hypothesis $c+a\hsame c+b$ then yields the same conclusion $a\hsame b$.
\end{proof}
\leanchecked{BEDC.Derived.RingUp.ring\_additive\_two\_sided\_cancellation}

\begin{theorem}[Ring negative of sum]
\label{thm:ring-neg-add}
For BEDC history addition and negation satisfying additive associativity,
commutativity, left zero, left inverse, and addition congruence, every
$a,b:\mathsf{BHist}$ satisfies
$$
\hsame(\mathsf{neg}(\mathsf{add}(a,b)),
\mathsf{add}(\mathsf{neg}(a),\mathsf{neg}(b))).
$$
The ten-field fit is: traditional target additive inverse of a sum in a ring;
BEDC source $\RingUp$ over $\mathsf{BHist}$; carrier $\mathsf{BHist}$;
classifier $\hsame$; stability from additive congruence and additive
cancellation; ledger the two addends $a,b$ and the inverse witnesses
$\mathsf{neg}(\mathsf{add}(a,b))$ and
$\mathsf{add}(\mathsf{neg}(a),\mathsf{neg}(b))$; core theorem this theorem;
exactness the displayed equality; downstream consumer CommRing signed-product
normalization and FieldUp denominator inverse normalization; status
$\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
The common element to cancel is $\mathsf{add}(a,b)$. By
Theorem~\ref{thm:ring-add-right-inverse}, the endpoint
$\mathsf{neg}(\mathsf{add}(a,b))$ is a right inverse of this common element.

It remains to see that
$\mathsf{add}(\mathsf{neg}(a),\mathsf{neg}(b))$ is also a right inverse of
$\mathsf{add}(a,b)$. Reassociate the sum
$(a+b)+((-a)+(-b))$, commute the middle terms so that $b+(-b)$ and
$a+(-a)$ appear as adjacent pairs, collapse those pairs using the right
inverse and left inverse rows, and remove the resulting zeros by the left and
right zero laws.

Both endpoints therefore give the same sum with the common left factor
$\mathsf{add}(a,b)$. Apply
Theorem~\ref{thm:ring-additive-two-sided-cancellation} to cancel that factor
and obtain the displayed $\hsame$ comparison.
\end{proof}

\begin{theorem}[Ring additive difference zero exactness]
\label{thm:ring-additive-difference-zero-exact}
For BEDC history addition satisfying additive associativity, commutativity,
left zero, left inverse, and addition congruence, every
$x,y:\mathsf{BHist}$ satisfies
$$
x+(-y)\hsame0 \Longleftrightarrow x\hsame y.
$$
The ten-field fit is: traditional target additive difference exactness; BEDC
source $\RingUp$ over $\mathsf{BHist}$ plus the $\GroupUp$ cancellation spine
already cited by \autoref{thm:ring-additive-two-sided-cancellation}; carrier
$\mathsf{BHist}$; classifier $\hsame$; stability from additive group laws and
addition congruence; ledger the hidden signed summand; core theorem this
theorem; exactness the displayed iff; standard bridge ordinary ring subtraction
exactness; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
First derive the additive right-zero law from
Theorem~\ref{thm:ring-add-right-zero} and the additive right-inverse law from
Theorem~\ref{thm:ring-add-right-inverse}. These give the removable zero and
inverse pairs used on both sides of the equivalence.

For the forward implication, add $y$ to the right of the comparison
$x+(-y)\hsame0$. Associativity rewrites $(x+(-y))+y$ as $x+((-y)+y)$,
commutativity and the right-inverse law collapse $(-y)+y$ to $0$, and the
right-zero law removes that terminal zero. The right endpoint $0+y$ reduces to
$y$ by the left-zero law, leaving $x\hsame y$.

For the reverse implication, transport $x\hsame y$ through addition congruence
against the common summand $-y$. The right side becomes $y+(-y)$, which
Theorem~\ref{thm:ring-add-right-inverse} collapses to $0$, yielding
$x+(-y)\hsame0$ by transitivity.
\end{proof}

\begin{theorem}[Ring multiplication zero absorption]
\label{thm:ring-mul-zero-absorption}
For BEDC history operations satisfying additive associativity, left zero,
left inverse, congruence, and both distributive laws, multiplication by zero on
either side is internally the same as zero.
\end{theorem}
\leanchecked{BEDC.Derived.RingUp.ring\_mul\_zero\_absorption}
\begin{proof}
First cancel a duplicated summand \(a\hsame a+a\) by adding the left inverse,
reassociating, and using the left-zero law. Then apply distributivity to
\(x\cdot(0+0)\) and \((0+0)\cdot x\), transport along \(0+0\hsame0\), and use
the duplicate-cancellation step on the two resulting products.
\end{proof}

\begin{theorem}[Ring zero-classifier factor absorption]
\label{thm:ring-zero-classifier-factor-absorption}
For BEDC history operations satisfying additive associativity, left zero, left
inverse, addition congruence, multiplication congruence, and both distributive
laws, if $\hsame(x,0)$, then every $y:\mathsf{BHist}$ satisfies
$$
\hsame(x\cdot y,0)\land \hsame(y\cdot x,0).
$$
\end{theorem}
\begin{proof}
Invoke \autoref{thm:ring-mul-zero-absorption} to obtain the two literal
zero-factor absorptions. For the left product, multiplication congruence
transports the left factor along $\hsame(x,0)$, comparing $x\cdot y$ with
$0\cdot y$. Transitivity with the left zero-factor absorption gives
$\hsame(x\cdot y,0)$, using symmetry if the congruence step is read in the
opposite direction.

For the right product, multiplication congruence transports the right factor
along the same classifier witness, comparing $y\cdot x$ with $y\cdot0$.
Transitivity with the right zero-factor absorption gives $\hsame(y\cdot x,0)$,
again using symmetry when the displayed comparison is oriented oppositely.
The two transported conclusions form the required conjunction.
\end{proof}
\leanchecked{BEDC.Derived.RingUp.ring\_zero\_classifier\_factor\_absorption}

\begin{theorem}[Ring equal-factor difference annihilation]
\label{thm:ring-equal-factor-difference-annihilation}
For BEDC history ring operations satisfying additive associativity,
commutativity, left zero, left inverse, addition congruence, multiplication
congruence, and both distributive laws, if $x\hsame y$, then every
$z:\mathsf{BHist}$ satisfies
$$
\hsame((x+(-y))\cdot z,0)\land \hsame(z\cdot(x+(-y)),0).
$$
The ten-field fit is: traditional target ordinary ring difference
annihilation; BEDC source $\RingUp$ over $\mathsf{BHist}$; carrier
$\mathsf{BHist}$; classifier $\hsame$; stability from additive group laws,
addition congruence, multiplication congruence, and both distributive laws;
ledger the equal-factor witness and the multiplier; core theorem this theorem;
exactness additive difference zero exactness feeding zero-factor absorption;
standard bridge ordinary ring equal-factor difference annihilation; downstream
use FieldUp and RatUp zero-product reasoning; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Apply the reverse direction of
\autoref{thm:ring-additive-difference-zero-exact} to the classifier witness
$x\hsame y$. This turns the displayed difference into a zero-classified factor,
namely $x+(-y)\hsame0$.

Now apply \autoref{thm:ring-zero-classifier-factor-absorption} with factor
$x+(-y)$ and multiplier $z$. The left component gives
$\hsame((x+(-y))\cdot z,0)$, and the right component gives
$\hsame(z\cdot(x+(-y)),0)$; these two components are exactly the displayed
conjunction.
\end{proof}

\begin{theorem}[Ring zero-classifier signed factor absorption]
\label{thm:ring-zero-classifier-signed-factor-absorption}
For BEDC history ring operations satisfying additive associativity,

$\hsame(-x,0)$.

Apply \autoref{thm:ring-zero-classifier-factor-absorption} with factor $-x$ and
multiplier $y$. This yields the two absorptions $\hsame((-x)y,0)$ and
$\hsame(y(-x),0)$.

Apply the same absorption theorem again with factor $-x$ and multiplier $-y$.
Its left and right conclusions are precisely $\hsame((-x)(-y),0)$ and
$\hsame((-y)(-x),0)$, and the four conclusions package into the stated

\begin{theorem}[Ring left-negative and right-negative products agree]
\label{thm:ring-mul-neg-left-right-same}
For BEDC history operations satisfying additive abelian-group laws, operation
congruence, and both distributive laws, the two negative products agree:
$( -x )y\hsame x(-y)$.

The left-negative and right-negative product sums both collapse to zero after
adding the common summand $xy$ on the right. The additive right-zero and
right-inverse laws supply the hypotheses for right cancellation in the
underlying additive group, so the common summand cancels.
\end{proof}
\leanchecked{BEDC.Derived.RingUp.ring\_mul\_neg\_left\_right\_same}

\begin{theorem}[Ring left-negative product equals negative product]
\label{thm:ring-mul-neg-left-eq-neg-mul}
For BEDC history ring operations satisfying additive associativity,
commutativity, left zero, left inverse, operation congruence, and both
distributive laws, every pair $x,y$ satisfies
$$
\hsame((-x)y,-(xy)).
$$
\end{theorem}
\begin{proof}
The left-negative product sum theorem gives $(-x)y+xy\hsame0$. The additive
right inverse theorem gives $xy+(-(xy))\hsame0$. Applying the additive
left/right inverse uniqueness argument identifies $(-x)y$ with $-(xy)$.
\end{proof}
\leanchecked{BEDC.Derived.RingUp.ring\_mul\_neg\_left\_eq\_neg\_mul}

\begin{theorem}[Ring right-negative product equals negative product]
\label{thm:ring-mul-neg-right-eq-neg-mul}
For BEDC history ring operations satisfying additive associativity,
commutativity, left zero, left inverse, operation congruence, and both
distributive laws, every pair $x,y$ satisfies
$$
\hsame(x(-y),-(xy)).
$$
\end{theorem}
\begin{proof}
Apply left distributivity backward to identify $x(-y)+xy$ with
$x((-y)+y)$. Transport the right factor across the additive left-inverse law,
use right zero absorption for multiplication, and combine the resulting
left-inverse witness with the additive right inverse of $xy$ by inverse
uniqueness.
\end{proof}
\leanchecked{BEDC.Derived.RingUp.ring\_mul\_neg\_right\_eq\_neg\_mul}

\begin{theorem}[Ring double-negative product]
\label{thm:ring-mul-neg-neg-eq-mul}
For BEDC history ring operations satisfying additive associativity,
commutativity, left zero, left inverse, operation congruence, and both
distributive laws, every pair $x,y$ satisfies
$$
\hsame((-x)(-y),xy).
$$
\end{theorem}
\begin{proof}
The left-negative product theorem identifies $(-x)(-y)$ with $-(x(-y))$.
The right-negative product theorem identifies $x(-y)$ with $-(xy)$. Additive
inverse congruence transports this equality through the outer inverse, and
additive inverse involutivity collapses $-(-(xy))$ to $xy$.
\end{proof}
\leanchecked{BEDC.Derived.RingUp.ring\_mul\_neg\_neg\_eq\_mul}
