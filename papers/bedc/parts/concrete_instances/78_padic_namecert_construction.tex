\chapter{A Concrete Naming Certificate for $\PadicUp$}
\label{ch:concrete-instances-padic-namecert}

The $p$-adic interface $\PadicUp$ packages the carrier of regular Cauchy sequences over $\NameCert_{\RatUp}$ (\autoref{ch:concrete-instances-rat-namecert}) under the $p$-adic absolute value, indexed by a prime $p$ from $\NameCert_{\PrimeUp}$ (\autoref{ch:concrete-instances-prime-namecert}). It is the non-archimedean sibling of $\NameCert_{\RealUp}$ (\autoref{ch:concrete-instances-real-namecert}).

\section{Carrier definition}
\label{sec:padic-carrier}

\begin{definition}[Prime scale over unary multiplication]
\label{def:padic-prime-scale}
For histories \(p,e,r\), the predicate \(\mathsf{PadicPrimeScale}(p,e,r)\)
holds when \(p\) is a \(\PrimeUp\)-prime and \(r\) is the unary multiplication
result of \(p\) by the exponent \(e\).
\end{definition}
\leandef{BEDC.Derived.PadicUp.PadicPrimeScale}

\begin{theorem}[Prime scale unary components]
\label{thm:padic-prime-scale-unary-components}
If \(\mathsf{PadicPrimeScale}(p,e,r)\), then the prime base \(p\), exponent
\(e\), and result \(r\) are all unary histories.
\end{theorem}
\begin{proof}
The scale predicate consists of a \(\mathsf{NatPrime}(p)\) witness and a
\(\mathsf{NatMul}(p,e,r)\) witness. The prime witness supplies the unary
carrier field for \(p\). The unary multiplication witness supplies the unary
field for \(e\), and together with the unary field for \(p\), supplies the
unary field for the result \(r\).
\end{proof}
\leanchecked{BEDC.Derived.PadicUp.PadicPrimeScale\_unary\_components}

\begin{theorem}[Prime scale commutes with the exponent]
\label{thm:padic-prime-scale-prime-exponent-comm}
If \(p\) and \(q\) are unary histories, and
\(\mathsf{PadicPrimeScale}(p,q,r)\) and \(\mathsf{PadicPrimeScale}(q,p,s)\)
both hold, then \(r\hsame s\).
\end{theorem}
\begin{proof}
Each prime-scale witness contains a concrete \(\mathsf{NatMul}\) field. The
commutativity theorem for unary multiplication applies to those two fields and
identifies the two scale results up to \(\hsame\).
\end{proof}
\leanchecked{BEDC.Derived.PadicUp.PadicPrimeScale\_prime\_exponent\_comm\_hsame}

\begin{theorem}[Prime scale totality]
\label{thm:padic-prime-scale-total}
If \(p\) is a \(\PrimeUp\)-prime and \(e\) is a unary history, then some unary
history \(r\) realizes \(\mathsf{PadicPrimeScale}(p,e,r)\).
\end{theorem}
\begin{proof}
The prime witness contains the unary carrier field for \(p\). Unary
multiplication totality applied to this carrier and to the exponent carrier
produces a unary result \(r\) together with the \(\mathsf{NatMul}(p,e,r)\)
witness. Pairing the original prime witness with that multiplication witness
is precisely \(\mathsf{PadicPrimeScale}(p,e,r)\).
\end{proof}
\leanchecked{BEDC.Derived.PadicUp.PadicPrimeScale\_total}

\begin{theorem}[Prime scale exists uniquely]
\label{thm:padic-prime-scale-exists-unique}
If \(p\) is a \(\PrimeUp\)-prime and \(e\) is a unary history, then there is a
history \(r\) with \(\mathsf{PadicPrimeScale}(p,e,r)\), and every other such
scale result is \(\hsame\)-equal to \(r\).
\end{theorem}
\begin{proof}
Prime-scale totality supplies a concrete result witness. Any competing
prime-scale result has the same prime base and exponent in its
\(\mathsf{NatMul}\) field, so unary multiplication functionality identifies
the two result histories.
\end{proof}
\leanchecked{BEDC.Derived.PadicUp.PadicPrimeScale\_exists\_unique}

\begin{theorem}[Prime scale empty result exactness]
\label{thm:padic-prime-scale-empty-result-iff-empty-exponent}
If \(\mathsf{PadicPrimeScale}(p,e,r)\), then \(r\hsame\emp\) exactly when
\(e\hsame\emp\).
\end{theorem}
\begin{proof}
The forward direction uses the \(\PrimeUp\) prime field to rule out
\(p\hsame\emp\), then applies the empty-result exactness theorem for
\(\mathsf{NatMul}\) with a nonempty multiplicand. The reverse direction
identifies the exponent with \(\emp\), so the zero constructor of
\(\mathsf{NatMul}\) identifies the result with \(\emp\).
\end{proof}
\leanchecked{BEDC.Derived.PadicUp.PadicPrimeScale\_empty\_result\_iff\_empty\_exponent}

\begin{lemma}[Prime scale successor exponent inversion]
\label{lem:padic-prime-scale-succ-exponent-inversion}
If \(\mathsf{PadicPrimeScale}(p,\Eone(q),r)\), then there is a predecessor
result \(n\) with \(\mathsf{PadicPrimeScale}(p,q,n)\) and \(\Cont(n,p,r)\).
\end{lemma}
\begin{proof}
The scale witness contains the prime field for \(p\) and a multiplication
witness \(\mathsf{NatMul}(p,\Eone(q),r)\). Successor inversion for unary
multiplication extracts the predecessor product \(n\) and the continuation
\(\Cont(n,p,r)\). Reusing the same prime field packages the predecessor as a
prime-scale witness.
\end{proof}
\leanchecked{BEDC.Derived.PadicUp.PadicPrimeScale\_succ\_exponent\_inversion}

\begin{theorem}[Prime scale positive-exponent continuation readback]
\label{thm:padic-prime-scale-positive-exponent-cont-readback}
If the exponent $e$ is unary, not history-same as $\emp$, and
\(\mathsf{PadicPrimeScale}(p,e,r)\) holds, then \(e\) has a unary tail \(t\)
and the scale result is obtained from a predecessor scale by one continuation
with the prime base:
$$
\exists t,n.\ \Unary(t)\land e=\Eone(t)\land
\mathsf{PadicPrimeScale}(p,t,n)\land\Cont(n,p,r).
$$
\end{theorem}
\begin{proof}
The unary nonempty exponent has a one-headed tail. After substituting that
tail into the scale witness, successor-exponent inversion for
\(\mathsf{PadicPrimeScale}\) exposes the predecessor scale and the continuation
from the predecessor result by \(p\).
\end{proof}
\leanchecked{BEDC.Derived.PadicUp.PadicPrimeScale\_positive\_exponent\_cont\_readback}

\begin{theorem}[Prime scale successor exponent factorization]
\label{thm:padic-prime-scale-succ-exponent-factorization}
A successor-exponent prime scale $\mathsf{PadicPrimeScale}(p,\Eone(q),r)$
holds exactly when some predecessor result $n$ carries
$\mathsf{PadicPrimeScale}(p,q,n)$ and a continuation $\Cont(n,p,r)$.
\end{theorem}
\begin{proof}
The forward direction is successor-exponent inversion. Conversely, the
predecessor scale gives the prime field and a concrete
$\mathsf{NatMul}(p,q,n)$ witness; the successor constructor for
$\mathsf{NatMul}$, applied to $\Cont(n,p,r)$, rebuilds the scale over
$\Eone(q)$.
\end{proof}
\leanchecked{BEDC.Derived.PadicUp.PadicPrimeScale\_succ\_exponent\_factorization\_iff}

\begin{lemma}[Prime scale exponent-result cases]
\label{lem:padic-prime-scale-exponent-result-cases}
Every \(\mathsf{PadicPrimeScale}(p,e,r)\) witness is either the empty exponent
case, with \(e\hsame\emp\) and \(r\hsame\emp\), or a successor exponent case:
there are histories \(e_0\) and \(r_0\) such that
\(e=\Eone(e_0)\), \(\mathsf{PadicPrimeScale}(p,e_0,r_0)\), and
\(\Cont(r_0,p,r)\).
\end{lemma}
\begin{proof}
Apply the exponent-result case split for the underlying
\(\mathsf{NatMul}(p,e,r)\) field. In the successor branch, the same prime
witness for \(p\) packages the predecessor multiplication witness as a
predecessor prime-scale witness.
\end{proof}
\leanchecked{BEDC.Derived.PadicUp.PadicPrimeScale\_exponent\_result\_cases}

\begin{theorem}[Prime scale successor predecessor uniqueness]
\label{thm:padic-prime-scale-succ-exponent-predecessor-unique}
If $\mathsf{PadicPrimeScale}(p,\Eone(q),r)$, then there is a predecessor
result $n$ with $\mathsf{PadicPrimeScale}(p,q,n)$ and $\Cont(n,p,r)$, and
every other predecessor scale over $q$ is history-same to $n$.
\end{theorem}
\begin{proof}
Successor-exponent inversion supplies the predecessor result $n$, its
prime-scale witness over $q$, and the continuation witness into $r$. Any
competing predecessor scale over the same base $p$ and exponent $q$ has a
second $\mathsf{NatMul}(p,q,-)$ field. Unary multiplication functionality,
using the unary carrier field of the prime witness, identifies the two
predecessor histories.
\end{proof}
\leanchecked{BEDC.Derived.PadicUp.PadicPrimeScale\_succ\_exponent\_predecessor\_unique}

\begin{theorem}[Prime scale unit exponent reads back to the prime]
\label{thm:padic-prime-scale-unit-exponent-result-prime}
If $\mathsf{PadicPrimeScale}(p,\Eone(\emp),result)$ holds, then the scale
result is history-same to the prime base:
$$
result\hsame p.
$$
\end{theorem}
\begin{proof}
Successor-exponent inversion extracts a predecessor result over the empty
exponent and a continuation from that predecessor through $p$ to $result$.
Empty-result exactness for prime scales identifies the predecessor with
$\emp$. Comparing the displayed continuation with the left-unit continuation
for $p$ identifies $result$ with $p$.
\end{proof}
\leanchecked{BEDC.Derived.PadicUp.PadicPrimeScale\_unit\_exponent\_result\_prime\_hsame}

\begin{theorem}[Prime scale append continuation closure]
\label{thm:padic-prime-scale-append-cont-closure}
If \(\mathsf{PadicPrimeScale}(p,w,n)\), \(\mathsf{PadicPrimeScale}(p,q,e)\),
and \(\mathsf{Cont}(n,e,r)\), then
\(\mathsf{PadicPrimeScale}(p,\mathsf{append}(w,q),r)\).
\end{theorem}
\begin{proof}
Both scale witnesses share the same prime field for \(p\). The multiplication
fields are concrete \(\mathsf{NatMul}\) derivations, so append continuation
closure for \(\mathsf{NatMul}\), applied to the supplied continuation witness,
gives the scale witness for the appended exponent.
\end{proof}
\leanchecked{BEDC.Derived.PadicUp.PadicPrimeScale\_append\_cont\_closure}

\begin{theorem}[Padic natural multiplication append exponent decomposition]
\label{thm:padic-natmul-append-exponent-decomposition}
For unary histories $d,w,q$, any multiplication witness over the appended
exponent decomposes into multiplication witnesses over the two exponent
factors and a continuation between their results:
$$
\mathsf{NatMul}(d,\mathsf{append}(w,q),r)
\quad\Longrightarrow\quad
\exists n,e.\ \mathsf{NatMul}(d,w,n)\land
\mathsf{NatMul}(d,q,e)\land\ContR(n,e,r).
$$
\end{theorem}
\begin{proof}
Induct on the unary right exponent $q$. For the empty exponent, take the left
result to be the displayed result $r$ and the right result to be $\emp$. In the
one-headed case, successor inversion for $\mathsf{NatMul}$ extracts the
previous product over $\mathsf{append}(w,q)$ and the final continuation by
$d$. The induction hypothesis decomposes the previous product into left and
right results; composing its continuation with the final continuation and
using append associativity gives the required continuation from the left
result to the successor right product.
\end{proof}
\leanchecked{BEDC.Derived.PadicUp.PadicNatMul\_append\_exponent\_decomposition}

\begin{theorem}[Prime scale append factorization]
\label{thm:padic-prime-scale-append-factorization}
For unary exponent histories $w,q$, a prime scale over the appended exponent
is exactly a pair of prime-scale factor results linked by continuation:
$$
\mathsf{PadicPrimeScale}(p,\mathsf{append}(w,q),r)
\Longleftrightarrow
\exists n,e.\ \mathsf{PadicPrimeScale}(p,w,n)\land
\mathsf{PadicPrimeScale}(p,q,e)\land\ContR(n,e,r).
$$
\end{theorem}
\begin{proof}
The forward direction applies append-exponent decomposition to the
$\mathsf{NatMul}$ field of the scale witness and reattaches the common prime
field to both factor multiplications. The reverse direction is precisely
prime-scale append continuation closure applied to the two factor scales and
their continuation row.
\end{proof}
\leanchecked{BEDC.Derived.PadicUp.PadicPrimeScale\_append\_factorization\_iff}

\label{thm:padic-prime-scale-append-exponent-decomposition}

\begin{theorem}[Prime scale append successor right factorization]
\label{thm:padic-prime-scale-append-succ-right-factorization}
For unary histories $w$ and $q$, a prime scale over
$\mathsf{append}(w,\Eone(q))$ is exactly a left prime-scale result over $w$,
a predecessor right prime-scale result over $q$, a one-step continuation by
the prime base, and the final continuation from the left result:
$$
\mathsf{PadicPrimeScale}(p,\mathsf{append}(w,\Eone(q)),r)
\Longleftrightarrow
\exists n,e,s.\,
\mathsf{PadicPrimeScale}(p,w,n)\land
\mathsf{PadicPrimeScale}(p,q,e)\land
\ContR(e,p,s)\land
\ContR(n,s,r).
$$
\end{theorem}
\begin{proof}
The forward direction first applies append-exponent decomposition with the
right factor $\Eone(q)$. The right factor scale is then inverted by the
successor-exponent factorization theorem, producing the predecessor scale over
$q$ and its continuation by $p$. Conversely, rebuild the right successor scale
from the predecessor row and then apply append continuation closure with the
final continuation row.
\end{proof}
\leanchecked{BEDC.Derived.PadicUp.PadicPrimeScale\_append\_succ\_right\_exponent\_factorization\_iff}

\end{proof}

$\hsame$, and continuation respects that endpoint comparison against the
canonical continuation $\ContR(n,p,\mathsf{append}(n,p))$.
\end{proof}

\begin{theorem}[Prime scale two unit exponents have nonempty result]
\label{thm:padic-prime-scale-append-unit-exponents-result-not-empty}
If
\(\mathsf{PadicPrimeScale}(p,\mathsf{append}(\Eone(\emp),\Eone(\emp)),result)\)

tail to be empty, a contradiction.

factor identifications across the displayed continuations and identifies the
two composite results.
\end{proof}

\begin{theorem}[Prime scale append factor results commute]
\label{thm:padic-prime-scale-append-factor-results-comm-hsame}
Let $\mathsf{PadicPrimeScale}(p,w,n)$ and
$\mathsf{PadicPrimeScale}(p,q,e)$ hold. If the two factor results are linked
by both continuation orders $\ContR(n,e,r)$ and $\ContR(e,n,s)$, then the
continuation results satisfy \(r\hsame s\).
\end{theorem}
\begin{proof}
The scale witnesses expose unary carriers for the two factor results \(n\) and
\(e\). Unary append commutativity identifies
\(\mathsf{append}(n,e)\) with \(\mathsf{append}(e,n)\), and the two displayed
continuation ledgers identify \(r\) and \(s\) with those append histories.
\end{proof}
\leanchecked{BEDC.Derived.PadicUp.PadicPrimeScale\_append\_factor\_results\_comm\_hsame}

\begin{theorem}[Prime scale append totality]

\(\mathsf{Cont}(n,e,\mathsf{append}(n,e))\) then feeds
\autoref{thm:padic-prime-scale-append-cont-closure}, producing the scale
witness for \(\mathsf{append}(w,q)\) together with the displayed factor
witnesses.
\end{proof}
