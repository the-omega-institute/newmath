\begin{theorem}[Metric distance witness empty-boundary visible-context distance empty]
\label{thm:metric-distance-witness-empty-boundary-visible-context-distance-empty}
For histories $p,q,d$, a visible-context distance witness whose central left and
right endpoints are both empty identifies the central distance with the empty
history and gives zero central distance depth:
$$
\mathsf{MetricDistanceWitness}(p\cdot\emp,\emp\cdot q,(p\cdot d)\cdot q)
\Longrightarrow
\hsame(d,\emp)\wedge\mathsf{MetricDistanceDepth}(d)=0.
$$
\end{theorem}
\begin{proof}
Apply \autoref{thm:metric-distance-witness-visible-context-iff} to peel away
the visible prefix and suffix. The remaining central witness is
$\mathsf{MetricDistanceWitness}(\emp,\emp,d)$. The empty-left boundary theorem
identifies $d$ with $\emp$; case analysis on that classifier gives the zero
depth equation.
\end{proof}
\leanchecked{BEDC.Derived.MetricUp.MetricDistanceWitness\_empty\_boundary\_visible\_context\_distance\_empty}

\begin{theorem}[Metric empty-boundary visible context append unit laws]
\label{thm:metric-distance-witness-empty-boundary-visible-context-append-unit-laws}
For histories $p,q,d,l,r$, a visible-context distance witness whose central
endpoints are both empty makes the central distance act as a left and right
append unit:
$$
\mathsf{MetricDistanceWitness}(p\cdot\emp,\emp\cdot q,(p\cdot d)\cdot q)
\Longrightarrow
\mathsf{append}(d,r)=r\ \wedge\ \mathsf{append}(l,d)=l.
$$
\end{theorem}
\begin{proof}
The empty-boundary distance theorem identifies $d$ with $\emp$. Case analysis
on that classifier reduces the two displayed equations to the left and right
unit laws for finite-kernel append.
\end{proof}
\leanchecked{BEDC.Derived.MetricUp.MetricDistanceWitness\_empty\_boundary\_visible\_context\_append\_unit\_laws}

\begin{theorem}[Metric empty-boundary visible context continuation depth add]
\label{thm:metric-distance-witness-empty-boundary-visible-context-continuation-depth-add}
For histories $p,q,d,l,r,mid,out$, an empty-boundary visible-context distance
witness lets a continuation through $d$ splice into the direct continuation
from $l$ to $r$, so the output depth is the sum of the endpoint depths:
$$
\mathsf{MetricDistanceWitness}(p\cdot\emp,\emp\cdot q,(p\cdot d)\cdot q)
\Longrightarrow
\mathsf{UnaryHistory}(l)\Longrightarrow\mathsf{UnaryHistory}(r)\Longrightarrow
\ContR(l,d,mid)\Longrightarrow\ContR(mid,r,out)\Longrightarrow
\mathsf{MetricDistanceDepth}(out)=
\mathsf{MetricDistanceDepth}(l)+\mathsf{MetricDistanceDepth}(r).
$$
\end{theorem}
\begin{proof}
The empty-boundary continuation splice removes the central empty distance from
the two-step ledger and gives $\ContR(l,r,out)$. Together with the unary
endpoint hypotheses, unary continuation closure supplies the output carrier,
so these fields form a concrete metric distance witness from $l$ to $r$.
Metric distance depth additivity then gives the displayed equation.
\end{proof}
\leanchecked{BEDC.Derived.MetricUp.MetricDistanceWitness\_empty\_boundary\_visible\_context\_continuation\_depth\_add}
