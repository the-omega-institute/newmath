\begin{theorem}[Group singleton two-conjugator orbit classifier]
\label{thm:group-singleton-two-conjugator-orbit-classifier}
For histories $s,t,x,y:\mathsf{BHist}$ in the concrete singleton history group,
write
$$
\mathsf{Conj}_u(z):=\mathsf{mul}(\mathsf{mul}(u,z),\mathsf{inv}(u)).
$$
If $A_{\emp}(s)$ and $A_{\emp}(t)$, then
$$
\mathsf{Conj}_s(x)\sim_{N_{\emp}}\mathsf{Conj}_t(y)
\Longleftrightarrow R_{\emp}(x,y).
$$
The ten-field fit is: traditional target terminal two-conjugator orbit
classifier; BEDC source concrete singleton $\GroupUp$; carrier $A_{\emp}$;
classifier $R_{\emp}$; stability from normalizer action coverage and terminal
classifier packaging; ledger the two conjugators and endpoint witnesses;
exactness the displayed iff; downstream consumer $\QuotientGroupUp$; status
$\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Apply \autoref{thm:group-singleton-two-conjugator-orbit-coverage} under the
two displayed conjugator carrier witnesses. This rewrites the normalizer orbit
relation between the acted endpoints as the endpoint carrier condition
$$
A_{\emp}(x)\land A_{\emp}(y).
$$

Apply the first row of
\autoref{thm:group-singleton-terminal-classifier-package} to the same endpoint
pair. It identifies $R_{\emp}(x,y)$ with
$A_{\emp}(x)\land A_{\emp}(y)$.

Composing the first equivalence with the reverse direction of the second gives
exactly
$$
\mathsf{Conj}_s(x)\sim_{N_{\emp}}\mathsf{Conj}_t(y)
\Longleftrightarrow R_{\emp}(x,y).
$$
\end{proof}

\begin{theorem}[Group singleton normalizer orbit action chain]
\label{thm:group-singleton-normalizer-orbit-action-chain}
For histories $s,x,y,z:\mathsf{BHist}$ carried by $A_{\emp}$ in the concrete
singleton history group, write
$$
\mathsf{Conj}_s(w):=\mathsf{mul}(\mathsf{mul}(s,w),\mathsf{inv}(s)).
$$
If $A_{\emp}(s)$, $A_{\emp}(x)$, $A_{\emp}(y)$, $A_{\emp}(z)$,
$x\sim_{N_{\emp}}y$, and $y\sim_{N_{\emp}}z$, then
$$
\mathsf{Conj}_s(x)\sim_{N_{\emp}}\mathsf{Conj}_s(y)
\land
\mathsf{Conj}_s(y)\sim_{N_{\emp}}\mathsf{Conj}_s(z)
\land
\mathsf{Conj}_s(x)\sim_{N_{\emp}}\mathsf{Conj}_s(z).
$$
The ten-field fit is: traditional target action congruence on an orbit
relation; BEDC source concrete singleton $\GroupUp$; carrier $A_{\emp}$;
classifier the normalizer orbit relation exact against $R_{\emp}$; stability
from the normalizer action certificate, orbit action invariance, and orbit
equivalence; ledger the conjugator and two orbit witnesses; core theorem this
block; exactness the displayed chain preservation; standard bridge terminal
group action on its one orbit; status $\mathsf{ProofTarget}$.
\end{theorem}
\leanchecked{BEDC.Derived.GroupUp.GroupSingletonNormalizerOrbit\_action\_chain}
\begin{proof}
First apply
\autoref{thm:group-singleton-normalizer-action-certificate} to the three pairs
$(s,x)$, $(s,y)$, and $(s,z)$. These instances carry
$\mathsf{Conj}_s(x)$, $\mathsf{Conj}_s(y)$, and $\mathsf{Conj}_s(z)$ inside
$A_{\emp}$.

Next apply
\autoref{thm:group-singleton-normalizer-orbit-action-invariance} to the input
edge $x\sim_{N_{\emp}}y$. Apply the same theorem to the input edge
$y\sim_{N_{\emp}}z$. This gives the two acted orbit edges
$$
\mathsf{Conj}_s(x)\sim_{N_{\emp}}\mathsf{Conj}_s(y)
\quad\text{and}\quad
\mathsf{Conj}_s(y)\sim_{N_{\emp}}\mathsf{Conj}_s(z).
$$

Finally use the transitivity row of
\autoref{thm:group-singleton-normalizer-orbit-equivalence} on the three acted
endpoints. The carrier rows supplied by the action certificate and the two
acted orbit edges compose to
$\mathsf{Conj}_s(x)\sim_{N_{\emp}}\mathsf{Conj}_s(z)$, and the three orbit
relations form the displayed conjunction.
\end{proof}

\begin{theorem}[Group singleton normalizer orbit action path iff]
\label{thm:group-singleton-normalizer-orbit-action-path-iff}
For histories $s,x,y,z:\mathsf{BHist}$ carried by $A_{\emp}$ in the concrete
singleton history group, write
$$
\mathsf{Conj}_s(w):=\mathsf{mul}(\mathsf{mul}(s,w),\mathsf{inv}(s)).
$$
If $A_{\emp}(s)$, $A_{\emp}(x)$, $A_{\emp}(y)$, and $A_{\emp}(z)$, then the
two-edge normalizer orbit path is reflected exactly by the terminal action:
$$
\bigl(\mathsf{Conj}_s(x)\sim_{N_{\emp}}\mathsf{Conj}_s(y)
\land
\mathsf{Conj}_s(y)\sim_{N_{\emp}}\mathsf{Conj}_s(z)\bigr)
\Longleftrightarrow
\bigl(x\sim_{N_{\emp}}y\land y\sim_{N_{\emp}}z\bigr).
$$
The transitive edge is reflected exactly as well:
$$
\mathsf{Conj}_s(x)\sim_{N_{\emp}}\mathsf{Conj}_s(z)
\Longleftrightarrow
x\sim_{N_{\emp}}z.
$$
The ten-field fit is: traditional target terminal action path exactness; BEDC
source concrete singleton $\GroupUp$; carrier $A_{\emp}$; classifier
$\sim_{N_{\emp}}$; stability from action identity iff, action invariance, and
orbit equivalence; ledger the conjugator plus the two source orbit witnesses;
core theorem this block; exactness the two displayed iff clauses; downstream
consumer $\QuotientGroupUp$ action-path classifier; status
$\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
First apply \autoref{thm:group-singleton-normalizer-action-certificate} to the
three pairs $(s,x)$, $(s,y)$, and $(s,z)$. This supplies the carrier rows for
$\mathsf{Conj}_s(x)$, $\mathsf{Conj}_s(y)$, and $\mathsf{Conj}_s(z)$.

For the forward direction of the two-edge path equivalence, assume the two
acted orbit edges. Apply
\autoref{thm:group-singleton-normalizer-action-orbit-identity-iff} to
$\mathsf{Conj}_s(x)\sim_{N_{\emp}}\mathsf{Conj}_s(y)$, with the carrier row for
$\mathsf{Conj}_s(y)$ supplied by the action certificate, to replace the acted
left endpoint by $x$.

Use the symmetry row of
\autoref{thm:group-singleton-normalizer-orbit-equivalence} to turn this edge
around. The same identity iff, now with source endpoint $y$ and target endpoint
$x$, replaces $\mathsf{Conj}_s(y)$ by $y$; symmetry turns the result back to
$x\sim_{N_{\emp}}y$.

The second acted edge is handled in the same way. First replace the acted left
endpoint $\mathsf{Conj}_s(y)$ by $y$, then use orbit symmetry, the identity iff
for $z$, and symmetry again. This gives $y\sim_{N_{\emp}}z$.

For the reverse direction of the two-edge path equivalence, assume
$x\sim_{N_{\emp}}y$ and $y\sim_{N_{\emp}}z$. Apply
\autoref{thm:group-singleton-normalizer-orbit-action-invariance} to each source
edge. The two outputs are exactly

$$
\mathsf{Conj}_s(x)\sim_{N_{\emp}}\mathsf{Conj}_s(y)
\quad\text{and}\quad
\mathsf{Conj}_s(y)\sim_{N_{\emp}}\mathsf{Conj}_s(z).
$$

The transitivity row of
\autoref{thm:group-singleton-normalizer-orbit-equivalence} also composes these
acted edges to the acted transitive edge.

For the forward direction of the transitive-edge equivalence, assume
$\mathsf{Conj}_s(x)\sim_{N_{\emp}}\mathsf{Conj}_s(z)$. The action identity iff
replaces the acted left endpoint by $x$, again using the carrier row for
$\mathsf{Conj}_s(z)$, and gives $x\sim_{N_{\emp}}\mathsf{Conj}_s(z)$.

Apply the symmetry row, the identity-iff row for $z$ with target $x$, and
symmetry once more to replace the remaining acted endpoint by $z$. The result is
$x\sim_{N_{\emp}}z$.

For the reverse direction of the transitive-edge equivalence, assume
$x\sim_{N_{\emp}}z$. Apply
\autoref{thm:group-singleton-normalizer-orbit-action-invariance} directly to
this source edge. It gives
$\mathsf{Conj}_s(x)\sim_{N_{\emp}}\mathsf{Conj}_s(z)$, which is the acted
transitive edge.
\end{proof}

\begin{theorem}[Group singleton normalizer orbit action conjugator independence]
\label{thm:group-singleton-normalizer-orbit-action-conjugator-independence}
For histories $s,t,x,y,z:\mathsf{BHist}$ carried by $A_{\emp}$ in the concrete
singleton history group, write
$$
\mathsf{Conj}_u(w):=\mathsf{mul}(\mathsf{mul}(u,w),\mathsf{inv}(u)).
$$
If $A_{\emp}(s)$, $A_{\emp}(t)$, $A_{\emp}(x)$, $A_{\emp}(y)$, and
$A_{\emp}(z)$, then the terminal two-edge normalizer action path is independent
of the carried conjugator:
$$
\bigl(
\mathsf{Conj}_s(x)\sim_{N_{\emp}}\mathsf{Conj}_s(y)
\land
\mathsf{Conj}_s(y)\sim_{N_{\emp}}\mathsf{Conj}_s(z)
\bigr)
\Longleftrightarrow
\bigl(
\mathsf{Conj}_t(x)\sim_{N_{\emp}}\mathsf{Conj}_t(y)
\land
\mathsf{Conj}_t(y)\sim_{N_{\emp}}\mathsf{Conj}_t(z)
\bigr).
$$
The induced transitive edge is independent of the carried conjugator as well:
$$
\mathsf{Conj}_s(x)\sim_{N_{\emp}}\mathsf{Conj}_s(z)
\Longleftrightarrow
\mathsf{Conj}_t(x)\sim_{N_{\emp}}\mathsf{Conj}_t(z).
$$
The ten-field fit is: traditional target conjugator-independent terminal action
paths; BEDC source concrete singleton $\GroupUp$; carrier $A_{\emp}$;
classifier $\sim_{N_{\emp}}$; stability from
\autoref{thm:group-singleton-normalizer-orbit-action-path-iff}; ledger the two
conjugators and the two-edge source path; core theorem this block; exactness
the displayed iff clauses; downstream consumer $\QuotientGroupUp$ action-path
classifier; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Apply \autoref{thm:group-singleton-normalizer-orbit-action-path-iff} with
conjugator $s$. Its first iff identifies the $s$-acted two-edge path with the
source path $x\sim_{N_{\emp}}y\land y\sim_{N_{\emp}}z$, and its second iff
identifies the $s$-acted transitive edge with the source edge
$x\sim_{N_{\emp}}z$.

Apply the same theorem with conjugator $t$. Its two iff clauses identify the
$t$-acted two-edge path and the $t$-acted transitive edge with the same source
path and source edge.

Compose the two iff chains through those shared source classifiers. The
two-edge chain gives the displayed path equivalence, and the transitive-edge
chain gives the displayed endpoint equivalence.
\end{proof}

\begin{theorem}[Group singleton two-conjugator orbit path classifier]
\label{thm:group-singleton-two-conjugator-orbit-path-classifier}
For histories $s,t,u,x,y,z:\mathsf{BHist}$ in the concrete singleton history
group, write
$$
\mathsf{Conj}_q(r):=\mathsf{mul}(\mathsf{mul}(q,r),\mathsf{inv}(q)).
$$
If $A_{\emp}(s)$, $A_{\emp}(t)$, and $A_{\emp}(u)$, then the two-edge acted
normalizer path with independent conjugators is classified exactly by the two
terminal classifier edges:
$$
\bigl(\mathsf{Conj}_s(x)\sim_{N_{\emp}}\mathsf{Conj}_t(y)
\land
\mathsf{Conj}_t(y)\sim_{N_{\emp}}\mathsf{Conj}_u(z)\bigr)
\Longleftrightarrow
\bigl(R_{\emp}(x,y)\land R_{\emp}(y,z)\bigr).
$$
When either side of this equivalence is available, the transitive acted edge
also holds:
$$
\mathsf{Conj}_s(x)\sim_{N_{\emp}}\mathsf{Conj}_u(z).
$$
The ten-field fit is: traditional target terminal mixed-conjugator orbit-path
exactness; BEDC source concrete singleton $\GroupUp$; carrier $A_{\emp}$;
classifier the normalizer orbit relation exact against $R_{\emp}$; stability
from two-conjugator orbit classification and orbit transitivity; ledger the
three conjugators and two endpoint classifier witnesses; core theorem this
block; exactness the displayed iff plus transitive edge; downstream consumer
$\QuotientGroupUp$; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
For the forward direction of the path equivalence, assume the two acted orbit
edges. Apply
\autoref{thm:group-singleton-two-conjugator-orbit-classifier} to the
$(s,t,x,y)$ edge, using the carrier witnesses for $s$ and $t$, to obtain
$R_{\emp}(x,y)$.

Apply the same classifier to the $(t,u,y,z)$ edge, using the carrier witnesses
for $t$ and $u$, to obtain $R_{\emp}(y,z)$. These two outputs are exactly the
right-hand side of the displayed equivalence.

For the reverse direction, assume $R_{\emp}(x,y)$ and $R_{\emp}(y,z)$. Apply
\autoref{thm:group-singleton-two-conjugator-orbit-classifier} in the reverse
direction to $(s,t,x,y)$, and then to $(t,u,y,z)$. The two results are precisely
the acted path edges.

It remains to produce the transitive acted edge. If the acted path side is
available, use the transitivity row of
\autoref{thm:group-singleton-normalizer-orbit-equivalence} on the three acted
endpoints. The two input edges compose to
$\mathsf{Conj}_s(x)\sim_{N_{\emp}}\mathsf{Conj}_u(z)$.

If the terminal classifier side is available instead, first use the reverse
direction of the two equivalence steps above to obtain the two acted path edges.
The same transitivity row of
\autoref{thm:group-singleton-normalizer-orbit-equivalence} then composes them to
the displayed transitive acted edge.
\end{proof}

\begin{theorem}[Group singleton three-conjugator orbit path compression]
\label{thm:group-singleton-three-conjugator-orbit-path-compression}
For histories $s,t,u,v,x,y,z,w:\mathsf{BHist}$ in the concrete singleton
history group, write
$$
\mathsf{Conj}_a(b):=\mathsf{mul}(\mathsf{mul}(a,b),\mathsf{inv}(a)).
$$
If $A_{\emp}(s)$, $A_{\emp}(t)$, $A_{\emp}(u)$, and $A_{\emp}(v)$, then the
three acted normalizer edges with independent conjugators are classified
exactly by the three terminal classifier edges:
$$
\bigl(
\mathsf{Conj}_s(x)\sim_{N_{\emp}}\mathsf{Conj}_t(y)
\land
\mathsf{Conj}_t(y)\sim_{N_{\emp}}\mathsf{Conj}_u(z)
\land
\mathsf{Conj}_u(z)\sim_{N_{\emp}}\mathsf{Conj}_v(w)
\bigr)
\Longleftrightarrow
\bigl(R_{\emp}(x,y)\land R_{\emp}(y,z)\land R_{\emp}(z,w)\bigr).
$$
Moreover, any such path compresses to the endpoint acted edge and the terminal
classifier edge:
$$
\bigl(
\mathsf{Conj}_s(x)\sim_{N_{\emp}}\mathsf{Conj}_t(y)
\land
\mathsf{Conj}_t(y)\sim_{N_{\emp}}\mathsf{Conj}_u(z)
\land
\mathsf{Conj}_u(z)\sim_{N_{\emp}}\mathsf{Conj}_v(w)
\bigr)
\implies
\bigl(\mathsf{Conj}_s(x)\sim_{N_{\emp}}\mathsf{Conj}_v(w)
\land R_{\emp}(x,w)\bigr).
$$
The ten-field fit is: traditional target terminal three-step conjugation orbit
path; BEDC source concrete singleton $\GroupUp$ over $\mathsf{BHist}$; carrier
$A_{\emp}$; classifier $R_{\emp}$ and the normalizer orbit relation; stability
from adjacent two-conjugator orbit classification plus terminal classifier
transitivity; ledger the four conjugators and three path edges; core theorem
this block; exactness the displayed iff plus compressed endpoint edge;
downstream consumer $\QuotientGroupUp$ path compression; status
$\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
For the forward direction of the path equivalence, assume the three acted orbit
edges. Apply
\autoref{thm:group-singleton-two-conjugator-orbit-classifier} to the
$(s,t,x,y)$ edge, then to the $(t,u,y,z)$ edge, and then to the $(u,v,z,w)$
edge. The carrier witnesses for the corresponding conjugator pairs supply
$R_{\emp}(x,y)$, $R_{\emp}(y,z)$, and $R_{\emp}(z,w)$.

For the reverse direction, assume $R_{\emp}(x,y)$, $R_{\emp}(y,z)$, and
$R_{\emp}(z,w)$. Apply the same two-conjugator orbit classifier in the reverse
direction to $(s,t,x,y)$, $(t,u,y,z)$, and $(u,v,z,w)$. These are precisely the
three acted normalizer edges.

It remains to compress a given acted path. Use
\autoref{thm:group-singleton-two-conjugator-orbit-path-classifier} on the first
two acted edges to obtain $R_{\emp}(x,z)$, and use the adjacent classifier on
the third edge to obtain $R_{\emp}(z,w)$.

Apply the terminal transitivity supplied by
\autoref{thm:group-singleton-terminal-classifier-package}. The classifier row
reads the two terminal edges as singleton carrier rows for $x$, $z$, and $w$,
and repacks them as $R_{\emp}(x,w)$.

Finally apply
\autoref{thm:group-singleton-two-conjugator-orbit-classifier} in the reverse
direction to $(s,v,x,w)$ with the carrier witnesses for $s$ and $v$. This gives
$\mathsf{Conj}_s(x)\sim_{N_{\emp}}\mathsf{Conj}_v(w)$, and together with
$R_{\emp}(x,w)$ it is the displayed compressed conclusion.
\end{proof}

\begin{theorem}[Group singleton four-conjugator orbit path compression]
\label{thm:group-singleton-four-conjugator-orbit-path-compression}

history group, write
$$
\mathsf{Conj}_a(b):=\mathsf{mul}(\mathsf{mul}(a,b),\mathsf{inv}(a)).

$$
\bigl(
\mathsf{Conj}_s(x)\sim_{N_{\emp}}\mathsf{Conj}_t(y)
\land
\mathsf{Conj}_t(y)\sim_{N_{\emp}}\mathsf{Conj}_u(z)
\land
\mathsf{Conj}_u(z)\sim_{N_{\emp}}\mathsf{Conj}_v(w)
\land
\mathsf{Conj}_v(w)\sim_{N_{\emp}}\mathsf{Conj}_r(q)
\bigr)
\Longleftrightarrow
\bigl(
R_{\emp}(x,y)\land R_{\emp}(y,z)\land R_{\emp}(z,w)\land R_{\emp}(w,q)
\bigr).
$$
Moreover, any such path compresses to the endpoint acted edge and the terminal
classifier edge:
$$
\bigl(
\mathsf{Conj}_s(x)\sim_{N_{\emp}}\mathsf{Conj}_t(y)
\land
\mathsf{Conj}_t(y)\sim_{N_{\emp}}\mathsf{Conj}_u(z)
\land
\mathsf{Conj}_u(z)\sim_{N_{\emp}}\mathsf{Conj}_v(w)

\mathsf{Conj}_v(w)\sim_{N_{\emp}}\mathsf{Conj}_r(q)

\implies

\begin{proof}

edges. Apply
\autoref{thm:group-singleton-three-conjugator-orbit-path-compression} to the
first three acted edges, using the carrier witnesses for $s$, $t$, $u$, and

Apply
\autoref{thm:group-singleton-two-conjugator-orbit-classifier} to the fourth

is the right-hand side of the displayed equivalence.

For the reverse direction, assume $R_{\emp}(x,y)$, $R_{\emp}(y,z)$,
$R_{\emp}(z,w)$, and $R_{\emp}(w,q)$. Apply
\autoref{thm:group-singleton-two-conjugator-orbit-classifier} in the reverse
direction to $(s,t,x,y)$, $(t,u,y,z)$, $(u,v,z,w)$, and $(v,r,w,q)$. These are
precisely the four acted normalizer edges.

It remains to compress a given acted path. The first paragraph supplies
$R_{\emp}(x,w)$ from the first three acted edges, and the fourth acted edge
supplies $R_{\emp}(w,q)$ by the two-conjugator classifier.

Apply the terminal transitivity supplied by
\autoref{thm:group-singleton-terminal-classifier-package}. The two terminal
edges $R_{\emp}(x,w)$ and $R_{\emp}(w,q)$ compose to $R_{\emp}(x,q)$.

Finally apply
\autoref{thm:group-singleton-two-conjugator-orbit-classifier} in the reverse
direction to $(s,r,x,q)$ with the carrier witnesses for $s$ and $r$. This gives
$\mathsf{Conj}_s(x)\sim_{N_{\emp}}\mathsf{Conj}_r(q)$, and together with
$R_{\emp}(x,q)$ it is the displayed compressed conclusion.
