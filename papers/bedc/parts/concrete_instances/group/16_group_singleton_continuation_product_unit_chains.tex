\begin{theorem}[Group singleton double continuation product-unit chain coverage]
\label{thm:group-singleton-double-continuation-product-unit-chain-coverage}
For histories $P,Q,S,T,U,V:\mathsf{BHist}$ with $\ContR(P,Q,S)$ and
$\ContR(S,T,U)$, the concrete singleton $\GroupUp$ instance satisfies
$$
R_{\emp}(\mathsf{append}(U,V),\emp)\Longleftrightarrow
A_{\emp}(P)\land A_{\emp}(Q)\land A_{\emp}(T)\land A_{\emp}(V).
$$
The ten-field fit names the traditional target terminal-group two-step
continuation product-unit coverage; BEDC source concrete singleton $\GroupUp$
over $\mathsf{BHist}$ plus two $\ContR$ witnesses and append; carrier
$A_{\emp}$; classifier $R_{\emp}$; stability from continuation product-unit
coverage, continuation terminal collapse, and terminal classifier packaging;
ledger the two continuation witnesses and suffix endpoint; exactness the
displayed iff; downstream consumer $\QuotientGroupUp$ path compression; and
status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Apply \autoref{thm:group-singleton-continuation-product-unit-triple-coverage}
to the continuation witness $\ContR(S,T,U)$. This rewrites the left side as
$$
A_{\emp}(S)\land A_{\emp}(T)\land A_{\emp}(V).
$$

Use \autoref{thm:group-singleton-continuation-terminal-collapse} under the
continuation witness $\ContR(P,Q,S)$ to identify $A_{\emp}(S)$ with
$R_{\emp}(P,Q)$. The intermediate condition becomes
$$
R_{\emp}(P,Q)\land A_{\emp}(T)\land A_{\emp}(V).
$$

Finally, apply the first row of
\autoref{thm:group-singleton-terminal-classifier-package} to rewrite
$R_{\emp}(P,Q)$ as $A_{\emp}(P)\land A_{\emp}(Q)$. Reassociating conjunctions
gives exactly
$$
A_{\emp}(P)\land A_{\emp}(Q)\land A_{\emp}(T)\land A_{\emp}(V).
$$
\end{proof}
\leanchecked{BEDC.Derived.GroupUp.GroupSingletonClassifier\_double\_continuation\_product\_unit\_chain\_coverage\_iff}

\begin{theorem}[Group singleton double continuation product-unit visible exclusion]
\label{thm:group-singleton-double-continuation-product-unit-visible-exclusion}
For histories $P,Q,S,T,U,V,r:\mathsf{BHist}$ with $\ContR(P,Q,S)$ and
$\ContR(S,T,U)$, if
$$
R_{\emp}(\mathsf{append}(U,V),\emp)
$$
holds, then the concrete singleton $\GroupUp$ instance excludes visible results
for both continuation witnesses:
$$
\bigl(\ContR(P,Q,\Ezero(r))\Rightarrow\mathsf{False}\bigr)\land
\bigl(\ContR(P,Q,\Eone(r))\Rightarrow\mathsf{False}\bigr)\land
\bigl(\ContR(S,T,\Ezero(r))\Rightarrow\mathsf{False}\bigr)\land
\bigl(\ContR(S,T,\Eone(r))\Rightarrow\mathsf{False}\bigr).
$$
The ten-field template names the traditional target terminal-group two-step
continuation visibility guard; BEDC source concrete singleton $\GroupUp$ over
$\mathsf{BHist}$ with two $\ContR$ witnesses and append; carrier $A_{\emp}$;
classifier $R_{\emp}$; stability from double continuation product-unit coverage,
terminal classifier packaging, and continuation terminal collapse; ledger the
two continuation witnesses, suffix endpoint, and visible result endpoint; core
theorem this block; exactness the four displayed contradictions; downstream
consumer $\QuotientGroupUp$ path compression and subgroup visibility guards; and
status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Apply
\autoref{thm:group-singleton-double-continuation-product-unit-chain-coverage} to
the two continuation witnesses and the product-unit classifier. The forward
direction gives
$$
A_{\emp}(P)\land A_{\emp}(Q)\land A_{\emp}(T)\land A_{\emp}(V).
$$

Keep the first two carrier components. Use the reverse direction of the first
row of \autoref{thm:group-singleton-terminal-classifier-package} to repack
$A_{\emp}(P)\land A_{\emp}(Q)$ as $R_{\emp}(P,Q)$.

Apply the visible-result part of
\autoref{thm:group-singleton-continuation-terminal-collapse} under
$\ContR(P,Q,S)$. The repacked input classifier excludes
$\ContR(P,Q,\Ezero(r))$ and $\ContR(P,Q,\Eone(r))$, and the same collapse
identifies the terminal condition $A_{\emp}(S)$.

Combine this $A_{\emp}(S)$ with the retained factor $A_{\emp}(T)$. The reverse
direction of the first row of
\autoref{thm:group-singleton-terminal-classifier-package} repacks
$A_{\emp}(S)\land A_{\emp}(T)$ as $R_{\emp}(S,T)$.

Apply the visible-result part of
\autoref{thm:group-singleton-continuation-terminal-collapse} under
$\ContR(S,T,U)$. The repacked second input classifier excludes
$\ContR(S,T,\Ezero(r))$ and $\ContR(S,T,\Eone(r))$, giving the four displayed
contradictions.
\end{proof}
\leanchecked{BEDC.Derived.GroupUp.GroupSingletonClassifier\_double\_continuation\_product\_unit\_visible\_exclusion}

\begin{theorem}[Group singleton triple continuation product-unit chain coverage]
\label{thm:group-singleton-triple-continuation-product-unit-chain-coverage}
For histories $P,Q,S,T,U,V,W,X:\mathsf{BHist}$ with $\ContR(P,Q,S)$,
$\ContR(S,T,U)$, and $\ContR(U,V,W)$, the concrete singleton $\GroupUp$
instance satisfies
$$
R_{\emp}(\mathsf{append}(W,X),\emp)\Longleftrightarrow
A_{\emp}(P)\land A_{\emp}(Q)\land A_{\emp}(T)\land A_{\emp}(V)\land
A_{\emp}(X).
$$
The ten-field fit names terminal-group three-step continuation product-unit
coverage; BEDC source concrete singleton $\GroupUp$ over $\mathsf{BHist}$ plus
three $\ContR$ witnesses and append; carrier $A_{\emp}$; classifier
$R_{\emp}$; stability from continuation product-unit triple coverage,
continuation terminal collapse, and terminal classifier packaging; ledger the
three continuation witnesses and suffix endpoint; exactness the displayed iff;
downstream consumer $\QuotientGroupUp$ path compression and group orbit
compression; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
The displayed equivalence is obtained by chaining reversible classifier
identifications. Apply
\autoref{thm:group-singleton-continuation-product-unit-triple-coverage} to
$\ContR(U,V,W)$, which identifies the left side with
$$
A_{\emp}(U)\land A_{\emp}(V)\land A_{\emp}(X).
$$

Use \autoref{thm:group-singleton-continuation-terminal-collapse} under
$\ContR(S,T,U)$ to identify $A_{\emp}(U)$ with $R_{\emp}(S,T)$. Then apply the
first row of \autoref{thm:group-singleton-terminal-classifier-package} to
replace $R_{\emp}(S,T)$ by
$$
A_{\emp}(S)\land A_{\emp}(T).
$$

It remains to expand the new terminal factor $A_{\emp}(S)$. Use
\autoref{thm:group-singleton-continuation-terminal-collapse} under
$\ContR(P,Q,S)$ to identify $A_{\emp}(S)$ with $R_{\emp}(P,Q)$, and apply the
same terminal classifier package to replace it by
$$
A_{\emp}(P)\land A_{\emp}(Q).
$$

Reassociating the conjunctions gives exactly
$$
A_{\emp}(P)\land A_{\emp}(Q)\land A_{\emp}(T)\land A_{\emp}(V)\land
A_{\emp}(X).
$$
\end{proof}
