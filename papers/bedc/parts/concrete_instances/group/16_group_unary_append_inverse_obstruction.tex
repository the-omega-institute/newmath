\section{Unary append inverse obstruction}
\label{sec:group-unary-append-inverse-obstruction}

\begin{theorem}[Unary append unit-product factor exactness]
\label{thm:unary-append-unit-product-factor-exactness}
For histories $h,k:\mathsf{BHist}$, the concrete unary-history append product
is classified as the monoid unit exactly when both factors are classified as
the unit:
$$
\mathsf{MonoidHistoryClassifier}(\mathsf{UnaryHistory})
(\mathsf{append}(h,k),\emp)
\Longleftrightarrow
\mathsf{MonoidHistoryClassifier}(\mathsf{UnaryHistory})(h,\emp)
\land
\mathsf{MonoidHistoryClassifier}(\mathsf{UnaryHistory})(k,\emp).
$$
Equivalently, the append operation in the full unary monoid cannot supply a
non-singleton inverse pair for the concrete singleton $\GroupUp$ instance.
\end{theorem}
\begin{proof}
Unfold the classifier from
\autoref{def:concrete-history-monoid-instance}. The forward direction gives
the unary carrier witnesses for $\mathsf{append}(h,k)$ and $\emp$ together
with $\hsame(\mathsf{append}(h,k),\emp)$. Apply the BEDC kernel theorem
\texttt{append\_eq\_empty\_iff} to this empty-result comparison; it splits the
append endpoint into $h\hsame\emp$ and $k\hsame\emp$. Repacking these two
singleton endpoint witnesses with the unary carrier data gives the two
displayed monoid-classifier factors.

For the reverse direction, assume the two displayed classifier factors. Their
carrier fields give $\mathsf{UnaryHistory}(h)$ and $\mathsf{UnaryHistory}(k)$,
while their sameness fields identify both histories with $\emp$. Unary closure
of append from \autoref{thm:concrete-history-monoid-laws} supplies
$\mathsf{UnaryHistory}(\mathsf{append}(h,k))$, and the constructor direction
of \texttt{append\_eq\_empty\_iff} rebuilds
$\hsame(\mathsf{append}(h,k),\emp)$. Together with
$\mathsf{UnaryHistory}(\emp)$, these fields repack the displayed
$\mathsf{MonoidHistoryClassifier}(\mathsf{UnaryHistory})
(\mathsf{append}(h,k),\emp)$.
\end{proof}

\begin{theorem}[Unary append group subcarrier unit collapse]
\label{thm:unary-append-group-subcarrier-unit-collapse}
Let $C:\mathsf{BHist}\to\mathsf{Prop}$ be a concrete unary subcarrier, and let
$i:\mathsf{BHist}\to\mathsf{BHist}$ be an inverse witness. Assume
$$
\begin{array}{ll}
\text{\rm carrier:} &
\forall h:\mathsf{BHist},\ C(h)\Rightarrow \mathsf{UnaryHistory}(h),\\
\text{\rm unit:} & C(\emp),\\
\text{\rm append closure:} &
\forall h,k:\mathsf{BHist},\
C(h)\Rightarrow C(k)\Rightarrow C(\mathsf{append}(h,k)),\\
\text{\rm inverse closure:} &
\forall h:\mathsf{BHist},\
C(h)\Rightarrow C(i(h)),\\
\text{\rm inverse products:} &
\forall h:\mathsf{BHist},\
C(h)\Rightarrow{}\\
&\quad
\mathsf{MonoidHistoryClassifier}(\mathsf{UnaryHistory})
(\mathsf{append}(h,i(h)),\emp)\land{}\\
&\quad
\mathsf{MonoidHistoryClassifier}(\mathsf{UnaryHistory})
(\mathsf{append}(i(h),h),\emp).
\end{array}
$$
Then every carried element lies in the empty-unit classifier fiber:
$$
\forall h:\mathsf{BHist},\
C(h)\Rightarrow
\mathsf{MonoidHistoryClassifier}(\mathsf{UnaryHistory})(h,\emp).
$$
\end{theorem}
\begin{proof}
Fix $h:\mathsf{BHist}$ with $C(h)$. The inverse-product hypothesis gives the
right-inverse classifier row
$$
\mathsf{MonoidHistoryClassifier}(\mathsf{UnaryHistory})
(\mathsf{append}(h,i(h)),\emp).
$$

Apply \autoref{thm:unary-append-unit-product-factor-exactness} in the forward
direction to this row. The resulting conjunction has left factor
$\mathsf{MonoidHistoryClassifier}(\mathsf{UnaryHistory})(h,\emp)$, which is
the required unit-fiber conclusion for the chosen carried element.

The left-inverse row, append closure, inverse closure, and explicit unit field
remain part of the concrete group-interface scope. They certify that $C$ is
being tested as a carried unary append instance, while the collapse argument
itself only needs the right inverse and the unit-product factor exactness
theorem.
\end{proof}

Fit: the traditional target is the append-inverse group interface; the BEDC
source is $C\subseteq\mathsf{UnaryHistory}$; the carrier is $C$; the classifier
is inherited from $\mathsf{MonoidHistoryClassifier}(\mathsf{UnaryHistory})$;
stability is supplied by append closure and inverse closure; the ledger is the
inverse witness $i$; the core theorem is this collapse block together with
\autoref{thm:unary-append-unit-product-factor-exactness}; exactness is
unit-fiber collapse; the downstream consumers are $\GroupUp$ and
$\QuotientGroupUp$; the status is $\mathsf{ProofTarget}$.

\begin{theorem}[Full unary append inverse obstruction]
\label{thm:full-unary-append-inverse-obstruction}
There is no total inverse operation on the full unary-history append carrier:
for every operation $\mathsf{inv}:\mathsf{BHist}\to\mathsf{BHist}$, if
$$
\begin{array}{l}
\forall h:\mathsf{BHist},\
\mathsf{UnaryHistory}(h)\Rightarrow{}\\
\qquad
\mathsf{UnaryHistory}(\mathsf{inv}(h))\land{}\\
\qquad
\mathsf{MonoidHistoryClassifier}(\mathsf{UnaryHistory})
(\mathsf{append}(h,\mathsf{inv}(h)),\emp),
\end{array}
$$
then $\mathsf{False}$.
\end{theorem}
\begin{proof}
Assume such an operation $\mathsf{inv}$ and its displayed total inverse law.
The history $\Eone(\emp)$ is a concrete nonempty unary history, so the law
applies to $h=\Eone(\emp)$ and gives the unit-product classifier for
$\mathsf{append}(\Eone(\emp),\mathsf{inv}(\Eone(\emp)))$.

Apply \autoref{thm:unary-append-unit-product-factor-exactness} to that
classifier. The left factor of the resulting conjunction gives

$$
\mathsf{MonoidHistoryClassifier}(\mathsf{UnaryHistory})(\Eone(\emp),\emp).
$$

Unfold \autoref{def:concrete-history-monoid-instance}: the classifier contains
the endpoint carrier witnesses together with an internal history-sameness
witness $\Eone(\emp)\hsame\emp$. This is impossible by
\autoref{thm:history-no-confusion}, since an $\Eone$-history and $\emp$ have
different outer constructors. Hence the assumed total inverse law yields
$\mathsf{False}$.
\end{proof}

\begin{theorem}[Unary append inverse completion singleton boundary]
\label{thm:group-unary-append-inverse-completion-singleton-boundary}
Let $C:\mathsf{BHist}\to\mathsf{Prop}$ be a concrete subcarrier of
$\mathsf{UnaryHistory}$, and let $i:\mathsf{BHist}\to\mathsf{BHist}$ be an
inverse operation. Assume
$$
\begin{array}{ll}
\text{\rm carrier:} &
\forall h:\mathsf{BHist},\ C(h)\Rightarrow \mathsf{UnaryHistory}(h),\\
\text{\rm unit:} & C(\emp),\\
\text{\rm append closure:} &
\forall h,k:\mathsf{BHist},\
C(h)\Rightarrow C(k)\Rightarrow C(\mathsf{append}(h,k)),\\
\text{\rm inverse closure:} &
\forall h:\mathsf{BHist},\
C(h)\Rightarrow C(i(h)),\\
\text{\rm inverse products:} &
\forall h:\mathsf{BHist},\
C(h)\Rightarrow{}\\
&\quad
\mathsf{MonoidHistoryClassifier}(\mathsf{UnaryHistory})
(\mathsf{append}(h,i(h)),\emp)\land{}\\
&\quad
\mathsf{MonoidHistoryClassifier}(\mathsf{UnaryHistory})
(\mathsf{append}(i(h),h),\emp).
\end{array}
$$
Then every carried element lies in the singleton unit fiber:
$$
\forall h:\mathsf{BHist},\
C(h)\Rightarrow
\mathsf{MonoidHistoryClassifier}(\mathsf{UnaryHistory})(h,\emp).
$$
Moreover, if $C(\Eone(\emp))$ is also supplied, these same rows imply
$\mathsf{False}$.
\end{theorem}
\begin{proof}
Apply \autoref{thm:unary-append-group-subcarrier-unit-collapse} to the
displayed carrier, unit, append-closure, inverse-closure, and inverse-product
rows. This gives the singleton unit-fiber classifier for every carried history
in $C$.

Now assume the additional nonempty carried row $C(\Eone(\emp))$. Instantiate
the unit-fiber conclusion at $\Eone(\emp)$ to obtain

$$
\mathsf{MonoidHistoryClassifier}(\mathsf{UnaryHistory})(\Eone(\emp),\emp).
$$

Unfold the concrete monoid classifier as in
\autoref{thm:full-unary-append-inverse-obstruction}. The classifier contains a
history-sameness witness $\Eone(\emp)\hsame\emp$ between histories with
different outer constructors.

Apply \autoref{thm:history-no-confusion}. The $\Eone$ constructor cannot be
identified with $\emp$, so the additional carried nonempty row contradicts the
same inverse-completion package. Thus the only valid concrete unary append
group carrier is the singleton unit fiber.
\end{proof}
