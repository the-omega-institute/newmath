\section{Concrete singleton equation solver}
\label{sec:group-singleton-equation-solver}

\begin{theorem}[Group singleton left equation exactness]
\label{thm:group-singleton-left-equation-exact}
In the concrete singleton history group of
Definition~\ref{def:concrete-singleton-history-group-instance}, with carrier
$A_{\emp}$, classifier $R_{\emp}$, multiplication
$\mathsf{mul}:=\mathsf{append}$, unit $\emp$, and inverse
$\mathsf{inv}(h):=\emp$, the left equation row is exact for histories
$a,x,b:\mathsf{BHist}$. If $A_{\emp}(a)$, $A_{\emp}(x)$, and
$A_{\emp}(b)$, then
$$
R_{\emp}(\mathsf{mul}(a,x),b)
\Longleftrightarrow
R_{\emp}(x,\mathsf{mul}(\mathsf{inv}(a),b)).
$$
\end{theorem}
\begin{proof}
Theorem~\ref{thm:concrete-singleton-history-group-laws} supplies the concrete
group rows for the same carrier, classifier, multiplication, unit, and inverse.
Multiplication closure and inverse closure place both displayed endpoints in
$A_{\emp}$.

For the forward direction, unpack
$R_{\emp}(\mathsf{mul}(a,x),b)$ to the $\hsame$ component
$\mathsf{mul}(a,x)\hsame b$. Apply
Theorem~\ref{thm:group-left-mul-equation-exactness}; the resulting
$\hsame(x,\mathsf{mul}(\mathsf{inv}(a),b))$ repacks with the carrier rows as
$R_{\emp}(x,\mathsf{mul}(\mathsf{inv}(a),b))$.

For the reverse direction, start from the unpacked component
$\hsame(x,\mathsf{mul}(\mathsf{inv}(a),b))$. The backward direction of
Theorem~\ref{thm:group-left-mul-equation-exactness} gives
$\mathsf{mul}(a,x)\hsame b$, and the same endpoint carrier rows repack it as
$R_{\emp}(\mathsf{mul}(a,x),b)$.
\end{proof}
\leanchecked{BEDC.Derived.GroupUp.GroupSingletonClassifier\_left\_equation\_exact\_iff}

\begin{proof}

\begin{theorem}[Group singleton two-sided cancellation]
\label{thm:group-singleton-two-sided-cancel}
In the concrete singleton history group of
Definition~\ref{def:concrete-singleton-history-group-instance}, with carrier
$A_{\emp}$, classifier $R_{\emp}$, multiplication
$\mathsf{mul}:=\mathsf{append}$, unit $\emp$, and inverse
$\mathsf{inv}(h):=\emp$, two-sided cancellation is exact for histories
$a,b,c,d:\mathsf{BHist}$. If $A_{\emp}(a)$, $A_{\emp}(b)$, $A_{\emp}(c)$, and
$A_{\emp}(d)$, then
$$
R_{\emp}(\mathsf{mul}(\mathsf{mul}(a,b),c),
\mathsf{mul}(\mathsf{mul}(a,d),c))
\Longrightarrow R_{\emp}(b,d).
$$
\end{theorem}
\begin{proof}
Theorem~\ref{thm:concrete-singleton-history-group-laws} supplies the concrete
group rows for the same carrier, classifier, multiplication, unit, and inverse.
Repeated multiplication closure places both displayed products in $A_{\emp}$.

Unpacking the assumed classifier gives the $\hsame$ component between those two
products. Apply Theorem~\ref{thm:group-two-sided-cancellation}; the result is
$\hsame(b,d)$, and the original carrier hypotheses $A_{\emp}(b)$ and
$A_{\emp}(d)$ repack it as $R_{\emp}(b,d)$.
\end{proof}
\leanchecked{BEDC.Derived.GroupUp.GroupSingletonClassifier\_two\_sided\_cancel}

\begin{theorem}[Group singleton equation-cancellation concrete solver]
\label{thm:group-singleton-equation-cancellation-concrete-solver}
In the concrete singleton history group of
Definition~\ref{def:concrete-singleton-history-group-instance}, the package of
Theorem~\ref{thm:group-singleton-left-equation-exact},
Theorem~\ref{thm:group-singleton-right-equation-exact}, and
Theorem~\ref{thm:group-singleton-two-sided-cancel} supplies the concrete
equation-cancellation solver over $\mathsf{BHist}$.

The ten-field fit is: traditional target terminal-group equation solving;
BEDC source concrete singleton $\GroupUp$ over $\mathsf{BHist}$; carrier
$A_{\emp}$; classifier $R_{\emp}$; stability from concrete group laws plus
generic left and right equation exactness and two-sided cancellation; ledger
equation factors and inverse endpoints; core theorem list
Theorem~\ref{thm:concrete-singleton-history-group-laws},
Theorem~\ref{thm:group-left-mul-equation-exactness},
Theorem~\ref{thm:group-right-mul-equation-exactness}, and
Theorem~\ref{thm:group-two-sided-cancellation}; exactness the three displayed
rows; downstream consumers $\QuotientGroupUp$ and $\FieldUp$ algebra-reduct
packages; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
The left equation row is Theorem~\ref{thm:group-singleton-left-equation-exact},
the right equation row is Theorem~\ref{thm:group-singleton-right-equation-exact},
and the cancellation row is Theorem~\ref{thm:group-singleton-two-sided-cancel}.

These three rows use the same carrier, classifier, multiplication, unit, inverse,
and concrete group-law source, so they assemble into one solver package with the
listed source, stability, ledger, exactness, consumer, and status fields.
\end{proof}
\leanchecked{BEDC.Derived.GroupUp.GroupSingletonClassifier\_equation\_cancellation\_solver}
