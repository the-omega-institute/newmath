\section{Group concrete history instance surface}
\label{sec:group-concrete-history-instance-unlock}

\begin{definition}[Concrete GroupUp history instance]
\label{def:group-concrete-history-instance}
The concrete $\GroupUp$ history instance is the singleton empty-history package
of \autoref{def:group-singleton-empty-history-instance} read as a direct
$\mathsf{BHist}$ surface. Its carrier is
$$
A_{\emp}(h) := \hsame(h,\emp).
$$
Its classifier is $R_{\emp}$, its multiplication is the append continuation
row
$$
\mathsf{mul}_{\emp}(h,k):=\mathsf{append}(h,k),
$$
its unit is $e_{\emp}:=\emp$, and its inverse sends every carried endpoint to
the constant empty endpoint
$$
\mathsf{inv}_{\emp}(h):=\emp.
$$
Thus the instance is not an abstract carrier with external operations. It is
the displayed $\mathsf{BHist}$ singleton carrier, append $\Cont$ ledger,
empty endpoint, $\hsame$ classifier, and constant inverse row.
\end{definition}

\begin{theorem}[Group concrete inverse surface]
\label{thm:group-concrete-inverse-surface}
For the concrete $\GroupUp$ history instance of
\autoref{def:group-concrete-history-instance}, inverse closure and the two
inverse laws are read entirely from the singleton empty-history laws and the
append unit rows. If $A_{\emp}(h)$, then
$A_{\emp}(\mathsf{inv}_{\emp}(h))$ holds, and both products
$\mathsf{mul}_{\emp}(\mathsf{inv}_{\emp}(h),h)$ and
$\mathsf{mul}_{\emp}(h,\mathsf{inv}_{\emp}(h))$ classify with the unit
$e_{\emp}$ through $R_{\emp}$.
\end{theorem}
\begin{proof}
Start from the concrete definition. The inverse endpoint is the constant
empty history, so inverse closure is the reflexive row
$\hsame(\emp,\emp)$.

For the left inverse law, expand
$\mathsf{mul}_{\emp}(\mathsf{inv}_{\emp}(h),h)$ as
$\mathsf{append}(\emp,h)$. The left append unit row identifies this endpoint
with $h$, and the carrier witness $A_{\emp}(h)$ identifies $h$ with $\emp$.
These two $\hsame$ rows package the required $R_{\emp}$ comparison with
$e_{\emp}$.

For the right inverse law, expand
$\mathsf{mul}_{\emp}(h,\mathsf{inv}_{\emp}(h))$ as
$\mathsf{append}(h,\emp)$. The right append unit row identifies this endpoint
with $h$, and the same carrier witness repacks it as an $R_{\emp}$ comparison
with the unit.

These are exactly the inverse rows already included in
\autoref{thm:group-singleton-empty-history-laws}; the concrete surface only
names the $\mathsf{BHist}$ rows that carry them.
\end{proof}

\begin{theorem}[Group concrete instance schema surface]
\label{thm:group-concrete-instance-schema-unlock}
The concrete $\GroupUp$ history instance of
\autoref{def:group-concrete-history-instance} supplies a non-parametric target
surface for the algebra hierarchy. Its carrier, classifier, multiplication,
unit, inverse, closure rows, unit rows, associativity row, congruence row, and
inverse rows are all the displayed singleton $\mathsf{BHist}$, append
$\Cont$, $\hsame$, and empty-endpoint rows of
\autoref{thm:group-singleton-empty-history-laws}. No external carrier family,
operation table, quotient presentation, or host-level group object is part of
the target surface.
\end{theorem}
\begin{proof}
Read the certificate rows from
\autoref{thm:group-singleton-empty-history-namecert}. The semantic carrier and
classifier are $A_{\emp}$ and $R_{\emp}$, and the algebraic fields are the
append multiplication, empty unit, and constant empty inverse.

Apply \autoref{thm:group-singleton-empty-history-laws}. It supplies
inhabitance, multiplication closure, the two unit laws, associativity,
congruence, inverse closure, and the two inverse laws for these exact rows.

Finally apply \autoref{thm:group-concrete-inverse-surface} to make the inverse
rows visible as part of the concrete $\mathsf{BHist}$ surface. Since every
field has been read from the singleton carrier and append ledger, the target
is the concrete instance itself rather than a parameterized group schema.
\end{proof}

\begin{definition}[Group finite-word history instance]
\label{def:group-finite-word-history-instance}
The finite-word $\GroupUp$ history instance has carrier
$\mathsf{WordCarrier}_{\GroupUp}$ consisting of finite $\ListUp$ word spines
whose entries are singleton carried endpoints of
\autoref{def:group-singleton-empty-history-instance}. Thus a word
$w=\mathsf{cons}(h_1,\ldots,\mathsf{cons}(h_n,\mathsf{nil}))$ is carried when
each $h_i$ satisfies $A_{\emp}(h_i)$, with the empty word carried by the nil
constructor.

The classifier
$\mathsf{WordClass}_{\GroupUp}(w,w')$ is the componentwise lift of $\hsame$,
together with equality of the finite word spine. The product is concatenation
of finite words through the displayed append continuation ledger
$$
\mathsf{mul}_{\mathrm{word}}(w,w'):=w\mathbin{+\!\!+}w',
$$
the unit is $\mathsf{nil}$, and inverse is reversal with the singleton inverse
row applied to each component:
$$
\mathsf{inv}_{\mathrm{word}}(h_1,\ldots,h_n)
:=(\mathsf{inv}_{\emp}(h_n),\ldots,\mathsf{inv}_{\emp}(h_1)).
$$
No host group operation is exported; every endpoint remains a finite
$\mathsf{BHist}$ word with its componentwise singleton ledger.
\end{definition}

\begin{theorem}[Group finite-word operation closure]
\label{thm:group-finite-word-operation-closure}
For the finite-word $\GroupUp$ history instance of
\autoref{def:group-finite-word-history-instance}, concatenation of two carried
finite-word histories is carried. If $w\sim_{\mathrm{word}}w'$ and
$v\sim_{\mathrm{word}}v'$, then
$\mathsf{mul}_{\mathrm{word}}(w,v)\sim_{\mathrm{word}}
\mathsf{mul}_{\mathrm{word}}(w',v')$. The empty word is a two-sided unit, and
associativity is the list-spine append associativity row together with the
singleton append laws of \autoref{thm:group-singleton-empty-history-laws}.
\end{theorem}
\begin{proof}
Induct on the first finite word. In the nil case, concatenation returns the
second carried word, and the left unit law is the nil append equation. The
right unit law follows by the corresponding induction on the carried word.

In the cons step, the head is carried by the componentwise carrier hypothesis
and the tail product is carried by the induction hypothesis. Reattaching the
head gives a carried concatenation spine, so no endpoint leaves the singleton
$\mathsf{BHist}$ surface.

Classifier transport is componentwise. The head comparison is the displayed
$\hsame$ row, while the tail comparison is transported by the induction
hypothesis. The list-spine equality part records that concatenation preserves
the finite shape on both sides.

Associativity is the standard finite-list append reassociation, with each
component endpoint governed by
\autoref{thm:group-singleton-empty-history-laws}. The local concrete schema
surface of \autoref{thm:group-concrete-instance-schema-unlock} supplies the same
append ledger, so the finite-word operation remains the concrete
$\mathsf{BHist}$ instance rather than an abstract group law.
\end{proof}

\begin{theorem}[Group finite-word inverse boundary]
\label{thm:group-finite-word-inverse-boundary}
For every carried finite-word history $w$, the reversed componentwise singleton
inverse $\mathsf{inv}_{\mathrm{word}}(w)$ is carried. Moreover,
$\mathsf{mul}_{\mathrm{word}}(w,\mathsf{inv}_{\mathrm{word}}(w))$ and
$\mathsf{mul}_{\mathrm{word}}(\mathsf{inv}_{\mathrm{word}}(w),w)$ expose exactly
the displayed finite cancellation ledger obtained by pairing each component
with its singleton inverse row; the boundary does not export a host group
object or an external quotient.
\end{theorem}
\begin{proof}
Induct on the finite word. The nil case is the empty word, whose inverse is
again nil; both products are the nil append row.

In the cons case, apply
\autoref{thm:group-singleton-empty-history-laws} to the head endpoint and
\autoref{thm:group-singleton-empty-history-namecert} to keep the transported
classifier row inside the singleton certificate. The induction hypothesis gives
the reversed inverse word for the tail and its carriedness.

Concatenating the word with its reversed inverse produces the visible sequence
of componentwise singleton cancellation rows. Reassociating this finite append
spine uses the same append ledger as
\autoref{thm:group-concrete-instance-schema-unlock}; it only rearranges the
displayed continuation evidence.

The opposite product is identical after reversing the induction order. Thus the
inverse boundary is a carried $\ListUp$ word endpoint with a finite cancellation
ledger, not a transported theorem about an ambient group object.
\end{proof}
