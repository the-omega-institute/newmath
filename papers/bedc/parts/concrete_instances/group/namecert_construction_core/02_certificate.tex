\section{The certificate}

\begin{definition}[Group source specification]
\label{def:group-source-specification}
The source is a certified monoid carrier plus inverse data.
\end{definition}

\begin{definition}[Group pattern specification]
\label{def:group-pattern-specification}
The pattern is monoid expression formation extended by inverse formation.
\end{definition}

\begin{definition}[Group classifier specification]
\label{def:group-classifier-specification}
The classifier is the monoid carrier classifier, with inverse congruence.
\end{definition}

\begin{definition}[Group stability certificate]
\label{def:group-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item all monoid stability fields
  \item inverse congruence
  \item left inverse law
  \item right inverse law
  \item closure under inverse
\end{enumerate}
\end{definition}

\begin{theorem}[Group inverse of identity]
\label{thm:group-inverse-identity}
For a history-level group operation with right identity and left inverses, the
inverse of the identity is internally the same as the identity:
\(\hsame(e^{-1},e)\).
\end{theorem}
\begin{proof}
Chain \(e^{-1}\) backward through the right identity
\(e^{-1}\cdot e\hsame e^{-1}\), then apply the left inverse law at \(e\).
\end{proof}

\begin{theorem}[Left inverse involutivity]
\label{thm:group-left-inverse-involutive}

by \(x^{-1}x\), reassociate, collapse

identity.

\begin{theorem}[Left and right inverse witnesses agree]
\label{thm:group-left-right-inverse-uniqueness}
For a history-level group operation with associativity, left and right identity,
and multiplication congruence, if histories $y$ and $z$ satisfy
$$
\hsame(yx,e)
$$
and
$$
\hsame(xz,e),
$$
then $\hsame(y,z)$.
\end{theorem}
\begin{proof}
Chain $y$ through the right identity, replace the identity $e$ by $xz$,
reassociate to $(yx)z$, replace $yx$ by $e$, and finish with the left identity
on $z$. Each replacement is licensed by multiplication congruence and the
displayed inverse hypotheses.
\end{proof}

$(xy)^{-1}$ with $y^{-1}x^{-1}$.

\label{thm:group-conjugation-equation-exactness}

$$

\begin{proof}

multiplication with $a$. Reassociate $((a x)a^{-1})a$ to

$a^{-1}(a x)$ to $(a^{-1}a)x$, collapse $a^{-1}a$, and use the left identity
to obtain $x\hsame (a^{-1}b)a$.

For the reverse direction, transport $x\hsame (a^{-1}b)a$ by left
multiplication with $a$. Reassociate $a((a^{-1}b)a)$ through
$(aa^{-1})b$ and then back to $b a$, using the right inverse law for
$aa^{-1}$ and the left identity on $b$.

$(a x)a^{-1}$ as the transported left side and reassociate

$$
\mathsf{comm}(a,b)\hsame e \Longleftrightarrow

congruence; ledger conjugating factor, inverse witness, and commutator spine;
core theorem this fixed-point equivalence; exactness predicate fixed point iff
commutator trivial iff commutation; standard bridge ordinary group
centralizers; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Apply Theorem~\ref{thm:group-conjugation-equation-exactness} with middle
factor $b$ and target $b$. The fixed-point assertion
$\hsame((a\cdot b)\cdot a^{-1},b)$ is therefore equivalent to the comparison
$\hsame(b,(a^{-1}\cdot b)\cdot a)$.

For the forward comparison, multiply the right-side equation by $a$ on the
left and reassociate $a\cdot((a^{-1}\cdot b)\cdot a)$ through
$(a\cdot a^{-1})\cdot b$. The inverse law collapses $a\cdot a^{-1}$ to the
unit, and the unit laws identify the transported comparison with
$\hsame(a\cdot b,b\cdot a)$.

For the reverse comparison, multiply $a\cdot b\hsame b\cdot a$ on the left by
$a^{-1}$. Associativity moves the inverse next to $a$, the inverse and unit
laws collapse the left factor, and the resulting comparison is exactly
$\hsame(b,(a^{-1}\cdot b)\cdot a)$.

Theorem~\ref{thm:group-commutator-trivial-iff-commutes} gives the second
equivalence, so transitivity of equivalences yields the displayed
fixed-point, commutator, and commutation criterion.
\end{proof}

\begin{definition}[Group centralizer carrier]
\label{def:group-centralizer-carrier}
Fix a BEDC history-level group operation $\mathsf{mul}$ over
$\mathsf{BHist}$, unit $e$, inverse operation $\mathsf{inv}$, and fixed
element $a:\mathsf{BHist}$. The centralizer carrier of $a$ is the classifier
$Z_a$ on histories defined by
$$
Z_a(x) :\Longleftrightarrow
\hsame(\mathsf{mul}(x,a),\mathsf{mul}(a,x)).
$$
\end{definition}

\begin{theorem}[Group centralizer inverse closure]
\label{thm:group-centralizer-inverse-closure}
For a BEDC history-level group operation $\mathsf{mul}$ over $\mathsf{BHist}$
with empty history unit $e=\mathsf{BHist.Empty}$, associativity, empty left
and right unit laws, multiplication congruence, and two-sided inverse laws,
the centralizer carrier of \autoref{def:group-centralizer-carrier} is closed
under inverse:
$$
\hsame(\mathsf{mul}(x,a),\mathsf{mul}(a,x))
\Rightarrow
\hsame(\mathsf{mul}(\mathsf{inv}(x),a),
\mathsf{mul}(a,\mathsf{inv}(x))).
$$
Equivalently, $Z_a(x)$ implies $Z_a(\mathsf{inv}(x))$.
The ten-field fit is: traditional target centralizer inverse closure; BEDC
source $\GroupUp$ over $\mathsf{BHist}$; carrier $Z_a$; classifier $\hsame$;
stability obligations associativity, empty unit laws, multiplication
congruence, and two-sided inverse laws; ledger fixed element $a$ and commuting
witness for $x$; core theorem inverse closure of the centralizer carrier;
exactness predicate commutation with $a$; standard bridge ordinary group
centralizers; status $\mathsf{Proof}$.
\end{theorem}
\begin{proof}
For inverse closure, first use \autoref{thm:group-inverse-congruence-from-laws}
on the witness $xa\hsame ax$. Apply
\autoref{thm:group-inverse-mul-reverse} to both products. The inverse of
$xa$ is represented by $a^{-1}x^{-1}$, and the inverse of $ax$ is represented
by $x^{-1}a^{-1}$; transporting along the inverse-congruence witness gives
$a^{-1}x^{-1}\hsame x^{-1}a^{-1}$.

Multiplying on the left by $a$ and on the right by $a$, then using
associativity and the two inverse laws, cancels the outer $a^{-1}$ factors and
yields $x^{-1}a\hsame ax^{-1}$.
\end{proof}
\leanchecked{BEDC.Derived.GroupUp.group\_centralizer\_inv\_closed\_from\_empty\_unit}

\begin{theorem}[Group centralizer multiplication closure from empty unit]
\label{thm:group-centralizer-mul-closed-from-empty-unit}
For a BEDC history-level group operation $\mathsf{mul}$ over $\mathsf{BHist}$
with associativity, right empty-unit law, and multiplication congruence, the
centralizer carrier of \autoref{def:group-centralizer-carrier} is closed under
multiplication:
$$
\hsame(\mathsf{mul}(x,a),\mathsf{mul}(a,x))
\Rightarrow
\hsame(\mathsf{mul}(y,a),\mathsf{mul}(a,y))
\Rightarrow
\hsame(\mathsf{mul}(\mathsf{mul}(x,y),a),
\mathsf{mul}(a,\mathsf{mul}(x,y))).
$$
\end{theorem}
\begin{proof}
The empty-context multiplication closure compares the two target products
after multiplying both by $\emp$ on the right. The concrete right empty-unit
law identifies each right-extended product with its unextended endpoint, and
transitivity removes the empty context from the comparison.
\end{proof}
\leanchecked{BEDC.Derived.GroupUp.group\_centralizer\_mul\_closed\_from\_empty\_unit}

\begin{theorem}[Group centralizer subgroup and commutator fiber]
\label{thm:group-centralizer-subgroup}
For a BEDC history-level group operation over $\mathsf{BHist}$ with
associativity, left and right identity laws for $e$, multiplication
congruence, and two-sided inverse laws, the carrier $Z_a$ of
\autoref{def:group-centralizer-carrier} is closed under the group operations:
\begin{enumerate}
\item $Z_a(e)$.
\item If $Z_a(x)$ and $Z_a(y)$, then $Z_a(\mathsf{mul}(x,y))$.
\item If $Z_a(x)$, then $Z_a(\mathsf{inv}(x))$.
\item For every $x:\mathsf{BHist}$,
$$
Z_a(x) \Longleftrightarrow \mathsf{comm}(x,a)\hsame e,
$$
where $\mathsf{comm}$ is the commutator operation of
\autoref{thm:group-commutator-trivial-iff-commutes}.
\end{enumerate}
The ten-field fit is: traditional target centralizer subgroup; BEDC source
$\GroupUp$ over $\mathsf{BHist}$; carrier $Z_a$; classifier $\hsame$;
stability obligations unit, product, inverse closure, and commutator-fiber
exactness; ledger fixed element $a$ and commuting witnesses for the active
factors; core theorem list \autoref{thm:group-inverse-mul-reverse},
\autoref{thm:group-inverse-congruence-from-laws}, and
\autoref{thm:group-commutator-trivial-iff-commutes}, together with
associativity, unit laws, inverse laws, and multiplication congruence;
exactness predicate $Z_a(x)$ iff trivial commutator; standard bridge ordinary
group centralizers; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
The unit belongs to $Z_a$ because the left and right identity laws identify
$ea$ and $ae$ with the same history $a$. Symmetry and transitivity of
$\hsame$ close the comparison.
\leanchecked{BEDC.Derived.GroupUp.group\_centralizer\_empty\_unit\_mem}

For product closure, start from witnesses $xa\hsame ax$ and $ya\hsame ay$.
Associativity rewrites $(xy)a$ as $x(ya)$; multiplication congruence transports
the second witness to $x(ay)$, and associativity rewrites this as $(xa)y$.
The first witness then transports the left factor to $(ax)y$, and
associativity gives $a(xy)$.

For inverse closure, apply
\autoref{thm:group-centralizer-inverse-closure} to the witness $Z_a(x)$.

The commutator fiber is exactly the displayed carrier by
\autoref{thm:group-commutator-trivial-iff-commutes} with the ordered pair
$(x,a)$. Its right side is precisely the definition of $Z_a(x)$, and symmetry
of the displayed equivalence supplies the stated orientation.
\end{proof}
\leanchecked{BEDC.Derived.GroupUp.group\_centralizer\_mul\_closed\_empty\_context}
