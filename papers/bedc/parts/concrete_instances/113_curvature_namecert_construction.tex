\chapter{A Concrete Naming Certificate for $\CurvatureUp$}
\label{ch:concrete-instances-curvature-namecert}

The curvature interface $\CurvatureUp$ packages the antisymmetric bracket of a connection, certifying the Bianchi identities and the tensorial behaviour under section change. It depends on $\NameCert_{\ConnectionUp}$ (\autoref{ch:concrete-instances-connection-namecert}) and supplies the local invariants on which Riemannian and gauge-field certificates rest.

\section{Carrier definition}
\label{sec:curvature-carrier}

\begin{definition}[Curvature bracket carrier]
\label{def:curvature-bhist-bracket-carrier}
A $\CurvatureUp$ carrier packet is a displayed $\mathsf{BHist}$ tuple
$$
\mathcal{R}=(\mathcal{C},b,f,s,u,v,d_{uv},d_{vu},\kappa,\beta,\rho)
$$
where $\mathcal{C}$ is a source $\NameCert_{\ConnectionUp}$ packet, $b:\mathsf{BHist}$ is the base row, $f:\mathsf{BHist}$ is the fibre row over $b$, $s:\mathsf{BHist}$ is the section row, $u,v:\mathsf{BHist}$ are the two tangent-direction rows, $d_{uv},d_{vu}:\mathsf{BHist}$ are the two iterated derivative endpoint rows, $\kappa:\mathsf{BHist}$ is the displayed commutator boundary supplied by \autoref{thm:connection-curvature-boundary-obligation}, $\beta$ is the finite $\Cont$ ledger reading that boundary from the two derivative orders, and $\rho$ records the $\mathsf{Pkg}$ provenance of the connection packet.

The carrier is accepted only when the base, fibre, section, tangent, derivative endpoint, commutator-boundary, ledger, and provenance rows all project from the same connection certificate. Thus curvature is a bracket package over $\NameCert_{\ConnectionUp}$, not an ambient tensor object added beside the BEDC history rows.
\end{definition}

\begin{theorem}[Curvature boundary source obligation]
\label{thm:curvature-boundary-source-obligation}
A $\CurvatureUp$ source row is read from a $\ConnectionUp$ carrier packet, two tangent-direction $\mathsf{BHist}$ rows, the two ordered derivative-composite $\Cont$ ledgers, and the commutator-boundary $\hsame$ row exposed by \autoref{thm:connection-curvature-boundary-obligation}. The resulting source packet carries no coordinate chart and no host curvature operator; it is exactly the displayed boundary row together with its connection, tangent, derivative, ledger, and $\mathsf{Pkg}$ provenance projections.
\end{theorem}
\leanchecked{BEDC.Derived.CurvatureUp.CurvatureBracketCarrier\_boundary\_source\_obligation}
\begin{proof}
Project the source connection packet through \autoref{thm:connection-curvature-boundary-obligation}. This gives the base, fibre, section, tangent-pair, derivative-endpoint, commutator-boundary, and provenance rows already accepted by the connection certificate.

Read the two derivative orderings as the ordered $\Cont$ ledgers carried by the same connection packet. Their displayed endpoint comparison is the commutator-boundary $\hsame$ row.

Repack these projected rows as the $\CurvatureUp$ source packet. Since every component is obtained by projection from the connection boundary obligation, the curvature carrier adds no independent coordinate datum and no further quotient of the history ledger.
\end{proof}

\begin{theorem}[Curvature connection classifier stability]
\label{thm:curvature-connection-classifier-stability}
Transport of the source connection packet by $\mathsf{ConnCls}$, together with $\ManifoldUp$ transport of the tangent pair, transports the $\CurvatureUp$ boundary source row componentwise by $\hsame$. In particular, classified replacements of the connection packet and of the two tangent-direction rows preserve the two derivative-composite ledgers and the commutator boundary endpoint row read by $\CurvatureUp$.
\end{theorem}
\begin{proof}
Start with the $\mathsf{ConnCls}$ comparison between the two connection packets. It transports the base, fibre, section, derivative endpoint, and provenance components by the classifier rows already exposed by $\NameCert_{\ConnectionUp}$.

Use the $\ManifoldUp$ classifier on the two tangent-direction rows. Componentwise transport sends the ordered tangent pair to the classified ordered tangent pair, so both derivative composites retain the same source and target projections after the connection transport.

Apply endpoint transport to the two $\Cont$ ledgers and then to the displayed commutator boundary. The result is an $\hsame$ comparison of the transported boundary rows, which is exactly the classifier stability required of the $\CurvatureUp$ source packet.
\end{proof}

\label{thm:curvature-classifier-transport-row}

\end{theorem}
\begin{proof}
Unpack the two curvature carrier packets. The connection classifier supplies the comparisons for the base, fibre, section, tangent-direction, and derivative-endpoint components, while the common provenance row supplies the package comparison.

Reassociate the finite $\Cont$ ledger so that the two derivative orders are read with the same source grouping. This reassociation does not change any endpoint; it only exposes the two commutator-boundary rows as comparable $\mathsf{BHist}$ entries in the same provenance context.

Use the assumed $\hsame$ comparison between the two boundary rows. Repack the transported connection classifier rows, the boundary $\hsame$ witness, and the provenance ledger as the curvature classifier witness.
\end{proof}

\begin{theorem}[Curvature Bianchi ledger row]
\label{thm:curvature-bianchi-ledger-row}
For any cyclic triple of tangent-direction rows $u,v,w:\mathsf{BHist}$ over the same curvature carrier packet, the three commutator boundaries for $(u,v)$, $(v,w)$, and $(w,u)$ carry a visible $\mathsf{SigRel}$ ledger witnessing the Bianchi boundary. This ledger is a finite $\Cont$ consequence of the three inherited connection boundary rows and their shared $\mathsf{Pkg}$ provenance, and it does not assert coordinate equality, exterior-calculus equality, or a host-level curvature identity.
\end{theorem}
\begin{proof}
Read $u$, $v$, and $w$ through the same source connection packet and the same finite $\Cont$ spine. For each cyclic ordered pair, apply \autoref{thm:connection-curvature-boundary-obligation} to obtain the displayed commutator boundary row with the common base, fibre, section, and provenance projections.

Collect the three boundary rows into the signature ledger whose entries are the cyclic boundary readings. The $\mathsf{SigRel}$ witness records that the three readings belong to one finite certificate surface, while the $\Cont$ ledger records the append order of the boundary rows.

The Bianchi row is this ledger. It is visible because every entry is a carried $\mathsf{BHist}$ or $\mathsf{SigRel}$ row, and it is conservative because no equality outside those displayed rows is used.
\end{proof}

\begin{theorem}[Curvature antisymmetric boundary classifier]
\label{thm:curvature-antisymmetric-boundary-classifier}
For a $\NameCert_{\ConnectionUp}$ boundary row whose ordered tangent pair is $(t_1,t_2)$, swapping the displayed tangent rows to $(t_2,t_1)$ reverses the commutator boundary classifier by the inherited $\hsame$ endpoint comparisons and the $\mathsf{SigRel}$ orientation row. The reversed row has the same base, fibre, section, derivative-endpoint carrier rows, and $\mathsf{Pkg}$ provenance rows, but its classifier is the opposite boundary relation in $\mathsf{SigRel}$, not a judgmental or host equality between the two commutator histories.
\end{theorem}
\begin{proof}
Start with the two finite $\Cont$ composites that define the accepted boundary: first $t_1$ then $t_2$, and first $t_2$ then $t_1$. Swapping the tangent rows interchanges these two composites while leaving their base, fibre, section, and provenance rows unchanged.

Use symmetry of $\hsame$ on the derivative endpoint comparison to reverse the displayed endpoint relation. The reversal is recorded as a $\mathsf{SigRel}$ orientation row attached to the boundary classifier.

Repack the same source rows with the opposite $\mathsf{SigRel}$ classifier. Since no host equality of histories is asserted, the antisymmetry belongs to the BEDC classifier data exposed by $\NameCert_{\ConnectionUp}$ and consumed by $\CurvatureUp$.
\end{proof}

Transport the two derivative endpoint rows along those ledgers and compare the transported endpoints by $\hsame$. The commutator boundary row is therefore carried to a boundary row with the same ordered tangent pair and the same source provenance.

The $\mathsf{Pkg}$ rows of the dependency certificates are copied through the transport. Hence the $\CurvatureUp$ source surface sees tensorial section and fibre transport as classifier-preserving readback, not as a change of the underlying boundary object.
