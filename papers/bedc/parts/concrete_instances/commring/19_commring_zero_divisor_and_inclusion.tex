\begin{definition}[CommRing left zero-divisor data]
\label{def:commring-left-zero-divisor-data}
For a BEDC history-level $\CommRingUp$ operation tuple with multiplication
$\cdot$ and zero $0$, write
$$
\mathsf{ApartZero}_R(x):=(x\hsame0\to\bot).
$$
The strict left zero-divisor predicate is
$$
\mathsf{LeftZD}_R(x):=
\mathsf{ApartZero}_R(x)\land
\exists c : \mathsf{BHist},\
\mathsf{ApartZero}_R(c)\land x\cdot c\hsame0.
$$
\end{definition}

\leandef{BEDC.Derived.CommRingUp.CommRingLeftZeroDivisor}

\begin{theorem}[CommRing apart-zero visible pair]
\label{thm:commring-apartzero-visible-pair}
For every history $h:\mathsf{BHist}$, both visible constructors are apart from
the empty zero history:
$$
\mathsf{ApartZero}_R(\Ezero(h))\land
\mathsf{ApartZero}_R(\Eone(h)).
$$
\end{theorem}
\begin{proof}
Each conjunct is the corresponding BEDC history no-confusion law: neither an
$\Ezero$-headed history nor an $\Eone$-headed history can be history-same as
$\emp$.
\end{proof}

\label{thm:commring-singleton-mul-visible-strict-zero-divisor}
For every BEDC history $h$, both visible singleton histories

\mathsf{StrictZD}_{\mathsf{singleton}}(\Ezero(h))

both visible histories, while both ordered products reduce to $\emp$.
\end{proof}

\begin{theorem}[CommRing left zero-divisor strictness from commutativity]
\label{thm:commring-left-zero-divisor-strict-of-mul-comm}
For any BEDC history multiplication $mul$ satisfying multiplicative
commutativity under $\hsame$, every left zero-divisor carries a two-sided
annihilator witness:
$$
\mathsf{LeftZD}_R(x)
\Longrightarrow
\mathsf{ApartZero}_R(x)\land
\exists c:\mathsf{BHist},\
\mathsf{ApartZero}_R(c)\land x\cdot c\hsame0\land c\cdot x\hsame0.
$$
\end{theorem}
\begin{proof}
Unpack the left zero-divisor data. It gives the apartness of $x$ and a witness
$c$ with $\mathsf{ApartZero}_R(c)$ and $x\cdot c\hsame0$. Apply multiplicative
commutativity to obtain $c\cdot x\hsame x\cdot c$, then compose by transitivity
with the displayed left annihilation row. The same witness $c$ therefore supplies
both annihilator orientations.
\end{proof}

\begin{theorem}[CommRing left zero-divisor strictness iff under commutativity]
\label{thm:commring-left-zero-divisor-strict-iff-of-mul-comm}
For any BEDC history multiplication $mul$, written as $\cdot$, satisfying
multiplicative commutativity under $\hsame$, every $x:\mathsf{BHist}$
satisfies
$$
(\forall u v:\mathsf{BHist},\,u\cdot v\hsame v\cdot u)
\Longrightarrow
\bigl(\mathsf{LeftZD}_R(x)\Longleftrightarrow\mathsf{StrictZD}_R(x)\bigr).
$$
\end{theorem}
\begin{proof}
For the forward implication, assume $\mathsf{LeftZD}_R(x)$ and the displayed
multiplicative commutativity row. Apply
\autoref{thm:commring-left-zero-divisor-strict-of-mul-comm}. Its conclusion is
the apartness of $x$ together with an apart annihilator $c$ satisfying both
$x\cdot c\hsame0$ and $c\cdot x\hsame0$, which is exactly
$\mathsf{StrictZD}_R(x)$ by
\autoref{def:commring-strict-zero-divisor-data}.

For the reverse implication, unfold $\mathsf{StrictZD}_R(x)$. The strict
two-sided data gives $\mathsf{ApartZero}_R(x)$ and a witness
$c:\mathsf{BHist}$ with $\mathsf{ApartZero}_R(c)$, $x\cdot c\hsame0$, and
$c\cdot x\hsame0$. Dropping the right-annihilation conjunct leaves precisely
the apartness of $x$, the same witness $c$, its apartness, and
$x\cdot c\hsame0$, hence $\mathsf{LeftZD}_R(x)$.
\end{proof}
\leanchecked{BEDC.Derived.CommRingUp.CommRingLeftZeroDivisor\_strictZeroDivisor\_of\_mul\_comm\_iff}

equivalence of one-sided and two-sided annihilator data; standard bridge
ordinary commutative-ring zero divisors; status $\mathsf{ProofTarget}$.

\begin{theorem}[CommRing zero-divisor product closure]
\label{thm:commring-zero-divisor-product-closure}
For BEDC history operations $add$, $mul$, $neg$, and $zero$ satisfying the
$\RingUp$ additive abelian-group laws, multiplicative associativity, addition
and multiplication congruence, left distributivity, and the $\CommRingUp$
multiplicative commutativity field, every $a,b:\mathsf{BHist}$ satisfies
$$
\mathsf{LeftZD}_R(a)
\Longrightarrow
\bigl((a\cdot b)\hsame0\to\bot\bigr)
\Longrightarrow
\mathsf{LeftZD}_R(a\cdot b).
$$
\end{theorem}
\begin{proof}
First derive right distributivity from the supplied left distributivity by
\autoref{thm:commring-right-distrib-from-left}. With this row, the additive
group laws, and multiplication congruence, the hypotheses of
\autoref{thm:ring-zero-classifier-factor-absorption} are available.

Unpack $\mathsf{LeftZD}_R(a)$. Its right component gives a history
$c:\mathsf{BHist}$ with $\mathsf{ApartZero}_R(c)$ and $a\cdot c\hsame0$. Let
$P$ be the displayed proof that $a\cdot b$ is apart from zero.

The same $c$ annihilates $a\cdot b$. Multiplicative associativity gives
$(a\cdot b)\cdot c\hsame a\cdot(b\cdot c)$. Multiplicative commutativity gives
$b\cdot c\hsame c\cdot b$, and multiplication congruence transports this inside
left multiplication by $a$. Symmetry of associativity then identifies
$a\cdot(c\cdot b)$ with $(a\cdot c)\cdot b$.

Apply \autoref{thm:ring-zero-classifier-factor-absorption} to the zero
classifier $a\cdot c\hsame0$ and multiplier $b$. Its left product component
gives $(a\cdot c)\cdot b\hsame0$. Transitivity through the associativity and
commutativity rows yields
$$
(a\cdot b)\cdot c\hsame0.
$$

The strict left zero-divisor data for $a\cdot b$ is therefore the pair
consisting of $P$ and the same annihilator witness $c$ with its apart-zero proof
and the displayed product-zero classifier.
\end{proof}
\leanchecked{BEDC.Derived.CommRingUp.CommRingLeftZeroDivisor\_product\_closed}

\bigl((a\cdot b)\hsame0\to\bot\bigr)

\mathsf{LeftZD}_R(a\cdot b).
$$
\end{theorem}
\begin{proof}
Assume $\mathsf{ApartZero}_R(c)$, $a\cdot c\hsame0$, and
$P:(a\cdot b)\hsame0\to\bot$. We reuse the displayed $c$ as the annihilator
witness and show that it annihilates $a\cdot b$.

Multiplicative associativity gives
$(a\cdot b)\cdot c\hsame a\cdot(b\cdot c)$. Multiplicative commutativity gives
$b\cdot c\hsame c\cdot b$, and multiplication congruence transports this
comparison inside left multiplication by $a$. Symmetry of associativity then
identifies $a\cdot(c\cdot b)$ with $(a\cdot c)\cdot b$.

As in \autoref{thm:commring-zero-divisor-product-closure}, derive right
distributivity from \autoref{thm:commring-right-distrib-from-left} and use the
zero-classifier factor absorption theorem. Applied to $a\cdot c\hsame0$ and
the multiplier $b$, it yields $(a\cdot c)\cdot b\hsame0$. Transitivity through
the associativity, commutativity, and congruence chain gives
$$
(a\cdot b)\cdot c\hsame0.
$$

The strict left zero-divisor data for $a\cdot b$ is the apart-zero component
$P$ together with the same witness $c$, its apart-zero proof, and the displayed
product-zero classifier.
\end{proof}

\begin{definition}[CommRing subring-inclusion name certificate]
\label{def:commring-subring-inclusion-name-certificate}
Let $S$ and $T$ be history-level $\CommRingUp$ packages as in \autoref{def:commring-certificate-obligations}. Write their carriers and classifiers as $C_S,\sim_S$ and $C_T,\sim_T$, and write their operations as $0_S,1_S,+_S,-_S,\cdot_S$ and $0_T,1_T,+_T,-_T,\cdot_T$. For a history map $\iota:\mathsf{BHist}\to\mathsf{BHist}$, define the target image carrier
$$
\mathsf{Im}_{\iota}(y)\Longleftrightarrow \exists x:\mathsf{BHist},\ C_S(x)\land \iota(x)\sim_T y.
$$
Define the target-image classifier by
$$
y\sim_{\iota} z\Longleftrightarrow \mathsf{Im}_{\iota}(y)\land\mathsf{Im}_{\iota}(z)\land y\sim_T z.
$$
A subring-inclusion name certificate for $\iota$ over $T$ is the naming certificate of \autoref{def:typed-naming-certificate} with source $\mathsf{Im}_{\iota}$, classifier $\sim_{\iota}$, stability rows inherited from the target $\CommRingUp$ package, and ledger rows recording the carrier reflection, classifier reflection, and operation-preservation data for $0$, $1$, $+$, and $\cdot$. Its stability component includes target-carrier inclusion, classifier transport, closure under $0_T$, $1_T$, $+_T$, $-_T$, and $\cdot_T$, and the inherited additive, multiplicative, distributive, congruence, and commutativity laws under $\sim_{\iota}$.
\end{definition}

\begin{theorem}[CommRing carrier reflection gives subring inclusion]
\label{thm:commring-carrier-reflection-subring-inclusion-certificate}
Let $S$ and $T$ be history-level $\CommRingUp$ packages as in \autoref{def:commring-certificate-obligations}, with notation as in \autoref{def:commring-subring-inclusion-name-certificate}. Let $\iota:\mathsf{BHist}\to\mathsf{BHist}$ satisfy the following rows:
\begin{enumerate}
  \item Carrier reflection: $C_S(x)\Longrightarrow C_T(\iota(x))$ and $C_T(\iota(x))\Longrightarrow C_S(x)$ for every $x$.
  \item Classifier reflection on carried source endpoints:
$$
C_S(x)\land C_S(y)\Longrightarrow \bigl(x\sim_S y\Longleftrightarrow \iota(x)\sim_T\iota(y)\bigr).
$$
  \item Constant preservation: $\iota(0_S)\sim_T0_T$ and $\iota(1_S)\sim_T1_T$.
  \item Operation preservation for carried inputs:
$$
\iota(x+_S y)\sim_T\iota(x)+_T\iota(y)\qquad\text{and}\qquad \iota(x\cdot_S y)\sim_T\iota(x)\cdot_T\iota(y).
$$
\end{enumerate}
Then the image carrier $\mathsf{Im}_{\iota}$ and classifier $\sim_{\iota}$ of \autoref{def:commring-subring-inclusion-name-certificate} form a subring-inclusion name certificate over the target package $T$.
\end{theorem}
\begin{proof}
The target classifier is a carrier naming certificate by \autoref{def:lean-carrier-naming-certificate}. Hence it supplies reflexivity, symmetry, transitivity, and carrier transport for $\sim_T$. If $\mathsf{Im}_{\iota}(y)$ is witnessed by a carried source endpoint $x$ with $\iota(x)\sim_T y$, carrier preservation gives $C_T(\iota(x))$, and target carrier transport gives $C_T(y)$. If also $y\sim_T y'$, transitivity gives $\iota(x)\sim_T y'$, so the same source witness proves $\mathsf{Im}_{\iota}(y')$.
The restricted classifier $\sim_{\iota}$ is a $\NameCert$ on $\mathsf{Im}_{\iota}$. Inhabitedness is witnessed by $0_T$: the source certificate carries $0_S$, and the constant row gives $\iota(0_S)\sim_T0_T$. Reflexivity, symmetry, and transitivity are the corresponding target classifier fields, with the image endpoints retained in the first two components of the definition of $\sim_{\iota}$. Carrier transport for $\mathsf{Im}_{\iota}$ is immediate from the right image component of a $\sim_{\iota}$ witness.
The constants are carried by the image for the same reason. The zero row uses $0_S$ and $\iota(0_S)\sim_T0_T$; the unit row uses $1_S$ and $\iota(1_S)\sim_T1_T$.
For addition closure, choose image witnesses $x,u$ for $y,z$, so $C_S(x)$, $C_S(u)$, $\iota(x)\sim_T y$, and $\iota(u)\sim_T z$. Source additive closure, part of the $\RingUp$ component in \autoref{def:ring-stability-certificate}, gives $C_S(x+_S u)$. Operation preservation gives
$$
\iota(x+_S u)\sim_T\iota(x)+_T\iota(u).
$$
Target additive congruence transports the two image comparisons to
$$
\iota(x)+_T\iota(u)\sim_T y+_T z.
$$
Transitivity gives $\iota(x+_S u)\sim_T y+_T z$, so $x+_S u$ witnesses $\mathsf{Im}_{\iota}(y+_T z)$. Multiplication closure is identical, using source multiplicative closure, target multiplication congruence, and the product-preservation row.
It remains to record closure under target additive inverse. Let $x$ witness $\mathsf{Im}_{\iota}(y)$. Source inverse closure gives $C_S(-_S x)$. The source left-inverse row gives $(-_S x)+_S x\sim_S0_S$, and classifier preservation sends this to
$$
\iota((-_S x)+_S x)\sim_T\iota(0_S).
$$
Additive operation preservation and zero preservation turn this into
$$
\iota(-_S x)+_T\iota(x)\sim_T0_T.
$$
Using target additive congruence with $\iota(x)\sim_T y$ gives
$$
\iota(-_S x)+_T y\sim_T0_T.
$$
The target right-inverse row from \autoref{thm:ring-add-right-inverse} gives $y+_T(-_T y)\sim_T0_T$. With target additive associativity, left zero, the right-zero row from \autoref{thm:ring-add-right-zero}, and target addition congruence, \autoref{thm:group-left-right-inverse-uniqueness} identifies the two inverse witnesses:
$$
\iota(-_S x)\sim_T -_T y.
$$
Thus $-_S x$ witnesses $\mathsf{Im}_{\iota}(-_T y)$. The additive inverse row is derived from $+$ and $0$ preservation, so no separate preservation assumption for $-_S$ is required.
All remaining $\CommRingUp$ laws restrict from $T$. Associativity, commutativity of addition, additive inverse laws, multiplicative unit and associativity, distributivity, multiplication congruence, and multiplicative commutativity are target stability rows from \autoref{def:commring-stability-certificate}; their endpoints stay inside $\mathsf{Im}_{\iota}$ by the closure rows just proved, and their classifier comparisons are exactly the third component of $\sim_{\iota}$. The reverse carrier and classifier reflection implications are recorded as the exactness part of the inclusion ledger, showing that the target image does not identify carried source endpoints beyond $\sim_S$.
These rows supply the source, classifier, stability, and ledger components required by \autoref{def:commring-subring-inclusion-name-certificate}, hence they induce the advertised subring-inclusion name certificate over $T$.
\end{proof}
