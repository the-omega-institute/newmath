\begin{theorem}[CommRing equal-squares commuted signed product annihilator]
\label{thm:commring-equal-squares-commuted-signed-product-annihilator}
Under the hypotheses of
\autoref{thm:commring-equal-squares-commuted-signed-factor-zero}, every square
equality gives the commuted signed product as a two-sided annihilator:
$$
a\cdot a\hsame b\cdot b
\Longrightarrow
(a+(-b))\cdot(a+b)\hsame0
\land
\forall c,
((a+(-b))\cdot(a+b))\cdot c\hsame0
\land
c\cdot((a+(-b))\cdot(a+b))\hsame0 .
$$
\end{theorem}
\begin{proof}
Apply \autoref{thm:commring-equal-squares-commuted-signed-factor-zero} to get
the commuted signed-product zero witness. Right distributivity follows from
left distributivity and multiplicative commutativity, and the ring zero
absorption rows then identify multiplication by the zero product on either
side with zero. Multiplication congruence transports the zero-product witness
through the two multiplier positions.
\end{proof}
\leanchecked{BEDC.Derived.CommRingUp.commring\_equal\_squares\_commuted\_signed\_product\_annihilator}

\begin{theorem}[CommRing equal-squares bilateral signed-product annihilator package]
\label{thm:commring-equal-squares-bilateral-signed-product-annihilator-package}
Under the joint hypotheses of
\autoref{thm:commring-equal-squares-signed-product-annihilator} and
\autoref{thm:commring-equal-squares-commuted-signed-product-annihilator},
every $a,b : \mathsf{BHist}$ satisfies
$$
a\cdot a\hsame b\cdot b
\Longrightarrow
\left(
\left(
(a+b)\cdot(a+(-b))\hsame0
\land
\forall c : \mathsf{BHist},\
\left(
((a+b)\cdot(a+(-b)))\cdot c\hsame0
\land
c\cdot((a+b)\cdot(a+(-b)))\hsame0
\right)
\right)
\land
\left(
(a+(-b))\cdot(a+b)\hsame0
\land
\forall c : \mathsf{BHist},\
\left(
((a+(-b))\cdot(a+b))\cdot c\hsame0
\land
c\cdot((a+(-b))\cdot(a+b))\hsame0
\right)
\right)
\right).
$$
\end{theorem}
\begin{proof}
Let $Q$ be the square-equality hypothesis $a\cdot a\hsame b\cdot b$.

Apply \autoref{thm:commring-equal-squares-signed-product-annihilator} to $Q$.
This gives the forward signed-product zero row together with its universal
two-sided multiplier annihilator row.

Apply
\autoref{thm:commring-equal-squares-commuted-signed-product-annihilator} to
the same $Q$. This gives the commuted signed-product zero row together with
its universal two-sided multiplier annihilator row.

Introduce the outer conjunction and place the forward package in the left
component and the commuted package in the right component.
\end{proof}

\begin{theorem}[CommRing equal-squares bilateral signed-product context annihilator]
\label{thm:commring-equal-squares-bilateral-signed-product-context-annihilator}
Under the joint hypotheses of
\autoref{thm:commring-equal-squares-bilateral-signed-product-annihilator-package}
and \autoref{thm:ring-mul-zero-absorption}, every $a,b : \mathsf{BHist}$ satisfies
$$
a\cdot a\hsame b\cdot b
\Longrightarrow
\forall c,d : \mathsf{BHist},\
\left(
((c\cdot((a+b)\cdot(a+(-b))))\cdot d)\hsame0
\land
((c\cdot((a+(-b))\cdot(a+b)))\cdot d)\hsame0
\right).
$$
\end{theorem}
\begin{proof}
Let $Q$ be the square-equality hypothesis $a\cdot a\hsame b\cdot b$.

Apply
\autoref{thm:commring-equal-squares-bilateral-signed-product-annihilator-package}
to $Q$. Evaluating the two universal left-multiplier rows at $c$ gives
$c\cdot((a+b)\cdot(a+(-b)))\hsame0$ and
$c\cdot((a+(-b))\cdot(a+b))\hsame0$.

For the forward signed product, multiplication congruence transports the
first zero witness through the common right multiplier $d$, comparing
$(c\cdot((a+b)\cdot(a+(-b))))\cdot d$ with $0\cdot d$. The corresponding
zero-absorption row from \autoref{thm:ring-mul-zero-absorption} collapses
$0\cdot d$ to $0$.

For the commuted signed product, the same argument transports the second
left-multiplier zero witness through $d$ and collapses the endpoint
$0\cdot d$ by zero absorption. These two transported zero witnesses form the
displayed conjunction.
\end{proof}

\begin{theorem}[CommRing equal-squares bilateral signed-product two-sided context annihilator]
\label{thm:commring-equal-squares-bilateral-signed-product-two-sided-context-annihilator}
Under the joint hypotheses of
\autoref{thm:commring-equal-squares-bilateral-signed-product-annihilator-package}
and \autoref{thm:ring-mul-zero-absorption}, every
$a,b,c,d : \mathsf{BHist}$ satisfies the following implication. Writing
$P=(a+b)\cdot(a+(-b))$ and $Q=(a+(-b))\cdot(a+b)$,
$$
a\cdot a\hsame b\cdot b
\Longrightarrow
\left(
\begin{array}{l}
(P\cdot c)\cdot d\hsame0
\land
d\cdot(P\cdot c)\hsame0
\land
(c\cdot P)\cdot d\hsame0
\land
d\cdot(c\cdot P)\hsame0
\\
{}\land
(Q\cdot c)\cdot d\hsame0
\land
d\cdot(Q\cdot c)\hsame0
\land
(c\cdot Q)\cdot d\hsame0
\land
d\cdot(c\cdot Q)\hsame0
\end{array}
\right).
$$
\end{theorem}
\begin{proof}
Let $S$ be the square-equality witness $a\cdot a\hsame b\cdot b$. Apply
\autoref{thm:commring-equal-squares-bilateral-signed-product-annihilator-package}
to $S$. The forward package supplies the universal two-sided rows for $P$,
and the commuted package supplies the universal two-sided rows for $Q$.

Specializing those universal rows at $c$ gives four zero factors:
$P\cdot c\hsame0$, $c\cdot P\hsame0$, $Q\cdot c\hsame0$, and
$c\cdot Q\hsame0$.

Apply the two rows of \autoref{thm:ring-mul-zero-absorption} to each of these
four zero factors with multiplier $d$. The right-context row gives
$(P\cdot c)\cdot d\hsame0$, $(c\cdot P)\cdot d\hsame0$,
$(Q\cdot c)\cdot d\hsame0$, and $(c\cdot Q)\cdot d\hsame0$; the
left-context row gives $d\cdot(P\cdot c)\hsame0$, $d\cdot(c\cdot P)\hsame0$,
$d\cdot(Q\cdot c)\hsame0$, and $d\cdot(c\cdot Q)\hsame0$. These eight
witnesses are exactly the displayed context rows.
\end{proof}
