\begin{theorem}[CommRing equal-squares commuted signed product annihilator]
\label{thm:commring-equal-squares-commuted-signed-product-annihilator}
Under the hypotheses of
\autoref{thm:commring-equal-squares-commuted-signed-factor-zero}, every square
equality gives the commuted signed product as a two-sided annihilator:
$$
a\cdot a\hsame b\cdot b
\Longrightarrow
(a+(-b))\cdot(a+b)\hsame0
\land
\forall c,
((a+(-b))\cdot(a+b))\cdot c\hsame0
\land
c\cdot((a+(-b))\cdot(a+b))\hsame0 .
$$
\end{theorem}
\begin{proof}
Apply \autoref{thm:commring-equal-squares-commuted-signed-factor-zero} to get
the commuted signed-product zero witness. Right distributivity follows from
left distributivity and multiplicative commutativity, and the ring zero
absorption rows then identify multiplication by the zero product on either
side with zero. Multiplication congruence transports the zero-product witness
through the two multiplier positions.
\end{proof}

\begin{theorem}[CommRing equal-squares bilateral signed-product annihilator package]
\label{thm:commring-equal-squares-bilateral-signed-product-annihilator-package}
Under the joint hypotheses of
\autoref{thm:commring-equal-squares-signed-product-annihilator} and
\autoref{thm:commring-equal-squares-commuted-signed-product-annihilator},
every $a,b : \mathsf{BHist}$ satisfies
$$
a\cdot a\hsame b\cdot b
\Longrightarrow
\left(
\left(
(a+b)\cdot(a+(-b))\hsame0
\land
\forall c : \mathsf{BHist},\
\left(
((a+b)\cdot(a+(-b)))\cdot c\hsame0
\land
c\cdot((a+b)\cdot(a+(-b)))\hsame0
\right)
\right)
\land
\left(
(a+(-b))\cdot(a+b)\hsame0
\land
\forall c : \mathsf{BHist},\
\left(
((a+(-b))\cdot(a+b))\cdot c\hsame0
\land
c\cdot((a+(-b))\cdot(a+b))\hsame0
\right)
\right)
\right).
$$
\end{theorem}
\begin{proof}
Let $Q$ be the square-equality hypothesis $a\cdot a\hsame b\cdot b$.

Apply \autoref{thm:commring-equal-squares-signed-product-annihilator} to $Q$.
This gives the forward signed-product zero row together with its universal
two-sided multiplier annihilator row.

Apply
\autoref{thm:commring-equal-squares-commuted-signed-product-annihilator} to
the same $Q$. This gives the commuted signed-product zero row together with
its universal two-sided multiplier annihilator row.

Introduce the outer conjunction and place the forward package in the left
component and the commuted package in the right component.
\end{proof}
