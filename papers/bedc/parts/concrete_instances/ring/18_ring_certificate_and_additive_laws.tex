\section{The certificate}

\begin{definition}[Ring source specification]
\label{def:ring-source-specification}
The source is an additive \(\AbGroupUp\) and multiplicative \(\MonoidUp\) on the same carrier and classifier.
\end{definition}

\begin{definition}[Ring pattern specification]
\label{def:ring-pattern-specification}
The pattern is expression formation from \(0,1,+,-,\cdot\).
\end{definition}

\begin{definition}[Ring classifier specification]
\label{def:ring-classifier-specification}
The classifier is the shared carrier sameness relation, congruent for both operations.
\end{definition}

\begin{definition}[Ring stability certificate]
\label{def:ring-stability-certificate}
The stability certificate consists of:
\begin{enumerate}
  \item additive abelian-group laws
  \item multiplicative monoid laws
  \item addition and multiplication congruence
  \item left distributivity
  \item right distributivity
  \item zero multiplication laws derived or supplied constructively
\end{enumerate}
\end{definition}
\leanchecked{BEDC.Derived.RingUp.ring\_stability\_certificate\_fields}

\begin{theorem}[Ring additive right inverse]
\label{thm:ring-add-right-inverse}
For BEDC history addition, additive commutativity and the supplied left inverse
law imply the matching right inverse law \(x+(-x)\hsame0\).
\end{theorem}
\begin{proof}
Commute \(x+(-x)\) to \((-x)+x\), then apply the left inverse law.
\end{proof}
\leanchecked{BEDC.Derived.RingUp.ring\_add\_right\_inverse}

\begin{theorem}[Ring additive inverse of zero]
\label{thm:ring-neg-zero}
For BEDC history addition, additive commutativity, the left-zero law, and the
left inverse law imply \(\hsame(-0,0)\).
\end{theorem}
\begin{proof}
Chain \(-0\) backward through the left-zero equation \(0+(-0)\hsame -0\),
commute \(0+(-0)\) to \((-0)+0\), and apply the left inverse law at \(0\).
\end{proof}
\leanchecked{BEDC.Derived.RingUp.ring\_neg\_zero}

\begin{theorem}[Ring duplicate addition collapses to zero]
\label{thm:ring-add-duplicate-eq-zero}
For BEDC history addition, if an element is the same as its duplicate sum
\(a+a\), then the element is the same as zero.
\end{theorem}
\begin{proof}
Add the left inverse of \(a\) on the left, reassociate to bring
\((-a)+a\) together, collapse that pair to zero, and use the left-zero law.
\end{proof}
\leanchecked{BEDC.Derived.RingUp.ring\_add\_duplicate\_eq\_zero}

\begin{theorem}[Ring additive duplicate zero exactness]
\label{thm:ring-add-duplicate-zero-iff}
For BEDC history addition satisfying additive associativity, left zero, left
inverse, and addition congruence, every history $a$ satisfies
$$
a\hsame a+a \Longleftrightarrow a\hsame0.
$$
\end{theorem}
\begin{proof}
The forward implication is exactly the duplicate-addition collapse: apply
\autoref{thm:ring-add-duplicate-eq-zero} to the hypothesis $a\hsame a+a$.

For the reverse implication, use symmetry of $a\hsame0$ to transport the left
summand from $a$ to $0$, and use the hypothesis itself to transport the right
summand from $a$ to $0$ by addition congruence. The resulting comparison
$a+a\hsame0+0$ reduces by the left-zero law to $0$, and transitivity with the
original hypothesis gives $a\hsame a+a$.
\end{proof}
\leanchecked{BEDC.Derived.RingUp.ring\_add\_duplicate\_empty\_iff}

\begin{theorem}[Ring additive right zero]
\label{thm:ring-add-right-zero}
For BEDC history addition, additive commutativity and the supplied left-zero
law imply the matching right-zero law $x+0\hsame x$.
\end{theorem}
\begin{proof}
Commute $x+0$ to $0+x$, then apply the left-zero law.
\end{proof}
\leanchecked{BEDC.Derived.RingUp.ring\_add\_right\_zero}

\begin{theorem}[Ring additive two-sided cancellation]
\label{thm:ring-additive-two-sided-cancellation}
For BEDC history addition satisfying additive associativity, commutativity,
left-zero, left-inverse, and addition congruence, every
$a,b,c:\mathsf{BHist}$ satisfies both cancellation implications:
$$
a+c\hsame b+c \Longrightarrow a\hsame b
$$
and
$$
c+a\hsame c+b \Longrightarrow a\hsame b.
$$
The ten-field fit is: traditional target additive cancellation in a ring; BEDC
source $\RingUp$ over $\mathsf{BHist}$ plus $\GroupUp$ cancellation; carrier
$\mathsf{BHist}$; classifier $\hsame$; stability from additive group laws and
addition congruence; ledger the common addend; core theorem this theorem;
exactness predicate the two displayed implications; standard bridge ordinary
ring additive cancellation; status $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
First derive the additive right-zero law from
Theorem~\ref{thm:ring-add-right-zero} and the additive right-inverse law from
Theorem~\ref{thm:ring-add-right-inverse}. Together with associativity,
commutativity, left zero, left inverse, and addition congruence, these are the
group-cancellation hypotheses for the additive operation.

For the first implication, read addition as the group operation and apply
Theorem~\ref{thm:group-right-cancellation} to the common right summand $c$.
The hypothesis $a+c\hsame b+c$ then yields $a\hsame b$.

For the second implication, apply
Theorem~\ref{thm:group-left-cancellation} to the common left summand $c$.
The hypothesis $c+a\hsame c+b$ then yields the same conclusion $a\hsame b$.
\end{proof}
\leanchecked{BEDC.Derived.RingUp.ring\_additive\_two\_sided\_cancellation}

\begin{theorem}[Ring negative of sum]
\label{thm:ring-neg-add}
For BEDC history addition and negation satisfying additive associativity,
commutativity, left zero, left inverse, and addition congruence, every
$a,b:\mathsf{BHist}$ satisfies
$$
\hsame(\mathsf{neg}(\mathsf{add}(a,b)),
\mathsf{add}(\mathsf{neg}(a),\mathsf{neg}(b))).
$$
The ten-field fit is: traditional target additive inverse of a sum in a ring;
BEDC source $\RingUp$ over $\mathsf{BHist}$; carrier $\mathsf{BHist}$;
classifier $\hsame$; stability from additive congruence and additive
cancellation; ledger the two addends $a,b$ and the inverse witnesses
$\mathsf{neg}(\mathsf{add}(a,b))$ and
$\mathsf{add}(\mathsf{neg}(a),\mathsf{neg}(b))$; core theorem this theorem;
exactness the displayed equality; downstream consumer CommRing signed-product
normalization and FieldUp denominator inverse normalization; status
$\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
The common element to cancel is $\mathsf{add}(a,b)$. By
Theorem~\ref{thm:ring-add-right-inverse}, the endpoint
$\mathsf{neg}(\mathsf{add}(a,b))$ is a right inverse of this common element.

It remains to see that
$\mathsf{add}(\mathsf{neg}(a),\mathsf{neg}(b))$ is also a right inverse of
$\mathsf{add}(a,b)$. Reassociate the sum
$(a+b)+((-a)+(-b))$, commute the middle terms so that $b+(-b)$ and
$a+(-a)$ appear as adjacent pairs, collapse those pairs using the right
inverse and left inverse rows, and remove the resulting zeros by the left and
right zero laws.

Both endpoints therefore give the same sum with the common left factor
$\mathsf{add}(a,b)$. Apply
Theorem~\ref{thm:ring-additive-two-sided-cancellation} to cancel that factor
and obtain the displayed $\hsame$ comparison.
\end{proof}
\leanchecked{BEDC.Derived.RingUp.ring\_neg\_add}

\begin{definition}[Opposite ring certificate data]
\label{def:opposite-ring-certificate-data}
Let $C$ carry a $\RingUp(C)$ certificate with classifier $\sim_C$, addition
$+_C$, zero $0_C$, additive inverse operation $-_C$, multiplication $\cdot_C$,
and multiplicative identity $1_C$. Define the reversed multiplication by
$$
a\cdot_C^{\mathrm{op}} b:=b\cdot_C a.
$$
The opposite ring data $C^{\mathrm{op}}$ have the same carrier $C$, the same
classifier $\sim_C$, the same addition $+_C$, the same zero $0_C$, the same
additive inverse operation $-_C$, the same multiplicative identity $1_C$, and
multiplication $\cdot_C^{\mathrm{op}}$.
\end{definition}

\begin{theorem}[Opposite ring certificate]
\label{thm:opposite-ring-certificate}
If the displayed data on $C$ satisfy the $\RingUp(C)$ certificate obligations,
then the data of \autoref{def:opposite-ring-certificate-data} also satisfy the
$\RingUp(C)$ certificate obligations. Thus the same carrier, additive
structure, multiplicative identity, and classifier carry a ring certificate with
multiplication $\cdot_C^{\mathrm{op}}$.
\end{theorem}
\begin{proof}
Unpack the $\RingUp(C)$ obligations through
\autoref{def:ring-certificate-obligations}. The carrier row is unchanged by
\autoref{def:ring-carrier}. The carrier-and-classifier core is the same
$\NameCert$ data, so classifier reflexivity, symmetry, transitivity, and
carrier transport are copied from \autoref{def:lean-carrier-naming-certificate}.

The additive source required by \autoref{def:ring-source-specification} is
unchanged: the $\AbGroupUp$ carrier uses the same $+_C$, $0_C$, $-_C$, and
$\sim_C$, with the same additive stability and ledger rows. It remains to read
the multiplicative source as a $\MonoidUp$ certificate for
$\cdot_C^{\mathrm{op}}$. Closure follows from the original multiplication
closure applied to the ordered pair $(b,a)$. The opposite left identity row is
$$
1_C\cdot_C^{\mathrm{op}}a=a\cdot_C1_C\sim_C a,
$$
which is the original right identity row. The opposite right identity row is
$$
a\cdot_C^{\mathrm{op}}1_C=1_C\cdot_Ca\sim_C a,
$$
which is the original left identity row. For associativity, the two opposite
bracketings unfold to
$$
(a\cdot_C^{\mathrm{op}}b)\cdot_C^{\mathrm{op}}c
=c\cdot_C(b\cdot_Ca),
\qquad
 a\cdot_C^{\mathrm{op}}(b\cdot_C^{\mathrm{op}}c)
=(c\cdot_Cb)\cdot_Ca.
$$
The associativity-under-classifier row in
\autoref{def:monoid-stability-certificate}, applied to the original
multiplication at the triple $(c,b,a)$, supplies
$$
(c\cdot_Cb)\cdot_Ca\sim_C c\cdot_C(b\cdot_Ca).
$$
Classifier symmetry gives the orientation
$c\cdot_C(b\cdot_Ca)\sim_C(c\cdot_Cb)\cdot_Ca$ needed to compare the two
unfolded opposite bracketings. Multiplication congruence for
$\cdot_C^{\mathrm{op}}$ is the original multiplication congruence with the two
input comparisons supplied in exchanged order. These are precisely the monoid
stability rows of \autoref{def:monoid-stability-certificate}, hence the
multiplicative source satisfies \autoref{def:monoid-certificate-obligations}.

The pattern row of \autoref{def:ring-pattern-specification} is unchanged except
that the multiplication node is interpreted by $\cdot_C^{\mathrm{op}}$. The
classifier row of \autoref{def:ring-classifier-specification} is unchanged for
addition, and the preceding paragraph supplies congruence for the reversed
multiplication.

For left distributivity in the opposite data, unfold the reversed product and
use original right distributivity:
$$
a\cdot_C^{\mathrm{op}}(b+_C c)
=(b+_C c)\cdot_C a
\sim_C
(b\cdot_C a)+_C(c\cdot_C a)
=(a\cdot_C^{\mathrm{op}}b)+_C(a\cdot_C^{\mathrm{op}}c).
$$
For right distributivity in the opposite data, unfold the reversed product and
use original left distributivity:
$$
(a+_C b)\cdot_C^{\mathrm{op}}c
=c\cdot_C(a+_C b)
\sim_C
(c\cdot_C a)+_C(c\cdot_C b)
=(a\cdot_C^{\mathrm{op}}c)+_C(b\cdot_C^{\mathrm{op}}c).
$$
The two zero multiplication rows are exchanged in the same way:
$0_C\cdot_C^{\mathrm{op}}a=a\cdot_C0_C$ is the original right-zero row, and
$a\cdot_C^{\mathrm{op}}0_C=0_C\cdot_Ca$ is the original left-zero row.
Together with the copied additive laws and congruence rows, these are exactly
the stability fields listed in \autoref{def:ring-stability-certificate}.

Finally, the ledger policy of \autoref{def:ring-ledger-policy} uses the same
carrier witnesses and the same additive operation trees. A multiplication node
with opposite inputs $(a,b)$ records the original multiplication ledger for
$(b,a)$. Therefore the carrier, source, pattern, classifier, stability, and
ledger rows all satisfy the $\RingUp(C)$ certificate obligations for the
opposite ring data.
\end{proof}

\begin{theorem}[Ring negation classifier exactness]
\label{thm:ring-negation-classifier-exactness}
For BEDC history addition and negation satisfying additive associativity,
commutativity, left zero, left inverse, addition congruence, and additive
inverse congruence, every $x,y:\mathsf{BHist}$ satisfies
$$
-x\hsame -y \Longleftrightarrow x\hsame y.
$$
The ten-field fit is: traditional target additive inverse injectivity in a
ring; BEDC source $\RingUp$ over $\mathsf{BHist}$ with the additive
$\GroupUp$ cancellation spine cited by
\autoref{thm:ring-additive-difference-zero-exact}; carrier $\mathsf{BHist}$;
classifier $\hsame$; stability from addition congruence and additive inverse
congruence; ledger the two negated endpoints; core theorem this theorem;
exactness the displayed iff; standard bridge ordinary ring additive inverse
injectivity; status $\mathsf{ProofTarget}$.
\end{theorem}
\leanchecked{BEDC.Derived.RingUp.ring\_negation\_classifier\_exactness}
\begin{proof}
Assume first that $-x\hsame -y$. By classifier symmetry, $-y\hsame -x$.
Addition congruence, with reflexivity on the left summand $x$, transports this
to
$$
x+(-y)\hsame x+(-x).
$$
The right-inverse law of \autoref{thm:ring-add-right-inverse} gives
$x+(-x)\hsame0$, so transitivity gives $x+(-y)\hsame0$. The forward direction
of \autoref{thm:ring-additive-difference-zero-exact} then yields
$x\hsame y$.

Conversely, assume $x\hsame y$. The additive inverse congruence field
transports this classifier witness directly across negation, giving
$-x\hsame -y$.
\end{proof}

\begin{theorem}[Double-opposite ring certificate data identity]
\label{thm:double-opposite-ring-certificate-data-identity}
Let $R$ carry a $\RingUp(R)$ certificate satisfying the obligations of
\autoref{def:ring-certificate-obligations}. Present its data by a carrier
predicate $\mathcal C$, classifier $\sim_R$, addition $+_R$, zero $0_R$,
additive inverse operation $-_R$, multiplication $\cdot_R$, multiplicative
identity $1_R$, stability table $\mathcal S_R$ as in
\autoref{def:ring-stability-certificate}, and ledger table $\mathcal L_R$ as in
\autoref{def:ring-ledger-policy}. Let
$R^{\mathrm{opop}}=(R^{\mathrm{op}})^{\mathrm{op}}$ be obtained by applying
\autoref{def:opposite-ring-certificate-data} twice. Then
$R^{\mathrm{opop}}$ is a $\RingUp(R)$ certificate, and its carrier,
classifier, additive data, multiplicative identity, multiplication, stability
rows, and ledger rows are componentwise identical to those of $R$: for carried
endpoints $a,b$,
$$
\begin{aligned}
\mathcal C^{\mathrm{opop}}(a)&=\mathcal C(a),&
a\sim_R^{\mathrm{opop}} b&=a\sim_R b,\\
a+_R^{\mathrm{opop}} b&=a+_R b,&
0_R^{\mathrm{opop}}&=0_R,\\
(-_R)^{\mathrm{opop}}(a)&=-_R a,&
1_R^{\mathrm{opop}}&=1_R,\\
a\cdot_R^{\mathrm{opop}} b&=a\cdot_R b,&
\mathcal S_R^{\mathrm{opop}}&=\mathcal S_R,\\
\mathcal L_R^{\mathrm{opop}}&=\mathcal L_R.
\end{aligned}
$$
The equalities involving $\mathcal S_R$ and $\mathcal L_R$ mean equality of the
named rows and their displayed endpoint data.
\end{theorem}
\begin{proof}
Because $R$ satisfies the $\RingUp(R)$ obligations,
\autoref{thm:opposite-ring-certificate} makes $R^{\mathrm{op}}$ a certified
$\RingUp(R)$ instance. Applying the same theorem to that certified opposite
data makes $R^{\mathrm{opop}}$ a certified $\RingUp(R)$ instance.

It remains to compare the displayed fields. By
\autoref{def:opposite-ring-certificate-data}, one opposite keeps the carrier,
classifier, addition, zero endpoint, additive inverse operation, and
multiplicative identity fixed. Two unfoldings therefore give
$$
\mathcal C^{\mathrm{opop}}(a)=\mathcal C(a),
\qquad
 a\sim_R^{\mathrm{opop}} b=a\sim_R b,
$$
$$
a+_R^{\mathrm{opop}} b=a+_R b,
\qquad
0_R^{\mathrm{opop}}=0_R,
\qquad
(-_R)^{\mathrm{opop}}(a)=-_R a,
\qquad
1_R^{\mathrm{opop}}=1_R.
$$
For multiplication, the first opposite reverses the two inputs and the second
opposite reverses them back:
$$
a\cdot_R^{\mathrm{opop}} b
=b\cdot_R^{\mathrm{op}} a
=a\cdot_R b.
$$
Thus the raw ring data of the double opposite certificate are the raw ring data
of $R$.

The source row of \autoref{def:ring-source-specification} is restored from
these identities: the additive $\AbGroupUp$ source is fixed in each opposite,
and the multiplicative $\MonoidUp$ source is typed over the restored
multiplication and the same identity. The pattern row of
\autoref{def:ring-pattern-specification} forms expressions from the same
$0_R$, $1_R$, $+_R$, $-_R$, and restored multiplication. The classifier row of
\autoref{def:ring-classifier-specification} is the same relation $\sim_R$, with
addition congruence fixed and multiplication congruence read over the restored
multiplication.

For the stability table, \autoref{def:ring-stability-certificate} lists the
additive abelian-group laws, multiplicative monoid laws, operation congruence,
left distributivity, right distributivity, and zero multiplication laws. The
additive rows and addition congruence are fixed by each opposite construction.
In the multiplicative part, one opposite exchanges the two unit rows, reads
associativity with the triple reversed, and exchanges the two congruence inputs;
the second opposite performs the same exchanges again, restoring the unit,
associativity, and multiplication-congruence rows for $\cdot_R$. The left and
right distributivity rows are exchanged once and then restored, and the two
zero multiplication rows are exchanged in the same way. Hence
$\mathcal S_R^{\mathrm{opop}}=\mathcal S_R$ row by row.

Finally, \autoref{def:ring-ledger-policy} records the shared carrier ledger and
the operation tree for ring expressions. Carrier, additive, zero, inverse, and
unit entries are fixed by each opposite. A multiplication node records the same
two operands with their child order exchanged in one opposite reading; the
second opposite exchanges that child order again. Therefore every ledger entry
of $R^{\mathrm{opop}}$ is the corresponding ledger entry of $R$, so
$\mathcal L_R^{\mathrm{opop}}=\mathcal L_R$. Together with the certified status
above, these identities identify the double-opposite ring certificate data with
the original ring certificate data.
\end{proof}
