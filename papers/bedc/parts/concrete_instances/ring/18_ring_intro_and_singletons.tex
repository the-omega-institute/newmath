\chapter{A Concrete Naming Certificate for \(\RingUp\)}
\label{ch:concrete-instances-ring-namecert}

The ring interface \(\RingUp\) is licensed by a carrier with two binary operations \(+\) and \(\cdot\) such that \((C, +)\) satisfies \autoref{ch:concrete-instances-abgroup-namecert} and \((C, \cdot)\) satisfies \autoref{ch:concrete-instances-monoid-namecert}, together with left and right distributivity of \(\cdot\) over \(+\). It is not a primitive structure; it is licensed only by composing the additive abelian group and multiplicative monoid certificates with the distributive law.

\begin{remark}[Concrete stability fields]
\label{rem:concrete-instances-ring-namecert-roadmap}
This chapter records the shared additive, multiplicative, congruence, and distributivity fields for the \(\RingUp\) naming certificate over BEDC histories.
\end{remark}

\section{Carrier definition}

\begin{definition}[Ring carrier]
\label{def:ring-carrier}
Use a setup with carrier \(C\), classifier, addition, zero, additive inverse, multiplication, one, and ring law fields.
\end{definition}

\section{Concrete singleton history ring instance}
\label{sec:ring-concrete-singleton-history-instance}

\begin{definition}[Concrete singleton history ring instance]
\label{def:concrete-singleton-history-ring-instance}
Let $A_{\emp}$ and $R_{\emp}$ be the singleton carrier and classifier from
Definition~\ref{def:concrete-singleton-history-group-instance}. Define the
operations by
$$
\mathsf{add}(h,k):=\mathsf{append}(h,k),\qquad
\mathsf{neg}(h):=\emp,\qquad
0:=\emp,
$$
and
$$
\mathsf{mul}(h,k):=\mathsf{append}(h,k),\qquad
1:=\emp.
$$
The resulting tuple
$(A_{\emp},R_{\emp},\mathsf{add},0,\mathsf{neg},\mathsf{mul},1)$ is the
concrete singleton history ring instance.
\end{definition}

\begin{theorem}[Concrete singleton history ring laws]
\label{thm:concrete-singleton-history-ring-laws}
The tuple
$(A_{\emp},R_{\emp},\mathsf{add},0,\mathsf{neg},\mathsf{mul},1)$ satisfies
the additive abelian-group laws, multiplicative monoid laws, addition and
multiplication congruence, left distributivity, and right distributivity under
$R_{\emp}$.
\end{theorem}
\begin{proof}
The additive component is the singleton history abelian group of
Theorem~\ref{thm:concrete-singleton-history-abgroup-laws}, read with its
binary operation as $\mathsf{add}$. Thus additive closure, zero, inverse,
associativity, commutativity, and additive congruence are already supplied over
$A_{\emp}$ and $R_{\emp}$.

The multiplicative component is the monoid part contained in
Theorem~\ref{thm:concrete-singleton-history-group-laws}, read with
$\mathsf{mul}(h,k):=\mathsf{append}(h,k)$ and $1:=\emp$. This gives
multiplication closure, left and right unit laws, associativity, and
multiplication congruence under the same classifier.

For left distributivity, assume $A_{\emp}(h)$, $A_{\emp}(k)$, and
$A_{\emp}(l)$. Closure for $\mathsf{add}$ and $\mathsf{mul}$ puts both
$\mathsf{mul}(h,\mathsf{add}(k,l))$ and
$\mathsf{add}(\mathsf{mul}(h,k),\mathsf{mul}(h,l))$ back in $A_{\emp}$.
The singleton witnesses identify each side with $\emp$, and transitivity of
$\hsame$ composes the two endpoint witnesses into the required
$R_{\emp}$ relation.

For right distributivity, the same argument applies to
$\mathsf{mul}(\mathsf{add}(h,k),l)$ and
$\mathsf{add}(\mathsf{mul}(h,l),\mathsf{mul}(k,l))$. Closure gives the two
carrier witnesses, and the singleton endpoint $\emp$ supplies the composed
$\hsame$ witness required by $R_{\emp}$.
\end{proof}
\leanchecked{BEDC.Derived.RingUp.concrete\_singleton\_history\_ring\_laws}

The RingUp schema is instantiated by this concrete $\mathsf{BHist}$ carrier and
operations, so later formalization can target BHist-anchored ring laws rather
than abstract operator fields.

\section{Concrete singleton empty-history ring instance}
\label{sec:ring-concrete-singleton-instance}

\begin{definition}[Singleton empty-history ring instance]
\label{def:singleton-empty-history-ring-instance}
Let $A_{\emp}$ be the singleton history source of
Definition~\ref{def:list-singleton-history-source} at $a_0=\emp$. Define the
classifier
$$
\rho_{\emp}(h,k)\Longleftrightarrow
A_{\emp}(h)\land A_{\emp}(k)\land\hsame(h,k).
$$
Define the concrete ring operations by
$$
\mathsf{add}(h,k):=\emp,\qquad
\mathsf{neg}(h):=\emp,\qquad
0:=\emp,\qquad
\mathsf{mul}(h,k):=\emp,\qquad
1:=\emp.
$$
This is the one-element BEDC history ring over the concrete carrier
$A_{\emp}:\mathsf{BHist}\to\mathsf{Prop}$ and classifier $\rho_{\emp}$.
\end{definition}

\begin{theorem}[Singleton empty-history ring laws]
\label{thm:singleton-empty-history-ring-laws}
The tuple
$(A_{\emp},\rho_{\emp},\mathsf{add},0,\mathsf{neg},\mathsf{mul},1)$ satisfies
carrier inhabitance, additive abelian-group laws, multiplicative monoid laws,
addition and multiplication congruence, left and right distributivity, and zero
multiplication laws under $\rho_{\emp}$.
\end{theorem}
\begin{proof}
Carrier inhabitance is the reflexive endpoint witness $\hsame(\emp,\emp)$.
By Definition~\ref{def:list-singleton-history-source} at $a_0=\emp$, this is
exactly the witness that $\emp$ lies in $A_{\emp}$.

Every operation result in the displayed tuple is definitionally $\emp$. Hence
addition closure, additive inverse closure, multiplication closure, and the
closedness of $0$ and $1$ all use the same singleton carrier witness
$\hsame(\emp,\emp)$.

The additive unit, inverse, associativity, and commutativity laws reduce both
sides to $\emp$. The required endpoint facts for $\rho_{\emp}$ are the
singleton carrier witnesses; the relation component is reflexivity of
$\hsame$, with symmetry and transitivity used only to package classifier
endpoints already identified with $\emp$.

The multiplicative left and right unit laws, associativity, and operation
congruence are handled in the same way. Both source and target products are
definitionally $\emp$, so the classifier relation is $\hsame(\emp,\emp)$.

For left and right distributivity, both the product over a sum and the sum of
products reduce to $\emp$. The zero multiplication laws also reduce both sides
to $\emp$, so each law is supplied by reflexivity of $\hsame$ together with the
two singleton endpoint witnesses.
\end{proof}
\leanchecked{BEDC.Derived.RingUp.RingSingletonEmptyHistory\_laws}

\begin{theorem}[Ring singleton negation empty endpoint]
\label{thm:ring-singleton-neg-empty-endpoint}
For histories $h,t:\mathsf{BHist}$ with $A_{\emp}(h)$,
$$
\rho_{\emp}(\mathsf{RingSingletonNeg}(h),t)
\Longleftrightarrow
A_{\emp}(t)\land
\hsame(\mathsf{RingSingletonNeg}(h),\emp).
$$
\end{theorem}
\begin{proof}
The singleton ring negation endpoint is definitionally $\emp$. In the forward
direction, unfold $\rho_{\emp}$ and read off the right carrier witness for
$t$ together with the left endpoint witness for
$\mathsf{RingSingletonNeg}(h)$.

Conversely, the displayed pair supplies the two carrier endpoints. Symmetry of
$\hsame$ turns the right carrier endpoint into an endpoint comparison back
from $\emp$ to $t$, and transitivity with the left endpoint witness packages
the classifier relation required by $\rho_{\emp}$.
\end{proof}
\leanchecked{BEDC.Derived.RingUp.RingSingletonClassifier\_neg\_empty\_endpoint\_iff}

\begin{theorem}[Ring singleton classifier multiplication empty endpoint]
\label{thm:ring-singleton-classifier-mul-empty-endpoint}
For histories $h,k,out:\mathsf{BHist}$,
$$
\rho_{\emp}(\mathsf{RingSingletonMul}(h,k),out)\Longleftrightarrow
\hsame(out,\emp).
$$
\end{theorem}
\begin{proof}
The multiplication endpoint is definitionally $\emp$. Forward projection reads the
right carrier field of $\rho_{\emp}$. Conversely, the displayed endpoint witness,
the reflexive left carrier at $\emp$, and symmetry rebuild the classifier.
\end{proof}
\leanchecked{BEDC.Derived.RingUp.RingSingletonClassifier\_mul\_empty\_endpoint\_iff}

\begin{theorem}[Ring singleton classifier visible-right exclusion]
\label{thm:ring-singleton-classifier-append-visible-right-absurd}
For histories $h,p,q:\mathsf{BHist}$, neither
$\rho_{\emp}(h,\mathsf{append}(p,\mathsf{e0}(q)))$ nor
$\rho_{\emp}(h,\mathsf{append}(p,\mathsf{e1}(q)))$ can hold.
\end{theorem}
\begin{proof}
Unfold the singleton ring classifier. In either case, the right carrier field
identifies the displayed append endpoint with $\emp$. Empty-result inversion
for append then forces the visible right component, either $\mathsf{e0}(q)$ or
$\mathsf{e1}(q)$, to be $\emp$. The closed constructors of $\mathsf{BHist}$ are
disjoint, so both classifier assumptions are contradictory.
\end{proof}
\leanchecked{BEDC.Derived.RingUp.RingSingletonClassifier\_append\_visible\_right\_absurd}

\begin{theorem}[Ring singleton classifier visible-left exclusion]
\label{thm:ring-singleton-classifier-append-visible-left-absurd}
For histories $p,q,h:\mathsf{BHist}$, neither
$\rho_{\emp}(\mathsf{append}(p,\mathsf{e0}(q)),h)$ nor
$\rho_{\emp}(\mathsf{append}(p,\mathsf{e1}(q)),h)$ can hold.
\end{theorem}
\begin{proof}
Unfold the singleton ring classifier. In either case, the left carrier field
identifies the displayed append endpoint with $\emp$. Empty-result inversion
for append then forces the visible right component, either $\mathsf{e0}(q)$ or
$\mathsf{e1}(q)$, to be $\emp$. Constructor separation for
$\mathsf{BHist}$ rules out both alternatives.
\end{proof}
\leanchecked{BEDC.Derived.RingUp.RingSingletonClassifier\_append\_visible\_left\_absurd}

\begin{theorem}[Ring singleton classifier append split empty]
\label{thm:ring-singleton-classifier-append-split-empty}
For histories $p,q,h:\mathsf{BHist}$,
$$
\rho_{\emp}(\mathsf{append}(p,q),h)
\Longleftrightarrow
A_{\emp}(p)\land A_{\emp}(q)\land A_{\emp}(h).
$$
This certificate has the following ten-field fit: the traditional target is
singleton ring endpoint exactness; the BEDC source is the concrete singleton
$\RingUp$ over $\mathsf{BHist}$; the carrier is $A_{\emp}$; the classifier is
$\rho_{\emp}$; stability comes from empty-result inversion and $\hsame$
transport; the ledger records append-factor endpoints; the core theorem is the
displayed split; exactness is the displayed equivalence; the standard bridge is
terminal ring endpoint equality; and the status is $\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
Unfold the singleton ring classifier. For the forward direction, the left
carrier field identifies $\mathsf{append}(p,q)$ with $\emp$. Empty-result
inversion for append splits this endpoint into the two factor witnesses
$A_{\emp}(p)$ and $A_{\emp}(q)$, while the right carrier field gives
$A_{\emp}(h)$.

For the reverse direction, the two factor witnesses rebuild the empty appended
endpoint. Together with the witness $A_{\emp}(h)$, transitivity through $\emp$
supplies the required classifier comparison between
$\mathsf{append}(p,q)$ and $h$.
\end{proof}
\leanchecked{BEDC.Derived.RingUp.RingSingletonClassifier\_append\_split\_empty}

\begin{theorem}[Ring singleton classifier append commutation]
\label{thm:ring-singleton-classifier-append-comm}
For singleton ring carrier histories $h,k:\mathsf{BHist}$, the classifier
relates the two appended endpoints with the factors exchanged:
$$
\rho_{\emp}(\mathsf{append}(h,k),\mathsf{append}(k,h)).
$$
\end{theorem}
\begin{proof}
The two carrier witnesses identify both $h$ and $k$ with $\emp$.
Empty-result construction for append gives singleton carrier witnesses for
$\mathsf{append}(h,k)$ and $\mathsf{append}(k,h)$. Since both endpoints are
classified through $\emp$, transitivity and symmetry of $\hsame$ package the
classifier relation.
\end{proof}
\leanchecked{BEDC.Derived.RingUp.RingSingletonClassifier\_append\_comm}

\begin{theorem}[Ring singleton classifier append associativity carrier exactness]
\label{thm:ring-singleton-classifier-append-assoc-carrier-iff}
For histories $a,b,c:\mathsf{BHist}$, the singleton ring classifier relates
the two append bracketings exactly when all three factors are singleton
carriers:
$$
\rho_{\emp}(\mathsf{append}(\mathsf{append}(a,b),c),
\mathsf{append}(a,\mathsf{append}(b,c)))
\Longleftrightarrow
A_{\emp}(a)\land A_{\emp}(b)\land A_{\emp}(c).
$$
\end{theorem}
\begin{proof}
The forward direction projects the left carrier field from the classifier.
Nested empty-result inversion for append first splits
$\mathsf{append}(\mathsf{append}(a,b),c)$ into the left bracketing and $c$,
then splits the left bracketing into $a$ and $b$.

Conversely, the three carrier witnesses rebuild singleton carrier witnesses
for both append bracketings. Since both endpoints are classified through
$\emp$, transitivity and symmetry of $\hsame$ give the classifier relation.
\end{proof}
\leanchecked{BEDC.Derived.RingUp.RingSingletonClassifier\_append\_assoc\_carrier\_iff}

\begin{theorem}[Ring singleton continuation empty result exactness]\label{thm:ring-singleton-classifier-cont-result-empty}
If $C_{\emp}(h,k)$ and $\Cont(h,p,r)$, then $r\hsame\emp$ holds exactly when $p$ is in the singleton carrier.
\end{theorem}
\begin{proof}
Transporting an empty result into the continuation and inverting empty concatenation gives $p\hsame\emp$. Conversely, the classifier makes $h\hsame\emp$; with $p\hsame\emp$, determinacy of continuation compares $\Cont(h,p,r)$ with the right-unit continuation at $\emp$.
\end{proof}
\leanchecked{BEDC.Derived.RingUp.RingSingletonClassifier\_cont\_result\_empty\_iff}

\begin{theorem}[Ring singleton multiplication continuation classifier]
\label{thm:ring-singleton-classifier-mul-continuation-classifier}
If $\Cont(\mathsf{RingSingletonMul}(x,y),z,r)$, then
$$
C_{\emp}(r,h)\Longleftrightarrow A_{\emp}(z)\wedge A_{\emp}(h).
$$
\end{theorem}
\begin{proof}
The singleton multiplication endpoint is $\emp$. In the forward direction,
transport the classified empty result into the continuation and invert the
empty continuation to obtain $z\hsame\emp$; the right classifier carrier gives
$h\hsame\emp$. Conversely, $z\hsame\emp$ compares the continuation result with
the right-unit continuation at $\mathsf{RingSingletonMul}(x,y)$, so $r$ is
empty and classifies against $h$.
\end{proof}
\leanchecked{BEDC.Derived.RingUp.RingSingletonClassifier\_mul\_continuation\_classifier\_iff}

\begin{theorem}[Ring singleton continuation visible result exclusion]\label{thm:ring-singleton-classifier-continuation-visible-result-absurd}
If $C_{\emp}(h,k)$, $A_{\emp}(p)$, and $\Cont(h,p,-)$ is formed, then no
visible endpoint can be the continuation result:
$$
\Cont(h,p,\Ezero(r))\Longrightarrow\bot,
\qquad
\Cont(h,p,\Eone(r))\Longrightarrow\bot.
$$
\end{theorem}
\begin{proof}
The classifier supplies $h\hsame\emp$, while the carrier hypothesis supplies
$p\hsame\emp$. Continuation respects these two history-same witnesses, so a
displayed continuation result would be history-same to the right-unit result
at $\emp$. Constructor disjointness excludes both $\Ezero(r)$ and $\Eone(r)$
from being history-same to $\emp$.
\end{proof}
\leanchecked{BEDC.Derived.RingUp.RingSingletonClassifier\_continuation\_visible\_result\_absurd}
