\begin{definition}[Rat source specification]
\label{def:rat-source-specification}
The source is \(\IntUp \times \NatUp\) restricted to normalized denominator-positive pairs.
\end{definition}
\leandef{BEDC.Derived.RatUp.RatSourceSpec}

\begin{theorem}[Rat source specification yields the carrier]
\label{thm:rat-source-spec-to-carrier}
A rational source witness determines a rational carrier witness: the numerator
is an integer carrier, the denominator is unary, and the positive one-tail
witness excludes internal sameness with \(\emp\).
\end{theorem}
\leanchecked{BEDC.Derived.RatUp.RatSourceSpec\_to\_RatCarrier}
\begin{proof}
Unpack the source witness. The positive-denominator component gives a unary
tail \(t\) with \(d \hsame e_1(t)\). Unary closure under one-extension and
transport along \(\hsame\) give \(\mathsf{Unary}(d)\). If \(d\hsame \emp\),
then symmetry and transitivity give \(e_1(t)\hsame \emp\), contradicting
finite-kernel history no-confusion.
\end{proof}

\begin{definition}[Rat pattern specification]
\label{def:rat-pattern-specification}
The pattern is normalized numerator/denominator construction via a total normalization function on integer and positive-natural inputs.
\end{definition}

\begin{definition}[Rat classifier specification]
\label{def:rat-classifier-specification}
The classifier is structural sameness of normalized numerator and denominator representatives, tracked by mark sameness on signs and history sameness on the numerator and denominator histories.
\end{definition}
\leandef{BEDC.Derived.RatUp.RatClassifierSpec}

\begin{theorem}[Rat classifier reflexivity]
\label{thm:rat-classifier-reflexivity}
Every rational carrier witness classifies its representative with itself.
\end{theorem}
\begin{proof}
Reuse the carrier witness for both endpoints. The sign component is witnessed by reflexivity of \(\msame\), and the numerator and denominator components are witnessed by reflexivity of \(\hsame\).
\end{proof}
\leanchecked{BEDC.Derived.RatUp.RatClassifierSpec\_refl}

\begin{theorem}[Rat classifier denominator append closure]
\label{thm:rat-classifier-append-unary-denominators-closed}

\(\hsame\)-matching unary histories to their denominators preserves the

\end{theorem}

reflexivity to the resulting append history.
\end{proof}

\begin{theorem}[Rat history classifier one-tail exactness]
\label{thm:rat-history-classifier-e1-tail-unary-iff}

relates the one-extension denominators exactly when both tails are unary and
internally the same:
$$
\mathsf{RatHistoryClassifier}(\Eone(d),\Eone(e))
\Longleftrightarrow
\mathsf{Unary}(d)\land \mathsf{Unary}(e)\land \hsame(d,e).
$$
\end{theorem}
\begin{proof}
For the forward direction, unpack the classifier into two rational history
carrier witnesses and the endpoint \(\hsame\) link. The one-tail carrier
equivalence extracts unary membership of both tails, and \(\Eone\)-constructor
inversion gives \(\hsame(d,e)\). For the reverse direction, rebuild both
rational history carriers from the unary tails and lift \(\hsame(d,e)\) by
\(\Eone\)-congruence.
\end{proof}
\leanchecked{BEDC.Derived.RatUp.RatHistoryClassifier\_e1\_tail\_unary\_iff}

\begin{theorem}[Rat history classifier zero-extension endpoint exclusion]
\label{thm:rat-history-classifier-zero-extension-endpoint-exclusion}
For histories $t,d:\mathsf{BHist}$, zero-extension cannot appear as either
endpoint of the rational history classifier:
$$
(\mathsf{RatHistoryClassifier}(\Ezero(t),d)\to\mathsf{False})\land
(\mathsf{RatHistoryClassifier}(d,\Ezero(t))\to\mathsf{False}).
$$
\end{theorem}
\begin{proof}
For the left endpoint, apply
\autoref{thm:rat-history-classifier-positive-denominators} to the classifier
hypothesis. Its first positive-denominator witness belongs to $\Ezero(t)$, so
\autoref{thm:rat-positive-denominator-e0-absurd} gives the contradiction.

For the right endpoint, apply
\autoref{thm:rat-history-classifier-positive-denominators} to the classifier
hypothesis again. Its second positive-denominator witness belongs to
$\Ezero(t)$, and
\autoref{thm:rat-positive-denominator-e0-absurd} rules out that endpoint.
\end{proof}

\label{thm:rat-history-classifier-unary-context-zero-extension-endpoint-exclusion}

$\hsame(p_d,p_e)$, and $t_d,t_e$ are unary tails with $\hsame(t_d,t_e)$, then
the two contextual denominator endpoints cannot be internally the same as
zero-extension histories:
$$
(\hsame(\mathsf{append}(p_d,\mathsf{append}(d,t_d)),\Ezero(z))\to
\mathsf{False})\land
(\hsame(\mathsf{append}(p_e,\mathsf{append}(e,t_e)),\Ezero(z'))\to
\mathsf{False}).
$$
\end{theorem}
\begin{proof}
Apply
\autoref{thm:rat-history-classifier-unary-denominator-context-positive-denominators}
to the displayed classifier, the matching unary prefix context, and the
matching unary tail context. This produces positive unary denominator witnesses
for $\mathsf{append}(p_d,\mathsf{append}(d,t_d))$ and
$\mathsf{append}(p_e,\mathsf{append}(e,t_e))$.

For the left endpoint, transport the positive unary denominator witness across
the displayed $\hsame$ link from
$\mathsf{append}(p_d,\mathsf{append}(d,t_d))$ to $\Ezero(z)$. Then apply
\autoref{thm:rat-positive-denominator-e0-absurd} to rule out that
zero-extension endpoint.

For the right endpoint, repeat the same transport across the displayed
$\hsame$ link from $\mathsf{append}(p_e,\mathsf{append}(e,t_e))$ to
$\Ezero(z')$. The positive-denominator zero-head absurdity theorem gives the
second contradiction.
\end{proof}

$$

ledger witnesses now share the raw endpoint
$\mathsf{append}(p,\mathsf{append}(\rho,t))$.

Finally apply
\autoref{thm:rat-history-ledger-policy-shared-raw-visible-classifier} to those
two contextual ledger witnesses. Its conclusion is exactly the displayed
classifier between the two contextual visible denominator endpoints.
\end{proof}

\begin{theorem}[Rat ledger shared-raw contextual visible endpoint equivalence]
\label{thm:rat-history-ledger-shared-raw-context-visible-endpoint-equivalence}
If both $\mathsf{RatHistoryLedgerPolicy}(\rho,v)$ and
$\mathsf{RatHistoryLedgerPolicy}(\rho,v')$ hold, unary prefixes $p,p_v,p_{v'}$
satisfy $\hsame(p,p_v)$ and $\hsame(p,p_{v'})$, and unary tails
$t,t_v,t_{v'}$ satisfy $\hsame(t,t_v)$ and $\hsame(t,t_{v'})$, then every
target history $w$ sees the same classifier fiber over the two contextual
visible denominator endpoints:
$$
\mathsf{RatHistoryClassifier}(\mathsf{append}(p_v,\mathsf{append}(v,t_v)),w)
\Longleftrightarrow
\mathsf{RatHistoryClassifier}(\mathsf{append}(p_{v'},\mathsf{append}(v',t_{v'})),w).
$$
\end{theorem}
\begin{proof}
Apply
\autoref{thm:rat-history-ledger-shared-raw-context-visible-classifier} to the
two ledger witnesses and the matching unary prefix and tail hypotheses. This
gives the classifier edge from
$\mathsf{append}(p_v,\mathsf{append}(v,t_v))$ to
$\mathsf{append}(p_{v'},\mathsf{append}(v',t_{v'}))$.

For the forward direction, assume the left contextual visible endpoint is
classified with $w$. Reverse the contextual endpoint edge by rational classifier
symmetry, then compose it with the left-endpoint-to-$w$ classifier by
transitivity. The result is the right-endpoint-to-$w$ classifier.

For the reverse direction, assume the right contextual visible endpoint is
classified with $w$. Compose the contextual endpoint edge itself with the
right-endpoint-to-$w$ classifier by transitivity. The result is the
left-endpoint-to-$w$ classifier.
\end{proof}

\begin{theorem}[Rat ledger shared-raw context one-tail readback]
\label{thm:rat-history-ledger-shared-raw-context-e1-pair-readback}
If both $\mathsf{RatHistoryLedgerPolicy}(\rho,v)$ and
$\mathsf{RatHistoryLedgerPolicy}(\rho,v')$ hold, unary prefixes
$p,p_v,p_{v'}$ satisfy $\hsame(p,p_v)$ and $\hsame(p,p_{v'})$, unary tails
$t,t_v,t_{v'}$ satisfy $\hsame(t,t_v)$ and $\hsame(t,t_{v'})$, and the two
contextual visible endpoints display as one-extension histories,
$$
\hsame(\mathsf{append}(p_v,\mathsf{append}(v,t_v)),\Eone(a))
\quad\text{and}\quad
\hsame(\mathsf{append}(p_{v'},\mathsf{append}(v',t_{v'})),\Eone(b)),
$$
then the displayed tails are unary and internally the same:
$$
\mathsf{Unary}(a)\land \mathsf{Unary}(b)\land\hsame(a,b).
$$

Its ten-field fit is: traditional target shared-denominator visible-tail
determinacy; BEDC source $\RatUp$ ledger over $\mathsf{BHist}$; carrier positive
unary denominators; classifier $\mathsf{RatHistoryClassifier}$; stability from
unary context closure and classifier transport; ledger the shared raw
denominator plus two visible contexts; core theorem this block; exactness the
displayed tail readback; standard bridge normalized denominator readback; status
$\mathsf{ProofTarget}$.
\end{theorem}
\begin{proof}
First invoke
\autoref{thm:rat-history-ledger-shared-raw-context-visible-classifier} with the
two ledger witnesses and the matching raw-to-visible unary contexts. This gives
the rational history classifier between the two contextual visible endpoints.

Then apply
\autoref{thm:rat-history-classifier-unary-context-e1-pair-readback} to that
classifier and to the two displayed one-extension witnesses. The readback
conclusion is exactly $\mathsf{Unary}(a)\land \mathsf{Unary}(b)\land
\hsame(a,b)$.
\end{proof}

\begin{theorem}[Rat history ledger classifier endpoint equivalence]
\label{thm:rat-history-ledger-classifier-endpoint-equivalence}
If $\mathsf{RatHistoryLedgerPolicy}(\rho,v)$ holds, then every target history
$w$ is classified from $\rho$ exactly when it is classified from $v$:
$$
\mathsf{RatHistoryClassifier}(\rho,w)\leftrightarrow
\mathsf{RatHistoryClassifier}(v,w).
$$
\end{theorem}
\leanchecked{BEDC.Derived.RatUp.RatHistoryLedgerPolicy\_classifier\_endpoint\_equivalence}
\begin{proof}
The ledger supplies the raw-visible rational classifier. For the raw-to-visible
direction, reverse that classifier by symmetry and compose it with the
raw-target classifier by transitivity. The visible-to-raw direction is the
ledger classifier extension theorem.
\end{proof}

\begin{theorem}[Rat history ledger displayed context classifier endpoint equivalence]
\label{thm:rat-history-ledger-cont-unary-context-classifier-endpoint-equivalence}
If $\mathsf{RatHistoryLedgerPolicy}(\rho,v)$ holds, unary prefixes
$p_\rho,p_v$ satisfy $\hsame(p_\rho,p_v)$, unary tails $t_\rho,t_v$ satisfy
$\hsame(t_\rho,t_v)$, and the displayed continuation witnesses
$\mathsf{Cont}(p_\rho,\rho,m_\rho)$,
$\mathsf{Cont}(m_\rho,t_\rho,out_\rho)$,
$\mathsf{Cont}(p_v,v,m_v)$, and
$\mathsf{Cont}(m_v,t_v,out_v)$ expose the two contextual endpoints, then
every target history $w$ is classified from the displayed raw endpoint exactly
when it is classified from the displayed visible endpoint:
$$
\mathsf{RatHistoryClassifier}(out_\rho,w)\leftrightarrow
\mathsf{RatHistoryClassifier}(out_v,w).
$$
\end{theorem}
\begin{proof}
First close the ledger under the matching unary denominator context. Apply
\autoref{thm:rat-ledger-unary-denominator-context-closed} to the original
ledger, the prefix match, and the tail match. This gives a ledger between
$\mathsf{append}(p_\rho,\mathsf{append}(\rho,t_\rho))$ and
$\mathsf{append}(p_v,\mathsf{append}(v,t_v))$.

The four continuation witnesses identify the displayed endpoints with those
normalized append endpoints. Associativity of $\mathsf{append}$ turns the
two-step displays through $m_\rho$ and $m_v$ into $\hsame$ links from
$out_\rho$ and $out_v$ to the corresponding normalized contextual histories.

Transport the contextual ledger across these two endpoint identifications by
\autoref{thm:rat-ledger-hsame-transport}. The transported ledger now relates
$out_\rho$ and $out_v$ directly.

Finally apply
\autoref{thm:rat-history-ledger-classifier-endpoint-equivalence} to the
transported ledger. Its conclusion is the displayed classifier endpoint
equivalence for every target history $w$.
\end{proof}
