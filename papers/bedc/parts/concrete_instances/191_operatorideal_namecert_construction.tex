\chapter{A Concrete Naming Certificate for $\OperatorIdealUp$}
\label{ch:concrete-instances-operatorideal-namecert}

The operator-ideal interface $\OperatorIdealUp$ packages a two-sided ideal of bounded operators on a Hilbert space carrier (compact, trace class, Hilbert-Schmidt). It depends on $\NameCert_{\HilbertUp}$ (\autoref{ch:concrete-instances-hilbert-namecert}) and $\NameCert_{\IdealUp}$ (\autoref{ch:concrete-instances-ideal-namecert}).

\section{Carrier definition}
\label{sec:operatorideal-carrier}

\begin{definition}[OperatorIdeal Hilbert source surface]
\label{def:operatorideal-hilbert-source-surface}
Let $H$ be a $\HilbertUp$ carrier supplied by $\NameCert_{\HilbertUp}$, with Hilbert carrier predicate $C_H$, classifier $\sim_H$, vector addition, additive inverse, zero, scalar action, and inner-product norm rows. Let $\mathsf{Op}_H(T)$ be a $\mathsf{BHist}$ predicate whose endpoints are bounded $\LinearMapUp(H,H)$ map-codes. The operator-ideal source surface over $H$ is the ten-field $\NameCert_{\OperatorIdealUp}$ inventory with the following rows.
\begin{enumerate}
\item The traditional target is a two-sided ideal of bounded Hilbert operators on $H$.
\item The BEDC source is $\NameCert_{\HilbertUp}$ together with $\NameCert_{\IdealUp}$ read over the bounded-operator composition ring on $H$.
\item The carrier is the predicate $\mathsf{Op}_H$ on bounded-operator rows $H\to H$.
\item The classifier is inherited from the Hilbert and linear-map classifier: $T\sim_{\mathsf{Op}(H)}S$ means pointwise $\sim_H$ agreement on every carried Hilbert endpoint after both map-codes are certified bounded and linear.
\item The stability rows are the bounded zero operator, addition, negation, scalar action where the Hilbert module row is consumed, and composition on the left and on the right by every carried bounded operator.
\item The ledger records the Hilbert source certificate, the $\LinearMapUp(H,H)$ row, the ideal-membership witness for the selected predicate, and the two composition witnesses for left and right multiplication.
\item The core theorem rows are the carrier support, additive closure, two-sided absorption, and downstream trace-class boundary rows in \autoref{sec:operatorideal-certificate}.
\item Exactness is row exhaustion: an accepted operator-ideal endpoint exposes only the Hilbert source, bounded linear-map row, ideal membership, classifier transport, additive rows, scalar rows, and the two bounded composition witnesses displayed here.
\item The standard bridge row is absent at this paper site.
\item The boundary package is the trace-class-ready operator-ideal surface consumed by $\DensityMatrixUp$, $\NuclearUp$, and $\QuantumChannelUp$.
\end{enumerate}
\end{definition}

\begin{definition}[Trace-class operator-ideal carrier]
\label{def:operatorideal-trace-class-carrier}
Let $H$ be a $\HilbertUp$ carrier, and let $\mathsf{Op}_H$ be the bounded-operator carrier from \autoref{def:operatorideal-hilbert-source-surface}. A trace-class carrier over $H$ is the subcarrier
$$
\mathsf{TC}(H)(T)
:\Longleftrightarrow
\mathsf{Op}_H(T)\land \mathsf{TraceClassIdeal}_H(T),
$$
where $\mathsf{TraceClassIdeal}_H$ is an $\IdealUp$ predicate over the bounded-operator composition ring carried by $\mathsf{Op}_H$. Its classifier is the restriction of $\sim_{\mathsf{Op}(H)}$, and its ledger contains the bounded-operator witness for $T$ together with the $\IdealUp$ membership witness $\mathsf{TraceClassIdeal}_H(T)$.
\end{definition}

\section{The certificate}
\label{sec:operatorideal-certificate}

\begin{theorem}[OperatorIdeal carrier support row]
\label{thm:operatorideal-carrier-support-row}
Let $H$ be a $\HilbertUp$ carrier, and let $\mathsf{TC}(H)$ be the trace-class carrier of \autoref{def:operatorideal-trace-class-carrier}. If $\mathsf{TC}(H)(T)$, then $T$ is carried by the bounded-operator carrier over $H$:
$$
\mathsf{TC}(H)(T)\Longrightarrow \mathsf{Op}_H(T).
$$
\end{theorem}
\begin{proof}
Unpack \autoref{def:operatorideal-trace-class-carrier}. A trace-class witness is a pair consisting of the bounded-operator row $\mathsf{Op}_H(T)$ and an $\IdealUp$ membership witness $\mathsf{TraceClassIdeal}_H(T)$.

Projecting the first component gives the displayed bounded-operator carrier support. No analytic trace row is used; the result is exactly the support projection from the carrier definition.
\end{proof}

\begin{theorem}[OperatorIdeal additive closure row]
\label{thm:operatorideal-additive-closure-row}
Let $H$ be a $\HilbertUp$ carrier, and let $\mathsf{TC}(H)$ be the trace-class carrier of \autoref{def:operatorideal-trace-class-carrier}. The $\OperatorIdealUp$ certificate supplies the trace-class zero, additive, and negation rows:
$$
\mathsf{TC}(H)(0_H),
$$
$$
\mathsf{TC}(H)(T)\land\mathsf{TC}(H)(S)\Longrightarrow \mathsf{TC}(H)(T+_H S),
$$
and
$$
\mathsf{TC}(H)(T)\Longrightarrow \mathsf{TC}(H)(-_H T).
$$
\end{theorem}
\begin{proof}
Read $\mathsf{TC}(H)$ as the $\IdealUp$ predicate $\mathsf{TraceClassIdeal}_H$ over the bounded-operator composition ring from \autoref{def:operatorideal-trace-class-carrier}. The zero row of the $\IdealUp$ predicate gives membership of the bounded zero operator; the bounded-operator surface of \autoref{def:operatorideal-hilbert-source-surface} supplies its $\mathsf{Op}_H$ carrier witness.

For addition, two trace-class witnesses project to two ideal-membership witnesses. The additive closure row of $\NameCert_{\IdealUp}$ gives membership for $T+_H S$, while the bounded linear-map rows inherited from $\HilbertUp$ and $\LinearMapUp$ carry the sum as a bounded operator.

For negation, project the additive inverse row of the same $\IdealUp$ predicate and pair it with the bounded negation row from the operator surface. These paired rows are precisely the three displayed $\mathsf{TC}(H)$ clauses.
\end{proof}

\begin{theorem}[OperatorIdeal two-sided absorption row]
\label{thm:operatorideal-two-sided-absorption-row}
Let $H$ be a $\HilbertUp$ carrier, and let $\mathsf{TC}(H)$ be the trace-class carrier of \autoref{def:operatorideal-trace-class-carrier}. If $A$ is a carried bounded operator on $H$ and $T$ is trace-class on $H$, then both composites are trace-class:
$$
\mathsf{Op}_H(A)\land\mathsf{TC}(H)(T)
\Longrightarrow
\mathsf{TC}(H)(A\circ T)\land\mathsf{TC}(H)(T\circ A).
$$
\end{theorem}
\begin{proof}
Unpack the trace-class witness for $T$. It supplies the bounded-operator row $\mathsf{Op}_H(T)$ and the ideal-membership row $\mathsf{TraceClassIdeal}_H(T)$.

The input $\mathsf{Op}_H(A)$ is exactly the ambient carried endpoint required by the two-sided absorption field of $\NameCert_{\IdealUp}$ over the bounded-operator composition ring. Projecting that field gives ideal membership for both $A\circ T$ and $T\circ A$.

The bounded composition rows of \autoref{def:operatorideal-hilbert-source-surface} carry the two composites as bounded $\LinearMapUp(H,H)$ map-codes. Pairing these carrier witnesses with the two projected ideal-membership witnesses gives $\mathsf{TC}(H)(A\circ T)$ and $\mathsf{TC}(H)(T\circ A)$.
\end{proof}

\begin{theorem}[DensityMatrix trace-class dependency boundary]
\label{thm:operatorideal-densitymatrix-trace-class-boundary}
The downstream $\DensityMatrixUp$ and $\QuantumChannelUp$ chapters may consume from $\OperatorIdealUp(H)$ only the trace-class carrier $\mathsf{TC}(H)$ of \autoref{def:operatorideal-trace-class-carrier}, its classifier transport from $\sim_{\mathsf{Op}(H)}$, the zero, addition, negation, and real scalar/module rows inherited through $\HilbertUp$ and $\LinearMapUp$, and the two-sided absorption row of \autoref{thm:operatorideal-two-sided-absorption-row}. Positivity rows, trace rows, and trace-one rows belong to the downstream chapters themselves. This boundary does not claim compactness, a spectral theorem, trace existence, or a standard operator-algebra bridge.
\end{theorem}
\begin{proof}
Inspect the ten-field surface in \autoref{def:operatorideal-hilbert-source-surface}. Its exactness field says that an operator-ideal endpoint exposes only the Hilbert source, bounded linear-map row, ideal membership, classifier transport, additive rows, scalar rows, and left and right bounded-composition witnesses.

The trace-class carrier in \autoref{def:operatorideal-trace-class-carrier} further restricts that surface to the $\IdealUp$ predicate $\mathsf{TraceClassIdeal}_H$. Therefore the support, additive, and absorption theorems above are the available projections for downstream use.

The positivity predicate $\mathsf{Pos}_H$, the trace functional $\operatorname{Tr}_H$, and the trace-one equation are not fields of either displayed OperatorIdeal inventory. They are introduced only when $\DensityMatrixUp$ and $\QuantumChannelUp$ state their own carrier and certificate rows, so none of compactness, spectral decomposition, trace existence, or a standard bridge is exported by this chapter.
\end{proof}

\begin{theorem}[OperatorIdeal trace-class scalar closure]
\label{thm:operatorideal-trace-class-scalar-closure}
Let $H$ be a $\HilbertUp$ carrier, and let $\mathsf{TC}(H)$ be the trace-class carrier of \autoref{def:operatorideal-trace-class-carrier}. Let $a$ be a carried $\RealUp$ scalar endpoint for the Hilbert module action. Write $M_a:H\to H$ for the bounded Hilbert scalar-action endomorphism carried by the scalar row of \autoref{def:operatorideal-hilbert-source-surface},
$$
M_a(x):=a\cdot_H x,
$$
and let the induced trace-class scalar action on operators be
$$
a\odot_H T:=M_a\circ T.
$$
Then trace-class membership is closed under this scalar action:
$$
\mathsf{TC}(H)(T)\Longrightarrow\mathsf{TC}(H)(a\odot_H T).
$$
\end{theorem}
\begin{proof}
The carried scalar hypothesis gives a certified scalar-action row on the Hilbert module surface. Read as a map in the bounded-operator composition ring, that row carries the homothety $M_a$ as a bounded $\LinearMapUp(H,H)$ endpoint; in the notation of \autoref{def:operatorideal-hilbert-source-surface}, this is the ambient witness $\mathsf{Op}_H(M_a)$.

Now assume $\mathsf{TC}(H)(T)$. By \autoref{def:operatorideal-trace-class-carrier}, this witness consists of a bounded-operator carrier row $\mathsf{Op}_H(T)$ and an ideal-membership row $\mathsf{TraceClassIdeal}_H(T)$ inside the bounded-operator composition ring.

Apply \autoref{thm:operatorideal-two-sided-absorption-row} with the carried ambient operator $A=M_a$. The left absorption component gives
$$
\mathsf{TC}(H)(M_a\circ T),
$$
while the right component gives $\mathsf{TC}(H)(T\circ M_a)$ and is not needed here.

The displayed definition of the inherited scalar action identifies $M_a\circ T$ with $a\odot_H T$. The classifier-transport row for the ideal predicate in \autoref{thm:ideal-classifier-transport-obligation} carries the obtained trace-class witness across that endpoint identity, giving
$$
\mathsf{TC}(H)(a\odot_H T).
$$
\end{proof}

\begin{theorem}[Trace-class binary linear-combination closure]
\label{thm:operatorideal-trace-class-binary-linear-combination-closure}
Let $H$ be a $\HilbertUp$ carrier, and let $\mathsf{TC}(H)$ be the trace-class carrier of \autoref{def:operatorideal-trace-class-carrier}. Write $+_{\mathsf{TC}(H)}$ for the trace-class addition inherited from the operator-ideal additive row, and use the real trace-class scalar action $\odot_H$ of \autoref{thm:operatorideal-trace-class-scalar-closure}. Let $T$ and $S$ be operator endpoints, and let $a$ and $b$ be carried $\RealUp$ scalar endpoints for the Hilbert module action. If
$$
\mathsf{TC}(H)(T)\land\mathsf{TC}(H)(S),
$$
then
$$
\mathsf{TC}(H)\bigl((a\odot_H T)+_{\mathsf{TC}(H)}(b\odot_H S)\bigr).
$$
\end{theorem}
\begin{proof}
Assume the two trace-class witnesses. Applying \autoref{thm:operatorideal-trace-class-scalar-closure} to the carried scalar $a$ and the endpoint $T$ gives
$$
\mathsf{TC}(H)(a\odot_H T).
$$
Applying the same scalar-closure row to the carried scalar $b$ and the endpoint $S$ gives
$$
\mathsf{TC}(H)(b\odot_H S).
$$

The operation $+_{\mathsf{TC}(H)}$ is the restriction of the bounded-operator addition row named in \autoref{def:operatorideal-hilbert-source-surface} to endpoints carrying the trace-class predicate of \autoref{def:operatorideal-trace-class-carrier}. Therefore the additive closure row of \autoref{thm:operatorideal-additive-closure-row}, applied to the two scalar-multiple witnesses displayed above, gives
$$
\mathsf{TC}(H)\bigl((a\odot_H T)+_{\mathsf{TC}(H)}(b\odot_H S)\bigr).
$$
Only the trace-class ideal membership, scalar closure, and additive closure rows are used; positivity, trace, and trace-one rows remain outside this operator-ideal claim.
\end{proof}
