
\chapter{Inductive Package-Token Policy for Signature Packages}
\label{ch:v152-inductive-package-token-policy}

Chapter~\ref{ch:v151-package-token-policy} introduced \(\mathsf{PackageTokenPolicy}(\Pi)\). This chapter removes one remaining ambiguity: package-sameness for the concrete signature-token instance should be generated by one rule only. Package reflection should then be a constructor-inversion theorem, not an extra informal assumption.

\begin{definition}[Concrete signature-token introduction]
For the concrete signature-token instance, the judgment
\[
  \TokIntro(\Pi,s,p)
\]
means that token \(p\) is introduced as a visible token for answer history \(s\) under probe bundle \(\Pi\).
\end{definition}

\begin{definition}[Inductive package-sameness for signature tokens]
For the concrete signature-token instance, \(\psame_{\Pi}^{\mathrm{sig}}(p,q)\) is generated only by the following rule:
\[
\frac{
  \TokIntro(\Pi,s,p)
  \qquad
  \TokIntro(\Pi,t,q)
  \qquad
  \hsame(s,t)
}{
  \psame_{\Pi}^{\mathrm{sig}}(p,q)
}.
\]
No other package-sameness constructors are admitted for this concrete instance.
\end{definition}

\begin{lemma}[Package-sameness soundness]
\label{lem:v152-psame-sound}
If
\[
  \TokIntro(\Pi,s,p),\quad \TokIntro(\Pi,t,q),\quad \hsame(s,t),
\]
then
\[
  \psame_{\Pi}^{\mathrm{sig}}(p,q).
\]
\end{lemma}

\begin{proof}
This is the single constructor of \(\psame_{\Pi}^{\mathrm{sig}}\).
\end{proof}

\begin{lemma}[Package-sameness reflection]
\label{lem:v152-psame-reflect}
If
\[
  \TokIntro(\Pi,s,p),\quad \TokIntro(\Pi,t,q),\quad
  \psame_{\Pi}^{\mathrm{sig}}(p,q),
\]
then
\[
  \hsame(s,t).
\]
\end{lemma}

\begin{proof}
By inversion on the derivation of \(\psame_{\Pi}^{\mathrm{sig}}(p,q)\). Since the concrete package-sameness relation has only the displayed constructor, any such derivation is generated from introduced signatures that are \(\hsame\). Therefore \(\hsame(s,t)\), up to the token-introduction premises used to identify the displayed tokens.
\end{proof}

\begin{theorem}[Concrete PackageTokenPolicy]
\label{thm:v152-concrete-package-token-policy}
The concrete signature-token instance satisfies \(\mathsf{PackageTokenPolicy}(\Pi)\) whenever token introduction is available for generated signatures and package-sameness is interpreted as \(\psame_{\Pi}^{\mathrm{sig}}\).
\end{theorem}

\begin{proof}
Token introduction is supplied by the chosen concrete token-introduction rule. Package-sameness introduction is Lemma~\ref{lem:v152-psame-sound}. Package reflection is Lemma~\ref{lem:v152-psame-reflect}. The no-extra-constructor clause holds by definition of \(\psame_{\Pi}^{\mathrm{sig}}\).
\end{proof}

\begin{corollary}[Exact signature package]
For the concrete signature-token instance, \(\psame_{\Pi}^{\mathrm{sig}}(p,q)\) between introduced tokens is equivalent to internal sameness of the answer histories that introduced them.
\end{corollary}

\begin{proof}
The forward direction is Lemma~\ref{lem:v152-psame-reflect}; the reverse direction is Lemma~\ref{lem:v152-psame-sound}.
\end{proof}

\begin{remark}[Implementation note]
In a machine implementation, this chapter should be realized by defining package-sameness as an inductive relation with exactly one constructor. Reflection should then be proved by case analysis on that constructor.
\end{remark}
