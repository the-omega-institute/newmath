
\chapter{Concrete Domain Policy for the Unary Source}
\label{ch:v152-unary-domain-policy}

Chapter~\ref{ch:v151-domain-policy} introduced \(\mathsf{DomainPolicy}(D)\) as the condition allowing \(\InDom(D,h)\) to be transported along internal history-sameness. This chapter records the first intended concrete instance: the unary source.

\begin{definition}[Unary source domain]
Let \(\mathsf{UnarySource}\) be the source-domain name whose internal domain judgment is
\[
  \InDom(\mathsf{UnarySource},h)
  \quad\text{exactly when}\quad
  \Unary(h).
\]
This is not a set-membership statement. It is a domain judgment licensed by the unary-generation rules.
\end{definition}

\begin{lemma}[Unary is transported by history-sameness]
\label{lem:v152-unary-transport}
If
\[
  \Unary(h),\qquad \hsame(h,k),
\]
then
\[
  \Unary(k).
\]
\end{lemma}

\begin{proof}
Proceed by induction on the derivation of \(\Unary(h)\).

For the empty case, \(h\) is internally the empty history. Since \(\hsame(h,k)\), closed generation and no-confusion for \(\hsame\) force \(k\) to be internally the empty history. Hence \(\Unary(k)\).

For the step case, the unary derivation has the form \(\Unary(h_0)\Rightarrow\Unary(\Eone(h_0))\). From \(\hsame(\Eone(h_0),k)\), constructor inversion for \(\hsame\) yields a history \(k_0\) such that \(k\) is internally \(\Eone(k_0)\) and \(\hsame(h_0,k_0)\). By the induction hypothesis, \(\Unary(k_0)\). Therefore \(\Unary(\Eone(k_0))\), and by internal sameness this is the required unary judgment for \(k\).
\end{proof}

\begin{theorem}[Unary source satisfies DomainPolicy]
\label{thm:v152-unary-domain-policy}
\[
  \mathsf{DomainPolicy}(\mathsf{UnarySource}).
\]
\end{theorem}

\begin{proof}
Assume \(\InDom(\mathsf{UnarySource},h)\) and \(\hsame(h,k)\). By definition of \(\mathsf{UnarySource}\), the first premise is \(\Unary(h)\). Lemma~\ref{lem:v152-unary-transport} gives \(\Unary(k)\), hence \(\InDom(\mathsf{UnarySource},k)\).
\end{proof}

\begin{corollary}[Unary-domain gaps may transport sources]
Any concrete gap theorem whose source domain is \(\mathsf{UnarySource}\) may use domain transport along \(\hsame\), provided no additional source-domain restriction is imposed.
\end{corollary}

\begin{proof}
Immediate from Theorem~\ref{thm:v152-unary-domain-policy} and Lemma~\ref{thm:v151-domain-transport}.
\end{proof}
