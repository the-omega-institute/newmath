\chapter{Unary Continuation Commutativity and the Additive Name}
\label{sec:concrete-hardening-unary-add-commutativity}

The \(\AddUp\) certificate establishes a unary continuation interface with unit, associativity, determinacy, and closure. To justify the additive name rather than merely a monoid-like continuation name, one more paper-level obligation is needed: unary continuation commutativity.

\begin{definition}[Unary histories]
The unary judgment \(\Unary(h)\) is generated by
\[
  \Unary(\emp)
  \qquad\text{and}\qquad
  \Unary(h)\Rightarrow \Unary(\Eone(h)).
\]
No rule generates \(\Unary(\Ezero(h))\).
\end{definition}

\begin{definition}[Unary continuation result]
\(\UnaryCont(h,k,r)\) means
\[
  \Unary(h),\qquad \Unary(k),\qquad \ContR(h,k,r).
\]
\end{definition}

\begin{lemma}[Unary continuation closure]
\label{prop:unary-repetition-continuation-closed}
If \(\UnaryCont(h,k,r)\), then \(\Unary(r)\).
\end{lemma}
\leanchecked{BEDC.FKernel.Examples.Unary.unary\_append\_closed}
\leanchecked{BEDC.FKernel.Examples.Unary.unary\_cont\_closed}
\leanchecked{BEDC.FKernel.Examples.Unary.unary\_cont\_right\_factor}

\begin{proof}
Induct on the derivation of \(\ContR(h,k,r)\). The empty case gives \(r\) internally same as \(k\), hence unary. The \(\Eone\)-case preserves unary by the constructor rule. The \(\Ezero\)-case is incompatible with \(\Unary(h)\) by unary no-zero-extension.
\end{proof}

\begin{theorem}[Unary continuation commutativity]
\label{thm:unary-cont-comm}
If
\[
  \Unary(h),\quad \Unary(k),\quad
  \ContR(h,k,r),\quad \ContR(k,h,r'),
\]
then
\[
  \hsame(r,r').
\]
\end{theorem}
\leanchecked{BEDC.FKernel.Examples.Unary.unary\_cont\_comm}

\begin{proof}
Proceed by induction on the unary derivation of \(h\).

If \(h=\emp\), then \(\ContR(\emp,k,r)\) gives \(\hsame(r,k)\), while \(\ContR(k,\emp,r')\) gives \(\hsame(r',k)\) by the right-unit theorem for continuation. Thus \(\hsame(r,r')\).

For the step case, write \(h=\Eone(h_0)\) with \(\Unary(h_0)\). From \(\ContR(\Eone(h_0),k,r)\), inversion yields a result \(u\) such that \(\ContR(h_0,k,u)\) and \(\hsame(r,\Eone(u))\). The other branch \(\ContR(k,\Eone(h_0),r')\) is shown, by induction on the unary derivation of \(k\), to be internally same as \(\Eone(v)\), where \(\ContR(k,h_0,v)\). Applying the outer induction hypothesis to \(h_0,k\) gives \(\hsame(u,v)\), hence \(\hsame(\Eone(u),\Eone(v))\), and therefore \(\hsame(r,r')\).

This proof uses only unary constructors, continuation inversion, and history-sameness congruence under \(\Eone\).
\end{proof}

\begin{corollary}[Upgrade of the additive naming certificate]
The certificate for \(\AddUp\) is upgraded from a unary continuation monoid-like certificate to a commutative unary additive certificate once Theorem~\ref{thm:unary-cont-comm} is included among its stability fields.
\end{corollary}

\begin{proof}
The certificate already contains unit, associativity, determinacy, closure, and composite ledger policy. Theorem~\ref{thm:unary-cont-comm} supplies commutativity up to internal history-sameness.
\end{proof}

\begin{remark}[Naming boundary]
Without Theorem~\ref{thm:unary-cont-comm}, the conservative public name is \(\mathsf{UnaryContMonoid}^{\uparrow}\). With it, the name \(\AddUp\) is licensed as a commutative unary addition-like interface.
\end{remark}
