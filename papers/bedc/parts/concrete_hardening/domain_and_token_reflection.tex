\chapter{Internal Domain Judgments and Signature Token Reflection}
\label{sec:concrete-hardening-domain-token-reflection}

This chapter replaces two external-looking concrete notations by internal judgments:
$$
  h\in D,
  \qquad
  \Tok_{\Pi}(s).
$$
They are readable, but both carry too much external mathematical flavor. This chapter replaces them with internal judgments.

\begin{definition}[Internal domain judgment]
\label{def:internal-domain-judgment}
A concrete source domain is not treated as a set. It is a source name \(D\) equipped with a judgment
$$
  \InDom(D,h),
$$
read as: the history \(h\) is admitted by the source domain name \(D\).
\end{definition}
\leanchecked{BEDC.FKernel.Gap.InDom\_iff\_setup\_field}

\begin{remark}[Layer discipline]
The notation \(\InDom(D,h)\) is still written in the metalanguage, but it is the public BEDC judgment to be used in all concrete gap statements. The notation \(h\in D\) is not used in the active concrete presentation.
\end{remark}

\begin{definition}[Signature token introduction]
\label{def:signature-token-introduction}
For a probe bundle \(\Pi\) and an answer history \(s\), a visible package token is introduced by a judgment
$$
  \TokIntro(\Pi,s,p).
$$
It means that \(p\) is a package token introduced from the signature history \(s\) under bundle \(\Pi\).
\end{definition}
\leanchecked{BEDC.FKernel.Package.TokIntro\_iff\_setup\_field}
\leandef{BEDC.FKernel.Package.SignatureTokenIntro}
\leanvariant{BEDC.FKernel.Package.SignatureTokenIntro\_iff\_TokIntro}

\begin{definition}[Concrete signature package]
\label{def:concrete-signature-package}
The concrete signature package relation is defined by token introduction:
$$
  \PkgSig(\Pi,s,p)
  \quad\Longleftrightarrow\quad
  \TokIntro(\Pi,s,p).
$$
\end{definition}
\leandef{BEDC.FKernel.Package.PkgSig}
\leanchecked{BEDC.FKernel.Package.PkgSig\_iff\_TokIntro}
\leanvariant{BEDC.FKernel.Package.TokIntro\_iff\_PkgSig}
\leanvariant{BEDC.FKernel.Package.PkgSig\_intro\_from\_token}
\leanvariant{BEDC.FKernel.Package.TokIntro\_nonempty\_pkg}

\begin{definition}[Concrete package-sameness policy]
For the signature-token instance, package-sameness is generated only from history-sameness of source signatures:
$$
  \TokIntro(\Pi,s,p),\quad
  \TokIntro(\Pi,t,q),\quad
  \hsame(s,t)
  \quad\Rightarrow\quad
  \psame(p,q).
$$
No additional package-sameness constructors are admitted for the concrete signature-token instance.
\end{definition}

\begin{theorem}[Package reflection for signature tokens]
\label{thm:package-reflection}
Assume $\PkgTokPol(\Pi)$ for the concrete signature-token package policy. If
$$
  \TokIntro(\Pi,s,p),\qquad
  \TokIntro(\Pi,t,q),\qquad
  \psame(p,q),
$$
then
$$
  \hsame(s,t).
$$
\end{theorem}
\leanchecked{BEDC.FKernel.Package.package\_reflection}
\leanvariant{BEDC.FKernel.Package.package\_reflection\_forward}
\leanvariant{BEDC.FKernel.Package.package\_reflection\_roundtrip\_pair}
\leanvariant{BEDC.FKernel.Package.package\_reflection\_iff\_hsame}
\leanvariant{BEDC.FKernel.Package.psame\_reflection\_witness\_chain\_under\_tok\_unique}

\begin{proof}
Apply the reflection field of $\PkgTokPol(\Pi)$ to the displayed token-introduction judgments and the package-sameness judgment. In the base-relation presentation, inversion on the constructor exposes introducing signatures for $p$ and $q$; the policy reflection field, or a token-uniqueness theorem supplying that field, transports those constructor witnesses to the displayed introductions and yields $\hsame(s,t)$.
\end{proof}

\begin{corollary}[Concrete package equivalence is signature-grounded]
\label{cor:concrete-package-equivalence-signature-grounded}
Assume $\PkgTokPol(\Pi)$ for the concrete signature-token package policy. If
$$
\TokIntro(\Pi,s,p)
\qquad\text{and}\qquad
\TokIntro(\Pi,t,q),
$$
then
$$
\psame(p,q)
\quad\Longleftrightarrow\quad
\hsame(s,t).
$$
\end{corollary}
\leanchecked{BEDC.FKernel.Package.package\_reflection\_iff\_hsame}

\begin{proof}
Apply the two fields of $\PkgTokPol(\Pi)$ to the displayed token-introduction judgments. The reflection field sends $\psame(p,q)$ to $\hsame(s,t)$, and the soundness field sends $\hsame(s,t)$ back to $\psame(p,q)$.
\end{proof}
