
\chapter{Extension and Continuation Proof Spine}
\label{ch:proof-standing-ext-cont-proof-spine}

This chapter records the proof-standing status of the extension and continuation relations. Both are relation-generated behaviors, not primitive functions.

\section{Relational extension}

\begin{definition}[Extension relation]
The relation \(\ExtR(m,h,r)\) is generated by:
$$
\ExtR(\bzero,h,\Ezero(h)),
\qquad
\ExtR(\bone,h,\Eone(h)).
$$
It expresses that emitting mark \(m\) after history \(h\) generates result \(r\).
\end{definition}

\begin{theorem}[Extension determinacy up to history-sameness]
\label{thm:extension-determinacy-up-to-history-sameness}
If
$$
\ExtR(m,h,r)
\quad\text{and}\quad
\ExtR(m,h,r'),
$$
then \(\HSame(r,r')\).
\end{theorem}
\leanchecked{BEDC.FKernel.Ext.ext\_determinacy\_up\_to\_hsame\_spine}
\begin{proof}
By case analysis on the generated extension proof. If \(m=\bzero\), both generated results have the form \(\Ezero(h)\), hence they are history-same by reflexivity. If \(m=\bone\), both generated results have the form \(\Eone(h)\), hence they are history-same by reflexivity. No cross mark case is generated for the same mark input.
\end{proof}

\begin{theorem}[Extension respects internal sameness]
\label{thm:extension-respects-internal-sameness-spine}
Suppose \(\MSame(m,n)\), \(\HSame(h,k)\), \(\ExtR(m,h,r)\), and \(\ExtR(n,k,s)\). Then \(\HSame(r,s)\).
\end{theorem}
\leanchecked{BEDC.FKernel.Ext.ext\_respects\_internal\_sameness\_spine}
\begin{proof}
By induction on the generated proof of \(\MSame(m,n)\). There are only two cases. In the \(\bzero\) case, the extension results are generated with outer constructor \(\Ezero\), and the conclusion follows by applying the \(\Ezero\) history-sameness constructor to \(\HSame(h,k)\). The \(\bone\) case is identical with \(\Eone\).
\end{proof}

\begin{theorem}[Extension result is nonempty]
\label{thm:ext-result-ne-empty}
If $\ExtR(m,h,r)$, then $r$ is not history-same to $\emp$.
\end{theorem}
\begin{proof}
Case-split the generated extension proof. The $\bzero$ case has result
$\Ezero(h)$, which cannot be history-same to $\emp$ by history constructor
no-confusion. The $\bone$ case is identical with $\Eone(h)$.
\end{proof}
\leanchecked{BEDC.FKernel.Ext.ext\_result\_ne\_empty}

\section{Relational continuation}

\begin{definition}[Continuation relation]
The relation $\ContR(h,k,r)$ is generated by:
\begin{enumerate}
  \item $\ContR(h,\emp,h)$;
  \item if $\ContR(h,k,r)$, then $\ContR(h,\Ezero(k),\Ezero(r))$;
  \item if $\ContR(h,k,r)$, then $\ContR(h,\Eone(k),\Eone(r))$.
\end{enumerate}
It expresses continuation of history $h$ by history $k$ without treating continuation as a primitive function.
\end{definition}

\begin{theorem}[Continuation determinacy up to history-sameness]
\label{thm:continuation-determinacy-up-to-history-sameness}
If \(\ContR(h,k,r)\) and \(\ContR(h,k,r')\), then \(\HSame(r,r')\).
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_determinacy\_up\_to\_hsame\_spine}
\begin{proof}
By induction on the generated right-history structure in the first continuation proof. In the empty case, the second proof must also be the empty case, hence both results are $h$. In the $\Ezero$ case, the second proof must also have right input and result with outer $\Ezero$ by closed generation; apply the induction hypothesis to the sub-results and then apply the $\Ezero$ constructor for $\HSame$. The $\Eone$ case is analogous.
\end{proof}

\begin{theorem}[Continuation result history-sameness tag separation]
\label{thm:cont-result-hsame-tag-separation}
If $\ContR(h,k,r_0)$ and $\ContR(h,k,r_1)$, and if
$\HSame(r_0,\Ezero(z))$ while $\HSame(r_1,\Eone(o))$, then contradiction.
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_result\_hsame\_tag\_separation}
\begin{proof}
Continuation determinacy gives $\HSame(r_0,r_1)$.  Composing this equality with
the two displayed tag classifications yields
$\HSame(\Ezero(z),\Eone(o))$, contradicting generated-history no-confusion
between the zero and one constructors.
\end{proof}

\begin{theorem}[Continuation step result history-sameness no-confusion pair]
\label{thm:cont-step-result-hsame-no-confusion-pair}
The two one-step continuation result shapes are disjoint up to history-sameness:
$$
\ContR(h,\Ezero(k_0),r_0),\quad
\ContR(h,\Eone(k_1),r_1),\quad
\HSame(r_0,r_1)
$$
is impossible, and the same holds with the displayed zero and one steps
exchanged.
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_step\_result\_hsame\_no\_confusion\_pair}
\begin{proof}
Invert the zero-step continuation witness to expose
$r_0=\Ezero(a)$ and invert the one-step witness to expose $r_1=\Eone(b)$.
The displayed $\HSame(r_0,r_1)$ then becomes
$\HSame(\Ezero(a),\Eone(b))$, which is ruled out by generated-history
no-confusion.  The exchanged direction is identical with the zero and one
constructors swapped.
\end{proof}

\begin{theorem}[Continuation unit laws]
\label{thm:continuation-unit-laws-spine}
For every generated history $h$, $\ContR(h,\emp,h)$. Moreover, for every generated history $k$, $\ContR(\emp,k,k)$, and if $\ContR(h,\emp,r)$, then $\HSame(r,h)$.
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_right\_unit\_iff}
\leanvariant{BEDC.FKernel.Cont.cont\_unit\_laws\_spine}
\begin{proof}
The right unit is the base constructor. The left unit is proved by induction on $k$. The empty case is immediate. If $k=\Ezero(k')$, use the induction hypothesis for $k'$ and rebuild with the $\Ezero$ constructor. The $\Eone$ case is analogous. The uniqueness form follows by continuation determinacy against the base constructor.
\end{proof}

\begin{theorem}[Continuation right-unit law endpoint emptiness]
\label{thm:cont-right-unit-law-endpoint-empty-iff}
For every generated history $u$, the universal right-unit result law
$$
\forall d,r:\mathsf{BHist}.\; \ContR(d,u,r)\to \HSame(r,d)
$$
holds if and only if $u\hsame\emp$.
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_right\_unit\_law\_endpoint\_empty\_iff}
\begin{proof}
If the universal law holds, apply it to the left-unit continuation
$\ContR(\emp,u,u)$ to obtain $u\hsame\emp$. Conversely, after the endpoint is
identified with $\emp$, any continuation $\ContR(d,\emp,r)$ is a right-unit
continuation, so right-unit result uniqueness gives $\HSame(r,d)$.
\end{proof}

\begin{theorem}[Continuation left-unit law endpoint emptiness]
\label{thm:cont-left-unit-law-endpoint-empty-iff}
For every generated history $u$, the universal left-unit result law
$$
\forall d,r:\mathsf{BHist}.\; \ContR(u,d,r)\to \HSame(r,d)
$$
holds if and only if $u\hsame\emp$.
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_left\_unit\_law\_endpoint\_empty\_iff}
\begin{proof}
If the universal law holds, apply it to the right-unit continuation
$\ContR(u,\emp,u)$ to obtain $u\hsame\emp$. Conversely, after the endpoint is
identified with $\emp$, any continuation $\ContR(\emp,d,r)$ is a left-unit
continuation, so the left-unit result theorem gives $\HSame(r,d)$.
\end{proof}

\begin{theorem}[Mutual continuation collapses to history-sameness]
\label{thm:cont-mutual-extension-hsame}
If \(\ContR(h,l,k)\) and \(\ContR(k,r,h)\), then \(\HSame(h,k)\).
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_mutual\_extension\_hsame}
\begin{proof}
Compose the two continuation equations into a cycle from $h$ back to $h$ with tail $\mathsf{append}(l,r)$. Right-unit uniqueness forces this composite tail to be empty. Empty append inversion forces $l$ to be empty, so the first continuation reduces to $\HSame(h,k)$.
\end{proof}

\begin{theorem}[Mutual continuation has empty tails]
\label{thm:cont-mutual-extension-tails-empty}
If $\ContR(h,l,k)$ and $\ContR(k,r,h)$, then
$$
\HSame(l,\emp)
\quad\text{and}\quad
\HSame(r,\emp).
$$
\end{theorem}
\begin{proof}
Compose the two continuations into a cycle from $h$ back to $h$ whose tail is
$\mathsf{append}(l,r)$. Right-unit uniqueness forces that composite tail to be
empty, and empty-append inversion identifies both component tails with
$\emp$.
\end{proof}
\leanchecked{BEDC.FKernel.Cont.cont\_mutual\_extension\_tails\_empty}

\begin{theorem}[Mutual continuation forbids visible right tails]
\label{thm:cont-mutual-extension-right-tail-absurd}
If $\ContR(h,l,k)$ holds, then neither $\ContR(k,\Ezero(r),h)$ nor
$\ContR(k,\Eone(r),h)$ can hold.
\end{theorem}
\begin{proof}
Apply the empty-tail theorem to the two mutual continuations. In the
$\Ezero(r)$ branch it would identify $\Ezero(r)$ with $\emp$, contradicting
history constructor no-confusion. The $\Eone(r)$ branch is identical with the
one-constructor no-confusion theorem.
\end{proof}
\leanchecked{BEDC.FKernel.Cont.cont\_mutual\_extension\_right\_tail\_absurd}

\begin{theorem}[Triangular continuation cycle forbids a visible left tail]
\label{thm:cont-triangle-cycle-left-visible-tail-absurd}
If
$$
\ContR(a,\Eone(k),b),\qquad
\ContR(b,g,c),\qquad
\ContR(c,h,a)
$$
all hold, then contradiction follows.
\end{theorem}
\begin{proof}
Compose the first two continuations into
$\ContR(a,\mathsf{append}(\Eone(k),g),c)$. This composite and the return
continuation $\ContR(c,h,a)$ form a mutual continuation cycle, so the composite
tail is empty. Empty-append inversion identifies $\Eone(k)$ with $\emp$,
contradicting history constructor no-confusion.
\end{proof}
\leanchecked{BEDC.FKernel.Cont.cont\_triangle\_cycle\_left\_visible\_tail\_absurd}

\begin{theorem}[Associativity up to history-sameness]
\label{thm:continuation-associativity-up-to-hsame-proof-standing}
Suppose
$$
\ContR(a,b,ab),\quad \ContR(ab,c,left),\quad
\ContR(b,c,bc),\quad \ContR(a,bc,right).
$$
Then \(\HSame(left,right)\).
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_assoc\_up\_to\_hsame\_spine}
\begin{proof}
Use the right-recursive append associativity spine underlying $\ContR$: first identify each continuation proof with its append result, then apply associativity for recursion on the right history $c$. The two displayed composites are therefore the same append tree up to $\HSame$.
\end{proof}

\begin{theorem}[Composite continuation tail uniqueness]
\label{thm:cont-composite-tail-unique}
If
$$
\ContR(h,f,k),\quad \ContR(k,g,r),\quad \ContR(h,tail,r),
$$
then $\HSame(tail,\mathsf{append}(f,g))$.
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_composite\_tail\_unique}
\begin{proof}
Compose the first two continuation witnesses using associativity to obtain a
direct witness from $h$ to $r$ with tail $\mathsf{append}(f,g)$.  Left
cancellation for continuation then identifies this composite tail with the
given direct tail.
\end{proof}

\begin{theorem}[Composite continuation left factor]
\label{thm:cont-composite-left-factor}
If
$$
\ContR(b,g,c),\quad \ContR(f,g,fg),\quad \ContR(a,fg,c),
$$
then $\ContR(a,f,b)$.
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_composite\_left\_factor}
\begin{proof}
The displayed composite continuation $\ContR(a,fg,c)$ and the factor
continuation $\ContR(f,g,fg)$ rewrite $c$ as
$\mathsf{append}(\mathsf{append}(a,f),g)$. Compare this with
$\ContR(b,g,c)$ and cancel the common right suffix $g$; the middle object is
therefore $\hsame$ to $\mathsf{append}(a,f)$, and the reflexive continuation
from $a$ by $f$ transports to $b$.
\end{proof}

\begin{theorem}[Composite continuation right factor]
\label{thm:cont-composite-right-factor}
If
$$
\ContR(a,f,b),\quad \ContR(f,g,fg),\quad \ContR(a,fg,c),
$$
then $\ContR(b,g,c)$.
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_composite\_right\_factor}
\begin{proof}
Invert the three continuation witnesses. The result endpoint $c$ is
$\mathsf{append}(a,\mathsf{append}(f,g))$, while the desired right factor
endpoint is $\mathsf{append}(\mathsf{append}(a,f),g)$. BEDC append
associativity identifies these two histories and supplies the displayed
continuation from $b$ by $g$ to $c$.
\end{proof}

\begin{corollary}[Addition-like behavior requires no primitive addition]
\label{cor:addition-like-behavior-requires-no-primitive-addition}
The continuation relation supplies a finite-kernel seed for addition-like derived interfaces. The additive interface is not primitive; it may be licensed only by a naming certificate whose stability component cites continuation unit and associativity.
\end{corollary}
\leanchecked{BEDC.FKernel.Cont.continuation\_addition\_like\_seed\_has\_unit\_assoc}
