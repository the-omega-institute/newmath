
\chapter{Signature Proof Spine}
\label{ch:proof-standing-signature-proof-spine}

This chapter turns asking events and generated bundles into signature events. No probe is treated as a function.

\begin{definition}[Bundle generation]
Probe bundles are generated by \(\Bnil\) and \(\Bcons(\pi,\Pi)\). They are not imported as external finite lists.
\end{definition}

\begin{definition}[Asking policy]
$\AskPol(\Pi,D)$ is paper notation for using the checked $\mathsf{AskPolicy}(D)$ certificate at bundle $\Pi$.
The certificate gives totality of asking, determinacy of asking up to $\MSame$, and respect for $\HSame$ on histories in $D$.
Its Lean fields quantify over each probe name available to signature generation, so $\Pi$ marks the bundle site where the certificate is used rather than a field that restricts the certificate.
\end{definition}

\begin{definition}[Signature event]
\label{def:proof-standing-signature-event}
\(\Sig(\Pi,h,s,\Delta)\) means that asking bundle \(\Pi\) against history \(h\) generates answer history \(s\) with evidence ledger \(\Delta\). It is generated by the empty-bundle rule and the cons-bundle rule.
\end{definition}
\leanchecked{BEDC.FKernel.Sig.signature\_event\_generation\_pack}

\begin{theorem}[Signature determinacy]
\label{thm:signature-determinacy-proof-spine}
\label{thm:signature-determinacy-spine}
Assume \(\AskPol(\Pi,D)\). If \(h\in D\), \(\Sig(\Pi,h,s,\Delta)\), and \(\Sig(\Pi,h,t,\Theta)\), then \(\HSame(s,t)\).
\end{theorem}
\leanchecked{BEDC.FKernel.Sig.signature\_determinacy\_proof\_spine}
\leanvariant{BEDC.FKernel.Sig.signature\_determinacy\_spine}
\leanvariant{BEDC.FKernel.Sig.sig\_cons\_full\_witness\_determinacy}
\begin{proof}
Induct on the generated bundle \(\Pi\). For the empty bundle, both answer histories are empty. For a cons bundle, determinacy of the first asking event gives internally same output marks, while the induction hypothesis gives internally same tail signatures. The result follows by the matching history constructor.
\end{proof}

\begin{definition}[Signature sameness]
\(\SameSig(\Pi,h,k)\) holds when there exist generated signatures \(s,t\) for \(h,k\) under \(\Pi\) with \(\HSame(s,t)\).
\end{definition}

\begin{theorem}[Signature sameness equivalence]
\label{thm:signature-sameness-equivalence-proof-standing}
Assume \(\AskPol(\Pi,D)\). Then \(\SameSig(\Pi,-,-)\) is reflexive, symmetric, and transitive on histories in \(D\).
\end{theorem}
\leanchecked{BEDC.FKernel.Sig.signature\_sameness\_equivalence\_proof\_spine}
\begin{proof}
For reflexivity, generate a signature \(s\) of \(h\) by totality and use \(\HSame(s,s)\). Symmetry swaps the two witnessing signatures and uses symmetry of \(\HSame\). For transitivity, choose witnesses \(s_h,s_k\) for \(h,k\) and \(t_k,t_l\) for \(k,l\). Since both \(s_k\) and \(t_k\) are signatures of \(k\), signature determinacy gives \(\HSame(s_k,t_k)\). Transitivity of \(\HSame\) gives \(\HSame(s_h,t_l)\).
\end{proof}
