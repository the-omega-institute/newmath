
\chapter{Naming Certificate Proof Instances}
\label{ch:proof-standing-namecert-proof-instances}

This chapter records the first proof-standing instances of the naming protocol. These are still interface seeds; they are not full reconstructions of traditional arithmetic.

\section{The unary repetition certificate}

\begin{definition}[Unary repetition source]
The unary source consists of histories generated from \(\emp\) by repeated \(\Eone\)-extension.
\end{definition}

\begin{definition}[Unary classifier]
Two unary histories are classified together when their generation pattern is internally the same under \(\HSame\). The classifier is intentionally weaker than a full arithmetic equality; it only records the repetition pattern available at this stage.
\end{definition}

\begin{theorem}[Unary repetition closure]
\label{thm:unary-repetition-closure}
The unary source is closed under \(\Eone\)-extension.
\end{theorem}
\leanchecked{BEDC.FKernel.Unary.unary\_e1\_closed}
\begin{proof}
By definition, unary histories are generated by repeated \(\Eone\)-extension. Extending such a history by \(\Eone\) gives another history generated by repeated \(\Eone\)-extension.
\end{proof}

\begin{corollary}[License for \(\NatUp\)]
A name \(\NatUp\) is licensed once its certificate records unary source, unary repetition pattern, unary classifier, closure under \(\Eone\), and the ledger of the underlying histories.
\end{corollary}

\section{The continuation certificate for addition-like behavior}

\begin{definition}[Unary continuation pattern]
The unary continuation pattern is the restriction of \(\ContR\) to unary source histories.
\end{definition}

\begin{theorem}[Unary continuation closure]
\label{thm:unary-continuation-closure-up-to-hsame}
If \(h\) and \(k\) are unary histories and \(\ContR(h,k,r)\), then \(r\) is unary up to \(\HSame\).
\end{theorem}
\leanchecked{BEDC.FKernel.Unary.unary\_continuation\_closure\_up\_to\_hsame}
\begin{proof}
Induct on the generation of \(h\). If \(h=\emp\), then \(r\) is internally same as \(k\), hence unary. If \(h=\Eone(h')\), the continuation rule reduces the result to \(\Eone(r')\) where \(r'\) is obtained by continuing \(h'\) by \(k\). Apply the induction hypothesis and close under \(\Eone\). The \(\Ezero\)-case is absent for unary histories.
\end{proof}

\begin{theorem}[Addition-like unit]
\label{thm:unary-addition-like-unit-spine}
\label{thm:addition-like-unit}
The empty history acts as a left unit for unary continuation, and as a right unit up to \(\HSame\).
\end{theorem}
\leanchecked{BEDC.FKernel.Unary.unary\_addition\_like\_unit\_with\_certificate}
\leanvariant{BEDC.FKernel.Unary.unary\_addition\_like\_unit\_spine}
\leanvariant{BEDC.FKernel.Unary.addition\_like\_unit\_laws}
\begin{proof}
The left unit is the continuation left-unit theorem. The right unit is continuation right unit up to \(\HSame\), restricted to unary histories.
\end{proof}

\begin{theorem}[Addition-like associativity]
\label{thm:addition-like-associativity}
Unary continuation is associative up to \(\HSame\).
\end{theorem}
\leanchecked{BEDC.FKernel.Unary.unary\_cont\_assoc\_with\_closure}
\begin{proof}
Restrict the continuation associativity theorem to unary histories. Unary continuation closure ensures that all intermediate histories remain within the unary source up to \(\HSame\).
\end{proof}

\begin{corollary}[License for \(\AddUp\)]
\label{cor:addup-license-proof-standing}
A name \(\AddUp\) is licensed once its certificate records unary source, unary continuation pattern, descent to unary classes, unit behavior, associativity behavior, and composite ledger policy.
\end{corollary}
\leanchecked{BEDC.FKernel.Unary.add\_up\_licensed\_not\_primitive}

\begin{remark}[Interface seed, not full arithmetic]
The licenses for \(\NatUp\) and \(\AddUp\) are first interface seeds. They do not yet assert a full induction principle, full recursion scheme, commutativity of addition, multiplication, order, or semiring structure.
\end{remark}
