
\chapter{Closed Generation and Internal Sameness}
\label{ch:proof-standing-internal-sameness-no-confusion}

The finite kernel relies on a closed-generation reading: judgments are generated only by their displayed constructors. This is not an extra mathematical object inside the theory; it is the proof discipline governing the kernel rules.

\begin{principle}[Closed generation]
For each primitive judgment family introduced in the finite kernel, the only valid derivations are those generated by its displayed rules. In particular, there are no hidden conversion, coercion, or extensionality rules.
\end{principle}

\begin{definition}[Mark constructors]
The raw marks are generated by two constructors, \(\bzero\) and \(\bone\). The internal mark-sameness judgment \(\MSame(m,n)\) is generated by the two displayed reflexive constructors:
\[
\MSame(\bzero,\bzero),\qquad \MSame(\bone,\bone).
\]
No cross-constructor rule is present.
\end{definition}

\begin{theorem}[Mark no-confusion]
The following judgments are not derivable:
\[
\MSame(\bzero,\bone),\qquad \MSame(\bone,\bzero).
\]
\end{theorem}
\begin{proof}
By closed generation, every derivation of \(\MSame\) must end in one of the two mark-sameness constructors. The first constructor has target \(\MSame(\bzero,\bzero)\), and the second has target \(\MSame(\bone,\bone)\). Neither constructor has target \(\MSame(\bzero,\bone)\) or \(\MSame(\bone,\bzero)\). Hence the cross judgments are not generated.
\end{proof}

\begin{theorem}[\(\MSame\) is an internal equivalence]
The judgment \(\MSame\) is reflexive, symmetric, and transitive on generated marks.
\end{theorem}
\begin{proof}
Reflexivity follows by cases on the two generated marks. Symmetry follows because each generated proof of \(\MSame(m,n)\) has identical endpoints. Transitivity follows by case analysis on the first proof and mark no-confusion on the second.
\end{proof}

\begin{definition}[History constructors]
The generated histories are built from an empty history \(\emp\) and two extension constructors \(\Ezero\) and \(\Eone\). The internal history-sameness judgment \(\HSame(h,k)\) is generated structurally from matching constructors.
\end{definition}

\begin{theorem}[History no-confusion]
\label{thm:history-no-confusion}
The following judgments are not derivable:
\[
\HSame(\emp,\Ezero(h)),\quad
\HSame(\emp,\Eone(h)),\quad
\HSame(\Ezero(h),\Eone(k)),\quad
\HSame(\Eone(h),\Ezero(k)).
\]
\end{theorem}
\begin{proof}
By closed generation, a derivation of \(\HSame\) must follow the displayed structural constructors. Empty histories can only be matched with empty histories. A history beginning with \(\Ezero\) can only be matched with a history beginning with \(\Ezero\), and similarly for \(\Eone\). There is no displayed constructor crossing \(\Ezero\) with \(\Eone\), nor a constructor crossing either extension with \(\emp\). Therefore the listed judgments are not generated.
\end{proof}

\begin{theorem}[\(\HSame\) is an internal equivalence]
The judgment \(\HSame\) is reflexive, symmetric, and transitive on generated histories.
\end{theorem}
\begin{proof}
Reflexivity is proved by induction on generated histories. Symmetry is proved by induction on the displayed derivation of \(\HSame\). Transitivity is proved by induction on the first derivation, using history no-confusion to invert the second derivation into the matching constructor case.
\end{proof}

\begin{corollary}[Constructor inversion]
\label{cor:constructor-inversion}
If \(\HSame(\Ezero(h),x)\) is derivable, then \(x\) is generated through an \(\Ezero\)-constructor up to the displayed structural form. If \(\HSame(\Eone(h),x)\) is derivable, then \(x\) is generated through an \(\Eone\)-constructor up to the displayed structural form.
\end{corollary}
