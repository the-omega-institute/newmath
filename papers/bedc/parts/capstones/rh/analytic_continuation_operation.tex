\chapter{Analytic Continuation as an Operation on \(\HolomorphicUp\)}
\label{ch:concrete-instances-analytic-continuation}

This chapter licenses analytic continuation as the operation \(\AnalyticContUp\) that, given a holomorphic \(f\) on a domain \(D_1\) and an overlapping holomorphic \(g\) on \(D_2\) with \(f = g\) on a non-trivial overlap region \(U\), produces the unique holomorphic extension \(h\) on \(D_1 \cup D_2\). The chapter is an OPERATION chapter rather than a pure namecert: the structure is input/output classifier-aware predicate plus operation laws (functoriality, composition, \(\hsame\)-respect) instead of the standard 5-field NameCert. Builds on \(\HolomorphicUp\) (\autoref{ch:concrete-instances-holomorphic-namecert}) and the constructive identity theorem of that chapter. Branch obstruction (the constructive face of monodromy / multi-valued continuation around branch points) is flagged as a SCHEMA-ONLY HORIZON: the chapter exposes the constructive interface but admits that closed witnesses for general branch behaviour do not in general live inside the framework. The construction is fully constructive on the parts that close; uses no \(\mathsf{Classical.choice}\), no \(\mathsf{Quot.sound}\), and no \(\mathsf{propext}\), so the operation's Lean targets fall under the project's axiom-purity gate.

\section{Domain compatibility}

\begin{definition}[Domain compatibility]
\label{def:dom-compat}
For two open domains \(D_1, D_2\) of \(\ComplexUp\), the predicate \(\mathsf{DomCompat}(D_1, D_2, U)\) is satisfied by an explicit overlap region \(U\) which contains an \(\mathsf{OpenDisk}(z_c, r_c)\) inside \(D_1 \cap D_2\) together with a quantitative density bound at a center point \(z_c\) (i.e.\ a sequence of points \((z_n)_n\) clustering at \(z_c\) inside \(U\) with \(|z_n - z_c| \leq 2^{-n}\)). The density witness is required for the constructive identity theorem (\autoref{ch:concrete-instances-holomorphic-namecert}) to apply.
\end{definition}

\begin{lemma}[Domain compatibility witness]
\label{lem:dom-compat-witness}
The predicate \(\mathsf{DomCompat}\) is decidable when \(D_1, D_2\) are concrete unions of open disks with rational centers and radii.
\end{lemma}
\begin{proof}
For unions of open disks, intersection is decidable from radii comparisons; the inner disk witness is obtained explicitly.
\end{proof}

\section{The continuation predicate}

\begin{definition}[Analytic continuation predicate]
\label{def:ana-cont}
For \(f\) holomorphic on \(D_1\), \(g\) holomorphic on \(D_2\), with \(\mathsf{DomCompat}(D_1, D_2, U)\), the predicate \(\mathsf{AnaCont}(f, D_1, g, D_2, h)\) holds when \(h\) is a holomorphic function on \(D_1 \cup D_2\) such that:
\begin{enumerate}
  \item \(h\) restricted to \(D_1\) equals \(f\) under \(\hsame\) pointwise;
  \item \(h\) restricted to \(D_2\) equals \(g\) under \(\hsame\) pointwise;
  \item \(f\) and \(g\) agree on the overlap region \(U\) under \(\hsame\) pointwise (a precondition for existence).
\end{enumerate}
\end{definition}

\begin{theorem}[Continuation is unique]
\label{thm:ana-cont-unique}
If \(\mathsf{AnaCont}(f, D_1, g, D_2, h)\) and \(\mathsf{AnaCont}(f, D_1, g, D_2, h')\) both hold, then \(h = h'\) under \(\hsame\) pointwise on \(D_1 \cup D_2\).
\end{theorem}
\begin{proof}[Sketch]
Constructive identity theorem (\autoref{ch:concrete-instances-holomorphic-namecert}) applied at the cluster sequence in \(U\): both \(h\) and \(h'\) agree with \(f\) and \(g\) on dense subsets, hence on the union.
\end{proof}

\section{Chain continuation}

\begin{theorem}[Chain continuation]
\label{thm:chain-continuation}
If \(\mathsf{AnaCont}(f_1, D_1, f_2, D_2, h_{12})\) and \(\mathsf{AnaCont}(h_{12}, D_1 \cup D_2, f_3, D_3, h_{123})\) hold (with \(\mathsf{DomCompat}(D_1 \cup D_2, D_3, U_{23})\)), then composing the two continuations gives a single \(\mathsf{AnaCont}\) witness on \(D_1 \cup D_2 \cup D_3\) with explicit modulus inheritance.
\end{theorem}
\begin{proof}[Sketch]
Direct composition of holomorphic restrictions; uniqueness from \autoref{thm:ana-cont-unique} applied at each step.
\end{proof}

\section{Branch obstruction (schema-only horizon)}

\begin{definition}[Branch obstruction]
\label{def:branch-obstruction-schema}
When the chain continuation closes a loop around a topological obstruction (e.g.\ continuing \(\sqrt{z}\) along a circle around the origin), the resulting two extensions through different paths can disagree. The branch obstruction predicate \(\mathsf{BranchObstruction}\) records this disagreement as a SCHEMA-ONLY structure: it codifies the constructive interface (input: two chain continuations; output: a quantitative non-agreement witness) without producing closed witnesses for general topological branches inside the BEDC framework.
\end{definition}

\begin{remark}[Schema-only nature of branch obstruction]
\label{rem:branch-obstruction-horizon}
The branch obstruction interface is intentionally a SCHEMA-ONLY HORIZON: BEDC can express the obstruction predicate, can produce concrete witnesses on specific examples (\(\sqrt{z}\), \(\log z\) around 0), but cannot in general decide whether a given chain configuration produces obstruction witnesses. This is consistent with the BEDC project's general schema-only horizon protocol --- the chapter exposes the interface so future framework extensions can codify the closed witnesses, while the current chapter holds the schema open.
\end{remark}

\section{Concrete continuations}

\begin{theorem}[Schwartz reflection]
\label{thm:schwartz-reflection}
If \(f\) is holomorphic on the upper half-plane \(\{z : \Im(z) > 0\}\) and continuous on the closure including a real boundary segment, with real values on the real boundary, then the function \(\tilde{f}(z) = \overline{f(\bar{z})}\) defines a holomorphic continuation across the real boundary into the lower half-plane. The continuation is given as an explicit function with explicit modulus.
\end{theorem}
\begin{proof}[Sketch]
Direct verification using the Cauchy-Riemann equations (\autoref{ch:concrete-instances-complex-diff-namecert}) and the symmetry of the boundary trace.
\end{proof}

\section{Operation laws}

\begin{lemma}[Functoriality]
\label{lem:ana-cont-functorial}
\(\mathsf{AnaCont}\) respects \(\hsame\) on inputs: if \(\hsame\) on \(f, f'\) and on \(g, g'\) (pointwise on the respective domains), then \(\mathsf{AnaCont}(f, D_1, g, D_2, h)\) implies \(\mathsf{AnaCont}(f', D_1, g', D_2, h)\) for the same \(h\) up to \(\hsame\) pointwise.
\end{lemma}
\begin{proof}
Direct transport of equality witnesses under \(\hsame\).
\end{proof}

\section{The operation certificate}

\begin{definition}[Analytic continuation operation certificate]
\label{def:ana-cont-op-cert}
The operation certificate consists of: (i) input spec --- pairs \((f, D_1)\) and \((g, D_2)\) with overlap witness \(\mathsf{DomCompat}(D_1, D_2, U)\); (ii) output spec --- holomorphic \(h\) on \(D_1 \cup D_2\) restricting compatibly; (iii) classifier --- \(\hsame\) pointwise on the union; (iv) operation laws --- functoriality, chain composition, schwartz reflection as a concrete instance, and branch obstruction as the schema-only horizon; (v) ledger policy --- uniqueness and existence under domain compatibility.
\end{definition}

\begin{theorem}[Concrete \(\AnalyticContUp\) operation certificate]
\label{thm:analytic-continuation-operation}
The data above forms an operation certificate on \(\HolomorphicUp\) for the analytic-continuation interface \(\AnalyticContUp\).
\end{theorem}
\begin{proof}[Sketch]
Each operation law is verified by direct construction in the preceding sections; the schema-only branch obstruction is exposed as a horizon, not closed. The certificate uses no \(\mathsf{Classical.choice}\), no \(\mathsf{Quot.sound}\), and no \(\mathsf{propext}\), so the operation's Lean targets fall under the project's axiom-purity gate.
\end{proof}

\begin{theorem}[Semantic \(\AnalyticContUp\) operation certificate]

\begin{lemma}[Domain compatibility is symmetric]
\label{lem:dom-compat-symmetric}
If $\mathsf{DomCompat}(D_1,D_2,U)$ holds, then $\mathsf{DomCompat}(D_2,D_1,U)$ holds for the same overlap region $U$.
\end{lemma}
\begin{proof}
Unpack \autoref{def:dom-compat}. The displayed witness consists of an open disk $\mathsf{OpenDisk}(z_c,r_c)$ contained in $D_1\cap D_2$ and a quantitative sequence in $U$ clustering at $z_c$ with the stated unary bound. The same center, radius, open-disk witness, overlap region, sequence, and density estimates witness compatibility in the reversed order. Only the containment record changes: membership in $D_1\cap D_2$ is transported to membership in $D_2\cap D_1$ by commutativity of intersection.
\end{proof}

\begin{theorem}[Analytic continuation symmetry]
\label{thm:ana-cont-symmetric}
For any overlap region $U$,
$$
\mathsf{DomCompat}(D_1,D_2,U)\land\mathsf{AnaCont}(f,D_1,g,D_2,h)
\Longrightarrow
\mathsf{DomCompat}(D_2,D_1,U)\land\mathsf{AnaCont}(g,D_2,f,D_1,h).
$$
\end{theorem}
\begin{proof}
The first conclusion is \autoref{lem:dom-compat-symmetric}. For the continuation conclusion, unfold \autoref{def:ana-cont}. The holomorphic carrier for $h$ is unchanged because $D_1\cup D_2$ and $D_2\cup D_1$ are the same domain by commutativity of union. The restriction of $h$ to $D_2$ is the second restriction clause of the given continuation witness, and becomes the first restriction clause after the swap. The restriction of $h$ to $D_1$ is the first restriction clause of the given witness, and becomes the second restriction clause after the swap. On the common overlap $U$, the given pointwise agreement of $f$ with $g$ transports to pointwise agreement of $g$ with $f$ by symmetry of the classifier $\hsame$. These are exactly the three clauses required for $\mathsf{AnaCont}(g,D_2,f,D_1,h)$ under the reversed compatibility witness.
\end{proof}

\begin{theorem}[Analytic continuation triple-chain associativity]
\label{thm:ana-cont-triple-chain-associativity}
Let $D_1,D_2,D_3$ be open domains of $\ComplexUp$. Suppose
$$
\mathsf{AnaCont}(f_1,D_1,f_2,D_2,h_{12})\land
\mathsf{AnaCont}(f_2,D_2,f_3,D_3,h_{23})\land
\mathsf{AnaCont}(h_{12},D_1\cup D_2,f_3,D_3,h_{(12)3}).
$$
Then there is a holomorphic carrier $h_{1(23)}$ such that
$$
\mathsf{AnaCont}(f_1,D_1,h_{23},D_2\cup D_3,h_{1(23)})
$$
and
$$
\hsame(h_{(12)3},h_{1(23)})
$$
pointwise on $D_1\cup D_2\cup D_3$.
\end{theorem}
\begin{proof}
Unpack the three continuation witnesses using \autoref{def:ana-cont}. The first witness says that $h_{12}$ restricts to $f_1$ on $D_1$ and to $f_2$ on $D_2$. The second says that $h_{23}$ restricts to $f_2$ on $D_2$ and to $f_3$ on $D_3$. The third says that $h_{(12)3}$ restricts to $h_{12}$ on $D_1\cup D_2$ and to $f_3$ on $D_3$.

The domain-compatibility witness for $D_1$ with $D_2\cup D_3$ is obtained from the first overlap witness. Its open disk, center, radius, and cluster sequence are unchanged; the containment clause is transported from $U_{12}\subseteq D_1\cap D_2$ to $U_{12}\subseteq D_1\cap(D_2\cup D_3)$ by the canonical inclusion $D_2\subseteq D_2\cup D_3$. This is exactly the data required by \autoref{def:dom-compat}, equivalently by the topological form in \autoref{def:cplx-topo-domain-compat}.

Apply \autoref{thm:chain-continuation} to the first two links, with the union parenthesized as $D_1\cup(D_2\cup D_3)$. This gives a holomorphic carrier $h_{1(23)}$ whose restrictions are $f_1$ on $D_1$ and $h_{23}$ on $D_2\cup D_3$, hence
$$
\mathsf{AnaCont}(f_1,D_1,h_{23},D_2\cup D_3,h_{1(23)}).
$$

It remains to identify this carrier with the left-grouped carrier. The same restriction clauses show that $h_{(12)3}$ is also a continuation of $f_1$ against $h_{23}$ over $D_1$ and $D_2\cup D_3$. On $D_1$, $h_{(12)3}$ restricts to $h_{12}$ and $h_{12}$ restricts to $f_1$. On $D_2$, both $h_{(12)3}$ and $h_{23}$ restrict to $f_2$: the former through $h_{12}$, the latter by the second continuation witness. On $D_3$, both restrict to $f_3$. Transitivity and symmetry of the classifier $\hsame$ combine these pointwise restriction witnesses. The overlap agreement is the inherited $D_1$-$D_2$ agreement, transported along $D_2\subseteq D_2\cup D_3$.

Thus $h_{(12)3}$ and $h_{1(23)}$ are two witnesses for the same predicate
$$
\mathsf{AnaCont}(f_1,D_1,h_{23},D_2\cup D_3,-).
$$
By uniqueness of analytic continuation, \autoref{thm:ana-cont-unique}, they are classified by $\hsame$ pointwise on $D_1\cup(D_2\cup D_3)$. Transporting along associativity of union identifies this domain with $D_1\cup D_2\cup D_3$.
\end{proof}

\begin{theorem}[Analytic continuation identity law]
\label{thm:ana-cont-identity-law}
Let $D$ be an open domain of $\ComplexUp$, let $f$ be holomorphic on $D$, and let $U$ be an overlap region with $\mathsf{DomCompat}(D,D,U)$. Suppose the compatibility witness is non-trivial in the sense of \autoref{def:dom-compat}: its displayed open disk and quantitative cluster sequence lie inside the overlap region contained in $D\cap D$. Then
$$
\mathsf{AnaCont}(f,D,f,D,f)
$$
holds. Moreover, for every $h$,
$$
\mathsf{AnaCont}(f,D,f,D,h)\Longrightarrow
\forall z,\ z\in D\Longrightarrow \hsame(h(z),f(z)).
$$
Equivalently, $h$ and $f$ are pointwise $\hsame$ on $D\cup D$, transported along the canonical idempotence equivalence $D\cup D=D$.
\end{theorem}
\begin{proof}
Unpack \autoref{def:ana-cont}. The output domain for the pair $(D,D)$ is $D\cup D$. This domain is constructively equivalent to $D$: membership in $D$ maps into $D\cup D$ by the left inclusion, and membership in $D\cup D$ returns membership in $D$ by case analysis on the two identical summands. Thus the given holomorphic carrier of $f$ on $D$ also presents $f$ as a holomorphic carrier on $D\cup D$.

The first restriction clause is the restriction of $f$ from $D\cup D$ to the first copy of $D$, which is pointwise $\hsame$ to $f$ by reflexivity of the classifier. The second restriction clause is identical for the second copy of $D$. The overlap clause asks that the two input functions, both equal to $f$, agree on $U$; this is again pointwise reflexivity of $\hsame$. The non-trivial disk and cluster data in the compatibility witness supply exactly the admissible overlap record required by \autoref{def:dom-compat}. These three clauses give $\mathsf{AnaCont}(f,D,f,D,f)$.

Now assume $\mathsf{AnaCont}(f,D,f,D,h)$. Comparing this witness with the identity witness just constructed, \autoref{thm:ana-cont-unique} gives pointwise $\hsame(h,f)$ on $D\cup D$. Transporting this classifier through the same idempotence equivalence $D\cup D=D$ yields, for every $z\in D$, the pointwise comparison $\hsame(h(z),f(z))$.
\end{proof}
