\chapter{A Concrete Naming Certificate for \(\ZCarryUp\)}
\label{ch:concrete-instances-zeckendorf-carry-classifier}

This chapter introduces the Zeckendorf carry classifier \(\mathsf{ZCarry}(h, k)\), generated by the local rewrite \(\mathtt{word\_011} \to \mathtt{word\_100}\) (cf.~Definition~\ref{def:axiszeckendorf-display-words}). The crucial property of this classifier is that it is NOT history-sameness: \(\mathsf{ZCarry}(h, k)\) does not entail \(\HSame(h, k)\). The Fibonacci value-preservation reading is a horizon target.

\section{The carry classifier}

\begin{definition}[Zeckendorf carry classifier]
\label{def:zeckendorf-carry-classifier}
The classifier \(\mathsf{ZCarry}(h, k)\) is generated by the single base rule
$$
  \mathsf{ZCarry}(\mathtt{word\_011}, \mathtt{word\_100}).
$$
\end{definition}
\leandef{BEDC.Derived.AxisZeckendorf.Carry.ZCarry}

\begin{lemma}[The display carry is generated]
\label{lem:zcarry-011-100}
\(\mathsf{ZCarry}(\mathtt{word\_011}, \mathtt{word\_100})\) holds.
\end{lemma}
\leandef{BEDC.Derived.AxisZeckendorf.Carry.zCarry\_011\_100}
\begin{proof}
First (and only) generator of \(\mathsf{ZCarry}\).
\end{proof}

\section{Carry is not history-sameness}

\begin{theorem}[\(\mathtt{word\_011}\) and \(\mathtt{word\_100}\) are not \(\HSame\)]
\label{thm:zcarry-011-100-not-hsame}
\(\HSame(\mathtt{word\_011}, \mathtt{word\_100})\) does not hold.
\end{theorem}
\leandef{BEDC.Derived.AxisZeckendorf.Carry.zCarry\_011\_100\_not\_hsame}
\begin{proof}
Recall that \(\HSame\) is propositional history identity. The first history has top constructor \(\Eone\); the second has top constructor \(\Ezero\). The two top constructors are distinct, so the histories are unequal, and \(\HSame\) does not hold.
\end{proof}

\begin{theorem}[Carry is not history-sameness]
\label{thm:zcarry-not-hsame}
For every \(h\) and \(k\), \(\mathsf{ZCarry}(h, k)\) implies \(\neg \HSame(h, k)\).
\end{theorem}
\leandef{BEDC.Derived.AxisZeckendorf.Carry.zCarry\_not\_hsame}
\begin{proof}
By the unique generator, any \(\mathsf{ZCarry}(h, k)\) instance has \(h = \mathtt{word\_011}\) and \(k = \mathtt{word\_100}\). The conclusion follows from Theorem~\ref{thm:zcarry-011-100-not-hsame}.
\end{proof}

\section{Carry transports normal-form status}

\begin{theorem}[Carry target is normal]
\label{thm:zcarry-target-normal}
For every \(h, k\), if \(\mathsf{ZCarry}(h, k)\), then \(\mathsf{ZNormal}(k)\).
\end{theorem}
\leandef{BEDC.Derived.AxisZeckendorf.Carry.zCarry\_target\_normal}
\begin{proof}
The unique generator pins \(k = \mathtt{word\_100}\); apply Theorem~\ref{thm:znormal-word-100}.
\end{proof}

\begin{theorem}[Carry source is non-normal]
\label{thm:zcarry-source-not-normal}
For every \(h, k\), if \(\mathsf{ZCarry}(h, k)\), then \(\neg \mathsf{ZNormal}(h)\).
\end{theorem}
\leandef{BEDC.Derived.AxisZeckendorf.Carry.zCarry\_source\_not\_normal}
\begin{proof}
The unique generator pins \(h = \mathtt{word\_011}\); apply Theorem~\ref{thm:znormal-word-011-absurd}.
\end{proof}

\section{The certificate}

\begin{definition}[Carry source specification]
\label{def:zcarry-source-specification}
The source specification \(\mathsf{ZCarrySource}\) consists of histories \(h\) with \(\neg \mathsf{ZNormal}(h)\).
\end{definition}
\leandef{BEDC.Derived.AxisZeckendorf.Carry.ZCarrySourceSpec}

\begin{definition}[Carry pattern specification]
\label{def:zcarry-pattern-specification}
The pattern specification \(\mathsf{ZCarryPattern}\) is the carry generator \(\mathsf{ZCarry}(h, k)\).
\end{definition}
\leandef{BEDC.Derived.AxisZeckendorf.Carry.ZCarryPatternSpec}

\begin{definition}[Carry classifier specification]
\label{def:zcarry-classifier-specification}
The classifier specification \(\mathsf{ZCarryClassifier}\) records, on each carry pair, that the target is normal, the source is non-normal, and the source is not \(\HSame\) to the target.
\end{definition}
\leandef{BEDC.Derived.AxisZeckendorf.Carry.ZCarryClassifierSpec}

\begin{definition}[Carry stability certificate]
\label{def:zcarry-stability-certificate}
The stability certificate \(\mathsf{ZCarryStability}\) consists of:
\begin{enumerate}
  \item carry-not-\(\HSame\) (Theorem~\ref{thm:zcarry-not-hsame});
  \item target normal (Theorem~\ref{thm:zcarry-target-normal});
  \item source non-normal (Theorem~\ref{thm:zcarry-source-not-normal});
  \item Fibonacci value preservation, horizon obligation.
\end{enumerate}
\end{definition}
\leandef{BEDC.Derived.AxisZeckendorf.Carry.ZCarryStabilityCert}
\leanvariant{BEDC.Derived.AxisZeckendorf.Carry.zCarryStabilityCert}

\begin{definition}[Carry ledger policy]
\label{def:zcarry-ledger-policy}
The ledger policy \(\mathsf{ZCarryLedger}\) records the non-normal preimage and the carry witness behind any visible carry-classification.
\end{definition}
\leandef{BEDC.Derived.AxisZeckendorf.Carry.ZCarryLedgerPolicy}

\begin{theorem}[Concrete \(\ZCarryUp\) naming certificate]
\label{thm:concrete-zcarry-namecert}
The tuple
$$
  \mathsf{NameCert}_{\ZCarryUp}
  :=
  (\mathsf{ZCarrySource},
   \mathsf{ZCarryPattern},
   \mathsf{ZCarryClassifier},
   \mathsf{ZCarryStability},
   \mathsf{ZCarryLedger})
$$
forms a valid naming certificate for the derived classifier interface \(\ZCarryUp\).
\end{theorem}
\leandef{BEDC.Derived.AxisZeckendorf.Carry.zCarry\_namecert}
\begin{proof}
The five fields are supplied by the preceding definitions. The stability field includes three checked theorems and one explicit horizon obligation.
\end{proof}

\begin{corollary}[\(\ZCarryUp\) is licensed, not primitive]
\label{cor:zcarry-licensed-not-primitive}
The interface \(\ZCarryUp\) is licensed by \(\mathsf{NameCert}_{\ZCarryUp}\). It is a classifier, not a kernel equality, and it is not a primitive object of the finite kernel.
\end{corollary}

\begin{remark}[Carry must remain a classifier]
The crucial discipline of this chapter is that \(\mathsf{ZCarry}\) is a classifier between non-\(\HSame\) histories. Any future extension that promotes the carry to a kernel-level identification—either by replacing \(\HSame\) on \(\mathsf{ZNormal}\) histories or by inserting a global rewrite into the kernel—conflicts with Theorem~\ref{thm:zcarry-not-hsame} and is rejected by the bridge discipline.
\end{remark}
