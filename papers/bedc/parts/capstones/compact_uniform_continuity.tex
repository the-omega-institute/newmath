\chapter{Compactness Implies Uniform Continuity}
\label{ch:capstones-compact-uniform-continuity}

A continuous function on a compact metric space is uniformly continuous, constructively. This chapter assembles the components already isolated in the kernel: the metric carrier with its located distance (\autoref{ch:concrete-instances-metric-namecert}), the pointwise modulus chain encoded by $\Cont$ on continuous naming certificates (\autoref{ch:concrete-instances-continuous-namecert}), and the compactness witness packaged as subset $+$ located $+$ finite $\varepsilon$-net $+$ intermediate $+$ compact (\autoref{ch:concrete-instances-compact-namecert}). The argument identifies the compactness data as a finite cover indexed by an $\varepsilon$-net, takes the minimum over the finitely many pointwise moduli supplied by $\Cont$, and reads off a uniform modulus depending only on $\varepsilon$. The main theorem states this in BEDC form: from a continuous naming certificate together with a compact naming certificate on the same metric carrier, the kernel produces a uniform-continuity modulus, with no appeal to $\mathsf{Classical.choice}$ and no excluded middle.

\section{Uniform modulus from located finite covers}
\label{sec:compact-uniform-continuity-uniform-modulus}

\begin{definition}[Pointwise modulus on $\NameCert_{\ContinuousFunctionUp}$]
\label{def:compact-uniform-pointwise-modulus}
Let $X, Y$ be metric naming certificates and let $\NameCert_f$ be a continuous-function naming certificate witnessing $f \colon X \to Y$. The \emph{pointwise modulus} extracted from $\NameCert_f$ is the function $\delta_f \colon X \times \mathbb{Q}_{>0} \to \mathbb{Q}_{>0}$ supplied by the certificate's $\Cont$ chain: for each base point $x$ and each rational tolerance $\varepsilon$, $\delta_f(x, \varepsilon)$ is a positive rational such that every $x'$ with $d_X(x, x') < \delta_f(x, \varepsilon)$ produces $d_Y(f\,x, f\,x') < \varepsilon$.
\end{definition}

\begin{definition}[Uniform modulus]
\label{def:compact-uniform-uniform-modulus}
A \emph{uniform modulus} for $f \colon X \to Y$ is a function $\delta_f^{\mathrm{u}} \colon \mathbb{Q}_{>0} \to \mathbb{Q}_{>0}$, depending only on the tolerance, such that for every $x, x' \in X$ with $d_X(x, x') < \delta_f^{\mathrm{u}}(\varepsilon)$, $d_Y(f\,x, f\,x') < \varepsilon$.
\end{definition}

\begin{definition}[Located finite $\varepsilon$-net]
\label{def:compact-uniform-finite-net}
A \emph{located finite $\varepsilon$-net} for a metric carrier $X$ is a finite list $N_\varepsilon = [x_1, \ldots, x_n]$ of base points together with a witness that for every $x \in X$ there exists $i \le n$ with $d_X(x, x_i) < \varepsilon$. The compact certificate $\NameCert_{\CompactUp}$ supplies $N_\varepsilon$ as a kernel-level operation indexed by $\varepsilon \in \mathbb{Q}_{>0}$.
\end{definition}

\section{The kernel construction}
\label{sec:compact-uniform-continuity-construction}

\begin{lemma}[Subordinate finite modulus cover]
\label{lem:compact-uniform-subordinate-modulus-cover}
Let $X$ be a compact metric carrier and let $f \colon X \to Y$ have pointwise modulus $\delta_f$. For every rational tolerance $\varepsilon > 0$, the compact certificate supplies a finite nonempty family
$$
(x_1,\rho_1), \ldots, (x_n,\rho_n)
$$
such that each $\rho_i$ is positive, $\rho_i \le \delta_f(x_i,\varepsilon/3)$, and every $x \in X$ has an index $i$ with $d_X(x,x_i) < \rho_i/2$.
\end{lemma}

\begin{proof}[Certificate reading]
Apply compactness to the located positive-radius family whose member at $x$ is the pointwise-modulus ball of radius $\delta_f(x,\varepsilon/3)$. The finite-cover certificate returns centers and radii already subordinate to those balls, so the inequalities $\rho_i \le \delta_f(x_i,\varepsilon/3)$ are part of the returned data.

The located half-radius clause is the cover witness used by the kernel construction below: for each input point $x$, it chooses one of the named centers with $d_X(x,x_i) < \rho_i/2$. No scale depending on $\delta_f^{\mathrm{u}}(\varepsilon)$ is used to obtain this family.
\end{proof}

\begin{theorem}[Capstone: finite BEDC minimum supports compact uniform modulus]
\label{thm:compact-uniform-finite-minimum-certificate}
Let $X$ and $Y$ be metric BEDC carriers whose distance rows are supplied by $\mathsf{MetricDistanceWitness}$, and let a BEDC history carry
$$
\mathsf{CompactWitnessCarrier}(\mathsf{subset},\mathsf{located},\mathsf{finite},\mathsf{intermediate},\mathsf{compact})
$$
for $X$. Let another history carry
$$
\mathsf{ContinuousFunctionCarrier}(\mathsf{source},\mathsf{map},\mathsf{target},\mathsf{modulus},\mathsf{cert})
$$
for a graph $f \colon X \to Y$ over the same source metric. For each rational $\varepsilon>0$, suppose \autoref{lem:compact-uniform-subordinate-modulus-cover} returns a nonempty $\mathsf{ListCarrier}$ spine
$$
R_\varepsilon=[\rho_1,\ldots,\rho_n]
$$
of positive $\RatUp$ radius histories, together with centers $x_i$ and the subordinate half-radius cover rows. Then the finite-minimum selector of \autoref{thm:totalorder-prefix-finite-minimum-selector}, applied to the $\RatUp$ order readback of $R_\varepsilon$, produces a unary history $\mu_\varepsilon$ such that $\mu_\varepsilon$ is the classifier-stable minimum of the radius ledger and the function $\varepsilon\mapsto \mu_\varepsilon/2$ satisfies \autoref{def:compact-uniform-uniform-modulus} for the same continuous graph.
\end{theorem}

\begin{proof}[Certificate assembly]
Read the finite family from the compact witness: the $\mathsf{finite}$ and $\mathsf{located}$ rows give a nonempty list of centers, positive $\RatUp$ radius histories, and for every point of $X$ a ledger position whose center lies within the displayed half-radius.

Transport the radius histories through the $\RatUp$ total-order readback. Since the list is nonempty and every entry is carried by the rational positive-radius certificate, \autoref{thm:totalorder-prefix-finite-minimum-selector} selects a ledger entry $\mu_\varepsilon$ below every radius entry. The minimum-uniqueness row of \autoref{def:totalorder-prefix-finite-minimum-witness} makes this selection stable under the classifier, so the selected lower bound is a BEDC history rather than a host-level choice.

At each selected center, the $\mathsf{ContinuousFunctionCarrier}$ modulus row reads back the local implication from the graph certificate: radius below $\delta_f(x_i,\varepsilon/3)$ gives output distance below $\varepsilon/3$. The subordinate-cover rows guarantee that the same center can be used for any input point, and the selected minimum guarantees that the proposed uniform radius is below the chosen local radius.

Finally apply the metric semantic certificate: symmetry converts $d_X(x,x')$ to the orientation needed at the center, and the triangle row combines $d_X(x,x_i)$ with $d_X(x,x')$ to put $x'$ inside the same local ball. Applying the target-metric triangle row to $f(x)$, $f(x_i)$, and $f(x')$ yields the displayed uniform estimate. Thus the map $\varepsilon\mapsto\mu_\varepsilon/2$ is the uniform modulus certified by the finite BEDC minimum ledger.
\end{proof}

\begin{theorem}[Continuous on compact yields uniform modulus]
\label{thm:compact-uniform-continuity}
Let $X$ be a metric carrier with a compact naming certificate $\NameCert_X$ and let $f \colon X \to Y$ be witnessed by a continuous naming certificate $\NameCert_f$ over $X$. Then the kernel produces a uniform modulus $\delta_f^{\mathrm{u}}$ for $f$, expressible as
$$
\delta_f^{\mathrm{u}}(\varepsilon) \;=\; \min_{1 \le i \le n} \rho_i/2,
$$
where $(x_1,\rho_1), \ldots, (x_n,\rho_n)$ is the finite subordinate family supplied by \autoref{lem:compact-uniform-subordinate-modulus-cover} for $\varepsilon$, and $\delta_f$ is the pointwise modulus of \autoref{def:compact-uniform-pointwise-modulus}.
\end{theorem}

\begin{proof}[Sketch]
Fix $\varepsilon > 0$. Invoke \autoref{lem:compact-uniform-subordinate-modulus-cover} to obtain a finite nonempty family $(x_i,\rho_i)_{i=1}^n$ with $\rho_i \le \delta_f(x_i,\varepsilon/3)$ and the half-radius covering property. Then \autoref{thm:compact-uniform-finite-minimum-certificate} supplies the classifier-stable finite minimum ledger whose readback is the positive rational
$$
\delta_f^{\mathrm{u}}(\varepsilon)=\min_i \rho_i/2.
$$

Let $x,x' \in X$ satisfy $d_X(x,x') < \delta_f^{\mathrm{u}}(\varepsilon)$. The subordinate cover gives an index $i$ with $d_X(x,x_i) < \rho_i/2$. Because $\delta_f^{\mathrm{u}}(\varepsilon) \le \rho_i/2$, the triangle inequality gives
$$
d_X(x',x_i) \le d_X(x',x)+d_X(x,x_i) < \rho_i.
$$
Also $d_X(x,x_i) < \rho_i/2 < \rho_i$.

The bound $\rho_i \le \delta_f(x_i,\varepsilon/3)$ lets the pointwise modulus at $x_i$ apply to both $x$ and $x'$. Hence $d_Y(f\,x_i,f\,x) < \varepsilon/3$ and $d_Y(f\,x_i,f\,x') < \varepsilon/3$. The triangle inequality on $Y$ gives $d_Y(f\,x,f\,x') < \varepsilon$. The construction uses only the finite subordinate cover, the finite BEDC minimum certificate, and rational arithmetic.
\end{proof}

\section{Axiom-purity of the uniform-modulus construction}
\label{sec:compact-uniform-continuity-axiom-purity}

\begin{remark}[No appeal to non-constructive principles]
\label{rem:compact-uniform-axiom-purity}
The construction in \autoref{thm:compact-uniform-continuity} uses only primitive recursion on the finite net (a $\Cont$-shaped chain on a closed inductive list of base points), arithmetic on $\mathbb{Q}_{>0}$ (rational addition, multiplication, and division by $2$, all kernel-level operations on $\NameCert_{\RatUp}$ from \autoref{ch:concrete-instances-rat-namecert}), and the located-distance comparison built into $\NameCert_{\MetricSpaceUp}$. None of these operations passes through $\mathsf{Classical.choice}$, $\mathsf{Quot.sound}$, or $\mathsf{propext}$. The chapter therefore falls under the same axiom-purity gate as the rest of the manuscript.
\end{remark}

\begin{remark}[Heine--Cantor as a kernel theorem]
\label{rem:compact-uniform-heine-cantor}
\autoref{thm:compact-uniform-continuity} is the BEDC form of the Heine--Cantor theorem of classical analysis. The classical proof uses sequential compactness and an extracted convergent subsequence; the BEDC proof uses the located finite $\varepsilon$-net and an explicit minimum over a finite list. The two routes deliver the same uniform modulus when the metric carrier is taken as $\NameCert_{\RealUp}$ restricted to a closed bounded interval (\autoref{ch:concrete-instances-interval-namecert}), making the classical result a derived corollary of the kernel-level statement and not the other way around.
\end{remark}

\begin{remark}[Composition with the real-completeness capstone]
\label{rem:compact-uniform-real-completeness-composition}
The construction composes with the real-completeness capstone (\autoref{ch:capstones-real-completeness}): once $\NameCert_{\RealUp}$ is upgraded to a regular Cauchy carrier, every closed bounded interval $[a, b] \subset \RealUp$ inherits a compact naming certificate via the standard located $\varepsilon$-net of dyadic rationals at scale $\lceil 1/\varepsilon \rceil$, and \autoref{thm:compact-uniform-continuity} then gives uniform continuity of every continuous $f \colon [a, b] \to \RealUp$ with an explicit modulus. The chapter is therefore the canonical bridge from the analytic certificate layer to the constructive analysis literature.
\end{remark}
