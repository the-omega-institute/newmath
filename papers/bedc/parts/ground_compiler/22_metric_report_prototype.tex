\chapter{Metric-Report Prototype}
\label{ch:ground-compiler-metric-report-prototype}

\autoref{ch:ground-compiler-motif-report-prototype} defines $P_3$, the motif-report prototype. It separates candidate motifs, recognized motifs, motif ledgers, recognized motif counts, missing recognizers, and cannot-claim annotations. This chapter defines the next prototype level:
$$
  \boxed{P_4=P_3+\text{admissible metric computation}+\text{metric report}.}
$$
The task of $P_4$ is to compute structural indicators from recognized motif reports: prefix overlap, motif counts, seal depth, carry index, ledger depth, classifier compression ratio, reuse depth, bridge depth, normal-address distance, motif distance, and flow signature vectors.

The governing rule is:
$$
  \boxed{\text{metrics are explanatory summaries, not proofs, certificates, or object identities}.}
$$
Thus $P_4$ cannot infer that equal metric profiles give equal theories, equal signature vectors give equal objects, distance zero gives theorem equivalence, high overlap gives a bridge, positive carry index gives dimension, or positive seal depth gives a completion object.

\section{Metric protocols and admissibility}
\label{sec:metric-report-protocols-admissibility}

\begin{definition}[Metric-report prototype]
\label{def:metric-report-prototype}
A \emph{metric-report prototype} is an executable prototype $P_4$ satisfying:
\begin{enumerate}
\item $P_4$ includes all $P_3$ functions;
\item $P_4$ receives a decoded event flow $S$ and a recognized motif report;
\item $P_4$ uses a recognized metric protocol flow;
\item $P_4$ computes admissible metrics;
\item $P_4$ emits a metric report;
\item $P_4$ does not treat metric reports as proofs, certificates, bridges, name-certificate flows, derivation-certificate flows, or acceptance gates.
\end{enumerate}
Write
$$
  \mathsf{MetricReportPrototype}(P_4).
$$
\end{definition}

\begin{definition}[Metric candidate]
\label{def:metric-report-candidate}
\leandef{BEDC.GroundCompiler.MetricsFlow.MetricCandidate}
A \emph{metric candidate} is a proposed numerical value, sequence, set, or summary to compute on an event flow. Examples include seal depth, carry index, ledger depth, reuse depth, bridge depth, motif count, motif distance, prefix overlap, and compression ratio. A metric candidate is not yet an admissible metric.
\end{definition}

\begin{definition}[Generated metric recognizer]
\label{def:metric-report-generated-metric-recognizer}
\leandef{BEDC.GroundCompiler.MetricsFlow.GeneratedMetricRecognizer}
A \emph{generated metric recognizer} is an event flow
$$
  R_{\mathrm{met}}\in\mathsf{EventFlow}
$$
supporting
$$
  \mathsf{RecognizesMetric}(R_{\mathrm{met}},m),
$$
where $m$ is a metric specification flow. The specification is not an external function name, Python function, JSON field, or hand-written metric table.
\end{definition}

\begin{definition}[Metric protocol flow]
\label{def:metric-report-protocol-flow}
\leandef{BEDC.GroundCompiler.MetricsFlow.MetricProtocolFlow}
A \emph{metric protocol flow} is an event flow
$$
  P_{\mathrm{met}}\in\mathsf{EventFlow}
$$
recognized as a metric-computation protocol. It contains metric-selection flow, metric-domain flow, recognizer-family flow, normalization-policy flow, ledger-policy flow, undefined-value-policy flow, comparison-policy flow, report-policy flow, cannot-claim-policy flow, and protocol seal flow. Write
$$
  \mathsf{MetricProtocolFlow}(P_{\mathrm{met}}).
$$
\end{definition}

\begin{principle}[No metric without protocol]
\label{prin:metric-report-no-metric-without-protocol}
External metric configuration, weight files, manual metric tables, host metric functions, and external similarity functions are not formal inputs. A metric is BEDC-admissible only when specified by recognized metric protocol flow.
\end{principle}

\begin{definition}[Admissible metric]
\label{def:metric-report-admissible}
A metric $M$ for an event flow $S$ is \emph{admissible} relative to protocol $P_{\mathrm{met}}$ when:
\begin{enumerate}
\item $M$ is given by a recognized metric specification flow;
\item its input is the decoded event flow $S$;
\item all motifs used by $M$ come from a recognized motif report;
\item all ledgers used by $M$ are recognized ledger flows;
\item undefined cases are governed by the protocol flow;
\item cannot-claim entries are generated by the report policy;
\item $M$ does not depend on channel substring search, external object names, host tags, YAML fields, manual grouping, or unrecognized semantic labels.
\end{enumerate}
Write
$$
  \mathsf{AdmissibleMetric}(M,S,P_{\mathrm{met}}).
$$
\end{definition}

\begin{theorem}[Unrecognized metric is inadmissible]
\label{thm:metric-report-unrecognized-inadmissible}
If there is no recognized metric specification flow and no recognized metric protocol flow, then a value or summary cannot be used as a BEDC-admissible metric.
\end{theorem}

\begin{proof}
Admissibility requires recognized specification and protocol flow. Without them, the domain, computation rule, undefined cases, recognizer family, ledger policy, and cannot-claim policy are externally supplied. That is hidden input.
\end{proof}

\begin{corollary}[Host similarity score is not admissible]
\label{cor:metric-report-host-score-not-admissible}
A similarity score produced only by a host program, embedding model, external statistic, or manual choice is not a BEDC-admissible metric without recognized metric protocol flow.
\end{corollary}

\section{Basic metrics}
\label{sec:metric-report-basic-metrics}

\begin{definition}[Recognized motif count]
\label{def:metric-report-recognized-motif-count}
Given an event flow $S$, a recognized motif report, and motif role $\mu$, define
$$
  \mathsf{MotifCount}(S,\mu)
$$
as the number of recognized $\mu$-role occurrences listed in the recognized motif report. If the report contains only candidate motifs and no recognized section, then $\mathsf{MotifCount}(S,\mu)$ is undefined, not zero.
\end{definition}

\begin{definition}[Candidate motif count]
\label{def:metric-report-candidate-motif-count}
The \emph{candidate motif count} for role $\mu$ is the number of $\mu$-like candidates listed in the candidate section. It is distinct from recognized motif count.
\end{definition}

\begin{proposition}[Candidate count is weaker than recognized count]
\label{prop:metric-report-candidate-count-weaker}
Candidate motif count cannot replace recognized motif count.
\end{proposition}

\begin{proof}
Candidate count records shape or candidate policy. Recognized count requires generated recognizer and motif ledger. Treating candidate count as recognized count promotes unverified patterns to recognized motifs.
\end{proof}

\begin{corollary}[Missing recognizer does not imply zero]
\label{cor:metric-report-missing-not-zero}
If a motif recognizer is missing, the corresponding recognized motif count is undefined or recognizer-missing. It is not zero.
\end{corollary}

\begin{definition}[Seal depth]
\label{def:metric-report-seal-depth}
The \emph{seal depth} of $S$ is the number of recognized seal motif occurrences in $S$. If the seal recognizer is missing, seal depth is undefined. Candidate seal events may be counted separately as candidate seal count.
\end{definition}

\begin{definition}[Carry index]
\label{def:metric-report-carry-index}
The \emph{carry index} of $S$ is the number of recognized carry motif occurrences in $S$. If only candidate pairs such as $011\mid 100$ or $0011\mid 0100$ are present, but no recognized carry motif is supplied, carry index is undefined.
\end{definition}

\begin{definition}[Ledger depth]
\label{def:metric-report-ledger-depth}
The \emph{ledger depth} of $S$ is the maximum nesting depth among recognized ledger flows in $S$. If there is no recognized ledger flow, ledger depth is $0$. If there are suspected ledger candidates without recognizer, the report must separately mark ledger-recognizer missing; it must not treat the value $0$ as a structural conclusion.
\end{definition}

\begin{definition}[Reuse depth]
\label{def:metric-report-reuse-depth}
The \emph{reuse depth} of $S$ is the length of the longest dependency chain of recognized reuse motifs. If the reuse recognizer is missing, reuse depth is undefined.
\end{definition}

\begin{definition}[Bridge depth]
\label{def:metric-report-bridge-depth}
The \emph{bridge depth} of $S$ is the maximum bridge-chain length among recognized bridge motifs or bridge flows. If only bridge-candidate motifs are present, bridge depth is undefined, while bridge-candidate count may be reported separately.
\end{definition}

\begin{definition}[Classifier compression ratio]
\label{def:metric-report-compression-ratio}
If a recognized classifier quotient in $S$ sends $k$ raw support flows into $r$ recognized classes with $r>0$, define
$$
  \mathsf{CompressionRatio}(S)=\frac{k}{r}.
$$
This metric requires classifier recognizer, exactness recognizer, and ledger recognizer.
\end{definition}

\begin{proposition}[Compression ratio requires exactness]
\label{prop:metric-report-compression-requires-exactness}
Without classifier exactness flow, admissible compression ratio cannot be computed.
\end{proposition}

\begin{proof}
Compression ratio counts how raw supports are classified. If classifier exactness is not recognized, the classes may not be sound, may not respect the intended sameness, and may lack ledger completeness. The metric is therefore inadmissible.
\end{proof}

\section{Comparison metrics}
\label{sec:metric-report-comparison-metrics}

\begin{definition}[Raw prefix overlap]
\label{def:metric-report-raw-prefix-overlap}
For event flows
$$
  S=(s_0,\ldots,s_{m-1}),
  \qquad
  T=(t_0,\ldots,t_{n-1}),
$$
the \emph{raw prefix overlap length}
$$
  \mathsf{RawPrefixLen}(S,T)
$$
is the largest $k$ such that $s_i=t_i$ for every $0\le i<k$.
\end{definition}

\begin{definition}[Recognized prefix overlap]
\label{def:metric-report-recognized-prefix-overlap}
The \emph{recognized prefix overlap} of $S$ and $T$ is the set of shared recognized prefix motif records. Sharing requires the same support shape, same motif role, ledger-compatible evidence, and either the same recognizer family or recognized compatible recognizers.
\end{definition}

\begin{proposition}[Raw prefix overlap does not imply recognized prefix overlap]
\label{prop:metric-report-raw-prefix-not-recognized-prefix}
Two event flows can share raw prefix without sharing recognized semantic prefix.
\end{proposition}

\begin{proof}
The same raw prefix may be recognized as different motif roles in different contexts, or may be unrecognized in one flow. Therefore raw prefix overlap alone does not imply recognized prefix overlap.
\end{proof}

\begin{definition}[Motif overlap]
\label{def:metric-report-motif-overlap}
The \emph{motif overlap} of two flows is the intersection of their recognized motif profiles:
$$
  \mathsf{MotifOverlap}(S,T)
  =
  \mathsf{MotifProfile}(S)\cap\mathsf{MotifProfile}(T),
$$
where intersection is computed by motif role, support shape, and ledger-compatible equivalence.
\end{definition}

\begin{definition}[Motif distance]
\label{def:metric-report-motif-distance}
When the protocol defines the denominator, the \emph{motif distance} is
$$
  \mathsf{MotifDist}(S,T)
  =
  1-
  \frac{
    |\mathsf{MotifProfile}(S)\cap\mathsf{MotifProfile}(T)|
  }{
    |\mathsf{MotifProfile}(S)\cup\mathsf{MotifProfile}(T)|
  }.
$$
If the denominator is $0$, the undefined-value policy in the metric protocol determines the report entry.
\end{definition}

\begin{definition}[Normal-address overlap and distance]
\label{def:metric-report-normal-address-distance}
Given normalizer recognizers, the \emph{normal-address profile} of $S$ is the set of recognized records $(w,v,L_{w\to v})$. The overlap of $S$ and $T$ is the intersection of these profiles. When defined, the normal-address distance is
$$
  \mathsf{NormDist}(S,T)
  =
  1-
  \frac{|\mathsf{NormOverlap}(S,T)|}
       {|\mathsf{NormalAddressProfile}(S)\cup\mathsf{NormalAddressProfile}(T)|}.
$$
Zero denominator behavior is protocol-defined.
\end{definition}

\begin{definition}[Flow signature vector]
\label{def:metric-report-signature-vector}
The \emph{flow signature vector} of $S$ is
$$
  \mathsf{Sig}(S)
  =
  (
  \mathsf{SealDepth},
  \mathsf{CarryIndex},
  \mathsf{LedgerDepth},
  \mathsf{CompressionRatio},
  \mathsf{ReuseDepth},
  \mathsf{BridgeDepth}
  )_S.
$$
It is defined only when each component is admissible or when the protocol explicitly handles undefined components.
\end{definition}

\begin{theorem}[Signature equality does not imply object equality]
\label{thm:metric-report-signature-not-object-equality}
If
$$
  \mathsf{Sig}(S)=\mathsf{Sig}(T),
$$
one cannot infer that $S$ and $T$ represent the same theorem, chapter, object, classifier class, or bridge target.
\end{theorem}

\begin{proof}
A signature vector is a finite summary. Distinct event flows may have equal carry count, seal count, ledger depth, or other component values while having different supports, ledgers, classifiers, proofs, and certificates. Object equivalence or theorem equivalence requires classifier, proof, bridge, or certificate flow.
\end{proof}

\begin{corollary}[Signature vector is explanatory summary]
\label{cor:metric-report-signature-explanatory}
Signature vectors may support comparison and clustering, but they are not identity criteria.
\end{corollary}

\section{Metric reports}
\label{sec:metric-report-reports}

\begin{definition}[Metric report]
\label{def:metric-report-report}
Given event flow $S$, metric protocol flow $P_{\mathrm{met}}$, and a recognized motif report, a \emph{metric report} is an output view containing the input flow, recognized motif summary, candidate motif summary, metric protocol, computed admissible metrics, undefined metrics, missing recognizers, ledger requirements, warnings, and cannot-claim annotations. Write
$$
  \mathsf{MetricReport}(S,P_{\mathrm{met}}).
$$
\end{definition}

\begin{definition}[Undefined metric item]
\label{def:metric-report-undefined-item}
An \emph{undefined metric item} records why a metric is undefined: recognizer missing, ledger missing, classifier exactness missing, normalizer missing, division by zero unresolved by protocol, candidate-only evidence, or unresolved bootstrap obligation.
\end{definition}

\begin{principle}[Undefined metric is explicit]
\label{prin:metric-report-undefined-explicit}
If a metric is inadmissible or undefined, the report must record an undefined metric item. It must not silently omit the metric or treat missing data as $0$.
\end{principle}

\begin{theorem}[Silent zero is unsound]
\label{thm:metric-report-silent-zero-unsound}
If a metric is undefined because a recognizer is missing, but the report records it as $0$, then the metric report is unsound.
\end{theorem}

\begin{proof}
Missing recognizer means the system cannot determine whether the motif exists. Recording $0$ asserts absence of the motif. That turns epistemic absence into structural absence, which is unsound.
\end{proof}

\begin{definition}[Metric report soundness]
\label{def:metric-report-soundness}
A metric report is \emph{sound} when every computed metric is admissible, every undefined metric has an undefined metric item, every recognizer and ledger dependency is recorded, candidate counts and recognized counts are separated, cannot-claim annotations are complete, and the report does not assert proof, certificate, object equality, or bridge.
\end{definition}

\begin{theorem}[Sound metric report theorem]
\label{thm:metric-report-sound-report}
If $P_4$ outputs a sound metric report, then every numerical metric in the report is supported by recognized event-flow structure, and every undefined metric is explicitly recorded.
\end{theorem}

\begin{proof}
By soundness, computed metrics must be admissible. Admissible metrics depend only on decoded event flow, recognized motifs, recognized ledgers, recognized classifiers, and protocol flow. Undefined metrics are recorded by undefined metric items.
\end{proof}

\section{Metric claims and cannot-claims}
\label{sec:metric-report-claims-cannot-claims}

\begin{definition}[Metric claim]
\label{def:metric-report-claim}
A \emph{metric claim} is a statement based on a metric report, such as: $S$ and $T$ share a finite-repetition motif, $S$ has carry index $1$, $S$ has seal depth $2$, $S$ and $T$ have motif distance $0.25$, or $S$ has classifier compression ratio $3$. Such a claim is admissible only when the relevant metric report is sound and the metric is admissible.
\end{definition}

\begin{definition}[Forbidden metric claim]
\label{def:metric-report-forbidden-claim}
The following metric claims are forbidden unless supported by additional certificate flows:
\begin{itemize}
\item same metric profile implies same object;
\item same signature vector implies same theorem;
\item motif distance zero implies proof equivalence;
\item high motif overlap implies bridge;
\item positive carry index implies dimension;
\item positive seal depth implies completion object;
\item compression ratio greater than one implies quotient exactness;
\item positive reuse depth implies a reuse certificate.
\end{itemize}
\end{definition}

\begin{theorem}[Forbidden metric claims need certificates]
\label{thm:metric-report-forbidden-claims-need-certificates}
Every forbidden metric claim is invalid without additional classifier, proof, bridge, name-certificate, derivation-certificate, or acceptance-gate flow.
\end{theorem}

\begin{proof}
Metric reports summarize event-flow structure. Forbidden claims assert object equivalence, proof equivalence, bridge existence, dimension, completion, quotient exactness, or reuse certification. These require independent certificate flows. Metrics alone do not supply them.
\end{proof}

\begin{corollary}[Metric reports require cannot-claim sections]
\label{cor:metric-report-require-cannot-claims}
Every nontrivial metric report must contain a cannot-claim section listing the stronger claims that the metrics do not support.
\end{corollary}

\section{$P_4$ pipeline and examples}
\label{sec:metric-report-pipeline-examples}

\begin{definition}[$P_4$ pipeline]
\label{def:metric-report-p4-pipeline}
Given channel code $c$, the $P_4$ pipeline consists of $P_0$ legality check, $P_1$ source-event report, $P_2$ normalizer ledger report, $P_3$ motif report, metric protocol recognition, admissible metric computation, undefined metric audit, cannot-claim audit, and metric report generation.
\end{definition}

\begin{theorem}[$P_4$ pipeline is source-determined relative to protocol]
\label{thm:metric-report-source-determined}
If $c\in\mathsf{LegalZStream}$, $P_{\mathrm{met}}$ is a recognized metric protocol flow, and all selected recognizers are deterministic, then the $P_4$ output is uniquely determined.
\end{theorem}

\begin{proof}
The channel bijection decodes $c$ to a unique event flow $S$. The recognized protocol fixes metric selection, recognizer family, undefined-value policy, and report policy. Deterministic recognizers determine motif reports and ledgers. Metrics are computed from those data.
\end{proof}

\begin{corollary}[Different protocols may yield different reports]
\label{cor:metric-report-different-protocols}
Different metric protocol flows may choose different recognizer families, metric selections, weights, undefined-value policies, or report policies. Different reports under different protocols are therefore not contradictory.
\end{corollary}

\begin{example}[Completion skeleton metric report]
\label{ex:metric-report-completion}
For input
$$
  011001100011010110101011100011,
$$
the decoded flow is
$$
  0\mid 00\mid 000\mid 01\mid 011\mid 100.
$$
Assume the recognized motif report includes finite repetition $0\mid 00\mid 000$, carry $011\mid 100$, and seal $011$. The metric report may record motif counts for finite repetition, carry, and seal as $1$, seal depth $1$, carry index $1$, and ledger depth at least $1$ depending on the carry ledger. Completion-object status and full-axis export remain undefined. Cannot-claims say that seal depth $1$ does not imply completion, carry index $1$ does not imply dimension-lift, and the report does not establish $011=100$.
\end{example}

\begin{example}[Fold skeleton metric report]
\label{ex:metric-report-fold}
For input
$$
  0110011001011001010110100011,
$$
the decoded flow is
$$
  0\mid 00\mid 001\mid 0011\mid 0100.
$$
If recognized motifs include additive recursion $001$ and fold normalizer $0011\mid 0100$, the report may count each as $1$. Carry index is $1$ only if the fold normalizer is recognized under carry role. Seal depth is undefined unless a seal recognizer recognizes $0011$, and reuse depth is undefined unless a reuse recognizer is supplied.
\end{example}

\begin{example}[Completion/fold comparison]
\label{ex:metric-report-completion-fold-comparison}
For
$$
  S_{\mathrm{Comp}}=0\mid 00\mid 000\mid 01\mid 011\mid 100
$$
and
$$
  S_{\mathrm{Fold}}=0\mid 00\mid 001\mid 0011\mid 0100,
$$
the raw prefix overlap is $0\mid 00$. A recognized shared finite-source motif exists only if recognizer evidence is supplied. Motif distance is protocol-relative. Cannot-claims record that shared prefix does not imply bridge and distance values do not imply object identity.
\end{example}

\section{Adequacy and conservativity}
\label{sec:metric-report-adequacy-conservativity}

\begin{definition}[$P_4$ audit checklist]
\label{def:metric-report-p4-audit}
A $P_4$ prototype must satisfy:
\begin{enumerate}
\item all $P_3$ audit requirements;
\item recognized metric protocol flow is required;
\item candidate counts are separated from recognized counts;
\item missing recognizers are marked undefined, not zero;
\item metric dependencies are recorded;
\item ledger requirements are recorded;
\item undefined metric items are recorded;
\item cannot-claim sections are included;
\item object equality is not inferred from metrics;
\item theorem equivalence is not inferred from metrics;
\item bridge is not inferred from motif overlap;
\item external metric tables are not formal inputs.
\end{enumerate}
\end{definition}

\begin{theorem}[$P_4$ adequacy theorem]
\label{thm:metric-report-p4-adequacy}
If a prototype satisfies the $P_4$ audit checklist, uses sound $P_3$ reports, and computes only admissible metrics under recognized metric protocol flow, then it is adequate as a metric-report prototype.
\end{theorem}

\begin{proof}
The checklist ensures that metrics are computed only from recognized event-flow structures, that candidate and recognized evidence are separated, undefined cases are explicit, and cannot-claims are included. Sound $P_3$ reports validate motif data. Recognized metric protocol prevents metric selection and policies from becoming hidden inputs.
\end{proof}

\begin{corollary}[$P_4$ adequacy does not imply higher adequacy]
\label{cor:metric-report-p4-not-higher}
A $P_4$ prototype does not recognize packages, name certificates, derivation certificates, theorems, chapters, or accepted objects unless those recognizers are separately supplied.
\end{corollary}

\begin{theorem}[$P_4$ is conservative over the finite kernel]
\label{thm:metric-report-p4-conservative}
Adding metric reports, signature vectors, distances, counts, compression ratios, and undefined metric items adds no raw marks, history constructors, sameness rules, continuation rules, derived-interface formation rules, theorem rules, or accepted-object export rules.
\end{theorem}

\begin{proof}
$P_4$ computes summaries over decoded event flows and recognized motif reports. It does not generate new $\bzero,\bone$, add $\msame$, $\hsame$, or $\psame$, alter $\ExtR$ or $\ContR$, or supply name certificates, derivation certificates, proof flows, or acceptance gates.
\end{proof}

\begin{corollary}[$P_4$ remains an analysis layer]
\label{cor:metric-report-p4-analysis-layer}
$P_4$ provides metric analysis. It is not a mathematical object generator, proof checker, or certificate generator.
\end{corollary}

\section{Full reports}
\label{sec:metric-report-full-reports}

\begin{example}[$P_4$ report for completion skeleton]
\label{ex:metric-report-full-completion}
For
$$
  0\mid 00\mid 000\mid 01\mid 011\mid 100,
$$
assume recognized finite repetition, seal, and carry motifs with supports $0\mid 00\mid 000$, $011$, and $011\mid 100$. The report records
$$
  \mathsf{MotifCount}(\text{finite repetition})=1,\quad
  \mathsf{MotifCount}(\text{seal})=1,\quad
  \mathsf{MotifCount}(\text{carry})=1,
$$
and seal depth $1$, carry index $1$, and ledger depth $1$ when the ledger is one level. Compression ratio, reuse depth, and bridge depth are undefined if the corresponding recognizers are missing. Cannot-claims say that seal depth does not imply completion, carry index does not imply dimension-lift, classifier exactness is not supplied, no name-certificate flow is supplied, no derivation-certificate flow is supplied, no acceptance-gate flow is supplied, and the report is not proof.
\end{example}

\begin{example}[$P_4$ comparison report]
\label{ex:metric-report-full-comparison}
For the completion and fold skeletons, the report may record raw prefix $0\mid 00$, recognized shared finite-source motif when recognizer evidence exists, completion-specific seal and carry motifs, and fold-specific additive recursion and fold normalizer motifs. Raw prefix length is $2$, while shared recognized motif count and motif distance are protocol-relative. Cannot-claims say that shared prefix does not imply subobject, motif distance does not imply theorem equivalence, and no bridge certificate is supplied.
\end{example}

\begin{remark}[Summary]
\label{rem:metric-report-summary}
This chapter defines $P_4$, the metric-report prototype. It adds metric candidates, generated metric recognizers, metric protocol flows, admissible metrics, recognized and candidate motif counts, seal depth, carry index, ledger depth, reuse depth, bridge depth, classifier compression ratio, prefix overlap, motif overlap, motif distance, normal-address distance, signature vectors, undefined metric items, metric claims, forbidden metric claims, the $P_4$ pipeline, and the $P_4$ audit checklist. It proves that unrecognized metrics are inadmissible, host similarity scores are not admissible metrics, candidate counts are weaker than recognized counts, missing recognizers do not imply zero, compression ratio requires exactness, raw prefix overlap does not imply recognized prefix overlap, signature equality does not imply object equality, silent zero is unsound, sound metric reports are backed by recognized structures, forbidden metric claims require certificates, the pipeline is source-determined relative to protocol, and $P_4$ is conservative over the finite kernel.
\end{remark}
