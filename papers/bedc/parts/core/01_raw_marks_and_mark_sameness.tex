\chapter{Raw Marks and Mark-Sameness}
\label{ch:core-raw-marks-and-mark-sameness}

\begin{definition}[Raw marks]
\label{def:raw-marks}
The only raw mark forms are
\[
\bzero,\qquad \bone.
\]
They are not the numbers \(0\) and \(1\), not truth values, and not elements of a pre-existing set.
\end{definition}
\leandef{BEDC.FKernel.Mark.BMark}
\leanchecked{BEDC.FKernel.Mark.BMark\_generated\_cases}

\begin{kernelrule}[Mark formation]
The judgment \(m:\Mark\) is generated by exactly two formation rules:
\[
\frac{}{\bzero:\Mark}
\qquad
\frac{}{\bone:\Mark}.
\]
\end{kernelrule}

\begin{definition}[Internal mark-sameness]
\label{def:internal-mark-sameness}
The judgment \(\msame(m,n)\) is generated by exactly two rules:
\[
\frac{}{\msame(\bzero,\bzero)}
\qquad
\frac{}{\msame(\bone,\bone)}.
\]
No rule generates \(\msame(\bzero,\bone)\) or \(\msame(\bone,\bzero)\).
\end{definition}
\leanchecked{BEDC.FKernel.Mark.msame\_constructor\_characterization}
\leanvariant{BEDC.FKernel.Mark.msame\_iff\_eq}
\leanvariant{BEDC.FKernel.Mark.msame\_generated\_rules}
\leanvariant{BEDC.FKernel.Mark.msame\_cases}

\begin{axiomlike}[Closed generation for marks]
The only generated marks are those generated by the two mark-formation rules. The only generated mark-sameness derivations are those generated by the two mark-sameness rules.
\end{axiomlike}

\begin{lemma}[Mark-sameness reflexivity]
For every generated mark \(m\), \(\msame(m,m)\).
\end{lemma}
\leanchecked{BEDC.FKernel.Mark.msame\_refl}
\begin{proof}
By closed generation, \(m\) is either \(\bzero\) or \(\bone\). Use the corresponding mark-sameness rule.
\end{proof}

\begin{lemma}[Mark-sameness symmetry]
If \(\msame(m,n)\), then \(\msame(n,m)\).
\end{lemma}
\leanchecked{BEDC.FKernel.Mark.msame\_symm}
\begin{proof}
The premise is generated only in the two reflexive cases. In either case, the conclusion is generated by the same rule.
\end{proof}

\begin{lemma}[Mark-sameness transitivity]
If \(\msame(m,n)\) and \(\msame(n,p)\), then \(\msame(m,p)\).
\end{lemma}
\leanchecked{BEDC.FKernel.Mark.msame\_trans}
\begin{proof}
Closed generation leaves only two cases. If the middle mark is \(\bzero\), then all generated mark-sameness premises force \(m\) and \(p\) to be \(\bzero\). The \(\bone\) case is identical.
\end{proof}

\begin{theorem}[Mark-sameness equivalence]
\label{thm:mark-sameness-equivalence}
\label{thm:msame-internal-equivalence-spine}
The internal judgment \(\msame\) is reflexive, symmetric, and transitive on generated raw marks.
\end{theorem}
\leanchecked{BEDC.FKernel.Mark.msame\_equivalence}
\leanvariant{BEDC.FKernel.Mark.msame\_internal\_equivalence\_spine}
\begin{proof}
Immediate from the preceding three lemmas.
\end{proof}

\begin{theorem}[Raw mark no-confusion]
\label{thm:mark-no-confusion}
The kernel does not generate \(\msame(\bzero,\bone)\) or \(\msame(\bone,\bzero)\).
\end{theorem}
\leanchecked{BEDC.FKernel.Mark.msame\_no\_confusion}
\leanvariant{BEDC.FKernel.Mark.msame\_cross\_iff\_false}
\leanvariant{BEDC.FKernel.Mark.msame\_cross\_impossible}
\begin{proof}
By the closed-generation principle, mark-sameness has only the two displayed constructors. Neither has cross-mark conclusion.
\end{proof}
