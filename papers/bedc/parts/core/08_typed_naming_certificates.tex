\chapter{Typed Naming Certificates}
\label{ch:core-typed-naming-certificates}

\begin{definition}[Source specification]
A source specification \(\SourceSpec\) records the generated material from which a proposed name is licensed. It may refer to raw marks, histories, bundles, packages, gaps, or previously licensed names.
\end{definition}

\begin{definition}[Pattern specification]
A pattern specification \(\PatternSpec\) records the stable behavior observed in the source. Examples include repetition, continuation, signature classification, package stability, or compatible threading.
\end{definition}

\begin{definition}[Classifier specification]
A classifier specification \(\ClassifierSpec\) records how source material is grouped. At the finite kernel level this is normally given by a bundle, an asking policy, a package policy, and a gap policy.
\end{definition}

\begin{definition}[Stability certificates]
\label{def:naming-certificate-stability-field}
The stability component \(\StabCert\) is typed by modes:
$$
\begin{aligned}
\ClosureCert &: \text{the source is closed under the named behavior},\\
\ReuseCert &: \text{the named behavior can be reused as a later tool},\\
\DescentCert &: \text{the behavior descends through classification},\\
\CompositionCert &: \text{the behavior composes with previously named behavior},\\
\SealCert &: \text{the behavior supports compatible-thread sealing}.
\end{aligned}
$$
A name may require one or more of these modes.
\end{definition}

\begin{definition}[Ledger policy]
A ledger policy \(\LedgerPolicy\) records what the name hides and how the hidden material is tracked. It must specify whether the ledger is a gap, a composite gap, a tail ledger, a phase ledger, a source ledger, or another derived ledger.
\end{definition}

\begin{definition}[Paper naming certificate template]
\label{def:typed-naming-certificate}
\label{def:naming-certificate}
A paper naming certificate for a proposed derived interface \(N\) is the five-field template
$$
\NameCert(N)=
(\SourceSpec_N,\PatternSpec_N,\ClassifierSpec_N,\StabCert_N,\LedgerPolicy_N).
$$
The proposed name \(N\) is licensed only if all required fields are supplied.
\end{definition}

\begin{definition}[Lean carrier naming certificate]
\label{def:lean-carrier-naming-certificate}
At the formal finite-kernel layer, the Lean structure
\(\mathsf{NameCert}(\mathsf{Carrier},\mathsf{Equiv})\) is a carrier-and-equivalence certificate over histories. It supplies inhabitedness of the carrier, reflexivity, symmetry, and transitivity for the equivalence relation, and transport of carrier membership along that relation.
\end{definition}

\begin{theorem}[Naming certificate classifier transitivity]
\label{thm:namecert-equivalence-transitivity}
For every naming certificate over a history carrier and classifier, the classifier is transitive on finite-kernel histories.
\end{theorem}
\begin{proof}
The certificate record contains the classifier transitivity field. Projecting that field supplies the required composition of two classifier witnesses.
\end{proof}
\leanchecked{BEDC.FKernel.NameCert.nameCert\_equiv\_trans}

\begin{theorem}[Derived interfaces require certificates]
\label{thm:derived-interfaces-require-certificates}
No interface marked by an upward arrow is licensed without a \(\NameCert\).
\end{theorem}
\begin{proof}
This is the formation rule for the paper template: derived names are not object-language primitives. They enter the theory only by the five supplied certificate fields.
\end{proof}
\leanchecked{BEDC.FKernel.NameCert.derived\_interfaces\_require\_certificates}

\begin{principle}[Limit-like interfaces carry a ledger]
\label{prin:limit-like-interfaces-derived}
Every certified derived interface carries a ledger policy witness.
\end{principle}
\begin{proof}
Case analysis on the semantic naming certificate exposes the source witness in the core certificate. Applying the ledger soundness field to that same source witness gives the required ledger witness.
\end{proof}
\leanchecked{BEDC.FKernel.NameCert.semanticNameCert\_ledger\_policy\_witness}

\begin{theorem}[Semantic NameCert presentation weakening]
\label{thm:semantic-namecert-presentation-weakening}
Let \(S,P,L,P',L'\colon \mathsf{BHist}\to\mathsf{Prop}\) and
\(C\colon \mathsf{BHist}\to\mathsf{BHist}\to\mathsf{Prop}\). If
\[
\mathsf{SemanticNameCert}(S,P,L,C)
\]
holds, and for every history \(h\) the implications \(P(h)\to P'(h)\) and
\(L(h)\to L'(h)\) hold, then
\[
\mathsf{SemanticNameCert}(S,P',L',C)
\]
holds. The source specification and classifier specification are unchanged.
\end{theorem}
\begin{proof}
By \autoref{def:typed-naming-certificate}, the naming template separates the
source, pattern, classifier, stability, and ledger roles. The core field of the
semantic certificate is the carrier-and-classifier \(\NameCert(S,C)\) described
by \autoref{def:lean-carrier-naming-certificate}; it mentions only the source
predicate and the classifier relation. Keep this core field. For a source
witness \(S(h)\), the original pattern soundness field gives \(P(h)\), and the
pointwise pattern weakening gives \(P'(h)\). The same source witness gives
\(L(h)\) by the original ledger soundness field, and the pointwise ledger
weakening gives \(L'(h)\). These three fields form the semantic certificate for
\(S,P',L',C\).
\end{proof}
