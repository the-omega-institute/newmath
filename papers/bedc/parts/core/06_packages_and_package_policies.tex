\chapter{Packages and Package Policies}
\label{ch:core-packages-and-package-policies}
\origin{human}

\begin{definition}[Package token and package-sameness]
\label{def:core-package-token-and-package-sameness}
\(\Pkg\) is a judgment
$$
\Pkg(\Pi,s,p)
$$
meaning that answer-history \(s\) under bundle \(\Pi\) is represented by package token \(p\). The internal package-sameness judgment is \(\psame(p,q)\).
\end{definition}
\leanchecked{BEDC.FKernel.Package.package\_token\_and\_psame\_generated\_pair}

\begin{definition}[Package policy]
A package policy \(\PkgPol(\Pi)\) consists of the following certified fields:
\begin{enumerate}
  \item existence: each generated answer-history \(s\) has a package token \(p\) with \(\Pkg(\Pi,s,p)\);
  \item extensionality: if \(\hsame(s,t)\), \(\Pkg(\Pi,s,p)\), and \(\Pkg(\Pi,t,q)\), then \(\psame(p,q)\);
  \item grounding: if \(\psame(p,q)\), then there are answer histories \(s,t\) with \(\Pkg(\Pi,s,p)\), \(\Pkg(\Pi,t,q)\), and \(\hsame(s,t)\);
  \item visibility: only the package token and its ledger are visible at the package interface.
\end{enumerate}
\end{definition}

\begin{theorem}[Packages classify signatures]
\label{thm:packages-classify-signatures-core}
Under $\PkgPol(\Pi)$, if $\Pkg(\Pi,s,p)$ and $\Pkg(\Pi,t,q)$, then
$\hsame(s,t)$ implies $\psame(p,q)$. Conversely, if $\psame(p,q)$, then
there exist answer histories $u,v$ such that $\Pkg(\Pi,u,p)$,
$\Pkg(\Pi,v,q)$, and $\hsame(u,v)$.
\end{theorem}
\leanchecked{BEDC.FKernel.Package.packagePolicy\_classifies\_signature\_witnesses}
\begin{proof}
The first direction is the extensionality field. The second direction is the
grounding field.
\end{proof}

\begin{theorem}[Package representative for generated answer-history]
\label{thm:package-representative-for-generated-answer-history}
Assume $\PkgPol(\Pi)$. For every generated answer-history $s$, there exists a package token $p$ such that $\Pkg(\Pi,s,p)$ and, for every answer-history $t$ and package token $q$, if $\hsame(s,t)$ and $\Pkg(\Pi,t,q)$, then $\psame(p,q)$.
\end{theorem}
\begin{proof}
Let $s$ be a generated answer-history. The existence field of $\PkgPol(\Pi)$ supplies a package token $p$ with $\Pkg(\Pi,s,p)$. It remains to verify the displayed universal property. Let $t$ and $q$ be such that $\hsame(s,t)$ and $\Pkg(\Pi,t,q)$. Applying the first direction of \autoref{thm:packages-classify-signatures-core} to $\Pkg(\Pi,s,p)$, $\Pkg(\Pi,t,q)$, and $\hsame(s,t)$ gives $\psame(p,q)$.
\end{proof}
