\chapter{Emission Histories and History-Sameness}
\label{ch:core-emission-histories-and-history-sameness}

\begin{definition}[Emission histories]
Emission histories are generated by:
\[
\frac{}{\emp:\Hist}
\]
\[
\frac{h:\Hist}{\Ezero(h):\Hist}
\qquad
\frac{h:\Hist}{\Eone(h):\Hist}.
\]
The symbols \(\emp\), \(\Ezero\), and \(\Eone\) are object-language history constructors, not imported list constructors.
\end{definition}

\begin{definition}[History-sameness]
The judgment \(\hsame(h,k)\) is generated by:
\[
\frac{}{\hsame(\emp,\emp)}
\]
\[
\frac{\hsame(h,k)}{\hsame(\Ezero(h),\Ezero(k))}
\qquad
\frac{\hsame(h,k)}{\hsame(\Eone(h),\Eone(k))}.
\]
There is no rule relating \(\emp\) to a nonempty history, nor any rule relating \(\Ezero\)-histories to \(\Eone\)-histories.
\end{definition}

\begin{axiomlike}[Closed generation for histories]
History formation and history-sameness are generated only by their displayed rules. Inversion and no-confusion principles are justified by this closed generation.
\end{axiomlike}

\begin{lemma}[History-sameness reflexivity]
For every generated history \(h\), \(\hsame(h,h)\).
\end{lemma}
\begin{proof}
Induct on the generation of \(h\). The empty case uses the empty rule. The two extension cases use the induction hypothesis and the corresponding constructor rule for \(\hsame\).
\end{proof}

\begin{lemma}[History-sameness symmetry]
If \(\hsame(h,k)\), then \(\hsame(k,h)\).
\end{lemma}
\begin{proof}
Induct on the derivation of \(\hsame(h,k)\). Each constructor is symmetric by applying the induction hypothesis under the same history constructor.
\end{proof}

\begin{lemma}[History-sameness transitivity]
If \(\hsame(h,k)\) and \(\hsame(k,l)\), then \(\hsame(h,l)\).
\end{lemma}
\begin{proof}
Induct on the first derivation. The empty case forces the second derivation to be the empty case by no-confusion. The \(\Ezero\) case forces the second derivation to also be an \(\Ezero\) case; apply the induction hypothesis to the inner histories and rebuild. The \(\Eone\) case is analogous.
\end{proof}

\begin{theorem}[History-sameness equivalence]
The internal judgment \(\hsame\) is reflexive, symmetric, and transitive on generated histories.
\end{theorem}
\begin{proof}
Combine the preceding lemmas.
\end{proof}

\begin{theorem}[History no-confusion]
The kernel does not generate any of the following cross-constructor identifications:
\[
\hsame(\emp,\Ezero(h)),\quad
\hsame(\emp,\Eone(h)),\quad
\hsame(\Ezero(h),\Eone(k)),
\]
or their symmetric variants.
\end{theorem}
\begin{proof}
By closed generation, the only conclusions of \(\hsame\) have matching outer constructors. The displayed cross-constructor conclusions are not produced by any rule.
\end{proof}
