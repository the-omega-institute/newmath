\chapter{Thin Seal Interface}
\label{ch:core-thin-seal-interface}

\begin{definition}[Stage interface]
A stage interface \(\Stage\) is a generated visibility level in an iterative observation process. This chapter treats \(\Stage\) as a placeholder for the completion module.
\end{definition}
\leandef{BEDC.FKernel.NameCert.StageInterface}

\begin{definition}[Thread]
\label{def:thread}
A thread \(\Thread\) is a compatible family of stage-level visible data. Compatibility is not developed fully in this finite kernel; it is reserved for the completion module.
\end{definition}
\leandef{BEDC.FKernel.NameCert.ThreadFamily}

\begin{definition}[Seal interface]
\label{def:seal-interface}
\label{def:thin-seal-interface}
A seal event \(\Seal(T,N)\) licenses a name \(N\) for a compatible thread \(T\), provided a \(\SealCert\) and a ledger policy are supplied.
\end{definition}
\leandef{BEDC.FKernel.NameCert.SealEvent}
\leandef{BEDC.FKernel.NameCert.SealInterface}
\leanchecked{BEDC.FKernel.NameCert.sealEvent\_field\_witnesses}
\leanvariant{BEDC.FKernel.NameCert.sealInterface\_thread\_witness}
\leanvariant{BEDC.FKernel.NameCert.sealInterface\_field\_witnesses}
\leanvariant{BEDC.FKernel.NameCert.sealInterface\_certificate\_witnesses}
\leanvariant{BEDC.FKernel.NameCert.sealInterface\_stability\_witness}

\begin{corollary}[Limit-like interfaces are derived]
\label{cor:limit-like-interfaces-are-derived}
A \(\LimitUp\) interface is a derived name for seal-certified compatible-thread behavior. It is not primitive in the finite kernel.
\end{corollary}
\leanchecked{BEDC.FKernel.NameCert.sealInterface\_nameCert\_stability\_ledger}
\leanvariant{BEDC.FKernel.NameCert.limit\_like\_interfaces\_are\_derived}

\begin{remark}
The seal layer is intentionally thin in this chapter. Real-like, circle-like, and completion-like derived interfaces require a separate completion module.
\end{remark}
