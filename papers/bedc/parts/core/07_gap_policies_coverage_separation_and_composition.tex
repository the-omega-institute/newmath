\section{Gap policy}

\begin{definition}[Gap membership]
The judgment \(\InGap(\Pi,p,h)\) means that source history \(h\) lies in the gap ledger of package token \(p\) under bundle \(\Pi\).
\end{definition}

\begin{definition}[Gap policy]
\label{def:gap-policy}
\label{def:gap-policy-interface}
\label{def:gap-policy-generation}
A gap policy \(\GapPol(\Pi,D)\) over source specification \(D\) consists of:
\begin{enumerate}
  \item generation: \(\InGap(\Pi,p,h)\) is generated only from a signature \(\Sig(\Pi,h,s,\Delta)\) and a package \(\Pkg(\Pi,s,p)\);
  \item coverage: every admitted source history belongs to some package gap;
  \item separation: if the same admitted source history belongs to the gaps of \(p\) and \(q\), then \(\psame(p,q)\);
  \item ledger visibility: a package is not complete without its gap ledger.
\end{enumerate}
\end{definition}

\begin{theorem}[Gap coverage]
Assume \(\GapPol(\Pi,D)\). Every admitted source history belongs to at least one generated package gap.
\end{theorem}
\leanchecked{BEDC.FKernel.Gap.gap\_coverage}
\begin{proof}
This is the coverage field of \(\GapPol\), which directly supplies a package gap containing each admitted source history.
\end{proof}

\begin{theorem}[Gap separation]
Assume \(\GapPol(\Pi,D)\). If an admitted source history belongs to the gaps of package tokens \(p\) and \(q\), then \(\psame(p,q)\).
\end{theorem}
\leanchecked{BEDC.FKernel.Gap.gap\_separation}
\begin{proof}
This is the separation field of \(\GapPol\), applied to the two gap memberships of the same admitted source history.
\end{proof}

\section{Composite gaps}

\begin{definition}[Composite gap]
\label{def:composite-gap-ledger}
Suppose an interface \(C\) packages sources into intermediate visible tokens and an interface \(D\) packages those intermediate tokens into final visible tokens. The composite gap judgment
\[
\CompGap_{D\circ C}(z,x)
\]
means that source \(x\) lies behind final visible token \(z\) through some intermediate token.
Externally this is described by the existence of an intermediate token \(y\) with
\[
\InGap_C(y,x)
\quad\text{and}\quad
\InGap_D(z,y).
\]
This is a metalanguage description of the composite ledger policy.
\end{definition}

\begin{theorem}[Composite gap coverage]
\label{thm:composite-coverage}
\label{thm:composite-gap-coverage-spine}
If the first interface has gap coverage in the form that every admitted source $h$ has an intermediate token $y$ with $\InGap_C(y,h)$ and an admissibility certificate $\mathsf{InterOk}(y)$, and if the second interface has gap coverage for every intermediate token satisfying $\mathsf{InterOk}$, then the composite interface has gap coverage.
\end{theorem}
\leanchecked{BEDC.FKernel.Gap.compGap\_coverage\_from\_layers}
\leanvariant{BEDC.FKernel.Gap.compGap\_composite\_coverage\_witness}
\leanvariant{BEDC.FKernel.Gap.composite\_gap\_coverage\_spine}
\leanvariant{BEDC.FKernel.Gap.compGap\_coverage\_intermediate\_and\_final}
\leanvariant{BEDC.FKernel.Gap.compGap\_coverage\_with\_intermediate\_ok}
\leanvariant{BEDC.FKernel.Gap.compGap\_coverage\_final\_witness\_pair}
\begin{proof}
Given an admitted source, first coverage supplies both an intermediate token whose first gap contains the source and the admissibility certificate for that intermediate token. Second coverage applies to that certificate and produces a final token whose second gap contains the intermediate token. Thus the source belongs to the composite gap of the final token.
\end{proof}

\begin{theorem}[Composite gap separation]
\label{thm:composite-gap-separation}
If the first gap separates two intermediate witnesses over the same source by $\mathsf{interSame}$, and the second gap sends $\mathsf{interSame}$ intermediate witnesses to $\mathsf{finalSame}$ final tokens, then two composite memberships over the same source imply $\mathsf{finalSame}(z_1,z_2)$.
\end{theorem}
\leanchecked{BEDC.FKernel.Gap.compGap\_separation\_from\_layers}
\begin{proof}
Unpack the first composite membership into an intermediate witness $y_1$ with $\InGap_C(y_1,h)$ and $\InGap_D(z_1,y_1)$. Unpack the second composite membership into $y_2$ with $\InGap_C(y_2,h)$ and $\InGap_D(z_2,y_2)$. First-layer separation gives $\mathsf{interSame}(y_1,y_2)$, and second-layer separation carries this relation across the two second-layer memberships to obtain $\mathsf{finalSame}(z_1,z_2)$.
\end{proof}

\begin{theorem}[Composite exactness from layered exactness]
\label{thm:composite-exactness-from-layers}
Let $\mathsf{Source}$, $\mathsf{Inter}$, and $\mathsf{Final}$ be the source, intermediate, and final token types. Let $\mathsf{SourceOk}$ and $\mathsf{InterOk}$ be admissibility predicates, let $\mathsf{firstGap}$ and $\mathsf{secondGap}$ be the first and second gap relations, and let $\mathsf{interSame}$ and $\mathsf{finalSame}$ be the intermediate and final sameness relations. Assume:
$$
\forall x,\ \mathsf{SourceOk}(x) \to
\exists y,\ \mathsf{firstGap}(y,x)\ \text{and}\ \mathsf{InterOk}(y),
$$
$$
\forall y,\ \mathsf{InterOk}(y) \to
\exists z,\ \mathsf{secondGap}(z,y),
$$
$$
\forall x,y_1,y_2,\ \mathsf{SourceOk}(x) \to
\mathsf{firstGap}(y_1,x) \to
\mathsf{firstGap}(y_2,x) \to
\mathsf{interSame}(y_1,y_2),
$$
and
$$
\forall y_1,y_2,z_1,z_2,\ \mathsf{interSame}(y_1,y_2) \to
\mathsf{secondGap}(z_1,y_1) \to
\mathsf{secondGap}(z_2,y_2) \to
\mathsf{finalSame}(z_1,z_2).
$$
Then the composite gap satisfies both conclusions:
$$
\forall x,\ \mathsf{SourceOk}(x) \to
\exists z,\ \CompGap(\mathsf{firstGap},\mathsf{secondGap},z,x)
$$
and
$$
\forall x,z_1,z_2,\ \mathsf{SourceOk}(x) \to
\CompGap(\mathsf{firstGap},\mathsf{secondGap},z_1,x) \to
\CompGap(\mathsf{firstGap},\mathsf{secondGap},z_2,x) \to
\mathsf{finalSame}(z_1,z_2).
$$
\end{theorem}
\begin{proof}
The first conclusion is Theorem~\ref{thm:composite-coverage}: choose the first-layer intermediate witness and its $\mathsf{InterOk}$ certificate, then apply second-layer coverage to obtain a final token. The second conclusion is Theorem~\ref{thm:composite-gap-separation}: unpack the two composite witnesses over the same source, use first-layer separation to identify the intermediate witnesses by $\mathsf{interSame}$, and apply second-layer transport to obtain $\mathsf{finalSame}$ for the final tokens.
\end{proof}

\begin{principle}[Ledger composition]
\label{cor:ledger-composition-principle}
The ledger of a composite compression is not the last ledger alone. It is the first ledger together with the pullback of the second ledger along the first visible interface.
\end{principle}
