\begin{theorem}[Prefixed result common-suffix source determinacy]
\label{thm:cont-prefixed-result-common-suffix-source-deterministic}
For histories $p,a,b,k,r,s:\mathsf{BHist}$, if $\ContR(a,k,r)$,
$\ContR(b,k,s)$, and
$$
\hsame(\mathsf{append}(p,r),\mathsf{append}(p,s)),
$$
then $\hsame(a,b)$.
\end{theorem}
\begin{proof}
Left cancellation removes the visible prefix $p$ from the result comparison.
The two continuations then have a common right suffix $k$ and history-same
results, so right-context continuation cancellation identifies the sources.
\end{proof}
\leanchecked{BEDC.FKernel.Cont.cont\_prefixed\_result\_common\_suffix\_source\_deterministic}

\begin{theorem}[Prefixed result common-source tail determinacy]
\label{thm:cont-prefixed-result-common-source-tail-deterministic}
For histories $p,a,k,l,r,s:\mathsf{BHist}$, if $\ContR(a,k,r)$,
$\ContR(a,l,s)$, and
$$
\hsame(\mathsf{append}(p,r),\mathsf{append}(p,s)),
$$
then $\hsame(k,l)$.
\end{theorem}
\begin{proof}
Left cancellation removes the common visible prefix $p$ from the result
comparison. The two continuation witnesses now share source $a$ and have
history-same results, so left-context continuation cancellation identifies the
two tails.
\end{proof}
\leanchecked{BEDC.FKernel.Cont.cont\_prefixed\_result\_common\_source\_tail\_deterministic}

\begin{theorem}[Continuation fixed right-unit probe forces empty tail]
\label{thm:cont-fixed-right-unit-probe-empty-tail}
For all histories $p,u:\mathsf{BHist}$, if
$$
\forall r:\mathsf{BHist}.\; \ContR(p,u,r)\Rightarrow \HSame(r,p),
$$
then $\HSame(u,\emp)$.
\end{theorem}
\begin{proof}
Let $r_0=\mathsf{append}(p,u)$ be the canonical result representative for the displayed continuation tail. By the append presentation of $\ContR$ in \autoref{def:continuation-relation}, there is a continuation witness
$$
\ContR(p,u,r_0).
$$
The fixed-probe hypothesis applied to this witness gives $\HSame(r_0,p)$. The base rule of \autoref{def:continuation-relation} also gives the right-unit continuation
$$
\ContR(p,\emp,p).
$$
Apply \autoref{thm:cont-cancel-hsame-left-context} to these two continuations, using reflexivity for $\HSame(p,p)$ and the obtained result comparison $\HSame(r_0,p)$. The conclusion is $\HSame(u,\emp)$.
\end{proof}
\leanchecked{BEDC.FKernel.Cont.cont\_fixed\_right\_unit\_probe\_empty\_tail}

\begin{theorem}[Continuation fixed left-unit probe forces empty source]
\label{thm:cont-fixed-left-unit-probe-empty-source}
For all histories $u,p:\mathsf{BHist}$, if
$$
\forall r:\mathsf{BHist}.\; \ContR(u,p,r)\Rightarrow \HSame(r,p),
$$
then $\HSame(u,\emp)$.
\end{theorem}
\begin{proof}
Let $r_0=\mathsf{append}(u,p)$ be the canonical result representative for continuing source $u$ by tail $p$. By the append presentation of $\ContR$ in \autoref{def:continuation-relation}, there is a continuation witness
$$
\ContR(u,p,r_0).
$$
The fixed-probe hypothesis applied to this witness gives $\HSame(r_0,p)$. Transporting the canonical witness along this result comparison by \autoref{thm:cont-result-hsame-transport} gives
$$
\ContR(u,p,p).
$$
By the left-unit clause of \autoref{thm:continuation-unit-laws}, there is also a continuation witness
$$
\ContR(\emp,p,p).
$$
Apply the right-cancellation clause of \autoref{thm:continuation-right-cancellation} to these two continuations and the reflexive comparison $\HSame(p,p)$. The conclusion is $\HSame(u,\emp)$.
\end{proof}
\leanchecked{BEDC.FKernel.Cont.cont\_fixed\_left\_unit\_probe\_empty\_source}
