\section{Bundle grammar}

\begin{definition}[Probe bundle]
\label{def:probe-bundle}
\label{def:bundle-generation}
Probe bundles are generated by:
$$
\frac{}{\Bnil:\Bundle}
$$
$$
\frac{\pi:\PN\qquad \Pi:\Bundle}{\Bcons(\pi,\Pi):\Bundle}.
$$
Bundle membership is the closed predicate generated by cons membership: the head probe of
\(\Bcons(\pi,\Pi)\) is a member of that bundle, and tail membership propagates through cons.
\end{definition}

\begin{definition}[Bundle append]
\label{def:bundle-append}
The append operation $\mathsf{BAppend}(\Pi,\Theta)$ is recursion on the left bundle:
$$
\mathsf{BAppend}(\Bnil,\Theta)=\Theta,
\qquad
\mathsf{BAppend}(\Bcons(\pi,\Pi),\Theta)
=\Bcons(\pi,\mathsf{BAppend}(\Pi,\Theta)).
$$
\end{definition}
\leandef{BEDC.FKernel.Bundle.bundleAppend}

\begin{definition}[Probe bundle length]
\label{def:bundle-length}
The length $\mathsf{BLEN}(\Pi)$ of a generated probe bundle is recursion on the bundle:
$$
\mathsf{BLEN}(\Bnil)=0,
\qquad
\mathsf{BLEN}(\Bcons(\pi,\Pi))=1+\mathsf{BLEN}(\Pi).
$$
\end{definition}
\leandef{BEDC.FKernel.Bundle.bundleLength}

\begin{theorem}[Probe bundle zero length equivalence]
\label{thm:probe-bundle-length-zero-iff-nil}
A generated probe bundle has length zero exactly when it is the empty bundle
$\Bnil$.
\end{theorem}
\begin{proof}
Eliminate the generated bundle. The empty case reduces by computation. In the
cons case the length reduces to a successor, which cannot equal zero.
\end{proof}
\leanchecked{BEDC.FKernel.Bundle.bundleLength\_eq\_zero\_iff\_nil}

\begin{lemma}[Probe bundle zero length exactly empty]
\label{lem:probe-bundle-length-zero-iff-empty}
For every generated probe bundle $\Pi$,
$$
\mathsf{BLEN}(\Pi)=0
\quad\Longleftrightarrow\quad
\Pi=\Bnil.
$$
\end{lemma}
\begin{proof}
Case split on the generated bundle. The empty case is definitional. In the cons
case, the length has successor form, contradicting equality with zero. The
reverse implication follows by substituting $\Bnil$.
\end{proof}
\leanchecked{BEDC.FKernel.Bundle.bundleLength\_zero\_iff\_nil}

\begin{theorem}[Probe bundle length one singleton equivalence]
\label{thm:probe-bundle-length-one-iff-singleton}
A generated probe bundle has length one exactly when it is a singleton
$\Bcons(\pi,\Bnil)$ for some probe $\pi$.
\end{theorem}
\begin{proof}
Eliminate the generated bundle. The empty case has length zero, so it cannot have length one. In the cons case, injectivity of successor forces the tail length to be zero; Theorem~\ref{thm:probe-bundle-length-zero-iff-nil} identifies the tail with $\Bnil$. Conversely, substituting a singleton bundle reduces the length equation by computation.
\end{proof}
\leanchecked{BEDC.FKernel.Bundle.bundleLength\_eq\_one\_iff\_singleton}

\begin{theorem}[Probe bundle successor length cons decomposition]
\label{thm:bundle-length-succ-iff-cons}
A generated probe bundle has successor length exactly when it is a cons whose
tail has the predecessor length.
\end{theorem}
\begin{proof}
Eliminate the bundle. The empty case contradicts successor length, while the
cons case uses successor injectivity; the converse substitutes the cons equation.
\end{proof}
\leanchecked{BEDC.FKernel.Bundle.bundleLength\_eq\_succ\_iff\_cons}

\begin{theorem}[Probe bundle append length]
\label{thm:probe-bundle-append-length}
For all probe bundles $\Pi,\Theta$,
$$
\mathsf{BLEN}(\mathsf{BAppend}(\Pi,\Theta))=\mathsf{BLEN}(\Pi)+\mathsf{BLEN}(\Theta).
$$
\end{theorem}
\begin{proof}
Induct on $\Pi$. In the empty case, the append term reduces to $\Theta$ and the length equation reduces to the left identity of addition. In the cons case, both sides reduce by one successor, and the tail equation is the induction hypothesis.
\end{proof}
\leanchecked{BEDC.FKernel.Bundle.bundleLength\_append}
\leanchecked{BEDC.FKernel.Bundle.bundleAppend\_length}
\leanchecked{BEDC.FKernel.Bundle.bundleLength\_bundleAppend}

\begin{lemma}[Probe bundle nonempty prefix length separation]
\label{lem:probe-bundle-nonempty-prefix-length-separation}
For every probe $\pi$ and probe bundles $\Pi_0,\Omega$,
$$
\mathsf{BLEN}(\mathsf{BAppend}(\Bcons(\pi,\Pi_0),\Omega))
\neq
\mathsf{BLEN}(\Omega).
$$
Consequently,
$$
\mathsf{BAppend}(\Bcons(\pi,\Pi_0),\Omega)\neq\Omega.
$$
\end{lemma}
\begin{proof}
Theorem~\ref{thm:probe-bundle-append-length} rewrites the left length as
$1+\mathsf{BLEN}(\Pi_0)+\mathsf{BLEN}(\Omega)$. A natural number with an exposed successor constructor cannot be equal to the unextended suffix length by natural-number constructor no-confusion. If the appended bundle itself were equal to $\Omega$, applying $\mathsf{BLEN}$ to that equality would contradict the displayed length separation.
\end{proof}
\leanchecked{BEDC.FKernel.Bundle.bundleAppend\_nonempty\_prefix\_length\_separation}

\begin{theorem}[Probe bundle append fixes the right side exactly at nil left]
\label{thm:bundle-append-eq-right-iff-left-nil}
For probe bundles $\Pi$ and $\Omega$,
$$
\mathsf{BAppend}(\Pi,\Omega)=\Omega
$$
if and only if $\Pi=\Bnil$.
\end{theorem}
\begin{proof}
Split on $\Pi$. The nil case is definitional. In the cons case, the
nonempty-prefix length separation theorem says that appending a cons-prefix to
$\Omega$ cannot equal $\Omega$. The reverse implication follows by reducing
the append definition at $\Bnil$.
\end{proof}
\leanchecked{BEDC.FKernel.Bundle.bundleAppend\_eq\_right\_iff\_left\_nil}

\begin{theorem}[Probe bundle append right unit]
\label{thm:bundle-append-right-unit}
For every probe bundle \(\Pi\),
$$
\mathsf{BAppend}(\Pi,\Bnil)=\Pi .
$$
\end{theorem}
\begin{proof}
Induct on \(\Pi\). The empty bundle case is definitional. In the cons case,
both sides expose the same head probe, and the tail equality is the induction
hypothesis.
\end{proof}
\leanchecked{BEDC.FKernel.Bundle.bundleAppend\_nil\_right}

\begin{theorem}[Probe bundle append associativity]
\label{thm:probe-bundle-append-associativity}
For all probe bundles \(\Pi,\Theta,\Omega\),
$$
\mathsf{BAppend}(\mathsf{BAppend}(\Pi,\Theta),\Omega)
=
\mathsf{BAppend}(\Pi,\mathsf{BAppend}(\Theta,\Omega)).
$$
\end{theorem}
\begin{proof}
Induct on \(\Pi\). The empty bundle case is definitional. In the cons case,
both sides expose the same head probe, and the tail equality is the induction
hypothesis.
\end{proof}
\leanchecked{BEDC.FKernel.Bundle.bundleAppend\_assoc}

\begin{theorem}[Bundle append membership]
\label{thm:bundle-append-membership}
For every probe $p$ and bundles $\Pi,\Theta$,
$$
p\in\mathsf{BAppend}(\Pi,\Theta)
\quad\Longleftrightarrow\quad
p\in\Pi \ \text{or}\ p\in\Theta .
$$
\end{theorem}
\leanchecked{BEDC.FKernel.Bundle.inBundle\_bundleAppend\_iff}
\begin{proof}
Induct on $\Pi$. The empty case reduces membership to $\Theta$. The cons case splits head
membership from tail membership, then applies the induction hypothesis to the tail.
\end{proof}

\begin{theorem}[Bundle append membership left residual]
\label{thm:bundle-append-membership-left-of-not-right}
For every probe \(p\) and bundles \(\Pi,\Theta\), if
$$
p\in\mathsf{BAppend}(\Pi,\Theta)
$$
and \(p\notin\Theta\), then \(p\in\Pi\).
\end{theorem}
\begin{proof}
Induct on \(\Pi\). In the empty case, appended membership is membership in
\(\Theta\), contradicting the exclusion premise. In the cons case, either the
probe is the head of \(\Pi\), or it lies in the appended tail. The head branch is
immediate, and the tail branch follows from the induction hypothesis with the
same exclusion premise.
\end{proof}
\leanchecked{BEDC.FKernel.Bundle.inBundle\_bundleAppend\_left\_of\_not\_right}

\begin{theorem}[Bundle-local policy append restriction]
\label{thm:bundle-local-policy-append-restriction}
If $\mathsf{BundleAskPolicy}(\mathsf{BAppend}(\Pi,\Theta),D)$ holds, then both $\mathsf{BundleAskPolicy}(\Pi,D)$ and $\mathsf{BundleAskPolicy}(\Theta,D)$ hold.
\end{theorem}
\leanchecked{BEDC.FKernel.Ask.bundleAskPolicy\_append\_restriction}
\begin{proof}
Use Theorem~\ref{thm:bundle-append-membership}. A member of $\Pi$ maps into $\mathsf{BAppend}(\Pi,\Theta)$ through the left disjunct, and a member of $\Theta$ maps into $\mathsf{BAppend}(\Pi,\Theta)$ through the right disjunct. Restrict the totality, mark determinacy, and history-respect fields of the appended bundle-local policy along these two membership maps.
\end{proof}

\begin{theorem}[Bundle-local policy append gluing]
\label{thm:bundle-local-policy-append-gluing}
If $\mathsf{BundleAskPolicy}(\Pi,D)$ and $\mathsf{BundleAskPolicy}(\Theta,D)$ hold, then $\mathsf{BundleAskPolicy}(\mathsf{BAppend}(\Pi,\Theta),D)$ holds.
\end{theorem}
\leanchecked{BEDC.FKernel.Ask.bundleAskPolicy\_append\_gluing}
\begin{proof}
Use Theorem~\ref{thm:bundle-append-membership}. For each probe membership in $\mathsf{BAppend}(\Pi,\Theta)$, split it into a left membership in $\Pi$ or a right membership in $\Theta$. In the left case, take totality, determinacy, and history-respect from the policy on $\Pi$; in the right case, take the same three fields from the policy on $\Theta$. These case analyses assemble the bundle-local asking policy for the appended bundle.
\end{proof}

\begin{theorem}[Bundle-local policy append equivalence]
\label{thm:bundle-local-policy-append-iff}
For all probe bundles $\Pi,\Theta$ and domain $D$,
$$
\mathsf{BundleAskPolicy}(\mathsf{BAppend}(\Pi,\Theta),D)
\Longleftrightarrow
\bigl(\mathsf{BundleAskPolicy}(\Pi,D)\ \text{and}\ \mathsf{BundleAskPolicy}(\Theta,D)\bigr).
$$
\end{theorem}
\leanchecked{BEDC.FKernel.Ask.bundleAskPolicy\_append\_iff}
\begin{proof}
The forward direction is exactly Theorem~\ref{thm:bundle-local-policy-append-restriction}. The reverse direction is exactly Theorem~\ref{thm:bundle-local-policy-append-gluing}.
\end{proof}

\begin{theorem}[Probe bundle prefix cancellation]
\label{thm:probe-bundle-prefix-cancellation}
For any fixed probe bundle $\Pi$ and any probe bundles $\Theta,\Omega$, if
$$
\mathsf{BAppend}(\Pi,\Theta)=\mathsf{BAppend}(\Pi,\Omega),
$$
then $\Theta=\Omega$.
\end{theorem}
\leanchecked{BEDC.FKernel.Bundle.bundleAppend\_prefix\_cancel}
\begin{proof}
Induct on $\Pi$. The empty case is immediate because both displayed append terms
reduce to their suffix bundles. In the cons case, constructor injectivity reduces
the equality of cons bundles to equality of their tails, and the induction
hypothesis cancels the remaining common prefix.
\end{proof}

\begin{theorem}[Probe bundle suffix cancellation]
\label{thm:probe-bundle-suffix-cancellation}
For any probe bundles $\Pi,\Theta,\Omega$, if
$$
\mathsf{BAppend}(\Pi,\Omega)=\mathsf{BAppend}(\Theta,\Omega),
$$
then $\Pi=\Theta$.
\end{theorem}
\begin{proof}
Induct on $\Pi$ and then case-split $\Theta$. The empty-empty case is immediate. In the empty-cons and cons-empty cases, Lemma~\ref{lem:probe-bundle-nonempty-prefix-length-separation} rules out equality between a bundle obtained by appending a nonempty prefix to $\Omega$ and the suffix $\Omega$. In the cons-cons case, the displayed equality is an equality of cons bundles. Constructor injectivity identifies the two head probes and reduces the tail equality to
$$
\mathsf{BAppend}(\Pi_0,\Omega)=\mathsf{BAppend}(\Theta_0,\Omega).
$$
The induction hypothesis gives $\Pi_0=\Theta_0$, and the head equality rebuilds $\Pi=\Theta$.
\end{proof}
\leanchecked{BEDC.FKernel.Bundle.bundleAppend\_suffix\_cancel}

\begin{lemma}[Probe bundle append empty-result inversion]
\label{lem:probe-bundle-append-empty-result-inversion}
For probe bundles $\Pi,\Theta$, if
$$
\mathsf{BAppend}(\Pi,\Theta)=\Bnil,
$$
then $\Pi=\Bnil$ and $\Theta=\Bnil$.
\end{lemma}
\leanchecked{BEDC.FKernel.Bundle.bundleAppend\_empty\_result\_inversion}
\begin{proof}
Proceed by cases on $\Pi$. In the empty case the displayed equality reduces to $\Theta=\Bnil$. In the cons case the append definition exposes a cons bundle equal to $\Bnil$, contradicting probe-bundle constructor no-confusion.
\end{proof}

\begin{lemma}[Probe bundle append cons-result inversion]
\label{lem:probe-bundle-append-cons-result-inversion}
For probe bundles $\Pi,\Theta,\Omega$ and probe $\pi$, if
$$
\mathsf{BAppend}(\Pi,\Theta)=\Bcons(\pi,\Omega),
$$
then either $\Pi=\Bnil$ and $\Theta=\Bcons(\pi,\Omega)$, or there exists a bundle $\Pi_0$ such that $\Pi=\Bcons(\pi,\Pi_0)$ and
$$
\mathsf{BAppend}(\Pi_0,\Theta)=\Omega.
$$
\end{lemma}
\begin{proof}
Proceed by cases on $\Pi$. In the empty case the displayed equality reduces to $\Theta=\Bcons(\pi,\Omega)$, giving the left alternative. In the cons case the append definition exposes a cons bundle on the left; constructor injectivity identifies its head with $\pi$ and its appended tail with $\Omega$, giving the required tail bundle $\Pi_0$.
\end{proof}
\leanchecked{BEDC.FKernel.Bundle.bundleAppend\_cons\_result\_inversion}

\begin{theorem}[Probe bundle append split uniqueness at fixed left length]
\label{thm:bundle-append-split-unique-fixed-length}
For probe bundles $\Pi,\Theta,\Pi',\Theta'$, if
$$
\mathsf{BAppend}(\Pi,\Theta)=\mathsf{BAppend}(\Pi',\Theta')
$$
and
$$
\mathsf{BLEN}(\Pi)=\mathsf{BLEN}(\Pi'),
$$
then $\Pi=\Pi'$ and $\Theta=\Theta'$.
\end{theorem}
\begin{proof}
Proceed by simultaneous cases on the two left bundles. The empty-empty case reduces the append equality to the suffix equality $\Theta=\Theta'$. The empty-cons and cons-empty cases contradict the displayed length equality by natural-number constructor no-confusion. In the cons-cons case, constructor injectivity on the append equality identifies the two head probes and gives equality of the appended tails; constructor injectivity on the length equality gives equal tail lengths. The induction hypothesis applied to the tails gives equality of the left tails and suffixes, and the head equality rebuilds $\Pi=\Pi'$.
\end{proof}
\leanchecked{BEDC.FKernel.Bundle.bundleAppend\_split\_unique\_fixed\_length}
