\section{Signature generation}

\begin{definition}[Signature event]
\label{def:signature-event}
The signature event \(\Sig(\Pi,h,s,\Delta)\) means that bundle \(\Pi\) applied to history \(h\) generates answer-history \(s\) with evidence \(\Delta\). It is generated by the rules below.
In the checked kernel, the logical predicate is the three-place $\mathsf{SigRel}(\Pi,h,s)$ on bundle, source history, and answer history. The displayed evidence ledger is a paper-side annotation attached to the generating derivation.
\end{definition}

\begin{kernelrule}[Empty signature]
$$
\frac{}{\Sig(\Bnil,h,\emp,\Delta_0)}
$$
where \(\Delta_0\) is the neutral signature evidence token.
\end{kernelrule}

\begin{lemma}[Empty signature inversion]
\label{lem:empty-signature-inversion}
If \(\Sig(\Bnil,h,r,\Delta_0)\), then \(\hsame(r,\emp)\).
\end{lemma}
\leanchecked{BEDC.FKernel.Sig.sig\_empty\_inversion}
\begin{proof}
Invert the signature derivation. For the empty bundle, only the empty-signature rule applies, so the result history is \(\emp\).
\end{proof}

\begin{kernelrule}[Cons signature]
If
$$
\Ask(\pi,h,m,\delta),
\qquad
\Sig(\Pi,h,s,\Delta),
\qquad
\ExtR(m,s,r),
$$
then
$$
\Sig(\Bcons(\pi,\Pi),h,r,\delta\star\Delta).
$$
Here \(\delta\star\Delta\) records the head asking evidence together with the tail derivation ledger; it is not a field of the asking policy.
\end{kernelrule}

\begin{theorem}[Cons signature result is nonempty]
\label{thm:signature-cons-result-ne-empty}
If a cons bundle \(\Bcons(\pi,\Pi)\) generates a result history \(r\), then \(r\)
cannot be internally the same as \(\emp\).
\end{theorem}
\begin{proof}
Invert the cons signature derivation. The final step is an extension witness
\(\ExtR(m,s,r)\). Each extension constructor produces either an \(\Ezero\) or
an \(\Eone\) result, and BEDC history no-confusion separates both constructors
from \(\emp\).
\end{proof}
\leanchecked{BEDC.FKernel.Sig.sig\_cons\_result\_ne\_empty}

\begin{theorem}[Signature result length equals bundle length]
\label{thm:signature-result-length-bundle-length}
If \(\Sig(\Pi,h,r,\Delta)\), then the external binary length of the answer
history \(r\) is the probe-bundle length of \(\Pi\).
\end{theorem}
\leanchecked{BEDC.FKernel.Sig.sigRel\_result\_length\_eq\_bundleLength}
\begin{proof}
Induct on the signature derivation. The empty rule gives zero length on both
sides. In the cons rule, the final extension adds exactly one outer history
constructor, and the bundle is one cons longer than its tail.
\end{proof}

\begin{theorem}[Empty signature result forces empty bundle]
\label{thm:signature-empty-result-bundle-nil}
If \(\Sig(\Pi,h,\emp,\Delta)\), then \(\Pi=\Bnil\).
\end{theorem}
\leanchecked{BEDC.FKernel.Sig.sigRel\_empty\_result\_bundle\_nil}
\begin{proof}
The result-length theorem gives bundle length zero because \(\emp\) has zero
external binary length. Probe-bundle length zero characterizes \(\Bnil\).
\end{proof}

\begin{theorem}[Signature generation over bundle append]
\label{thm:signature-generation-bundle-append}
For probe bundles $\Pi,\Theta$ and source history $h$, if $\Sig(\Pi,h,s,\Delta)$ and $\Sig(\Theta,h,t,\Gamma)$, then there exist a history $u$ and evidence $\Lambda$ such that
$$
\ContR(t,s,u)
\quad\text{and}\quad
\Sig(\mathsf{BAppend}(\Pi,\Theta),h,u,\Lambda).
$$
\end{theorem}
\leanchecked{BEDC.FKernel.Sig.signature\_generation\_bundle\_append}
\begin{proof}
Induct on $\Pi$. In the empty bundle case, empty-signature inversion identifies $s$ with $\emp$; take $u=t$, use the right-unit continuation constructor, and reuse the given signature for $\Theta$. In the cons case, invert the left signature. The tail signature for $\Pi$ and the signature for $\Theta$ feed the induction hypothesis, producing a tail result $u_0$ with $\ContR(t,s_0,u_0)$ and a signature for $\mathsf{BAppend}(\Pi,\Theta)$. Rebuild the cons signature using the original head asking event and the tail signature. The final extension branch supplies either the $\Ezero$ or $\Eone$ continuation step, yielding $\ContR(t,s,u)$ for the rebuilt result.
\end{proof}

\begin{theorem}[Signature generation append inversion]
\label{thm:signature-generation-bundle-append-inversion}
For probe bundles $\Pi,\Theta$ and source history $h$, if
$\Sig(\mathsf{BAppend}(\Pi,\Theta),h,u,\Lambda)$, then there exist histories
$s,t$ and evidence tokens $\Delta,\Gamma$ such that
$\Sig(\Pi,h,s,\Delta)$, $\Sig(\Theta,h,t,\Gamma)$, and $\ContR(t,s,u)$.
\end{theorem}
\leanchecked{BEDC.FKernel.Sig.signature\_generation\_bundle\_append\_inversion}
\begin{proof}
Induct on $\Pi$. In the empty bundle case, the appended bundle is $\Theta$;
generate the empty signature for $\Bnil$, take $s=\emp$ and $t=u$, reuse the
given signature for $\Theta$, and apply the continuation right-unit
constructor. In the cons case, invert the appended cons signature. The
inversion exposes the head asking event, the tail signature for
$\mathsf{BAppend}(\Pi,\Theta)$, and the final extension witness. Apply the
induction hypothesis to the tail append signature to obtain signatures for
$\Pi$ and $\Theta$ together with a continuation $\ContR(t,s_0,u_0)$. Rebuild
the left cons signature from the exposed head asking event, the recovered tail
signature for $\Pi$, and the final extension witness. The same extension
constructor lifts the recovered tail continuation to $\ContR(t,s,u)$ for the
whole appended signature.
\end{proof}

\begin{theorem}[Signature bundle-append decomposition]
\label{thm:signature-bundleappend-decomposition}
For probe bundles $\Pi,\Theta$ and source history $h$,
$$
\Sig(\mathsf{BAppend}(\Pi,\Theta),h,u,\Lambda)
\Longleftrightarrow
\exists s,t,\Delta,\Gamma.\;
\Sig(\Pi,h,s,\Delta)\land
\Sig(\Theta,h,t,\Gamma)\land
\ContR(t,s,u).
$$
\end{theorem}
\leanchecked{BEDC.FKernel.Sig.sigRel\_bundleAppend\_decomposition\_iff}
\begin{proof}
The forward implication is append-signature inversion. Conversely, generate
the canonical appended signature from the two component signatures. Its
continuation has the same displayed endpoints as $\ContR(t,s,u)$, so
continuation determinacy identifies the generated result with $u$.
\end{proof}

\begin{theorem}[Signature generation append component determinacy]
\label{thm:signature-generation-append-component-determinacy}
For probe bundles $\Pi,\Theta$, domain $D$, and source history $h$, assume
$\mathsf{BundleAskPolicy}(\Pi,D)$, $\mathsf{BundleAskPolicy}(\Theta,D)$, and
$h\in D$. If
$$
\Sig(\Pi,h,s,\Delta),
\qquad
\Sig(\Theta,h,t,\Gamma),
\qquad
\Sig(\mathsf{BAppend}(\Pi,\Theta),h,u,\Lambda),
$$
then there exists a history $v$ such that
$$
\ContR(t,s,v)
\quad\text{and}\quad
\hsame(v,u).
$$
\end{theorem}
\leanchecked{BEDC.FKernel.Sig.signature\_generation\_append\_component\_determinacy}
\begin{proof}
Theorem~\ref{thm:bundle-local-policy-append-gluing} gives
$\mathsf{BundleAskPolicy}(\mathsf{BAppend}(\Pi,\Theta),D)$ from the two component policies. Theorem~\ref{thm:signature-generation-bundle-append} applied to the two component signatures produces a history $v$ with $\ContR(t,s,v)$ and a generated signature for the appended bundle. Theorem~\ref{thm:bundle-local-signature-determinacy}, applied to the glued policy, the premise $h\in D$, that constructed appended signature, and the given appended signature for $u$, gives $\hsame(v,u)$.
\end{proof}

\begin{theorem}[Signature generation bundle append recomposition determinacy]
\label{thm:sigrel-bundleappend-recompose-hsame}
For probe bundles \(\Pi,\Theta\), domain \(D\), and source history \(h\), assume
an asking policy on \(D\) and \(h\in D\). If
$$
\Sig(\Pi,h,s,\Delta),
\qquad
\Sig(\Theta,h,t,\Gamma),
\qquad
\ContR(t,s,u),
$$
and \(\Sig(\mathsf{BAppend}(\Pi,\Theta),h,v,\Lambda)\), then
$$
\hsame(u,v).
$$
\end{theorem}
\begin{proof}
Generate the canonical appended signature from the two component signatures by
Theorem~\ref{thm:signature-generation-bundle-append}. Its continuation witness
has the same endpoints as the displayed \(\ContR(t,s,u)\), so continuation
determinacy identifies the displayed recomposition with the canonical result.
The asking-policy signature determinacy theorem compares the canonical appended
signature with the supplied appended signature for \(v\). Transitivity of
\(\hsame\) gives the displayed result.
\end{proof}
\leanchecked{BEDC.FKernel.Sig.sigRel\_bundleAppend\_recompose\_hsame}

\begin{theorem}[Three-bundle signature generation coherence]
\label{thm:signature-generation-three-bundle-coherence}
For probe bundles $\Pi,\Theta,\Omega$ and source history $h$, assume
$\Sig(\Pi,h,s,\Delta)$, $\Sig(\Theta,h,t,\Gamma)$, and
$\Sig(\Omega,h,w,\Xi)$. Then there exist histories $a,b,u_L,u_R$ and evidence
ledgers $\Lambda_{\Pi\Theta}$, $\Lambda_L$, $\Lambda_{\Theta\Omega}$, and
$\Lambda_R$ such that
\begin{enumerate}
  \item $\Sig(\mathsf{BAppend}(\Pi,\Theta),h,a,\Lambda_{\Pi\Theta})$ and $\ContR(t,s,a)$;
  \item $\Sig(\mathsf{BAppend}(\mathsf{BAppend}(\Pi,\Theta),\Omega),h,u_L,\Lambda_L)$ and $\ContR(w,a,u_L)$;
  \item $\Sig(\mathsf{BAppend}(\Theta,\Omega),h,b,\Lambda_{\Theta\Omega})$ and $\ContR(w,t,b)$;
  \item $\Sig(\mathsf{BAppend}(\Pi,\mathsf{BAppend}(\Theta,\Omega)),h,u_R,\Lambda_R)$ and $\ContR(b,s,u_R)$;
  \item $\hsame(u_L,u_R)$.
\end{enumerate}
\end{theorem}
\leanchecked{BEDC.FKernel.Sig.signature\_generation\_three\_bundle\_coherence}
\begin{proof}
Apply Theorem~\ref{thm:signature-generation-bundle-append} to the $\Pi$ and
$\Theta$ signatures to obtain $a$ and $\Lambda_{\Pi\Theta}$ with
$\ContR(t,s,a)$. Apply the same theorem to the resulting
$\mathsf{BAppend}(\Pi,\Theta)$ signature and the $\Omega$ signature to obtain
$u_L$ and $\Lambda_L$ with $\ContR(w,a,u_L)$. For the other grouping, first
apply Theorem~\ref{thm:signature-generation-bundle-append} to the $\Theta$ and
$\Omega$ signatures to obtain $b$ and $\Lambda_{\Theta\Omega}$ with
$\ContR(w,t,b)$, then apply it to the $\Pi$ signature and the
$\mathsf{BAppend}(\Theta,\Omega)$ signature to obtain $u_R$ and $\Lambda_R$ with
$\ContR(b,s,u_R)$. Theorem~\ref{thm:continuation-associativity}, with
$h=w$, $k=t$, $l=s$, middle result $b$, and alternative middle result $a$,
gives $\hsame(u_R,u_L)$. The history-sameness symmetry lemma converts this to
the displayed $\hsame(u_L,u_R)$.
\end{proof}

\begin{theorem}[Signature-sameness closure under bundle append]
\label{thm:samesig-bundle-append-closure}
For probe bundles $\Pi,\Theta$ and histories $h,k$, if $\sameSig_\Pi(h,k)$ and $\sameSig_\Theta(h,k)$, then
$$
\sameSig_{\mathsf{BAppend}(\Pi,\Theta)}(h,k).
$$
\end{theorem}
\begin{proof}
Unpack the two signature-sameness witnesses. The $\Pi$ witnesses give signatures from $h$ and $k$ whose answer histories are $\hsame$; the $\Theta$ witnesses give another such pair. Apply Theorem~\ref{thm:signature-generation-bundle-append} to the two $h$-side signatures and separately to the two $k$-side signatures. Continuation respect for $\hsame$ transports the two answer-history sameness witnesses across the appended continuations, so the appended signatures have $\hsame$ results. These appended signatures are the witnesses for $\sameSig_{\mathsf{BAppend}(\Pi,\Theta)}(h,k)$.
\end{proof}
\leanchecked{BEDC.FKernel.Sig.sameSig\_bundleAppend\_closure}

\begin{theorem}[Signature-sameness left cancellation under bundle append]
\label{thm:samesig-bundle-append-left-cancellation}
Assume $\mathsf{BundleAskPolicy}(\Pi,D)$ and $h,k\in D$. If
$$
\sameSig_{\mathsf{BAppend}(\Pi,\Theta)}(h,k)
$$
and $\sameSig_\Pi(h,k)$, then
$$
\sameSig_\Theta(h,k).
$$
\end{theorem}
\begin{proof}
Invert the appended signature-sameness witnesses using Theorem~\ref{thm:signature-generation-bundle-append-inversion}. This recovers $\Pi$ signatures, $\Theta$ signatures, and continuation witnesses on the $h$ and $k$ sides. Compare the recovered left $\Pi$ signatures with the witnesses from $\sameSig_\Pi(h,k)$ using Theorem~\ref{thm:bundle-local-signature-determinacy} at $h$ and at $k$. Transport one continuation along that left-result sameness. The common right continuation can then be cancelled against the appended result-sameness, yielding $\hsame$ for the recovered $\Theta$ results. Those recovered $\Theta$ signatures are the witnesses for $\sameSig_\Theta(h,k)$.
\end{proof}
\leanchecked{BEDC.FKernel.Sig.sameSig\_bundleAppend\_left\_cancellation}

\begin{theorem}[Signature-sameness right cancellation under bundle append]
\label{thm:samesig-bundle-append-right-cancellation}
Assume $\mathsf{BundleAskPolicy}(\Theta,D)$ and $h,k\in D$. If
$$
\sameSig_{\mathsf{BAppend}(\Pi,\Theta)}(h,k)
$$
and $\sameSig_\Theta(h,k)$, then
$$
\sameSig_\Pi(h,k).
$$
\end{theorem}
\begin{proof}
Invert the appended signature-sameness witnesses using Theorem~\ref{thm:signature-generation-bundle-append-inversion}. This recovers $\Pi$ signatures, $\Theta$ signatures, and continuation witnesses on the $h$ and $k$ sides. Compare the recovered $\Theta$ signatures with the supplied witnesses from $\sameSig_\Theta(h,k)$ using Theorem~\ref{thm:bundle-local-signature-determinacy} at $h$ and at $k$. Determinacy for the supplied and recovered $\Theta$ signatures gives $\hsame$ between the two recovered $\Theta$ continuation contexts. Continuation result transport then aligns one continuation so both continuations have the same left context and the same result up to $\hsame$. The left-context cancellation clause of Theorem~\ref{thm:continuation-right-cancellation} applies to these aligned continuations and yields $\hsame$ for the recovered $\Pi$ results. Those recovered $\Pi$ signatures are the witnesses for $\sameSig_\Pi(h,k)$.
\end{proof}
\leanchecked{BEDC.FKernel.Sig.sameSig\_bundleAppend\_right\_cancellation}

\begin{theorem}[Known component residual for signature-sameness under bundle append]
\label{thm:samesig-bundle-append-residual-iff}
For probe bundles $\Pi,\Theta$ and histories $h,k$, the following residual equivalences hold.
\begin{enumerate}
  \item If $\mathsf{BundleAskPolicy}(\Pi,D)$, $h,k\in D$, and $\sameSig_\Pi(h,k)$, then
  $$
  \sameSig_{\mathsf{BAppend}(\Pi,\Theta)}(h,k)\Longleftrightarrow\sameSig_\Theta(h,k).
  $$
  \item If $\mathsf{BundleAskPolicy}(\Theta,D)$, $h,k\in D$, and $\sameSig_\Theta(h,k)$, then
  $$
  \sameSig_{\mathsf{BAppend}(\Pi,\Theta)}(h,k)\Longleftrightarrow\sameSig_\Pi(h,k).
  $$
\end{enumerate}
\end{theorem}
\begin{proof}
For the first equivalence, the forward direction is Theorem~\ref{thm:samesig-bundle-append-left-cancellation}, using the known $\Pi$ component and the bundle-local policy on $\Pi$. The reverse direction combines the known $\sameSig_\Pi(h,k)$ with the supplied $\sameSig_\Theta(h,k)$ by Theorem~\ref{thm:samesig-bundle-append-closure}. The second equivalence is symmetric: its forward direction is Theorem~\ref{thm:samesig-bundle-append-right-cancellation}, and its reverse direction is again the append closure theorem.
\end{proof}
\leanchecked{BEDC.FKernel.Sig.sameSig\_bundleAppend\_residual\_iff}

\begin{theorem}[Signature-sameness common-context residual under bundle append]
\label{thm:samesig-bundleappend-common-context-iff}
Assume $\mathsf{BundleAskPolicy}(\Pi,D)$, $\mathsf{BundleAskPolicy}(\Omega,D)$,
$h,k\in D$, $\sameSig_\Pi(h,k)$, and $\sameSig_\Omega(h,k)$. Then
$$
\sameSig_{\mathsf{BAppend}(\Pi,\mathsf{BAppend}(\Theta,\Omega))}(h,k)
\Longleftrightarrow
\sameSig_\Theta(h,k).
$$
\end{theorem}
\begin{proof}
For the forward direction, cancel the known left component $\Pi$ from the outer
append using the residual equivalence, leaving
$\sameSig_{\mathsf{BAppend}(\Theta,\Omega)}(h,k)$. Then cancel the known right
component $\Omega$ using the right residual equivalence, yielding
$\sameSig_\Theta(h,k)$.

For the reverse direction, first append the supplied middle witness with the
known right witness $\sameSig_\Omega(h,k)$, then append the known left witness
$\sameSig_\Pi(h,k)$ to rebuild the full common-context signature-sameness row.
\end{proof}
\leanchecked{BEDC.FKernel.Sig.sameSig\_bundleAppend\_common\_context\_iff}

\begin{theorem}[Cons tail signatures from result-sameness]
\label{thm:signature-cons-tail-hsame-from-result-hsame}
If two derivations of \(\Sig(\Bcons(\pi,\Pi),h,r,\Delta)\) and
\(\Sig(\Bcons(\pi,\Pi),h,r',\Theta)\) have \(\hsame(r,r')\), then there exist
tail results \(s,t\) such that \(\Sig(\Pi,h,s,\Delta_s)\), \(\Sig(\Pi,h,t,\Delta_t)\), and
\(\hsame(s,t)\).
\end{theorem}
\leanchecked{BEDC.FKernel.Sig.sig\_cons\_tail\_hsame\_from\_result\_hsame}
\begin{proof}
Invert both cons-signature derivations and their final extension witnesses. Cross-mark extension branches contradict result-sameness by history no-confusion, while matching marks expose matching outer history constructors; constructor injectivity yields \(\hsame\) of the tail results.
\end{proof}

\begin{theorem}[Bundle-local signature existence]
\label{thm:bundle-local-signature-existence}

\end{theorem}
\begin{proof}
Induct on $\Pi$. The empty bundle emits the empty signature. In the cons case, the head occurrence satisfies $\pi\in\Bcons(\pi,\Pi)$, so bundle-local totality supplies the head asking event. The tail policy is obtained by restricting membership from $\rho\in\Pi$ to $\rho\in\Bcons(\pi,\Pi)$, and the induction hypothesis supplies the tail signature. The emitted head mark then selects the matching extension constructor.
\end{proof}
\leanchecked{BEDC.FKernel.Sig.sig\_total\_from\_bundle\_policy}

\begin{theorem}[Bundle-local history-sameness gives signature-sameness]
\label{thm:bundle-local-samesig-of-hsame}
Assume $\mathsf{BundleAskPolicy}(\Pi,D)$. If $h,k\in D$ and $\hsame(h,k)$, then $\sameSig_\Pi(h,k)$.
\end{theorem}
\begin{proof}
Apply bundle-local signature existence to $h$ and to $k$, obtaining generated signatures
$$
\Sig(\Pi,h,s,\Delta),
\qquad
\Sig(\Pi,k,t,\Theta).
$$
Bundle-local signature source respect transports the premise $\hsame(h,k)$ across these two generated signatures and yields $\hsame(s,t)$. The two signature witnesses together with this generated-history sameness form $\sameSig_\Pi(h,k)$.
\end{proof}
\leanchecked{BEDC.FKernel.Sig.bundle\_local\_sameSig\_of\_hsame}

\begin{theorem}[Bundle-local signature-sameness equivalence]
\label{thm:bundle-local-signature-sameness-equivalence}
Assume $\mathsf{BundleAskPolicy}(\Pi,D)$. If $h\in D$, then
$\sameSig_\Pi(h,h)$; if $\sameSig_\Pi(h,k)$, then $\sameSig_\Pi(k,h)$;
and if $k\in D$, $\sameSig_\Pi(h,k)$, and $\sameSig_\Pi(k,l)$, then
$\sameSig_\Pi(h,l)$.
\end{theorem}
\begin{proof}
For reflexivity, bundle-local signature existence gives a signature generated
from the admitted history $h$, and $\hsame$-reflexivity relates that generated
answer history to itself. For symmetry, unpack the two signature witnesses,
swap their order, and apply $\hsame$-symmetry. For transitivity through $k$,
unpack both signature-sameness hypotheses. The admitted middle history $k$
has two generated signature results; bundle-local signature determinacy
identifies them by $\hsame$. Composing with $\hsame$-transitivity connects the
left answer history to the right answer history, yielding $\sameSig_\Pi(h,l)$.
\end{proof}
\leanchecked{BEDC.FKernel.Sig.bundle\_local\_signature\_sameness\_equivalence}

\Sig(\Pi,h,s,\Delta),

The induction hypothesis gives $\hsame$ of the tail signatures. Extension respects mark-sameness and history-sameness, so the resulting answer histories are $\hsame$.
\end{proof}
