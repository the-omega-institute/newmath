\section{Purpose of the model}
The following model is external. It is a metalanguage interpretation showing that the kernel rules are not empty. It is not part of the object-language.

\begin{definition}[External model \(\Mbin\)]
\label{def:external-model-mbin}
The external model \(\Mbin\) interprets:
\begin{itemize}
  \item raw marks as two external tokens;
  \item histories as finite binary words;
  \item \(\msame\) as literal token equality;
  \item \(\hsame\) as literal word equality;
  \item \(\ExtR\) as appending one token;
  \item \(\ContR\) as word concatenation;
  \item asking events as externally chosen binary observations;
  \item signatures as finite words of answers;
  \item packages as signature classes;
  \item gaps as inverse images of package tokens;
  \item composite gaps as relational composites of inverse images.
\end{itemize}
\end{definition}

\begin{theorem}[External append respects history-sameness]
\label{thm:external-append-hsame-congruence}
For all external words $a,a',b,b'$, if $\hsame(a,a')$ and $\hsame(b,b')$, then
$$
\hsame(append(a,b),append(a',b')).
$$
\end{theorem}
\begin{proof}
Unfold external words as $BHist$. The two $\hsame$ hypotheses identify the left and right inputs. After those identifications, both appended histories are the same continuation append result, so $\hsame$-reflexivity gives the conclusion.
\end{proof}

\begin{theorem}[External append cancellation soundness]
\label{thm:external-append-cancellation-soundness}
Right and left cancellation for external append are sound for \(\hsame\): if appending the same right word or the same left word gives internally same results, then the uncancelled words are internally same.
\end{theorem}

\begin{theorem}[External append bit result inversion]
\label{thm:external-append-bit-result-inversion}
If an external append result is a zero-bit word, then either the right word is empty and the left word is that zero-bit word, or the right word starts with zero and the tail append is the result tail. The same alternative holds for one-bit words.
\end{theorem}

\leanvariant{BEDC.FKernel.ExternalBinary.external\_append\_bit0\_result\_inversion}
\leanvariant{BEDC.FKernel.ExternalBinary.external\_append\_bit1\_result\_inversion}
\leanvariant{BEDC.FKernel.ExternalBinary.external\_append\_bit\_result\_tail\_hsame}
\leanvariant{BEDC.FKernel.ExternalBinary.external\_append\_bit\_result\_cross\_impossible}
\leanvariant{BEDC.FKernel.ExternalBinary.external\_append\_bit\_result\_cases\_by\_right\_word}
\leanvariant{BEDC.FKernel.ExternalBinary.external\_append\_bit\_result\_iff}
\leanvariant{BEDC.FKernel.ExternalBinary.external\_append\_constructor\_result\_inversion}
\leanvariant{BEDC.FKernel.ExternalBinary.external\_append\_length\_zero\_pair}
\begin{proof}
Case-analyze the right appended word. The empty case returns the left word. Matching bit constructors expose the tail equality by constructor injectivity, and opposite bit constructors are impossible by constructor disjointness.
\end{proof}

\begin{corollary}[External append bit tail equality]
\label{cor:external-append-bit-tail-equality}
If $append(a,b)$ is a zero-bit result $0::r$ and $b$ is a zero-bit word $0::b_0$, then $append(a,b_0)=r$. If $append(a,b)$ is a one-bit result $1::r$ and $b$ is a one-bit word $1::b_1$, then $append(a,b_1)=r$.
\end{corollary}
\leanchecked{BEDC.FKernel.ExternalBinary.external\_append\_bit\_result\_tail\_eq\_pair}
\begin{proof}
In each bit case, substitute the displayed shape of $b$ into the append equality. Constructor injectivity for the matching bit exposes equality of the tails.
\end{proof}
