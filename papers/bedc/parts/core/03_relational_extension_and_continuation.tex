\section{Relational extension}

\begin{definition}[Extension relation]
\label{def:extension-relation}
The extension relation \(\ExtR(m,h,r)\) is generated by the two rules:
$$
\frac{}{\ExtR(\bzero,h,\Ezero(h))}
\qquad
\frac{}{\ExtR(\bone,h,\Eone(h))}.
$$
It is a relation, not a function.
\end{definition}

\begin{theorem}[Extension result constructor characterization]
\label{thm:extension-result-constructor-characterization}
For all histories $h,t$ and mark $m$,
$$
\ExtR(m,h,\Ezero(t))\Longleftrightarrow (m=\bzero\ \text{and}\ \hsame(h,t)).
$$
For the one-constructor case,
$$
\ExtR(m,h,\Eone(t))\Longleftrightarrow (m=\bone\ \text{and}\ \hsame(h,t)).
$$
\end{theorem}
\leanchecked{BEDC.FKernel.Ext.ext\_result\_constructor\_iff}
\begin{proof}
Invert the generated extension witness. In the zero-result branch, closed
generation forces the emitted mark to be \(\bzero\) and the source to be the
tail up to history sameness; the converse reconstructs the zero extension after
eliminating those equalities. The one-result branch is identical with the
\(\Eone\) constructor.
\end{proof}

\begin{lemma}[Cross-mark extension results are separated]
\label{lem:extension-cross-mark-result-separation}
If \(\ExtR(\bzero,h,r_0)\), \(\ExtR(\bone,h,r_1)\), and \(\hsame(r_0,r_1)\), then contradiction follows.
\end{lemma}
\leanchecked{BEDC.FKernel.Ext.ext\_same\_source\_cross\_mark\_results\_not\_hsame}
\begin{proof}
Invert the two extension witnesses. The result histories are \(\Ezero(h)\) and \(\Eone(h)\), and history no-confusion excludes \(\hsame(\Ezero(h),\Eone(h))\).
\end{proof}

\begin{theorem}[Extension result is not the source]
\label{thm:extension-result-source-not-hsame}
If \(\ExtR(m,h,r)\), then \(\hsame(r,h)\) is impossible.
\end{theorem}
\leanchecked{BEDC.FKernel.Ext.ext\_result\_source\_not\_hsame}
\begin{proof}
Invert the generated extension witness. The result is either \(\Ezero(h)\) or
\(\Eone(h)\), and both cases are excluded by history constructor
no-confusion for a one-constructor self-extension.
\end{proof}

\begin{theorem}[Extension two-step cycle is impossible]
\label{thm:extension-two-step-cycle-absurd}
If \(\ExtR(m,h,r)\) and \(\ExtR(n,r,h)\), then contradiction follows.
\end{theorem}
\leanchecked{BEDC.FKernel.Ext.ext\_two\_step\_cycle\_absurd}
\begin{proof}
Use the finite constructor length of histories. Each extension increases this
length by one. Two extensions returning from \(h\) to \(h\) would force the
length of \(h\) to equal its own double successor, which is impossible by
natural-number constructor no-confusion.
\end{proof}

\begin{theorem}[Extension result injectivity]
\label{thm:ext-result-injective-pair}
If \(\ExtR(m,h,r)\) and \(\ExtR(m',h',r)\), then \(\hsame(h,h')\) and \(\msame(m,m')\).
\end{theorem}
\leanchecked{BEDC.FKernel.Ext.ext\_result\_injective\_pair}
\begin{proof}
Inspect the two generated extension witnesses. A common result can only arise from matching extension constructors, so the source histories and emitted marks are internally same.
\end{proof}

\begin{theorem}[Extension result hsame injectivity]
\label{thm:extension-result-hsame-injective-pair}
If \(\ExtR(m,h,r)\), \(\ExtR(m',h',r')\), and \(\hsame(r,r')\), then \(\hsame(h,h')\) and \(\msame(m,m')\).
\end{theorem}
\leanchecked{BEDC.FKernel.Ext.ext\_result\_hsame\_injective\_pair}
\begin{proof}
Invert both generated extension witnesses. Matching outer constructors reduce
the result-sameness witness to source history sameness and diagonal
mark-sameness. Mixed outer constructors contradict history no-confusion.
\end{proof}

\begin{lemma}[Extension respects mark-sameness]
If \(\msame(m,n)\), \(\ExtR(m,h,r)\), and \(\ExtR(n,h,s)\), then \(\hsame(r,s)\).
\end{lemma}
\leanchecked{BEDC.FKernel.Ext.ext\_respects\_mark\_sameness}
\begin{proof}
By closed generation for \(\msame\), either both marks are \(\bzero\) or both are \(\bone\). In the first case both extension results are \(\Ezero(h)\); in the second case both are \(\Eone(h)\). Use reflexivity of \(\hsame\).
\end{proof}

\begin{theorem}[Extension determinacy up to history-sameness]
If \(\ExtR(m,h,r)\) and \(\ExtR(m,h,s)\), then \(\hsame(r,s)\).
\end{theorem}
\leanchecked{BEDC.FKernel.Ext.ext\_deterministic}
\begin{proof}
Apply the previous lemma with \(\msame(m,m)\), using mark-sameness reflexivity.
\end{proof}

\begin{definition}[Continuation relation]
\label{def:continuation-relation}
\label{def:continuation-step-rules}
The continuation relation \(\ContR(h,k,r)\) is generated by:
$$
\frac{}{\ContR(h,\emp,h)}
$$
$$
\frac{\ContR(h,k,r)}{\ContR(h,\Ezero(k),\Ezero(r))}
\qquad
\frac{\ContR(h,k,r)}{\ContR(h,\Eone(k),\Eone(r))}.
$$
It is a relation, not a primitive function.
\end{definition}

\begin{theorem}[Append nonempty factor equivalence]
\label{thm:append-nonempty-iff}
For histories $h$ and $k$, the appended history is not internally empty
exactly when at least one factor is not internally empty:
$$
\neg\HSame(\mathsf{append}(h,k),\emp)
\Longleftrightarrow
\neg\HSame(h,\emp)\vee\neg\HSame(k,\emp).
$$
\end{theorem}
\begin{proof}
The forward direction splits on the left factor: the empty branch reduces the
obstruction to the right factor, while visible left constructors are separated
from $\emp$. Conversely, empty-result inversion contradicts either nonempty
disjunct.
\end{proof}
\leanchecked{BEDC.FKernel.Cont.append\_nonempty\_iff}

\begin{theorem}[Append visible-left empty exclusion]
\label{thm:append-visible-left-empty-absurd}
For histories $h,k:\mathsf{BHist}$, appending any right tail to a visible left endpoint cannot produce an internally empty history:
both $\HSame(\mathsf{append}(\Ezero(h),k),\emp)\to\bot$ and $\HSame(\mathsf{append}(\Eone(h),k),\emp)\to\bot$ hold.
\end{theorem}
\begin{proof}
Split on the right tail $k$. If it is empty, the visible left endpoint remains
the outer constructor. If it is visible, append exposes that right-tail
constructor as the outer constructor. In every branch, constructor separation
from $\emp$ rules out the displayed history sameness.
\end{proof}
\leanchecked{BEDC.FKernel.Cont.append\_visible\_left\_empty\_absurd}

\begin{theorem}[Append visible-left same-tail tag separation]
\label{thm:append-visible-left-same-tail-tag-separation}
For histories $h_0,h_1,k:\mathsf{BHist}$, common right-tail append preserves
separation between zero-headed and one-headed visible left endpoints:
$$
\HSame(\mathsf{append}(\Ezero(h_0),k),
\mathsf{append}(\Eone(h_1),k))\to\bot.
$$
\end{theorem}
\begin{proof}
Induct on the common right tail $k$. The empty case is direct zero/one
constructor no-confusion. In each visible tail step, both appended histories
gain the same outer constructor; constructor injectivity peels that shared
outer layer and the induction hypothesis applies to the shorter tail.
\end{proof}
\leanchecked{BEDC.FKernel.Cont.append\_visible\_left\_same\_tail\_tag\_separation}

\begin{theorem}[Append visible-left same-tag tail cancellation]
\label{thm:append-visible-left-same-tag-tail-cancel}
For histories $h_0,h_1,k:\mathsf{BHist}$, common right-tail append cancellation identifies the payloads of equal zero-headed visible-left endpoints, and likewise for equal one-headed visible-left endpoints.
\end{theorem}
\begin{proof}
Apply append right-cancellation to the common tail, then use constructor injectivity for the matching \(\Ezero\) or \(\Eone\) head.
\end{proof}
\leanchecked{BEDC.FKernel.Cont.append\_visible\_left\_same\_tag\_tail\_cancel}

\begin{lemma}[Continuation respects a matching right extension step]
\label{lem:cont-ext-right-step}
If $\ContR(h,k,r)$, $\ExtR(m,k,k')$, and $\ExtR(m,r,r')$, then
$\ContR(h,k',r')$.
\end{lemma}
\leanchecked{BEDC.FKernel.Cont.cont\_ext\_right\_step}
\begin{proof}
Case split on the extension of $k$ by $m$. In the $\Ezero$ branch, the matching extension of $r$ is also the zero branch, and the zero continuation step applies. The $\Eone$ branch is analogous.
\end{proof}

\begin{theorem}[Extension as singleton continuation]
\label{thm:extension-as-one-step-continuation}
For every history $h$ and result $r$, $\ExtR(\bzero,h,r)$ if and only if
$\ContR(h,\Ezero(\emp),r)$, and $\ExtR(\bone,h,r)$ if and only if
$\ContR(h,\Eone(\emp),r)$.
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.extension\_as\_singleton\_continuation}
\leanchecked{BEDC.FKernel.Cont.ext\_singleton\_continuation\_iff\_pair}
\begin{proof}
For the forward zero branch, invert the extension witness to identify
$r$ with $\Ezero(h)$, use the empty-continuation rule
$\ContR(h,\emp,h)$, and apply the $\Ezero$ continuation step rule. The one
branch is the same argument with $\Eone$.

For the reverse zero branch, invert the continuation witness for
$\ContR(h,\Ezero(\emp),r)$. Step-result inversion gives a predecessor
continued by $\emp$, and history constructor no-confusion forces the
singleton history to be the displayed $\Ezero(\emp)$ branch. The
empty-continuation rule identifies the predecessor with $h$, so the
$\bzero$ extension constructor yields $\ExtR(\bzero,h,r)$. The one branch
uses the corresponding $\Eone$ inversion and the $\bone$ extension
constructor.
\end{proof}

\begin{theorem}[Continuation step result no-confusion]
\label{thm:cont-step-result-no-confusion-pair}
No generated history \(r\) can be both the result of continuing by
\(\Ezero(k_0)\) and the result of continuing by \(\Eone(k_1)\) from the same
source history. The converse ordering of the two step constructors is equally
impossible.
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_step\_result\_no\_confusion\_pair}
\begin{proof}
Invert the two continuation witnesses. The zero-step witness forces the common
result to have outer constructor \(\Ezero\), while the one-step witness forces
the same result to have outer constructor \(\Eone\). Closed generation of
histories gives constructor no-confusion. The converse direction is symmetric.
\end{proof}

\begin{theorem}[Continuation step result is nonempty]
\label{thm:continuation-step-result-not-empty}
For all histories $h,k$, neither $\ContR(h,\Ezero(k),\emp)$ nor
$\ContR(h,\Eone(k),\emp)$ can hold.
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_step\_result\_not\_empty\_pair}
\begin{proof}
Unfold the step result shape for $\ContR$. The zero branch would identify $\emp$ with
$\Ezero(r)$, and the one branch would identify $\emp$ with $\Eone(r)$. Both contradict the
generated history constructors.
\end{proof}

\begin{theorem}[Zero-headed continuation result characterization]
\label{thm:cont-result-e0-cases-iff}
For histories $h,k,r:\mathsf{BHist}$,
$$
\ContR(h,k,\Ezero(r))
\Longleftrightarrow
(k=\emp\land \hsame(h,\Ezero(r)))
\lor
(\exists k_0:\mathsf{BHist}.\; k=\Ezero(k_0)\land \ContR(h,k_0,r)).
$$
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_result\_e0\_cases\_iff}
\begin{proof}
Classify the right continuation history $k$. The empty branch is exactly the
right-unit endpoint equality between $h$ and $\Ezero(r)$. A zero-headed tail
exposes its predecessor and removes the matching outer constructor from the
continuation equation. A one-headed tail would identify zero-headed and
one-headed generated histories. Conversely, the empty branch is the right-unit
continuation after the displayed history sameness is eliminated, and the
zero-headed branch is rebuilt by the zero step rule.
\end{proof}

\begin{theorem}[Continuation result tag separation]
\label{thm:continuation-result-tag-separation}
There are no histories $h,k,r_0,r_1$ such that both
$\ContR(h,k,\Ezero(r_0))$ and $\ContR(h,k,\Eone(r_1))$ hold.
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_result\_tag\_separation}
\begin{proof}
Assume both continuation witnesses for the same inputs $h$ and $k$.
Continuation determinacy gives $\hsame(\Ezero(r_0),\Eone(r_1))$. History
no-confusion rules out any generated sameness derivation between a zero-tagged
history and a one-tagged history, so the two witnesses cannot coexist.
\end{proof}

\begin{theorem}[Left one-headed continuation result cases]
\label{thm:cont-left-e1-result-cases}
If $\ContR(\Eone(h),k,\Eone(r))$, then either $k=\emp$ and
$\HSame(h,r)$, or $k=\Eone(k_0)$ for some $k_0$ and

one-headed history to equal a zero-headed history, contradicting constructor
no-confusion. A one-headed $k$ exposes its predecessor $k_0$ and removes the
matching outer constructor from the continuation equation.
\end{proof}

\begin{theorem}[Left cross-tag continuation result cases]
\label{thm:cont-left-tag-cross-result-cases}
If $\ContR(\Ezero(h),k,\Eone(r))$, then $k=\Eone(k_0)$ for some $k_0$

\end{proof}

\begin{theorem}[Left cross-tag continuation result characterization]
\label{thm:cont-left-tag-cross-result-iff-pair}
For histories \(h,k,r\), the two cross-tag result shapes are characterized by

$$
\ContR(\Ezero(h),k,\Eone(r))
\Longleftrightarrow
\exists k_0:\mathsf{BHist}.\;k=\Eone(k_0)\land\ContR(\Ezero(h),k_0,r),
$$
and

\ContR(\Eone(h),k,\Ezero(r))

\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_left\_tag\_cross\_result\_iff\_pair}
\begin{proof}

result constructor using the corresponding continuation step rule.
\end{proof}

\begin{theorem}[Left same-tag continuation tail alignment]
\label{thm:cont-left-same-tag-tail-alignment}
If two continuations have the same visible left endpoint and the same
same-tag visible result, then their right tails align in the same shape. For
zero-headed endpoints, both tails are empty or both are zero-headed with
$\HSame$ exposed tails; for one-headed endpoints, both tails are empty or both
are one-headed with $\HSame$ exposed tails.
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_left\_same\_tag\_tail\_alignment}
\begin{proof}
Apply continuation left cancellation to the two displayed continuations to
obtain $\HSame(k,l)$ between their right tails. Split the first continuation by
the same-tag result case theorem. In the empty branch, the tail sameness forces
the second tail to be empty. In the visible branch, tail sameness transports
the second tail into the matching visible constructor and yields the exposed
$\HSame$ relation between payload tails.
\end{proof}

\begin{theorem}[Left cross-tag continuation tail alignment]
\label{thm:cont-left-tag-cross-tail-alignment}
If two continuations have the same visible left endpoint and the same
cross-tag visible result, then both right tails carry the opposite visible tag,
and the exposed tails are $\HSame$.
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_left\_tag\_cross\_tail\_alignment}
\begin{proof}
Continuation left cancellation compares the two right tails. The cross-tag
result case theorem applied to the first continuation forces that tail to have
the opposite visible constructor. Transporting the second tail through the
tail sameness forces the same constructor there as well, and constructor
inversion gives the exposed $\HSame$ relation.
\end{proof}

\begin{theorem}[Continuation right-tail result cases]
\label{thm:cont-right-tail-result-cases}
For histories $h,k,r:\mathsf{BHist}$, if $\ContR(h,k,r)$, then the right
continuation tail has one of the following displayed forms:
$$
\HSame(k,\emp)\land\HSame(r,h),
$$
or there exist histories $t,u$ such that
$$
\HSame(k,\Ezero(t))\land\HSame(r,\Ezero(u))\land\ContR(h,t,u),
$$
or there exist histories $t,u$ such that
$$
\HSame(k,\Eone(t))\land\HSame(r,\Eone(u))\land\ContR(h,t,u).
$$
\end{theorem}
\begin{proof}
Classify the right continuation tail $k$. The empty case is the right-unit
continuation, so the result is classified with the source. In the zero-headed
and one-headed cases, step-result inversion removes the displayed outer
constructor from the result and exposes the residual continuation through the
tail payload.
\end{proof}
\leanchecked{BEDC.FKernel.Cont.cont\_right\_tail\_result\_cases}

\begin{lemma}[Continuation right unit]
For every history \(h\), \(\ContR(h,\emp,h)\).
\end{lemma}
\begin{proof}
This is the base rule of \(\ContR\).
\end{proof}

\begin{lemma}[Continuation left unit]

\begin{theorem}[Continuation determinacy up to history-sameness]
If \(\ContR(h,k,r)\) and \(\ContR(h,k,s)\), then \(\hsame(r,s)\).
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_deterministic}
\begin{proof}
Induct on the first derivation. In the empty case the second derivation is also the empty case by no-confusion, and the result follows by reflexivity. In the two extension cases, no-confusion forces the second derivation to use the same outer continuation constructor; apply the induction hypothesis and rebuild \(\hsame\) with the matching history constructor.
\end{proof}

\begin{theorem}[Continuation result transport]
\label{thm:cont-result-hsame-transport}
If \(\ContR(h,k,r)\) and \(\hsame(r,r')\), then \(\ContR(h,k,r')\).
\end{theorem}
\begin{proof}
Unfold \(\ContR\) as the append-result equation. Symmetry of the result sameness followed by the original continuation equation gives the required append-result equation for \(r'\).
\end{proof}
\leanchecked{BEDC.FKernel.Cont.cont\_result\_hsame\_transport}

\begin{theorem}[Continuation hsame transport]
\label{thm:cont-hsame-transport}
If $\hsame(h,h')$, $\hsame(k,k')$, $\hsame(r,r')$, and $\ContR(h,k,r)$, then
$\ContR(h',k',r')$.
\end{theorem}
\begin{proof}
Eliminate the two input $\hsame$ witnesses to align the left and right
continuation inputs. The remaining result endpoint is transported by
Theorem~\ref{thm:cont-result-hsame-transport}.
\end{proof}
\leanchecked{BEDC.FKernel.Cont.cont\_hsame\_transport}

\begin{theorem}[Continuation transport result classification]

\end{theorem}
\begin{proof}

\begin{theorem}[Continuation left-context cancellation up to history-sameness]
\label{thm:cont-cancel-hsame-left-context}
If $\ContR(a,b,r)$, $\ContR(a',d,r')$, $\hsame(a,a')$, and $\hsame(r,r')$,
then $\hsame(b,d)$.
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_cancel\_hsame\_left\_context}
\begin{proof}
Use $\hsame(a,a')$ to align the left context and $\hsame(r,r')$ to align the
result. The two continuation derivations then have the same left context and
the same result, so left-context cancellation gives $\hsame(b,d)$.
\end{proof}

\begin{theorem}[Continuation prefix cancellation]
\label{thm:continuation-prefix-cancel}
If $\ContR(\mathsf{append}(p,a),f,\mathsf{append}(p,b))$, then
$\ContR(a,f,b)$.
\end{theorem}
\begin{proof}
Read the displayed continuation as equality of append representatives. Append
associativity rewrites the right side as $\mathsf{append}(p,\mathsf{append}(a,f))$.
Left cancellation of the common prefix $p$ identifies $b$ with
$\mathsf{append}(a,f)$, which is exactly the unprefixed continuation.
\end{proof}
\leanchecked{BEDC.FKernel.Cont.cont\_prefix\_cancel}

\begin{theorem}[Continuation prefix equivalence]
\label{thm:cont-prefix-iff}
For histories $p,a,b,f$,
$$
\ContR(\mathsf{append}(p,a),f,\mathsf{append}(p,b))
\Longleftrightarrow
\ContR(a,f,b).
$$
\end{theorem}
\begin{proof}
The forward direction is continuation prefix cancellation. For the reverse
direction, prefix the equality witness for $\ContR(a,f,b)$ by
$\mathsf{append}(p,-)$ and use associativity to identify
$\mathsf{append}(p,\mathsf{append}(a,f))$ with
$\mathsf{append}(\mathsf{append}(p,a),f)$.
\end{proof}
\leanchecked{BEDC.FKernel.Cont.cont\_prefix\_iff}

\begin{theorem}[Append right-context cancellation up to history-sameness]
\label{thm:append-hsame-right-context-cancel}
For histories $P,Q,a,b:\mathsf{BHist}$, if $P\hsame Q$, then appending the
history-same right contexts $P$ and $Q$ preserves and reflects endpoint
history-sameness:
$$
P\hsame Q\Rightarrow
\bigl(
\hsame(\mathsf{append}(a,P),\mathsf{append}(b,Q))
\Longleftrightarrow
a\hsame b
\bigr).
$$
\end{theorem}
\begin{proof}
Transport the right context along $P\hsame Q$. The forward direction is right append cancellation for the common suffix. Conversely, after the endpoints $a$ and $b$ are identified, the two appended histories are definitionally the same.
\end{proof}
\leanchecked{BEDC.FKernel.Cont.append\_hsame\_right\_context\_cancel\_iff}

\begin{theorem}[Append common-context cancellation up to history-sameness]
\label{thm:append-hsame-common-context-cancel}
For histories $L,R,P,Q,a,b:\mathsf{BHist}$, if $L\hsame R$ and $P\hsame Q$,
then the two-sided contextual append comparison is exactly the comparison of
the middle histories:
$$
L\hsame R\Rightarrow P\hsame Q\Rightarrow
\bigl(
\hsame(\mathsf{append}(L,\mathsf{append}(a,P)),
       \mathsf{append}(R,\mathsf{append}(b,Q)))
\Longleftrightarrow
a\hsame b
\bigr).
$$
\end{theorem}
\begin{proof}
First transport the left and right contexts along the two $\hsame$ witnesses.
In the forward direction, left append cancellation removes the common outer
context and right append cancellation removes the common suffix. The reverse
direction is immediate after the middle histories are identified.
\end{proof}
\leanchecked{BEDC.FKernel.Cont.append\_hsame\_common\_context\_cancel\_iff}

\begin{theorem}[Continuation cancellation laws]
\label{thm:continuation-right-cancellation}
Continuation admits cancellation on both sides:
\begin{enumerate}
  \item if $\ContR(h,k,r)$, $\ContR(h',k,r')$, and $\hsame(r,r')$, then $\hsame(h,h')$;
  \item if $\ContR(h,k,r)$ and $\ContR(h,k',r)$, then $\hsame(k,k')$;
  \item if $\ContR(a,b,ab)$, $\ContR(ab,c,left)$, $\ContR(a,d,ad)$, $\ContR(ad,c,right)$, and $\hsame(left,right)$, then $\hsame(b,d)$.
\end{enumerate}
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_right\_cancel}
\leanvariant{BEDC.FKernel.Cont.cont\_right\_cancel\_eq}
\leanvariant{BEDC.FKernel.Cont.cont\_common\_suffix\_cancellation}
\leanvariant{BEDC.FKernel.Cont.cont\_right\_cancel\_hsame\_result}
\leanvariant{BEDC.FKernel.Cont.cont\_cancel\_common\_context}
\leanvariant{BEDC.FKernel.Cont.continuation\_right\_cancellation}
\leanvariant{BEDC.FKernel.Cont.cont\_left\_cancel}
\leanvariant{BEDC.FKernel.Cont.append\_right\_cancel}
\leanvariant{BEDC.FKernel.Cont.append\_left\_cancel}
\begin{proof}
For the first clause, use right-suffix cancellation after transporting the result by $\hsame$. The second clause is left-context cancellation: compare the two derivations with the same left history, peel the shared left context, and conclude that the right histories are equal up to $\hsame$. For the third clause, associate the two continuations so both sides expose the same outer context $c$ over the inner continuations from $a$; the common outer context cancels against the result-sameness, and the remaining left-context cancellation yields $\hsame(b,d)$.
\end{proof}

\begin{theorem}[Continuation cancellation with history-same right context]
\label{thm:cont-cancel-hsame-right-context}
For histories $a,a',b,b',r,r':\mathsf{BHist}$, if $\ContR(a,b,r)$,
$\ContR(a',b',r')$, $\hsame(b,b')$, and $\hsame(r,r')$, then $\hsame(a,a')$.
\end{theorem}
\begin{proof}
Transport the second right context and result along the two $\hsame$ witnesses.
The continuations now share right context and result, so right cancellation
identifies the two sources.
\end{proof}

\begin{theorem}[Self-extension by a visible tail is impossible]
\label{thm:cont-self-extension-tail-absurd}
For any histories $h$ and $t$, neither $\ContR(h,\Ezero(t),h)$ nor
$\ContR(h,\Eone(t),h)$ can hold.
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_self\_extension\_tail\_absurd}
\begin{proof}
Any such continuation would make a visible right tail behave as the empty
right unit at $h$. Right-unit uniqueness identifies the tail with $\emp$,
contradicting constructor no-confusion for $\Ezero(t)$ and $\Eone(t)$.
\end{proof}

\begin{theorem}[Mutual continuations cannot start with a visible left tail]
\label{thm:cont-mutual-extension-left-tail-absurd}
For histories $h,k,t,u$, a continuation $\ContR(h,\Ezero(t),k)$ together
with a return continuation $\ContR(k,u,h)$ is impossible; the same holds with
$\Eone(t)$ in place of $\Ezero(t)$.
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_mutual\_extension\_left\_tail\_absurd}
\begin{proof}
Mutual continuations force both displayed tails to be empty by the concrete
tail-empty cancellation theorem. A left tail with outer constructor
$\Ezero$ or $\Eone$ cannot be internally same as $\emp$.
\end{proof}
