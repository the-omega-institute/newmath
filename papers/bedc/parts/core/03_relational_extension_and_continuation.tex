\section{Relational extension}

\begin{definition}[Extension relation]
\label{def:extension-relation}
The extension relation \(\ExtR(m,h,r)\) is generated by the two rules:
$$
\frac{}{\ExtR(\bzero,h,\Ezero(h))}
\qquad
\frac{}{\ExtR(\bone,h,\Eone(h))}.
$$
It is a relation, not a function.
\end{definition}

\begin{theorem}[Extension result constructor characterization]
\label{thm:extension-result-constructor-characterization}
For all histories $h,t$ and mark $m$,
$$
\ExtR(m,h,\Ezero(t))\Longleftrightarrow (m=\bzero\ \text{and}\ \hsame(h,t)).
$$
For the one-constructor case,
$$
\ExtR(m,h,\Eone(t))\Longleftrightarrow (m=\bone\ \text{and}\ \hsame(h,t)).
$$
\end{theorem}
\leanchecked{BEDC.FKernel.Ext.ext\_result\_constructor\_iff}
\begin{proof}
Invert the generated extension witness. In the zero-result branch, closed
generation forces the emitted mark to be \(\bzero\) and the source to be the
tail up to history sameness; the converse reconstructs the zero extension after
eliminating those equalities. The one-result branch is identical with the
\(\Eone\) constructor.
\end{proof}

\begin{lemma}[Cross-mark extension results are separated]
\label{lem:extension-cross-mark-result-separation}
If \(\ExtR(\bzero,h,r_0)\), \(\ExtR(\bone,h,r_1)\), and \(\hsame(r_0,r_1)\), then contradiction follows.
\end{lemma}
\leanchecked{BEDC.FKernel.Ext.ext\_same\_source\_cross\_mark\_results\_not\_hsame}
\begin{proof}
Invert the two extension witnesses. The result histories are \(\Ezero(h)\) and \(\Eone(h)\), and history no-confusion excludes \(\hsame(\Ezero(h),\Eone(h))\).
\end{proof}

\begin{theorem}[Extension result is not the source]
\label{thm:extension-result-source-not-hsame}
If \(\ExtR(m,h,r)\), then \(\hsame(r,h)\) is impossible.
\end{theorem}
\leanchecked{BEDC.FKernel.Ext.ext\_result\_source\_not\_hsame}
\begin{proof}
Invert the generated extension witness. The result is either \(\Ezero(h)\) or
\(\Eone(h)\), and both cases are excluded by history constructor
no-confusion for a one-constructor self-extension.
\end{proof}

\begin{theorem}[Extension two-step cycle is impossible]
\label{thm:extension-two-step-cycle-absurd}
If \(\ExtR(m,h,r)\) and \(\ExtR(n,r,h)\), then contradiction follows.
\end{theorem}
\leanchecked{BEDC.FKernel.Ext.ext\_two\_step\_cycle\_absurd}
\begin{proof}
Use the finite constructor length of histories. Each extension increases this
length by one. Two extensions returning from \(h\) to \(h\) would force the
length of \(h\) to equal its own double successor, which is impossible by
natural-number constructor no-confusion.
\end{proof}

\begin{theorem}[Extension result injectivity]
\label{thm:ext-result-injective-pair}
If \(\ExtR(m,h,r)\) and \(\ExtR(m',h',r)\), then \(\hsame(h,h')\) and \(\msame(m,m')\).
\end{theorem}
\leanchecked{BEDC.FKernel.Ext.ext\_result\_injective\_pair}
\begin{proof}
Inspect the two generated extension witnesses. A common result can only arise from matching extension constructors, so the source histories and emitted marks are internally same.
\end{proof}

\begin{theorem}[Extension result hsame injectivity]
\label{thm:extension-result-hsame-injective-pair}
If \(\ExtR(m,h,r)\), \(\ExtR(m',h',r')\), and \(\hsame(r,r')\), then \(\hsame(h,h')\) and \(\msame(m,m')\).
\end{theorem}
\leanchecked{BEDC.FKernel.Ext.ext\_result\_hsame\_injective\_pair}
\begin{proof}
Invert both generated extension witnesses. Matching outer constructors reduce
the result-sameness witness to source history sameness and diagonal
mark-sameness. Mixed outer constructors contradict history no-confusion.
\end{proof}

\begin{lemma}[Extension respects mark-sameness]
If \(\msame(m,n)\), \(\ExtR(m,h,r)\), and \(\ExtR(n,h,s)\), then \(\hsame(r,s)\).
\end{lemma}
\leanchecked{BEDC.FKernel.Ext.ext\_respects\_mark\_sameness}
\begin{proof}
By closed generation for \(\msame\), either both marks are \(\bzero\) or both are \(\bone\). In the first case both extension results are \(\Ezero(h)\); in the second case both are \(\Eone(h)\). Use reflexivity of \(\hsame\).
\end{proof}

\begin{theorem}[Extension determinacy up to history-sameness]
If \(\ExtR(m,h,r)\) and \(\ExtR(m,h,s)\), then \(\hsame(r,s)\).
\end{theorem}
\leanchecked{BEDC.FKernel.Ext.ext\_deterministic}
\begin{proof}
Apply the previous lemma with \(\msame(m,m)\), using mark-sameness reflexivity.
\end{proof}

\section{Relational continuation}

\begin{definition}[Continuation relation]
\label{def:continuation-relation}
\label{def:continuation-step-rules}
The continuation relation \(\ContR(h,k,r)\) is generated by:
$$
\frac{}{\ContR(h,\emp,h)}
$$
$$
\frac{\ContR(h,k,r)}{\ContR(h,\Ezero(k),\Ezero(r))}
\qquad
\frac{\ContR(h,k,r)}{\ContR(h,\Eone(k),\Eone(r))}.
$$
It is a relation, not a primitive function.
\end{definition}

\begin{lemma}[Continuation respects a matching right extension step]
\label{lem:cont-ext-right-step}
If $\ContR(h,k,r)$, $\ExtR(m,k,k')$, and $\ExtR(m,r,r')$, then
$\ContR(h,k',r')$.
\end{lemma}
\leanchecked{BEDC.FKernel.Cont.cont\_ext\_right\_step}
\begin{proof}
Case split on the extension of $k$ by $m$. In the $\Ezero$ branch, the matching
extension of $r$ is also the zero branch, and the zero continuation step
applies. The $\Eone$ branch is analogous.
\end{proof}

\begin{theorem}[Extension as singleton continuation]
\label{thm:extension-as-one-step-continuation}
For every history $h$ and result $r$, $\ExtR(\bzero,h,r)$ if and only if
$\ContR(h,\Ezero(\emp),r)$, and $\ExtR(\bone,h,r)$ if and only if
$\ContR(h,\Eone(\emp),r)$.
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.extension\_as\_singleton\_continuation}
\leanchecked{BEDC.FKernel.Cont.ext\_singleton\_continuation\_iff\_pair}
\begin{proof}
For the forward zero branch, invert the extension witness to identify
$r$ with $\Ezero(h)$, use the empty-continuation rule
$\ContR(h,\emp,h)$, and apply the $\Ezero$ continuation step rule. The one
branch is the same argument with $\Eone$.

For the reverse zero branch, invert the continuation witness for
$\ContR(h,\Ezero(\emp),r)$. Step-result inversion gives a predecessor
continued by $\emp$, and history constructor no-confusion forces the
singleton history to be the displayed $\Ezero(\emp)$ branch. The
empty-continuation rule identifies the predecessor with $h$, so the
$\bzero$ extension constructor yields $\ExtR(\bzero,h,r)$. The one branch
uses the corresponding $\Eone$ inversion and the $\bone$ extension
constructor.
\end{proof}

\begin{theorem}[Continuation step result no-confusion]
\label{thm:cont-step-result-no-confusion-pair}
No generated history \(r\) can be both the result of continuing by
\(\Ezero(k_0)\) and the result of continuing by \(\Eone(k_1)\) from the same
source history. The converse ordering of the two step constructors is equally
impossible.
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_step\_result\_no\_confusion\_pair}
\begin{proof}
Invert the two continuation witnesses. The zero-step witness forces the common
result to have outer constructor \(\Ezero\), while the one-step witness forces
the same result to have outer constructor \(\Eone\). Closed generation of
histories gives constructor no-confusion. The converse direction is symmetric.
\end{proof}

\begin{theorem}[Continuation step result is nonempty]
\label{thm:continuation-step-result-not-empty}
For all histories $h,k$, neither $\ContR(h,\Ezero(k),\emp)$ nor
$\ContR(h,\Eone(k),\emp)$ can hold.
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_step\_result\_not\_empty\_pair}
\begin{proof}
Unfold the step result shape for $\ContR$. The zero branch would identify $\emp$ with
$\Ezero(r)$, and the one branch would identify $\emp$ with $\Eone(r)$. Both contradict the
generated history constructors.
\end{proof}

\begin{theorem}[Continuation result tag separation]
\label{thm:continuation-result-tag-separation}
There are no histories $h,k,r_0,r_1$ such that both
$\ContR(h,k,\Ezero(r_0))$ and $\ContR(h,k,\Eone(r_1))$ hold.
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_result\_tag\_separation}
\begin{proof}
Assume both continuation witnesses for the same inputs $h$ and $k$.
Continuation determinacy gives $\hsame(\Ezero(r_0),\Eone(r_1))$. History
no-confusion rules out any generated sameness derivation between a zero-tagged
history and a one-tagged history, so the two witnesses cannot coexist.
\end{proof}

\begin{theorem}[Left one-headed continuation result cases]
\label{thm:cont-left-e1-result-cases}
If $\ContR(\Eone(h),k,\Eone(r))$, then either $k=\emp$ and
$\HSame(h,r)$, or $k=\Eone(k_0)$ for some $k_0$ and
$\ContR(\Eone(h),k_0,r)$.
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_left\_e1\_result\_cases}
\begin{proof}
Classify the right continuation history $k$. The empty case reduces the result
equation to equality of the inner histories. A zero-headed $k$ would force a
one-headed history to equal a zero-headed history, contradicting constructor
no-confusion. A one-headed $k$ exposes its predecessor $k_0$ and removes the
matching outer constructor from the continuation equation.
\end{proof}

\begin{theorem}[Left cross-tag continuation result cases]
\label{thm:cont-left-tag-cross-result-cases}
If $\ContR(\Ezero(h),k,\Eone(r))$, then $k=\Eone(k_0)$ for some $k_0$
and $\ContR(\Ezero(h),k_0,r)$. If $\ContR(\Eone(h),k,\Ezero(r))$, then
$k=\Ezero(k_0)$ for some $k_0$ and $\ContR(\Eone(h),k_0,r)$.
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_left\_tag\_cross\_result\_cases}
\begin{proof}
In each direction, classify $k$ by generated-history constructors. The empty
case and the branch whose tag matches the left source produce impossible
outer-constructor equalities. The opposite-tag branch exposes the predecessor
of $k$ and peels the matching outer constructor from the result equation.
\end{proof}

\begin{theorem}[Continuation determinacy up to history-sameness]
If \(\ContR(h,k,r)\) and \(\ContR(h,k,s)\), then \(\hsame(r,s)\).
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_deterministic}
\begin{proof}
Induct on the first derivation. In the empty case the second derivation is also the empty case by no-confusion, and the result follows by reflexivity. In the two extension cases, no-confusion forces the second derivation to use the same outer continuation constructor; apply the induction hypothesis and rebuild \(\hsame\) with the matching history constructor.
\end{proof}

\begin{theorem}[Continuation result transport]
\label{thm:cont-result-hsame-transport}
If \(\ContR(h,k,r)\) and \(\hsame(r,r')\), then \(\ContR(h,k,r')\).
\end{theorem}
\begin{proof}
Unfold \(\ContR\) as the append-result equation. Symmetry of the result sameness followed by the original continuation equation gives the required append-result equation for \(r'\).
\end{proof}
\leanchecked{BEDC.FKernel.Cont.cont\_result\_hsame\_transport}

\begin{theorem}[Continuation hsame transport]
\label{thm:cont-hsame-transport}
If $\hsame(h,h')$, $\hsame(k,k')$, $\hsame(r,r')$, and $\ContR(h,k,r)$, then
$\ContR(h',k',r')$.
\end{theorem}
\begin{proof}
Eliminate the two input $\hsame$ witnesses to align the left and right
continuation inputs. The remaining result endpoint is transported by
Theorem~\ref{thm:cont-result-hsame-transport}.
\end{proof}

\begin{theorem}[Continuation left-context cancellation up to history-sameness]
\label{thm:cont-cancel-hsame-left-context}
If $\ContR(a,b,r)$, $\ContR(a',d,r')$, $\hsame(a,a')$, and $\hsame(r,r')$,
then $\hsame(b,d)$.
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_cancel\_hsame\_left\_context}
\begin{proof}
Use $\hsame(a,a')$ to align the left context and $\hsame(r,r')$ to align the
result. The two continuation derivations then have the same left context and
the same result, so left-context cancellation gives $\hsame(b,d)$.
\end{proof}

\begin{theorem}[Continuation cancellation laws]
\label{thm:continuation-right-cancellation}
Continuation admits cancellation on both sides:
\begin{enumerate}
  \item if $\ContR(h,k,r)$, $\ContR(h',k,r')$, and $\hsame(r,r')$, then $\hsame(h,h')$;
  \item if $\ContR(h,k,r)$ and $\ContR(h,k',r)$, then $\hsame(k,k')$;
  \item if $\ContR(a,b,ab)$, $\ContR(ab,c,left)$, $\ContR(a,d,ad)$, $\ContR(ad,c,right)$, and $\hsame(left,right)$, then $\hsame(b,d)$.
\end{enumerate}
\end{theorem}

\begin{corollary}[Continuation gives an addition-like seed]
