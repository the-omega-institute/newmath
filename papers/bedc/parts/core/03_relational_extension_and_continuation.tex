\section{Relational extension}

\begin{definition}[Extension relation]
\label{def:extension-relation}
The extension relation \(\ExtR(m,h,r)\) is generated by the two rules:
\[
\frac{}{\ExtR(\bzero,h,\Ezero(h))}
\qquad
\frac{}{\ExtR(\bone,h,\Eone(h))}.
\]
It is a relation, not a function.
\end{definition}

\leanchecked{BEDC.FKernel.Ext.ext\_total}
\leanchecked{BEDC.FKernel.Ext.ext\_constructor\_inversion}
\leanchecked{BEDC.FKernel.Ext.ext\_constructor\_characterization}
\leanchecked{BEDC.FKernel.Ext.ext\_mark\_deterministic\_from\_result}

\begin{lemma}[Extension respects mark-sameness]
If \(\msame(m,n)\), \(\ExtR(m,h,r)\), and \(\ExtR(n,h,s)\), then \(\hsame(r,s)\).
\end{lemma}
\begin{proof}
By closed generation for \(\msame\), either both marks are \(\bzero\) or both are \(\bone\). In the first case both extension results are \(\Ezero(h)\); in the second case both are \(\Eone(h)\). Use reflexivity of \(\hsame\).
\end{proof}

\begin{theorem}[Extension determinacy up to history-sameness]
If \(\ExtR(m,h,r)\) and \(\ExtR(m,h,s)\), then \(\hsame(r,s)\).
\end{theorem}
\leanchecked{BEDC.FKernel.Ext.ext\_deterministic}
\begin{proof}
Apply the previous lemma with \(\msame(m,m)\), using mark-sameness reflexivity.
\end{proof}

\section{Relational continuation}

\begin{definition}[Continuation relation]
\label{def:continuation-relation}
The continuation relation \(\ContR(h,k,r)\) is generated by:
\[
\frac{}{\ContR(\emp,k,k)}
\]
\[
\frac{\ContR(h,k,r)}{\ContR(\Ezero(h),k,\Ezero(r))}
\qquad
\frac{\ContR(h,k,r)}{\ContR(\Eone(h),k,\Eone(r))}.
\]
It is a relation, not a primitive function.
\end{definition}
\leanchecked{BEDC.FKernel.Cont.cont\_iff\_append}
\leanchecked{BEDC.FKernel.Cont.cont\_right\_constructor\_inversion}

\begin{lemma}[Continuation right unit]
For every history \(h\), \(\ContR(h,\emp,h)\).
\end{lemma}
\leanchecked{BEDC.FKernel.Cont.cont\_right\_unit}
\begin{proof}
Induct on \(h\). The empty case is the base rule. The extension cases follow by applying the corresponding continuation constructor to the induction hypothesis.
\end{proof}

\begin{lemma}[Continuation left unit]
For every history \(h\), \(\ContR(\emp,h,h)\).
\end{lemma}
\leanchecked{BEDC.FKernel.Cont.cont\_left\_unit}
\begin{proof}
This is the base rule of \(\ContR\).
\end{proof}

\begin{theorem}[Continuation unit laws]
\label{thm:continuation-unit-laws}
The empty history is a left unit for continuation, and continuation by the empty history returns a history same as the source.
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_unit\_laws}
\leanchecked{BEDC.FKernel.Cont.append\_empty\_left}
\begin{proof}
The left-unit statement is the left-unit lemma. For the right-unit statement, compare the given continuation proof with the right-unit lemma by continuation determinacy.
\end{proof}

\begin{theorem}[Continuation determinacy up to history-sameness]
If \(\ContR(h,k,r)\) and \(\ContR(h,k,s)\), then \(\hsame(r,s)\).
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_deterministic}
\begin{proof}
Induct on the first derivation. In the empty case the second derivation is also the empty case by no-confusion, and the result follows by reflexivity. In the two extension cases, no-confusion forces the second derivation to use the same outer continuation constructor; apply the induction hypothesis and rebuild \(\hsame\) with the matching history constructor.
\end{proof}

\begin{theorem}[Continuation associativity]
\label{thm:continuation-associativity}
If \(\ContR(h,k,u)\), \(\ContR(u,l,v)\), \(\ContR(k,l,w)\), and \(\ContR(h,w,z)\), then \(\hsame(v,z)\).
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_assoc\_relational}
\leanchecked{BEDC.FKernel.Cont.cont\_assoc\_middle\_exists}
\leanchecked{BEDC.FKernel.Cont.cont\_assoc\_left\_exists}
\leanchecked{BEDC.FKernel.Cont.cont\_assoc\_unique}
\begin{proof}
Induct on the derivation of \(\ContR(h,k,u)\). The empty case reduces both sides to the determinacy of \(\ContR(k,l,-)\). In the \(\Ezero\) and \(\Eone\) cases, no-confusion forces matching outer constructors in the relevant continuation derivations; use the induction hypothesis on inner histories and rebuild \(\hsame\) under the same outer constructor.
\end{proof}

\begin{corollary}[Continuation gives an addition-like seed]
Unary repetition histories with relational continuation license an addition-like derived interface \(\AddUp\), but no primitive addition is introduced into the kernel.
\end{corollary}
