\section{Bundle grammar}

\begin{definition}[Probe bundle]
\label{def:probe-bundle}
Probe bundles are generated by:
\[
\frac{}{\Bnil:\Bundle}
\]
\[
\frac{\pi:\PN\qquad \Pi:\Bundle}{\Bcons(\pi,\Pi):\Bundle}.
\]
Bundle membership is the closed predicate generated by cons membership: the head probe of
\(\Bcons(\pi,\Pi)\) is a member of that bundle, and tail membership propagates through cons.
\end{definition}

\section{Signature generation}

\begin{definition}[Signature event]
\label{def:signature-event}
The signature event \(\Sig(\Pi,h,s,\Delta)\) means that bundle \(\Pi\) applied to history \(h\) generates answer-history \(s\) with evidence \(\Delta\). It is generated by the rules below.
\end{definition}
\leanchecked{BEDC.FKernel.Sig.sig\_empty\_constructor}
\leanvariant{BEDC.FKernel.Sig.sig\_cons\_constructor}
\leanvariant{BEDC.FKernel.Sig.sig\_cons\_inversion}
\leanvariant{BEDC.FKernel.Sig.sig\_cons\_result\_inversion}
\leanvariant{BEDC.FKernel.Sig.sig\_cons\_tail\_deterministic}
\leanvariant{BEDC.FKernel.Sig.sigRel\_cons\_inversion}
\leanvariant{BEDC.FKernel.Sig.sig\_empty\_inversion}

\begin{kernelrule}[Empty signature]
\[
\frac{}{\Sig(\Bnil,h,\emp,\Delta_0)}
\]
where \(\Delta_0\) is the neutral signature evidence token.
\end{kernelrule}

\begin{lemma}[Empty signature inversion]
\label{lem:empty-signature-inversion}
If \(\Sig(\Bnil,h,r,\Delta_0)\), then \(\hsame(r,\emp)\).
\end{lemma}
\leanchecked{BEDC.FKernel.Sig.sig\_empty\_inversion}
\begin{proof}
Invert the signature derivation. For the empty bundle, only the empty-signature rule applies, so the result history is \(\emp\).
\end{proof}

\begin{kernelrule}[Cons signature]
If
\[
\Ask(\pi,h,m,\delta),
\qquad
\Sig(\Pi,h,s,\Delta),
\qquad
\ExtR(m,s,r),
\]
then
\[
\Sig(\Bcons(\pi,\Pi),h,r,\delta\star\Delta).
\]
Here \(\delta\star\Delta\) is the evidence-composition operation licensed by the relevant asking policy.
\end{kernelrule}

\begin{theorem}[Cons head mark determinacy]
\label{thm:signature-cons-head-mark-determinacy}
Assume an asking policy on \(D\). If \(h\in D\) and two cons-signature derivations for \(\Bcons(\pi,\Pi)\) start from the same source history \(h\), then their head asking events emit internally same marks.
\end{theorem}
\leanchecked{BEDC.FKernel.Sig.sig\_cons\_head\_mark\_determinacy}
\begin{proof}
Invert both cons-signature derivations. Their head premises are asking events for the same probe and source history. Asking-policy determinacy gives internal sameness of the emitted marks, while the tail signature witnesses are retained from the two inverted derivations.
\end{proof}

\begin{theorem}[Signature existence under an asking policy]
\label{thm:signature-totality-under-policy}
Assume \(\AskPol(\Pi,D)\). If \(h\) is admitted by \(D\), then bundle \(\Pi\) generates some signature event from \(h\).
\end{theorem}
\leanchecked{BEDC.FKernel.Sig.sig\_total\_from\_policy}
\leanvariant{BEDC.FKernel.Sig.sigTotalOn\_from\_policy}
\begin{proof}
Induct on the generated bundle \(\Pi\). The empty bundle produces the empty answer-history. In the cons case, asking-policy totality supplies the first emitted mark and evidence token; the induction hypothesis supplies the tail signature; the emitted mark chooses the corresponding extension constructor.
\end{proof}

\begin{theorem}[Signature determinacy under an asking policy]
Assume \(\AskPol(\Pi,D)\). If \(h\) is admitted by \(D\) and
\[
\Sig(\Pi,h,s,\Delta),
\qquad
\Sig(\Pi,h,t,\Theta),
\]
then \(\hsame(s,t)\).
\end{theorem}
\leanchecked{BEDC.FKernel.Sig.sig\_deterministic}
\begin{proof}
Induct on the generated bundle \(\Pi\). The empty case uses reflexivity of \(\hsame\). In the cons case, asking determinacy gives \(\msame\) of the first emitted marks. The induction hypothesis gives \(\hsame\) of the tail signatures. Extension respects mark-sameness and history-sameness, so the resulting answer histories are \(\hsame\).
\end{proof}

\section{Signature sameness}

\begin{definition}[Signature-sameness]
\label{def:signature-sameness-witnesses}
\label{def:signature-sameness}
Given a bundle \(\Pi\), histories \(h,k\) are signature-same, written \(\sameSig_\Pi(h,k)\), if there exist generated signatures
\[
\Sig(\Pi,h,s,\Delta),
\qquad
\Sig(\Pi,k,t,\Theta),
\]
such that \(\hsame(s,t)\).
\end{definition}
\leandef{BEDC.FKernel.Sig.SameSig}
\leanchecked{BEDC.FKernel.Sig.sameSig\_witnesses}
\leanvariant{BEDC.FKernel.Sig.sameSig\_refl\_from\_witness}

\begin{theorem}[History-sameness gives signature-sameness under policy]
\label{thm:samesig-of-hsame-under-policy}
Assume an asking policy and total signature generation on \(D\). If \(h,k\in D\) and \(\hsame(h,k)\), then \(\sameSig_\Pi(h,k)\).
\end{theorem}
\leanchecked{BEDC.FKernel.Sig.sameSig\_of\_hsame\_under\_policy}
\begin{proof}
Choose generated signatures for \(h\) and \(k\) by totality. The asking policy transports \(\hsame(h,k)\) across signature generation, yielding \(\hsame\) of the generated signatures. These witnesses form \(\sameSig_\Pi(h,k)\).
\end{proof}

\begin{theorem}[Signature-sameness equivalence]
\label{thm:signature-sameness-equivalence}
\label{thm:signature-sameness-equivalence-policy-spine}
Under an asking policy for \(\Pi\) over a source specification \(D\), \(\sameSig_\Pi\) is reflexive, symmetric, and transitive on admitted histories.
\end{theorem}
\leanchecked{BEDC.FKernel.Sig.sameSig\_equivalence}
\leanvariant{BEDC.FKernel.Sig.sameSig\_equivalence\_under\_policy}
\leanvariant{BEDC.FKernel.Sig.sameSig\_refl\_under\_policy}
\leanvariant{BEDC.FKernel.Sig.sameSig\_trans\_under\_policy}
\leanvariant{BEDC.FKernel.Sig.sameSig\_trans\_witnesses\_under\_policy}
\leanvariant{BEDC.FKernel.Sig.sameSig\_trans\_from\_middle\_determinacy}
\begin{proof}
Reflexivity follows from totality and \(\hsame\)-reflexivity. Symmetry follows from \(\hsame\)-symmetry. For transitivity, suppose \(h\) and \(k\) share a signature, and \(k\) and \(l\) share a signature. Signature determinacy for \(k\) ensures the two signatures generated from \(k\) are \(\hsame\). Transitivity of \(\hsame\) then connects a signature of \(h\) with a signature of \(l\).
\end{proof}
