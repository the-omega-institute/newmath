\section{Bundle grammar}

\begin{definition}[Probe bundle]
\label{def:probe-bundle}
\label{def:bundle-generation}
Probe bundles are generated by:
\[
\frac{}{\Bnil:\Bundle}
\]
\[
\frac{\pi:\PN\qquad \Pi:\Bundle}{\Bcons(\pi,\Pi):\Bundle}.
\]
Bundle membership is the closed predicate generated by cons membership: the head probe of
\(\Bcons(\pi,\Pi)\) is a member of that bundle, and tail membership propagates through cons.
\end{definition}

\begin{definition}[Bundle append]
\label{def:bundle-append}
The append operation $\mathsf{BAppend}(\Pi,\Theta)$ is recursion on the left bundle:
$$
\mathsf{BAppend}(\Bnil,\Theta)=\Theta,
\qquad
\mathsf{BAppend}(\Bcons(\pi,\Pi),\Theta)
=\Bcons(\pi,\mathsf{BAppend}(\Pi,\Theta)).
$$
\end{definition}
\leandef{BEDC.FKernel.Bundle.bundleAppend}

\begin{definition}[Probe bundle length]
\label{def:bundle-length}
The length $\mathsf{BLEN}(\Pi)$ of a generated probe bundle is recursion on the bundle:
$$
\mathsf{BLEN}(\Bnil)=0,
\qquad
\mathsf{BLEN}(\Bcons(\pi,\Pi))=1+\mathsf{BLEN}(\Pi).
$$
\end{definition}
\leandef{BEDC.FKernel.Bundle.bundleLength}

\begin{theorem}[Probe bundle zero length equivalence]
\label{thm:probe-bundle-length-zero-iff-nil}
A generated probe bundle has length zero exactly when it is the empty bundle
$\Bnil$.
\end{theorem}
\begin{proof}
Eliminate the generated bundle. The empty case reduces by computation. In the
cons case the length reduces to a successor, which cannot equal zero.
\end{proof}
\leanchecked{BEDC.FKernel.Bundle.bundleLength\_eq\_zero\_iff\_nil}

\begin{theorem}[Probe bundle append length]
\label{thm:probe-bundle-append-length}
For all probe bundles $\Pi,\Theta$,
$$
\mathsf{BLEN}(\mathsf{BAppend}(\Pi,\Theta))
=
\mathsf{BLEN}(\Pi)+\mathsf{BLEN}(\Theta).
$$
\end{theorem}
\begin{proof}
Induct on $\Pi$. In the empty case, the append term reduces to $\Theta$ and the length equation reduces to the left identity of addition. In the cons case, both sides reduce by one successor, and the tail equation is the induction hypothesis.
\end{proof}
\leanchecked{BEDC.FKernel.Bundle.bundleLength\_append}
\leanchecked{BEDC.FKernel.Bundle.bundleAppend\_length}
\leanchecked{BEDC.FKernel.Bundle.bundleLength\_bundleAppend}

\begin{lemma}[Probe bundle nonempty prefix length separation]
\label{lem:probe-bundle-nonempty-prefix-length-separation}
For every probe $\pi$ and probe bundles $\Pi_0,\Omega$,
$$
\mathsf{BLEN}(\mathsf{BAppend}(\Bcons(\pi,\Pi_0),\Omega))
\neq
\mathsf{BLEN}(\Omega).
$$
Consequently,
$$
\mathsf{BAppend}(\Bcons(\pi,\Pi_0),\Omega)\neq\Omega.
$$
\end{lemma}
\begin{proof}
Theorem~\ref{thm:probe-bundle-append-length} rewrites the left length as
$1+\mathsf{BLEN}(\Pi_0)+\mathsf{BLEN}(\Omega)$. A natural number with an exposed successor constructor cannot be equal to the unextended suffix length by natural-number constructor no-confusion. If the appended bundle itself were equal to $\Omega$, applying $\mathsf{BLEN}$ to that equality would contradict the displayed length separation.
\end{proof}
\leanchecked{BEDC.FKernel.Bundle.bundleAppend\_nonempty\_prefix\_length\_separation}

\begin{theorem}[Probe bundle append right unit]
\label{thm:bundle-append-right-unit}
For every probe bundle \(\Pi\),
$$
\mathsf{BAppend}(\Pi,\Bnil)=\Pi .
$$
\end{theorem}
\begin{proof}
Induct on \(\Pi\). The empty bundle case is definitional. In the cons case,
both sides expose the same head probe, and the tail equality is the induction
hypothesis.
\end{proof}
\leanchecked{BEDC.FKernel.Bundle.bundleAppend\_nil\_right}

\begin{theorem}[Probe bundle append associativity]
\label{thm:probe-bundle-append-associativity}
For all probe bundles \(\Pi,\Theta,\Omega\),
$$
\mathsf{BAppend}(\mathsf{BAppend}(\Pi,\Theta),\Omega)
=
\mathsf{BAppend}(\Pi,\mathsf{BAppend}(\Theta,\Omega)).
$$
\end{theorem}
\begin{proof}
Induct on \(\Pi\). The empty bundle case is definitional. In the cons case,
both sides expose the same head probe, and the tail equality is the induction
hypothesis.
\end{proof}
\leanchecked{BEDC.FKernel.Bundle.bundleAppend\_assoc}

\begin{theorem}[Bundle append membership]
\label{thm:bundle-append-membership}
For every probe $p$ and bundles $\Pi,\Theta$,
$$
p\in\mathsf{BAppend}(\Pi,\Theta)
\quad\Longleftrightarrow\quad
p\in\Pi \ \text{or}\ p\in\Theta .
$$
\end{theorem}
\leanchecked{BEDC.FKernel.Bundle.inBundle\_bundleAppend\_iff}
\begin{proof}
Induct on $\Pi$. The empty case reduces membership to $\Theta$. The cons case splits head
membership from tail membership, then applies the induction hypothesis to the tail.
\end{proof}

\begin{theorem}[Bundle append membership left residual]
\label{thm:bundle-append-membership-left-of-not-right}
For every probe \(p\) and bundles \(\Pi,\Theta\), if
$$
p\in\mathsf{BAppend}(\Pi,\Theta)
$$
and \(p\notin\Theta\), then \(p\in\Pi\).
\end{theorem}
\begin{proof}
Induct on \(\Pi\). In the empty case, appended membership is membership in
\(\Theta\), contradicting the exclusion premise. In the cons case, either the
probe is the head of \(\Pi\), or it lies in the appended tail. The head branch is
immediate, and the tail branch follows from the induction hypothesis with the
same exclusion premise.
\end{proof}
\leanchecked{BEDC.FKernel.Bundle.inBundle\_bundleAppend\_left\_of\_not\_right}

\begin{theorem}[Bundle-local policy append restriction]
\label{thm:bundle-local-policy-append-restriction}
If $\mathsf{BundleAskPolicy}(\mathsf{BAppend}(\Pi,\Theta),D)$ holds, then both $\mathsf{BundleAskPolicy}(\Pi,D)$ and $\mathsf{BundleAskPolicy}(\Theta,D)$ hold.
\end{theorem}
\leanchecked{BEDC.FKernel.Ask.bundleAskPolicy\_append\_restriction}
\begin{proof}
Use Theorem~\ref{thm:bundle-append-membership}. A member of $\Pi$ maps into $\mathsf{BAppend}(\Pi,\Theta)$ through the left disjunct, and a member of $\Theta$ maps into $\mathsf{BAppend}(\Pi,\Theta)$ through the right disjunct. Restrict the totality, mark determinacy, and history-respect fields of the appended bundle-local policy along these two membership maps.
\end{proof}

\begin{theorem}[Bundle-local policy append gluing]
\label{thm:bundle-local-policy-append-gluing}
If $\mathsf{BundleAskPolicy}(\Pi,D)$ and $\mathsf{BundleAskPolicy}(\Theta,D)$ hold, then $\mathsf{BundleAskPolicy}(\mathsf{BAppend}(\Pi,\Theta),D)$ holds.
\end{theorem}
\leanchecked{BEDC.FKernel.Ask.bundleAskPolicy\_append\_gluing}
\begin{proof}
Use Theorem~\ref{thm:bundle-append-membership}. For each probe membership in $\mathsf{BAppend}(\Pi,\Theta)$, split it into a left membership in $\Pi$ or a right membership in $\Theta$. In the left case, take totality, determinacy, and history-respect from the policy on $\Pi$; in the right case, take the same three fields from the policy on $\Theta$. These case analyses assemble the bundle-local asking policy for the appended bundle.
\end{proof}

\begin{theorem}[Bundle-local policy append equivalence]
\label{thm:bundle-local-policy-append-iff}
For all probe bundles $\Pi,\Theta$ and domain $D$,
$$
\mathsf{BundleAskPolicy}(\mathsf{BAppend}(\Pi,\Theta),D)
\Longleftrightarrow
\bigl(\mathsf{BundleAskPolicy}(\Pi,D)\ \text{and}\ \mathsf{BundleAskPolicy}(\Theta,D)\bigr).
$$
\end{theorem}
\leanchecked{BEDC.FKernel.Ask.bundleAskPolicy\_append\_iff}
\begin{proof}
The forward direction is exactly Theorem~\ref{thm:bundle-local-policy-append-restriction}. The reverse direction is exactly Theorem~\ref{thm:bundle-local-policy-append-gluing}.
\end{proof}

\begin{theorem}[Probe bundle prefix cancellation]
\label{thm:probe-bundle-prefix-cancellation}
For any fixed probe bundle $\Pi$ and any probe bundles $\Theta,\Omega$, if
$$
\mathsf{BAppend}(\Pi,\Theta)=\mathsf{BAppend}(\Pi,\Omega),
$$
then $\Theta=\Omega$.
\end{theorem}
\leanchecked{BEDC.FKernel.Bundle.bundleAppend\_prefix\_cancel}
\begin{proof}
Induct on $\Pi$. The empty case is immediate because both displayed append terms
reduce to their suffix bundles. In the cons case, constructor injectivity reduces
the equality of cons bundles to equality of their tails, and the induction
hypothesis cancels the remaining common prefix.
\end{proof}

\begin{theorem}[Probe bundle suffix cancellation]
\label{thm:probe-bundle-suffix-cancellation}
For any probe bundles $\Pi,\Theta,\Omega$, if
$$
\mathsf{BAppend}(\Pi,\Omega)=\mathsf{BAppend}(\Theta,\Omega),
$$
then $\Pi=\Theta$.
\end{theorem}
\begin{proof}
Induct on $\Pi$ and then case-split $\Theta$. The empty-empty case is immediate. In the empty-cons and cons-empty cases, Lemma~\ref{lem:probe-bundle-nonempty-prefix-length-separation} rules out equality between a bundle obtained by appending a nonempty prefix to $\Omega$ and the suffix $\Omega$. In the cons-cons case, the displayed equality is an equality of cons bundles. Constructor injectivity identifies the two head probes and reduces the tail equality to
$$
\mathsf{BAppend}(\Pi_0,\Omega)=\mathsf{BAppend}(\Theta_0,\Omega).
$$
The induction hypothesis gives $\Pi_0=\Theta_0$, and the head equality rebuilds $\Pi=\Theta$.
\end{proof}
\leanchecked{BEDC.FKernel.Bundle.bundleAppend\_suffix\_cancel}

\begin{lemma}[Probe bundle append empty-result inversion]
\label{lem:probe-bundle-append-empty-result-inversion}
For probe bundles $\Pi,\Theta$, if
$$
\mathsf{BAppend}(\Pi,\Theta)=\Bnil,
$$
then $\Pi=\Bnil$ and $\Theta=\Bnil$.
\end{lemma}
\leanchecked{BEDC.FKernel.Bundle.bundleAppend\_empty\_result\_inversion}
\begin{proof}
Proceed by cases on $\Pi$. In the empty case the displayed equality reduces to $\Theta=\Bnil$. In the cons case the append definition exposes a cons bundle equal to $\Bnil$, contradicting probe-bundle constructor no-confusion.
\end{proof}

\begin{lemma}[Probe bundle append cons-result inversion]
\label{lem:probe-bundle-append-cons-result-inversion}
For probe bundles $\Pi,\Theta,\Omega$ and probe $\pi$, if
$$
\mathsf{BAppend}(\Pi,\Theta)=\Bcons(\pi,\Omega),
$$
then either $\Pi=\Bnil$ and $\Theta=\Bcons(\pi,\Omega)$, or there exists a bundle $\Pi_0$ such that $\Pi=\Bcons(\pi,\Pi_0)$ and
$$
\mathsf{BAppend}(\Pi_0,\Theta)=\Omega.
$$
\end{lemma}
\begin{proof}
Proceed by cases on $\Pi$. In the empty case the displayed equality reduces to $\Theta=\Bcons(\pi,\Omega)$, giving the left alternative. In the cons case the append definition exposes a cons bundle on the left; constructor injectivity identifies its head with $\pi$ and its appended tail with $\Omega$, giving the required tail bundle $\Pi_0$.
\end{proof}
\leanchecked{BEDC.FKernel.Bundle.bundleAppend\_cons\_result\_inversion}

\begin{theorem}[Probe bundle append split uniqueness at fixed left length]
\label{thm:bundle-append-split-unique-fixed-length}
For probe bundles $\Pi,\Theta,\Pi',\Theta'$, if
$$
\mathsf{BAppend}(\Pi,\Theta)=\mathsf{BAppend}(\Pi',\Theta')
$$
and
$$
\mathsf{BLEN}(\Pi)=\mathsf{BLEN}(\Pi'),
$$
then $\Pi=\Pi'$ and $\Theta=\Theta'$.
\end{theorem}
\begin{proof}
Proceed by simultaneous cases on the two left bundles. The empty-empty case reduces the append equality to the suffix equality $\Theta=\Theta'$. The empty-cons and cons-empty cases contradict the displayed length equality by natural-number constructor no-confusion. In the cons-cons case, constructor injectivity on the append equality identifies the two head probes and gives equality of the appended tails; constructor injectivity on the length equality gives equal tail lengths. The induction hypothesis applied to the tails gives equality of the left tails and suffixes, and the head equality rebuilds $\Pi=\Pi'$.
\end{proof}
\leanchecked{BEDC.FKernel.Bundle.bundleAppend\_split\_unique\_fixed\_length}

\section{Signature generation}

\begin{definition}[Signature event]
\label{def:signature-event}
The signature event \(\Sig(\Pi,h,s,\Delta)\) means that bundle \(\Pi\) applied to history \(h\) generates answer-history \(s\) with evidence \(\Delta\). It is generated by the rules below.
In the checked kernel, the logical predicate is the three-place $\mathsf{SigRel}(\Pi,h,s)$ on bundle, source history, and answer history. The displayed evidence ledger is a paper-side annotation attached to the generating derivation.
\end{definition}

\begin{kernelrule}[Empty signature]
\[
\frac{}{\Sig(\Bnil,h,\emp,\Delta_0)}
\]
where \(\Delta_0\) is the neutral signature evidence token.
\end{kernelrule}

\begin{lemma}[Empty signature inversion]
\label{lem:empty-signature-inversion}
If \(\Sig(\Bnil,h,r,\Delta_0)\), then \(\hsame(r,\emp)\).
\end{lemma}
\leanchecked{BEDC.FKernel.Sig.sig\_empty\_inversion}
\begin{proof}
Invert the signature derivation. For the empty bundle, only the empty-signature rule applies, so the result history is \(\emp\).
\end{proof}

\begin{kernelrule}[Cons signature]
If
\[
\Ask(\pi,h,m,\delta),
\qquad
\Sig(\Pi,h,s,\Delta),
\qquad
\ExtR(m,s,r),
\]
then
\[
\Sig(\Bcons(\pi,\Pi),h,r,\delta\star\Delta).
\]
Here \(\delta\star\Delta\) records the head asking evidence together with the tail derivation ledger; it is not a field of the asking policy.
\end{kernelrule}

\begin{theorem}[Signature generation over bundle append]
\label{thm:signature-generation-bundle-append}
For probe bundles $\Pi,\Theta$ and source history $h$, if $\Sig(\Pi,h,s,\Delta)$ and $\Sig(\Theta,h,t,\Gamma)$, then there exist a history $u$ and evidence $\Lambda$ such that
$$
\ContR(t,s,u)
\quad\text{and}\quad
\Sig(\mathsf{BAppend}(\Pi,\Theta),h,u,\Lambda).
$$
\end{theorem}
\leanchecked{BEDC.FKernel.Sig.signature\_generation\_bundle\_append}
\begin{proof}
Induct on $\Pi$. In the empty bundle case, empty-signature inversion identifies $s$ with $\emp$; take $u=t$, use the right-unit continuation constructor, and reuse the given signature for $\Theta$. In the cons case, invert the left signature. The tail signature for $\Pi$ and the signature for $\Theta$ feed the induction hypothesis, producing a tail result $u_0$ with $\ContR(t,s_0,u_0)$ and a signature for $\mathsf{BAppend}(\Pi,\Theta)$. Rebuild the cons signature using the original head asking event and the tail signature. The final extension branch supplies either the $\Ezero$ or $\Eone$ continuation step, yielding $\ContR(t,s,u)$ for the rebuilt result.
\end{proof}

\begin{theorem}[Signature generation append inversion]
\label{thm:signature-generation-bundle-append-inversion}
For probe bundles $\Pi,\Theta$ and source history $h$, if
$\Sig(\mathsf{BAppend}(\Pi,\Theta),h,u,\Lambda)$, then there exist histories
$s,t$ and evidence tokens $\Delta,\Gamma$ such that
$\Sig(\Pi,h,s,\Delta)$, $\Sig(\Theta,h,t,\Gamma)$, and $\ContR(t,s,u)$.
\end{theorem}
\leanchecked{BEDC.FKernel.Sig.signature\_generation\_bundle\_append\_inversion}
\begin{proof}
Induct on $\Pi$. In the empty bundle case, the appended bundle is $\Theta$;
generate the empty signature for $\Bnil$, take $s=\emp$ and $t=u$, reuse the
given signature for $\Theta$, and apply the continuation right-unit
constructor. In the cons case, invert the appended cons signature. The
inversion exposes the head asking event, the tail signature for
$\mathsf{BAppend}(\Pi,\Theta)$, and the final extension witness. Apply the
induction hypothesis to the tail append signature to obtain signatures for
$\Pi$ and $\Theta$ together with a continuation $\ContR(t,s_0,u_0)$. Rebuild
the left cons signature from the exposed head asking event, the recovered tail
signature for $\Pi$, and the final extension witness. The same extension
constructor lifts the recovered tail continuation to $\ContR(t,s,u)$ for the
whole appended signature.
\end{proof}

\begin{theorem}[Signature generation append component determinacy]
\label{thm:signature-generation-append-component-determinacy}
For probe bundles $\Pi,\Theta$, domain $D$, and source history $h$, assume
$\mathsf{BundleAskPolicy}(\Pi,D)$, $\mathsf{BundleAskPolicy}(\Theta,D)$, and
$h\in D$. If
$$
\Sig(\Pi,h,s,\Delta),
\qquad
\Sig(\Theta,h,t,\Gamma),
\qquad
\Sig(\mathsf{BAppend}(\Pi,\Theta),h,u,\Lambda),
$$
then there exists a history $v$ such that
$$
\ContR(t,s,v)
\quad\text{and}\quad
\hsame(v,u).
$$
\end{theorem}
\leanchecked{BEDC.FKernel.Sig.signature\_generation\_append\_component\_determinacy}
\begin{proof}
Theorem~\ref{thm:bundle-local-policy-append-gluing} gives
$\mathsf{BundleAskPolicy}(\mathsf{BAppend}(\Pi,\Theta),D)$ from the two component policies. Theorem~\ref{thm:signature-generation-bundle-append} applied to the two component signatures produces a history $v$ with $\ContR(t,s,v)$ and a generated signature for the appended bundle. Theorem~\ref{thm:bundle-local-signature-determinacy}, applied to the glued policy, the premise $h\in D$, that constructed appended signature, and the given appended signature for $u$, gives $\hsame(v,u)$.
\end{proof}

\begin{theorem}[Three-bundle signature generation coherence]
\label{thm:signature-generation-three-bundle-coherence}
For probe bundles $\Pi,\Theta,\Omega$ and source history $h$, assume
$\Sig(\Pi,h,s,\Delta)$, $\Sig(\Theta,h,t,\Gamma)$, and
$\Sig(\Omega,h,w,\Xi)$. Then there exist histories $a,b,u_L,u_R$ and evidence
ledgers $\Lambda_{\Pi\Theta}$, $\Lambda_L$, $\Lambda_{\Theta\Omega}$, and
$\Lambda_R$ such that
\begin{enumerate}
  \item $\Sig(\mathsf{BAppend}(\Pi,\Theta),h,a,\Lambda_{\Pi\Theta})$ and $\ContR(t,s,a)$;
  \item $\Sig(\mathsf{BAppend}(\mathsf{BAppend}(\Pi,\Theta),\Omega),h,u_L,\Lambda_L)$ and $\ContR(w,a,u_L)$;
  \item $\Sig(\mathsf{BAppend}(\Theta,\Omega),h,b,\Lambda_{\Theta\Omega})$ and $\ContR(w,t,b)$;
  \item $\Sig(\mathsf{BAppend}(\Pi,\mathsf{BAppend}(\Theta,\Omega)),h,u_R,\Lambda_R)$ and $\ContR(b,s,u_R)$;
  \item $\hsame(u_L,u_R)$.
\end{enumerate}
\end{theorem}
\leanchecked{BEDC.FKernel.Sig.signature\_generation\_three\_bundle\_coherence}
\begin{proof}
Apply Theorem~\ref{thm:signature-generation-bundle-append} to the $\Pi$ and
$\Theta$ signatures to obtain $a$ and $\Lambda_{\Pi\Theta}$ with
$\ContR(t,s,a)$. Apply the same theorem to the resulting
$\mathsf{BAppend}(\Pi,\Theta)$ signature and the $\Omega$ signature to obtain
$u_L$ and $\Lambda_L$ with $\ContR(w,a,u_L)$. For the other grouping, first
apply Theorem~\ref{thm:signature-generation-bundle-append} to the $\Theta$ and
$\Omega$ signatures to obtain $b$ and $\Lambda_{\Theta\Omega}$ with
$\ContR(w,t,b)$, then apply it to the $\Pi$ signature and the
$\mathsf{BAppend}(\Theta,\Omega)$ signature to obtain $u_R$ and $\Lambda_R$ with
$\ContR(b,s,u_R)$. Theorem~\ref{thm:continuation-associativity}, with
$h=w$, $k=t$, $l=s$, middle result $b$, and alternative middle result $a$,
gives $\hsame(u_R,u_L)$. The history-sameness symmetry lemma converts this to
the displayed $\hsame(u_L,u_R)$.
\end{proof}

\begin{theorem}[Signature-sameness closure under bundle append]
\label{thm:samesig-bundle-append-closure}
For probe bundles $\Pi,\Theta$ and histories $h,k$, if $\sameSig_\Pi(h,k)$ and $\sameSig_\Theta(h,k)$, then
$$
\sameSig_{\mathsf{BAppend}(\Pi,\Theta)}(h,k).
$$
\end{theorem}
\begin{proof}
Unpack the two signature-sameness witnesses. The $\Pi$ witnesses give signatures from $h$ and $k$ whose answer histories are $\hsame$; the $\Theta$ witnesses give another such pair. Apply Theorem~\ref{thm:signature-generation-bundle-append} to the two $h$-side signatures and separately to the two $k$-side signatures. Continuation respect for $\hsame$ transports the two answer-history sameness witnesses across the appended continuations, so the appended signatures have $\hsame$ results. These appended signatures are the witnesses for $\sameSig_{\mathsf{BAppend}(\Pi,\Theta)}(h,k)$.
\end{proof}
\leanchecked{BEDC.FKernel.Sig.sameSig\_bundleAppend\_closure}

\begin{theorem}[Signature-sameness left cancellation under bundle append]
\label{thm:samesig-bundle-append-left-cancellation}
Assume $\mathsf{BundleAskPolicy}(\Pi,D)$ and $h,k\in D$. If
$$
\sameSig_{\mathsf{BAppend}(\Pi,\Theta)}(h,k)
$$
and $\sameSig_\Pi(h,k)$, then
$$
\sameSig_\Theta(h,k).
$$
\end{theorem}
\begin{proof}
Invert the appended signature-sameness witnesses using Theorem~\ref{thm:signature-generation-bundle-append-inversion}. This recovers $\Pi$ signatures, $\Theta$ signatures, and continuation witnesses on the $h$ and $k$ sides. Compare the recovered left $\Pi$ signatures with the witnesses from $\sameSig_\Pi(h,k)$ using Theorem~\ref{thm:bundle-local-signature-determinacy} at $h$ and at $k$. Transport one continuation along that left-result sameness. The common right continuation can then be cancelled against the appended result-sameness, yielding $\hsame$ for the recovered $\Theta$ results. Those recovered $\Theta$ signatures are the witnesses for $\sameSig_\Theta(h,k)$.
\end{proof}

\begin{theorem}[Signature-sameness right cancellation under bundle append]
\label{thm:samesig-bundle-append-right-cancellation}
Assume $\mathsf{BundleAskPolicy}(\Theta,D)$ and $h,k\in D$. If
$$
\sameSig_{\mathsf{BAppend}(\Pi,\Theta)}(h,k)
$$
and $\sameSig_\Theta(h,k)$, then
$$
\sameSig_\Pi(h,k).
$$
\end{theorem}
\begin{proof}
Invert the appended signature-sameness witnesses using Theorem~\ref{thm:signature-generation-bundle-append-inversion}. This recovers $\Pi$ signatures, $\Theta$ signatures, and continuation witnesses on the $h$ and $k$ sides. Compare the recovered $\Theta$ signatures with the supplied witnesses from $\sameSig_\Theta(h,k)$ using Theorem~\ref{thm:bundle-local-signature-determinacy} at $h$ and at $k$. Determinacy for the supplied and recovered $\Theta$ signatures gives $\hsame$ between the two recovered $\Theta$ continuation contexts. Continuation result transport then aligns one continuation so both continuations have the same left context and the same result up to $\hsame$. The left-context cancellation clause of Theorem~\ref{thm:continuation-right-cancellation} applies to these aligned continuations and yields $\hsame$ for the recovered $\Pi$ results. Those recovered $\Pi$ signatures are the witnesses for $\sameSig_\Pi(h,k)$.
\end{proof}

\begin{theorem}[Known component residual for signature-sameness under bundle append]
\label{thm:samesig-bundle-append-residual-iff}
For probe bundles $\Pi,\Theta$ and histories $h,k$, the following residual equivalences hold.
\begin{enumerate}
  \item If $\mathsf{BundleAskPolicy}(\Pi,D)$, $h,k\in D$, and $\sameSig_\Pi(h,k)$, then
  $$
  \sameSig_{\mathsf{BAppend}(\Pi,\Theta)}(h,k)\Longleftrightarrow\sameSig_\Theta(h,k).
  $$
  \item If $\mathsf{BundleAskPolicy}(\Theta,D)$, $h,k\in D$, and $\sameSig_\Theta(h,k)$, then
  $$
  \sameSig_{\mathsf{BAppend}(\Pi,\Theta)}(h,k)\Longleftrightarrow\sameSig_\Pi(h,k).
  $$
\end{enumerate}
\end{theorem}
\begin{proof}
For the first equivalence, the forward direction is Theorem~\ref{thm:samesig-bundle-append-left-cancellation}, using the known $\Pi$ component and the bundle-local policy on $\Pi$. The reverse direction combines the known $\sameSig_\Pi(h,k)$ with the supplied $\sameSig_\Theta(h,k)$ by Theorem~\ref{thm:samesig-bundle-append-closure}. The second equivalence is symmetric: its forward direction is Theorem~\ref{thm:samesig-bundle-append-right-cancellation}, and its reverse direction is again the append closure theorem.
\end{proof}

\begin{theorem}[Bundle-local history-sameness gives signature-sameness]

\Sig(\Pi,h,s,\Delta),

\Sig(\Pi,k,t,\Theta).

\begin{theorem}[Bundle-local signature-sameness equivalence]
\label{thm:bundle-local-signature-sameness-equivalence}
Assume $\mathsf{BundleAskPolicy}(\Pi,D)$. If $h\in D$, then
$\sameSig_\Pi(h,h)$; if $\sameSig_\Pi(h,k)$, then $\sameSig_\Pi(k,h)$;
and if $k\in D$, $\sameSig_\Pi(h,k)$, and $\sameSig_\Pi(k,l)$, then
$\sameSig_\Pi(h,l)$.
\end{theorem}
\begin{proof}
For reflexivity, bundle-local signature existence gives a signature generated
from the admitted history $h$, and $\hsame$-reflexivity relates that generated
answer history to itself. For symmetry, unpack the two signature witnesses,
swap their order, and apply $\hsame$-symmetry. For transitivity through $k$,
unpack both signature-sameness hypotheses. The admitted middle history $k$
has two generated signature results; bundle-local signature determinacy
identifies them by $\hsame$. Composing with $\hsame$-transitivity connects the
left answer history to the right answer history, yielding $\sameSig_\Pi(h,l)$.
\end{proof}

\begin{theorem}[Signature-sameness endpoint replacement]
\label{thm:signature-sameness-endpoint-replacement}
Assume $\AskPol(\Pi,D)$. If $h,h',k,k'\in D$, $\hsame(h,h')$, $\sameSig_\Pi(h,k)$, and $\hsame(k,k')$, then
$$
  \sameSig_\Pi(h',k').
$$
\end{theorem}

\begin{proof}
By Theorem~\ref{thm:samesig-of-hsame-under-policy}, the left endpoint hypothesis gives $\sameSig_\Pi(h,h')$, and the right endpoint hypothesis gives $\sameSig_\Pi(k,k')$. Symmetry from Theorem~\ref{thm:signature-sameness-equivalence} turns the first relation into $\sameSig_\Pi(h',h)$. Transitivity through the middle $h$ yields $\sameSig_\Pi(h',k)$, and transitivity through the middle $k$ yields $\sameSig_\Pi(h',k')$.
\end{proof}
