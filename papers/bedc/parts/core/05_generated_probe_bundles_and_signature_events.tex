\section{Bundle grammar}

\begin{definition}[Probe bundle]
\label{def:probe-bundle}
\label{def:bundle-generation}
Probe bundles are generated by:
\[
\frac{}{\Bnil:\Bundle}
\]
\[
\frac{\pi:\PN\qquad \Pi:\Bundle}{\Bcons(\pi,\Pi):\Bundle}.
\]
Bundle membership is the closed predicate generated by cons membership: the head probe of
\(\Bcons(\pi,\Pi)\) is a member of that bundle, and tail membership propagates through cons.
\end{definition}

\begin{definition}[Bundle append]
\label{def:bundle-append}
The append operation $\mathsf{BAppend}(\Pi,\Theta)$ is recursion on the left bundle:
$$
\mathsf{BAppend}(\Bnil,\Theta)=\Theta,
\qquad
\mathsf{BAppend}(\Bcons(\pi,\Pi),\Theta)
=\Bcons(\pi,\mathsf{BAppend}(\Pi,\Theta)).
$$
\end{definition}
\leandef{BEDC.FKernel.Bundle.bundleAppend}

\begin{theorem}[Probe bundle append right unit]
\label{thm:bundle-append-right-unit}
For every probe bundle \(\Pi\),
$$
\mathsf{BAppend}(\Pi,\Bnil)=\Pi .
$$
\end{theorem}
\begin{proof}
Induct on \(\Pi\). The empty bundle case is definitional. In the cons case,
both sides expose the same head probe, and the tail equality is the induction
hypothesis.
\end{proof}
\leanchecked{BEDC.FKernel.Bundle.bundleAppend\_nil\_right}

\begin{theorem}[Probe bundle append associativity]
\label{thm:probe-bundle-append-associativity}
For all probe bundles \(\Pi,\Theta,\Omega\),
$$
\mathsf{BAppend}(\mathsf{BAppend}(\Pi,\Theta),\Omega)
=
\mathsf{BAppend}(\Pi,\mathsf{BAppend}(\Theta,\Omega)).
$$
\end{theorem}
\begin{proof}
Induct on \(\Pi\). The empty bundle case is definitional. In the cons case,
both sides expose the same head probe, and the tail equality is the induction
hypothesis.
\end{proof}
\leanchecked{BEDC.FKernel.Bundle.bundleAppend\_assoc}

\begin{theorem}[Probe bundle append right unit]
\label{thm:probe-bundle-append-right-unit}
For every probe bundle \(\Pi\),
$$
\mathsf{BAppend}(\Pi,\Bnil)=\Pi .
$$
\end{theorem}
\begin{proof}
Induct on \(\Pi\). The empty bundle case is definitional. In the cons case,
both sides expose the same head probe, and the tail equality is the induction
hypothesis.
\end{proof}
\leanchecked{BEDC.FKernel.Bundle.bundleAppend\_right\_nil}

\begin{theorem}[Bundle append membership]
\label{thm:bundle-append-membership}
For every probe $p$ and bundles $\Pi,\Theta$,
$$
p\in\mathsf{BAppend}(\Pi,\Theta)
\quad\Longleftrightarrow\quad
p\in\Pi \ \text{or}\ p\in\Theta .
$$
\end{theorem}
\leanchecked{BEDC.FKernel.Bundle.inBundle\_bundleAppend\_iff}
\begin{proof}
Induct on $\Pi$. The empty case reduces membership to $\Theta$. The cons case splits head
membership from tail membership, then applies the induction hypothesis to the tail.
\end{proof}

\section{Signature generation}

\begin{definition}[Signature event]
\label{def:signature-event}
The signature event \(\Sig(\Pi,h,s,\Delta)\) means that bundle \(\Pi\) applied to history \(h\) generates answer-history \(s\) with evidence \(\Delta\). It is generated by the rules below.
\end{definition}
\leanchecked{BEDC.FKernel.Sig.sig\_empty\_constructor}
\leanvariant{BEDC.FKernel.Sig.sig\_cons\_constructor}
\leanvariant{BEDC.FKernel.Sig.sig\_cons\_inversion}
\leanvariant{BEDC.FKernel.Sig.sig\_cons\_result\_inversion}
\leanvariant{BEDC.FKernel.Sig.sig\_cons\_tail\_deterministic}
\leanvariant{BEDC.FKernel.Sig.sigRel\_cons\_inversion}
\leanvariant{BEDC.FKernel.Sig.sigRel\_generation\_cases}
\leanvariant{BEDC.FKernel.Sig.sig\_empty\_inversion}

\begin{kernelrule}[Empty signature]
\[
\frac{}{\Sig(\Bnil,h,\emp,\Delta_0)}
\]
where \(\Delta_0\) is the neutral signature evidence token.
\end{kernelrule}

\begin{lemma}[Empty signature inversion]
\label{lem:empty-signature-inversion}
If \(\Sig(\Bnil,h,r,\Delta_0)\), then \(\hsame(r,\emp)\).
\end{lemma}
\leanchecked{BEDC.FKernel.Sig.sig\_empty\_inversion}
\begin{proof}
Invert the signature derivation. For the empty bundle, only the empty-signature rule applies, so the result history is \(\emp\).
\end{proof}

\begin{kernelrule}[Cons signature]
If
\[
\Ask(\pi,h,m,\delta),
\qquad
\Sig(\Pi,h,s,\Delta),
\qquad
\ExtR(m,s,r),
\]
then
\[
\Sig(\Bcons(\pi,\Pi),h,r,\delta\star\Delta).
\]
Here \(\delta\star\Delta\) is the evidence-composition operation licensed by the relevant asking policy.
\end{kernelrule}

\begin{theorem}[Signature generation over bundle append]
\label{thm:signature-generation-bundle-append}
For probe bundles $\Pi,\Theta$ and source history $h$, if $\Sig(\Pi,h,s,\Delta)$ and $\Sig(\Theta,h,t,\Gamma)$, then there exist a history $u$ and evidence $\Lambda$ such that
$$
\ContR(t,s,u)
\quad\text{and}\quad
\Sig(\mathsf{BAppend}(\Pi,\Theta),h,u,\Lambda).
$$
\end{theorem}
\begin{proof}
Induct on $\Pi$. In the empty bundle case, empty-signature inversion identifies $s$ with $\emp$; take $u=t$, use the right-unit continuation constructor, and reuse the given signature for $\Theta$. In the cons case, invert the left signature. The tail signature for $\Pi$ and the signature for $\Theta$ feed the induction hypothesis, producing a tail result $u_0$ with $\ContR(t,s_0,u_0)$ and a signature for $\mathsf{BAppend}(\Pi,\Theta)$. Rebuild the cons signature using the original head asking event and the tail signature. The final extension branch supplies either the $\Ezero$ or $\Eone$ continuation step, yielding $\ContR(t,s,u)$ for the rebuilt result.
\end{proof}

\begin{theorem}[Cons tail signatures from result-sameness]
\label{thm:signature-cons-tail-hsame-from-result-hsame}
If two derivations of \(\Sig(\Bcons(\pi,\Pi),h,r,\Delta)\) and
\(\Sig(\Bcons(\pi,\Pi),h,r',\Theta)\) have \(\hsame(r,r')\), then there exist
tail results \(s,t\) such that \(\Sig(\Pi,h,s,\Delta_s)\), \(\Sig(\Pi,h,t,\Delta_t)\), and
\(\hsame(s,t)\).
\end{theorem}
\leanchecked{BEDC.FKernel.Sig.sig\_cons\_tail\_hsame\_from\_result\_hsame}
\begin{proof}
Invert both cons-signature derivations and their final extension witnesses. Cross-mark extension branches contradict result-sameness by history no-confusion, while matching marks expose matching outer history constructors; constructor injectivity yields \(\hsame\) of the tail results.
\end{proof}

Assume an asking policy on \(D\). If \(h\in D\) and two cons-signature derivations for \(\Bcons(\pi,\Pi)\) start from the same source history \(h\), then their head asking events emit internally same marks.
\end{theorem}

\begin{theorem}[Signature determinacy under an asking policy]
Assume \(\AskPol(\Pi,D)\). If \(h\) is admitted by \(D\) and
\[
\Sig(\Pi,h,s,\Delta),
\qquad
\Sig(\Pi,h,t,\Theta),
\]
then \(\hsame(s,t)\).
\end{theorem}
\leanchecked{BEDC.FKernel.Sig.sig\_deterministic}
\leanvariant{BEDC.FKernel.Sig.sig\_cons\_determinacy\_spine}
\begin{proof}
Induct on the generated bundle \(\Pi\). The empty case uses reflexivity of \(\hsame\). In the cons case, asking determinacy gives \(\msame\) of the first emitted marks. The induction hypothesis gives \(\hsame\) of the tail signatures. Extension respects mark-sameness and history-sameness, so the resulting answer histories are \(\hsame\).
\end{proof}

\section{Signature sameness}

\begin{definition}[Signature-sameness]
\label{def:signature-sameness-witnesses}
\label{def:signature-sameness}
Given a bundle \(\Pi\), histories \(h,k\) are signature-same, written \(\sameSig_\Pi(h,k)\), if there exist generated signatures
\[
\Sig(\Pi,h,s,\Delta),
\qquad
\Sig(\Pi,k,t,\Theta),
\]
such that \(\hsame(s,t)\).
\end{definition}

\begin{theorem}[History-sameness gives signature-sameness under policy]
\label{thm:samesig-of-hsame-under-policy}
Assume an asking policy and total signature generation on \(D\). If \(h,k\in D\) and \(\hsame(h,k)\), then \(\sameSig_\Pi(h,k)\).
\end{theorem}
\leanchecked{BEDC.FKernel.Sig.sameSig\_of\_hsame\_under\_policy}
\leanvariant{BEDC.FKernel.Sig.sameSig\_of\_hsame\_witnesses\_under\_policy}

\leanvariant{BEDC.FKernel.Sig.signature\_totality\_witness\_chain}
\begin{proof}
Choose generated signatures for \(h\) and \(k\) by totality. The asking policy transports \(\hsame(h,k)\) across signature generation, yielding \(\hsame\) of the generated signatures. These witnesses form \(\sameSig_\Pi(h,k)\).
\end{proof}
