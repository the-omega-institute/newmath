\section{Bundle grammar}

\begin{definition}[Probe bundle]
\label{def:probe-bundle}
\label{def:bundle-generation}
Probe bundles are generated by:
\[
\frac{}{\Bnil:\Bundle}
\]
\[
\frac{\pi:\PN\qquad \Pi:\Bundle}{\Bcons(\pi,\Pi):\Bundle}.
\]
Bundle membership is the closed predicate generated by cons membership: the head probe of
\(\Bcons(\pi,\Pi)\) is a member of that bundle, and tail membership propagates through cons.
\end{definition}

\begin{definition}[Bundle append]
\label{def:bundle-append}
The append operation $\mathsf{BAppend}(\Pi,\Theta)$ is recursion on the left bundle:
$$
\mathsf{BAppend}(\Bnil,\Theta)=\Theta,
\qquad
\mathsf{BAppend}(\Bcons(\pi,\Pi),\Theta)
=\Bcons(\pi,\mathsf{BAppend}(\Pi,\Theta)).
$$
\end{definition}
\leandef{BEDC.FKernel.Bundle.bundleAppend}

\begin{theorem}[Probe bundle append right unit]
\label{thm:bundle-append-right-unit}
For every probe bundle \(\Pi\),
$$
\mathsf{BAppend}(\Pi,\Bnil)=\Pi .
$$
\end{theorem}
\begin{proof}
Induct on \(\Pi\). The empty bundle case is definitional. In the cons case,
both sides expose the same head probe, and the tail equality is the induction
hypothesis.
\end{proof}
\leanchecked{BEDC.FKernel.Bundle.bundleAppend\_nil\_right}

\begin{theorem}[Probe bundle append associativity]
\label{thm:probe-bundle-append-associativity}
For all probe bundles \(\Pi,\Theta,\Omega\),
$$
\mathsf{BAppend}(\mathsf{BAppend}(\Pi,\Theta),\Omega)
=
\mathsf{BAppend}(\Pi,\mathsf{BAppend}(\Theta,\Omega)).
$$
\end{theorem}
\begin{proof}
Induct on \(\Pi\). The empty bundle case is definitional. In the cons case,
both sides expose the same head probe, and the tail equality is the induction
hypothesis.
\end{proof}
\leanchecked{BEDC.FKernel.Bundle.bundleAppend\_assoc}

\begin{theorem}[Bundle append membership]
\label{thm:bundle-append-membership}
For every probe $p$ and bundles $\Pi,\Theta$,
$$
p\in\mathsf{BAppend}(\Pi,\Theta)
\quad\Longleftrightarrow\quad
p\in\Pi \ \text{or}\ p\in\Theta .
$$
\end{theorem}
\leanchecked{BEDC.FKernel.Bundle.inBundle\_bundleAppend\_iff}
\begin{proof}
Induct on $\Pi$. The empty case reduces membership to $\Theta$. The cons case splits head
membership from tail membership, then applies the induction hypothesis to the tail.
\end{proof}

\begin{theorem}[Probe bundle prefix cancellation]
\label{thm:probe-bundle-prefix-cancellation}
For any fixed probe bundle $\Pi$ and any probe bundles $\Theta,\Omega$, if
$$
\mathsf{BAppend}(\Pi,\Theta)=\mathsf{BAppend}(\Pi,\Omega),
$$
then $\Theta=\Omega$.
\end{theorem}
\leanchecked{BEDC.FKernel.Bundle.bundleAppend\_prefix\_cancel}
\begin{proof}
Induct on $\Pi$. The empty case is immediate because both displayed append terms
reduce to their suffix bundles. In the cons case, constructor injectivity reduces
the equality of cons bundles to equality of their tails, and the induction
hypothesis cancels the remaining common prefix.
\end{proof}

\begin{lemma}[Probe bundle append cons-result inversion]
\label{lem:probe-bundle-append-cons-result-inversion}
For probe bundles $\Pi,\Theta,\Omega$ and probe $\pi$, if
$$
\mathsf{BAppend}(\Pi,\Theta)=\Bcons(\pi,\Omega),
$$
then either $\Pi=\Bnil$ and $\Theta=\Bcons(\pi,\Omega)$, or there exists a bundle $\Pi_0$ such that $\Pi=\Bcons(\pi,\Pi_0)$ and
$$
\mathsf{BAppend}(\Pi_0,\Theta)=\Omega.
$$
\end{lemma}
\begin{proof}
Proceed by cases on $\Pi$. In the empty case the displayed equality reduces to $\Theta=\Bcons(\pi,\Omega)$, giving the left alternative. In the cons case the append definition exposes a cons bundle on the left; constructor injectivity identifies its head with $\pi$ and its appended tail with $\Omega$, giving the required tail bundle $\Pi_0$.
\end{proof}

\section{Signature generation}

\begin{definition}[Signature event]
\label{def:signature-event}
The signature event \(\Sig(\Pi,h,s,\Delta)\) means that bundle \(\Pi\) applied to history \(h\) generates answer-history \(s\) with evidence \(\Delta\). It is generated by the rules below.
In the checked kernel, the logical predicate is the three-place $\mathsf{SigRel}(\Pi,h,s)$ on bundle, source history, and answer history. The displayed evidence ledger is a paper-side annotation attached to the generating derivation.
\end{definition}

\begin{kernelrule}[Empty signature]
\[
\frac{}{\Sig(\Bnil,h,\emp,\Delta_0)}
\]
where \(\Delta_0\) is the neutral signature evidence token.
\end{kernelrule}

\begin{lemma}[Empty signature inversion]
\label{lem:empty-signature-inversion}
If \(\Sig(\Bnil,h,r,\Delta_0)\), then \(\hsame(r,\emp)\).
\end{lemma}
\leanchecked{BEDC.FKernel.Sig.sig\_empty\_inversion}
\begin{proof}
Invert the signature derivation. For the empty bundle, only the empty-signature rule applies, so the result history is \(\emp\).
\end{proof}

\begin{kernelrule}[Cons signature]
If
\[
\Ask(\pi,h,m,\delta),
\qquad
\Sig(\Pi,h,s,\Delta),
\qquad
\ExtR(m,s,r),
\]
then
\[
\Sig(\Bcons(\pi,\Pi),h,r,\delta\star\Delta).
\]
Here \(\delta\star\Delta\) records the head asking evidence together with the tail derivation ledger; it is not a field of the asking policy.
\end{kernelrule}

\begin{theorem}[Signature generation over bundle append]
\label{thm:signature-generation-bundle-append}
For probe bundles $\Pi,\Theta$ and source history $h$, if $\Sig(\Pi,h,s,\Delta)$ and $\Sig(\Theta,h,t,\Gamma)$, then there exist a history $u$ and evidence $\Lambda$ such that
$$
\ContR(t,s,u)
\quad\text{and}\quad
\Sig(\mathsf{BAppend}(\Pi,\Theta),h,u,\Lambda).
$$
\end{theorem}
\begin{proof}
Induct on $\Pi$. In the empty bundle case, empty-signature inversion identifies $s$ with $\emp$; take $u=t$, use the right-unit continuation constructor, and reuse the given signature for $\Theta$. In the cons case, invert the left signature. The tail signature for $\Pi$ and the signature for $\Theta$ feed the induction hypothesis, producing a tail result $u_0$ with $\ContR(t,s_0,u_0)$ and a signature for $\mathsf{BAppend}(\Pi,\Theta)$. Rebuild the cons signature using the original head asking event and the tail signature. The final extension branch supplies either the $\Ezero$ or $\Eone$ continuation step, yielding $\ContR(t,s,u)$ for the rebuilt result.
\end{proof}

\begin{theorem}[Signature generation append inversion]
\label{thm:signature-generation-bundle-append-inversion}
For probe bundles $\Pi,\Theta$ and source history $h$, if
$\Sig(\mathsf{BAppend}(\Pi,\Theta),h,u,\Lambda)$, then there exist histories
$s,t$ and evidence tokens $\Delta,\Gamma$ such that
$\Sig(\Pi,h,s,\Delta)$, $\Sig(\Theta,h,t,\Gamma)$, and $\ContR(t,s,u)$.
\end{theorem}
\begin{proof}
Induct on $\Pi$. In the empty bundle case, the appended bundle is $\Theta$;
generate the empty signature for $\Bnil$, take $s=\emp$ and $t=u$, reuse the
given signature for $\Theta$, and apply the continuation right-unit
constructor. In the cons case, invert the appended cons signature. The
inversion exposes the head asking event, the tail signature for
$\mathsf{BAppend}(\Pi,\Theta)$, and the final extension witness. Apply the
induction hypothesis to the tail append signature to obtain signatures for
$\Pi$ and $\Theta$ together with a continuation $\ContR(t,s_0,u_0)$. Rebuild
the left cons signature from the exposed head asking event, the recovered tail
signature for $\Pi$, and the final extension witness. The same extension
constructor lifts the recovered tail continuation to $\ContR(t,s,u)$ for the
whole appended signature.
\end{proof}

\begin{theorem}[Signature-sameness closure under bundle append]
\label{thm:samesig-bundle-append-closure}
For probe bundles $\Pi,\Theta$ and histories $h,k$, if $\sameSig_\Pi(h,k)$ and $\sameSig_\Theta(h,k)$, then
$$
\sameSig_{\mathsf{BAppend}(\Pi,\Theta)}(h,k).
$$
\end{theorem}
\begin{proof}
Unpack the two signature-sameness witnesses. The $\Pi$ witnesses give signatures from $h$ and $k$ whose answer histories are $\hsame$; the $\Theta$ witnesses give another such pair. Apply Theorem~\ref{thm:signature-generation-bundle-append} to the two $h$-side signatures and separately to the two $k$-side signatures. Continuation respect for $\hsame$ transports the two answer-history sameness witnesses across the appended continuations, so the appended signatures have $\hsame$ results. These appended signatures are the witnesses for $\sameSig_{\mathsf{BAppend}(\Pi,\Theta)}(h,k)$.
\end{proof}

\begin{theorem}[Cons tail signatures from result-sameness]
\label{thm:signature-cons-tail-hsame-from-result-hsame}
If two derivations of \(\Sig(\Bcons(\pi,\Pi),h,r,\Delta)\) and
\(\Sig(\Bcons(\pi,\Pi),h,r',\Theta)\) have \(\hsame(r,r')\), then there exist
tail results \(s,t\) such that \(\Sig(\Pi,h,s,\Delta_s)\), \(\Sig(\Pi,h,t,\Delta_t)\), and
\(\hsame(s,t)\).
\end{theorem}
\leanchecked{BEDC.FKernel.Sig.sig\_cons\_tail\_hsame\_from\_result\_hsame}
\begin{proof}
Invert both cons-signature derivations and their final extension witnesses. Cross-mark extension branches contradict result-sameness by history no-confusion, while matching marks expose matching outer history constructors; constructor injectivity yields \(\hsame\) of the tail results.
\end{proof}

\begin{theorem}[Signature determinacy under an asking policy]
Assume \(\AskPol(\Pi,D)\). If \(h\) is admitted by \(D\) and
\[
\Sig(\Pi,h,s,\Delta),
\qquad
\Sig(\Pi,h,t,\Theta),
\]
then \(\hsame(s,t)\).
\end{theorem}
\leanchecked{BEDC.FKernel.Sig.sig\_deterministic}
\leanvariant{BEDC.FKernel.Sig.sig\_cons\_determinacy\_spine}
\begin{proof}
Induct on the generated bundle \(\Pi\). The empty case uses reflexivity of \(\hsame\). In the cons case, asking determinacy gives \(\msame\) of the first emitted marks. The induction hypothesis gives \(\hsame\) of the tail signatures. Extension respects mark-sameness and history-sameness, so the resulting answer histories are \(\hsame\).
\end{proof}

\section{Signature sameness}

\begin{definition}[Signature-sameness]
\label{def:signature-sameness-witnesses}
\label{def:signature-sameness}
Given a bundle \(\Pi\), histories \(h,k\) are signature-same, written \(\sameSig_\Pi(h,k)\), if there exist generated signatures
\[
\Sig(\Pi,h,s,\Delta),
\qquad
\Sig(\Pi,k,t,\Theta),
\]

\begin{theorem}[Signature-sameness endpoint replacement]
\label{thm:signature-sameness-endpoint-replacement}
Assume $\AskPol(\Pi,D)$. If $h,h',k,k'\in D$, $\hsame(h,h')$, $\sameSig_\Pi(h,k)$, and $\hsame(k,k')$, then
$$
  \sameSig_\Pi(h',k').
$$
\end{theorem}

\begin{proof}
By Theorem~\ref{thm:samesig-of-hsame-under-policy}, the left endpoint hypothesis gives $\sameSig_\Pi(h,h')$, and the right endpoint hypothesis gives $\sameSig_\Pi(k,k')$. Symmetry from Theorem~\ref{thm:signature-sameness-equivalence} turns the first relation into $\sameSig_\Pi(h',h)$. Transitivity through the middle $h$ yields $\sameSig_\Pi(h',k)$, and transitivity through the middle $k$ yields $\sameSig_\Pi(h',k')$.
\end{proof}
