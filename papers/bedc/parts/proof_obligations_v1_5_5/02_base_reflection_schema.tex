\chapter{Base Reflection Schema}
\label{ch:v155-base-reflection-schema}

This chapter isolates the smallest constructor-inversion chain needed for base package reflection. It is the first proof-check target after v1.5.5.

\begin{definition}[Base package witness data]
\label{def:v155-psame-base-data}
For fixed \(\Pi,p,q\), define the witness predicate
\[
\PBaseData(\Pi,p,q) :\Longleftrightarrow
\exists s_0,t_0,\;\TokIntro(\Pi,s_0,p)\wedge\TokIntro(\Pi,t_0,q)\wedge\hsame(s_0,t_0).
\]
\end{definition}

\begin{lemma}[Base constructor]
\label{lem:v155-psame-base-constructor}
If
\[
  \TokIntro(\Pi,s,p),\quad \TokIntro(\Pi,t,q),\quad \hsame(s,t),
\]
then
\[
  \psame_{\Pi}^{\mathrm{base}}(p,q).
\]
\end{lemma}

\begin{proof}
This is the single constructor of Definition~\ref{def:v154-base-psame}. The important checker rule is that no second public constructor may be added without changing the versioned contract.
\end{proof}

\begin{lemma}[Base inversion]
\label{lem:v155-psame-base-inversion}
If \(\psame_{\Pi}^{\mathrm{base}}(p,q)\), then \(\PBaseData(\Pi,p,q)\).
\end{lemma}

\begin{proof}
Invert the one-constructor inductive proof of \(\psame_{\Pi}^{\mathrm{base}}(p,q)\). The inversion exposes introducing signatures \(s_0,t_0\) and a proof \(\hsame(s_0,t_0)\). This is exactly \(\PBaseData(\Pi,p,q)\).
\end{proof}

\begin{lemma}[Token replacement]
\label{lem:v155-token-replacement}
Assume \(\TokUnique(\Pi)\). If
\[
  \TokIntro(\Pi,s,p),\quad \TokIntro(\Pi,t,p),
\]
then \(\hsame(s,t)\).
\end{lemma}

\begin{proof}
This is the token-uniqueness field applied at the same visible package token \(p\). In a canonical-constructor implementation it may instead be obtained by the no-confusion theorem described in Definition~\ref{def:v154-canonical-mode}.
\end{proof}

\begin{theorem}[Base package reflection]
\label{thm:v155-base-package-reflection}
Assume \(\TokUnique(\Pi)\). If
\[
  \TokIntro(\Pi,s,p),\quad \TokIntro(\Pi,t,q),\quad \psame_{\Pi}^{\mathrm{base}}(p,q),
\]
then
\[
  \hsame(s,t).
\]
\end{theorem}

\begin{proof}
By Lemma~\ref{lem:v155-psame-base-inversion}, choose witnesses \(s_0,t_0\) such that
\[
  \TokIntro(\Pi,s_0,p),\quad \TokIntro(\Pi,t_0,q),\quad \hsame(s_0,t_0).
\]
Token replacement at \(p\) gives \(\hsame(s,s_0)\). Token replacement at \(q\) gives \(\hsame(t,t_0)\), hence by symmetry \(\hsame(t_0,t)\). Transitivity of \(\hsame\) gives
\[
  \hsame(s,s_0),\quad \hsame(s_0,t_0),\quad \hsame(t_0,t)
  \quad\Longrightarrow\quad \hsame(s,t).
\]
\end{proof}
