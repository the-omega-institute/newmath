\chapter{Optional Closure Reflection Target}
\label{ch:v155-closure-reflection-target}

The equivalence closure of the base package relation is useful for later package algebra, but it must not be confused with the active exact relation. This chapter records the optional proof target without exporting it.

\begin{definition}[Equivalence closure target]
\label{def:v155-psame-eq-target}
The optional relation remains
\[
  \psame_{\Pi}^{\mathrm{eq}}(p,q) := \EqCl(\psame_{\Pi}^{\mathrm{base}})(p,q).
\]
It is not part of \(\ExactGlobalizeBase(\Pi,D)\).
\end{definition}

\begin{protocol}[Closure-reflection proof route]
\label{prot:v155-closure-reflection-route}
A future theorem \texttt{PackageReflection.eqClosure} should prove \(\ClosureReflect(\Pi)\) by induction over \(\EqCl(\psame_{\Pi}^{\mathrm{base}})\). The intended cases are:
\begin{enumerate}
  \item reflexive case: use two introductions of the same token and \(\TokUnique(\Pi)\);
  \item base-edge case: use Theorem~\ref{thm:v155-base-package-reflection};
  \item symmetric case: use symmetry of \(\hsame\);
  \item transitive case: use transitivity of \(\hsame\).
\end{enumerate}
\end{protocol}

\begin{proposition}[Conditional closure-reflection schema]
\label{prop:v155-conditional-closure-reflection-schema}
Assume \(\hsame\) is an equivalence relation, assume \(\TokUnique(\Pi)\), and assume \(\psame_{\Pi}^{\mathrm{eq}}\) is implemented as the ordinary reflexive-symmetric-transitive closure of \(\psame_{\Pi}^{\mathrm{base}}\). Then the proof route of Protocol~\ref{prot:v155-closure-reflection-route} is sufficient to prove
\[
\begin{aligned}
  &\TokIntro(\Pi,s,p),\quad \TokIntro(\Pi,t,q),\quad \psame_{\Pi}^{\mathrm{eq}}(p,q)\\
  &\hspace{3cm}\Longrightarrow\quad \hsame(s,t).
\end{aligned}
\]
\end{proposition}

\begin{proof}
Induct on the derivation of \(\psame_{\Pi}^{\mathrm{eq}}(p,q)\). The reflexive case compares two introductions of one token by token uniqueness. The base-edge case is Theorem~\ref{thm:v155-base-package-reflection}. The symmetric case reverses the induction result by symmetry of \(\hsame\). The transitive case composes the two induction results by transitivity of \(\hsame\).
\end{proof}

\begin{corollary}[Still not active]
\label{cor:v155-closure-still-not-active}
Proposition~\ref{prop:v155-conditional-closure-reflection-schema} is a paper-level proof schema. It does not activate \(\psame_{\Pi}^{\mathrm{eq}}\)-exactness unless a future checked report records the actual theorem and imports it into the export contract.
\end{corollary}

\begin{proof}
The active export discipline is Protocol~\ref{prot:v154-export-discipline} together with Corollary~\ref{cor:v155-no-closure-export}. A proof schema is not a checked artifact.
\end{proof}
