\chapter{Checker-Friendly Exactness Predicate}
\label{ch:v155-checker-friendly-exactness}

The v1.5.4 exact concrete Globalize theorem intentionally stated the active relation as \(\psame_{\Pi}^{\mathrm{base}}\). The remaining problem was not the relation boundary, but the phrase ``classify admitted histories exactly''. This chapter replaces that phrase by a predicate with explicit fields.

\begin{definition}[Generated signature sameness]
\label{def:v155-generated-same-sig}
For a generated probe bundle \(\Pi\), define
\[
\begin{aligned}
\GeneratedSameSig(\Pi,h,k) :\Longleftrightarrow\;&
\exists s,t,\Delta,\Theta,\;\Sig(\Pi,h,s,\Delta)\;\wedge\;\Sig(\Pi,k,t,\Theta)\;\wedge\;\hsame(s,t).
\end{aligned}
\]
This is the checker-facing predicate corresponding to ``the two admitted histories generate the same visible signature''.
\end{definition}

\begin{definition}[Base exact Globalize structure]
\label{def:v155-exact-globalize-base}
For a bundle \(\Pi\) and a concrete domain policy \(D\), the proposition
\[
  \ExactGlobalizeBase(\Pi,D)
\]
is the conjunction of the following four fields.
\begin{enumerate}
  \item \textbf{Coverage.}
  \[
    \forall h,\; \InDom(D,h)\Rightarrow \exists p,\;\InGapSig(\Pi,D,p,h).
  \]
  \item \textbf{Base soundness.}
  \[
  \begin{aligned}
    \forall h,k,p,q,\;&\InGapSig(\Pi,D,p,h)\wedge\InGapSig(\Pi,D,q,k)\wedge\GeneratedSameSig(\Pi,h,k)\;\Rightarrow\\
    &\psame_{\Pi}^{\mathrm{base}}(p,q).
  \end{aligned}
  \]
  \item \textbf{Base completeness.}
  \[
  \begin{aligned}
    \forall h,k,p,q,\;&\InGapSig(\Pi,D,p,h)\wedge\InGapSig(\Pi,D,q,k)\wedge\psame_{\Pi}^{\mathrm{base}}(p,q)\;\Rightarrow\\
    &\GeneratedSameSig(\Pi,h,k).
  \end{aligned}
  \]
  \item \textbf{Classification iff.}
  \[
  \begin{aligned}
    \forall h,k,p,q,\;&\InGapSig(\Pi,D,p,h)\wedge\InGapSig(\Pi,D,q,k)\;\Rightarrow\\
    &\bigl(\psame_{\Pi}^{\mathrm{base}}(p,q)\Longleftrightarrow \GeneratedSameSig(\Pi,h,k)\bigr).
  \end{aligned}
  \]
\end{enumerate}
The fourth field may be implemented as a derived theorem from soundness and completeness; it is recorded here because it is the public export shape.
\end{definition}

\begin{proposition}[Classification is derivable from the two directions]
\label{prop:v155-iff-derived}
If coverage, base soundness, and base completeness hold, then for all \(h,k,p,q\) satisfying the two gap premises,
\[
  \psame_{\Pi}^{\mathrm{base}}(p,q)\Longleftrightarrow \GeneratedSameSig(\Pi,h,k).
\]
\end{proposition}

\begin{proof}
The forward implication is base completeness. The reverse implication is base soundness. Coverage is not needed for this local equivalence, but it remains part of the global exactness structure because exact Globalize must also classify every admitted history, not only pairs already carrying visible packages.
\end{proof}

\begin{protocol}[Export replacement for prose exactness]
\label{prot:v155-export-replacement}
Future modules should cite \(\ExactGlobalizeBase(\Pi,D)\) rather than a prose phrase such as ``visible packages classify histories exactly''. If a theorem needs only the local iff, it should cite the classification field. If it needs existence of a visible package for every admitted history, it must also cite coverage.
\end{protocol}
