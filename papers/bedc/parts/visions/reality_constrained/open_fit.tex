\section{Finite observation and open fit}
\label{sec:finite-observation-open-fit}

\begin{definition}[Reality trace]
\label{def:reality-trace}
For a physical system $R$, a reality trace of length $n$ is a finite sequence
$$
H_n(R)=(o_0,o_1,\ldots,o_n)
$$
of observations generated by interacting with $R$ through an admissible observation protocol.
\end{definition}

The trace is not reality itself. It is the finite observational history through which reality becomes available to a model.

\begin{definition}[Observation bundle]
\label{def:observation-bundle}
An observation bundle $\Pi$ is a finite or recursively extendable family of admissible questions. Given $R$, $M$, and a trace $H_n$, the observed and model-generated signatures are written
$$
\operatorname{ObsSig}_{\Pi}(R\mid H_n),
\qquad
\operatorname{ModSig}_{\Pi}(M\mid H_n).
$$
\end{definition}

\begin{definition}[Finite observational fit]
\label{def:finite-observational-fit}
A model $M$ has finite observational fit to a system $R$ at trace length $n$, relative to $\Pi$, when
$$
\operatorname{ModSig}_{\Pi}(M\mid H_n)
\sim_{\Pi}
\operatorname{ObsSig}_{\Pi}(R\mid H_n).
$$
This judgment is denoted $\operatorname{Fit}_n(M,R;\Pi)$.
\end{definition}

Finite fit says only that $M$ has not been separated from $R$ by the finite observations available up to $H_n$. It does not assert $M=R$.

\begin{definition}[Open physical fit]
\label{def:open-physical-fit}
Open physical fit, written $\operatorname{OpenFit}(M,R;\Pi)$, is the continuing condition that every actually obtained finite trace carries a finite fit certificate, while no internal final certificate is assumed for
$$
\forall n,\operatorname{Fit}_n(M,R;\Pi).
$$
\end{definition}

\concretizedIn{ch:concrete-instances-open-fit-residue-gate-namecert}{Open physical fit becomes a finite residue-gate NameCert packet whose refutation row stays visible.}

\begin{definition}[Physical induction certificate]
\label{def:physical-induction-certificate}
A physical induction certificate is a stability certificate
$$
\operatorname{PhysInd}(M,R;\Pi)
$$
licensing a transition from previously observed fit to predicted continuation fit:
$$
\operatorname{Fit}_n(M,R;\Pi)
\leadsto
\operatorname{PredFit}_{n+1}(M,R;\Pi).
$$
\end{definition}

This differs from mathematical induction. Mathematical induction applies after the carrier and its constructors are already closed. Physical induction is a stability claim about future observations of a system whose complete generative law is not externally given.

\begin{definition}[Final-fit oracle]
\label{def:final-fit-oracle}
A final-fit oracle for $M$ and $R$ is a hypothetical judgment $\Omega(M,R;\Pi)$ such that
$$
\Omega(M,R;\Pi)
\Longleftrightarrow
\forall H,\;
\operatorname{ModSig}_{\Pi}(M\mid H)
\sim_{\Pi}
\operatorname{ObsSig}_{\Pi}(R\mid H),
$$
where $H$ ranges over all admissible actual and counterfactual observation histories.
\end{definition}

The framework rejects such an oracle as an internally available physical object. It would be an observer above all observers, contrary to the no-global-standpoint discipline of \autoref{ch:visions-inter-hist-locality}.

\section{Open-fit theorems}
\label{sec:open-fit-theorems}

\begin{theorem}[Finite fit does not imply total fit]
\label{thm:finite-fit-does-not-imply-total-fit}
For any finite trace length $n$,
$$
\operatorname{Fit}_n(M,R;\Pi)
$$
does not imply
$$
\forall k\geq n,\operatorname{Fit}_k(M,R;\Pi).
$$
\end{theorem}

\begin{proof}
By \autoref{def:finite-observational-fit}, $\operatorname{Fit}_n(M,R;\Pi)$ compares model and system only on the finite trace $H_n$. The judgment contains no information about observations not yet made. Two continuations
$$
H_n\cdot o_{n+1}
\quad\text{and}\quad
H_n\cdot o'_{n+1}
$$
may agree on the prefix and separate at the next observation. A model can match $R$ on the prefix and fail on one continuation. To infer future fit one must add a stability or physical induction certificate.
\end{proof}

\begin{corollary}[Empirical truth is not model identity]
\label{cor:empirical-truth-not-model-identity}
A successful finite model fit supports undefeatedness by the displayed trace, not model-reality identity.
\end{corollary}

\begin{proof}
By \autoref{thm:finite-fit-does-not-imply-total-fit}, finite fit cannot imply total fit. Since identity would require agreement across all admissible observations and all reality-facing properties, finite fit is weaker than identity.
\end{proof}

\begin{theorem}[Mathematical induction and physical induction have different closures]
\label{thm:math-induction-physical-induction-different-closures}
Mathematical induction is valid when the target carrier is closed under known constructors. Physical induction requires an additional stability certificate because the complete future observation carrier of reality is not internally given.
\end{theorem}

\begin{proof}
Mathematical induction over a carrier such as $\mathbb{N}$ relies on a closed generation rule:
$$
0\in\mathbb{N},
\qquad
n\in\mathbb{N}\Longrightarrow S(n)\in\mathbb{N}.
$$
The base case and successor step range over the closure of that carrier. Physical observation begins from a finite trace $H_n=(o_0,\ldots,o_n)$, but the observer does not possess a constructor theorem saying that every future reality-facing observation arises by a known successor rule compatible with the model. The transition from $\operatorname{Fit}_n$ to future fit therefore requires $\operatorname{PhysInd}$ as additional stability data.
\end{proof}

\begin{corollary}[Physical intuition is a classifier prior]
\label{cor:physical-intuition-classifier-prior}
What is often called physical intuition is a pre-formal expectation that future observations will remain within the current classifier.
\end{corollary}

\begin{proof}
By \autoref{thm:math-induction-physical-induction-different-closures}, future fit does not follow from finite fit alone. Expectations based on prior experience, symmetry, scale, or dynamical regularity function as priors over admissible continuations. They await certification by further observation or by a displayed stability argument.
\end{proof}

\begin{corollary}[Physical law as stability certificate]
\label{cor:physical-law-as-stability-certificate}
A physical law is not merely an equation. It is a certified stability claim that a model-generated signature continues to match observed signatures under admissible extensions of observation.
\end{corollary}

\begin{proof}
By \autoref{thm:math-induction-physical-induction-different-closures}, finite fit alone does not justify future prediction. Prediction requires a certificate licensing continuation from current fit to later fit. Since a physical law is used precisely to support such continuation, its certificate role is stability over an open observation process, together with an open surface on which future observations can refute the licensed continuation.
\end{proof}

\begin{theorem}[No internal final-fit oracle]
\label{thm:no-internal-final-fit-oracle}
No observer internal to the physical inquiry process can supply a final-fit oracle $\Omega(M,R;\Pi)$.
\end{theorem}

\begin{proof}
An internal observer has a finite history and local observational constraints. To supply $\Omega(M,R;\Pi)$ it would have to certify agreement across all admissible observations, including not-yet-performed observations and observations from other possible perspectives. If the certifying observer is external to all observation histories, it is not internal to the inquiry system. If it is internal, it is itself a history subject to the same locality and no-global-synchronization constraints recorded in \autoref{ch:visions-inter-hist-locality}. In neither case is an internal final-fit oracle obtained.
\end{proof}

\begin{corollary}[Final theory is not an internal scientific certificate]
\label{cor:final-theory-not-internal-certificate}
A physical theory may be undefeated by every available observation, but the judgment that it is final cannot be internally certified.
\end{corollary}

\begin{proof}
Such a judgment would require a final-fit oracle over all possible observations. \autoref{thm:no-internal-final-fit-oracle} excludes that oracle as an internal object.
\end{proof}

\begin{theorem}[Model fit is itself an approximation tower]
\label{thm:model-fit-is-approximation-tower}
The judgment that a model approximates reality is itself obtained through a sequence of approximations.
\end{theorem}

\begin{proof}
To assert that $M$ approximates $R$, one observes $R$, obtaining a finite trace $H_n$ rather than $R$ itself. The trace is converted into $\operatorname{ObsSig}_{\Pi}(R\mid H_n)$. The model produces $\operatorname{ModSig}_{\Pi}(M\mid H_n)$. The two signatures are compared by $\sim_{\Pi}$, yielding a finite fit certificate. Thus the adequacy judgment factors as
$$
R
\longrightarrow
H_n
\longrightarrow
\operatorname{ObsSig}_{\Pi}
\longrightarrow
\operatorname{Fit}_n
\longrightarrow
\operatorname{OpenFit}.
$$
Each arrow hides information and therefore requires a ledger when later use may depend on the hidden layer.
\end{proof}

\begin{corollary}[Evidence for approximation is approximate]
\label{cor:evidence-for-approximation-is-approximate}
There are two distinct approximation judgments: the model's approximation to the system, and the finite evidence that licenses that approximation judgment. The second is more primitive, because the first is accessible only through it.
\end{corollary}

\begin{proof}
By \autoref{thm:model-fit-is-approximation-tower}, the claim that $M$ approximates $R$ is produced through finite traces, signature extraction, classifier comparison, and fit certificates. These mediating structures are finite and revisable. Therefore the evidence for approximation is itself approximation-relative.
\end{proof}

\begin{corollary}[Falsification is finite and confirmation is open]
\label{cor:falsification-finite-confirmation-open}
A model can fail by one finite observation, but it cannot be finally confirmed by finitely many observations.
\end{corollary}

\begin{proof}
A finite trace may produce
$$
\operatorname{ModSig}_{\Pi}(M\mid H_n)
\not\sim_{\Pi}
\operatorname{ObsSig}_{\Pi}(R\mid H_n),
$$
which refutes finite fit. Final confirmation would require all admissible finite prefixes, which no single finite trace supplies.
\end{proof}
