\section{Physical induction and open continuation}
\label{sec:physical-induction-open-continuation}

An approximation tower makes a model usable only by certification. The next question is how an endpoint can be reused beyond the observations that licensed it. This is the problem of induction. Mathematical induction is licensed by a closed generated carrier. Physical induction is different: future observation is not already present as a completed carrier inside the current finite trace. The observer has a finite history, a model extracted from that history, and a stability certificate claiming that the model remains adequate under specified continuations.

\begin{definition}[Closed mathematical carrier]
\label{def:closed-mathematical-carrier}
A closed mathematical carrier is a domain $A$ equipped with constructors such that every element of $A$ is generated by finitely many applications of those constructors, and no further elements are admitted except by the stated generation rules. A proof principle over $A$ is closed when it reasons only over this generated carrier.
\end{definition}

\begin{definition}[Open observation carrier]
\label{def:open-observation-carrier}
An open observation carrier for a physical system $R$ is an extendable family
$$
\mathcal H(R)=\{H_0,H_1,H_2,\ldots\}
$$
of possible finite observation histories, where each $H_n$ is finite and no internal final object is assumed to contain all admissible future histories.
\end{definition}

\begin{definition}[Observational prefix]
\label{def:observational-prefix}
For finite histories $H_n=(o_0,\ldots,o_n)$ and $H_m$, $H_n$ is an observational prefix of $H_m$, written
$$
H_n\preceq H_m,
$$
when $m\geq n$ and the first $n+1$ observations of $H_m$ agree with $H_n$.
\end{definition}

\begin{definition}[Predictive continuation]
\label{def:predictive-continuation}
A predictive continuation of a model $M$ from a finite fit $\operatorname{Fit}_n(M,R;\Pi)$ is a licensed claim
$$
\operatorname{PredFit}_{n\to n+k}(M,R;\Pi)
$$
or an analogous claim over an explicitly stated class of continuations. It is not a theorem of finite fit alone; it requires a stability certificate.
\end{definition}

\begin{definition}[Physical induction certificate]
\label{def:physical-induction-certificate-refined}
A physical induction certificate is a stability certificate
$$
\operatorname{PhysInd}(M,R;\Pi,\mathcal C)
$$
for a class $\mathcal C$ of admissible continuations of the current observation history. It licenses
$$
\operatorname{Fit}_n(M,R;\Pi)
\leadsto
\operatorname{PredFit}_{n\to\mathcal C}(M,R;\Pi).
$$
\end{definition}

\begin{definition}[Induction ledger and failure surface]
\label{def:induction-ledger-failure-surface}
The induction ledger attached to $\operatorname{PhysInd}(M,R;\Pi,\mathcal C)$ is
$$
\operatorname{IndLedger}(M,R;\Pi,\mathcal C)
=
(H_n,\Pi,\mathcal C,\operatorname{FitCert}_n,
\operatorname{StabilityBasis},
\operatorname{FailureSurface}).
$$
Its failure surface is the class of continuations that would separate model-generated and observed signatures:
$$
\operatorname{Fail}(M,R;\Pi,\mathcal C)
=
\{\,H'\in\mathcal C
\mid
\operatorname{ModSig}_{\Pi}(M\mid H')
\not\sim_{\Pi}
\operatorname{ObsSig}_{\Pi}(R\mid H')\,\}.
$$
\end{definition}

\section{Mathematical and physical induction}
\label{sec:mathematical-physical-induction}

\begin{theorem}[Mathematical induction requires closed carrier]
\label{thm:mathematical-induction-requires-closed-carrier}
A mathematical induction principle is licensed only when its target carrier is closed under the constructors used in the induction.
\end{theorem}

\begin{proof}
An induction proof shows that a property holds at the base constructors and is preserved by each generating constructor. This proves the property for all elements only because every element of the carrier is generated by those constructors. If the carrier were not closed, an ungenerated element could exist outside the reach of the induction step. Thus induction requires closed generation.
\end{proof}

\begin{corollary}[Induction is a closure theorem]
\label{cor:induction-is-closure-theorem}
Mathematical induction is not repeated verification. It is a theorem about a closed generated domain.
\end{corollary}

\begin{proof}
Repeated successful cases do not exclude ungenerated cases. The decisive premise is that the carrier has no elements outside the constructor closure.
\end{proof}

\begin{theorem}[Physical induction is not mathematical induction]
\label{thm:physical-induction-not-mathematical-induction}
The inference from finite observational success to future predictive success is not licensed by mathematical induction alone.
\end{theorem}

\begin{proof}
Let $\operatorname{Fit}_n(M,R;\Pi)$ hold. This says that $M$ agrees with $R$ on the finite prefix $H_n$ under $\Pi$. Future success would require a principle covering admissible continuations of $H_n$. But the future observation carrier is open: it is not given as a closed carrier whose elements are generated by a known successor rule that preserves the model. Therefore the transition from finite fit to future prediction requires an additional stability certificate. That certificate may be physically justified, experimentally supported, or theoretically motivated, but it is not mathematical induction over a closed carrier.
\end{proof}

\begin{corollary}[Repeated success is not necessity]
\label{cor:repeated-success-not-necessity}
No finite sequence of successful predictions entails that a model must succeed on every future admissible observation.
\end{corollary}

\begin{proof}
A finite sequence of successes extends fit to a larger prefix. By \autoref{thm:physical-induction-not-mathematical-induction}, no finite prefix closes the open carrier of future observations.
\end{proof}

\begin{corollary}[Physical induction is certificate-relative]
\label{cor:physical-induction-certificate-relative}
A physical inductive inference is valid only relative to the continuation class and stability certificate that license it.
\end{corollary}

\begin{proof}
\autoref{def:physical-induction-certificate-refined} includes a continuation class $\mathcal C$. The certificate licenses prediction over that class. A different continuation class may expose hidden residue or violate the stability basis.
\end{proof}

\section{Law certificates, reuse, and descent}
\label{sec:law-certificates-reuse-descent}

\begin{definition}[Law certificate]
\label{def:law-certificate}
A law certificate for a model $M$ of a system $R$ relative to $\Pi$ is
$$
\operatorname{LawCert}(M,R;\Pi)
=
(\operatorname{SourceFit},
\operatorname{Pattern},
\operatorname{Classifier},
\operatorname{Stability},
\operatorname{FailureSurface},
\operatorname{Ledger}).
$$
It records the finite observations supporting the law, the claimed invariant or regularity, the equivalence under which fit is judged, the continuation principle licensing reuse, the observations that would defeat the law, and the hidden assumptions or tower levels.
\end{definition}

\begin{definition}[Lawlike stability]
\label{def:lawlike-stability}
A model $M$ has lawlike stability for $R$ under $\Pi$ and continuation class $\mathcal C$ when finite fit holds and a law certificate licenses
$$
\forall H'\in\mathcal C,\;
\operatorname{PredFit}(M,R;\Pi\mid H').
$$
The universal statement is certificate-bounded, not an oracle over all possible reality.
\end{definition}

\begin{theorem}[Physical law is not merely equation]
\label{thm:physical-law-not-merely-equation}
A physical law requires a law certificate. An equation alone is insufficient.
\end{theorem}

\begin{proof}
An equation supplies formal syntax and a calculation rule. A physical law must also specify the observations from which it is licensed, the physical pattern it preserves, the classifier comparing model and reality, the stability condition for reuse, the observations that would count as failure, and the ledger of idealizations or hidden assumptions. Without these fields, the equation is not tied to physical reality as an open observation process.
\end{proof}

\begin{corollary}[Same equation may carry different laws]
\label{cor:same-equation-different-laws}
The same mathematical equation may express different physical laws when attached to different source histories, classifiers, stability domains, or failure surfaces.
\end{corollary}

\begin{proof}
By \autoref{def:law-certificate}, the equation-like pattern is only one field. Different source fit, classifier, stability, failure surface, or ledger data determine different law certificates.
\end{proof}

\begin{corollary}[Law without failure surface is oracle claim]
\label{cor:law-without-failure-surface-oracle}
If a claimed law has no specified failure surface, it is not a scientific law certificate but an unbounded oracle assertion.
\end{corollary}

\begin{proof}
A law certificate must specify observations or refinements that would separate model from reality. Removing this surface removes empirical separability and turns the claim into a final-fit oracle, which the present framework rejects.
\end{proof}

\begin{definition}[Predictive reuse and descent]
\label{def:predictive-reuse-descent}
Predictive reuse applies a model fitted on $H_n$ to a later or counterfactual history $H'$ with $H_n\preceq H'$ or $H'\in\mathcal C$. A prediction operation $F$ has predictive descent from the observed classifier $\sim_n$ to the continuation classifier $\sim_{\mathcal C}$ when
$$
M_1\sim_n M_2
\quad\Longrightarrow\quad
F(M_1)\sim_{\mathcal C}F(M_2).
$$
\end{definition}

\begin{theorem}[Prediction requires reuse stability and descent]
\label{thm:prediction-requires-reuse-stability-descent}
A prediction from a finite-fit model is licensed only when the model has reuse stability over the relevant continuation class and the prediction operation descends through the classifier used to form the model.
\end{theorem}

\begin{proof}
Prediction applies the model beyond the finite trace on which it was fitted. Without reuse stability, finite fit gives no reason to expect future fit. Since the model was formed by classification and compression, a prediction operation that does not descend through that classifier may depend on hidden representative data rather than on the visible model. Both reuse stability and descent are therefore required.
\end{proof}

\begin{corollary}[Predictive failure can be non-descent]
\label{cor:predictive-failure-non-descent}
A model may fail predictively because the prediction operation does not descend through the approximation that produced the model, even when the visible equation is internally correct.
\end{corollary}

\begin{proof}
By \autoref{thm:prediction-requires-reuse-stability-descent}, prediction requires descent. If descent fails, the failure lies in transferring the equation through an invalid compression, not necessarily in the equation's algebraic form.
\end{proof}

\section{Open empirical continuation}
\label{sec:open-empirical-continuation}

\begin{definition}[Local empirical closure]
\label{def:local-empirical-closure}
Local empirical closure is the capacity, for a fixed finite trace $H_n$, bundle $\Pi$, and classifier $\sim_{\Pi}$, to determine or certify whether $\operatorname{Fit}_n(M,R;\Pi)$ holds.
\end{definition}

\begin{definition}[Global empirical openness]
\label{def:global-empirical-openness}
Global empirical openness is the absence of any internal finite procedure supplying a final certificate for
$$
\forall H,\;
\operatorname{Fit}(M,R;\Pi\mid H)
$$
over all admissible actual and counterfactual observation histories.
\end{definition}

\begin{definition}[Final induction oracle]
\label{def:final-induction-oracle}
A final induction oracle is a hypothetical judgment $\Omega_{\mathrm{ind}}(M,R;\Pi)$ implying
$$
\forall H,\;
\operatorname{Fit}(M,R;\Pi\mid H).
$$
It would convert finite or current model success into unrestricted future validity.
\end{definition}

\begin{theorem}[Local closure does not imply global closure]
\label{thm:local-closure-not-global-closure}
A finite inquiry may close fit judgments at each observed stage without closing the universal judgment over all future stages.
\end{theorem}

\begin{proof}
Local empirical closure concerns a fixed finite trace and observed data. Global empirical closure quantifies over observations not yet performed and refinements not yet introduced. No finite local fit judgment contains those future traces.
\end{proof}

\begin{corollary}[Science requires local closure and global openness]
\label{cor:science-local-closure-global-openness}
Scientific inquiry requires both the ability to settle finite tests and the refusal to close final fit.
\end{corollary}

\begin{proof}
If local closure fails, experiments cannot decide whether a model fits current observations. If global openness fails, the theory becomes immune to future correction.
\end{proof}

\begin{theorem}[Final induction oracle collapses inquiry]
\label{thm:final-induction-oracle-collapses-inquiry}
If a final induction oracle is admitted, physical inquiry loses its open continuation structure.
\end{theorem}

\begin{proof}
A final induction oracle asserts that a model will fit all admissible future observations. This removes the failure surface and closes the continuation horizon. Future observations can no longer genuinely revise the model, so open inquiry is replaced by a meta-level assertion of total validity.
\end{proof}

\section{Falsification, intuition, and tower induction}
\label{sec:falsification-intuition-tower-induction}

\begin{definition}[Finite refutation and open confirmation]
\label{def:finite-refutation-open-confirmation}
A finite refutation of $M$ relative to $R$ and $\Pi$ is a finite trace $H_n$ such that
$$
\operatorname{ModSig}_{\Pi}(M\mid H_n)
\not\sim_{\Pi}
\operatorname{ObsSig}_{\Pi}(R\mid H_n).
$$
Open confirmation is an extendable family of finite fit certificates together with a physical induction certificate licensing specified continuations.
\end{definition}

\begin{theorem}[Falsification finite, confirmation open]
\label{thm:falsification-finite-confirmation-open-refined}
A model can be refuted by a finite observation, but it cannot be finally confirmed by finitely many observations.
\end{theorem}

\begin{proof}
A finite refutation is one positive witness of mismatch. Final confirmation would require agreement over all admissible traces, and no finite collection of traces exhausts the open observation carrier.
\end{proof}

\begin{definition}[Stability prior and certified intuition]
\label{def:stability-prior-certified-intuition}
A stability prior is a pre-certificate expectation that an observed pattern will persist under a specified class of continuations. It becomes certified intuition when upgraded to a physical induction certificate by specifying the stability basis and failure surface.
\end{definition}

\begin{theorem}[Physical intuition is not oracle]
\label{thm:physical-intuition-not-oracle}
A stability prior does not license prediction unless it is converted into a physical induction certificate.
\end{theorem}

\begin{proof}
A stability prior expresses expectation. Prediction requires a licensed transition from finite fit to future fit. Without a physical induction certificate, the prior does not specify the continuation class, classifier, descent conditions, or failure surface.
\end{proof}

\begin{definition}[Tower induction]
\label{def:tower-induction}
For a tower $\mathcal T:L_0\to L_1\to\cdots\to L_n$, tower induction is a physical induction certificate licensing prediction at level $L_j$ from fit established at level $L_i$, with $i\leq j$, together with descent or reflection data relating the relevant levels.
\end{definition}

\begin{theorem}[Cross-level prediction requires tower induction]
\label{thm:cross-level-prediction-requires-tower-induction}
If a prediction is made at a different tower level from the level at which fit was established, then a tower induction certificate is required.
\end{theorem}

\begin{proof}
Fit at $L_i$ is judged by the classifier and visible signatures of $L_i$. Prediction at $L_j$ may use a different classifier, visible structure, and hidden residue. The finite fit at $L_i$ does not by itself license the prediction at $L_j$. Tower induction supplies the missing transfer data: signature preservation between levels, descent of operations, and inactivity or ledgered handling of hidden structure.
\end{proof}

\begin{corollary}[Effective laws are level-indexed induction certificates]
\label{cor:effective-laws-level-indexed-induction}
An effective law is valid only within the approximation level and continuation class for which its induction certificate holds.
\end{corollary}

\begin{proof}
Effective laws compress lower-level structure into higher-level variables. Moving across levels requires the certification described in \autoref{thm:cross-level-prediction-requires-tower-induction}.
\end{proof}

\begin{theorem}[Physical induction has closed-ground and open-continuation form]
\label{thm:physical-induction-closed-ground-open-continuation}
Physical induction requires local closure of finite fit judgments and global openness of future observation.
\end{theorem}

\begin{proof}
Local finite fit must be decidable or certifiable; otherwise the model cannot be compared with the observed trace. The future observation horizon cannot be closed by the same finite procedure. If it were closed, the model would acquire a final-fit oracle over all future observations. Physical induction therefore has the form of closed finite fit together with open continuation, parallel to BEDC's closed-ground and open-meta polarity.
\end{proof}
