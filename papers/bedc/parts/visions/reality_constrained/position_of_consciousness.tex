\section{The Position of Consciousness}
\label{sec:position-of-consciousness}
\origin{ai}

\begin{remark}[Vision seed status]
\label{rem:position-consciousness-vision-seed-status}
This section has vision-seed status: it is a schema horizon and has no
Lean target. It does not add consciousness as a new BEDC kernel object and
does not introduce a subject-substance carrier. It records a location
principle: on the records side, consciousness appears as the self-proxy
surface of a focused $\Hist$; on the inscription side, the self-center
appears as the local inscription point of $\mathsf{T}$ on that focused
observational chain.

The section forbids the strong equations
$$
\mathsf{Consciousness}=\mathsf{T},
\qquad
\mathsf{Consciousness}=\mathsf{SubjectSubstance},
\qquad
\mathsf{Consciousness}=\mathsf{GlobalTime},
\qquad
\mathsf{SelfCenter}(C,t)=\mathsf{T}.
$$
The safe reading is
$$
\mathsf{Consciousness}
=
\mathsf{RecordsSide}(\mathsf{FocusedHistSelfProxy})
+
\mathsf{InscriptionSide}(\mathsf{LocalTInscription}),
$$
or, more concretely,
$$
\mathsf{Consciousness}
=
(H(F),\Theta(F),S(F),\rho)
+
\mathsf{InscriptionPoint}(\mathsf{T},C_F,t).
$$
The first summand is the object-layer recordable carrier. The second is an
aspectual boundary reading and must not be internalized as a BEDC object.
\end{remark}

\begin{definition}[Focused observational chain]
\label{def:focused-observational-chain}
Given a finite focused multi-$\Hist$ ledger
$$
L_N=(I,H,\Theta,F,S,D,G,R,J,Q,\rho,A),
$$
where $I$ is a finite index ledger, $H:I\to\Hist$ assigns histories,
$\Theta$ records local continuation traces, $F\in I$ is the focus index,
$S$ is the selector ledger, $\rho$ is the focused self-proxy, and $A$ is
the audit row, the \emph{focused observational chain} is
$$
C_F:=(H(F),\Theta(F),S(F),\rho).
$$
Here $H(F)$ is the focused generated $\Hist$, $\Theta(F)$ is the local
$\Cont$ trace from $\emp$ to $H(F)$, $S(F)$ records which visible
transitions are selected into the chain, and $\rho$ is the endpoint and
trace self-proxy.
\end{definition}

\begin{definition}[Records-side conscious carrier]
\label{def:records-side-conscious-carrier}
For a focused ledger $L_N$ and focus $F$, the records-side conscious
carrier is
$$
\mathsf{ConsciousRecord}(L_N,F)
:=
(H(F),\Theta(F),S(F),\rho).
$$
It is the records-side carrier of a consciousness reading, not an
object-level definition of phenomenal essence. Thus the recordable BEDC
location consists of focused generated history, local continuation trace,
selector ledger, and self-proxy.
\end{definition}

\begin{definition}[Inscription-side self-center]
\label{def:inscription-side-self-center}
Let $\mathsf{T}$ denote the non-internalizable apophatic far end. For a
focused observational chain $C_F$ at local stage $t$, its
inscription-side self-center is
$$
\mathsf{SelfCenter}(C_F,t)
:=
\mathsf{InscriptionPoint}(\mathsf{T},C_F,t).
$$
This definition is local and aspectual. It says that a local accumulation
event of the focused chain is read, on the inscription side, as a local
inscription point of $\mathsf{T}$. It does not say
$\mathsf{SelfCenter}(C_F,t)=\mathsf{T}$.
\end{definition}

\begin{definition}[Position of consciousness]
\label{def:position-of-consciousness}
The position of consciousness is the double-aspect location
$$
\mathsf{Pos}_{\mathsf{Con}}(L_N,F,t)
:=
\bigl(
\mathsf{ConsciousRecord}(L_N,F),
\mathsf{InscriptionPoint}(\mathsf{T},C_F,t)
\bigr).
$$
The first component is the records-side carrier
$$
\mathsf{ConsciousRecord}(L_N,F)
=
(H(F),\Theta(F),S(F),\rho).
$$
The second component is the inscription-side reading
$$
\mathsf{InscriptionPoint}(\mathsf{T},C_F,t).
$$
Thus the position of consciousness is not a single entity-location. It is
the double-aspect position consisting of a records-side focused self-proxy
surface and an inscription-side local $\mathsf{T}$-inscription point.
\end{definition}

\begin{definition}[Role discipline]
\label{def:position-consciousness-role-discipline}
This vision-seed schema keeps the following roles separate:
$$
\begin{gathered}
H(F)=\text{focused generated }\Hist,\\
\Theta(F)=\text{local continuation trace},\\
\rho=\text{focused self-proxy/readback},\\
\mathsf{T}=\text{non-internalizable apophatic far end}.
\end{gathered}
$$
The equation
$$
\mathsf{SelfCenter}(C_F,t)
=
\mathsf{InscriptionPoint}(\mathsf{T},C_F,t)
$$
expresses a local inscription reading, not object-level equality with
$\mathsf{T}$. Any presentation that collapses these roles into one
object-level equality is a forbidden collapse.
\end{definition}

\begin{theorem}[Consciousness adds no subject-substance]
\label{thm:consciousness-adds-no-subject-substance}
$\mathsf{ConsciousRecord}(L_N,F)$ does not add a new subject-substance to
BEDC.
\end{theorem}

\begin{proof}
By \autoref{def:records-side-conscious-carrier},
$$
\mathsf{ConsciousRecord}(L_N,F)
=
(H(F),\Theta(F),S(F),\rho).
$$
All four components are already contained in the finite focused ledger
$L_N$: $H(F)$ is an existing $\Hist$, $\Theta(F)$ is an existing
$\Cont$ trace, $S(F)$ is a selector ledger, and $\rho$ is a focused
self-proxy/readback row. The definition adds no subject parameter outside
$\Hist$ and no new identity criterion. Observer identity remains governed
by $\hsame$. An additional subject-substance carrier would require
transport laws across $\Cont$, which is exactly the structure refused by
the no-subject discipline of \autoref{thm:no-subject-parameter}. Thus
$\mathsf{ConsciousRecord}(L_N,F)$ is a records-side focused carrier, not a
new subject-substance.
\end{proof}

\begin{corollary}[Consciousness is not reified]
\label{cor:consciousness-is-not-reified}
BEDC does not reify consciousness as a new ontological object:
$$
\mathsf{Consciousness}\neq\mathsf{SubjectSubstance}.
$$
\end{corollary}

\begin{proof}
Reifying consciousness as a subject-substance would require a carrier
independent of $\Hist$ and transport laws for that carrier through
continuation, selector, self-proxy, and inter-$\Hist$ relation rows. But
\autoref{def:records-side-conscious-carrier} writes the records-side
carrier entirely as the four fields $H(F),\Theta(F),S(F),\rho$ already
inside $L_N$. No additional carrier is introduced.
\end{proof}

\begin{theorem}[Consciousness is local Hist-time, not global time]
\label{thm:consciousness-local-hist-time-not-global-time}
The position of consciousness is a focused readback surface of local
Hist-time, not a global time object:
$$
\mathsf{Consciousness}\neq\mathsf{GlobalTime}.
$$
\end{theorem}

\begin{proof}
BEDC kernel time is given by $\Hist$ growth. A time step is one
$\Eone/\Ezero$ distinction applied to a history, and a local time stream is
a $\Cont$ trace from $\emp$ to a generated history. For the focused ledger
$L_N$, the conscious carrier uses $H(F)$ and its local trace $\Theta(F)$.
This structure does not synchronize all histories, does not define a
global clock, and does not supply a unified time axis across all observers.
The focus $F$ selects one local chain; it does not generate a global
observer time. Hence consciousness is local Hist-time read through a
focused self-proxy, not global time.
\end{proof}

\begin{corollary}[Other-consciousness comparison requires relation rows]
\label{cor:other-consciousness-comparison-requires-relation-rows}
Comparison between distinct focused chains cannot be made through a global
consciousness time. It requires inter-$\Hist$ relation, invariant,
dependence, and audit rows such as $R,J,Q,A$.
\end{corollary}

\begin{proof}
By \autoref{thm:consciousness-local-hist-time-not-global-time},
consciousness is a focused local Hist-time surface. Comparing two focuses
$F_i$ and $F_j$ therefore requires displayed rows relating
$H(F_i),\Theta(F_i)$ to $H(F_j),\Theta(F_j)$: relation rows, classifier-stable
invariant rows, dependence rows, or rate rows. Without these ledger fields,
BEDC does not license a direct comparison inside one global consciousness
clock.
\end{proof}

\begin{theorem}[SelfCenter is local inscription, not T]
\label{thm:selfcenter-local-inscription-not-t}
The equation
$$
\mathsf{SelfCenter}(C_F,t)
=
\mathsf{InscriptionPoint}(\mathsf{T},C_F,t)
$$
does not imply
$$
\mathsf{SelfCenter}(C_F,t)=\mathsf{T}.
$$
\end{theorem}

\begin{proof}
\autoref{def:inscription-side-self-center} contains three roles: the far
end $\mathsf{T}$, the focused chain $C_F$, and the local stage $t$. It is a
local inscription point of $\mathsf{T}$ on a particular chain at a
particular event. It is not $\mathsf{T}$ itself. Writing
$\mathsf{SelfCenter}(C_F,t)=\mathsf{T}$ would internalize the apophatic far
end as an object-level term and would erase the local chain and local
stage. This is precisely the collapse refused by
\autoref{def:position-consciousness-role-discipline}.
\end{proof}

\begin{corollary}[Consciousness exposes T-sockets but is not T]
\label{cor:consciousness-exposes-t-sockets-not-t}
Consciousness may expose $\mathsf{T}$-supply sockets, but consciousness is
not $\mathsf{T}$:
$$
\mathsf{Consciousness}\neq\mathsf{T}.
$$
\end{corollary}

\begin{proof}
A focused self-proxy can carry forward-binding commitments: that the trace
continues, that the trace is read as mine, that the local experience has
meaning beyond the displayed record, or that an unrecorded next moment is
expected. Such commitments exceed the current finite record and therefore
expose $\mathsf{T}$-supply sockets. But exposing a socket is not becoming
the socket's far end. $\mathsf{T}$ is the non-internalizable supply
position; $\mathsf{ConsciousRecord}(L_N,F)$ is a records-side carrier.
They occupy different strata.
\end{proof}

\begin{theorem}[Consciousness has a double-aspect position]
\label{thm:consciousness-double-aspect-position}
The position of consciousness is
$$
\mathsf{Pos}_{\mathsf{Con}}(L_N,F,t)
=
\bigl(
\mathsf{ConsciousRecord}(L_N,F),
\mathsf{InscriptionPoint}(\mathsf{T},C_F,t)
\bigr).
$$
\end{theorem}

\begin{proof}
From the records side, consciousness must be written as finite ledger rows.
\autoref{def:records-side-conscious-carrier} gives
$$
\mathsf{ConsciousRecord}(L_N,F)
=
(H(F),\Theta(F),S(F),\rho),
$$
which records the focused $\Hist$, local time trace, selector ledger, and
self-proxy. From the inscription side, the same local accumulation event is
read as a local actualization at the boundary:
$$
\mathsf{SelfCenter}(C_F,t)
=
\mathsf{InscriptionPoint}(\mathsf{T},C_F,t).
$$
This does not add a second event. It gives an aspectual reading of the same
event: records-side entry into the ledger and inscription-side local
inscription of $\mathsf{T}$. Therefore consciousness is not located in a
single object but in the double-aspect position formed by these two
readings.
\end{proof}

\begin{corollary}[Brain-state and experience are two readings of one event]
\label{cor:brain-state-experience-two-readings}
In this vision-seed schema, brain state or bodily state is not the external
causal maker of consciousness, and consciousness is not an external mover of
the brain. They are two readings of one accumulation event:
$$
\begin{gathered}
\mathsf{BrainState}(C_F,t)=\text{records-side observable row},\\
\mathsf{Experience}(C_F,t)=\text{inscription-side reading of the same event}.
\end{gathered}
$$
\end{corollary}

\begin{proof}
Brain states, bodily states, neural activity, and behavioral outputs enter
BEDC only as observation, history, signature, classifier, or ledger rows.
These are records-side rows. First-person experience is not another
records-side object beside them; it is the inscription-side reading of the
same local event. Therefore BEDC does not need either
$\mathsf{Brain}\to\mathsf{Consciousness}$ or
$\mathsf{Consciousness}\to\mathsf{Brain}$ as a relation between two
substances.
\end{proof}

\begin{theorem}[Consciousness is not identical with observation, distinction, or time]
\label{thm:consciousness-not-identical-observation-distinction-time}
Although the kernel reading identifies observation, distinction, and local
time-step through $\Cont$ and $\Eone/\Ezero$ growth, consciousness is not
identical with bare observation, bare distinction, or bare time-step.
\end{theorem}

\begin{proof}
Observation, distinction, and local time-step give the minimal occurrence
form
$$
h\xrightarrow{\Ezero/\Eone}h'.
$$
Consciousness requires additional structure:
$$
H(F),\Theta(F),S(F),\rho.
$$
There must be focus, local trace, selector ledger, and self-proxy/readback.
A history may grow without being selected as a focused conscious record; a
distinction may occur without a self-proxy; a local time step may be
recorded without forming a conscious reading. Thus the kernel-facing
identity of observation, distinction, and local time-step must not be
promoted to the vision-layer equation
$\mathsf{Consciousness}=\mathsf{Time}$.
\end{proof}

\begin{corollary}[Safe slogan]
\label{cor:position-consciousness-safe-slogan}
The safe compressed readings are
$$
\mathsf{Consciousness}
=
\mathsf{FocusedSelfProxy}(\mathsf{HistTime})
+
\mathsf{LocalTInscription},
$$
and
$$
\mathsf{Consciousness}
=
\mathsf{RecordsSide}(H(F),\Theta(F),S(F),\rho)
+
\mathsf{InscriptionSide}
\bigl(\mathsf{InscriptionPoint}(\mathsf{T},C_F,t)\bigr).
$$
The forbidden slogans are
$$
\mathsf{Consciousness}=\mathsf{T},
\qquad
\mathsf{Consciousness}=\mathsf{Subject},
\qquad
\mathsf{Consciousness}=\mathsf{BrainObject},
\qquad
\mathsf{Consciousness}=\mathsf{GlobalTime}.
$$
\end{corollary}

\begin{proof}
The safe readings preserve records-side carrier fields and
inscription-side boundary reading. The forbidden slogans respectively
violate no-internal-$\mathsf{T}$, no-subject-substance, the
records/inscription two-aspect discipline, and no-global-clock.
\end{proof}

\begin{remark}[Reading]
\label{rem:position-consciousness-reading}
The BEDC answer is neither that consciousness is in the brain, nor that it
is in a soul-substance, nor that it is in $\mathsf{T}$, nor that it is in a
global time. On the records side, consciousness is located at
$$
\mathsf{ConsciousRecord}(L_N,F)
=
(H(F),\Theta(F),S(F),\rho),
$$
the self-proxy surface of a focused $\Hist$. On the inscription side, the
self-center is located at
$$
\mathsf{SelfCenter}(C_F,t)
=
\mathsf{InscriptionPoint}(\mathsf{T},C_F,t),
$$
the local inscription point of $\mathsf{T}$ on that focused chain.

Thus consciousness is the double-aspect position where local Hist-time
reads itself as focused self-proxy and is locally inscribed by
$\mathsf{T}$. This preserves first-person centeredness without adding a
subject-substance; it acknowledges the $\mathsf{T}$ boundary without
identifying consciousness with $\mathsf{T}$; and it allows brain and body
records to matter without reducing experience to a bare records-side
object.
\end{remark}
