\subsection{Registry layer}
\label{subsec:philosophy-registry-layer}

\begin{definition}[Registering for export]
\label{def:philosophy-registering-for-export}
To register a claim for export is to assign it to the appropriate
registry surface: concept registry, theorem registry, gap registry,
tradition registry, science registry, cannot-claim registry, upgrade
registry, subject index, closure-status record, or formalization
target.
\end{definition}

\begin{definition}[Export-control surface]
\label{def:philosophy-export-control-surface}
The registry layer is an export-control surface when every exported
claim either appears in a registry entry or is blocked by a registry
entry.
\end{definition}

\begin{theorem}[Registry layer is not supplementary]
\label{thm:philosophy-registry-layer-not-supplementary}
For the BEDC philosophy package, the registry layer is part of the
claim surface rather than supplementary documentation.
\end{theorem}
\begin{proof}
The package's claims are licensed only by their carriers,
classifiers, ledgers, certificates, gap reports, bridge status, and
cannot-claim boundaries. Those fields are recorded in registries. A
prose claim not represented there has no audit surface.
\end{proof}

\begin{corollary}[No registry, no package claim]
\label{cor:philosophy-no-registry-no-package-claim}
If a philosophical claim is absent from the registry layer and not
blocked by it, it is not a package-level BEDC claim.
\end{corollary}

\subsection{Manuscript form}
\label{subsec:philosophy-manuscript-form}

\begin{definition}[Two-surface manuscript]
\label{def:philosophy-two-surface-manuscript}
A mature BEDC philosophy manuscript has two surfaces: the synthesis
surface and the registry surface. The synthesis surface develops the
concepts, theorem readings, and research programme. The registry
surface records their export status, refusal boundaries, and machine
interface.
\end{definition}

\begin{definition}[Reference module]
\label{def:philosophy-reference-module}
A reference module is a stable registry-facing unit for one family of
claims: concepts, theorems, gaps, traditions, scientific claims,
cannot-claim entries, upgrade entries, subject terms, closure-status
records, or formal targets.
\end{definition}

\begin{theorem}[Manuscript completion requires both surfaces]
\label{thm:philosophy-manuscript-completion-two-surfaces}
The philosophy manuscript is package-complete only when prose synthesis
is paired with registry reference modules.
\end{theorem}
\begin{proof}
The synthesis surface alone can state the philosophical reading, but
cannot by itself show every carrier, forbidden reading, gap type,
refusal entry, upgrade path, and formalization target. The registry
surface supplies those fields.
\end{proof}

\begin{corollary}[Reference modules are not lower status]
\label{cor:philosophy-reference-modules-not-lower-status}
Reference modules should not be read as lower-status material. They
are the audit side of the same package.
\end{corollary}
