\subsection{Interface synthesis}
\label{subsec:philosophy-interface-synthesis}

\begin{theorem}[Registry and machine-interface synthesis]
\label{thm:philosophy-registry-machine-interface-synthesis}
A mature BEDC philosophy package consists of a synthesis surface and a
registry-machine surface. The synthesis surface develops the
philosophical structure. The registry-machine surface registers every
exportable claim, blocks forbidden claims, classifies gaps, records
upgrade paths, constrains formalization targets, and preserves
amendment continuity.
\end{theorem}
\begin{proof}
The registry layer provides export-control entries. The export
interface bounds prose by status and cannot-claim conflicts. Registry
consistency ensures that related entries agree. The machine-facing
interface identifies future structures and no-smuggling predicates.
The amendment ledger records active-reading changes without preserving
parallel readings. Closure-status references bound the whole surface.
\end{proof}

\begin{corollary}[After registry layer]
\label{cor:philosophy-after-registry-layer}
Further work should populate the reference modules, check registry
consistency, link each prose theorem to the theorem registry, link each
strong claim to closure status, link each boundary to the gap and
cannot-claim registries, link each future claim to the upgrade
registry, and prepare the \(\mathsf{BEDC.Philosophy}\) namespace.
\end{corollary}

\begin{corollary}[Manuscript architecture]
\label{cor:philosophy-manuscript-architecture}
The final manuscript can be organized as philosophical synthesis,
registry and audit reference modules, and machine-facing interface
with amendment governance. In a shorter article, the synthesis may
occupy the main body while registries and machine interface form the
reference layer.
\end{corollary}

\subsection{Interface status}
\label{subsec:philosophy-interface-status}

\begin{definition}[Machine-interface status]
\label{def:philosophy-machine-interface-status}
The present interface is paper synthesis export. The reference-module
architecture is defined, but the registries are not fully populated,
the structures are not machine checked, registry consistency is not
proved in Lean, and the interface does not close gaps or turn roadmaps
into theorems.
\end{definition}

\begin{definition}[Interface upgrade path]
\label{def:philosophy-interface-upgrade-path}
The upgrade path is to populate the reference modules, define Lean
structures for registry entries and closure references, encode registry
consistency, prove that no exported claim lacks a registry entry, prove
that no registered claim conflicts with cannot-claim entries, prove
that roadmap claims have upgrade entries and refusal boundaries, and
prove no-smuggling for forbidden readings.
\end{definition}

\begin{definition}[Interface compression]
\label{def:philosophy-interface-compression}
The registry layer is the export-control surface of BEDC philosophy:
prose explains, registries license, cannot-claim entries refuse,
closure status bounds, and the machine interface turns these fields
into future formal targets.
\end{definition}
