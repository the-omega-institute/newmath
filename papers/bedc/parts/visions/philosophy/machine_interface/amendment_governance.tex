\subsection{Amendment ledger}
\label{subsec:philosophy-amendment-ledger}

\begin{definition}[Amendment ledger entry]
\label{def:philosophy-amendment-ledger-entry}
An amendment ledger entry records target, change type, old status, new
status, reason, affected registries, cannot-claim impact, and gap
impact.
\end{definition}

\begin{definition}[Change type]
\label{def:philosophy-amendment-change-type}
Change types include add entry, replace entry, downgrade, upgrade,
reclassify gap, remove cannot-claim, add cannot-claim, revise
forbidden reading, and amend formal target.
\end{definition}

\begin{theorem}[Amendment ledger does not create parallel readings]
\label{thm:philosophy-amendment-ledger-no-parallel-readings}
The amendment ledger records audit continuity, but it does not preserve
multiple active readings.
\end{theorem}
\begin{proof}
The governing manuscript remains the active source of truth. When a
reading is replaced, the active formulation takes the citation target.
The ledger records the status change and affected registries; it does
not keep the displaced formulation active.
\end{proof}

\begin{corollary}[Amendment ledger is not an archive]
\label{cor:philosophy-amendment-ledger-not-archive}
The amendment ledger explains the current active reading; it is not a
repository of prior prose.
\end{corollary}

\subsection{Closure-status reference layer}
\label{subsec:philosophy-closure-status-reference-layer}

\begin{definition}[Closure-status reference]
\label{def:philosophy-closure-status-reference}
A closure-status reference records constructive story, theory closure,
scope closed, formal status, Lean target if any, bridge status, not
claimed surface, and upgrade path for a package, part, concept, or
registered theorem.
\end{definition}

\begin{definition}[Closure-status bound]
\label{def:philosophy-closure-status-bound}
The closure-status reference is the strongest export status allowed
for its target.
\end{definition}

\begin{theorem}[Closure status bounds registry exports]
\label{thm:philosophy-closure-status-bounds-registry}
Registry entries may not export a claim stronger than the relevant
closure-status reference.
\end{theorem}
\begin{proof}
If the closure reference records paper synthesis, a registry entry
cannot report whole-package machine checking. If the not-claimed field
rejects a global truth predicate, a theorem entry cannot reintroduce
it under another name.
\end{proof}

\begin{corollary}[Closure references belong to the registry layer]
\label{cor:philosophy-closure-references-belong-registry-layer}
Closure-status references are not marginal notes. They are binding
entries in the export-control surface.
\end{corollary}
