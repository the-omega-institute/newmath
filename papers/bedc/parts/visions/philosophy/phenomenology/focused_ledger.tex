\subsection{Finite focused multi-Hist ledgers}
\label{subsec:philosophy-focused-multi-hist-ledgers}

\begin{definition}[Finite focused multi-Hist ledger]
\label{def:philosophy-focused-multi-hist-ledger}
For a finite horizon \(N\), a finite focused multi-Hist ledger is a
records-side packet
$$
\begin{gathered}
L_N=(I,H,T,F,S,D,G,R,J,Q,\rho,A).
\end{gathered}
$$
Here \(I\) is a finite index ledger; \(H:I\to\Hist\) assigns histories;
\(T\) gives local continuation chains; \(F\) is the focus index;
\(S\) is the selector ledger; \(D\) is a digest ledger; \(G\) is the
gap and provenance row; \(R\) contains inter-history relation rows;
\(J\) contains inter-history invariant rows; \(Q\) contains dependence
rows or optional maximum-rate readings; \(\rho\) is a focused
self-proxy; and \(A\) is the audit row.
\end{definition}

\begin{definition}[Audit row]
\label{def:philosophy-focused-ledger-audit-row}
The audit row \(A\) records the refusals that keep the packet closed:
no subject parameter, no global clock, no hidden chooser, no
unledgered compression, no object-level \(T\) field, and no hidden
dependence on the three meta-closure axioms.
\end{definition}

\begin{theorem}[Closed records-side carrier]
\label{thm:philosophy-closed-records-side-carrier}
\(L_N\) is a closed records-side carrier when every object has a
generator, every branch has selector rows, every compression has
provenance, every inter-history relation has a finite witness, every
invariant is classifier-stable, and every boundary pressure remains
outside the object layer.
\end{theorem}
\begin{proof}
Each component of \(L_N\) is a finite row or finite family of rows. The
packet closes only when its histories are generated, its branch choices
are ledgered, its compressions remain traceable, its relations are
witnessed, and its invariants respect the relevant classifiers. The
audit row blocks the extra structures that would turn the packet into a
subject substance, global universe object, synchronized clock, hidden
choice, or internal \(T\)-object.
\end{proof}

\begin{corollary}[Phenomenological objects require records-side form]
\label{cor:philosophy-phenomenological-objects-records-side}
Consciousness, universe, space, time, identity, and related
phenomenological objects can enter the BEDC object layer only as rows,
relations, invariants, self-proxies, or gap-tagged boundary readings of
a packet such as \(L_N\).
\end{corollary}
\begin{proof}
The object layer accepts generated records, witnesses, certificates,
and ledgers. A phenomenological term that cannot be written as such a
finite row or as an exposed boundary reading has no object-layer
standing.
\end{proof}

\subsection{Records-side consciousness}
\label{subsec:philosophy-records-side-consciousness}

\begin{definition}[Conscious record]
\label{def:philosophy-conscious-record}
For a focused ledger \(L_N\), the records-side carrier of a
consciousness reading is
$$
\begin{gathered}
\mathsf{ConsciousRecord}(L_N,F)=(H(F),T(F),S(F),\rho).
\end{gathered}
$$
It consists of the focused generated history, its local continuation
trace, its selector witnesses, and its endpoint or trace self-proxy.
\end{definition}

\begin{definition}[Consciousness reading]
\label{def:philosophy-consciousness-reading}
\(\mathsf{ConsciousRecord}(L_N,F)\) is a carrier for consciousness
readings. It is not an object-level definition of phenomenal essence.
\end{definition}

\begin{theorem}[Consciousness carrier adds no subject]
\label{thm:philosophy-consciousness-carrier-adds-no-subject}
\(\mathsf{ConsciousRecord}(L_N,F)\) introduces no new
subject-substance.
\end{theorem}
\begin{proof}
The four components are already present in \(L_N\): focused history,
local continuation trace, selector ledger, and self-proxy. The focus
index selects a reading surface but does not create a hidden
first-person carrier or an identity principle beyond \(\hsame\) and the
trace ledger.
\end{proof}

\begin{corollary}[Consciousness is not reified]
\label{cor:philosophy-consciousness-not-reified}
BEDC does not add consciousness as a new ontological substance.
\end{corollary}
\begin{proof}
A substance reading would require a new carrier and transport laws
relating it to histories and continuations. The records-side carrier
uses only focused history data already licensed by the finite ledger.
\end{proof}

\begin{corollary}[Consciousness becomes a closure question]
\label{cor:philosophy-consciousness-closure-question}
The BEDC form of the consciousness question asks which focused history,
local continuation trace, selector ledger, self-proxy, and gap rows
constitute the records-side carrier of a consciousness reading.
\end{corollary}
\begin{proof}
The object layer can close the carrier and its audit rows. It does not
claim to capture phenomenal essence as an additional object.
\end{proof}
