\subsection{T-supply sockets}
\label{subsec:philosophy-t-supply-sockets}

\begin{definition}[T-supply socket]
\label{def:philosophy-t-supply-socket}
A \(T\)-supply socket is a gap tag whose requested discharge cannot be
generated from displayed records, closed inductive generators, naming
certificate fields, or certified continuation laws. It is well-formed
only when the missing shape, consumption site, and preventing audit
gate are stated.
\end{definition}

\begin{definition}[Supply kind]
\label{def:philosophy-supply-kind}
A supply kind is an equivalence class of \(T\)-supply sockets. Two
sockets have the same kind when accepting either one would authorize
the same substrate-internal shortcut.
\end{definition}

\begin{theorem}[Gap sockets must be classified]
\label{thm:philosophy-gap-sockets-classified}
Not all gap sockets have the same shape; each must be classified by
the kind of \(T\)-supply it requests.
\end{theorem}
\begin{proof}
Unrestricted induction, missing witnesses, extensional collapse,
future-law projection, cross-history recognition, inscription
actuality, and self-truth closure consume different shortcuts at
different sites. Treating them as one gap kind hides the requested
supply.
\end{proof}

\begin{corollary}[Many sockets, one far end]
\label{cor:philosophy-many-sockets-one-far-end}
The taxonomy has many socket shapes but does not multiply the ultimate
transcendental supply position.
\end{corollary}
\begin{proof}
The distinction lies in how the closed substrate experiences pressure
at its boundary. The far end remains the source-position of
forward-binding supply.
\end{proof}

\subsection{Seven socket kinds}
\label{subsec:philosophy-seven-socket-kinds}

\begin{definition}[Type I socket]
\label{def:philosophy-type-i-socket}
A Type I socket requests unrestricted induction or universal closure
beyond the displayed authorized generator. It turns generator closure
for a shown carrier into a claim over an unrestricted intended range.
\end{definition}

\begin{theorem}[Type I gate]
\label{thm:philosophy-type-i-gate}
Type I sockets are governed by the generator-closure ledger.
\end{theorem}
\begin{proof}
An induction claim must cite the exact carrier, constructor list,
recursor, and closure certificate. If the claim ranges beyond that
displayed generator, it requests supply not present in the ledger.
\end{proof}

\begin{definition}[Type II socket]
\label{def:philosophy-type-ii-socket}
A Type II socket requests a positive existential witness without
displaying the constructor-level witness. It converts inhabitedness
pressure into a usable element.
\end{definition}

\begin{theorem}[Type II gate]
\label{thm:philosophy-type-ii-gate}
Type II sockets are governed by witness visibility.
\end{theorem}
\begin{proof}
If a later construction consumes an element, the element or the
function selecting it must be displayed. Otherwise the construction
uses an unwitnessed selector.
\end{proof}

\begin{definition}[Type III socket]
\label{def:philosophy-type-iii-socket}
A Type III socket requests the conversion of a relation into identity:
bidirectional implication into proposition equality, quotient
equivalence into equality, classifier sameness into rewrite equality,
or observational indistinguishability into identity.
\end{definition}

\begin{theorem}[Type III gate]
\label{thm:philosophy-type-iii-gate}
Type III sockets are governed by refusal of extensional collapse.
\end{theorem}
\begin{proof}
A relation may support transport only when the transport is displayed.
Treating the relation as equality hides the transport witness and
therefore crosses the boundary.
\end{proof}

\begin{definition}[Type IV socket]
\label{def:philosophy-type-iv-socket}
A Type IV socket requests causal continuity: projection of observed
regularity into not-yet-observed history.
\end{definition}

\begin{theorem}[Type IV gate]
\label{thm:philosophy-type-iv-gate}
Type IV sockets are governed by separation between records and
prediction.
\end{theorem}
\begin{proof}
Observed records and regularity summaries may be stored. The claim that
the same regularity persists into future unobserved cases is a
forward-binding commitment, not a record.
\end{proof}

\begin{definition}[Type V socket]
\label{def:philosophy-type-v-socket}
A Type V socket requests cross-history coordination: treating a
behavior stream recorded by one history as another inscription-bearing
closed history rather than merely as an external object.
\end{definition}

\begin{theorem}[Type V gate]
\label{thm:philosophy-type-v-gate}
Type V sockets are governed by locality discipline.
\end{theorem}
\begin{proof}
Communication records and observable behavior may be ledgered. The
claim that the behavior is another closed observational history is an
additional locality commitment and must be exposed as such.
\end{proof}

\begin{definition}[Type VI socket]
\label{def:philosophy-type-vi-socket}
A Type VI socket requests the actuality of an inscription event inside
a history: the arrival of \(O_{t+1}\) from \(O_t\) as more than a
replay of already recorded derivation.
\end{definition}

\begin{theorem}[Type VI gate]
\label{thm:philosophy-type-vi-gate}
Type VI sockets are governed by aspect separation between proof replay
and inscription novelty.
\end{theorem}
\begin{proof}
The record-side derivation can be checked. The event by which a record
enters a local history has a different aspect. BEDC may name this
aspect, but it must not turn it into an extra proof rule.
\end{proof}

\begin{definition}[Type VII socket]
\label{def:philosophy-type-vii-socket}
A Type VII socket requests self-reference closure: a total truth
predicate, runtime validator, or self-certifying checker for the
system's own structural description.
\end{definition}

\begin{theorem}[Type VII gate]
\label{thm:philosophy-type-vii-gate}
Type VII sockets are governed by reflection stratification.
\end{theorem}
\begin{proof}
Structural self-description, runtime validation, truth classification,
and external checking must remain distinct. Collapsing them would close
the meta loop.
\end{proof}

\subsection{Non-collapse of socket kinds}
\label{subsec:philosophy-socket-non-collapse}

\begin{definition}[Socket collapse]
\label{def:philosophy-socket-collapse}
A socket kind \(A\) collapses into a socket kind \(B\) when every
instance of \(A\) can be reported as \(B\) without losing audit
information. Mutual collapse makes the kinds indistinguishable.
\end{definition}

\begin{theorem}[The seven socket kinds do not collapse]
\label{thm:philosophy-seven-sockets-do-not-collapse}
The seven \(T\)-supply socket kinds are not mutually reducible.
\end{theorem}
\begin{proof}
Their consumption sites and audit gates differ. Type I asks for
unrestricted universal closure; Type II asks for a missing witness;
Type III asks for relation-to-identity collapse; Type IV asks for
future persistence; Type V asks for cross-history locality; Type VI
asks for inscription actuality; Type VII asks for self-truth or runtime
totalization.
\end{proof}

\begin{corollary}[Induction is not choice]
\label{cor:philosophy-induction-not-choice}
Type I is not Type II.
\end{corollary}
\begin{proof}
Universal closure over a range does not provide a particular missing
witness, and a particular witness does not provide total generator
closure over an unrestricted range.
\end{proof}

\begin{corollary}[Choice is not extensional identity]
\label{cor:philosophy-choice-not-extensional-identity}
Type II is not Type III.
\end{corollary}
\begin{proof}
A missing witness produces term-level content. Extensional collapse
produces an equality or rewrite token. These are different shortcuts.
\end{proof}

\begin{corollary}[Identity collapse is not causal continuity]
\label{cor:philosophy-identity-not-causal-continuity}
Type III is not Type IV.
\end{corollary}
\begin{proof}
An equality token does not supply a future law, and a persistent
regularity does not supply proposition or quotient equality.
\end{proof}

\begin{corollary}[Causality is not other-history recognition]
\label{cor:philosophy-causality-not-other-history}
Type IV is not Type V.
\end{corollary}
\begin{proof}
A behavior stream may be regular while still being treated as an
external object. Recognizing it as another closed history is a
different commitment.
\end{proof}

\begin{corollary}[Other-history recognition is not inscription novelty]
\label{cor:philosophy-other-history-not-inscription-novelty}
Type V is not Type VI.
\end{corollary}
\begin{proof}
Cross-history locality relates histories. Inscription novelty concerns
the actuality of a transition within a history. Neither subsumes the
other.
\end{proof}

\begin{corollary}[Inscription novelty is not self-truth closure]
\label{cor:philosophy-inscription-novelty-not-self-truth}
Type VI is not Type VII.
\end{corollary}
\begin{proof}
The actuality of a local record event is not a total truth predicate or
runtime validator for the whole system.
\end{proof}

\begin{corollary}[Self-truth closure is not induction]
\label{cor:philosophy-self-truth-not-induction}
Type VII is not Type I.
\end{corollary}
\begin{proof}
Even when induction over a displayed generator is authorized,
truth-totalization for the system's own syntax and runtime remains a
separate boundary.
\end{proof}
