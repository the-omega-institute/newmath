\subsection{Structural adjacency}
\label{subsec:philosophy-structural-adjacency}

\begin{definition}[Philosophical genealogy]
\label{def:philosophy-philosophical-genealogy}
A traditional philosophical claim \(P\) has a genealogical relation to
a BEDC structure \(B\) when a structural core of \(P\) can be translated
as a BEDC carrier, relation, classifier, ledger, certificate, gap, or
\(T\)-position. The relation does not say that BEDC belongs to the
tradition containing \(P\), nor that BEDC accepts that tradition's full
metaphysical commitments.
\end{definition}

\begin{definition}[Structural adjacency]
\label{def:philosophy-structural-adjacency}
Two traditional claims are structurally adjacent when they land on the
same BEDC site. For example, bundle theories of the self,
atman-negation readings, and second-order cybernetics all land on the
observer-as-history site when restricted to the shared conclusion that
there is no observer-substance behind the trace.
\end{definition}

\concretizedIn{ch:concrete-instances-structuraladjacency-namecert}{Structural adjacency is realized as a finite genealogy, BEDC-role, adjacency-witness, transport, route, provenance, and NameCert packet.}

\begin{theorem}[BEDC is not one tradition]
\label{thm:philosophy-bedc-not-one-tradition}
BEDC cannot be reduced to any single inherited philosophical
tradition.
\end{theorem}
\begin{proof}
The object-language content of BEDC is generated by the finite kernel,
internal sameness judgments, continuations, signatures, packages, gaps,
and naming certificates. Traditional philosophical vocabulary belongs
to interpretive metalanguage. Since the kernel chain does not take any
traditional concept as primitive, no single tradition supplies the
structure of BEDC.
\end{proof}

\begin{corollary}[Comparisons need two labels]
\label{cor:philosophy-comparisons-two-labels}
Every comparison with a philosophical tradition must name both the
traditional claim and the BEDC site on which its structural core lands.
\end{corollary}
\begin{proof}
Naming only the tradition projects external doctrine into BEDC. Naming
only the BEDC site loses the explanatory genealogy. The two labels keep
tradition as reading and BEDC as structure.
\end{proof}

\subsection{Spencer-Brown and primitive distinction}
\label{subsec:philosophy-spencer-brown-distinction}

\begin{definition}[Distinction genealogy]
\label{def:philosophy-distinction-genealogy}
The Spencer-Brown-style claim that a formal system begins by drawing a
distinction lands in BEDC on the primitive mark distinction:
\(\bzero\), \(\bone\), and constructor no-confusion.
\end{definition}

\begin{theorem}[BEDC internalizes the distinction site]
\label{thm:philosophy-bedc-internalizes-distinction-site}
The thought that form begins with a distinction is internalized in
BEDC as the constructor distinction between the two primitive marks.
\end{theorem}
\begin{proof}
The two marks and their no-confusion are not derived from an earlier
object. Writing the two constructors is already the minimal act of
distinction needed for generation to begin. This is the BEDC site
adjacent to the distinction-first reading.
\end{proof}

\begin{corollary}[Distinction becomes certified generation]
\label{cor:philosophy-distinction-certified-generation}
BEDC does not stop at the act of distinction; it sends distinction into
histories, continuations, classifiers, ledgers, and naming
certificates.
\end{corollary}
\begin{proof}
The primitive distinction is the ground. The remaining BEDC layers are
the certified development of that ground into trace, observation,
object license, and boundary accounting.
\end{proof}

\subsection{Brouwer and time intuition}
\label{subsec:philosophy-brouwer-time}

\begin{definition}[Brouwer time site]
\label{def:philosophy-brouwer-time-site}
The Brouwer-style claim that mathematics arises from the perception of
time lands in BEDC on \(\Ezero/\Eone\) application, history growth, and
local continuation streams.
\end{definition}

\begin{theorem}[The Brouwer structural claim becomes Hist-time]
\label{thm:philosophy-brouwer-structural-claim-hist-time}
The structural content of the Brouwer time reading is captured in BEDC
by elementary history extension and local continuation.
\end{theorem}
\begin{proof}
A history step is an elementary distinction entering a generated trace.
A local continuation stream orders such steps from \(\emp\) to a
displayed history. Constructive mathematics can then be read as the
certified unfolding of this finite temporal trace structure.
\end{proof}

\begin{corollary}[BEDC constructivism is not psychologism]
\label{cor:philosophy-bedc-constructivism-not-psychologism}
BEDC does not reduce mathematics to psychological time experience.
\end{corollary}
\begin{proof}
\(\Hist\) is an inductive carrier and \(\Cont\) is a formal relation.
The Brouwer adjacency concerns the formal structure of time as
distinction-order, not a mental state as mathematical substrate.
\end{proof}
