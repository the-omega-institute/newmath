\subsection{Definitions}
\label{subsec:philosophy-definition-discipline}

\begin{definition}[BEDC definition]
\label{def:philosophy-bedc-definition-discipline}
A BEDC definition states carrier, constructor or generator, classifier,
ledger, scope, forbidden readings, and status. It names generated and
stabilized behavior rather than creating an object from nothing.
\end{definition}

\begin{theorem}[Definitions require carriers]
\label{thm:philosophy-definitions-require-carriers}
A definition without a carrier is explanatory vocabulary, not a BEDC
definition.
\end{theorem}
\begin{proof}
Without a carrier the definition has no generated source, no identity
surface, and no place where certificate obligations can be checked.
\end{proof}

\begin{theorem}[Definitions require forbidden readings]
\label{thm:philosophy-definitions-require-forbidden-readings}
Every BEDC definition must state what it is not.
\end{theorem}
\begin{proof}
Many BEDC names reuse mature language such as universe, consciousness,
law, real, or God. Without forbidden readings, mature external meanings
are projected back into the kernel.
\end{proof}

\begin{corollary}[Definition template]
\label{cor:philosophy-definition-template}
A robust definition states: carrier, source, classifier, ledger, scope,
forbidden readings, and status.
\end{corollary}

\subsection{Theorems and corollaries}
\label{subsec:philosophy-theorem-corollary-discipline}

\begin{definition}[BEDC theorem]
\label{def:philosophy-bedc-theorem-discipline}
A BEDC theorem states explicit carrier, certificate, and gap
assumptions; an exact scoped conclusion; a finite witness-chain proof;
and an audit line recording theory and verification grade.
\end{definition}

\begin{definition}[Theorem scope]
\label{def:philosophy-theorem-scope}
The scope of a theorem is the exact class of carriers, certificates, or
objects for which the theorem is licensed.
\end{definition}

\begin{theorem}[Unscoped theorems are refused]
\label{thm:philosophy-unscoped-theorems-refused}
A theorem without explicit scope cannot be used as a BEDC theorem.
\end{theorem}
\begin{proof}
Without scope, the carrier, classifier, ledger, and closure conditions
are unknown. A claim over all objects, observers, worlds, or
mathematics typically requests unrestricted universal closure and must
be downgraded or socketed.
\end{proof}

\begin{corollary}[Theorem titles must not overstate]
\label{cor:philosophy-theorem-titles-not-overstate}
Theorem titles should name the exact carrier and conclusion rather than
claiming that BEDC proves causality, consciousness, or all mathematics.
\end{corollary}

\begin{definition}[BEDC corollary]
\label{def:philosophy-bedc-corollary-discipline}
A BEDC corollary is a direct consequence of prior theorems that adds no
new carrier, witness, or gap.
\end{definition}

\begin{theorem}[Corollaries do not increase strength]
\label{thm:philosophy-corollaries-do-not-increase-strength}
The closure and verification grade of a corollary cannot exceed the
weakest grade of the results it depends on.
\end{theorem}
\begin{proof}
The corollary contributes no new witness. It reuses prior proof
surfaces, so it cannot raise theory closure or formal verification.
\end{proof}

\begin{corollary}[Philosophical corollaries are conservative]
\label{cor:philosophy-philosophical-corollaries-conservative}
Philosophical corollaries may explain, restrict, or translate; they may
not add new ontological commitments.
\end{corollary}

\subsection{Proofs}
\label{subsec:philosophy-proof-discipline}

\begin{definition}[BEDC proof discipline]
\label{def:philosophy-bedc-proof-discipline}
A BEDC proof is a finite witness chain passing through source witness,
classifier witness, transport or exactness witness, ledger witness,
stability witness, and conclusion.
\end{definition}

\concretizedIn{ch:concrete-instances-trace-verdict-boundary-namecert}{Trace-verdict boundaries are realized as finite proof-trace, refusal, verdict, interface-target, provenance, and local NameCert rows.}

\begin{definition}[Forbidden proof resources]
\label{def:philosophy-forbidden-proof-resources}
Forbidden proof resources include hidden choice, host equality as BEDC
equality, quotient collapse, propositional extensional rewrite, global
truth predicate, global clock, unledgered compression, and unscoped
universal claim.
\end{definition}

\begin{theorem}[Proofs must display transport]
\label{thm:philosophy-proofs-display-transport}
Any proof crossing a classifier, bridge, package, or history-identity
surface must display transport.
\end{theorem}
\begin{proof}
Without displayed transport, the proof treats a relation as equality.
That is an extensional-identification socket rather than an internal
proof.
\end{proof}

\begin{corollary}[Obviousness is not a witness]
\label{cor:philosophy-obviousness-not-witness}
Words such as obvious, natural, standard, or by the usual definition do
not close a BEDC proof unless accompanied by the relevant witness
chain.
\end{corollary}
