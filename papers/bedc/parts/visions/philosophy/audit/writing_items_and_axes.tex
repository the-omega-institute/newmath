\subsection{Writing items}
\label{subsec:philosophy-writing-items}

\begin{definition}[Writing item]
\label{def:philosophy-writing-item}
A BEDC writing item is an eight-field packet:
kind, carrier, classifier, ledger, theory-closure status,
formal-verification status, gap report, and claim text. The kind
records whether the item is a definition, theorem, corollary, proof,
reading, schema, roadmap, or refusal. The claim text is the strongest
sentence the item permits the reader to believe.
\end{definition}

\begin{definition}[Minimal legality]
\label{def:philosophy-writing-item-minimal-legality}
A writing item is minimally legal when it states the type of claim, the
carrier it lives on, the way identity or classification is compared,
whether hidden material is ledgered, its theory and verification
status, whether it has a gap, and what it does not claim.
\end{definition}

\begin{theorem}[Writing items must be auditable]
\label{thm:philosophy-writing-items-auditable}
Every concept entering the BEDC philosophy framework must be written as
an auditable writing item.
\end{theorem}
\begin{proof}
Objects require naming certificates. Positive content requires
witnesses. Boundary commitments require gap exposure. Model claims
require bridge discipline. Tradition comparisons require a BEDC site.
Without carrier, classifier, ledger, status, and gap or refusal
surface, the claim cannot be audited.
\end{proof}

\begin{corollary}[Philosophical concepts are not bare prose]
\label{cor:philosophy-concepts-not-bare-prose}
Claims such as consciousness, universe, freedom, law, or mature
mathematics must be rewritten as carriers, slices, socket types,
certificate surfaces, or refusal entries.
\end{corollary}
\begin{proof}
The bare word does not state scope, carrier, witness, gap, or status.
The rewrite supplies those audit fields.
\end{proof}

\subsection{Theory closure and formal verification}
\label{subsec:philosophy-theory-formal-axes}

\begin{definition}[Theory-closure axis]
\label{def:philosophy-theory-closure-axis}
The theory-closure axis records paper-side closure: seed, obligation,
scoped closure, public closure, bridged closure, and mature package
closure. It concerns source, classifier, exactness, ledger, stability,
scope, public object surface, bridge obligations, and theorem
inventory.
\end{definition}

\begin{definition}[Formal-verification axis]
\label{def:philosophy-formal-verification-axis}
The formal-verification axis records Lean-side and audit-side status:
unformalized, formal target, encoded definition, checked scaffold,
checked theorem, audit clean, axiom clean, and checked bridge. It
concerns declarations, encoded carriers, theorem proofs, marker audit,
transitive axiom purity, and no-host-leak checking.
\end{definition}

\begin{definition}[Theory gate]
\label{def:philosophy-theory-gate}
\(\mathsf{TheoryGate}(N,c)\) means that \(N\) has the naming and
closure certificates needed for theory-closure grade \(c\).
\end{definition}

\begin{definition}[Verified gate]
\label{def:philosophy-verified-gate}
\(\mathsf{VerifiedGate}(N,c,v)\) is the conjunction of
\(\mathsf{TheoryGate}(N,c)\) and formal-verification status \(v\) for
the relevant target surface.
\end{definition}

\begin{theorem}[Theory closure and formal verification are independent]
\label{thm:philosophy-theory-formal-independent}
Theory closure does not imply formal verification, and formal
verification does not imply theory closure.
\end{theorem}
\begin{proof}
Theory closure depends on paper-side source, classifier, exactness,
ledger, stability, scope, bridge, and public-surface witnesses. Formal
verification depends on declarations, type checking, proof checking,
marker resolution, and axiom-dependency audits. A paper proof may be
closed without a checked theorem, and a checked theorem may exist
without the surrounding paper-side object being closed.
\end{proof}

\begin{corollary}[Checked is not closed]
\label{cor:philosophy-checked-not-closed}
A checked Lean theorem reports formal-verification status only; it does
not by itself close the corresponding object theory.
\end{corollary}
\begin{proof}
Formal status lacks the full paper-side scope, ledger, closure, and
bridge obligations unless those are separately supplied.
\end{proof}

\begin{corollary}[Paper closed is not machine checked]
\label{cor:philosophy-paper-closed-not-machine-checked}
Paper closure reports theory-closure status only; it does not imply
machine checking.
\end{corollary}
\begin{proof}
The paper witness may fill the theory fields while no canonical Lean
target has yet been checked.
\end{proof}

\subsection{Closure status blocks}
\label{subsec:philosophy-closure-status-blocks}

\begin{definition}[Closure status block]
\label{def:philosophy-closure-status-block}
A closure status block is the binding chapter record containing theory
closure, exact closed scope, formal status, canonical target when
present, bridge status, not-claimed surface, and upgrade path.
\end{definition}

\begin{definition}[Binding commitment]
\label{def:philosophy-binding-commitment}
The closure status block is the strongest export claim the chapter may
make.
\end{definition}

\begin{theorem}[Closure status bounds prose]
\label{thm:philosophy-closure-status-bounds-prose}
If prose claims more than the closure status block permits, the prose
claim is rejected.
\end{theorem}
\begin{proof}
The block states grade, scope, formal status, bridge status, refusal
surface, and upgrade path. A stronger prose claim would bypass the
binding commitment.
\end{proof}

\begin{corollary}[Prose strength has an upper bound]
\label{cor:philosophy-prose-strength-upper-bound}
The allowed claim of a chapter is bounded above by its closure status
block.
\end{corollary}
