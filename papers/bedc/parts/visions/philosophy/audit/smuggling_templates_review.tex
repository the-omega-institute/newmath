\subsection{Philosophical comparison}
\label{subsec:philosophy-comparison-discipline}

\begin{definition}[Tradition map]
\label{def:philosophy-tradition-map-discipline}
A comparison between BEDC and a philosophical tradition is a packet of
traditional claim, BEDC site, preserved structure, rejected surplus, and
audit status.
\end{definition}

\begin{theorem}[Comparisons are not membership claims]
\label{thm:philosophy-comparisons-not-membership}
BEDC comparisons with traditions must not become claims that BEDC is
Kantian, Husserlian, Buddhist, Whiteheadian, or otherwise a member of
the compared tradition.
\end{theorem}
\begin{proof}
The object language contains BEDC carriers, relations, certificates,
ledgers, and gaps, not the tradition's full metaphysical vocabulary.
The comparison maps a structural question to a BEDC site.
\end{proof}

\begin{corollary}[Rejected surplus is mandatory]
\label{cor:philosophy-rejected-surplus-mandatory}
Every philosophical comparison must state what BEDC does not accept
from the tradition.
\end{corollary}

\subsection{Smuggling}
\label{subsec:philosophy-smuggling-discipline}

\begin{definition}[Philosophical smuggling]
\label{def:philosophy-philosophical-smuggling}
Philosophical smuggling introduces an external philosophical entity as
a BEDC internal object: subject substance, world-in-itself, global
truth, absolute now, mental substance, or God as object.
\end{definition}

\begin{definition}[Mathematical smuggling]
\label{def:philosophy-mathematical-smuggling}
Mathematical smuggling introduces mature mathematical objects such as
natural numbers, real numbers, sets, functions, limits, quotients,
topologies, or categories as BEDC primitives.
\end{definition}

\begin{definition}[Host smuggling]
\label{def:philosophy-host-smuggling}
Host smuggling exposes host equality, host numbers, host lists, host
function spaces, quotient soundness, choice, or propositional
extensionality on the BEDC public surface.
\end{definition}

\begin{theorem}[The three smuggling forms share one shape]
\label{thm:philosophy-three-smuggling-forms-one-shape}
Philosophical, mathematical, and host smuggling all disguise
uncertified external supply as internal content.
\end{theorem}
\begin{proof}
The philosophical case imports an unwitnessed entity, the mathematical
case imports a mature object without certificate, and the host case
imports implementation primitives without bridge. All bypass naming,
closure, formal status, gap, or bridge discipline.
\end{proof}

\begin{corollary}[External concepts have two routes]
\label{cor:philosophy-external-concepts-two-routes}
An external concept can enter BEDC only by certificate and gate, or by
gap, socket report, and cannot-claim entry.
\end{corollary}

\subsection{Research template}
\label{subsec:philosophy-research-template}

\begin{definition}[Ten-field research template]
\label{def:philosophy-ten-field-research-template}
Every proposed object or philosophical concept should state traditional
target, BEDC source, carrier choice, classifier choice, stability
obligations, ledger, core theorem list, exactness predicate, standard
bridge status, and combined theory and formal status.
\end{definition}

\begin{theorem}[The template prevents overclaiming]
\label{thm:philosophy-template-prevents-overclaiming}
The ten-field template blocks the common overclaims.
\end{theorem}
\begin{proof}
The traditional target controls terminology. The source blocks
primitive import. Carrier and classifier block identity smuggling.
Stability blocks unusable behavior. Ledger blocks amnesic compression.
The theorem list blocks empty naming. Exactness blocks weak
correspondence becoming identity. Bridge status blocks host fusion.
Status blocks grade inflation.
\end{proof}

\begin{corollary}[Philosophical objects use the same template]
\label{cor:philosophy-objects-use-template}
Concepts such as freedom or consciousness use the same template if
they are to become object-layer interfaces; otherwise they remain
readings, gaps, sockets, or roadmap items.
\end{corollary}

\subsection{Review discipline}
\label{subsec:philosophy-review-discipline}

\begin{definition}[Review questions]
\label{def:philosophy-review-questions}
Each claim is reviewed by asking for carrier, classifier, ledger,
theory grade, formal status, bridge host if present, gap socket if
present, forbidden readings, cannot-claim conflicts, and axis
confusion.
\end{definition}

\begin{definition}[Automatic downgrade]
\label{def:philosophy-automatic-downgrade}
When a claim cannot answer a required audit question, it must be
downgraded from theorem to proposition, proposition to schema, schema
to roadmap, roadmap to gap, or gap to cannot-claim.
\end{definition}

\begin{theorem}[Unauditable claims cannot be strengthened]
\label{thm:philosophy-unauditable-claims-not-strengthened}
A claim that cannot be audited cannot be strengthened into a theorem.
\end{theorem}
\begin{proof}
A theorem requires a proof witness, and proof witnesses require
carrier, classifier, ledger, scope, and status. If these cannot be
given, theorem status is unavailable.
\end{proof}

\subsection{Synthesis}
\label{subsec:philosophy-audit-methodology-synthesis}

\begin{theorem}[Audit methodology synthesis]
\label{thm:philosophy-audit-methodology-synthesis}
A BEDC claim is admissible when it has a carrier, classifier, ledger,
theory-closure grade, formal-verification grade, gap or refusal surface
when incomplete, and no-leakage discipline.
\end{theorem}
\begin{proof}
Writing items provide the audit surface. The two status axes remain
independent. Closure status blocks bind chapter claims. Definitions,
theorems, corollaries, and proofs have fixed fields. Gaps are typed.
Cannot-claim entries and failure certificates record negative export
surfaces. Bridges prevent host fusion. Tradition maps prevent
philosophical overidentification. Smuggling checks block external
primitive import.
\end{proof}

\begin{corollary}[Audit-first writing principle]
\label{cor:philosophy-audit-first-writing-principle}
The primary question is not whether a concept may be mentioned, but by
which carrier, classifier, ledger, gate, gap, and status it is allowed
to be mentioned.
\end{corollary}

\begin{corollary}[Passage to final synthesis]
\label{cor:philosophy-passage-final-synthesis}
With audit methodology fixed, the final layer can state the complete
auditable-generativity theorem and research programme.
\end{corollary}
