\subsection{BEDC certificate machines, allegory machines, and biological branches}
\label{subsec:philosophy-certificate-machine-allegory-branch}

The DNA/window-six bridge also clarifies a larger methodological
separation. BEDC is best treated as a Turing-realizable certificate
machine; an allegory machine generates candidate symbolic patterns; DNA
enters only as an external biological bridge branch. The slogan is:
allegory proposes figures, BEDC certifies names, and DNA supplies a
natural branch whose records can be encoded, classified, audited, and
compared without becoming a BEDC kernel primitive.

\begin{definition}[Finite certificate kernel]
\label{def:philosophy-bedc-finite-certificate-kernel}
The BEDC finite certificate kernel is the dependency spine
$$
\begin{gathered}
  b_{0},b_{1}
  \to \mathsf{Hist}
  \to \mathsf{msame},\mathsf{hsame}
  \to \mathsf{Ext},\mathsf{Cont}\\
  \to \mathsf{Ask},\mathsf{Sig},\mathsf{sameSig}
  \to \mathsf{Pkg},\mathsf{Gap}
  \to \mathsf{NameCert}.
\end{gathered}
$$
It generates finite marks, histories, continuation structure, question
and signature structure, package/gap classification, and certified name
admission.
\end{definition}

\begin{definition}[BEDC certificate machine]
\label{def:philosophy-bedc-certificate-machine}
A BEDC certificate machine is the finite certificate kernel together
with classifier formation, ledger policy, NameCert admission, derivation
strength records, and cannot-claim boundaries. It accepts a candidate
name $N$ only when the relevant source, pattern, classifier, stability,
and ledger fields have been supplied at a declared strength.
\end{definition}

\begin{definition}[Turing-realizable certificate machine]
\label{def:philosophy-turing-realizable-certificate-machine}
A certificate machine is Turing-realizable when its public derivations
can be encoded as finite strings and its certificate checks can be
executed by a finite, Turing-computable procedure. For the BEDC finite
kernel, the intended external model sends raw marks to finite binary
tokens, histories to finite binary words, $\mathsf{Ext}$ to word
extension, $\mathsf{Cont}$ to concatenation, query and signature rows to
finite strings, package and gap rows to finite classifier fibers, and
NameCert fields to finite certificate records.
\end{definition}

\begin{definition}[Plain Turing machine and certificate machine]
\label{def:philosophy-plain-turing-machine-certificate-machine}
A plain Turing-machine description specifies an input, a transition
operation, and an output. A BEDC certificate-machine description records
source, pattern, classifier, stability, ledger, exactness, derivation
strength, and cannot-claim boundaries. Thus BEDC is Turing-realizable
without being exhausted by the input-output image of one plain machine.
\end{definition}

\begin{definition}[Allegory machine and seed]
\label{def:philosophy-allegory-machine-seed}
An allegory machine is a semantic-pattern generator: it compresses
mythic, metaphorical, numerical, geometric, linguistic, and biological
motifs into candidate patterns. An allegory seed is such a candidate
pattern before BEDC admission. Examples include phased creation motifs,
five-phase systems, seven-day ledgers, twelve-orientation ledgers,
four-base biological alphabets, window-six stop boundaries, local
$Q_{2}$ tile dynamics, and codon-window inverse spectra.
\end{definition}

\begin{definition}[Admission of an allegory seed]
\label{def:philosophy-allegory-seed-admission}
An allegory seed $P$ is BEDC-admissible at strength $s$ only when there
is a derived name $N_{P}$ with an acceptance gate
$$
  \mathsf{AcceptGate}(N_{P},s)
  :=
  \mathsf{NameCert}(N_{P})\wedge\mathsf{DerivCert}(N_{P},s).
$$
The strength $s$ records whether the object remains a seed, a paper
certificate target, a checked certificate target, or a bridge certificate
target.
\end{definition}

\begin{definition}[Biological bridge branch]
\label{def:philosophy-biological-bridge-branch}
A biological bridge branch is an external natural system $B$ admitted as
a derived BEDC interface through source, pattern, classifier, stability,
and ledger fields, together with explicit external biological tables or
models. A biological branch is not a finite-kernel primitive.
\end{definition}

\begin{definition}[DNA window-six biological branch]
\label{def:philosophy-dna-window-six-biological-branch}
The DNA window-six branch has source objects including a four-phase
alphabet, a codon cube $Q_{6}$, a translation classifier
$$
  Q_{6}\to\mathsf{AminoAcidClass}\oplus\mathsf{Stop},
$$
a codon ledger, and a context ledger. Its dynamical context records
sequence, reading frame, current codon window, ribosome or release-factor
state, mutation and repair state, epigenetic or chromatin accessibility,
and cellular context.
\end{definition}

\begin{definition}[Human DNA branch]
\label{def:philosophy-human-dna-branch}
The human DNA branch is a species-specific instance of the DNA
window-six branch for Homo sapiens. Its source and ledger fields include
genome sequence, assembly and coordinate system, gene intervals,
transcripts, protein-coding frames, genetic-code table provenance,
variant alleles, epigenetic marks, cell-type context, and annotation
provenance. It is an external biological instance, not a kernel object.
\end{definition}

\begin{theorem}[BEDC finite kernel is Turing-realizable]
\label{thm:philosophy-bedc-finite-kernel-turing-realizable}
The BEDC finite certificate kernel has a Turing-realizable external
model.
\end{theorem}
\begin{proof}
Interpret $b_{0}$ and $b_{1}$ as finite binary tokens, histories as
finite binary words, sameness as literal token or word equality,
$\mathsf{Ext}$ as word extension, $\mathsf{Cont}$ as word concatenation,
query and signature rows as finite string operations, package and gap
rows as finite classifier fibers and inverse images, and NameCert fields
as finite records. Equality, extension, concatenation, lookup,
classification, and certificate-field checking are all finitary and
Turing-computable.
\end{proof}

\begin{theorem}[BEDC is not merely a plain Turing machine]
\label{thm:philosophy-bedc-not-plain-turing-machine}
BEDC is Turing-realizable, but it is not merely a plain input-output
Turing machine.
\end{theorem}
\begin{proof}
A plain Turing machine produces an output from a transition system. BEDC
requires public certificate fields, ledger policy, derivation strength,
and cannot-claim boundaries for admitted names. These obligations are
not merely output values; they are auditable conditions on why a name is
licensed and what it is not allowed to claim.
\end{proof}

\begin{corollary}[BEDC as certificate machine]
\label{cor:philosophy-bedc-as-certificate-machine}
BEDC should be described as a certificate machine rather than merely as
a computational automaton.
\end{corollary}
\begin{proof}
This follows from
\autoref{thm:philosophy-bedc-not-plain-turing-machine}: BEDC public
output is a licensed name with certificate and ledger obligations, not
only a computed string.
\end{proof}

\begin{theorem}[Allegory does not certify itself]
\label{thm:philosophy-allegory-does-not-certify-itself}
An allegory machine can generate candidate patterns, but it does not by
itself produce BEDC-certified names.
\end{theorem}
\begin{proof}
An allegory machine supplies semantic compression, analogy, and symbolic
direction. These do not by themselves supply SourceSpec, PatternSpec,
ClassifierSpec, StabilityCert, LedgerPolicy, or a derivation-strength
record. Thus an allegory seed has no BEDC certificate force until it is
converted into a certified name.
\end{proof}

\begin{corollary}[Allegory proposes and BEDC certifies]
\label{cor:philosophy-allegory-proposes-bedc-certifies}
The relation between the allegory machine and BEDC is: allegory proposes
candidate symbolic patterns, while BEDC tests, certifies, or rejects
them.
\end{corollary}
\begin{proof}
This is \autoref{thm:philosophy-allegory-does-not-certify-itself}
together with \autoref{def:philosophy-allegory-seed-admission}.
\end{proof}

\begin{theorem}[Allegory seeds enter only through NameCert]
\label{thm:philosophy-allegory-seeds-enter-through-namecert}
If an allegory seed $P$ becomes an admissible BEDC object, it must do so
through a derived name $N_{P}$ equipped with NameCert and a recorded
derivation strength.
\end{theorem}
\begin{proof}
By \autoref{def:philosophy-allegory-seed-admission}, admissibility at
strength $s$ requires $\mathsf{AcceptGate}(N_{P},s)$, which includes both
NameCert and DerivCert. An allegory seed has neither field by itself.
\end{proof}

\begin{theorem}[DNA is not a BEDC kernel primitive]
\label{thm:philosophy-dna-not-kernel-primitive}
DNA cannot be a finite-kernel primitive of BEDC.
\end{theorem}
\begin{proof}
The finite kernel contains marks, histories, sameness, extension,
continuation, asking/signature, package/gap, and NameCert structure.
DNA requires an external four-base alphabet, molecular complementarity,
codon windows, translation classifiers, mutation and repair processes,
cellular execution machinery, and biological context. These are not
finite-kernel constructors. DNA can therefore enter only as a derived
biological bridge interface.
\end{proof}

\begin{corollary}[DNA as biological bridge branch]
\label{cor:philosophy-dna-as-biological-bridge-branch}
If DNA is admitted into BEDC, it is admitted as a biological bridge
branch, with $\mathsf{DNAWindow6Dynamics}^{\uparrow}$ as a principal
candidate interface.
\end{corollary}
\begin{proof}
DNA is not primitive by \autoref{thm:philosophy-dna-not-kernel-primitive}.
External natural systems enter by the bridge discipline of
\autoref{def:philosophy-biological-bridge-branch}.
\end{proof}

\begin{theorem}[Human DNA is species-specific bridge data]
\label{thm:philosophy-human-dna-species-specific-branch}
Human DNA is a species-specific instance of the DNA biological bridge,
not a universal BEDC kernel object.
\end{theorem}
\begin{proof}
The human branch requires Homo sapiens-specific genome assemblies,
transcripts, gene intervals, alleles, regulatory context, epigenetic
states, and annotation provenance. These are contingent external
biological records. Kernel objects are generated by the finite BEDC
kernel without such biological contingencies, so the human branch is a
bridge instance rather than a kernel object.
\end{proof}

\begin{corollary}[Human DNA requires ledger]
\label{cor:philosophy-human-dna-requires-ledger}
The valid BEDC expression is that human DNA is a BiologicalBridge branch
with source, classifier, context, and ledger fields. A genome string
without context and ledger is not a complete BEDC object.
\end{corollary}
\begin{proof}
This is the ledger reading of
\autoref{thm:philosophy-human-dna-species-specific-branch} and
\autoref{def:philosophy-human-dna-branch}.
\end{proof}

\begin{theorem}[DNA window-six results are bridge theorems]
\label{thm:philosophy-dna-window-six-results-bridge-theorems}
DNA/window-six reconstructions are biological-bridge theorems, not
proofs that biology is forced by BEDC alone.
\end{theorem}
\begin{proof}
The reconstructions require external biological data: the four-base
alphabet, codon length three, genetic-code tables, translation and
termination behavior, and species or code context. These are not
generated solely by the finite BEDC kernel. Conversely, the occurrence
of window-six structure in DNA supplies a strong bridge instance, not a
proof of BEDC as a whole.
\end{proof}

\begin{definition}[Research surface]
\label{def:philosophy-certificate-allegory-branch-research-surface}
The research surface has five parts. First, formalize BEDC as a
Turing-realizable certificate machine and prove non-reduction to a plain
input-output machine. Second, define allegory machines, allegory seeds,
and NameCert admission gates for symbolic patterns. Third, define
biological branch obligations: source, pattern, classifier, stability,
ledger, bridge table, and cannot-claim boundary. Fourth, develop
$\mathsf{DNAWindow6Dynamics}^{\uparrow}$ and the human DNA branch as
external biological branches. Fifth, connect the branch to local
window-six results: stop codons as $\mathrm{U}+\mathrm{RR}+X_{2}$,
$\mathrm{UGG}$ as a reserved $11$ corner, dual $X_{2}/\{11\}$ control
tiles, $00$-singleton sense-expansion tiles, codon-window inverse
spectra, and alternative genetic-code reassignments.
\end{definition}

\begin{definition}[Empirical bridge surface]
\label{def:philosophy-certificate-allegory-branch-empirical-surface}
The empirical biological surface compares standard and alternative
genetic-code tables, stop-codon usage, readthrough dynamics, codon
optimality, mRNA stability, ribosome-profiling pause rows, synthetic
reporters for $\mathrm{UAA}$, $\mathrm{UAG}$, $\mathrm{UGA}$, and
$\mathrm{UGG}$, and context-specific human variant annotations. These
records test biological adequacy but do not replace NameCert or bridge
certificate obligations.
\end{definition}

\begin{theorem}[Certificate-allegory-branch cannot-claim boundary]
\label{thm:philosophy-certificate-allegory-branch-cannot-claim}
The certificate/allegory/branch triad does not license any claim that
BEDC is exactly one ordinary Turing machine, that allegories are
automatically true, that an allegorical pattern is already a BEDC
theorem, that DNA is a BEDC kernel primitive, that human DNA is generated
by the finite kernel alone, that DNA/window-six proves BEDC, that BEDC
alone proves the biological genetic code, that the human DNA branch is
already a checked certificate, that a genome string without context and
ledger is a complete BEDC object, or that BiologicalBridge data replace
NameCert obligations.
\end{theorem}
\begin{proof}
The Turing-machine refusal follows from
\autoref{thm:philosophy-bedc-not-plain-turing-machine}. The allegory
refusals follow from
\autoref{thm:philosophy-allegory-does-not-certify-itself} and
\autoref{thm:philosophy-allegory-seeds-enter-through-namecert}. The DNA
and human-branch refusals follow from
\autoref{thm:philosophy-dna-not-kernel-primitive} and
\autoref{thm:philosophy-human-dna-species-specific-branch}. The bridge
overclaim refusal follows from
\autoref{thm:philosophy-dna-window-six-results-bridge-theorems}.
\end{proof}

\begin{corollary}[Allowed triad claim]
\label{cor:philosophy-certificate-allegory-branch-allowed-claim}
The allowed current-state formulation is: BEDC is Turing-realizable as a
finite certificate machine; the allegory machine generates patterns that
may be certified or rejected; DNA is a BiologicalBridge branch rather
than a kernel primitive; and human DNA is a species-specific external
branch requiring source, classifier, context, and ledger fields.
\end{corollary}
