\begin{definition}[Outer reassignment boundary]
\label{def:philosophy-outer-reassignment-boundary}
The outer boundary of the reassigned support is
$$
  \partial\mathcal{R}
  =
  \{z\notin\mathcal{R}:\exists r\in\mathcal{R},\ d_{H}(z,r)=1\}.
$$
For $z\in\partial\mathcal{R}$ define the boundary degree and boundary
pressure by
$$
\begin{aligned}
  d_{\partial}(z)
    &=|\{r\in\mathcal{R}:d_{H}(z,r)=1\}|,\\
  P_{\partial}(z)
    &=\sum_{r\in\mathcal{R},\ d_{H}(z,r)=1} w(r).
\end{aligned}
$$
\end{definition}

\begin{theorem}[Outer boundary size]
\label{thm:philosophy-outer-boundary-size}
The reassigned support has $46$ boundary edges and $33$ distinct outer
boundary codons:
$$
  |\mathcal{R}|=13,\qquad
  |E(G_{\mathcal{R}})|=16,\qquad
  |\partial\mathcal{R}|=33.
$$
\end{theorem}
\begin{proof}
Each vertex of $Q_{6}$ has degree $6$, so the $13$ support vertices
have $78$ incident half-edges. The $16$ internal edges of
\autoref{thm:philosophy-reassignment-support-graph} consume $32$ of
these half-edges. The remaining $46$ half-edges leave the support.
Enumerating their external endpoints gives $33$ distinct codons.
\end{proof}

\begin{corollary}[Inactive envelope corners]
\label{cor:philosophy-inactive-envelope-corners}
The three false-positive codons in the three-module envelope are exactly
$$
  \mathrm{UGG},\qquad \mathrm{AAG},\qquad \mathrm{ACA}.

They are inactive boundary codons:

  \mathrm{WRR}\cup\mathrm{CUN}\cup\mathrm{WYA}

\end{proof}

\begin{definition}[$Q_{2}$ completion frontier]
\label{def:philosophy-q2-completion-frontier}
The first $Q_{2}$-completion frontier is the set of codons
$z\notin\mathcal{R}$ for which some $Q_{2}$ face contains exactly
three vertices of $\mathcal{R}$ and has $z$ as its fourth vertex:
$$
  C_{1}(\mathcal{R})
  =
  \{z\notin\mathcal{R}:\exists Q_{2}\text{ face }F,\ 
    |F\cap\mathcal{R}|=3,\ z\in F\setminus\mathcal{R}\}.
$$
\end{definition}

\begin{theorem}[First completion frontier]
\label{thm:philosophy-first-completion-frontier}
The first $Q_{2}$-completion frontier is
$$
  C_{1}(\mathcal{R})
  =
  \{\mathrm{UGG},\mathrm{AAG},\mathrm{ACA},\mathrm{UUG},
    \mathrm{CAA},\mathrm{CCA},\mathrm{GUA},\mathrm{UUU}\}.
$$
It is obtained from $13$ three-out-of-four $Q_{2}$ faces.
\end{theorem}
\begin{proof}
Enumerate the $240$ axis-aligned $Q_{2}$ faces of $Q_{6}$ and select
those with exactly three vertices in $\mathcal{R}$. There are $13$ such
faces. Their missing vertices, after removing duplicates, are exactly
the eight codons displayed above.
\end{proof}

\begin{theorem}[$Q_{2}$ completion percolation]
\label{thm:philosophy-q2-completion-percolation}
Starting from $S_{0}=\mathcal{R}$ and iterating the rule
$$
  S_{n+1}
  =
  S_{n}\cup
  \{z:\exists Q_{2}\text{ face }F,\ |F\cap S_{n}|\ge 3,\ z\in F\},
$$
the process reaches all $64$ codons in five stages. The stage sizes are
$$
\begin{array}{c|c|l}
  \text{stage} & \text{added} & \text{codons}\\
  \hline
  1 & 8 &
    \mathrm{AAG},\mathrm{ACA},\mathrm{CAA},\mathrm{CCA},
    \mathrm{GUA},\mathrm{UGG},\mathrm{UUG},\mathrm{UUU}\\
  2 & 16 &
    \mathrm{ACG},\mathrm{AUG},\mathrm{AUU},\mathrm{CAG},
    \mathrm{CAU},\mathrm{CCG},\mathrm{CCU},\mathrm{CGA},
    \mathrm{GAA},\mathrm{GCA},\mathrm{GUG},\mathrm{GUU},
    \mathrm{UAU},\mathrm{UCG},\mathrm{UCU},\mathrm{UUC}\\
  3 & 16 &
    \mathrm{AAU},\mathrm{ACU},\mathrm{AUC},\mathrm{CAC},
    \mathrm{CCC},\mathrm{CGG},\mathrm{CGU},\mathrm{GAG},
    \mathrm{GAU},\mathrm{GCG},\mathrm{GCU},\mathrm{GGA},
    \mathrm{GUC},\mathrm{UAC},\mathrm{UCC},\mathrm{UGU}\\
  4 & 9 &
    \mathrm{AAC},\mathrm{ACC},\mathrm{AGU},\mathrm{CGC},
    \mathrm{GAC},\mathrm{GCC},\mathrm{GGG},\mathrm{GGU},
    \mathrm{UGC}\\
  5 & 2 &
    \mathrm{AGC},\mathrm{GGC}.
\end{array}
$$
\end{theorem}
\begin{proof}
At each stage enumerate all $Q_{2}$ faces and adjoin every missing
vertex of each face already containing at least three vertices of the
current set. The displayed stage table is the direct finite iteration;
after the fifth stage the set has size $64$.
\end{proof}

\end{proof}

\begin{theorem}[Boundary-closure structure]
\label{thm:philosophy-boundary-closure-structure}
The reassigned support is a cubical seed with inactive boundary
anchors:
$$
  \mathcal{R}
  \quad+\quad
  \{\mathrm{UGG},\mathrm{AAG},\mathrm{ACA}\}
  \quad=\quad
  \text{the active part of }
  \mathrm{WRR}\cup\mathrm{CUN}\cup\mathrm{WYA}.
$$
The three inactive codons are high-pressure boundary codons, the first
$Q_{2}$-completion frontier contains eight codons, and repeated
$Q_{2}$-completion from $\mathcal{R}$ percolates through $Q_{6}$.
\end{theorem}
\begin{proof}
Combine \autoref{cor:philosophy-inactive-envelope-corners},
\autoref{thm:philosophy-first-completion-frontier}, and
\autoref{thm:philosophy-q2-completion-percolation}.
\end{proof}
