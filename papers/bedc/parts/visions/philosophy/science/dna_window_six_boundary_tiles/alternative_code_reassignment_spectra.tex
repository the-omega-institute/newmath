\begin{definition}[Alternative-code reassignment spectrum]
\label{def:philosophy-alternative-code-reassignment-spectrum}
Let $E_{1}:Q_{6}\to C_{1}$ be the standard genetic-code classifier and
let $E_{t}:Q_{6}\to C_{t}$ be another curated genetic-code table on the
same codon carrier. The reassignment set is
$$
  \Delta_{t}:=\{c\in Q_{6}:E_{t}(c)\neq E_{1}(c)\}.
$$
For a family of tables, the reassignment spectrum is the finite multiset
of all codon rows in all $\Delta_{t}$, grouped by local codon module.
\end{definition}
\concretizedIn{ch:concrete-instances-alternative-code-reassignment-namecert}{Alternative-code reassignment spectra are realized as finite standard-code comparison, module ledger, sparse-support, transport, provenance, and NameCert rows.}

\begin{definition}[Four reassignment modules]
\label{def:philosophy-four-reassignment-modules}
The comparison ledger uses four local modules:
$$
\begin{aligned}
  M_{\mathrm{Stop/Trp}}
    &=\{\mathrm{UAA},\mathrm{UAG},\mathrm{UGA},\mathrm{UGG}\},\\
  M_{\mathrm{Ile/Met}}
    &=\{\mathrm{AUU},\mathrm{AUC},\mathrm{AUA},\mathrm{AUG}\},\\
  M_{\mathrm{Phe/Leu}}
    &=\{\mathrm{UUU},\mathrm{UUC},\mathrm{UUA},\mathrm{UUG},
      \mathrm{CUU},\mathrm{CUC},\mathrm{CUA},\mathrm{CUG}\},\\
  M_{\mathrm{AGA/AGG}}
    &=\{\mathrm{AGA},\mathrm{AGG}\}.
\end{aligned}
$$
The first two are the complete global $3+1$ control tiles of
\autoref{thm:philosophy-complete-global-fiber-tiles-standard-code};
the third is the unique Phe/Leu $6+2$ cube of
\autoref{cor:philosophy-phe-leu-unique-six-two-cube}; the fourth is a
two-codon Arg-side pair that becomes visible in comparative tables.
\end{definition}

\begin{theorem}[Alternative-code reassignment concentration]
\label{thm:philosophy-alternative-code-reassignment-concentration}
Across the curated non-standard genetic-code tables compared with the
standard table, the reassignment spectrum has $65$ codon rows. Grouped
by the modules of \autoref{def:philosophy-four-reassignment-modules},
the counts are
$$
\begin{array}{c|r|r}
  \text{module} & \text{reassignment rows} & \text{tables involved}\\
\hline
  M_{\mathrm{Stop/Trp}} & 33 & 22\\
  M_{\mathrm{AGA/AGG}} & 16 & 8\\
  M_{\mathrm{Phe/Leu}} & 7 & 4\\
  M_{\mathrm{Ile/Met}} & 5 & 5\\
  \text{other codons} & 4 & 4.
\end{array}
$$
\end{theorem}
\begin{proof}
For each non-standard table, compare all $64$ codon assignments with
the standard classifier and record exactly those codons whose label
changes. Classifying the resulting $65$ rows by
\autoref{def:philosophy-four-reassignment-modules} gives the displayed
finite table. The enumeration is bridge-side data over curated
genetic-code tables; it is not a BEDC derivation of those biological
assignments.
\end{proof}

\begin{corollary}[Stop/Trp is the main deformation face]
\label{cor:philosophy-stop-trp-main-deformation-face}
The Stop/Trp tile contains only four codons, but it accounts for
$33$ of the $65$ reassignment rows. Thus it is the dominant local
deformation face in the comparative spectrum.
\end{corollary}
\begin{proof}
This is the first row of
\autoref{thm:philosophy-alternative-code-reassignment-concentration}.
\end{proof}

\begin{theorem}[Control half-cube concentration]
\label{thm:philosophy-control-half-cube-reassignment-concentration}
Let
$$
  Q_{5}^{\mathrm{ctrl}}=\{x\in Q_{6}:b_{1}^{f}(x)=0\}.
$$
Among the $65$ reassignment rows in the comparison family, $59$ lie in
$Q_{5}^{\mathrm{ctrl}}$ and $6$ lie outside it:
$$
\begin{array}{c|r}
  \text{region} & \text{reassignment rows}\\
\hline
  b_{1}^{f}=0 & 59\\
  b_{1}^{f}=1 & 6.
\end{array}
$$
Equivalently, $59/65$ of the reassignment rows lie in the same
half-cube that contains the dual Ile/Met and Stop/Trp control tiles.
\end{theorem}
\begin{proof}
For each reassignment row, decode the codon to its six-bit word and
read the first-base family bit. The finite count gives $59$ rows with
$b_{1}^{f}=0$ and $6$ rows with $b_{1}^{f}=1$. The identification of
this half-cube with the shared dual-control half-cube is
\autoref{thm:philosophy-unique-shared-control-half-cube}.
\end{proof}

\begin{corollary}[The shared half-cube is the comparative working region]
\label{cor:philosophy-shared-half-cube-comparative-working-region}
The half-cube selected by the standard-code dual control tiles is also
the main region of observed comparative reassignment. In this sense
$b_{1}^{f}=0$ is both a standard-code control half-cube and the
dominant comparative deformation region.
\end{corollary}
\begin{proof}
The standard-code half-cube is
\autoref{thm:philosophy-unique-shared-control-half-cube}. Its
comparative concentration is
\autoref{thm:philosophy-control-half-cube-reassignment-concentration}.
\end{proof}

\begin{theorem}[Connected reassignment support]
\label{thm:philosophy-connected-reassignment-support}
Let $R$ be the set of the $13$ codons that are ever reassigned in the
comparison family. The induced Hamming subgraph $Q_{6}[R]$ is connected
and has $16$ internal edges.
\end{theorem}
\begin{proof}
Take the support set from
\autoref{thm:philosophy-sparse-reassignment-codon-support}. Enumerate
all Hamming edges whose two endpoints lie in this set. The finite
enumeration gives $16$ such edges. A breadth-first search from any
vertex reaches all $13$ codons.
\end{proof}

\begin{corollary}[Reassignment support is a connected zone]

The reassigned codons are not scattered as isolated vertices of $Q_{6}$.

\begin{theorem}[Coupled $10$-to-$11$ control-tile reassignment]
\label{thm:philosophy-coupled-ten-to-eleven-control-tile-reassignment}
The only observed Ile/Met tile reassignment is
$$
  \mathrm{AUA}:\mathrm{Ile}\to\mathrm{Met}.
$$
It occurs in exactly the tables
$$
  2,\quad 3,\quad 5,\quad 13,\quad 21.
$$
Each of these tables also has
$$
  \mathrm{UGA}:\mathrm{Stop}\to\mathrm{Trp}.
$$
In local coordinates, both changes have the form
$$
  10\ \text{joins the class containing}\ 11.
$$
\end{theorem}
\begin{proof}
In the Ile/Met tile, $\mathrm{AUA}=\mathrm{AU}+10$ and
$\mathrm{AUG}=\mathrm{AU}+11$. In the Stop/Trp tile,
$\mathrm{UGA}=\mathrm{U}+\mathrm{RR}+10$ and
$\mathrm{UGG}=\mathrm{U}+\mathrm{RR}+11$. The finite comparison of
non-standard tables shows that the AUA-to-Met row occurs exactly in the
displayed five table identifiers, and each of those identifiers also
contains the UGA-to-Trp row. The local coordinate action is therefore
the same in both complete control tiles.
\end{proof}

\begin{corollary}[Dual control tiles are coupled in observed variation]
\label{cor:philosophy-dual-control-tiles-coupled-variation}
The Ile/Met and Stop/Trp complete control tiles are not merely parallel
standard-code structures. In the comparative spectrum, the observed
Ile/Met deformation is always accompanied by the corresponding
Stop/Trp $10$-to-$11$ deformation.
\end{corollary}
\begin{proof}
This is the inclusion of table identifiers established in
\autoref{thm:philosophy-coupled-ten-to-eleven-control-tile-reassignment}.
\end{proof}

\begin{theorem}[Alternative-code edge smoothness remains local]
\label{thm:philosophy-alternative-code-edge-smoothness-remains-local}
Across the standard table together with the curated non-standard
genetic-code tables, the total number of same-class Hamming edges lies
between $47$ and $52$, with mean approximately $50.07$.
The third-base family bit remains the dominant stable direction:
its same-class edge count ranges from $28$ to $32$ out of $32$, with
mean approximately $30.04$. The first-base shape bit has zero same-class
edges in every table.
\end{theorem}
\begin{proof}
For each table, enumerate the same $192$ Hamming edges used in
\autoref{thm:philosophy-standard-code-q-one-edge-spectrum} and count
same-class endpoints, both in total and by bit direction. The displayed
range and means are the finite counts over the comparison family.
\end{proof}

\begin{corollary}[Alternative codes are local deformations]
\label{cor:philosophy-alternative-codes-local-deformations}
The comparative tables preserve the broad Hamming-local smoothness
profile of the standard code while moving a small number of codon
assignments, concentrated in the Stop/Trp tile, the Ile/Met tile, the
Phe/Leu cube, and the AGA/AGG pair.
\end{corollary}
\begin{proof}
The edge-count range is narrow by
\autoref{thm:philosophy-alternative-code-edge-smoothness-remains-local},
and the reassignment rows are concentrated by
\autoref{thm:philosophy-alternative-code-reassignment-concentration}.
\end{proof}

\begin{theorem}[Nucleotide-substitution smoothness across the comparison]
\label{thm:philosophy-nucleotide-substitution-smoothness-comparison}
For ordinary single-nucleotide substitution edges, the mean same-class
edge count across the comparison family is approximately $69.04$ out of
$288$. By codon position, the mean same-class counts are approximately
$$
  3.59,\qquad 0.67,\qquad 64.78
$$
out of $96$ edges per position.
\end{theorem}
\begin{proof}
For each table and each codon position, fix the other two bases and
connect the four choices at the chosen position by the complete graph
on four bases. Counting same-class endpoints over the resulting
$288$ substitution edges gives the displayed means.
\end{proof}

\begin{theorem}[Comparative tile-dynamics cannot-claim boundary]
\label{thm:philosophy-comparative-tile-dynamics-cannot-claim-boundary}
The comparative reassignment spectrum does not license any claim that
alternative genetic codes are forced by BEDC, that the observed
curated tables exhaust all possible biological translation systems, that
codon identity alone determines ribosome dynamics, that reassignment
frequency is a mechanistic cause, or that the AGA/AGG pair has already
been reduced to a complete local control-tile certificate.
\end{theorem}
\begin{proof}
The spectrum is a finite bridge-side comparison of curated external
tables. It records where assignment rows differ from the standard
classifier. It does not supply biochemical mechanism, evolutionary
necessity, exhaustive biological coverage, or a checked BEDC certificate
for the AGA/AGG pair.
\end{proof}

\begin{corollary}[Allowed comparative tile-dynamics claim]
\label{cor:philosophy-comparative-tile-dynamics-allowed-claim}
Under the two-axis codon bridge, comparative genetic-code variation is
strongly concentrated in local modules already visible in the standard
codon-cube spectra. The Stop/Trp tile is the main deformation face;
the shared $b_{1}^{f}=0$ half-cube is the dominant reassignment region;
the Stop/Trp $11$ corner remains a stable continuation corner; and the
observed Ile/Met deformation is coupled to the corresponding Stop/Trp
$10$-to-$11$ deformation.
\end{corollary}
\begin{proof}
Combine
\autoref{thm:philosophy-alternative-code-reassignment-concentration},
\autoref{thm:philosophy-control-half-cube-reassignment-concentration},
\autoref{thm:philosophy-stop-trp-eleven-corner-invariant}, and
\autoref{thm:philosophy-coupled-ten-to-eleven-control-tile-reassignment}.
\end{proof}
