\begin{definition}[Conditional trigger ensemble]
\label{def:philosophy-conditional-trigger-ensemble}
Let $E=Q_{6}\setminus M^{\star}$ and weight external subsets by
$$
  W(B)=t^{|B|}y^{\mathcal{E}(B)}.
$$
The full-trigger partition function is
$$
  Z_{\mathrm{trig}}(t,y)
  =
  Z_{\mathrm{all}}(t,y)-Z_{\mathrm{block}}(t,y).
$$
The conditional trigger ensemble is the probability distribution on
full-trigger subsets defined by
$$
  \Pr(B\mid\mathrm{trig})
  =
  \frac{t^{|B|}y^{\mathcal{E}(B)}}{Z_{\mathrm{trig}}(t,y)}.
$$
\end{definition}

\begin{theorem}[Conditional trigger derivatives]
\label{thm:philosophy-conditional-trigger-derivatives}
In the conditional trigger ensemble,
$$
  \mathbb{E}[|B|\mid\mathrm{trig}]
  =
  t\frac{\partial}{\partial t}\log Z_{\mathrm{trig}}(t,y),
$$
and
$$
  \mathbb{E}[\mathcal{E}(B)\mid\mathrm{trig}]
  =
  y\frac{\partial}{\partial y}\log Z_{\mathrm{trig}}(t,y).
$$
\end{theorem}
\begin{proof}
Differentiating $Z_{\mathrm{trig}}$ with $t\partial_{t}$ multiplies
each monomial $t^{|B|}y^{\mathcal{E}(B)}$ by $|B|$. Dividing by
$Z_{\mathrm{trig}}$ gives the conditional expectation. The same
argument with $y\partial_{y}$ gives the conditional expected energy.
\end{proof}

\begin{theorem}[Low-energy conditional distribution]
\label{thm:philosophy-low-energy-conditional-distribution}
As $y\to0$, the conditional trigger ensemble is supported on
partition-minimal triggers. If
$$
  P(t)=2t+28t^{2}+48t^{3}+16t^{4},
$$
then the limiting cardinality distribution is
$$
  \Pr_{\min}(|B|=m)=\frac{a_{m}t^{m}}{P(t)},
$$
where
$$
  a_{1}=2,\qquad
  a_{2}=28,\qquad
  a_{3}=48,\qquad
  a_{4}=16.
$$
The limiting conditional mean cardinality is
$$
  \mathbb{E}_{\min}[|B|]
  =
  \frac{
  2t+56t^{2}+144t^{3}+64t^{4}
  }{
  2t+28t^{2}+48t^{3}+16t^{4}
  }.
$$
\end{theorem}
\begin{proof}
By \autoref{cor:philosophy-energy-biased-low-energy-expansion}, the
lowest full-trigger energy is $4$, with cardinality counts
$2,28,48,16$. Dividing the four leading terms by their sum gives the
limiting conditional distribution.
\end{proof}

\begin{corollary}[Activity crossover points]
\label{cor:philosophy-activity-crossover-points}
The adjacent leading terms in the low-energy ensemble cross at
$$
  t=\frac{1}{14},\qquad
  t=\frac{7}{12},\qquad
  t=3.
$$
Thus the dominant low-energy trigger type passes from singleton
antipodal triggers, to two-codon partition triggers, to three-codon
partition triggers, and finally to four unit-direction triggers.
\end{corollary}
\begin{proof}
The three equalities are
$$
  2t=28t^{2},
  \qquad
  28t^{2}=48t^{3},
  \qquad
  48t^{3}=16t^{4}.
$$
Solving them gives the displayed activities.
\end{proof}

\begin{theorem}[Conditional trigger ensemble theorem]
\label{thm:philosophy-conditional-trigger-ensemble-theorem}
Beyond $M^{\star}=\mathrm{NNR}\cup\mathrm{CUY}$, the full-trigger event
has an explicit conditional ensemble. Its conditional mean cardinality
and energy are logarithmic derivatives of
$Z_{\mathrm{trig}}(t,y)$, and its low-energy limit is governed by the
partition-minimal trigger polynomial
$$
  2t+28t^{2}+48t^{3}+16t^{4}.
$$
\end{theorem}
\begin{proof}
The derivative formulas are
\autoref{thm:philosophy-conditional-trigger-derivatives}. The
low-energy limiting distribution is
\autoref{thm:philosophy-low-energy-conditional-distribution}, and the
activity crossovers are
\autoref{cor:philosophy-activity-crossover-points}.
\end{proof}
