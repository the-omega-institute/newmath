\begin{definition}[Induced blocker nerve]
\label{def:philosophy-induced-blocker-nerve}
For $W\subseteq E$, let $\mathcal{B}_{W}$ be the induced subcomplex of
$\mathcal{B}$ on $W$. Define a simplicial complex $N_{W}$ on the
vertex set $\Omega$ by declaring $T\subseteq\Omega$ to be a face when
there exists $p\in W$ such that
$$
  \sigma(p)\cap T=\varnothing.
$$
Equivalently, $T$ is a face exactly when some support in $W$ is
contained in $\Omega\setminus T$.
\end{definition}

\begin{theorem}[Hochster--nerve reduction]
\label{thm:philosophy-hochster-nerve-reduction}
For every $W\subseteq E$,
$$
  \mathcal{B}_{W}\simeq N_{W}.
$$
\end{theorem}
\begin{proof}
The induced complex is covered by the four simplices
$$
  W\cap B_{\neg s_{1}},
  \quad
  W\cap B_{\neg f_{1}},
  \quad
  W\cap B_{\neg s_{2}},
  \quad
  W\cap B_{\neg f_{2}}.
$$
For $T\subseteq\Omega$, the corresponding intersection is nonempty
exactly when some $p\in W$ omits every coordinate in $T$, that is,
$\sigma(p)\cap T=\varnothing$. All nonempty intersections are
simplices. The nerve theorem identifies $\mathcal{B}_{W}$ with this
four-vertex nerve.
\end{proof}

\begin{corollary}[Hochster homology range]
\label{cor:philosophy-hochster-homology-range}
All reduced homology groups appearing in Hochster's formula for
$k[\mathcal{B}]$ are computed on four-vertex nerves. Hence they occur
only in degrees
$$
  -1,\ 0,\ 1,\ 2.
$$
Consequently the Betti table of $k[\mathcal{B}]$ lies on the four
diagonals
$$
  j-i=0,\ 1,\ 2,\ 3,
$$
and
$$
  \operatorname{reg}k[\mathcal{B}]=3.
$$
\end{corollary}
\begin{proof}
Hochster's formula is
$$
  \beta_{i,j}(k[\mathcal{B}])
  =
  \sum_{|W|=j}
  \dim_{k}\widetilde{H}_{j-i-1}(\mathcal{B}_{W};k).
$$
By \autoref{thm:philosophy-hochster-nerve-reduction},
$\mathcal{B}_{W}$ has the homology of a complex on at most four
vertices. Such a complex has no reduced homology above degree $2$.
The full complex has nonzero $\widetilde{H}_{2}$ by
\autoref{cor:philosophy-blocker-thickened-sphere-dimension}, so the
regularity bound is sharp.
\end{proof}

\begin{theorem}[Top Betti strand]
\label{thm:philosophy-top-betti-strand}
For $4\leq j\leq30$, the highest Betti strand is
$$
  \beta_{j-3,j}(k[\mathcal{B}])
  =
  [x^{j}](2x+x^{2})^{4}(1+x)^{22}.
$$
\end{theorem}
\begin{proof}
The term $\widetilde{H}_{2}(N_{W})$ is nonzero precisely when
$N_{W}=\partial\Delta^{3}$. Since the full four-coordinate face is
never present in $N_{W}$, this is equivalent to the presence of every
proper face. It is enough and necessary to have every three-coordinate
face; the face $\Omega\setminus\{i\}$ requires a codon in $W$ with
support contained in $\{i\}$, hence one of the two singleton-support
lifts for coordinate $i$.

Thus $W$ must choose at least one of the two singleton lifts for each
of the four coordinates. Each coordinate contributes
$(1+x)^{2}-1=2x+x^{2}$. The remaining $30-8=22$ external codons are
unrestricted and contribute $(1+x)^{22}$. In Hochster's formula,
$\widetilde{H}_{2}$ appears when $j-i-1=2$, namely $i=j-3$.
\end{proof}

\begin{definition}[Hochster diagonal series]
\label{def:philosophy-hochster-diagonal-series}
For $q=-1,0,1,2$, define
$$
  B_{q}(x)
  =
  \sum_{j}\beta_{j-q-1,j}(k[\mathcal{B}])x^{j}.
$$
\end{definition}

\begin{theorem}[Complete Hochster diagonal series]
\label{thm:philosophy-complete-hochster-diagonal-series}
The four Hochster diagonal generating functions are
$$
  B_{-1}(x)=(1+x)^{2},
$$
$$
  B_{0}(x)=x^{2}(x+1)^{2}(x+2)^{2}P_{0}(x),
$$
where
$$
\begin{aligned}
  P_{0}(x)=&
  4x^{12}+48x^{11}+268x^{10}+920x^{9}+2167x^{8}\\
  &+3704x^{7}+4736x^{6}+4592x^{5}+3373x^{4}\\
  &+1844x^{3}+720x^{2}+184x+25,
\end{aligned}
$$
$$
  B_{1}(x)=x^{3}(x+1)^{10}(x+2)^{3}P_{1}(x),
$$
where
$$
\begin{aligned}
  P_{1}(x)=&
  6x^{10}+60x^{9}+278x^{8}+784x^{7}+1491x^{6}\\
  &+2002x^{5}+1928x^{4}+1320x^{3}+617x^{2}\\
  &+178x+22,
\end{aligned}
$$
and
$$
  B_{2}(x)=x^{4}(x+1)^{22}(x+2)^{4}.
$$
\end{theorem}
\begin{proof}
Enumerate support-type subsets
$$
  A\subseteq 2^{\Omega}\setminus\{\varnothing\}.
$$
Each chosen support type has two codon lifts, so its nonempty lift
selection contributes $(1+x)^{2}-1=2x+x^{2}$. For each $A$, form the
four-vertex nerve $N_{A}$ as in
\autoref{def:philosophy-induced-blocker-nerve}. Hochster's formula
therefore gives
$$
  B_{q}(x)
  =
  \sum_{A}
  \dim_{k}\widetilde{H}_{q}(N_{A};k)
  (2x+x^{2})^{|A|}.
$$
There are only $2^{15}$ support-type subsets. Computing the reduced
homology of their four-vertex nerves gives the displayed factorizations.
\end{proof}

\begin{definition}[Support-level Betti polynomials]
\label{def:philosophy-support-level-betti-polynomials}
Let
$$
  \mathcal{S}=2^{\Omega}\setminus\{\varnothing\}.
$$
For $A\subseteq\mathcal{S}$, define the support nerve
$$
  N_{A}
  =
  \bigcup_{S\in A}2^{\Omega\setminus S}.
$$
For $q=-1,0,1,2$, set
$$
  Q_{q}(z)
  =
  \sum_{A\subseteq\mathcal{S}}
  \dim_{k}\widetilde{H}_{q}(N_{A};k)z^{|A|}.
$$
\end{definition}

\begin{theorem}[Support-level Betti polynomials]
\label{thm:philosophy-support-level-betti-polynomials}
The support-level Hochster polynomials are
$$
  Q_{-1}(z)=1+z,
$$
$$
  Q_{0}(z)
  =
  z^{2}(1+z)
  \left(
  4z^{6}+28z^{5}+87z^{4}+152z^{3}
  +157z^{2}+92z+25
  \right),
$$
$$
  Q_{1}(z)
  =
  z^{3}(1+z)^{5}
  \left(
  6z^{5}+38z^{4}+99z^{3}+132z^{2}+89z+22
  \right),
$$
and
$$
  Q_{2}(z)=z^{4}(1+z)^{11}.
$$
\end{theorem}
\begin{proof}
Enumerate the $2^{15}$ subsets of the support-type set
$\mathcal{S}$. For each $A$, the nerve is the four-vertex complex
$N_{A}$. Computing its reduced homology and summing by $|A|$ gives the
four displayed polynomials. The highest polynomial has the direct
form $Q_{2}(z)=z^{4}(1+z)^{11}$ because $\widetilde{H}_{2}$ occurs
exactly when the four singleton supports are present, while the other
$11$ support types are unrestricted.
\end{proof}

\begin{theorem}[Support-lift Betti transform]
\label{thm:philosophy-support-lift-betti-transform}
If each support type has $\ell$ lifts, then the lifted Hochster
diagonal series is
$$
  B_{q}^{(\ell)}(x)
  =
  Q_{q}\left((1+x)^{\ell}-1\right).
$$
In the codon case $\ell=2$, so
$$
  B_{q}(x)=Q_{q}(2x+x^{2}).
$$
\end{theorem}
\begin{proof}
For a fixed support type, either no lift is chosen, or a nonempty
subset of its $\ell$ lifts is chosen. The nonempty choice generating
function is $(1+x)^{\ell}-1$. If the support-type set is $A$, its lift
contribution is therefore $\left((1+x)^{\ell}-1\right)^{|A|}$, while
the homology factor is $\dim\widetilde{H}_{q}(N_{A})$. Summing over
$A$ gives the formula.
\end{proof}

\begin{corollary}[Codon support-lift compression]
\label{cor:philosophy-codon-support-lift-compression}
The complete $30$-variable Betti table of $k[\mathcal{B}]$ is
determined by the four support-level polynomials
$$
  Q_{-1},\ Q_{0},\ Q_{1},\ Q_{2}
$$
and the substitution
$$
  z\mapsto (1+x)^{2}-1=2x+x^{2}.
$$
Equivalently,
$$
  30\text{-variable Betti data}
  \longrightarrow
  15\text{-support-type nerves}
  \longrightarrow
  4\text{ support polynomials}.
$$
\end{corollary}
\begin{proof}
The four codon-level diagonal series are
\autoref{thm:philosophy-complete-hochster-diagonal-series}. By
\autoref{thm:philosophy-support-lift-betti-transform}, each is obtained
from the corresponding support-level polynomial by the codon
substitution $z=2x+x^{2}$.
\end{proof}

\begin{theorem}[Universal top Betti strand]
\label{thm:philosophy-universal-top-betti-strand}
For the coverage blocker complex $\mathcal{B}_{n,\ell}$, the highest
Betti strand is
$$
  \beta_{j-(n-1),j}(k[\mathcal{B}_{n,\ell}])
  =
  [x^{j}]
  \left((1+x)^{\ell}-1\right)^{n}
  (1+x)^{\ell(2^{n}-1-n)}.
$$
\end{theorem}
\begin{proof}
The induced nerve is the boundary $\partial\Delta^{n-1}$ exactly when
each singleton coordinate support has at least one selected lift. Each
of the $n$ singleton supports has $\ell$ lifts, contributing
$(1+x)^{\ell}-1$. All other elements number
$\ell(2^{n}-1-n)$ and may be chosen freely. The boundary sphere has
homology degree $n-2$, so Hochster gives $j-i-1=n-2$, or
$i=j-(n-1)$.
\end{proof}

\begin{theorem}[Hochster--nerve Betti theorem]
\label{thm:philosophy-hochster-nerve-betti-theorem}
Every induced subcomplex $\mathcal{B}_{W}$ of the post-$M^{\star}$
blocker complex has the homology of a four-coordinate nerve $N_{W}$.
Therefore the Betti table of $k[\mathcal{B}]$ is governed by the nerve
homology degrees $\widetilde{H}_{-1},\widetilde{H}_{0},
\widetilde{H}_{1},\widetilde{H}_{2}$, has regularity $3$, has
projective dimension $27$, and has depth $3$. The four Hochster
diagonal generating functions are those of
\autoref{thm:philosophy-complete-hochster-diagonal-series}, and they
are obtained from the support-level polynomials of
\autoref{thm:philosophy-support-level-betti-polynomials} by the
substitution $z=2x+x^{2}$. In particular, its top strand is
$$
  \beta_{j-3,j}
  =
  [x^{j}](2x+x^{2})^{4}(1+x)^{22}.
$$
\end{theorem}
\begin{proof}
The nerve reduction is
\autoref{thm:philosophy-hochster-nerve-reduction}. The homology range
and regularity are
\autoref{cor:philosophy-hochster-homology-range}. The top strand is
\autoref{thm:philosophy-top-betti-strand}, and the four diagonal
series are \autoref{thm:philosophy-complete-hochster-diagonal-series}.
The support-lift compression is
\autoref{thm:philosophy-support-lift-betti-transform}.
The projective dimension and depth are
\autoref{thm:philosophy-blocker-depth-regularity}.
\end{proof}
