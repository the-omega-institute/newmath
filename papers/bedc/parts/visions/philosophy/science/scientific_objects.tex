\subsection{Scientific objects}
\label{subsec:philosophy-scientific-objects}

\begin{definition}[Scientific object]
\label{def:philosophy-scientific-object}
A scientific object \(S\) is not a primitive. It is a certificate
packet consisting of a naming certificate, a derivation certificate at
a declared strength, an empirical ledger, an optional model bridge, and
a gap report for unclosed commitments.
\end{definition}

\begin{definition}[Five sources of a scientific object]
\label{def:philosophy-scientific-object-five-sources}
A scientific object must specify records, classifier, invariant,
ledger, and bridge data. The records state which histories,
continuations, or measurement rows generate the object. The classifier
states which records count as the same object or state. The invariant
states what remains stable across perspectives. The ledger states what
has been compressed, averaged, fitted, projected, or sealed. The bridge
states how the object corresponds to a mature scientific or
mathematical model.
\end{definition}

\begin{theorem}[Scientific objects are not primitives]
\label{thm:philosophy-scientific-objects-not-primitives}
Scientific objects cannot enter BEDC as bare primitives.
\end{theorem}
\begin{proof}
Every mature object requires a naming certificate. Every positive
claim requires a witness. Every commitment beyond closed records must
remain exposed as a gap. A scientific object combines object naming,
observational records, classification, stability, prediction, and
bridge obligations, so it is subject to all of these gates rather than
exempt from them.
\end{proof}

\begin{corollary}[Scientific realism becomes audited entry]
\label{cor:philosophy-scientific-realism-audited-entry}
BEDC does not assert bare existence of electrons, fields, spacetime,
or laws. It asks whether the corresponding upward interface has naming
data, empirical rows, bridge fields, and exposed gap reports.
\end{corollary}
\begin{proof}
To enter as an object, the name must be certified. To enter as a
scientific object, it must also expose records, invariants,
measurement surfaces, prediction commitments, and bridge status.
\end{proof}

\subsection{Physical records and relational content}
\label{subsec:philosophy-physical-records-relational-content}

\begin{definition}[Physical record]
\label{def:philosophy-physical-record}
A physical record is a finite, ledgered, source-visible row: a history
row, continuation row, measurement row, instrument row, calibration
row, error row, relation row, or falsification row.
\end{definition}

\begin{definition}[Frame-independent physical content]
\label{def:philosophy-frame-independent-physical-content}
Physical content is frame-independent when it is carried by an
observer-symmetric and history-sameness-stable inter-history invariant.
\end{definition}

\begin{theorem}[Physical content is inter-history invariant]
\label{thm:philosophy-physical-content-inter-history-invariant}
In a multi-history BEDC frame with no global synchronization,
frame-independent physical content can only be carried by inter-history
invariants.
\end{theorem}
\begin{proof}
Individual histories carry local records and local continuation
streams. The kernel supplies no shared base history, global time-step
index, or privileged frame. A claim that survives a change of
perspective must therefore be expressed through witnessed relations and
classifier-stable invariant rows between histories.
\end{proof}

\begin{corollary}[Physical objects are not single-history objects]
\label{cor:philosophy-physical-objects-not-single-history}
A physical object with frame-independent meaning cannot be identified
with a single history alone.
\end{corollary}
\begin{proof}
A single history gives only local record content. Frame-independent
content requires stability across histories, supplied by inter-history
relations and invariants.
\end{proof}

\begin{corollary}[Relational physical realism]
\label{cor:philosophy-relational-physical-realism}
BEDC physical realism is relational: physical reality is stable
inter-history relational content rather than objects in a global
spacetime container.
\end{corollary}
\begin{proof}
A global container would require global synchronization. BEDC refuses
that primitive and lets physical content stabilize through witnessed
relations.
\end{proof}

\subsection{Mature mathematics and mature science}
\label{subsec:philosophy-mature-math-science}

\begin{definition}[Mature mathematical object]
\label{def:philosophy-mature-mathematical-object}
A mature mathematical object has a naming certificate, a derivation
certificate at a declared strength, and optionally a conservative
bridge to a host theory.
\end{definition}

\begin{definition}[Mature scientific object]
\label{def:philosophy-mature-scientific-object}
A mature scientific object adds an empirical ledger, inter-history
invariant rows, prediction gaps, and falsification surfaces to the
mathematical certificate surface.
\end{definition}

\begin{theorem}[Science adds an empirical layer]
\label{thm:philosophy-science-adds-empirical-layer}
Mature science requires the mathematical certificate surface plus
empirical record and prediction discipline.
\end{theorem}
\begin{proof}
Mathematical maturity concerns carrier, classifier, stability, ledger,
strength, and optional bridge. Scientific maturity also concerns
measurements, instruments, errors, fits, future projections, and
records of failure. Prediction adds a Type IV gap, and falsification
adds records-side failure rows.
\end{proof}

\begin{corollary}[Theoretical physics has two certificate faces]
\label{cor:philosophy-theoretical-physics-two-certificate-faces}
A theoretical physics object needs a mathematical certificate for
well-formedness and a physical certificate tying it to records,
invariants, measurements, and falsification surfaces.
\end{corollary}
\begin{proof}
A carrier may be mathematically coherent without empirical
interpretation. An empirical fit may exist without a mature BEDC
carrier. A theoretical physical object needs both surfaces.
\end{proof}
