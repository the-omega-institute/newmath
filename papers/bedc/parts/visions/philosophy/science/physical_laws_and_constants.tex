\subsection{Physical laws as records and gaps}
\label{subsec:philosophy-physical-laws-records-gaps}

\begin{definition}[Empirical regularity]
\label{def:philosophy-empirical-regularity}
An empirical regularity is a finite records-side pattern: observed rows
satisfy a relation under a displayed measurement, classifier, and error
ledger.
\end{definition}

\begin{definition}[Physical law]
\label{def:philosophy-physical-law}
A physical law, in BEDC, is a two-layer object consisting of an
empirical regularity over recorded cases and a Type IV gap asserting
future persistence into not-yet-observed continuation.
\end{definition}

\begin{theorem}[Physical laws are not substrate primitives]
\label{thm:philosophy-physical-laws-not-primitives}
Physical laws cannot be substrate primitives; they combine records-side
regularity with a forward-binding gap.
\end{theorem}
\begin{proof}
Finite records can show regularity among observed rows. They cannot
generate a universal future claim. The law-like projection from records
to not-yet-observed continuation is therefore a Type IV commitment, not
a closed substrate theorem.
\end{proof}

\begin{corollary}[Scientific theories are not closed truth sets]
\label{cor:philosophy-scientific-theories-not-truth-sets}
A scientific theory is a packet of records, regularity summaries,
inter-history invariants, model bridges, prediction gaps, and
falsification ledgers, not a closed set of true laws.
\end{corollary}
\begin{proof}
The packet exposes what is recorded, what is invariant, what is bridged
to a mature model, what is projected into the future, and what would
count as failure. A closed truth set would require the total truth
predicate refused by the meta-loop discipline.
\end{proof}

\subsection{Maximum cross-history rate}
\label{subsec:philosophy-maximum-cross-history-rate}

\begin{definition}[Cross-history causal dependence]
\label{def:philosophy-cross-history-causal-dependence}
A cross-history causal dependence is a proposed relation saying that
recent continuation steps of one history depend on recent continuation
steps of another. At the current philosophical layer, it is a schema
horizon until constructive predicates and concrete witnesses are
provided.
\end{definition}

\begin{definition}[Maximum cross-history causal rate]
\label{def:philosophy-maximum-cross-history-causal-rate}
A maximum cross-history causal rate is a unary-valued invariant form
bounding the ratio at which displayed causal dependence can propagate
between histories, once the underlying dependence relation is
certified.
\end{definition}

\begin{theorem}[Maximum rate is invariant only after witnesses]
\label{thm:philosophy-maximum-rate-invariant-after-witnesses}
If cross-history causal dependence and the maximum-rate construction
are closed by witnesses, the resulting rate must be an inter-history
invariant.
\end{theorem}
\begin{proof}
A rate depending on a privileged frame would not be frame-independent
physical content. Since BEDC supplies no global synchronization, the
rate can become physical content only as an observer-symmetric and
history-sameness-stable invariant over inter-history rows.
\end{proof}

\begin{corollary}[Light-speed form is structural skeleton]
\label{cor:philosophy-light-speed-form-structural-skeleton}
At this layer, BEDC can describe the structural form of an
observer-independent maximum-rate constant, not the empirical numerical
value of the speed of light.
\end{corollary}
\begin{proof}
The structural form follows from inter-history invariance once the
dependence relation is certified. The numerical value requires
measurement records, unit conventions, calibration, and empirical
bridges.
\end{proof}

\begin{corollary}[Physics roadmap is not proved physics]
\label{cor:philosophy-physics-roadmap-not-proved-physics}
The multi-history physics route is a schema horizon until each carrier,
dependence relation, rate invariant, geometric structure, and empirical
bridge receives its own certificate.
\end{corollary}
\begin{proof}
Skipping any layer would leave source, classifier, stability, ledger,
or bridge obligations unsupplied.
\end{proof}

\subsection{Formal constants and empirical values}
\label{subsec:philosophy-formal-constants-empirical-values}

\begin{definition}[Formal constant]
\label{def:philosophy-formal-constant}
A formal constant is an invariant form or bound supplied by a
structural theorem, such as the form of an observer-independent
maximum-rate bound.
\end{definition}

\begin{definition}[Empirical value]
\label{def:philosophy-empirical-value}
An empirical value is supported by measurement records, calibration,
unit bridge, uncertainty, instrument ledger, and reproducibility
ledger.
\end{definition}

\begin{theorem}[Formal constants are not empirical values]
\label{thm:philosophy-formal-constants-not-empirical-values}
BEDC may derive structural forms for constants, but the numerical
empirical values of physical constants require empirical ledgers.
\end{theorem}
\begin{proof}
An invariant form says what kind of value could be observer-independent.
It does not supply the measurement protocol, unit convention,
calibration, error estimate, or reproducibility record needed for a
physical numerical value.
\end{proof}

\begin{corollary}[Units and dimensions are ledgered]
\label{cor:philosophy-units-dimensions-ledgered}
Numerical physical constants require ledgers for units, dimensions,
measurement protocol, uncertainty, calibration, and bridge to standard
units.
\end{corollary}
\begin{proof}
Numerical values are not pure kernel outputs. They depend on empirical
practice and unit systems, which must be auditable.
\end{proof}
