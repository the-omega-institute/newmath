\subsection{DNA window-six boundary tiles and reserved corners}
\label{subsec:philosophy-dna-window-six-boundary-tiles}

The stop-codon reconstruction of
\autoref{subsec:philosophy-stop-codon-zeckendorf-suffix-bridge} uses a
local two-bit no-$11$ tile inside the $\mathrm{U}+\mathrm{RR}$ face. The
same local tile shape also appears in the $\mathrm{AU}$ face of the
standard genetic-code bridge. This subsection records the shared finite
pattern:
$$
  Q_{2}=X_{2}\sqcup\{11\},
  \qquad
  X_{2}=\{00,01,10\}.
$$
At the biological bridge layer, the local $11$ corner is not interpreted
as deletion or biological absence. It is a separated singleton or control
corner.

\begin{definition}[Boundary-tile bridge status]
\label{def:philosophy-boundary-tile-bridge-status}
The boundary-tile bridge is a
$\mathsf{VisionSeed}/\mathsf{BiologicalBridge}$ target. Its finite
enumerations may become checked finite targets after the codon alphabet,
the local faces, and the $X_{2}$ partition are formalized. The biological
labels $\mathrm{Ile}$, $\mathrm{Met}$, $\mathrm{Start}$, $\mathrm{Stop}$,
and $\mathrm{Trp}$ remain external standard-code bridge data.
\end{definition}

\begin{definition}[Local two-bit tile]
\label{def:philosophy-local-two-bit-tile}
The two-bit square and its no-$11$ part are
$$
\begin{aligned}
  Q_{2}&:=\{00,01,10,11\},\\
  X_{2}&:=\{00,01,10\}.
\end{aligned}
$$
Thus $Q_{2}$ decomposes as the disjoint displayed union
$$
  Q_{2}=X_{2}\sqcup\{11\}.
$$
\end{definition}

\begin{definition}[Local codon $Q_{2}$ face]
\label{def:philosophy-local-codon-q-two-face}
A local codon $Q_{2}$ face is a four-codon face of the codon atlas in
which the varying part is displayed by a two-bit coordinate. The two
distinguished faces in this bridge are
$$
\begin{aligned}
  \mathsf{AUFace}
    &:=\{\mathrm{AUU},\mathrm{AUC},\mathrm{AUA},\mathrm{AUG}\},\\
  \mathsf{StopTrpFace}
    &:=\{\mathrm{UAA},\mathrm{UAG},\mathrm{UGA},\mathrm{UGG}\}.
\end{aligned}
$$
The second face is the $\mathrm{U}+\mathrm{RR}$ square of
\autoref{def:philosophy-purine-purine-stop-square}.
\end{definition}

\begin{definition}[Reserved local corner]
\label{def:philosophy-reserved-local-corner}
For a local codon $Q_{2}$ face $F$, a reserved local corner is the codon
in $F$ whose displayed local coordinate is $11$ and whose biological
bridge row is separated from the three-point $X_{2}$ class. Reserved here
means singleton or control-like under the biological classifier; it does
not mean absent, deleted, or identified with the BEDC channel terminator.
\end{definition}

\begin{definition}[Boundary tile]
\label{def:philosophy-boundary-tile}
A boundary tile is a pair
$$
  \mathsf{Tile}(F):=(F_{X_{2}},F_{11})
$$
where $F\simeq Q_{2}$, $F_{X_{2}}$ is the three-point part indexed by
$X_{2}$, and $F_{11}$ is the singleton part indexed by $11$.
\end{definition}

\begin{theorem}[Local two-bit partition]
\label{thm:philosophy-local-two-bit-partition}
The two-bit square satisfies
$$
  Q_{2}=X_{2}\sqcup\{11\},
  \qquad
  X_{2}=\{00,01,10\}.
$$
\end{theorem}
\begin{proof}
The four two-bit words are exactly $00$, $01$, $10$, and $11$. The
length-two words with no adjacent occurrence of $11$ are $00$, $01$, and
$10$. The remaining word is $11$, and the displayed union is disjoint.
\end{proof}

\begin{theorem}[Stop and tryptophan boundary tile]
\label{thm:philosophy-stop-trp-boundary-tile}
In the $\mathrm{U}+\mathrm{RR}$ face,
$$
\begin{aligned}
  \mathsf{StopTrpFace}_{X_{2}}
    &=\{\mathrm{UAA},\mathrm{UAG},\mathrm{UGA}\},\\
  \mathsf{StopTrpFace}_{11}
    &=\{\mathrm{UGG}\}.
\end{aligned}
$$
\end{theorem}
\begin{proof}
By \autoref{def:philosophy-family-suffix}, the family suffixes in the
$\mathrm{U}+\mathrm{RR}$ face are
$$
\begin{aligned}
  \mathrm{UAA} &\longmapsto 00,\\
  \mathrm{UAG} &\longmapsto 01,\\
  \mathrm{UGA} &\longmapsto 10,\\
  \mathrm{UGG} &\longmapsto 11.
\end{aligned}
$$
The $X_{2}$ part is therefore
$\{\mathrm{UAA},\mathrm{UAG},\mathrm{UGA}\}$, and the local $11$ corner is
$\mathrm{UGG}$.
\end{proof}

\begin{corollary}[Standard-code stop and tryptophan reading]
\label{cor:philosophy-standard-code-stop-trp-reading}
Under the standard-code biological bridge,
$$
\begin{aligned}
  \mathsf{StopTrpFace}_{X_{2}}
    &=\{\mathrm{UAA},\mathrm{UAG},\mathrm{UGA}\}
      =\mathrm{Stop},\\
  \mathsf{StopTrpFace}_{11}
    &=\{\mathrm{UGG}\}
      =\mathrm{Trp}.
\end{aligned}
$$
\end{corollary}
\begin{proof}
The finite partition is \autoref{thm:philosophy-stop-trp-boundary-tile}.
The biological labels are supplied by the external standard genetic-code
bridge.
\end{proof}

\begin{theorem}[Isoleucine and methionine boundary tile]
\label{thm:philosophy-ile-met-boundary-tile}
In the $\mathrm{AU}$ face,
$$
\begin{aligned}
  \mathsf{AUFace}_{X_{2}}
    &=\{\mathrm{AUU},\mathrm{AUC},\mathrm{AUA}\},\\
  \mathsf{AUFace}_{11}
    &=\{\mathrm{AUG}\}.
\end{aligned}
$$
\end{theorem}
\begin{proof}
The first two bases are fixed as $\mathrm{A}$ and $\mathrm{U}$. The third
base ranges over $\mathrm{U}$, $\mathrm{C}$, $\mathrm{A}$, and
$\mathrm{G}$, whose two-axis displays are
$$
\begin{aligned}
  \mathrm{U} &\longmapsto 00,\\
  \mathrm{C} &\longmapsto 01,\\
  \mathrm{A} &\longmapsto 10,\\
  \mathrm{G} &\longmapsto 11.
\end{aligned}
$$
Thus $\mathrm{AUU}$, $\mathrm{AUC}$, and $\mathrm{AUA}$ form the $X_{2}$
part, while $\mathrm{AUG}$ is the local $11$ corner.
\end{proof}

\begin{corollary}[Standard-code isoleucine and methionine reading]
\label{cor:philosophy-standard-code-ile-met-reading}
Under the standard-code biological bridge,
$$
\begin{aligned}
  \mathsf{AUFace}_{X_{2}}
    &=\{\mathrm{AUU},\mathrm{AUC},\mathrm{AUA}\}
      =\mathrm{Ile},\\
  \mathsf{AUFace}_{11}
    &=\{\mathrm{AUG}\}
      =\mathrm{Met}.
\end{aligned}
$$
In an initiation context, the same $\mathrm{AUG}$ row is also the
canonical start-codon bridge row.
\end{corollary}
\begin{proof}
The finite partition is \autoref{thm:philosophy-ile-met-boundary-tile}.
The $\mathrm{Ile}$, $\mathrm{Met}$, and canonical-start labels are
external standard-code bridge rows.
\end{proof}

\begin{corollary}[Dual boundary tiles]
\label{cor:philosophy-dual-boundary-tiles}
Under the standard-code biological bridge, the two distinguished local
tiles have the same displayed $X_{2}\sqcup\{11\}$ form:
$$
\begin{aligned}
  \mathrm{Ile}/\mathrm{Met}
    &=
      \mathrm{AU}+X_{2}\;/\;11,\\
  \mathrm{Stop}/\mathrm{Trp}
    &=
      \mathrm{U}+\mathrm{RR}+X_{2}\;/\;11.
\end{aligned}
$$
\end{corollary}
\begin{proof}
Combine \autoref{cor:philosophy-standard-code-stop-trp-reading} and
\autoref{cor:philosophy-standard-code-ile-met-reading}.
\end{proof}

\begin{theorem}[Reserved-corner interpretation]
\label{thm:philosophy-reserved-corner-interpretation}
In the DNA window-six biological bridge, the local $11$ corner is not a
claim of biological absence. It is a separated singleton or control
corner.
\end{theorem}
\begin{proof}
The two local $11$ corners displayed above are actual standard-code
codons:
$$
\begin{aligned}
  \mathrm{AUG} &= \mathrm{Met}
    \quad\text{and may carry the canonical start row},\\
  \mathrm{UGG} &= \mathrm{Trp}.
\end{aligned}
$$
Therefore the bridge-level reading cannot be that the local $11$ corner
is deleted from the biological code. The licensed interpretation is that
the $11$ corner is separated from the ordinary three-point $X_{2}$ class
and retained as a singleton or control-sensitive row.
\end{proof}

\begin{theorem}[Local not global boundary tile]
\label{thm:philosophy-boundary-tile-local-not-global}
The boundary-tile construction is local $X_{2}$ inside selected codon
faces, not global $X_{6}$ on the flattened codon.
\end{theorem}
\begin{proof}
The global no-$11$ claim is already refused by
\autoref{thm:philosophy-global-no-eleven-not-stop-classifier}. For the
stop tile, the flattened codons $\mathrm{UAG}$ and $\mathrm{UGA}$ contain
adjacent $11$ in the six-bit display, even though they belong to the
standard stop class. The active finite pattern is the selected two-bit
face coordinate, not the entire six-bit word.
\end{proof}

\begin{proposition}[Boundary reassignment schema]
\label{prop:philosophy-boundary-reassignment-schema}
Alternative genetic-code tables can be compared by how they move leaves
or singleton corners inside the local tile:
$$
  \{00,01,10\}/\{11\}
  \rightsquigarrow
  \{00,01\}/\{10,11\}.
$$
In the $\mathrm{AU}$ tile this records the possible reassignment of
$\mathrm{AUA}$ from the isoleucine side to the methionine side. In the
$\mathrm{U}+\mathrm{RR}$ tile it records the possible reassignment of
$\mathrm{UGA}$ from the stop side to the tryptophan side.
\end{proposition}
\begin{proof}
In the $\mathrm{AU}$ tile, $\mathrm{AUA}$ carries local coordinate $10$
and $\mathrm{AUG}$ carries local coordinate $11$. In the
$\mathrm{U}+\mathrm{RR}$ tile, $\mathrm{UGA}$ carries local coordinate
$10$ and $\mathrm{UGG}$ carries local coordinate $11$. Moving the
$10$ leaf into the class containing the $11$ corner therefore has the
same local displayed form in both tiles. The fact that a particular
organism or organelle uses such a table is external biological bridge
data.
\end{proof}

\begin{corollary}[Variant-code comparison table]
\label{cor:philosophy-variant-code-comparison-table}
For each biological genetic-code table $T$, the boundary-tile bridge
records the roles
$$
\begin{gathered}
  \mathsf{role}_{T}(\mathrm{AUU}),\quad
  \mathsf{role}_{T}(\mathrm{AUC}),\quad
  \mathsf{role}_{T}(\mathrm{AUA}),\quad
  \mathsf{role}_{T}(\mathrm{AUG}),\\
  \mathsf{role}_{T}(\mathrm{UAA}),\quad
  \mathsf{role}_{T}(\mathrm{UAG}),\quad
  \mathsf{role}_{T}(\mathrm{UGA}),\quad
  \mathsf{role}_{T}(\mathrm{UGG}).
\end{gathered}
$$
The comparison target is not exception-handling; it is a finite record of
which local leaves and singleton corners are grouped by the chosen
biological bridge.
\end{corollary}

\begin{definition}[Boundary-tile finite-state skeleton]
\label{def:philosophy-boundary-tile-finite-state-skeleton}
A minimal translation-state skeleton for the boundary-tile bridge has
states
$$
  S=\{\mathsf{scan},\mathsf{elongate},\mathsf{terminate},
      \mathsf{readthrough},\mathsf{repair},\mathsf{mutate}\}
$$
over the alphabet $\{\mathrm{U},\mathrm{C},\mathrm{A},\mathrm{G}\}$ and
current codon window $w_{t}\in\mathsf{Codon}_{64}^{\uparrow}$. Under the
standard-code bridge, its local transition rows include
$$
\begin{aligned}
  \delta(\mathsf{elongate},\mathrm{UAA})
    &=\mathsf{terminate},\\
  \delta(\mathsf{elongate},\mathrm{UAG})
    &=\mathsf{terminate},\\
  \delta(\mathsf{elongate},\mathrm{UGA})
    &=\mathsf{terminate},\\
  \delta(\mathsf{elongate},\mathrm{UGG})
    &=\mathsf{elongate}\text{-}\mathrm{Trp},\\
  \delta(\mathsf{elongate},\mathrm{AUG})
    &=\mathsf{elongate}\text{-}\mathrm{Met}.
\end{aligned}
$$
An initiation-context bridge may also attach a start role to the
$\mathrm{AUG}$ row. The skeleton is not a closed autonomous biological
dynamics; its transitions depend on cellular machinery and context.
\end{definition}

\begin{theorem}[Boundary-tile cannot-claim boundary]
\label{thm:philosophy-boundary-tile-cannot-claim-boundary}
The boundary-tile bridge does not license any claim that Zeckendorf alone
proves the genetic code, that DNA biology is forced by window six, that
the whole genetic code is global $X_{6}$ or a window-six fold, that
standard stop codons are universal across all genetic-code tables, that
$\mathrm{UGG}$ is tryptophan by BEDC logic alone, that $\mathrm{AUG}$ is a
start codon by BEDC logic alone, that a local biological $11$ corner is
the BEDC channel terminator, that $11$ is biologically absent, or that
the present boundary-tile bridge is already a checked certificate.
\end{theorem}
\begin{proof}
The global $X_{6}$ refusal follows from
\autoref{thm:philosophy-boundary-tile-local-not-global}. The channel-token
refusal follows from
\autoref{thm:philosophy-zstop-no-channel-token-collapse}. The biological
label refusals follow because $\mathrm{Ile}$, $\mathrm{Met}$,
$\mathrm{Start}$, $\mathrm{Stop}$, and $\mathrm{Trp}$ are external
standard-code bridge rows. The checked-status refusal follows from
\autoref{def:philosophy-boundary-tile-bridge-status}.
\end{proof}

\begin{corollary}[Allowed boundary-tile claim]
\label{cor:philosophy-boundary-tile-allowed-claim}
Under the two-axis RNA or DNA readback and the standard-code biological
bridge, the $\mathrm{AU}$ face and the $\mathrm{U}+\mathrm{RR}$ face are
candidate dual boundary tiles:
$$
\begin{aligned}
  \mathrm{AU}+X_{2}
    &=\{\mathrm{AUU},\mathrm{AUC},\mathrm{AUA}\},&
  \mathrm{AU}+11
    &=\{\mathrm{AUG}\},\\
  \mathrm{U}+\mathrm{RR}+X_{2}
    &=\{\mathrm{UAA},\mathrm{UAG},\mathrm{UGA}\},&
  \mathrm{U}+\mathrm{RR}+11
    &=\{\mathrm{UGG}\}.
\end{aligned}
$$
The biological interpretation reads the three-point parts as
$\mathrm{Ile}$ and $\mathrm{Stop}$, and reads the $11$ corners as
$\mathrm{Met}$ with a possible start row and $\mathrm{Trp}$. This is a
conditional structural bridge, not a theorem that Zeckendorf logic alone
determines biology.
\end{corollary}
