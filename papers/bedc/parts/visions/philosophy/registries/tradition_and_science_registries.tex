\subsection{Tradition registry}
\label{subsec:philosophy-tradition-registry-table}

\begin{definition}[Tradition registry entry]
\label{def:philosophy-tradition-registry-entry}
A tradition entry records tradition, traditional claim, BEDC site,
preserved structure, rejected surplus, and status.
\end{definition}

\begin{theorem}[Rejected surplus is required]
\label{thm:philosophy-tradition-registry-rejected-surplus}
A tradition comparison without rejected surplus is refused.
\end{theorem}
\begin{proof}
Without rejected surplus, structural adjacency becomes doctrinal
identity. The comparison would then import more metaphysics than the
BEDC site licenses.
\end{proof}

\subsection{Science registry}
\label{subsec:philosophy-science-registry-table}

\begin{definition}[Science registry entry]
\label{def:philosophy-science-registry-entry}
A science entry records name, records carrier, invariant carrier,
empirical ledger, prediction gap, bridge status, claim status, and
failure surface.
\end{definition}

\begin{theorem}[Science roadmaps require failure surfaces]
\label{thm:philosophy-science-roadmaps-require-failure}
Every scientific roadmap claim must list a failure surface.
\end{theorem}
\begin{proof}
Without a failure surface, a roadmap becomes a one-way promise of
progress. BEDC roadmaps must state what would block promotion, such as
host leakage, choice dependence, propositional extensionality, missing
invariant, or missing empirical ledger.
\end{proof}

\begin{corollary}[Science registry exports discipline]
\label{cor:philosophy-science-registry-exports-discipline}
The current science registry exports audit discipline, not a completed
physical theory.
\end{corollary}
\begin{proof}
Several entries remain roadmap or schema-only and lack full naming
certificates, empirical ledgers, or standard bridges.
\end{proof}
