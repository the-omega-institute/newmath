\subsection{Theorem registry}
\label{subsec:philosophy-theorem-registry-table}

\begin{definition}[Theorem registry entry]
\label{def:philosophy-theorem-registry-entry}
A theorem entry records name, statement, BEDC dependencies, proof mode,
strength, whether a gap is used, forbidden strengthening, and upgrade
path.
\end{definition}

\begin{definition}[Proof mode]
\label{def:philosophy-proof-mode-registry}
Proof modes include kernel theorem, paper proof, structural reading,
schema-only, roadmap, and gap-boundary statement.
\end{definition}

\begin{theorem}[Theorem strength is bounded by dependencies]
\label{thm:philosophy-theorem-strength-bounded}
Unless a further witness is supplied, the strength of a theorem entry
cannot exceed the weakest strength among its dependencies.
\end{theorem}
\begin{proof}
The theorem obtains its status from the proof chain it cites. A
stronger status would assert an obligation not present in the
dependencies.
\end{proof}

\begin{theorem}[Theorem registry controls strength]
\label{thm:philosophy-theorem-registry-controls-strength}
The theorem registry prevents structural readings from being reported
as machine-checked theorems.
\end{theorem}
\begin{proof}
Each row records proof mode, strength, and forbidden strengthening.
This makes the difference between component support, paper synthesis,
roadmap, and checked theorem visible.
\end{proof}

\subsection{Gap registry}
\label{subsec:philosophy-gap-registry-table}

\begin{definition}[Gap registry entry]
\label{def:philosophy-gap-registry-entry}
A gap entry records name, site, requested supply, socket type, gate,
consequence if internalized, current status, upgrade path, and related
cannot-claim entry.
\end{definition}

\begin{definition}[Socket type registry]
\label{def:philosophy-socket-type-registry}
Socket types are unrestricted induction, unwitnessed existential,
extensional identification, causal continuity, inter-history
coordination, inscription novelty, self-reference closure, and stacked
socket.
\end{definition}

\begin{theorem}[Gap entries require gates]
\label{thm:philosophy-gap-entries-require-gates}
Every gap entry must state the gate that blocks internalization.
\end{theorem}
\begin{proof}
Without a gate, a gap is only an ordinary pending proof note. A BEDC gap
is a boundary claim and must state why the requested supply cannot be
used as an internal rule.
\end{proof}

\begin{theorem}[Gap registry coverage]
\label{thm:philosophy-gap-registry-coverage}
The core gap registry covers the main open boundaries of the philosophy
package.
\end{theorem}
\begin{proof}
The package's central open boundaries are universal closure,
unwitnessed existence, identity collapse, causal future, other minds,
inscription novelty, self-reference, and physical completion. The table
registers each with socket type and gate.
\end{proof}
