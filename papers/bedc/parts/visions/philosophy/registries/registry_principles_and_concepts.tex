\subsection{Registry principles}
\label{subsec:philosophy-registry-principles}

\begin{definition}[Registry]
\label{def:philosophy-registry}
A registry is a finite collection of entries. Each entry records name,
BEDC site, status, forbidden readings, and upgrade path.
\end{definition}

\begin{definition}[Registry family]
\label{def:philosophy-registry-family}
The registry family of the philosophy package consists of concept
registry, theorem registry, gap registry, tradition registry, science
registry, cannot-claim registry, and upgrade registry.
\end{definition}

\begin{theorem}[Registry family as export surface]
\label{thm:philosophy-registry-family-export-surface}
The registry family is the export surface of the philosophy package.
\end{theorem}
\begin{proof}
The philosophy package cannot export unregistered prose as a BEDC
claim. Concepts enter the concept registry, theorems enter the theorem
registry, gaps enter the gap registry, comparisons enter the tradition
registry, scientific claims enter the science registry, refusals enter
the cannot-claim registry, and future promotions enter the upgrade
registry. A claim absent from all registries has no carrier,
dependency, socket, status, or refusal surface.
\end{proof}

\begin{corollary}[No registry, no export]
\label{cor:philosophy-no-registry-no-export}
An unregistered item is not exportable as part of the philosophy
package.
\end{corollary}

\subsection{Concept registry}
\label{subsec:philosophy-concept-registry-table}

\begin{definition}[Concept registry entry]
\label{def:philosophy-concept-registry-entry}
A concept entry contains name, BEDC site, carrier, classifier, ledger,
positive reading, forbidden readings, gap status, closure status, and
formal status.
\end{definition}

\begin{definition}[Concept status]
\label{def:philosophy-concept-status}
Concept statuses include kernel site, certificate site, boundary site,
records-side reading, scientific schema, tradition reading, roadmap,
and cannot-claim.
\end{definition}

\begin{theorem}[Concepts require BEDC sites]
\label{thm:philosophy-concepts-require-bedc-sites}
Every philosophical concept entering BEDC prose requires a BEDC site.
\end{theorem}
\begin{proof}
Without a site, the concept remains external vocabulary. Subject talk
without \(\Hist\), \(\hsame\), and \(\Cont\) becomes subject-substance;
universe talk without the focused universe slice becomes a global
world object; far-end talk without gap position becomes an object-level
entity.
\end{proof}

\begin{theorem}[Core concept registry closure]
\label{thm:philosophy-core-concept-registry-closure}
The core concept registry supplies the minimum registration surface for
the central concepts used by the philosophy package.
\end{theorem}
\begin{proof}
Each row gives a BEDC site, a positive reading, and forbidden readings.
The full entry also carries carrier, classifier, ledger, gap status,
closure status, and formal status. The table is therefore a
paper-synthesis registry, not a machine-checked claim.
\end{proof}

\begin{corollary}[Concept registry extension]
\label{cor:philosophy-concept-registry-extension}
A further concept enters the philosophy package only by supplying the
same fields.
\end{corollary}
