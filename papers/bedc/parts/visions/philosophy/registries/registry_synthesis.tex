\subsection{Registry synthesis}
\label{subsec:philosophy-registry-synthesis}

\begin{theorem}[Registry synthesis]
\label{thm:philosophy-registry-synthesis}
The registry family gives the philosophy package an auditable export
surface: every concept has a BEDC site, every theorem has dependency
and strength, every gap has socket type and gate, every tradition
comparison has rejected surplus, every scientific claim has status and
failure surface, every refusal has reason and upgrade path, every
upgrade path distinguishes current status from next status, and every
ordinary theme returns to a registered entry.
\end{theorem}
\begin{proof}
The concept registry supplies sites. The theorem registry supplies
dependencies and strength. The gap registry supplies socket type and
gate. The tradition registry supplies preserved structure and rejected
surplus. The science registry supplies carriers, status, and failure
surface. The cannot-claim registry supplies present refusals. The
upgrade registry supplies promotion conditions without claiming
completion. The subject index routes ordinary vocabulary back to
registries.
\end{proof}

\begin{corollary}[Registries replace prose expansion]
\label{cor:philosophy-registries-replace-prose-expansion}
Further development of BEDC philosophy should proceed mainly by
registry extension rather than explanatory prose expansion.
\end{corollary}
\begin{proof}
Prose can repeat, drift, or overclaim. Registry entries require
carrier, dependency, socket, status, refusal, and upgrade path.
\end{proof}

\begin{corollary}[Registry appendix route]
\label{cor:philosophy-registry-appendix-route}
The natural publication form of these tables is a set of appendices:
concept registry, theorem registry, gap registry, tradition registry,
science registry, cannot-claim registry, upgrade registry, and subject
index.
\end{corollary}

\subsection{Registry status}
\label{subsec:philosophy-registry-status}

\begin{definition}[Registry package status]
\label{def:philosophy-registry-package-status}
The registry package is a paper-synthesis export. It fixes registry
formats and core entries at the paper level, but it is not a
machine-checked registry structure and does not prove all registered
claims. It does not close gap sockets, convert roadmaps into theorems,
or establish a whole-package no-smuggling theorem.
\end{definition}

\begin{definition}[Registry package upgrade]
\label{def:philosophy-registry-package-upgrade}
To upgrade the registry package, one would define formal structures for
concept, theorem, gap, tradition, science, cannot-claim, upgrade, and
subject-index entries; encode status monotonicity; prove that exported
claims do not conflict with the cannot-claim registry; prove that every
philosophy export points to at least one registry entry; and prove
no-smuggling for forbidden readings.
\end{definition}

\begin{definition}[Registry compression]
\label{def:philosophy-registry-compression}
The registry discipline compresses the philosophy package into a
single rule: no concept without registry, no theorem without
dependency, no gap without socket type, no comparison without rejected
surplus, no scientific claim without status, no export without
cannot-claim boundary, and no upgrade without current status.
\end{definition}
