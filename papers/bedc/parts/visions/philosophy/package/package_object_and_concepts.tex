\subsection{Object of the philosophy package}
\label{subsec:philosophy-package-object}

\begin{definition}[BEDC philosophy package]
\label{def:philosophy-bedc-package}
A BEDC philosophy package is a paper-level structure containing core
thesis, concept registry, theorem map, gap map, tradition map,
scientific map, cannot-claim registry, closure status, and amendment
policy. Its core thesis is auditable generativity; its registries and
maps record how philosophical language is attached to BEDC carriers,
classifiers, ledgers, gaps, bridges, and audit gates.
\end{definition}

\begin{definition}[Not a kernel extension]
\label{def:philosophy-package-not-kernel-extension}
The philosophy package is not a kernel primitive, axiom, object theory,
proof checker, or truth predicate. It is reading discipline, export
discipline, mapping discipline, and audit discipline.
\end{definition}

\begin{theorem}[No primitive added]
\label{thm:philosophy-package-adds-no-primitive}
The BEDC philosophy package adds no primitive to the BEDC kernel.
\end{theorem}
\begin{proof}
All fields of the package refer to existing BEDC surfaces: marks,
histories, continuation, history sameness, naming certificates, gaps,
standard bridges, failure certificates, and cannot-claim entries. The
package organizes and reports these surfaces; it does not introduce a
constructor in the object language.
\end{proof}

\begin{corollary}[Philosophy cannot overrule the kernel]
\label{cor:philosophy-cannot-overrule-kernel}
Any philosophical reading that conflicts with the kernel signature,
naming-certificate discipline, gap discipline, or axiom-purity gate is
refused or downgraded.
\end{corollary}
\begin{proof}
The package has only reading and export authority. It cannot create a
new subject entity, global truth predicate, host equality, or
unwitnessed existence claim.
\end{proof}

\subsection{Concept registry}
\label{subsec:philosophy-concept-registry}

\begin{definition}[Concept entry]
\label{def:philosophy-concept-entry}
A concept entry for \(X\) records name, BEDC carrier, classifier,
ledger, positive reading, forbidden readings, gap status, theory
closure, and formal status.
\end{definition}

\begin{definition}[Core concept registry]
\label{def:philosophy-core-concept-registry}
The core registry contains at least distinction, history, information,
observation, observer, time, object, knowledge, truth boundary, gap,
far-end position, consciousness, universe, space, causality, freedom,
other minds, scientific object, physical law, standard bridge,
tradition mapping, and audit method.
\end{definition}

\begin{definition}[Observer entry]
\label{def:philosophy-observer-concept-entry}
The observer entry has name observer, carrier \(\Hist\), classifier
\(\hsame\) with the licensed observational classifier, ledger the
observer-history identity ledger, positive reading observer equals
history, forbidden readings subject-substance, hidden self-entity, and
global viewer, no gap for records-side observer identity, possible gap
for phenomenal excess, paper-level structural closure, and component
formal status only.
\end{definition}

\begin{definition}[Far-end entry]
\label{def:philosophy-far-end-concept-entry}
The far-end entry has no object-level carrier, no internal classifier,
gap-socket reports as ledger, positive reading as the far end of
forward-binding supply, forbidden readings object-level deity,
theorem-deity, substrate axiom, and intervention entity, boundary
status as its gap status, apophatic theory reading, and no object-level
formal target.
\end{definition}

\begin{theorem}[Core concepts require registry entries]
\label{thm:philosophy-core-concepts-require-entries}
Every core concept entering BEDC philosophy prose requires a concept
entry.
\end{theorem}
\begin{proof}
Without the entry, the reader cannot tell carrier, classifier, ledger,
boundary status, or forbidden readings. Terms such as universe,
consciousness, and far-end position are especially vulnerable to
primitive import without those fields.
\end{proof}

\begin{corollary}[Unregistered concepts are helpers]
\label{cor:philosophy-unregistered-concepts-helpers}
An unregistered concept may serve as explanatory prose, but not as a
BEDC philosophical object.
\end{corollary}
