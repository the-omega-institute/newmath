\subsection{Theorem map}
\label{subsec:philosophy-theorem-map}

\begin{definition}[Theorem map entry]
\label{def:philosophy-theorem-map-entry}
A theorem map entry records philosophical theorem, BEDC dependency,
proof mode, claim strength, and not-claimed surface.
\end{definition}

\begin{definition}[Core theorem map]
\label{def:philosophy-core-theorem-map}
The core theorem map includes: observer equals history, depending on
the continuation signature, history sameness, and absence of a subject
parameter; observation equals continuation, depending on continuation
over histories; time as history growth, depending on elementary history
steps and local continuation chains; object as licensed name, depending
on naming-certificate discipline; knowledge as witness, depending on
classifier, certificate, and ledger witnesses; gap and far-end boundary,
depending on closed observational systems and forward-binding supply;
scientific object as certified records, invariant, bridge, and gap;
and tradition mapping as structural adjacency.
\end{definition}

\begin{theorem}[Philosophical theorems depend on BEDC sites]
\label{thm:philosophy-theorems-depend-bedc-sites}
A philosophical theorem in the package must depend on a BEDC theorem,
definition, or certificate site.
\end{theorem}
\begin{proof}
The philosophy package is not a primitive source of theorems. Without a
BEDC dependency, a philosophical theorem would be an external doctrine
rather than a BEDC reading.
\end{proof}

\begin{corollary}[Theorem-map strength is bounded]
\label{cor:philosophy-theorem-map-strength-bounded}
A philosophical theorem cannot be stronger than the status of the BEDC
sites on which it depends.
\end{corollary}
\begin{proof}
If a dependency is schema-only, statement-only, or roadmap-level, the
mapped philosophical theorem inherits that limitation.
\end{proof}

\subsection{Gap map}
\label{subsec:philosophy-gap-map}

\begin{definition}[Gap map]
\label{def:philosophy-gap-map}
A gap map is a finite table of entries. Each entry records name, site,
requested supply, socket type, gate, current status, upgrade path, and
forbidden internalization.
\end{definition}

\begin{definition}[Core philosophical gaps]
\label{def:philosophy-core-gaps}
The core gap map contains phenomenal-excess gap, induction gap,
causality gap, other-minds gap, freedom gap, self-reference gap,
physics gap, and bridge gap. These mark, respectively, the failure of
records-side conscious records to define phenomenal essence, finite
observation to imply unrestricted universal closure, observed
regularity to imply future persistence, external behavior to yield
another conscious system, derivation to erase inscription actuality,
self-description to become a truth predicate, multi-history physics to
close its witnesses, and host recognition to become checked
no-host-leak bridge.
\end{definition}

\begin{theorem}[Gap map prevents closure illusion]
\label{thm:philosophy-gap-map-prevents-closure-illusion}
The gap map prevents philosophical prose from presenting boundary
questions as closed theorems.
\end{theorem}
\begin{proof}
Consciousness, causality, other minds, freedom, self-reference, and
physics invite overclosed prose. A typed gap entry records the missing
supply and the gate that blocks internalization.
\end{proof}

\begin{corollary}[Gap map is necessary]
\label{cor:philosophy-gap-map-necessary}
A philosophy package without a gap map cannot be exported.
\end{corollary}

\subsection{Tradition map}
\label{subsec:philosophy-package-tradition-map}

\begin{definition}[Tradition map entry]
\label{def:philosophy-package-tradition-map-entry}
A tradition map entry records tradition, traditional claim, BEDC site,
preserved structure, rejected surplus, and audit status.
\end{definition}

\begin{definition}[Core tradition map]
\label{def:philosophy-core-tradition-map}
The core tradition map registers Spencer-Brown at primitive
distinction, Brouwer at history growth and elementary time step,
bundle theory, no-self, and second-order cybernetics at observer equals
history, Kant at certificate and gap conditions of possibility,
Husserl at focused records-side stabilization and naming,
Wittgenstein at certified use, Whitehead at continuation as universal
verb, Tarski, Godel, and Muenchhausen at the open meta loop, and
relational physics at inter-history invariants.
\end{definition}

\begin{theorem}[Tradition map is not membership]
\label{thm:philosophy-tradition-map-not-membership}
The tradition map does not state that BEDC belongs to any mapped
tradition.
\end{theorem}
\begin{proof}
Each entry contains preserved structure and rejected surplus. If the
map were doctrinal membership, rejected surplus would not be a required
field.
\end{proof}

\begin{corollary}[Tradition comparisons are retractable]
\label{cor:philosophy-tradition-comparisons-retractable}
If a comparison imports excessive metaphysics, the entry is removed or
downgraded without affecting the kernel.
\end{corollary}
\begin{proof}
Tradition mapping is a reading layer. Removing it does not alter
\(\Hist\), \(\Cont\), or \(\NameCert\).
\end{proof}

\subsection{Scientific map}
\label{subsec:philosophy-package-scientific-map}

\begin{definition}[Scientific map entry]
\label{def:philosophy-scientific-map-entry}
A scientific map entry records name, records carrier, invariant
carrier, empirical ledger, prediction gap, bridge status, claim
status, and failure surface.
\end{definition}

\begin{definition}[Core scientific map]
\label{def:philosophy-core-scientific-map}
The core scientific map records physical records as history,
measurement, and continuation rows; physical laws as empirical
regularities plus Type IV prediction gaps; space as locality-cell
relation geometry rather than spacetime substrate; maximum causal rate
as schema-only inter-history invariant until cross-history causality
closes; light-speed-like constant as structural skeleton without
empirical value; and Lorentz geometry, curved spacetime, and quantum
state as roadmap naming-certificate targets.
\end{definition}

\begin{theorem}[Scientific map separates structure, model, and value]
\label{thm:philosophy-scientific-map-separates}
Scientific entries must distinguish structure form, model bridge,
empirical value, and prediction commitment.
\end{theorem}
\begin{proof}
A maximum-rate form may be structural, a light-speed number must come
from measurement ledger, a Lorentz geometry is a model or certificate
target, and future persistence is a Type IV gap. Confusing these fields
inflates schemas into empirical theorems.
\end{proof}

\begin{corollary}[Physical claims default downward]
\label{cor:philosophy-physical-claims-default-downward}
Without full naming certificate, invariant surface, empirical ledger,
and bridge, a physical claim remains schema, roadmap, gap, or
failure-surface statement.
\end{corollary}
