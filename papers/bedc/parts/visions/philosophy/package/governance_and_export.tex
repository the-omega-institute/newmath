\subsection{Philosophy cannot-claim registry}
\label{subsec:philosophy-package-cannot-claim}

\begin{definition}[Philosophy cannot-claim registry]
\label{def:philosophy-package-cannot-claim-registry}
The philosophy package cannot-claim registry includes refusals that
BEDC has completed all mathematical reconstruction, proved a complete
physical theory, defined phenomenal consciousness as essence,
internalized the far end as an object-level entity, internalized its
own truth predicate, supplied a global clock or God's-eye observer,
used host equality or host naturals as public surface, made doctrinal
identity claims about traditions, equated roadmap with theorem, or
equated paper synthesis with a checked formal package.
\end{definition}

\begin{definition}[Registry entry format]
\label{def:philosophy-package-registry-entry-format}
Each cannot-claim entry records claim text, status, why it is not
claimable, and required upgrade. The status is forbidden, deferred, or
partial.
\end{definition}

\begin{theorem}[Cannot-claim registry is required]
\label{thm:philosophy-package-cannot-claim-required}
Without a philosophy cannot-claim registry, the philosophy package
cannot be exported.
\end{theorem}
\begin{proof}
Philosophical language easily turns structural readings into completed
metaphysical claims. The registry explicitly blocks those exports and
records the missing upgrade or principled refusal.
\end{proof}

\begin{corollary}[Refusal is content]
\label{cor:philosophy-package-refusal-content}
In the philosophy package, a refusal supported by reason and upgrade
condition is typed negative knowledge.
\end{corollary}

\subsection{Amendment policy}
\label{subsec:philosophy-package-amendment-policy}

\begin{definition}[Philosophy package amendment]
\label{def:philosophy-package-amendment}
A philosophy package amendment is an in-place change to the concept
registry, theorem map, gap map, tradition map, scientific map,
cannot-claim registry, or closure status.
\end{definition}

\begin{definition}[In-place amendment]
\label{def:philosophy-package-in-place-amendment}
Amendments are made in the thematic chapter whose subject they advance.
Replaced phrasing is removed when the active formulation takes its
place; parallel active readings are not retained.
\end{definition}

\begin{definition}[Active label]
\label{def:philosophy-package-active-label}
When an amendment changes the active reading of an object, the active
label remains the citation target, and definition or theorem labels
attach to the active formulation.
\end{definition}

\begin{theorem}[Amendment preserves audit continuity]
\label{thm:philosophy-package-amendment-audit-continuity}
An amendment must preserve audit continuity by identifying the affected
registry entry, claim strength, gap classification, cannot-claim
status, and tradition-map status.
\end{theorem}
\begin{proof}
In-place replacement keeps the theory source of truth clean, but it can
otherwise obscure whether a claim was strengthened, weakened,
reclassified, or refused. Audit continuity records the status effect of
the replacement.
\end{proof}

\begin{corollary}[No stacked interpretations]
\label{cor:philosophy-package-no-stacked-interpretations}
Inactive philosophical readings should not remain as parallel active
versions.
\end{corollary}
\begin{proof}
Parallel active readings would make concept and theorem map entries
non-unique.
\end{proof}

\subsection{Capstone policy}
\label{subsec:philosophy-package-capstone-policy}

\begin{definition}[Philosophy capstone]
\label{def:philosophy-package-capstone}
A philosophy capstone is a unifying claim spanning at least three
registered domains.
\end{definition}

\begin{definition}[Capstone is not commentary]
\label{def:philosophy-package-capstone-not-commentary}
A capstone provides structural unification rather than ordinary
summary or meta-commentary.
\end{definition}

\begin{theorem}[Capstones require dependencies]
\label{thm:philosophy-package-capstones-require-dependencies}
A philosophy capstone must list its cross-domain dependencies.
\end{theorem}
\begin{proof}
The point of a capstone is structural unity. Without the domains,
carriers, classifiers, ledgers, and gaps it spans, the claim cannot be
audited as a capstone.
\end{proof}

\begin{corollary}[Observer-history capstone]
\label{cor:philosophy-package-observer-history-capstone}
The current core philosophy capstone is the observer-history identity:
observer equals history, observation equals continuation, time equals
history growth, and no observer-substance is present. Its dependencies
are \(\Hist\), \(\hsame\), \(\Cont\), elementary history steps, absence
of a subject parameter, and the axiom-purity gate.
\end{corollary}

\subsection{Export interface}
\label{subsec:philosophy-package-export-interface}

\begin{definition}[Philosophy export]
\label{def:philosophy-export}
A philosophical claim exports when it has a concept entry, a theorem
map, gap map, or tradition map entry, a recorded closure status, and no
active cannot-claim conflict.
\end{definition}

\begin{definition}[Export grade]
\label{def:philosophy-export-grade}
Export grades are reading export, paper synthesis export, component
checked export, formal package target, and formal package checked.
\end{definition}

\begin{theorem}[Current export grade]
\label{thm:philosophy-current-export-grade}
The present philosophy framework has paper synthesis export status, not
formal package checked status.
\end{theorem}
\begin{proof}
The preceding parts supply organized concepts, theorem readings, gaps,
tradition maps, scientific maps, research route, and audit method.
They do not supply checked Lean structures for the whole philosophy
package or a whole-package checked target.
\end{proof}

\begin{corollary}[Upgrade path for export]
\label{cor:philosophy-export-upgrade-path}
To reach formal package target status, the project must define
structures for concept entries, theorem map entries, gap map entries,
tradition map entries, scientific map entries, cannot-claim entries,
and the philosophy package, together with well-formedness,
no-smuggling, status monotonicity, cannot-claim consistency, and gap
classification predicates.
\end{corollary}
