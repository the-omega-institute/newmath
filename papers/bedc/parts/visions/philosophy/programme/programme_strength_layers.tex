\subsection{Definition of the research programme}
\label{subsec:philosophy-programme-definition}

\begin{definition}[BEDC research programme]
\label{def:philosophy-bedc-research-programme}
The BEDC research programme starts from the finite kernel
$$
\bzero,\bone \to \Hist \to \msame,\hsame \to
\mathsf{Ext},\Cont \to \mathsf{Ask},\mathsf{Sig}
\to \mathsf{Pkg},\Gap \to \NameCert,
$$
and, for each mature name \(N\), constructs the five-field certificate
consisting of source specification, pattern specification, classifier
specification, stability certificate, and ledger policy. The resulting
name is then assigned theory closure, formal verification, bridge, and
refusal status.
\end{definition}

\begin{definition}[Research target]
\label{def:philosophy-programme-research-target}
The target of research is not to collect standard objects as
assumptions. It is to display the exact generative behavior by which a
standard name earns a licensed place in BEDC.
\end{definition}

\begin{theorem}[Research is not expansionism]
\label{thm:philosophy-programme-not-expansionism}
The BEDC research programme is not the outward expansion of an object
library, but the inward completion of eligibility proofs for names.
\end{theorem}
\begin{proof}
If the programme were expansionist, it could import mature objects such
as numbers, reals, sets, functions, topologies, and categories as
assumptions. The roadmap instead treats mature constructions as
licensed names and treats host naturals, host lists, definitional
equality, quotient collapse, choice, and propositional extensionality
as contamination when exposed on the public BEDC surface. Hence the
programme proves license rather than importing objects.
\end{proof}

\concretizedIn{ch:concrete-instances-name-eligibility-proof-namecert}{Name eligibility proofs are realized as finite source, classifier, strength, audit, refusal, transport, replay, provenance, and NameCert packets.}

\begin{corollary}[The object of research is the certificate]
\label{cor:philosophy-programme-researches-certificates}
When BEDC studies \(\mathsf{Real}^{\uparrow}\),
\(\mathsf{Cat}^{\uparrow}\), \(\mathsf{Topology}^{\uparrow}\), or
\(\mathsf{Zeta}^{\uparrow}\), it studies their naming certificates,
classifiers, ledgers, stability surfaces, bridges, and failure
surfaces.
\end{corollary}
\begin{proof}
The mature name is admitted only through the certificate. Without that
certificate, the mature object remains an external target or roadmap
name.
\end{proof}

\subsection{Strength gradient}
\label{subsec:philosophy-programme-strength-gradient}

\begin{definition}[Research strength]
\label{def:philosophy-programme-research-strength}
Each derived mature-object claim lies on a strength gradient:
$$
\mathsf{naming\ seed}
< \mathsf{paper\ certificate}
< \mathsf{machine\ checked\ certificate}
< \mathsf{conservative\ bridge}.
$$
Lower strength may not be reported as higher strength.
\end{definition}

\begin{definition}[Four strengths]
\label{def:philosophy-programme-four-strengths}
A naming seed proposes a paper-side carrier and classifier while
obligations remain open. A paper certificate supplies the five naming
certificate fields and a prose proof. A machine-checked certificate
formalizes the relevant certificate fields and stability obligations in
the Lean development and resolves the cited target markers. A
conservative bridge gives an external two-way translation to a mature
standard object while preserving no-host-leak.
\end{definition}

\begin{theorem}[No strength inflation]
\label{thm:philosophy-programme-no-strength-inflation}
If \(N\) has current strength \(s\), BEDC cannot claim for \(N\) any
strength strictly above \(s\).
\end{theorem}
\begin{proof}
Each higher strength adds an undischarged obligation: full certificate
fields above seed, machine verification above paper certificate, and
host-relative conservative translation above machine checking.
Reporting a higher strength would present missing obligations as
complete.
\end{proof}

\begin{corollary}[Philosophical claims report strength]
\label{cor:philosophy-programme-philosophy-reports-strength}
Claims about consciousness, freedom, universe, and physical law must
be reported as structural readings, paper certificates, schemas, gap
sockets, machine targets, or bridge targets, rather than as unqualified
proofs.
\end{corollary}
\begin{proof}
Philosophical claims also carry naming, classification, stability,
ledger, and boundary obligations. Without strength reporting, readings,
schemas, gaps, and theorems are confused.
\end{proof}

\subsection{Layer route}
\label{subsec:philosophy-programme-layer-route}

\begin{definition}[Layer route]
\label{def:philosophy-programme-layer-route}
The BEDC reconstruction route is layered:
$$
\begin{gathered}
L0\ \mathsf{Marks},\quad
L1\ \mathsf{Histories},\quad
L2\ \mathsf{GenerativeDynamics},\quad
L3\ \mathsf{AskingAndSignature},\\
L4\ \mathsf{PackageRelationsAndGaps},\quad
L5\ \mathsf{NamingAndStability},\quad
L6\ \mathsf{FiniteData},\quad
L7\ \mathsf{Arithmetic},\\
L8\ \mathsf{Algebra},\quad
L9\ \mathsf{OrderLogicSetLikeTypeLike},\quad
L10\ \mathsf{TopologyMetricCompletionReal},\\
L11\ \mathsf{Analysis},\quad
L12\ \mathsf{GeometryCategoryHomology},\quad
L13\ \mathsf{NumberTheoryAndSpectra},\quad
L14\ \mathsf{MatureSystemBridge}.
\end{gathered}
$$
Each layer may depend only on lower layers.
\end{definition}

\begin{definition}[Layer dependency]
\label{def:philosophy-programme-layer-dependency}
\(\mathsf{Depends}(L_i,L_j)\) is licensed only when \(i>j\). Cyclic
dependence between layers is not part of the route.
\end{definition}

\begin{theorem}[Lowest open layer first]
\label{thm:philosophy-programme-lowest-open-layer-first}
The research route advances the lowest open layer first.
\end{theorem}
\begin{proof}
A high-layer claim imports all unclosed lower-layer obligations into
its vocabulary. For example, analytic or spectral claims hide
completion, algebraic descent, order, bridge, and continuity
obligations unless the lower layers are already closed. The layer index
therefore functions as review priority.
\end{proof}

\begin{corollary}[Over-layer work is roadmap]
\label{cor:philosophy-programme-over-layer-roadmap}
If a high-layer target depends on an open lower layer, the high-layer
target is a roadmap, schema, proof target, deferred claim, or gap, not
a closed theorem.
\end{corollary}
\begin{proof}
The missing lower layer withholds source, classifier, stability,
exactness, or bridge data needed by the high-layer certificate.
\end{proof}

\concretizedIn{ch:concrete-instances-programme-strength-ledger-namecert}{Programme strength ledgers realize the seed, paper-certificate, machine-check, bridge, dependency, refusal, and no-strength-inflation rows as one finite NameCert packet.}
