\subsection{Finite kernel baseline}
\label{subsec:philosophy-programme-kernel-baseline}

\begin{definition}[Finite kernel baseline]
\label{def:philosophy-programme-finite-kernel-baseline}
The finite kernel baseline contains marks, histories, extension,
continuation, asking, signatures, signature sameness, packages, gaps,
composite gaps, and naming certificates, with dependency shape
$$
\mathsf{Mark}
\to \Hist
\to \mathsf{Ext},\Cont
\to \mathsf{Ask},\mathsf{Sig}
\to \mathsf{sameSig}
\to \mathsf{Pkg},\Gap
\to \mathsf{CompGap}
\to \NameCert.
$$
\end{definition}

\begin{definition}[Kernel exit condition]
\label{def:philosophy-programme-kernel-exit-condition}
The L0--L5 exit condition is that the kernel theorems are checked at
their cited targets and remain free of axiom-purity contamination. The
surface includes sameness equivalences and no-confusion, extension and
continuation laws, asking and signature determinacy under policies,
package and gap soundness, coverage, and separation, and naming
certificate admission.
\end{definition}

\begin{theorem}[Finite kernel precedes mature objects]
\label{thm:philosophy-programme-kernel-precedes-mature}
Every mature-object certificate depends on the finite kernel baseline.
\end{theorem}
\begin{proof}
\(\NameCert\) is the universal admission rule. It depends on package
and gap structure, which depends on signatures, which depends on
continuation, histories, and marks. Without this baseline, no mature
name has a proof boundary.
\end{proof}

\begin{corollary}[The philosophy package depends on L0--L5]
\label{cor:philosophy-programme-philosophy-depends-kernel}
Any formalization of the philosophy package must return to L0--L5:
distinction to marks, history to \(\Hist\), observation and time to
\(\Cont\), object license to \(\NameCert\), and boundary memory to
\(\Gap\).
\end{corollary}

\subsection{Phase bundles}
\label{subsec:philosophy-programme-phase-bundles}

\begin{definition}[Phase bundle]
\label{def:philosophy-programme-phase-bundle}
A phase bundle is the collection of objects sanctioned for priority
during a research period, together with a precise exit criterion.
\end{definition}

\begin{definition}[Near-term phases]
\label{def:philosophy-programme-near-term-phases}
The near-term route is organized as follows: kernel-checked baseline;
finite data from booleans through lists; full arithmetic through
rationals with induction, recursion, ordered-ring, and ordered-field
structure; algebra up to fields; real and completion through dyadic
rational core, stream names, regular rational sequences, and real
names; real algebra and geometry; compactness and continuity; linear
algebra and analysis seeds; categories and bridges for already checked
objects.
\end{definition}

\begin{theorem}[No phase skipping]
\label{thm:philosophy-programme-no-phase-skipping}
While a phase has not exited, targets belonging to later phases are
deferred.
\end{theorem}
\begin{proof}
The phase determines the current sanctioned advancement direction.
Later-phase targets depend on obligations the current phase is meant
to discharge. Advancing them as closed would hide that dependency.
\end{proof}

\begin{corollary}[Dependency before attraction]
\label{cor:philosophy-programme-dependency-before-attraction}
High-profile goals such as the Riemann Hypothesis, quantum states,
curved spacetime, or spectral theorems cannot bypass lower certificates.
\end{corollary}
\begin{proof}
Such goals depend on real numbers, analysis, topology, algebra,
bridges, or multi-history semantics. Until those dependencies are
closed, the goals remain roadmaps or proof targets.
\end{proof}

\subsection{Off-the-table claims}
\label{subsec:philosophy-programme-off-table}

\begin{definition}[Off-the-table claim]
\label{def:philosophy-programme-off-table-claim}
An off-the-table claim is a claim explicitly not made until its named
milestone holds at the required strength.
\end{definition}

\begin{definition}[Current refusal list]
\label{def:philosophy-programme-current-refusal-list}
The current refusal surface includes claims that BEDC has re-derived
all mature mathematics; that real numbers, the unit circle, or folds
are complete; that a natural-number seed is full natural-number theory;
that an addition seed is full addition; that paper certificates are
machine-checked certificates; that scaffolds are verified
implementations; that external models replace BEDC proof obligations;
or that host equality, host natural numbers, or host functions belong
to the BEDC public surface.
\end{definition}

\begin{theorem}[Off-the-table is a boundary condition]
\label{thm:philosophy-programme-off-table-boundary}
The off-the-table list is a boundary condition of the research
programme, not a rhetorical caution.
\end{theorem}
\begin{proof}
Without explicit refusals, seeds, paper certificates, scaffolds, and
external models could be misread as mature exports. The refusal list
prevents that inference and records which claim remains unsupported.
\end{proof}

\begin{corollary}[Philosophical synthesis must not overclaim]
\label{cor:philosophy-programme-synthesis-not-overclaim}
The philosophy package cannot claim that BEDC has completed all
mathematics, proved physics, internalized consciousness as essence, or
supplied a global truth predicate. It can claim a certificate and audit
discipline for any such target to enter.
\end{corollary}
