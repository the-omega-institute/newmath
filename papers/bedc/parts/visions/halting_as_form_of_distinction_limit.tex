\chapter{The Halting Problem as a Limit of the Form of Distinction}
\label{ch:visions-halting-as-form-of-distinction-limit}

Reading the form of distinction (\autoref{def:form-of-distinction}) as the primitive computational atom and the continuation $\Cont$ as the small-step relation (\autoref{ch:visions-computation-as-continuation}), the halting problem becomes the question: is there a BEDC classifier on naming certificates that decides, for an arbitrary certificate $\NameCert_{P}$ and history $h$, whether the continuation chain from $h$ in $\NameCert_{P}$ terminates? This chapter records the kernel fixed-point interface behind the negative answer as a Tarski-style obstruction in BEDC, locates it on the open-meta-loop side of the two-loops structure (\autoref{thm:two-loops}), and notes that the obstruction is, structurally, the same as the form of distinction's irreducibility (\autoref{thm:form-of-distinction-irreducible}) read at the meta level.

\section{Halting as fixed-point obstruction}
\label{sec:halting-as-form-of-distinction-fixed-point}

\begin{definition}[Termination of a $\Cont$-chain]
\label{def:halting-termination}
The $\Cont$-chain from $h$ in $\NameCert_{P}$ \emph{terminates} when there exists a finite list of marks $[m_1, \ldots, m_n]$ and a sequence $h = h_0, h_1, \ldots, h_n$ with $\Cont_P\,h_i\,m_{i+1}\,h_{i+1}$ for $0 \le i < n$ and no $m, h'$ with $\Cont_P\,h_n\,m\,h'$. Termination is a positive existential at the kernel level.
\end{definition}

\begin{definition}[Halting predicate]
\label{def:halting-predicate}
A \emph{halting predicate} is a naming certificate $\NameCert_H$ over the carrier $\mathcal{C} \times \Hist$ of pairs (certificate, history) whose $\Ext_H$ on $(\NameCert_P, h)$ holds exactly when the $\Cont_P$-chain from $h$ terminates in the sense of \autoref{def:halting-termination}, and whose $\Cont_H$ supplies the kernel-level evidence required by the certificate framework.
\end{definition}

\begin{theorem}[Halting via fixed point of the form of distinction]
\label{thm:halting-as-form-of-distinction-fixed-point}
Let $\NameCert_H$ be a naming certificate for a predicate $\mathsf{Halts}$ and classifier $\sim_H$, and suppose that $\mathsf{Halts}(h)$ is exactly the positive kernel fact that some $\Cont$-continuation from $h$ exists. Let $\mathsf{SelfRef}$ be a self-reference predicate whose histories are produced by one $\Ext$ step, and let a diagonal policy say that whenever a self-referential $h$ continues with empty fuel to $r$, any classification $h\sim_H r$ yields $\mathsf{Diverges}(r)$. Then $\sim_H$ transports $\mathsf{Halts}$, every history satisfies $\mathsf{Halts}$ by the empty $\Cont$-continuation, and every self-referential history carries an $\Ext$ witness together with the fixed point $\Cont\,h\,\emptyset\,h$ and the diagonal implication $\sim_H h h \Rightarrow \mathsf{Diverges}(h)$. This is the kernel form of the halting obstruction: the contradiction is not a separate primitive but appears only when a classifier is also required to close that diagonal divergence claim.
\leanchecked{BEDC.Reflection.halting\_as\_form\_of\_distinction\_fixed\_point\_proof}
\leanvariant{BEDC.Reflection.halting\_as\_form\_of\_distinction\_fixed\_point}
\end{theorem}

\begin{proof}[Sketch]
The certificate field \texttt{carrier\_respects\_equiv} gives transport of $\mathsf{Halts}$ along $\sim_H$. The empty continuation law gives $\Cont\,h\,\emptyset\,h$ for every $h$, hence $\mathsf{Halts}(h)$ under the stated equivalence. If $h$ is self-referential, the self-reference hypothesis supplies an $\Ext$ predecessor and mark for $h$; using the empty continuation as the fixed point gives the witness $r=h$. The diagonal implication is then exactly the supplied policy applied to this empty-fuel continuation.
\end{proof}

\section{Halting as open-meta-loop closure}
\label{sec:halting-meta-loop-closure}

\begin{theorem}[Halting decidability is open-meta-loop closure]
\label{thm:halting-meta-loop-closure}
Accepting a halting predicate as in \autoref{def:halting-predicate} closes the open meta loop of \autoref{ch:visions-reflection-and-limits} at the certificate stratum: it supplies a $\NameCert$-level decision for a predicate over the carrier of all naming certificates, which is precisely the stratum the meta loop refuses to settle.
\end{theorem}

\begin{proof}[Sketch]
A halting predicate is, by \autoref{def:halting-predicate}, a $\NameCert$-level decision procedure indexed by the certificate-of-certificates carrier $\mathcal{C}$. By the unification theorem of \autoref{thm:reasoning-forbidden-implements-meta-closure}, supplying a $\NameCert$-level decision indexed by $\mathcal{C}$ closes the open meta loop. The diagonal interface of \autoref{thm:halting-as-form-of-distinction-fixed-point} is the kernel-level witness showing where such a closure must also account for self-reference.
\end{proof}

\section{Connection to the form of distinction}
\label{sec:halting-form-of-distinction}

\begin{theorem}[Halting irreducibility is form-of-distinction irreducibility at the meta level]
\label{thm:halting-form-of-distinction-meta}
The fixed-point interface for the halting predicate (\autoref{thm:halting-as-form-of-distinction-fixed-point}) is the meta-level reading of the irreducibility of the form of distinction (\autoref{thm:form-of-distinction-irreducible}): both expose a primitive distinction that a classifier must already respect. The form of distinction is the irreducible primitive at the constructor stratum (telling $\bzero$ and $\bone$ apart); halting is the corresponding pressure point at the certificate stratum (telling terminating from non-terminating $\Cont$-chains apart).
\end{theorem}

\begin{proof}[Sketch]
\autoref{thm:form-of-distinction-irreducible} shows that any classifier $\sim$ on a carrier $G$ presupposes the form of distinction in its very statement. Read $G$ as $\mathcal{C}$ and $\sim$ as the would-be halting classifier: deciding whether $(\NameCert_P, h) \sim (\NameCert_Q, k)$ in the sense of having the same termination status is an instance of the form of distinction at the certificate stratum. The halting interface is therefore the form-of-distinction irreducibility specialised to $G = \mathcal{C}$ and $\sim$ = halting equivalence; the diagonal fixed point is the explicit witness that this instance does not collapse to a simpler kernel primitive.
\end{proof}

\begin{remark}[Two-loops architecture]
\label{rem:halting-two-loops}
\autoref{thm:halting-as-form-of-distinction-fixed-point} and \autoref{thm:form-of-distinction-irreducible} sit on the two opposite faces of \autoref{thm:two-loops}: the form of distinction must close at the bottom for the kernel to start (closed ground loop), and the same closure exposes the self-reference burden carried by halting decidability at the top (open meta loop). The two-loops structure is, read through this chapter, the discipline that the closure paid at the bottom is not silently transferred to the top.
\end{remark}

\begin{theorem}[Halting predicate exists iff the meta loop closes at the certificate stratum]
\label{thm:halting-predicate-iff-meta-loop-certificate-stratum}
Let
$$
\mathsf{Halts}(\NameCert_P,h)
$$
denote the predicate that the $\Cont_P$-chain from $h$ terminates in the sense of \autoref{def:halting-termination}. The following are equivalent:
\begin{enumerate}
\item there exists a halting predicate $\NameCert_H$ in the sense of \autoref{def:halting-predicate};
\item there exists a $\NameCert$-level decision procedure indexed by the certificate-of-certificates carrier $\mathcal{C}$, with carrier $\mathcal{C}\times\Hist$, whose extension predicate decides $\mathsf{Halts}(\NameCert_P,h)$ for every pair $(\NameCert_P,h)$.
\end{enumerate}
In this situation the open meta loop is closed at the certificate stratum precisely by that deciding certificate.
\end{theorem}

\begin{proof}
Assume (1). By \autoref{def:halting-predicate}, the certificate $\NameCert_H$ is already a naming certificate over $\mathcal{C}\times\Hist$ whose extension predicate on $(\NameCert_P,h)$ holds exactly when the $\Cont_P$-chain from $h$ terminates. Its continuation field is the kernel-level evidence required by the certificate framework. Therefore $\NameCert_H$ is exactly a $\NameCert$-level decision procedure of the form stated in (2). By \autoref{thm:halting-meta-loop-closure}, this deciding certificate is exactly what closes the open meta loop at the certificate stratum for the halting predicate.

Assume (2). Let $\NameCert_D$ be the stated $\NameCert$-level decision procedure. By hypothesis, $\NameCert_D$ lives over the carrier $\mathcal{C}\times\Hist$, its extension predicate on $(\NameCert_P,h)$ holds exactly when the $\Cont_P$-chain from $h$ terminates, and its continuation field supplies the certificate-level evidence attached to that decision. But these are precisely the clauses of \autoref{def:halting-predicate}. Therefore $\NameCert_D$ is a halting predicate; rename it $\NameCert_H$.

Thus (1) and (2) are equivalent, and the final sentence follows from \autoref{thm:halting-meta-loop-closure} applied to that same deciding certificate.
\end{proof}
