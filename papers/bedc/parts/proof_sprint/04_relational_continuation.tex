\section{Relational continuation}

\begin{definition}[Continuation relation]
Continuation is a relation, not a function.  The judgment
\[
\Cont(h,k,r)
\]
means that continuing history \(h\) by history \(k\) may generate result \(r\).  It is generated by:
\[
\Cont(\e,k,k),
\]
\[
\Cont(h,k,r)\Rightarrow \Cont(\Ezero(h),k,\Ezero(r)),
\]
\[
\Cont(h,k,r)\Rightarrow \Cont(\Eone(h),k,\Eone(r)).
\]
\end{definition}

\begin{theorem}[Continuation determinacy]
\label{thm:continuation-determinacy}
If \(\Cont(h,k,r)\) and \(\Cont(h,k,r')\), then \(\HSame(r,r')\).
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_deterministic}
\begin{proof}
By induction on the first continuation derivation.  In the empty case, both results must be internally the same as \(k\).  In the \(\Ezero\)-case, closed generation forces the second derivation to be an \(\Ezero\)-case, and the induction hypothesis applies to the subresults.  The \(\Eone\)-case is analogous.
\end{proof}

\begin{theorem}[Continuation left unit]
\label{thm:continuation-left-unit-proof-sprint}
For every generated history \(k\), \(\Cont(\e,k,k)\).
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_left\_unit\_induction\_spine}
\begin{proof}
This is the first generator of \(\Cont\).
\end{proof}

\begin{theorem}[Continuation right unit]
For every generated history \(h\), \(\Cont(h,\e,h)\).
\end{theorem}
\begin{proof}
By induction on \(h\).  The empty case is \(\Cont(\e,\e,\e)\).  If \(h=\Ezero(h_0)\), apply the \(\Ezero\)-continuation rule to the induction hypothesis.  The \(\Eone\)-case is analogous.
\end{proof}

\begin{theorem}[Continuation associativity, relational form]
If \(\Cont(a,b,ab)\) and \(\Cont(b,c,bc)\), then there exists a generated history \(abc\) such that
\[
\Cont(ab,c,abc)
\qquad\text{and}\qquad
\Cont(a,bc,abc).
\]
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_assoc\_exists}
\begin{proof}
By induction on the derivation of \(\Cont(a,b,ab)\).  In the empty case, take \(abc=bc\).  In the \(\Ezero\)-case, apply the induction hypothesis to the subderivation and prefix the resulting witness by \(\Ezero\).  The \(\Eone\)-case is analogous.
\end{proof}
