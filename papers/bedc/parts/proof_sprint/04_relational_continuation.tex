\section{Relational continuation}

\begin{definition}[Continuation relation]
Continuation is a relation, not a function.  The judgment
$$
\Cont(h,k,r)
$$
means that continuing history $h$ by history $k$ may generate result $r$.  It is generated by:
$$
\Cont(h,\e,h),
$$
$$
\Cont(h,k,r)\Rightarrow \Cont(h,\Ezero(k),\Ezero(r)),
$$
$$
\Cont(h,k,r)\Rightarrow \Cont(h,\Eone(k),\Eone(r)).
$$
\end{definition}

\begin{theorem}[Continuation determinacy]
\label{thm:continuation-determinacy}
If $\Cont(h,k,r)$ and $\Cont(h,k,r')$, then $\HSame(r,r')$.
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_deterministic}
\begin{proof}
Induct on the right-history structure exposed by the first continuation derivation.  In the empty case, both results are internally the same as $h$.  In the $\Ezero$-case, closed generation forces the second derivation to be an $\Ezero$-case over the same right-history predecessor, and the induction hypothesis applies to the subresults.  The $\Eone$-case is analogous.
\end{proof}

\begin{theorem}[Continuation left unit]
\label{thm:continuation-left-unit-proof-sprint}
For every generated history $k$, $\Cont(\e,k,k)$.
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_left\_unit\_induction\_spine}
\begin{proof}
Induct on $k$.  The empty case is the base generator $\Cont(\e,\e,\e)$.  The $\Ezero$ and $\Eone$ cases apply the corresponding right-recursive continuation step to the induction hypothesis.
\end{proof}

\begin{theorem}[Continuation right unit]
\label{thm:continuation-right-unit}
For every generated history $h$, $\Cont(h,\e,h)$.
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_right\_unit}
\begin{proof}
This is the base generator of $\Cont$.
\end{proof}

\begin{theorem}[Continuation associativity, relational form]
If $\Cont(a,b,ab)$ and $\Cont(b,c,bc)$, then there exists a generated history $abc$ such that
$$
\Cont(ab,c,abc)
\qquad\text{and}\qquad
\Cont(a,bc,abc).
$$
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_assoc\_exists}
\begin{proof}
Use the checked right-recursive append characterization of continuation and the append associativity spine.  Equivalently, induct on the right continuation by $c$: the empty case takes $abc=ab$, and the $\Ezero$ and $\Eone$ cases lift the witness with the matching right-recursive continuation step.
\end{proof}
