\section{Relational continuation}

\begin{definition}[Continuation relation]
Continuation is a relation, not a function.  The judgment
$$
\Cont(h,k,r)
$$
means that continuing history $h$ by history $k$ may generate result $r$.  It is generated by:
$$
\Cont(h,\e,h),
$$
$$
\Cont(h,k,r)\Rightarrow \Cont(h,\Ezero(k),\Ezero(r)),
$$
$$
\Cont(h,k,r)\Rightarrow \Cont(h,\Eone(k),\Eone(r)).
$$
\end{definition}

\begin{theorem}[Continuation determinacy]
\label{thm:continuation-determinacy}
If $\Cont(h,k,r)$ and $\Cont(h,k,r')$, then $\HSame(r,r')$.
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_deterministic}
\begin{proof}
Induct on the right-history structure exposed by the first continuation derivation.  In the empty case, both results are internally the same as $h$.  In the $\Ezero$-case, closed generation forces the second derivation to be an $\Ezero$-case over the same right-history predecessor, and the induction hypothesis applies to the subresults.  The $\Eone$-case is analogous.
\end{proof}

\begin{theorem}[Continuation left unit]
\label{thm:continuation-left-unit-proof-sprint}
For every generated history $k$, $\Cont(\e,k,k)$.
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_left\_unit\_induction\_spine}
\begin{proof}
Induct on $k$.  The empty case is the base generator $\Cont(\e,\e,\e)$.  The $\Ezero$ and $\Eone$ cases apply the corresponding right-recursive continuation step to the induction hypothesis.
\end{proof}

\begin{theorem}[Continuation right unit]
\label{thm:continuation-right-unit}
For every generated history $h$, $\Cont(h,\e,h)$.
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_right\_unit}
\begin{proof}
This is the base generator of $\Cont$.
\end{proof}

\begin{theorem}[Append right nonempty self contradiction]
\label{thm:append-right-nonempty-self-absurd}
For histories $h,k:\mathsf{BHist}$, a nonempty right factor cannot leave the
append result internally the same as the left factor:
$$
(\HSame(k,\emp)\to\bot)\Rightarrow
\HSame(\mathsf{append}(h,k),h)\Rightarrow\bot.
$$
\end{theorem}
\begin{proof}
Compare $\mathsf{append}(h,k)$ with $\mathsf{append}(h,\emp)$ using the
displayed self-sameness and the right unit for append. Left cancellation
forces $k\hsame\emp$, contradicting the nonempty right-factor hypothesis.
\end{proof}
\leanchecked{BEDC.FKernel.Cont.append\_right\_nonempty\_self\_absurd}

\begin{theorem}[Append left nonempty self contradiction]
\label{thm:append-left-nonempty-self-absurd}
For histories $h,k:\mathsf{BHist}$, a nonempty left factor cannot leave the
append result internally the same as the right factor:
$$
(\HSame(h,\emp)\to\bot)\Rightarrow
\HSame(\mathsf{append}(h,k),k)\Rightarrow\bot.
$$
\end{theorem}
\begin{proof}
Compare $\mathsf{append}(h,k)$ with $\mathsf{append}(\emp,k)$ using the
displayed self-sameness and the left unit for append. Right cancellation
forces $h\hsame\emp$, contradicting the nonempty left-factor hypothesis.
\end{proof}
\leanchecked{BEDC.FKernel.Cont.append\_left\_nonempty\_self\_absurd}

\begin{theorem}[Continuation associativity, relational form]
If $\Cont(a,b,ab)$ and $\Cont(b,c,bc)$, then there exists a generated history $abc$ such that
$$
\Cont(ab,c,abc)
\qquad\text{and}\qquad
\Cont(a,bc,abc).
$$
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_assoc\_exists}
\begin{proof}
Use the checked right-recursive append characterization of continuation and the append associativity spine.  Equivalently, induct on the right continuation by $c$: the empty case takes $abc=ab$, and the $\Ezero$ and $\Eone$ cases lift the witness with the matching right-recursive continuation step.
\end{proof}

\begin{theorem}[Continuation suffix iff]
\label{thm:continuation-suffix-iff}
For histories $a,b,f,p$, continuation from $a$ by the suffixed history
$\mathsf{append}(f,p)$ to the suffixed result $\mathsf{append}(b,p)$ is
equivalent to the unsuffixed continuation from $a$ by $f$ to $b$:
$$
\Cont(a,\mathsf{append}(f,p),\mathsf{append}(b,p))
\Longleftrightarrow
\Cont(a,f,b).
$$
\end{theorem}
\leanchecked{BEDC.FKernel.Cont.cont\_suffix\_iff}
\begin{proof}
For the forward direction, rewrite the suffixed continuation through
associativity so that both sides share the right suffix $p$, then cancel that
common suffix. For the reverse direction, expand the unsuffixed continuation and
use associativity of history append to rebuild the suffixed continuation.
\end{proof}
