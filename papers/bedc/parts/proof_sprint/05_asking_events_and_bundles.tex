\section{Asking events and bundles}

\begin{definition}[Asking event]
\label{def:proof-sprint-asking-event}
An asking event is a judgment
\[
\Ask(\pi,h,m,\delta),
\]
where \(\pi\) is a probe-name, \(h\) is a generated history, \(m\) is a raw mark emitted as answer, and \(\delta\) is evidence produced by the asking event.  This is not a function application.
\end{definition}
\leandef{BEDC.FKernel.Ask.Ask}

\begin{definition}[Probe bundles]
Probe bundles are generated by:
\[
\mathsf{Bnil},
\qquad
\Pi\mapsto \mathsf{Bcons}(\pi,\Pi).
\]
They are internal generated records of probe-names, not imported finite lists.
\end{definition}

\begin{definition}[Ask policy]
\label{def:proof-sprint-ask-policy}
An \(\mathsf{AskPolicy}(\Pi,D)\) states that, on source domain \(D\), bundle \(\Pi\) has enough asking events to generate signatures and that those asking events are deterministic up to \(\MSame\), respect \(\HSame\), and have composable evidence ledgers.
\end{definition}
