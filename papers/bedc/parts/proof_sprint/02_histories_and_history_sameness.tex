\section{Histories and history-sameness}

\begin{definition}[Emission histories]
Emission histories are generated by:
\[
\e : \Hist,
\qquad
h:\Hist \Rightarrow \Ezero(h):\Hist,
\qquad
h:\Hist \Rightarrow \Eone(h):\Hist.
\]
The symbol \(\e\) denotes the empty emission history.  The constructors \(\Ezero\) and \(\Eone\) record one further emission of \(\bzero\) or \(\bone\), respectively.
\end{definition}

\begin{definition}[Internal history-sameness]
The judgment \(\HSame(h,k)\) is generated by:
\[
\HSame(\e,\e),
\]
\[
\HSame(h,k)\Rightarrow \HSame(\Ezero(h),\Ezero(k)),
\]
\[
\HSame(h,k)\Rightarrow \HSame(\Eone(h),\Eone(k)).
\]
There are no cross-constructor generators.
\end{definition}

\begin{theorem}[History-sameness is an equivalence]
The judgment \(\HSame\) is reflexive, symmetric, and transitive on generated histories.
\end{theorem}
\begin{proof}
Reflexivity is by induction on generated histories.  The empty history uses \(\HSame(\e,\e)\).  The \(\Ezero\) and \(\Eone\) cases use the corresponding closure rule.

Symmetry is by induction on a derivation of \(\HSame(h,k)\).  The empty case is immediate.  The constructor cases reduce to the induction hypothesis.

Transitivity is by induction on the first sameness derivation and inversion on the second.  Closed generation prevents cross-constructor cases.  Thus the only non-empty cases are matching \(\Ezero\)-with-\(\Ezero\) or \(\Eone\)-with-\(\Eone\), where the induction hypothesis applies to the subhistories.
\end{proof}

\begin{theorem}[History no-confusion]
The following judgments are not generated:
\[
\HSame(\e,\Ezero(h)),\quad
\HSame(\e,\Eone(h)),\quad
\HSame(\Ezero(h),\Eone(k)),
\]
with symmetric variants.
\end{theorem}
\begin{proof}
Each would require a generator for a cross-constructor history-sameness judgment.  The closed generation clauses for \(\HSame\) contain no such generator.
\end{proof}
