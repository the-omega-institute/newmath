\section{Signatures and signature-sameness}

\begin{definition}[Generated signature]
The signature judgment
\[
\Sig(\Pi,h,s,\Delta)
\]
means that asking bundle \(\Pi\) against history \(h\) generates answer history \(s\) with evidence ledger \(\Delta\).  It is generated by an empty rule for \(\mathsf{Bnil}\) and a cons rule using one \(\Ask\)-event followed by the signature of the remaining bundle.
\end{definition}

\begin{definition}[Signature-sameness]
For a bundle \(\Pi\), histories \(h,k\) are signature-same, written \(\SameSig(\Pi,h,k)\), when there exist generated signatures
\[
\Sig(\Pi,h,s,\Delta),
\qquad
\Sig(\Pi,k,t,\Theta),
\]
with \(\HSame(s,t)\).
\end{definition}

\begin{theorem}[Signature determinacy under policy]
Assume \(\mathsf{AskPolicy}(\Pi,D)\).  If \(h\in D\), \(\Sig(\Pi,h,s,\Delta)\), and \(\Sig(\Pi,h,t,\Theta)\), then \(\HSame(s,t)\).
\end{theorem}
\begin{proof}
By induction on the generated bundle.  The empty bundle produces empty signatures.  For a cons bundle, determinacy of the first asking event gives \(\MSame\) of the first answer marks, and the induction hypothesis gives \(\HSame\) of the remaining answer histories.  The history constructor closure rules then yield \(\HSame(s,t)\).
\end{proof}

\begin{theorem}[Signature-sameness is an equivalence under policy]
\label{thm:proof-sprint-signature-sameness-equivalence-under-policy}
Assume \(\mathsf{AskPolicy}(\Pi,D)\).  On histories in \(D\), \(\SameSig(\Pi,-,-)\) is reflexive, symmetric, and transitive.
\end{theorem}
\leanchecked{BEDC.FKernel.Sig.proof\_sprint\_signature\_sameness\_equivalence\_under\_policy}
\begin{proof}
Reflexivity uses totality of signature generation and reflexivity of \(\HSame\).  Symmetry uses symmetry of \(\HSame\).  For transitivity, suppose \(h\) is signature-same to \(k\), and \(k\) is signature-same to \(l\).  The two signatures generated for \(k\) are internally the same by signature determinacy.  Transitivity of \(\HSame\) then composes the answer-history sameness witnesses.
\end{proof}
