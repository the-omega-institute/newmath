\section{Relational extension}

\begin{definition}[Extension relation]
Extension is a relation, not a function.  The judgment
\[
\Ext(m,h,r)
\]
means that extending history \(h\) by raw mark \(m\) may generate result history \(r\).  It is generated by:
\[
\Ext(\bzero,h,\Ezero(h)),
\qquad
\Ext(\bone,h,\Eone(h)).
\]
\end{definition}

\begin{theorem}[Extension determinacy]
If \(\Ext(m,h,r)\) and \(\Ext(m,h,r')\), then \(\HSame(r,r')\).
\end{theorem}
\begin{proof}
By cases on the generated extension judgment.  If \(m=\bzero\), both results are internally the same as \(\Ezero(h)\).  If \(m=\bone\), both results are internally the same as \(\Eone(h)\).  The claim follows by reflexivity of \(\HSame\).
\end{proof}

\begin{theorem}[Extension respects internal sameness]
If \(\MSame(m,m')\), \(\HSame(h,h')\), \(\Ext(m,h,r)\), and \(\Ext(m',h',r')\), then \(\HSame(r,r')\).
\end{theorem}
\leansorryd{BEDC.FKernel.Ext.ext\_respects\_sameness}{stub awaiting v0.2+ proof}
\begin{proof}
By closed generation of \(\MSame\), the two marks are internally the same diagonal mark.  If that mark is \(\bzero\), the extension results have the form \(\Ezero(h)\) and \(\Ezero(h')\); the conclusion follows from \(\HSame(h,h')\) and the \(\Ezero\)-closure rule.  The \(\bone\) case is analogous.
\end{proof}
